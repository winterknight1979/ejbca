49    @Test
50    public void testSingleThreaded() throws Exception {
51      LOG.trace(">testSingleThreaded");
52      final ConcurrentCache<String, Integer> cache = new ConcurrentCache<>();
53      ConcurrentCache<String, Integer>.Entry entry;
54  
55      // Create new entry
56      entry = cache.openCacheEntry("A", 1);
57      assertNotNull("openCacheEntry timed out in single-threaded test", entry);
58      assertFalse(
59          "isInCache should return false for non-existent entry",
60          entry.isInCache());
61      try {
62        entry.getValue();
63        fail("getValue should throw for non-existent entry");
64      } catch (IllegalStateException e) { // NOPMD
65      }
66      entry.putValue(111);
67      entry.setCacheValidity(60 * 1000);
68      entry.close();
69  
70      // Read it and modify the value
71      entry = cache.openCacheEntry("A", 1);
72      assertNotNull("openCacheEntry timed out in single-threaded test", entry);
73      assertTrue(
74          "isInCache should return true for existing entry", entry.isInCache());
75      assertEquals(Integer.valueOf(111), entry.getValue());
76      entry.putValue(222);
77      assertEquals(Integer.valueOf(222), entry.getValue());
78      entry.setCacheValidity(60 * 1000);
79      entry.close();
80  
81      // Read it again and modify the timeout
82      entry = cache.openCacheEntry("A", 1);
83      assertNotNull("openCacheEntry timed out in single-threaded test", entry);
84      assertTrue(
85          "isInCache should return true for existing entry", entry.isInCache());
86      assertEquals(Integer.valueOf(222), entry.getValue());
87      entry.setCacheValidity(1);
88      entry.close();
89  
90      Thread.sleep(5);
91  
92      // Try to read it now. It should have expired and a fresh entry should be
93      // returned.
94      entry = cache.openCacheEntry("A", 1);
95      assertFalse(
96          "isInCache should return false for expired entry", entry.isInCache());
97      try {
98        entry.getValue();
99        fail("getValue should throw for expired entry");
100     } catch (IllegalStateException e) { // NOPMD
101     }
102     entry.close();
103     LOG.trace("<testSingleThreaded");
104   }

109   @Test
110   public void testDisabled() throws Exception {
111     LOG.trace(">testDisabled");
112     final ConcurrentCache<String, Integer> cache = new ConcurrentCache<>();
113     cache.setEnabled(false);
114     ConcurrentCache<String, Integer>.Entry entry;
115 
116     // Create entries. All operations should be "no-ops"
117     for (int i = 0; i < 10; i++) {
118       long timeBefore = System.currentTimeMillis();
119       entry = cache.openCacheEntry("A", 1);
120       assertTrue(
121           "openCacheEntry took too long. It should be a no-op when cache is"
122               + " disabled",
123           System.currentTimeMillis() < timeBefore + 10);
124       assertNotNull("openCacheEntry timed out when cache is disabled", entry);
125       assertFalse(
126           "isInCache should return false when cache is disabled",
127           entry.isInCache());
128       try {
129         entry.getValue();
130         fail("getValue should throw when caching is disabled");
131       } catch (IllegalStateException e) { // NOPMD
132       }
133       entry.putValue(111);
134       entry.setCacheValidity(60 * 1000);
135       entry.close();
136 
137       // This should have been a no-op
138       cache.checkNumberOfEntries(0, 0);
139     }
140     LOG.trace("<testDisabled");
141   }

164   /**
165    * Test.
166    * @throws Exception fail

168   @Test
169   public void testExpiryBehavior() throws Exception {
170     LOG.trace(">testExpiryBehavior");
171     final ConcurrentCache<String, Integer> cache = new ConcurrentCache<>();
172     cache.setCleanupInterval(60000L); // don't run cleanup during test
173 
174     // Create an entry that will expire
175     final ConcurrentCache<String, Integer>.Entry entry1 =
176         cache.openCacheEntry("A", 1000);
177     entry1.putValue(456);
178     entry1.setCacheValidity(1L);
179     entry1.close();
180 
181     Thread.sleep(5);
182 
183     // We should get an expired entry now. Start updating it but don't finish
184     final ConcurrentCache<String, Integer>.Entry entry2 =
185         cache.openCacheEntry("A", 1000);
186     assertFalse(
187         "Should be forced to update the cache.",
188         entry2.isInCache()); // technically, it is still in the cache, but
189     // ConcurrentCache will pretend it's not, so the
190     // caller will update it
191 
192     // Start a new request for the same entry. Should use the old expired entry
193     final long startTime = System.currentTimeMillis();
194     final ConcurrentCache<String, Integer>.Entry entry3 =
195         cache.openCacheEntry("A", 1000);
196     final long stopTime = System.currentTimeMillis();
197     assertTrue("openCacheEntry should not block.", stopTime - startTime < 500L);
198     assertTrue(
199         "Should be able to use the existing cached data, because someone is"
200             + " updating the new value",
201         entry3.isInCache());
202     assertEquals(Integer.valueOf(456), entry3.getValue());
203     entry3.close();
204 
205     // Now write the new value
206     entry2.putValue(768);
207     entry2.setCacheValidity(500);
208     entry2.close();
209 
210     Thread.sleep(5);
211 
212     // We should see the new entry now
213     final ConcurrentCache<String, Integer>.Entry entry4 =
214         cache.openCacheEntry("A", 1000);
215     assertTrue("Entry should still be in cache.", entry4.isInCache());
216     assertEquals(
217         "Entry should have been updated.",
218         Integer.valueOf(768),
219         entry4.getValue());
220     entry4.close();
221 
222     LOG.trace("<testExpiryBehavior");
223   }

228   @Test(timeout = 6000)
229   public void testRandomMultiThreaded() throws InterruptedException {
230     LOG.trace(">testRandomMultiThreaded");
231     // This tests outputs as LOT of debug/trace messages. JUnit even runs out of
232     // heap space if those are enabled.
233     Logger.getRootLogger().setLevel(Level.INFO);
234     Logger.getLogger(ConcurrentCache.class).setLevel(Level.INFO);
235     try {
236       final ConcurrentCache<String, Integer> cache = new ConcurrentCache<>();
237       cache.setMaxEntries(20);
238       cache.setCleanupInterval(100); // To stress the system a bit more
239 
240       final Thread[] threads = new Thread[NUM_RANDOM_THREADS];
241       final CacheTestRunner[] runners = new CacheTestRunner[NUM_RANDOM_THREADS];
242       for (int i = 0; i < threads.length; i++) {
243         runners[i] = new CacheTestRunner(cache, THREAD_RANDOM);
244         threads[i] = new Thread(runners[i], "CacheTestThread" + i);
245       }
246       try {
247         LOG.info("Now starting the threads");
248         for (int i = 0; i < threads.length; i++) {
249           threads[i].start();
250         }
251 
252         LOG.info("All threads started");
253         Thread.sleep(2000);
254       } finally {
255         LOG.info("Asking threads to stop");
256         for (int i = 0; i < threads.length; i++) {
257           runners[i].shouldExit = true;
258         }
259 
260         LOG.info("Waiting for join of first thread (1 sec timeout)");
261         threads[0].join(1000);
262         LOG.info("Waiting for other threads");
263         Thread.sleep(100);
264         for (int i = 1; i < threads.length; i++) {
265           threads[i].join(1);
266         }
267       }
268 
269       long timeout =
270           System.currentTimeMillis()
271               + 2000; // if a thread stops for more than 2 s in cleanup() then
272       // that's a problem by itself
273       for (int i = 0; i < threads.length; i++) {
274         if (threads[i].isAlive() && System.currentTimeMillis() < timeout) {
275           threads[i].join(2000);
276         }
277         assertFalse("Thread " + i + " was still alive", threads[i].isAlive());
278       }
279     } finally {
280       // Preferably, the log level should be restored to the original value,
281       // but neither of the getLevel/getEffectiveLevel methods return the
282       // correct value
283       // so we hard-code Level.TRACE here.
284       Logger.getRootLogger().setLevel(Level.TRACE);
285       Logger.getLogger(ConcurrentCache.class).setLevel(Level.TRACE);
286       LOG.trace("<testRandomMultiThreaded");
287     }
288   }

290   /** config. */
291   private static final int MAXENTRIES = 1000000;
292   /** config. */
293   private static final int OVERSHOOT =
294       MAXENTRIES + (MAXENTRIES / 2) + 1; // overshoot by 50%
295   /** config. */
296   private static final int MIN_ENTRIES_AFTER_CLEANUP =
297       MAXENTRIES - (MAXENTRIES / 4); // 75% of maxentries
298   /**
299    * Test.
300    * @throws Exception fail
301    */
302   @Test
303   public void testMaxEntries() throws Exception {
304     LOG.trace(">testMaxEntries");
305     final ConcurrentCache<String, Integer> cache = new ConcurrentCache<>();
306     cache.setMaxEntries(MAXENTRIES);
307     cache.setCleanupInterval(100);
308     ConcurrentCache<String, Integer>.Entry entry;
309 
310     try {
311       Logger.getLogger(ConcurrentCache.class).setLevel(Level.INFO);
312       // Create initial entries
313       LOG.debug("Creating initial entries");
314       for (int i = 0; i < MAXENTRIES; i++) {
315         entry = cache.openCacheEntry(String.valueOf(i), 1);
316         assertNotNull("openCacheEntry timed out", entry);
317         assertFalse(
318             "isInCache should return false for non-existent entry",
319             entry.isInCache());
320         entry.putValue(i);
321         entry.setCacheValidity(60 * 1000);
322         entry.close();
323       }
324 
325       cache.checkNumberOfEntries(MAXENTRIES, MAXENTRIES);
326 
327       // Add some more. Cleanup is guaranteed to run if we overshoot by 50%
328       LOG.debug("Creating more entries (to overshoot the limit)");
329       for (int i = MAXENTRIES; i <= OVERSHOOT; i++) {
330         entry = cache.openCacheEntry(String.valueOf(i), 1);
331         assertNotNull("openCacheEntry timed out", entry);
332         assertFalse(
333             "isInCache should return false for non-existent entry",
334             entry.isInCache());
335         entry.putValue(i);
336         entry.setCacheValidity(60 * 1000);
337         entry.close();
338       }
339 
340       LOG.debug("Sleeping to allow for cleanup to run again");
341       Thread.sleep(100);
342 
343       // Access the cache once more to trigger a cleanup
344       entry = cache.openCacheEntry(String.valueOf("x"), 1);
345       assertNotNull("openCacheEntry timed out", entry);
346       assertFalse(
347           "isInCache should return false for non-existent entry",
348           entry.isInCache());
349       entry.putValue(-123456);
350       entry.setCacheValidity(60 * 1000);
351       entry.close();
352 
353       LOG.debug("Done creating entries");
354 
355       // Cleanup should have run now
356       cache.checkNumberOfEntries(MIN_ENTRIES_AFTER_CLEANUP, MAXENTRIES - 1);
357     } finally {
358       Logger.getLogger(ConcurrentCache.class).setLevel(Level.TRACE);
359       LOG.trace("<testMaxEntries");
360     }

363   private static final class CacheTestRunner implements Runnable {
364     /** ID. */
365     private final int id;
366     /** cache. */
367     private final ConcurrentCache<String, Integer> cache;
368 
369     /** keys. */
370     private static final String[] KEYS = {
371       "A", "B", "C", "D", "E", "F", "G", "H", "I", "J", "K", "L", "M", "N", "O",
372       "P", "Q", "R", "S", "T", "U", "V", "W", "X", "Y", "Z"
373     };
374     /** bool. */
375    private volatile boolean shouldExit = false;
376 
377     CacheTestRunner(
378         final ConcurrentCache<String, Integer> aCache, final int anId) {
379       this.cache = aCache;
380       this.id = anId;
381     }
382 
383     @Override
384     public void run() {
385       ConcurrentCache<String, Integer>.Entry entry;
386 
387       try {
388         switch (id) {
389           case THREAD_ONE:
390             // 0 ms
391             entry = cache.openCacheEntry("A", 1);
392             assertFalse(
393                 "non-existent entry should not be in cache", entry.isInCache());
394             Thread.sleep(200);
395             entry.putValue(123);
396             entry.setCacheValidity(200); // = valid to ~400 ms
397             entry.close();
398             break;
399           case THREAD_TWO:
400             Thread.sleep(100);
401             // 100 ms
402             entry = cache.openCacheEntry("A", CONCURRENT_OPEN_TIMEOUT);
403             // 200 ms
404             assertTrue("existing entry should be in cache", entry.isInCache());
405             assertEquals(
406                 "wrong value of cache entry",
407                 Integer.valueOf(123),
408                 entry.getValue());
409             entry.close();
410 
411             Thread.sleep(300);
412 
413             // 500 ms
414             entry = cache.openCacheEntry("A", CONCURRENT_OPEN_TIMEOUT);
415             assertFalse("entry should have expired", entry.isInCache());
416             entry.close();
417             break;
418           case THREAD_RANDOM:
419             final Random random = new Random(id);
420             final Integer value = Integer.valueOf(id);
421             while (!shouldExit) {
422               final String key = KEYS[random.nextInt(KEYS.length)];
423               entry = cache.openCacheEntry(key, CONCURRENT_OPEN_TIMEOUT);
424               if (entry.isInCache()) {
425                 assertNotNull(
426                     "got null value in cache entry", entry.getValue());
427                 if (random.nextInt(10000) == 123) {
428                   entry.setCacheValidity(2000 + random.nextInt(100));
429                 }
430                 entry.close();
431               } else {
432                 Thread.sleep(1 + random.nextInt(50));
433                 entry.putValue(value);
434                 entry.setCacheValidity(2000 + random.nextInt(100));
435                 entry.close();
436               }
437             }
438             break;
439           default:
440             throw new IllegalStateException("invalid test thread id");
441         }
442       } catch (InterruptedException ie) {
443         throw new RuntimeException(ie);
444       }
445     }
