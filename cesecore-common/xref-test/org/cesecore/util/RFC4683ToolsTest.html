65      @Test
66      public void testGenerateSimForInternalSanFormat() throws NoSuchAlgorithmException, NoSuchProviderException {
67          // 1. Test SANs.
68          // 1a. Test empty SAN -> nothing happens.
69          String san = "";
70          assertEquals(RFC4683Tools.generateSimForInternalSanFormat(san), san);
71          // 1b. Test SAN without 'subjectIdentificationMethod' -> nothing happens.
72          san = "DNSNAME=localhost";
73          assertEquals(RFC4683Tools.generateSimForInternalSanFormat(san), san);
74          // 1c. Test SAN with 'subjectIdentificationMethod' but without SIM parameters -> nothing happens.        
75          san = "SUBJECTIDENTIFICATIONMETHOD=, DNSNAME=localhost";
76          assertEquals(RFC4683Tools.generateSimForInternalSanFormat(san), san);
77          // 1d: Test SAN with 'subjectIdentificationMethod' but with wrong number of SIM parameters.
78          san = "SUBJECTIDENTIFICATIONMETHOD=2.16.840.1.101.3.4.2.1::MyStrongPassword::SsiType::abc::abc, DNSNAME=localhost";
79          try {
80              RFC4683Tools.generateSimForInternalSanFormat(san);
81              fail("An illegal number SIM parameters should throw an IllegalArgumentException: " + san);
82          } catch (IllegalArgumentException e) {
83              assertTrue(e.getMessage().startsWith("Wrong SIM input string with "));
84          }
85          // SIM is calculated (4 tokens).
86          san = "SUBJECTIDENTIFICATIONMETHOD=2.16.840.1.101.3.4.2.1::MyStrongPassword::SsiType::SsiValue, DNSNAME=localhost";
87          RFC4683Tools.generateSimForInternalSanFormat(san);
88          // Calculated SIM (3 tokens) -> nothing happens
89          san = "SUBJECTIDENTIFICATIONMETHOD=2.16.840.1.101.3.4.2.1::CB3AE7FBFFFD9C85A3FB234E51FFFD2190B1F8F161C0A2873B998EFAC067B03A::6D9E6264DDBD0FC997B9B40524247C8BC319D02A583F4B499DD3ECAF06C786DF, DNSNAME=localhost";
90          assertEquals(RFC4683Tools.generateSimForInternalSanFormat(san), san);
91      }

93      @Test
94      public void testGenerateInternalSimString() {
95          // 1. Use different illegal SIM parameters.
96          String[] simParameters = new String[] { "2.16.840.1.101.3.4.2.1", "MyStrongPassword", "SsiType", "Sensitive identifier information" };
97          // 2a. Use invalid hash algorithm OID (unknown | empty | null) -> IAE("Hash algorithm OID string must not be null or empty.")
98          simParameters = new String[] { null, "MyStrongPassword", "SsiType", "Sensitive identifier information" };
99          assertIAEForGenerateInternalSimString(simParameters, "Hash algorithm OID string must not be null or empty");
100         simParameters = new String[] { "2.16.840.1.101a.A.B.C", "MyStrongPassword", "SsiType", "Sensitive identifier information" };
101         assertIAEForGenerateInternalSimString(simParameters, "Hash algorithm with OID");
102         // 2b. Use invalid password ( empty | null | ... ) -> FIPS 112 and FIPS 180-1 compliance is not tested -> IAE("The user chosen password must not be null or empty")
103         simParameters = new String[] { "2.16.840.1.101.3.4.2.1", null, "SsiType", "Sensitive identifier information" };
104         assertIAEForGenerateInternalSimString(simParameters, "The user chosen password must not be null or empty");
105         // 2b. User chosen password must have at least 8 characters.
106         simParameters = new String[] { "2.16.840.1.101.3.4.2.1", "weakPwd", "SsiType", "Sensitive identifier information" };
107         assertIAEForGenerateInternalSimString(simParameters, "The user chosen password must not be null or empty");
108         // 2c. Use invalid SSI type (empty | null) -> IAE expected
109         simParameters = new String[] { "2.16.840.1.101.3.4.2.1", "MyStrongPassword", null, "Sensitive identifier information" };
110         assertIAEForGenerateInternalSimString(simParameters, "The sensitve identification information type must not be null or empty");
111         // 2d. Use invalid SSI (empty | null) -> IAE expected
112         simParameters = new String[] { "2.16.840.1.101.3.4.2.1", "MyStrongPassword", "SsiType", null };
113         assertIAEForGenerateInternalSimString(simParameters, "The sensitve identification information must not be null or empty");
114 
115     }

117     private final void assertIAEForGenerateInternalSimString(final String[] parameters, final String message) {
118         try {
119             RFC4683Tools.generateInternalSimString(parameters[0], parameters[1], parameters[2], parameters[3]);
120             fail("An IllegalArgumentException should have been thrown: " + message);
121         } catch (IllegalArgumentException | NoSuchAlgorithmException | NoSuchProviderException e) {
122             assertTrue(e.getMessage().startsWith(message));
123         }
124     }

126     @Test
127     public void testAsn1ReadWrite() throws NoSuchAlgorithmException, NoSuchProviderException, IOException {
128         final String[] simParameters = new String[] { "2.16.840.1.101.3.4.2.1", "MyStrongPassword", "SsiType", "Sensitive identifier information" };
129         final String internalSimString = RFC4683Tools.generateInternalSimString(simParameters[0], simParameters[1], simParameters[2],
130                 simParameters[3]);
131         final String[] simTokens = internalSimString.split("(::)");
132 
133         final ASN1Primitive generalName = testCreateSimGeneralName(simParameters, simTokens);
134         testGetSimStringSequence(internalSimString, generalName);
135     }
136 
137     private ASN1Primitive testCreateSimGeneralName(final String[] simParameters, final String[] simTokens)
138             throws NoSuchAlgorithmException, NoSuchProviderException, IOException {
139         final ASN1Primitive generalName = RFC4683Tools.createSimGeneralName(simTokens[0], simTokens[1], simTokens[2]);
140         final DERSequence simSequence = ((DERSequence) ((DERTaggedObject) ((DERSequence) ((DERTaggedObject) generalName.toASN1Primitive())
141                 .getObjectParser(0, true)).getObjectAt(1)).getObjectParser(0, true));
142         final String algorithmIdentifier = ((AlgorithmIdentifier) simSequence.getObjectAt(0)).getAlgorithm().getId();
143         final DEROctetString authorityRandom = (DEROctetString) simSequence.getObjectAt(1);
144         final DEROctetString sim = (DEROctetString) simSequence.getObjectAt(2);
145         assertEquals("The SIM algorithm identifier must match.", simParameters[0], algorithmIdentifier);
146         assertEquals("The SIM authority random must match.", simTokens[1], new String(authorityRandom.getOctets()));
147         assertEquals("The SIM algorithm identifier must match.", simTokens[2], new String(sim.getOctets()));
148         return generalName;
149     }

151     private void testGetSimStringSequence(String simString, ASN1Primitive generalName) {
152         final ASN1Sequence otherName = ASN1Sequence.getInstance(((DERTaggedObject) generalName.toASN1Primitive()).getObject());
153         final String simStringBySequence = RFC4683Tools.getSimStringSequence(otherName);
154         assertEquals("The SIM string, extracted by its ASN.1 structure must match the original.", simString, simStringBySequence);
155     }
