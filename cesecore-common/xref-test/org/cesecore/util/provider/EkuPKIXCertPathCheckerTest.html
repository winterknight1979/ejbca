103   @Test
104   public void testNoEkuInCert() throws Exception {
105     LOG.trace(">testNoEkuInCert");
106     /*
107      * When no EKU is present in the certificate, the PKIXCertPathChecker
108      *  should never be invoked.
109      * This just documents the actual behavior in such a case.
110      */
111     assertTrue(validateCert(keyPair, LEAF, null, null));
112     assertFalse(validateCert(keyPair, CA, null, null));
113     assertTrue(validateCert(keyPair, LEAF, null, ekusEmpty));
114     assertFalse(validateCert(keyPair, CA, null, ekusEmpty));
115     assertFalse(validateCert(keyPair, LEAF, null, ekus2));
116     assertFalse(validateCert(keyPair, CA, null, ekus2));
117     assertFalse(validateCert(keyPair, LEAF, null, ekus3));
118     assertFalse(validateCert(keyPair, CA, null, ekus3));
119     LOG.trace("<testNoEkuInCert");
120   }

125   @Test
126   public void testEmptyCriticalEkuInCert() throws Exception {
127     /*
128      * When an empty EKU is present in the certificate, the PKIXCertPathChecker
129      * will perform the check for required.

133     assertFalse(validateCert(keyPair, CA, ekusEmpty, null));
134     assertTrue(validateCert(keyPair, LEAF, ekusEmpty, ekusEmpty));
135     assertFalse(validateCert(keyPair, CA, ekusEmpty, ekusEmpty));
136     assertFalse(validateCert(keyPair, LEAF, ekusEmpty, ekus2));
137     assertFalse(validateCert(keyPair, CA, ekusEmpty, ekus2));
138     assertFalse(validateCert(keyPair, LEAF, ekusEmpty, ekus3));
139     assertFalse(validateCert(keyPair, CA, ekusEmpty, ekus3));

176   /**
177    * @param aKeyPair keys
178    * @param isCa CA
179    * @param actualOids OIDs
180    * @param requiredOids OIDs
181    * @return true if the extendedKeyUsage was accepted
182    * @throws Exception fail
183    */
184   private boolean validateCert(
185       final KeyPair aKeyPair,
186       final boolean isCa,
187       final List<String> actualOids,
188       final List<String> requiredOids)
189       throws Exception {
190 
191     final long now = System.currentTimeMillis();
192     final List<Extension> additionalExtensions = new ArrayList<Extension>();
193     if (actualOids != null) {
194       List<KeyPurposeId> actual = new ArrayList<KeyPurposeId>();
195       for (final String oid : actualOids) {
196         actual.add(KeyPurposeId.getInstance(new ASN1ObjectIdentifier(oid)));
197       }
198       final ExtendedKeyUsage extendedKeyUsage =
199           new ExtendedKeyUsage(actual.toArray(new KeyPurposeId[0]));
200       final ASN1Sequence seq =
201           ASN1Sequence.getInstance(extendedKeyUsage.toASN1Primitive());
202       final Extension extension =
203           new Extension(Extension.extendedKeyUsage, true, seq.getEncoded());
204       additionalExtensions.add(extension);
205     }
206     final int ku;
207     if (isCa) {
208       ku = X509KeyUsage.cRLSign | X509KeyUsage.keyCertSign;
209     } else {
210       ku = X509KeyUsage.digitalSignature | X509KeyUsage.keyEncipherment;
211     }
212     final X509Certificate cert =
213         CertTools.genSelfCertForPurpose(
214             "CN=dummy",
215             new Date(now - 3600000L),
216             new Date(now + 3600000L),
217             null,
218             aKeyPair.getPrivate(),
219             aKeyPair.getPublic(),
220             AlgorithmConstants.SIGALG_SHA1_WITH_RSA,
221             isCa,
222             ku,
223             null,
224             null,
225             BouncyCastleProvider.PROVIDER_NAME,
226             true,
227             additionalExtensions);
228     final PKIXCertPathChecker pkixCertPathChecker =
229         new EkuPKIXCertPathChecker(requiredOids);
230     final Collection<String> unresolvedCritExts =
231         new ArrayList<String>(
232             Arrays.asList(new String[] {Extension.extendedKeyUsage.getId()}));
233     pkixCertPathChecker.check(cert, unresolvedCritExts);
234     return !unresolvedCritExts.contains(Extension.extendedKeyUsage.getId());
235   }
