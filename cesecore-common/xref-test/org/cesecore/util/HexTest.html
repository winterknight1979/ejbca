90    @Test
91    public void test01HexSmall() throws Exception {
92      // Testcert is on long line of base 64 encoded stuff
93      byte[] certBytes = Base64.decode(testcertOneline.getBytes());
94      assertNotNull(certBytes);
95      // This should be a cert
96      Certificate cert =
97          CertTools.getCertfromByteArray(certBytes, Certificate.class);
98      assertNotNull(cert);
99      byte[] hexBytes = Hex.decode(hexCert.getBytes());
100     assertEquals(
101         new String(Base64.encode(certBytes)),
102         new String(Base64.encode(hexBytes)));
103     Certificate cert1 =
104         CertTools.getCertfromByteArray(hexBytes, Certificate.class);
105     assertEquals(CertTools.getSubjectDN(cert), CertTools.getSubjectDN(cert1));
106     byte[] hexBytes2 = Hex.encode(cert1.getEncoded());
107     assertEquals(new String(hexBytes2), hexCert);
108   }
