1888         v.add(new ASN1ObjectIdentifier(CertTools.PERMANENTIDENTIFIER_OBJECTID));
1889

1891         ASN1EncodableVector piSeq = new ASN1EncodableVector();
1892         if (identifierValue != null) {
1893             piSeq.add(new DERUTF8String(identifierValue));
1894         }
1895         if (assigner != null) {
1896             piSeq.add(new ASN1ObjectIdentifier(assigner));
1897         }
1898         v.add(new DERTaggedObject(true, 0, new DERSequence(piSeq)));
1899         result = new DERSequence(v);

1901         log.info(ASN1Dump.dumpAsString(result));
1902         return result;
