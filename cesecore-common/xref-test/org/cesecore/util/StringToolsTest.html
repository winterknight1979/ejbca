83      @Test
84      public void test04Strip() throws Exception {
85          log.trace(">test04Strip()");
86          String strip1 = "CN=foo, O=Acme\\, Inc, OU=;\\/\\<\\>bar";
87          String stripped = StringTools.strip(strip1);
88          assertFalse("String has chars that should be stripped!", StringTools.hasSqlStripChars(strip1).isEmpty());
89          assertEquals("String not stripped correctly! " + stripped, "CN=foo, O=Acme\\, Inc, OU=//\\<\\>bar", stripped);
90  
91          strip1 = "CN=foo, O=Acme\\, Inc, OU=;\\/<>\"bar";
92          stripped = StringTools.strip(strip1);
93          assertFalse("String has chars that should be stripped!", StringTools.hasSqlStripChars(strip1).isEmpty());
94          assertEquals("String not stripped correctly! " + stripped, "CN=foo, O=Acme\\, Inc, OU=//<>\"bar", stripped);
95          strip1 = "CN=foo\\+bar, O=Acme\\, Inc";
96          stripped = StringTools.strip(strip1);
97          assertTrue("String does not have chars to be stripped!", StringTools.hasSqlStripChars(strip1).isEmpty());
98          assertEquals("String not stripped correctly! " + stripped, "CN=foo\\+bar, O=Acme\\, Inc", stripped);
99  
100         // Multi-valued.. not supported by EJBCA yet.. let it through for backwards compatibility.
101         strip1 = "CN=foo+CN=bar, O=Acme\\, Inc";
102         stripped = StringTools.strip(strip1);
103         assertTrue("String does not have chars to be stripped!", StringTools.hasSqlStripChars(strip1).isEmpty());
104         assertEquals("String not stripped correctly! " + stripped, "CN=foo+CN=bar, O=Acme\\, Inc", stripped);
105 
106         log.trace("<test04Strip()");
107     }

109     final static String FORBIDDEN_CHARS_KEY = "forbidden.characters";
110     private static void forbiddenTest( final String forbidden, final String input, final String output ) {
111         ConfigurationHolder.instance().setProperty(FORBIDDEN_CHARS_KEY, forbidden);
112         StringTools.CharSet.reset();
113         final String stripped = StringTools.strip(input);
114         if ( input.equals(output) ) {
115             assertTrue("The string do NOT have chars that should be stripped!", StringTools.hasStripChars(input).isEmpty());
116         } else {
117             assertFalse("The string DO have chars that should be stripped!", StringTools.hasStripChars(input).isEmpty());
118         }
119         assertEquals("String not stripped correctly!", output, stripped);
120     }

272         assertTrue(StringTools.hasSqlStripChars(str).isEmpty());
273 
274         str = "select * from Table; delete from password";
275         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
276 
277         str = "select * from User where username like 'foo\\%'";
278         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
279 
280         // check that we can escape commas
281         str = "foo\\,";
282         assertTrue(StringTools.hasSqlStripChars(str).isEmpty());
283 
284         str = "foo\\;";
285         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
286 
287         // Check that escaping does not work for other characters
288         str = "foo\\?";
289         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
290 
291         str = "foo\\?bar";
292         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
293 
294         str = "\\?bar";
295         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
296 
297         // Check special case that a slash at the end also returns bad
298         str = "foo\\";
299         assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
