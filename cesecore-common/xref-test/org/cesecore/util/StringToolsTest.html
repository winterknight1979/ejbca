96      String stripped = StringTools.strip(strip1);
97      assertFalse(
98          "String has chars that should be stripped!",
99          StringTools.hasSqlStripChars(strip1).isEmpty());
100     assertEquals(
101         "String not stripped correctly! " + stripped,
102         "CN=foo, O=Acme\\, Inc, OU=//\\<\\>bar",
103         stripped);
104 
105     strip1 = "CN=foo, O=Acme\\, Inc, OU=;\\/<>\"bar";
106     stripped = StringTools.strip(strip1);
107     assertFalse(
108         "String has chars that should be stripped!",
109         StringTools.hasSqlStripChars(strip1).isEmpty());
110     assertEquals(
111         "String not stripped correctly! " + stripped,
112         "CN=foo, O=Acme\\, Inc, OU=//<>\"bar",
113         stripped);
114     strip1 = "CN=foo\\+bar, O=Acme\\, Inc";
115     stripped = StringTools.strip(strip1);
116     assertTrue(
117         "String does not have chars to be stripped!",
118         StringTools.hasSqlStripChars(strip1).isEmpty());
119     assertEquals(
120         "String not stripped correctly! " + stripped,
121         "CN=foo\\+bar, O=Acme\\, Inc",
122         stripped);
123 
124     // Multi-valued.. not supported by EJBCA yet.. let it through for backwards
125     // compatibility.
126     strip1 = "CN=foo+CN=bar, O=Acme\\, Inc";
127     stripped = StringTools.strip(strip1);
128     assertTrue(
129         "String does not have chars to be stripped!",
130         StringTools.hasSqlStripChars(strip1).isEmpty());
131     assertEquals(
132         "String not stripped correctly! " + stripped,
133         "CN=foo+CN=bar, O=Acme\\, Inc",
134         stripped);
135 
136     log.trace("<test04Strip()");
137   }

140   static final String FORBIDDEN_CHARS_KEY = "forbidden.characters";
141 
142   private static void forbiddenTest(
143       final String forbidden, final String input, final String output) {
144     ConfigurationHolder.instance().setProperty(FORBIDDEN_CHARS_KEY, forbidden);
145     StringTools.CharSet.reset();
146     final String stripped = StringTools.strip(input);
147     if (input.equals(output)) {
148       assertTrue(
149           "The string do NOT have chars that should be stripped!",
150           StringTools.hasStripChars(input).isEmpty());
151     } else {
152       assertFalse(
153           "The string DO have chars that should be stripped!",
154           StringTools.hasStripChars(input).isEmpty());
155     }
156     assertEquals("String not stripped correctly!", output, stripped);
157   }

167     try {
168       final String input =

381     String str = "select * from Table";
382     assertTrue(StringTools.hasSqlStripChars(str).isEmpty());
383 
384     str = "select * from Table; delete from password";
385     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
386 
387     str = "select * from User where username like 'foo\\%'";
388     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
389 
390     // check that we can escape commas
391     str = "foo\\,";
392     assertTrue(StringTools.hasSqlStripChars(str).isEmpty());
393 
394     str = "foo\\;";
395     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
396 
397     // Check that escaping does not work for other characters
398     str = "foo\\?";
399     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
400 
401     str = "foo\\?bar";
402     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
403 
404     str = "\\?bar";
405     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
406 
407     // Check special case that a slash at the end also returns bad
408     str = "foo\\";
409     assertFalse(StringTools.hasSqlStripChars(str).isEmpty());
410   }

519         StringTools.splitURIs(";;;;;aa;bb;;;;cc;"));

522         StringTools.splitURIs(";;\"aa\";;;b;c;;;;d;\"e\";;;"));
