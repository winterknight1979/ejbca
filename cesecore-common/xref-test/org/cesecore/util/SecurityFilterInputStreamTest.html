26  /**
27   * Tests SecurityFilterInputStream class that can be used to prevent java
28   * heap-overflow during reading from input stream.
29   *
30   * @version $Id: SecurityFilterInputStreamTest.java 26057 2017-06-22 08:08:34Z
31   *     anatom $
32   */
33  public class SecurityFilterInputStreamTest {
34  
35      /** Logger. */
36    private static final Logger LOG =
37        Logger.getLogger(SecurityFilterInputStreamTest.class);
38  
39    /**
40     * Prepares output stream with exploit string.
41     *
42     * @param outputStream stream where exploit String objects will be write into
43     * @param payloadSize payloadSize of string that is going to be generated; use
44     *     high value for exploit (etc. 0x1FFFFFFF); also be aware that some high
45     *     values could throw heap error on attacker's JVM
46     * @throws IOException fail
47     */
48    public static void prepareExploitStream(
49        final OutputStream outputStream, final long payloadSize)
50        throws IOException {
51  
52      outputStream.write(ObjectOutputStream.STREAM_MAGIC >>> 8);
53      outputStream.write(ObjectOutputStream.STREAM_MAGIC);
54      outputStream.write(0); // don't need the high bits set for the version
55      outputStream.write(ObjectOutputStream.STREAM_VERSION);
56  
57      if (payloadSize <= 0xFFFF) {
58        outputStream.write(ObjectOutputStream.TC_STRING);
59        outputStream.write((int) payloadSize >>> 8);
60        outputStream.write((int) payloadSize);
61      } else {
62        outputStream.write(ObjectOutputStream.TC_LONGSTRING);
63        outputStream.write((int) (payloadSize >>> 56));
64        outputStream.write((int) (payloadSize >>> 48));
65        outputStream.write((int) (payloadSize >>> 40));
66        outputStream.write((int) (payloadSize >>> 32));
67        outputStream.write((int) (payloadSize >>> 24));
68        outputStream.write((int) (payloadSize >>> 16));
69        outputStream.write((int) (payloadSize >>> 8));
70        outputStream.write((int) (payloadSize >>> 0));
71      }
72  
73      for (long i = 0; i < payloadSize; i++) {
74        outputStream.write((byte) 'B');
75      }
76      outputStream.flush();
77    }
78  
79    /**
80     * Test preventing exploit by checking its size with
81     * SecurityFilterInputStream.
82     *
83     * @throws Exception fail
84     */
85    @Test
86    public void testPreventingTheStringBuilderExploit() throws Exception {
87      LOG.trace(">testPreventingTheStringBuilderExploit");
88      ObjectInputStream objectInputStream = null;
89      ByteArrayOutputStream byteArrayOutputStream = null;
90      try {
91        byteArrayOutputStream = new ByteArrayOutputStream();
92        prepareExploitStream(
93            byteArrayOutputStream,
94            0x1FFFFF); // 0x1FFFFF just simulates exploit stream
95  
96        objectInputStream =
97            new ObjectInputStream(
98                new SecurityFilterInputStream(
99                    new ByteArrayInputStream(byteArrayOutputStream.toByteArray()),
100                   0xFFFFF));
101       objectInputStream.readObject(); // would throw Java heap error if
102       // SecurityFilterInputStream is not applied
103       fail(
104           "No Java heap error happened for StringBuilder exploit (MaxHeap = "
105               + Runtime.getRuntime().maxMemory() / (1024 * 1024)
106               + "MB) and SecurityFilterInputStream hasn't limited the size of"
107               + " input stream during testPreventingTheStringBuilderExploit");
108     } catch (SecurityException e) {
109       // Good
110     } catch (Exception e) {
111       fail(
112           "Unexpected exception: "
113               + e.getMessage()
114               + " during testPreventingTheStringBuilderExploit");
115     } finally {
116       if (byteArrayOutputStream != null) {
117         byteArrayOutputStream.close();
118       }
119       if (objectInputStream != null) {
120         objectInputStream.close();
121       }
122     }
123     LOG.trace("<testPreventingTheStringBuilderExploit");
124   }

126   /**
127    * Test if good input stream (size &lt; SecurityFilterInputStream.maxBytes)
128    * can be filtered.
129    *
130    * @throws Exception fail
131    */
132   @Test
133   public void testAcceptedSizeInputStream() throws Exception {
134     LOG.trace(">testAcceptedSizeInputStream");
135     ObjectInputStream objectInputStream = null;
136     ByteArrayOutputStream byteArrayOutputStream = null;
137     try {
138       byteArrayOutputStream = new ByteArrayOutputStream();
139       prepareExploitStream(
140           byteArrayOutputStream, 0xFFFF); // 0xFFFF simulates safe stream
141 
142       objectInputStream =
143           new ObjectInputStream(
144               new SecurityFilterInputStream(
145                   new ByteArrayInputStream(byteArrayOutputStream.toByteArray()),
146                   0xFFFFF));
147       objectInputStream.readObject();
148     } catch (Exception e) {
149       fail(
150           "Unexpected exception: "
151               + e.getMessage()
152               + " during testAcceptedSizeInputStream");
153     } finally {
154       if (byteArrayOutputStream != null) {
155         byteArrayOutputStream.close();
156       }
157       if (objectInputStream != null) {
158         objectInputStream.close();
159       }
160     }
161     LOG.trace("<testAcceptedSizeInputStream");
162   }
