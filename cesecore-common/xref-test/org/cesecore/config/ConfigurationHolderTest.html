35  public class ConfigurationHolderTest {
36  
37      @Before
38      public void setUp() {
39          ConfigurationHolder.instance().clear();
40      }
41  
42      @Test
43      public void testGetString() throws Exception {
44          File f = File.createTempFile("cesecore", "test");
45          try {
46              FileWriter fw = new FileWriter(f);
47  
48              // First the value "property1" should not exists in configuration
49              // Test null default value and a default value
50              String val = ConfigurationHolder.getString("property1");
51              assertNull(val);
52              // Create a configuration file
53              fw.write("property1=foo\n");
54              fw.write("property2=${property1}bar\n");
55              // Make sure we handle comma in values
56              fw.write("property3=EN,DE,FR\n");
57              fw.close();
58              // We haven't read it so it should still not contain our property
59              val = ConfigurationHolder.getString("property1");
60              assertEquals(null, val);
61              // Add the config file to configuration, now the property should be visible
62              ConfigurationHolder.addConfigurationFile(f.getAbsolutePath());
63              val = ConfigurationHolder.getString("property1");
64              assertEquals("foo", val);
65              // An expanded string "${property1}bar" will be expanded with the value from "property1" (foo)
66              val = ConfigurationHolder.getString("property2");
67              assertEquals("foobar", val);
68              // Make sure we handle comma in values
69              val = ConfigurationHolder.getString("property3");
70              assertEquals("EN,DE,FR", val);
71          } finally {
72              f.deleteOnExit();
73          }
74      }
75  
76      @Test
77      public void testGetDefaultValuesWithCommas() throws SecurityException, NoSuchFieldException, IllegalArgumentException, IllegalAccessException, IOException, ConfigurationException {        
78          // Make sure we handle comma in default values
79          String val = ConfigurationHolder.getString("intresources.preferredlanguage");
80          assertEquals("en", val);
81          // A little reflection magic just to avoid dumping a test value in defaultvalues.properties file.
82          Field field = ConfigurationHolder.class.getDeclaredField("defaultValues");
83          field.setAccessible(true);
84          CompositeConfiguration defaultValues = (CompositeConfiguration) field.get(null);
85          val = ConfigurationHolder.getString("test.comma.in.defaultvalue");
86          assertNull(val);
87          File f = File.createTempFile("cesecore", "test");
88          try {
89              FileWriter fw = new FileWriter(f);
90              fw.write("test.comma.in.defaultvalue=EN,DE,FR\n");
91              fw.close();
92              defaultValues.addConfiguration(new PropertiesConfiguration(f));
93              val = ConfigurationHolder.getString("test.comma.in.defaultvalue");
94              assertEquals("EN,DE,FR", val);
95          } finally {
96              f.deleteOnExit();
97          }
98      }
99  }
