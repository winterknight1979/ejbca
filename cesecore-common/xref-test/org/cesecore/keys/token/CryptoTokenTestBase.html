101     // without error

436     // Should still be ACTIVE now, because we run activate
437     assertEquals(CryptoToken.STATUS_ACTIVE, cryptoToken.getTokenStatus());
438     cryptoToken.deleteEntry("rsatest00001");

445     assertEquals(1024, KeyTools.getKeyLength(pub));

447     // Get a key that does not exist

449       pub = cryptoToken.getPublicKey("sdfsdf77474");
450       assertTrue("Should throw", false);
451     } catch (CryptoTokenOfflineException e) {
452       assertTrue(

455               .contains(
456                   INTRES.getLocalizedMessage(
457                       "token.errornosuchkey", "sdfsdf77474")));
458     }
459     // We have not set auto activate, so the internal key storage in CryptoToken

461     cryptoToken.deactivate();
462     try {
463       priv = cryptoToken.getPrivateKey("rsatest00001");
464       assertTrue("Should throw", false);
465     } catch (CryptoTokenOfflineException e) {
466       assertTrue(
467           e.getMessage(), e.getMessage().contains("keyStore (111) == null"));

470       pub = cryptoToken.getPublicKey("rsatest00001");
471       assertTrue("Should throw", false);

473       assertTrue(
474           e.getMessage(), e.getMessage().contains("keyStore (111) == null"));

478       cryptoToken.activate("gfhf56564".toCharArray());

485     }
486     cryptoToken.activate(TOKEN_PIN.toCharArray());

488     pub = cryptoToken.getPublicKey("rsatest00001");
