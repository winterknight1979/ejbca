13  package org.cesecore.keys.token.p11;
14  
15  import static org.junit.Assert.assertFalse;
16  import static org.junit.Assert.assertTrue;
17  
18  import org.junit.Test;
19  
20  /**
21   * @version $Id: Pkcs11SlotLabelTypeTest.java 19902 2014-09-30 14:32:24Z anatom
22   *     $
23   */
24  public class Pkcs11SlotLabelTypeTest {
25      /**
26         * Test.
27         */
28    @Test
29    public void testNumberTypeValidator() {
30      Pkcs11SlotLabelType type = Pkcs11SlotLabelType.SLOT_NUMBER;
31      assertTrue("Validator did not validate a number.", type.validate("4711"));
32      assertFalse("Validator falsely validated a label", type.validate("foo"));
33      assertFalse("Validator falsely validated an index", type.validate("i7"));
34    }
35    /**
36     * Test.
37     */
38    @Test
39    public void testIndexTypeValidator() {
40      Pkcs11SlotLabelType type = Pkcs11SlotLabelType.SLOT_INDEX;
41      assertFalse("Validator falsely validated a number.", type.validate("4711"));
42      assertFalse("Validator falsely validated a label", type.validate("foo"));
43      assertTrue(
44          "Validator did not correctly validate an index", type.validate("i7"));
45    }
46    /**
47     * Test.
48     */
49    @Test
50    public void testLabelTypeValidator() {
51      Pkcs11SlotLabelType type = Pkcs11SlotLabelType.SLOT_LABEL;
52      assertTrue(
53          "Validator did not correctly validate a label", type.validate("4711"));
54      assertTrue(
55          "Validator did not correctly validate a label", type.validate("foo"));
56      assertTrue(
57          "Validator did not correctly validate a label", type.validate("i7"));
58    }
59    /**
60     * Test.
61     */
62    @Test
63    public void testLongLabelValidation() {
64      assertFalse(
65          "Label of >32 characters validated.",
66          Pkcs11SlotLabelType.SLOT_LABEL.validate(
67              "YeahbutIcantusethewordsoberbecausethatsatermfromthose"
68              + "peopleandIhavecleansedmyself"));
69    }
70  }
