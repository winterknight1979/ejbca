62    private void testGenerateUseDelete(final boolean cache) throws Exception {
63      final KeyStore keyStore =
64          KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
65      keyStore.load(null, null);
66      final CachingKeyStoreWrapper cachingKeyStoreWrapper =
67          new CachingKeyStoreWrapper(keyStore, cache);

89    private void testPersistLoad(final boolean cache) throws Exception {
90      // Create a key store with some content
91      final KeyStore keyStore =
92          KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
93      keyStore.load(null, null);
94      final CachingKeyStoreWrapper cachingKeyStoreWrapper =
95          new CachingKeyStoreWrapper(keyStore, cache);
96      testGenerate(cachingKeyStoreWrapper, ALIAS);
97      // "Persist" and load it back from storage
98      final ByteArrayOutputStream baos2 = new ByteArrayOutputStream();
99      cachingKeyStoreWrapper.store(baos2, PASSWORD);
100     final KeyStore keyStore2 =
101         KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
102     keyStore2.load(new ByteArrayInputStream(baos2.toByteArray()), PASSWORD);
103     // Test loaded key store
104     final CachingKeyStoreWrapper cachingKeyStoreWrapper2 =
105         new CachingKeyStoreWrapper(keyStore2, cache);
106     testUse(cachingKeyStoreWrapper2, ALIAS);
107     // "Persist" and load it back from storage
108     final ByteArrayOutputStream baos3 = new ByteArrayOutputStream();
109     cachingKeyStoreWrapper2.store(baos3, PASSWORD);
110     final KeyStore keyStore3 =
111         KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
112     keyStore3.load(new ByteArrayInputStream(baos3.toByteArray()), PASSWORD);
113     // Test loaded key store
114     final CachingKeyStoreWrapper cachingKeyStoreWrapper3 =
115         new CachingKeyStoreWrapper(keyStore3, cache);
116     testRemove(cachingKeyStoreWrapper3, ALIAS);
117   }

119   private void testGenerate(
120       final CachingKeyStoreWrapper cachingKeyStoreWrapper, final String alias)
121       throws Exception {
122     final int aliasCountBefore = getAliasCount(cachingKeyStoreWrapper);
123     // Generate a key pair
124     final KeyStoreTools keyStoreTools =
125         new KeyStoreTools(
126             cachingKeyStoreWrapper,
127             cachingKeyStoreWrapper.getProvider().getName());
128     keyStoreTools.generateKeyPair("secp256r1", alias);
129     // Verify that key store contains the key pair
130     Assert.assertEquals(
131         "Number of aliases should have increased by 1 after generation.",
132         aliasCountBefore + 1,
133         getAliasCount(cachingKeyStoreWrapper));
134     Assert.assertTrue(
135         "Generated key pair alias was not found.",
136         isContainsAlias(cachingKeyStoreWrapper, alias));
137     final Key key = cachingKeyStoreWrapper.getKey(alias, null);
138     Assert.assertNotNull(
139         "No private key for generated key pair could be found.", key);
140     Assert.assertTrue(key instanceof PrivateKey);
141     final Certificate certificate =
142         cachingKeyStoreWrapper.getCertificate(alias);
143     Assert.assertNotNull(
144         "No certificate for generated key pair could be found.", certificate);
145   }
146 
147   private void testUse(
148       final CachingKeyStoreWrapper cachingKeyStoreWrapper, final String alias)
149       throws Exception {
150     final Key key = cachingKeyStoreWrapper.getKey(alias, null);
151     final Certificate certificate =
152         cachingKeyStoreWrapper.getCertificate(alias);
153     // Verify that key pair from the key store is usable
154     KeyTools.testKey(
155         (PrivateKey) key,
156         certificate.getPublicKey(),
157         cachingKeyStoreWrapper.getProvider().getName());
158   }
159 
160   private void testRemove(
161       final CachingKeyStoreWrapper cachingKeyStoreWrapper, final String alias)
162       throws Exception {
163     final int aliasCountBefore = getAliasCount(cachingKeyStoreWrapper);
164     // Remove the key pair
165     final KeyStoreTools keyStoreTools =
166         new KeyStoreTools(
167             cachingKeyStoreWrapper,
168             cachingKeyStoreWrapper.getProvider().getName());
169     keyStoreTools.deleteEntry(alias);
170     Assert.assertEquals(
171         "Number of aliases should have decreased by 1 after removal.",
172         aliasCountBefore - 1,
173         getAliasCount(cachingKeyStoreWrapper));
174     Assert.assertFalse(
175         "Alias of removed key pair still exist.",
176         isContainsAlias(cachingKeyStoreWrapper, alias));
177   }

179   /**
180    * @param cachingKeyStoreWrapper wrapper
181    * @return the count of aliases the wrapped key store claims to exist
182    * @throws KeyStoreException fail
183    */
184   private int getAliasCount(final CachingKeyStoreWrapper cachingKeyStoreWrapper)
185       throws KeyStoreException {
186     final Enumeration<String> aliasEnumeration =
187         cachingKeyStoreWrapper.aliases();
188     int aliasCount = 0;
189     while (aliasEnumeration.hasMoreElements()) {
190       aliasEnumeration.nextElement();
191       aliasCount++;
192     }
193     return aliasCount;
194   }
195 
196   /**
197    * @param cachingKeyStoreWrapper wrapper
198    * @param alias alias
199    * @return true if the wrapped key store claims the alias exist
200    * @throws KeyStoreException fail
201    */
202   private boolean isContainsAlias(
203       final CachingKeyStoreWrapper cachingKeyStoreWrapper, final String alias)
204       throws KeyStoreException {
205     final Enumeration<String> aliasEnumeration =
206         cachingKeyStoreWrapper.aliases();
207     while (aliasEnumeration.hasMoreElements()) {
208       final String currentAlias = aliasEnumeration.nextElement();
209       if (alias.equals(currentAlias)) {
210         return true;
211       }
212     }
213     return false;
214   }
