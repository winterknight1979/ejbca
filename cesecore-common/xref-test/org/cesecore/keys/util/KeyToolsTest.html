165     @Test
166     public void testGenKeysRSA() throws Exception {
167         KeyPair keys = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA);
168         assertNotNull("keys must not be null", keys);
169         String b64private = new String(Base64.encode(keys.getPrivate().getEncoded()));
170         assertNotNull("b64private must not be null", b64private);
171         // log.debug(b64private);
172         X509Certificate cert = CertTools.genSelfCert("C=SE,O=Test,CN=Test", 365, null, keys.getPrivate(), keys.getPublic(),
173                 AlgorithmConstants.SIGALG_SHA1_WITH_RSA, true);
174         assertNotNull("cert must not be null", cert);
175         String b64cert = new String(Base64.encode(cert.getEncoded()));
176         assertNotNull("b64cert cannot be null", b64cert);
177         // log.debug(b64cert);
178         KeyTools.testKey(keys.getPrivate(), keys.getPublic(), BouncyCastleProvider.PROVIDER_NAME);
179         // Test that fails
180         PKCS8EncodedKeySpec pkKeySpec = new PKCS8EncodedKeySpec(keys1024bit);
181         KeyFactory keyFactory = KeyFactory.getInstance("RSA");
182         PrivateKey pk = keyFactory.generatePrivate(pkKeySpec);
183         try {
184         	KeyTools.testKey(pk, keys.getPublic(), BouncyCastleProvider.PROVIDER_NAME);
185         	assertTrue(false);
186         } catch (InvalidKeyException e) {
187         	assertEquals("Signature was not correctly verified.", e.getMessage());
188         }
189         
190         AlgorithmParameterSpec paramspec = KeyTools.getKeyGenSpec(keys.getPublic());
191         assertTrue((paramspec instanceof RSAKeyGenParameterSpec));
192         RSAKeyGenParameterSpec rsaspec = (RSAKeyGenParameterSpec)paramspec;
193         assertEquals(512, rsaspec.getKeysize());
194         
195         assertTrue(KeyTools.isPrivateKeyExtractable(keys.getPrivate()));
196         
197         ByteArrayOutputStream out = new ByteArrayOutputStream();
198         PrintStream ps = new PrintStream(out);
199         KeyTools.printPublicKeyInfo(keys.getPublic(), ps);
200         ps.close();
201         String str = out.toString();
202         assertTrue(str.contains("RSA key"));
203     }

209         KeyFactory keyFactory = KeyFactory.getInstance("RSA");

214         // If password below is more than 7 chars, strong crypto is needed

216         assertTrue("baos size must not be 0", baos.size() > 0);
217         Certificate cert1 = ks.getCertificate("Foo");
218         assertNotNull(cert1);
219         byte[] bytes = KeyTools.getSinglePemFromKeyStore(ks, "foo123".toCharArray());
220         assertNotNull(bytes);
221         String str = new String(bytes);
222         assertTrue(str.contains("-----BEGIN PRIVATE KEY-----"));
223         assertTrue(str.contains("-----BEGIN CERTIFICATE-----"));
224     }

245         KeyPair keys = KeyTools.genKeys("prime192v1", AlgorithmConstants.KEYALGORITHM_ECDSA);
246         // Verify that the keys are using maned curves, and not explicit parameters
247         PrivateKeyInfo priv2 = PrivateKeyInfo.getInstance(keys.getPrivate().getEncoded());

249         SubjectPublicKeyInfo pub2 = SubjectPublicKeyInfo.getInstance(keys.getPublic().getEncoded());
250         assertTrue("Public key is not encoded with named curves, but using explicit parameters", X962Parameters.getInstance(pub2.getAlgorithm().getParameters()).isNamedCurve());
251 
252         assertNotNull("keys must not be null", keys);
253         String b64private = new String(Base64.encode(keys.getPrivate().getEncoded()));
254         assertNotNull("b64private must not be null", b64private);
255         // log.debug(b64private);
256         X509Certificate cert = CertTools.genSelfCert("C=SE,O=Test,CN=Test", 365, null, keys.getPrivate(), keys.getPublic(),
257                 AlgorithmConstants.SIGALG_SHA256_WITH_ECDSA, true);
258         // log.debug(cert);
259         assertNotNull("cert must not be null", cert);
260         String b64cert = new String(Base64.encode(cert.getEncoded()));
261         assertNotNull("b64cert cannot be null", b64cert);
262         // log.debug(b64cert);
263         KeyTools.testKey(keys.getPrivate(), keys.getPublic(), BouncyCastleProvider.PROVIDER_NAME);
264         // Test that fails
265         KeyPair keys1 = KeyTools.genKeys("prime192v1", AlgorithmConstants.KEYALGORITHM_ECDSA);
266         try {
267         	KeyTools.testKey(keys1.getPrivate(), keys.getPublic(), BouncyCastleProvider.PROVIDER_NAME);

618         final byte[] pem = CertTools.getPEMFromPublicKey(der);

639             fail("getBytesFromPublicKeyFile on corrupt data should throw");
