283     assertTrue(str.contains("RSA key"));

289   @Test
290   public void testCreateP12() throws Exception {
291     Certificate cert =
292         CertTools.getCertfromByteArray(CERT_BYTES, Certificate.class);
293     PKCS8EncodedKeySpec pkKeySpec = new PKCS8EncodedKeySpec(KEYS_1024_BIT);
294     KeyFactory keyFactory = KeyFactory.getInstance("RSA");
295     PrivateKey pk = keyFactory.generatePrivate(pkKeySpec);
296     KeyStore ks = KeyTools.createP12("Foo", pk, cert, (X509Certificate) null);
297     assertNotNull("ks must not be null", ks);
298     ByteArrayOutputStream baos = new ByteArrayOutputStream();
299     // If password below is more than 7 chars, strong crypto is needed
300     ks.store(baos, "foo123".toCharArray());
301     assertTrue("baos size must not be 0", baos.size() > 0);
302     Certificate cert1 = ks.getCertificate("Foo");
303     assertNotNull(cert1);
304     byte[] bytes =
305         KeyTools.getSinglePemFromKeyStore(ks, "foo123".toCharArray());
306     assertNotNull(bytes);
307     String str = new String(bytes);
308     assertTrue(str.contains("-----BEGIN PRIVATE KEY-----"));
309     assertTrue(str.contains("-----BEGIN CERTIFICATE-----"));
310   }
