44  @FixMethodOrder(MethodSorters.NAME_ASCENDING)
45  public class SignWithWorkingAlgorithmTest {
46    /**
47     * This list starts on purpose with long hashing algorithms that will not be
48     * handled by a 512bit key. Since first test is the 512 bit key some
49     * algorithms must be tried before finding the working one. If the test should
50     * have started with 2048 bit key then the signing with 512 should have failed
51     * since the chosen algorithm had to long hash for this key. So in real life
52     * the list should start with the short algorithms.
53     */
54    public static final List<String> SIG_ALGS_RSA =
55        Collections.unmodifiableList(
56            Arrays.asList(
57                "someBogusName",
58                "SHA512withRSAandMGF1",
59                AlgorithmConstants.SIGALG_SHA512_WITH_RSA,
60                AlgorithmConstants.SIGALG_SHA256_WITH_RSA_AND_MGF1,
61                AlgorithmConstants.SIGALG_SHA1_WITH_RSA_AND_MGF1,
62                AlgorithmConstants.SIGALG_SHA256_WITH_RSA,
63                AlgorithmConstants.SIGALG_SHA1_WITH_RSA));
64  
65    private class SignOperation implements ISignOperation {
66  
67      SignOperation(final KeyPair kp) {
68        this.keyPair = kp;
69      }
70  
71      /** Config. */
72      private final byte[] bvOriginal = "Här är orginalet!".getBytes();
73      /** Config. */
74      private final KeyPair keyPair;
75      /** Config. */
76      private byte[] bvSignature;
77      /** Config. */
78      private String usedAlgorithm;
79      /** Config. */
80      private int nrOfCalls = 0;
81  
82      @Override
83      public void taskWithSigning(final String algorithm, final Provider provider)
84          throws TaskWithSigningException {
85        this.nrOfCalls++;
86        final Signature signature;
87        try {
88          signature = Signature.getInstance(algorithm, provider);
89          signature.initSign(this.keyPair.getPrivate());
90          signature.update(this.bvOriginal);
91          this.bvSignature = signature.sign();
92        } catch (NoSuchAlgorithmException
93            | InvalidKeyException
94            | SignatureException e) {
95          throw new TaskWithSigningException("Signing failed", e);
96        }
97        this.usedAlgorithm = algorithm;
98      }
99  
100     public boolean verifySignature() throws GeneralSecurityException {
101       final Signature signature =
102           Signature.getInstance(
103               this.usedAlgorithm, BouncyCastleProvider.PROVIDER_NAME);
104       signature.initVerify(this.keyPair.getPublic());
105       signature.update(this.bvOriginal);
106       return signature.verify(this.bvSignature);
107     }
108 
109     public String getUsedAlgorithm() {
110       return this.usedAlgorithm;
111     }
112 
113     public int getNrOfCalls() {
114       return this.nrOfCalls;
115     }
116   }
117 
118   private static KeyPair generateKeyPair(final int size)
119       throws NoSuchAlgorithmException {
120     final KeyPairGenerator kpg = KeyPairGenerator.getInstance("RSA");
121     kpg.initialize(size);
122     return kpg.generateKeyPair();
123   }
124   /** Register BC provider before starting the test. */
125   @BeforeClass
126   public static void before() {
127     if (Security.getProvider(BouncyCastleProvider.PROVIDER_NAME) != null) {
128       return;
129     }
130     Security.addProvider(new BouncyCastleProvider());
131   }
132   /**
133    * First time for BC provider. Check that 5 tries are made until
134    * SIGALG_SHA1_WITH_RSA_AND_MGF1 that is working is found. The key is too
135    * short for first algorithms.
136    *
137    * @throws NoSuchProviderException fail
138    * @throws GeneralSecurityException fail
139    * @throws TaskWithSigningException fail
140    */
141   @Test
142   public void n1BC512()
143       throws NoSuchProviderException, GeneralSecurityException,
144           TaskWithSigningException {
145     final SignOperation operation = new SignOperation(generateKeyPair(512));
146     assertTrue(
147         SignWithWorkingAlgorithm.doSignTask(
148             SIG_ALGS_RSA, BouncyCastleProvider.PROVIDER_NAME, operation));
149     assertTrue(operation.verifySignature());
150     assertEquals(5, operation.getNrOfCalls());
151     assertEquals(
152         AlgorithmConstants.SIGALG_SHA1_WITH_RSA_AND_MGF1,
153         operation.getUsedAlgorithm());
154   }
