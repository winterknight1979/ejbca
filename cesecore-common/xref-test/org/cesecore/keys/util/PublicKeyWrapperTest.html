13  package org.cesecore.keys.util;
14  
15  import static org.junit.Assert.assertEquals;
16  
17  import java.io.ByteArrayInputStream;
18  import java.io.ByteArrayOutputStream;
19  import java.io.IOException;
20  import java.io.ObjectInputStream;
21  import java.io.ObjectOutputStream;
22  import java.security.InvalidAlgorithmParameterException;
23  import java.security.PublicKey;
24  import org.cesecore.certificates.util.AlgorithmConstants;
25  import org.cesecore.util.CryptoProviderTools;
26  import org.junit.BeforeClass;
27  import org.junit.Test;
28  
29  /**
30   * @version $Id: PublicKeyWrapperTest.java 20836 2015-03-04 14:32:00Z anatom $
31   */
32  public class PublicKeyWrapperTest {
33  /** Key. */
34    private static PublicKey testKey;
35  /**
36   * setup.
37   * @throws InvalidAlgorithmParameterException fail
38   */
39    @BeforeClass
40    public static void beforeClass() throws InvalidAlgorithmParameterException {
41      CryptoProviderTools.installBCProviderIfNotAvailable();
42      testKey =
43          KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA)
44              .getPublic();
45    }
46    /**
47     * Test.
48     */
49    @Test
50    public void testGetPublicKey() {
51      PublicKeyWrapper testWrapper = new PublicKeyWrapper(testKey);
52      assertEquals(
53          "Decoded PublicKey was not identical to encoded.",
54          testKey,
55          testWrapper.getPublicKey());
56    }
57    /**
58     * Test.
59   * @throws IOException fail
60   * @throws ClassNotFoundException fail
61     */
62    @Test
63    public void testPublicKeySerialization()
64        throws IOException, ClassNotFoundException {
65      ByteArrayOutputStream baos = new ByteArrayOutputStream();
66      ObjectOutputStream oos = new ObjectOutputStream(baos);
67      oos.writeObject(testKey);
68      oos.close();
69      byte[] bytes = baos.toByteArray();
70      baos.close();
71      ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
72      ObjectInputStream ois = new ObjectInputStream(bais);
73      Object obj = ois.readObject();
74      PublicKey pk = (PublicKey) obj;
75      final String str = pk.getClass().getName();
76      assertEquals(
77          "Deserialized class should be a BC PublicKey",
78          "org.bouncycastle.jcajce.provider.asymmetric.rsa.BCRSAPublicKey",
79          str);
80      assertEquals(
81          "Deserialized PublicKey was not identical to encoded.", testKey, pk);
82    }
83  }
