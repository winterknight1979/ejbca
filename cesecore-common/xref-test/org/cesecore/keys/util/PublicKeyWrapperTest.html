13  package org.cesecore.keys.util;
14  
15  import static org.junit.Assert.assertEquals;
16  
17  import java.io.ByteArrayInputStream;
18  import java.io.ByteArrayOutputStream;
19  import java.io.IOException;
20  import java.io.ObjectInputStream;
21  import java.io.ObjectOutputStream;
22  import java.security.InvalidAlgorithmParameterException;
23  import java.security.PublicKey;
24  
25  import org.cesecore.certificates.util.AlgorithmConstants;
26  import org.cesecore.util.CryptoProviderTools;
27  import org.junit.BeforeClass;
28  import org.junit.Test;
29  
30  /**
31   * @version $Id: PublicKeyWrapperTest.java 20836 2015-03-04 14:32:00Z anatom $
32   *
33   */
34  public class PublicKeyWrapperTest {
35  
36      public static PublicKey testKey;
37      @BeforeClass
38      public static void beforeClass() throws InvalidAlgorithmParameterException {
39          CryptoProviderTools.installBCProviderIfNotAvailable();
40          testKey = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA).getPublic();
41      }
42      
43      @Test
44      public void testGetPublicKey() {
45          PublicKeyWrapper testWrapper = new PublicKeyWrapper(testKey);
46          assertEquals("Decoded PublicKey was not identical to encoded.", testKey, testWrapper.getPublicKey());
47      }
48  
49      @Test
50      public void testPublicKeySerialization() throws IOException, ClassNotFoundException {
51          ByteArrayOutputStream baos = new ByteArrayOutputStream();
52          ObjectOutputStream oos = new ObjectOutputStream(baos);
53          oos.writeObject(testKey);
54          oos.close();
55          byte[] bytes = baos.toByteArray();
56          baos.close();
57          ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
58          ObjectInputStream ois = new ObjectInputStream(bais);
59          Object obj = ois.readObject();
60          PublicKey pk = (PublicKey)obj;
61          final String str = pk.getClass().getName();
62          assertEquals("Deserialized class should be a BC PublicKey", "org.bouncycastle.jcajce.provider.asymmetric.rsa.BCRSAPublicKey", str);
63          assertEquals("Deserialized PublicKey was not identical to encoded.", testKey, pk);
64      }
65  
66  }
