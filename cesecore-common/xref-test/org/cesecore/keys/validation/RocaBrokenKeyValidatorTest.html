54      /**
55       * Run a test on a known bad key (from https://github.com/crocs-muni/roca/blob/master/roca/tests/data/csr05.pem)
56       * @throws NoSuchAlgorithmException fail
57       * @throws InvalidKeySpecException fail
58       */
59      @Test
60      public void testKnownBadKey() throws NoSuchAlgorithmException, InvalidKeySpecException {        
61          X509EncodedKeySpec spec = new X509EncodedKeySpec(
62                  KeyTools.getBytesFromPEM(KNOWN_VULNERABLE_KEY, CertTools.BEGIN_PUBLIC_KEY, CertTools.END_PUBLIC_KEY));
63          KeyFactory kf = KeyFactory.getInstance("RSA");
64          RSAPublicKey knownBadPublicKey = (RSAPublicKey) kf.generatePublic(spec);
65          assertTrue("Known vulnerable public key was not caught", RocaBrokenKey.isAffected(knownBadPublicKey.getModulus()));
66      }
67      
68      /**
69       * Run a test on a known good key 
70       * @throws NoSuchAlgorithmException fail
71       * @throws InvalidKeySpecException fail
72       */
73      @Test
74      public void testKnownGoodKey() throws NoSuchAlgorithmException, InvalidKeySpecException {        
75          X509EncodedKeySpec spec = new X509EncodedKeySpec(
76                  KeyTools.getBytesFromPEM(KNOWN_GOOD_KEY, CertTools.BEGIN_PUBLIC_KEY, CertTools.END_PUBLIC_KEY));
77          KeyFactory kf = KeyFactory.getInstance("RSA");
78          RSAPublicKey knownBadPublicKey = (RSAPublicKey) kf.generatePublic(spec);
79          assertFalse("Known good public key was falsely caught", RocaBrokenKey.isAffected(knownBadPublicKey.getModulus()));
80      }
