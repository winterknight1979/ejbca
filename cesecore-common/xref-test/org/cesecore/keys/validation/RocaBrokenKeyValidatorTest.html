57    /**
58     * Run a test on a known bad key (from
59     * https://github.com/crocs-muni/roca/blob/master/roca/tests/data/csr05.pem).
60     *
61     * @throws NoSuchAlgorithmException fail
62     * @throws InvalidKeySpecException fail
63     */
64    @Test
65    public void testKnownBadKey()
66        throws NoSuchAlgorithmException, InvalidKeySpecException {
67      X509EncodedKeySpec spec =
68          new X509EncodedKeySpec(
69              KeyTools.getBytesFromPEM(
70                  knownVulnerableKey,
71                  CertTools.BEGIN_PUBLIC_KEY,
72                  CertTools.END_PUBLIC_KEY));
73      KeyFactory kf = KeyFactory.getInstance("RSA");
74      RSAPublicKey knownBadPublicKey = (RSAPublicKey) kf.generatePublic(spec);
75      assertTrue(
76          "Known vulnerable public key was not caught",
77          RocaBrokenKey.isAffected(knownBadPublicKey.getModulus()));
78    }
79  
80    /**
81     * Run a test on a known good key.
82     *
83     * @throws NoSuchAlgorithmException fail
84     * @throws InvalidKeySpecException fail
85     */
86    @Test
87    public void testKnownGoodKey()
88        throws NoSuchAlgorithmException, InvalidKeySpecException {
89      X509EncodedKeySpec spec =
90          new X509EncodedKeySpec(
91              KeyTools.getBytesFromPEM(
92                  knownGoodKey,
93                  CertTools.BEGIN_PUBLIC_KEY,
94                  CertTools.END_PUBLIC_KEY));
95      KeyFactory kf = KeyFactory.getInstance("RSA");
96      RSAPublicKey knownBadPublicKey = (RSAPublicKey) kf.generatePublic(spec);
97      assertFalse(
98          "Known good public key was falsely caught",
99          RocaBrokenKey.isAffected(knownBadPublicKey.getModulus()));
100   }
