82          log.trace(">testNoNegativeNumbers()");

86          String numStringPos = "2";
87          String numStringNeg = "-4";

91          keyValidator.setPublicKeyExponentMin(exponent);

96          exponent = new BigInteger(numStringPos);
97          keyValidator.setPublicKeyExponentMax(exponent);
98          exponent = new BigInteger(numStringNeg);
99          keyValidator.setPublicKeyExponentMax(exponent);
100         Assert.assertEquals("Test should not fail, validator field can not be set to a negative value ",
101                 keyValidator.getPublicKeyExponentMax(), new BigInteger(numStringPos)  );
102         // Test that a negative number can not be set for setPublicKeyModulusMin
103         exponent = new BigInteger(numStringPos);
104         keyValidator.setPublicKeyModulusMin(exponent);
105         exponent = new BigInteger(numStringNeg);
106         keyValidator.setPublicKeyModulusMin(exponent);
107         Assert.assertEquals("Test should not fail, validator field can not be set to a negative value ",
108                 keyValidator.getPublicKeyModulusMin(),  new BigInteger(numStringPos));
109         // Test that a negative number can not be set for setPublicKeyModulusMax
110         exponent = new BigInteger(numStringPos);
111         keyValidator.setPublicKeyModulusMax(exponent);
112         exponent = new BigInteger(numStringNeg);
113         keyValidator.setPublicKeyModulusMax(exponent);
114         Assert.assertEquals("Test should not fail, validator field can not be set to a negative value ",
115                 keyValidator.getPublicKeyModulusMax(), new BigInteger(numStringPos) );
116         // Test that a negative number can not be set for setPublicKeyModulusMinFactor
117         keyValidator.setPublicKeyModulusMinFactor(2);
118         keyValidator.setPublicKeyModulusMinFactor(-4);
119         Assert.assertEquals("Test should not fail, validator field can not be set to a negative value ",

262     @Test
263     public void test03RsaParameterValidations() throws Exception {
264         log.trace(">test03RsaParameterValidations()");
265 
266         final KeyFactory keyFactory = KeyFactory.getInstance(AlgorithmConstants.KEYALGORITHM_RSA, BouncyCastleProvider.PROVIDER_NAME);
267 
268         // A-1: Test RSA key validation OK with default settings except key size.
269         BigInteger modulus = BigInteger.valueOf(15);
270         BigInteger exponent = BigInteger.valueOf(3);
271         PublicKey publicKey = keyFactory.generatePublic(new RSAPublicKeySpec(modulus, exponent));
272         RsaKeyValidator keyValidator = (RsaKeyValidator) ValidatorTestUtil.createKeyValidator(RsaKeyValidator.class,
273                 "rsa-parameter-validation-test-1", "Description", null, -1, null, -1, -1, new Integer[] {});
274         keyValidator.setSettingsTemplate(KeyValidatorSettingsTemplate.USE_CUSTOM_SETTINGS.getOption());
275         // Set custom bit length.
276         List<String> bitLengths = new ArrayList<String>();
277         bitLengths.add(Integer.toString(modulus.bitLength()));
278         keyValidator.setBitLengths(bitLengths);
279         List<String> messages = keyValidator.validate(publicKey, null);
280         log.trace("Key validation error messages: " + messages);
281         Assert.assertTrue("Key validation should have been successful.", messages.size() == 0);
282 
283         // A-2: Test RSA key validation failed RSA parameter bounds with even parameters.
284         modulus = BigInteger.valueOf(16);
285         exponent = BigInteger.valueOf(4);
286         publicKey = keyFactory.generatePublic(new RSAPublicKeySpec(modulus, exponent));
287         keyValidator.setPublicKeyExponentMin(exponent.add(BigInteger.ONE));
288         keyValidator.setPublicKeyExponentOnlyAllowOdd(true);
289         keyValidator.setPublicKeyModulusMin(modulus.add(BigInteger.ONE));
290         keyValidator.setPublicKeyModulusMax(modulus.subtract(BigInteger.ONE));
291         keyValidator.setPublicKeyModulusOnlyAllowOdd(true);
292         bitLengths = new ArrayList<String>();
293         bitLengths.add(Integer.toString(modulus.bitLength()));
294         keyValidator.setBitLengths(bitLengths);
295         messages = keyValidator.validate(publicKey, null);
296         log.trace("Key validation error messages: " + messages);
297         Assert.assertTrue("Key valildation should have failed because of even RSA parameter and outside parameter bounds.",
298                 messages.size() == 4);
299         Assert.assertEquals("RSA parameters bounds failure message isn't right",
300                 "Invalid: RSA public key exponent is odd.", messages.get(0));
301         Assert.assertEquals("RSA parameters bounds failure message isn't right",
302                 "Invalid: RSA public key exponent is smaller than 5", messages.get(1));
303         Assert.assertEquals("RSA parameters bounds failure message isn't right",
304                 "Invalid: RSA public key modulus is odd.", messages.get(2));
305         Assert.assertEquals("RSA parameters bounds failure message isn't right",
306                 "Invalid: RSA public key modulus is smaller than 17", messages.get(3));
307         // Need to set min to null before lowering max
308         keyValidator.setPublicKeyExponentMin(null);
309         keyValidator.setPublicKeyExponentMax(exponent.subtract(BigInteger.ONE));
310         keyValidator.setPublicKeyModulusMin(null);
311         keyValidator.setPublicKeyModulusMax(modulus.subtract(BigInteger.ONE));
312         keyValidator.validate(publicKey, null);
313         Assert.assertEquals("RSA parameters bounds failure message isn't right",
314                 "Invalid: RSA public key exponent is greater than 3", keyValidator.validate(publicKey, null).get(1));
315         Assert.assertEquals("RSA parameters bounds failure message isn't right",
316                 "Invalid: RSA public key modulus is greater than 15", keyValidator.validate(publicKey, null).get(3));
317 
318         // A-3: Test RSA key validation failed because of modulus factor restriction.
319         modulus = BigInteger.valueOf(25);
320         exponent = BigInteger.valueOf(3);
321         publicKey = keyFactory.generatePublic(new RSAPublicKeySpec(modulus, exponent));
322         keyValidator.setPublicKeyExponentMin(exponent);
323         keyValidator.setPublicKeyExponentMax(exponent);
324         keyValidator.setPublicKeyExponentOnlyAllowOdd(true);
325         keyValidator.setPublicKeyModulusMin(modulus);
326         keyValidator.setPublicKeyModulusMax(modulus);
327         keyValidator.setPublicKeyModulusOnlyAllowOdd(true);
328         keyValidator.setPublicKeyModulusMinFactor(6); // smallest factor = 5
329         messages = keyValidator.validate(publicKey, null);
330         log.trace("Key validation error messages: " + messages);
331         Assert.assertTrue("Key valildation should have failed because of smallest factor restriction for modulus.",
332                 messages.size() == 1);
333         Assert.assertEquals("smallest factor failure message isn't right",
334                 "Invalid: RSA public key modulus smallest factor is less than 6", messages.get(0));
335 
336         // A-4: Test RSA key validation failed because of modulus power of prime restriction.
337         keyValidator.setPublicKeyModulusMinFactor(5); // smallest factor = 5
338         keyValidator.setPublicKeyModulusDontAllowPowerOfPrime(true);
339         messages = keyValidator.validate(publicKey, null);
340         log.trace("Key validation error messages: " + messages);
341         Assert.assertTrue("Key valildation should have failed because of power of prime restriction for modulus.",
342                 messages.size() == 1);
343         Assert.assertEquals("Power of prime failure message isn't right.",
344                 "Invalid: RSA public key modulus is not allowed to be the power of a prime.", messages.get(0));
345 
346         log.trace("<test03RsaParameterValidations()");
347     }

349     /** Tests public key validation for the ROCA vulnerable key generation. CVE-2017-15361
350      * @throws CertificateParsingException fail
351      * @throws InstantiationException fail
352      * @throws IllegalAccessException fail
353      * @throws ValidatorNotApplicableException fail 
354      * @throws ValidationException fail
355      * @throws SecurityException fail
356      * @throws NoSuchMethodException fail
357      * @throws InvocationTargetException fail
358      * @throws IllegalArgumentException fail
359      */
360     @Test
361     public void testRocaWeakKeys() throws CertificateParsingException, InstantiationException, IllegalAccessException, ValidatorNotApplicableException, ValidationException, IllegalArgumentException, InvocationTargetException, NoSuchMethodException, SecurityException {
362         log.trace(">testRocaWeakKeys()");
363         X509Certificate noroca = CertTools.getCertfromByteArray(noRocaCert, X509Certificate.class);
364         X509Certificate roca = CertTools.getCertfromByteArray(rocaCert, X509Certificate.class);
365 
366         RsaKeyValidator keyValidator = (RsaKeyValidator) ValidatorTestUtil.createKeyValidator(RsaKeyValidator.class,
367                 "rsa-parameter-validation-test-1", "Description", null, -1, null, -1, -1, new Integer[] {});
368         keyValidator.setSettingsTemplate(KeyValidatorSettingsTemplate.USE_CUSTOM_SETTINGS.getOption());
369         keyValidator.setPublicKeyModulusDontAllowRocaWeakKeys(false);
370 
371         // With disabled validation of ROCA, everything should pass
372         List<String> bitLengths = new ArrayList<String>();
373         bitLengths.add("1024");
374         bitLengths.add("2048");
375         bitLengths.add("2050"); // The positive sample ROCA cert is 2050 bits
376         keyValidator.setBitLengths(bitLengths);
377         List<String> messages = keyValidator.validate(noroca.getPublicKey(), null);
378         log.trace("Key validation error messages: " + messages);
379         assertEquals("Key validation should have been successful: "+messages, 0, messages.size());
380         messages = keyValidator.validate(roca.getPublicKey(), null);
381         log.trace("Key validation error messages: " + messages);
382         assertEquals("Key validation should have been successful: "+messages, 0, messages.size());
383 
384         // Check for ROCA weak keys
385         keyValidator.setPublicKeyModulusDontAllowRocaWeakKeys(true);
386         messages = keyValidator.validate(noroca.getPublicKey(), null);
387         log.trace("Key validation error messages: " + messages);
388         assertEquals("Key validation should have been successful: "+messages, 0, messages.size());
389         messages = keyValidator.validate(roca.getPublicKey(), null);
390         log.trace("Key validation error messages: " + messages);
391         assertEquals("Key validation should have failes", 1, messages.size());
392         assertEquals("It should have been a ROCA failure.",
393                 "Invalid: RSA public key modulus is a weak key according to CVE-2017-15361.", messages.get(0));
394 
395         log.trace("<testRocaWeakKeys()");
396 
397     }

408                 log.trace(">ms spent for " + size + " bit odd modulus with factor " + factor + ": " + (System.currentTimeMillis() - time));
409             }
410         }
411 
412         log.trace("<profileHasSmallerFactor()");
413     }
