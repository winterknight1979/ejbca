52    /** Test vasic ops.
53     *
54     * @throws InterruptedException fail
55     * @throws AuthenticationFailedException fail
56     */
57    @Test
58    public void testBasicOperations()
59        throws InterruptedException, AuthenticationFailedException {
60      LOG.trace(">testBasicOperations");
61      AuthorizationCache.INSTANCE.reset();
62      final AtomicInteger updateNumber = new AtomicInteger(0);
63      final AtomicLong keepUnusedEntriesFor = new AtomicLong(3600000L);
64      final HashMap<String, Boolean> accessRules1 = new HashMap<>();
65      accessRules1.put("/rule1", Boolean.TRUE);
66      final AuthenticationToken at1 =
67          new AlwaysAllowLocalAuthenticationToken(
68              new UsernamePrincipal("AuthorizationCacheTest1"));
69      final AuthorizationCacheCallback callback =
70          new AuthorizationCacheCallback() {
71            @Override
72            public AuthorizationResult loadAuthorization(
73                final AuthenticationToken authenticationToken)
74                throws AuthenticationFailedException {
75              if (at1 == authenticationToken) {
76                return new AuthorizationResult(accessRules1, updateNumber.get());
77              }
78              return new AuthorizationResult(null, updateNumber.get());
79            }
80  
81            @Override
82            public long getKeepUnusedEntriesFor() {
83              return keepUnusedEntriesFor.get();
84            }
85  
86            @Override
87            public void subscribeToAuthorizationCacheReload(
88                final AuthorizationCacheReloadListener
89                    authorizationCacheReloadListener) {
90              // Not needed for this test
91            }
92          };
93      // The callback should only populate the cache for one of the tokens
94      assertNotNull(AuthorizationCache.INSTANCE.get(at1, callback));
95      assertEquals(1, AuthorizationCache.INSTANCE.get(at1, callback).size());
96      assertNotNull(AuthorizationCache.INSTANCE.get(null, callback));
97      assertEquals(0, AuthorizationCache.INSTANCE.get(null, callback).size());
98      assertNotNull(AuthorizationCache.INSTANCE.get(at1, null));
99      assertEquals(0, AuthorizationCache.INSTANCE.get(at1, null).size());
100     // From a caching perspective all AlwaysAllowLocalAuthenticationToken are
101     // the same and the cached object should be returned
102     final AuthenticationToken at2 =
103         new AlwaysAllowLocalAuthenticationToken(
104             new UsernamePrincipal("AuthorizationCacheTest2"));
105     final AuthorizationCacheCallback callbackEmpty =
106         new AuthorizationCacheCallback() {
107           @Override
108           public AuthorizationResult loadAuthorization(
109               final AuthenticationToken authenticationToken)
110               throws AuthenticationFailedException {
111             return new AuthorizationResult(null, updateNumber.get());
112           }
113 
114           @Override
115           public long getKeepUnusedEntriesFor() {
116             return keepUnusedEntriesFor.get();
117           }
118 
119           @Override
120           public void subscribeToAuthorizationCacheReload(
121               final AuthorizationCacheReloadListener
122                   authorizationCacheReloadListener) {
123             // Not needed for this test
124           }
125         };
126     assertNotNull(AuthorizationCache.INSTANCE.get(at2, callbackEmpty));
127     assertEquals(1, AuthorizationCache.INSTANCE.get(at2, callbackEmpty).size());
128     // Without increasing the updateNumer, nothing should happen when the cache
129     // is refreshed
130     accessRules1.put("/rule2", Boolean.TRUE);
131     AuthorizationCache.INSTANCE.refresh(callback, updateNumber.get());
132     assertNotNull(AuthorizationCache.INSTANCE.get(at1, callback));
133     assertEquals(1, AuthorizationCache.INSTANCE.get(at1, callback).size());
134     updateNumber.incrementAndGet();
135     AuthorizationCache.INSTANCE.refresh(callback, updateNumber.get());
136     assertNotNull(AuthorizationCache.INSTANCE.get(at1, callback));
137     assertEquals(2, AuthorizationCache.INSTANCE.get(at1, callback).size());
138     // When the cache is refreshed, even if the updateNumber is not changed,
139     // unused entries will be removed
140     accessRules1.clear();
141     Thread.sleep(100L);
142     AuthorizationCache.INSTANCE.refresh(callback, updateNumber.get());
143     assertEquals(2, AuthorizationCache.INSTANCE.get(at1, callback).size());
144     keepUnusedEntriesFor.set(0L);
145     Thread.sleep(100L);
146     AuthorizationCache.INSTANCE.refresh(callback, updateNumber.get());
147     assertEquals(0, AuthorizationCache.INSTANCE.get(at1, callback).size());
148     LOG.trace("<testBasicOperations");
149   }

151   /**
152    * Test already cached entries are reloaded if there is an update to the
153    * authorization system.
154    *
155    * @throws InterruptedException fail
156    * @throws AuthenticationFailedException fail
157    */
158   @Test
159   public void testSubscribeToAuthorizationCacheReload()
160       throws InterruptedException, AuthenticationFailedException {
161     LOG.trace(">testSubscribeToAuthorizationCacheReload");
162     AuthorizationCache.INSTANCE.reset();
163     final AtomicInteger updateNumber = new AtomicInteger(0);
164     final AtomicLong keepUnusedEntriesFor = new AtomicLong(3600000L);
165     final HashMap<String, Boolean> accessRules1 = new HashMap<>();
166     accessRules1.put("/rule1", Boolean.TRUE);
167     final AuthenticationToken at1 =
168         new AlwaysAllowLocalAuthenticationToken(
169             new UsernamePrincipal("AuthorizationCacheTest1"));
170     final Set<AuthorizationCacheReloadListener>
171         authorizationCacheReloadListeners = new HashSet<>();
172     final AuthorizationCacheCallback callback =
173         new AuthorizationCacheCallback() {
174           @Override
175           public AuthorizationResult loadAuthorization(
176               final AuthenticationToken authenticationToken)
177               throws AuthenticationFailedException {
178             if (at1 == authenticationToken) {
179               return new AuthorizationResult(accessRules1, updateNumber.get());
180             }
181             return new AuthorizationResult(null, updateNumber.get());
182           }
183 
184           @Override
185           public long getKeepUnusedEntriesFor() {
186             return keepUnusedEntriesFor.get();
187           }
188 
189           @Override
190           public void subscribeToAuthorizationCacheReload(
191               final AuthorizationCacheReloadListener
192                   authorizationCacheReloadListener) {
193             authorizationCacheReloadListeners.add(
194                 authorizationCacheReloadListener);
195           }
196         };
197     // The callback should only populate the cache for one of the tokens
198     assertNotNull(AuthorizationCache.INSTANCE.get(at1, callback));
199     assertEquals(1, AuthorizationCache.INSTANCE.get(at1, callback).size());
200     assertNotNull(AuthorizationCache.INSTANCE.get(null, callback));
201     assertEquals(0, AuthorizationCache.INSTANCE.get(null, callback).size());
202     assertNotNull(AuthorizationCache.INSTANCE.get(at1, null));
203     assertEquals(0, AuthorizationCache.INSTANCE.get(at1, null).size());
204     assertEquals(
205         "Cache never subscribed to updates.",
206         1,
207         authorizationCacheReloadListeners.size());
208     // Update the underlying access rules for this token
209     final HashMap<String, Boolean> accessRules2 = new HashMap<>(accessRules1);
210     accessRules2.put("/rule2", Boolean.TRUE);
211     updateNumber.incrementAndGet();
212     final AuthorizationCacheCallback callback2 =
213         new AuthorizationCacheCallback() {
214           @Override
215           public AuthorizationResult loadAuthorization(
216               final AuthenticationToken authenticationToken)
217               throws AuthenticationFailedException {
218             if (at1 == authenticationToken) {
219               return new AuthorizationResult(accessRules2, updateNumber.get());
220             }
221             return new AuthorizationResult(null, updateNumber.get());
222           }
223 
224           @Override
225           public long getKeepUnusedEntriesFor() {
226             return keepUnusedEntriesFor.get();
227           }
228 
229           @Override
230           public void subscribeToAuthorizationCacheReload(
231               final AuthorizationCacheReloadListener
232                   authorizationCacheReloadListener) {
233             fail("This should only be invoked once on first from the cache.");
234           }
235         };
236     // Before AuthorizationCacheReload notification or cache refresh it the
237     // change should not be detected
238     assertEquals(1, AuthorizationCache.INSTANCE.get(at1, callback2).size());
239     // After AuthorizationCacheReload notification it should be detected
240     authorizationCacheReloadListeners
241         .iterator()
242         .next()
243         .onReload(new AuthorizationCacheReload(updateNumber.get()));
244     assertEquals(2, AuthorizationCache.INSTANCE.get(at1, callback2).size());
245     LOG.trace("<testSubscribeToAuthorizationCacheReload");
246   }

342   /**
343    * Verify that only one of many calling threads for a cache entry will do the
344    * actual database lookup.
345    *
346    * @throws InterruptedException fail
347    */
348   @Test
349   public void testConcurrentRead() throws InterruptedException {
350     LOG.trace(">testConcurrentRead");
351     AuthorizationCache.INSTANCE.reset();
352     final AtomicInteger updateNumber = new AtomicInteger(0);
353     final AtomicLong keepUnusedEntriesFor = new AtomicLong(3600000L);
354     final HashMap<String, Boolean> accessRules1 = new HashMap<>();
355     accessRules1.put("/rule1", Boolean.TRUE);
356     final AuthenticationToken at1 =
357         new AlwaysAllowLocalAuthenticationToken(
358             new UsernamePrincipal("AuthorizationCacheTest.testConcurrentRead"));
359     final int threadCount = 10;
360     final CountDownLatch countDownLatchThreadsStarted =
361         new CountDownLatch(threadCount + 1);
362     final CountDownLatch readTrigger = new CountDownLatch(1);
363     final AtomicInteger loadInvocations = new AtomicInteger(0);
364     final AuthorizationCacheCallback callback =
365         new AuthorizationCacheCallback() {
366           @Override
367           public AuthorizationResult loadAuthorization(
368               final AuthenticationToken authenticationToken)
369               throws AuthenticationFailedException {
370             loadInvocations.incrementAndGet();
371             if (at1 == authenticationToken) {
372               countDownLatchThreadsStarted.countDown();
373               try {
374                 // Wait for trigger until proceeding
375                 readTrigger.await();
376               } catch (InterruptedException e) {
377                 Thread.currentThread().interrupt();
378               }
379               return new AuthorizationResult(accessRules1, updateNumber.get());
380             }
381             return new AuthorizationResult(null, updateNumber.get());
382           }
383 
384           @Override
385           public long getKeepUnusedEntriesFor() {
386             return keepUnusedEntriesFor.get();
387           }
388 
389           @Override
390           public void subscribeToAuthorizationCacheReload(
391               final AuthorizationCacheReloadListener
392                   authorizationCacheReloadListener) {
393             // Not needed for this test
394           }
395         };
396     final List<CacheReaderThread> cacheReaderThreads = new ArrayList<>();
397     for (int i = 0; i < threadCount; i++) {
398       cacheReaderThreads.add(
399           new CacheReaderThread(countDownLatchThreadsStarted, at1, callback));
400     }
401     for (final CacheReaderThread cacheReaderThread : cacheReaderThreads) {
402       cacheReaderThread.start();
403     }
404     countDownLatchThreadsStarted.await();
405     // Before the latch is released, none of the treads should have any result
406     for (final CacheReaderThread cacheReaderThread : cacheReaderThreads) {
407       assertNull(cacheReaderThread.result);
408     }
409     // Only a single thread should have entered the callback section where the
410     // access rules are loaded
411     assertEquals(1, loadInvocations.get());
412     // Release the hounds!
413     readTrigger.countDown();
414     for (final CacheReaderThread cacheReaderThread : cacheReaderThreads) {
415       cacheReaderThread.join();
416     }
417     // The EXACT same object should have been returned by all threads
418     HashMap<String, Boolean> lastResult = null;
419     for (final CacheReaderThread cacheReaderThread : cacheReaderThreads) {
420       assertNotNull(cacheReaderThread.result);
421       assertTrue(lastResult == null || lastResult == cacheReaderThread.result);
422       lastResult = cacheReaderThread.result;
423     }
424     LOG.trace("<testConcurrentRead");
425   }

427   /** Helper class for retrieving a cache entry in a background thread. */
428   private class CacheReaderThread extends Thread {
429     /** Latch. */
430     private final CountDownLatch countDownLatch;
431     /** Token. */
432     private final AuthenticationToken authenticationToken;
433     /** Callback. */
434     private final AuthorizationCacheCallback authorizationCacheCallback;
435     /** Map. */
436     private HashMap<String, Boolean> result = null;
437 
438     CacheReaderThread(
439         final CountDownLatch aCountDownLatch,
440         final AuthenticationToken anAuthenticationToken,
441         final AuthorizationCacheCallback anAuthorizationCacheCallback) {
442       this.countDownLatch = aCountDownLatch;
443       this.authenticationToken = anAuthenticationToken;
444       this.authorizationCacheCallback = anAuthorizationCacheCallback;
445     }
446 
447     @Override
448     public void run() {
449       countDownLatch.countDown();
450       try {
451         result =
452             AuthorizationCache.INSTANCE.get(
453                 authenticationToken, authorizationCacheCallback);
454       } catch (AuthenticationFailedException e) {
455         LOG.debug(e.getMessage());
456       }
457     }
458   }
