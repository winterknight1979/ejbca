13  package org.cesecore.authorization.user;
14  
15  import static org.junit.Assert.assertEquals;
16  
17  import org.junit.Test;
18  
19  /**
20   * Unit tests for the AccessMatchType enums.
21   *
22   * @version $Id: AccessMatchTypeTest.java 26812 2017-10-15 03:30:26Z anatom $
23   */
24  public class AccessMatchTypeTest {
25  
26      /** Match test.
27       *
28       * @throws SecurityException fail
29       */
30    @Test
31    public void testAccessMatchEnum() throws SecurityException {
32      AccessMatchType type0 = AccessMatchType.matchFromDatabase(0);
33      AccessMatchType type1000 = AccessMatchType.matchFromDatabase(1000);
34      AccessMatchType type1001 = AccessMatchType.matchFromDatabase(1001);
35      AccessMatchType type1002 = AccessMatchType.matchFromDatabase(1002);
36      AccessMatchType type1003 = AccessMatchType.matchFromDatabase(1003);
37      AccessMatchType type1999 = AccessMatchType.matchFromDatabase(1999);
38      AccessMatchType type2000 = AccessMatchType.matchFromDatabase(2000);
39      AccessMatchType type2001 = AccessMatchType.matchFromDatabase(2001);
40      AccessMatchType type2002 = AccessMatchType.matchFromDatabase(2002);
41      AccessMatchType type2003 = AccessMatchType.matchFromDatabase(2003);
42      AccessMatchType type2004 = AccessMatchType.matchFromDatabase(2004);
43      AccessMatchType type2005 = AccessMatchType.matchFromDatabase(2005);
44      assertEquals("name is incorrect", "TYPE_UNUSED", type0.name());
45      assertEquals("name is incorrect", "TYPE_EQUALCASE", type1000.name());
46      assertEquals("name is incorrect", "TYPE_EQUALCASEINS", type1001.name());
47      // Types before the role rewrite in EJBCA 6.9, must be kept for upgrade
48      // reasons,
49      // these types are bound to be hanging around in the database
50      assertEquals("name is incorrect", "TYPE_NOT_EQUALCASE", type1002.name());
51      assertEquals("name is incorrect", "TYPE_NOT_EQUALCASEINS", type1003.name());
52      assertEquals("name is incorrect", "TYPE_NONE", type1999.name());
53      // Types that were used earlier than EJBCA 4, must also be kept for upgrade
54      // reasons,
55      // since we support 100% uptime upgrades, we can not NPE if these are in the
56      // database
57      assertEquals(
58          "name is incorrect", "SPECIALADMIN_PUBLICWEBUSER", type2000.name());
59      assertEquals(
60          "name is incorrect",
61          "SPECIALADMIN_CACOMMANDLINEADMIN",
62          type2001.name());
63      assertEquals("name is incorrect", "SPECIALADMIN_RAADMIN", type2002.name());
64      assertEquals(
65          "name is incorrect",
66          "SPECIALADMIN_BATCHCOMMANDLINEADMIN",
67          type2003.name());
68      assertEquals(
69          "name is incorrect", "SPECIALADMIN_INTERNALUSER", type2004.name());
70      assertEquals("name is incorrect", "SPECIALADMIN_NOUSER", type2005.name());
71    }
72  }
