17  public class UpgradeableDataHashMapTest {
18      
19      static final Logger log = Logger.getLogger(UpgradeableDataHashMapTest.class);
20      
21      /**
22       * Test if UpgradeableDataHashMap is vulnerable to CVE-2010-4476 through
23       * the XML Serialization we use for storing UpgradeableDataHashMap.
24       * 
25       * When "2.2250738585072012e-308" is converted to a float, the code toggles
26       * between two values causing the the Thread to hang.
27       * 
28       * UpgradeableDataHashMap.VERSION is normally stored as a Float.
29       */
30      @Test
31      public void testCVE_2010_4476() {
32          final String XML_W_BADFLOAT = "<java version=\"1.6.0_21\" class=\"java.beans.XMLDecoder\">"
33              + "<object class=\"java.util.HashMap\"><void method=\"put\">"
34              + "<string>version</string><float>2.2250738585072012e-308</float>"
35              + "</void></object></java>";
36          final String XML_W_BADDERFLOAT = "<java version=\"1.6.0_21\" class=\"java.beans.XMLDecoder\">"
37              + "<object class=\"java.util.HashMap\"><void method=\"put\">"
38              + "<string>version</string><double>2.2250738585072012e-308</double>"
39              + "</void></object></java>";
40          final String FAIL_MESSAGE = "JDK is vulnerable to CVE-2010-4476 (requires write access to EJBCA database to exploit).";
41          assertTrue(FAIL_MESSAGE, new DecoderThread(XML_W_BADFLOAT).execute());
42          assertTrue(FAIL_MESSAGE, new DecoderThread(XML_W_BADDERFLOAT).execute());
43      }
44      
45      /** Separate thread for test that might hang. */
46      class DecoderThread implements Runnable {   // NOPMD this is a stand-alone test, not a part of a JEE application
47          final String decodeXML;
48  
49          DecoderThread(String decodeXML) {
50              this.decodeXML = decodeXML;
51          }
52          
53          protected boolean execute() {
54              Thread t = new Thread(this);    // NOPMD this is a stand-alone test, not a part of a JEE application
55              t.start();
56              try {
57                  t.join(4000);   //Wait 5 seconds for thread to complete
58              } catch (InterruptedException e) {
59                  e.printStackTrace();
60              } 
61              if (!t.getState().equals(State.TERMINATED)) {
62                  t.interrupt();
63                  return false;
64              }
65              return true;
66          }
67  
68          @SuppressWarnings("unchecked")
69          @Override
70          public void run() {
71              try (final SecureXMLDecoder decoder = new SecureXMLDecoder(new java.io.ByteArrayInputStream(decodeXML.getBytes(StandardCharsets.UTF_8)))) {
72                  final HashMap<Object,Object> h = (HashMap<Object,Object>) decoder.readObject();
73                  for (Object o : h.keySet()) {
74                      log.info(o.toString() + ": " + h.get(o));
75                  }
76              } catch (IOException e) {
77                  log.error("Failed to decode XML", e);
78                  throw new IllegalStateException(e);
79              }
80          }
81      }
