126     @Test
127     public void test02() {
128         String dn = CertTools.stringToBCDNString("uri=fff,CN=oid,SN=12345,NAME=name,C=se");
129         final X500Name name = CertTools.stringToBcX500Name(dn);
130         ASN1ObjectIdentifier[] oids = name.getAttributeTypes();
131         assertEquals(BCStyle.CN, oids[0]);
132         assertEquals(BCStyle.NAME, oids[1]);
133         assertEquals(BCStyle.SERIALNUMBER, oids[2]);
134         assertEquals(BCStyle.C, oids[3]);
135         assertEquals("CN=oid,Name=name,SN=12345,C=se", dn);
136 
137         String dn1 = CertTools.stringToBCDNString("SURNAME=Json,=fff,Description=test,CN=oid,SN=12345,NAME=name,C=se");
138         final X500Name name1 = CertTools.stringToBcX500Name(dn1);
139         ASN1ObjectIdentifier[] oids1 = name1.getAttributeTypes();
140         assertEquals(CeSecoreNameStyle.DESCRIPTION, oids1[0]);
141         assertEquals(BCStyle.CN, oids1[1]);
142         assertEquals(BCStyle.NAME, oids1[2]);
143         assertEquals(BCStyle.SERIALNUMBER, oids1[3]);
144         assertEquals(BCStyle.SURNAME, oids1[4]);
145         assertEquals(BCStyle.C, oids1[5]);
146         assertEquals("description=test,CN=oid,Name=name,SN=12345,SURNAME=Json,C=se", dn1);
147 
148         String dn2 = CertTools.stringToBCDNString("jurisdictionCountry=SE,jurisdictionState=Stockholm,SURNAME=Json,=fff,CN=oid,jurisdictionLocality=Solna,SN=12345,unstructuredname=foo.bar.com,unstructuredaddress=1.2.3.4,NAME=name,C=se");
149         final X500Name name2 = CertTools.stringToBcX500Name(dn2);
150         ASN1ObjectIdentifier[] oids2 = name2.getAttributeTypes();
151         assertEquals(CeSecoreNameStyle.JURISDICTION_COUNTRY, oids2[0]);
152         assertEquals(CeSecoreNameStyle.JURISDICTION_STATE, oids2[1]);
153         assertEquals(CeSecoreNameStyle.JURISDICTION_LOCALITY, oids2[2]);
154         assertEquals(CeSecoreNameStyle.UnstructuredAddress, oids2[3]);
155         assertEquals(CeSecoreNameStyle.UnstructuredName, oids2[4]);
156         assertEquals(BCStyle.CN, oids2[5]);
157         assertEquals(BCStyle.NAME, oids2[6]);
158         assertEquals(BCStyle.SERIALNUMBER, oids2[7]);
159         assertEquals(BCStyle.SURNAME, oids2[8]);
160         assertEquals(BCStyle.C, oids2[9]);
161         assertEquals("JurisdictionCountry=SE,JurisdictionState=Stockholm,JurisdictionLocality=Solna,unstructuredAddress=1.2.3.4,unstructuredName=foo.bar.com,CN=oid,Name=name,SN=12345,SURNAME=Json,C=se", dn2);
162 
163     }
