106 	@Test
107     public void test04TestAbsoluteValidityWithSecondsPrecision() throws Exception {
108 	    LOG.trace(">test04TestAbsoluteValidityWithSecondsPrecision");
109 	    final EndEntityInformation subject = new EndEntityInformation();
110         final CertificateProfile profile = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
111         CertificateValidity validity;
112         profile.setEncodedValidity(ValidityDate.formatAsISO8601(absolulteTestDate, ValidityDate.TIMEZONE_SERVER));
113         Date notBefore;
114         Date notAfter;
115         
116         // A: Tests inside nested CA certificate validity.
117         // 1. Test validity without offset in certificate profile: default offset by cesecore.properties with '-10m'.
118         profile.setUseCertificateValidityOffset(false);
119         profile.setUseExpirationRestrictionForWeekdays(false);
120         profile.setAllowValidityOverride(false);
121         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
122         notBefore = new Date(now.getTime() + CertificateValidity.getValidityOffset());
123         notAfter = absolulteTestDate;
124         assertTrue("1. NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'", validity.getNotBefore().equals(notBefore));
125         assertTrue("1. NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'", equals(validity.getNotAfter(), notAfter));
126         
127         // 2. Test validity with offset in certificate profile: '-1mo-2d-3h-4m-5s'.
128         profile.setUseCertificateValidityOffset(true);
129         profile.setCertificateValidityOffset("-1mo-2d-3h-4m-5s");
130         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
131         notBefore = new Date(now.getTime() + SimpleTime.parseMillies(profile.getCertificateValidityOffset()));
132         notAfter = absolulteTestDate;
133         assertTrue("2. NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'", validity.getNotBefore().getTime() == notBefore.getTime());
134         assertTrue("2. NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'", equals(validity.getNotAfter(), notAfter));
135         
136         // 3. ECA-5330 Prove that expiration restrictions does not affect a fix date validity.
137         profile.setUseExpirationRestrictionForWeekdays(true);
138         GregorianCalendar calendar = new GregorianCalendar();
139         calendar.setTime(absolulteTestDate);
140         profile.setExpirationRestrictionWeekday(calendar.get(Calendar.DAY_OF_WEEK), true);
141         // now the end date caused by relative times must be rolled on day forward, but not with absolute times.
142         profile.setExpirationRestrictionForWeekdaysExpireBefore(false);
143         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
144         notBefore = new Date(now.getTime() + SimpleTime.parseMillies(profile.getCertificateValidityOffset()));
145         notAfter = absolulteTestDate;
146         assertTrue("3. NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'.", validity.getNotBefore().getTime() == notBefore.getTime());
147         assertTrue("3. NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'.", equals(validity.getNotAfter(), notAfter));
148         
149         // 4.1 Overwrite start date with fixed date end entity information (which has first priority)
150         // 5.1 Overwrite end date with fixed date end entity information (which has first priority)
151         profile.setAllowValidityOverride(true);
152         profile.setUseCertificateValidityOffset(true); // is overwritten by the extended information date!
153         profile.setUseExpirationRestrictionForWeekdays(false); // otherwise it could overwrite the end date again!
154         String extendedInformationStartDate = ValidityDate.formatAsUTC(now);
155         String extendedInformationEndDate = ValidityDate.formatAsUTC(absolulteTestDate);
156         ExtendedInformation extendedInformation = new ExtendedInformation();
157         extendedInformation.setCustomData(ExtendedInformation.CUSTOM_STARTTIME, extendedInformationStartDate);
158         extendedInformation.setCustomData(ExtendedInformation.CUSTOM_ENDTIME, extendedInformationEndDate);
159         subject.setExtendedInformation(extendedInformation);
160         Date methodParmameterStartDate = new Date( now.getTime() + 2L * SimpleTime.MILLISECONDS_PER_DAY);
161         Date methodParmameterEndDate = new Date( absolulteTestDate.getTime() - 2L * SimpleTime.MILLISECONDS_PER_DAY);
162         validity = new CertificateValidity(now, subject, profile, methodParmameterStartDate, methodParmameterEndDate, caCertificate, false, false);
163         // seconds MUST be cut here!
164         notBefore = ValidityDate.parseAsUTC(extendedInformationStartDate);
165         notAfter = new Date((absolulteTestDate.getTime()/(60*1000))*60*1000);
166         assertTrue("4.1 Start date '"+validity.getNotBefore()+"' was overwritten by extended information start date'"+notBefore+"'.", equals(validity.getNotBefore(), notBefore));
167         assertTrue("5.1 NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'", equals(validity.getNotAfter(), notAfter));
168         
169         // 4.2 Overwrite start date by method parameter
170         // 5.2 Overwrite end date by method parameter
171         subject.getExtendedInformation().setCustomData(ExtendedInformation.CUSTOM_STARTTIME, null); // First priority -> set to null.
172         subject.getExtendedInformation().setCustomData(ExtendedInformation.CUSTOM_ENDTIME, null); // First priority -> set to null.
173         validity = new CertificateValidity(now, subject, profile, methodParmameterStartDate, methodParmameterEndDate, caCertificate, false, false);
174         notBefore = methodParmameterStartDate;
175         notAfter = methodParmameterEndDate;
176         assertTrue("4.2 Start date '"+validity.getNotBefore()+"' was overwritten by method parameter start date'"+notBefore+"'.", equals(validity.getNotBefore(), notBefore));
177         assertTrue("5.2 NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'", equals(validity.getNotAfter(), notAfter));
178        
179         // 6. Don't exceed CA too late expire date or an IllegalValidityException will be thrown! This is the last step!
180         // There is a clause in certificate validity that prevents notAfter to be after the certificate profiles encodedValidity, so it MUST be set to a date after the date to test! 
181         // Let away CA certificate to avoid time nesting.
182         final Date exceededEndDate = new Date(tooLateExpireTestDate.getTime() + 2L * SimpleTime.MILLISECONDS_PER_DAY);
183         try {
184             profile.setEncodedValidity(ValidityDate.formatAsISO8601(new Date(tooLateExpireTestDate.getTime() + 4L * SimpleTime.MILLISECONDS_PER_DAY), ValidityDate.TIMEZONE_SERVER));
185             validity = new CertificateValidity(now, subject, profile, null, exceededEndDate, null, false, false);
186             fail("6. Exceeding the latest validity '"+tooLateExpireTestDate+"' with the end date '"+exceededEndDate+
187                     "' MUST cause an IllegalValidityException! But may be the notAfter value was calculated wrong: '" + validity.getNotAfter() + "'");
188         }
189         catch(IllegalValidityException e) {
190             // NOOP
191         }
192         LOG.trace("<test04TestAbsoluteValidityWithSecondsPrecision");
193     }

221     public void test05TestRelativeValidityWithSecondsPrecision() throws Exception {
222 	    LOG.trace(">test05TestRealtiveValidityWithSecondsPrecision");
223         final EndEntityInformation subject = new EndEntityInformation();
224         final CertificateProfile profile = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
225         CertificateValidity validity;
226         profile.setEncodedValidity("1mo");
227         Date notBefore;
228         Date notAfter;
229         
230         // A: Tests inside nested CA certificate validity.
231         // 1. ECA-5141 Test relative time with seconds precision
232         final String relativeTestTime = "7d8h9m10s";
233         profile.setUseCertificateValidityOffset(false); // use default offset '-10m'
234         profile.setUseExpirationRestrictionForWeekdays(false);
235         profile.setEncodedValidity(relativeTestTime);
236         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
237         notBefore = new Date(now.getTime() + CertificateValidity.getValidityOffset());
238         notAfter = new Date(now.getTime() + CertificateValidity.getValidityOffset() + SimpleTime.parseMillies(relativeTestTime));
239         assertTrue("1. NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'.", equals( validity.getNotBefore(), notBefore));
240         assertTrue("1. NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'.", equals(validity.getNotAfter(), notAfter));
241                 
242         // 2. ECA-5330 Expiration restrictions for a relative validity is applied during issuance.
243         // 2.1 Roll forward ("but expire" -> "After")
244         profile.setUseCertificateValidityOffset(false); // use default offset '-10m'
245         profile.setUseExpirationRestrictionForWeekdays(true);
246         GregorianCalendar calendar = new GregorianCalendar();
247         calendar.setTime(notAfter);
248         setExpirationRestrictionForWeekdays(profile, calendar.get(Calendar.DAY_OF_WEEK));
249         // now the relative time validity end date must be rolled on day forward
250         profile.setExpirationRestrictionForWeekdaysExpireBefore(false);
251         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
252         notBefore = new Date(now.getTime() + CertificateValidity.getValidityOffset());
253         notAfter = new Date(now.getTime() + CertificateValidity.getValidityOffset() + SimpleTime.parseMillies(relativeTestTime));
254         calendar = new GregorianCalendar();
255         calendar.setTime(notAfter);
256         calendar.add(Calendar.DAY_OF_MONTH, 1); 
257         notAfter = calendar.getTime();
258         assertTrue("2.1 NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'.", validity.getNotBefore().getTime() == notBefore.getTime());
259         assertTrue("2.1 NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'.", equals(validity.getNotAfter(), notAfter));
260         
261         // 2.2 Roll backward("but expire" -> "Before")
262         profile.setExpirationRestrictionForWeekdaysExpireBefore(true);
263         validity = new CertificateValidity(now, subject, profile, null, null, caCertificate, false, false);
264         notBefore = new Date(now.getTime() + CertificateValidity.getValidityOffset());
265         notAfter = new Date(now.getTime() + CertificateValidity.getValidityOffset() + SimpleTime.parseMillies(relativeTestTime));
266         calendar = new GregorianCalendar();
267         calendar.setTime(notAfter);
268         calendar.add(Calendar.DAY_OF_MONTH, -1); 
269         notAfter = calendar.getTime();
270         assertTrue("3.1 NotBefore '"+validity.getNotBefore()+"'matches start date '"+notBefore+"'.", validity.getNotBefore().getTime() == notBefore.getTime());
271         assertTrue("3.1 NotAfter '"+validity.getNotAfter()+"' matches end date '"+notAfter+"'.", equals(validity.getNotAfter(), notAfter));
272         LOG.trace("<test05TestRealtiveValidityWithSecondsPrecision");
273     }

386     private void testBaseTestCertificateValidity(String encodedValidity) throws Exception {
387 		final Date caFrom = new Date();
388 		caFrom.setTime(caFrom.getTime() - 20L*(24L*60L*60L*1000L));
389 		final Date caTo = new Date();
390 		caTo.setTime(caTo.getTime() + 100L*(24L * 60L * 60L * 1000L));
391 		
392     	X509Certificate cacert = CertTools.genSelfCertForPurpose("CN=dummy2", caFrom, caTo, null, keyPair.getPrivate(), keyPair.getPublic(),
393     			AlgorithmConstants.SIGALG_SHA1_WITH_RSA, true, X509KeyUsage.cRLSign|X509KeyUsage.keyCertSign,
394     			null, null, "BC", true, null);
395 
396         final EndEntityInformation subject = new EndEntityInformation();
397     	final CertificateProfile cp = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
398     	cp.setEncodedValidity(encodedValidity);
399     	cp.setUseExpirationRestrictionForWeekdays(false);
400     	cp.setAllowValidityOverride(false);
401     
402     	// First see that when we don't have a specified time requested and validity override is not allowed, the end time should be ruled by the certificate profile.
403     	
404     	CertificateValidity cv = new CertificateValidity(subject, cp, null, null, cacert, false, false);
405     	Date notBefore = cv.getNotBefore();
406     	Date notAfter = cv.getNotAfter();
407     	Date now = new Date();
408         Calendar cal1 = Calendar.getInstance();

615     /**
616      * Returns true if the encoded validity given is no ISO8601 date.
617      * @param encodedValidity valid
618      * @return bool
619      */
620     private final boolean isRelativeTime(final String encodedValidity) {
621         try {
622             ValidityDate.parseAsIso8601(encodedValidity);
623             return false;
624         } catch(ParseException e) {
625             // must be SimpleTime here
626             return true;
627         }
628     }

630     /** Compares the dates without milliseconds. 
631      * @param leftSide date
632      * @param rightSide date
633      * @return bool */
634     private final boolean equals(final Date leftSide, final Date rightSide) {
635         return leftSide.getTime() / 1000 == rightSide.getTime() / 1000;
636     }
