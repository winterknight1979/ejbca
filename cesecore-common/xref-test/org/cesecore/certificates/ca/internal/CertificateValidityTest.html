173   @Test
174   public void test04TestAbsoluteValidityWithSecondsPrecision()
175       throws Exception {
176     LOG.trace(">test04TestAbsoluteValidityWithSecondsPrecision");
177     final EndEntityInformation subject = new EndEntityInformation();
178     final CertificateProfile profile =
179         new CertificateProfile(
180             CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
181     CertificateValidity validity;
182     profile.setEncodedValidity(
183         ValidityDate.formatAsISO8601(
184             absolulteTestDate, ValidityDate.TIMEZONE_SERVER));
185     Date notBefore;
186     Date notAfter;
187 
188     // A: Tests inside nested CA certificate validity.
189     // 1. Test validity without offset in certificate profile: default offset by
190     // cesecore.properties with '-10m'.
191     profile.setUseCertificateValidityOffset(false);
192     profile.setUseExpirationRestrictionForWeekdays(false);
193     profile.setAllowValidityOverride(false);
194     validity =
195         new CertificateValidity(
196             now, subject, profile, null, null, caCertificate, false, false);
197     notBefore =
198         new Date(now.getTime() + CertificateValidity.getValidityOffset());
199     notAfter = absolulteTestDate;
200     assertTrue(
201         "1. NotBefore '"
202             + validity.getNotBefore()
203             + "'matches start date '"
204             + notBefore
205             + "'",
206         validity.getNotBefore().equals(notBefore));
207     assertTrue(
208         "1. NotAfter '"
209             + validity.getNotAfter()
210             + "' matches end date '"
211             + notAfter
212             + "'",
213         equals(validity.getNotAfter(), notAfter));
214 
215     // 2. Test validity with offset in certificate profile: '-1mo-2d-3h-4m-5s'.
216     profile.setUseCertificateValidityOffset(true);
217     profile.setCertificateValidityOffset("-1mo-2d-3h-4m-5s");
218     validity =
219         new CertificateValidity(
220             now, subject, profile, null, null, caCertificate, false, false);
221     notBefore =
222         new Date(
223             now.getTime()
224                 + SimpleTime.parseMillies(
225                     profile.getCertificateValidityOffset()));
226     notAfter = absolulteTestDate;
227     assertTrue(
228         "2. NotBefore '"
229             + validity.getNotBefore()
230             + "'matches start date '"
231             + notBefore
232             + "'",
233         validity.getNotBefore().getTime() == notBefore.getTime());
234     assertTrue(
235         "2. NotAfter '"
236             + validity.getNotAfter()
237             + "' matches end date '"
238             + notAfter
239             + "'",
240         equals(validity.getNotAfter(), notAfter));
241 
242     // 3. ECA-5330 Prove that expiration restrictions does not affect a fix date
243     // validity.
244     profile.setUseExpirationRestrictionForWeekdays(true);
245     GregorianCalendar calendar = new GregorianCalendar();
246     calendar.setTime(absolulteTestDate);
247     profile.setExpirationRestrictionWeekday(
248         calendar.get(Calendar.DAY_OF_WEEK), true);
249     // now the end date caused by relative times must be rolled on day forward,
250     // but not with absolute times.
251     profile.setExpirationRestrictionForWeekdaysExpireBefore(false);
252     validity =
253         new CertificateValidity(
254             now, subject, profile, null, null, caCertificate, false, false);
255     notBefore =
256         new Date(
257             now.getTime()
258                 + SimpleTime.parseMillies(
259                     profile.getCertificateValidityOffset()));
260     notAfter = absolulteTestDate;
261     assertTrue(
262         "3. NotBefore '"
263             + validity.getNotBefore()
264             + "'matches start date '"
265             + notBefore
266             + "'.",
267         validity.getNotBefore().getTime() == notBefore.getTime());
268     assertTrue(
269         "3. NotAfter '"
270             + validity.getNotAfter()
271             + "' matches end date '"
272             + notAfter
273             + "'.",
274         equals(validity.getNotAfter(), notAfter));
275 
276     // 4.1 Overwrite start date with fixed date end entity information (which
277     // has first priority)
278     // 5.1 Overwrite end date with fixed date end entity information (which has
279     // first priority)
280     profile.setAllowValidityOverride(true);
281     profile.setUseCertificateValidityOffset(
282         true); // is overwritten by the extended information date!
283     profile.setUseExpirationRestrictionForWeekdays(
284         false); // otherwise it could overwrite the end date again!
285     String extendedInformationStartDate = ValidityDate.formatAsUTC(now);
286     String extendedInformationEndDate =
287         ValidityDate.formatAsUTC(absolulteTestDate);
288     ExtendedInformation extendedInformation = new ExtendedInformation();
289     extendedInformation.setCustomData(
290         ExtendedInformation.CUSTOM_STARTTIME, extendedInformationStartDate);
291     extendedInformation.setCustomData(
292         ExtendedInformation.CUSTOM_ENDTIME, extendedInformationEndDate);
293     subject.setExtendedInformation(extendedInformation);
294     Date methodParmameterStartDate =
295         new Date(now.getTime() + 2L * SimpleTime.MILLISECONDS_PER_DAY);
296     Date methodParmameterEndDate =
297         new Date(
298             absolulteTestDate.getTime() - 2L * SimpleTime.MILLISECONDS_PER_DAY);
299     validity =
300         new CertificateValidity(
301             now,
302             subject,
303             profile,
304             methodParmameterStartDate,
305             methodParmameterEndDate,
306             caCertificate,
307             false,
308             false);
309     // seconds MUST be cut here!
310     notBefore = ValidityDate.parseAsUTC(extendedInformationStartDate);
311     notAfter =
312         new Date((absolulteTestDate.getTime() / (60 * 1000)) * 60 * 1000);
313     assertTrue(
314         "4.1 Start date '"
315             + validity.getNotBefore()
316             + "' was overwritten by extended information start date'"
317             + notBefore
318             + "'.",
319         equals(validity.getNotBefore(), notBefore));
320     assertTrue(
321         "5.1 NotAfter '"
322             + validity.getNotAfter()
323             + "' matches end date '"
324             + notAfter
325             + "'",
326         equals(validity.getNotAfter(), notAfter));
327 
328     // 4.2 Overwrite start date by method parameter
329     // 5.2 Overwrite end date by method parameter
330     subject
331         .getExtendedInformation()
332         .setCustomData(
333             ExtendedInformation.CUSTOM_STARTTIME,
334             null); // First priority -> set to null.
335     subject
336         .getExtendedInformation()
337         .setCustomData(
338             ExtendedInformation.CUSTOM_ENDTIME,
339             null); // First priority -> set to null.
340     validity =
341         new CertificateValidity(
342             now,
343             subject,
344             profile,
345             methodParmameterStartDate,
346             methodParmameterEndDate,
347             caCertificate,
348             false,
349             false);
350     notBefore = methodParmameterStartDate;
351     notAfter = methodParmameterEndDate;
352     assertTrue(
353         "4.2 Start date '"
354             + validity.getNotBefore()
355             + "' was overwritten by method parameter start date'"
356             + notBefore
357             + "'.",
358         equals(validity.getNotBefore(), notBefore));
359     assertTrue(
360         "5.2 NotAfter '"
361             + validity.getNotAfter()
362             + "' matches end date '"
363             + notAfter
364             + "'",
365         equals(validity.getNotAfter(), notAfter));
366 
367     // 6. Don't exceed CA too late expire date or an IllegalValidityException
368     // will be thrown! This is the last step!
369     // There is a clause in certificate validity that prevents notAfter to be
370     // after the certificate profiles encodedValidity, so it MUST be set to a
371     // date after the date to test!
372     // Let away CA certificate to avoid time nesting.
373     final Date exceededEndDate =
374         new Date(
375             tooLateExpireTestDate.getTime()
376                 + 2L * SimpleTime.MILLISECONDS_PER_DAY);
377     try {
378       profile.setEncodedValidity(
379           ValidityDate.formatAsISO8601(
380               new Date(
381                   tooLateExpireTestDate.getTime()
382                       + 4L * SimpleTime.MILLISECONDS_PER_DAY),
383               ValidityDate.TIMEZONE_SERVER));
384       validity =
385           new CertificateValidity(
386               now, subject, profile, null, exceededEndDate, null, false, false);
387       fail(
388           "6. Exceeding the latest validity '"
389               + tooLateExpireTestDate
390               + "' with the end date '"
391               + exceededEndDate
392               + "' MUST cause an IllegalValidityException! But may be the"
393               + " notAfter value was calculated wrong: '"
394               + validity.getNotAfter()
395               + "'");
396     } catch (IllegalValidityException e) {
397       // NOOP
398     }
399     LOG.trace("<test04TestAbsoluteValidityWithSecondsPrecision");
400   }

460   public void test05TestRelativeValidityWithSecondsPrecision()
461       throws Exception {
462     LOG.trace(">test05TestRealtiveValidityWithSecondsPrecision");
463     final EndEntityInformation subject = new EndEntityInformation();
464     final CertificateProfile profile =
465         new CertificateProfile(
466             CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
467     CertificateValidity validity;
468     profile.setEncodedValidity("1mo");
469     Date notBefore;
470     Date notAfter;
471 
472     // A: Tests inside nested CA certificate validity.
473     // 1. ECA-5141 Test relative time with seconds precision
474     final String relativeTestTime = "7d8h9m10s";
475     profile.setUseCertificateValidityOffset(false); // use default offset '-10m'
476     profile.setUseExpirationRestrictionForWeekdays(false);
477     profile.setEncodedValidity(relativeTestTime);
478     validity =
479         new CertificateValidity(
480             now, subject, profile, null, null, caCertificate, false, false);
481     notBefore =
482         new Date(now.getTime() + CertificateValidity.getValidityOffset());
483     notAfter =
484         new Date(
485             now.getTime()
486                 + CertificateValidity.getValidityOffset()
487                 + SimpleTime.parseMillies(relativeTestTime));
488     assertTrue(
489         "1. NotBefore '"
490             + validity.getNotBefore()
491             + "'matches start date '"
492             + notBefore
493             + "'.",
494         equals(validity.getNotBefore(), notBefore));
495     assertTrue(
496         "1. NotAfter '"
497             + validity.getNotAfter()
498             + "' matches end date '"
499             + notAfter
500             + "'.",
501         equals(validity.getNotAfter(), notAfter));
502 
503     // 2. ECA-5330 Expiration restrictions for a relative validity is applied
504     // during issuance.
505     // 2.1 Roll forward ("but expire" -> "After")
506     profile.setUseCertificateValidityOffset(false); // use default offset '-10m'
507     profile.setUseExpirationRestrictionForWeekdays(true);
508     GregorianCalendar calendar = new GregorianCalendar();
509     calendar.setTime(notAfter);
510     setExpirationRestrictionForWeekdays(
511         profile, calendar.get(Calendar.DAY_OF_WEEK));
512     // now the relative time validity end date must be rolled on day forward
513     profile.setExpirationRestrictionForWeekdaysExpireBefore(false);
514     validity =
515         new CertificateValidity(
516             now, subject, profile, null, null, caCertificate, false, false);
517     notBefore =
518         new Date(now.getTime() + CertificateValidity.getValidityOffset());
519     notAfter =
520         new Date(
521             now.getTime()
522                 + CertificateValidity.getValidityOffset()
523                 + SimpleTime.parseMillies(relativeTestTime));
524     calendar = new GregorianCalendar();
525     calendar.setTime(notAfter);
526     calendar.add(Calendar.DAY_OF_MONTH, 1);
527     notAfter = calendar.getTime();
528     assertTrue(
529         "2.1 NotBefore '"
530             + validity.getNotBefore()
531             + "'matches start date '"
532             + notBefore
533             + "'.",
534         validity.getNotBefore().getTime() == notBefore.getTime());
535     assertTrue(
536         "2.1 NotAfter '"
537             + validity.getNotAfter()
538             + "' matches end date '"
539             + notAfter
540             + "'.",
541         equals(validity.getNotAfter(), notAfter));
542 
543     // 2.2 Roll backward("but expire" -> "Before")
544     profile.setExpirationRestrictionForWeekdaysExpireBefore(true);
545     validity =
546         new CertificateValidity(
547             now, subject, profile, null, null, caCertificate, false, false);
548     notBefore =
549         new Date(now.getTime() + CertificateValidity.getValidityOffset());
550     notAfter =
551         new Date(
552             now.getTime()
553                 + CertificateValidity.getValidityOffset()
554                 + SimpleTime.parseMillies(relativeTestTime));
555     calendar = new GregorianCalendar();
556     calendar.setTime(notAfter);
557     calendar.add(Calendar.DAY_OF_MONTH, -1);
558     notAfter = calendar.getTime();
559     assertTrue(
560         "3.1 NotBefore '"
561             + validity.getNotBefore()
562             + "'matches start date '"
563             + notBefore
564             + "'.",
565         validity.getNotBefore().getTime() == notBefore.getTime());
566     assertTrue(
567         "3.1 NotAfter '"
568             + validity.getNotAfter()
569             + "' matches end date '"
570             + notAfter
571             + "'.",
572         equals(validity.getNotAfter(), notAfter));
573     LOG.trace("<test05TestRealtiveValidityWithSecondsPrecision");
574   }
