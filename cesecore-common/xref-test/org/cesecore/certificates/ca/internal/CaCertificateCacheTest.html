116 	@Test
117 	public void test01CACertificates() throws Exception {
118 		// Prepare the certificate cache with some test certificates
119 		Collection<Certificate> certs = new ArrayList<Certificate>();
120 		X509Certificate testrootcert = CertTools.getCertfromByteArray(testroot, X509Certificate.class);
121 		certs.add(testrootcert);
122 		X509Certificate testrootnewcert = CertTools.getCertfromByteArray(testrootnew, X509Certificate.class);
123 		certs.add(testrootnewcert);
124 		X509Certificate testsubcert = CertTools.getCertfromByteArray(testsub, X509Certificate.class);
125 		certs.add(testsubcert);
126 		Certificate testcvccert = CertTools.getCertfromByteArray(testcvc, Certificate.class);
127 		certs.add(testcvccert);
128 		X509Certificate testscepcert = CertTools.getCertfromByteArray(testscepca, X509Certificate.class);
129 		certs.add(testscepcert);
130 		CaCertificateCache cache = CaCertificateCache.INSTANCE;
131 		cache.loadCertificates(certs);
132 		
133 		// Test lookup of not existing cert
134 		X509Certificate cert = cache.findLatestBySubjectDN(HashID.getFromDNString("CN=Foo,C=SE"));
135 		assertNull(cert);
136 		// Old root cert should not be found, we only store the latest to be found by subjectDN
137 		X509Certificate rootcert = cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testrootnewcert));
138 		assertNotNull(rootcert);
139 		X509Certificate subcert = cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testsubcert));
140 		// This old subcert should not be possible to verify with the new root cert
141 		try {
142 			subcert.verify(rootcert.getPublicKey());
143 			fail("verification should have failed");
144 		} catch (SignatureException e) {} // NOPMD: BC 1.47
145 		// CVC certificate should not be part of OCSP certificate cache
146 		cert = cache.findLatestBySubjectDN(HashID.getFromDNString(CertTools.getSubjectDN(testcvccert)));
147 		assertNull(cert);
148 		cert = cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testscepcert));
149 		assertEquals(CertTools.getSubjectDN(testscepcert), CertTools.getSubjectDN(cert));
150 
151 	}

228 	private static class CacheTester implements Runnable { // NOPMD, this is not a JEE app, only a test
229 	    private CaCertificateCache cache = null;
230 	    private String dn;
231 	    public CacheTester(CaCertificateCache cache, String lookfor) {
232 	        this.cache = cache;
233 	        this.dn = lookfor;
234 	    }
235 	    public void run() {
236 	        for (int i=0; i<1000;i++) {
237 	            X509Certificate cert = cache.findLatestBySubjectDN(HashID.getFromDNString(dn));
238 	            // The cache tests will not return any CV Certificates because this OCSP cache 
239 	            // only handles X.509 Certificates.
240 	            if (!StringUtils.contains(dn, "CVCTest")) {
241 	                cert.getSubjectDN(); // just to see that we did receive a cert, will throw NPE if no cert was returned              
242 	            }
243 	        }               
244 	    }
245 	}
