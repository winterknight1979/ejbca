134   @Test
135   public void test01CACertificates() throws Exception {
136     // Prepare the certificate cache with some test certificates
137     Collection<Certificate> certs = new ArrayList<Certificate>();
138     X509Certificate testrootcert =
139         CertTools.getCertfromByteArray(testroot, X509Certificate.class);
140     certs.add(testrootcert);
141     X509Certificate testrootnewcert =
142         CertTools.getCertfromByteArray(testrootnew, X509Certificate.class);
143     certs.add(testrootnewcert);
144     X509Certificate testsubcert =
145         CertTools.getCertfromByteArray(testsub, X509Certificate.class);
146     certs.add(testsubcert);
147     Certificate testcvccert =
148         CertTools.getCertfromByteArray(testcvc, Certificate.class);
149     certs.add(testcvccert);
150     X509Certificate testscepcert =
151         CertTools.getCertfromByteArray(testscepca, X509Certificate.class);
152     certs.add(testscepcert);
153     CaCertificateCache cache = CaCertificateCache.INSTANCE;
154     cache.loadCertificates(certs);
155 
156     // Test lookup of not existing cert
157     X509Certificate cert =
158         cache.findLatestBySubjectDN(HashID.getFromDNString("CN=Foo,C=SE"));
159     assertNull(cert);
160     // Old root cert should not be found, we only store the latest to be found
161     // by subjectDN
162     X509Certificate rootcert =
163         cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testrootnewcert));
164     assertNotNull(rootcert);
165     X509Certificate subcert =
166         cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testsubcert));
167     // This old subcert should not be possible to verify with the new root cert
168     try {
169       subcert.verify(rootcert.getPublicKey());
170       fail("verification should have failed");
171     } catch (SignatureException e) {
172     } // NOPMD: BC 1.47
173     // CVC certificate should not be part of OCSP certificate cache
174     cert =
175         cache.findLatestBySubjectDN(
176             HashID.getFromDNString(CertTools.getSubjectDN(testcvccert)));
177     assertNull(cert);
178     cert = cache.findLatestBySubjectDN(HashID.getFromSubjectDN(testscepcert));
179     assertEquals(
180         CertTools.getSubjectDN(testscepcert), CertTools.getSubjectDN(cert));
181   }

299   private static class CacheTester
300       implements Runnable { // NOPMD, this is not a JEE app, only a test
301       /** Cache. */
302     private CaCertificateCache cache = null;
303     /** DN .*/
304     private final String dn;
305 
306     CacheTester(final CaCertificateCache aCache, final String lookfor) {
307       this.cache = aCache;
308       this.dn = lookfor;
309     }
310 
311     public void run() {
312       for (int i = 0; i < 1000; i++) {
313         X509Certificate cert =
314             cache.findLatestBySubjectDN(HashID.getFromDNString(dn));
315         // The cache tests will not return any CV Certificates because this OCSP
316         // cache
317         // only handles X.509 Certificates.
318         if (!StringUtils.contains(dn, "CVCTest")) {
319           cert.getSubjectDN(); // just to see that we did receive a cert, will
320           // throw NPE if no cert was returned
321         }
322       }
323     }
324   }
