754               catoken.getAliasFromPurpose(
755                   CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT));

757       // There exist an RSA encryption key
758       pub =
759           cryptoToken.getPublicKey(
760               catoken.getAliasFromPurpose(
761                   CATokenConstants.CAKEYPURPOSE_KEYENCRYPT));
762       assertEquals(1024, KeyTools.getKeyLength(pub));
763     } finally {

777   /** activate / deactivate.
778    *
779    * @param keySpecification Key
780    * @param cryptoToken Token
781    * @param caTokenProperties Props
782    * @throws KeyStoreException fail
783    * @throws NoSuchAlgorithmException fail
784    * @throws CertificateException fail
785    * @throws IOException fail
786    * @throws CryptoTokenOfflineException fail
787    * @throws NoSuchProviderException fail
788    * @throws InvalidKeyException fail
789    * @throws SignatureException fail
790    * @throws CryptoTokenAuthenticationFailedException fail
791    * @throws InvalidAlgorithmParameterException fail
792    */
793   protected void doActivateDeactivate(
794       final String keySpecification,
795       final CryptoToken cryptoToken,
796       final Properties caTokenProperties)
797       throws KeyStoreException, NoSuchAlgorithmException, CertificateException,
798           IOException, CryptoTokenOfflineException, NoSuchProviderException,
799           InvalidKeyException, SignatureException,
800           CryptoTokenAuthenticationFailedException,
801           InvalidAlgorithmParameterException {
802     LOG.trace(">" + Thread.currentThread().getStackTrace()[1].getMethodName());
803     // Remove auto activate
804     Properties prop = cryptoToken.getProperties();
805     prop.remove(CryptoToken.AUTOACTIVATE_PIN_PROPERTY);
806     cryptoToken.setProperties(prop);
807 
808     CAToken catoken = new CAToken(cryptoToken.getId(), caTokenProperties);
809     try {
810       // Set key sequence so that next sequence will be 00001 (this is the
811       // default though so not really needed here)
812       catoken.setKeySequence(CAToken.DEFAULT_KEYSEQUENCE);
813       catoken.setKeySequenceFormat(StringTools.KEY_SEQUENCE_FORMAT_NUMERIC);
814       catoken.setSignatureAlgorithm(AlgorithmConstants.SIGALG_SHA1_WITH_RSA);
815       catoken.setEncryptionAlgorithm(AlgorithmConstants.SIGALG_SHA1_WITH_RSA);
816 
817       // First we start by deleting all old entries
818       try {
819         cryptoToken.deleteEntry("rsatest00001");
820         assertTrue("Should throw", false);
821       } catch (CryptoTokenOfflineException e) {
822         // NOPMD
823       }
824       cryptoToken.activate(TOKEN_PIN.toCharArray());
825       for (int i = 0; i < 4; i++) {
826         cryptoToken.deleteEntry("rsatest0000" + i);
827       }
828       // Before this there are no keys.
829       final String nextSignKeyAlias = catoken.generateNextSignKeyAlias();
830       cryptoToken.generateKeyPair(keySpecification, nextSignKeyAlias);
831       catoken.activateNextSignKey();
832       KeyTools.testKey(
833           cryptoToken.getPrivateKey(
834               catoken.getAliasFromPurpose(
835                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
836           cryptoToken.getPublicKey(
837               catoken.getAliasFromPurpose(
838                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
839           cryptoToken.getSignProviderName());
840 
841       // We have not set auto activate, so the internal key storage in
842       // CryptoToken is emptied
843       cryptoToken.deactivate();
844       try {
845         KeyTools.testKey(
846             cryptoToken.getPrivateKey(
847                 catoken.getAliasFromPurpose(
848                     CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
849             cryptoToken.getPublicKey(
850                 catoken.getAliasFromPurpose(
851                     CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
852             cryptoToken.getSignProviderName());
853         assertTrue(false);
854       } catch (CryptoTokenOfflineException e) {
855         assertEquals(
856             "Can not instantiate "
857                 + getProvider()
858                 + ". keyStore (111) == null.",
859             e.getMessage());
860       }
861       // Activate with wrong PIN should not work
862       try {
863         cryptoToken.activate((TOKEN_PIN + "x").toCharArray());
864         assertTrue("should throw", false);
865       } catch (CryptoTokenAuthenticationFailedException e) {
866         String strsoft =
867             "PKCS12 key store mac invalid - wrong password or corrupted file.";
868         String strp11 = "Failed to initialize PKCS11 provider slot '1'.";
869         assert (e.getMessage().equals(strsoft)
870             || e.getMessage().equals(strp11));
871       }
872       cryptoToken.activate(TOKEN_PIN.toCharArray());
873       KeyTools.testKey(
874           cryptoToken.getPrivateKey(
875               catoken.getAliasFromPurpose(
876                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
877           cryptoToken.getPublicKey(
878               catoken.getAliasFromPurpose(
879                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
880           cryptoToken.getSignProviderName());
881     } finally {
882       // End by deleting all old entries
883       cryptoToken.activate(TOKEN_PIN.toCharArray());
884       for (int i = 0; i < 4; i++) {
885         cryptoToken.deleteEntry("rsatest0000" + i);
886       }
887     }
888     LOG.trace("<" + Thread.currentThread().getStackTrace()[1].getMethodName());
889   }

892    *
893    * @param keySpecification Key
894    * @param cryptoToken TOken
895    * @param caTokenProperties Props
896    * @throws InvalidKeyException Fail
897    * @throws CryptoTokenAuthenticationFailedException fail
898    * @throws CryptoTokenOfflineException fail
899    * @throws NoSuchAlgorithmException fail
900    * @throws CertificateException fail
901    * @throws KeyStoreException fail
902    * @throws NoSuchProviderException fail
903    * @throws InvalidAlgorithmParameterException fail
904    * @throws SignatureException fail
905    * @throws IOException fail
906    * @throws IllegalCryptoTokenException fail
907    */
908   protected void doSaveAndLoad(
909       final String keySpecification,
910       final CryptoToken cryptoToken,
911       final Properties caTokenProperties)
912       throws InvalidKeyException, CryptoTokenAuthenticationFailedException,
913           CryptoTokenOfflineException, NoSuchAlgorithmException,
914           CertificateException, KeyStoreException, NoSuchProviderException,
915           InvalidAlgorithmParameterException, SignatureException, IOException,
916           IllegalCryptoTokenException {
917     LOG.trace(">" + Thread.currentThread().getStackTrace()[1].getMethodName());
918     CAToken catoken = new CAToken(cryptoToken.getId(), caTokenProperties);
919     try {
920       cryptoToken.activate(TOKEN_PIN.toCharArray());
921       cryptoToken.generateKeyPair(keySpecification, DEFAULT_KEY);
922       catoken.setNextCertSignKey(DEFAULT_KEY);
923       catoken.activateNextSignKey();
924       PublicKey publicKey =
925           cryptoToken.getPublicKey(
926               catoken.getAliasFromPurpose(
927                   CATokenConstants.CAKEYPURPOSE_CERTSIGN));
928       KeyTools.testKey(
929           cryptoToken.getPrivateKey(
930               catoken.getAliasFromPurpose(
931                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
932           publicKey,
933           cryptoToken.getSignProviderName());
934       String keyhash = CertTools.getFingerprintAsString(publicKey.getEncoded());
935       HashMap<?, ?> data = (HashMap<?, ?>) catoken.saveData();
936       CAToken newcatoken = new CAToken(data);
937       assertEquals(cryptoToken.getId(), newcatoken.getCryptoTokenId());
938       PublicKey newPublicKey =
939           cryptoToken.getPublicKey(
940               newcatoken.getAliasFromPurpose(
941                   CATokenConstants.CAKEYPURPOSE_CERTSIGN));
942       KeyTools.testKey(
943           cryptoToken.getPrivateKey(
944               newcatoken.getAliasFromPurpose(
945                   CATokenConstants.CAKEYPURPOSE_CERTSIGN)),
946           newPublicKey,
947           cryptoToken.getSignProviderName());
948       String newkeyhash =
949           CertTools.getFingerprintAsString(newPublicKey.getEncoded());
950       assertEquals(keyhash, newkeyhash);
951     } finally {
952       // End by deleting all old entries
953       cryptoToken.activate(TOKEN_PIN.toCharArray());
954       cryptoToken.deleteEntry("rsatest00000");
955       cryptoToken.deleteEntry("rsatest00001");
956     }
957     LOG.trace("<" + Thread.currentThread().getStackTrace()[1].getMethodName());
958   }
