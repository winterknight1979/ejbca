51  	@Test
52      public void testExtendedInformationCSR() {
53          // Test basic function, latest version->latest version
54          ExtendedInformation ei = new ExtendedInformation();
55          ei.setCertificateRequest(Base64.decode(pkcs10.getBytes(StandardCharsets.UTF_8)));
56          assertEquals(pkcs10, new String(Base64.encode(ei.getCertificateRequest()), StandardCharsets.UTF_8));
57          // Test by hardcoding in the base64 encoded data as string, to ensure we can read such data
58          // (even if the current implementation changed)
59          ExtendedInformation ei2 = new ExtendedInformation();
60          ei2.setMapData("CERTIFICATE_REQUEST", pkcs10);
61          assertEquals(pkcs10, new String(Base64.encode(ei2.getCertificateRequest()), StandardCharsets.UTF_8));
62          // Test null
63          ExtendedInformation ei3 = new ExtendedInformation();
64          ei3.setMapData("FOO", pkcs10);
65          assertNull(ei3.getCertificateRequest());
66          // Test by hardcoding in the binary encoded data, to ensure we can read such data
67          // As it was stored before EJBCA 6.8.0
68          ExtendedInformation ei4 = new ExtendedInformation();
69          LinkedHashMap<String,Object> map = new LinkedHashMap<String,Object>();
70          map.put(UpgradeableDataHashMap.VERSION, Float.valueOf(4));
71          map.put("CERTIFICATE_REQUEST", Base64.decode(pkcs10.getBytes(StandardCharsets.UTF_8)));
72          ei4.setData(map);
73          assertEquals(pkcs10, new String(Base64.encode(ei4.getCertificateRequest()), StandardCharsets.UTF_8));
74      }

76      @Test
77      public void testSetGetCopyFields() {
78          final Integer approvalRequestId = 1;
79          final String certificateEndTime = "2020-01-01T12:00:00+00:00";
80          final byte[] certificateRequest = new byte[] { 0x1, 0x2, 0x3 };
81          final BigInteger certificateSerialNumber = new BigInteger("13813134560128420686215030629780757563");
82          final String customDataKey = "customDataKey";
83          final String customDataValue = "customDataValue";
84          final String customExtensionDataKey = "customExtensionDataKey";
85          final String customExtensionDataValue = "customExtensionDataValue";
86          final Integer issuanceRevocationReason = 2;
87          final String keyStoreSubAlgorithmType = "4096";
88          final String keyStoreAlgorithm = "RSA";
89          final String mapDataKey = "mapDataKey";
90          final String mapDataValue = "mapDataValue";
91          final Integer maxLoginAttempts = 3;
92          final List<String> nameConstraintsExcluded = Arrays.asList(new String[] { "a", "b", "c" });
93          final List<String> nameConstraintsPermitted = Arrays.asList(new String[] { "d", "e", "f" });
94          final Integer remainingLoginAttempts = 5;
95          final String subjectDirectoryAttributes = "subjectDirectoryAttributes";
96  
97          final ExtendedInformation extendedInformation = new ExtendedInformation();
98          extendedInformation.setAddEndEntityApprovalRequestId(approvalRequestId);
99          extendedInformation.setCertificateEndTime(certificateEndTime);
100         extendedInformation.setCertificateRequest(certificateRequest);
101         extendedInformation.setCertificateSerialNumber(certificateSerialNumber);
102         extendedInformation.setCustomData(customDataKey, customDataValue);
103         extendedInformation.setExtensionData(customExtensionDataKey, customExtensionDataValue);
104         extendedInformation.setIssuanceRevocationReason(issuanceRevocationReason);
105         extendedInformation.setKeyStoreAlgorithmSubType(keyStoreSubAlgorithmType);
106         extendedInformation.setKeyStoreAlgorithmType(keyStoreAlgorithm);
107         extendedInformation.setMapData(mapDataKey, mapDataValue);
108         extendedInformation.setMaxLoginAttempts(maxLoginAttempts);
109         extendedInformation.setNameConstraintsExcluded(nameConstraintsExcluded);
110         extendedInformation.setNameConstraintsPermitted(nameConstraintsPermitted);
111         extendedInformation.setRemainingLoginAttempts(remainingLoginAttempts);
112         extendedInformation.setSubjectDirectoryAttributes(subjectDirectoryAttributes);
113 
114         assertEquals(approvalRequestId, extendedInformation.getAddEndEntityApprovalRequestId());
115         assertEquals(certificateEndTime, extendedInformation.getCertificateEndTime());
116         assertArrayEquals(certificateRequest, extendedInformation.getCertificateRequest());
117         assertEquals(certificateSerialNumber, extendedInformation.certificateSerialNumber());
118         assertEquals(customDataValue, extendedInformation.getCustomData(customDataKey));
119         assertEquals(customExtensionDataValue, extendedInformation.getExtensionData(customExtensionDataKey));
120         assertEquals(issuanceRevocationReason, (Integer) extendedInformation.getIssuanceRevocationReason());
121         assertEquals(keyStoreSubAlgorithmType, extendedInformation.getKeyStoreAlgorithmSubType());
122         assertEquals(keyStoreAlgorithm, extendedInformation.getKeyStoreAlgorithmType());
123         assertEquals(mapDataValue, extendedInformation.getMapData(mapDataKey));
124         assertEquals(maxLoginAttempts, (Integer) extendedInformation.getMaxLoginAttempts());
125         assertEquals(nameConstraintsExcluded, extendedInformation.getNameConstraintsExcluded());
126         assertEquals(nameConstraintsPermitted, extendedInformation.getNameConstraintsPermitted());
127         assertEquals(remainingLoginAttempts, (Integer) extendedInformation.getRemainingLoginAttempts());
128         assertEquals(subjectDirectoryAttributes, extendedInformation.getSubjectDirectoryAttributes());
129 
130         final ExtendedInformation extendedInformation2 = new ExtendedInformation(extendedInformation);
131         assertEquals(approvalRequestId, extendedInformation2.getAddEndEntityApprovalRequestId());
132         assertEquals(certificateEndTime, extendedInformation2.getCertificateEndTime());
133         assertArrayEquals(certificateRequest, extendedInformation2.getCertificateRequest());
134         assertEquals(certificateSerialNumber, extendedInformation2.certificateSerialNumber());
135         assertEquals(customDataValue, extendedInformation2.getCustomData(customDataKey));
136         assertEquals(customExtensionDataValue, extendedInformation2.getExtensionData(customExtensionDataKey));
137         assertEquals(issuanceRevocationReason, (Integer) extendedInformation2.getIssuanceRevocationReason());
138         assertEquals(keyStoreSubAlgorithmType, extendedInformation2.getKeyStoreAlgorithmSubType());
139         assertEquals(keyStoreAlgorithm, extendedInformation2.getKeyStoreAlgorithmType());
140         assertEquals(mapDataValue, extendedInformation2.getMapData(mapDataKey));
141         assertEquals(maxLoginAttempts, (Integer) extendedInformation2.getMaxLoginAttempts());
142         assertEquals(nameConstraintsExcluded, extendedInformation2.getNameConstraintsExcluded());
143         assertEquals(nameConstraintsPermitted, extendedInformation2.getNameConstraintsPermitted());
144         assertEquals(remainingLoginAttempts, (Integer) extendedInformation2.getRemainingLoginAttempts());
145         assertEquals(subjectDirectoryAttributes, extendedInformation2.getSubjectDirectoryAttributes());
146     }

148     @Test
149     public void testIntegerFieldsAsStrings() {
150         final ExtendedInformation extendedInformation = new ExtendedInformation();
151         final String remainingLoginAttempts = "1";
152         final String maxFailedLoginAttempts = "2";
153         extendedInformation.getRawData().put("remainingloginattempts", remainingLoginAttempts);
154         extendedInformation.getRawData().put("maxfailedloginattempts", maxFailedLoginAttempts);
155         assertEquals(Integer.valueOf(remainingLoginAttempts), (Integer) extendedInformation.getRemainingLoginAttempts());
156         assertEquals(Integer.valueOf(maxFailedLoginAttempts), (Integer) extendedInformation.getMaxLoginAttempts());
157     }
