48    @Test
49    public void testExtendedInformationCSR() {
50      // Test basic function, latest version->latest version
51      ExtendedInformation ei = new ExtendedInformation();
52      ei.setCertificateRequest(
53          Base64.decode(PKCS10.getBytes(StandardCharsets.UTF_8)));
54      assertEquals(
55          PKCS10,
56          new String(
57              Base64.encode(ei.getCertificateRequest()), StandardCharsets.UTF_8));
58      // Test by hardcoding in the base64 encoded data as string, to ensure we can
59      // read such data
60      // (even if the current implementation changed)
61      ExtendedInformation ei2 = new ExtendedInformation();
62      ei2.setMapData("CERTIFICATE_REQUEST", PKCS10);
63      assertEquals(
64          PKCS10,
65          new String(
66              Base64.encode(ei2.getCertificateRequest()),
67              StandardCharsets.UTF_8));
68      // Test null
69      ExtendedInformation ei3 = new ExtendedInformation();
70      ei3.setMapData("FOO", PKCS10);
71      assertNull(ei3.getCertificateRequest());
72      // Test by hardcoding in the binary encoded data, to ensure we can read such
73      // data
74      // As it was stored before EJBCA 6.8.0
75      ExtendedInformation ei4 = new ExtendedInformation();
76      LinkedHashMap<String, Object> map = new LinkedHashMap<String, Object>();
77      map.put(UpgradeableDataHashMap.VERSION, Float.valueOf(4));
78      map.put(
79          "CERTIFICATE_REQUEST",
80          Base64.decode(PKCS10.getBytes(StandardCharsets.UTF_8)));
81      ei4.setData(map);
82      assertEquals(
83          PKCS10,
84          new String(
85              Base64.encode(ei4.getCertificateRequest()),
86              StandardCharsets.UTF_8));
87    }

212   @Test
213   public void testIntegerFieldsAsStrings() {
214     final ExtendedInformation extendedInformation = new ExtendedInformation();
215     final String remainingLoginAttempts = "1";
216     final String maxFailedLoginAttempts = "2";
217     extendedInformation
218         .getRawData()
219         .put("remainingloginattempts", remainingLoginAttempts);
220     extendedInformation
221         .getRawData()
222         .put("maxfailedloginattempts", maxFailedLoginAttempts);
223     assertEquals(
224         Integer.valueOf(remainingLoginAttempts),
225         (Integer) extendedInformation.getRemainingLoginAttempts());
226     assertEquals(
227         Integer.valueOf(maxFailedLoginAttempts),
228         (Integer) extendedInformation.getMaxLoginAttempts());
229   }
