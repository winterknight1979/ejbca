38  /** @version $Id: QcStatementTest.java 26380 2017-08-21 22:41:48Z anatom $ */
39  public class QcStatementTest {
40      /** Logger. */
41    private static Logger log = Logger.getLogger(QcStatementTest.class);
42    /**
43     * @throws CertificateExtensionException fail
44   * @throws IOException Fail
45     */
46    @Test
47    public void testQcStatement()
48        throws CertificateExtensionException, IOException {
49      CertificateProfile prof =
50          new CertificateProfile(
51              CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
52      prof.setUseQCStatement(true);
53      prof.setUseQCEtsiQCCompliance(true);
54      prof.setUseQCEtsiSignatureDevice(true);
55      // QC ETSI type from eIDAS EN 319 412-5, eSign = 0.4.0.1862.1.6.1, eseal =
56      // 0.4.0.1862.1.6.2, web = 0.4.0.1862.1.6.3
57      prof.setQCEtsiType("0.4.0.1862.1.6.1");
58      prof.setQCEtsiPds(
59          Arrays.asList(
60              new PKIDisclosureStatement("http://qcs.localhost/QcPDS", "en")));
61      // id-qcs-pkixQCSyntax-v2 (OID = 1.3.6.1.55.7.11.2) with SematicsID =
62      // 0.4.0.194121.1.2
63      // SemanticsID = "Legal Person", according to eIDAS EN 319 412-1
64      prof.setUsePkixQCSyntaxV2(true);
65      prof.setQCSemanticsId("0.4.0.194121.1.2");
66      QcStatement statement = new QcStatement();
67      byte[] value =
68          statement.getValueEncoded(null, null, prof, null, null, null);
69      @SuppressWarnings("resource")
70      final String dump =
71          ASN1Dump.dumpAsString(new ASN1InputStream(value).readObject(), true);
72      log.info(dump);
73      // Hex dump can be used in Custom Certificate Extensions
74      final String hexEncodedQcStatements = new String(Hex.encode(value));
75      log.info(hexEncodedQcStatements);
76      // Parsing the Hex encoded QCStatements that we creted above, will it parse?
77      final ASN1Sequence seq = (ASN1Sequence) ASN1Sequence.fromByteArray(value);
78      // Dump included IDs
79      // This is just a loop to get all the statement IDs in the QcStatements
80      // extension, so we can view them and count them
81      ArrayList<String> oids = new ArrayList<>();
82      for (int i = 0; i < seq.size(); i++) {
83        final QCStatement qc = QCStatement.getInstance(seq.getObjectAt(i));
84        final ASN1ObjectIdentifier oid = qc.getStatementId();
85        if (oid != null) {
86          oids.add(oid.getId());
87        } else {
88          fail("QC statements have empty statement");
89        }
90      }
91      log.info(oids);
92      // Check that all OIDs we set exist
93      assertEquals("Not all QC statement Ids were included", 5, oids.size());
94      assertTrue(
95          oids.contains(
96              ETSIQCObjectIdentifiers.id_etsi_qcs_QcCompliance.getId()));
97      assertTrue(
98          oids.contains(ETSIQCObjectIdentifiers.id_etsi_qcs_QcSSCD.getId()));
99      assertTrue(
100         oids.contains(
101             RFC3739QCObjectIdentifiers.id_qcs_pkixQCSyntax_v2.getId()));
102     assertTrue(
103         oids.contains(
104             "0.4.0.1862.1.6")); // ETSIQCObjectIdentifiers.id_etsi_qcs_QcType
105     assertTrue(
106         oids.contains(
107             "0.4.0.1862.1.5")); // ETSIQCObjectIdentifiers.id_etsi_qcs_QcPds
108     // Check the values we set
109     assertEquals(
110         "0.4.0.1862.1.6.1",
111         QCStatementExtension.getStatementStringValue(seq, "0.4.0.1862.1.6", 0));
112     assertEquals(
113         "[http://qcs.localhost/QcPDS, en]",
114         QCStatementExtension.getStatementStringValue(seq, "0.4.0.1862.1.5", 0));
115   }
