28  /**
29   * Unit test of {@link AvailableCustomCertificateExtensionsConfiguration}.
30   * see org.cesecore.certificates.certificate.AvailableCustomCertExtensionsConfigTest AvailableCustomCertExtensionsConfigTest (related system test)
31   * @version $Id: AvailableCustomCertificateExtensionsConfigurationUnitTest.java 30621 2018-11-26 09:44:13Z samuellb $
32   */
33  public class AvailableCustomCertificateExtensionsConfigurationUnitTest {
34  
35      private static final Logger log = Logger.getLogger(AvailableCustomCertificateExtensionsConfigurationUnitTest.class);
36  
37      /** Encoded DEROctetString with 2 byte value: AB CD */
38      private static final byte[] EXPECTED_ENCODED_VALUE = { 0x04, 0x02, (byte) 0xAB, (byte) 0xCD };
39  
40      /**
41       * Tests the method that is used for parsing old certextensions.properties files from EJBCA 6.3.x and earlier.
42       * @throws Exception fail 
43       */
44      @Test
45      public void testUpgradeParsing() throws Exception {
46          log.trace(">testUpgradeParsing");
47          final Properties oldFileProperties = new Properties();
48          oldFileProperties.put("id123.oid", " 2.999.123 ");
49          oldFileProperties.put("id123.classpath", " org.cesecore.certificates.certificate.certextensions.BasicCertificateExtension ");
50          oldFileProperties.put("id123.displayname", " My Extension ");
51          oldFileProperties.put("id123.used", " tRuE ");
52          oldFileProperties.put("id123.translatable", " tRuE ");
53          oldFileProperties.put("id123.critical", " FaLsE ");
54          // BasicCertificateExtension properties
55          oldFileProperties.put("id123.property.encoding", " DeRoCtEtStRiNg  ");
56          oldFileProperties.put("id123.property.dynamic", " FaLsE ");
57          oldFileProperties.put("id123.property.value", "aBcD"); // value did not allow padding
58          CertificateExtension upgraded = AvailableCustomCertificateExtensionsConfiguration.getCertificateExtensionFromFile(123, oldFileProperties);
59          log.debug("Properties after upgrade: " + upgraded.getProperties());
60          assertEquals("2.999.123", upgraded.getOID());
61          assertEquals("My Extension", upgraded.getDisplayName());
62          assertTrue("'Required' is the default. Was false.", upgraded.isRequiredFlag());
63          assertFalse("Wrong value of 'Critical'", upgraded.isCriticalFlag());
64          assertEquals("Wrong value of 'translatable' property", true, upgraded.getProperties().get("translatable"));
65          final EndEntityInformation dummyEndEntity = new EndEntityInformation();
66          dummyEndEntity.setDN("CN=Dummy AvailableCustomCertificateExtensionsConfigurationUnitTest");
67          final byte[] valueEncoded = upgraded.getValueEncoded(dummyEndEntity, null, new CertificateProfile(), /*userPublicKey*/null, /*caPublicKey*/null, /*validity*/null);
68          assertArrayEquals("Wrong encoded value.", EXPECTED_ENCODED_VALUE, valueEncoded);
69          log.trace("<testUpgradeParsing");
70          
71      }
