278     private PKCS10CertificationRequest createP10(final String subjectDN) throws IOException, OperatorCreationException {
279         // Create a P10 with extensions, in this case altNames with a DNS name
280         ASN1EncodableVector altnameattr = new ASN1EncodableVector();
281         altnameattr.add(PKCSObjectIdentifiers.pkcs_9_at_extensionRequest);
282         // AltNames
283         // String[] namearray = altnames.split(",");
284         GeneralNames san = CertTools.getGeneralNamesFromAltName("dNSName=foo1.bar.com");
285         ExtensionsGenerator extgen = new ExtensionsGenerator();
286         extgen.addExtension(Extension.subjectAlternativeName, false, san );
287         Extensions exts = extgen.generate();
288         altnameattr.add(new DERSet(exts));
289 
290         // Add a challenge password as well
291         ASN1EncodableVector pwdattr = new ASN1EncodableVector();
292         pwdattr.add(PKCSObjectIdentifiers.pkcs_9_at_challengePassword); 
293         ASN1EncodableVector pwdvalues = new ASN1EncodableVector();
294         pwdvalues.add(new DERUTF8String("foo123"));
295         pwdattr.add(new DERSet(pwdvalues));
296 
297         // Complete the Attribute section of the request, the set (Attributes)
298         // contains one sequence (Attribute)
299         ASN1EncodableVector v = new ASN1EncodableVector();
300         v.add(new DERSequence(altnameattr));
301         v.add(new DERSequence(pwdattr));
302         DERSet attributes = new DERSet(v);
303 
304         // Create the PKCS10
305         X500Name dn = new X500Name(subjectDN);
306         PKCS10CertificationRequest basicpkcs10 = CertTools.genPKCS10CertificationRequest("SHA1WithRSA", dn, 
307                 keyPair.getPublic(), attributes, keyPair.getPrivate(), null);
308         return basicpkcs10;
309     }
