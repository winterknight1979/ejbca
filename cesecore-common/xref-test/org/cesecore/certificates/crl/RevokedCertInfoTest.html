22  import org.cesecore.util.CompressedCollection;

26  /**
27   * @version $Id: RevokedCertInfoTest.java 34193 2020-01-07 15:18:15Z samuellb $

29  public class RevokedCertInfoTest {
30  
31      private static final RevokedCertInfo REVINFO_1_NOTREVOKED = revCertInfo("2017-01-15", RevocationReasons.NOT_REVOKED);
32      private static final RevokedCertInfo REVINFO_2_ONHOLD = revCertInfo("2017-01-31", RevocationReasons.CERTIFICATEHOLD);
33      private static final RevokedCertInfo REVINFO_3_UNSPECIFIED = revCertInfo("2017-02-20", RevocationReasons.UNSPECIFIED);
34      private static final RevokedCertInfo REVINFO_4_UNSPECIFIED = revCertInfo("2017-03-31", RevocationReasons.UNSPECIFIED);
35      private static final RevokedCertInfo REVINFO_5_REMOVEFROMCRL = revCertInfo("2017-05-31", RevocationReasons.REMOVEFROMCRL);
36      
37      private static final RevokedCertInfo revCertInfo(final String revocationDate, final RevocationReasons reason) {
38          return new RevokedCertInfo(new byte[] { 1,2,3,4 }, BigInteger.valueOf(0x1234ABCDL).toByteArray(), date(revocationDate), reason.getDatabaseValue(), date("2017-12-31"));
39      }
40      
41      private static final long date(final String ymd) {
42          try {
43              return ValidityDate.parseAsIso8601(ymd).getTime();
44          } catch (ParseException e) {
45              throw new IllegalArgumentException(e);
46          }
47      }
48      
49      private static void assertRCIEquals(final String message, final RevokedCertInfo expected, final RevokedCertInfo actual) {
50          assertEquals(message, expected.getRevocationDate(), actual.getRevocationDate());
51          assertEquals(message, expected.getReason(), actual.getReason());
52      }
53      
54      @Test
55      public void mergeEmpty() {
56          final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
57          a.add(REVINFO_1_NOTREVOKED);
58          final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
59          assertSame("Empty 'b' collection should cause 'a' to be returned.", a, RevokedCertInfo.mergeByDateAndStatus(a, b, 0));
60      }
61      
62      @Test
63      public void mergeWithDuplicates() {
64          final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
65          final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
66          b.add(REVINFO_1_NOTREVOKED);
67          b.add(REVINFO_4_UNSPECIFIED); // permanent revocation => wins
68          b.add(REVINFO_5_REMOVEFROMCRL);
69          final Collection<RevokedCertInfo> res = RevokedCertInfo.mergeByDateAndStatus(a, b, 0);
70          assertEquals("Items should have been de-duplicated.", 1, res.size());
71          assertRCIEquals("Should contain entry REVINFO_4_UNSPECIFIED.", REVINFO_4_UNSPECIFIED, res.iterator().next());
72      }
73      
74      @Test
75      public void mergeWithDuplicates2() {
76          final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
77          a.add(REVINFO_4_UNSPECIFIED); // permanent revocation => wins
78          final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
79          b.add(REVINFO_1_NOTREVOKED);
80          b.add(REVINFO_5_REMOVEFROMCRL);
81          final Collection<RevokedCertInfo> res = RevokedCertInfo.mergeByDateAndStatus(a, b, System.currentTimeMillis());
82          assertEquals("Items should have been de-duplicated.", 1, res.size());
83          assertRCIEquals("Should contain entry REVINFO_4_UNSPECIFIED.", REVINFO_4_UNSPECIFIED, res.iterator().next());
84      }
85      
86      @Test
87      public void mergeWithDuplicates3() {
88          final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
89          final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
90          b.add(REVINFO_2_ONHOLD); // only temporary revocations, and most recent entry => wins
91          b.add(REVINFO_1_NOTREVOKED);
92          final Collection<RevokedCertInfo> res = RevokedCertInfo.mergeByDateAndStatus(a, b, 0);
93          assertEquals("Items should have been de-duplicated.", 1, res.size());
94          assertRCIEquals("Should contain entry REVINFO_2_ONHOLD.", REVINFO_2_ONHOLD, res.iterator().next());
95      }
96      
97      @Test
98      public void mergeWithDuplicates4() {
99          final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
100         final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
101         b.add(REVINFO_4_UNSPECIFIED);
102         b.add(REVINFO_3_UNSPECIFIED); // permanent revocation with oldest date wins
103         final Collection<RevokedCertInfo> res = RevokedCertInfo.mergeByDateAndStatus(a, b, 0);
104         assertEquals("Items should have been de-duplicated.", 1, res.size());
105         assertRCIEquals("Should contain entry REVINFO_3_UNSPECIFIED.", REVINFO_3_UNSPECIFIED, res.iterator().next());
106     }
107     
108     @Test
109     public void mergeWithRemoveFromCrl1() {
110         final CompressedCollection<RevokedCertInfo> a = new CompressedCollection<>(RevokedCertInfo.class);
111         final CompressedCollection<RevokedCertInfo> b = new CompressedCollection<>(RevokedCertInfo.class);
112         b.add(REVINFO_5_REMOVEFROMCRL); // should be skipped
113         Collection<RevokedCertInfo> res = RevokedCertInfo.mergeByDateAndStatus(a, b, 0);
114         assertEquals("REMOVEFROMCRL should be removed in Base CRL.", 0, res.size());
115         
116         res = RevokedCertInfo.mergeByDateAndStatus(a, b, date("2001-01-01"));
117         assertEquals("REMOVEFROMCRL should not be removed when Base CRL is older than revocation date.", 1, res.size());
118         assertRCIEquals("Should contain entry REVINFO_5_REMOVEFROMCRL", REVINFO_5_REMOVEFROMCRL, res.iterator().next());
119         
120         res = RevokedCertInfo.mergeByDateAndStatus(a, b, date("2017-12-31"));
121         assertEquals("REMOVEFROMCRL should be removed when Base CRL is more recent than revocation date.", 0, res.size());
122     }
