13  package org.cesecore.internal;
14  
15  import java.util.List;
16  import java.util.Map;
17  
18  /**
19   * Object and name to id lookup cache interface.
20   * 
21   * @version $Id: CommonCache.java 28332 2018-02-20 14:40:52Z anatom $
22   * @param <T> Type
23   */
24  public interface CommonCache<T> {
25  
26      /** @param id ID
27       * @return cached Object or null if none is present. */
28      T getEntry(Integer id);
29  
30      /** @param id ID
31       * @return cached Object or null if none is present. */
32      T getEntry(int id);
33  
34      /** @param id ID
35       * @return true when the cache for this object has expired or the token is non-existing. */
36      boolean shouldCheckForUpdates(int id);
37  
38      /** Update the cache with the current version read from the database. 
39       *
40       * @param id id of the object, typically database id
41       * @param digest typically getProtectString(0).hashCode() of the object;
42       * @param name String name of the object, typically database name
43       * @param object the object to cache
44       */
45      void updateWith(int id, int digest, String name, T object);
46  
47      /** this method exposes the loginc inside updateWith that determines if an object is the same as the cached object.
48       * If the objects are the same, updateWith will not update the object as an optimization. This method can be used to skip expensive object creation
49       * if the objects are anyway the same and updateWith will not end up doing anything
50       * @param id id of the object, typically database id
51       * @param digest typically getProtectString(0).hashCode() of the object;
52       * @return true if a call to updateWith will perform an actual cache update
53       */
54      boolean willUpdate(int id, int digest);
55  
56      /** Remove the specified entry from the cache and mapping if it exists. 
57       * @param id ID*/
58      void removeEntry(int id);
59  
60      /** Provides functionality of an IdToNameMap. 
61       * @param id ID
62       * @return return the name from the entry, given the id, or null if it does not exist.
63       */
64      String getName(int id);
65  
66      /** @return a copy of the id to name map as a name to id map */
67      Map<String, Integer> getNameToIdMap();
68  
69      /** Remove all references from this cache */
70      void flush();
71      
72      /** Replace cache with the entries referenced in keys 
73       * @param keys keys*/
74      void replaceCacheWith(List<Integer> keys);
75  }
