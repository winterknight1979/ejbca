13  package org.cesecore.certificates.certificate;
14  
15  import org.cesecore.CesecoreException;
16  
17  /**
18   * Error due to malformed key. The cause of failure can be related to illegal key length etc.
19   *
20   * @version $Id: IllegalKeyException.java 18578 2014-03-10 09:44:39Z anatom $
21   */
22  public class IllegalKeyException extends CesecoreException {
23  
24      private static final long serialVersionUID = -3144774253953346584L;
25      
26      public IllegalKeyException(String message, Throwable cause) {
27          super(message, cause);
28      }
29     
30      /**
31       * Constructor used to create exception with an errormessage. Calls the same constructor in
32       * baseclass <code>Exception</code>.
33       *
34       * @param message Human redable error message, can not be NULL.
35       */
36      public IllegalKeyException(String message) {
37          super(message);
38      }
39      /**
40       * Constructs an instance of <code>IllegalKeyException</code> with the specified cause.
41       * @param e the detail message.
42       */
43      public IllegalKeyException(Exception e) {
44          super(e);
45      }
46  }
