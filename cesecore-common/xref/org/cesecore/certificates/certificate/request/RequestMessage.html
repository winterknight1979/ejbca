13  package org.cesecore.certificates.certificate.request;
14  
15  import java.io.Serializable;
16  import java.math.BigInteger;
17  import java.security.InvalidKeyException;
18  import java.security.NoSuchAlgorithmException;
19  import java.security.NoSuchProviderException;
20  import java.security.PrivateKey;
21  import java.security.PublicKey;
22  import java.security.cert.Certificate;
23  import java.util.Date;
24  import java.util.List;
25  
26  import org.bouncycastle.asn1.x500.X500Name;
27  import org.bouncycastle.asn1.x509.Extensions;
28  
29  
30  /**
31   * Base interface for request messages sent to the CA. Implementors of this interface must also
32   * implement Serializable if they are to be sent to any EJB business methods.
33   *
34   * @version $Id: RequestMessage.java 28201 2018-02-07 08:33:29Z andresjakobs $
35   */
36  public interface RequestMessage extends Serializable {    
37      /**
38       * Get the username used to request a certificate from EJBCA.
39       *
40       * @return The username from the certification request.
41       */
42      String getUsername();
43  
44      /**
45       * Get the password used to request a certificate from EJBCA.
46       *
47       * @return The password from the certification request.
48       */
49      String getPassword();
50  
51      /**
52       * Gets the issuer DN if contained in the request (the CA the request is targeted at).
53       *
54       * @return issuerDN of receiving CA or null.
55       */
56      String getIssuerDN();
57  
58      /**
59       * Gets the number (of CA cert) from IssuerAndSerialNumber. Combined with getIssuerDN to identify
60       * the CA-certificate of the CA the request is targeted for.
61       *
62       * @return serial number of CA certificate for CA issuing CRL or null.
63       */
64      BigInteger getSerialNo();
65      
66      /**
67       * Gets the requested DN if contained in the request (the desired DN for the user).
68       *
69       * @return requested DN or null.
70       */
71      String getRequestDN();
72  
73      /**
74       * Gets the requested DN in the pure form of an X508Name (the desired DN for the user).
75       *
76       * @return requested DN or null.
77       */
78      X500Name getRequestX500Name();
79  
80      /**
81       * Gets the requested altNames if contained in the request (the desired altNames for the user).
82       *
83       * @return requested altNames or null.
84       */
85      String getRequestAltNames();
86      
87      /**
88       * Gets a validity date from the request, if the request contains a desired validity.
89       * The requested validity may, or may not be used, it depends if allowValidityOverride is set in 
90       * the certificate profile.
91       * 
92       * @return A date now or in the future for notBefore validity in the certificate, or null if no desired validity is in the certificate.
93       */
94  	Date getRequestValidityNotBefore();
95  	
96      /**
97       * Gets a validity date from the request, if the request contains a desired validity.
98       * The requested validity may, or may not be used, it depends if allowValidityOverride is set in 
99       * the certificate profile.
100      * 
101      * @return A date in the future for notAfter validity in the certificate, or null if no desired validity is in the certificate.
102      */
103 	Date getRequestValidityNotAfter();
104 
105 	/** 
106 	 * Gets any requested extensions, if the request message type is able to contain request extensions
107 	 * and if there are any. Requested extensions are (currently) in the form of X509Extensions.
108 	 * 
109 	 * @return X509Extensions or null
110 	 */
111 	Extensions getRequestExtensions();
112 
113     /**
114      * Gets the issuer DN (of CA cert) from IssuerAndSerialNumber when this is a CRL request.
115      *
116      * @return issuerDN of CA issuing CRL or null.
117      */
118     String getCRLIssuerDN();
119 
120     /**
121      * Gets the number (of CA cert) from IssuerAndSerialNumber when this is a CRL request.
122      *
123      * @return serial number of CA certificate for CA issuing CRL or null.
124      */
125     BigInteger getCRLSerialNo();
126 
127     /**
128      * Get the key from a certification request.
129      *
130      * @return The key from a certification request. Returns null if request content hasn't been set, or there is no public key in the request (can be ok if server generated keys are requested).
131      *
132      * @throws InvalidKeyException If the key is invalid.
133      * @throws NoSuchProviderException if there is an error with the Provider.
134      * @throws NoSuchAlgorithmException if the key uses an unhandled algorithm.
135      */
136     PublicKey getRequestPublicKey()
137             throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException;
138 
139     /**
140      * Verifies signatures, popo etc on the request message. If verification fails the request
141      * should be considered invalid.
142      *
143      * @return true if verification was successful, false if it failed.
144      *
145      * @throws InvalidKeyException If the key used for verification is invalid.
146      * @throws NoSuchProviderException if there is an error with the Provider.
147      * @throws NoSuchAlgorithmException if the signature on the request is done with an unhandled
148      *         algorithm.
149      */
150     boolean verify()
151             throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException;
152 
153     /**
154      * indicates if this message needs recipients and private key to verify, decrypt etc. If
155      * this returns true, setKeyInfo() should be called.
156      *
157      * @return True if and private key is needed.
158      */
159     boolean requireKeyInfo();
160 
161     /**
162      * Sets the and private key needed to decrypt/verify the message. Must be set if
163      * requireKeyInfo() returns true.
164      *
165      * @param cert certificate containing the key.
166      * @param key private key.
167      * @param provider the provider to use, if the private key is on a HSM you must use a special provider. If null is given, the default BC provider is used.
168      *
169      * @see #requireKeyInfo()
170      */
171     void setKeyInfo(Certificate cert, PrivateKey key, String provider);
172 
173     /**
174      * Returns an error number after an error has occurred processing the request
175      *
176      * @return class specific error number
177      */
178     int getErrorNo();
179 
180     /**
181      * Returns an error message after an error has occurred processing the request
182      *
183      * @return class specific error message
184      */
185     String getErrorText();
186 
187     /**
188      * Returns a senderNonce if present in the request
189      *
190      * @return senderNonce as a string of base64 encoded bytes
191      */
192     String getSenderNonce();
193 
194     /**
195      * Returns a transaction identifier if present in the request
196      *
197      * @return transaction id
198      */
199     String getTransactionId();
200 
201     /**
202      * Returns requesters key info, key id or similar
203      *
204      * @return request key info
205      */
206     byte[] getRequestKeyInfo();
207     
208     /**
209      * Returns the name of the preferred Digest algorithm to be used in the response if applicable.
210      * Defaults to CMSSignedGenerator.DIGEST_SHA1 for normal messages.
211      *  
212      * @return oid of digest algorithm ex CMSSignedGenerator.DIGEST_SHA1, SHA256 etc
213      */
214     String getPreferredDigestAlg(); 
215     
216     
217     /** If the CA certificate should be included in the response or not, default to true = yes.
218      * Not applicable for all request/response types.
219      * 
220      * @return true or false
221      */
222     boolean includeCACert();
223 
224     /** Sometimes (CMP) the response identifier sent depends on which request identifier was used, 
225      * even if the messages themselves are the same messages.
226      * 
227      * @return which type of request message this response is in response to
228      */ 
229     int getRequestType();
230     
231     /**
232      * For some types of request-responses there is a need for a requestId to match the request and the
233      * response together.
234      * @return the id from the request matching to this response
235      */
236     int getRequestId();
237     
238     /**
239      * Sets the private key needed to create a response message.
240      *
241      * @param key private key.
242      * @param provider the provider to use, if the private key is on a HSM you must use a special provider. If null is given, the default BC provider is used.
243      */
244     void setResponseKeyInfo(PrivateKey key, String provider);
245 
246     /**
247      * Gets the list of additional CA certificates
248      * (i.e. to be appended to the user certificates CA certificate returned in the CMP response message caPubs field).
249      * @return the list of CA certificates.
250      */
251     public List<Certificate> getAdditionalCaCertificates();
252 
253     /**
254      * Sets the list of additional CA certificates 
255      * (i.e. to be appended to the user certificates CA certificate returned in the CMP response message caPubs field).
256      * @param certificates the list of CA certificates.
257      */
258     public void setAdditionalCaCertificates(final List<Certificate> certificates);
259     
260     /**
261      * Gets the list of additional CA certificates to be appended to the PKI response message extraCerts field.
262      * @return the list of CA certificates.
263      */
264     public List<Certificate> getAdditionalExtraCertsCertificates();
265     
266     /**
267      * Sets the list of additional CA certificates to be appended to the PKI response message extraCerts field.
268      * @param certificates the list of CA certificates.
269      */
270     public void setAdditionalExtraCertsCertificates(final List<Certificate> certificates);
271 }
