96    private void init() {
97      try {
98        CVCObject parsedObject;
99        parsedObject = CertificateParser.parseCVCObject(cvcmsg);
100       if (parsedObject instanceof CVCertificate) {
101         cvcert = (CVCertificate) parsedObject;
102       } else if (parsedObject instanceof CVCAuthenticatedRequest) {
103         CVCAuthenticatedRequest authreq =
104             (CVCAuthenticatedRequest) parsedObject;
105         cvcert = authreq.getRequest();
106       }
107     } catch (ParseException e) {
108       LOG.error("Error in init for CVC request: ", e);
109       throw new IllegalArgumentException(e);
110     } catch (ConstructionException e) {
111       LOG.error("Error in init for CVC request: ", e);
112       throw new IllegalArgumentException(e);
113     } catch (NoSuchFieldException e) {
114       LOG.error("Error in init for CVC request: ", e);
115       throw new IllegalArgumentException(e);
116     }
117   }

119   @Override
120   public PublicKey getRequestPublicKey()
121       throws InvalidKeyException, NoSuchAlgorithmException,
122           NoSuchProviderException {
123     try {
124       if (cvcert == null) {
125         init();
126       }
127     } catch (IllegalArgumentException e) {
128       LOG.error("CVC not inited!");
129       return null;
130     }
131 
132     PublicKey pk;
133     try {
134       pk = cvcert.getCertificateBody().getPublicKey();
135     } catch (NoSuchFieldException e) {
136       throw new InvalidKeyException(e);
137     }
138     return pk;
139   }

164   @Override
165   public String getUsername() {
166     if (username != null) {
167       return username;
168     }
169     String subject = null;
170     try {
171       HolderReferenceField hr =
172           cvcert.getCertificateBody().getHolderReference();
173       subject = hr.getMnemonic() + hr.getCountry();
174     } catch (NoSuchFieldException e) {
175       LOG.error(e);
176     }
177     return subject;
178   }

251   private boolean verify(final PublicKey pubKey)
252       throws InvalidKeyException, NoSuchAlgorithmException,
253           NoSuchProviderException {
254     LOG.trace(">verify()");
255 
256     boolean ret = false;
257 
258     try {
259       CardVerifiableCertificate cc = getCardVerifiableCertificate();
260       if (cc != null) {
261         if (pubKey == null) {
262           cc.verify(cvcert.getCertificateBody().getPublicKey());
263           ret = true; // If we came here verification was successful
264         } else {
265           cc.verify(pubKey);
266           ret = true; // If we came here verification was successful
267         }
268       }
269     } catch (NoSuchFieldException e) {
270       LOG.error("CVC error!", e);
271     } catch (InvalidKeyException e) {
272       LOG.error("Error in CVC-request:", e);
273       throw e;
274     } catch (CertificateException e) {
275       LOG.error("Error in CVC-signature:", e);
276     } catch (SignatureException e) {
277       LOG.error("Error in CVC-signature:", e);
278     }
279 
280     LOG.trace("<verify()");
281 
282     return ret;
283   }

378    * Specific to CVC request messages, EAC requests contains a sequence.
379    *
380    * @return sequence

382   public String getKeySequence() {
383     String ret = null;
384     try {
385       if (cvcert.getCertificateBody().getHolderReference() != null) {
386         ret = cvcert.getCertificateBody().getHolderReference().getSequence();
387       }
388     } catch (NoSuchFieldException e) {
389       // No sequence found...
390     }
391     return ret;
392   }

394   private CardVerifiableCertificate getCardVerifiableCertificate() {
395     try {
396       if (cvcert == null) {
397         init();
398       }
399     } catch (IllegalArgumentException e) {
400       LOG.error("CVC not inited!", e);
401       return null;
402     }
403     CardVerifiableCertificate cc = new CardVerifiableCertificate(cvcert);
404     return cc;
405   }
