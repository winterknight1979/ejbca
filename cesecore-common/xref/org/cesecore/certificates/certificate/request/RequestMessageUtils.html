60  	/** Tries to parse the byte array to create a request message of the correct type.
61  	 * Currently handles PKCS10 request messages and CVC request messages.
62  	 * @param request byte array
63  	 * 
64  	 * @return IRequestMessage
65  	 */
66  	public static RequestMessage parseRequestMessage(byte[] request) {
67  		RequestMessage ret = null;
68  		try {
69  			ret = genPKCS10RequestMessage(request);			
70  		} catch (IllegalArgumentException e) {
71  			log.debug("Can not parse PKCS10 request, trying CVC instead: "+ e.getMessage());
72  			ret = genCVCRequestMessage(request);
73  		}
74  		return ret;
75  	}

77  	public static PKCS10RequestMessage genPKCS10RequestMessage(byte[] bytes) {
78  		byte[] buffer = getDecodedBytes(bytes);
79  		if (buffer == null) {
80  			return null;
81  		}		
82  		return new PKCS10RequestMessage(buffer);
83  	} // genPKCS10RequestMessageFromPEM

86  	public static CVCRequestMessage genCVCRequestMessage(byte[] bytes) { 
87  		byte[] buffer = getDecodedBytes(bytes);
88  		if (buffer == null) {
89  			return null;
90  		}		
91  		return new CVCRequestMessage(buffer);
92  	} // genCvcRequestMessageFromPEM

94  	/** Tries to get decoded, if needed, bytes from a certificate request or certificate
95  	 * 
96  	 * @param bytes pem (with headers), plain base64, or binary bytes with a CSR of certificate 
97  	 * @return binary bytes
98  	 */
99  	public static byte[] getDecodedBytes(byte[] bytes) {
100 		byte[] buffer = null;
101 		try {
102 			 buffer = getRequestBytes(bytes); 
103 		} catch (IOException e) {
104 			log.debug("Message not base64 encoded? Trying as binary: "+e.getMessage());
105 			buffer = bytes;
106 		}
107 		return buffer;
108 	}

110 	/** Tries to get decoded bytes from a certificate request or certificate
111 	 * 
112 	 * @param b64Encoded pem (with headers) or plain base64 with a CSR of certificate 
113 	 * @return binary bytes
114 	 * @throws IOException on error
115 	 */
116 	public static byte[] getRequestBytes(byte[] b64Encoded) throws IOException {
117 		byte[] buffer = null;
118 		if (b64Encoded != null && b64Encoded.length > 0) {
119 		    String str = new String(b64Encoded);
120             // A real PKCS10 PEM request
121             String beginKey = CertTools.BEGIN_CERTIFICATE_REQUEST;
122             String endKey = CertTools.END_CERTIFICATE_REQUEST;
123             if (!str.contains(beginKey)) {
124                 // Keytool PKCS10 PEM request
125                 beginKey = CertTools.BEGIN_KEYTOOL_CERTIFICATE_REQUEST;
126                 endKey = CertTools.END_KEYTOOL_CERTIFICATE_REQUEST;
127                 if (!str.contains(beginKey)) {
128                     // CSR can be a PEM encoded certificate instead of "certificate request"
129                     beginKey = CertTools.BEGIN_CERTIFICATE;
130                     endKey = CertTools.END_CERTIFICATE;
131                     if (!str.contains(beginKey)) {
132                         // IE PKCS10 Base64 coded request
133                         try {
134                             buffer = Base64.decode(b64Encoded);
135                             if (buffer == null) {
136                                 throw new IOException("Base64 decode of buffer returns null");
137                             }
138                         } catch (DecoderException de) {
139                             throw new IOException("Base64 decode fails, message not base64 encoded: "+de.getMessage());
140                         }
141                     }
142                 }
143             }
144             if (buffer == null) {
145                 buffer = FileTools.getBytesFromPEM(b64Encoded, beginKey, endKey);                
146             }
147 		} else {
148             throw new IOException("Base64 decode fails, message is empty");		    
149 		}
150 		return buffer;
151 	}

153     public static RequestMessage getRequestMessageFromType(final String username, final String password, final String req, final int reqType)
154             throws SignRequestSignatureException, InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException, IOException,
155             SignatureException, InvalidKeySpecException, ParseException, ConstructionException, NoSuchFieldException {
156 	    RequestMessage ret = null;
157         if (reqType == CertificateConstants.CERT_REQ_TYPE_PKCS10) {
158             final PKCS10RequestMessage pkcs10RequestMessage = RequestMessageUtils.genPKCS10RequestMessage(req.getBytes());
159             pkcs10RequestMessage.setUsername(username);
160             pkcs10RequestMessage.setPassword(password);
161             ret = pkcs10RequestMessage;
162         } else if (reqType == CertificateConstants.CERT_REQ_TYPE_SPKAC) {
163             byte[] reqBytes = req.getBytes();
164             if (reqBytes != null) {
165                 if (log.isDebugEnabled()) {
166                     log.debug("Received NS request: "+new String(reqBytes));
167                 }
168                 byte[] buffer = Base64.decode(reqBytes);
169                 if (buffer == null) {
170                     return null;
171                 }
172                 ASN1InputStream in = new ASN1InputStream(new ByteArrayInputStream(buffer));
173                 ASN1Sequence spkacSeq = (ASN1Sequence) in.readObject();
174                 in.close();
175                 NetscapeCertRequest nscr = new NetscapeCertRequest(spkacSeq);
176                 // Verify POPO, we don't care about the challenge, it's not important.
177                 nscr.setChallenge("challenge");
178                 if (nscr.verify("challenge") == false) {
179                     if (log.isDebugEnabled()) {
180                         log.debug("SPKAC POPO verification Failed");
181                     }
182                     throw new SignRequestSignatureException("Invalid signature in NetscapeCertRequest, popo-verification failed.");
183                 }
184                 if (log.isDebugEnabled()) {
185                     log.debug("POPO verification successful");
186                 }
187                 PublicKey pubKey = nscr.getPublicKey();
188                 ret = new SimpleRequestMessage(pubKey, username, password);
189             }       
190         } else if (reqType == CertificateConstants.CERT_REQ_TYPE_CRMF) {
191             final byte[] certificateRequestMessages = Base64.decode(req.getBytes());
192             final CertReqMsg certReqMsg = CertReqMsg.getInstance(((ASN1Sequence)ASN1Sequence.fromByteArray(certificateRequestMessages)).getObjectAt(0));
193             final JcaCertificateRequestMessage jcrm = new JcaCertificateRequestMessage(certReqMsg);
194             try {
195                 final PublicKey publicKey = jcrm.getPublicKey();
196                 if (jcrm.hasProofOfPossession()) {
197                     switch (jcrm.getProofOfPossessionType()) {
198                     case JcaCertificateRequestMessage.popRaVerified: {
199                         // The requestor claims that it is verified by an RA
200                         break;
201                     }
202                     case JcaCertificateRequestMessage.popSigningKey: {
203                         // RFC 4211 Section 4.1
204                         final POPOSigningKey popoSigningKey = POPOSigningKey.getInstance(jcrm.toASN1Structure().getPopo().getObject());
205                         if (log.isDebugEnabled()) {
206                             if (popoSigningKey!=null) {
207                                 log.debug("CRMF POPOSigningKey poposkInput:                      " + popoSigningKey.getPoposkInput());
208                                 if (popoSigningKey.getPoposkInput()!=null) {
209                                     log.debug("CRMF POPOSigningKey poposkInput PublicKey:            " + popoSigningKey.getPoposkInput().getPublicKey());
210                                     log.debug("CRMF POPOSigningKey poposkInput PublicKeyMAC:         " + popoSigningKey.getPoposkInput().getPublicKeyMAC());
211                                 }
212                                 log.debug("CRMF POPOSigningKey algorithmIdentifier.algorithm.id: " + popoSigningKey.getAlgorithmIdentifier().getAlgorithm().getId());
213                                 log.debug("CRMF POPOSigningKey signature:                        " + popoSigningKey.getSignature());
214                             } else {
215                                 log.debug("CRMF POPOSigningKey is not defined even though POP type is popSigningKey. Validation will fail.");
216                             }
217                         }
218                         final ContentVerifierProvider cvp = CertTools.genContentVerifierProvider(publicKey);
219                         // Work around for bug in BC where jcrm.hasSigningKeyProofOfPossessionWithPKMAC() will throw NPE if PoposkInput is null
220                         if (popoSigningKey.getPoposkInput()!=null && jcrm.hasSigningKeyProofOfPossessionWithPKMAC()) {
221                             final PKMACBuilder pkmacBuilder = new PKMACBuilder(new JcePKMACValuesCalculator().setProvider(BouncyCastleProvider.PROVIDER_NAME));
222                             if (!jcrm.isValidSigningKeyPOP(cvp, pkmacBuilder, password.toCharArray())) {
223                                 throw new SignRequestSignatureException("CRMF POP with PKMAC failed signature or MAC validation.");
224                             } else {
225                                 if (log.isDebugEnabled()) {
226                                     log.debug("CRMF POP with PKMAC passed signature and PKMAC validation.");
227                                 }
228                             }
229                         } else {
230                             if (!jcrm.isValidSigningKeyPOP(cvp)) {
231                                 throw new SignRequestSignatureException("CRMF POP failed signature validation.");
232                             } else {
233                                 if (log.isDebugEnabled()) {
234                                     log.debug("CRMF POP passed signature validation.");
235                                 }
236                             }
237                         }
238                         break;
239                     }
240                     case JcaCertificateRequestMessage.popKeyEncipherment: {
241                         // RFC 4211 Section 4.2 (Not implemented)
242                         log.info("CRMF RFC4211 Section 4.2 KeyEncipherment POP validation is not implemented. Will try to use the request's public key anyway.");
243                         break;
244                     }
245                     case JcaCertificateRequestMessage.popKeyAgreement: {
246                         // RFC 4211 Section 4.3 (Not implemented)
247                         log.info("CRMF RFC4211 Section 4.3 KeyAgreement POP validation is not implemented. Will try to use the request's public key anyway.");
248                         break;
249                     }
250                     default:
251                         throw new SignRequestSignatureException("CRMF POP of type "+jcrm.getProofOfPossessionType()+" is unknown.");
252                     }
253                 }

255                 simpleRequestMessage.setRequestExtensions(jcrm.getCertTemplate().getExtensions());
256                 ret = simpleRequestMessage;
257             } catch (CRMFException e) {
258                 throw new SignRequestSignatureException("CRMF POP verification failed.", e);
259             } catch (OperatorCreationException e) {
260                 throw new SignRequestSignatureException("CRMF POP verification failed.", e);
261             }
262         } else if (reqType == CertificateConstants.CERT_REQ_TYPE_PUBLICKEY) {
263             byte[] request;
264             // Request can be Base64 encoded or in PEM format
265             try {
266                 request = FileTools.getBytesFromPEM(req.getBytes(), CertTools.BEGIN_PUBLIC_KEY, CertTools.END_PUBLIC_KEY);
267             } catch (IOException ex) {
268                 try {
269                     request = Base64.decode(req.getBytes());
270                     if (request == null) {
271                         throw new IOException("Base64 decode of buffer returns null");
272                     }
273                 } catch (DecoderException de) {
274                     throw new IOException("Base64 decode fails, message not base64 encoded: " + de.getMessage());
275                 }
276             }
277             final PublicKey pubKey = KeyTools.getPublicKeyFromBytes(request);
278             ret = new SimpleRequestMessage(pubKey, username, password);
279         } else if (reqType == CertificateConstants.CERT_REQ_TYPE_CVC) {
280             CVCObject parsedObject = CertificateParser.parseCVCObject(Base64.decode(req.getBytes()));
281             // We will handle both the case if the request is an authenticated request, i.e. with an outer signature
282             // and when the request is missing the (optional) outer signature.
283             CVCertificate cvccert = null;
284             if (parsedObject instanceof CVCAuthenticatedRequest) {
285                 CVCAuthenticatedRequest cvcreq = (CVCAuthenticatedRequest)parsedObject;
286                 cvccert = cvcreq.getRequest();
287             } else {
288                 cvccert = (CVCertificate)parsedObject;
289             }

291             reqmsg.setUsername(username);
292             reqmsg.setPassword(password);
293             // Popo is really actually verified by the CA (in SignSessionBean) as well
294             if (reqmsg.verify() == false) {
295                 if (log.isDebugEnabled()) {
296                     log.debug("CVC POPO verification Failed");
297                 }
298                 throw new SignRequestSignatureException("Invalid inner signature in CVCRequest, popo-verification failed.");
299             } else {
300                 if (log.isDebugEnabled()) {
301                     log.debug("POPO verification successful");
302                 }
303             }
304             ret = reqmsg;
305         }
306         return ret;
