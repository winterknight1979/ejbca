60    /**
61     * Tries to parse the byte array to create a request message of the correct
62     * type. Currently handles PKCS10 request messages and CVC request messages.
63     *
64     * @param request byte array
65     * @return IRequestMessage
66     */
67    public static RequestMessage parseRequestMessage(final byte[] request) {
68      RequestMessage ret = null;
69      try {
70        ret = genPKCS10RequestMessage(request);
71      } catch (IllegalArgumentException e) {
72        LOG.debug(
73            "Can not parse PKCS10 request, trying CVC instead: "
74                + e.getMessage());
75        ret = genCVCRequestMessage(request);
76      }
77      return ret;
78    }

84    public static PKCS10RequestMessage genPKCS10RequestMessage(
85            final byte[] bytes) {
86      byte[] buffer = getDecodedBytes(bytes);
87      if (buffer == null) {
88        return null;
89      }
90      return new PKCS10RequestMessage(buffer);

105   /**
106    * Tries to get decoded, if needed, bytes from a certificate request or
107    * certificate.
108    *
109    * @param bytes pem (with headers), plain base64, or binary bytes with a CSR
110    *     of certificate
111    * @return binary bytes
112    */
113   public static byte[] getDecodedBytes(final byte[] bytes) {
114     byte[] buffer = null;
115     try {
116       buffer = getRequestBytes(bytes);
117     } catch (IOException e) {
118       LOG.debug(
119           "Message not base64 encoded? Trying as binary: " + e.getMessage());
120       buffer = bytes;
121     }
122     return buffer;
123   }
124 
125   /**
126    * Tries to get decoded bytes from a certificate request or certificate.
127    *
128    * @param b64Encoded pem (with headers) or plain base64 with a CSR of
129    *     certificate
130    * @return binary bytes
131    * @throws IOException on error
132    */
133   public static byte[] getRequestBytes(
134           final byte[] b64Encoded) throws IOException {
135     byte[] buffer = null;
136     if (b64Encoded != null && b64Encoded.length > 0) {
137       String str = new String(b64Encoded);
138       // A real PKCS10 PEM request
139       String beginKey = CertTools.BEGIN_CERTIFICATE_REQUEST;
140       String endKey = CertTools.END_CERTIFICATE_REQUEST;
141       if (!str.contains(beginKey)) {
142         // Keytool PKCS10 PEM request
143         beginKey = CertTools.BEGIN_KEYTOOL_CERTIFICATE_REQUEST;
144         endKey = CertTools.END_KEYTOOL_CERTIFICATE_REQUEST;
145         if (!str.contains(beginKey)) {
146           // CSR can be a PEM encoded certificate instead of "certificate
147           // request"
148           beginKey = CertTools.BEGIN_CERTIFICATE;
149           endKey = CertTools.END_CERTIFICATE;
150           if (!str.contains(beginKey)) {
151             // IE PKCS10 Base64 coded request
152             try {
153               buffer = Base64.decode(b64Encoded);
154               if (buffer == null) {
155                 throw new IOException("Base64 decode of buffer returns null");
156               }
157             } catch (DecoderException de) {
158               throw new IOException(
159                   "Base64 decode fails, message not base64 encoded: "
160                       + de.getMessage());
161             }
162           }
163         }
164       }
165       if (buffer == null) {
166         buffer = FileTools.getBytesFromPEM(b64Encoded, beginKey, endKey);
167       }
168     } else {
169       throw new IOException("Base64 decode fails, message is empty");
170     }
171     return buffer;
172   }

174   /**
175    *
176    * @param username User
177    * @param password Pass
178    * @param req Request
179    * @param reqType Type
180    * @return Request message
181    * @throws SignRequestSignatureException faik
182    * @throws InvalidKeyException fail
183    * @throws NoSuchAlgorithmException fail
184    * @throws NoSuchProviderException fail
185    * @throws IOException fail
186    * @throws SignatureException fail
187    * @throws InvalidKeySpecException fail
188    * @throws ParseException fail
189    * @throws ConstructionException fail
190    * @throws NoSuchFieldException fail
191    */
192   public static RequestMessage getRequestMessageFromType(
193       final String username,
194       final String password,
195       final String req,
196       final int reqType)
197       throws SignRequestSignatureException, InvalidKeyException,
198           NoSuchAlgorithmException, NoSuchProviderException, IOException,
199           SignatureException, InvalidKeySpecException, ParseException,
200           ConstructionException, NoSuchFieldException {
201     RequestMessage ret = null;
202     if (reqType == CertificateConstants.CERT_REQ_TYPE_PKCS10) {
203       final PKCS10RequestMessage pkcs10RequestMessage =
204           RequestMessageUtils.genPKCS10RequestMessage(req.getBytes());
205       pkcs10RequestMessage.setUsername(username);
206       pkcs10RequestMessage.setPassword(password);
207       ret = pkcs10RequestMessage;
208     } else if (reqType == CertificateConstants.CERT_REQ_TYPE_SPKAC) {
209       byte[] reqBytes = req.getBytes();
210       if (reqBytes != null) {
211         if (LOG.isDebugEnabled()) {
212           LOG.debug("Received NS request: " + new String(reqBytes));
213         }
214         byte[] buffer = Base64.decode(reqBytes);
215         if (buffer == null) {
216           return null;
217         }
218         ASN1InputStream in =
219             new ASN1InputStream(new ByteArrayInputStream(buffer));
220         ASN1Sequence spkacSeq = (ASN1Sequence) in.readObject();
221         in.close();
222         NetscapeCertRequest nscr = new NetscapeCertRequest(spkacSeq);
223         // Verify POPO, we don't care about the challenge, it's not important.
224         nscr.setChallenge("challenge");
225         if (!nscr.verify("challenge")) {
226           if (LOG.isDebugEnabled()) {
227             LOG.debug("SPKAC POPO verification Failed");
228           }
229           throw new SignRequestSignatureException(
230               "Invalid signature in NetscapeCertRequest, popo-verification"
231                   + " failed.");
232         }
233         if (LOG.isDebugEnabled()) {
234           LOG.debug("POPO verification successful");
235         }
236         PublicKey pubKey = nscr.getPublicKey();
237         ret = new SimpleRequestMessage(pubKey, username, password);
238       }
239     } else if (reqType == CertificateConstants.CERT_REQ_TYPE_CRMF) {
240       final byte[] certificateRequestMessages = Base64.decode(req.getBytes());
241       final CertReqMsg certReqMsg =
242           CertReqMsg.getInstance(
243               ((ASN1Sequence)
244                       ASN1Sequence.fromByteArray(certificateRequestMessages))
245                   .getObjectAt(0));
246       final JcaCertificateRequestMessage jcrm =
247           new JcaCertificateRequestMessage(certReqMsg);
248       try {
249         final PublicKey publicKey = jcrm.getPublicKey();
250         if (jcrm.hasProofOfPossession()) {
251           switch (jcrm.getProofOfPossessionType()) {
252             case JcaCertificateRequestMessage.popRaVerified:
253 
254                 // The requestor claims that it is verified by an RA
255                 break;
256 
257             case JcaCertificateRequestMessage.popSigningKey:
258 
259                 // RFC 4211 Section 4.1
260                 final POPOSigningKey popoSigningKey =
261                     POPOSigningKey.getInstance(
262                         jcrm.toASN1Structure().getPopo().getObject());
263                 if (LOG.isDebugEnabled()) {
264                   if (popoSigningKey != null) {
265                     LOG.debug(
266                         "CRMF POPOSigningKey poposkInput:                      "
267                             + popoSigningKey.getPoposkInput());
268                     if (popoSigningKey.getPoposkInput() != null) {
269                       LOG.debug(
270                           "CRMF POPOSigningKey poposkInput PublicKey:         "
271                               + "   "
272                               + popoSigningKey.getPoposkInput().getPublicKey());
273                       LOG.debug(
274                           "CRMF POPOSigningKey poposkInput PublicKeyMAC:      "
275                               + "   "
276                               + popoSigningKey
277                                   .getPoposkInput()
278                                   .getPublicKeyMAC());
279                     }
280                     LOG.debug(
281                         "CRMF POPOSigningKey algorithmIdentifier.algorithm.id: "
282                             + popoSigningKey
283                                 .getAlgorithmIdentifier()
284                                 .getAlgorithm()
285                                 .getId());
286                     LOG.debug(
287                         "CRMF POPOSigningKey signature:                        "
288                             + popoSigningKey.getSignature());
289                   } else {
290                     LOG.debug(
291                         "CRMF POPOSigningKey is not defined even though POP"
292                             + " type is popSigningKey. Validation will fail.");
293                   }
294                 }
295                 final ContentVerifierProvider cvp =
296                     CertTools.genContentVerifierProvider(publicKey);
297                 // Work around for bug in BC where
298                 // jcrm.hasSigningKeyProofOfPossessionWithPKMAC() will throw NPE
299                 // if PoposkInput is null
300                 if (popoSigningKey.getPoposkInput() != null
301                     && jcrm.hasSigningKeyProofOfPossessionWithPKMAC()) {
302                   final PKMACBuilder pkmacBuilder =
303                       new PKMACBuilder(
304                           new JcePKMACValuesCalculator()
305                               .setProvider(BouncyCastleProvider.PROVIDER_NAME));
306                   if (!jcrm.isValidSigningKeyPOP(
307                       cvp, pkmacBuilder, password.toCharArray())) {
308                     throw new SignRequestSignatureException(
309                         "CRMF POP with PKMAC failed signature or MAC"
310                             + " validation.");
311                   } else {
312                     if (LOG.isDebugEnabled()) {
313                       LOG.debug(
314                           "CRMF POP with PKMAC passed signature and PKMAC"
315                               + " validation.");
316                     }
317                   }
318                 } else {
319                   if (!jcrm.isValidSigningKeyPOP(cvp)) {
320                     throw new SignRequestSignatureException(
321                         "CRMF POP failed signature validation.");
322                   } else {
323                     if (LOG.isDebugEnabled()) {
324                       LOG.debug("CRMF POP passed signature validation.");
325                     }
326                   }
327                 }
328                 break;
329 
330             case JcaCertificateRequestMessage.popKeyEncipherment:
331 
332                 // RFC 4211 Section 4.2 (Not implemented)
333                 LOG.info(
334                     "CRMF RFC4211 Section 4.2 KeyEncipherment POP validation"
335                         + " is not implemented. Will try to use the request's"
336                         + " public key anyway.");
337                 break;
338 
339             case JcaCertificateRequestMessage.popKeyAgreement:
340 
341                 // RFC 4211 Section 4.3 (Not implemented)
342                 LOG.info(
343                     "CRMF RFC4211 Section 4.3 KeyAgreement POP validation is"
344                         + " not implemented. Will try to use the request's"
345                         + " public key anyway.");
346                 break;
347 
348             default:
349               throw new SignRequestSignatureException(
350                   "CRMF POP of type "
351                       + jcrm.getProofOfPossessionType()
352                       + " is unknown.");
353           }
354         }
355         final SimpleRequestMessage simpleRequestMessage =
356             new SimpleRequestMessage(publicKey, username, password);
357         simpleRequestMessage.setRequestExtensions(
358             jcrm.getCertTemplate().getExtensions());
359         ret = simpleRequestMessage;
360       } catch (CRMFException e) {
361         throw new SignRequestSignatureException(
362             "CRMF POP verification failed.", e);
363       } catch (OperatorCreationException e) {
364         throw new SignRequestSignatureException(
365             "CRMF POP verification failed.", e);
366       }
367     } else if (reqType == CertificateConstants.CERT_REQ_TYPE_PUBLICKEY) {
368       byte[] request;
369       // Request can be Base64 encoded or in PEM format
370       try {
371         request =
372             FileTools.getBytesFromPEM(
373                 req.getBytes(),
374                 CertTools.BEGIN_PUBLIC_KEY,
375                 CertTools.END_PUBLIC_KEY);
376       } catch (IOException ex) {
377         try {
378           request = Base64.decode(req.getBytes());
379           if (request == null) {
380             throw new IOException("Base64 decode of buffer returns null");
381           }
382         } catch (DecoderException de) {
383           throw new IOException(
384               "Base64 decode fails, message not base64 encoded: "
385                   + de.getMessage());
386         }
387       }
388       final PublicKey pubKey = KeyTools.getPublicKeyFromBytes(request);
389       ret = new SimpleRequestMessage(pubKey, username, password);
390     } else if (reqType == CertificateConstants.CERT_REQ_TYPE_CVC) {
391       CVCObject parsedObject =
392           CertificateParser.parseCVCObject(Base64.decode(req.getBytes()));
393       // We will handle both the case if the request is an authenticated
394       // request, i.e. with an outer signature
395       // and when the request is missing the (optional) outer signature.
396       CVCertificate cvccert = null;
397       if (parsedObject instanceof CVCAuthenticatedRequest) {
398         CVCAuthenticatedRequest cvcreq = (CVCAuthenticatedRequest) parsedObject;
399         cvccert = cvcreq.getRequest();
400       } else {
401         cvccert = (CVCertificate) parsedObject;
402       }
403       CVCRequestMessagee/request/CVCRequestMessage.html#CVCRequestMessage">CVCRequestMessage reqmsg = new CVCRequestMessage(cvccert.getDEREncoded());
404       reqmsg.setUsername(username);
405       reqmsg.setPassword(password);
406       // Popo is really actually verified by the CA (in SignSessionBean) as well
407       if (!reqmsg.verify()) {
408         if (LOG.isDebugEnabled()) {
409           LOG.debug("CVC POPO verification Failed");
410         }
411         throw new SignRequestSignatureException(
412             "Invalid inner signature in CVCRequest, popo-verification failed.");
413       } else {
414         if (LOG.isDebugEnabled()) {
415           LOG.debug("POPO verification successful");
416         }
417       }
418       ret = reqmsg;
419     }
420     return ret;
421   }
