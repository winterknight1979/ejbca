28  /**
29   * An object of this class is identifying one or several certificates or a CRL.
30   *
31   * @version $Id: HashID.java 26020 2017-06-14 12:41:08Z mikekushner $
32   */
33  public final class HashID {
34    /** Log4j instance for Base. */
35    private static final Logger LOG = Logger.getLogger(HashID.class);
36    /** True if the ID confirms to a RFC4387 ID. */
37    private final boolean isOK;
38    /**
39     * The RFC4387 identification string. This is a base64 encoding of the hash.
40     */
41    private final String b64;
42  
43    /** The b64 with \ substituted with %2B to be used for URLs. */
44    private final String b64url;
45    /** Key to be used for hash tables. */
46    private final Integer key;
47  
48    private HashID(final byte[] hash) {
49      final String b64padded = new String(Base64.encode(hash));
50      if (b64padded.length() != HASH_LENGTH + 1
51              || b64padded.charAt(HASH_LENGTH) != '=') {
52        this.isOK = false;
53        this.b64 = b64padded;
54      } else {
55        this.isOK = true;
56        this.b64 = b64padded.substring(0, HASH_LENGTH);
57      }
58      this.b64url = this.b64.replaceAll("\\+", "%2B");
59      this.key = Integer.valueOf(new BigInteger(hash).hashCode());
60    }
61  
62    private static byte[] hashFromPrincipalDN(final X500Principal principal) {
63      return CertTools.generateSHA1Fingerprint(principal.getEncoded());
64    }
65  
66    private static HashID getFromDN(final X500Principal principal) {
67      final HashIDs/certificate/HashID.html#HashID">HashID id = new HashID(hashFromPrincipalDN(principal));
68      if (id.isOK) {
69        if (LOG.isDebugEnabled()) {
70          LOG.debug(
71              "The DN '"
72                  + principal.getName()
73                  + "' is identified by the Hash string '"
74                  + id.b64
75                  + "' when accessing the VA.");
76        }
77      } else {
78        LOG.error(
79            "The DN '"
80                + principal.getName()
81                + "' has a non valid Hash identification string: "
82                + id.b64);
83      }
84      return id;
85    }
86    /**
87     * @param cert The subject DN of the certificate should be the identifier.
88     * @return the ID
89     */
90    public static HashID getFromSubjectDN(final X509Certificate cert) {
91      return getFromDN(cert.getSubjectX500Principal());
92    }
93    /**
94     * @param cert The issuer DN of the certificate should be the identifier.
95     * @return the ID
96     */
97    public static HashID getFromIssuerDN(final X509Certificate cert) {
98      return getFromDN(cert.getIssuerX500Principal());
99    }

101   /**
102    * @param certificateHolder The issuer DN of the certificate should be the
103    *     identifier.
104    * @return the ID
105    * @throws CertificateException on failure
106    */
107   public static HashID getFromIssuerDN(
108           final X509CertificateHolder certificateHolder)
109       throws CertificateException {
110     return getFromIssuerDN(
111         new JcaX509CertificateConverter().getCertificate(certificateHolder));
112   }
113   /**
114    * @param sDN A string representation of a DN to be as ID. The DN will not be
115    *     transformed in any way.
116    * @return the ID.
117    */
118   public static HashID getFromDNString(final String sDN) {
119     // Note that the DN string has to be encoded to an ASN1 with the BC lib. BC
120     // endcoding is EJBCA standard.
121     try {
122       return getFromDN(
123           new X500Principal(
124               new X500Name(
125                       CertTools.isDNReversed(sDN)
126                           ? CertTools.reverseDN(sDN)
127                           : sDN)
128                   .getEncoded()));
129     } catch (IOException e) {
130       throw new IllegalStateException(e);
131     }
132   }
133 
134   /** Length. */
135   private static final int HASH_LENGTH = 27;
136   /**
137    * @param s The hash base64 encoded. See RFC4387
138    * @return the ID.
139    */
140   public static HashID getFromB64(final String s) {
141     return new HashID(Base64.decode(s.length() == HASH_LENGTH ? s + '=' : s));
142   }
143   /**
144    * @param cert The public key of the certificate will be used as ID.
145    * @return the ID.
146    */
147   public static HashID getFromKeyID(final X509Certificate cert) {
148     final HashID id =
149         new HashID(
150             KeyTools.createSubjectKeyId(cert.getPublicKey())
151                 .getKeyIdentifier());
152     if (id.isOK) {
153       if (LOG.isDebugEnabled()) {
154         LOG.debug(
155             "The certificate with subject DN '"
156                 + cert.getSubjectX500Principal().getName()
157                 + "' can be fetched with 'search.cgi?sKIDHash="
158                 + id.b64
159                 + "' from the VA.");
160       }
161     } else {
162       LOG.error(
163           "The certificate with subject DN '"
164               + cert.getSubjectX500Principal().getName()
165               + "' gives a sKIDHash with a not valid format: "
166               + id.b64);
167     }
168     return id;
169   }
170 
171   /**
172    * @param cert certificate
173    * @return ID
174    * @throws IOException on fail*/
175   public static HashID getFromAuthorityKeyId(final X509Certificate cert)
176       throws IOException {
177     final byte[] hash = CertTools.getAuthorityKeyId(cert);
178     if (hash == null) {
179       return null;
180     }
181     final HashIDs/certificate/HashID.html#HashID">HashID id = new HashID(CertTools.getAuthorityKeyId(cert));
182     if (!id.isOK) {
183       LOG.error(
184           "The certificate with subject DN '"
185               + cert.getSubjectX500Principal().getName()
186               + "' don't have a valid AuthorityKeyId: "
187               + id.b64);
188     }
189     return id;
190   }

193   public String getB64url() {

195   }

198   public Integer getKey() {
199     return key;
200   }
201 
202   /** @return base64 encoded. */
203   public String getB64() {
204     return b64;
205   }
