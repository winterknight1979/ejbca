13  package org.cesecore.certificates.certificate;
14  
15  import java.io.Serializable;
16  import java.math.BigInteger;
17  import java.util.Date;
18  
19  /**
20   * Holds information about a certificate but not the certificate itself.
21   *
22   * @version $Id: CertificateInfo.java 23611 2016-06-03 13:34:45Z jeklund $
23   */
24  public class CertificateInfo implements Serializable {
25  
26    private static final long serialVersionUID = -1973602951994928833L;
27    /** FP. */
28    private String fingerprint;
29    /** FP. */
30    private String cafingerprint;
31    /** Serial. */
32    private String serno;
33    /** DN. */
34    private String issuerdn;
35    /** DN. */
36    private String subjectdn;
37    /** Status. */
38    private int status;
39    /** Type. */
40    private int type;
41    /** Start. */
42    private Date notBefore;
43    /** Expiry. */
44    private Date expiredate;
45    /** Date. */
46    private Date revocationdate;
47    /** Reason. */
48    private int revocationreason;
49    /** Username. */
50    private String username;
51    /** Tag. */
52    private String tag;
53    /** Profile ID. */
54    private int certificateProfileId;
55    /** Profile ID. */
56    private Integer endEntityProfileId;
57    /** Updated. */
58    private Date updateTime;
59    /** ID. */
60    private String subjectKeyId;
61    /** Name. */
62    private String subjectAltName;
63    /**
64     * Constructor.
65     * @param aFingerprint FP
66     * @param aCafingerprint FP
67     * @param aSerno SN
68     * @param aIssuerdn DN
69     * @param aSubjectdn DN
70     * @param aStatus Status
71     * @param aType Type
72     * @param aNotBefore Start
73     * @param aExpiredate End
74     * @param aRevocationdate Revoke
75     * @param aRevocationreason Reason
76     * @param aUsername User
77     * @param aTag Tag
78     * @param aCertificateProfileId ID
79     * @param aEndEntityProfileId ID
80     * @param aUpdateTime Time
81     * @param aSubjectKeyId Key
82     * @param aSubjectAltName Name
83     */
84    public CertificateInfo(
85        final String aFingerprint,
86        final String aCafingerprint,
87        final String aSerno,
88        final String aIssuerdn,
89        final String aSubjectdn,
90        final int aStatus,
91        final int aType,
92        final Long aNotBefore,
93        final long aExpiredate,
94        final long aRevocationdate,
95        final int aRevocationreason,
96        final String aUsername,
97        final String aTag,
98        final int aCertificateProfileId,
99        final Integer aEndEntityProfileId,
100       final long aUpdateTime,
101       final String aSubjectKeyId,
102       final String aSubjectAltName) {
103     this.fingerprint = aFingerprint;
104     this.cafingerprint = aCafingerprint;
105     this.serno = aSerno;
106     this.issuerdn = aIssuerdn;
107     this.subjectdn = aSubjectdn;
108     this.status = aStatus;
109     this.type = aType;
110     this.notBefore = aNotBefore == null ? null : new Date(aNotBefore);
111     this.expiredate = new Date(aExpiredate);
112     this.revocationdate = new Date(aRevocationdate);
113     this.revocationreason = aRevocationreason;
114     this.username = aUsername;
115     this.tag = aTag;
116     this.certificateProfileId = aCertificateProfileId;
117     this.endEntityProfileId = aEndEntityProfileId;
118     this.updateTime = new Date(aUpdateTime);
119     this.subjectKeyId = aSubjectKeyId;
120     this.subjectAltName = aSubjectAltName;
121   }
122 
123   /** @return FP */
124   public String getFingerprint() {
125     return fingerprint;
126   }
127 
128   /** @param fp FP */
129   public void setFingerprint(final String fp) {
130     this.fingerprint = fp;
131   }
132 
133   /** @return FP */
134   public String getCAFingerprint() {
135     return cafingerprint;
136   }
137 
138   /** @return serno */
139   public BigInteger getSerialNumber() {
140     return new BigInteger(serno);
141   }
142 
143   /** @return DN */
144   public String getSubjectDN() {
145     return subjectdn;
146   }
147 
148   /** @return DN */
149   public String getIssuerDN() {
150     return issuerdn;
151   }
152   /**
153    * @return One of the CertificateConstants.CERT_ constants, for example
154    *     CertificateConstants.CERT_ACTIVE
155    */
156   public int getStatus() {
157     return status;
158   }
159   /**
160    * @param s One of the CertificateConstants.CERT_ constants, for example
161    *     CertificateConstants.CERT_ACTIVE
162    */
163   public void setStatus(final int s) {
164     this.status = s;
165   }
166 
167   /** @return type */
168   public int getType() {
169     return type;
170   }
171 
172   /** @return date */
173   public Date getNotBefore() {
174     return notBefore;
175   }
176 
177   /** @return date */
178   public Date getExpireDate() {
179     return expiredate;
180   }
181 
182   /** @return date */
183   public Date getRevocationDate() {
184     return revocationdate;
185   }
186 
187   /** @param d date */
188   public void setRevocationDate(final Date d) {
189     this.revocationdate = d;
190   }
191 
192   /** @return reason */
193   public int getRevocationReason() {
194     return revocationreason;
195   }
196 
197   /** @return tag */
198   public String getTag() {
199     return tag;
200   }
201 
202   /** @param aTag tag */
203   public void setTag(final String aTag) {
204     this.tag = aTag;
205   }
206 
207   /** @return ID */
208   public int getCertificateProfileId() {
209     return certificateProfileId;
210   }
211 
212   /** @return profile */
213   public int getEndEntityProfileIdOrZero() {
214     return endEntityProfileId == null ? 0 : endEntityProfileId;
215   }
216 
217   /** @param aCertificateProfileId ID */
218   public void setCertificateProfileId(final int aCertificateProfileId) {
219     this.certificateProfileId = aCertificateProfileId;
220   }
221 
222   /** @return time */
223   public Date getUpdateTime() {
224     return updateTime;
225   }
226 
227   /** @param aUpdateTime time  */
228   public void setUpdateTime(final Date aUpdateTime) {
229     this.updateTime = aUpdateTime;
230   }
231 
232   /** @return name */
233   public String getUsername() {
234     return username;
235   }
236 
237   /** @param aUsername name  */
238   public void setUsername(final String aUsername) {
239     this.username = aUsername;
240   }
241 
242   /** @return ID */
243   public String getSubjectKeyId() {
244     return subjectKeyId;
245   }
246 
247   /** @return name */
248   public String getSubjectAltName() {
249     return subjectAltName;
250   }
251 }
