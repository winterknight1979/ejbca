13  package org.cesecore.certificates.certificate;
14  
15  import java.io.Serializable;
16  import java.math.BigInteger;
17  import java.util.Date;
18  
19  /**
20   * Holds information about a certificate but not the certificate itself.
21   *
22   * @version $Id: CertificateInfo.java 23611 2016-06-03 13:34:45Z jeklund $
23   */
24  public class CertificateInfo implements Serializable {
25  
26  	private static final long serialVersionUID = -1973602951994928833L;
27  	
28  	private String fingerprint;
29      private String cafingerprint;
30      private String serno;
31      private String issuerdn;
32      private String subjectdn;
33      private int status;
34      private int type;
35      private Date notBefore;
36      private Date expiredate;
37      private Date revocationdate;
38      private int revocationreason;
39      private String username;
40  	private String tag;
41  	private int certificateProfileId;
42      private Integer endEntityProfileId;
43      private Date updateTime;
44      private String subjectKeyId;
45      private String subjectAltName;
46      
47      public CertificateInfo(String fingerprint, String cafingerprint, String serno,  String issuerdn, String subjectdn, int status, int type, 
48              Long notBefore, long expiredate, long revocationdate, int revocationreason, String username, String tag, int certificateProfileId,
49              Integer endEntityProfileId, long updateTime, String subjectKeyId, String subjectAltName) {
50          this.fingerprint = fingerprint;
51          this.cafingerprint = cafingerprint;
52          this.serno = serno;
53          this.issuerdn = issuerdn;
54          this.subjectdn = subjectdn;
55          this.status = status;
56          this.type = type;
57          this.notBefore = notBefore==null ? null : new Date(notBefore);
58          this.expiredate = new Date(expiredate);
59          this.revocationdate = new Date(revocationdate);
60          this.revocationreason = revocationreason;
61          this.username = username;
62          this.tag = tag;
63          this.certificateProfileId = certificateProfileId;
64          this.endEntityProfileId = endEntityProfileId;
65          this.updateTime = new Date(updateTime);
66          this.subjectKeyId = subjectKeyId;
67          this.subjectAltName = subjectAltName;
68      }
69      
70      public String getFingerprint() {return fingerprint;}
71      public void setFingerprint(String fp) {this.fingerprint=fp;}
72      public String getCAFingerprint() {return cafingerprint;}
73      public BigInteger getSerialNumber() {return new BigInteger(serno);}
74      public String getSubjectDN() {return subjectdn;}
75      public String getIssuerDN() {return issuerdn;}
76      /**@return One of the CertificateConstants.CERT_ constants, for example CertificateConstants.CERT_ACTIVE 
77       */
78      public int getStatus() { return status; }
79      /** @param s One of the CertificateConstants.CERT_ constants, for example CertificateConstants.CERT_ACTIVE */
80      public void setStatus(int s) { this.status=s; }
81      public int getType() { return type; }
82      public Date getNotBefore() { return notBefore; }
83      public Date getExpireDate() { return expiredate; }
84      public Date getRevocationDate() { return revocationdate; }
85      public void setRevocationDate(Date d) { this.revocationdate=d; }
86      public int getRevocationReason() { return revocationreason; }
87      public String getTag() {
88  		return tag;
89  	}
90  
91  	public void setTag(String tag) {
92  		this.tag = tag;
93  	}
94  
95  	public int getCertificateProfileId() {
96  		return certificateProfileId;
97  	}
98  
99  	public int getEndEntityProfileIdOrZero() {
100 	    return endEntityProfileId==null ? 0 : endEntityProfileId;
101 	}
102 
103 	public void setCertificateProfileId(int certificateProfileId) {
104 		this.certificateProfileId = certificateProfileId;
105 	}
106 
107 	public Date getUpdateTime() {
108 		return updateTime;
109 	}
110 
111 	public void setUpdateTime(Date updateTime) {
112 		this.updateTime = updateTime;
113 	}
114 
115     public String getUsername() {
116 		return username;
117 	}
118 
119 	public void setUsername(String username) {
120 		this.username = username;
121 	}
122 
123     public String getSubjectKeyId() {
124         return subjectKeyId;
125     }
126 	
127     public String getSubjectAltName() {
128         return subjectAltName;
129     }
130 }
