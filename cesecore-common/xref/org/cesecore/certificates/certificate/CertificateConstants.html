14  package org.cesecore.certificates.certificate;
15  
16  /**
17   * Constants for users and certificates. Constants for Type of user: Type is
18   * constructed as a mask since one user can be of several types. To test a user
19   * type:
20   *
21   * <pre>
22   * if (((type &amp; USER_ENDUSER) == USER_ENDUSER) &amp;&amp;
23   *     ((type &amp; USER_CAADMIN) == USER_ADMINISTOR) || ...
24   *    ...
25   * </pre>
26   *
27   * Bit usage: bits 0-7 (1:st byte): user types bits 8-15 (2:nd byte): unused
28   * bits 16-23 (3:rd byte): unused bits 24-30 (4:th byte): unused Constants for
29   * certificates are simple integer types. Constants for Token Types Token type
30   * is constructed of integer constants since only one token type can be
31   * generated.
32   *
33   * <p>Based on EJBCA (SecConst) version: SecConst.java 9321 2010-06-30 12:49:32Z
34   * jeklund
35   *
36   * @version $Id: CertificateConstants.java 30396 2018-11-05 17:00:50Z
37   *     mikekushner $
38   */
39  public final class CertificateConstants {
40  
41    /** Prevents creation of new class. */
42    private CertificateConstants() { }
43  
44    // Certificate status representations
45    /** Certificate doesn't belong to anyone. */
46    public static final int CERT_UNASSIGNED = 0;
47    /** Assigned, but not yet active. */
48    public static final int CERT_INACTIVE = 10;
49    /**
50     * Certificate is a new CA certificate which will become valid in the future.
51     */
52    public static final int CERT_ROLLOVERPENDING = 11;
53    /** Certificate is active and assigned. */
54    public static final int CERT_ACTIVE = 20;
55    /**
56     * Certificate is still active and the user is notified that it will soon
57     * expire.
58     */
59    public static final int CERT_NOTIFIEDABOUTEXPIRATION = 21;
60    // there was previously a status 30, CERT_TEMP_REVOKED here as well, but it
61    // was not used so
62    // it was removed to avoid misunderstandings.
63    /** Certificate is blocked (can be permanently or temporarily). */
64    public static final int CERT_REVOKED = 40;
65    // there was previously a status 50, EXPIRED here as well, but it was not used
66    // so
67    // it was removed to avoid misunderstandings.
68    /** Certificate is expired and kept for archive purpose. */
69    public static final int CERT_ARCHIVED = 60;
70  
71    // Constants used in certificate generation and publication.
72    /** Certificate type is unknown. */
73    public static final int CERTTYPE_UNKNOWN = 0x0;
74    /** Certificate belongs to an end entity. */
75    public static final int CERTTYPE_ENDENTITY = 0x1;
76    /** Certificate belongs to a sub ca. */
77    public static final int CERTTYPE_SUBCA = 0x2;
78    /** Certificate belongs to a root ca. */
79    public static final int CERTTYPE_ROOTCA = 0x8;
80    /** Certificate belongs on a hard token. */
81    public static final int CERTTYPE_HARDTOKEN = 0x16;
82  
83    // Certificate types used to create certificates
84    /** Certificate used for encryption. */
85    public static final int CERT_TYPE_ENCRYPTION = 0x1;
86    /** Certificate used for digital signatures. */
87    public static final int CERT_TYPE_SIGNATURE = 0x2;
88    /** Certificate used for both encryption and signatures. */
89    public static final int CERT_TYPE_ENCSIGN = 0x3;
90  
91    // Certificate request types
92    /** PKCS#10. */
93    public static final int CERT_REQ_TYPE_PKCS10 = 0;
94    /** CMRF. */
95    public static final int CERT_REQ_TYPE_CRMF = 1;
96    /** SPKAC. */
97    public static final int CERT_REQ_TYPE_SPKAC = 2;
98    /** Public key. */
99    public static final int CERT_REQ_TYPE_PUBLICKEY = 3;
100   /** CVC. */
101   public static final int CERT_REQ_TYPE_CVC = 4;
102 
103   // Certificate response types
104   /** Single non-PKCS cert. */
105   public static final int CERT_RES_TYPE_CERTIFICATE = 0;
106   /** PKCS#7 single cert. */
107   public static final int CERT_RES_TYPE_PKCS7 = 1;
108   /** PKCS#7 Chain. */
109   public static final int CERT_RES_TYPE_PKCS7WITHCHAIN = 2;
110 
111   /* KeyUsage constants */
112   /** Signature. */
113   public static final int DIGITALSIGNATURE = 0;
114   /** Nonrepudiation. */
115   public static final int NONREPUDIATION = 1;
116   /** Incrypt keys. */
117   public static final int KEYENCIPHERMENT = 2;
118   /** Encrypt data. */
119   public static final int DATAENCIPHERMENT = 3;
120   /** Key agreement. */
121   public static final int KEYAGREEMENT = 4;
122   /** Sign certificates. */
123   public static final int KEYCERTSIGN = 5;
124   /** Sign CRLs. */
125   public static final int CRLSIGN = 6;
126   /** Only encrypt. */
127   public static final int ENCIPHERONLY = 7;
128   /** Only decrypt. */
129   public static final int DECIPHERONLY = 8;
130   /** Boundary. */
131   public static final int HIGN_REASON_BOUNDRARY = 11;
132 }
