45  /**
46   * Class parsing the modules/ejbca-common/src/certextensions.properties file 
47   * and maintains a set of available custom extensions to the system.
48   * 
49   * It is also responsible for creating the actual CertificateExtensions
50   * used in certificate generation.
51   * 
52   * It also keep a list of standard (not custom) built in extensions.
53   *
54   * @version $Id: CertificateExtensionFactory.java 22142 2015-11-03 14:15:51Z mikekushner $
55   */
56  public class CertificateExtensionFactory {
57  
58  	private static final Logger log = Logger.getLogger(CertificateExtensionFactory.class);
59  	private static final InternalResources intres = InternalResources.getInstance();
60  	
61  	private static CertificateExtensionFactory instance = null;
62  	
63  	private HashMap<String, String> standardCertificateExtensions = new HashMap<String, String>();
64  	{
65  		standardCertificateExtensions.put(Extension.basicConstraints.getId(), BasicConstraint.class.getName());
66  		standardCertificateExtensions.put(Extension.subjectKeyIdentifier.getId(), SubjectKeyIdentifier.class.getName());
67  		standardCertificateExtensions.put(Extension.authorityKeyIdentifier.getId(), AuthorityKeyIdentifier.class.getName());
68  		standardCertificateExtensions.put(Extension.keyUsage.getId(), KeyUsage.class.getName());
69  		standardCertificateExtensions.put(Extension.extendedKeyUsage.getId(), ExtendedKeyUsage.class.getName());
70  		standardCertificateExtensions.put(Extension.subjectAlternativeName.getId(), SubjectAltNames.class.getName());
71  		standardCertificateExtensions.put(Extension.issuerAlternativeName.getId(), IssuerAltNames.class.getName());
72  		standardCertificateExtensions.put("2.23.136.1.1.6.2", DocumentTypeList.class.getName());
73  		standardCertificateExtensions.put(Extension.cRLDistributionPoints.getId(), CrlDistributionPoints.class.getName());
74  		standardCertificateExtensions.put(Extension.freshestCRL.getId(), FreshestCrl.class.getName());
75  		standardCertificateExtensions.put(Extension.certificatePolicies.getId(), CertificatePolicies.class.getName());
76  		standardCertificateExtensions.put(Extension.subjectDirectoryAttributes.getId(), SubjectDirectoryAttributes.class.getName());
77  		standardCertificateExtensions.put(Extension.authorityInfoAccess.getId(), AuthorityInformationAccess.class.getName());
78  		standardCertificateExtensions.put(Extension.qCStatements.getId(), QcStatement.class.getName());
79  		standardCertificateExtensions.put(Extension.nameConstraints.getId(), NameConstraint.class.getName());
80  		standardCertificateExtensions.put(OCSPObjectIdentifiers.id_pkix_ocsp_nocheck.getId(), OcspNoCheck.class.getName());
81  		standardCertificateExtensions.put(CertTools.OID_MSTEMPLATE, MsTemplate.class.getName());
82  		standardCertificateExtensions.put(SeisCardNumber.OID_CARDNUMBER, SeisCardNumber.class.getName());
83  		standardCertificateExtensions.put(Extension.privateKeyUsagePeriod.getId(), PrivateKeyUsagePeriod.class.getName());
84  	}
85  	
86  	private CertificateExtensionFactory(){}
87  	
88  	/**
89  	 * Method used to get the instance of the factory.
90  	 * If it is the first time the method is called will
91  	 * the configuration file be parsed.
92  	 * @return The factory instance
93  	 */
94  	public static CertificateExtensionFactory getInstance(){
95  		if(instance == null){
96  		    instance = new CertificateExtensionFactory();
97  		}
98  		
99  		return instance;
100 	}
101 	
102 	/**
103 	 * Method returning the instance of the standard CertificateExtension
104 	 * given its object identifier
105 	 * @param oid OID
106 	 * @param certProf Profile 
107 	 * 
108 	 * @return null if the CertificateExtension doesn't exist
109 	 */
110 	public CertificateExtension getStandardCertificateExtension(final String oid, final CertificateProfile certProf){
111 		StandardCertificateExtension ret = null;
112 		final String classPath = (String)standardCertificateExtensions.get(oid);
113 		if (classPath != null) {
114 			try {
115 				final Class<?> implClass = Class.forName(classPath);
116 				ret = (StandardCertificateExtension)implClass.getConstructor().newInstance();					
117 				ret.init(certProf);                    
118 			} catch (ClassNotFoundException e) {
119 				log.error(intres.getLocalizedMessage("certext.noextensionforid", oid), e);			
120 			} catch (InstantiationException e) {
121 				log.error(intres.getLocalizedMessage("certext.noextensionforid", oid), e);			
122 			} catch (IllegalAccessException e) {
123 				log.error(intres.getLocalizedMessage("certext.noextensionforid", oid), e);			
124 			}	catch (InvocationTargetException e) {
125 				log.error(intres.getLocalizedMessage("certext.noextensionforid", oid), e);			
126 			}	catch (NoSuchMethodException e) {
127 				log.error(intres.getLocalizedMessage("certext.noextensionforid", oid), e);			
128 			}					
129 		}
130 		if (ret == null) {
131 			log.error(intres.getLocalizedMessage("certext.noextensionforid", oid));			
132 		}
133 		return ret;
134 	}
135 
136 	/** Method used for testing to be able to reset class between tests
137 	 */
138 	protected static void resetExtensions() {
139 	    instance = null;
140 	}
141 }
