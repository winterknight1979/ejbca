96    /**
97     * @param id ID
98     * @param oid OID
99     * @param displayName Name
100    * @param classPath Classpath
101    * @param critical true if Critical
102    * @param required reue if required
103    * @param properties Properties
104    * @throws CertificateExtentionConfigurationException on error
105    */
106   public void addCustomCertExtension(
107       final int id,
108       final String oid,
109       final String displayName,
110       final String classPath,
111       final boolean critical,
112       final boolean required,
113       final Properties properties)
114       throws CertificateExtentionConfigurationException {
115     try {
116       Class<?> implClass = Class.forName(classPath);
117       CertificateExtension certificateExtension =
118           (CertificateExtension) implClass.getConstructor().newInstance();
119       certificateExtension.init(
120           id, oid.trim(), displayName, critical, required, properties);
121       data.put(id, certificateExtension);
122     } catch (ClassNotFoundException e) {
123       throw new CertificateExtentionConfigurationException(
124           "Cannot add custom certificate extension. "
125               + e.getLocalizedMessage());
126     } catch (InstantiationException e) {
127       throw new CertificateExtentionConfigurationException(
128           "Cannot add custom certificate extension. "
129               + e.getLocalizedMessage());
130     } catch (IllegalAccessException e) {
131       throw new CertificateExtentionConfigurationException(
132           "Cannot add custom certificate extension. "
133               + e.getLocalizedMessage());
134     } catch (InvocationTargetException e) {
135       throw new CertificateExtentionConfigurationException(
136           "Cannot add custom certificate extension. "
137               + e.getLocalizedMessage());
138     } catch (NoSuchMethodException e) {
139       throw new CertificateExtentionConfigurationException(
140           "Cannot add custom certificate extension. "
141               + e.getLocalizedMessage());
142     }
143   }

204   /**
205    * Imports CustomCertExtensions from certextensions.properties
206    * into the database.
207    *
208    * TODO Remove this method when support for EJBCA 6.4.0 is dropped.
209    */
210   @Deprecated
211   private void addAvailableCustomCertExtensionsFromFile() {
212     // If the file has already been removed, no need to go further
213     InputStream is =
214         CertificateExtensionFactory.class.getResourceAsStream(
215             "/certextensions.properties");
216     if (is == null) {
217       return;
218     }
219 
220     try {
221       Properties props = new Properties();
222       try {
223         props.load(is);
224       } finally {
225         is.close();
226       }
227 
228       int count = 0;
229       for (int i = 1; i < 255; i++) {
230         if (props.get("id" + i + ".oid") != null) {
231           if (LOG.isDebugEnabled()) {
232             LOG.debug("found " + props.get("id" + i + ".oid"));
233           }
234           CertificateExtension ce = getCertificateExtensionFromFile(i, props);
235           addCustomCertExtension(ce);
236           count++;
237         }
238       }
239       if (LOG.isDebugEnabled()) {
240         LOG.debug(
241             "Nr of read Custom Certificate Extensions from file: " + count);
242       }
243     } catch (IOException e) {
244       LOG.error("Error parsing the 'certextensions.properties' file.", e);
245     } catch (CertificateExtentionConfigurationException e) {
246       LOG.error(e.getMessage(), e);
247     }
248   }

250   /**
251    * Used for upgrading from old certextensions.properties files.
252    * Package-internal to allow for testing.
253    *
254    * @param id ID
255    * @param propertiesInFile Properties
256    * @return Certificate extension
257    * @throws CertificateExtentionConfigurationException on fail
258    */
259   static CertificateExtension getCertificateExtensionFromFile(
260       final int id, final Properties propertiesInFile)
261       throws CertificateExtentionConfigurationException {
262     String propertyId = "id";
263     String propertyOid = ".oid";
264     String propertyClassPath = ".classpath";
265     String propertyDisplayName = ".displayname";
266     String propertyUsed = ".used";
267     String propertyTranslatable = ".translatable";
268     String propertyCritical = ".critical";
269 
270     try {
271       String oid =
272           StringUtils.trim(
273               propertiesInFile.getProperty(propertyId + id + propertyOid));
274       String classPath =
275           StringUtils.trim(
276               propertiesInFile.getProperty(
277                   propertyId + id + propertyClassPath));
278       String displayName =
279           StringUtils.trim(
280               propertiesInFile.getProperty(
281                   propertyId + id + propertyDisplayName));
282       LOG.debug(
283           propertyId
284               + id
285               + propertyUsed
286               + ":"
287               + propertiesInFile.getProperty(propertyId + id + propertyUsed));
288       boolean used =
289           propertiesInFile
290               .getProperty(propertyId + id + propertyUsed)
291               .trim()
292               .equalsIgnoreCase("TRUE");
293       boolean translatable =
294           propertiesInFile
295               .getProperty(propertyId + id + propertyTranslatable)
296               .trim()
297               .equalsIgnoreCase("TRUE");
298       boolean critical =
299           propertiesInFile
300               .getProperty(propertyId + id + propertyCritical)
301               .trim()
302               .equalsIgnoreCase("TRUE");
303       LOG.debug(
304           id
305               + ", "
306               + used
307               + ", "
308               + oid
309               + ", "
310               + critical
311               + ", "
312               + translatable
313               + ", "
314               + displayName);
315       if (used) {
316         if (oid != null && classPath != null && displayName != null) {
317           Class<?> implClass = Class.forName(classPath);
318           CertificateExtension certificateExtension =
319               (CertificateExtension) implClass.getConstructor().newInstance();
320           Properties extensionProperties =
321               getExtensionProperties(id, propertiesInFile);
322           if (translatable) {
323             extensionProperties.put("translatable", true);
324           }
325           certificateExtension.init(
326               id,
327               oid.trim(),
328               displayName,
329               critical,
330               /*required=*/ true,
331               extensionProperties);
332           return certificateExtension;
333 
334         } else {
335           throw new CertificateExtentionConfigurationException(
336               "Certificate Extension "
337                   + Integer.valueOf(id)
338                   + " seems to be misconfigured in the"
339                   + " certextensions.properties");
340         }
341       }
342 
343     } catch (Exception e) {
344       throw new CertificateExtentionConfigurationException(
345           "Certificate Extension "
346               + Integer.valueOf(id)
347               + " seems to be misconfigured in the certextensions.properties",
348           e);
349     }
350     return null;
351   }

353   private static Properties getExtensionProperties(
354       final int id, final Properties propertiesInFile) {
355     Properties extProps = new Properties();
356     Iterator<Object> keyIter = propertiesInFile.keySet().iterator();
357     String matchString = "id" + id + ".property.";
358     while (keyIter.hasNext()) {
359       String nextKey = (String) keyIter.next();
360       if (nextKey.startsWith(matchString)) {
361         if (nextKey.length() > matchString.length()) {
362           extProps.put(
363               nextKey.substring(matchString.length()),
364               propertiesInFile.get(nextKey));
365         }
366       }
367     }
368     return extProps;
369   }
