13  package org.cesecore.certificates.certificate.certextensions;
14  
15  import java.io.IOException;
16  import java.io.Serializable;
17  import java.security.PublicKey;
18  import java.util.Properties;
19  
20  import org.bouncycastle.asn1.ASN1Encodable;
21  import org.cesecore.certificates.ca.CA;
22  import org.cesecore.certificates.ca.internal.CertificateValidity;
23  import org.cesecore.certificates.certificateprofile.CertificateProfile;
24  import org.cesecore.certificates.endentity.EndEntityInformation;

26  /**
27   * Base class for a certificate extension.
28   * All extensions should inherit this class.
29   * 
30   * The methods that need implementation is getValue
31   * 
32   * @version $Id: CertificateExtension.java 29873 2018-09-13 09:36:27Z aminkh $
33   */
34  public abstract class CertificateExtension implements Serializable {
35  	
36  	private static final long serialVersionUID = -7708267512352607118L;
37  	
38  	// This attribute should be removed when support for EJBCA 6.4.0 is dropped. This attribute has been replaced by OID as an extension's identifying attribute.
39  	// By the time we dropped support for Ejbca 6.4.0, this attribute will not be significant in any way and any references to it can be removed without replacement.
40  	@Deprecated
41      private int id;
42  	private String oID;
43  	private String displayName;
44  	private boolean criticalFlag;
45  	private boolean requiredFlag;
46  	private Properties properties;
47  	
48  	/**
49  	 * Constructor for creating a Certificate Extension. 
50  	 */
51  	public CertificateExtension() {
52  		super();
53  	}
54  
55  	/**
56  	 * @return the unique id of the extension
57  	 */
58  	public int getId() {
59  		return id;
60  	}
61  
62  	/**
63  	 * @return The unique OID of the extension
64  	 */
65  	public String getOID() {
66  		return oID;
67  	}
68  	
69  	/**
70  	 * @param oID The unique OID of the extension
71  	 */
72  	public void setOID(String oID) {
73  		this.oID = oID.trim();
74  	}
75  	
76  	/**
77  	 * @return This extension's readable name
78  	 */
79  	public String getDisplayName() {
80  	    return displayName;
81  	}
82  	
83  	/**
84  	 * @param displayName The extension's readable name
85  	 */
86  	public void setDisplayName(String displayName) {
87  	    this.displayName = displayName;
88  	}
89  	/**
90  	 * @return flag indicating if the extension should be marked as critical or not.
91  	 */
92  	public boolean isCriticalFlag() {
93  		return criticalFlag;
94  	}
95  
96  	/**
97  	 * @param criticalFlag flag indicating if the extension should be marked as critical or not.
98  	 */
99  	public void setCriticalFlag(boolean criticalFlag) {
100 		this.criticalFlag = criticalFlag;
101 	}
102 	
103 	/**
104      * @return flag indicating if the extension should be marked as required or not.
105      */
106     public boolean isRequiredFlag() {
107         return requiredFlag;
108     }
109 
110     /**
111      * @param requiredFlag flag indicating if the extension should be marked as required or not.
112      */
113     public void setRequiredFlag(final boolean requiredFlag) {
114         this.requiredFlag = requiredFlag;
115     }
116 	
117 
118 	/**
119 	 * The properties configured for this extension. The properties are stripped
120 	 * of the beginning "idX.property.". So searching for the property
121 	 * "id1.property.value" only the key "value" should be used in the returned property.
122 	 * 
123 	 * @return the properties configured for this certificate extension.
124 	 */
125 	public Properties getProperties() {
126 		return properties;
127 	}
128 	
129 	/**
130 	 * Method that initializes the CertificateExtension
131 	 * 
132 	 * @param id the uniqueID of the extension
133 	 * @param oID the OID 
134 	 * @param displayName the DN
135 	 * @param criticalFlag if the extension should be marked as critical or not.
136 	 * @param requiredFlag if the extension should be marked as required or not. 
137 	 * @param extensionProperties the complete configuration property file.
138 	 */
139 	public void init(int id, String oID, String displayName, boolean criticalFlag, final boolean requiredFlag, Properties extensionProperties){
140 		this.id = id;
141 		this.oID = oID.trim();
142 		this.displayName = displayName;
143 		this.criticalFlag = criticalFlag;
144 		this.requiredFlag = requiredFlag;
145 		this.properties = extensionProperties;
146 	}
147 	
148 	/**
149 	 * Method that should return the ASN1Encodable value used in the extension
150 	 * this is the method at all implementors must implement.
151 	 * 
152 	 * @param userData the userdata of the issued certificate.
153 	 * @param ca the CA data with access to all the keys etc. For CA certificates, this is the CA itself.
154 	 * @param certProfile the certificate profile
155 	 * @param userPublicKey public key of the user, or null if not available
156 	 * @param caPublicKey public key of the CA, or null if not available
157 	 * @param val validity of certificate where the extension will be added
158 	 * @return a ASN1Encodable or null, if this extension should not be used, which was determined from the values somehow.
159 	 * @throws CertificateExtensionException if there was an error constructing the certificate extension
160 	 * 
161      * @deprecated Callers should use the getValueEncoded method as this method might not be supported by all implementations. Implementors can
162      * still implement this method if they prefer as it gets called from getValueEncoded.
163 	 */
164     public abstract ASN1Encodable getValue(EndEntityInformation userData, CA ca, CertificateProfile certProfile, PublicKey userPublicKey,
165             PublicKey caPublicKey, CertificateValidity val) throws CertificateExtensionException;
166 
167 	/**
168 	 * Method that should return the byte[] value used in the extension. 
169 	 * 
170 	 * The default implementation of this method first calls the getValue() 
171 	 * method and then encodes the result as an byte array. 
172 	 * CertificateExtension implementors has the choice of overriding this 
173 	 * method if they want to include byte[] data in the certificate that
174 	 * is not necessarily an ASN.1 structure otherwise the getValue method 
175 	 * can be implemented as before.
176 	 * 
177 	 * @param userData the userdata of the issued certificate.
178 	 * @param ca the CA data with access to all the keys etc
179 	 * @param certProfile the certificate profile
180 	 * @param userPublicKey public key of the user, or null if not available
181 	 * @param caPublicKey public key of the CA, or null if not available
182      * @param val validity of certificate where the extension will be added
183 	 * @return a byte[] or null, if this extension should not be used, which was determined from the values somehow.
184 	 * @throws CertificateExtensionException if there was an error constructing the certificate extensio
185      *
186 	 */
187     public byte[] getValueEncoded(EndEntityInformation userData, CA ca, CertificateProfile certProfile, PublicKey userPublicKey,
188             PublicKey caPublicKey, CertificateValidity val) throws CertificateExtensionException {
189 		final byte[] result;
190 		final ASN1Encodable value = getValue(userData, ca, certProfile, userPublicKey, caPublicKey, val);
191 		if (value == null) {
192 			result = null;
193 		} else {
194 		    try {
195 		        result = value.toASN1Primitive().getEncoded();
196 		    } catch (IOException ioe) {
197 		        throw new CertificateExtensionException(ioe.getMessage(), ioe);
198 		    }
199 		}
200 		return result;
201 	}
202 	
203 }
