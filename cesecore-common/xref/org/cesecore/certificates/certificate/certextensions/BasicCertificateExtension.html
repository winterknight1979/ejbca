44  /**
45   * The default basic certificate extension that has two property.
46   *
47   * <p>'value' : The value returned 'encoding' : How the value is encoded.
48   *
49   * <p>Optionally, a new property can be defined:
50   *
51   * <p>'nvalues' : number of values of type 'encoding'
52   *
53   * <p>Thus, the extension will be of type 'SEQUENCE OF ENCODING' with a size of
54   * nvalues. The members will be: 'value1', 'value2' and so on.
55   *
56   * <p>Optionally, an other property can be defined:
57   *
58   * <p>'dynamic' : true/false if the extension value(s) should be allowed to be
59   * overridden by value(s) put as extensiondata in ExtendedInformation. Default
60   * is 'false'.
61   *
62   * <p>See documentation for more information.
63   *
64   * @version $Id: BasicCertificateExtension.java 30583 2018-11-22 17:32:11Z
65   *     samuellb $
66   */
67  public class BasicCertificateExtension extends CertificateExtension
68      implements CustomCertificateExtension {

84    private enum Encoding {
85      /** Bitstring. */
86      ENCODING_DERBITSTRING("DERBITSTRING"),
87      /** Int. */
88      ENCODING_DERINTEGER("DERINTEGER"),
89      /** ASCII. */
90      ENCODING_DEROCTETSTRING("DEROCTETSTRING"),
91      /** Boolean. */
92      ENCODING_DERBOOLEAN("DERBOOLEAN"),
93      /** Printable. */
94      ENCODING_DERPRINTABLESTRING("DERPRINTABLESTRING"),
95      /** UTF-8. */
96      ENCODING_DERUTF8STRING("DERUTF8STRING"),
97      /** IA5 String. */
98      ENCODING_DERIA5STRING("DERIA5STRING"),
99      /** DER Null. */
100     ENCODING_DERNULL("DERNULL"),
101     /** DER Object. */
102     ENCODING_DEROBJECT("DEROBJECT"),
103     /** DER OID. */
104     ENCODING_DEROID("DERBOJECTIDENTIFIER");
105 
106     /** lookup map. */
107     private static final Map<String, Encoding> LOOKUP_MAP =
108         new HashMap<String, Encoding>();
109 
110     static {
111       for (Encoding encoding : Encoding.values()) {
112         LOOKUP_MAP.put(encoding.value(), encoding);
113       }
114     }
115 
116     /** Value. */
117     private final String value;
118 
119     Encoding(final String avalue) {
120       this.value = avalue;
121     }
122 
123     public String value() {
124       return value;
125     }
126 
127     public boolean equals(final Encoding otherValue) {
128       if (otherValue == null) {
129         return false;
130       }
131       return value.equalsIgnoreCase(otherValue.value());
132     }
133 
134     public static Encoding fromString(final String value) {
135       return LOOKUP_MAP.get(StringUtils.upperCase(value, Locale.ROOT));
136     }
137   }

161   static {
162     Encoding[] encodings = Encoding.values();
163     // +1 because we need to add RAW as well in the end

172     PROPERTIES_MAP.put(PROPERTY_VALUE, new String[] {});
173     PROPERTIES_MAP.put(PROPERTY_DYNAMIC, CustomCertificateExtension.BOOLEAN);
174   }

201   /**
202    * Returns the defined property 'value' in the encoding specified in
203    * 'encoding'.
204    *
205    * <p>This certificate extension implementations overrides this method as it
206    * want to be able to return a byte[] with the extension value. Otherwise the
207    * implementation could have been put in the getValue method as the super
208    * class CertificateExtension has a default implementation for getValueEncoded
209    * which calls getValue.
210    *
211    * @param userData Used to lookup extension data
212    * @param ca not used
213    * @param certProfile not used
214    * @see
215    *     org.cesecore.certificates.certificate.certextensions.CertificateExtension#getValueEncoded(EndEntityInformation,
216    *     CA, CertificateProfile, PublicKey, PublicKey, CertificateValidity)
217    */
218   @Override
219   public byte[] getValueEncoded(
220       final EndEntityInformation userData,
221       final CA ca,
222       final CertificateProfile certProfile,
223       final PublicKey userPublicKey,
224       final PublicKey caPublicKey,
225       final CertificateValidity val)
226       throws CertificateExtensionException {
227     String[] values = getValues(userData, null);
228     return handleValues(values);
229   }

245   private byte[] handleValues(final String[] values)
246       throws CertificateExtensionException {
247     final byte[] result;
248     String encoding =
249         StringUtils.trim(getProperties().getProperty(PROPERTY_ENCODING));
250     if (values == null
251         || values.length == 0
252         || (values[0] == null
253             && !encoding.equalsIgnoreCase(ENCODING_DERNULL))) {
254       if (!isRequiredFlag()) {
255         return null;
256       }
257       throw new CertificateExtensionException(
258           INT_RES.getLocalizedMessage(
259               "certext.basic.incorrectvalue",
260               Integer.valueOf(getId()),
261               getOID()));
262     }
263 
264     if (encoding.equalsIgnoreCase(ENCODING_RAW)) {
265       if (values.length > 1) {
266         // nvalues can not be used together with encoding=RAW
267         throw new CertificateExtensionException(
268             INT_RES.getLocalizedMessage(
269                 "certext.certextmissconfigured", Integer.valueOf(getId())));
270       } else {
271         result = parseRaw(values[0]);
272       }
273     } else {
274       try {
275         if (values.length > 1) {
276           ASN1EncodableVector ev = new ASN1EncodableVector();
277           for (String value : values) {
278             ASN1Encodable derval = parseValue(encoding, value);
279             ev.add(derval);
280           }
281           result = new DERSequence(ev).getEncoded();
282         } else {
283           result =
284               parseValue(encoding, values[0]).toASN1Primitive().getEncoded();
285         }
286       } catch (IOException ioe) {
287         throw new CertificateExtensionException(ioe.getMessage(), ioe);
288       }
289     }
290     return result;
291   }

293   /**
294    * Get the extension value by first looking in the ExtendedInformation (if
295    * dynamic is enabled) and then in the static configuration.
296    *
297    * @param userData The userdata to get the ExtendedInformation from
298    * @param oid OID
299    * @return The value(s) for the extension (usually 1) or null if no value
300    *     found
301    */
302   private String[] getValues(
303           final EndEntityInformation userData, final String oid) {
304     String[] result = null;
305 
306     boolean dynamic =
307         Boolean.parseBoolean(
308             StringUtils.trim(
309                 getProperties()
310                     .getProperty(PROPERTY_DYNAMIC, Boolean.FALSE.toString())));
311 
312     String strnvalues = getProperties().getProperty(PROPERTY_NVALUES);
313     int nvalues;
314 
315     if (strnvalues == null || strnvalues.trim().equals("")) {
316       nvalues = 0;
317     } else {
318       nvalues = Integer.parseInt(strnvalues);
319     }
320 
321     if (dynamic) {
322       final ExtendedInformation ei = userData.getExtendedInformation();
323       if (ei == null) {
324         result = null;
325       } else {
326         if (nvalues < 1) {
327           String value = null;
328           if (oid != null) {
329             value = userData.getExtendedInformation().getExtensionData(oid);
330             if (value == null || value.trim().isEmpty()) {
331               value =
332                   userData
333                       .getExtendedInformation()
334                       .getExtensionData(oid + "." + PROPERTY_VALUE);
335             }
336           } else {
337             value =
338                 userData.getExtendedInformation().getExtensionData(getOID());
339             if (value == null || value.trim().isEmpty()) {
340               value =
341                   userData
342                       .getExtendedInformation()
343                       .getExtensionData(getOID() + "." + PROPERTY_VALUE);
344             }
345           }
346           if (value == null) {
347             result = null;
348           } else {
349             result = new String[] {value};
350           }
351         } else {
352           for (int i = 1; i <= nvalues; i++) {
353             String value = null;
354             if (oid != null) {
355               value =
356                   userData
357                       .getExtendedInformation()
358                       .getExtensionData(
359                           oid + "." + PROPERTY_VALUE + Integer.toString(i));
360             } else {
361               value =
362                   userData
363                       .getExtendedInformation()
364                       .getExtensionData(
365                           getOID()
366                               + "."
367                               + PROPERTY_VALUE
368                               + Integer.toString(i));
369             }
370             if (value != null) {
371               if (result == null) {
372                 result = new String[nvalues];
373               }
374               result[i - 1] = value;
375             }
376           }
377         }
378       }
379     }
380     if (result == null) {
381       if (nvalues < 1) {
382         String value = getProperties().getProperty(PROPERTY_VALUE);
383         if (value == null || value.trim().equals("")) {
384           value = getProperties().getProperty(PROPERTY_VALUE + "1");
385         }
386         result = new String[] {value};
387       } else {
388         result = new String[nvalues];
389         for (int i = 1; i <= nvalues; i++) {
390           result[i - 1] =
391               getProperties().getProperty(PROPERTY_VALUE + Integer.toString(i));
392         }
393       }
394     }
395     return result;
396   }

398   private ASN1Encodable parseValue(final String encoding, final String value)
399       throws CertificateExtensionException {
400 
401     ASN1Encodable toret = null;
402 
403     Encoding encodingType = Encoding.fromString(encoding);
404 
405     if (encodingType == null) {
406       throw new CertificateExtensionException(
407           INT_RES.getLocalizedMessage(
408               "certext.basic.incorrectenc",
409               encoding,
410               Integer.valueOf(getId())));
411     }
412 
413     if (!Encoding.ENCODING_DERNULL.equals(encodingType)
414         && ((value == null || value.trim().equals("")) && isRequiredFlag())) {
415       throw new CertificateExtensionException(
416           INT_RES.getLocalizedMessage(
417               "certext.basic.incorrectvalue",
418               Integer.valueOf(getId()),
419               getOID()));
420     }
421 
422     switch (encodingType) {
423       case ENCODING_DERBITSTRING:
424         toret = parseDERBitString(value);
425         break;
426       case ENCODING_DERINTEGER:
427         toret = parseDERInteger(value);
428         break;
429       case ENCODING_DEROCTETSTRING:
430         toret = parseDEROctetString(value);
431         break;
432       case ENCODING_DERBOOLEAN:
433         toret = parseDERBoolean(value);
434         break;
435       case ENCODING_DEROID:
436         toret = parseDEROID(value);
437         break;
438       case ENCODING_DERPRINTABLESTRING:
439         toret = parseDERPrintableString(value);
440         break;
441       case ENCODING_DERUTF8STRING:
442         toret = parseDERUTF8String(value);
443         break;
444       case ENCODING_DERIA5STRING:
445         toret = parseDERIA5String(value);
446         break;
447       case ENCODING_DERNULL:
448         toret = DERNull.INSTANCE;
449         break;
450       case ENCODING_DEROBJECT:
451         toret = parseHexEncodedDERObject(value);
452         break;
453       default:
454         throw new CertificateExtensionException(
455             INT_RES.getLocalizedMessage(
456                 "certext.basic.incorrectenc",
457                 encoding,
458                 Integer.valueOf(getId())));
459     }
460     return toret;
461   }

463   private ASN1Encodable parseDERBitString(final String value)
464       throws CertificateExtensionException {
465     ASN1Encodable retval = null;
466     try {
467       BigInteger bigInteger = new BigInteger(value, 2);
468       int padBits = value.length() - 1 - value.lastIndexOf("1");
469       if (padBits == 8) {
470         padBits = 0;
471       }
472       byte[] byteArray = bigInteger.toByteArray();
473       if (byteArray[0] == 0) {
474         // Remove empty extra byte
475         // System.arraycopy handles creating of temporary array when destinatio
476         // is the same
477         System.arraycopy(byteArray, 1, byteArray, 0, byteArray.length - 1);
478       }
479       retval = new DERBitString(byteArray, padBits);
480     } catch (NumberFormatException e) {
481       throw new CertificateExtensionException(
482           INT_RES.getLocalizedMessage(
483               "certext.basic.illegalvalue",
484               value,
485               Integer.valueOf(getId()),
486               getOID()));
487     }
488 
489     return retval;
490   }

509   private ASN1Encodable parseDERInteger(final String value)
510       throws CertificateExtensionException {
511     ASN1Encodable retval = null;

519               value,
520               Integer.valueOf(getId()),
521               getOID()));
522     }

527   private ASN1Encodable parseDEROctetString(final String value)
528       throws CertificateExtensionException {
529     ASN1Encodable retval = null;
530     if (value.matches("^\\p{XDigit}*")) {
531       byte[] bytes = Hex.decode(value);
532       retval = new DEROctetString(bytes);
533     } else {
534       throw new CertificateExtensionException(
535           INT_RES.getLocalizedMessage(
536               "certext.basic.illegalvalue",
537               value,
538               Integer.valueOf(getId()),
539               getOID()));
540     }
541     return retval;
542   }

544   /**
545    * Tries to read the hex-string as an DERObject. If it contains more than one
546    * ASN1Encodable object, return a DERSequence of the objects.
547    *
548    * @param value value
549    * @return DER sequence
550    * @throws CertificateExtensionException on fail
551    */
552   private ASN1Encodable parseHexEncodedDERObject(final String value)
553       throws CertificateExtensionException {
554     ASN1Encodable retval = null;
555     if (value.matches("^\\p{XDigit}*")) {
556       byte[] bytes = Hex.decode(value);
557       try {
558         ASN1InputStream ais = new ASN1InputStream(bytes);
559         ASN1Encodable firstObject = ais.readObject();
560         if (ais.available() > 0) {
561           ASN1EncodableVector ev = new ASN1EncodableVector();
562           ev.add(firstObject);
563           while (ais.available() > 0) {
564             ev.add(ais.readObject());
565           }
566           retval = new DERSequence(ev);
567         } else {
568           retval = firstObject;
569         }
570         ais.close();
571       } catch (Exception e) {
572         throw new CertificateExtensionException(
573             INT_RES.getLocalizedMessage(
574                 "certext.basic.illegalvalue",
575                 value,
576                 Integer.valueOf(getId()),
577                 getOID()));
578       }
579     } else {
580       throw new CertificateExtensionException(
581           INT_RES.getLocalizedMessage(
582               "certext.basic.illegalvalue",
583               value,
584               Integer.valueOf(getId()),
585               getOID()));
586     }
587     return retval;
588   }

590   private ASN1Encodable parseDERBoolean(final String value)
591       throws CertificateExtensionException {
592     ASN1Encodable retval = null;
593     if (value.equalsIgnoreCase("TRUE")) {
594       retval = ASN1Boolean.TRUE;
595     }
596 
597     if (value.equalsIgnoreCase("FALSE")) {
598       retval = ASN1Boolean.FALSE;
599     }
600 
601     if (retval == null) {
602       throw new CertificateExtensionException(
603           INT_RES.getLocalizedMessage(
604               "certext.basic.illegalvalue",
605               value,
606               Integer.valueOf(getId()),
607               getOID()));
608     }
609 
610     return retval;
611   }

645   private byte[] parseRaw(
646           final String value) throws CertificateExtensionException {
647     if (value == null) {
648       throw new CertificateExtensionException(
649           INT_RES.getLocalizedMessage(
650               "certext.basic.incorrectvalue",
651               Integer.valueOf(getId()),
652               getOID()));
653     }
654     return Hex.decode(value);
655   }
