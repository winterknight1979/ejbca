13  package org.cesecore.certificates.certificate.certextensions.standard;
14  
15  import java.security.PublicKey;
16  
17  import org.apache.commons.lang.StringUtils;
18  import org.apache.log4j.Logger;
19  import org.bouncycastle.asn1.ASN1Encodable;
20  import org.bouncycastle.asn1.x509.Extension;
21  import org.bouncycastle.asn1.x509.GeneralNames;
22  import org.cesecore.certificates.ca.CA;
23  import org.cesecore.certificates.ca.X509CAInfo;
24  import org.cesecore.certificates.ca.internal.CertificateValidity;
25  import org.cesecore.certificates.certificate.CertificateConstants;
26  import org.cesecore.certificates.certificateprofile.CertificateProfile;
27  import org.cesecore.certificates.endentity.EndEntityInformation;
28  import org.cesecore.util.CertTools;
29  
30  /**
31   * Class for standard X509 certificate extension. 
32   * See rfc3280 or later for spec of this extension.
33   * 
34   * @version $Id: IssuerAltNames.java 24542 2016-10-18 15:32:48Z samuellb $
35   */
36  public class IssuerAltNames extends StandardCertificateExtension {
37      private static final long serialVersionUID = 1L;
38      private static final Logger log = Logger.getLogger(IssuerAltNames.class);
39  
40      @Override
41      public void init(final CertificateProfile certProf) {
42          super.setOID(Extension.issuerAlternativeName.getId());
43          super.setCriticalFlag(certProf.getIssuerAlternativeNameCritical());
44      }
45      
46      @Override
47      public ASN1Encodable getValue(final EndEntityInformation subject, final CA ca, final CertificateProfile certProfile,
48              final PublicKey userPublicKey, final PublicKey caPublicKey, CertificateValidity val) {
49          GeneralNames ret = null;
50          String altName = null;
51          if (ca.getCACertificate() != null) {
52              altName = CertTools.getSubjectAlternativeName(ca.getCACertificate());
53          } else {
54              // If we have a new Root CA (not renewing), we go here
55              if (certProfile.getType() == CertificateConstants.CERTTYPE_ROOTCA && ca.getCAInfo() instanceof X509CAInfo) {
56                  altName = ((X509CAInfo)ca.getCAInfo()).getSubjectAltName();
57                  if (StringUtils.isNotEmpty(altName) && certProfile.getUseSubjectAltNameSubSet()) {
58                      altName = certProfile.createSubjectAltNameSubSet(altName);
59                  }
60                  if (log.isDebugEnabled()) {
61                      log.debug("Using the SAN value as the IssuerAltName value for Root CA. SAN='" + altName + "'");
62                  }
63              } else if (log.isDebugEnabled()) {
64                  // This should never happen
65                  log.debug("Missing CA certificate in CA " + ca.getCAId() + ", and subject is not an X509 Root CA. Will not add IssuerAltName extension");
66              }
67          }
68          if (StringUtils.isNotEmpty(altName)) {
69              ret = CertTools.getGeneralNamesFromAltName(altName);
70          }
71          if (ret == null) {
72              if (log.isDebugEnabled()) {
73                  log.debug("No altnames (SubjectAltName in issuing CA certificate) trying to make IssuerAltName extension: "+altName);
74              }
75          }
76          return ret;
77      }   
78  }
