45  /**
46   * QCStatement (rfc3739)
47   *
48   * <p>Class for standard X509 certificate extension. This extension have some
49   * basics defined in RFC 3739, but the majority of fields are used in EU
50   * purposes and specified in EU standards. ETSI EN 319 412-5 (v2.1.1, 2016-02 or
51   * later)
52   * https://www.etsi.org/deliver/etsi_en/319400_319499/31941205/02.01.01_60/en_31941205v020101p.pdf
53   * ETSI TS 101 862 (v1.3.3, 2006-01 or later)
54   * https://www.etsi.org/deliver/etsi_ts/101800_101899/101862/01.03.03_60/ts_101862v010303p.pdf
55   * ETSI TS 119 495 (v1.1.2, 2018-07 or later)
56   * https://www.etsi.org/deliver/etsi_ts/119400_119499/119495/01.01.02_60/ts_119495v010102p.pdf
57   *
58   * <p>qcStatements EXTENSION ::= { SYNTAX QCStatements IDENTIFIED BY
59   * id-pe-qcStatements } id-pe-qcStatements OBJECT IDENTIFIER ::= { id-pe 3 }
60   *
61   * <p>QCStatements ::= SEQUENCE OF QCStatement QCStatement ::= SEQUENCE {
62   * statementId QC-STATEMENT.&amp;Id({SupportedStatements}), statementInfo
63   * QC-STATEMENT.&amp;Type ({SupportedStatements}{@literal @}statementId})
64   * OPTIONAL }
65   *
66   * <p>SupportedStatements QC-STATEMENT ::= { qcStatement-1,...}
67   *
68   * @version $Id: QcStatement.java 29674 2018-08-17 10:26:59Z anatom $
69   */
70  public class QcStatement extends StandardCertificateExtension {
71    private static final long serialVersionUID = 1L;
72    /** Logger. */
73    private static final Logger LOG = Logger.getLogger(QcStatement.class);
74  
75    @Override
76    public void init(final CertificateProfile certProf) {
77      super.setOID(Extension.qCStatements.getId());
78      super.setCriticalFlag(certProf.getQCStatementCritical());
79    }
80  
81    @Override
82    public ASN1Encodable getValue(
83        final EndEntityInformation subject,
84        final CA ca,
85        final CertificateProfile certProfile,
86        final PublicKey userPublicKey,
87        final PublicKey caPublicKey,
88        final CertificateValidity val)
89        throws CertificateExtensionException {
90      DERSequence ret = null;
91      final String names = certProfile.getQCStatementRAName();
92      final GeneralNames san = CertTools.getGeneralNamesFromAltName(names);
93      SemanticsInformation si = null;
94      if (san != null) {
95        if (StringUtils.isNotEmpty(certProfile.getQCSemanticsId())) {
96          si =
97              new SemanticsInformation(
98                  new ASN1ObjectIdentifier(certProfile.getQCSemanticsId()),
99                  san.getNames());
100       } else {
101         si = new SemanticsInformation(san.getNames());
102       }
103     } else if (StringUtils.isNotEmpty(certProfile.getQCSemanticsId())) {
104       si =
105           new SemanticsInformation(
106               new ASN1ObjectIdentifier(certProfile.getQCSemanticsId()));
107     }
108     final ArrayList<QCStatement> qcs = new ArrayList<QCStatement>();
109     QCStatement qc = null;
110     // First the standard rfc3739 QCStatement with an optional
111     // SematicsInformation
112     // We never add RFC3739QCObjectIdentifiers.id_qcs_pkixQCSyntax_v1. This is
113     // so old so we think it has never been used in the wild basically.
114     // That means no need to have code we have to maintain for that.
115     if (certProfile.getUsePkixQCSyntaxV2()) {
116       ASN1ObjectIdentifier pkixQcSyntax =
117           RFC3739QCObjectIdentifiers.id_qcs_pkixQCSyntax_v2;
118       if ((si != null)) {
119         qc = new QCStatement(pkixQcSyntax, si);
120         qcs.add(qc);
121       } else {
122         qc = new QCStatement(pkixQcSyntax);
123         qcs.add(qc);
124       }
125     }
126     // ETSI Statement that the certificate is a Qualified Certificate
127     if (certProfile.getUseQCEtsiQCCompliance()) {
128       qc = new QCStatement(ETSIQCObjectIdentifiers.id_etsi_qcs_QcCompliance);
129       qcs.add(qc);
130     }
131     // ETSI Statement regarding limit on the value of transactions
132     // Both value and currency must be available for this extension
133     if (certProfile.getUseQCEtsiValueLimit()
134         && (certProfile.getQCEtsiValueLimit() >= 0)
135         && (certProfile.getQCEtsiValueLimitCurrency() != null)) {
136       final int limit = certProfile.getQCEtsiValueLimit();
137       // The exponent should be default 0
138       final int exponent = certProfile.getQCEtsiValueLimitExp();
139       final MonetaryValue value =
140           new MonetaryValue(
141               new Iso4217CurrencyCode(
142                   certProfile.getQCEtsiValueLimitCurrency()),
143               limit,
144               exponent);
145       qc =
146           new QCStatement(
147               ETSIQCObjectIdentifiers.id_etsi_qcs_LimiteValue, value);
148       qcs.add(qc);
149     }
150 
151     if (certProfile.getUseQCEtsiRetentionPeriod()) {
152       final ASN1Integer years =
153           new ASN1Integer(((Integer) certProfile.getQCEtsiRetentionPeriod()));
154       qc =
155           new QCStatement(
156               ETSIQCObjectIdentifiers.id_etsi_qcs_RetentionPeriod, years);
157       qcs.add(qc);
158     }
159 
160     // ETSI Statement claiming that the private key resides in a Signature
161     // Creation Device
162     if (certProfile.getUseQCEtsiSignatureDevice()) {
163       qc = new QCStatement(ETSIQCObjectIdentifiers.id_etsi_qcs_QcSSCD);
164       qcs.add(qc);
165     }
166     // ETSI QC Type and PDS is new fields in EN 319 412-05 (2016)
167     if (StringUtils.isNotEmpty(certProfile.getQCEtsiType())) {
168       final ASN1EncodableVector vec = new ASN1EncodableVector();
169       vec.add(new ASN1ObjectIdentifier(certProfile.getQCEtsiType()));
170       ASN1Sequence seq = new DERSequence(vec);
171       qc =
172           new QCStatement(
173               new ASN1ObjectIdentifier("0.4.0.1862.1.6"),
174               seq); // ETSIQCObjectIdentifiers.id_etsi_qcs_QcType in BC > 1.54
175       qcs.add(qc);
176     }
177     if (certProfile.getQCEtsiPds() != null) {
178       final ASN1EncodableVector locations = new ASN1EncodableVector();
179       for (PKIDisclosureStatement pds : certProfile.getQCEtsiPds()) {
180         final ASN1EncodableVector location = new ASN1EncodableVector();
181         location.add(new DERIA5String(pds.getUrl()));
182         location.add(new DERPrintableString(pds.getLanguage()));
183         locations.add(new DERSequence(location));
184       }
185       qc =
186           new QCStatement(
187               new ASN1ObjectIdentifier("0.4.0.1862.1.5"),
188               new DERSequence(
189                   locations)); // ETSIQCObjectIdentifiers.id_etsi_qcs_QcPds in
190                                // BC > 1.54
191       qcs.add(qc);
192     }
193     // Custom UTF8String QC-statement:
194     // qcStatement-YourCustom QC-STATEMENT ::= { SYNTAX YourCustomUTF8String
195     //   IDENTIFIED BY youroid }
196     //   -- This statement gives you the possibility to define your own
197     // QC-statement
198     //   -- using an OID and a simple UTF8String, with describing text. A sample
199     // text could for example be:
200     //   -- This certificate, according to Act. No. xxxx Electronic Signature
201     // Law is a qualified electronic certificate
202     //
203     // YourCustomUTF8String ::= UTF8String
204     if (certProfile.getUseQCCustomString()
205         && !StringUtils.isEmpty(certProfile.getQCCustomStringOid())
206         && !StringUtils.isEmpty(certProfile.getQCCustomStringText())) {
207       final DERUTF8String str =
208           new DERUTF8String(certProfile.getQCCustomStringText());
209       final ASN1ObjectIdentifier oid =
210           new ASN1ObjectIdentifier(certProfile.getQCCustomStringOid());
211       qc = new QCStatement(oid, str);
212       qcs.add(qc);
213     }
214     if (!qcs.isEmpty()) {
215       final ASN1EncodableVector vec = new ASN1EncodableVector();
216       final Iterator<QCStatement> iter = qcs.iterator();
217       while (iter.hasNext()) {
218         final QCStatement q = (QCStatement) iter.next();
219         vec.add(q);
220       }
221       ret = new DERSequence(vec);
222     }
223     if (ret == null) {
224       LOG.error(
225           "Qualified certificate statements extension has been enabled, but no"
226               + " statements were included!");
227       throw new CertificateExtensionException(
228           "If qualified certificate statements extension has been enabled, at"
229               + " least one statement must be included!");
230     }
231     return ret;
232   }
