40  /**
41   * Class for standard X509 certificate extension. See rfc3280 or later for spec
42   * of this extension.
43   *
44   * @version $Id: CertificatePolicies.java 24112 2016-08-04 12:54:30Z anatom $
45   */
46  public class CertificatePolicies extends StandardCertificateExtension {
47    private static final long serialVersionUID = 1L;
48    /** Logger. */
49    private static final Logger LOG = Logger.getLogger(CertificatePolicies.class);
50  
51    @Override
52    public void init(final CertificateProfile certProf) {
53      super.setOID(Extension.certificatePolicies.getId());
54      super.setCriticalFlag(certProf.getCertificatePoliciesCritical());
55    }
56  
57    @Override
58    public ASN1Encodable getValue(
59        final EndEntityInformation subject,
60        final CA ca,
61        final CertificateProfile certProfile,
62        final PublicKey userPublicKey,
63        final PublicKey caPublicKey,
64        final CertificateValidity val)
65        throws CertificateExtensionException {
66      DERSequence ret = null;
67      // The UserNotice policy qualifier can have two different character
68      // encodings,
69      // the correct one (UTF8) or the wrong one (BMP) used by IE < 7.
70      final X509CA../../../../../../org/cesecore/certificates/ca/X509CA.html#X509CA">X509CA x509ca = (X509CA) ca;
71      int displayencoding = DisplayText.CONTENT_TYPE_BMPSTRING;
72      if (x509ca.getUseUTF8PolicyText()) {
73        displayencoding = DisplayText.CONTENT_TYPE_UTF8STRING;
74      }
75      // Iterate through policies and add oids and policy qualifiers if they exist
76      final List<CertificatePolicy> policies =
77          certProfile.getCertificatePolicies();
78      final Map<ASN1ObjectIdentifier, ASN1EncodableVector> policiesMap =
79          new LinkedHashMap<ASN1ObjectIdentifier, ASN1EncodableVector>();
80      // Each Policy OID can be entered several times, with different qualifiers,
81      // because of this we make a map of oid and qualifiers, and we can add a new
82      // qualifier
83      // in each round of this for loop
84      for (final Iterator<CertificatePolicy> it = policies.iterator();
85          it.hasNext();) {
86        final CertificatePolicy policy = it.next();
87        final ASN1ObjectIdentifier oid =
88            new ASN1ObjectIdentifier(policy.getPolicyID());
89        final ASN1EncodableVector qualifiers;
90        if (policiesMap.containsKey(oid)) {
91          qualifiers = policiesMap.get(oid);
92        } else {
93          qualifiers = new ASN1EncodableVector();
94        }
95        final PolicyQualifierInfo pqi =
96            getPolicyQualifierInformation(policy, displayencoding);
97        if (pqi != null) {
98          qualifiers.add(pqi);
99        }
100       policiesMap.put(oid, qualifiers);
101     }
102     final ASN1EncodableVector seq = new ASN1EncodableVector();
103     for (final Iterator<ASN1ObjectIdentifier> it =
104             policiesMap.keySet().iterator();
105         it.hasNext();) {
106       final ASN1ObjectIdentifier oid = it.next();
107       final ASN1EncodableVector qualifiers = policiesMap.get(oid);
108       if (qualifiers.size() == 0) {
109         seq.add(new PolicyInformation(oid, null));
110       } else {
111         seq.add(new PolicyInformation(oid, new DERSequence(qualifiers)));
112       }
113     }
114     if (seq.size() > 0) {
115       ret = new DERSequence(seq);
116     }
117     if (ret == null) {
118       LOG.warn("Certificate policies missconfigured, no policies present!");
119     }
120     return ret;
121   }
122 
123   /**
124    * Obtains the Policy Qualifier Information object.
125    *
126    * @param policy CertificatePolicy with oid, user notice and cps uri
127    * @param displayencoding the encoding used for UserNotice text,
128    *     DisplayText.CONTENT_TYPE_BMPSTRING, CONTENT_TYPE_UTF8STRING,
129    *     CONTENT_TYPE_IA5STRING or CONTENT_TYPE_VISIBLESTRING
130    * @return PolicyQualifierInfo
131    */
132   private PolicyQualifierInfo getPolicyQualifierInformation(
133       final CertificatePolicy policy, final int displayencoding) {
134     PolicyQualifierInfo pqi = null;
135     final String qualifierId = policy.getQualifierId();
136     if ((qualifierId != null) && !StringUtils.isEmpty(qualifierId.trim())) {
137       final String qualifier = policy.getQualifier();
138       if ((qualifier != null) && !StringUtils.isEmpty(qualifier.trim())) {
139         if (qualifierId.equals(PolicyQualifierId.id_qt_cps.getId())) {
140           pqi = new PolicyQualifierInfo(qualifier);
141         } else if (qualifierId.equals(
142             PolicyQualifierId.id_qt_unotice.getId())) {
143           // Normally we would just use 'DisplayText(unotice)' here. IE has
144           // problems with UTF8 though, so lets stick with BMSSTRING to satisfy
145           // Bills sick needs.
146           final UserNotice un =
147               new UserNotice(null, new DisplayText(displayencoding, qualifier));
148           pqi = new PolicyQualifierInfo(PolicyQualifierId.id_qt_unotice, un);
149         }
150       }
151     }
152     return pqi;
153   }
