13  package org.cesecore.certificates.certificate.certextensions.standard;
14  
15  import java.security.PublicKey;
16  
17  import org.bouncycastle.asn1.ASN1Encodable;
18  import org.bouncycastle.asn1.DERNull;
19  import org.bouncycastle.asn1.ocsp.OCSPObjectIdentifiers;
20  import org.cesecore.certificates.ca.CA;
21  import org.cesecore.certificates.ca.internal.CertificateValidity;
22  import org.cesecore.certificates.certificateprofile.CertificateProfile;
23  import org.cesecore.certificates.endentity.EndEntityInformation;
24  
25  /** OCSP nocheck extension (rfc 2560)
26   * 
27   * Class for standard X509 certificate extension. 
28   * See rfc3280 or later for spec of this extension.      
29   * 
30   * @version $Id: OcspNoCheck.java 22092 2015-10-26 13:58:55Z mikekushner $
31   */
32  public class OcspNoCheck extends StandardCertificateExtension {
33  	
34      private static final long serialVersionUID = 1L;
35  
36      @Override
37  	public void init(final CertificateProfile certProf) {
38  		super.setOID(OCSPObjectIdentifiers.id_pkix_ocsp_nocheck.getId());
39  		super.setCriticalFlag(false);
40  	}
41      
42      @Override
43      public ASN1Encodable getValue(final EndEntityInformation subject, final CA ca, final CertificateProfile certProfile,
44              final PublicKey userPublicKey, final PublicKey caPublicKey, CertificateValidity val) {
45  		return DERNull.INSTANCE;
46  	}	
47  }
