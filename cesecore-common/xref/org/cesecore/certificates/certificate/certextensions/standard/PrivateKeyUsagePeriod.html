14  package org.cesecore.certificates.certificate.certextensions.standard;
15  
16  import java.security.PublicKey;
17  import java.util.Date;
18  
19  import org.apache.log4j.Logger;
20  import org.bouncycastle.asn1.ASN1Encodable;
21  import org.bouncycastle.asn1.ASN1EncodableVector;
22  import org.bouncycastle.asn1.DERGeneralizedTime;
23  import org.bouncycastle.asn1.DERSequence;
24  import org.bouncycastle.asn1.DERTaggedObject;
25  import org.bouncycastle.asn1.x509.Extension;
26  import org.cesecore.certificates.ca.CA;
27  import org.cesecore.certificates.ca.internal.CertificateValidity;
28  import org.cesecore.certificates.certificate.certextensions.CertificateExtensionException;
29  import org.cesecore.certificates.certificateprofile.CertificateProfile;
30  import org.cesecore.certificates.endentity.EndEntityInformation;
31  
32  /**
33   * Class for standard X509 certificate extension. See rfc3280 or later for spec of this extension.
34   * 
35   * @version $Id: PrivateKeyUsagePeriod.java 24737 2016-11-15 13:53:25Z anatom $
36   */
37  public class PrivateKeyUsagePeriod extends StandardCertificateExtension {
38  
39      private static final long serialVersionUID = 1L;
40      /** Logger for this class. */
41      private static final Logger LOG = Logger.getLogger(PrivateKeyUsagePeriod.class);
42  
43      @Override
44      public void init(CertificateProfile certProf) {
45          super.setOID(Extension.privateKeyUsagePeriod.getId());
46          super.setCriticalFlag(false);
47      }
48  
49      @Override
50      public ASN1Encodable getValue(final EndEntityInformation subject, final CA ca, final CertificateProfile certProfile,
51              final PublicKey userPublicKey, final PublicKey caPublicKey, CertificateValidity val) throws 
52              CertificateExtensionException {
53          // Construct the start and end dates of PrivateKeyUsagePeriod
54          // As start date, use the same as the start date of the certificate
55          long start;
56          if (val != null) {
57              start = val.getNotBefore().getTime();
58          } else {
59              if (LOG.isDebugEnabled()) {
60                  LOG.debug("No validity passed in to getValue, using default 'Date().getTime() - CertificateValidity.getValidityOffset()'");
61              }
62              start = new Date().getTime() - CertificateValidity.getValidityOffset();
63          }
64          if (certProfile.isUsePrivateKeyUsagePeriodNotBefore()) {
65              start += certProfile.getPrivateKeyUsagePeriodStartOffset() * 1000;
66          }
67          Date notBefore = null;
68          Date notAfter = null;
69  
70          if (certProfile.isUsePrivateKeyUsagePeriodNotBefore()) {
71              notBefore = new Date(start);
72          }
73          if (certProfile.isUsePrivateKeyUsagePeriodNotAfter()) {
74              final long validity = certProfile.getPrivateKeyUsagePeriodLength(); // seconds
75              notAfter = new Date(start + validity * 1000);
76          }
77          if (LOG.isDebugEnabled()) {
78              LOG.debug("PrivateKeyUsagePeriod.notBefore: " + notBefore);
79              LOG.debug("PrivateKeyUsagePeriod.notAfter: " + notAfter);
80          }
81  
82          return privateKeyUsagePeriod(notBefore, notAfter);
83      }
84  
85      private static DERSequence privateKeyUsagePeriod(final Date notBefore, final Date notAfter) throws CertificateExtensionException {
86          // Create the extension.
87          // PrivateKeyUsagePeriod ::= SEQUENCE {
88          // notBefore [0] GeneralizedTime OPTIONAL,
89          // notAfter [1] GeneralizedTime OPTIONAL }
90          final ASN1EncodableVector v = new ASN1EncodableVector();
91          if (notBefore != null) {
92              v.add(new DERTaggedObject(false, 0, new DERGeneralizedTime(notBefore)));
93          }
94          if (notAfter != null) {
95              v.add(new DERTaggedObject(false, 1, new DERGeneralizedTime(notAfter)));
96          }
97          if (v.size() == 0) {
98              throw new CertificateExtensionException("At least one of notBefore and notAfter must be specified!");
99          }
100         return new DERSequence(v);
101     }
102 }
