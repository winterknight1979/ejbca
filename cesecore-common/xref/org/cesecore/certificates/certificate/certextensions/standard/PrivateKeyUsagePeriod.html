14  package org.cesecore.certificates.certificate.certextensions.standard;
15  
16  import java.security.PublicKey;
17  import java.util.Date;
18  import org.apache.log4j.Logger;
19  import org.bouncycastle.asn1.ASN1Encodable;
20  import org.bouncycastle.asn1.ASN1EncodableVector;
21  import org.bouncycastle.asn1.DERGeneralizedTime;
22  import org.bouncycastle.asn1.DERSequence;
23  import org.bouncycastle.asn1.DERTaggedObject;
24  import org.bouncycastle.asn1.x509.Extension;
25  import org.cesecore.certificates.ca.CA;
26  import org.cesecore.certificates.ca.internal.CertificateValidity;
27  import org.cesecore.certificates.certificate.certextensions.CertificateExtensionException;
28  import org.cesecore.certificates.certificateprofile.CertificateProfile;
29  import org.cesecore.certificates.endentity.EndEntityInformation;
30  
31  /**
32   * Class for standard X509 certificate extension. See rfc3280 or later for spec
33   * of this extension.
34   *
35   * @version $Id: PrivateKeyUsagePeriod.java 24737 2016-11-15 13:53:25Z anatom $
36   */
37  public class PrivateKeyUsagePeriod extends StandardCertificateExtension {
38    /** milliseconds. */
39    private static final int MS_PER_S = 1000;
40  
41    private static final long serialVersionUID = 1L;
42    /** Logger for this class. */
43    private static final Logger LOG =
44        Logger.getLogger(PrivateKeyUsagePeriod.class);
45  
46    @Override
47    public void init(final CertificateProfile certProf) {
48      super.setOID(Extension.privateKeyUsagePeriod.getId());
49      super.setCriticalFlag(false);
50    }
51  
52    @Override
53    public ASN1Encodable getValue(
54        final EndEntityInformation subject,
55        final CA ca,
56        final CertificateProfile certProfile,
57        final PublicKey userPublicKey,
58        final PublicKey caPublicKey,
59        final CertificateValidity val)
60        throws CertificateExtensionException {
61      // Construct the start and end dates of PrivateKeyUsagePeriod
62      // As start date, use the same as the start date of the certificate
63      long start;
64      if (val != null) {
65        start = val.getNotBefore().getTime();
66      } else {
67        if (LOG.isDebugEnabled()) {
68          LOG.debug(
69              "No validity passed in to getValue, using default"
70                  + " 'Date().getTime() -"
71                  + " CertificateValidity.getValidityOffset()'");
72        }
73        start = new Date().getTime() - CertificateValidity.getValidityOffset();
74      }
75      if (certProfile.isUsePrivateKeyUsagePeriodNotBefore()) {
76        start += certProfile.getPrivateKeyUsagePeriodStartOffset() * MS_PER_S;
77      }
78      Date notBefore = null;
79      Date notAfter = null;
80  
81      if (certProfile.isUsePrivateKeyUsagePeriodNotBefore()) {
82        notBefore = new Date(start);
83      }
84      if (certProfile.isUsePrivateKeyUsagePeriodNotAfter()) {
85        final long validity =
86            certProfile.getPrivateKeyUsagePeriodLength(); // seconds
87        notAfter = new Date(start + validity * MS_PER_S);
88      }
89      if (LOG.isDebugEnabled()) {
90        LOG.debug("PrivateKeyUsagePeriod.notBefore: " + notBefore);
91        LOG.debug("PrivateKeyUsagePeriod.notAfter: " + notAfter);
92      }
93  
94      return privateKeyUsagePeriod(notBefore, notAfter);
95    }
96  
97    private static DERSequence privateKeyUsagePeriod(
98        final Date notBefore, final Date notAfter)
99        throws CertificateExtensionException {
100     // Create the extension.
101     // PrivateKeyUsagePeriod ::= SEQUENCE {
102     // notBefore [0] GeneralizedTime OPTIONAL,
103     // notAfter [1] GeneralizedTime OPTIONAL }
104     final ASN1EncodableVector v = new ASN1EncodableVector();
105     if (notBefore != null) {
106       v.add(new DERTaggedObject(false, 0, new DERGeneralizedTime(notBefore)));
107     }
108     if (notAfter != null) {
109       v.add(new DERTaggedObject(false, 1, new DERGeneralizedTime(notAfter)));
110     }
111     if (v.size() == 0) {
112       throw new CertificateExtensionException(
113           "At least one of notBefore and notAfter must be specified!");
114     }
115     return new DERSequence(v);
116   }
117 }
