38  /**
39   * Extension for Name Constraints. <a
40   * href="https://tools.ietf.org/html/rfc5280#section-4.2.1.10">RFC 5280</a>
41   *
42   * <p>For storing Name Constraints, an internal encoded form is used. The format
43   * is "type-id:data" where data is either a regular string or hex-encoded data,
44   * depending on the type. Use parseNameConstraintList to convert human-readable
45   * strings into encoded strings.
46   *
47   * @version $Id: NameConstraint.java 29904 2018-09-26 13:14:30Z samuellb $
48   */
49  public class NameConstraint extends StandardCertificateExtension {
50  
51    private static final long serialVersionUID = 1L;
52  
53    @Override
54    public void init(final CertificateProfile certProf) {
55      super.setOID(Extension.nameConstraints.getId());
56      super.setCriticalFlag(certProf.getNameConstraintsCritical());
57    }
58  
59    @Override
60    public ASN1Encodable getValue(
61        final EndEntityInformation userData,
62        final CA ca,
63        final CertificateProfile certProfile,
64        final PublicKey userPublicKey,
65        final PublicKey caPublicKey,
66        final CertificateValidity val)
67        throws CertificateExtensionException {
68      NameConstraints nc = null;
69  
70      if (!(ca instanceof X509CA)) {
71        throw new CertificateExtensionException(
72            "Can't issue non-X509 certificate with Name Constraint");
73      }
74  
75      final ExtendedInformation ei = userData.getExtendedInformation();
76      if (ei != null) {
77        final List<String> permittedNames = ei.getNameConstraintsPermitted();
78        final List<String> excludedNames = ei.getNameConstraintsExcluded();
79  
80        // Do not include an empty name constraints extension
81        if (permittedNames != null || excludedNames != null) {
82          final GeneralSubtree[] permitted = toGeneralSubtrees(permittedNames);
83          final GeneralSubtree[] excluded = toGeneralSubtrees(excludedNames);
84  
85          nc = new NameConstraints(permitted, excluded);
86        }
87      }
88  
89      return nc;
90    }

92    /**
93     * Converts a list of encoded strings of Name Constraints into ASN1
94     * GeneralSubtree objects. This is needed when creating an BouncyCastle ASN1
95     * NameConstraint object for inclusion in a certificate.
96     *
97     * @param list encoded strings
98     * @return ASN1 subtree
99     */
100   public static GeneralSubtree[] toGeneralSubtrees(final List<String> list) {
101     if (list == null) {
102       return null;
103     }
104 
105     GeneralSubtree[] ret = new GeneralSubtree[list.size()];
106     int i = 0;
107     for (String entry : list) {
108       int type = getNameConstraintType(entry);
109       Object data = getNameConstraintData(entry);
110       GeneralName genname;
111       switch (type) {
112         case GeneralName.dNSName:
113         case GeneralName.rfc822Name:
114           genname = new GeneralName(type, (String) data);
115           break;
116         case GeneralName.directoryName:
117           genname =
118               new GeneralName(
119                   new X500Name(CeSecoreNameStyle.INSTANCE, (String) data));
120           break;
121         case GeneralName.iPAddress:
122           genname = new GeneralName(type, new DEROctetString((byte[]) data));
123           break;
124         default:
125           throw new UnsupportedOperationException(
126               "Encoding of name constraint type "
127                   + type
128                   + " is not implemented.");
129       }
130       ret[i++] = new GeneralSubtree(genname);
131     }
132     return ret;
133   }
134 
135   /**
136    * @param encoded name constraint
137    * @return the GeneralName type code for an encoded Name Constraint.
138    */
139   private static int getNameConstraintType(final String encoded) {
140     String typeString = encoded.split(":", 2)[0];
141     if ("iPAddress".equals(typeString)) {
142       return GeneralName.iPAddress;
143     }
144     if ("dNSName".equals(typeString)) {
145       return GeneralName.dNSName;
146     }
147     if ("directoryName".equals(typeString)) {
148       return GeneralName.directoryName;
149     }
150     if ("rfc822Name".equals(typeString)) {
151       return GeneralName.rfc822Name;
152     }
153     throw new UnsupportedOperationException(
154         "Unsupported name constraint type " + typeString);
155   }
156 
157   /**
158    * Returns the GeneralName data (as a byte array or String) from an encoded
159    * string.
160    *
161    * @param encoded string
162    * @return GeneralName
163    */
164   private static Object getNameConstraintData(final String encoded) {
165     int type = getNameConstraintType(encoded);
166     String data = encoded.split(":", 2)[1];
167 
168     switch (type) {
169       case GeneralName.dNSName:
170       case GeneralName.directoryName:
171       case GeneralName.rfc822Name:
172         return data;
173       case GeneralName.iPAddress:
174         try {
175           return Hex.decodeHex(data.toCharArray());
176         } catch (DecoderException e) {
177           throw new IllegalStateException(
178               "internal name constraint data could not be decoded as hex", e);
179         }
180       default:
181         throw new UnsupportedOperationException(
182             "Unsupported name constraint type " + type);
183     }
184   }

186   /**
187    * Parses a single name constraint entry in human-readable form into an
188    * encoded string for database storage etc. The intention is to make it
189    * possible to change the human readable form at a later point.
190    *
191    * <p>This format is essentially a hex string representation of a RFC 5280
192    * GeneralName, but only DNS Names and IP Addresses are supported so far.
193    *
194    * @param str constraint
195    * @return encoded string
196    * @throws CertificateExtensionException if the string can not be parsed.
197    */
198   public static String parseNameConstraintEntry(final String str)
199       throws CertificateExtensionException {
200     if (str.matches("^([0-9]+\\.){3,3}([0-9]+)/[0-9]+$")
201         || str.matches(
202             "^[0-9a-fA-F]{0,4}:[0-9a-fA-F]{0,4}:[0-9a-fA-F:]*/[0-9]+$")) {
203       // IPv4 or IPv6 address
204       try {
205         String[] pieces = str.split("/", 2);
206         byte[] addr = InetAddress.getByName(pieces[0]).getAddress();
207         byte[] encoded =
208             new byte[2 * addr.length]; // will hold address and netmask
209         System.arraycopy(addr, 0, encoded, 0, addr.length);
210 
211         // The second half in the encoded form is the netmask
212         int netmask = Integer.parseInt(pieces[1]);
213         if (netmask > 8 * addr.length) {
214           throw new CertificateExtensionException(
215               "Netmask is too large: " + str);
216         }
217         for (int i = 0; i < netmask; i++) {
218           encoded[addr.length + i / 8] |= 1 << (7 - i % 8);
219         }
220         // Clear host part from IP address
221         for (int i = netmask; i < 8 * addr.length; i++) {
222           encoded[i / 8] &= ~(1 << (7 - i % 8));
223         }
224         return "iPAddress:" + Hex.encodeHexString(encoded);
225       } catch (UnknownHostException e) {
226         throw new CertificateExtensionException(
227             "Failed to parse IP address in name constraint: " + str, e);
228       }
229     } else if (str.matches("^([0-9]+\\.){3,3}([0-9]+)$")) {
230       // IP address without netmask. This is not a valid DNS name, so catch it
231       // here.
232       throw new CertificateExtensionException(
233           "Name constraint entry with IP address is missing a netmask: "
234               + str
235               + ". Use /32 to match only this address.");
236     } else if (str.matches("^\\.?([a-zA-Z0-9_-]+\\.)*[a-zA-Z0-9_-]+$")) {
237       // DNS name (it can start with a ".", this means "all subdomains")
238       return "dNSName:" + str;
239     } else if (str.matches("^[^=,]*@[a-zA-Z0-9_.\\[\\]:-]+$")) {
240       String email = str;
241       // RFC 822 Name (i.e. e-mail)
242       if (str.startsWith("@")) {
243         // In EJBCA, rfc822Names without a user part start with @ to distinguish
244         // them from domain names.
245         // This is not the case in the encoded form.
246         email = email.substring(1);
247       }
248       return "rfc822Name:" + email;
249     } else if (str.contains("=")) {
250       // Directory name
251       return "directoryName:"
252           + new X500Name(CeSecoreNameStyle.INSTANCE, str).toString();
253     } else {
254       throw new CertificateExtensionException(
255           "Cannot parse name constraint entry (only DNS Name, RFC 822 Name,"
256               + " Directory Name, IPv4/Netmask and IPv6/Netmask are"
257               + " supported): "
258               + str);
259     }
260   }

262   /**
263    * Parses human readable name constraints, one entry per line, into a list of
264    * encoded name constraints.
265    *
266    * @param input Constraints
267    * @return encoded constraints
268    * @throws CertificateExtensionException if the string cannot be parsed
269    * @see #parseNameConstraintEntry
270    */
271   public static List<String> parseNameConstraintsList(final String input)
272       throws CertificateExtensionException {
273     List<String> encodedNames = new ArrayList<>();
274     if (input != null) {
275       String[] pieces = input.split("\n");
276       for (String piece : pieces) {
277         piece = piece.trim();
278         if (!piece.isEmpty()) {
279           encodedNames.add(NameConstraint.parseNameConstraintEntry(piece));
280         }
281       }
282     }
283     return encodedNames;
284   }

286   /**
287    * Formats an encoded name constraint from parseNameConstraintEntry into
288    * human-readable form.
289    *
290    * @param encoded encoded string
291    * @return human-readable
292    */
293   public static String formatNameConstraintEntry(final String encoded) {
294     if (encoded == null) {
295       return "";
296     }
297 
298     int type = getNameConstraintType(encoded);
299     Object data = getNameConstraintData(encoded);
300 
301     switch (type) {
302       case GeneralName.dNSName:
303       case GeneralName.directoryName:
304         return (String) data; // not changed during encoding
305       case GeneralName.iPAddress:
306         byte[] bytes = (byte[]) data;
307         byte[] ip = new byte[bytes.length / 2];
308         byte[] netmaskBytes = new byte[bytes.length / 2];
309         System.arraycopy(bytes, 0, ip, 0, ip.length);
310         System.arraycopy(
311             bytes, ip.length, netmaskBytes, 0, netmaskBytes.length);
312 
313         int netmask = 0;
314         for (int i = 0; i < 8 * netmaskBytes.length; i++) {
315           final boolean one = (netmaskBytes[i / 8] >> (7 - i % 8) & 1) == 1;
316           if (one && netmask == i) {
317             netmask++; // leading ones
318           } else if (one) {
319             // trailings ones = error!
320             throw new IllegalArgumentException(
321                 "Unsupported netmask with mixed ones/zeros");
322           }
323         }
324 
325         try {
326           return InetAddress.getByAddress(ip).getHostAddress() + "/" + netmask;
327         } catch (UnknownHostException e) {
328           throw new IllegalArgumentException(e);
329         }
330       case GeneralName.rfc822Name:
331         // Prepend @ is it's only the domain part to distinguish from DNS names
332         String str = (String) data;
333         return (str.contains("@") ? str : "@" + str);
334       default:
335         throw new UnsupportedOperationException(
336             "Unsupported name constraint type " + type);
337     }
338   }

340   /**
341    * Formats an encoded list of name constraints into a human-readable list,
342    * with one entry per line.
343    *
344    * @param encodedList List of constraints
345    * @return a newline-separated string of encoded name constraints
346    */
347   public static String formatNameConstraintsList(
348       final List<String> encodedList) {
349     final StringBuilder sb = new StringBuilder();
350     if (encodedList != null) {
351       boolean first = true;
352       for (String encodedName : encodedList) {
353         if (!first) {
354           sb.append('\n');
355         }
356         first = false;
357         sb.append(
358             formatNameConstraintEntry(
359                 encodedName)); // notice the call to formatNameConstraintEntry,
360                                // so this is different from StringUtils.join
361       }
362     }
363     return sb.toString();
364   }
