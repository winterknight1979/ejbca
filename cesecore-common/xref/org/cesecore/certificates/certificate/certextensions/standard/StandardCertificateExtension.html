13  package org.cesecore.certificates.certificate.certextensions.standard;
14  
15  import org.cesecore.certificates.certificate.certextensions.CertificateExtension;
16  import org.cesecore.certificates.certificateprofile.CertificateProfile;
17  
18  /**
19   * Base class for a standard certificate extension.
20   * All standard extensions should inherit this class.
21   * 
22   * The methods that need implementation is init frmm here and getValue from the super class.
23   * init should call setOID and setCriticalFlag from the super class.
24   * Implementing class must have a default constructor, calling super constructor.
25   * 
26   * @version $Id: StandardCertificateExtension.java 21979 2015-10-07 14:06:49Z mikekushner $
27   */
28  public abstract class StandardCertificateExtension extends CertificateExtension {
29  	
30  	private static final long serialVersionUID = 1L;
31  
32      /**
33  	 * Method that initializes the CertificateExtension
34  	 * 
35  	 * @param certProf certificateprofile that defines if this extension is used and critical
36  	 */
37  	public abstract void init(CertificateProfile certProf);
38  	
39  }
