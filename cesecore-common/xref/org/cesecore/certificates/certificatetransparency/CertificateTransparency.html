13  package org.cesecore.certificates.certificatetransparency;
14  
15  import java.security.cert.Certificate;
16  import java.security.cert.X509Certificate;
17  import java.util.Collection;
18  import java.util.List;
19  import org.bouncycastle.cert.X509v3CertificateBuilder;
20  import org.cesecore.certificates.certificate.CertificateCreateException;
21  import org.cesecore.certificates.certificateprofile.CertificateProfile;
22  
23  /**
24   * X509CA will attempt to load
25   * org.cesecore.certificates.certificatetransparency.CertificateTransparencyImpl
26   * that must implement this interface if it exists.
27   *
28   * <p>The reason why this is an interface is because the CT support is a
29   * separate feature not included in standard EJBCA.
30   *
31   * @version $Id: CertificateTransparency.java 29261 2018-06-18 10:54:26Z
32   *     bastianf $
33   */
34  public interface CertificateTransparency {
35    /** OID. */
36    String SCTLIST_OID = "1.3.6.1.4.1.11129.2.4.2";
37  
38    /**
39     * Controls which parameters are used for minimum and maximum number of SCTs.
40     * Also, for OCSP and PUBLISHER, the {@link
41     * CertificateProfile#isUseCTSubmitExisting} will be respected.
42     */
43    enum UsageMode {
44      /** Requesting a SCT for a new certificate to be issued. */
45      CERTIFICATE,
46      /** Requesting a SCT for inclusion in a OCSP response. */
47      OCSP,
48      /** Publishing a certificate to all CT logs. */
49      PUBLISHER;
50    };
51  
52    /**
53     * Overloaded method with usageMode = UsageMode.CERTIFICATE.
54     *
55     * @param chain Certificate chain
56     * @param certProfile Certificate Profile
57     * @param config Config
58     * @return SCT List
59     * @throws CTLogException If too many servers are down to satisfy the
60     *     certificate profile.
61     * @see CertificateTransparency#fetchSCTList(List, CertificateProfile,
62     *     CTSubmissionConfigParams, UsageMode)
63     */
64    byte[] fetchSCTList(
65        List<Certificate> chain,
66        CertificateProfile certProfile,
67        CTSubmissionConfigParams config)
68        throws CTLogException;
69  
70    /**
71     * Tries to add a certificate to CT logs and obtain SCTs (Signed Certificate
72     * Timestamps). The configuration is taken from the certificate profile.
73     *
74     * @param chain Certificate chain including any CT signer and the leaf
75     *     pre-certificate
76     * @param certProfile Certificate profile with CT configuration
77     * @param config Configuration parameters, that are not specific to the
78     *     certificate profile.
79     * @param usageMode Why we are fetching SCTs. The minimum and maximum number
80     *     of SCTs are different depending on this.
81     * @return A "SCT List" structure, for inclusion in e.g. the CT certificate
82     *     extension, or null if no logs have been configured.
83     * @throws CTLogException If too many servers are down to satisfy the
84     *     certificate profile.
85     */
86    byte[] fetchSCTList(
87        List<Certificate> chain,
88        CertificateProfile certProfile,
89        CTSubmissionConfigParams config,
90        UsageMode usageMode)
91        throws CTLogException;
92  
93    /**
94     * Adds a critical extension to prevent the certificate from being used.
95     *
96     * @param precertbuilder builder
97     */
98    void addPreCertPoison(X509v3CertificateBuilder precertbuilder);
99  
100   /**
101    * @return true if the given certificate has an SCT extension with at least
102    *     one entry.
103    * @param cert certificate
104    */
105   boolean hasSCTs(Certificate cert);
106 
107   /**
108    * Clears the URL availability status (fast fail) cache. To clear the OCSP CT
109    * cache, use OcspExtensionsCache.reloadCache()
110    */
111   void clearCaches();
112 
113   /**
114    * Ensure that all SCTs in the certificate are valid or throw an exception. No
115    * checks are performed if the certificate given as input does not have an SCT
116    * extension present.
117    *
118    * @param leafCertificate The final certificate to check
119    * @param issuerCertificates Other certs in the chain
120    * @param ctLogs Logs
121    * @throws CertificateCreateException if one of the SCTs are invalid or if the
122    *     certificate could not be parsed
123    */
124   void allSctsAreValidOrThrow(
125       X509Certificate leafCertificate,
126       List<Certificate> issuerCertificates,
127       Collection<CTLogInfo> ctLogs)
128       throws CertificateCreateException;
129 }
