14  package org.cesecore.certificates.ocsp.extension;
15  
16  
17  import java.security.cert.X509Certificate;
18  import java.util.Map;
19  import java.util.Set;
20  
21  import org.bouncycastle.asn1.ASN1ObjectIdentifier;
22  import org.bouncycastle.asn1.x509.Extension;
23  import org.bouncycastle.cert.ocsp.CertificateStatus;
24  import org.cesecore.keybind.InternalKeyBinding;
25  
26  /**
27   * Interface that must be implemented by OCSP extensions that are added to the OCSPServlet
28   * 
29   * @version $Id: OCSPExtension.java 28881 2018-05-08 18:18:11Z anatom $
30   */
31  public interface OCSPExtension {
32  
33      /** Called after construction
34       * 
35       */
36      void init();
37  
38      /**
39       * Called by OCSP responder when the configured extension is found in the request.
40       * 
41       * @param requestCertificates
42       *            A certificate array from the original HttpServletRequest, used for authorization.
43       * @param remoteAddress
44       *            Extracted from the HttpServletRequest.
45       * @param remoteHost
46       *            DEPRECATED. Currently set to the same as remoteAddress. An extension that relies on this value must perform the remote lookup by itself.
47       * @param cert
48       *            X509Certificate the caller asked for in the OCSP request
49       * @param status
50       *            CertificateStatus the status the certificate has according to the OCSP responder, null means the cert is good
51       * 
52       * @param internalKeyBinding
53       *            Used to get the trusted ca cert etc.
54       * 
55       * @return Hashtable with X509Extensions &lt;String oid, X509Extension ext&gt; that will be added to responseExtensions by OCSP responder, or null if an
56       *         error occurs
57       */
58      Map<ASN1ObjectIdentifier, Extension> process(final X509Certificate[] requestCertificates, final String remoteAddress, final String remoteHost,
59              final X509Certificate cert, final CertificateStatus status, final InternalKeyBinding internalKeyBinding);
60  
61      /**
62       * Returns the last error that occured during process(), when process returns null
63       * 
64       * @return error code as defined by implementing class
65       */
66      int getLastErrorCode();
67      
68      /**
69       * Returns a set of valid positions where the extension may be added.
70       * 
71       * @return a set of valid positions where the extension may be added. The returned Set should contain at least one value. Never returns <code>null</code>.
72       */
73      Set<OCSPExtensionType> getExtensionType();
74      
75      /**
76       * @return Extension OID
77       */
78      String getOid();
79      
80      /**
81       * @return Human readable extension name
82       */
83      String getName();
84  }
