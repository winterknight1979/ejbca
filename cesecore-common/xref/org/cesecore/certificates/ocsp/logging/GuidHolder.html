13  package org.cesecore.certificates.ocsp.logging;
14  
15  import org.cesecore.util.GUIDGenerator;
16  
17  /**
18   * Keeps track of a GUID which identifies this instance.
19   *
20   * @version $Id: GuidHolder.java 17625 2013-09-20 07:12:06Z netmackan $
21   */
22  public enum GuidHolder {
23    /** Singleton instance. */
24      INSTANCE;
25  
26      /** Constructor. */
27    GuidHolder() {
28      guid = GUIDGenerator.generateGUID(this);
29    }
30  
31    /**
32     * @return GUID
33     */
34    public String getGlobalUid() {
35      return guid;
36    }
37  
38    /** GUID.*/
39    private final String guid;
40  }
