30  /**
31   * This class can be extended to create highly configurable log classes. Values
32   * that are to be logged are stored in a Hashmap and the output is configured
33   * using a Java.util.regex.Matcher and a sortString. The extending classes also
34   * need to supply a Logger and a String specifying how to log Dates.
35   *
36   * <p>Use paramPut(String key, String value) to add values, Use writeln() to log
37   * all the stored values and then use flush() to store them to file.
38   *
39   * <p>Roughly based on PatternLogger.java 8663 2010-02-17 10:42:41Z anatom from
40   * EJBCA
41   *
42   * @version $Id: PatternLogger.java 31242 2019-01-17 22:41:24Z jeklund $
43   */
44  public abstract class PatternLogger implements Serializable {
45  
46    private static final long serialVersionUID = 8486004615125959046L;
47  /** The Date and time the request. */
48    public static final String LOG_TIME =
49        "LOG_TIME";
50  /** An integer identifying a log entry for a request. */
51    public static final String LOG_ID =
52        "LOG_ID";
53  /** A random 32 bit number identifying a log entry for a request. */
54    public static final String SESSION_ID =
55        "SESSION_ID";
56  
57    /** Time. */
58    public static final String REPLY_TIME = "REPLY_TIME";
59  
60    /** Hash of the issuer DN. */
61    public static final String ISSUER_NAME_HASH = "ISSUER_NAME_HASH";
62    /** The public key of the issuer of a requested certificate. */
63    public static final String ISSUER_KEY = "ISSUER_KEY";
64    /** Serial number of the requested certificate. */
65    public static final String SERIAL_NOHEX = "SERIAL_NOHEX";
66    /** IP of the client making the reques. */
67    public static final String CLIENT_IP = "CLIENT_IP";
68    /**
69     * The status of the OCSP-Request. SUCCESSFUL = 0;MALFORMED_REQUEST =
70     * 1;INTERNAL_ERROR = 2;
71     */
72    public static final String STATUS = "STATUS";
73  
74    /**
75     * PROCESS_TIME is a marker that is used to record the total time a request
76     * takes to process, excluding reading the request. It is replaced with the
77     * correct value when the log entry is written. the time measurement start
78     * when this param is set in the logger with:
79     *
80     * <pre>
81     * ocspTransactionLogger(PROCESS_TIME, PROCESS_TIME);
82     * </pre>
83     *
84     * This means that this variable can be used to measure any time you want to
85     * measure in your code.
86     */
87    public static final String PROCESS_TIME = "PROCESS_TIME";

89    /** Values. */
90    private final Map<String, String> valuepairs = new HashMap<String, String>();
91    /** Matcher. */
92    private final String matchString;
93    /** Match. */
94    private final String matchPattern;
95    // Matcher is not Serializable
96    /** Matcher. */
97    private transient Matcher m;
98    /** Order. */
99    private final String orderString;
100   /** Rime. */
101   private final Date startTime;
102   /** Time. */
103   private Date startProcessTime = null;
104   /** Logger. */
105   private boolean doLogging;
106   /** Logger. */
107   private final Class<?> loggerClass;
108   // Logger is not Serializable
109   /** Logger. */
110   private transient Logger logger;
111 
112   // for writing the output
113   /** Writer. */
114   private transient StringWriter sw;
115   /** Writer. */
116   private transient PrintWriter pw;
117 
118   /**
119    * @param adoLogging True if you want this pattern logger to do anything upon
120    *     flush.
121    * @param aLoggerClass The Class to create Log4j logger for, to log to if
122    *     doLogging is true
123    * @param aMatchPattern A string to create a matcher
124    *     that is used together with
125    *     matchString to determine how output is formatted
126    * @param aMatchString A string that matches the pattern
127    *     in m and specifies the
128    *     order in which values are logged by the logger
129    * @param logDateFormat A string that specifies how the log-time is formatted
130    * @param timeZone Time Zone
131    */
132   protected PatternLogger(
133       final boolean adoLogging,
134       final Class<?> aLoggerClass,
135       final String aMatchPattern,
136       final String aMatchString,
137       final String logDateFormat,
138       final String timeZone) {
139     this.doLogging = adoLogging;
140     this.matchString = aMatchString;
141     this.matchPattern = aMatchPattern;
142     this.orderString = aMatchString;
143     this.loggerClass = aLoggerClass;
144     this.startTime = new Date();
145     final FastDateFormat dateformat;
146     if (timeZone == null) {
147       dateformat = FastDateFormat.getInstance(logDateFormat);
148     } else {
149       dateformat =
150           FastDateFormat.getInstance(
151               logDateFormat, TimeZone.getTimeZone(timeZone));
152     }
153     paramPut(LOG_TIME, dateformat.format(new Date()));
154     this.paramPut(REPLY_TIME, REPLY_TIME);
155     this.paramPut(LOG_ID, "0");
156   }
157 
158   private Matcher getMatcher() {
159     if (this.m == null) {
160       // We have to instantiate the Matcher in the class and can not have it as
161       // an instance variable.
162       // This is because we are sending this object to a remote EJB (at least in
163       // system tests)
164       this.m = Pattern.compile(matchPattern).matcher(matchString);
165     }
166     return this.m;
167   }
168 
169   private Logger getLogger() {
170     if (this.logger == null) {
171       // We have to instantiate the logger in the class and can not have it as
172       // an instance variable.
173       // This is because we are sending this object to a remote EJB (at least in
174       // system tests) and org.apache.log4j.Logger is not serializable.
175       this.logger = Logger.getLogger(loggerClass);
176     }
177     return this.logger;
178   }
179 
180   private PrintWriter getPrintWriter() {
181     if (this.pw == null) {
182       this.sw = new StringWriter();
183       this.pw = new PrintWriter(this.sw);
184     }
185     return pw;
186   }
187 
188   /** @return output to be logged */
189   private String interpolate() {
190     final StringBuffer sb = new StringBuffer(this.orderString.length());
191     final Matcher matcher = getMatcher();
192     matcher.reset();
193     while (matcher.find()) {
194       // when the pattern is ${identifier}, group 0 is 'identifier'
195       final String key = matcher.group(1);
196       final String value = this.valuepairs.get(key);
197 
198       // if the pattern does exists, replace it by its value
199       // otherwise keep the pattern ( it is group(0) )
200       if (value != null) {
201         matcher.appendReplacement(sb, value);
202       } else {
203         // I'm doing this to avoid the backreference problem as there will be a
204         // $
205         // if I replace directly with the group 0 (which is also a pattern)
206         matcher.appendReplacement(sb, "");
207         final String unknown = matcher.group(0);
208         sb.append(unknown);
209       }
210     }
211     matcher.appendTail(sb);
212     return sb.toString();
213   }

215   /**
216    * Hex-encodes the bytes. method that makes sure that a "" is inserted instead
217    * of null.
218    *
219    * @param key key
220    * @param value value
221    */
222   public void paramPut(final String key, final byte[] value) {
223     paramPut(key, new String(Hex.encode(value)));
224   }
225 
226   /**
227    * method that makes sure that a "" is inserted instead of null.
228    *
229    * @param key key
230    * @param value value
231    */
232   public void paramPut(final String key, final String value) {
233     // logger.debug("paramput: "+ key+ ";" +value +";" +valuepairs.toString());
234     if (value == null) {
235       this.valuepairs.put(key, "");
236     } else {
237       this.valuepairs.put(key, value);
238     }
239     if (StringUtils.equals(key, PROCESS_TIME)) {
240       startProcessTime = new Date();
241     }
242   }
243 
244   /**
245    * method that makes sure that a "" is inserted instead of null.
246    *
247    * @param key key
248    * @param value value
249    */
250   public void paramPut(final String key, final Integer value) {
251     if (value == null) {
252       this.valuepairs.put(key, "");
253     } else {
254       this.valuepairs.put(key, value.toString());
255     }
256   }
257 
258   /** Method used for creating a log row of all added values. */
259   public void writeln() {
260     if (doLogging) {
261       getPrintWriter().println(interpolate());
262     }
263   }
264 
265   /** Writes all the rows created by writeln() to the Logger. */
266   public void flush() {
267     if (doLogging) {
268       getPrintWriter().flush();
269       String output = this.sw.toString();
270       output =
271           output.replaceAll(
272               REPLY_TIME,
273               String.valueOf(new Date().getTime() - this.startTime.getTime()));
274       if (startProcessTime != null) {
275         output =
276             output.replaceAll(
277                 PROCESS_TIME,
278                 String.valueOf(
279                     new Date().getTime() - this.startProcessTime.getTime()));
280       }
281       // Remove an extra linebreak between flushes.
282       if (output.endsWith(System.getProperty("line.separator"))) {
283         output = output.substring(0, output.length() - 1);
284       }
285       getLogger()
286           .debug(output); // Finally output the log row to the logging device
287     }
288   }
289 
290   /** @return true if this logger is enabled */
291   public boolean isEnabled() {
292     return doLogging && getLogger().isDebugEnabled();
293   }
