13  package org.cesecore.certificates.ocsp.logging;
14  
15  import org.cesecore.certificates.certificateprofile.CertificateProfileConstants;
16  import org.cesecore.certificates.crl.RevocationReasons;
17  import org.cesecore.config.OcspConfiguration;
18  
19  /**
20   * OCSP transaction logger.
21   * 
22   * @version $Id: TransactionLogger.java 25218 2017-02-08 13:45:12Z anatom $
23   *
24   */
25  public class TransactionLogger extends PatternLogger {
26  
27      private static final long serialVersionUID = 1722398387512931482L;
28      /**
29       * The BC normalized subject Distinguished Name of the client making the request
30       */
31      public static final String REQ_NAME = "REQ_NAME";
32      /** 
33       * The Unnormalized subject Distinguished Name of the client making the request
34       */
35      public static final String REQ_NAME_RAW = "REQ_NAME_RAW";
36      /**
37       * The BC normalized issuer Distinguished Name of the certificate used to sign the request.
38       */
39      public static final String SIGN_ISSUER_NAME_DN = "SIGN_ISSUER_NAME_DN";
40      /**
41       * The BC normalized Subject Distinguished Name of the certificate used to sign the request.
42       */
43      public static final String SIGN_SUBJECT_NAME = "SIGN_SUBJECT_NAME";
44      /**
45       * Certificate serial number of the certificate used to sign the request.
46       */
47      public static final String SIGN_SERIAL_NO = "SIGN_SERIAL_NO";
48      /**
49       * The BC normalized issuer Distinguished Name of the requested certificate
50       */
51      public static final String ISSUER_NAME_DN = "ISSUER_NAME_DN";
52      /**
53       * The unnormalized issuer Distinguished Name of the requested certificate
54       */
55      public static final String ISSUER_NAME_DN_RAW = "ISSUER_NAME_DN_RAW";
56      
57      /**
58       * Algorithm used by requested certificate to hash issuer key and issuer name
59       */
60      public static final String DIGEST_ALGOR = "DIGEST_ALGOR";
61      
62      /**
63       * The number of certificates to check revocation status for
64       */
65      public static final String NUM_CERT_ID = "NUM_CERT_ID";
66  
67      /**
68       * The requested certificate revocation status, 0=good, 1=revoked, 2=unknown (see OCSPResponseItem).
69       */
70      public static final String CERT_STATUS = "CERT_STATUS";
71  
72      /**
73       * The requested certificate revocation reason, or NOT_REVOKED, if unrevoked (see RevocationReasons).
74       * Set to 6 (certificateHold) when certificate is unknown, even if status returned is good.
75       */
76      public static final String REV_REASON = "REV_REASON";
77  
78      /** The id of the certificate profile that was used to issue the requested certificate. */
79      public static final String CERT_PROFILE_ID = "CERT_PROFILE_ID";
80  
81      /** The HTTP X-Forwarded-For header value. */
82      public static final String FORWARDED_FOR = "FORWARDED_FOR";
83  
84      public TransactionLogger(Integer logId, String sessionId, String clientIp) {
85          super(OcspConfiguration.getTransactionLog(), TransactionLogger.class, OcspConfiguration.getTransactionLogPattern(),
86                  OcspConfiguration.getTransactionLogOrder(), OcspConfiguration.getLogDateFormat(), OcspConfiguration.getLogTimeZone());
87          paramPut(PatternLogger.LOG_ID, logId);
88          paramPut(PatternLogger.SESSION_ID, sessionId);
89          paramPut(PatternLogger.CLIENT_IP, clientIp);
90          paramPut(PatternLogger.STATUS, "0");
91          paramPut(REQ_NAME, "0");
92          paramPut(REQ_NAME_RAW, "0");
93          paramPut(SIGN_ISSUER_NAME_DN, "0");
94          paramPut(SIGN_SUBJECT_NAME, "0");
95          paramPut(SIGN_SERIAL_NO, "0");
96          paramPut(NUM_CERT_ID, "0");
97          paramPut(ISSUER_NAME_DN, "0");
98          paramPut(ISSUER_NAME_DN_RAW, "0");
99          paramPut(PatternLogger.ISSUER_NAME_HASH, "0");
100         paramPut(PatternLogger.ISSUER_KEY, "0");
101         paramPut(DIGEST_ALGOR, "0");
102         paramPut(PatternLogger.SERIAL_NOHEX, "0");
103         paramPut(CERT_STATUS, "0");
104         paramPut(PatternLogger.PROCESS_TIME, "-1");
105         paramPut(CERT_PROFILE_ID, String.valueOf(CertificateProfileConstants.CERTPROFILE_NO_PROFILE));
106         paramPut(FORWARDED_FOR, "");
107         paramPut(REV_REASON, String.valueOf(RevocationReasons.NOT_REVOKED.getDatabaseValue()));
108     }
109 }
