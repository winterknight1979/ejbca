23  public class AuditLogger extends PatternLogger {
24      
25      private static final long serialVersionUID = 4474243795289459488L;
26      /**
27       * The byte[] ocsp-request that came with the http-request
28       */
29      public static final String OCSPREQUEST = "OCSPREQUEST";
30      /**
31       * The byte[] ocsp-response that was included in the http-response
32       */
33      public static final String OCSPRESPONSE = "OCSPRESPONSE";
34  
35      public AuditLogger(String ocspRequest, Integer logId, String sessionId, String clientIp) {
36          super(OcspConfiguration.getAuditLog(), AuditLogger.class, OcspConfiguration.getAuditLogPattern(), OcspConfiguration.getAuditLogOrder(), OcspConfiguration.getLogDateFormat(), OcspConfiguration.getLogTimeZone());
37  
38          paramPut(OCSPREQUEST, ocspRequest);
39          paramPut(PatternLogger.LOG_ID, logId);
40          paramPut(PatternLogger.SESSION_ID, sessionId);
41          paramPut(PatternLogger.CLIENT_IP, clientIp);
42          
43       //   paramPut(PatternLogger.CLIENT_IP, "0");
44          paramPut(OCSPREQUEST, "0");
45          paramPut(OCSPRESPONSE, "0");
46          paramPut(PatternLogger.STATUS, "-1");
47          paramPut(PatternLogger.PROCESS_TIME, "-1");
48      }
49  
50  }
