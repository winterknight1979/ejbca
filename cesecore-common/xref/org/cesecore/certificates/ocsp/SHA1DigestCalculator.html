13  package org.cesecore.certificates.ocsp;
14  
15  import java.io.ByteArrayOutputStream;
16  import java.io.OutputStream;
17  import java.security.MessageDigest;
18  import java.security.NoSuchAlgorithmException;
19  import org.bouncycastle.asn1.x509.AlgorithmIdentifier;
20  import org.bouncycastle.cert.ocsp.RespID;
21  import org.bouncycastle.operator.DigestCalculator;
22  import org.cesecore.certificates.ocsp.exception.OcspFailureException;
23  
24  /**
25   * @version $Id: SHA1DigestCalculator.java 18437 2014-02-03 12:46:08Z
26   *     mikekushner $
27   */
28  public class SHA1DigestCalculator implements DigestCalculator {
29      /** OS. */
30    private ByteArrayOutputStream bOut = new ByteArrayOutputStream();
31    /** Digest. */
32    private MessageDigest digest;
33   /**
34    * @param adigest digest
35    */
36    public SHA1DigestCalculator(final MessageDigest adigest) {
37      this.digest = adigest;
38    }
39  
40    @Override
41    public AlgorithmIdentifier getAlgorithmIdentifier() {
42      return RespID.HASH_SHA1;
43    }
44  
45    @Override
46    public OutputStream getOutputStream() {
47      return bOut;
48    }
49  
50  
51    @Override
52    public byte[] getDigest() {
53      byte[] bytes = digest.digest(bOut.toByteArray());
54  
55      bOut.reset();
56  
57      return bytes;
58    }
59  
60    /**
61     * @return instance
62     */
63    public static SHA1DigestCalculator buildSha1Instance() {
64      try {
65        return new SHA1DigestCalculator(MessageDigest.getInstance("SHA1"));
66      } catch (NoSuchAlgorithmException e) {
67        throw new OcspFailureException(e);
68      }
69    }
70  }
