124     /**
125      * Gets the name of matching key algorithm from a public key as defined by
126      * <i>AlgorithmConstants</i>.
127      * @param publickey Public key to find matching key algorithm for.
128      * @return Name of the matching key algorithm or null if no match.
129      * @see AlgorithmConstants#KEYALGORITHM_RSA
130      * @see AlgorithmConstants#KEYALGORITHM_DSA
131      * @see AlgorithmConstants#KEYALGORITHM_ECDSA
132      */
133     public static String getKeyAlgorithm(final PublicKey publickey) {
134         String keyAlg = null;
135         if ( publickey instanceof RSAPublicKey ) {
136             keyAlg  = AlgorithmConstants.KEYALGORITHM_RSA;
137         } else if ( publickey instanceof DSAPublicKey ) {
138             keyAlg = AlgorithmConstants.KEYALGORITHM_DSA;
139         } else if ( publickey instanceof ECPublicKey ) {
140             final String algo = publickey.getAlgorithm();
141             if (StringUtils.equals(algo, AlgorithmConstants.KEYALGORITHM_ECGOST3410)) {
142                 keyAlg = AlgorithmConstants.KEYALGORITHM_ECGOST3410;
143             } else if (StringUtils.equals(algo, AlgorithmConstants.KEYALGORITHM_DSTU4145)) {
144                 keyAlg = AlgorithmConstants.KEYALGORITHM_DSTU4145;
145             } else {
146                 keyAlg = AlgorithmConstants.KEYALGORITHM_ECDSA;
147             }
148         }
149         return keyAlg;
150     }

162     /**
163      * Get unique available named elliptic curves and their aliases.
164      *
165      * @param hasToBeKnownByDefaultProvider if the curve name needs to be known by the default provider (e.g. so Sun PKCS#11 can use it)
166      * @return a Map with elliptic curve names as key and the key+any alias as the value.
167      */
168     public static Map<String,List<String>> getNamedEcCurvesMap(final boolean hasToBeKnownByDefaultProvider) {
169         // Map of curve name and aliases which are the same curve
170         final Map<String,List<String>> processedCurveNames = new HashMap<>();
171         @SuppressWarnings("unchecked")
172         final Enumeration<String> ecNamedCurvesStandard = ECNamedCurveTable.getNames();
173         // Process standard curves, removing blacklisted ones and those not supported by the provider
174         while (ecNamedCurvesStandard.hasMoreElements()) {
175             final String ecNamedCurve = ecNamedCurvesStandard.nextElement();
176             processCurveName(hasToBeKnownByDefaultProvider, processedCurveNames, ecNamedCurve);
177         }
178 
179         // Process additional curves that we specify
180         for (String ecNamedCurve : AlgorithmConstants.EXTRA_EC_CURVES) {
181             processCurveName(hasToBeKnownByDefaultProvider, processedCurveNames, ecNamedCurve);
182         }
183         return processedCurveNames;
184     }

186     /**
187      * Get unique available named elliptic curves and their aliases.
188      *  
189      * @param hasToBeKnownByDefaultProvider if the curve name needs to be known by the default provider (e.g. so Sun PKCS#11 can use it)
190      * @return a Map with elliptic curve names as key and the list of alias separated by '/' as the value.
191      */
192     public static TreeMap<String,String> getFlatNamedEcCurvesMap(final boolean hasToBeKnownByDefaultProvider) {
193         final TreeMap<String,String> result = new TreeMap<String,String>();
194         final Map<String, List<String>> map = getNamedEcCurvesMap(hasToBeKnownByDefaultProvider);
195         final String[] keys = map.keySet().toArray(new String[map.size()]);
196         Arrays.sort(keys);
197         for (final String name : keys) {
198             result.put(name, StringTools.getAsStringWithSeparator(" / ", map.get(name)));
199         }
200         return result;
201     }

216         list.addAll(AlgorithmTools.getEcKeySpecAliases("P-521"));
217         return list;

220     private static void processCurveName(final boolean hasToBeKnownByDefaultProvider, final Map<String, List<String>> processedCurveNames,
221             final String ecNamedCurve) {
222         if (AlgorithmConstants.BLACKLISTED_EC_CURVES.contains(ecNamedCurve)) {
223             return;
224         }
225         // Only add it if the key-length is sufficient
226         try {
227             final ECNamedCurveParameterSpec parameterSpec = ECNamedCurveTable.getParameterSpec(ecNamedCurve);
228             final int bitLength = parameterSpec.getN().bitLength();
229             KeyTools.checkValidKeyLength(AlgorithmConstants.KEYALGORITHM_ECDSA, bitLength);
230             // Check if this exists under another alias
231             boolean added = false;
232             for (final String name : processedCurveNames.keySet()) {
233                 final ECNamedCurveParameterSpec parameterSpec2 = ECNamedCurveTable.getParameterSpec(name);
234                 if (parameterSpec.equals(parameterSpec2)) {
235                     // We have already listed this curve under another name
236                     added = true;
237                     break;
238                 }
239             }
240             if (!added) {
241                 if (hasToBeKnownByDefaultProvider) {
242                     if (AlgorithmTools.isNamedECKnownInDefaultProvider(ecNamedCurve)) {
243                         processedCurveNames.put(ecNamedCurve, getEcKeySpecAliases(ecNamedCurve));
244                     }
245                 } else {
246                     processedCurveNames.put(ecNamedCurve, getEcKeySpecAliases(ecNamedCurve));
247                 }
248             }
249         } catch (InvalidKeyException e) {
250             // Ignore very silently
251             if (log.isTraceEnabled()) {
252                 log.trace("Not adding keys that are not allowed to key list: "+e.getMessage());
253             }
254         } catch (RuntimeException e) {
255             // Ignore
256             if (log.isDebugEnabled()) {
257                 log.debug(e.getMessage());
258             }
259         }
260     }

262     /** @param ecNamedCurve Curve
263      * @return the number of bits a named elliptic curve has or 0 if the curve name is unknown. */
264     public static int getNamedEcCurveBitLength(final String ecNamedCurve) {
265         ECNamedCurveParameterSpec ecNamedCurveParameterSpec = ECGOST3410NamedCurveTable.getParameterSpec(ecNamedCurve);
266         if (ecNamedCurveParameterSpec!=null) {
267             // This always returns 0, so try to use the field size as an estimate of the bit strength
268             //return ECGOST3410NamedCurveTable.getParameterSpec(ecNamedCurve).getN().bitLength()
269             return ecNamedCurveParameterSpec.getCurve().getFieldSize();
270         }
271         ecNamedCurveParameterSpec = ECNamedCurveTable.getParameterSpec(ecNamedCurve);
272         if (ecNamedCurveParameterSpec==null) {
273             return 0;
274         }
275         return ecNamedCurveParameterSpec.getN().bitLength();
276     }

278     /**
279      * Gets a collection of signature algorithm names supported by the given
280      * key.
281      * @param publickey key to find supported algorithms for.
282      * @return Collection of zero or more signature algorithm names
283      * @see AlgorithmConstants
284      */
285     public static List<String> getSignatureAlgorithms(final PublicKey publickey) {
286         if ( publickey instanceof RSAPublicKey ) {
287             return SIG_ALGS_RSA;
288         }
289         if ( publickey instanceof DSAPublicKey ) {
290             return SIG_ALGS_DSA;
291         }
292         if ( publickey instanceof ECPublicKey ) {
293             final String algo = publickey.getAlgorithm();
294             if (StringUtils.equals(algo, AlgorithmConstants.KEYALGORITHM_ECGOST3410)) {
295                 return SIG_ALGS_ECGOST3410;
296             }
297             if (StringUtils.equals(algo, AlgorithmConstants.KEYALGORITHM_DSTU4145)) {
298                 return SIG_ALGS_DSTU4145;
299             }
300             return SIG_ALGS_ECDSA;
301         }
302         return Collections.emptyList();
303     }

328     /**
329      * Gets the key specification from a public key. Example: "2048" for a RSA
330      * or DSA key or "secp256r1" for EC key. The EC curve is only detected
331      * if <i>publickey</i> is an object known by the bouncy castle provider.
332      * @param publicKey The public key to get the key specification from
333      * @return The key specification, "unknown" if it could not be determined and
334      * null if the key algorithm is not supported
335      */
336     public static String getKeySpecification(final PublicKey publicKey) {
337         if (log.isTraceEnabled()) {
338             log.trace(">getKeySpecification");
339         }
340         String keyspec = null;
341         if ( publicKey instanceof RSAPublicKey ) {
342             keyspec = Integer.toString( ((RSAPublicKey) publicKey).getModulus().bitLength() );
343         } else if ( publicKey instanceof DSAPublicKey ) {
344             keyspec = Integer.toString( ((DSAPublicKey) publicKey).getParams().getP().bitLength() );
345         } else if ( publicKey instanceof ECPublicKey) {
346             final ECPublicKey ecPublicKey = (ECPublicKey) publicKey;
347             if ( ecPublicKey.getParams() instanceof ECNamedCurveSpec ) {
348                 keyspec = ((ECNamedCurveSpec) ecPublicKey.getParams()).getName();
349                 // Prefer to return a curve name alias that also works with the default and BC provider
350                 for (String keySpecAlias : getEcKeySpecAliases(keyspec)) {
351                     if (isNamedECKnownInDefaultProvider(keySpecAlias)) {
352                         keyspec = keySpecAlias;
353                         break;
354                     }
355                 }
356             } else {
357                 keyspec = KEYSPEC_UNKNOWN;
358                 // Try to detect if it is a curve name known by BC even though the public key isn't a BC key
359                 final ECParameterSpec namedCurve = ecPublicKey.getParams();
360                 if (namedCurve!=null) {
361                     final int c1 = namedCurve.getCofactor();
362                     final EllipticCurve ec1 = namedCurve.getCurve();
363                     final BigInteger a1 = ec1.getA();
364                     final BigInteger b1 = ec1.getB();
365                     final int fs1 = ec1.getField().getFieldSize();
366                     //final byte[] s1 = ec1.getSeed();
367                     final ECPoint g1 = namedCurve.getGenerator();
368                     final BigInteger ax1 = g1.getAffineX();
369                     final BigInteger ay1 = g1.getAffineY();
370                     final BigInteger o1 = namedCurve.getOrder();
371                     if (log.isDebugEnabled()) {
372                         log.debug("a1=" + a1 + " b1=" + b1 + " fs1=" + fs1 + " ax1=" + ax1 + " ay1=" + ay1 + " o1=" + o1 + " c1="+c1);
373                     }
374                     @SuppressWarnings("unchecked")
375                     final Enumeration<String> ecNamedCurves = ECNamedCurveTable.getNames();
376                     while (ecNamedCurves.hasMoreElements()) {
377                         final String ecNamedCurveBc = ecNamedCurves.nextElement();
378                         final ECNamedCurveParameterSpec parameterSpec2 = ECNamedCurveTable.getParameterSpec(ecNamedCurveBc);
379                         final ECCurve ec2 = parameterSpec2.getCurve();
380                         final BigInteger a2 = ec2.getA().toBigInteger();
381                         final BigInteger b2 = ec2.getB().toBigInteger();
382                         final int fs2 = ec2.getFieldSize();
383                         final org.bouncycastle.math.ec.ECPoint g2 = parameterSpec2.getG();
384                         final BigInteger ax2 = g2.getAffineXCoord().toBigInteger();
385                         final BigInteger ay2 = g2.getAffineYCoord().toBigInteger();
386                         final BigInteger h2 = parameterSpec2.getH();
387                         final BigInteger n2 = parameterSpec2.getN();
388                         if (a1.equals(a2) && ax1.equals(ax2) && b1.equals(b2) && ay1.equals(ay2) && fs1==fs2 && o1.equals(n2) && c1==h2.intValue()) {
389                             // We have a matching curve here!
390                             if (log.isDebugEnabled()) {
391                                 log.debug("a2=" + a2 + " b2=" + b2 + " fs2=" + fs2 + " ax2=" + ax2 + " ay2=" + ay2 + " h2=" + h2 + " n2=" + n2 + " " + ecNamedCurveBc);
392                             }
393                             // Since this public key is a SUN PKCS#11 pub key if we get here, we only return an alias if it is recognized by the provider
394                             if (isNamedECKnownInDefaultProvider(ecNamedCurveBc)) {
395                                 keyspec = ecNamedCurveBc;
396                                 break;
397                             }
398                         }
399                     }
400                 }
401             }
402         }
403         if (log.isTraceEnabled()) {
404             log.trace("<getKeySpecification: "+keyspec);
405         }
406         return keyspec;
407     }

409     /** Check if the curve name is known by the first found PKCS#11 provider or default (BC) (if no EC capable PKCS#11 provider were found)
410      * @param ecNamedCurveBc Curve
411      * @return boolean */
412     public static boolean isNamedECKnownInDefaultProvider(String ecNamedCurveBc) {
413         final Provider[] providers = Security.getProviders("KeyPairGenerator.EC");
414         String providerName = providers[0].getName();
415         try {
416             for (Provider ecProvider : providers) {
417                 //This will list something like: SunPKCS11-NSS, BC, SunPKCS11-<library>-slot<slotnumber>
418                 if (log.isDebugEnabled()) {
419                     log.debug("Found EC capable provider named: " + ecProvider.getName());
420                 }
421                 if (ecProvider.getName().startsWith("SunPKCS11-") && !ecProvider.getName().startsWith("SunPKCS11-NSS") ) {
422                     // Sometimes the P11 provider will not even know about EC, skip these providers. As an example the SunP11
423                     // provider in some version/installations will throw a:
424                     // java.lang.RuntimeException: Cannot load SunEC provider
425                     //   at sun.security.pkcs11.P11ECKeyFactory.getSunECProvider(P11ECKeyFactory.java:55)
426                     // This was a bug of non upgraded NSS in RHEL at some point in time.
427                     try {
428                         KeyPairGenerator.getInstance("EC", ecProvider.getName());
429                         providerName = ecProvider.getName();
430                         break;
431                     } catch (RuntimeException e) {
432                         log.info("Provider "+ecProvider.getName()+" bailed out on EC, ignored.", e);
433                     }
434                 }
435             }
436             final KeyPairGenerator kpg = KeyPairGenerator.getInstance("EC", providerName);
437             kpg.initialize(new ECGenParameterSpec(getEcKeySpecOidFromBcName(ecNamedCurveBc)));
438             return true;
439         } catch (InvalidAlgorithmParameterException e) {
440             if (log.isDebugEnabled()) {
441                 log.debug(ecNamedCurveBc + " is not available in provider " + providerName);
442             }
443         } catch (NoSuchAlgorithmException e) {
444             throw new RuntimeException("EC capable provider " + providerName + " could no longer handle elliptic curve algorithm.." ,e);
445         } catch (NoSuchProviderException e) {
446             throw new RuntimeException("EC capable provider " + providerName + " disappeard unexpectedly." ,e);
447         }
448         return false;
449     }

451     /**
452      * Convert from BC ECC curve names to the OID.
453      *
454      * @param ecNamedCurveBc the name as BC reports it
455      * @return the OID of the curve or the input curve name if it is unknown by BC
456      */
457     public static String getEcKeySpecOidFromBcName(final String ecNamedCurveBc) {
458         // Although the below class is in x9 package, it handles all different curves, including TeleTrust (brainpool)
459         final ASN1ObjectIdentifier oid = org.bouncycastle.asn1.x9.ECNamedCurveTable.getOID(ecNamedCurveBc);
460         if (oid==null) {
461             return ecNamedCurveBc;
462         }
463         return oid.getId();
464     }

466     /** @param namedEllipticCurve Curve name
467      * @return a list of aliases for the provided curve name (including the provided name) */
468     public static List<String> getEcKeySpecAliases(final String namedEllipticCurve) {
469         final ECNamedCurveParameterSpec parameterSpec = ECNamedCurveTable.getParameterSpec(namedEllipticCurve);
470         final List<String> ret = new ArrayList<>();
471         ret.add(namedEllipticCurve);
472 
473         if (parameterSpec != null) { // GOST and DSTU aren't present in ECNamedCurveTable (and don't have aliases)
474             @SuppressWarnings("unchecked")
475             final Enumeration<String> ecNamedCurves = ECNamedCurveTable.getNames();
476             while (ecNamedCurves.hasMoreElements()) {
477                 final String currentCurve = ecNamedCurves.nextElement();
478                 if (!namedEllipticCurve.equals(currentCurve)) {
479                     final ECNamedCurveParameterSpec parameterSpec2 = ECNamedCurveTable.getParameterSpec(currentCurve);
480                     if (parameterSpec.equals(parameterSpec2)) {
481                         ret.add(currentCurve);
482                     }
483                 }
484             }
485         }
486         return ret;
487     }

489     /**
490      * Gets the algorithm to use for encryption given a specific signature algorithm.
491      * Some signature algorithms (i.e. DSA and ECDSA) can not be used for
492      * encryption so they are instead substituted with RSA with equivalent hash
493      * algorithm.
494      * @param signatureAlgorithm to find a encryption algorithm for
495      * @return an other encryption algorithm or same as signature algorithm if it
496      * can be used for encryption
497      */
498     public static String getEncSigAlgFromSigAlg(final String signatureAlgorithm) {
499         String encSigAlg = signatureAlgorithm;
500         if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA512_WITH_ECDSA)) {
501             encSigAlg = AlgorithmConstants.SIGALG_SHA256_WITH_RSA;
502         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA384_WITH_ECDSA)) {
503             // Even though SHA384 is used for ECDSA, play it safe and use SHA256 for RSA since we do not trust all PKCS#11 implementations
504             // to be so new to support SHA384WithRSA
505             encSigAlg = AlgorithmConstants.SIGALG_SHA256_WITH_RSA;
506         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA256_WITH_ECDSA)) {
507             encSigAlg = AlgorithmConstants.SIGALG_SHA256_WITH_RSA;
508         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA3_256_WITH_ECDSA)) {
509             encSigAlg = AlgorithmConstants.SIGALG_SHA3_256_WITH_RSA;
510         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA3_384_WITH_ECDSA)) {
511             encSigAlg = AlgorithmConstants.SIGALG_SHA3_384_WITH_RSA;
512         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA3_512_WITH_ECDSA)) {
513             encSigAlg = AlgorithmConstants.SIGALG_SHA3_512_WITH_RSA;
514         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA224_WITH_ECDSA)) {
515             encSigAlg = AlgorithmConstants.SIGALG_SHA256_WITH_RSA;
516         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA1_WITH_ECDSA)) {
517             encSigAlg = AlgorithmConstants.SIGALG_SHA1_WITH_RSA;
518         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_SHA1_WITH_DSA)) {
519             encSigAlg = AlgorithmConstants.SIGALG_SHA1_WITH_RSA;
520         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_GOST3411_WITH_ECGOST3410)) {
521             encSigAlg = AlgorithmConstants.SIGALG_SHA1_WITH_RSA;
522         } else if (signatureAlgorithm.equals(AlgorithmConstants.SIGALG_GOST3411_WITH_DSTU4145)) {
523             encSigAlg = AlgorithmConstants.SIGALG_SHA1_WITH_RSA;
524         }
525         return encSigAlg;
526     }

528     /**
529      * Answers if the key can be used together with the given signature algorithm.
530      * @param publicKey public key to use
531      * @param signatureAlgorithm algorithm to test
532      * @return true if signature algorithm can be used with the public key algorithm
533      */
534     public static boolean isCompatibleSigAlg(final PublicKey publicKey, final String signatureAlgorithm) {
535         String algname = publicKey.getAlgorithm();
536         if (algname == null) {
537             algname = "";
538         }
539         boolean isGost3410 = algname.contains("GOST3410");
540         boolean isDstu4145 = algname.contains("DSTU4145");
541         boolean isSpecialECC = isGost3410 || isDstu4145;
542 
543         boolean ret = false;
544         if (StringUtils.contains(signatureAlgorithm, AlgorithmConstants.KEYALGORITHM_RSA)) {
545             if (publicKey instanceof RSAPublicKey) {
546                 ret = true;
547             }
548         } else if (StringUtils.contains(signatureAlgorithm, AlgorithmConstants.KEYALGORITHM_ECDSA)) {
549             if (publicKey instanceof ECPublicKey && !isSpecialECC) {
550                 ret = true;
551             }
552         } else if (StringUtils.contains(signatureAlgorithm, AlgorithmConstants.KEYALGORITHM_DSA)) {
553             if (publicKey instanceof DSAPublicKey) {
554                 ret = true;
555             }
556         } else if (StringUtils.contains(signatureAlgorithm, AlgorithmConstants.KEYALGORITHM_ECGOST3410)) {
557              if (publicKey instanceof ECPublicKey && isGost3410) {
558                  ret = true;
559              }
560          } else if (StringUtils.contains(signatureAlgorithm, AlgorithmConstants.KEYALGORITHM_DSTU4145)) {
561             if (publicKey instanceof ECPublicKey && isDstu4145) {
562                 ret = true;
563             }
564          }
565         return ret;
566     }

568     /**
569      * Simple methods that returns the signature algorithm value from the certificate. Not usable for setting signature algorithms names in EJBCA,
570      * only for human presentation.
571      * @param cert certificate
572      *
573      * @return Signature algorithm name from the certificate as a human readable string, for example SHA1WithRSA.
574      */
575     public static String getCertSignatureAlgorithmNameAsString(Certificate cert) {
576         final String certSignatureAlgorithm;
577         {
578             final String certSignatureAlgorithmTmp;
579             if (cert instanceof X509Certificate) {
580                 final X509Certificate x509cert = (X509Certificate) cert;
581                 certSignatureAlgorithmTmp = x509cert.getSigAlgName();
582                 if (log.isDebugEnabled()) {
583                     log.debug("certSignatureAlgorithm is: " + certSignatureAlgorithmTmp);
584                 }
585             } else if (StringUtils.equals(cert.getType(), "CVC")) {
586                 final CardVerifiableCertificate cvccert = (CardVerifiableCertificate) cert;
587                 final CVCPublicKey cvcpk;
588                 try {
589                     cvcpk = cvccert.getCVCertificate().getCertificateBody().getPublicKey();
590                     final OIDField oid = cvcpk.getObjectIdentifier();
591                     certSignatureAlgorithmTmp = AlgorithmUtil.getAlgorithmName(oid);
592                 } catch (NoSuchFieldException e) {
593                     throw new AlgorithmsToolRuntimeException("Not a valid CVC certificate", e);
594                 }
595             } else {
596                 throw new AlgorithmsToolRuntimeException("Certificate type neither X509 nor CVS.");
597             }
598             // Try to make it easier to display some signature algorithms that cert.getSigAlgName() does not have a good string for.
599             if (certSignatureAlgorithmTmp.equalsIgnoreCase("1.2.840.113549.1.1.10") && cert instanceof X509Certificate) {
600                 // Figure out if it is SHA1 or SHA256
601                 // If we got this value we should have a x509 cert
602                 final X509Certificate x509cert = (X509Certificate) cert;
603                 final byte[] params = x509cert.getSigAlgParams();
604                 if ((params != null) && (params.length == 2)) {
605                     certSignatureAlgorithm = AlgorithmConstants.SIGALG_SHA1_WITH_RSA_AND_MGF1;
606                 } else {
607                     certSignatureAlgorithm = AlgorithmConstants.SIGALG_SHA256_WITH_RSA_AND_MGF1;
608                 }
609             } else {
610                 certSignatureAlgorithm = certSignatureAlgorithmTmp;
611             }
612         }
613         // SHA256WithECDSA does not work to be translated in JDK5.
614         if (certSignatureAlgorithm.equalsIgnoreCase("1.2.840.10045.4.3.2")) {
615             return AlgorithmConstants.SIGALG_SHA256_WITH_ECDSA;
616         }
617         // GOST3410
618         if(isGost3410Enabled() && certSignatureAlgorithm.equalsIgnoreCase(CesecoreConfiguration.getOidGost3410())) {
619             return AlgorithmConstants.SIGALG_GOST3411_WITH_ECGOST3410;
620         }
621         // DSTU4145
622         if(isDstu4145Enabled() && certSignatureAlgorithm.startsWith(CesecoreConfiguration.getOidDstu4145()+".")) {
623             return AlgorithmConstants.SIGALG_GOST3411_WITH_DSTU4145;
624         }
625         return certSignatureAlgorithm;
626     }

628     /**
629      * Simple method that looks at the certificate and determines, from EJBCA's standpoint, which signature algorithm it is
630      *
631      * @param cert the cert to examine
632      * @return Signature algorithm name from AlgorithmConstants.SIGALG_SHA1_WITH_RSA etc.
633      */
634     public static String getSignatureAlgorithm(Certificate cert) {
635         String signatureAlgorithm = null;
636         String certSignatureAlgorithm = getCertSignatureAlgorithmNameAsString(cert);
637 
638         // The signature string returned from the certificate is often not usable as the signature algorithm we must
639         // specify for a CA in EJBCA, for example SHA1WithECDSA is returned as only ECDSA, so we need some magic to fix it up.
640         PublicKey publickey = cert.getPublicKey();
641         if (publickey instanceof RSAPublicKey) {
642             if (certSignatureAlgorithm.contains("SHA3-")) {
643                 if (certSignatureAlgorithm.contains("256")) {
644                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA3_256_WITH_RSA;
645                 } else if (certSignatureAlgorithm.contains("384")) {
646                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA3_384_WITH_RSA;
647                 } else if (certSignatureAlgorithm.contains("512")) {
648                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA3_512_WITH_RSA;
649                 }
650             } else if (certSignatureAlgorithm.indexOf("MGF1") == -1) {
651                 if (certSignatureAlgorithm.indexOf("MD5") != -1) {
652                     signatureAlgorithm = "MD5WithRSA";
653                 } else if (certSignatureAlgorithm.indexOf("SHA1") != -1) {
654                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA1_WITH_RSA;
655                 } else if (certSignatureAlgorithm.indexOf("256") != -1) {
656                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA256_WITH_RSA;
657                 } else if (certSignatureAlgorithm.indexOf("384") != -1) {
658                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA384_WITH_RSA;
659                 } else if (certSignatureAlgorithm.indexOf("512") != -1) {
660                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA512_WITH_RSA;
661                 }
662             } else {
663                 if (certSignatureAlgorithm.indexOf("SHA1") != -1) {
664                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA1_WITH_RSA_AND_MGF1;
665                 } else {
666                     signatureAlgorithm = AlgorithmConstants.SIGALG_SHA256_WITH_RSA_AND_MGF1;
667                 }
668             }
669         } else if (publickey instanceof DSAPublicKey) {
670             signatureAlgorithm = AlgorithmConstants.SIGALG_SHA1_WITH_DSA;
671         } else {
672             if (certSignatureAlgorithm.contains("SHA3-")) {
673                 if (certSignatureAlgorithm.contains("256")) {
674                     return AlgorithmConstants.SIGALG_SHA3_256_WITH_ECDSA;
675                 } else if (certSignatureAlgorithm.contains("384")) {
676                     return AlgorithmConstants.SIGALG_SHA3_384_WITH_ECDSA;
677                 } else if (certSignatureAlgorithm.contains("512")) {
678                     return AlgorithmConstants.SIGALG_SHA3_512_WITH_ECDSA;
679                 }
680             } else if (certSignatureAlgorithm.indexOf("256") != -1) {
681                 signatureAlgorithm = AlgorithmConstants.SIGALG_SHA256_WITH_ECDSA;
682             } else if (certSignatureAlgorithm.indexOf("224") != -1) {
683                 signatureAlgorithm = AlgorithmConstants.SIGALG_SHA224_WITH_ECDSA;
684             } else if (certSignatureAlgorithm.indexOf("384") != -1) {
685                 signatureAlgorithm = AlgorithmConstants.SIGALG_SHA384_WITH_ECDSA;
686             } else if (certSignatureAlgorithm.indexOf("512") != -1) {
687                 signatureAlgorithm = AlgorithmConstants.SIGALG_SHA512_WITH_ECDSA;
688             } else if (certSignatureAlgorithm.indexOf("ECDSA") != -1) {
689                 // From x509cert.getSigAlgName(), SHA1withECDSA only returns name ECDSA
690                 signatureAlgorithm = AlgorithmConstants.SIGALG_SHA1_WITH_ECDSA;
691             } else if (isGost3410Enabled() && certSignatureAlgorithm.equalsIgnoreCase(AlgorithmConstants.SIGALG_GOST3411_WITH_ECGOST3410)) {
692                 signatureAlgorithm = AlgorithmConstants.SIGALG_GOST3411_WITH_ECGOST3410;
693             } else if (isDstu4145Enabled() && certSignatureAlgorithm.equalsIgnoreCase(AlgorithmConstants.SIGALG_GOST3411_WITH_DSTU4145)) {
694                 signatureAlgorithm = AlgorithmConstants.SIGALG_GOST3411_WITH_DSTU4145;
695             }
696         }
697         if (log.isDebugEnabled()) {
698             log.debug("getSignatureAlgorithm: " + signatureAlgorithm);
699         }
700         return signatureAlgorithm;
701     } // getSignatureAlgorithm

703     /**
704      * Get the digest algorithm corresponding to the signature algorithm. This is used for the creation of
705      * PKCS7 file. SHA1 shall always be used, but it is not working with GOST which needs GOST3411 digest.
706      * @param sigAlg Algorithm
707      * @return name
708      */
709     public static String getDigestFromSigAlg(String sigAlg) {
710         if (sigAlg.toUpperCase().contains("GOST") || sigAlg.toUpperCase().contains("DSTU")) {
711             return CMSSignedGenerator.DIGEST_GOST3411;
712         }
713         if(sigAlg.equals(X9ObjectIdentifiers.ecdsa_with_SHA1.getId()) || sigAlg.equals(PKCSObjectIdentifiers.sha1WithRSAEncryption.getId())) {
714             return CMSSignedGenerator.DIGEST_SHA1;
715         }
716         if(sigAlg.equals(X9ObjectIdentifiers.ecdsa_with_SHA224.getId()) || sigAlg.equals(PKCSObjectIdentifiers.sha224WithRSAEncryption.getId())) {
717             return CMSSignedGenerator.DIGEST_SHA224;
718         }
719         if(sigAlg.equals(X9ObjectIdentifiers.ecdsa_with_SHA256.getId()) ||
720                 sigAlg.equals(PKCSObjectIdentifiers.sha256WithRSAEncryption.getId())) {
721             return CMSSignedGenerator.DIGEST_SHA256;
722         }
723         if (sigAlg.equals(NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_256.getId())) {
724             return NISTObjectIdentifiers.id_sha3_256.getId();
725         }
726         if (sigAlg.equals(NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_384.getId())) {
727             return NISTObjectIdentifiers.id_sha3_384.getId();
728         }
729         if (sigAlg.equals(NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_512.getId())) {
730             return NISTObjectIdentifiers.id_sha3_512.getId();
731         }
732         if (sigAlg.equals(NISTObjectIdentifiers.id_ecdsa_with_sha3_256.getId())) {
733             return NISTObjectIdentifiers.id_sha3_256.getId();
734         }
735         if (sigAlg.equals(NISTObjectIdentifiers.id_ecdsa_with_sha3_384.getId())) {
736             return NISTObjectIdentifiers.id_sha3_384.getId();
737         }
738         if (sigAlg.equals(NISTObjectIdentifiers.id_ecdsa_with_sha3_512.getId())) {
739             return NISTObjectIdentifiers.id_sha3_512.getId();
740         }
741         if(sigAlg.equals(X9ObjectIdentifiers.ecdsa_with_SHA384.getId()) || sigAlg.equals(PKCSObjectIdentifiers.sha384WithRSAEncryption.getId())) {
742             return CMSSignedGenerator.DIGEST_SHA384;
743         }
744         if(sigAlg.equals(X9ObjectIdentifiers.ecdsa_with_SHA512.getId()) || sigAlg.equals(PKCSObjectIdentifiers.sha512WithRSAEncryption.getId())) {
745             return CMSSignedGenerator.DIGEST_SHA512;
746         }
747         if(sigAlg.equals(PKCSObjectIdentifiers.md5WithRSAEncryption.getId())) {
748             return CMSSignedGenerator.DIGEST_MD5;
749         }
750         if(sigAlg.equals(CryptoProObjectIdentifiers.gostR3411_94_with_gostR3410_2001.getId()) ) {
751             return CMSSignedGenerator.DIGEST_GOST3411;
752         }
753         return CMSSignedGenerator.DIGEST_SHA1;
754     }

756     /**
757      * Calculates which signature algorithm to use given a key type and a digest algorithm
758      *
759      * @param digestAlg objectId of a digest algorithm, CMSSignedGenerator.DIGEST_SHA256 etc
760      * @param keyAlg RSA, EC, DSA
761      * @return ASN1ObjectIdentifier with the id of PKCSObjectIdentifiers.sha1WithRSAEncryption, X9ObjectIdentifiers.ecdsa_with_SHA1, X9ObjectIdentifiers.id_dsa_with_sha1, etc
762      */
763     public static ASN1ObjectIdentifier getSignAlgOidFromDigestAndKey(final String digestAlg, final String keyAlg) {
764         if (log.isTraceEnabled()) {
765             log.trace(">getSignAlg("+digestAlg+","+keyAlg+")");
766         }
767         // Default to SHA1WithRSA if everything else fails
768         ASN1ObjectIdentifier oid = PKCSObjectIdentifiers.sha1WithRSAEncryption;
769         if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA)) {
770             oid = X9ObjectIdentifiers.ecdsa_with_SHA1;
771         } else if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_DSA)) {
772             oid = X9ObjectIdentifiers.id_dsa_with_sha1;
773         } else if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECGOST3410)) {
774             oid = CryptoProObjectIdentifiers.gostR3411_94_with_gostR3410_2001;
775         } else if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_DSTU4145)) {
776             oid = new ASN1ObjectIdentifier(CesecoreConfiguration.getOidDstu4145());
777         }
778         if (digestAlg != null) {
779             if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA256) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
780                 oid = PKCSObjectIdentifiers.sha256WithRSAEncryption;
781             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA512) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
782                 oid = PKCSObjectIdentifiers.sha512WithRSAEncryption;
783             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_MD5) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
784                 oid = PKCSObjectIdentifiers.md5WithRSAEncryption;
785             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA256) && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC)) ) {
786                 oid = X9ObjectIdentifiers.ecdsa_with_SHA256;
787             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA224) && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC)) ) {
788                 oid = X9ObjectIdentifiers.ecdsa_with_SHA224;
789             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA384) && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC)) ) {
790                 oid = X9ObjectIdentifiers.ecdsa_with_SHA384;
791             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA512) && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC)) ) {
792                 oid = X9ObjectIdentifiers.ecdsa_with_SHA512;
793             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA256) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_DSA)) {
794                 oid = NISTObjectIdentifiers.dsa_with_sha256;
795             } else if (digestAlg.equals(CMSSignedGenerator.DIGEST_SHA512) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_DSA)) {
796                 oid = NISTObjectIdentifiers.dsa_with_sha512;
797             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_256.toString()) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
798                 oid = NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_256;
799             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_384.toString()) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
800                 oid = NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_384;
801             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_512.toString()) && keyAlg.equals(AlgorithmConstants.KEYALGORITHM_RSA)) {
802                 oid = NISTObjectIdentifiers.id_rsassa_pkcs1_v1_5_with_sha3_512;
803             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_256.toString())
804                     && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC))) {
805                 oid = NISTObjectIdentifiers.id_ecdsa_with_sha3_256;
806             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_384.toString())
807                     && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC))) {
808                 oid = NISTObjectIdentifiers.id_ecdsa_with_sha3_384;
809             } else if (digestAlg.equals(NISTObjectIdentifiers.id_sha3_512.toString())
810                     && (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECDSA) || keyAlg.equals(AlgorithmConstants.KEYALGORITHM_EC))) {
811                 oid = NISTObjectIdentifiers.id_ecdsa_with_sha3_512;
812             }
813         }
814         if (log.isDebugEnabled()) {
815             log.debug("getSignAlgOidFromDigestAndKey: " + oid.getId());
816         }
817         return oid;

851     /**
852      * <p>Perform a case-insensitive lookup of all known aliases for an elliptic curve given one known alias.</p>
853      * <p>To determine whether an alias is known, see {@link #isKnownAlias}.</p>
854      * @param alias alias
855      * @return a sorted list of aliases for the elliptic curve specified, never null
856      */
857     public static List<String> getAllCurveAliasesFromAlias(final String alias) {
858         final String lowerCaseAlias = alias.toLowerCase();
859         for (final Entry<String, List<String>> name : getNamedEcCurvesMap(false).entrySet()) {
860             final String lowerCaseCanonicalName = name.getKey().toLowerCase();
861             final List<String> lowerCaseAliases = StringTools.toLowerCase(name.getValue());
862             if (StringUtils.equals(lowerCaseAlias, lowerCaseCanonicalName) || lowerCaseAliases.contains(lowerCaseAlias)) {
863                 final List<String> aliases = new ArrayList<String>(name.getValue());
864                 aliases.add(name.getKey());
865                 Collections.sort(aliases);
866                 return aliases;
867             }
868         }
869         return new ArrayList<>();
870     }
