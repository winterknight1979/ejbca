183     if (type == DNFieldExtractor.TYPE_SUBJECTDN) {
184       return DnComponents.getDnDnIds();
185     } else if (type == DNFieldExtractor.TYPE_SUBJECTALTNAME) {
186       return DnComponents.getAltNameDnIds();
187     } else if (type == DNFieldExtractor.TYPE_SUBJECTDIRATTR) {
188       return DnComponents.getDirAttrDnIds();
189     } else {
190       return new ArrayList<Integer>();

194   /**
195    * Returns the valid components for the given DN type (Subject DN, Subject
196    * Alternative Name or Subject Directory Attributes).
197    *
198    * @param dnType DNFieldExtractor.TYPE_*
199    * @return List of valid components from DnComponents.*
200    */
201   public static List<String> getValidFieldComponents(final int dnType) {
202     switch (dnType) {
203       case DNFieldExtractor.TYPE_SUBJECTDN:
204         return DnComponents.getDnProfileFields();
205       case DNFieldExtractor.TYPE_SUBJECTALTNAME:
206         return DnComponents.getAltNameFields();
207       case DNFieldExtractor.TYPE_SUBJECTDIRATTR:
208         return DnComponents.getDirAttrFields();
209       default:
210         throw new IllegalStateException("Invalid DN type");
211     }
212   }

231   /**
232    * Looks up a DN Id (for use with DnComponents functions etc.) from a DN
233    * component.
234    *
235    * @param dnComponent Component, e.g. "CN". Not case sensitive.
236    * @param dnType DN type, e.g. DNFieldExtractor.TYPE_SUBJECTDN
237    * @return DN Id, or null if no such component exists for the given DN type.
238    */
239   public static Integer getDnIdFromComponent(
240       final String dnComponent, final int dnType) {
241     switch (dnType) {
242       case DNFieldExtractor.TYPE_SUBJECTDN:
243         return DnComponents.getDnIdFromDnName(dnComponent);
244       case DNFieldExtractor.TYPE_SUBJECTALTNAME:
245         return DnComponents.getDnIdFromAltName(dnComponent);
246       case DNFieldExtractor.TYPE_SUBJECTDIRATTR:
247         DnComponents.getDnIdFromDirAttr(dnComponent);
248       default:
249         throw new IllegalStateException("Invalid DN type");
250     }
251   }

253   /**
254    * Fills the dnfields variable with dn (or altname or subject dir attrs)
255    * numerical ids and the value of the components (i.e. the value of CN). Also
256    * populates fieldnumbers with number of occurances in dn
257    *
258    * @param dn DOCUMENT ME!
259    * @param aType DOCUMENT ME!
260    */
261   public final void setDN(final String dn, final int aType) {
262 
263     this.type = aType;
264     final ArrayList<Integer> ids;
265     if (aType == TYPE_SUBJECTDN) {
266       ids = DnComponents.getDnDnIds();
267     } else if (aType == TYPE_SUBJECTALTNAME) {
268       ids = DnComponents.getAltNameDnIds();
269     } else if (aType == TYPE_SUBJECTDIRATTR) {
270       ids = DnComponents.getDirAttrDnIds();
271     } else {
272       ids = new ArrayList<>();
273     }
274     fieldnumbers = new HashMap<>();
275     for (Integer id : ids) {
276       fieldnumbers.put(id, 0);
277     }
278 
279     if ((dn != null) && !dn.equalsIgnoreCase("null")) {
280       dnfields = new HashMap<>();
281       try {
282         final String[] dnexploded = LDAPDN.explodeDN(dn, false);
283         for (int i = 0; i < dnexploded.length; i++) {
284           boolean exists = false;
285 
286           for (Integer id : ids) {
287             Integer number = fieldnumbers.get(id);
288             String field;
289             if (aType == TYPE_SUBJECTDN) {
290               field = DnComponents.getDnExtractorFieldFromDnId(id.intValue());
291             } else if (aType == TYPE_SUBJECTALTNAME) {
292               field =
293                   DnComponents.getAltNameExtractorFieldFromDnId(id.intValue());
294             } else {
295               field =
296                   DnComponents.getDirAttrExtractorFieldFromDnId(id.intValue());
297             }
298             final String dnex = dnexploded[i].toUpperCase();
299             if (id.intValue() == DNFieldExtractor.URI) {
300               // Fix up URI, which can have several forms
301               if (dnex.indexOf(CertTools.URI.toUpperCase(Locale.ENGLISH) + "=")
302                   > -1) {
303                 field = CertTools.URI.toUpperCase(Locale.ENGLISH) + "=";
304               }
305               if (dnex.indexOf(CertTools.URI1.toUpperCase(Locale.ENGLISH) + "=")
306                   > -1) {
307                 field = CertTools.URI1.toUpperCase(Locale.ENGLISH) + "=";
308               }
309             }
310 
311             if (dnex.startsWith(field)) {
312 
313               exists = true;
314               final String rdn;
315               final String tmp;
316               // LDAPDN.unescapeRDN don't like fields with just a key but no
317               // contents. Example: 'OU='
318               if (dnexploded[i].charAt(dnexploded[i].length() - 1) != '=') {
319                 tmp = LDAPDN.unescapeRDN(dnexploded[i]);
320               } else {
321                 tmp = dnexploded[i];
322               }
323               // We don't want the CN= (or whatever) part of the RDN
324               if (tmp.toUpperCase().startsWith(field)) {
325                 rdn = tmp.substring(field.length(), tmp.length());
326               } else {
327                 rdn = tmp;
328               }
329 
330               // Same code for TYPE_SUBJECTDN, TYPE_SUBJECTALTNAME and
331               // TYPE_SUBJECTDIRATTR and we will never get here
332               // if it is not one of those types
333               dnfields.put(
334                   Integer.valueOf(
335                       (id.intValue() * BOUNDRARY) + number.intValue()),
336                   rdn);
337 
338               number = Integer.valueOf(number.intValue() + 1);
339               fieldnumbers.put(id, number);
340             }
341           }
342           if (!exists) {
343             existsother = true;
344           }
345         }
346       } catch (Exception e) {
347         LOG.warn("setDN: ", e);
348         illegal = true;
349         if (aType == TYPE_SUBJECTDN) {
350           dnfields.put(Integer.valueOf((CN * BOUNDRARY)), "Illegal DN : " + dn);
351         } else if (aType == TYPE_SUBJECTALTNAME) {
352           dnfields.put(
353               Integer.valueOf((RFC822NAME * BOUNDRARY)),
354               "Illegal Subjectaltname : " + dn);
355         } else if (aType == TYPE_SUBJECTDIRATTR) {
356           dnfields.put(
357               Integer.valueOf((PLACEOFBIRTH * BOUNDRARY)),
358               "Illegal Subjectdirectory attribute : " + dn);
359         }
360       }
361     }
362   }

364   /**
365    * Returns the value of a certain DN component.
366    *
367    * @param field the DN component, one of the constants DNFieldExtractor.CN,
368    *     ...
369    * @param number the number of the component if several entries for this
370    *     component exists, normally 0 fir the first
371    * @return A String for example "PrimeKey" if DNFieldExtractor.O and 0 was
372    *     passed, "PrimeKey" if DNFieldExtractor.DC and 0 was passed or "com" if
373    *     DNFieldExtractor.DC and 1 was passed. Returns an empty String "", if no
374    *     such field with the number exists.
375    */
376   public String getField(final int field, final int number) {
377     String returnval =
378         dnfields.get(Integer.valueOf((field * BOUNDRARY) + number));
379 
380     if (returnval == null) {
381       returnval = "";
382     }
383 
384     return returnval;
385   }

387   /**
388    * Returns a string representation of a certain DN component.
389    *
390    * @param field the DN component, one of the constants DNFieldExtractor.CN,
391    *     ...
392    * @return A String for example "CN=Tomas Gustavsson" if DNFieldExtractor.CN
393    *     was passed, "DC=PrimeKey,DC=com" if DNFieldExtractor.DC was passed.
394    *     This string is escaped so it can be used in a DN string.
395    */
396   public String getFieldString(final int field) {
397     String retval = "";
398     String fieldname = DnComponents.getDnExtractorFieldFromDnId(field);
399     if (type != TYPE_SUBJECTDN) {
400       fieldname = DnComponents.getAltNameExtractorFieldFromDnId(field);
401     }
402     final int num = getNumberOfFields(field);
403     for (int i = 0; i < num; i++) {
404       if (retval.length() == 0) {
405         retval += LDAPDN.escapeRDN(fieldname + getField(field, i));
406       } else {
407         retval += "," + LDAPDN.escapeRDN(fieldname + getField(field, i));
408       }
409     }
410     return retval;
411   }

422   /**
423    * Returns the number of one kind of dn field.
424    *
425    * @param field the DN component, one of the constants DNFieldExtractor.CN,
426    *     ...
427    * @return number of components available for a field, for example 1 if DN is
428    *     "dc=primekey" and 2 if DN is "dc=primekey,dc=com"
429    */
430   public int getNumberOfFields(final int field) {
431     Integer ret = fieldnumbers.get(Integer.valueOf(field));
432     if (ret == null) {
433       LOG.error("Not finding fieldnumber value for " + field);
434       ret = Integer.valueOf(0);
435     }
436     return ret.intValue();
437   }
