40  /**
41   * A class for reading values from SubjectDirectoryAttributes extension.
42   *
43   * @version $Id: SubjectDirAttrExtension.java 25905 2017-05-27 00:53:42Z jeklund $
44   */
45  public class SubjectDirAttrExtension extends CertTools {
46  
47      private static final Logger log = Logger.getLogger(SubjectDirAttrExtension.class);
48      
49      /**
50       * inhibits creation of new SubjectDirAttrExtension
51       */
52      private SubjectDirAttrExtension() {
53      }
54  
55      /**
56  	 * SubjectDirectoryAttributes ::= SEQUENCE SIZE (1..MAX) OF Attribute
57  	 *
58  	 * Attribute ::= SEQUENCE {
59       *  type AttributeType,
60       *  values SET OF AttributeValue }
61       *  -- at least one value is required
62       * 
63       * AttributeType ::= OBJECT IDENTIFIER
64       * AttributeValue ::= ANY
65       * 
66  	 * SubjectDirectoryAttributes is of form 
67  	 * dateOfBirth=&lt;19590927&gt;, placeOfBirth=&lt;string&gt;, gender=&lt;M/F&gt;, countryOfCitizenship=&lt;two letter ISO3166&gt;, countryOfResidence=&lt;two letter ISO3166&gt;
68       * 
69       * Supported subjectDirectoryAttributes are the ones above 
70  	 *
71  	 * @param certificate containing subject directory attributes
72  	 * @return String containing directoryAttributes of form the form specified above or null if no directoryAttributes exist. 
73  	 *   Values in returned String is from CertTools constants. 
74  	 *   DirectoryAttributes not supported are simply not shown in the resulting string.  
75       *  
76  	 * @throws java.text.ParseException when id_pda_dateOfBirth is malformed
77  	 */
78  	public static String getSubjectDirectoryAttributes(Certificate certificate) throws ParseException {
79  		log.debug("Search for SubjectDirectoryAttributes");
80          String result = "";
81          if (certificate instanceof X509Certificate) {
82  			X509Certificate x509cert = (X509Certificate) certificate;
83  	        ASN1Primitive obj = CertTools.getExtensionValue(x509cert, Extension.subjectDirectoryAttributes.getId());
84  	        if (obj == null) {
85  	            return null;
86  	        }
87  	        ASN1Sequence seq = (ASN1Sequence)obj;
88  	        
89  	        String prefix = "";
90  			SimpleDateFormat dateF = new SimpleDateFormat("yyyyMMdd");
91  	        for (int i = 0; i < seq.size(); i++) {
92  	        	Attribute attr = Attribute.getInstance(seq.getObjectAt(i));
93  	        	if (!StringUtils.isEmpty(result)) {
94  	        		prefix = ", ";
95  	        	}
96  	        	if (attr.getAttrType().getId().equals(id_pda_dateOfBirth)) {
97  	        		ASN1Set set = attr.getAttrValues();
98  	        		// Come on, we'll only allow one dateOfBirth, we're not allowing such frauds with multiple birth dates
99  	        		ASN1GeneralizedTime time = ASN1GeneralizedTime.getInstance(set.getObjectAt(0));
100 	        		Date date = time.getDate();
101 	        		String dateStr = dateF.format(date);
102 	        		result += prefix + "dateOfBirth="+dateStr; 
103 	        	}
104 	        	if (attr.getAttrType().getId().equals(id_pda_placeOfBirth)) {
105 	        		ASN1Set set = attr.getAttrValues();
106 	        		// same here only one placeOfBirth
107 	        		String pb = ((ASN1String)set.getObjectAt(0)).getString();
108 	        		result += prefix + "placeOfBirth="+pb;        			
109 	        	}
110 	        	if (attr.getAttrType().getId().equals(id_pda_gender)) {
111 	        		ASN1Set set = attr.getAttrValues();
112 	        		// same here only one gender
113 	        		String g = ((ASN1String)set.getObjectAt(0)).getString();
114 	        		result += prefix + "gender="+g;        			
115 	        	}
116 	        	if (attr.getAttrType().getId().equals(id_pda_countryOfCitizenship)) {
117 	        		ASN1Set set = attr.getAttrValues();
118 	        		// same here only one citizenship
119 	        		String g = ((ASN1String)set.getObjectAt(0)).getString();
120 	        		result += prefix + "countryOfCitizenship="+g;        			
121 	        	}
122 	        	if (attr.getAttrType().getId().equals(id_pda_countryOfResidence)) {
123 	        		ASN1Set set = attr.getAttrValues();
124 	        		// same here only one residence
125 	        		String g = ((ASN1String)set.getObjectAt(0)).getString();
126 	        		result += prefix + "countryOfResidence="+g;        			
127 	        	}
128 	        }
129         }
130         if (StringUtils.isEmpty(result)) {
131             return null;
132         }
133         return result;            
134 	}

136     /**
137      * From subjectDirAttributes string as defined in getSubjectDirAttribute 
138      * @param dirAttr string of SubjectDirectoryAttributes
139      * @return A Collection of ASN.1 Attribute (org.bouncycastle.asn1.x509), or an empty Collection, never null
140      * @see #getSubjectDirectoryAttributes(Certificate)
141      */
142     public static Collection<Attribute> getSubjectDirectoryAttributes(String dirAttr) {
143     	ArrayList<Attribute> ret = new ArrayList<Attribute>();
144     	Attribute attr = null;
145         String value = CertTools.getPartFromDN(dirAttr, "countryOfResidence");
146         if (!StringUtils.isEmpty(value)) {
147         	ASN1EncodableVector vec = new ASN1EncodableVector();
148         	vec.add(new DERPrintableString(value));
149         	attr = new Attribute(new ASN1ObjectIdentifier(id_pda_countryOfResidence),new DERSet(vec));
150         	ret.add(attr);
151         }
152         value = CertTools.getPartFromDN(dirAttr, "countryOfCitizenship");
153         if (!StringUtils.isEmpty(value)) {
154         	ASN1EncodableVector vec = new ASN1EncodableVector();
155         	vec.add(new DERPrintableString(value));
156         	attr = new Attribute(new ASN1ObjectIdentifier(id_pda_countryOfCitizenship),new DERSet(vec));
157         	ret.add(attr);
158         }
159         value = CertTools.getPartFromDN(dirAttr, "gender");
160         if (!StringUtils.isEmpty(value)) {
161         	ASN1EncodableVector vec = new ASN1EncodableVector();
162         	vec.add(new DERPrintableString(value));
163         	attr = new Attribute(new ASN1ObjectIdentifier(id_pda_gender),new DERSet(vec));
164         	ret.add(attr);
165         }
166         value = CertTools.getPartFromDN(dirAttr, "placeOfBirth");
167         if (!StringUtils.isEmpty(value)) {
168         	ASN1EncodableVector vec = new ASN1EncodableVector();
169         	X509DefaultEntryConverter conv = new X509DefaultEntryConverter();
170         	ASN1Primitive obj = conv.getConvertedValue(new ASN1ObjectIdentifier(id_pda_placeOfBirth), value);
171         	vec.add(obj);
172         	attr = new Attribute(new ASN1ObjectIdentifier(id_pda_placeOfBirth),new DERSet(vec));
173         	ret.add(attr);
174         }        
175         // dateOfBirth that is a GeneralizedTime
176         // The correct format for this is YYYYMMDD, it will be padded to YYYYMMDD120000Z
177         value = CertTools.getPartFromDN(dirAttr, "dateOfBirth");
178         if (!StringUtils.isEmpty(value)) {
179             if (value.length() == 8) {
180                 value += "120000Z"; // standard format according to rfc3739
181             	ASN1EncodableVector vec = new ASN1EncodableVector();
182                 vec.add(new DERGeneralizedTime(value));
183                 attr = new Attribute(new ASN1ObjectIdentifier(id_pda_dateOfBirth),new DERSet(vec));
184                 ret.add(attr);                
185             } else {
186                 log.error("Wrong length of data for 'dateOfBirth', should be of format YYYYMMDD, skipping...");
187             }
188         }
189         return ret;
190     }
