225   /**

227    * @return bool
228    */
229   public static boolean isRevoked(final int revocationReason) {
230     return revocationReason != NOT_REVOKED
231         && revocationReason != REVOCATION_REASON_REMOVEFROMCRL;
232   }

234   /**
235    * @param revocationReason reason
236    * @return bool
237    */
238   public static boolean isPermanentlyRevoked(final int revocationReason) {
239     return isRevoked(revocationReason)
240         && revocationReason != REVOCATION_REASON_CERTIFICATEHOLD;
241   }

261   /**
262    * Merges two collections of RevokedCertInfo. Note that the parameters are
263    * slightly different. Duplicates are removed according to these rules:
264    *
265    * <ul>
266    *   <li>Older permanent status changes win over newer ones
267    *   <li>Permanent status changes always win over temporary ones ("on hold" /
268    *       "re-activate")
269    *   <li>More recent temporary status changes win over older ones
270    * </ul>
271    *
272    * @param a First collection of RevokedCertInfo. May <b>not</b> contain
273    *     duplicates for the same serial number.
274    * @param b Second collection of RevokedCertInfo. May contain duplicates
275    * @param lastBaseCrlDate Entries in unrevoked state will only be included if
276    *     they are more recent than this date. (&lt;= 0 means never include them)
277    * @return CompressionCollection of certificates. May simply be a reference to
278    *     <code>a</code> if <code>b</code> is empty, or a new merged
279    *     CompressedCollection with any duplicates removed.
280    */
281   public static Collection<RevokedCertInfo> mergeByDateAndStatus(
282       final Collection<RevokedCertInfo> a,
283       final Collection<RevokedCertInfo> b,
284       final long lastBaseCrlDate) {
285     // We can optimize this case, but not the reverse, since b can contain
286     // duplicates that should be filtered.
287     if (b.isEmpty()) {
288       return a;
289     }
290     // Merge revocation information
291     final Map<BigInteger, RevokedCertInfo> permRevoked = new HashMap<>();
292     final Map<BigInteger, RevokedCertInfo> tempRevoked = new HashMap<>();
293     for (final RevokedCertInfo revoked : a) {
294       final BigInteger serial = revoked.getUserCertificate();
295       if (revoked.isPermanentlyRevoked()) {
296         permRevoked.put(serial, revoked);
297       } else {
298         tempRevoked.put(serial, revoked);
299       }
300     }
301     for (final RevokedCertInfo revoked : b) {
302       final BigInteger serial = revoked.getUserCertificate();
303       final Date revdate = revoked.getRevocationDate();
304       final RevokedCertInfo permDate = permRevoked.get(serial);
305       if (permDate != null) {
306         // Older permanent status changes win over newer ones
307         if (permDate.getRevocationDate().after(revdate)
308             && revoked.isPermanentlyRevoked()) {
309           permRevoked.put(serial, revoked);
310           tempRevoked.remove(serial);
311         }
312         continue;
313       }
314       if (revoked.isPermanentlyRevoked()) {
315         // Permanently revoked wins over temporary revoked/re-activated
316         permRevoked.put(serial, revoked);
317         tempRevoked.remove(serial);
318         continue;
319       }
320       final RevokedCertInfo tempDate = tempRevoked.get(serial);
321       // More recent temporary status changes win over older ones
322       if (tempDate == null || tempDate.getRevocationDate().before(revdate)) {
323         tempRevoked.put(serial, revoked);
324       }
325     }
326     final CompressedCollection<RevokedCertInfo> mergedRevokedData =
327         new CompressedCollection<>(RevokedCertInfo.class);
328     mergedRevokedData.addAll(
329         permRevoked.values()); // Permanently revoked entries are always added
330     for (final RevokedCertInfo revoked : tempRevoked.values()) {
331       if (!revoked.isRevoked()
332           && (lastBaseCrlDate <= 0
333               || revoked.getRevocationDate().getTime() <= lastBaseCrlDate)) {
334         continue; // REMOVEFROMCRL entries are not added in Base CRLs
335                   // (lastBaseCrlDate=0) or if already removed from the latest
336                   // Base CRL
337       }
338       mergedRevokedData.add(revoked);
339     }
340     mergedRevokedData.closeForWrite();
341     if (LOG.isDebugEnabled()) {
342       LOG.debug(
343           "mergeByDateAndStatus: Merged to "
344               + mergedRevokedData.size()
345               + " entries");
346     }
347     return mergedRevokedData;
348   }
