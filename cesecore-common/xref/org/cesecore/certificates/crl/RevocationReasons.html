13  package org.cesecore.certificates.crl;
14  
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  /**
19   * An enum for revocation reasons, with a numerical database value and a String value for CLI applications.
20   * 
21   * Based on RFC 5280 Section 5.3.1
22   * 
23   * @version $Id: RevocationReasons.java 28485 2018-03-14 12:59:48Z anatom $
24   */
25  public enum RevocationReasons {
26      NOT_REVOKED(-1, "NOT_REVOKED", "The Certificate Is Not Revoked"),
27      UNSPECIFIED(0, "UNSPECIFIED", "Unspecified"),
28      KEYCOMPROMISE(1, "KEY_COMPROMISE", "Key Compromise"),
29      CACOMPROMISE(2, "CA_COMPROMISE", "CA Compromise"),
30      AFFILIATIONCHANGED(3, "AFFILIATION_CHANGED", "Affiliation Changed"),
31      SUPERSEDED(4, "SUPERSEDED", "Superseded"),
32      CESSATIONOFOPERATION(5, "CESSATION_OF_OPERATION", "Cessation of Operation"),
33      CERTIFICATEHOLD(6, "CERTIFICATE_HOLD", "Certificate Hold"),
34      REMOVEFROMCRL(8, "REMOVE_FROM_CRL", "Remove from CRL"),
35      PRIVILEGESWITHDRAWN(9, "PRIVILEGES_WITHDRAWN", "Privileges Withdrawn"),
36      AACOMPROMISE(10, "AA_COMPROMISE", "AA Compromise");
37      
38      private final int databaseValue;
39      private final String stringValue;
40      private final String humanReadable;
41  
42      private static final Map<Integer, RevocationReasons> databaseLookupMap = new HashMap<Integer, RevocationReasons>();
43      private static final Map<String, RevocationReasons> cliLookupMap = new HashMap<String, RevocationReasons>();
44  
45      
46      static {
47          for(RevocationReasons reason : RevocationReasons.values()) {
48              databaseLookupMap.put(reason.getDatabaseValue(), reason);
49              cliLookupMap.put(reason.getStringValue(), reason);
50          }
51      }
52  
53      private RevocationReasons(final int databaseValue, final String stringValue, String humanReadable) {
54          this.databaseValue = databaseValue;
55          this.stringValue = stringValue;
56          this.humanReadable = humanReadable;
57      }
58      
59      public int getDatabaseValue() {
60          return databaseValue;
61      }
62      
63      public String getHumanReadable() {
64          return humanReadable;
65      }
66      
67      public String getStringValue() {
68          return stringValue;
69      }
70      
71      /**
72       * 
73       * @param databaseValue the database value
74       * @return the relevant RevocationReasons object, null if none found. 
75       */
76      public static RevocationReasons getFromDatabaseValue(int databaseValue) {
77          return databaseLookupMap.get(databaseValue);
78      }
79      
80      /**
81       * 
82       * @param cliValue the database value
83       * @return the relevant RevocationReasons object, null if none found. 
84       */
85      public static RevocationReasons getFromCliValue(String cliValue) {
86          if(cliValue == null) {
87              return null;
88          }
89          return cliLookupMap.get(cliValue);
90      }
91  }
