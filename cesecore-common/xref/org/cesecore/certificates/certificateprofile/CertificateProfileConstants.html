14  package org.cesecore.certificates.certificateprofile;
15  
16  /**
17   * Constants for users and certificates. Constants for Type of user: Type is constructed as a mask
18   * since one user can be of several types. To test a user type:
19   * <pre>
20   * if (((type &amp; USER_ENDUSER) == USER_ENDUSER) &amp;&amp; ((type &amp; USER_CAADMIN) == USER_ADMINISTOR) || ...
21   *    ...
22   * </pre>
23   * Bit usage: bits 0-7   (1:st byte):  user types bits 8-15  (2:nd byte):  unused bits 16-23 (3:rd
24   * byte):  unused bits 24-30 (4:th byte):  unused Constants for certificates are simple integer
25   * types. Constants for Token Types Token type is constructed of integer constants since only one
26   * token type can be generated.
27   *
28   * Based on EJBCA (SecConst) version: SecConst.java 9321 2010-06-30 12:49:32Z jeklund
29   *
30   * @version $Id: CertificateProfileConstants.java 28251 2018-02-12 17:04:15Z samuellb $
31   */
32  public final class CertificateProfileConstants {
33      /**
34       * The id of a non-existing certificate profile.
35       */
36      public static final int NO_CERTIFICATE_PROFILE = 0;
37  
38      // Certificate profiles.
39  
40      /** Used when no certificate profile id value is available */
41      public static final int CERTPROFILE_NO_PROFILE            = 0;
42      public static final int CERTPROFILE_FIXED_ENDUSER         = 1;
43      public static final int CERTPROFILE_FIXED_SUBCA           = 2;
44      public static final int CERTPROFILE_FIXED_ROOTCA          = 3;
45  	public static final int CERTPROFILE_FIXED_OCSPSIGNER      = 4;
46  	public static final int CERTPROFILE_FIXED_HARDTOKENAUTH   = 5;
47  	public static final int CERTPROFILE_FIXED_HARDTOKENAUTHENC= 6;
48  	public static final int CERTPROFILE_FIXED_HARDTOKENENC    = 7;
49  	public static final int CERTPROFILE_FIXED_HARDTOKENSIGN   = 8;
50      public static final int CERTPROFILE_FIXED_SERVER          = 9;
51  
52      /**
53       * Constants defining range of id's reserved for fixed certificate types. Observe fixed
54       * certificates cannot have value 0.
55       */
56      public static final int FIXED_CERTIFICATEPROFILE_BOUNDRY = 1000;
57  
58      /**
59       * @param profileId ID
60       * @return true is certificate profile identified by profileId is fixed
61       */
62      public static boolean isFixedCertificateProfile(final int profileId) {
63      	return (
64      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER ||
65      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_SUBCA ||
66      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_ROOTCA ||
67      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_HARDTOKENAUTH ||
68      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_HARDTOKENAUTHENC ||
69      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_HARDTOKENENC ||
70      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_HARDTOKENSIGN ||
71      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_OCSPSIGNER ||
72      			profileId == CertificateProfileConstants.CERTPROFILE_FIXED_SERVER );
73      }
74  
75      /**
76       * Default key lengths. Users are allowed to choose from these key lengths when
77       * lookup of available bit lengths fails.
78       * @see "org.ejbca.ui.web.pub.ApplyBean (in publicweb-gui)"
79       */
80      public static final int[] DEFAULT_KEY_LENGTHS = new int[] {512, 1024, 2048};
81      
82      /** QC ETSI Type OID: Electronic Signature */
83      public static final String QC_ETSI_TYPE_ESIGN = "0.4.0.1862.1.6.1";
84      /** QC ETSI Type OID: Electronic Seal */
85      public static final String QC_ETSI_TYPE_ESEAL = "0.4.0.1862.1.6.2";
86      /** QC ETSI Type OID: Website Authentication */
87      public static final String QC_ETSI_TYPE_WEBAUTH = "0.4.0.1862.1.6.3";
88  
89      /**
90       * Prevents creation of new class
91       */
92      private CertificateProfileConstants() {
93      }
94  
95  }
