183     public void setDN(String dn) {
184         if (dn==null) {
185             dn = "";
186         }
187     	final StringBuilder removedAllEmpties = new StringBuilder(dn.length());
188         final StringBuilder removedTrailingEmpties = DNFieldsUtil.removeEmpties(dn, removedAllEmpties, true);
189         if (removedTrailingEmpties == null) {
190         	this.subjectDNClean=StringTools.putBase64String(removedAllEmpties.toString());
191         	this.subjectDN=this.subjectDNClean;
192     	} else {
193         	this.subjectDNClean=StringTools.putBase64String(removedAllEmpties.toString());
194         	this.subjectDN=StringTools.putBase64String(removedTrailingEmpties.toString());
195     	}
196     }

381     /**
382      * @return an information map about this end entity, listing all general fields.
383      */
384     public Map<String, String> getDetailMap() {
385         @SuppressWarnings("unchecked")
386         Map<String, String> details = new Base64GetHashMap();
387         details.put("caid", Integer.toString(caid));
388         details.put("cardnumber", cardNumber);
389         details.put("certificateprofileid", Integer.toString(certificateprofileid));
390         details.put("endentityprofileid", Integer.toString(endentityprofileid));
391         if (extendedinformation != null) {
392             StringBuilder extendedInformationDump = new StringBuilder("{");
393             LinkedHashMap<Object, Object> rawData = extendedinformation.getRawData();
394             for (Object key : rawData.keySet()) {
395                 if (rawData.get(key) != null) {
396                     extendedInformationDump.append(", [" + (String) key + ":" + rawData.get(key).toString() + "]");
397                 }
398             }
399             extendedInformationDump.append("}");
400             details.put("extendedInformation", extendedInformationDump.substring(2));
401         }
402         details.put("hardtokenissuerid", Integer.toString(hardtokenissuerid));
403         details.put("status", Integer.toString(status));
404         details.put("subjectAltName", subjectAltName);
405         details.put("subjectDN", subjectDN);
406         details.put("subjectEmail", subjectEmail);
407         if (timecreated != null) {
408             details.put("timecreated", timecreated.toString());
409         }
410         if (timemodified != null) {
411             details.put("timemodified", timemodified.toString());
412         }
413         details.put("tokentype", Integer.toString(tokentype));
414         details.put("type", Integer.toString(type));
415         details.put("username", username);
416         return details;
417     }

420      *
421      *
422      * @param other another {@link EndEntityInformation}
423      * @return the differences between this map and the parameter, as &lt;key, [thisValue, otherValue]&gt;
424      */
425     public Map<String, String[]> getDiff(EndEntityInformation other) {
426         Map<String, String[]> changedValues = new LinkedHashMap<>();
427         Map<String, String> thisValues = getDetailMap();
428         Map<String, String> otherValues = other.getDetailMap();
429         List<String> thisKeySet = new ArrayList<>(thisValues.keySet());
430         for (String key : thisKeySet) {
431             String thisValue = thisValues.get(key);
432             String otherValue = otherValues.get(key);
433             if (thisValue == null) {
434                 if (otherValue != null) {
435                     changedValues.put(key, new String[] { "<null>", otherValue });
436                 }
437             } else if (!thisValue.equals(otherValue)) {
438                 changedValues.put(key, new String[] { thisValue, otherValue });
439             }
440             thisValues.remove(key);
441             otherValues.remove(key);
442         }
443         //Add in any values that may have been in otherValues but not here
444         for (String otherKey : otherValues.keySet()) {
445             changedValues.put(otherKey, new String[] { "<null>", otherValues.get(otherKey) });
446         }
447         return changedValues;
448     }
