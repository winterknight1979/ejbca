13  package org.cesecore.certificates.endentity;
14  
15  import java.util.ArrayList;
16  import java.util.Collection;
17  import java.util.Collections;
18  import java.util.HashMap;
19  import java.util.List;
20  
21  /** Constants for End Entity types
22   *
23   * @version $Id: EndEntityConstants.java 27816 2018-01-09 16:19:36Z samuellb $
24   */
25  public final class EndEntityConstants {
26      /**
27       * The id of a non-existing end entity profile.
28       */
29      public static final int NO_END_ENTITY_PROFILE = 0;
30      /**
31       * The id for the built-in EMPTY end entity profile.
32       */
33      public static final int EMPTY_END_ENTITY_PROFILE = 1;
34      public static final String EMPTY_ENDENTITYPROFILENAME = "EMPTY";
35  
36      //
37      // User status codes
38      //
39      public static final int STATUS_NEW = 10;        // New user
40      public static final int STATUS_FAILED = 11;     // Generation of user certificate failed
41      public static final int STATUS_INITIALIZED = 20;// User has been initialized
42      public static final int STATUS_INPROCESS = 30;  // Generation of user certificate in process
43      public static final int STATUS_GENERATED = 40;  // A certificate has been generated for the user
44      public static final int STATUS_REVOKED = 50;  // The user has been revoked and should not have any more certificates issued
45      public static final int STATUS_HISTORICAL = 60; // The user is old and archived
46      public static final int STATUS_KEYRECOVERY  = 70; // The user is should use key recovery functions in next certificate generation.
47      public static final int STATUS_WAITINGFORADDAPPROVAL = 80; // the operation is waiting to be approved before execution. This status
48                                                              // is never stored in the database, but is used transiently when a request
49                                                              // is not stored because it's waiting for approval. This status is primarily
50                                                              // used to send the right notification note when a request is waiting for approval.
51  
52      //
53      // Token types.
54      //
55      /** Indicates that a user generated token should be used, i.e not token generated but we expect a request and will create a certificate */
56      public static final int TOKEN_USERGEN = 1;
57      /** Indicates that a p12 token should be generated. */
58      public static final int TOKEN_SOFT_P12 = 2;
59      /** Indicates that a jks token should be generated. */
60      public static final int TOKEN_SOFT_JKS = 3;
61      /** Indicates that a pem token should be generated. */
62      public static final int TOKEN_SOFT_PEM = 4;
63      /** All values equal or below this constant should be treated as a soft token. */
64      public static final int TOKEN_SOFT = 100;
65  
66      //
67      // Names and language strings of statuses
68      //
69      /** These string values maps a status code to a language string in the admin GUI language property files */
70      private static final HashMap<Integer, String> STATUS_TEXT_TRANS = new HashMap<Integer, String>();
71      static {
72          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_NEW),"STATUSNEW");
73          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_FAILED),"STATUSFAILED");
74          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_INITIALIZED),"STATUSINITIALIZED");
75          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_INPROCESS),"STATUSINPROCESS");
76          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_GENERATED),"STATUSGENERATED");
77          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_REVOKED),"STATUSREVOKED");
78          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_HISTORICAL),"STATUSHISTORICAL");
79          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_KEYRECOVERY),"STATUSKEYRECOVERY");
80          STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_WAITINGFORADDAPPROVAL),"STATUSWAITINGFORADDAPPROVAL");
81      }
82  
83      public static String getTranslatableStatusText(int status) {
84          String ret = null;
85          Object o =  STATUS_TEXT_TRANS.get(Integer.valueOf(status));
86          if (o != null) {
87              ret = (String)o;
88          }
89          return ret;
90      }
91  
92      /** These string values maps a status code to a plain string */
93      private static final HashMap<Integer, String> STATUS_TEXT = new HashMap<Integer, String>();
94      static {
95          STATUS_TEXT.put(Integer.valueOf(STATUS_NEW),"NEW");
96          STATUS_TEXT.put(Integer.valueOf(STATUS_FAILED),"FAILED");
97          STATUS_TEXT.put(Integer.valueOf(STATUS_INITIALIZED),"INITIALIZED");
98          STATUS_TEXT.put(Integer.valueOf(STATUS_INPROCESS),"INPROCESS");
99          STATUS_TEXT.put(Integer.valueOf(STATUS_GENERATED),"GENERATED");
100         STATUS_TEXT.put(Integer.valueOf(STATUS_REVOKED),"REVOKED");
101         STATUS_TEXT.put(Integer.valueOf(STATUS_HISTORICAL),"HISTORICAL");
102         STATUS_TEXT.put(Integer.valueOf(STATUS_KEYRECOVERY),"KEYRECOVERY");
103     }
104 
105     public static String getStatusText(int status) {
106         String ret = null;
107         Object o =  STATUS_TEXT.get(Integer.valueOf(status));
108         if (o != null) {
109             ret = (String)o;
110         }
111         return ret;
112     }
113 
114     public static Collection<Integer> getAllStatusCodes() {
115         List<Integer> statuses = new ArrayList<Integer>(STATUS_TEXT.keySet());
116         Collections.sort(statuses);
117         return statuses;
118     }
119 
120 }
