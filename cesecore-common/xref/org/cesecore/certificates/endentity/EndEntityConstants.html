13  package org.cesecore.certificates.endentity;
14  
15  import java.util.ArrayList;
16  import java.util.Collection;
17  import java.util.Collections;
18  import java.util.HashMap;
19  import java.util.List;
20  
21  /**
22   * Constants for End Entity types.
23   *
24   * @version $Id: EndEntityConstants.java 27816 2018-01-09 16:19:36Z samuellb $
25   */
26  public final class EndEntityConstants {
27  
28    /** don't instantiate. */
29    private EndEntityConstants() { }
30  
31    /** The id of a non-existing end entity profile. */
32    public static final int NO_END_ENTITY_PROFILE = 0;
33    /** The id for the built-in EMPTY end entity profile. */
34    public static final int EMPTY_END_ENTITY_PROFILE = 1;
35    /** Empty profile. */
36    public static final String EMPTY_ENDENTITYPROFILENAME = "EMPTY";
37  
38    //
39    // User status codes
40    //
41    /** New user.*/
42    public static final int STATUS_NEW = 10;
43    /** Generation of user certificate failed. */
44    public static final int STATUS_FAILED =
45            11;
46    /** User has been initialized. */
47    public static final int STATUS_INITIALIZED = 20;
48    /** Generation of user certificate in process. */
49    public static final int STATUS_INPROCESS =
50            30;
51    /** A certificate has been generated for the user. */
52    public static final int STATUS_GENERATED =
53            40;
54    /** The user has been revoked and should not have any more certificates
55     * issued. */
56    public static final int STATUS_REVOKED =
57            50;
58    /** The user is old and archived. */
59    public static final int STATUS_HISTORICAL =
60            60;
61    /** The user is should use key recovery functions in next certificate
62     * generation. */
63    public static final int STATUS_KEYRECOVERY =
64            70;
65  /** The operation is waiting to be approved before execution. This
66    * status
67    * is never stored in the database, but is used transiently when a request
68    * is not stored because it's waiting for approval. This status is primarily
69    * used to send the right notification note when a request is waiting for
70    * approval.*/
71    public static final int STATUS_WAITINGFORADDAPPROVAL =
72        80;
73  
74    //
75    // Token types.
76    //
77    /**
78     * Indicates that a user generated token should be used, i.e not token
79     * generated but we expect a request and will create a certificate.
80     */
81    public static final int TOKEN_USERGEN = 1;
82    /** Indicates that a p12 token should be generated. */
83    public static final int TOKEN_SOFT_P12 = 2;
84    /** Indicates that a jks token should be generated. */
85    public static final int TOKEN_SOFT_JKS = 3;
86    /** Indicates that a pem token should be generated. */
87    public static final int TOKEN_SOFT_PEM = 4;
88    /**
89     * All values equal or below this constant should be treated as a soft token.
90     */
91    public static final int TOKEN_SOFT = 100;
92  
93    //
94    // Names and language strings of statuses
95    //
96    /**
97     * These string values maps a status code to a language string in the admin
98     * GUI language property files.
99     */
100   private static final HashMap<Integer, String> STATUS_TEXT_TRANS =
101       new HashMap<Integer, String>();
102 
103   static {
104     STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_NEW), "STATUSNEW");
105     STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_FAILED), "STATUSFAILED");
106     STATUS_TEXT_TRANS.put(
107         Integer.valueOf(STATUS_INITIALIZED), "STATUSINITIALIZED");
108     STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_INPROCESS), "STATUSINPROCESS");
109     STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_GENERATED), "STATUSGENERATED");
110     STATUS_TEXT_TRANS.put(Integer.valueOf(STATUS_REVOKED), "STATUSREVOKED");
111     STATUS_TEXT_TRANS.put(
112         Integer.valueOf(STATUS_HISTORICAL), "STATUSHISTORICAL");
113     STATUS_TEXT_TRANS.put(
114         Integer.valueOf(STATUS_KEYRECOVERY), "STATUSKEYRECOVERY");
115     STATUS_TEXT_TRANS.put(
116         Integer.valueOf(STATUS_WAITINGFORADDAPPROVAL),
117         "STATUSWAITINGFORADDAPPROVAL");
118   }
119 
120   /**
121    * @param status Status code
122    * @return Translatable text
123    */
124   public static String getTranslatableStatusText(final int status) {
125     String ret = null;
126     Object o = STATUS_TEXT_TRANS.get(Integer.valueOf(status));
127     if (o != null) {
128       ret = (String) o;
129     }
130     return ret;
131   }
132 
133   /** These string values maps a status code to a plain string. */
134   private static final HashMap<Integer, String> STATUS_TEXT =
135       new HashMap<Integer, String>();
136 
137   static {
138     STATUS_TEXT.put(Integer.valueOf(STATUS_NEW), "NEW");
139     STATUS_TEXT.put(Integer.valueOf(STATUS_FAILED), "FAILED");
140     STATUS_TEXT.put(Integer.valueOf(STATUS_INITIALIZED), "INITIALIZED");
141     STATUS_TEXT.put(Integer.valueOf(STATUS_INPROCESS), "INPROCESS");
142     STATUS_TEXT.put(Integer.valueOf(STATUS_GENERATED), "GENERATED");
143     STATUS_TEXT.put(Integer.valueOf(STATUS_REVOKED), "REVOKED");
144     STATUS_TEXT.put(Integer.valueOf(STATUS_HISTORICAL), "HISTORICAL");
145     STATUS_TEXT.put(Integer.valueOf(STATUS_KEYRECOVERY), "KEYRECOVERY");
146   }
147 
148   /**
149    * @param status Status code
150    * @return Status text
151    */
152   public static String getStatusText(final int status) {
153     String ret = null;
154     Object o = STATUS_TEXT.get(Integer.valueOf(status));
155     if (o != null) {
156       ret = (String) o;
157     }
158     return ret;
159   }
160 
161   /**
162    * @return Collection of all status codes.
163    */
164   public static Collection<Integer> getAllStatusCodes() {
165     List<Integer> statuses = new ArrayList<Integer>(STATUS_TEXT.keySet());
166     Collections.sort(statuses);
167     return statuses;
168   }
169 }
