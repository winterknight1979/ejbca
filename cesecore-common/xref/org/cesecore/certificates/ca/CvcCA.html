82      private static CvcPlugin createCAImpl(final String type) {
83          // type can be used to differentiate between different types of CVC CA implementations as there
84          // can be several different types of CVC: EAC, Tachograph, eID etc.
85          ServiceLoader<? extends CvcPlugin> serviceLoader = getImplementationClasses();
86          for (CvcPlugin cvcPlugin : serviceLoader) {
87          	if (log.isDebugEnabled()) {
88          	    log.debug("ServiceLoader found CvcPlugin implementation: "+cvcPlugin.getCvcType());
89          	}
90              if (type.equals(cvcPlugin.getCvcType())) {
91                  return cvcPlugin;
92              }            
93          }
94          // No implementation found, it is probably an Enterprise only feature
95          log.info("CVC CA is not available in this version of EJBCA.");
96          return null;
97      }

171 	/** Implementation of UpgradableDataHashMap function upgrade. 
172 	 */
173     @SuppressWarnings("deprecation")
174     @Override
175 	public void upgrade(){
176 		if(Float.compare(LATEST_VERSION, getVersion()) != 0) {
177 			// New version of the class, upgrade
178             log.info("Upgrading CVCCA with version "+getVersion());
179 
180 			// Put upgrade code here...
181             
182             // v1->v2 is only an upgrade in order to upgrade CA token
183             // v2->v3 is a upgrade of X509CA that has to be adjusted here too, due to the common heritage
184             if (data.get(CRLPERIOD) instanceof Integer) {
185             	setCRLPeriod(0L);
186             }
187             if (data.get(CRLISSUEINTERVAL) instanceof Integer) {
188             	setCRLIssueInterval(0L);
189             }
190             if (data.get(CRLOVERLAPTIME) instanceof Integer) {
191             	setCRLOverlapTime(0L);
192             }
193             if (data.get(DELTACRLPERIOD) instanceof Integer) {
194             	setDeltaCRLPeriod(0L);
195             }
196 
197             // v4.
198             // 'encodedValidity' MUST set to "" (Empty String) here. The initialization is done during post-upgrade of EJBCA 6.6.1.
199             if(null == data.get(ENCODED_VALIDITY) && null != data.get(VALIDITY)) {
200                 setEncodedValidity(getEncodedValidity());
201             }
202             
203             data.put(VERSION, new Float(LATEST_VERSION));
204 		}  
205 	}

207 	/**
208 	 * Method to upgrade new (or existing external caservices)
209 	 * This method needs to be called outside the regular upgrade
210 	 * since the CA isn't instantiated in the regular upgrade.
211 	 */
212     @Override
213 	@SuppressWarnings("deprecation")
214     public boolean upgradeExtendedCAServices() {
215 	    boolean retval = false;
216 	    Collection<Integer> externalServiceTypes = getExternalCAServiceTypes();
217         if (!CesecoreConfiguration.getCaKeepOcspExtendedService() && externalServiceTypes.contains(ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE)) {
218             //This type has been removed, so remove it from any CAs it's been added to as well.
219             externalServiceTypes.remove(ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE);
220             data.put(EXTENDEDCASERVICES, externalServiceTypes);
221             retval = true;
222         }
223 		return retval;
224 	}
