32  /**
33   * Base class for CVC CAs. Holds data specific for Certificate and CRL
34   * generation according to the CVC (Card Verifiable Certificate) standards,
35   * which are not real standards. There can be many different implementations of
36   * CVC CA which are quite different, for example EU EAC electronic passports,
37   * Tachographs and eIDs.
38   *
39   * @version $Id: CvcCA.java 30618 2018-11-26 07:27:19Z samuellb $
40   */

116   private static CvcPlugin createCAImpl(final String type) {
117     // type can be used to differentiate between different types of CVC CA
118     // implementations as there
119     // can be several different types of CVC: EAC, Tachograph, eID etc.
120     ServiceLoader<? extends CvcPlugin> serviceLoader =
121         getImplementationClasses();
122     for (CvcPlugin cvcPlugin : serviceLoader) {
123       if (LOG.isDebugEnabled()) {
124         LOG.debug(
125             "ServiceLoader found CvcPlugin implementation: "
126                 + cvcPlugin.getCvcType());
127       }
128       if (type.equals(cvcPlugin.getCvcType())) {
129         return cvcPlugin;
130       }
131     }
132     // No implementation found, it is probably an Enterprise only feature
133     LOG.info("CVC CA is not available in this version of EJBCA.");
134     return null;
135   }

137   /**
138    * Constructor used when retrieving existing CVCCA from database.
139    *
140    * @param data data
141    * @param caId ID
142    * @param subjectDN DN
143    * @param name name
144    * @param status status
145    * @param updateTime Update time
146    * @param expireTime expiry
147    */
148   @SuppressWarnings("deprecation")
149   public void init(
150       final HashMap<Object, Object> data,
151       final int caId,
152       final String subjectDN,
153       final String name,
154       final int status,
155       final Date updateTime,
156       final Date expireTime) {
157     super.init(data);
158     setExpireTime(expireTime);
159     final List<ExtendedCAServiceInfo> externalcaserviceinfos =
160         new ArrayList<>();
161     for (final Integer externalCAServiceType : getExternalCAServiceTypes()) {
162       // Type was removed in 6.0.0. It is removed from the database in the
163       // upgrade method in this class, but it needs to be ignored
164       // for instantiation.
165       if (externalCAServiceType
166           != ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE) {
167         final ExtendedCAServiceInfo info =
168             this.getExtendedCAServiceInfo(externalCAServiceType.intValue());
169         if (info != null) {
170           externalcaserviceinfos.add(info);
171         }
172       }
173     }
174 
175     final CVCCAInfo info =
176         new CVCCAInfo(
177             subjectDN,
178             name,
179             status,
180             updateTime,
181             getCertificateProfileId(),
182             getDefaultCertificateProfileId(),
183             getEncodedValidity(),
184             getExpireTime(),
185             getCAType(),
186             getSignedBy(),
187             getCertificateChain(),
188             getCAToken(),
189             getDescription(),
190             getRevocationReason(),
191             getRevocationDate(),
192             getCRLPeriod(),
193             getCRLIssueInterval(),
194             getCRLOverlapTime(),
195             getDeltaCRLPeriod(),
196             getCRLPublishers(),
197             getValidators(),
198             getFinishUser(),
199             externalcaserviceinfos,
200             getApprovals(),
201             getIncludeInHealthCheck(),
202             isDoEnforceUniquePublicKeys(),
203             isDoEnforceUniqueDistinguishedName(),
204             isDoEnforceUniqueSubjectDNSerialnumber(),
205             isUseCertReqHistory(),
206             isUseUserStorage(),
207             isUseCertificateStorage(),
208             isAcceptRevocationNonExistingEntry());
209     // These to settings were deprecated in 6.8.0, but are still set for upgrade
210     // reasons
211     info.setApprovalProfile(getApprovalProfile());
212     info.setApprovalSettings(getApprovalSettings());
213     super.setCAInfo(info);
214     setCAId(caId);
215   }

261   /** Implementation of UpgradableDataHashMap function upgrade. */
262   @SuppressWarnings("deprecation")
263   @Override
264   public void upgrade() {
265     if (Float.compare(LATEST_VERSION, getVersion()) != 0) {
266       // New version of the class, upgrade
267       LOG.info("Upgrading CVCCA with version " + getVersion());
268 
269       // Put upgrade code here...
270 
271       // v1->v2 is only an upgrade in order to upgrade CA token
272       // v2->v3 is a upgrade of X509CA that has to be adjusted here too, due to
273       // the common heritage
274       if (data.get(CRLPERIOD) instanceof Integer) {
275         setCRLPeriod(0L);
276       }
277       if (data.get(CRLISSUEINTERVAL) instanceof Integer) {
278         setCRLIssueInterval(0L);
279       }
280       if (data.get(CRLOVERLAPTIME) instanceof Integer) {
281         setCRLOverlapTime(0L);
282       }
283       if (data.get(DELTACRLPERIOD) instanceof Integer) {
284         setDeltaCRLPeriod(0L);
285       }
286 
287       // v4.
288       // 'encodedValidity' MUST set to "" (Empty String) here. The
289       // initialization is done during post-upgrade of EJBCA 6.6.1.
290       if (null == data.get(ENCODED_VALIDITY) && null != data.get(VALIDITY)) {
291         setEncodedValidity(getEncodedValidity());
292       }
293 
294       data.put(VERSION, new Float(LATEST_VERSION));
295     }
296   }

298   /**
299    * Method to upgrade new (or existing external caservices) This method needs
300    * to be called outside the regular upgrade since the CA isn't instantiated in
301    * the regular upgrade.
302    */
303   @Override
304   @SuppressWarnings("deprecation")
305   public boolean upgradeExtendedCAServices() {
306     boolean retval = false;
307     Collection<Integer> externalServiceTypes = getExternalCAServiceTypes();
308     if (!CesecoreConfiguration.getCaKeepOcspExtendedService()
309         && externalServiceTypes.contains(
310             ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE)) {
311       // This type has been removed, so remove it from any CAs it's been added
312       // to as well.
313       externalServiceTypes.remove(
314           ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE);
315       data.put(EXTENDEDCASERVICES, externalServiceTypes);
316       retval = true;
317     }
318     return retval;
319   }
