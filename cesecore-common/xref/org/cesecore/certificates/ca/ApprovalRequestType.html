13  package org.cesecore.certificates.ca;
14  
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  /**
19   * Represents a type of approval request.
20   *
21   * @version $Id: ApprovalRequestType.java 29813 2018-09-05 15:01:59Z bastianf $
22   */
23  public enum ApprovalRequestType {
24    /** Add. */
25    ADDEDITENDENTITY(1, "APPROVEADDEDITENDENTITY"),
26    /** Recover. */
27    KEYRECOVER(2, "APPROVEKEYRECOVER"),
28    /** Revoke. */
29    REVOCATION(3, "APPROVEREVOCATION"),
30    /** Activate. */
31    ACTIVATECA(4, "APPROVEACTIVATECA");
32  
33    /** Value. */
34    private final int integerValue;
35    /** Language. */
36    private final String languageString;
37    /** Map of request types. */
38    private static final Map<Integer, ApprovalRequestType> REVERSE_LOOKUP_MAP =
39        new HashMap<>();
40  
41    static {
42      for (ApprovalRequestType approvalRequestType
43          : ApprovalRequestType.values()) {
44        REVERSE_LOOKUP_MAP.put(
45            approvalRequestType.getIntegerValue(), approvalRequestType);
46      }
47    }
48  
49    /**
50     * Constructor.
51     *
52     * @param anIntegerValue Value
53     * @param aLanguageString Language
54     */
55    ApprovalRequestType(final int anIntegerValue, final String aLanguageString) {
56      this.integerValue = anIntegerValue;
57      this.languageString = aLanguageString;
58    }
59  
60    /** @return value */
61    public int getIntegerValue() {
62      return integerValue;
63    }
64  
65    /** @return language */
66    public String getLanguageString() {
67      return languageString;
68    }
69  
70    /**
71     * Get request type.
72     *
73     * @param integerValue Value.
74     * @return Type
75     */
76    public static ApprovalRequestType getFromIntegerValue(
77        final int integerValue) {
78      return REVERSE_LOOKUP_MAP.get(integerValue);
79    }
80  }
