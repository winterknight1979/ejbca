18  import java.util.Date;

20  import org.bouncycastle.asn1.ASN1Encodable;
21  import org.bouncycastle.asn1.x509.Extensions;
22  import org.cesecore.certificates.certificate.certextensions.AvailableCustomCertificateExtensionsConfiguration;
23  import org.cesecore.certificates.certificate.certextensions.CertificateExtensionException;
24  import org.cesecore.certificates.certificate.request.RequestMessage;
25  import org.cesecore.certificates.certificateprofile.CertificateProfile;
26  import org.cesecore.certificates.endentity.EndEntityInformation;

31  /**
32   * CvcPlugin is an interface for CVC CA implementation. There can be different types of CVCCAs.  
33   *
34   * @version $Id: CvcPlugin.java 27346 2017-11-28 16:14:30Z samuellb $
35   */
36  public interface CvcPlugin {
37  
38      /** @return A string representing the type of CVC CA this is */ 
39      String getCvcType();
40      
41      /** 
42       * @param cryptoToken Token
43       * @param attributes Attrs
44       * @param signAlg Algorithm
45       * @param cacert CA Certificate
46       * @param signatureKeyPurpose Purpose
47       * @param certificateProfile Profile
48       * @param cceConfig Config
49       * @return Request
50       * @throws CryptoTokenOfflineException if offline 
51       * @throws CertificateExtensionException if certificate has invalid extensions
52       * @see org.cesecore.certificates.ca.CA#createRequest(CryptoToken, Collection, String, Certificate, int, CertificateProfile, AvailableCustomCertificateExtensionsConfiguration) 
53       */
54      byte[] createRequest(CryptoToken cryptoToken, Collection<ASN1Encodable> attributes, String signAlg, Certificate cacert, int signatureKeyPurpose,
55              CertificateProfile certificateProfile, AvailableCustomCertificateExtensionsConfiguration cceConfig) throws CryptoTokenOfflineException, CertificateExtensionException;
56  
57      /** 
58       * @param cryptoToken Token
59       * @param request Request
60       * @return Response
61       * @throws CryptoTokenOfflineException if offline 
62       * @see org.cesecore.certificates.ca.CA#createAuthCertSignRequest(CryptoToken, byte[]) 
63       */
64      byte[] createAuthCertSignRequest(CryptoToken cryptoToken, byte[] request) throws CryptoTokenOfflineException;
65      
66      /** 
67       * @param cryptoToken Token
68       * @param createLinkCertificate bool 
69       * @param certProfile Profile
70       * @param cceConfig Config
71       * @param oldCaCert Old certificate
72       * @throws CryptoTokenOfflineException if offline
73       * @see org.cesecore.certificates.ca.CA#createOrRemoveLinkCertificate(CryptoToken, boolean, CertificateProfile, AvailableCustomCertificateExtensionsConfiguration, Certificate) 
74       */
75      void createOrRemoveLinkCertificate(final CryptoToken cryptoToken, final boolean createLinkCertificate, final CertificateProfile certProfile, 
76              AvailableCustomCertificateExtensionsConfiguration cceConfig, Certificate oldCaCert) throws CryptoTokenOfflineException;   
77  
78      /** 
79       * @param cryptoToken Token
80       * @param subject Subject
81       * @param request Request
82       * @param publicKey Public key
83       * @param keyusage Usage
84       * @param notBefore start date
85       * @param notAfter End date
86       * @param certProfile Profile
87       * @param extensions extensions
88       * @param sequence sequence
89       * @param cceConfig config
90       * @return certificate
91       * @throws Exception on error
92       * @see org.cesecore.certificates.ca.CA#generateCertificate(CryptoToken, EndEntityInformation, RequestMessage, PublicKey, int, Date, Date, CertificateProfile, Extensions, String, CertificateGenerationParams, AvailableCustomCertificateExtensionsConfiguration) 
93       */
94      Certificate generateCertificate(CryptoToken cryptoToken, EndEntityInformation subject, 
95              RequestMessage request,
96              PublicKey publicKey, 
97              int keyusage, 
98              Date notBefore,
99              Date notAfter,
100             CertificateProfile certProfile,
101             Extensions extensions,
102             String sequence, 
103             AvailableCustomCertificateExtensionsConfiguration cceConfig) throws Exception;
104 }
