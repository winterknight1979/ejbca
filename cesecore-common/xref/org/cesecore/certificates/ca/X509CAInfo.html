67      /**
68       * This constructor can be used when creating a CA.
69       * This constructor uses defaults for the fields that are not specified.
70       * @param subjectdn DN
71       * @param name Name
72       * @param status Status
73       * @param certificateProfileId Profile
74       * @param encodedValidity Vlidity
75       * @param signedby Signer
76       * @param certificatechain Certs 
77       * @param catoken Token
78       */
79      public X509CAInfo(final String subjectdn, final String name, final int status,
80              final int certificateProfileId, final String encodedValidity, int signedby, final Collection<Certificate> certificatechain, final CAToken catoken) {
81          this(subjectdn,
82               name,
83               status, // CA status (CAConstants.CA_ACTIVE, etc.)
84               new Date(), // update time
85               "", // Subject Alternative name
86               certificateProfileId, // CA certificate profile
87                  0, // default ca profile
88               false, // default is certificate data table   
89               encodedValidity, null, // Expiretime
90               CAInfo.CATYPE_X509, // CA type (X509/CVC)
91               signedby, // Signed by CA
92               certificatechain, // Certificate chain
93               catoken, // CA Token
94               "", // Description
95               CesecoreConfiguration.getSerialNumberOctetSizeForNewCa(), // serial number octet size
96               -1, // Revocation reason
97               null, // Revocation date
98               null, // PolicyId
99               1 * SimpleTime.MILLISECONDS_PER_DAY, // CRLPeriod
100              0L, // CRLIssueInterval
101              10 * SimpleTime.MILLISECONDS_PER_MINUTE, // CRLOverlapTime
102              0L, // DeltaCRLPeriod
103              new ArrayList<Integer>(),
104              new ArrayList<Integer>(),
105              true, // Authority Key Identifier
106              false, // Authority Key Identifier Critical
107              true, // CRL Number
108              false, // CRL Number Critical
109              null, // defaultcrldistpoint
110              null, // defaultcrlissuer
111              null, // defaultocsplocator
112              null, // CRL Authority Information Access (AIA) extension
113              null, // Certificate AIA default CA issuer URI
114              null, null, // Name Constraints (permitted/excluded)
115              null, // defaultfreshestcrl
116              true, // Finish User
117              new ArrayList<ExtendedCAServiceInfo>(), // no extended services
118              false, // use default utf8 settings
119              new HashMap<ApprovalRequestType, Integer>(), //approvals
120              false, // Use UTF8 subject DN by default
121              true, // Use LDAP DN order by default
122              false, // Use CRL Distribution Point on CRL
123              false, // CRL Distribution Point on CRL critical
124              true, // Include in HealthCheck
125              true, // isDoEnforceUniquePublicKeys
126              true, // isDoEnforceUniqueDistinguishedName
127              false, // isDoEnforceUniqueSubjectDNSerialnumber
128              false, // useCertReqHistory
129              true, // useUserStorage
130              true, // useCertificateStorage
131              false, // acceptRevocationNonExistingEntry
132              null, // cmpRaAuthSecret
133              false // keepExpiredCertsOnCRL
134         );
135     }

155      * @param caSerialNumberOctetSize serial number octet size for this CA
156      * @param revocationReason the reason why this CA was revoked, or -1 if not revoked

158      * @param policies a policy OID
159      * @param crlperiod the CRL validity period in ms
160      * @param crlIssueInterval how often in ms the CRLs should be distributed, e.g. 3600000 will generate a new CRL every hour

162      * @param deltacrlperiod how often Delta CRLs should be distributed
163      * @param crlpublishers a collection of publisher IDs for this CA
164      * @param keyValidators a collection of key validator IDs for this CA
165      * @param useauthoritykeyidentifier bool
166      * @param authoritykeyidentifiercritical bool
167      * @param usecrlnumber bool
168      * @param crlnumbercritical bool
169      * @param defaultcrldistpoint the URI of the default CRL distribution point
170      * @param defaultcrlissuer Issuer
171      * @param defaultocspservicelocator Locator
172      * @param authorityInformationAccess Acces
173      * @param certificateAiaDefaultCaIssuerUri Issuer URI
174      * @param nameConstraintsPermitted a list of name constraints which should be permitted
175      * @param nameConstraintsExcluded a list of name constraints which should be excluded

196     private X509CAInfo(final String subjectDn, final String name, final int status, final Date updateTime, final String subjectaltname,
197             final int certificateprofileid, final int defaultCertprofileId, final boolean useNoConflictCertificateData, final String encodedValidity, final Date expiretime, final int catype, final int signedBy,
198             final Collection<Certificate> certificatechain, final CAToken catoken,
199     		final String description, final int caSerialNumberOctetSize, final int revocationReason, final Date revocationDate, final List<CertificatePolicy> policies,
200     		final long crlperiod, final long crlIssueInterval, final long crlOverlapTime, final long deltacrlperiod,
201     		final Collection<Integer> crlpublishers, final Collection<Integer> keyValidators, final boolean useauthoritykeyidentifier, final boolean authoritykeyidentifiercritical,
202     		final boolean usecrlnumber, final boolean crlnumbercritical, final String defaultcrldistpoint, final String defaultcrlissuer,
203     		final String defaultocspservicelocator,
204     		final List<String> authorityInformationAccess,
205     		final List<String> certificateAiaDefaultCaIssuerUri,
206     		final List<String> nameConstraintsPermitted, final List<String> nameConstraintsExcluded, final String cadefinedfreshestcrl,
207     		final boolean finishuser, final Collection<ExtendedCAServiceInfo> extendedcaserviceinfos,
208     		final boolean useUTF8PolicyText, final Map<ApprovalRequestType, Integer> approvals, final boolean usePrintableStringSubjectDN,
209     		final boolean useLdapDnOrder, final boolean useCrlDistributionPointOnCrl, final boolean crlDistributionPointOnCrlCritical, final boolean includeInHealthCheck,
210     		final boolean _doEnforceUniquePublicKeys, final boolean _doEnforceUniqueDistinguishedName, final boolean _doEnforceUniqueSubjectDNSerialnumber,
211     		final boolean _useCertReqHistory, final boolean _useUserStorage, final boolean _useCertificateStorage, final boolean _acceptRevocationNonExistingEntry,
212             final String _cmpRaAuthSecret, final boolean keepExpiredCertsOnCRL) {
213         this.subjectdn = CertTools.stringToBCDNString(StringTools.strip(subjectDn));
214         this.caid = CertTools.stringToBCDNString(this.subjectdn).hashCode();
215         this.name = name;
216         this.status = status;
217         this.updatetime = updateTime;
218         this.encodedValidity = encodedValidity;
219         this.expiretime = expiretime;
220         this.catype = catype;
221         this.signedby = signedBy;
222         // Due to a bug in Glassfish v1 (fixed in v2), we used to have to make sure all certificates in this
223         // Array were of SUNs own provider, using CertTools.SYSTEM_SECURITY_PROVIDER.
224         // As of EJBCA 3.9.3 we decided that we don't have to support Glassfish v1 anymore.
225 		try {
226 			if (certificatechain != null) {
227 		        X509Certificate[] certs = certificatechain.toArray(new X509Certificate[certificatechain.size()]);
228                 List<Certificate> list = CertTools.getCertCollectionFromArray(certs, null);
229 		        setCertificateChain(list);
230 			} else {
231 			    setCertificateChain(null);
232 			}
233 		} catch (CertificateException e) {
234 			throw new IllegalArgumentException(e);
235 		} catch (NoSuchProviderException e) {
236 			throw new IllegalArgumentException(e);
237 		}
238         this.catoken = catoken;
239         this.description = description;
240         setRevocationReason(revocationReason);
241         this.revocationDate = revocationDate;
242         this.policies = policies;
243         this.crlperiod = crlperiod;
244         this.crlIssueInterval = crlIssueInterval;
245         this.crlOverlapTime = crlOverlapTime;
246         this.deltacrlperiod = deltacrlperiod;
247         this.crlpublishers = crlpublishers;
248         this.validators = keyValidators;
249         this.useauthoritykeyidentifier = useauthoritykeyidentifier;
250         this.authoritykeyidentifiercritical = authoritykeyidentifiercritical;
251         this.usecrlnumber = usecrlnumber;
252         this.crlnumbercritical = crlnumbercritical;
253         this.defaultcrldistpoint = defaultcrldistpoint;
254         this.defaultcrlissuer = defaultcrlissuer;
255         this.defaultocsplocator = defaultocspservicelocator;
256         this.cadefinedfreshestcrl = cadefinedfreshestcrl;
257         this.finishuser = finishuser;
258         this.subjectaltname = subjectaltname;
259         this.certificateprofileid = certificateprofileid;
260         this.defaultCertificateProfileId = defaultCertprofileId;
261         this.extendedcaserviceinfos = extendedcaserviceinfos;
262         this.useUTF8PolicyText = useUTF8PolicyText;
263         setApprovals(approvals);
264         this.usePrintableStringSubjectDN = usePrintableStringSubjectDN;
265         this.useLdapDNOrder = useLdapDnOrder;
266         this.useCrlDistributionPointOnCrl = useCrlDistributionPointOnCrl;
267         this.crlDistributionPointOnCrlCritical = crlDistributionPointOnCrlCritical;
268         this.includeInHealthCheck = includeInHealthCheck;
269         this.doEnforceUniquePublicKeys = _doEnforceUniquePublicKeys;
270         this.doEnforceUniqueDistinguishedName = _doEnforceUniqueDistinguishedName;
271         this.doEnforceUniqueSubjectDNSerialnumber = _doEnforceUniqueSubjectDNSerialnumber;
272         this.useCertReqHistory = _useCertReqHistory;
273         this.useUserStorage = _useUserStorage;
274         this.useCertificateStorage = _useCertificateStorage;
275         this.acceptRevocationNonExistingEntry = _acceptRevocationNonExistingEntry;
276         setCmpRaAuthSecret(_cmpRaAuthSecret);
277         this.keepExpiredCertsOnCRL = keepExpiredCertsOnCRL;
278         this.authorityInformationAccess = authorityInformationAccess;
279         this.certificateAiaDefaultCaIssuerUri = certificateAiaDefaultCaIssuerUri;
280         this.nameConstraintsPermitted = nameConstraintsPermitted;
281         this.nameConstraintsExcluded = nameConstraintsExcluded;
282         this.useNoConflictCertificateData = useNoConflictCertificateData;
283         this.caSerialNumberOctetSize = caSerialNumberOctetSize;
284     }
