59    /**
60     * Issuing certificates with 'notAfter' greater than this value throws an
61     * exception.
62     */
63    private static Date tooLateExpireDate;
64  
65    static {
66      final String value = CesecoreConfiguration.getCaTooLateExpireDate();
67      try {
68        tooLateExpireDate = ValidityDate.parseCaLatestValidDateTime(value);
69      } catch (Exception e) {
70        final String newValue =
71            ValidityDate.formatAsISO8601(
72                new Date(Long.MAX_VALUE), ValidityDate.TIMEZONE_SERVER);
73        tooLateExpireDate = ValidityDate.parseCaLatestValidDateTime(newValue);
74        LOG.warn(
75            "cesecore.properties ca.toolateexpiredate '"
76                + value
77                + "' could not be parsed Using default value '"
78                + newValue
79                + "'.",
80            e);
81      }
82    }

84    /**
85     * Validity offset in milliseconds (offset for the 'notBefore' value) The
86     * default start date is set 10 minutes back to avoid some problems with
87     * unsynchronized clocks.
88     */
89    private static long defaultValidityOffset;
90  
91    static {
92      final String value = CesecoreConfiguration.getCertificateValidityOffset();
93      try {
94        defaultValidityOffset =
95            SimpleTime.getSecondsFormat().parseMillis(value);
96      } catch (Exception e) {
97        // Use old value for compatibility reasons!
98        defaultValidityOffset = -10L * SECS_PER_MIN * MS_PER_SEC;
99        LOG.warn(
100           "cesecore.properties certificate.validityoffset '"
101               + value
102               + "' could not be parsed as relative time string. Using default"
103               + " value '-10m' = -60000ms",
104           e);
105     }
106   }

174   /**
175    * Constructor that injects the reference point (now). This constructor mainly
176    * is used for unit testing.
177    *
178    * @param now Now
179    * @param subject Subject
180    * @param certProfile Profila
181    * @param notBefore Start date
182    * @param notAfter End date
183    * @param cacert CA Certificate
184    * @param isRootCA bool
185    * @param isLinkCertificate bool
186    * @throws IllegalValidityException on fail
187    */
188   public CertificateValidity(
189       final Date now,
190       final EndEntityInformation subject,
191       final CertificateProfile certProfile,
192       final Date notBefore,
193       final Date notAfter,
194       final Certificate cacert,
195       final boolean isRootCA,
196       final boolean isLinkCertificate)
197       throws IllegalValidityException {
198     if (LOG.isDebugEnabled()) {
199       LOG.debug("Requested notBefore: " + notBefore);
200       LOG.debug("Requested notAfter: " + notAfter);
201       if (null != subject.getExtendedInformation()) {
202         LOG.debug(
203             "End entity extended information 'notBefore': "
204                 + subject
205                     .getExtendedInformation()
206                     .getCustomData(ExtendedInformation.CUSTOM_STARTTIME));
207       }
208       if (null != subject.getExtendedInformation()) {
209         LOG.debug(
210             "End entity extended information 'notAfter': "
211                 + subject
212                     .getExtendedInformation()
213                     .getCustomData(ExtendedInformation.CUSTOM_ENDTIME));
214       }
215       LOG.debug("Default validty offset: " + defaultValidityOffset);
216       LOG.debug(
217           "Certificate profile validty: " + certProfile.getEncodedValidity());
218       LOG.debug(
219           "Certificate profile use validty offset: "
220               + certProfile.getUseCertificateValidityOffset());
221       LOG.debug(
222           "Certificate profile validty offset: "
223               + certProfile.getCertificateValidityOffset());
224       LOG.debug(
225           "Certificate profile use expiration restrictions for weekdays: "
226               + certProfile.getUseExpirationRestrictionForWeekdays());
227       LOG.debug(
228           "Certificate profile expiration restrictions weekdays: "
229               + Arrays.toString(
230                   certProfile.getExpirationRestrictionWeekdays()));
231       LOG.debug(
232           "Certificate profile expiration restrictions for weekdays before: "
233               + certProfile.getExpirationRestrictionForWeekdaysExpireBefore());
234     }
235     if (tooLateExpireDate == null) {
236       throw new IllegalStateException(
237           "ca.toolateexpiredate in cesecore.properties is not a valid date.");
238     }
239 
240     // ECA-3554 add the offset
241 
242     Date newNow = getNowWithOffset(now, certProfile);
243     if (LOG.isDebugEnabled()) {
244       LOG.debug("Using new start time including offset: " + newNow);
245     }
246 
247     // Find out what start and end time to actually use..
248     if (certProfile.getAllowValidityOverride()) {
249       // First Priority has information supplied in Extended information object.
250       // This allows RA-users to set the time-span.
251       // Second Priority has the information supplied in the method arguments
252       firstDate = getExtendedInformationStartTime(newNow, subject);
253       if (firstDate == null) {
254         firstDate = notBefore;
255       }
256       lastDate = getExtendedInformationEndTime(newNow, subject);
257       if (lastDate == null) {
258         lastDate = notAfter;
259       }
260       if (LOG.isDebugEnabled()) {
261         LOG.debug("Allow validity override, notBefore: " + firstDate);
262         LOG.debug("Allow validity override, notAfter: " + lastDate);
263       }
264     }
265     // Third priority: If nothing could be set by external information have the
266     // default  3 is default values
267     if (firstDate == null) {
268       firstDate = newNow;
269     }
270     Date certProfileLastDate =
271         new Date(getCertificateProfileValidtyEndDate(certProfile, firstDate));
272     // Limit validity: ECA-5330 Apply expiration restriction for weekdays
273     if (certProfile.getUseExpirationRestrictionForWeekdays()
274         && isRelativeTime(certProfile.getEncodedValidity())) {
275       LOG.info(
276           "Applying expiration restrictions for weekdays: "
277               + Arrays.asList(certProfile.getExpirationRestrictionWeekdays()));
278       try {
279         final Date newDate =
280             ValidityDate.applyExpirationRestrictionForWeekdays(
281                 certProfileLastDate,
282                 certProfile.getExpirationRestrictionWeekdays(),
283                 certProfile.getExpirationRestrictionForWeekdaysExpireBefore());
284         if (!firstDate.before(newDate)) {
285           LOG.warn(
286               "Expiration restriction of certificate profile could not be"
287                   + " applied because it's before start date!");
288         } else if (!tooLateExpireDate.after(newDate)) {
289           LOG.warn(
290               "Expiration restriction of certificate profile could not be"
291                   + " applied because it's after latest possible end date!");
292         } else {
293           certProfileLastDate = newDate;
294         }
295       } catch (Exception e) {
296         LOG.warn(
297             "Expiration restriction of certificate profile could not be"
298                 + " applied!");
299       }
300     }
301     // If it is a link certificate that we create, we use the old ca's expire
302     // date, as requested, as link certificate expire date
303     if (isLinkCertificate) {
304       lastDate = notAfter;
305     }
306     if (lastDate == null) {
307       lastDate = certProfileLastDate;
308     }
309     // Limit validity: Do not allow last date to be before first date
310     if (lastDate.before(firstDate)) {
311       LOG.info(
312           INTRES.getLocalizedMessage(
313               "createcert.errorinvalidcausality", firstDate, lastDate));
314       Date tmp = lastDate;
315       lastDate = firstDate;
316       firstDate = tmp;
317     }
318     // Limit validity: We do not allow a certificate to be valid before the
319     // current date, i.e. not back dated start dates
320     // Unless allowValidityOverride is set, then we allow everything
321     // So this check is probably completely unneeded and can never be true
322     if (firstDate.before(newNow) && !certProfile.getAllowValidityOverride()) {
323       LOG.error(
324           INTRES.getLocalizedMessage(
325               "createcert.errorbeforecurrentdate",
326               firstDate,
327               subject.getUsername()));
328       firstDate = newNow;
329       // Update valid length from the profile since the starting point has
330       // changed
331       certProfileLastDate =
332           new Date(getCertificateProfileValidtyEndDate(certProfile, firstDate));
333       // Update lastDate if we use maximum validity
334     }
335     // Limit validity: We do not allow a certificate to be valid after the the
336     // validity of the certificate profile
337     if (lastDate.after(certProfileLastDate)) {
338       LOG.info(
339           INTRES.getLocalizedMessage(
340               "createcert.errorbeyondmaxvalidity",
341               lastDate,
342               subject.getUsername(),
343               certProfileLastDate));
344       lastDate = certProfileLastDate;
345     }
346     // Limit validity: We do not allow a certificate to be valid after the the
347     // validity of the CA (unless it's RootCA during renewal)
348     if (cacert != null && !isRootCA) {
349       final Date caNotAfter = CertTools.getNotAfter(cacert);
350       if (lastDate.after(caNotAfter)) {
351         LOG.info(
352             INTRES.getLocalizedMessage(
353                 "createcert.limitingvalidity",
354                 lastDate.toString(),
355                 caNotAfter));
356         lastDate = caNotAfter;
357       }
358     }
359     // Limit validity: We do not allow a certificate to be valid before the the
360     // CA becomes valid (unless it's RootCA during renewal)
361     if (cacert != null && !isRootCA) {
362       final Date caNotBefore = CertTools.getNotBefore(cacert);
363       if (firstDate.before(caNotBefore)) {
364         LOG.info(
365             INTRES.getLocalizedMessage(
366                 "createcert.limitingvaliditystart",
367                 firstDate.toString(),
368                 caNotBefore));
369         firstDate = caNotBefore;
370       }
371     }
372     if (!lastDate.before(CertificateValidity.tooLateExpireDate)) {
373       String msg =
374           INTRES.getLocalizedMessage(
375               "createcert.errorbeyondtoolateexpiredate",
376               lastDate.toString(),
377               CertificateValidity.tooLateExpireDate.toString());
378       LOG.info(msg);
379       throw new IllegalValidityException(msg);
380     }
381   }

401   /**
402    * Gets the validity end date for the certificate using the certificate
403    * profiles encoded validity.
404    *
405    * @param profile the certificate profile
406    * @param aFirstDate the start time.
407    * @return the encoded validity.
408    */
409   @SuppressWarnings("deprecation")
410   private long getCertificateProfileValidtyEndDate(
411       final CertificateProfile profile, final  Date aFirstDate) {
412     final String encodedValidity = profile.getEncodedValidity();
413     Date date = null;
414     if (StringUtils.isNotBlank(encodedValidity)) {
415       date = ValidityDate.getDate(encodedValidity, aFirstDate);
416     } else {
417       date =
418           ValidityDate.getDateBeforeVersion661(
419               profile.getValidity(), aFirstDate);
420     }
421     return date.getTime();
422   }

424   /**
425    * Offsets the certificates 'notBefore' (reference point) with the global
426    * offset or the offset of the certificate profile.
427    *
428    * @param now the reference point
429    * @param profile the certificate profile
430    * @return the offset reference point
431    */
432   private Date getNowWithOffset(
433       final Date now, final CertificateProfile profile) {
434     Date result = null;
435     if (profile.getUseCertificateValidityOffset()) {
436       final String offset = profile.getCertificateValidityOffset();
437       try {
438         result = new Date(now.getTime() + SimpleTime.parseMillies(offset));
439         if (LOG.isDebugEnabled()) {
440           LOG.debug("Using validity offset by certificate profile: " + offset);
441         }
442       } catch (NumberFormatException e) {
443         LOG.warn(
444             "Could not parse certificate validity offset "
445                 + offset
446                 + "; using default "
447                 + defaultValidityOffset);
448       }
449     } else {
450       result = new Date(now.getTime() + defaultValidityOffset);
451       if (LOG.isDebugEnabled()) {
452         LOG.debug(
453             "Using validity offset by cesecore.properties: "
454                 + SimpleTime.toString(
455                     defaultValidityOffset, SimpleTime.TYPE_DAYS));
456       }
457     }
458     return result;
459   }

462    * Gets the start time by the extended entity information.
463    *
464    * @param now the reference point.
465    * @param subject the end entity information.
466    * @return Start time
467    */
468   private Date getExtendedInformationStartTime(
469       final Date now, final EndEntityInformation subject) {
470     Date result = null;
471     final ExtendedInformation extendedInformation =
472         subject.getExtendedInformation();
473     if (extendedInformation != null) {
474       result =
475           parseExtendedInformationEncodedValidity(
476               now,
477               extendedInformation.getCustomData(
478                   ExtendedInformation.CUSTOM_STARTTIME));
479     }
480     if (LOG.isDebugEnabled()) {
481       LOG.debug("Using ExtendedInformationStartTime: " + result);
482     }
483     return result;

524   /**
525    *
526    * @param cert Certificate
527    * @param checkDate Date
528    * @throws CAOfflineException Fail
529    */
530   public static void checkPrivateKeyUsagePeriod(
531       final X509Certificate cert, final Date checkDate)
532       throws CAOfflineException {
533     if (cert != null) {
534       final PrivateKeyUsagePeriod pku =
535           CertTools.getPrivateKeyUsagePeriod(cert);
536       if (pku != null) {
537         final ASN1GeneralizedTime notBefore = pku.getNotBefore();
538         final Date pkuNotBefore;
539         final Date pkuNotAfter;
540         try {
541           if (notBefore == null) {
542             pkuNotBefore = null;
543           } else {
544             pkuNotBefore = notBefore.getDate();
545           }
546           if (LOG.isDebugEnabled()) {
547             LOG.debug("PrivateKeyUsagePeriod.notBefore is " + pkuNotBefore);
548           }
549           if (pkuNotBefore != null && checkDate.before(pkuNotBefore)) {
550             final String msg =
551                 INTRES.getLocalizedMessage(
552                     "createcert.privatekeyusagenotvalid",
553                     pkuNotBefore.toString(),
554                     cert.getSubjectDN().toString());
555             if (LOG.isDebugEnabled()) {
556               LOG.debug(msg);
557             }
558             throw new CAOfflineException(msg);
559           }
560           final ASN1GeneralizedTime notAfter = pku.getNotAfter();
561 
562           if (notAfter == null) {
563             pkuNotAfter = null;
564           } else {
565             pkuNotAfter = notAfter.getDate();
566           }
567         } catch (ParseException e) {
568           throw new IllegalStateException("Could not parse dates.", e);
569         }
570         if (LOG.isDebugEnabled()) {
571           LOG.debug("PrivateKeyUsagePeriod.notAfter is " + pkuNotAfter);
572         }
573         if (pkuNotAfter != null && checkDate.after(pkuNotAfter)) {
574           final String msg =
575               INTRES.getLocalizedMessage(
576                   "createcert.privatekeyusageexpired",
577                   pkuNotAfter.toString(),
578                   cert.getSubjectDN().toString());
579           if (LOG.isDebugEnabled()) {
580             LOG.debug(msg);
581           }
582           throw new CAOfflineException(msg);
583         }
584       } else if (LOG.isDebugEnabled()) {
585         LOG.debug("No PrivateKeyUsagePeriod available in certificate.");
586       }
587     } else if (LOG.isDebugEnabled()) {
588       LOG.debug(
589           "No CA certificate available, not checking PrivateKeyUsagePeriod.");
590     }
591   }

593   /**
594    * Checks if the encoded validity is an ISO8601 date or a relative time.
595    *
596    * @param encodedValidity the validity
597    * @return Boolean.TRUE if it is a relative time, Boolean.FALSE if it is an
598    *     ISO8601 date, otherwise NULL. @See {@link
599    *     org.cesecore.util.ValidityDate ValidityDate} @See {@link
600    *     org.cesecore.util.SimpleTime SimpleTime}
601    */
602   private static Boolean isRelativeTime(final String encodedValidity) {
603     try {
604       // Try the most likely setting first, for fail-fast
605       SimpleTime.parseMillies(encodedValidity);
606       return Boolean.TRUE;
607     } catch (NumberFormatException nfe) {
608       // NOOP
609     }
610     try {
611       ValidityDate.parseAsIso8601(encodedValidity);
612       return Boolean.FALSE;
613     } catch (ParseException e) {
614       return null;
615     }
616   }

618   /**
619    * Parses the entity extended information start and end time format and
620    * offsets it with the reference point.
621    *
622    * @param now the reference point
623    * @param timeString the value in form of 'days:hours:minutes'
624    * @return the parse value offset with now (reference point).
625    */
626   private static Date parseExtendedInformationEncodedValidity(
627       final Date now, final String timeString) {
628     Date result = null;
629     if (timeString != null) {
630       if (timeString.matches("^\\d+:\\d?\\d:\\d?\\d$")) {
631         final String[] endTimeArray = timeString.split(":");
632         long relative =
633             (Long.parseLong(endTimeArray[0]) * HOURS_PER_DAY * MINS_PER_HOUR
634                     + Long.parseLong(endTimeArray[1]) * MINS_PER_HOUR
635                     + Long.parseLong(endTimeArray[2]))
636                 * SECS_PER_MIN
637                 * MS_PER_SEC;
638         result = new Date(now.getTime() + relative);
639       } else {
640         try {
641           // Try parsing data as "yyyy-MM-dd HH:mm" assuming UTC
642           result = ValidityDate.parseAsUTC(timeString);
643         } catch (ParseException e) {
644           LOG.error(
645               INTRES.getLocalizedMessage(
646                   "createcert.errorinvalidstarttime", timeString));
647         }
648       }
649       if ((LOG.isDebugEnabled())) {
650         LOG.debug("Time string by end entity extended Information: " + result);
651       }
652     }
653     return result;
654   }
