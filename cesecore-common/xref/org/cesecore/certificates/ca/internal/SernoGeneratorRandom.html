28  /**
29   * Implements a singleton serial number generator using SecureRandom. This
30   * generator generates random 8 octec (64 bits) serial numbers.
31   *
32   * <p>RFC3280 defines serialNumber be positive INTEGER, and X.690 defines
33   * INTEGER consist of one or more octets. X.690 also defines as follows:
34   *
35   * <p>If the contents octets of an integer value encoding consist of more than
36   * one octet, then the bits of the first octet and bit 8 of the second octet: a)
37   * shall not all be ones; and b) shall not all be zero.
38   *
39   * <p>Therefore, minimum 8 octets value is 0080000000000000 and maximum value is
40   * 7FFFFFFFFFFFFFFF."
41   *
42   * <p>Therefore, minimum 4 octets value is 00800000 and maximum value is
43   * 7FFFFFFF."
44   *
45   * <p>X.690:
46   *
47   * <p>8.3 Encoding of an integer value 8.3.1 The encoding of an integer value
48   * shall be primitive. The contents octets shall consist of one or more octets.
49   * 8.3.2 If the contents octets of an integer value encoding consist of more
50   * than one octet, then the bits of the first octet and bit 8 of the second
51   * octet: a) shall not all be ones; and b) shall not all be zero. NOTE – These
52   * rules ensure that an integer value is always encoded in the smallest possible
53   * number of octets. 8.3.3 The contents octets shall be a two's complement
54   * binary number equal to the integer value, and consisting of bits 8 to 1 of
55   * the first octet, followed by bits 8 to 1 of the second octet, followed by
56   * bits 8 to 1 of each octet in turn up to and including the last octet of the
57   * contents octets. NOTE – The value of a two's complement binary number is
58   * derived by numbering the bits in the contents octets, starting with bit 1 of
59   * the last octet as bit zero and ending the numbering with bit 8 of the first
60   * octet. Each bit is assigned a numerical value of 2N, where N is its position
61   * in the above numbering sequence. The value of the two's complement binary
62   * number is obtained by summing the numerical values assigned to each bit for
63   * those bits which are set to one, excluding bit 8 of the first octet, and then
64   * reducing this value by the numerical value assigned to bit 8 of the first
65   * octet if that bit is set to one.
66   *
67   * @version $Id: SernoGeneratorRandom.java 31966 2019-03-25 10:19:57Z anatom $
68   */
69  public class SernoGeneratorRandom implements SernoGenerator {
70    /** Log4j instance. */
71    private static final Logger LOG =
72        Logger.getLogger(SernoGeneratorRandom.class);
73    /** Internal localization of logs and errors. */
74    private static final InternalResources INTRES =
75        InternalResources.getInstance();
76  
77    /**
78     * RFC5280, section 4.1.2.2, specifies using max 20 octets for serial number.
79     */
80    private static final int SERNO_MAX_LENGTH = 20;
81  
82    /**
83     * random generator algorithm, defaults to FIPS approve SHA1PRNG in
84     * constructor The algorithm is specified globally in
85     * CesecoreConfiguration.getCaSerialNumberAlgorithm().
86     */
87    private String algorithm;
88  
89    /** number of bytes to generate, fixed size serial numbers. */
90    private int noOctets;
91  
92    /** random generator. */
93    private SecureRandom random;
94  
95    /**
96     * A registry of Singleton instances, to handle multiple octet sizes
97     * simultaneously.
98     */
99    private static Map<Integer, SernoGeneratorRandom> instances = new HashMap<>();
100   /**
101    * Creates (if needed) a serial number generator and returns the object.
102    *
103    * @param noOctets size
104    * @return An instance of the serial number generator.
105    */
106   public static synchronized SernoGenerator instance(final Integer noOctets) {
107     SernoGeneratorRandom instance = instances.get(noOctets);
108     if (instance == null) {
109       instance = new SernoGeneratorRandom(noOctets);
110       instances.put(noOctets, instance);
111     }
112     return instance;
113   }
114 
115   /**
116    * DO NOT USE: Protected only to do testing of this implementation use {@link
117    * #instance(Integer)} instead.
118    *
119    * @param aNoOctets size
120    */
121   protected SernoGeneratorRandom(final Integer aNoOctets) {
122     if (LOG.isTraceEnabled()) {
123       LOG.trace(">SernoGenerator()");
124     }
125     this.algorithm = CesecoreConfiguration.getCaSerialNumberAlgorithm();
126     if (this.algorithm == null) {
127       this.algorithm = "SHA1PRNG";
128     }
129     if ((aNoOctets > SERNO_MAX_LENGTH
130         || aNoOctets < 0)) { // We allow 0 octets for testing
131       throw new IllegalArgumentException(
132           "ca.serialnumberoctetsize must be between 0 and "
133               + SERNO_MAX_LENGTH
134               + " bytes for this serial number generator.");
135     }
136     this.noOctets = aNoOctets;
137     init();
138     if (LOG.isTraceEnabled()) {
139       LOG.trace("<SernoGenerator()");
140     }
141   }
142 
143   private void init() {
144     // Init random number generator for random serial numbers.
145     // SecureRandom provides a cryptographically strong random number generator
146     // (CSPRNG).
147     try {
148       // Use a specified algorithm if ca.rngalgorithm is provided and it's not
149       // set to default
150       if (!StringUtils.isEmpty(algorithm)
151           && !StringUtils.containsIgnoreCase(algorithm, "default")) {
152         random = SecureRandom.getInstance(algorithm);
153         LOG.info("Using " + algorithm + " serialNumber RNG algorithm.");
154       } else if (!StringUtils.isEmpty(algorithm)
155           && StringUtils.equalsIgnoreCase(algorithm, "defaultstrong")) {
156         // If defaultstrong is specified and we use >=JDK8 try the
157         // getInstanceStrong to get a guaranteed strong random number generator.
158         // Note that this may give you a generator that takes >30 seconds to
159         // create a single random number.
160         // On JDK8/Linux this gives you a NativePRNGBlocking, while
161         // SecureRandom.getInstance() gives a NativePRNG.
162         try {
163           final Method methodGetInstanceStrong =
164               SecureRandom.class.getDeclaredMethod("getInstanceStrong");
165           random = (SecureRandom) methodGetInstanceStrong.invoke(null);
166           LOG.info(
167               "Using SecureRandom.getInstanceStrong() with "
168                   + random.getAlgorithm()
169                   + " for serialNumber RNG algorithm.");
170         } catch (NoSuchMethodException
171             | SecurityException
172             | IllegalAccessException
173             | IllegalArgumentException
174             | InvocationTargetException e) {
175           throw new IllegalStateException(
176               "SecureRandom.getInstanceStrong() is not available or failed"
177                   + " invocation. (This method was added in Java 8.)");
178         }
179       } else if (!StringUtils.isEmpty(algorithm)
180           && StringUtils.equalsIgnoreCase(algorithm, "default")) {
181         // We entered "default" so let's use a good default SecureRandom this
182         // should be good enough for just about everyone (on Linux at least)
183         // On Linux the default Java implementation uses the (secure)
184         // /dev/(u)random, but on windows something else
185         // On JDK8/Linux this gives you a NativePRNG, while
186         // SecureRandom.getInstanceStrong() gives a NativePRNGBlocking.
187         random = new SecureRandom();
188         LOG.info(
189             "Using default "
190                 + random.getAlgorithm()
191                 + " serialNumber RNG algorithm.");
192       }
193     } catch (NoSuchAlgorithmException e) {
194       // This state is unrecoverable, and since algorithm is set in
195       // configuration requires a redeploy to handle
196       throw new IllegalStateException(
197           "Algorithm " + algorithm + " was not a valid algorithm.", e);
198     }
199     if (random == null) {
200       // This state is unrecoverable, and since algorithm is set in
201       // configuration requires a redeploy to handle
202       throw new IllegalStateException(
203           "Algorithm " + algorithm + " was not a valid algorithm.");
204     }
205     // Call nextBytes directly after in order to force seeding if not already
206     // done. SecureRandom typically seeds on first call.
207     random.nextBytes(new byte[0]);
208   }

210   @Override
211   public BigInteger getSerno() {
212     // This is only for testing, if size is set to 0 we will generate random
213     // number
214     // between 1 and 5, this will give collisions often...
215     if (noOctets == 0) {
216       final Random rand = new Random();
217       return BigInteger.valueOf(rand.nextInt(4) + 1); // value 1-5
218     }
219     while (true) {
220       /*
221       Note that initBitsOfEntropy are not left intact by the
222        following subsequent filtering operations:
223       - Values discarded to avoid encoding in less
224       than noOctets (including zero value).
225       - Serial numbers previously assigned to other
226       certificates (filtered later, not here).
227       So the real entropy provided for generated
228        serial numbers is always less than initBitsOfEntropy.
229        */
230       // initBitsOfEntropy is 1 less than octet size, because we always use
231       // positive integers, which in
232       // two complements representation always has the most significant bit 0,
233       // making 63 bits random
234       int initBitsOfEntropy = noOctets * 8 - 1;
235       // SecureRanom is thread safe. This will generate from (0 to
236       // 2^initBitsOfEntropy -1)
237       final BigInteger serno = new BigInteger(initBitsOfEntropy, random);
238       if (checkSernoValidity(serno)) {
239         return serno;
240       } else {
241         String msg = INTRES.getLocalizedMessage("sernogenerator.discarding");
242         LOG.info(msg);
243       }
244     }
245   }

248   /**
249    * This validates that the argument is a non-zero number to be encoded
250    * (according to X.690, "8.3 Encoding of an integer value") exactly in
251    * 'noOctets' bytes. For example, for an 8 bytes serial number it will
252    * validate that it falls within the range 0080000000000000 - 7FFFFFFFFFFFFFFF
253    * (both inclusive).
254    *
255    * @param serno serial
256    * @return bool
257    */
258   protected boolean checkSernoValidity(final BigInteger serno) {
259     return serno.compareTo(BigInteger.ZERO) != 0
260         && serno.bitLength() / 8 + 1 == noOctets;
261   }
262 
263   @Override
264   public int getNoSernoBytes() {
265     return noOctets;
266   }
267 
268   @Override
269   public void setSeed(final long seed) {
270     random.setSeed(seed);
271   }
272 
273   @Override
274   public void setAlgorithm(final String algo) throws NoSuchAlgorithmException {
275     // Since re-initialization is expensive, we only do it if we changed the
276     // algo
277     if (this.algorithm == null || !this.algorithm.equals(algo)) {
278       this.algorithm = algo;
279       // We must force re-init after choosing a new algorithm
280       this.random = null;
281       init();
282     }
283   }

285   /**
286    * Available for testing so we can compare that we actually use what we think.
287    *
288    * @return the random generator algorithm as reported by the underlying Java
289    *     random number generator.
290    */
291   protected String getAlgorithm() {
292     return random.getAlgorithm();
293   }
