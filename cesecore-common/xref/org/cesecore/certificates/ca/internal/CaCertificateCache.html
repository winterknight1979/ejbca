35  /**
36   * A cache for storing CA certificates
37   *
38   * @version $Id: CaCertificateCache.java 27419 2017-12-05 13:18:18Z anatom $
39   *
40   */
41  public enum CaCertificateCache  {
42      INSTANCE;
43      
44  	private final Logger log = Logger.getLogger(CaCertificateCache.class);
45  
46      /** Mapping from subjectDN to key in the certs HashMap. */
47      private Map<Integer, X509Certificate> certsFromSubjectDN = new HashMap<Integer, X509Certificate>();
48      /** Mapping from issuerDN to key in the certs HashMap. */
49      private Map<Integer, Set<X509Certificate>> certsFromIssuerDN = new HashMap<Integer, Set<X509Certificate>>();
50      /** Mapping from subject key identifier to key in the certs HashMap. */
51      private Map<Integer, X509Certificate> certsFromSubjectKeyIdentifier = new HashMap<Integer, X509Certificate>();
52      /** All root certificates. */
53      private Set<X509Certificate> rootCertificates = new HashSet<X509Certificate>();
54  
55  	/** Cache time counter, set and used by loadCertificates */
56  	private long certValidTo = 0;
57  
58      public X509Certificate findLatestBySubjectDN(final HashID id) {
59          final X509Certificate ret = certsFromSubjectDN.get(id.getKey());
60          if (ret==null && log.isDebugEnabled()) {
61              log.debug("Certificate not found from SubjectDN HashId in certsFromSubjectDN map. HashID=" + id.getB64());
62          }
63          return ret;
64  	}
65  
66  	public X509Certificate[] findLatestByIssuerDN(final HashID id) {	    
67          final Set<X509Certificate> sCert = certsFromIssuerDN.get(id.getKey());
68          if (sCert == null || sCert.isEmpty()) {
69              if (log.isDebugEnabled()) {
70                  log.debug("Certificate not found from IssuerDN HashId in certsFromIssuerDN map. HashID=" + id.getB64());
71              }
72              return null;
73          }
74          return sCert.toArray(new X509Certificate[sCert.size()]);
75  
76      }
77  
78      public X509Certificate[] getRootCertificates() {
79          return rootCertificates.toArray(new X509Certificate[0]);
80      }
81  
82      public X509Certificate findBySubjectKeyIdentifier(final HashID id) {
83          final X509Certificate ret = certsFromSubjectKeyIdentifier.get(id.getKey());
84          if (ret==null && log.isDebugEnabled()) {
85              log.debug("Certificate not found from SubjectKeyIdentifier HashId in certsFromSubjectKeyIdentifier map. HashID=" + id.getB64());
86          }
87          return ret;
88      }
89  
90      public boolean isCacheExpired() {
91          return certValidTo < System.currentTimeMillis();
92      }
93  
94  	/** Loads CA certificates but holds a cache so it's reloaded only every five minutes (configurable).
95  	 *
96  	 * We keep this method as synchronized, it should not take more than a few microseconds to complete if the cache does not have
97  	 * to be reloaded. If the cache must be reloaded, we must wait for it anyway to not have ConcurrentModificationException.
98  	 * We also only want one single thread to do the rebuilding.
99  	 * @param certs Certificates
100 	 * 
101 	 */
102     public synchronized void loadCertificates(final Collection<Certificate> certs) {
103         if (log.isDebugEnabled()) {
104             log.debug("Loaded " + (certs == null ? "0" : Integer.toString(certs.size())) + " ca certificates");
105         }
106 
107         Map<Integer, X509Certificate> newCertsFromSubjectDN = new HashMap<Integer, X509Certificate>();
108         Map<Integer, Set<X509Certificate>> newCertsFromIssuerDN = new HashMap<Integer, Set<X509Certificate>>();
109         Map<Integer, X509Certificate> newCertsFromSubjectKeyIdentifier = new HashMap<Integer, X509Certificate>();
110         Set<X509Certificate> newRootCertificates = new HashSet<X509Certificate>();
111         for (final Certificate tmp : certs) {
112             if (!(tmp instanceof X509Certificate)) {
113                 log.debug("Not adding CA certificate of type: " + tmp.getType());
114                 continue;
115             }
116             final X509Certificate cert = (X509Certificate) tmp;
117             try { // test if certificate is OK. we have experienced that BC could decode a certificate that later on could not be used.
118                 final Integer key = HashID.getFromKeyID(cert).getKey();
119                 final X509Certificate pastCert = newCertsFromSubjectKeyIdentifier.get(key);
120                 // Add the entry if it's the first, or if it is more recent than the one existing (in that case replace it)
121                 if ( pastCert == null || (pastCert != null && CertTools.getNotBefore(cert).after(CertTools.getNotBefore(pastCert))) ) {
122                     newCertsFromSubjectKeyIdentifier.put(key, cert);
123                 }
124             } catch (Throwable t) { // NOPMD: catch all to not break with an error here.
125                 if (log.isDebugEnabled()) {
126                     final StringWriter sw = new StringWriter();
127                     final PrintWriter pw = new PrintWriter(sw);
128                     pw.println("Erroneous certificate fetched from database.");
129                     pw.println("The public key can not be extracted from the certificate.");
130                     pw.println("Here follows a base64 encoding of the certificate:");
131                     try {
132                         final String b64encoded = new String(Base64.encode(cert.getEncoded()));
133                         pw.println(CertTools.BEGIN_CERTIFICATE);
134                         pw.println(b64encoded);
135                         pw.println(CertTools.END_CERTIFICATE);
136                     } catch (CertificateEncodingException e) {
137                         pw.println("Not possible to encode certificate.");
138                     }
139                     pw.flush();
140                     log.debug(sw.toString());
141                 }
142                 continue;
143             }
144             final Integer subjectDNKey = HashID.getFromSubjectDN(cert).getKey();
145             // Check if we already have a certificate from this issuer in the HashMap.
146             // We only want to store the latest cert from each issuer in this map
147             final X509Certificate pastCert = newCertsFromSubjectDN.get(subjectDNKey);
148             final boolean isLatest;
149             if (pastCert != null) {
150                 if (CertTools.getNotBefore(cert).after(CertTools.getNotBefore(pastCert))) {
151                     isLatest = true;
152                 } else {
153                     isLatest = false;
154                 }
155             } else {
156                 isLatest = true;
157             }
158             if (isLatest) {
159                 newCertsFromSubjectDN.put(subjectDNKey, cert);
160                 final Integer issuerDNKey = HashID.getFromIssuerDN(cert).getKey();
161                 if (!issuerDNKey.equals(subjectDNKey)) { // don't add roots to themselves
162                     Set<X509Certificate> sIssuer = newCertsFromIssuerDN.get(issuerDNKey);
163                     if (sIssuer == null) {
164                         sIssuer = new HashSet<X509Certificate>();
165                         newCertsFromIssuerDN.put(issuerDNKey, sIssuer);
166                     }
167                     sIssuer.add(cert);
168                     sIssuer.remove(pastCert);
169                 } else {
170                     newRootCertificates.add(cert);
171                     newRootCertificates.remove(pastCert);
172                 }
173             }
174         }
175         // Log what we have stored in the cache
176         if (log.isDebugEnabled()) {
177             final StringWriter sw = new StringWriter();
178             final PrintWriter pw = new PrintWriter(sw, true);
179             pw.println("Found the following CA certificates :");
180             for (Entry<Integer, X509Certificate> key : newCertsFromSubjectKeyIdentifier.entrySet()) {
181                 final Certificate cert = key.getValue();
182                 pw.print(CertTools.getSubjectDN(cert));
183                 pw.print(',');
184                 pw.println(CertTools.getSerialNumberAsString(cert));
185             }
186             log.debug(sw);
187         }
188         //Replace the old caches
189         certsFromSubjectKeyIdentifier = newCertsFromSubjectKeyIdentifier;
190         certsFromIssuerDN = newCertsFromIssuerDN;
191         certsFromSubjectDN = newCertsFromSubjectDN;
192         rootCertificates = newRootCertificates;
193         certValidTo = System.currentTimeMillis() + OcspConfiguration.getSigningCertsValidTimeInMilliseconds();
194     }
