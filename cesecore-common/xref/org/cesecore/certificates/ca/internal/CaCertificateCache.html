18  import java.security.cert.Certificate;
19  import java.security.cert.CertificateEncodingException;
20  import java.security.cert.X509Certificate;
21  import java.util.Collection;
22  import java.util.HashMap;
23  import java.util.HashSet;
24  import java.util.Map;
25  import java.util.Map.Entry;
26  import java.util.Set;
27  import org.apache.log4j.Logger;
28  import org.cesecore.certificates.certificate.HashID;
29  import org.cesecore.config.OcspConfiguration;
30  import org.cesecore.util.Base64;

33  /**
34   * A cache for storing CA certificates.
35   *
36   * @version $Id: CaCertificateCache.java 27419 2017-12-05 13:18:18Z anatom $
37   */
38  public enum CaCertificateCache {
39    /** Singleton instance. */
40    INSTANCE;
41  
42      /** Logger. */
43    private final Logger log = Logger.getLogger(CaCertificateCache.class);
44  
45    /** Mapping from subjectDN to key in the certs HashMap. */
46    private Map<Integer, X509Certificate> certsFromSubjectDN =
47        new HashMap<Integer, X509Certificate>();
48    /** Mapping from issuerDN to key in the certs HashMap. */
49    private Map<Integer, Set<X509Certificate>> certsFromIssuerDN =
50        new HashMap<Integer, Set<X509Certificate>>();
51    /** Mapping from subject key identifier to key in the certs HashMap. */
52    private Map<Integer, X509Certificate> certsFromSubjectKeyIdentifier =
53        new HashMap<Integer, X509Certificate>();
54    /** All root certificates. */
55    private Set<X509Certificate> rootCertificates =
56        new HashSet<X509Certificate>();
57  
58    /** Cache time counter, set and used by loadCertificates. */
59    private long certValidTo = 0;
60    /**
61     *
62     * @param id ID
63     * @return Cert
64     */
65    public X509Certificate findLatestBySubjectDN(final HashID id) {
66      final X509Certificate ret = certsFromSubjectDN.get(id.getKey());
67      if (ret == null && log.isDebugEnabled()) {
68        log.debug(
69            "Certificate not found from SubjectDN HashId in certsFromSubjectDN"
70                + " map. HashID="
71                + id.getB64());
72      }
73      return ret;
74    }
75  
76    /** @param id ID
77   * @return cert */
78    public X509Certificate[] findLatestByIssuerDN(final HashID id) {
79      final Set<X509Certificate> sCert = certsFromIssuerDN.get(id.getKey());
80      if (sCert == null || sCert.isEmpty()) {
81        if (log.isDebugEnabled()) {
82          log.debug(
83              "Certificate not found from IssuerDN HashId in certsFromIssuerDN"
84                  + " map. HashID="
85                  + id.getB64());
86        }
87        return null;
88      }
89      return sCert.toArray(new X509Certificate[sCert.size()]);
90    }
91  
92    /** @return root certs */
93    public X509Certificate[] getRootCertificates() {
94      return rootCertificates.toArray(new X509Certificate[0]);
95    }
96  
97    /**
98     *  @param id ID
99   * @return cert */
100   public X509Certificate findBySubjectKeyIdentifier(final HashID id) {
101     final X509Certificate ret = certsFromSubjectKeyIdentifier.get(id.getKey());
102     if (ret == null && log.isDebugEnabled()) {
103       log.debug(
104           "Certificate not found from SubjectKeyIdentifier HashId in"
105               + " certsFromSubjectKeyIdentifier map. HashID="
106               + id.getB64());
107     }
108     return ret;
109   }
110 
111   /** @return true if cache has expired */
112   public boolean isCacheExpired() {
113     return certValidTo < System.currentTimeMillis();
114   }
115 
116   /**
117    * Loads CA certificates but holds a cache so it's reloaded only every five
118    * minutes (configurable).
119    *
120    * <p>We keep this method as synchronized, it should not take more than a few
121    * microseconds to complete if the cache does not have to be reloaded. If the
122    * cache must be reloaded, we must wait for it anyway to not have
123    * ConcurrentModificationException. We also only want one single thread to do
124    * the rebuilding.
125    *
126    * @param certs Certificates
127    */
128   public synchronized void loadCertificates(
129       final Collection<Certificate> certs) {
130     if (log.isDebugEnabled()) {
131       log.debug(
132           "Loaded "
133               + (certs == null ? "0" : Integer.toString(certs.size()))
134               + " ca certificates");
135     }
136 
137     Map<Integer, X509Certificate> newCertsFromSubjectDN =
138         new HashMap<Integer, X509Certificate>();
139     Map<Integer, Set<X509Certificate>> newCertsFromIssuerDN =
140         new HashMap<Integer, Set<X509Certificate>>();
141     Map<Integer, X509Certificate> newCertsFromSubjectKeyIdentifier =
142         new HashMap<Integer, X509Certificate>();
143     Set<X509Certificate> newRootCertificates = new HashSet<X509Certificate>();
144     for (final Certificate tmp : certs) {
145       if (!(tmp instanceof X509Certificate)) {
146         log.debug("Not adding CA certificate of type: " + tmp.getType());
147         continue;
148       }
149       final X509Certificate cert = (X509Certificate) tmp;
150       try { // test if certificate is OK. we have experienced that BC could
151             // decode a certificate that later on could not be used.
152         final Integer key = HashID.getFromKeyID(cert).getKey();
153         final X509Certificate pastCert =
154             newCertsFromSubjectKeyIdentifier.get(key);
155         // Add the entry if it's the first, or if it is more recent than the one
156         // existing (in that case replace it)
157         if (pastCert == null
158             || (pastCert != null
159                 && CertTools.getNotBefore(cert)
160                     .after(CertTools.getNotBefore(pastCert)))) {
161           newCertsFromSubjectKeyIdentifier.put(key, cert);
162         }
163       } catch (
164           Throwable t) { // NOPMD: catch all to not break with an error here.
165         if (log.isDebugEnabled()) {
166           final StringWriter sw = new StringWriter();
167           final PrintWriter pw = new PrintWriter(sw);
168           pw.println("Erroneous certificate fetched from database.");
169           pw.println(
170               "The public key can not be extracted from the certificate.");
171           pw.println("Here follows a base64 encoding of the certificate:");
172           try {
173             final String b64encoded =
174                 new String(Base64.encode(cert.getEncoded()));
175             pw.println(CertTools.BEGIN_CERTIFICATE);
176             pw.println(b64encoded);
177             pw.println(CertTools.END_CERTIFICATE);
178           } catch (CertificateEncodingException e) {
179             pw.println("Not possible to encode certificate.");
180           }
181           pw.flush();
182           log.debug(sw.toString());
183         }
184         continue;
185       }
186       final Integer subjectDNKey = HashID.getFromSubjectDN(cert).getKey();
187       // Check if we already have a certificate from this issuer in the HashMap.
188       // We only want to store the latest cert from each issuer in this map
189       final X509Certificate pastCert = newCertsFromSubjectDN.get(subjectDNKey);
190       final boolean isLatest;
191       if (pastCert != null) {
192         if (CertTools.getNotBefore(cert)
193             .after(CertTools.getNotBefore(pastCert))) {
194           isLatest = true;
195         } else {
196           isLatest = false;
197         }
198       } else {
199         isLatest = true;
200       }
201       if (isLatest) {
202         newCertsFromSubjectDN.put(subjectDNKey, cert);
203         final Integer issuerDNKey = HashID.getFromIssuerDN(cert).getKey();
204         if (!issuerDNKey.equals(
205             subjectDNKey)) { // don't add roots to themselves
206           Set<X509Certificate> sIssuer = newCertsFromIssuerDN.get(issuerDNKey);
207           if (sIssuer == null) {
208             sIssuer = new HashSet<X509Certificate>();
209             newCertsFromIssuerDN.put(issuerDNKey, sIssuer);
210           }
211           sIssuer.add(cert);
212           sIssuer.remove(pastCert);
213         } else {
214           newRootCertificates.add(cert);
215           newRootCertificates.remove(pastCert);
216         }
217       }
218     }
219     // Log what we have stored in the cache
220     if (log.isDebugEnabled()) {
221       final StringWriter sw = new StringWriter();
222       final PrintWriter pw = new PrintWriter(sw, true);
223       pw.println("Found the following CA certificates :");
224       for (Entry<Integer, X509Certificate> key
225           : newCertsFromSubjectKeyIdentifier.entrySet()) {
226         final Certificate cert = key.getValue();
227         pw.print(CertTools.getSubjectDN(cert));
228         pw.print(',');
229         pw.println(CertTools.getSerialNumberAsString(cert));
230       }
231       log.debug(sw);
232     }
233     // Replace the old caches
234     certsFromSubjectKeyIdentifier = newCertsFromSubjectKeyIdentifier;
235     certsFromIssuerDN = newCertsFromIssuerDN;
236     certsFromSubjectDN = newCertsFromSubjectDN;
237     rootCertificates = newRootCertificates;
238     certValidTo =
239         System.currentTimeMillis()
240             + OcspConfiguration.getSigningCertsValidTimeInMilliseconds();
241   }
