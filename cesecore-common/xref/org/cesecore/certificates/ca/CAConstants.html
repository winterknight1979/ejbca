13  package org.cesecore.certificates.ca;
14  
15  /**
16   * Constants for CAs.
17   *
18   * @version $Id: CAConstants.java 27567 2017-12-18 10:12:34Z bastianf $
19   */
20  public final class CAConstants {
21    /**
22     * The state of a node-local CA with a keypair which is neither expired nor
23     * revoked. An active CA should be able to create signatures unless the crypto
24     * token associated with the CA is offline, in which case healthcheck will
25     * fail. A CA stays in this state until the certificate expires or is revoked.
26     */
27    public static final int CA_ACTIVE = 1;
28    /**
29     * The state of an external CA where a CSR has been created but the signed
30     * certificate has not yet been imported into EJBCA.
31     */
32    public static final int CA_WAITING_CERTIFICATE_RESPONSE = 2;
33    /**
34     * The state of a node-local or external CA whose certificate has expired.
35     * Once a CA's certificate has expired, it will stay in this state
36     * indefinitely.
37     */
38    public static final int CA_EXPIRED = 3;
39    /** The state of a node-local CA with a certificate which has been revoked. */
40    public static final int CA_REVOKED = 4;
41    /**
42     * The state of a node-local CA which has been purposely put offline by the
43     * user, i.e a CA whose "CA Service State" is "Offline". Healthcheck will be
44     * disabled for CAs in this state.
45     */
46    public static final int CA_OFFLINE = 5;
47    /**
48     * An external CA without a private key. A CA stays in this state until the
49     * certificate expires.
50     */
51    public static final int CA_EXTERNAL = 6;
52    /**
53     * The initial state of a CA imported using Statedump. In this state, the CA
54     * does not have a keypair. The CA can advance to the
55     * CA_WAITING_CERTIFICATE_RESPONSE state if a CSR is created for the CA, or it
56     * can advance to the CA_ACTIVE state directly, if a keypair is associated
57     * with it.
58     */
59    public static final int CA_UNINITIALIZED = 7;
60  
61    /** Statuses. */
62    private static final String[] STATUS_TEXTS = {
63      "",
64      "ACTIVE",
65      "WAITINGFORCERTRESPONSE",
66      "EXPIRED",
67      "REVOKED",
68      "OFFLINE",
69      "EXTERNALCA",
70      "UNINITIALIZED"
71    };
72  
73    /** Prevents creation of new CAConstants. */
74    private CAConstants() { }
75  
76    /**
77     * Constants used in the SignSessionBean indicating the userdata defined CA
78     * should be used.
79     */
80    public static final int CAID_USEUSERDEFINED = 0;
81  
82    /**
83     * Used in profiles and service workers to make the catch all every CA instead
84     * of listing individual CAs when operating on them This is duplicated in
85     * SecConst.
86     */
87    public static final int ALLCAS = 1;
88  
89    /**
90     * Convert status code to message .
91     *
92     * @param status Code
93     * @return Message
94     */
95    public static String getStatusText(final int status) {
96      return STATUS_TEXTS[status];
97    }
98  }
