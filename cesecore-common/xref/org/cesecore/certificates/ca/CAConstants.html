13  package org.cesecore.certificates.ca;
14  
15  /**
16   * Constants for CAs.
17   *
18   * @version $Id: CAConstants.java 27567 2017-12-18 10:12:34Z bastianf $
19   */
20  public final class CAConstants {
21      /**
22       * The state of a node-local CA with a keypair which is neither expired nor revoked.
23       * An active CA should be able to create signatures unless the crypto token associated
24       * with the CA is offline, in which case healthcheck will fail. A CA stays in this
25       * state until the certificate expires or is revoked.
26       */
27      public static final int CA_ACTIVE = 1;
28      /**
29       * The state of an external CA where a CSR has been created but the signed
30       * certificate has not yet been imported into EJBCA.
31       */
32      public static final int CA_WAITING_CERTIFICATE_RESPONSE = 2;
33      /**
34       * The state of a node-local or external CA whose certificate has expired. Once
35       * a CA's certificate has expired, it will stay in this state indefinitely.
36       */
37      public static final int CA_EXPIRED = 3;
38      /**
39       * The state of a node-local CA with a certificate which has been revoked.
40       */
41      public static final int CA_REVOKED = 4;
42      /**
43       * The state of a node-local CA which has been purposely put offline by the user, i.e
44       * a CA whose "CA Service State" is "Offline". Healthcheck will be disabled for CAs
45       * in this state.
46       */
47      public static final int CA_OFFLINE = 5;
48      /**
49       * An external CA without a private key. A CA stays in this state until
50       * the certificate expires.
51       */
52      public static final int CA_EXTERNAL = 6;
53      /**
54       * The initial state of a CA imported using Statedump. In this state, the CA does not have a keypair. The CA can advance to the
55       * CA_WAITING_CERTIFICATE_RESPONSE state if a CSR is created for the CA, or it can advance to the CA_ACTIVE state directly, if
56       * a keypair is associated with it.
57       */
58      public static final int CA_UNINITIALIZED = 7;
59  
60      private static final String[] statustexts = {"", "ACTIVE", "WAITINGFORCERTRESPONSE", "EXPIRED", "REVOKED", "OFFLINE","EXTERNALCA", "UNINITIALIZED"};
61  
62      /**
63       * Prevents creation of new CAConstants
64       */
65      private CAConstants() {
66      }
67  
68      /**
69       * Constants used in the SignSessionBean indicating the userdata defined CA should be used.
70       */
71      public static final int CAID_USEUSERDEFINED = 0;
72  
73      /** Used in profiles and service workers to make the catch all every CA instead of listing individual CAs when operating on them
74       * This is duplicated in SecConst */
75      public static final int ALLCAS = 1;
76  
77      public static String getStatusText(int status) {
78          return statustexts[status];
79      }
80  
81  }
