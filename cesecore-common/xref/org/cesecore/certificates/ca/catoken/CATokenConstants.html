13  package org.cesecore.certificates.ca.catoken;
14  
15  /**
16   * Purpose mapping constants to for CryptoToken aliases and related.
17   *
18   * @version $Id: CATokenConstants.java 27312 2017-11-24 10:00:33Z anatom $
19   */
20  public final class CATokenConstants {
21  
22    /** Private constructor - do not instantiate. */
23    private CATokenConstants() { }
24  
25    /** default key used when there is no specific setting for a purpose. */
26    public static final int CAKEYPURPOSE_DEFAULT = 0;
27    /** signs certificates issued by the CA. */
28    public static final int CAKEYPURPOSE_CERTSIGN = 1;
29    /** igns CRLs issues by the CA. */
30    public static final int CAKEYPURPOSE_CRLSIGN = 2;
31    /** encrypts entity keys stored in the database for key recovery. */
32    public static final int CAKEYPURPOSE_KEYENCRYPT = 3;
33    /** used for testing if the CA token is functioning and on-line. */
34    public static final int CAKEYPURPOSE_KEYTEST = 4;
35    /** encrypts hard token PIN/PUK codes etc. */
36    public static final int CAKEYPURPOSE_HARDTOKENENCRYPT = 5;
37    /** The CAs previous signing key, if any exists. */
38    public static final int CAKEYPURPOSE_CERTSIGN_PREVIOUS = 6;
39    /** The CAs next signing key, if any exists. */
40    public static final int CAKEYPURPOSE_CERTSIGN_NEXT = 7;
41  
42    /** Purposes. */
43    public static final int[] ALL_KEY_PURPOSES =
44        new int[] {
45          CATokenConstants.CAKEYPURPOSE_CERTSIGN,
46          CATokenConstants.CAKEYPURPOSE_CRLSIGN,
47          CATokenConstants.CAKEYPURPOSE_KEYENCRYPT,
48          CATokenConstants.CAKEYPURPOSE_KEYTEST,
49          CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT,
50          CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS,
51          CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT
52        };
53  
54    /**
55     * Key strings for token properties matching the above constants, i.e. when
56     * doing getPrivateKey(1) on a CAToken it will try to use the key configured
57     * with label certSignKey in the CryptoToken properties.
58     */
59    public static final String CAKEYPURPOSE_CERTSIGN_STRING = "certSignKey";
60    /** Key. */
61    public static final String CAKEYPURPOSE_CRLSIGN_STRING = "crlSignKey";
62    /** Key. */
63    public static final String CAKEYPURPOSE_KEYENCRYPT_STRING = "keyEncryptKey";
64    /** Test. */
65    public static final String CAKEYPURPOSE_TESTKEY_STRING = "testKey";
66    /** Default. */
67    public static final String CAKEYPURPOSE_DEFAULT_STRING = "defaultKey";
68    /** Hard token. */
69    public static final String CAKEYPURPOSE_HARDTOKENENCRYPT_STRING =
70        "hardTokenEncrypt";
71    /** Previous. */
72    public static final String CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS =
73        "previousCertSignKey";
74    /** Next. */
75    public static final String CAKEYPURPOSE_CERTSIGN_STRING_NEXT =
76        "nextCertSignKey";
77  
78    /**
79     * Previous sequence (matching
80     * CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS key) that can be
81     * set in CA token properties.
82     */
83    public static final String PREVIOUS_SEQUENCE_PROPERTY = "previousSequence";
84    /**
85     * Next sequence (matching
86     * CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT key) that can be set
87     * in CA token properties.
88     */
89    public static final String NEXT_SEQUENCE_PROPERTY = "nextSequence";
90  }
