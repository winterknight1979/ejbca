13  package org.cesecore.certificates.ca.catoken;
14  
15  /**
16   * Purpose mapping constants to for CryptoToken aliases and related.
17   * 
18   * @version $Id: CATokenConstants.java 27312 2017-11-24 10:00:33Z anatom $
19   */
20  public final class CATokenConstants {
21  
22      /** default key used when there is no specific setting for a purpose */
23      public static final int CAKEYPURPOSE_DEFAULT             = 0;
24      /** signs certificates issued by the CA */
25  	public static final int CAKEYPURPOSE_CERTSIGN            = 1;
26  	/** igns CRLs issues by the CA */
27  	public static final int CAKEYPURPOSE_CRLSIGN             = 2;
28  	/** encrypts entity keys stored in the database for key recovery */
29  	public static final int CAKEYPURPOSE_KEYENCRYPT          = 3;
30  	/** used for testing if the CA token is functioning and on-line */
31  	public static final int CAKEYPURPOSE_KEYTEST             = 4;
32  	/** encrypts hard token PIN/PUK codes etc */
33  	public static final int CAKEYPURPOSE_HARDTOKENENCRYPT    = 5;
34  	/** The CAs previous signing key, if any exists */
35  	public static final int CAKEYPURPOSE_CERTSIGN_PREVIOUS   = 6;
36  	/** The CAs next signing key, if any exists */
37  	public static final int CAKEYPURPOSE_CERTSIGN_NEXT       = 7;
38  
39      public final static int[] ALL_KEY_PURPOSES = new int[] {
40          CATokenConstants.CAKEYPURPOSE_CERTSIGN,
41          CATokenConstants.CAKEYPURPOSE_CRLSIGN,
42          CATokenConstants.CAKEYPURPOSE_KEYENCRYPT,
43          CATokenConstants.CAKEYPURPOSE_KEYTEST,
44          CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT,
45          CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS,
46          CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT
47      };
48  
49  	/** Key strings for token properties matching the above constants, i.e. when doing getPrivateKey(1)
50  	 * on a CAToken it will try to use the key configured with label certSignKey in the CryptoToken properties. */
51  	final static public String CAKEYPURPOSE_CERTSIGN_STRING = "certSignKey";
52  	final static public String CAKEYPURPOSE_CRLSIGN_STRING = "crlSignKey";
53  	final static public String CAKEYPURPOSE_KEYENCRYPT_STRING = "keyEncryptKey";
54  	final static public String CAKEYPURPOSE_TESTKEY_STRING = "testKey";
55  	final static public String CAKEYPURPOSE_DEFAULT_STRING = "defaultKey";
56  	final static public String CAKEYPURPOSE_HARDTOKENENCRYPT_STRING = "hardTokenEncrypt";
57  	final static public String CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS = "previousCertSignKey";
58  	final static public String CAKEYPURPOSE_CERTSIGN_STRING_NEXT = "nextCertSignKey";
59  
60      /** Previous sequence (matching CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS key) that can be set in CA token properties */
61      public static final String PREVIOUS_SEQUENCE_PROPERTY = "previousSequence";
62      /** Next sequence (matching CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT key) that can be set in CA token properties */
63      public static final String NEXT_SEQUENCE_PROPERTY = "nextSequence";
64  }
