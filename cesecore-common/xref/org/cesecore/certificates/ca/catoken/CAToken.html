33  /**
34   * The CAToken is keeps references to the CA's key aliases and the CryptoToken where the keys are stored.
35   * 
36   * The signing key can have 3 stages:
37   * - Next:     Can become the new current CA key when a valid signing certificate is present
38   * - Current:  Is used to issue certificates and has a CA certificate
39   * - Previous: The signing key before the latest CA renewal.
40   * 
41   * Each CA signing key "generation" has a corresponding key sequence number that is kept track of
42   * via this class. The key sequence also have the states next, current and previous.
43   * 
44   * The CA token stores a reference (an integer) to the CryptoToken where the CA keys are stored.
45   * 
46   * @version $Id: CAToken.java 26093 2017-06-28 15:05:13Z anatom $
47   */
48  public class CAToken extends UpgradeableDataHashMap {

99      /** Constructor used to initialize a stored CA token, when the UpgradeableHashMap has been stored as is.
100      * 
101      * @param tokendata LinkedHashMap
102      */
103     @SuppressWarnings("rawtypes")
104     public CAToken(final HashMap tokendata) {
105 		loadData(tokendata);
106 		final Object cryptoTokenIdObject = data.get(CAToken.CRYPTOTOKENID);
107 		if (cryptoTokenIdObject==null) {
108 		    log.warn("No CryptoTokenId in CAToken map. This can safely be ignored if shown during an upgrade from EJBCA 5.0.x or lower.");
109 		} else {
110             this.cryptoTokenId = Integer.parseInt((String) cryptoTokenIdObject);
111 		}
112         final Properties caTokenProperties = getProperties();
113         internalInit(caTokenProperties);
114     }

116     /** Verifies that the all the mapped keys are present in the CryptoToken and optionally that the test key is usable. 
117      * @param caTokenSignTest bool
118      * @param cryptoToken token
119      * @return status code*/
120     public int getTokenStatus(boolean caTokenSignTest, CryptoToken cryptoToken) {
121         if (log.isTraceEnabled()) {
122             log.trace(">getCATokenStatus");
123         }
124         int ret = CryptoToken.STATUS_OFFLINE;
125         // If we have no key aliases, no point in continuing...
126         try {
127         	if (keyStrings != null) {
128         		final String aliases[] = keyStrings.getAliases();
129         		final String aliasCertSignKeyPrevious = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS);
130                 final String aliasCertSignKeyNext = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT);
131                 final String aliasTestKey = getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_KEYTEST);
132         		int foundKeys = 0;
133                 // Loop that checks  if there all key aliases have keys
134         		if (cryptoToken!=null) {
135                     final HashMap<String, PrivateKey> aliasMap = new HashMap<String, PrivateKey>();
136                     for (final String alias : aliases) {
137                         PrivateKey privateKey = aliasMap.get(alias);
138                         if (privateKey==null) {
139                             try {
140                                 privateKey = cryptoToken.getPrivateKey(alias);
141                                 // Cache lookup to avoid having to retrieve the same key when used for multiple purposes
142                                 if (privateKey!=null) {
143                                     aliasMap.put(alias, privateKey);
144                                 }
145                             } catch (CryptoTokenOfflineException e) {
146                                 privateKey = null;
147                             }
148                         }
149                         if (privateKey==null) {
150                             // We don't consider it critical if currently unused certificate signing keys has been deleted (as long as it isn't mapped for any other purposes)
151                             if (alias.equals(aliasCertSignKeyPrevious) && keyStrings.isAliasMappedForSinglePurpose(aliasCertSignKeyPrevious)) {
152                                 foundKeys++;
153                                 if (log.isDebugEnabled()) {
154                                     log.debug("Missing private key for alias: "+alias + " (Not treated as an error, since it is only mapped as the previous CA signing key.)");
155                                 }
156                             } else if (alias.equals(aliasCertSignKeyNext) && keyStrings.isAliasMappedForSinglePurpose(aliasCertSignKeyNext)) {
157                                     foundKeys++;
158                                     if (log.isDebugEnabled()) {
159                                         log.debug("Missing private key for alias: "+alias + " (Not treated as an error, since it is only mapped as the next CA signing key.)");
160                                     }
161                             } else {
162                                 if (log.isDebugEnabled()) {
163                                     log.debug("Missing private key for alias: "+alias);
164                                 }
165                             }
166                         } else {
167                             foundKeys++;
168                         }
169                         if (alias.equals(aliasTestKey)) {
170                             PublicKey publicKey;
171                             try {
172                                 publicKey = cryptoToken.getPublicKey(aliasTestKey);
173                             } catch (CryptoTokenOfflineException e) {
174                                 publicKey = null;
175                             }
176                             if (publicKey == null) {
177                                 if (log.isDebugEnabled()) {
178                                     log.debug("Missing public key for alias: "+alias);
179                                 }
180                             }
181                             // Check that that the testkey is usable by doing a test signature.
182                             try {
183                                 if (caTokenSignTest) {
184                                     cryptoToken.testKeyPair(alias, publicKey, privateKey);
185                                 }
186                                 // If we can test the testkey, we are finally active!
187                                 ret = CryptoToken.STATUS_ACTIVE;
188                             } catch (Throwable th) { // NOPMD: we need to catch _everything_ when dealing with HSMs
189                                 log.error(intres.getLocalizedMessage("token.activationtestfail", cryptoToken.getId()), th);
190                             }
191                         }
192                     }
193         		}
194                 if (foundKeys < aliases.length) {
195         			if (log.isDebugEnabled()) {
196         				StringBuilder builder = new StringBuilder();
197         				for (int j = 0; j < aliases.length; j++) {
198         					builder.append(' ').append(aliases[j]);
199         				}
200         				log.debug("Not enough keys for the key aliases: "+builder.toString());
201         			}
202                     ret = CryptoToken.STATUS_OFFLINE;
203         		}
204         	}
205         } catch (CryptoTokenOfflineException e) {
206         	if (log.isDebugEnabled()) {
207         		log.debug("CryptoToken offline: "+e.getMessage());
208         	}
209         }
210 
211         if (log.isTraceEnabled()) {
212         	log.trace("<getCATokenStatus: " + ret);
213         }
214         return ret;
215     }

217     /** @param purpose purpose
218      * @return the key pair alias in the CryptoToken from the CATokenConstants.CAKEYPURPOSE_.. 
219      * @throws CryptoTokenOfflineException if offline */
220     public String getAliasFromPurpose(final int purpose) throws CryptoTokenOfflineException {
221         if (keyStrings==null) {
222             // keyStrings is transient and can be null after serialization
223             keyStrings = new PurposeMapping(getProperties());
224         }
225         final String alias = keyStrings.getAlias(purpose);
226         if (alias == null) {
227             throw new CryptoTokenOfflineException("No alias for key purpose " + purpose);
228         }
229         return alias;
230     }

252     /**
253      * Internal method just to get rid of the always present date that is part of the standard Properties.store().
254      * 
255      * @param caTokenProperties properties
256      * @return String that can be loaded by Properties.load
257      */
258     private String storeProperties(Properties caTokenProperties) {
259         this.keyStrings = new PurposeMapping(caTokenProperties);
260         final StringWriter sw = new StringWriter();
261         try ( final PrintWriter writer = new PrintWriter(sw); ) {
262             final Enumeration<Object> e = caTokenProperties.keys();
263             while (e.hasMoreElements()) {
264                 final Object s = e.nextElement();
265                 if (caTokenProperties.get(s) != null) {
266                     writer.println(s + "=" + caTokenProperties.get(s));
267                 }
268             }
269         }
270         return sw.toString();
271     }

283         }
284         return getPropertiesFromString(propertyStr);

287     public static Properties getPropertiesFromString(final String propertyStr) {
288         final Properties prop = new Properties();
289         if (StringUtils.isNotEmpty(propertyStr)) {
290             try {
291                 // If the input string contains \ (backslash on windows) we must convert it to \\
292                 // Otherwise properties.load will parse it as an escaped character, and that is not good
293                 final String propertyStrAdjusted = StringUtils.replace(propertyStr, "\\", "\\\\");
294                 prop.load(new StringReader(propertyStrAdjusted));
295                 // Trim whitespace in values
296                 for (final Object keyObj : prop.keySet()) {
297                     String key = (String)keyObj;
298                     String value = prop.getProperty(key);
299                     prop.setProperty(key, value.trim());
300                 }
301             } catch (IOException e) {
302                 log.error("Error getting PKCS#11 token properties: ", e);
303             }
304         }
305         return prop;
306     }

366     /** @see org.cesecore.internal.UpgradeableDataHashMap#upgrade() */
367     @Override
368     public void upgrade() {
369         if (Float.compare(LATEST_VERSION, getVersion()) != 0) {
370             // New version of the class, upgrade
371             String msg = intres.getLocalizedMessage("token.upgrade", Float.valueOf(getVersion()));
372             log.info(msg);
373             // Put upgrade stuff here
374             if (data.get(CAToken.SEQUENCE_FORMAT) == null) { // v7
375                 log.info("Adding new sequence format to CA Token data: " + StringTools.KEY_SEQUENCE_FORMAT_NUMERIC);
376                 data.put(CAToken.SEQUENCE_FORMAT, StringTools.KEY_SEQUENCE_FORMAT_NUMERIC);
377             }
378             if (data.get(CAToken.SEQUENCE) == null) { // v7
379                 log.info("Adding new default key sequence to CA Token data: " + CAToken.DEFAULT_KEYSEQUENCE);
380                 data.put(CAToken.SEQUENCE, CAToken.DEFAULT_KEYSEQUENCE);
381             }
382 
383             if (data.get(CAToken.CLASSPATH) != null) { // v8 upgrade of classpaths for CESeCore
384                 final String classpath = (String) data.get(CAToken.CLASSPATH);
385                 log.info("Upgrading CA token classpath: "+classpath);
386                 String newclasspath = classpath;
387                 if (StringUtils.equals(classpath, "org.ejbca.core.model.ca.catoken.SoftCAToken")) {
388                 	newclasspath = "org.cesecore.keys.token.SoftCryptoToken";
389                 	// Upgrade properties to set a default key, also for soft crypto tokens
390                 	Properties prop = getProperties();
391                     // A small unfortunate special property that we have to make in order to 
392                     // be able to use soft keystores that does not have a specific test or default key
393                     if ((prop.getProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING) == null) &&
394                     		(prop.getProperty(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING) == null)) {
395                         // The soft key alias was changed from privatesignkeyalias to signKey in EJBCA 6.4.1, which is long after
396                         // we changed the classpath. So if we come in here, we are upgrading a token that is way before 6.4.1, meaning 
397                         // that it uses the old key aliases
398                     	log.info("Setting CAKEYPURPOSE_CERTSIGN_STRING and CAKEYPURPOSE_CRLSIGN_STRING to privatesignkeyalias.");
399                     	prop.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING, CAToken.OLDPRIVATESIGNKEYALIAS);
400                     	prop.setProperty(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING, CAToken.OLDPRIVATESIGNKEYALIAS);
401                     }
402                     if ((prop.getProperty(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING) == null) &&
403                     		(prop.getProperty(CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING) == null)) {
404                         // Same as above regarding key aliases
405                         log.info("Setting CAKEYPURPOSE_DEFAULT_STRING to privatedeckeyalias.");
406                     	prop.setProperty(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING, CAToken.OLDPRIVATEDECKEYALIAS);
407                     }
408                     setCATokenPropertyData(storeProperties(prop)); // Stores property string in "data"
409                 } else if (StringUtils.equals(classpath, "org.ejbca.core.model.ca.catoken.PKCS11CAToken")) {
410                 	newclasspath = "org.cesecore.keys.token.PKCS11CryptoToken";
411                 } else if (StringUtils.equals(classpath, "org.ejbca.core.model.ca.catoken.NullCAToken")) {
412                 	newclasspath = "org.cesecore.keys.token.NullCryptoToken";
413                 } else if (StringUtils.equals(classpath, "org.ejbca.core.model.ca.catoken.NFastCAToken")) {
414                 	log.error("Upgrading of NFastCAToken not supported, you need to convert to using PKCS11CAToken before upgrading.");
415                 }
416                 data.put(CAToken.CLASSPATH, newclasspath);
417             }
418 
419             data.put(VERSION, Float.valueOf(LATEST_VERSION));
420         }
421     }

423     /**
424      * Use current key sequence to generate and store a "next" key sequence and "next" singing key alias.
425      * @return the next sign key alias.
426      */
427     public String generateNextSignKeyAlias() {
428         // Generate a new key sequence
429         final String currentKeySequence = getKeySequence();
430         final String newKeySequence = StringTools.incrementKeySequence(getKeySequenceFormat(), currentKeySequence);
431         if (log.isDebugEnabled()) {
432             log.debug("Current key sequence: " + currentKeySequence + "  New key sequence: " + newKeySequence);
433         }
434         // Generate a key alias based on the new key sequence
435         final String currentCertSignKeyLabel = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CERTSIGN);
436         final String newCertSignKeyLabel = StringUtils.removeEnd(currentCertSignKeyLabel, currentKeySequence) + newKeySequence;
437         if (log.isDebugEnabled()) {
438             log.debug("Current sign key alias: " + currentCertSignKeyLabel + "  New sign key alias: " + newCertSignKeyLabel);
439         }
440         // Store the new values in the properties of this token
441         setNextCertSignKey(newCertSignKeyLabel);
442         setNextKeySequence(newKeySequence);
443         return newCertSignKeyLabel;
444     }

446     /** Next sign key becomes current. Current becomes previous. Same goes for KeySequence. CRL sign key is updated if it is the same as cert sign key */
447     public void activateNextSignKey() {
448         final Properties caTokenProperties = getProperties();
449         // Replace certificate (and crl) signing key aliases (if present)
450         boolean swichedSigningKey = false;
451         final String nextCertSignKeyLabel = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT);
452         if (nextCertSignKeyLabel!=null) {
453             final String currentCertSignKeyLabel = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CERTSIGN);
454             final String currentCrlSignKeyLabel = keyStrings.getAlias(CATokenConstants.CAKEYPURPOSE_CRLSIGN);
455             if (log.isDebugEnabled()) {
456                 log.debug("CERTSIGN_NEXT: " + nextCertSignKeyLabel);
457                 log.debug("CERTSIGN:      " + currentCertSignKeyLabel);
458                 log.debug("CRLSIGN:       " + currentCrlSignKeyLabel);
459             }
460             if (StringUtils.equals(currentCertSignKeyLabel, currentCrlSignKeyLabel)) {
461                 log.info("Setting CRL signing key alias to: " + nextCertSignKeyLabel);
462                 caTokenProperties.setProperty(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING, nextCertSignKeyLabel);
463             }
464             log.info("Setting certificate signing key alias to: " + nextCertSignKeyLabel);
465             caTokenProperties.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS, currentCertSignKeyLabel);
466             caTokenProperties.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING, nextCertSignKeyLabel);
467             caTokenProperties.remove(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT);
468             swichedSigningKey = !StringUtils.equals(nextCertSignKeyLabel, currentCertSignKeyLabel);
469         }
470         // Replace key sequence (if present)
471         final String nextKeySequence = caTokenProperties.getProperty(CATokenConstants.NEXT_SEQUENCE_PROPERTY);
472         final String currentKeySequence = getKeySequence();
473         if (nextKeySequence != null) {
474             if (log.isDebugEnabled()) {
475                 log.debug("Current KeySequence: " + getKeySequence());
476             }
477             log.info("Set key sequence from nextSequence: " + nextKeySequence);
478             caTokenProperties.setProperty(CATokenConstants.PREVIOUS_SEQUENCE_PROPERTY, currentKeySequence);
479             setKeySequence(nextKeySequence);
480             caTokenProperties.remove(CATokenConstants.NEXT_SEQUENCE_PROPERTY);
481         } else if (swichedSigningKey) {
482             // If we did not have a next key sequence before this activation we generate one and push back the current.
483             final String newKeySequence = StringTools.incrementKeySequence(getKeySequenceFormat(), currentKeySequence);
484             caTokenProperties.setProperty(CATokenConstants.PREVIOUS_SEQUENCE_PROPERTY, currentKeySequence);
485             setKeySequence(newKeySequence);
486         } else {
487             // So there is no key sequence and we didn't switch singing key..
488             // ..let us just set the previous sequence to the current to at least match the singing key alias
489             caTokenProperties.setProperty(CATokenConstants.PREVIOUS_SEQUENCE_PROPERTY, currentKeySequence);
490         }
491         // Store changes in the CAToken's properties
492         setCATokenPropertyData(storeProperties(caTokenProperties));
493     }
