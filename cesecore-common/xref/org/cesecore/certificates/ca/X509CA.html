245   /**
246    * Creates a new instance of CA, this constructor should be used when a new CA
247    * is created.
248    *
249    * @param cainfo Info
250    */
251   public X509CA(final X509CAInfo cainfo) {
252     super(cainfo);
253     // Verify integrity if caInfo, either one SubjectDN or SubjectAltName needs
254     // to be filled in
255     if (StringUtils.isEmpty(
256             DNFieldsUtil.removeAllEmpties(cainfo.getSubjectDN()))
257         && StringUtils.isEmpty(cainfo.getSubjectAltName())) {
258       throw new IllegalArgumentException(
259           "Subject DN and Alt Name can't both be blank for an X509 CA.");
260     }
261     data.put(POLICIES, cainfo.getPolicies());
262     data.put(SUBJECTALTNAME, cainfo.getSubjectAltName());
263     setUseAuthorityKeyIdentifier(cainfo.getUseAuthorityKeyIdentifier());
264     setAuthorityKeyIdentifierCritical(
265         cainfo.getAuthorityKeyIdentifierCritical());
266     setUseCRLNumber(cainfo.getUseCRLNumber());
267     setCRLNumberCritical(cainfo.getCRLNumberCritical());
268     setDefaultCRLDistPoint(cainfo.getDefaultCRLDistPoint());
269     setDefaultCRLIssuer(cainfo.getDefaultCRLIssuer());
270     setCADefinedFreshestCRL(cainfo.getCADefinedFreshestCRL());
271     setDefaultOCSPServiceLocator(cainfo.getDefaultOCSPServiceLocator());
272     setUseUTF8PolicyText(cainfo.getUseUTF8PolicyText());
273     setUsePrintableStringSubjectDN(cainfo.getUsePrintableStringSubjectDN());
274     setUseLdapDNOrder(cainfo.getUseLdapDnOrder());
275     setUseCrlDistributionPointOnCrl(cainfo.getUseCrlDistributionPointOnCrl());
276     setCrlDistributionPointOnCrlCritical(
277         cainfo.getCrlDistributionPointOnCrlCritical());
278     setKeepExpiredCertsOnCRL(cainfo.getKeepExpiredCertsOnCRL());
279     setCmpRaAuthSecret(cainfo.getCmpRaAuthSecret());
280     // CA Issuer URI to put in CRLs (RFC5280 section 5.2.7, not the URI to put
281     // in certs
282     setAuthorityInformationAccess(cainfo.getAuthorityInformationAccess());
283     setCertificateAiaDefaultCaIssuerUri(
284         cainfo.getCertificateAiaDefaultCaIssuerUri());
285     setNameConstraintsPermitted(cainfo.getNameConstraintsPermitted());
286     setNameConstraintsExcluded(cainfo.getNameConstraintsExcluded());
287     data.put(CA.CATYPE, Integer.valueOf(CAInfo.CATYPE_X509));
288     data.put(VERSION, Float.valueOf(LATEST_VERSION));
289     setCaSerialNumberOctetSize(cainfo.getCaSerialNumberOctetSize());
290   }

322       // upgrade method in this class, but it needs to be ignored
323       // for instantiation.

326             this.getExtendedCAServiceInfo(type.intValue());

350             .setRevocationReason(getRevocationReason())

352             .setPolicies(getPolicies())
353             .setCrlPeriod(getCRLPeriod())
354             .setCrlIssueInterval(getCRLIssueInterval())
355             .setCrlOverlapTime(getCRLOverlapTime())
356             .setDeltaCrlPeriod(getDeltaCRLPeriod())
357             .setCrlPublishers(getCRLPublishers())
358             .setValidators(getValidators())
359             .setUseAuthorityKeyIdentifier(getUseAuthorityKeyIdentifier())
360             .setAuthorityKeyIdentifierCritical(
361                 getAuthorityKeyIdentifierCritical())
362             .setUseCrlNumber(getUseCRLNumber())
363             .setCrlNumberCritical(getCRLNumberCritical())
364             .setDefaultCrlDistPoint(getDefaultCRLDistPoint())
365             .setDefaultCrlIssuer(getDefaultCRLIssuer())

367             .setAuthorityInformationAccess(getAuthorityInformationAccess())
368             .setCertificateAiaDefaultCaIssuerUri(
369                 getCertificateAiaDefaultCaIssuerUri())
370             .setNameConstraintsPermitted(getNameConstraintsPermitted())
371             .setNameConstraintsExcluded(getNameConstraintsExcluded())
372             .setCaDefinedFreshestCrl(getCADefinedFreshestCRL())
373             .setFinishUser(getFinishUser())
374             .setExtendedCaServiceInfos(externalcaserviceinfos)
375             .setUseUtf8PolicyText(getUseUTF8PolicyText())
376             .setApprovals(getApprovals())
377             .setUsePrintableStringSubjectDN(getUsePrintableStringSubjectDN())
378             .setUseLdapDnOrder(getUseLdapDNOrder())
379             .setUseCrlDistributionPointOnCrl(getUseCrlDistributionPointOnCrl())
380             .setCrlDistributionPointOnCrlCritical(
381                 getCrlDistributionPointOnCrlCritical())
382             .setIncludeInHealthCheck(getIncludeInHealthCheck())
383             .setDoEnforceUniquePublicKeys(isDoEnforceUniquePublicKeys())
384             .setDoEnforceUniqueDistinguishedName(
385                 isDoEnforceUniqueDistinguishedName())
386             .setDoEnforceUniqueSubjectDNSerialnumber(
387                 isDoEnforceUniqueSubjectDNSerialnumber())
388             .setUseCertReqHistory(isUseCertReqHistory())
389             .setUseUserStorage(isUseUserStorage())

392                 isAcceptRevocationNonExistingEntry())

736   @Override
737   public byte[] createPKCS7(
738       final CryptoToken cryptoToken,
739       final X509Certificate cert,
740       final boolean includeChain)
741       throws SignRequestSignatureException {
742     // First verify that we signed this certificate
743     final X509Certificate cacert = (X509Certificate) getCACertificate();
744     if (cert != null) {
745       try {
746         final PublicKey verifyKey;
747         if (cacert != null) {
748           verifyKey = cacert.getPublicKey();
749         } else {
750           verifyKey =
751               cryptoToken.getPublicKey(
752                   getCAToken()
753                       .getAliasFromPurpose(
754                           CATokenConstants.CAKEYPURPOSE_CERTSIGN));
755         }
756         cert.verify(verifyKey);
757       } catch (CryptoTokenOfflineException e) {
758         throw new SignRequestSignatureException(
759             "The cryptotoken was not available, could not create a PKCS7", e);
760       } catch (InvalidKeyException e) {
761         throw new SignRequestSignatureException(
762             "The specified certificate contains the wrong public key.", e);
763       } catch (CertificateException e) {
764         throw new SignRequestSignatureException(
765             "An encoding error was encountered.", e);
766       } catch (NoSuchAlgorithmException e) {
767         throw new SignRequestSignatureException(
768             "The certificate provided was signed with an invalid algorithm.",
769             e);
770       } catch (NoSuchProviderException e) {
771         throw new SignRequestSignatureException(
772             "The crypto provider was not found for verification of the"
773                 + " certificate.",
774             e);
775       } catch (SignatureException e) {
776         throw new SignRequestSignatureException(
777             "Cannot verify certificate in createPKCS7(), did I sign this?", e);
778       }
779     }
780     final List<X509Certificate> x509Chain = new ArrayList<>();
781     if (cert != null) {
782       x509Chain.add(cert);
783     }
784     if (includeChain) {
785       x509Chain.addAll(
786           CertTools.convertCertificateChainToX509Chain(getCertificateChain()));
787     }
788     List<JcaX509CertificateHolder> certList;
789     try {
790       certList = CertTools.convertToX509CertificateHolder(x509Chain);
791     } catch (CertificateEncodingException e) {
792       throw new SignRequestSignatureException(
793           "Could not encode certificate", e);
794     }
795     try {
796       CMSTypedData msg = new CMSProcessableByteArray(new byte[0]);
797       CMSSignedDataGenerator gen = new CMSSignedDataGenerator();
798       final PrivateKey privateKey =
799           cryptoToken.getPrivateKey(
800               getCAToken()
801                   .getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN));
802       if (privateKey == null) {
803         String msg1 = "createPKCS7: Private key does not exist!";
804         LOG.debug(msg1);
805         throw new SignRequestSignatureException(msg1);
806       }
807       String signatureAlgorithmName =
808           AlgorithmTools.getAlgorithmNameFromDigestAndKey(
809               CMSSignedGenerator.DIGEST_SHA1, privateKey.getAlgorithm());
810       try {
811         ContentSigner contentSigner =
812             new JcaContentSignerBuilder(signatureAlgorithmName)
813                 .setProvider(cryptoToken.getSignProviderName())
814                 .build(privateKey);
815         JcaDigestCalculatorProviderBuilder calculatorProviderBuilder =
816             new JcaDigestCalculatorProviderBuilder()
817                 .setProvider(BouncyCastleProvider.PROVIDER_NAME);
818         JcaSignerInfoGeneratorBuilder builder =
819             new JcaSignerInfoGeneratorBuilder(
820                 calculatorProviderBuilder.build());
821         gen.addSignerInfoGenerator(builder.build(contentSigner, cacert));
822       } catch (OperatorCreationException e) {
823         throw new IllegalStateException(
824             "BouncyCastle failed in creating signature provider.", e);
825       }
826       gen.addCertificates(new CollectionStore<>(certList));
827       CMSSignedData s = null;
828       CAToken catoken = getCAToken();
829       if (catoken != null && !(cryptoToken instanceof NullCryptoToken)) {
830         LOG.debug(
831             "createPKCS7: Provider="
832                 + cryptoToken.getSignProviderName()
833                 + " using algorithm "
834                 + privateKey.getAlgorithm());
835         s = gen.generate(msg, true);
836       } else {
837         String msg1 = "CA Token does not exist!";
838         LOG.debug(msg1);
839         throw new SignRequestSignatureException(msg1);
840       }
841       return s.getEncoded();
842     } catch (CryptoTokenOfflineException e) {
843       throw new IllegalStateException(e);
844     } catch (Exception e) {
845       // FIXME: This right here is just nasty
846       throw new IllegalStateException(e);
847     }
848   }

850   @Override
851   public byte[] createPKCS7Rollover(final CryptoToken cryptoToken)
852       throws SignRequestSignatureException {
853     List<Certificate> nextChain = getRolloverCertificateChain();
854     if (nextChain == null) {
855       LOG.debug("CA does not have a rollover chain, returning empty PKCS#7");
856       nextChain = Collections.emptyList();
857     } else if (nextChain.isEmpty()) {
858       LOG.warn("next chain exists but is empty");
859     }
860 
861     ArrayList<X509CertificateHolder> certList = new ArrayList<>();
862     try {
863       for (Certificate certificate : nextChain) {
864         certList.add(
865             new JcaX509CertificateHolder((X509Certificate) certificate));
866       }
867     } catch (CertificateEncodingException e) {
868       throw new SignRequestSignatureException(
869           "Could not encode certificate", e);
870     }
871     if (LOG.isDebugEnabled()) {
872       LOG.debug(
873           "createPKCS7Rollover: Creating a rollover chain with "
874               + certList.size()
875               + " certificates.");
876     }
877     try {
878       CMSTypedData msg = new CMSProcessableByteArray(new byte[0]);
879       CMSSignedDataGenerator gen = new CMSSignedDataGenerator();
880       // We always sign with the current key, even during rollover, so the new
881       // key can be linked to the old key. SCEP draft 23, "4.6.1.  Get Next CA
882       // Response Message Format"
883       final PrivateKey privateKey =
884           cryptoToken.getPrivateKey(
885               getCAToken()
886                   .getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN));
887       if (privateKey == null) {
888         final String msg1 = "createPKCS7Rollover: Private key does not exist!";
889         LOG.debug(msg1);
890         throw new SignRequestSignatureException(msg1);
891       }
892       String signatureAlgorithmName =
893           AlgorithmTools.getAlgorithmNameFromDigestAndKey(
894               CMSSignedGenerator.DIGEST_SHA1, privateKey.getAlgorithm());
895       try {
896         ContentSigner contentSigner =
897             new JcaContentSignerBuilder(signatureAlgorithmName)
898                 .setProvider(cryptoToken.getSignProviderName())
899                 .build(privateKey);
900         JcaDigestCalculatorProviderBuilder calculatorProviderBuilder =
901             new JcaDigestCalculatorProviderBuilder()
902                 .setProvider(BouncyCastleProvider.PROVIDER_NAME);
903         JcaSignerInfoGeneratorBuilder builder =
904             new JcaSignerInfoGeneratorBuilder(
905                 calculatorProviderBuilder.build());
906         gen.addSignerInfoGenerator(
907             builder.build(contentSigner, (X509Certificate) getCACertificate()));
908       } catch (OperatorCreationException e) {
909         throw new IllegalStateException(
910             "BouncyCastle failed in creating signature provider.", e);
911       } catch (CertificateEncodingException e) {
912         throw new IllegalStateException(e);
913       }
914       gen.addCertificates(new CollectionStore<>(certList));
915       CMSSignedData s = null;
916       CAToken catoken = getCAToken();
917       if (catoken != null && !(cryptoToken instanceof NullCryptoToken)) {
918         LOG.debug(
919             "createPKCS7Rollover: Provider="
920                 + cryptoToken.getSignProviderName()
921                 + " using algorithm "
922                 + privateKey.getAlgorithm());
923         // Don't encapsulate any content, i.e. the bytes in the message. This
924         // makes data section of the PKCS#7 message completely empty.
925         // BER Sequence
926         //   ObjectIdentifier(1.2.840.113549.1.7.1)
927         // Instead of
928         // BER Sequence
929         //   ObjectIdentifier(1.2.840.113549.1.7.1)
930         //   BER Tagged [0]
931         //     BER Constructed Octet String[0]
932         s = gen.generate(msg, false);
933       } else {
934         String msg1 = "CA Token does not exist!";
935         LOG.debug(msg1);
936         throw new SignRequestSignatureException(msg1);
937       }
938       return s.getEncoded();
939     } catch (CryptoTokenOfflineException e) {
940       throw new IllegalStateException(e);
941     } catch (CMSException e) {
942       throw new IllegalStateException(e);
943     } catch (IOException e) {
944       throw new IllegalStateException("Failed to encode CMS data", e);
945     }
946   }

1832                 break;

1907                       .getParsedValue()); // adding nrOfRedactedLabels extension
1908                                           // to certbuilder
1909             }
1910           }
1911         } else { // if not a subjectAlternativeName extension, just add it to
1912                  // both certbuilder and precertbuilder

1915           // invalid extensions that is not constructed from proper ASN.1
1916           final byte[] value = extension.getExtnValue().getOctets();
1917           certbuilder.addExtension(extension.getExtnId(), isCritical, value);
1918           if (precertbuilder != null) {
1919             precertbuilder.addExtension(
1920                 extension.getExtnId(), isCritical, value);
1921           }
1922         }

1927       // the CA key so it has to be processed here inside X509CA.

2133           final String skistr = new String(Hex.encode(ski));

2135               INTRES.getLocalizedMessage(
2136                   "createcert.errorpathverifykeyid", akistr, skistr);

2138           throw new CertificateCreateException(msg);
2139         }
2140       }

2153         }
2154       }

2159     // be signed
2160     if (request != null) {
2161       request.setResponseKeyInfo(caPrivateKey, provider);
2162     }
2163     if (LOG.isDebugEnabled()) {
2164       LOG.debug(
2165           "X509CA: generated certificate, CA "
2166               + this.getCAId()
2167               + " for DN: "
2168               + subject.getCertificateDN());
2169     }
2170     return cert;
2171   }

2203    * Constructs the SubjectAlternativeName extension that will end up on the
2204    * generated certificate.
2205    *
2206    * <p>If the DNS values in the subjectAlternativeName extension contain
2207    * parentheses to specify labels that should be redacted, the parentheses are

2209    * added.

2211    * @param subAltNameExt Alt Name
2212    * @param publishToCT boolean
2213    * @return An extension generator containing the SubjectAlternativeName
2214    *     extension and an extension holding the number of redacted labels if the
2215    *     certificate is to be published to a CTLog
2216    * @throws IOException if read fails
2217    */
2218   protected ExtensionsGenerator getSubjectAltNameExtensionForCert(
2219       final Extension subAltNameExt,
2220       final boolean publishToCT) throws IOException {
2221     GeneralNames names = CertTools.getGeneralNamesFromExtension(subAltNameExt);
2222     GeneralName[] gns = names.getNames();
2223     boolean sanEdited = false;
2224     ASN1EncodableVector nrOfRecactedLables = new ASN1EncodableVector();
2225     for (int j = 0; j < gns.length; j++) {
2226       GeneralName generalName = gns[j];
2227       // Look for DNS name
2228       if (generalName.getTagNo() == 2) {
2229         final String str =
2230             CertTools.getGeneralNameString(2, generalName.getName());
2231         if (StringUtils.contains(str, "(")
2232             && StringUtils.contains(
2233                 str, ")")) { // if it contains parts that should be redacted
2234           // Remove the parentheses from the SubjectAltName that will end up on
2235           // the certificate
2236           String certBuilderDNSValue = StringUtils.remove(str, "dNSName=");
2237           certBuilderDNSValue = StringUtils.remove(certBuilderDNSValue, '(');
2238           certBuilderDNSValue = StringUtils.remove(certBuilderDNSValue, ')');
2239           // Replace the old value with the new
2240           gns[j] = new GeneralName(2, new DERIA5String(certBuilderDNSValue));
2241           sanEdited = true;
2242           if (publishToCT) {
2243             String redactedLable =
2244                 StringUtils.substring(
2245                     str,
2246                     StringUtils.indexOf(str, "("),
2247                     StringUtils.lastIndexOf(str, ")")
2248                         + 1); // tex. (top.secret).domain.se => redactedLable =
2249                               // (top.secret) aka. including the parentheses
2250             nrOfRecactedLables.add(
2251                 new ASN1Integer(
2252                     StringUtils.countMatches(redactedLable, ".") + 1));
2253           }
2254         } else {
2255           nrOfRecactedLables.add(new ASN1Integer(0));
2256         }
2257       }
2258       if (generalName.getTagNo() == 1) {
2259         final String str =
2260             CertTools.getGeneralNameString(1, generalName.getName());

2262             str,
2263             "\\+")) { // if it contains a '+' character that should be unescaped
2264           // Remove '\' from the email that will end up on the certificate
2265           String certBuilderEmailValue = StringUtils.remove(str, "rfc822name=");
2266           certBuilderEmailValue =
2267               StringUtils.remove(certBuilderEmailValue, '\\');
2268           // Replace the old value with the new
2269           gns[j] = new GeneralName(1, new DERIA5String(certBuilderEmailValue));
2270         }
2271       }
2272     }
2273     ExtensionsGenerator gen = new ExtensionsGenerator();

2278         subAltNameExt.isCritical(),
2279         new GeneralNames(gns));
2280     // If there actually are redacted parts, add the extension containing the
2281     // number of redacted labels to the certificate
2282     if (publishToCT && sanEdited) {
2283       ASN1Encodable seq = new DERSequence(nrOfRecactedLables);
2284       gen.addExtension(
2285           new ASN1ObjectIdentifier(CertTools.ID_CT_REDACTED_DOMAINS),
2286           false,
2287           seq);
2288     }
2289

2451             Extension.authorityKeyIdentifier,
2452             getAuthorityKeyIdentifierCritical(),
2453             aki);

2456             new JcaX509ExtensionUtils(SHA1DigestCalculator.buildSha1Instance());

2458             extensionUtils.createAuthorityKeyIdentifier(
2459                 cryptoToken.getPublicKey(
2460                     getCAToken()
2461                         .getAliasFromPurpose(
2462                             CATokenConstants.CAKEYPURPOSE_CRLSIGN)));

2464             Extension.authorityKeyIdentifier,
2465             getAuthorityKeyIdentifierCritical(),
2466             aki);
2467       }
2468     }

2475           GeneralName accessLocation =
2476               new GeneralName(
2477                   GeneralName.uniformResourceIdentifier, new DERIA5String(url));

2479               new AccessDescription(
2480                   AccessDescription.id_ad_caIssuers, accessLocation));
2481         }

2483     }

2489           Extension.authorityInfoAccess, false, authorityInformationAccess);

2499     // ExpiredCertsOnCRL extension (is always specified as not critical)
2500     // Date format to be used is: yyyyMMddHHmmss

2502     //
2503     // expiredCertsOnCRL EXTENSION ::= {
2504     //   SYNTAX         ExpiredCertsOnCRL
2505     //   IDENTIFIED BY  id-ce-expiredCertsOnCRL
2506     // }
2507     // ExpiredCertsOnCRL ::= GeneralizedTime
2508     // The ExpiredCertsOnCRL CRL extension is not specified by IETF-PKIX. It is
2509     // defined by the ITU-T Recommendation X.509 and
2510     // indicates that a CRL containing this extension will include revocation
2511     // status information for certificates that have
2512     // been already expired. When used, the ExpiredCertsOnCRL contains the date
2513     // on which the CRL starts to keep revocation
2514     // status information for expired certificates (i.e. revocation entries are
2515     // not removed from the CRL for any certificates
2516     // that expire at or after the date contained in the ExpiredCertsOnCRL
2517     // extension).

2519         new ASN1ObjectIdentifier("2.5.29.60");

2539       CRLNumber basecrlnum = new CRLNumber(BigInteger.valueOf(basecrlnumber));

2553                 false,

2555                 false,

2560         // handle the IDP extension and discards the CRL if it is critical.
2561         crlgen.addExtension(
2562             Extension.issuingDistributionPoint,

2565       }

2576 
2577           // According to the RFC, the Freshest CRL extension on a
2578           // CRL must not be marked as critical. Therefore it is
2579           // hardcoded as not critical and is independent of
2580           // getCrlDistributionPointOnCrlCritical().
2581           crlgen.addExtension(Extension.freshestCRL, false, ext);

2583       }

2589           "Signing CRL. Free memory=" + Runtime.getRuntime().freeMemory());
2590     }
2591     final String alias =
2592         getCAToken().getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CRLSIGN);
2593     try {
2594       final ContentSigner signer =
2595           new BufferingContentSigner(
2596               new JcaContentSignerBuilder(sigAlg)
2597                   .setProvider(cryptoToken.getSignProviderName())
2598                   .build(cryptoToken.getPrivateKey(alias)),
2599               20480);
2600       crl = crlgen.build(signer);
2601     } catch (OperatorCreationException e) {
2602       // Very fatal error
2603       throw new RuntimeException("Can not create Jca content signer: ", e);
2604     }
2605     if (LOG.isDebugEnabled()) {
2606       LOG.debug(
2607           "Finished signing CRL. Free memory="
2608               + Runtime.getRuntime().freeMemory());
2609     }

2612     // If we can not verify the issued CRL using the CA certificate we don't
2613     // want to issue this CRL
2614     // because something is wrong...
2615     final PublicKey verifyKey;
2616     if (cacert != null) {
2617       verifyKey = cacert.getPublicKey();
2618       if (LOG.isTraceEnabled()) {
2619         LOG.trace("Got the verify key from the CA certificate.");
2620       }
2621     } else {
2622       verifyKey = cryptoToken.getPublicKey(alias);
2623       if (LOG.isTraceEnabled()) {
2624         LOG.trace("Got the verify key from the CA token.");
2625       }
2626     }

2629           CertTools.genContentVerifierProvider(verifyKey);

2635       throw new RuntimeException("Can not create Jca content signer: ", e);
2636     } catch (CertException e) {
2637       throw new SignatureException(e.getMessage(), e);
2638     }
2639     if (LOG.isDebugEnabled()) {
2640       LOG.debug(
2641           "Returning CRL. Free memory=" + Runtime.getRuntime().freeMemory());
2642     }
2643     return crl;
2644   }
