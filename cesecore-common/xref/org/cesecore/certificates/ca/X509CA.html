209     /** Creates a new instance of CA, this constructor should be used when a new CA is created 
210      * @param cainfo Info*/
211     public X509CA(final X509CAInfo cainfo) {
212         super(cainfo);
213         //Verify integrity if caInfo, either one SubjectDN or SubjectAltName needs to be filled in
214         if(StringUtils.isEmpty(DNFieldsUtil.removeAllEmpties(cainfo.getSubjectDN())) && StringUtils.isEmpty(cainfo.getSubjectAltName())) {
215             throw new IllegalArgumentException("Subject DN and Alt Name can't both be blank for an X509 CA.");
216         }
217         data.put(POLICIES, cainfo.getPolicies());
218         data.put(SUBJECTALTNAME, cainfo.getSubjectAltName());
219         setUseAuthorityKeyIdentifier(cainfo.getUseAuthorityKeyIdentifier());
220         setAuthorityKeyIdentifierCritical(cainfo.getAuthorityKeyIdentifierCritical());
221         setUseCRLNumber(cainfo.getUseCRLNumber());
222         setCRLNumberCritical(cainfo.getCRLNumberCritical());
223         setDefaultCRLDistPoint(cainfo.getDefaultCRLDistPoint());
224         setDefaultCRLIssuer(cainfo.getDefaultCRLIssuer());
225         setCADefinedFreshestCRL(cainfo.getCADefinedFreshestCRL());
226         setDefaultOCSPServiceLocator(cainfo.getDefaultOCSPServiceLocator());
227         setUseUTF8PolicyText(cainfo.getUseUTF8PolicyText());
228         setUsePrintableStringSubjectDN(cainfo.getUsePrintableStringSubjectDN());
229         setUseLdapDNOrder(cainfo.getUseLdapDnOrder());
230         setUseCrlDistributionPointOnCrl(cainfo.getUseCrlDistributionPointOnCrl());
231         setCrlDistributionPointOnCrlCritical(cainfo.getCrlDistributionPointOnCrlCritical());
232         setKeepExpiredCertsOnCRL(cainfo.getKeepExpiredCertsOnCRL());
233         setCmpRaAuthSecret(cainfo.getCmpRaAuthSecret());
234         // CA Issuer URI to put in CRLs (RFC5280 section 5.2.7, not the URI to put in certs
235         setAuthorityInformationAccess(cainfo.getAuthorityInformationAccess());
236         setCertificateAiaDefaultCaIssuerUri(cainfo.getCertificateAiaDefaultCaIssuerUri());
237         setNameConstraintsPermitted(cainfo.getNameConstraintsPermitted());
238         setNameConstraintsExcluded(cainfo.getNameConstraintsExcluded());
239         data.put(CA.CATYPE, Integer.valueOf(CAInfo.CATYPE_X509));
240         data.put(VERSION, Float.valueOf(LATEST_VERSION));
241         setCaSerialNumberOctetSize(cainfo.getCaSerialNumberOctetSize());
242     }

698         List<Certificate> nextChain = getRolloverCertificateChain();
699         if (nextChain == null) {
700             log.debug("CA does not have a rollover chain, returning empty PKCS#7");
701             nextChain = Collections.emptyList();
702         } else if (nextChain.isEmpty()) {

704         }

706         ArrayList<X509CertificateHolder> certList = new ArrayList<>();

709                 certList.add(new JcaX509CertificateHolder((X509Certificate) certificate));

719             CMSSignedDataGenerator gen = new CMSSignedDataGenerator();

724                 log.debug(msg1);

726             }

728             try {
729                 ContentSigner contentSigner = new JcaContentSignerBuilder(signatureAlgorithmName).setProvider(cryptoToken.getSignProviderName()).build(privateKey);
730                 JcaDigestCalculatorProviderBuilder calculatorProviderBuilder = new JcaDigestCalculatorProviderBuilder().setProvider(BouncyCastleProvider.PROVIDER_NAME);
731                 JcaSignerInfoGeneratorBuilder builder = new JcaSignerInfoGeneratorBuilder(calculatorProviderBuilder.build());
732                 gen.addSignerInfoGenerator(builder.build(contentSigner, (X509Certificate) getCACertificate()));
733             } catch (OperatorCreationException e) {
734                 throw new IllegalStateException("BouncyCastle failed in creating signature provider.", e);
735             } catch (CertificateEncodingException e) {
736                 throw new IllegalStateException(e);
737             }
738             gen.addCertificates(new CollectionStore<>(certList));
739             CMSSignedData s = null;
740             CAToken catoken = getCAToken();
741             if (catoken != null && !(cryptoToken instanceof NullCryptoToken)) {
742                 log.debug("createPKCS7Rollover: Provider=" + cryptoToken.getSignProviderName() + " using algorithm "

749                 //   ObjectIdentifier(1.2.840.113549.1.7.1)

755                 log.debug(msg1);
756                 throw new SignRequestSignatureException(msg1);
757             }
758             return s.getEncoded();
759         } catch (CryptoTokenOfflineException e) {
760             throw new IllegalStateException(e);
761         } catch (CMSException e) {
762             throw new IllegalStateException(e);
763         } catch (IOException e) {
764             throw new IllegalStateException("Failed to encode CMS data", e);
765         }
766     }

1231                 if (log.isDebugEnabled()) {

1233                 }

1260                     wildcardExt.add(id);
1261                     continue;   
1262                 }

1264                 // from the request, if AllowExtensionOverride is enabled.

1276                 }

1300                             }

1340                     final byte[] value = extension.getExtnValue().getOctets();

1502         }

1507         }
1508         if (log.isDebugEnabled()) {
1509             log.debug("X509CA: generated certificate, CA " + this.getCAId() + " for DN: " + subject.getCertificateDN());

1512     }

1527     /**
1528      * Constructs the SubjectAlternativeName extension that will end up on the generated certificate.
1529      *
1530      * If the DNS values in the subjectAlternativeName extension contain parentheses to specify labels that should be redacted, the parentheses are removed and another extension
1531      * containing the number of redacted labels is added.
1532      *
1533      * @param subAltNameExt Alt Name
1534      * @param publishToCT boolean
1535      * @return An extension generator containing the SubjectAlternativeName extension and an extension holding the number of redacted labels if the certificate is to be published
1536      * to a CTLog
1537      * @throws IOException if read fails
1538      */
1539     protected ExtensionsGenerator getSubjectAltNameExtensionForCert(Extension subAltNameExt, boolean publishToCT) throws IOException {
1540         GeneralNames names = CertTools.getGeneralNamesFromExtension(subAltNameExt);
1541         GeneralName[] gns = names.getNames();
1542         boolean sanEdited = false;
1543         ASN1EncodableVector nrOfRecactedLables = new ASN1EncodableVector();
1544         for (int j = 0; j<gns.length; j++) {
1545             GeneralName generalName = gns[j];
1546             // Look for DNS name
1547             if (generalName.getTagNo() == 2) {
1548                 final String str = CertTools.getGeneralNameString(2, generalName.getName());
1549                 if(StringUtils.contains(str, "(") && StringUtils.contains(str, ")") ) { // if it contains parts that should be redacted
1550                     // Remove the parentheses from the SubjectAltName that will end up on the certificate
1551                     String certBuilderDNSValue = StringUtils.remove(str, "dNSName=");
1552                     certBuilderDNSValue = StringUtils.remove(certBuilderDNSValue, '(');
1553                     certBuilderDNSValue = StringUtils.remove(certBuilderDNSValue, ')');
1554                     // Replace the old value with the new
1555                     gns[j] = new GeneralName(2, new DERIA5String(certBuilderDNSValue));
1556                     sanEdited = true;
1557                     if (publishToCT) {
1558                         String redactedLable = StringUtils.substring(str, StringUtils.indexOf(str, "("), StringUtils.lastIndexOf(str, ")")+1); // tex. (top.secret).domain.se => redactedLable = (top.secret) aka. including the parentheses
1559                         nrOfRecactedLables.add(new ASN1Integer(StringUtils.countMatches(redactedLable, ".")+1));
1560                     }
1561                 } else {
1562                     nrOfRecactedLables.add(new ASN1Integer(0));
1563                 }
1564             }
1565             if(generalName.getTagNo() == 1) {
1566                 final String str = CertTools.getGeneralNameString(1, generalName.getName());
1567                 if(StringUtils.contains(str, "\\+") ) { // if it contains a '+' character that should be unescaped
1568                     // Remove '\' from the email that will end up on the certificate
1569                     String certBuilderEmailValue = StringUtils.remove(str, "rfc822name=");
1570                     certBuilderEmailValue = StringUtils.remove(certBuilderEmailValue, '\\');
1571                     // Replace the old value with the new
1572                     gns[j] = new GeneralName(1, new DERIA5String(certBuilderEmailValue));
1573                 }
1574             }
1575         }
1576         ExtensionsGenerator gen = new ExtensionsGenerator();
1577         // Use the GeneralName from original altName in order to not re-encode anything
1578         gen.addExtension(Extension.subjectAlternativeName, subAltNameExt.isCritical(), new GeneralNames(gns));
1579         // If there actually are redacted parts, add the extension containing the number of redacted labels to the certificate
1580         if(publishToCT && sanEdited) {
1581             ASN1Encodable seq = new DERSequence(nrOfRecactedLables);
1582             gen.addExtension(new ASN1ObjectIdentifier(CertTools.id_ct_redacted_domains), false, seq);
1583         }
1584 
1585         return gen;
1586     }

1631     /**
1632      * Generate a CRL or a deltaCRL
1633      *
1634      * @param cryptoToken the cryptoToken with keys used to sign the CRL
1635      * @param certs list of revoked certificates
1636      * @param crlPeriod the validity period of the generated CRL, the CRLs nextUpdate will be set to (currentTimeMillis + crlPeriod)
1637      * @param crlnumber CRLNumber for this CRL
1638      * @param isDeltaCRL true if we should generate a DeltaCRL
1639      * @param basecrlnumber caseCRLNumber for a delta CRL, use 0 for full CRLs
1640      * @return X509CRLHolder with the generated CRL
1641      * @throws CryptoTokenOfflineException If offline
1642      * @throws IllegalCryptoTokenException If token is invalid
1643      * @throws IOException If read/write fails
1644      * @throws SignatureException If signature is invalid
1645      * @throws NoSuchProviderException If provider cannot be found
1646      * @throws InvalidKeyException If key is invalid
1647      * @throws CRLException if CRL is invalid
1648      * @throws NoSuchAlgorithmException If algorithm is invalid
1649      */
1650     private X509CRLHolder generateCRL(CryptoToken cryptoToken, Collection<RevokedCertInfo> certs, long crlPeriod, int crlnumber, boolean isDeltaCRL, int basecrlnumber)
1651             throws CryptoTokenOfflineException, IllegalCryptoTokenException, IOException, SignatureException, NoSuchProviderException,
1652             InvalidKeyException, CRLException, NoSuchAlgorithmException {
1653         final String sigAlg = getCAInfo().getCAToken().getSignatureAlgorithm();
1654 
1655         if (log.isDebugEnabled()) {
1656             log.debug("generateCRL(" + certs.size() + ", " + crlPeriod + ", " + crlnumber + ", " + isDeltaCRL + ", " + basecrlnumber);
1657         }
1658 
1659         // Make DNs
1660         final X509Certificate cacert = (X509Certificate) getCACertificate();
1661         final X500Name issuer;
1662         if (cacert == null) {
1663             // This is an initial root CA, since no CA-certificate exists
1664             // (I don't think we can ever get here!!!)
1665             final X500NameStyle nameStyle;
1666             if (getUsePrintableStringSubjectDN()) {
1667                 nameStyle = PrintableStringNameStyle.INSTANCE;
1668             } else {
1669                 nameStyle = CeSecoreNameStyle.INSTANCE;
1670             }
1671             issuer = CertTools.stringToBcX500Name(getSubjectDN(), nameStyle, getUseLdapDNOrder());
1672         } else {
1673             issuer = X500Name.getInstance(cacert.getSubjectX500Principal().getEncoded());
1674         }
1675         final Date thisUpdate = new Date();
1676         final Date nextUpdate = new Date();
1677         nextUpdate.setTime(nextUpdate.getTime() + crlPeriod);
1678         final X509v2CRLBuilder crlgen = new X509v2CRLBuilder(issuer, thisUpdate);
1679         crlgen.setNextUpdate(nextUpdate);
1680         if (certs != null) {
1681             if (log.isDebugEnabled()) {
1682                 log.debug("Adding "+certs.size()+" revoked certificates to CRL. Free memory="+Runtime.getRuntime().freeMemory());
1683             }
1684             for (final RevokedCertInfo certinfo : certs) {
1685                 crlgen.addCRLEntry(certinfo.getUserCertificate(), certinfo.getRevocationDate(), certinfo.getReason());
1686             }
1687             if (log.isDebugEnabled()) {
1688                 log.debug("Finished adding "+certs.size()+" revoked certificates to CRL. Free memory="+Runtime.getRuntime().freeMemory());
1689             }
1690         }
1691 
1692 
1693         // Authority key identifier
1694         if (getUseAuthorityKeyIdentifier() == true) {
1695             byte[] caSkid = (cacert != null ? CertTools.getSubjectKeyId(cacert) : null);
1696             if (caSkid != null) {
1697                 // Use subject key id from CA certificate
1698                 AuthorityKeyIdentifier aki = new AuthorityKeyIdentifier(caSkid);
1699                 crlgen.addExtension(Extension.authorityKeyIdentifier, getAuthorityKeyIdentifierCritical(), aki);
1700             } else {
1701                 JcaX509ExtensionUtils extensionUtils = new JcaX509ExtensionUtils(SHA1DigestCalculator.buildSha1Instance());
1702                 AuthorityKeyIdentifier aki = extensionUtils.createAuthorityKeyIdentifier(cryptoToken.getPublicKey(getCAToken().getAliasFromPurpose(
1703                         CATokenConstants.CAKEYPURPOSE_CRLSIGN)));
1704                 crlgen.addExtension(Extension.authorityKeyIdentifier, getAuthorityKeyIdentifierCritical(), aki);
1705             }
1706         }
1707 
1708         // Authority Information Access
1709         final ASN1EncodableVector accessList = new ASN1EncodableVector();
1710         if (getAuthorityInformationAccess() != null) {
1711             for(String url :  getAuthorityInformationAccess()) {
1712                 if(StringUtils.isNotEmpty(url)) {
1713                     GeneralName accessLocation = new GeneralName(GeneralName.uniformResourceIdentifier, new DERIA5String(url));
1714                     accessList.add(new AccessDescription(AccessDescription.id_ad_caIssuers, accessLocation));
1715                 }
1716             }
1717         }
1718         if(accessList.size() > 0) {
1719             AuthorityInformationAccess authorityInformationAccess = AuthorityInformationAccess.getInstance(new DERSequence(accessList));
1720             // "This CRL extension MUST NOT be marked critical." according to rfc4325
1721             crlgen.addExtension(Extension.authorityInfoAccess, false, authorityInformationAccess);
1722         }
1723 
1724         // CRLNumber extension
1725         if (getUseCRLNumber() == true) {
1726             CRLNumber crlnum = new CRLNumber(BigInteger.valueOf(crlnumber));
1727             crlgen.addExtension(Extension.cRLNumber, this.getCRLNumberCritical(), crlnum);
1728         }
1729 
1730         // ExpiredCertsOnCRL extension (is always specified as not critical)
1731         // Date format to be used is: yyyyMMddHHmmss
1732         // https://www.itu.int/ITU-T/formal-language/itu-t/x/x509/2005/CertificateExtensions.html
1733         //
1734         // expiredCertsOnCRL EXTENSION ::= {
1735         //   SYNTAX         ExpiredCertsOnCRL
1736         //   IDENTIFIED BY  id-ce-expiredCertsOnCRL
1737         // }
1738         // ExpiredCertsOnCRL ::= GeneralizedTime
1739         // The ExpiredCertsOnCRL CRL extension is not specified by IETF-PKIX. It is defined by the ITU-T Recommendation X.509 and
1740         // indicates that a CRL containing this extension will include revocation status information for certificates that have
1741         // been already expired. When used, the ExpiredCertsOnCRL contains the date on which the CRL starts to keep revocation
1742         // status information for expired certificates (i.e. revocation entries are not removed from the CRL for any certificates
1743         // that expire at or after the date contained in the ExpiredCertsOnCRL extension).
1744         final ASN1ObjectIdentifier ExpiredCertsOnCRL = new ASN1ObjectIdentifier("2.5.29.60");
1745         boolean keepexpiredcertsoncrl = getKeepExpiredCertsOnCRL();
1746         if(keepexpiredcertsoncrl) {
1747             SimpleDateFormat sdf = new SimpleDateFormat();
1748             final String GMTdatePattern = "yyyyMMddHHmmss";
1749             sdf.setTimeZone(new SimpleTimeZone(0, "GMT"));
1750             sdf.applyPattern(GMTdatePattern);
1751             // For now force parameter with date equals NotBefore of CA certificate, or now
1752             final Date keepDate = cacert != null ? cacert.getNotBefore() : new Date();
1753             crlgen.addExtension(ExpiredCertsOnCRL, false, new DERGeneralizedTime(keepDate));
1754             if (log.isDebugEnabled()) {
1755                 log.debug("ExpiredCertsOnCRL extension added to CRL. Keep date: "+keepDate);
1756             }
1757         }
1758 
1759         if (isDeltaCRL) {
1760             // DeltaCRLIndicator extension
1761             CRLNumber basecrlnum = new CRLNumber(BigInteger.valueOf(basecrlnumber));
1762             crlgen.addExtension(Extension.deltaCRLIndicator, true, basecrlnum);
1763         }
1764         // CRL Distribution point URI and Freshest CRL DP
1765         if (getUseCrlDistributionPointOnCrl()) {
1766             String crldistpoint = getDefaultCRLDistPoint();
1767             List<DistributionPoint> distpoints = generateDistributionPoints(crldistpoint);
1768 
1769             if (distpoints.size() > 0) {
1770                 IssuingDistributionPoint idp = new IssuingDistributionPoint(distpoints.get(0).getDistributionPoint(), false, false, null, false,
1771                         false);
1772 
1773                 // According to the RFC, IDP must be a critical extension.
1774                 // Nonetheless, at the moment, Mozilla is not able to correctly
1775                 // handle the IDP extension and discards the CRL if it is critical.
1776                 crlgen.addExtension(Extension.issuingDistributionPoint, getCrlDistributionPointOnCrlCritical(), idp);
1777             }
1778 
1779             if (!isDeltaCRL) {
1780                 String crlFreshestDP = getCADefinedFreshestCRL();
1781                 List<DistributionPoint> freshestDistPoints = generateDistributionPoints(crlFreshestDP);
1782                 if (freshestDistPoints.size() > 0) {
1783                     CRLDistPoint ext = new CRLDistPoint(freshestDistPoints.toArray(new DistributionPoint[freshestDistPoints.size()]));
1784 
1785                     // According to the RFC, the Freshest CRL extension on a
1786                     // CRL must not be marked as critical. Therefore it is
1787                     // hardcoded as not critical and is independent of
1788                     // getCrlDistributionPointOnCrlCritical().
1789                     crlgen.addExtension(Extension.freshestCRL, false, ext);
1790                 }
1791 
1792             }
1793         }
1794 
1795         final X509CRLHolder crl;
1796         if (log.isDebugEnabled()) {
1797             log.debug("Signing CRL. Free memory="+Runtime.getRuntime().freeMemory());
1798         }
1799         final String alias = getCAToken().getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CRLSIGN);
1800         try {
1801             final ContentSigner signer = new BufferingContentSigner(new JcaContentSignerBuilder(sigAlg).setProvider(cryptoToken.getSignProviderName()).build(cryptoToken.getPrivateKey(alias)), 20480);
1802             crl = crlgen.build(signer);
1803         } catch (OperatorCreationException e) {
1804             // Very fatal error
1805             throw new RuntimeException("Can not create Jca content signer: ", e);
1806         }
1807         if (log.isDebugEnabled()) {
1808             log.debug("Finished signing CRL. Free memory="+Runtime.getRuntime().freeMemory());
1809         }
1810 
1811         // Verify using the CA certificate before returning
1812         // If we can not verify the issued CRL using the CA certificate we don't want to issue this CRL
1813         // because something is wrong...
1814         final PublicKey verifyKey;
1815         if (cacert != null) {
1816             verifyKey = cacert.getPublicKey();
1817             if (log.isTraceEnabled()) {
1818                 log.trace("Got the verify key from the CA certificate.");
1819             }
1820         } else {
1821             verifyKey = cryptoToken.getPublicKey(alias);
1822             if (log.isTraceEnabled()) {
1823                 log.trace("Got the verify key from the CA token.");
1824             }
1825         }
1826         try {
1827             final ContentVerifierProvider verifier = CertTools.genContentVerifierProvider(verifyKey);
1828             if (!crl.isSignatureValid(verifier)) {
1829                 throw new SignatureException("Error verifying CRL to be returned.");
1830             }
1831         } catch (OperatorCreationException e) {
1832             // Very fatal error
1833             throw new RuntimeException("Can not create Jca content signer: ", e);
1834         } catch (CertException e) {
1835             throw new SignatureException(e.getMessage(), e);
1836         }
1837         if (log.isDebugEnabled()) {
1838             log.debug("Returning CRL. Free memory="+Runtime.getRuntime().freeMemory());
1839         }
1840         return crl;
1841     }

1968             		setCertificateAiaDefaultCaIssuerUri( new ArrayList<String>());
1969             	}
1970             }
1971             // v22, 'encodedValidity' is derived by the former long value!
1972             if (null == data.get(ENCODED_VALIDITY)  && null != data.get(VALIDITY)) {
1973                 setEncodedValidity(getEncodedValidity());
1974             }
1975             // v23 'keyValidators' new empty list.

1977                 setValidators(new ArrayList<Integer>());
1978             }
1979             // v24 'serial number octet size' assign configured value (or default value if not configured)

1992     @SuppressWarnings({ "rawtypes", "deprecation" })
1993     @Override
1994     public boolean upgradeExtendedCAServices() {
1995         boolean retval = false;
1996         // call upgrade, if needed, on installed CA services
1997         Collection<Integer> externalServiceTypes = getExternalCAServiceTypes();
1998         if (!CesecoreConfiguration.getCaKeepOcspExtendedService() && externalServiceTypes.contains(ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE)) {
1999             //This type has been removed, so remove it from any CAs it's been added to as well.
2000             externalServiceTypes.remove(ExtendedCAServiceTypes.TYPE_OCSPEXTENDEDSERVICE);
2001             data.put(EXTENDEDCASERVICES, externalServiceTypes);
2002             retval = true;
2003         }
2004 
2005         for (Integer type : externalServiceTypes) {
2006             ExtendedCAService service = getExtendedCAService(type);
2007             if (service != null) {
2008                 if (Float.compare(service.getLatestVersion(), service.getVersion()) != 0) {
2009                     retval = true;
2010                     service.upgrade();
2011                     setExtendedCAServiceData(service.getExtendedCAServiceInfo().getType(), (HashMap) service.saveData());
2012                 } else if (service.isUpgraded()) {
2013                     // Also return true if the service was automatically upgraded by a UpgradeableDataHashMap.load, which calls upgrade automagically.
2014                     retval = true;
2015                     setExtendedCAServiceData(service.getExtendedCAServiceInfo().getType(), (HashMap) service.saveData());
2016                 }
2017             } else {
2018                 log.error("Extended service is null, can not upgrade service of type: " + type);
2019             }
2020         }
2021         return retval;
2022     }

2024     /**
2025      * Encryption method used to encrypt a key pair using a CA
2026      *
2027      * @param cryptoToken the crypto token where the encryption key is
2028      * @param alias the alias of the key on the crypto token to use for encryption
2029      * @param keypair the data to encrypt
2030      * @return encrypted data
2031      * @throws CryptoTokenOfflineException If crypto token is off-line so encryption key can not be used.
2032      */
2033     public static byte[] encryptKeys(final CryptoToken cryptoToken, final String alias, final KeyPair keypair) throws CryptoTokenOfflineException {
2034         try {
2035             ByteArrayOutputStream baos = new ByteArrayOutputStream();
2036             ObjectOutputStream os = new ObjectOutputStream(baos);
2037             os.writeObject(keypair);
2038             CMSEnvelopedDataGenerator edGen = new CMSEnvelopedDataGenerator();
2039             CMSEnvelopedData ed;
2040             // Creating the KeyId may just throw an exception, we will log this but store the cert and ignore the error
2041             final PublicKey pk = cryptoToken.getPublicKey(alias);
2042             byte[] keyId = KeyTools.createSubjectKeyId(pk).getKeyIdentifier();
2043             edGen.addRecipientInfoGenerator(new JceKeyTransRecipientInfoGenerator(keyId, pk));
2044             JceCMSContentEncryptorBuilder jceCMSContentEncryptorBuilder = new JceCMSContentEncryptorBuilder(NISTObjectIdentifiers.id_aes256_CBC).setProvider(BouncyCastleProvider.PROVIDER_NAME);
2045             ed = edGen.generate(new CMSProcessableByteArray(baos.toByteArray()), jceCMSContentEncryptorBuilder.build());
2046             log.info("Encrypted keys using key alias '"+alias+"' from Crypto Token "+cryptoToken.getId());
2047             return ed.getEncoded();
2048         } catch (IOException | CMSException e) {
2049             throw new IllegalStateException("Failed to encrypt keys: " + e.getMessage(), e);
2050         }
2051     }

2053     /**
2054      * Decryption method used to decrypt a key pair using a CA
2055      *
2056      * @param cryptoToken the crypto token where the decryption key is
2057      * @param alias the alias of the key on the crypto token to use for decryption
2058      * @param data the data to decrypt
2059      * @return a KeyPair
2060      * @throws CryptoTokenOfflineException If crypto token is off-line so decryption key can not be used.
2061      * @throws IOException In case reading/writing data streams failed during decryption, or parsing decrypted data into KeyPair.
2062      */
2063     public static KeyPair decryptKeys(final CryptoToken cryptoToken, final String alias, final byte[] data) throws IOException, CryptoTokenOfflineException {
2064         try {
2065             CMSEnvelopedData ed = new CMSEnvelopedData(data);
2066             RecipientInformationStore recipients = ed.getRecipientInfos();
2067             RecipientInformation recipient = recipients.getRecipients().iterator().next();
2068             ObjectInputStream ois = null;
2069             JceKeyTransEnvelopedRecipient rec = new JceKeyTransEnvelopedRecipient(cryptoToken.getPrivateKey(alias));
2070             rec.setProvider(cryptoToken.getEncProviderName());
2071             rec.setContentProvider(BouncyCastleProvider.PROVIDER_NAME);
2072             // Option we must set to prevent Java PKCS#11 provider to try to make the symmetric decryption in the HSM,
2073             // even though we set content provider to BC. Symm decryption in HSM varies between different HSMs and at least for this case is known
2074             // to not work in SafeNet Luna (JDK behavior changed in JDK 7_75 where they introduced imho a buggy behavior)
2075             rec.setMustProduceEncodableUnwrappedKey(true);
2076             byte[] recdata = recipient.getContent(rec);
2077             ois = new ObjectInputStream(new ByteArrayInputStream(recdata));
2078             log.info("Decrypted keys using key alias '"+alias+"' from Crypto Token "+cryptoToken.getId());
2079             return (KeyPair) ois.readObject();
2080         } catch (ClassNotFoundException e) {
2081             throw new IOException("Could not deserialize key pair after decrypting it due to missing class: " + e.getMessage(), e);
2082         } catch (CMSException e) {
2083             throw new IOException("Could not parse encrypted data: " + e.getMessage(), e);
2084         }
2085     }
