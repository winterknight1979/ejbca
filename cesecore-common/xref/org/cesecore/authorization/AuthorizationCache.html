28  /**
29   * Cache of the authorization granted to different AuthenticationTokens.
30   *
31   * <p>Features: - Concurrent cache misses for the same AuthenticationToken will
32   * only lead - to a single call-back while the other threads wait - Never return
33   * stale entries (when signaled that newer data - might be available) - Supports
34   * background reload via the rebuild(...) method - which also purges unused
35   * entries
36   *
37   * @version $Id: AuthorizationCache.java 25694 2017-04-13 15:28:04Z jeklund $
38   */
39  public enum AuthorizationCache {
40    /** Default instance. */
41    INSTANCE,
42    /** RA instance. */
43    RAINSTANCE;
44  
45    /** Logger. */
46    private final Logger log = Logger.getLogger(AuthorizationCache.class);
47  
48    /**
49     * The access available to an authentication token and corresponding version
50     * of the authorization systems updateNumber.
51     */
52    public static class AuthorizationResult {
53      /** Rules. */
54      private final HashMap<String, Boolean> accessRules;
55      /** Update number. */
56      private final int updateNumber;
57  
58      /**
59       * Constructor.
60       *
61       * @param theAccessRules Rules
62       * @param theUpdateNumber Update number.
63       */
64      public AuthorizationResult(
65          final HashMap<String, Boolean> theAccessRules,
66          final int theUpdateNumber) {
67        this.accessRules = theAccessRules;
68        this.updateNumber = theUpdateNumber;
69      }
70      /** @return rules */
71      public HashMap<String, Boolean> getAccessRules() {
72        return accessRules;
73      }
74      /** @return update number */
75      public int getUpdateNumeber() {
76        return updateNumber;
77      }
78    }

80    /** Call-back interface for loading access rules on cache miss. */
81    public interface AuthorizationCacheCallback {
82      /**
83       * @param authenticationToken Authentication
84       * @return the access rules and corresponding update number for the
85       *     specified authenticationToken
86       * @throws AuthenticationFailedException If authentication fails
87       */
88      AuthorizationResult loadAuthorization(
89          AuthenticationToken authenticationToken)
90          throws AuthenticationFailedException;
91  
92      /**
93       * @return the number of milliseconds to keep cache entries for after an
94       *     authentication token was last seen
95       */
96      long getKeepUnusedEntriesFor();
97  
98      /**
99       * Invoked by cache on first cache miss to start listening to authorization
100      * updates.
101      *
102      * @param authorizationCacheReloadListener Listener
103      */
104     void subscribeToAuthorizationCacheReload(
105         AuthorizationCacheReloadListener authorizationCacheReloadListener);
106   }
107 
108   private class AuthorizationCacheEntry {
109     /** Rules. */
110     private HashMap<String, Boolean> accessRules;
111     /** Update. */
112     private int updateNumber = 0;
113     /** Last used. */
114     private long timeOfLastUse = 0L;
115     /** Auth token. */
116     private AuthenticationToken authenticationToken;
117     /** Countdown. */
118     private final CountDownLatch countDownLatch = new CountDownLatch(1);
119   }
120 
121   /** Cache. */
122   private ConcurrentHashMap<String, AuthorizationCacheEntry> cacheMap =
123       new ConcurrentHashMap<>();
124 
125   /** Last update. */
126   private AtomicInteger latestUpdateNumber = new AtomicInteger(0);
127 
128   /** True if listener has been registered. */
129   private final AtomicBoolean authorizationCacheReloadListenerRegistered =
130       new AtomicBoolean(false);
131 
132   /** Reload listener. */
133   private final AuthorizationCacheReloadListener
134       authorizationCacheReloadListener =
135           new AuthorizationCacheReloadListener() {
136             @Override
137             public void onReload(final AuthorizationCacheReload event) {
138               setUpdateNumberIfLower(event.getAccessTreeUpdateNumber());
139             }
140 
141             @Override
142             public String getListenerName() {
143               return AuthorizationCache.class.getSimpleName();
144             }
145           };
146 
147   /**
148    * Clear stale cache.
149    *
150    * @param updateNumber Update number
151    */
152   public void clear(final int updateNumber) {
153     setUpdateNumberIfLower(updateNumber);
154     cacheMap.clear();
155   }
156 
157   /**
158    * Clear stale cache.
159    *
160    * @param updateNumber Update number
161    */
162   public void clearWhenStale(final int updateNumber) {
163     if (setUpdateNumberIfLower(updateNumber)) {
164       cacheMap.clear();
165     }
166   }
167 
168   /** Full reset should only be invoked by JUnit tests. */
169   protected void reset() {
170     cacheMap.clear();
171     latestUpdateNumber.set(0);
172     authorizationCacheReloadListenerRegistered.set(false);
173   }
174 
175   /**
176    * Re-build the authorization cache for all entries that been seen recently
177    * (as determined by authorizationCacheCallback.getKeepUnusedEntriesFor()).
178    *
179    * @param authorizationCacheCallback Callback
180    * @param refreshUpdateNumber Update No.
181    */
182   public void refresh(
183       final AuthorizationCacheCallback authorizationCacheCallback,
184       final int refreshUpdateNumber) {
185     // final int refreshUpdateNumber
186     // = authorizationCacheCallback.getUpdateNumber();
187     if (log.isTraceEnabled()) {
188       log.trace(
189           "Starting cache refresh when update number was "
190               + refreshUpdateNumber
191               + ".");
192     }
193     setUpdateNumberIfLower(refreshUpdateNumber);
194     final long purgeUnusedAuthorizationAfter =
195         authorizationCacheCallback.getKeepUnusedEntriesFor();
196     final long now = System.currentTimeMillis();
197     final HashSet<String> existingKeysWhenInvoked =
198         new HashSet<>(cacheMap.keySet());
199     for (final String key : existingKeysWhenInvoked) {
200       final AuthorizationCacheEntry entry = cacheMap.get(key);
201       if (entry != null) {
202         if (entry.updateNumber < latestUpdateNumber.get()) {
203           // Newer access rules might be available
204           if (cacheMap.remove(key, entry)) {
205             if (log.isDebugEnabled()) {
206               log.debug(
207                   "Removed entry for key '"
208                       + key
209                       + "' since its updateNumber was "
210                       + entry.updateNumber
211                       + ".");
212             }
213             // Recalculate the authorization right away if this
214             // AuthenticationToken was seen recently
215             if (entry.timeOfLastUse + purgeUnusedAuthorizationAfter < now) {
216               try {
217                 get(entry.authenticationToken, authorizationCacheCallback);
218               } catch (AuthenticationFailedException e) {
219                 log.debug(
220                     "Unexpected failure during refresh "
221                         + "if authroization cache: "
222                         + e.getMessage());
223               }
224             }
225           }
226         } else if (entry.timeOfLastUse + purgeUnusedAuthorizationAfter < now) {
227           // Remove the unused entry
228           if (cacheMap.remove(key, entry)) {
229             if (log.isDebugEnabled()) {
230               log.debug(
231                   "Removed entry for key '"
232                       + key
233                       + "' since it was last seen "
234                       + ValidityDate.formatAsUTC(entry.timeOfLastUse)
235                       + ".");
236             }
237           }
238         }
239       }
240     }
241   }

243   /**
244    * @param authenticationToken Authentication
245    * @param authorizationCacheCallback Callback
246    * @return the access rules granted to the specified authenticationToken using
247    *     the callback to load them if needed. Never null.
248    * @throws AuthenticationFailedException If authentication fails
249    */
250   public HashMap<String, Boolean> get(
251       final AuthenticationToken authenticationToken,
252       final AuthorizationCacheCallback authorizationCacheCallback)
253       throws AuthenticationFailedException {
254     return getAuthorizationResult(
255             authenticationToken, authorizationCacheCallback)
256         .accessRules;
257   }
258 
259   /**
260    * @param authenticationToken Auth token
261    * @param authorizationCacheCallback Callback
262    * @return the access rules granted to the specified authenticationToken and
263    *     corresponding update number using the callback to load them if needed.
264    *     Never null.
265    * @throws AuthenticationFailedException if authentication fails
266    */
267   public AuthorizationResult getAuthorizationResult(
268       final AuthenticationToken authenticationToken,
269       final AuthorizationCacheCallback authorizationCacheCallback)
270       throws AuthenticationFailedException {
271     if (authenticationToken == null || authorizationCacheCallback == null) {
272       return new AuthorizationResult(new HashMap<String, Boolean>(), 0);
273     }
274     final String key = authenticationToken.getUniqueId();
275     final AuthorizationCacheEntry authorizationCacheEntry =
276         new AuthorizationCacheEntry();
277     AuthorizationCacheEntry ret =
278         cacheMap.putIfAbsent(key, authorizationCacheEntry);
279     if (ret == null) {
280       // Start subscribing to authorization system updates on first
281       // cache miss (which happens on application startup)
282       if (!authorizationCacheReloadListenerRegistered.getAndSet(true)) {
283         authorizationCacheCallback.subscribeToAuthorizationCacheReload(
284             authorizationCacheReloadListener);
285       }
286       ret = authorizationCacheEntry;
287       try {
288         ret.authenticationToken = authenticationToken;
289         final AuthorizationResult authorizationResult =
290             authorizationCacheCallback.loadAuthorization(authenticationToken);
291         ret.updateNumber = authorizationResult.updateNumber;
292         setUpdateNumberIfLower(ret.updateNumber);
293         ret.accessRules = new HashMap<>();
294         if (authorizationResult.accessRules != null) {
295           // Cache a copy of the loaded access rules map
296           ret.accessRules.putAll(authorizationResult.accessRules);
297         }
298       } finally {
299         // Ensure that we release any waiting thread
300         ret.countDownLatch.countDown();
301       }
302       if (log.isDebugEnabled()) {
303         log.debug("Added entry for key '" + key + "'.");
304       }
305     } else {
306       if (log.isTraceEnabled()) {
307         log.trace("Cache hit for key '" + key + "'.");
308       }
309       try {
310         // Block while it is loading (if it is still loading)
311         ret.countDownLatch.await();
312       } catch (InterruptedException e) {
313         Thread.currentThread().interrupt();
314       }
315       // Check if the returned entry is stale
316       if (ret.updateNumber < latestUpdateNumber.get()) {
317         // Trigger an update on next get and recurse
318         if (cacheMap.remove(key, ret)) {
319           if (log.isDebugEnabled()) {
320             log.debug(
321                 "Removed entry for key '"
322                     + key
323                     + "' since its updateNumber was "
324                     + ret.updateNumber
325                     + ".");
326           }
327         }
328         return getAuthorizationResult(
329             authenticationToken, authorizationCacheCallback);
330       }
331       // Don't care about last time of use here, just be happy that
332       // it was found if it was found
333     }
334     // Weak indication of last use, so rebuild can
335     // eventually purge unused entries
336     ret.timeOfLastUse = System.currentTimeMillis();
337     return new AuthorizationResult(ret.accessRules, ret.updateNumber);
338   }
339 
340   /** @return Last Update Number */
341   public int getLastUpdateNumber() {
342     return latestUpdateNumber.get();
343   }
344 
345   /**
346    * Non-blocking atomic update of the last known update number.
347    *
348    * @param readUpdateNumber Update no.
349    * @return true if the number was updated, false if it was already set
350    */
351   private boolean setUpdateNumberIfLower(final int readUpdateNumber) {
352     int current;
353     while ((current = latestUpdateNumber.get()) < readUpdateNumber) {
354       if (latestUpdateNumber.compareAndSet(current, readUpdateNumber)) {
355         if (log.isDebugEnabled()) {
356           log.debug("latestUpdateNumber is now " + readUpdateNumber + ".");
357         }
358         return true;
359       }
360     }
361     return false;
362   }
