13  package org.cesecore.authorization.control;
14  
15  import java.util.Collections;
16  import java.util.HashMap;
17  import java.util.Map;
18  
19  /** @version $Id: AuditLogRules.java 25428 2017-03-09 14:45:59Z jeklund $ */
20  public enum AuditLogRules {
21    /** Base. */
22    BASE("/secureaudit"),
23    /** Configure. */
24    CONFIGURE(BASE.resource() + "/management/manage"),
25    /** Export. */
26    EXPORT_LOGS(BASE.resource() + "/auditor/export"),
27    /** View. */
28    VIEW(BASE.resource() + "/auditor/select"),
29    /** Verify. */
30    VERIFY(BASE.resource() + "/auditor/verify"),
31    /** Log. */
32    LOG(BASE.resource() + "/log"),
33    /** Custom log. */
34    LOG_CUSTOM(BASE.resource() + "/log_custom_events");
35  
36    /** Resource. */
37    private final String resource;
38    /** All resources. */
39    private static final Map<String, String> ALL_RESOURCES = new HashMap<>();
40  
41    static {
42      for (AuditLogRules rule : AuditLogRules.values()) {
43        ALL_RESOURCES.put(rule.resource(), rule.resource());
44      }
45    }
46  
47    /**
48     * Constructor.
49     *
50     * @param aResource Resource
51     */
52    AuditLogRules(final String aResource) {
53      this.resource = aResource;
54    }
55  
56    /** @return resource */
57    public String resource() {
58      return this.resource;
59    }
60  
61    @Override
62    public String toString() {
63      return this.resource;
64    }
65  
66    /** @return Map of all resources */
67    public static Map<String, String> getAllResources() {
68      return Collections.unmodifiableMap(ALL_RESOURCES);
69    }
70  }
