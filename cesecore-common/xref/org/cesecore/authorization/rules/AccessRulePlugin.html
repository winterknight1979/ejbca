13  package org.cesecore.authorization.rules;
14  
15  import java.util.Map;
16  
17  /**
18   * Marker interface to allow access rules to be plugged in.
19   *
20   * @version $Id: AccessRulePlugin.java 25428 2017-03-09 14:45:59Z jeklund $
21   */
22  public interface AccessRulePlugin {
23  
24    /**
25     * @return a map of resources (rules) as map keys and their human readable
26     *     counterpart as value (if available or otherwise the resource again)
27     */
28    Map<String, String> getRules();
29  
30    /** @return a category key this rule set belongs to */
31    String getCategory();
32  }
