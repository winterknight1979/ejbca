13  package org.cesecore.authorization.user;
14  
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  /**
19   * @version $Id: AccessMatchType.java 26811 2017-10-15 03:29:26Z anatom $
20   * 
21   */
22  public enum AccessMatchType {
23  
24      TYPE_UNUSED(0),
25      TYPE_EQUALCASE(1000),
26      /** Case insensitive. Can be used for textual match values, e.g. a Common Name. Do <b>NOT</b> use with serial numbers (a change since 6.8.0)  */
27      TYPE_EQUALCASEINS(1001),
28      @Deprecated
29      TYPE_NOT_EQUALCASE(1002),
30      @Deprecated
31      TYPE_NOT_EQUALCASEINS(1003),
32      @Deprecated
33      TYPE_NONE(1999),
34      /** Type 2000-2005 are old types used from before EJBCA 4, we must expect to find these in the database in old installations, even though we don't want to use them. 
35       * These have no meaning whatsoever in newer installations of EJBCA, and can safely be removed from the database if found (unless we need to run an old installation in parallel). */
36      @Deprecated
37      SPECIALADMIN_PUBLICWEBUSER(2000),
38      @Deprecated
39      SPECIALADMIN_CACOMMANDLINEADMIN(2001),
40      @Deprecated
41      SPECIALADMIN_RAADMIN(2002),
42      @Deprecated
43      SPECIALADMIN_BATCHCOMMANDLINEADMIN(2003),
44      @Deprecated
45      SPECIALADMIN_INTERNALUSER(2004),
46      @Deprecated
47      SPECIALADMIN_NOUSER(2005);
48  
49      private AccessMatchType(int numericValue) {
50          this.numericValue = numericValue;
51      }
52  
53      public int getNumericValue() {
54          return numericValue;
55      }
56  
57      public static AccessMatchType matchFromDatabase(int numericValue) {
58          return databaseLookup.get(numericValue);
59      }
60      
61      public static AccessMatchType matchFromName(String name) {
62          return nameLookup.get(name);
63      }
64  
65      private int numericValue;
66      private static Map<Integer, AccessMatchType> databaseLookup;
67      private static Map<String, AccessMatchType> nameLookup;
68  
69      static {
70          databaseLookup = new HashMap<Integer, AccessMatchType>();
71          nameLookup = new HashMap<String, AccessMatchType>();
72          for (AccessMatchType accessMatchType : AccessMatchType.values()) {
73              databaseLookup.put(accessMatchType.numericValue, accessMatchType);
74              nameLookup.put(accessMatchType.name(), accessMatchType);
75          }
76      }
77  
78  }
