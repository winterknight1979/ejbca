15  import java.io.Serializable;

17  /**
18   * You can have a method that has a parameter of this type with the
19   * &#64;Observes attribute to be notified when the access rules or roles have
20   * been changed. Please make the method &#64;Asynchronous so it doesn't block
21   * the caller (currently
22   * org.cesecore.authorization.cache.AccessTreeUpdateSessionBean#signalForAccessTreeUpdate()
23   * in cesecore-ejb.
24   *
25   * <p>Currently we can't use JEE events, because it requires that we use CDI.
26   * Instead the method
27   * org.cesecore.authorization.cache.AccessTreeUpdateSessionLocal.addReloadEvent(AuthorizationCacheReloadListener)
28   * can be used
29   *
30   * @version $Id: AuthorizationCacheReload.java 25591 2017-03-23 13:13:02Z
31   *     jeklund $
32   */
33  public final class AuthorizationCacheReload implements Serializable {
34  
35    private static final long serialVersionUID = 1L;
36  
37    /** Update number. */
38    private final int accessTreeUpdateNumber;
39  
40    /**
41     * Set tree number.
42     *
43     * @param anAccessTreeUpdateNumber Number
44     */
45    public AuthorizationCacheReload(final int anAccessTreeUpdateNumber) {
46      this.accessTreeUpdateNumber = anAccessTreeUpdateNumber;
47    }
48  
49    /** @return number */
50    public int getAccessTreeUpdateNumber() {
51      return accessTreeUpdateNumber;
52    }
53  }
