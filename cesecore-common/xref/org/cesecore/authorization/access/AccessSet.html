106   /**
107    * Is access to the resource authorised?
108    *
109    * @param resources Resource
110    * @return boolean
111    */
112   public boolean isAuthorized(final String... resources) {
113     // Note that "*SOME" rules are added when the rules for the
114     // AccessSet are built, and don't need to be handled here
115     NEXT_RESOURCE:
116     for (final String resource : resources) {
117       if (resource.charAt(0) != '/') {
118         throw new IllegalArgumentException("Resource must start with /");
119       } else if (resource.length() != 1
120           && resource.charAt(resource.length() - 1) == '/') {
121         throw new IllegalArgumentException("Resource should not end with /");
122       }
123 
124       // Check for exact rule
125       if (set.contains(resource)) {
126         continue NEXT_RESOURCE; // OK. Check next resource
127       }
128 
129       // Check for recursive rules
130       int depth = 0;
131       String parentResource = resource;
132       while (++depth < MAX_DEPTH) { // never split more than 100 times
133         int slash = parentResource.lastIndexOf('/');
134         if (slash == -1) {
135           break;
136         }
137         parentResource = parentResource.substring(0, slash);
138         if (LOG.isTraceEnabled()) {
139           LOG.trace(
140               "Checking for '"
141                   + parentResource
142                   + "/"
143                   + WILDCARD_RECURSIVE
144                   + "'");
145         }
146         if (set.contains(parentResource + "/" + WILDCARD_RECURSIVE)) {
147           continue NEXT_RESOURCE; // OK. Check next resource
148         }
149       }
150       if (depth == MAX_DEPTH && LOG.isDebugEnabled()) {
151         // Recursive rules are always accept rules, so it's safe to
152         // ignore some of them and continue
153         LOG.debug(
154             "Resource had more than 100 components, only the "
155                 + "first 100 were checked for recursive accept access: "
156                 + resource);
157       }
158 
159       if (LOG.isTraceEnabled()) {
160         LOG.trace(
161             "No access rule for "
162                 + resource
163                 + ". Denying access."
164                 + " Number of allowed resources="
165                 + set.size());
166       }
167       return false;
168     }
169     return true; // all resources match
170   }

217   /**
218    * Converts from EJBCA 6.8.0+ access rules to the old version of AccessSet
219    * introduced in EJBCA 6.6.0, for compatibility with old RA clients. The
220    * result of the conversion is written to "set", which is a HashSet of every
221    * single accepted resource enriched with "*SOME" but no "*RECURSIVE".
222    *
223    * <p>Before calling this method, the "accessMap" variable is expected to
224    * contain the 6.8.0+ access rule structure.
225    *
226    * <p>Note: - The legacy set created via this method will not grant access to
227    * a - configured rules that don't exist on the system. - ...and this means
228    * that access to non-existing resources will not - be granted to old RA
229    * clients.
230    *
231    * @param accessRules the EJBCA 6.8.0+ style access rules
232    * @param allResources whole universe of resources that exists
233    * @return an AccessSet of every single accepted resource enriched with
234    *     "*SOME", but no "*RECURSIVE"
235    */
236   public static AccessSet fromAccessRules(
237       final HashMap<String, Boolean> accessRules,
238       final Set<String> allResources) {
239     final Set<String> set = new HashSet<>();
240     for (final String current : allResources) {
241       // De-normalize if needed
242       final String resource =
243           (current.length() > 1 && current.charAt(current.length() - 1) == '/')
244               ? current.substring(0, current.length() - 1)
245               : current;
246       final boolean authorizedToResource =
247           AccessRulesHelper.hasAccessToResource(accessRules, resource);
248       if (authorizedToResource) {
249         set.add(resource);
250         // Check if we have an (integer) ID in the resource
251         final Matcher matcher = ID_IN_RULENAME.matcher(resource);
252         if (matcher.find()) {
253           // Add "*SOME" resource
254           final String someResource =
255               matcher.replaceFirst("/$1/" + WILDCARD_SOME + "$3");
256           set.add(someResource);
257         }
258       }
259     }
260     // Since expect the whole universe of rules to be provided,
261     // there should be no need to add the WILDCARD_RECURSIVE rule
262     return new AccessSet(set);
263   }
