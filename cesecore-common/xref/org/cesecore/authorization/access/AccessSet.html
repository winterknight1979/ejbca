88      public boolean isAuthorized(final String... resources) {
89          // Note that "*SOME" rules are added when the rules for the AccessSet are built, and don't need to be handled here
90          NEXT_RESOURCE: for (final String resource : resources) {
91              if (resource.charAt(0) != '/') {
92                  throw new IllegalArgumentException("Resource must start with /");
93              } else if (resource.length() != 1 && resource.charAt(resource.length() - 1) == '/') {
94                  throw new IllegalArgumentException("Resource should not end with /");
95              }
96  
97              // Check for exact rule
98              if (set.contains(resource)) {
99                  continue NEXT_RESOURCE; // OK. Check next resource
100             }
101 
102             // Check for recursive rules
103             int depth = 0;
104             String parentResource = resource;
105             while (++depth < 100) { // never split more than 100 times
106                 int slash = parentResource.lastIndexOf('/');
107                 if (slash == -1) {
108                     break;
109                 }
110                 parentResource = parentResource.substring(0, slash);
111                 if (log.isTraceEnabled()) {
112                     log.trace("Checking for '"+ parentResource + "/" + WILDCARD_RECURSIVE + "'");
113                 }
114                 if (set.contains(parentResource + "/" + WILDCARD_RECURSIVE)) {
115                     continue NEXT_RESOURCE; // OK. Check next resource
116                 }
117             }
118             if (depth == 100 && log.isDebugEnabled()) {
119                 // Recursive rules are always accept rules, so it's safe to ignore some of them and continue
120                 log.debug("Resource had more than 100 components, only the first 100 were checked for recursive accept access: " + resource);
121             }
122 
123             if (log.isTraceEnabled()) {
124                 log.trace("No access rule for " + resource + ". Denying access. Number of allowed resources=" + set.size());
125             }
126             return false;
127         }
128         return true; // all resources match
129     }

176     /**
177      * Converts from EJBCA 6.8.0+ access rules to the old version of AccessSet introduced in EJBCA 6.6.0,
178      * for compatibility with old RA clients. The result of the conversion is written to "set", which is a
179      * HashSet of every single accepted resource enriched with "*SOME" but no "*RECURSIVE".
180      *
181      * Before calling this method, the "accessMap" variable is expected to contain the 6.8.0+ access rule structure.
182      *
183      * Note:
184      * - The legacy set created via this method will not grant access to a configured rules that don't exist on the system.
185      * - ...and this means that access to non-existing resources will not be granted to old RA clients.
186      *
187      * @param accessRules the EJBCA 6.8.0+ style access rules
188      * @param allResources whole universe of resources that exists
189      * @return an AccessSet of every single accepted resource enriched with "*SOME", but no "*RECURSIVE"
190      */
191     public static AccessSet fromAccessRules(final HashMap<String, Boolean> accessRules, final Set<String> allResources) {
192         final Set<String> set = new HashSet<>();
193         for (final String current : allResources) {
194             // De-normalize if needed
195             final String resource = (current.length()>1 && current.charAt(current.length()-1)=='/') ? current.substring(0, current.length()-1) : current;
196             final boolean authorizedToResource = AccessRulesHelper.hasAccessToResource(accessRules, resource);
197             if (authorizedToResource) {
198                 set.add(resource);
199                 // Check if we have an (integer) ID in the resource
200                 final Matcher matcher = idInRulename.matcher(resource);
201                 if (matcher.find()) {
202                     // Add "*SOME" resource
203                     final String someResource = matcher.replaceFirst("/$1/" + WILDCARD_SOME + "$3");
204                     set.add(someResource);
205                 }
206             }
207         }
208         // Since expect the whole universe of rules to be provided, there should be no need to add the WILDCARD_RECURSIVE rule
209         return new AccessSet(set);
210     }
