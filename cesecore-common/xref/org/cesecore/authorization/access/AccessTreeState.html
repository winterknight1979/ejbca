14  package org.cesecore.authorization.access;
15  
16  /**
17   * Enum adapted from the constants in AccessTreeNode in EJBCA. Represents the
18   * state of an accessTree node.
19   *
20   * @version $Id: AccessTreeState.java 17625 2013-09-20 07:12:06Z netmackan $
21   */
22  public enum AccessTreeState {
23    /** Unknown. */
24    STATE_UNKNOWN(1),
25    /** Accept at top level. */
26    STATE_ACCEPT(2),
27    /** Accept at any level. */
28    STATE_ACCEPT_RECURSIVE(3),
29    /** Decline. */
30    STATE_DECLINE(4);
31  
32    /**
33     * Constructor.
34     *
35     * @param aLegacyNumber number
36     */
37    AccessTreeState(final int aLegacyNumber) {
38      this.legacyNumber = aLegacyNumber;
39    }
40  
41    /** @return legacyNumber. */
42    public int getLegacyNumber() {
43      return legacyNumber;
44    }
45  
46    /** Legacy number. */
47    private int legacyNumber;
48  }
