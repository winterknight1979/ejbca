79    /**
80     * Builds the platform dependent external command array: - field at index 0 is
81     * the interpreter, - field at index 1 is the one and only parameter of the
82     * interpreter, - field at index 2 must contain the complete external command,
83     * including pipes, chains, sub-shells, etc. and is appended later.
84     *
85     * @param cmd the external command or script
86     * @return the command array as list.
87     */
88    protected static List<String> buildShellCommand(final String cmd) {
89      final List<String> result = new ArrayList<String>();
90      if (SystemUtils.IS_OS_WINDOWS) {
91        result.add(WINDOWS_SHELL);
92        result.add(WINDOWS_SHELL_OPTIONS);
93      } else {
94        result.add(UNIX_SHELL);
95        result.add(UNIX_SHELL_OPTIONS);
96      }
97      if (LOG.isDebugEnabled()) {
98        LOG.debug(
99            "Use platform shell command for "
100               + SystemUtils.OS_NAME
101               + " : "
102               + result);
103     }
104     if (result.size() == 2) {
105       result.add(cmd);
106     }
107     return result;
108   }

110   /**
111    * Writes the byte-array to a temporary file or pipes the PEM certificate into
112    * the command, and launches the given external command, see {@link
113    * ExternalProcessTools#launchExternalCommand(String, byte[], boolean,
114    * boolean, boolean, boolean, List, String)}.
115    *
116    * @param cmd The command to run.
117    * @param bytes The buffer with content to write to the file.
118    * @param failOnCode Determines if the method should fail on a non-zero exit
119    *     code.
120    * @param failOnOutput Determines if the method should fail on output to
121    *     standard error.
122    * @param args Added to the command after the temporary files name
123    * @param filePrefix prefix
124    * @return output
125    * @throws ExternalProcessException if the temporary file could not be written
126    *     or the external process fails.
127    */
128   public static List<String> launchExternalCommand(
129       final String cmd,
130       final byte[] bytes,
131       final boolean failOnCode,
132       final boolean failOnOutput,
133       final List<String> args,
134       final String filePrefix)
135       throws ExternalProcessException {
136     return launchExternalCommand(
137         cmd, bytes, failOnCode, failOnOutput, false, false, args, filePrefix);
138   }

140   /**
141    * Writes the byte-array to a temporary file and launches the given external
142    * command with the file as argument at index positional parameter index 1 or
143    * the pipes the PEM certificate into the command. The function will,
144    * depending on its parameters, fail if output to standard error from the
145    * command was detected or the command returns with an non-zero exit code.
146    *
147    * @param cmd The command to run. If the parameter place holder {@link
148    *     #PLACE_HOLDER_CERTIFICATE} is used, the PEM certificate is piped into
149    *     the STDIN of the command (i.e. 'openssl x509 -text -noout %cert%').
150    * @param bytes The buffer with content to write to the file.
151    * @param failOnCode Determines if the method should fail on a non-zero exit
152    *     code.
153    * @param failOnOutput Determines if the method should fail on output to
154    *     standard error.
155    * @param logStdOut if the scripts STDOUT should be logged as info.
156    * @param logErrOut if the scripts ERROUT should be logged as info.
157    * @param arguments Added to the command after the temporary files name
158    * @param filePrefix prefix
159    * @return output
160    * @throws ExternalProcessException if the temporary file could not be written
161    *     or the external process fails.
162    */
163   public static List<String> launchExternalCommand(
164       final String cmd,
165       final byte[] bytes,
166       final boolean failOnCode,
167       final boolean failOnOutput,
168       final boolean logStdOut,
169       final boolean logErrOut,
170       final List<String> arguments,
171       final String filePrefix)
172       throws ExternalProcessException {
173     final long startTime = System.currentTimeMillis();
174     int exitStatus = -1;
175     final List<String> result = new ArrayList<String>();
176     final boolean writeFileToDisk =
177         !arguments.contains(PLACE_HOLDER_CERTIFICATE);
178     File file = null;
179     if (writeFileToDisk) {
180       file = writeTemporaryFileToDisk(bytes, filePrefix, ".tmp");
181     }
182     // Execute external script or command with PEM in STDIN or full path of
183     // temporary file as first argument.
184     String filename = null;
185     try {
186       final List<String> cmdTokens = Arrays.asList(cmd.split("\\s"));
187       // Write file to disk or process place holder with PEM certificates and
188       // build shell command.
189       if (writeFileToDisk) {
190         filename = file.getCanonicalPath();
191         arguments.add(0, filename);
192       } else {
193         // Only works with PEM X.509 certificates at the time as used in
194         // ExternalCommandCertificateValidator (not by CRL publishers).
195         final List<Certificate> certificates = new ArrayList<Certificate>();
196         certificates.add(
197             CertTools.getCertfromByteArray(bytes, Certificate.class));
198         final byte[] testPemBytes =
199             CertTools.getPemFromCertificateChain(certificates);
200         String pemString = new String(testPemBytes);
201         pemString =
202             pemString.substring(
203                 pemString.indexOf(LINE_SEPARATOR) + 1, pemString.length());
204         pemString =
205             pemString.substring(
206                 pemString.indexOf(LINE_SEPARATOR) + 1, pemString.length());
207         if (LOG.isDebugEnabled()) {
208           LOG.debug("Using certificates:\n" + pemString);
209         }
210         arguments.remove(arguments.indexOf(PLACE_HOLDER_CERTIFICATE));
211 
212         if (SystemUtils.IS_OS_WINDOWS) {
213           // Broken. Command cannot be executed.
214           cmdTokens.set(0, "echo \"" + pemString + "\" | " + cmdTokens.get(0));
215           /*
216            * Hack needed for Windows, where Runtime.exec won't consistently
217            *  encapsulate arguments, leading to arguments
218            * containing spaces (such as Subject DNs) sometimes being parsed
219            * as multiple arguments. Bash, on the other hand,
220            * won't parse quote surrounded arguments.
221            */
222           qouteArguments(arguments);
223         } else {
224           cmdTokens.set(
225               0, "echo -n \"" + pemString + "\" | " + cmdTokens.get(0));
226         }
227       }
228       List<String> cmdArray = new ArrayList<String>();
229       cmdArray.addAll(cmdTokens);
230       cmdArray.addAll(arguments);
231       if (!writeFileToDisk) {
232         cmdArray = buildShellCommand(StringUtils.join(cmdArray, " "));
233       }
234       if (LOG.isDebugEnabled()) {
235         LOG.debug(
236             "Process external command for "
237                 + getPlatformString()
238                 + ": "
239                 + cmdArray);
240       }
241       // Launch external process.
242       final Process externalProcess =
243           Runtime.getRuntime()
244               .exec(cmdArray.toArray(new String[] {}), null, null);
245       externalProcess
246           .getOutputStream()
247           .close(); // prevent process from trying to wait for user input (e.g.
248                     // prompt for overwrite, or similar)
249       final BufferedReader stdError =
250           new BufferedReader(
251               new InputStreamReader(externalProcess.getErrorStream()));
252       final BufferedReader stdOut =
253           new BufferedReader(
254               new InputStreamReader(externalProcess.getInputStream()));
255       String line = null;
256       while ((line = stdOut.readLine())
257           != null) { // NOPMD: Required under win32 to avoid lock
258         if (logStdOut) {
259           result.add(STDOUT_PREFIX + line);
260         }
261       }
262       String stdErrorOutput = null;
263       // Check error code and the external applications output to STDERR.
264       exitStatus = externalProcess.waitFor();
265       result.add(0, EXIT_CODE_PREFIX + exitStatus);
266       if (((exitStatus != 0) && failOnCode)
267           || (stdError.ready() && failOnOutput)) {
268         if (writeFileToDisk && file.exists()) {
269           file.delete();
270         }
271         String errTemp = null;
272         while (stdError.ready() && (errTemp = stdError.readLine()) != null) {
273           if (logErrOut) {
274             result.add(ERROUT_PREFIX + errTemp);
275           }
276           if (stdErrorOutput == null) {
277             stdErrorOutput = errTemp;
278           } else {
279             stdErrorOutput += "\n" + errTemp;
280           }
281         }
282         String msg =
283             INTRES.getLocalizedMessage("process.errorexternalapp", cmd);
284         if (stdErrorOutput != null) {
285           msg += " - " + stdErrorOutput + " - " + filename;
286         }
287         throw new ExternalProcessException(msg, result);
288       }
289     } catch (CertificateParsingException
290         | CertificateEncodingException
291             e) { // Should never happen (is only used for certificates not for
292                  // CRL.)
293       throw new ExternalProcessException(
294           "Certificate could not parsed or encoded." + cmd, e, result);
295     } catch (IOException e) { // if the command could not be found
296       result.add(0, EXIT_CODE_PREFIX + exitStatus);
297       if (logErrOut) {
298         result.add(ERROUT_PREFIX + e.getMessage());
299       }
300       throw new ExternalProcessException(
301           INTRES.getLocalizedMessage("process.errorexternalapp", cmd),
302           e,
303           result);
304     } catch (InterruptedException e) {
305       Thread.currentThread().interrupt();
306       throw new ExternalProcessException(
307           INTRES.getLocalizedMessage("process.errorexternalapp", cmd),
308           e,
309           result);
310     } finally {
311       if (writeFileToDisk && file != null && file.exists() && !file.delete()) {
312         // Remove temporary file or schedule for delete if delete fails.
313         file.deleteOnExit();
314         LOG.info(
315             INTRES.getLocalizedMessage(
316                 "process.errordeletetempfile", filename));
317       }
318     }
319     if (LOG.isTraceEnabled()) {
320       LOG.trace(
321           "Time spent to execute external command (writeFileToDisk="
322               + writeFileToDisk
323               + "): "
324               + (System.currentTimeMillis() - startTime)
325               + "ms.");
326     }
327     return result;
328   }

341   /**
342    * Writes the byte array into a temporary file with the prefix + "-" +
343    * System.currentTimeMillies + suffix into the user directory and returns it,
344    * or null.
345    *
346    * @param bytes the bytes to write.
347    * @param filePrefix the file prefix.
348    * @param fileSuffix the file suffix.
349    * @return the file or null.
350    * @throws ExternalProcessException any exception.
351    */
352   public static File writeTemporaryFileToDisk(
353       final byte[] bytes, final String filePrefix, final String fileSuffix)
354       throws ExternalProcessException {
355     File file = null;
356     try {
357       file =
358           File.createTempFile(
359               filePrefix + "-" + System.currentTimeMillis(), fileSuffix);
360     } catch (IOException e) {
361       final String msg = INTRES.getLocalizedMessage("process.errortempfile");
362       LOG.error(msg, e);
363     }
364     if (file != null) {
365       try (FileOutputStream fos = new FileOutputStream(file)) {
366         fos.write(bytes);
367       } catch (FileNotFoundException e) {
368         final String msg = INTRES.getLocalizedMessage("process.errortempfile");
369         LOG.error(msg, e);
370         throw new ExternalProcessException(msg);
371       } catch (IOException e) {
372         try {
373           file.delete();
374         } catch (Exception e1) {
375           // NOOP
376         }
377         final String msg = INTRES.getLocalizedMessage("process.errortempfile");
378         LOG.error(msg, e);
379         throw new ExternalProcessException(msg);
380       }
381     }
382     // Delete temp file on JVM exit.
383     file.deleteOnExit();
384     return file;
385   }

387   /**
388    * Extracts the exit code in the list (at index 0 prefixed with
389    * #EXIT_CODE_PREFIX).
390    *
391    * @param out the output of the external process.
392    * @return the exit code.
393    */
394   public static Integer extractExitCode(final List<String> out) {
395     Integer result = null;
396     if (CollectionUtils.isNotEmpty(out)) {
397       result =
398           Integer.parseInt(
399               out.get(0)
400                   .replaceFirst(
401                       ExternalProcessTools.EXIT_CODE_PREFIX,
402                       StringUtils.EMPTY));
403     }
404     return result;
405   }

407   /**
408    * Checks if the list contains logging to ERROUT.
409    *
410    * @param out the output of the external process.
411    * @return true if the list contains logging to ERROUT.
412    */
413   public static boolean containsErrout(final List<String> out) {
414     if (CollectionUtils.isNotEmpty(out) && out.size() > 1) {
415       for (int i = 1, j = out.size(); i < j; i++) {
416         if (out.get(i).startsWith(ERROUT_PREFIX)) {
417           return true;
418         }
419       }
420     }
421     return false;
422   }

424   /**
425    * Quotes the arguments list.
426    *
427    * @param arguments the arguments list.
428    */
429   private static void qouteArguments(final List<String> arguments) {
430     for (int i = 0; i < arguments.size(); i++) {
431       String argument = arguments.get(i);
432       // Add quotes to encapsulate argument.
433       if (!argument.startsWith("\"") && !argument.endsWith("\"")) {
434         arguments.set(i, "\"" + argument + "\"");
435       }
436     }
437   }
