38  /**
39   * Helper class to handle operations for RFC4683 certificate extension Subject
40   * Identification method (SIM) for including a privacy-sensitive identifier in
41   * the subjectAltName extension of a certificate. The SIM is an optional feature
42   * that may be used by relying parties to determine whether the subject of a
43   * particular certificate is also the person corresponding to a particular
44   * sensitive identifier (@see <a
45   * href="https://tools.ietf.org/html/rfc4683">RFC4683</a>.)
46   *
47   * @version $Id: RFC4683Tools.java 31882 2019-03-18 13:50:56Z anatom $
48   */
49  public final class RFC4683Tools {
50  
51    /**
52     * List separator to separate the SIM tokens in the internal storage format
53     * (also has to be entered by the user).
54     */
55    public static final String LIST_SEPARATOR = "::";
56  
57    /** Label for SIM rendered in the certificate. */
58    public static final String SUBJECTIDENTIFICATIONMETHOD =
59        "subjectIdentificationMethod";
60  
61    /** OID for SIM written into the certificate. */
62    public static final String SUBJECTIDENTIFICATIONMETHOD_OBJECTID =
63        "1.3.6.1.5.5.7.8.6";

65    /** Logger. */
66    private static final Logger LOG = Logger.getLogger(RFC4683Tools.class);

68    /**
69     * Gets the allowed hash algorithm object identifiers (@see <a
70     * href="https://tools.ietf.org/html/rfc4683#section-4.3">RFC4683</a>).
71     *
72     * @return a list of ASN1ObjectIdentifier {@link TSPAlgorithms#ALLOWED}.
73     */
74    @SuppressWarnings("unchecked")
75    public static List<ASN1ObjectIdentifier> getAllowedHashAlgorithms() {
76      return new ArrayList<ASN1ObjectIdentifier>(TSPAlgorithms.ALLOWED);
77    }
78  
79    /**
80     * Gets the allowed hash algorithm OID strings.
81     *
82     * @return a list of OID strings {@link TSPAlgorithms#ALLOWED}.
83     */
84    public static List<String> getAllowedHashAlgorithmOidStrings() {
85      final List<ASN1ObjectIdentifier> identifiers = getAllowedHashAlgorithms();
86      final List<String> result = new ArrayList<String>(identifiers.size());
87      for (ASN1ObjectIdentifier identifier : identifiers) {
88        result.add(identifier.getId());
89      }
90      return result;
91    }

93    /**
94     * This method reads the internal storage format for SAN. If the SAN contains
95     * SIM parameters (list of 4 tokens, separated by '::'), the parameters are
96     * replaced by the generated SIM strings (list of 3 tokens, separated by '::')
97     * {@link RFC4683Tools#generateInternalSimString(String, String, String,
98     * String)}
99     *
100    * @param osan the SAN string in internal storage format with SIM as user
101    *     parameters.
102    * @return SAN string in internal storage format with generated SIM strings.
103    * @throws IllegalArgumentException on illegal request
104    * @throws NoSuchProviderException If provider not found
105    * @throws NoSuchAlgorithmException If algorithm not found
106    */
107   public static String generateSimForInternalSanFormat(final String osan)
108       throws IllegalArgumentException, NoSuchProviderException,
109           NoSuchAlgorithmException {
110       final int hashIdx = 0;
111       final int tokIdx = 1;
112       final int typeIdx = 2;
113       final int ssiIdx = 3;
114       final int length = 4;
115       String san = osan;
116     if (StringUtils.isNotBlank(san)
117         && san.contains(DnComponents.SUBJECTIDENTIFICATIONMETHOD)) {
118       final List<String> sims =
119           CertTools.getPartsFromDN(
120               san, DnComponents.SUBJECTIDENTIFICATIONMETHOD);
121       for (String sim : sims) {
122         if (LOG.isDebugEnabled()) {
123           LOG.info("Store user SIM strings: " + sims);
124         }
125         if (StringUtils.isNotBlank(sim)) {
126           final String[] tokens = sim.split(LIST_SEPARATOR);
127           // was entered as hash, password, SSIType and SSI, so generate the SIM
128           if (tokens.length == length) {
129             final String newSim =
130                 generateInternalSimString(
131                     tokens[hashIdx], tokens[tokIdx],
132                     tokens[typeIdx], tokens[ssiIdx]);
133             san = san.replace(sim, newSim);
134           } else if (tokens.length == length - 1) {
135             // NOOP
136           } else {
137             throw new IllegalArgumentException(
138                 "Wrong SIM input string with " + tokens.length + " tokens.");
139           }
140         }
141       }
142     }
143     return san;
144   }

146   /**
147    * Creates a '::' separated string of hashAlogrithmOidString, Authority Random
148    * (R) and Privacy-Enhanced Protected Subject Information (PEPSI). Note:
149    * RFC4683 Subject Identification Method (SIM = R || PEPSI), and PEPSI = H(H(
150    * P || R || SIItype || SII)). The resulting String is used for internal
151    * storage.
152    *
153    * <p>Where R is the Authority Random hash and PEPSI the Privacy-Enhanced
154    * Protected Subject Information: PEPSI = H(H( P || R || SIItype || SII))
155    * Where P is the user chosen password, SSI the Sensitive Identification
156    * Information and SIIType its type.
157    *
158    * @param hashAlogrithmOidString i.e '1.3.14.3.2.26' for SHA-1
159    * @param userChosenPassword (P) FIPS 112 and FIPS 180-1 compliant password up
160    *     to 28 characters (see https://tools.ietf.org/html/rfc4683#section-4.2)
161    * @param ssiType OID string of an SSI type (see
162    *     https://tools.ietf.org/html/rfc4683#section-4.1).
163    * @param ssi Sensitive Identification Information (SII) (see
164    *     https://tools.ietf.org/html/rfc4683#section-4.1).
165    * @return a '::' separated string of hashAlogrithmOidString, R and PEPSI.
166    * @throws IllegalArgumentException on illegal request
167    * @throws NoSuchProviderException If provider not found
168    * @throws NoSuchAlgorithmException If algorithm not found
169    */
170   public static String generateInternalSimString(
171       final String hashAlogrithmOidString,
172       final String userChosenPassword,
173       final String ssiType,
174       final String ssi)
175       throws IllegalArgumentException, NoSuchProviderException,
176           NoSuchAlgorithmException {
177     if (StringUtils.isBlank(hashAlogrithmOidString)) {
178       throw new IllegalArgumentException(
179           "Hash algorithm OID string must not be null or empty: '"
180               + hashAlogrithmOidString
181               + "'.");
182     }
183     if (!getAllowedHashAlgorithmOidStrings().contains(hashAlogrithmOidString)) {
184       throw new IllegalArgumentException(
185           "Hash algorithm with OID '"
186               + hashAlogrithmOidString
187               + "' is not supparted for RFC4683 (SIM).");
188     }
189     // TODO Insert check for FIPS 180-1 compliant passwords (better current
190     // standards ...)
191     if (StringUtils.isBlank(userChosenPassword)
192         || userChosenPassword.length() < 8) {
193       throw new IllegalArgumentException(
194           "The user chosen password must not be null or empty: '"
195               + hashAlogrithmOidString
196               + "'.");
197     }
198     if (StringUtils.isBlank(ssiType)) {
199       throw new IllegalArgumentException(
200           "The sensitve identification information type must not be null or"
201               + " empty: '"
202               + ssiType
203               + "'.");
204     }
205     if (StringUtils.isBlank(ssi)) {
206       throw new IllegalArgumentException(
207           "The sensitve identification information must not be null or empty: '"
208               + ssi
209               + "'.");
210     }
211     final StringBuilder result = new StringBuilder();
212     result.append(hashAlogrithmOidString);
213 
214     // 1. Create authority random.
215     final String authorityRandomSource =
216         Long.toHexString(
217             SernoGeneratorRandom.instance(16).getSerno().longValue());
218     if (LOG.isDebugEnabled()) {
219       LOG.debug("Authority random source created: " + authorityRandomSource);
220     }
221 
222     // 1b. Get HEX by hash of authority random.
223     final MessageDigest digester =
224         MessageDigest.getInstance(
225             new ASN1ObjectIdentifier(hashAlogrithmOidString).getId(),
226             BouncyCastleProvider.PROVIDER_NAME);
227     digester.update(authorityRandomSource.getBytes());
228     final String authorityRandom = toHexString(digester.digest());
229     result.append(LIST_SEPARATOR).append(authorityRandom);
230     if (LOG.isDebugEnabled()) {
231       LOG.debug("Authority random hash created: " + authorityRandom);
232     }
233 
234     // 2. Create SIM HEX string, and hash 2 times.
235     digester.update(
236         new StringBuilder()
237             .append(userChosenPassword)
238             .append(authorityRandomSource)
239             .append(ssiType)
240             .append(ssi)
241             .toString()
242             .getBytes());
243     digester.update(digester.digest());
244     final String pepsi = toHexString(digester.digest());
245     result.append(LIST_SEPARATOR).append(pepsi);
246     if (LOG.isDebugEnabled()) {
247       LOG.debug("SIM string PEPSI created: " + pepsi);
248     }
249     return result.toString();
250   }

252   /**
253    * Creates a SIM GeneralName by the internal SIM storage format
254    * ('hashAlgorithmOIDString::R::PEPSI') SIM ::= SEQUENCE { hashAlg
255    * AlgorithmIdentifier, authorityRandom OCTET STRING, -- RA-chosen random
256    * number -- used in computation of -- pEPSI pEPSI OCTET STRING -- hash of
257    * HashContent -- with algorithm hashAlg }.
258    *
259    * @param hashAlgorithmIdentifier the OID string for the hash algorithm used
260    *     to hash R and PEPSI.
261    * @param authorityRandom the registration authority chosen random value,
262    *     hashed with hash of hashAlogrithmOidString (see
263    *     https://tools.ietf.org/html/rfc4683#section-4.3).
264    * @param pepsi Privacy-Enhanced Protected Subject Information (PEPSI), with
265    *     SIM = R || PEPSI.
266    * @return the RFC4683 SIM GeneralName (@see <a
267    *     href="https://tools.ietf.org/html/rfc4683">RFC4683</a>.).
268    */
269   public static ASN1Primitive createSimGeneralName(
270       final String hashAlgorithmIdentifier,
271       final String authorityRandom,
272       final String pepsi) {
273     if (LOG.isDebugEnabled()) {
274       LOG.debug(
275           "Creating SIM with hash algorithem identifier "
276               + hashAlgorithmIdentifier
277               + ", authority random "
278               + authorityRandom
279               + " and PEPSI "
280               + pepsi);
281     }
282     final ASN1EncodableVector otherName = new ASN1EncodableVector();
283     otherName.add(
284         new ASN1ObjectIdentifier(SUBJECTIDENTIFICATIONMETHOD_OBJECTID));
285     final ASN1EncodableVector simVector = new ASN1EncodableVector();
286     simVector.add(
287         new AlgorithmIdentifier(
288             new ASN1ObjectIdentifier(
289                 hashAlgorithmIdentifier))); // new DERTaggedObject(true, 0,
290     simVector.add(new DEROctetString((authorityRandom).getBytes()));
291     simVector.add(new DEROctetString((pepsi).getBytes()));
292     otherName.add(new DERTaggedObject(true, 0, new DERSequence(simVector)));
293     final ASN1Primitive generalName =
294         new DERTaggedObject(false, 0, new DERSequence(otherName));
295     if (LOG.isDebugEnabled()) {
296       LOG.debug(
297           "GeneralName (type 0 - OtherName) for SIM created "
298               + generalName.toString());
299     }
300     return generalName;
301   }

303   /**
304    * Helper method for getting the SIM name from SAN ASN.1 sequence.
305    *
306    * @param sequence the OtherName sequence
307    * @return the SIM string by the otherName.
308    */
309   public static String getSimStringSequence(final ASN1Sequence sequence) {
310     if (LOG.isDebugEnabled()) {
311       LOG.debug("Parsing RFC4683 (SIM) from SAN ASN.1 sequence: " + sequence);
312     }
313     String result = null;
314     if (sequence != null) {
315       // First in sequence is the object identifier, that we must check
316       final ASN1ObjectIdentifier id =
317           ASN1ObjectIdentifier.getInstance(sequence.getObjectAt(0));
318       if (SUBJECTIDENTIFICATIONMETHOD_OBJECTID.equals(id.getId())) {
319         final ASN1Sequence simVector =
320             (ASN1Sequence)
321                 ((DERTaggedObject) sequence.getObjectAt(1)).getObject();
322         // 1. After certificate issuance the method is called with an algorithm
323         // identifier in its ASN.1 sequence.
324         // 2. But after reading a stored certificate (PEM or DER) the ASN.1
325         // sequence contains a DERSeqence instead.
326         String algorithmIdentifier = null;
327         if (simVector.getObjectAt(0) instanceof AlgorithmIdentifier) {
328           algorithmIdentifier =
329               ((AlgorithmIdentifier) simVector.getObjectAt(0))
330                   .getAlgorithm()
331                   .getId();
332         } else {
333           final ASN1Encodable encodable =
334               ((ASN1Sequence) simVector.getObjectAt(0)).getObjectAt(0);
335           algorithmIdentifier = encodable.toASN1Primitive().toString();
336         }
337         final ASN1OctetString hash = (ASN1OctetString) simVector.getObjectAt(1);
338         final ASN1OctetString pepsi =
339             (ASN1OctetString) simVector.getObjectAt(2);
340         final String hashString = new String(hash.getOctets());
341         final String pepsiString = new String(pepsi.getOctets());
342         final StringBuilder builder = new StringBuilder();
343         result =
344             builder
345                 .append(algorithmIdentifier)
346                 .append(LIST_SEPARATOR)
347                 .append(hashString)
348                 .append(LIST_SEPARATOR)
349                 .append(pepsiString)
350                 .toString();
351       }
352     }
353     if (LOG.isDebugEnabled()) {
354       LOG.debug("SIM parsed from other name: " + result);
355     }
356     return result;
357   }

359   /**
360    * The method generates the HEX string by digestResult.
361    *
362    * @param digestResult the resulting byte[] of the digester.
363    * @return the HEX string.
364    */
365   public static String toHexString(final byte[] digestResult) {
366     final StringBuffer buf = new StringBuffer(digestResult.length * 2);
367     final int mask = 0xff;
368     for (int i = 0; i < digestResult.length; i++) {
369       int intVal = digestResult[i] & mask;
370       if (intVal < 0x10) {
371         buf.append("0");
372       }
373       buf.append(Integer.toHexString(intVal).toUpperCase());
374     }
375     return buf.toString();
376   }
