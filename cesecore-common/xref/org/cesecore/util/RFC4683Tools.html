39  /**
40   * Helper class to handle operations for RFC4683 certificate extension Subject Identification method (SIM) for 
41   * including a privacy-sensitive identifier in the subjectAltName extension of a certificate. The SIM is 
42   * an optional feature that may be used by relying parties to determine whether the subject of a particular 
43   * certificate is also the person corresponding to a particular sensitive identifier 
44   * (@see <a href="https://tools.ietf.org/html/rfc4683">RFC4683</a>.)
45   * 
46   * @version $Id: RFC4683Tools.java 31882 2019-03-18 13:50:56Z anatom $
47   */
48  public final class RFC4683Tools {
49  
50      /** List separator to separate the SIM tokens in the internal storage format (also has to be entered by the user). */
51      public static final String LIST_SEPARATOR = "::";
52  
53      /** Label for SIM rendered in the certificate. */
54      public static final String SUBJECTIDENTIFICATIONMETHOD = "subjectIdentificationMethod";
55  
56      /** OID for SIM written into the certificate. */
57      public static final String SUBJECTIDENTIFICATIONMETHOD_OBJECTID = "1.3.6.1.5.5.7.8.6";
58  
59      private static final Logger LOG = Logger.getLogger(RFC4683Tools.class);

61      /**
62       * Gets the allowed hash algorithm object identifiers (@see <a href="https://tools.ietf.org/html/rfc4683#section-4.3">RFC4683</a>).
63       * @return a list of ASN1ObjectIdentifier {@link TSPAlgorithms#ALLOWED}.
64       */
65      @SuppressWarnings("unchecked")
66      public static final List<ASN1ObjectIdentifier> getAllowedHashAlgorithms() {
67          return new ArrayList<ASN1ObjectIdentifier>(TSPAlgorithms.ALLOWED);
68      }
69  
70      /**
71       * Gets the allowed hash algorithm OID strings.
72       * @return a list of OID strings {@link TSPAlgorithms#ALLOWED}.
73       */
74      public static final List<String> getAllowedHashAlgorithmOidStrings() {
75          final List<ASN1ObjectIdentifier> identifiers = getAllowedHashAlgorithms();
76          final List<String> result = new ArrayList<String>(identifiers.size());
77          for (ASN1ObjectIdentifier identifier : identifiers) {
78              result.add(identifier.getId());
79          }
80          return result;
81      }

83      /** This method reads the internal storage format for SAN. 
84       * If the SAN contains SIM parameters (list of 4 tokens, separated by '::'), the parameters are replaced by 
85       * the generated SIM strings (list of 3 tokens, separated by '::') {@link RFC4683Tools#generateInternalSimString(String, String, String, String)}
86       * 
87       * @param san the SAN string in internal storage format with SIM as user parameters.
88       * @return SAN string in internal storage format with generated SIM strings.
89       * @throws IllegalArgumentException on illegal request
90       * @throws NoSuchProviderException If provider not found
91       * @throws NoSuchAlgorithmException If algorithm not found
92       */
93      public static final String generateSimForInternalSanFormat(String san)
94              throws IllegalArgumentException, NoSuchProviderException, NoSuchAlgorithmException {
95          if (StringUtils.isNotBlank(san) && san.contains(DnComponents.SUBJECTIDENTIFICATIONMETHOD)) {
96              final List<String> sims = CertTools.getPartsFromDN(san, DnComponents.SUBJECTIDENTIFICATIONMETHOD);
97              for (String sim : sims) {
98                  if (LOG.isDebugEnabled()) {
99                      LOG.info("Store user SIM strings: " + sims);
100                 }
101                 if (StringUtils.isNotBlank(sim)) {
102                     final String[] tokens = sim.split(LIST_SEPARATOR);
103                     // was entered as hash, password, SSIType and SSI, so generate the SIM
104                     if (tokens.length == 4) {
105                         final String newSim = generateInternalSimString(tokens[0], tokens[1], tokens[2], tokens[3]);
106                         san = san.replace(sim, newSim);
107                     } else if (tokens.length == 3) {
108                         // NOOP
109                     } else {
110                         throw new IllegalArgumentException("Wrong SIM input string with " + tokens.length + " tokens.");
111                     }
112                 }
113             }
114         }
115         return san;
116     }

118     /**
119      * Creates a '::' separated string of hashAlogrithmOidString, Authority Random (R) and Privacy-Enhanced Protected Subject Information (PEPSI). 
120      * Note: RFC4683 Subject Identification Method (SIM = R || PEPSI), and PEPSI = H(H( P || R || SIItype || SII)). The resulting String is used 
121      * for internal storage.
122      * 
123      * Where R is the Authority Random hash and PEPSI the Privacy-Enhanced Protected Subject Information:
124      * PEPSI = H(H( P || R || SIItype || SII))
125      * Where P is the user chosen password, SSI the Sensitive Identification Information and SIIType its type.
126      * 
127      * @param hashAlogrithmOidString i.e '1.3.14.3.2.26' for SHA-1
128      * @param userChosenPassword (P) FIPS 112 and FIPS 180-1 compliant password up to 28 characters (see https://tools.ietf.org/html/rfc4683#section-4.2)
129      * @param ssiType OID string of an SSI type (see https://tools.ietf.org/html/rfc4683#section-4.1).
130      * @param ssi Sensitive Identification Information (SII) (see https://tools.ietf.org/html/rfc4683#section-4.1).
131      * @return a '::' separated string of hashAlogrithmOidString, R and PEPSI.
132      * @throws IllegalArgumentException on illegal request
133      * @throws NoSuchProviderException If provider not found
134      * @throws NoSuchAlgorithmException If algorithm not found
135      */
136     public static final String generateInternalSimString(final String hashAlogrithmOidString, final String userChosenPassword, final String ssiType,
137             final String ssi) throws IllegalArgumentException, NoSuchProviderException, NoSuchAlgorithmException {
138         if (StringUtils.isBlank(hashAlogrithmOidString)) {
139             throw new IllegalArgumentException("Hash algorithm OID string must not be null or empty: '" + hashAlogrithmOidString + "'.");
140         }
141         if (!getAllowedHashAlgorithmOidStrings().contains(hashAlogrithmOidString)) {
142             throw new IllegalArgumentException("Hash algorithm with OID '" + hashAlogrithmOidString + "' is not supparted for RFC4683 (SIM).");
143         }
144         // TODO Insert check for FIPS 180-1 compliant passwords (better current standards ...)
145         if (StringUtils.isBlank(userChosenPassword) || userChosenPassword.length() < 8) {
146             throw new IllegalArgumentException("The user chosen password must not be null or empty: '" + hashAlogrithmOidString + "'.");
147         }
148         if (StringUtils.isBlank(ssiType)) {
149             throw new IllegalArgumentException("The sensitve identification information type must not be null or empty: '" + ssiType + "'.");
150         }
151         if (StringUtils.isBlank(ssi)) {
152             throw new IllegalArgumentException("The sensitve identification information must not be null or empty: '" + ssi + "'.");
153         }
154         final StringBuilder result = new StringBuilder();
155         result.append(hashAlogrithmOidString);
156 
157         // 1. Create authority random.
158         final String authorityRandomSource = Long.toHexString(SernoGeneratorRandom.instance(16).getSerno().longValue());
159         if (LOG.isDebugEnabled()) {
160             LOG.debug("Authority random source created: " + authorityRandomSource);
161         }
162 
163         // 1b. Get HEX by hash of authority random.
164         final MessageDigest digester = MessageDigest.getInstance(new ASN1ObjectIdentifier(hashAlogrithmOidString).getId(),
165                 BouncyCastleProvider.PROVIDER_NAME);
166         digester.update(authorityRandomSource.getBytes());
167         final String authorityRandom = toHexString(digester.digest());
168         result.append(LIST_SEPARATOR).append(authorityRandom);
169         if (LOG.isDebugEnabled()) {
170             LOG.debug("Authority random hash created: " + authorityRandom);
171         }
172 
173         // 2. Create SIM HEX string, and hash 2 times.
174         digester.update(
175                 new StringBuilder().append(userChosenPassword).append(authorityRandomSource).append(ssiType).append(ssi).toString().getBytes());
176         digester.update(digester.digest());
177         final String pepsi = toHexString(digester.digest());
178         result.append(LIST_SEPARATOR).append(pepsi);
179         if (LOG.isDebugEnabled()) {
180             LOG.debug("SIM string PEPSI created: " + pepsi);
181         }
182         return result.toString();
183     }

185     /**
186      * Creates a SIM GeneralName by the internal SIM storage format ('hashAlgorithmOIDString::R::PEPSI')
187      * SIM ::= SEQUENCE { hashAlg AlgorithmIdentifier, authorityRandom OCTET
188      * STRING, -- RA-chosen random number -- used in computation of -- pEPSI
189      * pEPSI OCTET STRING -- hash of HashContent -- with algorithm hashAlg }
190      * @param hashAlgorithmIdentifier the OID string for the hash algorithm used to hash R and PEPSI.
191      * @param authorityRandom the registration authority chosen random value, hashed with hash of hashAlogrithmOidString (see https://tools.ietf.org/html/rfc4683#section-4.3).
192      * @param pepsi Privacy-Enhanced Protected Subject Information (PEPSI), with SIM = R || PEPSI.
193      * @return the RFC4683 SIM GeneralName (@see <a href="https://tools.ietf.org/html/rfc4683">RFC4683</a>.).
194      */
195     public static final ASN1Primitive createSimGeneralName(final String hashAlgorithmIdentifier, final String authorityRandom, final String pepsi) {
196         if (LOG.isDebugEnabled()) {
197             LOG.debug("Creating SIM with hash algorithem identifier " + hashAlgorithmIdentifier + ", authority random " + authorityRandom
198                     + " and PEPSI " + pepsi);
199         }
200         final ASN1EncodableVector otherName = new ASN1EncodableVector();
201         otherName.add(new ASN1ObjectIdentifier(SUBJECTIDENTIFICATIONMETHOD_OBJECTID));
202         final ASN1EncodableVector simVector = new ASN1EncodableVector();
203         simVector.add(new AlgorithmIdentifier(new ASN1ObjectIdentifier(hashAlgorithmIdentifier))); // new DERTaggedObject(true, 0, 
204         simVector.add(new DEROctetString((authorityRandom).getBytes()));
205         simVector.add(new DEROctetString((pepsi).getBytes()));
206         otherName.add(new DERTaggedObject(true, 0, new DERSequence(simVector)));
207         final ASN1Primitive generalName = new DERTaggedObject(false, 0, new DERSequence(otherName));
208         if (LOG.isDebugEnabled()) {
209             LOG.debug("GeneralName (type 0 - OtherName) for SIM created " + generalName.toString());
210         }
211         return generalName;
212     }

214     /**
215      * Helper method for getting the SIM name from SAN ASN.1 sequence.
216      * 
217      * @param sequence the OtherName sequence
218      * @return the SIM string by the otherName.
219      */
220     public static String getSimStringSequence(final ASN1Sequence sequence) {
221         if (LOG.isDebugEnabled()) {
222             LOG.debug("Parsing RFC4683 (SIM) from SAN ASN.1 sequence: " + sequence);
223         }
224         String result = null;
225         if (sequence != null) {
226             // First in sequence is the object identifier, that we must check
227             final ASN1ObjectIdentifier id = ASN1ObjectIdentifier.getInstance(sequence.getObjectAt(0));
228             if (SUBJECTIDENTIFICATIONMETHOD_OBJECTID.equals(id.getId())) {
229                 final ASN1Sequence simVector = (ASN1Sequence) ((DERTaggedObject) sequence.getObjectAt(1)).getObject();
230                 // 1. After certificate issuance the method is called with an algorithm identifier in its ASN.1 sequence.
231                 // 2. But after reading a stored certificate (PEM or DER) the ASN.1 sequence contains a DERSeqence instead.
232                 String algorithmIdentifier = null;
233                 if (simVector.getObjectAt(0) instanceof AlgorithmIdentifier) {
234                     algorithmIdentifier = ((AlgorithmIdentifier) simVector.getObjectAt(0)).getAlgorithm().getId();
235                 } else {
236                     final ASN1Encodable encodable = ((ASN1Sequence) simVector.getObjectAt(0)).getObjectAt(0);
237                     algorithmIdentifier = encodable.toASN1Primitive().toString();
238                 }
239                 final ASN1OctetString hash = (ASN1OctetString) simVector.getObjectAt(1);
240                 final ASN1OctetString pepsi = (ASN1OctetString) simVector.getObjectAt(2);
241                 final String hashString = new String(hash.getOctets());
242                 final String pepsiString = new String(pepsi.getOctets());
243                 final StringBuilder builder = new StringBuilder();
244                 result = builder.append(algorithmIdentifier).append(LIST_SEPARATOR).append(hashString).append(LIST_SEPARATOR).append(pepsiString)
245                         .toString();
246             }
247         }
248         if (LOG.isDebugEnabled()) {
249             LOG.debug("SIM parsed from other name: " + result);
250         }
251         return result;
252     }

254     /**
255      * The method generates the HEX string by digestResult. 
256      * @param digestResult the resulting byte[] of the digester.
257      * @return the HEX string.
258      */
259     public static final String toHexString(final byte[] digestResult) {
260         final StringBuffer buf = new StringBuffer(digestResult.length * 2);
261         for (int i = 0; i < digestResult.length; i++) {
262             int intVal = digestResult[i] & 0xff;
263             if (intVal < 0x10) {
264                 buf.append("0");
265             }
266             buf.append(Integer.toHexString(intVal).toUpperCase());
267         }
268         return buf.toString();
269     }
