14  package org.cesecore.util;
15  
16  import java.util.LinkedHashMap;
17  import java.util.List;
18  import java.util.Map;
19  import java.util.Map.Entry;
20  import java.util.regex.Matcher;
21  import java.util.regex.Pattern;
22  import org.apache.commons.lang.StringUtils;
23  
24  /**
25   * Class to parse and format relative time values, i.e. '1y-2mo+3d-4h+5m-6s7ms'.
26   *
27   * @version $Id: TimeUnitFormat.java 24933 2016-12-20 08:41:57Z mikekushner $
28   */
29  public final class TimeUnitFormat {
30    /** Start. */
31    private static final String PATTERN_PREFIX = "\\s*(([+-]?\\d+)\\s*(";
32    /** End. */
33    private static final String PATTERN_SUFFIX = "))\\s*";
34    /** Zero. */
35    private static final String ZERO = "0";
36    /** Space. */
37    private static final String SPACE = " ";
38    /** Or. */
39    private static final String OR = "|";
40  
41    /** Error message. */
42    private static final String EXCEPTION_MESSAGE_ILLEGAL_CHARACTERS =
43        "Illegal characters.";
44    /** Error message. */
45    private static final String EXCEPTION_MESSAGE_BLANK_STRING =
46        "Cannot parse a blank string.";
47  
48    /** Pattern. */
49    private final Pattern pattern;
50    /** Defaults. */
51    private final Map<String, Long> defaultValues;
52    /** Factors. */
53    private final Map<String, Long> factors;
54    /** Units. */
55    private final List<String> units;
56  
57    /**
58     * Instantiates a new TimeUnitFormat and initializes it with the given units.
59     *
60     * @param theUnits List of units
61     *     (suffixes, i.e. 'ms', 'mo', 'y', 'd', 'h', 'm', and 's').
62     * @param theFactors factors
63     */
64    public TimeUnitFormat(
65        final List<String> theUnits, final Map<String, Long> theFactors) {
66      this.units = theUnits;
67      this.factors = theFactors;
68      this.defaultValues = new LinkedHashMap<String, Long>(theUnits.size());
69      final StringBuilder builder = new StringBuilder(PATTERN_PREFIX);
70      int index = 0;
71      for (String unit : theUnits) {
72        this.defaultValues.put(unit.toLowerCase(), 0L);
73        if (index++ > 0) {
74          builder.append(OR);
75        }
76        // allows the characters passed in as units. the regexp is compiled with
77        // Pattern.CASE_INSENSITIVE below.
78        builder.append(Pattern.quote(unit));
79      }
80      builder.append(PATTERN_SUFFIX);
81      pattern = Pattern.compile(builder.toString(), Pattern.CASE_INSENSITIVE);
82    }
83  
84    /**
85     * Parses a formatted time string.
86     *
87     * @param aformattedString time string, i.e '1y-2mo10d'.
88     * @return the milliseconds as long value from 0.
89     * @throws NumberFormatException if the string cannot be parsed, i.e. it
90     *     contains units not listed or other illegal characters or forms.
91     */
92    public long parseMillis(final String aformattedString)
93            throws NumberFormatException {
94      String formattedString = aformattedString;
95      long result = 0;
96      final int numlength = 3;
97      if (StringUtils.isNotBlank(formattedString)) {
98        formattedString = formattedString.trim();
99        final Matcher matcher = pattern.matcher(formattedString);
100       long parsedValue;
101       String unit = null;
102       int start = 0;
103       int end = 0;
104       while (matcher.find()) {
105         start = matcher.start();
106         if (start != end) {
107           throw new NumberFormatException(EXCEPTION_MESSAGE_ILLEGAL_CHARACTERS);
108         }
109         end = matcher.end();
110         for (int i = 0; i < matcher.groupCount(); i = i + numlength) {
111           parsedValue = Long.parseLong(matcher.group(i + 2));
112           unit = matcher.group(i + numlength).toLowerCase();
113           result += factors.get(unit) * parsedValue;
114         }
115       }
116       if (end != formattedString.length()) {
117         throw new NumberFormatException(EXCEPTION_MESSAGE_ILLEGAL_CHARACTERS);
118       }
119 
120     } else {
121       throw new NumberFormatException(EXCEPTION_MESSAGE_BLANK_STRING);
122     }
123 
124     return result;
125   }
126 
127   /**
128    * Formats the given period in milliseconds to a readable string.
129    *
130    * @param millis the milliseconds (count from 0 - not epoch).
131    * @param theFactors factors
132    * @param zeroType the unit if the result is 0.
133    * @return a readable string in form of the ordered value unit pairs (*y *mo
134    *     *d *h *m *s), separated by white space character. Milliseconds are
135    *     lost.
136    */
137   public String format(
138       final long millis,
139       final Map<String, Long> theFactors,
140       final String zeroType) {
141     long value = millis;
142     String unit = null;
143     long factor = 0;
144     long currentValue = 0;
145     final StringBuilder builder = new StringBuilder();
146     for (Entry<String, Long> entry : theFactors.entrySet()) {
147       unit = entry.getKey();
148       if (units.contains(unit)) {
149         factor = entry.getValue();
150         currentValue = value / factor;
151         value %= factor;
152         if (currentValue != 0) {
153           if (builder.length() > 0) {
154             builder.append(SPACE);
155           }
156           builder.append(Long.toString(currentValue)).append(unit);
157         }
158       }
159     }
160     if (builder.length() < 1) {
161       builder.append(ZERO).append(zeroType);
162     }
163     return builder.toString();
164   }
