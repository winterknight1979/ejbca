14  package org.cesecore.util;
15  
16  import java.util.Random;
17  import org.apache.log4j.Logger;
18  
19  /**
20   * Shamelessly ripped from generated XDoclet source, because I don't want to
21   * generate util classes.
22   *
23   * @version $Id: GUIDGenerator.java 20325 2014-11-25 19:22:49Z samuellb $
24   */
25  public final class GUIDGenerator {
26      private GUIDGenerator() { }
27      /** Logger. */
28    private static final Logger LOG = Logger.getLogger(GUIDGenerator.class);
29  
30    /** Cached per JVM server IP. */
31    private static String hexServerIP = null;
32  
33    /** Initialize the secure random instance. */
34    private static final Random SEEDER = new Random();
35  
36    /**
37     * A 32 byte GUID generator (Globally Unique ID). These artificial keys SHOULD
38     * <strong>NOT </strong> be seen by the user, not even touched by the DBA but
39     * with very rare exceptions, just manipulated by the database and the
40     * programs. NOTE: This guid is not cryptographically secure, and should not
41     * be used as such.
42     *
43     * <p>Usage: Add an id field (type java.lang.String) to your EJB, and add
44     * setId(XXXUtil.generateGUID(this)); to the ejbCreate method.
45     *
46     * @param o object
47     * @return guid
48     */
49    public static String generateGUID(final Object o) {
50  
51      if (hexServerIP == null) {
52        java.net.InetAddress localInetAddress = null;
53        try {
54          // get the inet address
55          localInetAddress = java.net.InetAddress.getLocalHost();
56        } catch (java.net.UnknownHostException uhe) {
57          LOG.error(
58              "Could not get the local IP address using"
59                  + " InetAddress.getLocalHost(): ",
60              uhe);
61          // todo: find better way to get around this...
62          return null;
63        }
64        byte[] serverIP = localInetAddress.getAddress();
65        hexServerIP = hexFormat(getInt(serverIP), 8);
66      }
67  
68      final String hashcode = hexFormat(System.identityHashCode(o), 8);
69      // Use a combination of milliseconds and nanoseconds:
70      // - milliseconds is not good enough for the test case (which makes a lot of
71      // calls in a quick succession).
72      // - nanoseconds is also not good enough since some of the low-order bits
73      // may be always zero (how many is system and JDK dependent).
74      long timeNow = System.currentTimeMillis() ^ System.nanoTime();
75      int timeLow = (int) timeNow & 0xFFFFFFFF;
76      int node = SEEDER.nextInt();
77      final StringBuilder guid = new StringBuilder(32);
78  
79      guid.append(hexFormat(timeLow, 8));
80      guid.append(hexServerIP);
81      guid.append(hashcode);
82      guid.append(hexFormat(node, 8));
83      return guid.toString();
84    }
85  
86    private static int getInt(final byte[] bytes) {
87      final int threebytes = 24;
88      int i = 0;
89      int j = threebytes;
90      final int mask = 0xff;
91      for (int k = 0; j >= 0; k++) {
92        int l = bytes[k] & mask;
93        i += l << j;
94        j -= 8;
95      }
96      return i;
97    }
98  
99    private static String hexFormat(final int i, final int j) {
100     final String s = Integer.toHexString(i);
101     return padHex(s, j) + s;
102   }
103 
104   private static String padHex(final String s, final int i) {
105     final StringBuilder tmpBuffer = new StringBuilder();
106     if (s.length() < i) {
107       for (int j = 0; j < i - s.length(); j++) {
108         tmpBuffer.append('0');
109       }
110     }
111     return tmpBuffer.toString();
112   }
