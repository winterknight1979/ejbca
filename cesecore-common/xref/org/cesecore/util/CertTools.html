450   /**
451    * @param odn DN
452    * @param nameStyle Style
453    * @return Name
454    */
455   public static X500Name stringToUnorderedX500Name(
456       final String odn, final X500NameStyle nameStyle) {
457     if (LOGGER.isTraceEnabled()) {
458       LOGGER.trace(">stringToUnorderedX500Name: " + odn);
459     }
460     if (odn == null) {
461       return null;
462     }
463     // If the entire DN is quoted (which is strange but legacy), we just remove
464     // these quotes and carry on
465    String dn;
466     if (odn.length() > 2
467         && odn.charAt(0) == '"'
468         && odn.charAt(odn.length() - 1) == '"') {
469       dn = odn.substring(1, odn.length() - 1);
470     } else {
471         dn = odn;
472     }
473     final X500NameBuilder nameBuilder = new X500NameBuilder(nameStyle);
474     boolean quoted = false;
475     boolean escapeNext = false;
476     int currentStartPosition = -1;
477     String currentPartName = null;
478     for (int i = 0; i < dn.length(); i++) {
479       final char current = dn.charAt(i);
480       // Toggle quoting for every non-escaped "-char
481       if (!escapeNext && current == '"') {
482         quoted = !quoted;
483       }
484       // If there is an unescaped and unquoted =-char the proceeding chars is a
485       // part name
486       if (currentStartPosition == -1
487           && !quoted
488           && !escapeNext
489           && current == '='
490           && 1 <= i) {
491         // Trim spaces (e.g. "O =value")
492         int endIndexOfPartName = i;
493         while (endIndexOfPartName > 0
494             && dn.charAt(endIndexOfPartName - 1) == ' ') {
495           endIndexOfPartName--;
496         }
497         int startIndexOfPartName = endIndexOfPartName - 1;
498         final String endOfPartNameSearchChars = ", +";
499         while (startIndexOfPartName > 0
500             && (endOfPartNameSearchChars.indexOf(
501                     dn.charAt(startIndexOfPartName - 1))
502                 == -1)) {
503           startIndexOfPartName--;
504         }
505         currentPartName =
506             dn.substring(startIndexOfPartName, endIndexOfPartName);
507         currentStartPosition = i + 1;
508       }
509       // When we have found a start marker, we need to be on the lookout for the
510       // ending marker
511       if (currentStartPosition != -1
512           && ((!quoted && !escapeNext && (current == ',' || current == '+'))
513               || i == dn.length() - 1)) {
514         int endPosition = (i == dn.length() - 1) ? dn.length() - 1 : i - 1;
515         // Remove white spaces from the end of the value
516         while (endPosition > currentStartPosition
517             && dn.charAt(endPosition) == ' ') {
518           endPosition--;
519         }
520         // Remove white spaces from the beginning of the value
521         while (endPosition > currentStartPosition
522             && dn.charAt(currentStartPosition) == ' ') {
523           currentStartPosition++;
524         }
525         // Only return the inner value if the part is quoted
526         if (currentStartPosition < dn.length()
527             && dn.charAt(currentStartPosition) == '"'
528             && dn.charAt(endPosition) == '"') {
529           currentStartPosition++;
530           endPosition--;
531         }
532         String currentValue =
533             dn.substring(currentStartPosition, endPosition + 1);
534         // Unescape value (except escaped #) since the nameBuilder will double
535         // each escape
536         currentValue =
537             unescapeValue(new StringBuilder(currentValue)).toString();
538         try {
539           // -- First search the OID by name in declared OID's
540           ASN1ObjectIdentifier oid = DnComponents.getOid(currentPartName);
541           // -- If isn't declared, we try to create it
542           if (oid == null) {
543             oid = new ASN1ObjectIdentifier(currentPartName);
544           }
545           nameBuilder.addRDN(oid, currentValue);
546         } catch (final IllegalArgumentException e) {
547           // If it is not an OID we will ignore it
548           LOGGER.warn(
549               "Unknown DN component ignored and silently dropped: "
550                   + currentPartName);
551         }
552         // Reset markers
553         currentStartPosition = -1;
554         currentPartName = null;
555       }
556       if (escapeNext) {
557         // This character was escaped, so don't escape the next one
558         escapeNext = false;
559       } else {
560         if (!quoted && current == '\\') {
561           // This escape character is not escaped itself, so the next one should
562           // be
563           escapeNext = true;
564         }
565       }
566     }
567     final X500Name x500Name = nameBuilder.build();
568     if (LOGGER.isTraceEnabled()) {
569       LOGGER.trace("<stringToUnorderedX500Name: x500Name="
570               + x500Name.toString());
571     }
572     return x500Name;
573   }

575   /**
576    * Removes any unescaped '\' character from the provided StringBuilder.
577    * Assumes that escaping quotes have been stripped. Special treatment of the #
578    * sign, which if not escaped will be treated as hex encoded DER value by BC.
579    *
580    * @param sb unescaped StringBuilder
581    * @return escaped StringBuilder
582    */
583   private static StringBuilder unescapeValue(final StringBuilder sb) {
584     boolean esq = false;
585     int index = 0;
586     while (index < (sb.length() - 1)) {
587       if (!esq && sb.charAt(index) == '\\' && sb.charAt(index + 1) != '#') {
588         esq = true;
589         sb.deleteCharAt(index);
590       } else {
591         esq = false;
592         index++;
593       }
594     }
595     return sb;
596   }

620   /**
621    * Check if the String contains any unescaped '+'. RFC 2253, section 2.2
622    * states that '+' is used for multi-valued RelativeDistinguishedName. BC
623    * (version 1.45) did not support multi-valued RelativeDistinguishedName, and
624    * automatically escaped them instead. Even though it is now (BC 1.49b15)
625    * supported, we want to keep ecaping '+' chars and warn that this might not
626    * be supported in the future.
627    *
628    * @param dn DN
629    * @return escaped DN
630    */
631   public static String handleUnescapedPlus(final String dn) {
632     if (dn == null) {
633       return dn;
634     }
635     final StringBuilder buf = new StringBuilder(dn);
636     int index = 0;
637     final int end = buf.length();
638     while (index < end) {
639       if (buf.charAt(index) == '+') {
640         // Found an unescaped '+' character.
641         LOGGER.warn(
642             "DN \""
643                 + dn
644                 + "\" contains an unescaped '+'-character that will be"
645                 + " automatically escaped. RFC 2253 reservs this for"
646                 + " multi-valued RelativeDistinguishedNames. Encourage clients"
647                 + " to use '\\+' instead, since future behaviour might"
648                 + " change.");
649         buf.insert(index, '\\');
650         index++;
651       } else if (buf.charAt(index) == '\\') {
652         // Found an escape character.
653         index++;
654       }
655       index++;
656     }
657     return buf.toString();
658   }

801   /**
802    * Tries to determine if a DN is in reversed form. It does this by taking the
803    * last attribute and the first attribute. If the last attribute comes before
804    * the first in the dNObjects array the DN is assumed to be in reversed order.
805    *
806    * <p>The default ordering is: "CN=Tomas, O=PrimeKey, C=SE" (dNObjectsForward
807    * ordering in EJBCA) a dn or form "C=SE, O=PrimeKey, CN=Tomas" is reversed.
808    *
809    * <p>If the string has only one component (e.g. "CN=example.com") then this
810    * method returns false. If the string does not contain any real DN
811    * components, it returns false.
812    *
813    * @param dn String containing DN to be checked, The DN string has the format
814    *     "C=SE, O=xx, OU=yy, CN=zz".
815    * @return true if the DN is believed to be in reversed order, false otherwise
816    */
817   public static boolean isDNReversed(final String dn) {
818     /*
819      * if (log.isTraceEnabled()) { log.trace(">isDNReversed: dn: " + dn); }
820      */
821     boolean ret = false;
822     if (dn != null) {
823       String first = null;
824       String last = null;
825       final X509NameTokenizer xt = new X509NameTokenizer(dn);
826       if (xt.hasMoreTokens()) {
827         first = xt.nextToken().trim();
828       }
829       while (xt.hasMoreTokens()) {
830         last = xt.nextToken().trim();
831       }
832       final String[] dNObjects = DnComponents.getDnObjects(true);
833       if ((first != null) && (last != null)) {
834         // Be careful for bad input, that may not have any = sign in it
835         final int fi = first.indexOf('=');
836         first = first.substring(0, (fi != -1 ? fi : (first.length() - 1)));
837         final int li = last.indexOf('=');
838         last = last.substring(0, (li != -1 ? li : (last.length() - 1)));
839         int firsti = 0;
840         int lasti = 0;
841         for (int i = 0; i < dNObjects.length; i++) {
842           if (first.equalsIgnoreCase(dNObjects[i])) {
843             firsti = i;
844           }
845           if (last.equalsIgnoreCase(dNObjects[i])) {
846             lasti = i;
847           }
848         }
849         if (lasti < firsti) {
850           ret = true;
851         }
852       }
853     }
854     /*
855      * if (log.isTraceEnabled()) { log.trace("<isDNReversed: " + ret); }
856      */
857     return ret;
858   } // isDNReversed

911   /**
912    * @param dn DN
913    * @param dnPart Part
914    * @param onlyReturnFirstMatch bool
915    * @return parts
916    */
917   public static List<String> getPartsFromDNInternal(
918       final String dn,
919       final String dnPart,
920       final boolean onlyReturnFirstMatch) {
921     if (LOGGER.isTraceEnabled()) {
922       LOGGER.trace(
923           ">getPartsFromDNInternal: dn:'"
924               + dn
925               + "', dnpart="
926               + dnPart
927               + ", onlyReturnFirstMatch="
928               + onlyReturnFirstMatch);
929     }
930     final List<String> parts = new ArrayList<String>();
931     if (dn != null && dnPart != null) {
932       final String dnPartLowerCase = dnPart.toLowerCase();
933       final int dnPartLenght = dnPart.length();
934       boolean quoted = false;
935       boolean escapeNext = false;
936       int currentStartPosition = -1;
937       for (int i = 0; i < dn.length(); i++) {
938         final char current = dn.charAt(i);
939         // Toggle quoting for every non-escaped "-char
940         if (!escapeNext && current == '"') {
941           quoted = !quoted;
942         }
943         // If there is an unescaped and unquoted =-char we need to investigate
944         // if it is a match for the sought after part
945         if (!quoted && !escapeNext && current == '=' && dnPartLenght <= i) {
946           // Check that the character before our expected partName isn't a
947           // letter (e.g. dnsName=.. should not match E=..)
948           if (i - dnPartLenght - 1 < 0
949               || !Character.isLetter(dn.charAt(i - dnPartLenght - 1))) {
950             boolean match = true;
951             for (int j = 0; j < dnPartLenght; j++) {
952               if (Character.toLowerCase(dn.charAt(i - dnPartLenght + j))
953                   != dnPartLowerCase.charAt(j)) {
954                 match = false;
955                 break;
956               }
957             }
958             if (match) {
959               currentStartPosition = i + 1;
960             }
961           }
962         }
963         // When we have found a start marker, we need to be on the lookout for
964         // the ending marker
965         if (currentStartPosition != -1
966             && ((!quoted && !escapeNext && (current == ',' || current == '+'))
967                 || i == dn.length() - 1)) {
968           int endPosition = (i == dn.length() - 1) ? dn.length() - 1 : i - 1;
969           // Remove white spaces from the end of the value
970           while (endPosition > currentStartPosition
971               && dn.charAt(endPosition) == ' ') {
972             endPosition--;
973           }
974           // Remove white spaces from the beginning of the value
975           while (endPosition > currentStartPosition
976               && dn.charAt(currentStartPosition) == ' ') {
977             currentStartPosition++;
978           }
979           // Only return the inner value if the part is quoted
980           if (currentStartPosition != dn.length()
981               && dn.charAt(currentStartPosition) == '"'
982               && dn.charAt(endPosition) == '"') {
983             currentStartPosition++;
984             endPosition--;
985           }
986           parts.add(
987               unescapeFieldValue(
988                   dn.substring(currentStartPosition, endPosition + 1)));
989           if (onlyReturnFirstMatch) {
990             break;
991           }
992           currentStartPosition = -1;
993         }
994         if (escapeNext) {
995           // This character was escaped, so don't escape the next one
996           escapeNext = false;
997         } else {
998           if (!quoted && current == '\\') {
999             // This escape character is not escaped itself, so the next one
1000             // should be
1001             escapeNext = true;
1002           }
1003         }
1004       }
1005     }
1006     if (LOGGER.isTraceEnabled()) {
1007       LOGGER.trace(
1008           "<getPartsFromDNInternal: resulting DN part=" + parts.toString());
1009     }
1010     return parts;
1011   }

1096   /**
1097    * Gets subject or issuer DN in the format we are sure about
1098    * (BouncyCastle),supporting UTF8.
1099    *
1100    * @param cert X509Certificate
1101    * @param which 1 = subjectDN, anything else = issuerDN
1102    * @return String containing the DN.
1103    */
1104   private static String getDN(final Certificate cert, final int which) {
1105     String ret = null;
1106     if (cert == null) {
1107       return null;
1108     }
1109     if (cert instanceof X509Certificate) {
1110       // cert.getType=X.509
1111       try {
1112         final String clazz = cert.getClass().getName();
1113         // The purpose of the below generateCertificate is to create a BC
1114         // certificate object, because there we know how DN components
1115         // are handled. If we already have a BC certificate however, we can save
1116         // a lot of time to not have to encode/decode it.
1117         final X509Certificate x509cert;
1118         if (clazz.contains("org.bouncycastle")) {
1119           x509cert = (X509Certificate) cert;
1120         } else {
1121           final CertificateFactory cf = CertTools.getCertificateFactory();
1122           x509cert =
1123               (X509Certificate)
1124                   cf.generateCertificate(
1125                       new ByteArrayInputStream(cert.getEncoded()));
1126         }
1127         String dn = null;
1128         if (which == 1) {
1129           dn = x509cert.getSubjectDN().toString();
1130         } else {
1131           dn = x509cert.getIssuerDN().toString();
1132         }
1133         ret = stringToBCDNString(dn);
1134       } catch (final CertificateException ce) {
1135         LOGGER.info("Could not get DN from X509Certificate. "
1136                 + ce.getMessage());
1137         LOGGER.debug("", ce);
1138         return null;
1139       }
1140     } else if (StringUtils.equals(cert.getType(), "CVC")) {
1141       final CardVerifiableCertificate cvccert =
1142           (CardVerifiableCertificate) cert;
1143       try {
1144         ReferenceField rf = null;
1145         if (which == 1) {
1146           rf =
1147               cvccert
1148                   .getCVCertificate()
1149                   .getCertificateBody()
1150                   .getHolderReference();
1151         } else {
1152           rf =
1153               cvccert
1154                   .getCVCertificate()
1155                   .getCertificateBody()
1156                   .getAuthorityReference();
1157         }
1158         if (rf != null) {
1159           // Construct a "fake" DN which can be used in EJBCA
1160           // Use only mnemonic and country, since sequence is more of a
1161           // serialnumber than a DN part
1162           String dn = "";

1231   /**
1232    * Gets a serial number in numeric form, it takes - either a hex encoded
1233    * integer with length != 5 (x.509 certificate) - 5 letter numeric string
1234    * (cvc), will convert the number to an int - 5 letter alfanumeric string vi
1235    * some numbers in it (cvc), will convert the numbers in it to a numeric
1236    * string (remove the letters) and convert to int - 5 letter alfanumeric
1237    * string with only letters (cvc), will convert to integer from string with
1238    * radix 36.
1239    *
1240    * @param sernoString serial
1241    * @return BigInteger
1242    */
1243   public static BigInteger getSerialNumberFromString(final String sernoString) {
1244     if (sernoString == null) {
1245       throw new IllegalArgumentException(
1246           "getSerialNumberFromString: cert is null");
1247     }
1248     BigInteger ret;
1249     try {
1250         final int cvcLength = 5;
1251       if (sernoString.length() != cvcLength) {
1252         // This can not be a CVC certificate sequence, so it must be a hex
1253         // encoded regular certificate serial number
1254         ret = new BigInteger(sernoString, 16);
1255       } else {
1256         // We try to handle the different cases of CVC certificate sequences,
1257         // see StringTools.KEY_SEQUENCE_FORMAT
1258         if (NumberUtils.isNumber(sernoString)) {
1259           ret = NumberUtils.createBigInteger(sernoString);
1260         } else {
1261           // check if input is hexadecimal
1262           LOGGER.info(
1263               "getSerialNumber: Sequence is not a numeric string, trying to"
1264                   + " extract numerical sequence part.");
1265           final StringBuilder buf = new StringBuilder();
1266           for (int i = 0; i < sernoString.length(); i++) {
1267             final char c = sernoString.charAt(i);
1268             if (CharUtils.isAsciiNumeric(c)) {
1269               buf.append(c);
1270             }
1271           }
1272           if (buf.length() > 0) {
1273             ret = NumberUtils.createBigInteger(buf.toString());
1274           } else {
1275             LOGGER.info(
1276                 "getSerialNumber: can not extract numeric sequence part,"
1277                     + " trying alfanumeric value (radix 36).");
1278             if (sernoString.matches("[0-9A-Z]{1,5}")) {
1279               final int numSeq = Integer.parseInt(sernoString, 36);
1280               ret = BigInteger.valueOf(numSeq);
1281             } else {
1282               LOGGER.info(
1283                   "getSerialNumber: Sequence does not contain any numeric"
1284                       + " parts, returning 0.");
1285               ret = BigInteger.valueOf(0);
1286             }
1287           }
1288         }
1289       }
1290     } catch (final NumberFormatException e) {
1291       // If we can't make the sequence into a serial number big integer, set it
1292       // to 0
1293       LOGGER.debug(
1294           "getSerialNumber: NumberFormatException for sequence: "
1295               + sernoString);
1296       ret = BigInteger.valueOf(0);
1297     }
1298     return ret;

1650    * Reads certificates in PEM-format from an InputStream. The stream may
1651    * contain other things between the different certificates.
1652    *
1653    * @param certstream the input stream containing the certificates in
1654    *     PEM-format
1655    * @param returnType specifies the desired certificate type. Certificate can
1656    *     be used if certificate type is unknown.
1657    * @param <T> type
1658    * @return Ordered List of Certificates, first certificate first, or empty
1659    *     List
1660    * @exception CertificateParsingException if the stream contains an incorrect
1661    *     certificate.
1662    */
1663   public static <T extends Certificate> List<T> getCertsFromPEM(
1664       final InputStream certstream, final Class<T> returnType)
1665       throws CertificateParsingException {
1666     if (LOGGER.isTraceEnabled()) {
1667       LOGGER.trace(">getCertfromPEM");
1668     }
1669     final ArrayList<T> ret = new ArrayList<>();
1670     final String beginKeyTrust = "-----BEGIN TRUSTED CERTIFICATE-----";
1671     final String endKeyTrust = "-----END TRUSTED CERTIFICATE-----";
1672     BufferedReader bufRdr = null;
1673     ByteArrayOutputStream ostr = null;
1674     PrintStream opstr = null;
1675     try {
1676       try {
1677         bufRdr =
1678             new BufferedReader(
1679                 new InputStreamReader(
1680                     new SecurityFilterInputStream(certstream)));
1681         while (bufRdr.ready()) {
1682           ostr = new ByteArrayOutputStream();
1683           opstr = new PrintStream(ostr);
1684           String temp;
1685           while ((temp = bufRdr.readLine()) != null
1686               && !(temp.equals(CertTools.BEGIN_CERTIFICATE)
1687                   || temp.equals(beginKeyTrust))) {
1688             continue;
1689           }
1690           if (temp == null) {
1691             if (ret.isEmpty()) {
1692               // There was no certificate in the file
1693               throw new CertificateParsingException(
1694                   "Error in "
1695                       + certstream.toString()
1696                       + ", missing "
1697                       + CertTools.BEGIN_CERTIFICATE
1698                       + " boundary");
1699             } else {
1700               // There were certificates, but some blank lines or something in
1701               // the end
1702               // anyhow, the file has ended so we can break here.
1703               break;
1704             }
1705           }
1706           while ((temp = bufRdr.readLine()) != null
1707               && !(temp.equals(CertTools.END_CERTIFICATE)
1708                   || temp.equals(endKeyTrust))) {
1709             opstr.print(temp);
1710           }
1711           if (temp == null) {
1712             throw new IllegalArgumentException(
1713                 "Error in "
1714                     + certstream.toString()
1715                     + ", missing "
1716                     + CertTools.END_CERTIFICATE
1717                     + " boundary");
1718           }
1719           opstr.close();
1720 
1721           final byte[] certbuf = Base64.decode(ostr.toByteArray());
1722           ostr.close();
1723           // Phweeew, were done, now decode the cert from file back to
1724           // Certificate object
1725           final T cert = getCertfromByteArray(certbuf, returnType);
1726           ret.add(cert);
1727         }
1728 
1729       } finally {
1730         if (bufRdr != null) {
1731           bufRdr.close();
1732         }
1733         if (opstr != null) {
1734           opstr.close();
1735         }
1736         if (ostr != null) {
1737           ostr.close();
1738         }
1739       }
1740     } catch (final IOException e) {
1741       throw new IllegalStateException(
1742           "Exception caught when attempting to read stream, see underlying"
1743               + " IOException",
1744           e);
1745     }
1746     if (LOGGER.isTraceEnabled()) {
1747       LOGGER.trace("<getcertfromPEM:" + ret.size());
1748     }
1749     return ret;
1750   }

3802         names.add(new DERGeneralString(principalName));
3803       }
3804       principals.add(new DERTaggedObject(true, 1, new DERSequence(names)));
3805       krb5p.add(new DERTaggedObject(true, 1, new DERSequence(principals)));
3806 
3807       v.add(new DERTaggedObject(true, 0, new DERSequence(krb5p)));

3809           new DERTaggedObject(false, 0, new DERSequence(v));

3815         : CertTools.getPartsFromDN(
3816             altName, RFC4683Tools.SUBJECTIDENTIFICATIONMETHOD)) {

3820             internalSimString.split(RFC4683Tools.LIST_SEPARATOR);

3825           vec.add(gn);
3826           if (LOGGER.isDebugEnabled()) {
3827             LOGGER.debug("SIM GeneralName added: " + gn.toString());
3828           }
3829         }
3830       }
3831     }
3832 
3833     // To support custom OIDs in altNames, they must be added as an OtherName of
3834     // plain type UTF8String
3835     for (final String oid : CertTools.getCustomOids(altName)) {
3836       for (final String oidValue : CertTools.getPartsFromDN(altName, oid)) {
3837         final ASN1EncodableVector v = new ASN1EncodableVector();
3838         v.add(new ASN1ObjectIdentifier(oid));
3839         v.add(new DERTaggedObject(true, 0, new DERUTF8String(oidValue)));
3840         final ASN1Primitive gn =
3841             new DERTaggedObject(false, 0, new DERSequence(v));
3842         vec.add(gn);
3843       }
3844     }
3845 
3846     if (vec.size() > 0) {
3847       return GeneralNames.getInstance(new DERSequence(vec));
3848     }
3849     return null;

4871     private final String value;
4872     /** Index. */

4899     public String nextToken() {
4900       if (index == value.length()) {
4901         return null;
4902       }
4903 
4904       int end = index + 1;
4905       boolean quoted = false;
4906       boolean escaped = false;
4907 
4908       buf.setLength(0);
4909 
4910       while (end != value.length()) {
4911         final char c = value.charAt(end);
4912 
4913         if (c == '"') {

4915             quoted = !quoted;
4916           } else {
4917             if (c == '#' && buf.charAt(buf.length() - 1) == '=') {
4918               buf.append('\\');
4919             } else if (c == '+' && separator != '+') {
4920               buf.append('\\');
4921             }

4923           }
4924           escaped = false;
4925         } else {
4926           if (escaped || quoted) {

4928               buf.append('\\');
4929             } else if (c == '+' && separator != '+') {
4930               buf.append('\\');
4931             }
4932             buf.append(c);
4933             escaped = false;
4934           } else if (c == '\\') {
4935             escaped = true;
4936           } else if (c == separator) {

4938           } else {

5792   /**
5793    * Checks that the given SubjectDN / SAN satisfies the Name Constraints of the
5794    * given issuer (if there are any). This method checks the Name Constraints in
5795    * the given issuer only. A complete implementation of name constraints should
5796    * check the whole certificate chain.
5797    *
5798    * @param issuer Issuing CA.
5799    * @param subjectDNName Subject DN to check. Optional.
5800    * @param subjectAltName Subject Alternative Name to check. Optional.
5801    * @throws IllegalNameException if the name(s) didn't pass naming constraints
5802    */
5803   public static void checkNameConstraints(
5804       final X509Certificate issuer,
5805       final X500Name subjectDNName,
5806       final GeneralNames subjectAltName)
5807       throws IllegalNameException {
5808     final byte[] ncbytes =
5809         issuer.getExtensionValue(Extension.nameConstraints.getId());
5810     final ASN1OctetString ncstr =
5811         (ncbytes != null ? DEROctetString.getInstance(ncbytes) : null);
5812     final ASN1Sequence ncseq =
5813         (ncbytes != null ? DERSequence.getInstance(ncstr.getOctets()) : null);
5814     final NameConstraints nc =
5815         (ncseq != null ? NameConstraints.getInstance(ncseq) : null);
5816 
5817     if (nc != null) {
5818       if (subjectDNName != null) {
5819         // Skip check for root CAs
5820         final X500Name issuerDNName =
5821             X500Name.getInstance(issuer.getSubjectX500Principal().getEncoded());
5822         if (issuerDNName.equals(subjectDNName)) {
5823           return;
5824         }
5825       }
5826 
5827       final PKIXNameConstraintValidator validator =
5828           new PKIXNameConstraintValidator();
5829 
5830       final GeneralSubtree[] permitted = nc.getPermittedSubtrees();
5831       final GeneralSubtree[] excluded = nc.getExcludedSubtrees();
5832 
5833       if (permitted != null) {
5834         validator.intersectPermittedSubtree(permitted);
5835       }
5836       if (excluded != null) {
5837         for (final GeneralSubtree subtree : excluded) {
5838           validator.addExcludedSubtree(subtree);
5839         }
5840       }
5841 
5842       if (subjectDNName != null) {
5843         final GeneralName dngn = new GeneralName(subjectDNName);
5844         try {
5845           validator.checkPermitted(dngn);
5846           validator.checkExcluded(dngn);
5847         } catch (final PKIXNameConstraintValidatorException e) {
5848           final String dnStr = subjectDNName.toString();
5849           final boolean isLdapOrder =
5850               dnHasMultipleComponents(dnStr) && !isDNReversed(dnStr);
5851           if (isLdapOrder) {
5852             final String msg =
5853                 INTRES.getLocalizedMessage(
5854                     "nameconstraints.x500dnorderrequired");
5855             throw new IllegalNameException(msg);
5856           } else {
5857             final String msg =
5858                 INTRES.getLocalizedMessage(
5859                     "nameconstraints.forbiddensubjectdn", subjectDNName);
5860             throw new IllegalNameException(msg, e);
5861           }
5862         }
5863       }
5864 
5865       if (subjectAltName != null) {
5866         for (final GeneralName sangn : subjectAltName.getNames()) {
5867           try {
5868             validator.checkPermitted(sangn);
5869             validator.checkExcluded(sangn);
5870           } catch (final PKIXNameConstraintValidatorException e) {
5871             final String msg =
5872                 INTRES.getLocalizedMessage(
5873                     "nameconstraints.forbiddensubjectaltname", sangn);
5874             throw new IllegalNameException(msg, e);
5875           }
5876         }
5877       }
5878     }
5879   }
