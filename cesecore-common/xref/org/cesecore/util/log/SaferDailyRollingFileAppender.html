14  package org.cesecore.util.log;
15  
16  import java.io.File;
17  import java.net.MalformedURLException;
18  import java.net.URL;
19  
20  import org.apache.log4j.DailyRollingFileAppender;
21  import org.apache.log4j.helpers.LogLog;
22  import org.apache.log4j.spi.LoggingEvent;
23  
24  /**
25   * The purpose of this extension is to notify the client of the this log appender that it isn't possible to log anymore.
26   * 
27   * @version $Id: SaferDailyRollingFileAppender.java 18195 2013-11-25 08:51:43Z mikekushner $
28   */
29  public class SaferDailyRollingFileAppender extends DailyRollingFileAppender {
30      private static SaferAppenderListener subscriber;
31  
32      @Override
33      public void append(LoggingEvent evt) {
34          super.append(evt);
35          File logfile;
36          try {
37              logfile = new File(super.getFile());
38              if ((subscriber != null) && (logfile != null)) {
39                  if (logfile.canWrite()) {
40                      subscriber.setCanlog(true);
41                  } else {
42                      subscriber.setCanlog(false);
43                  }
44              }
45          } catch (Exception e) {
46              if (subscriber != null) {
47                  subscriber.setCanlog(false);
48              }
49          }
50      }
51  
52      /** Sets the SaferAppenderListener that will be informed if a logging error occurs. 
53       * @param pSubscriber subscriber*/
54      public static void addSubscriber(SaferAppenderListener pSubscriber) {
55          subscriber = pSubscriber;
56      }
57  
58      @Override
59      public void setFile(final String filename) {
60          constructPath(filename);
61          super.setFile(filename);
62      }
63  
64      private void constructPath(final String filename) {
65          File dir;
66          try {
67              URL url = new URL(filename.trim());
68              dir = new File(url.getFile()).getParentFile();
69          } catch (MalformedURLException e) {
70              dir = new File(filename.trim()).getParentFile();
71          }
72          if (!dir.exists()) {
73              boolean success = dir.mkdirs();
74              if (!success) {
75                  LogLog.error("Failed to create directory structure: " + dir);
76              }
77          }
78      }
79  }
