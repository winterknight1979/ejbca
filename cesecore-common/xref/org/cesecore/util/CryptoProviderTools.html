32  /**
33   * Basic crypto provider helper methods.

38  @SuppressWarnings("deprecation")
39  public final class CryptoProviderTools {
40  	
41  	private static final Logger log = Logger.getLogger(CryptoProviderTools.class);
42  			
43      private CryptoProviderTools() {} // Not for instantiation
44  
45      /** Parameters used when generating or verifying ECDSA keys/certs using the "implicitlyCA" key encoding.
46       * The curve parameters is then defined outside of the key and configured in the BC provider.
47       */
48      private static final String IMPLICITLYCA_Q = CesecoreConfiguration.getEcdsaImplicitlyCaQ();
49      private static final String IMPLICITLYCA_A = CesecoreConfiguration.getEcdsaImplicitlyCaA(); 
50      private static final String IMPLICITLYCA_B = CesecoreConfiguration.getEcdsaImplicitlyCaB(); 
51      private static final String IMPLICITLYCA_G = CesecoreConfiguration.getEcdsaImplicitlyCaG(); 
52      private static final String IMPLICITLYCA_N = CesecoreConfiguration.getEcdsaImplicitlyCaN();
53  
54      /** System provider used to circumvent a bug in Glassfish. Should only be used by 
55       * X509CAInfo, OCSPCAService, CMSCAService. 
56       * Defaults to SUN but can be changed to IBM by the installBCProvider method.
57       */
58      public static String SYSTEM_SECURITY_PROVIDER = "SUN";
59      
60      /**
61       * Detect if "Unlimited Strength" Policy files has bean properly installed.
62       * 
63       * @return true if key strength is limited
64       */
65      public static boolean isUsingExportableCryptography() {
66      	boolean returnValue = true;
67      	try {
68      		final int keylen = Cipher.getMaxAllowedKeyLength("DES");
69      		if (log.isDebugEnabled()) {
70      			log.debug("MaxAllowedKeyLength for DES is: "+keylen);
71      		}
72  			if ( keylen == Integer.MAX_VALUE ) {
73  				returnValue = false;
74  			}
75  		} catch (NoSuchAlgorithmException e) {
76  			// NOPMD
77  		}
78  		return returnValue;
79      }
80      
81      public static synchronized void installBCProviderIfNotAvailable() {
82      	if (Security.getProvider(BouncyCastleProvider.PROVIDER_NAME) == null) {
83      		installBCProvider();
84      	}
85      }
86  
87      public static synchronized void removeBCProvider() {
88          Security.removeProvider(BouncyCastleProvider.PROVIDER_NAME);  
89          // Also remove the CVC provider
90          Security.removeProvider("CVC");
91      }
92      @SuppressWarnings({ "unchecked" })
93      public static synchronized void installBCProvider() {
94      	
95          // A flag that ensures that we install the parameters for implcitlyCA only when we have installed a new provider
96          boolean installImplicitlyCA = false;
97          if (Security.addProvider(new BouncyCastleProvider()) < 0) {
98              // If already installed, remove so we can handle redeploy
99              // Nope, we ignore re-deploy on this level, because it can happen
100             // that the BC-provider is uninstalled, in just the second another
101             // thread tries to use the provider, and then that request will fail.
102             if (CesecoreConfiguration.isDevelopmentProviderInstallation()) {
103                 removeBCProvider();
104                 if (Security.addProvider(new BouncyCastleProvider()) < 0) {
105                     log.error("Cannot even install BC provider again!");
106                 } else {
107                     installImplicitlyCA = true;
108                 }
109             }
110         } else {
111             installImplicitlyCA = true;
112         }
113         
114     	// Also install the CVC provider
115     	try {
116         	Security.addProvider(new CVCProvider());    		
117     	} catch (Exception e) {
118     		log.info("CVC provider can not be installed, CVC certificate will not work: ", e);
119     	}
120     	
121         if (installImplicitlyCA) {
122             // Install EC parameters for implicitlyCA encoding of EC keys, we have default curve parameters if no new ones have been given.
123             // The parameters are only used if implicitlyCA is used for generating keys, or verifying certs
124             final ECCurve curve = new ECCurve.Fp(
125                     new BigInteger(IMPLICITLYCA_Q), // q
126                     new BigInteger(IMPLICITLYCA_A, 16), // a
127                     new BigInteger(IMPLICITLYCA_B, 16)); // b
128             final org.bouncycastle.jce.spec.ECParameterSpec implicitSpec = new org.bouncycastle.jce.spec.ECParameterSpec(
129                     curve,
130                     curve.decodePoint(Hex.decode(IMPLICITLYCA_G)), // G
131                     new BigInteger(IMPLICITLYCA_N)); // n
132             final ConfigurableProvider config = (ConfigurableProvider)Security.getProvider("BC");
133             if (config != null) {
134                 config.setParameter(ConfigurableProvider.EC_IMPLICITLY_CA, implicitSpec);                                               
135             } else {
136                 log.error("Can not get ConfigurableProvider, implicitlyCA EC parameters NOT set!");
137             }                
138         }
139         
140         // 2007-05-25
141         // Finally we must configure SERIALNUMBER behavior in BC >=1.36 to be the same
142         // as the behavior in BC 1.35, it changed from SN to SERIALNUMBER in BC 1.36
143         // We must be backwards compatible
144         
145         X509Name.DefaultSymbols.put(X509Name.SN, "SN");
146         // We hard specify the system security provider in a few cases (see SYSTEM_SECURITY_PROVIDER). 
147         // If the SUN provider does not exist, we will always use BC.
148         final Provider p = Security.getProvider(SYSTEM_SECURITY_PROVIDER);
149         if (p == null) {
150         	log.debug("SUN security provider does not exist, using BC as system default provider.");
151         	SYSTEM_SECURITY_PROVIDER = BouncyCastleProvider.PROVIDER_NAME;
152         }
153         
154     }
