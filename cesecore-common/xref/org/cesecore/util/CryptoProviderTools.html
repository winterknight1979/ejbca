39      /** Logger. */
40    private static final Logger LOG = Logger.getLogger(CryptoProviderTools.class);
41  
42    private CryptoProviderTools() { } // Not for instantiation

44    /**
45     * Parameters used when generating or verifying ECDSA keys/certs using the
46     * "implicitlyCA" key encoding. The curve parameters is then defined outside
47     * of the key and configured in the BC provider.
48     */
49    private static final String IMPLICITLYCA_Q =
50        CesecoreConfiguration.getEcdsaImplicitlyCaQ();
51    /** Config. */
52    private static final String IMPLICITLYCA_A =
53        CesecoreConfiguration.getEcdsaImplicitlyCaA();
54    /** Config. */
55    private static final String IMPLICITLYCA_B =
56        CesecoreConfiguration.getEcdsaImplicitlyCaB();
57    /** Config. */
58    private static final String IMPLICITLYCA_G =
59        CesecoreConfiguration.getEcdsaImplicitlyCaG();
60    /** Config. */
61    private static final String IMPLICITLYCA_N =
62        CesecoreConfiguration.getEcdsaImplicitlyCaN();
63  
64    /**
65     * System provider used to circumvent a bug in Glassfish. Should only be used
66     * by X509CAInfo, OCSPCAService, CMSCAService. Defaults to SUN but can be
67     * changed to IBM by the installBCProvider method.
68     */
69    private static String systemSecurity = "SUN";
70  
71    /**
72     * Detect if "Unlimited Strength" Policy files has bean properly installed.
73     *
74     * @return true if key strength is limited
75     */
76    public static boolean isUsingExportableCryptography() {
77      boolean returnValue = true;
78      try {
79        final int keylen = Cipher.getMaxAllowedKeyLength("DES");
80        if (LOG.isDebugEnabled()) {
81          LOG.debug("MaxAllowedKeyLength for DES is: " + keylen);
82        }
83        if (keylen == Integer.MAX_VALUE) {
84          returnValue = false;
85        }
86      } catch (NoSuchAlgorithmException e) {
87        // NOPMD
88      }
89      return returnValue;
90    }
91  
92    /** Install Bouncy CAstle. */
93    public static synchronized void installBCProviderIfNotAvailable() {
94      if (Security.getProvider(BouncyCastleProvider.PROVIDER_NAME) == null) {
95        installBCProvider();
96      }
97    }
98  
99    /** Remove (deflate?) Bouncy CAstle. */
100   public static synchronized void removeBCProvider() {
101     Security.removeProvider(BouncyCastleProvider.PROVIDER_NAME);
102     // Also remove the CVC provider
103     Security.removeProvider("CVC");
104   }
105 
106   /** Install Bouncy Castle. */
107   @SuppressWarnings({"unchecked"})
108   public static synchronized void installBCProvider() {
109 
110     // A flag that ensures that we install the parameters for implcitlyCA only
111     // when we have installed a new provider
112     boolean installImplicitlyCA = false;
113     if (Security.addProvider(new BouncyCastleProvider()) < 0) {
114       // If already installed, remove so we can handle redeploy
115       // Nope, we ignore re-deploy on this level, because it can happen
116       // that the BC-provider is uninstalled, in just the second another
117       // thread tries to use the provider, and then that request will fail.
118       if (CesecoreConfiguration.isDevelopmentProviderInstallation()) {
119         removeBCProvider();
120         if (Security.addProvider(new BouncyCastleProvider()) < 0) {
121           LOG.error("Cannot even install BC provider again!");
122         } else {
123           installImplicitlyCA = true;
124         }
125       }
126     } else {
127       installImplicitlyCA = true;
128     }
129 
130     // Also install the CVC provider
131     try {
132       Security.addProvider(new CVCProvider());
133     } catch (Exception e) {
134       LOG.info(
135           "CVC provider can not be installed, CVC certificate will not work: ",
136           e);
137     }
138 
139     if (installImplicitlyCA) {
140       // Install EC parameters for implicitlyCA encoding of EC keys, we have
141       // default curve parameters if no new ones have been given.
142       // The parameters are only used if implicitlyCA is used for generating
143       // keys, or verifying certs
144       final ECCurve curve =
145           new ECCurve.Fp(
146               new BigInteger(IMPLICITLYCA_Q), // q
147               new BigInteger(IMPLICITLYCA_A, 16), // a
148               new BigInteger(IMPLICITLYCA_B, 16)); // b
149       final org.bouncycastle.jce.spec.ECParameterSpec implicitSpec =
150           new org.bouncycastle.jce.spec.ECParameterSpec(
151               curve,
152               curve.decodePoint(Hex.decode(IMPLICITLYCA_G)), // G
153               new BigInteger(IMPLICITLYCA_N)); // n
154       final ConfigurableProvider config =
155           (ConfigurableProvider) Security.getProvider("BC");
156       if (config != null) {
157         config.setParameter(
158             ConfigurableProvider.EC_IMPLICITLY_CA, implicitSpec);
159       } else {
160         LOG.error(
161             "Can not get ConfigurableProvider, implicitlyCA EC parameters NOT"
162                 + " set!");
163       }
164     }
165 
166     // 2007-05-25
167     // Finally we must configure SERIALNUMBER behavior in BC >=1.36 to be the
168     // same
169     // as the behavior in BC 1.35, it changed from SN to SERIALNUMBER in BC 1.36
170     // We must be backwards compatible
171 
172     X509Name.DefaultSymbols.put(X509Name.SN, "SN");
173     // We hard specify the system security provider in a few cases (see
174     // SYSTEM_SECURITY_PROVIDER).
175     // If the SUN provider does not exist, we will always use BC.
176     final Provider p = Security.getProvider(systemSecurity);
177     if (p == null) {
178       LOG.debug(
179           "SUN security provider does not exist, using BC as system default"
180               + " provider.");
181       systemSecurity = BouncyCastleProvider.PROVIDER_NAME;
182     }
183   }
