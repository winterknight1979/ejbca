14  package org.cesecore.util;
15  
16  import java.io.ByteArrayOutputStream;
17  
18  /**
19   * This class implements a BASE64 Character encoder/decoder as specified in
20   * RFC1521. It extends the bouncycastle implementation and adds the
21   * functionality to split lines with a '\n' after every 64 bytes.
22   *
23   * @version $Id: Base64.java 25733 2017-04-23 20:14:32Z joseromero $
24   */
25  public final class Base64 {
26  
27    private Base64() { } // Not for instantiation
28  
29    /**
30     * encode the input data producing a base 64 encoded byte array with the
31     * output lines be split by '\n' (64 byte rows).
32     *
33     * @param data data to be encoded
34     * @return a byte array containing the base 64 encoded data.
35     */
36    public static byte[] encode(final byte[] data) {
37      return encode(data, true);
38    }
39    /**
40     * encode the input data producing a base 64 encoded byte array.
41     *
42     * @param data the data to be encoded
43     * @param splitlines whether the output lines will be split by '\n' (64 byte
44     *     rows) or not.
45     * @return a byte array containing the base 64 encoded data.
46     */
47    public static byte[] encode(final byte[] data, final boolean splitlines) {
48  
49      final int lineLength = 64;
50      byte[] bytes = org.bouncycastle.util.encoders.Base64.encode(data);
51      if (splitlines) {
52        // make sure we get limited lines...
53        final ByteArrayOutputStream os = new ByteArrayOutputStream();
54        for (int i = 0; i < bytes.length; i += lineLength) {
55          if ((i + lineLength) < bytes.length) {
56            os.write(bytes, i, lineLength);
57            os.write('\n');
58          } else {
59            os.write(bytes, i, bytes.length - i);
60          }
61        }
62        bytes = os.toByteArray();
63      }
64      return bytes;
65    }
66  
67    /**
68     * @param token Token
69     * @return Decoded
70     */
71    public static byte[] decodeURLSafe(final String token) {
72      return org.apache.commons.codec.binary.Base64.decodeBase64(
73          token.getBytes());
74    }
75  
76    /**
77     * @param bytes Bytes
78     * @return Base64
79     */
80    public static byte[] decode(final byte[] bytes) {
81      return org.bouncycastle.util.encoders.Base64.decode(bytes);
82    }
83  }
