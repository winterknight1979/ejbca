14  package org.cesecore.util;
15  
16  import java.io.ByteArrayOutputStream;
17  
18  
19  /**
20   * This class implements a BASE64 Character encoder/decoder as specified in RFC1521. 
21   * It extends the bouncycastle implementation and adds the functionality to split lines 
22   * with a '\n' after every 64 bytes.
23   *
24   * @version $Id: Base64.java 25733 2017-04-23 20:14:32Z joseromero $
25   */
26  public final class Base64 {
27  
28      private Base64 () {} // Not for instantiation
29  
30      /**
31       * encode the input data producing a base 64 encoded byte array with the output lines be split by '\n' (64 byte rows).
32       *
33       * @param data data to be encoded
34       * @return a byte array containing the base 64 encoded data.
35       */
36      public static byte[] encode(final byte[] data) {
37          return encode(data, true);
38      }
39  
40  	/**
41       * encode the input data producing a base 64 encoded byte array.
42       *
43       * @param data the data to be encoded
44       * @param splitlines whether the output lines will be split by '\n' (64 byte rows) or not.
45       * @return a byte array containing the base 64 encoded data.
46       */
47      public static byte[] encode(final byte[] data, final boolean splitlines) {
48  		byte[] bytes = org.bouncycastle.util.encoders.Base64.encode(data);
49          if (splitlines) {
50              // make sure we get limited lines...
51              final ByteArrayOutputStream os = new ByteArrayOutputStream();
52              for (int i = 0; i < bytes.length; i += 64) {
53                  if ((i + 64) < bytes.length) {
54                      os.write(bytes, i, 64);
55                      os.write('\n');
56                  } else {
57                      os.write(bytes, i, bytes.length - i);
58                  }
59              }
60              bytes = os.toByteArray();
61          }
62          return bytes;
63  
64      }
65  
66      public static byte[] decodeURLSafe(String token){
67          return org.apache.commons.codec.binary.Base64.decodeBase64(token.getBytes());
68      }
69      public static byte[] decode(final byte[] bytes) {
70          return org.bouncycastle.util.encoders.Base64.decode(bytes);
71      }
72  
73  }
