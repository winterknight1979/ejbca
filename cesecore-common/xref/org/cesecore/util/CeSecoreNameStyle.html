201     /**
202      * This method is intended to be used in toString() in BCStyle classes. It is
203      * useful e.g. when the DefaultSymbols map is not the default inherited from
204      * BCStyle. It is public so it can be re-used by other classes as well
205      * (e.g. LdapNameStyle in EJBCA).
206      * @param defaultSymbols symbols
207      * @param name Name
208      * @return String
209      */
210     public static String buildString(Hashtable<ASN1ObjectIdentifier,String> defaultSymbols, X500Name name) {
211         StringBuffer buf = new StringBuffer();
212         boolean first = true;
213 
214         RDN[] rdns = name.getRDNs();
215 
216         for (int i = 0; i < rdns.length; i++) {
217             if (first) {
218                 first = false;
219             } else {
220                 buf.append(',');
221             }
222 
223             if (rdns[i].isMultiValued()) {
224                 AttributeTypeAndValue[] atv = rdns[i].getTypesAndValues();
225                 boolean firstAtv = true;
226 
227                 for (int j = 0; j != atv.length; j++) {
228                     if (firstAtv) {
229                         firstAtv = false;
230                     } else {
231                         buf.append('+');
232                     }
233 
234                     IETFUtils.appendTypeAndValue(buf, atv[j], defaultSymbols);
235                 }
236             } else {
237                 IETFUtils.appendTypeAndValue(buf, rdns[i].getFirst(), defaultSymbols);
238             }
239         }
240 
241         return buf.toString();
242     }
243 
244     @Override
245     public String toString(X500Name name) {
246         return buildString(DefaultSymbols, name);
247     }
248 
249     @Override
250     public ASN1Encodable stringToValue(ASN1ObjectIdentifier oid, String value) {
251         // JurisdictionCountry is not included in BC (at least up to and including 1.49), and must be PrintableString
252         if (oid.equals(CeSecoreNameStyle.JURISDICTION_COUNTRY)) {
253             return new DERPrintableString(value);
254         }
255         return super.stringToValue(oid, value);
256     }
257 
258     @Override
259     public ASN1ObjectIdentifier attrNameToOID(String attrName)
260     {
261         return IETFUtils.decodeAttrName(attrName, DefaultLookUp);
262     }
