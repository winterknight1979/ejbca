13  package org.cesecore.util.query;
14  
15  import java.io.Serializable;
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  /**
20   * Query Criteria DSL.
21   *
22   * <p>This object will not contain the generated query itself. Generation and
23   * validation will be done server-side.
24   *
25   * <p>This will not guarantee that the conjunction of all restrictions will
26   * result in a good query. Restrictions order should be guaranteed by the
27   * third-party lib. If not it will result in a malformed query.
28   *
29   * <p>TODO: For now there is no support for prioritizing certain conditions
30   * using parenthesis but with the current queyr generator this could be easily
31   * supported.
32   *
33   * <p><em>For usage example @see QueryCriteriaTest</em>
34   *
35   * @version $Id: QueryCriteria.java 17625 2013-09-20 07:12:06Z netmackan $
36   */
37  public final class QueryCriteria implements Serializable {
38  
39    private static final long serialVersionUID = 1823996498624633259L;
40    /** Elements. */
41    private final List<Elem> elements = new ArrayList<Elem>();
42  
43    private QueryCriteria() { }
44  
45    /** @return a new QueryCriteria */
46    public static QueryCriteria create() {
47      return new QueryCriteria();
48    }
49  
50    /**
51     * Adds a new Criteria (Restriction).
52     *
53     * @param elem element to be added
54     * @return QueryCriteria instance for chained calls.
55     */
56    public QueryCriteria add(final Elem elem) {
57      elements.add(elem);
58      return this;
59    }
60  
61    /**
62     * @return Elements.
63     */
64    public List<Elem> getElements() {
65      return elements;
66    }
67  }
