33  /**
34   * Class responsible for Query Generation.
35   *
36   * <p>Queries will be generated according to provided Criteria. Each criteria
37   * (composing Terms) will be subdued to validation.
38   *
39   * <p><em>For usage examples @see QueryCriteriaTest</em>
40   *
41   * @version $Id: QueryGenerator.java 27999 2018-01-18 09:17:13Z mikekushner $
42   */
43  public final class QueryGenerator implements Serializable {
44  
45    private static final long serialVersionUID = 1567027442267416376L;
46    /** Params. */
47    private final Map<String, Object> parameters =
48        new LinkedHashMap<String, Object>();
49    /** Qurey. */
50    private final Query query;
51    /** Fields. */
52    private final List<String> availableFields = new ArrayList<String>();
53    /** Criteria. */
54    private final QueryCriteria criteria;
55  
56    /** Class holding query construction logic. */
57    private final class Query {
58        /** query. */
59      private final StringBuilder query = new StringBuilder();
60      /** alias. */
61      private final String attrAlias;
62      /** Where. */
63      private static final String WHERE = "WHERE";
64      /** Sep. */
65      private static final String SEPARATOR = " ";
66  
67      Query(final String anAttrAlias) {
68        this.attrAlias = anAttrAlias;
69      }
70  
71      public Query where() {
72        query.append(SEPARATOR).append(WHERE);
73        return this;
74      }
75  
76      public Query attribute(final String name) {
77        query.append(SEPARATOR).append(attrAlias).append(".").append(name);
78        return this;
79      }
80  
81      public Query parameter(final String name) {
82        query.append(SEPARATOR).append(":" + name);
83        return this;
84      }
85  
86      private Query operator(final RelationalOperator op) {
87        final String operator =
88            op == RelationalOperator.EQ
89                ? "="
90                : op == RelationalOperator.GE
91                    ? ">="
92                    : op == RelationalOperator.GT
93                        ? ">"
94                        : op == RelationalOperator.LE
95                            ? "<="
96                            : op == RelationalOperator.LT
97                                ? "<"
98                                : op == RelationalOperator.NEQ
99                                    ? "!="
100                                   : op == RelationalOperator.BETWEEN
101                                       ? "BETWEEN"
102                                       : op == RelationalOperator.LIKE
103                                           ? "LIKE"
104                                           : op == RelationalOperator.NULL
105                                               ? "IS NULL"
106                                               : op == RelationalOperator.NOTNULL
107                                                   ? "IS NOT NULL"
108                                                   : "";
109       if (operator.isEmpty()) {
110         throw new QueryParameterException("operator not recognized");
111       }
112       query.append(SEPARATOR).append(operator);
113       return this;
114     }
115 
116     private Query operator(final LogicOperator op) {
117       query.append(SEPARATOR).append(op.toString());
118       return this;
119     }
120 
121     public Query order(final String name, final Order.Value order) {
122       query
123           .append(SEPARATOR)
124           .append("ORDER BY")
125           .append(SEPARATOR)
126           .append(attrAlias)
127           .append(".")
128           .append(name)
129           .append(SEPARATOR)
130           .append(order.toString());
131       return this;
132     }
133 
134     public boolean isEmpty() {
135       return query.length() == 0;
136     }
137 
138     public String toString() {
139       return query.toString();
140     }
141   }
142 
143   private QueryGenerator(
144       final Class<?> clazz,
145       final QueryCriteria aCriteria,
146       final String alias) {
147     query = new Query(alias);
148     this.criteria = aCriteria;
149     for (final Field f : clazz.getDeclaredFields()) {
150       availableFields.add(f.getName());
151     }
152   }
153 
154   /**
155    * Returns a new QueryGenerator or null if criteria is null.
156    *
157    * @param clazz class
158    * @param criteria QueryCriteria
159    * @param attrAlias alias
160    * @return QueryGenerator or null if criteria is null
161    */
162   public static QueryGenerator generator(
163       final Class<?> clazz,
164       final QueryCriteria criteria,
165       final String attrAlias) {
166     if (criteria == null) {
167       return null;
168     }
169     return new QueryGenerator(clazz, criteria, attrAlias);
170   }
171 
172   /**
173    * Generates the SQL query according to the criteria passed in generator.
174    *
175    * @return generated Query.
176    */
177   public String generate() {
178     if (query.isEmpty()) {
179       final List<Elem> elements = criteria.getElements();
180       final List<Elem> terms = new ArrayList<Elem>();
181       final List<Elem> clauses = new ArrayList<Elem>();
182 
183       CollectionUtils.selectRejected(
184           elements, PredicateUtils.instanceofPredicate(Order.class), terms);
185       CollectionUtils.select(
186           elements, PredicateUtils.instanceofPredicate(Order.class), clauses);
187 
188       if (terms.size() > 0) {
189         query.where();
190       }
191       termTraversal(terms);
192       clauseTraversal(clauses);
193     }
194     return query.toString();
195   }
196 
197   /**
198    * Traverses the terms list that is constructed according to the elements list
199    * in the QueryCriteria.
200    *
201    * @param elements elements
202    */
203   private void termTraversal(final List<Elem> elements) {
204     boolean first = true;
205     for (final Elem element : elements) {
206       if (!first) {
207         query.operator(LogicOperator.AND);
208       } else {
209         first = false;
210       }
211       generate(element);
212     }
213   }
214 
215   /**
216    * Traverses the clauses list that is constructed according to the elements
217    * list in the QueryCriteria.
218    *
219    * @param clauses clauses
220    */
221   private void clauseTraversal(final List<Elem> clauses) {
222     for (final Elem clause : clauses) {
223       generate(clause);
224     }
225   }
226 
227   /**
228    * Partial query generation according to the provided element.
229    *
230    * @param elem Term or Operation or Order object
231    */
232   private void generate(final Elem elem) {
233     if (elem instanceof Operation) {
234       generateRestriction((Operation) elem);
235     } else if (elem instanceof Term) {
236       generateRestriction((Term) elem);
237     } else if (elem instanceof Order) {
238       generateRestriction((Order) elem);
239     } else {
240       throw new QueryParameterException("No matched restriction");
241     }
242   }
243 
244   private void generateRestriction(final Operation op) {
245     generateRestriction(op.getTerm());
246     query.operator(op.getOperator());
247     final Elem elem = op.getElement();
248     if (elem != null) {
249       generate(elem);
250     }
251   }
252 
253   private void generateRestriction(final Term term) {
254     validate(term.getName());
255     query.attribute(term.getName()).operator(term.getOperator());
256     if (term.getOperator() == RelationalOperator.BETWEEN) {
257       @SuppressWarnings("unchecked")
258       final AbstractMap.SimpleEntry<Object, Object> values =
259           (AbstractMap.SimpleEntry<Object, Object>) term.getValue();
260       query
261           .parameter(genAndStoreParameter(term.getName(), values.getKey()))
262           .operator(LogicOperator.AND)
263           .parameter(genAndStoreParameter(term.getName(), values.getValue()));
264     } else if (term.getOperator() != RelationalOperator.NULL
265         && term.getOperator() != RelationalOperator.NOTNULL) {
266       query.parameter(genAndStoreParameter(term.getName(), term.getValue()));
267     }
268   }
269 
270   private void generateRestriction(final Order order) {
271     validate(order.getName());
272     query.order(order.getName(), order.getOrder());
273   }

275   /** @return all stored parameter names. */
276   public Set<String> getParameterKeys() {
277     return parameters.keySet();
278   }
279 
280   /**
281    * @param key key
282    * @return the value of a stored parameter
283    */
284   public Object getParameterValue(final String key) {
285     return parameters.get(key);
286   }
287 
288   /**
289    * Validates the provided name against our naming strategy ... columns with
290    * alphanumeric chars only.
291    *
292    * @param name to be validated
293    */
294   private void validate(final String name) {
295     if (!StringUtils.isAlphanumeric(name)) {
296       throw new QueryParameterException(
297           "parameter is not alphanumeric " + name);
298     }
299     if (!availableFields.contains(name)) {
300       throw new QueryParameterException("parameter is not valid field " + name);
301     }
302   }
303 
304   /**
305    * Generated a valid parameter name. Uses an internal store to associate the
306    * generated name (parameter) to a value.
307    *
308    * @param name that will be used as a seed to the parameter name
309    * @param value to be associated with the parameter name
310    * @return parameter name
311    */
312   private String genAndStoreParameter(final String name, final Object value) {
313     int i = 0;
314     while (true) {
315       final String parameter = name + i++;
316       if (!parameters.containsKey(parameter)) {
317         parameters.put(parameter, value);
318         return parameter;
319       }
320     }
321   }
