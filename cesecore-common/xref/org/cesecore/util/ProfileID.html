13  package org.cesecore.util;
14  
15  import java.util.Random;
16  import org.apache.log4j.Logger;
17  
18  /**
19   * Used to get an ID not already existing in a DB.
20   *
21   * @version $Id: ProfileID.java 23749 2016-06-30 11:12:39Z mikekushner $
22   */
23  public final class ProfileID {
24      private ProfileID() { }
25    /** Logger. */
26      private static final Logger LOG = Logger.getLogger(ProfileID.class);
27    /** Tandom. */
28    private static final Random RANDOM = new Random();
29    /** Minimum. */
30    private static final int MIN = 1 << 16;
31    /**
32     * Callback that is used from caller to check if the generated id can be used,
33     * or if we should generate a new one.
34     */
35    public interface DB {
36      /**
37       * Test if an ID is not already used in a DB.
38       *
39       * @param i id to test.
40       * @return true if not already used in DB
41       */
42      boolean isFree(int i);
43    }
44    /**
45     * @param db The {@link DB} implementation.
46     * @return The ID to be used. Always returning &gt;0xffff. IDs &lt;0x10000 are
47     *     reserved for constant defined in the code.
48     */
49    public static int getNotUsedID(final DB db) {
50      for (int i = 0; i < 10; i++) {
51        final int id = getRandomIdNumber();
52        if (db.isFree(id)) {
53          return id;
54        }
55        LOG.info("ID " + id + " already exist in DB generating a new random ID.");
56      }
57      // this throw is indicating an implementation error of the DB class and
58      // should never occur. If it does the code must be fixed.
59      throw new RuntimeException(
60          "Impossible to find a spare ID in the database for the class: "
61              + db.getClass().getCanonicalName());
62    }
63  
64    /**
65     * @return random ID
66     */
67    public static int getRandomIdNumber() {
68      return RANDOM.nextInt(Integer.MAX_VALUE - MIN) + MIN;
69    }
70  }
