22  /**
23   * Helper class for handling user friendly format of time intervals.
24   *
25   * <p>The format is in the form 'y* mo* *d *h *m *s *ms' where * is a decimal
26   * number and d=days, h=hours, m=minutes, s=seconds, ms=milliseconds. Spaces are
27   * optional.
28   *
29   * @version $Id: SimpleTime.java 24744 2016-11-15 14:36:16Z anatom $
30   */
31  public final class SimpleTime {
32  
33        /** Config. */
34    public static final long MILLISECONDS_PER_YEAR = 31536000000L; // 365 days
35    /** Config. */
36    public static final long MILLISECONDS_PER_MONTH = 2592000000L; // 30 days
37    /** Config. */
38    public static final long MILLISECONDS_PER_DAY = 86400000L;
39    /** Config. */
40    public static final long MILLISECONDS_PER_HOUR = 3600000L;
41    /** Config. */
42    public static final long MILLISECONDS_PER_MINUTE = 60000L;
43    /** Config. */
44    public static final long MILLISECONDS_PER_SECOND = 1000L;
45  
46    /** Config. */
47    public static final String TYPE_YEARS = "y";
48    /** Config. */
49    public static final String TYPE_MONTHS = "mo";
50    /** Config. */
51    public static final String TYPE_DAYS = "d";
52    /** Config. */
53    public static final String TYPE_HOURS = "h";
54    /** Config. */
55    public static final String TYPE_MINUTES = "m";
56    /** Config. */
57    public static final String TYPE_SECONDS = "s";
58    /** Config. */
59    public static final String TYPE_MILLISECONDS = "ms";
60    /** MS. */
61    public static final String PRECISION_MILLISECONDS = "milliseconds";
62    /** Secs. */
63    public static final String PRECISION_SECONDS = "seconds";
64    /** Days. */
65    public static final String PRECISION_DAYS = "days";
66  
67    /** Lost. */
68    public static final List<String> AVAILABLE_PRECISIONS =
69        Arrays.asList(
70            new String[] {
71              PRECISION_MILLISECONDS, PRECISION_SECONDS, PRECISION_DAYS
72            });

74    /** Map. */
75    private static final Map<String, Long> MILLISECONDS_FACTOR =
76        new LinkedHashMap<String, Long>();
77  
78    static {
79      MILLISECONDS_FACTOR.put(TYPE_YEARS, MILLISECONDS_PER_YEAR);
80      MILLISECONDS_FACTOR.put(TYPE_MONTHS, MILLISECONDS_PER_MONTH);
81      MILLISECONDS_FACTOR.put(TYPE_DAYS, MILLISECONDS_PER_DAY);
82      MILLISECONDS_FACTOR.put(TYPE_HOURS, MILLISECONDS_PER_HOUR);
83      MILLISECONDS_FACTOR.put(TYPE_MINUTES, MILLISECONDS_PER_MINUTE);
84      MILLISECONDS_FACTOR.put(TYPE_SECONDS, MILLISECONDS_PER_SECOND);
85      MILLISECONDS_FACTOR.put(TYPE_MILLISECONDS, 1L);
86    }

150   /**
151    * @param time AdBhCmDsEu meaning A days, B hours, C minutes, D seconds and E
152    *     milliseconds
153    * @throws Exception if unable to parse a String
154    */
155   private SimpleTime(final String time) throws Exception {

157   }

159   /**
160    * @param otime AdBhCmDsEu meaning A days, B hours, C minutes, D seconds and E
161    *     milliseconds
162    * @param defaultTime AdBhCmDsEu meaning A days, B hours, C minutes, D seconds
163    *     and E milliseconds
164    * @throws Exception if unable to parse a String
165    */
166   private SimpleTime(
167           final String otime, final String defaultTime) throws Exception {
168       String time = otime;
169     if (time == null || time.trim().length() == 0) {
170       time = defaultTime;
171     }
172     setTime(parseMillies(time));
173   }

260   /**
261    * Gets the TimeUnitFormat by precision.
262    *
263    * @param precision precision
264    * @return the TimeUnitFormat with the desired precision if existent.
265    * @throws IllegalArgumentException if invalid args
266    * @see SimpleTime#AVAILABLE_PRECISIONS
267    */
268   public static TimeUnitFormat getTimeUnitFormatOrThrow(
269       final String precision) throws IllegalArgumentException {
270     TimeUnitFormat result = null;
271     if (!AVAILABLE_PRECISIONS.contains(precision)) {
272       throw new IllegalArgumentException(
273           "Could not get TimeUnitForm for precision: " + precision);
274     }
275     switch (precision) {
276       case SimpleTime.PRECISION_MILLISECONDS:
277         result = SimpleTime.getMilliSecondsFormat();
278         break;
279       case SimpleTime.PRECISION_SECONDS:
280         result = SimpleTime.getSecondsFormat();
281         break;
282       case SimpleTime.PRECISION_DAYS:
283         result = SimpleTime.getDaysFormat();
284         break;
285       default:
286         result = SimpleTime.getSecondsFormat();
287     }
288     return result;
289   }

302   /**
303    * @param time MS
304    * @return time
305    * @throws NumberFormatException fail
306    */
307   public static long parseMillies(final String time)
308       throws NumberFormatException {
309     return SimpleTime.getMilliSecondsFormat().parseMillis(time);
310   }

312   private void setTime(final long otime) {
313     long time = otime;
314     longTime = time;
315     years = time / MILLISECONDS_PER_YEAR;
316     time %= MILLISECONDS_PER_YEAR;
317     months = time / MILLISECONDS_PER_MONTH;
318     time %= MILLISECONDS_PER_MONTH;
319     days = time / MILLISECONDS_PER_DAY;
320     time %= MILLISECONDS_PER_DAY;
321     hours = time / MILLISECONDS_PER_HOUR;
322     time %= MILLISECONDS_PER_HOUR;
323     minutes = time / MILLISECONDS_PER_MINUTE;
324     time %= MILLISECONDS_PER_MINUTE;
325     seconds = time / MILLISECONDS_PER_SECOND;
326     time %= MILLISECONDS_PER_SECOND;
327     milliSeconds = time;
328   }
