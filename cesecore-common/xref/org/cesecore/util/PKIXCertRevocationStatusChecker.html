73  /**
74   * A class to check whether a certificate is revoked or not using either OCSP or CRL. 
75   * The revocation status will first be obtained using OCSP. If it turned out that that was not possible for 
76   * some reason, a CRL will be used instead. If it was not possible to check the CRL for some reason, an 
77   * exception will be thrown.
78   * 
79   * @version $Id: PKIXCertRevocationStatusChecker.java 24567 2016-10-25 05:36:17Z anatom $
80   *
81   */
82  public class PKIXCertRevocationStatusChecker extends PKIXCertPathChecker {

100      * if 'null', the issuer certificate will be looked for among the certificates specified in 'cacerts'
101      * @param cacerts A collection of certificates where one of them is the certificate of the issuer of the certificate

170     /**
171      * Checks the revocation status of 'cert'; first by sending on OCSP request. If that fails for any reason, then through a CRL
172      */
173     @Override
174     public void check(Certificate cert, Collection<String> unresolvedCritExts) throws CertPathValidatorException {
175         
176         clearResult();
177         Certificate cacert = getCaCert(cert);
178         if(cacert == null) {
179             final String msg = "No issuer CA certificate was found. An issuer CA certificate is needed to create an OCSP request and to get the right CRL"; 
180             log.info(msg);
181             throw new CertPathValidatorException(msg);
182         }
183         
184         ArrayList<String> ocspurls = getOcspUrls(cert);
185         if(!ocspurls.isEmpty()) {
186             BigInteger certSerialnumber = CertTools.getSerialNumber(cert);
187             byte[] nonce = new byte[16];
188             final Random randomSource = new Random();
189             randomSource.nextBytes(nonce);
190             OCSPReq req = null;
191             try {
192                 req = getOcspRequest(cacert, certSerialnumber, nonce);
193             } catch (CertificateEncodingException | OCSPException e) {
194                 if(log.isDebugEnabled()) {
195                     log.debug("Failed to create OCSP request. " + e.getLocalizedMessage());
196                 }
197                 fallBackToCrl(cert, CertTools.getSubjectDN(cacert));
198                 return;
199                 
200             }
201             
202             SingleResp ocspResp = null;
203             for(String url : ocspurls) {
204                 ocspResp = getOCSPResponse(url, req, cert, nonce, OCSPRespBuilder.SUCCESSFUL, 200);
205                 if(ocspResp != null) {
206                     log.info("Obtained OCSP response from " + url);
207                     break;
208                 } else {
209                     if(log.isDebugEnabled()) {
210                         log.debug("Failed to obtain an OCSP reponse from " + url);
211                     }
212                 }
213             }
214             
215             if(ocspResp==null) {
216                 log.info("Failed to check certificate revocation status using OCSP. Falling back to check using CRL");
217                 fallBackToCrl(cert, CertTools.getSubjectDN(cacert));
218             } else {
219                 CertificateStatus status = ocspResp.getCertStatus();
220                 this.ocspResponse = ocspResp;
221                 if(log.isDebugEnabled()) {
222                     log.debug("The certificate status is: " + (status==null? "Good" : status.toString()));
223                 }
224                 if(status != null) { // status==null -> certificate OK
225                     throw new CertPathValidatorException("Certificate with serialnumber " + CertTools.getSerialNumberAsString(cert) + " was revoked");
226                 }
227                 
228                 if(unresolvedCritExts != null) {
229                     unresolvedCritExts.remove(Extension.authorityInfoAccess.getId());
230                 }
231             }
232 
233         } else {
234             fallBackToCrl(cert, CertTools.getSubjectDN(cacert));
235             
236             if(unresolvedCritExts != null) {
237                 unresolvedCritExts.remove(Extension.cRLDistributionPoints.getId());
238             }
239         }
240 
241     }

243     /**
244      * Check the revocation status of 'cert' using a CRL
245      * @param cert the certificate whose revocation status is to be checked
246      * @param issuerDN DN
247      * @throws CertPathValidatorException if invalid
248      */
249     private void fallBackToCrl(final Certificate cert, final String issuerDN) throws CertPathValidatorException {
250         final ArrayList<URL> crlUrls = getCrlUrl(cert);
251         if(crlUrls.isEmpty()) {
252             final String errmsg = "Failed to verify certificate status using the fallback CRL method. Could not find a CRL URL"; 
253             log.info(errmsg);
254             throw new CertPathValidatorException(errmsg);
255         }
256         if(log.isDebugEnabled()) {
257             log.debug("Found " + crlUrls.size() + " CRL URLs");
258         }
259         
260         CRL crl = null;
261         for(URL url : crlUrls) {
262             crl = getCRL(url);
263             if(crl != null) {
264                 if(isCorrectCRL(crl, issuerDN)) {
265                     final boolean isRevoked = crl.isRevoked(cert);
266                     this.crl = crl;
267                     if(isRevoked) {
268                         throw new CertPathValidatorException("Certificate with serialnumber " + CertTools.getSerialNumberAsString(cert) + " was revoked");
269                     }
270                     break;
271                 }
272             }
273         }
274         if(this.crl==null) {
275             throw new CertPathValidatorException("Failed to verify certificate status using CRL. Could not find a CRL issued by " + issuerDN + " reasonably lately");
276         }
277     }

279     private boolean isCorrectCRL(final CRL crl, final String issuerDN) {
280         if(!(crl instanceof X509CRL)) {
281             return false;
282         }
283         
284         X509CRL x509crl = (X509CRL) crl;
285         if(!StringUtils.equals(issuerDN, CertTools.getIssuerDN(x509crl))) {
286             return false;
287         }
288         
289         final Date now = new Date(System.currentTimeMillis());
290         final Date nextUpdate = x509crl.getNextUpdate();
291         if(nextUpdate!=null) {
292             if(nextUpdate.after(now)) {
293                 return true;
294             }
295             
296             if(log.isDebugEnabled()) {
297                 log.debug("CRL issued by " + issuerDN + " is out of date");
298             }
299             return false;
300         }
301         
302         final Date thisUpdate = x509crl.getThisUpdate();
303         if(thisUpdate!=null) {
304             final GregorianCalendar gc = new GregorianCalendar();
305             gc.setTime(now);
306             gc.add(Calendar.HOUR, 1);
307             final Date expire = gc.getTime();
308             
309             if(expire.before(now)) {
310                 if(log.isDebugEnabled()) {
311                     log.debug("Could not find when CRL issued by " + issuerDN + " should be updated and this CRL is over one hour old. Not using it");
312                 }
313                 return false;
314             }
315             
316             log.warn("Could not find when CRL issued by " + issuerDN + " should be updated, but this CRL was issued less than an hour ago, so we are using it");
317             return true;
318         }
319             
320         if(log.isDebugEnabled()) {
321             log.debug("Could not check issuance time for CRL issued by " + issuerDN);
322         }
323         return false;
324     }

326     private CRL getCRL(final URL url) {
327         CRL crl = null;
328         try {
329             final URLConnection con = url.openConnection();
330             final InputStream is = con.getInputStream();
331             final CertificateFactory cf = CertificateFactory.getInstance("X.509");
332             crl = cf.generateCRL(is);
333             is.close();
334             log.info("Downloaded CRL from " + url);
335         } catch(IOException | CertificateException | CRLException e) {
336             if(log.isDebugEnabled()) {
337                 log.debug("Fetching CRL from " + url.toString() + " failed. " + e.getLocalizedMessage());
338             }
339         }
340         return crl;
341     }

343     /**
344      * Construct an OCSP request
345      * @param cacert The certificate of the issuer of the certificate to be checked
346      * @param certSerialnumber the serialnumber of the certificate to be checked
347      * @param nonce random nonce to be included in the OCSP request (OCSP POST)
348      * @return OCSPReq
349      * @throws CertificateEncodingException if cert is corrupt
350      * @throws OCSPException if OCSP fails
351      */
352     private OCSPReq getOcspRequest(Certificate cacert, BigInteger certSerialnumber, final byte[] nonce) throws CertificateEncodingException, OCSPException {
353         OCSPReqBuilder gen = new OCSPReqBuilder();
354         gen.addRequest(new JcaCertificateID(SHA1DigestCalculator.buildSha1Instance(), (X509Certificate) cacert, certSerialnumber));
355         
356         Extension[] extensions = new Extension[1];
357         extensions[0] = new Extension(OCSPObjectIdentifiers.id_pkix_ocsp_nonce, false, new DEROctetString(nonce));
358         gen.setRequestExtensions(new Extensions(extensions));
359 
360         return gen.build();
361     }

365     /**
366      * Sends an OCSP request, gets a response and verifies the response as much as possible before returning it to the caller.
367      * @param ocspurl URL
368      * @param ocspRequest Request 
369      * @param cert Certificate
370      * @param nonce Nonce
371      * @param expectedOcspRespCode OCSP response 
372      * @param expectedHttpRespCode HTTP response
373      * 
374      * @return The OCSP response, or null of no correct response could be obtained.
375      */
376     private SingleResp getOCSPResponse(final String ocspurl, final OCSPReq ocspRequest, final Certificate cert, final byte[] nonce, int expectedOcspRespCode, int expectedHttpRespCode) {
377         if(log.isDebugEnabled()) {
378             log.debug("Sending OCSP request to " + ocspurl + " regarding certificate with SubjectDN: " + CertTools.getSubjectDN(cert) 
379                         + " - IssuerDN: " + CertTools.getIssuerDN(cert));
380         }
381         
382         //----------------------- Open connection and send the request --------------//
383         OCSPResp response = null;
384         HttpURLConnection con = null;
385         try {
386             final URL url = new URL(ocspurl);
387             con = (HttpURLConnection)url.openConnection();
388             // we are going to do a POST
389             con.setDoOutput(true);
390             con.setRequestMethod("POST");
391 
392             // POST it
393             con.setRequestProperty("Content-Type", "application/ocsp-request");
394             OutputStream os = con.getOutputStream();
395             os.write(ocspRequest.getEncoded());
396             os.close();
397         
398         
399             final int httpRespCode = ((HttpURLConnection)con).getResponseCode();
400             if (httpRespCode != expectedHttpRespCode) {
401                 log.info("HTTP response from OCSP request was " + httpRespCode + ". Expected " + expectedHttpRespCode );
402                 handleContentOfErrorStream(con.getErrorStream());
403                 return null; // if it is an http error code we don't need to test any more
404             }
405 
406             InputStream is = con.getInputStream();
407             response = new OCSPResp(IOUtils.toByteArray(is));
408             is.close();
409         
410         } catch(IOException e) {
411             log.info("Unable to get an OCSP response. " + e.getLocalizedMessage());
412             if(con != null) {
413                 handleContentOfErrorStream(con.getErrorStream());
414             }
415             return null;
416         }
417         
418         
419         
420         // ------------ Verify the response signature --------------//
421         BasicOCSPResp brep = null;
422         try {
423             brep = (BasicOCSPResp) response.getResponseObject();
424 
425             if ((expectedOcspRespCode != OCSPRespBuilder.SUCCESSFUL) && (brep!=null)) {
426                 log.warn("According to RFC 2560, responseBytes are not set on error, but we got some.");    
427                 return null; // it messes up testing of invalid signatures... but is needed for the unsuccessful responses
428             }
429         
430             if (brep==null) {
431                 log.warn("Cannot extract OCSP response object. OCSP response status: " + response.getStatus());
432                 return null;
433             }
434         
435             X509CertificateHolder[] chain = brep.getCerts();
436             boolean verify = brep.isSignatureValid(new JcaContentVerifierProviderBuilder().setProvider(BouncyCastleProvider.PROVIDER_NAME).build(chain[0]));
437             if(!verify) {
438                 log.warn("OCSP response signature was not valid");
439                 return null;
440             }
441         } catch (OCSPException | OperatorCreationException | CertificateException e) {
442             if(log.isDebugEnabled()) {
443                 log.debug("Failed to obtain or verify OCSP response. " + e.getLocalizedMessage());
444             }
445             return null;
446         }
447 
448         // ------------- Verify the nonce ---------------//
449         byte[] noncerep;
450         try {
451             noncerep = brep.getExtension(OCSPObjectIdentifiers.id_pkix_ocsp_nonce).getExtnValue().getEncoded();
452         } catch (IOException e) {
453             if(log.isDebugEnabled()) {
454                 log.debug("Failed to read extension from OCSP response. " + e.getLocalizedMessage());
455             }
456             return null;
457         }
458         if(noncerep == null) {
459             log.warn("Sent an OCSP request containing a nonce, but the OCSP response does not contain a nonce");
460             return null;
461         }
462         
463         try {
464             ASN1InputStream ain = new ASN1InputStream(noncerep);
465             ASN1OctetString oct = ASN1OctetString.getInstance(ain.readObject());
466             ain.close();
467             if(!Arrays.equals(nonce, oct.getOctets())) {
468                 log.warn("The nonce in the OCSP request and the OCSP response do not match");
469                 return null;
470             }
471         } catch (IOException e) {
472             if(log.isDebugEnabled()) {
473                 log.debug("Failed to read extension from OCSP response. " + e.getLocalizedMessage());
474             }
475             return null;
476         }
477         
478         
479         // ------------ Extract the single response and verify that it concerns a cert with the right serialnumber ----//
480         SingleResp[] singleResps = brep.getResponses();
481         if((singleResps==null) || (singleResps.length==0)) {
482             if(log.isDebugEnabled()) {
483                 log.debug("The OCSP response object contained no responses.");
484             }
485             return null;
486         }
487         
488         SingleResp singleResponse = singleResps[0];
489         CertificateID certId = singleResponse.getCertID();
490         if(!certId.getSerialNumber().equals(CertTools.getSerialNumber(cert))) {
491             if(log.isDebugEnabled()) {
492                 log.debug("Certificate serialnumber in response does not match certificate serialnumber in request.");
493             }
494             return null;
495         }
496         
497         // ------------ Return the single response ---------------//
498         return singleResponse;        
499     }

501     /**
502      * Reads the content of 'httpErrorStream' and ignores it. 
503      * @param httpErrorStream stream
504      */
505     private void handleContentOfErrorStream(final InputStream httpErrorStream) {
506         if (httpErrorStream != null) {
507             try {
508                 OutputStream os = new NullOutputStream();
509                 IOUtils.copy(httpErrorStream, os);
510                 httpErrorStream.close();
511                 os.close();
512             } catch(IOException ex) {}
513         }
514     }

516     private ArrayList<String> getOcspUrls(Certificate cert) {
517         ArrayList<String> urls = new ArrayList<String>();
518         if(StringUtils.isNotEmpty(this.ocspUrl)) {
519             urls.add(this.ocspUrl);
520         }
521         
522         urls.addAll(CertTools.getAuthorityInformationAccessOcspUrls((X509Certificate)cert));
523         
524         return urls;
525     }

527     private ArrayList<URL> getCrlUrl(final Certificate cert) {
528         
529         ArrayList<URL> urls = new ArrayList<URL>();
530         
531         if(StringUtils.isNotEmpty(this.crlUrl)) {
532             try {
533                 urls.add(new URL(this.crlUrl));
534             } catch (MalformedURLException e) {
535                 if(log.isDebugEnabled()) {
536                     log.debug("Failed to parse '" + this.crlUrl + "' as a URL. " + e.getLocalizedMessage());
537                 }
538             }
539         }
540         
541         Collection<URL> crlUrlFromExtension = CertTools.getCrlDistributionPoints((X509Certificate)cert); 
542         urls.addAll(crlUrlFromExtension);
543         
544         return urls; 
545     }
546     
547     private X509Certificate getCaCert(final Certificate targetCert) {
548         if(this.issuerCert != null) {
549             return issuerCert;
550         }
551         
552         if(this.caCerts==null) { // no CA specified
553             return null;
554         }
555 
556         for(final X509Certificate cacert : this.caCerts) {
557             if(isIssuerCA(targetCert, cacert)) {
558                 return cacert;
559             }
560         }
561         return null;
562     }
563     
564     private boolean isIssuerCA(final Certificate cert, final Certificate cacert) {
565         if(!StringUtils.equals(CertTools.getIssuerDN(cert), CertTools.getSubjectDN(cacert))) {
566             return false;
567         }
568         try {
569             cert.verify(cacert.getPublicKey());
570             return true;
571         } catch (InvalidKeyException | CertificateException | NoSuchAlgorithmException | NoSuchProviderException | SignatureException e) {
572             return false;
573         }
574     }
