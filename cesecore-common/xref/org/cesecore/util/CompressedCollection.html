20  import java.util.ArrayList;

27  import java.util.Set;
28  import java.util.zip.Deflater;
29  import java.util.zip.DeflaterOutputStream;
30  import java.util.zip.InflaterInputStream;
31  import org.apache.commons.collections.SetUtils;
32  import org.apache.log4j.Logger;

34  /**
35   * Special Collection that compresses all the added objects to a in memory byte
36   * array.
37   *
38   * <p>Objects are read from the collection by using a iterator over a
39   * decompression InputStream. To avoid memory leaks, the .clear() call should be
40   * used when the collection is no longer needed.
41   *
42   * <p>The implementation is not thread safe.
43   *
44   * <p>Example use-case: a RevokedCertInfo takes 248 bytes in serialized form,
45   * but averages at only 48 bytes in compressed serialized form.
46   *
47   * @version $Id: CompressedCollection.java 34193 2020-01-07 15:18:15Z samuellb $
48   * @param <T> type
49   */
50  public class CompressedCollection<T extends Serializable>
51      implements Collection<T>, Serializable {
52  
53    private static final long serialVersionUID = 1L;
54    /** Logger. */
55    private static final Logger LOG =
56        Logger.getLogger(CompressedCollection.class);
57  
58    /** Output. */
59    private ByteArrayOutputStream baos = null;
60    /** Output. */
61    private ObjectOutputStream oos = null;
62    /** Data. */
63    private byte[] compressedData = null;
64    /** Size. */
65    private int size = 0;
66    /** Streams. */
67    private final List<LookAheadObjectInputStream> oiss = new ArrayList<>();
68    /** Classes. */
69    private final Set<Class<? extends Serializable>> acceptedClasses;
70  
71    /**
72     * @param elementClass Class
73     * @param nestedClasses Classes
74     */
75    @SuppressWarnings("unchecked")
76    @SafeVarargs
77    public CompressedCollection(
78        final Class<T> elementClass,
79        final Class<? extends Serializable>... nestedClasses) {
80      acceptedClasses =
81          SetUtils.typedSet(
82              new HashSet<>(nestedClasses.length + 1),
83              Serializable.class); // generic fireworks are not type safe
84      acceptedClasses.add(elementClass);
85      acceptedClasses.addAll(Arrays.asList(nestedClasses));
86      clear();
87    }
88  
89    @Override
90    public boolean add(final T object) {
91      if (compressedData != null) {
92        throw new IllegalStateException(
93            "closeForWrite() has alread been called without clear() for this"
94                + " CompressedCollection.");
95      }
96      boolean ret = false;
97      if (object != null) {
98        try {
99          getObjectOutputStream().writeObject(object);
100         ret = true;
101         size++;
102       } catch (IOException e) {
103         throw new IllegalStateException(e);
104       }
105     }
106     return ret;
107   }
108 
109   /**
110    * Lazy initialization of our in memory object storage.
111    *
112    * @return stream
113    * @throws IOException on error
114    */
115   private ObjectOutputStream getObjectOutputStream() throws IOException {
116     if (oos == null) {
117       baos = new ByteArrayOutputStream();
118       final DeflaterOutputStream dos =
119           new DeflaterOutputStream(
120               baos, new Deflater(Deflater.BEST_COMPRESSION));
121       oos = new ObjectOutputStream(dos);
122     }
123     return oos;
124   }
125 
126   @Override
127   public boolean addAll(final Collection<? extends T> objects) {
128     for (final T object : objects) {
129       add(object);
130     }
131     return objects.size() != 0;
132   }
133 
134   @Override
135   public void clear() {
136     if (oos != null) {
137       // Clean up OutputStream, unless this has already been done
138       try {
139         oos.close();
140       } catch (IOException e) {
141         LOG.error(e.getMessage(), e);
142       }
143       oos = null;
144     }
145     size = 0;
146     compressedData = null;
147     // Clean up all InputStreams, unless this has already been done
148     for (final LookAheadObjectInputStream ois : oiss) {
149       try {
150         ois.close();
151       } catch (IOException e) {
152         LOG.error(e.getMessage(), e);
153       }
154     }
155     oiss.clear();
156   }
157 
158   @Override
159   public boolean contains(final Object object) {
160     final Iterator<T> iterator = iterator();
161     while (iterator.hasNext()) {
162       final T t = iterator.next();
163       if (t.equals(object)) {
164         try {
165           oiss.get(oiss.size() - 1).close();
166         } catch (IOException e) {
167           throw new IllegalStateException(e);
168         }
169         oiss.remove(oiss.size() - 1);
170         return true;
171       }
172     }
173     return false;
174   }
175 
176   @Override
177   public boolean containsAll(final Collection<?> objects) {
178     throw new UnsupportedOperationException(); // This is not really an optional
179                                                // operation for a Collection
180   }
181 
182   @Override
183   public boolean isEmpty() {
184     return size == 0;
185   }
186 
187   /**
188    * Signal that no more data will be added to this collection. Call before
189    * Serialization.
190    */
191   public void closeForWrite() {
192     if (compressedData == null) {
193       if (oos == null) {
194         // Nothing was added
195         compressedData = new byte[0];
196       } else {
197         // Clean up outputstream now when we are about to read the data
198         try {
199           oos.flush();
200           oos.close();
201           oos = null;
202           compressedData = baos.toByteArray();
203           baos = null;
204           if (LOG.isDebugEnabled()) {
205             LOG.debug(
206                 "Compressed data of "
207                     + size
208                     + " entries to "
209                     + compressedData.length
210                     + " bytes.");
211           }
212         } catch (IOException e) {
213           LOG.error(e.getMessage(), e);
214         }
215       }
216     }
217   }
218 
219   @Override
220   public Iterator<T> iterator() {
221     closeForWrite();
222     // Create a new decompression stream over the data that belongs to the
223     // Iterator
224     final ByteArrayInputStream bais = new ByteArrayInputStream(compressedData);
225     final InflaterInputStream iis = new InflaterInputStream(bais);
226     final LookAheadObjectInputStream ois;
227     if (compressedData.length == 0) {
228       ois = null;
229     } else {
230       try {
231         ois = new LookAheadObjectInputStream(iis);
232         ois.setAcceptedClasses(acceptedClasses);
233         ois.setEnabledMaxObjects(false);
234       } catch (IOException e) {
235         throw new IllegalStateException(e);
236       }
237       oiss.add(ois);
238     }
239     return new Iterator<T>() {
240       private T next = null;
241 
242       @Override
243       public boolean hasNext() {
244         try {
245           next = readNext();
246         } catch (NoSuchElementException e) {
247           return false;
248         }
249         return true;
250       }
251 
252       @Override
253       public T next() {
254         if (next == null) {
255           return readNext();
256         }
257         T ret = next;
258         next = null;
259         return ret;
260       }
261 
262       @Override
263       public void remove() {
264         throw new UnsupportedOperationException();
265       }
266 
267       @SuppressWarnings("unchecked")
268       private T readNext() {
269         if (ois == null) {
270           throw new NoSuchElementException();
271         }
272         try {
273           return (T) ois.readObject();
274         } catch (IOException e) {
275           cleanUp();
276           throw new NoSuchElementException();
277         } catch (ClassNotFoundException e) {
278           cleanUp();
279           throw new NoSuchElementException();
280         }
281       }
282 
283       /**
284        * Clean up InputStream right away if we reached the last entry in the
285        * stream
286        */
287       private void cleanUp() {
288         oiss.remove(ois);
289         try {
290           ois.close();
291         } catch (IOException e) {
292           LOG.error(e.getMessage(), e);
293         }
294       }
295     };
296   }
