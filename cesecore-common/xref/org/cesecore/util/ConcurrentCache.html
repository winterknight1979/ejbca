33  /**
34   * A concurrent cache allows multiple threads to cache data. Only one thread
35   * will be allowed to generate data for one particular key, and other threads
36   * will block.
37   * 
38   * All methods in this class and inner classes are thread-safe.
39   * 
40   * @version $Id: ConcurrentCache.java 27456 2017-12-07 10:59:06Z samuellb $
41   * @param <K> Key type
42   * @param <V> Value type 
43   *
44   */
45  public final class ConcurrentCache<K,V> {
46  
47      private static final Logger log = Logger.getLogger(ConcurrentCache.class);
48      
49      /**
50       * Internal entries are stored in the ConcurrentMap
51       * @param <V> Value type
52       */
53      private final static class InternalEntry<V> {
54          final V value;
55          volatile long expire;
56          
57          private InternalEntry(final V value) {
58              this.value = value;
59              this.expire = Long.MAX_VALUE;
60          }
61      }
62      
63      /**
64       * A reference to a cache entry, with a get and put method to read/write data from/to the cache.
65       * 
66       * All methods are thread safe, but only one thread should operate on an Entry object.
67       */
68      public final class Entry {
69          private final K key;
70          private InternalEntry<V> entry;
71          /** If non-null, then other threads are waiting on this semaphore for data on the same key in the cache. */
72          private final Object ourSemaphore;
73          
74          private Entry(final K key, final InternalEntry<V> entry) {
75              this.key = key;
76              this.entry = entry;
77              this.ourSemaphore = null;
78          }
79          
80          private Entry(final K key, final InternalEntry<V> entry, final Object ourSemaphore) {
81              if (ourSemaphore == null) {
82                  throw new IllegalArgumentException("ourSemaphore may not be null");
83              }
84              this.key = key;
85              this.entry = entry;
86              this.ourSemaphore = ourSemaphore;
87          }
88          
89          /**
90           * @return true if the key existed in the cache.
91           */
92          public boolean isInCache() {
93              return entry != null;
94          }
95          
96          /**
97           * @return the value read from the cache when the Entry was created. Calls to putValue() on this particular Entry change the return value.
98           */
99          public V getValue() {
100             if (entry == null) {
101                 throw new IllegalStateException("Tried to read from non-existent cache entry");
102             }
103             return entry.value;
104         }
105         
106         /**
107          * Updates the value in this Entry as well as in the underlying cache.
108          * The expire time is set to be "infinite". Thread-safe.
109          * @param value value
110          */
111         public void putValue(final V value) {
112             if (key != null) {
113                 entry = new InternalEntry<>(value);
114                 cache.put(key, entry);
115             }
116         }
117         
118         /**
119          * Sets the validity of the value. After the cache entry expires, the next request for it will
120          * fail (on purpose) so it can be updated. Requests that happen while the expired entry is being
121          * updated will still use the expired value, so they don't have to block.
122          * @param validFor Cache validity in milliseconds.
123          */
124         public void setCacheValidity(long validFor) {
125             if (entry != null) {
126                 entry.expire = System.currentTimeMillis() + validFor;
127             }
128         }
129         
130         /**
131          * Must be called if other threads might be waiting for this cache entry
132          * (i.e. if isInCache() returns false)
133          */
134         public void close() {
135             if (ourSemaphore != null) {
136                 synchronized (ourSemaphore) {
137                     semaphores.remove(key);
138                     ourSemaphore.notifyAll();
139                 }
140             }
141         }
142     }
143     
144     private final ConcurrentHashMap<K,InternalEntry<V>> cache = new ConcurrentHashMap<>();
145     private final ConcurrentMap<K,Object> semaphores = new ConcurrentHashMap<>();
146     
147     public static final long NO_LIMIT = -1L;
148     
149     /** @see #setEnabled */
150     private volatile boolean enabled = true;
151     /** @see #setCloseOnEviction */
152     private volatile boolean closeOnEviction = false;
153     /** @see #setMaxEntries */
154     private volatile long maxEntries = NO_LIMIT;
155     
156     private AtomicLong numEntries = new AtomicLong(0L);
157     private final Set<K> pendingRemoval = Collections.newSetFromMap(new ConcurrentHashMap<K,Boolean>());
158     private final Lock isCleaning = new ReentrantLock();
159     private volatile long lastCleanup = 0L;
160     private volatile long cleanupInterval = 1000L;
161     
162     /** Creates an empty concurrent cache */
163     public ConcurrentCache() {
164         // Do nothing
165     }
166     
167     /**
168      * Creates a concurrent cache initialized with the mapping defined in the given map.
169      * Can be used for rebuilding the cache in the background for instance.
170      * 
171      * @param map map
172      * @param validFor Time in milliseconds which the entry will be valid for, or -1L for forever.
173      * @see ConcurrentCache#getKeys()
174      */
175     public ConcurrentCache(final Map<? extends K, ? extends V> map, long validFor) {
176         for (Map.Entry<? extends K, ? extends V> mapEntry : map.entrySet()) {
177             final InternalEntry<V> intEntry = new InternalEntry<V>(mapEntry.getValue());
178             if (validFor != -1L) {
179                 intEntry.expire = System.currentTimeMillis() + validFor;
180             }
181             cache.put(mapEntry.getKey(), intEntry);
182         }
183     }
184     
185     /**
186      * "Opens" a cache entry. If the entry already exists, then an {@link Entry} that
187      * maps to the existing entry is returned. Otherwise, a semaphore is used
188      * to prevent multiple threads from creating the new cache entry. Only the
189      * first thread is returned an Entry with isInCache()==false, later threads
190      * will block and wait for the first thread.
191      * 
192      * For non-existent entries (i.e. isInCache()==false), the caller is expected to put
193      * a value in it and call close() on the Entry.
194      *  
195      * @param key      Key in the cache.
196      * @param timeout  Timeout in milliseconds. The call will only be allowed
197      *                 to block for (approximately) this amount of time.
198      * @return An Entry object that maps to an entry in the cache (existing
199      *         or blank), or null if a timeout occurred. 
200      * @throws NullPointerException if key is null.
201      */
202     public Entry openCacheEntry(final K key, final long timeout) {
203         final long timeAtEntry = System.currentTimeMillis();
204         if (key == null) {
205             throw new NullPointerException("key may not be null");
206         }
207         
208         if (!enabled) {
209             return new Entry(null, null);
210         }
211         
212         if (maxEntries != NO_LIMIT) {
213             pendingRemoval.remove(key); // always mark as used
214         }
215         
216         // Fast path if cached
217         InternalEntry<V> entry = cache.get(key);
218         final long toExpire = (entry != null ? entry.expire : 0L);
219         if (entry != null && toExpire > timeAtEntry) {
220             // Found valid entry in cache
221             if (log.isDebugEnabled()) {
222                 log.debug("Found valid entry in cache for key "+key);
223                 if (log.isTraceEnabled()) {
224                     log.debug("Value: "+entry.value);
225                     log.trace("<ConcurrentCache.openCacheEntry");
226                 }
227             }
228             cleanupIfNeeded();
229             return new Entry(key, entry);
230         } else if (entry != null) {
231             if (log.isDebugEnabled()) {
232                 log.debug("Cache entry has expired "+key+", expiry="+entry.expire);
233             }
234             numEntries.decrementAndGet();
235         } else {
236             if (log.isDebugEnabled()) {
237                 log.debug("Entry was not present in cache "+key);
238             }
239         }
240         
241         // Make sure only one thread enters "opens" the cache entry in write mode.
242         // Subsequent attempts to open it will block until the first cache entry has been closed.
243         final Object ourSemaphore = new Object();
244         final Object theirSemaphore = semaphores.putIfAbsent(key, ourSemaphore);
245         if (theirSemaphore == null) {
246             // We were first
247             numEntries.incrementAndGet();
248             cleanupIfHighlyNeeded();
249             return new Entry(key, null, ourSemaphore);
250         }
251         
252         // Someone else was first
253         
254         // Check if we can return an existing entry (ECA-4936)
255         if (entry != null) {
256             log.debug("Returning existing cache entry for now");
257             log.trace("<ConcurrentCache.openCacheEntry");
258             cleanupIfNeeded();
259             return new Entry(key, entry);
260         }
261         
262         // Wait for a fresh entry to be created
263         try {
264             synchronized (theirSemaphore) {
265                 if (!cache.containsKey(key)) {
266                     cleanupIfNeeded();
267                     theirSemaphore.wait(timeout);
268                     while (!cache.containsKey(key) && System.currentTimeMillis() < timeAtEntry+timeout) {
269                         theirSemaphore.wait(timeout/10L+1L);
270                     }
271                 }
272             }
273         } catch (InterruptedException e) {
274             Thread.currentThread().interrupt();
275             throw new RuntimeException(e); // should preferably not be catched
276         }
277         
278         // Return cached result from other thread, or null on failure
279         entry = cache.get(key);
280         if (log.isDebugEnabled()) {
281             log.debug("Got "+ (entry != null ? entry.value : "null") + " after waiting for cache");
282             log.trace("<ConcurrentCache.openCacheEntry");
283         }
284         return entry != null ? new Entry(key, entry) : null;
285     }

291      */

293         return new HashSet<>(cache.keySet());
294     }

297      * <p>Enables or disables caching. If disabled, nothing will be cached and openCacheEntry will
298      * always immediately return an non-existent entry (this may also cause concurrent attempts
299      * to fetch/build/etc the same object).</p>
300      * 
301      * <p>Disabling the cache doesn't stop any currently "open" cache entries from being written to.</p>
302      * 
303      * <p>The default is enabled.</p>
304      * @param enabled bool
305      */
306     public void setEnabled(boolean enabled) {
307         this.enabled = enabled;
308     }

311      * @see ConcurrentCache#setEnabled */
312     public boolean isEnabled() {
313         return enabled;
314     }

321      * <p>The default is false.</p>

329      * @see ConcurrentCache#setCloseOnEviction */
330     public boolean isCloseOnEviction() {
331         return closeOnEviction;

334     /**
335      * <p>Sets the desired maximum number of entries in the cache. This is not a
336      * strict limit, and the cache may temporarily exceed this number.</p>
337      * 
338      * <p>The value {@link ConcurrentCache#NO_LIMIT} (-1) is the default.</p>
339      * @param maxEntries Max number of entries
340      */
341     public void setMaxEntries(long maxEntries) {
342         if (maxEntries == NO_LIMIT || maxEntries > 0L) {
343             this.maxEntries = maxEntries;
344         } else {
345             throw new IllegalArgumentException("max entries must be either a positive value or -1");
346         }
347     }
348     
349     /** @return max number of entries
350      * @see ConcurrentCache#setMaxEntries */
351     public long getMaxEntries() {
352         return maxEntries;
353     }

355     /**
356      * Sets the minimum time in milliseconds between two cleanup runs.
357      * 
358      * The default is 1000 (= 1 second).
359      * @param milliseconds cleanup interval
360      */
361     public void setCleanupInterval(long milliseconds) {
362         cleanupInterval = milliseconds;
363     }

365     /** @return cleanup interval
366      * @see ConcurrentCache#setCleanupInterval */
367     public long getCleanupInterval() {
368         return cleanupInterval;
369     }
370     
371     private void cleanupIfNeeded() {
372         if (maxEntries != NO_LIMIT && numEntries.get() > maxEntries) {
373             cleanup();
374         }
375     }
376     
377     private void cleanupIfHighlyNeeded() {
378         // More than 1.5 times the limit 
379         if (maxEntries != NO_LIMIT && 2L*numEntries.get() > 3L*maxEntries) {
380             cleanup();
381         }
382     }

384     /** Used internally for testing. 
385      * @param min min
386      * @param max max */
387     void checkNumberOfEntries(long min, long max) {
388         long a = numEntries.get();
389         long b = cache.size();
390         if (a != b) {
391             throw new IllegalStateException("cache.size() and numEntries does not match ("+a+" and "+b+")");
392         }
393         if (a < min) {
394             throw new IllegalStateException("number of entries ("+a+") is less than minimum ("+min+").");
395         }
396         if (a > max) {
397             throw new IllegalStateException("number of entries ("+a+") is greater than maximum ("+max+").");
398         }
399     }

401     /**
402      * Removes expired entries, and randomly selected entries that have not been used since the last call.
403      */
404     private void cleanup() {
405         List<Closeable> valuesToClose = null;
406         final long startTime = System.currentTimeMillis();
407         if (startTime < lastCleanup+cleanupInterval || !isCleaning.tryLock()) {
408             return;
409         }
410         try {
411             final float ratioToRemove;
412             final Random random;
413             if (maxEntries == NO_LIMIT) {
414                 ratioToRemove = 0;
415                 random = null;
416             } else {
417                 // Remove a bit extra
418                 ratioToRemove = Math.max(0.0F, 1.0F-0.8F*maxEntries/numEntries.get());
419                 
420                 // Remove items that have not been accessed since they were last marked as "pending removal"
421                 if (closeOnEviction) { valuesToClose = new ArrayList<>(); }
422                 for (K key : pendingRemoval) {
423                     InternalEntry<V> evicted = cache.remove(key);
424                     if (closeOnEviction && evicted.value instanceof Closeable) {
425                         valuesToClose.add((Closeable)evicted.value);
426                     }
427                     numEntries.decrementAndGet();
428                 }
429                 pendingRemoval.clear();
430                 random = new Random(System.nanoTime());
431             }
432             
433             final long now = System.currentTimeMillis();
434             final Iterator<Map.Entry<K,InternalEntry<V>>> iter = cache.entrySet().iterator();
435             while (iter.hasNext()) {
436                 final Map.Entry<K,InternalEntry<V>> mapEntry = iter.next();
437                 if (mapEntry.getValue().expire <= now) {
438                     iter.remove();
439                     numEntries.decrementAndGet();
440                 } else if (maxEntries != NO_LIMIT && random.nextFloat() < ratioToRemove) {
441                     pendingRemoval.add(mapEntry.getKey());
442                 }
443             }
444         } finally {
445             isCleaning.unlock();
446             
447             final long endTime = System.currentTimeMillis();
448             lastCleanup = endTime;
449             if (log.isDebugEnabled()) {
450                 log.debug("Clean up took "+(endTime - startTime)+" ms");
451             }
452         }
453         
454         if (valuesToClose != null) {
455             for (final Closeable closable : valuesToClose) {
456                 try {
457                     closable.close();
458                 } catch (IOException e) {
459                     log.debug("Exception ocurring when closing evicted value.", e);
460                 }
461             }
462         }
463     }
464     
465     /**
466      * Removes all entries in the cache
467      */
468     public void clear() {
469         if (closeOnEviction) {
470             isCleaning.lock();
471             try {
472                 for (final InternalEntry<V> entry : cache.values()) {
473                     if (entry.value instanceof Closeable) {
474                         try {
475                             ((Closeable)entry.value).close();
476                         } catch (IOException e) {
477                             log.debug("Exception ocurring when closing value during cache clearing.", e);
478                         }
479                     }
480                 }
481                 cache.clear();
482             } finally {
483                 isCleaning.unlock();
484             }
485         } else {
486             cache.clear();
487         }
488         numEntries.set(0L);
489         pendingRemoval.clear();
490         lastCleanup = 0L;
491     }
