13  package org.cesecore.util;
14  
15  /**
16   * An enum where each enum value has an "index" value, which is guaranteed to not change between versions
17   * (unlike the ordinal() method which can change if enum values are added, removed or reordered).
18   * <p>
19   * Avoid building enums using this pattern, unless storage space is really a concern.
20   * Please store the enum name as a string instead.
21   * 
22   * @version $Id: IndexEnum.java 30982 2019-01-04 12:53:31Z samuellb $
23   */
24  public interface IndexEnum {
25      /** @return a stable "index" of the enum value, which is guaranteed to not change with new versions (unlike {@link Enum#ordinal}). 
26       * */
27      int getIndex();
28  }
