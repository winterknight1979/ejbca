63      /**
64       * @return The length in octets of certificate serial numbers generated for legacy CAs. (8 octets is a 64 bit serial number.)
65       */
66      public static int getSerialNumberOctetSizeForExistingCa() {
67          String value = ConfigurationHolder.getConfiguredString("ca.serialnumberoctetsize");
68          if (value == null) {
69              if (log.isDebugEnabled()) {
70                  log.debug("Using default value of " + DEFAULT_SERIAL_NUMBER_OCTET_SIZE_EXISTINGCA + " for existing CA's ca.serialnumberoctetsize");
71              }
72              value = DEFAULT_SERIAL_NUMBER_OCTET_SIZE_EXISTINGCA;
73          }
74          return Integer.parseInt(value);
75      }

239     private static long getLongValue(final String propertyName, final long defaultValue, final String unit) {
240         final String value = ConfigurationHolder.getString(propertyName);
241         long time = defaultValue;
242         try {
243             if (value!=null) {
244                 time = Long.valueOf(value);
245             }
246         } catch (NumberFormatException e) {
247             log.error("Invalid value for " + propertyName + ". Using default " + defaultValue + ". Value must be decimal number (" + unit + "): " + e.getMessage());
248         }
249         return time;
250     }

287     /**
288      * When we run in a cluster, each node should have it's own identifier. By default we use the DNS name.
289      * @return ID
290      */
291     public static String getNodeIdentifier() {
292     	final String PROPERTY_NAME = "cluster.nodeid";
293     	final String PROPERTY_VALUE = "undefined";
294         String value = ConfigurationHolder.getString(PROPERTY_NAME);
295         if (value == null) {
296         	try {
297 				value = InetAddress.getLocalHost().getHostName();
298 			} catch (UnknownHostException e) {
299 				log.warn(PROPERTY_NAME + " is undefined on this host and was not able to resolve hostname. Using " + PROPERTY_VALUE + " which is fine if use a single node.");
300 				value = PROPERTY_VALUE;
301 			}
302 			// Update configuration, so we don't have to make a hostname lookup each time we call this method.
303             ConfigurationHolder.updateConfigurationWithoutBackup(PROPERTY_NAME, value);
304         }
305         return value;
306     }

335     public static String getExtraAlgSubAlgTitle(String algName, String subAlg) {
336         String name = ConfigurationHolder.getString("extraalgs." + algName + ".subalgs." + subAlg + ".title");
337         if (name == null) {
338             // Show the algorithm name, if it has one
339             String end = ConfigurationHolder.getString("extraalgs." + algName + ".subalgs." + subAlg + ".name");
340             // Otherwise, show the key name in the configuration
341             if (end == null) { end = subAlg; }
342             name = ConfigurationHolder.getString("extraalgs." + algName + ".title") + " " + end;
343         }
344         return name;
345     }

347     public static String getExtraAlgSubAlgName(String algName, String subAlg) {
348         String name = ConfigurationHolder.getString("extraalgs." + algName + ".subalgs." + subAlg + ".name");
349         if (name == null) {
350             // Not a named algorithm
351             name = getExtraAlgSubAlgOid(algName, subAlg);

356     public static String getExtraAlgSubAlgOid(String algName, String subAlg) {
357         final String oidTree = ConfigurationHolder.getString("extraalgs." + algName + ".oidtree");
358         final String oidEnd = ConfigurationHolder.getString("extraalgs." + algName + ".subalgs." + subAlg + ".oid");
359 
360         if (oidEnd != null && oidTree != null) { return oidTree + "." + oidEnd; }
361         if (oidEnd != null) { return oidEnd; }
362         else { return null; }
363     }

407      * Used just in {@link #getForbiddenCharacters()}. The method is called very
408      * often so we declare this String in the class so it does not have to be

414      * @return all forbidden characters.

416     public static char[] getForbiddenCharacters() {
417         // Using 'instance().getString' instead of 'getString' since an empty
418         // String (size 0) must be returned when the property is defined without
419         // any value.
420         final String s = ConfigurationHolder.instance().getString(FORBIDDEN_CARACTERS_KEY);
421         if (s==null) {
422             return ConfigurationHolder.getDefaultValue(FORBIDDEN_CARACTERS_KEY).toCharArray();
423         }
424         return s.toCharArray();
425     }

440     /** @return a list of enabled TLS protocol versions and cipher suites */
441     /*
442      * Java 6: http://docs.oracle.com/javase/6/docs/technotes/guides/security/SunProviders.html#SunJSSEProvider
443      *  TLS versions: SSLv3, TLSv1, SSLv2Hello
444      * Java 7: http://docs.oracle.com/javase/7/docs/technotes/guides/security/SunProviders.html#SunJSSEProvider
445      *  TLS versions: SSLv3, TLSv1, TLSv1.1, TLSv1.2
446      *  Cipher suites with SHA384 and SHA256 are available only for TLS 1.2 or later.
447      * Java 8: http://docs.oracle.com/javase/8/docs/technotes/guides/security/SunProviders.html#SunJSSEProvider
448      *  TLS versions: SSLv3, TLSv1, TLSv1.1, TLSv1.2
449      *  Cipher suites with SHA384 and SHA256 are available only for TLS 1.2 or later.
450      */
451     public static String[] getAvailableCipherSuites() {
452         final List<String> availableCipherSuites = new ArrayList<String>();
453         for (int i=0; i<255; i++) {
454             final String key = "authkeybind.ciphersuite." + i;
455             final String value = ConfigurationHolder.getString(key);
456             if (value==null || value.indexOf(AVAILABLE_CIPHER_SUITES_SPLIT_CHAR)==-1) {
457                 continue;
458             }
459             availableCipherSuites.add(value);
460         }
461         return availableCipherSuites.toArray(new String[0]);
462     }
