128   /**
129    * @param dataobj Object
130    */
131   public AvailableExtendedKeyUsagesConfiguration(final Serializable dataobj) {
132     @SuppressWarnings("unchecked")
133     LinkedHashMap<Object, Object> d = (LinkedHashMap<Object, Object>) dataobj;
134     data = d;
135   }
136 
137   @Override
138   public String getConfigurationId() {
139     return CONFIGURATION_ID;
140   }
141 
142   /**
143    * @param oid OID
144    * @return Bool
145    */
146   public boolean isExtendedKeyUsageSupported(final String oid) {
147     return data.containsKey(oid.trim());
148   }
149 
150   /**
151    * @param oid OID
152    * @param name Name
153    */
154   public void addExtKeyUsage(final String oid, final String name) {
155     data.put(oid.trim(), name);
156   }
157 
158   /**
159    * @param oid OID
160    */
161   public void removeExtKeyUsage(final String oid) {
162     data.remove(oid.trim());
163   }
164 
165   /**
166    * @param ooid OID
167    * @return Name
168    */
169   public String getExtKeyUsageName(final String ooid) {
170     String oid = ooid.trim();
171     String name = (String) data.get(oid);
172     if (name == null) {
173       name = oid;
174     }
175     return name;
176   }
177 
178   /**
179    * @return List of OIDs.
180    */
181   public List<String> getAllOIDs() {
182     Set<Object> keyset = data.keySet();
183     ArrayList<String> keys = new ArrayList<String>();
184     for (Object k : keyset) {
185       if (!StringUtils.equalsIgnoreCase((String) k, "version")) {
186         keys.add((String) k);
187       }
188     }
189     return keys;
190   }
191 
192   /**
193    * @return Map of OIDs.
194    */
195   public Map<String, String> getAllEKUOidsAndNames() {
196     @SuppressWarnings("unchecked")
197     Map<String, String> ret = (Map<String, String>) saveData();
198     ret.remove("version");
199     return ret;
200   }
201 
202   /**
203    * @return Properties
204    */
205   public Properties getAsProperties() {
206     Properties properties = new Properties();
207     Map<String, String> allEkus = getAllEKUOidsAndNames();
208     for (Entry<String, String> eku : allEkus.entrySet()) {
209       properties.setProperty(eku.getKey(), eku.getValue());
210     }
211     return properties;
212   }
