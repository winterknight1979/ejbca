13  package org.cesecore.audit;
14  
15  import java.util.Map;
16  import org.cesecore.audit.enums.EventStatus;
17  import org.cesecore.audit.enums.EventType;
18  import org.cesecore.audit.enums.ModuleType;
19  import org.cesecore.audit.enums.ServiceType;
20  
21  /**
22   * Stored log dta that should be searchable should implement this interface.
23   *
24   * @version $Id: AuditLogEntry.java 17625 2013-09-20 07:12:06Z netmackan $
25   */
26  public interface AuditLogEntry {
27  
28    /* Database column names that AuditLogDevices
29     * might implement. Used to build custom queries. */
30    /** Timestamp. */
31    String FIELD_TIMESTAMP = "timeStamp";
32    /** Event type. */
33    String FIELD_EVENTTYPE = "eventType";
34    /** Event status. */
35    String FIELD_EVENTSTATUS = "eventStatus";
36    /** Authentication token. */
37    String FIELD_AUTHENTICATION_TOKEN = "authToken";
38    /** Service. */
39    String FIELD_SERVICE = "service";
40    /** Module. */
41    String FIELD_MODULE = "module";
42    /** Custom ID. */
43    String FIELD_CUSTOM_ID = "customId";
44    /** Searchable detail. */
45    String FIELD_SEARCHABLE_DETAIL1 = "searchDetail1";
46    /** Serchable detail. */
47    String FIELD_SEARCHABLE_DETAIL2 = "searchDetail2";
48    /** Extra details. */
49    String FIELD_ADDITIONAL_DETAILS = "additionalDetails";
50    /** Sequence number. */
51    String FIELD_SEQUENCENUMBER = "sequenceNumber";
52    /** Node ID. */
53    String FIELD_NODEID = "nodeId";
54  
55    /** @return epoch GMT timestamp when log was created. */
56    Long getTimeStamp();
57  
58    /** @return event type. @see EventTypes */
59    EventType getEventTypeValue();
60  
61    /** @return event status. @see EventStatus */
62    EventStatus getEventStatusValue();
63  
64    /** @return the authentication token that triggered the creation of a log. */
65    String getAuthToken();
66  
67    /** @return service type. @see ServiceTypes */
68    ServiceType getServiceTypeValue();
69  
70    /** @return module type. @see ModuleTypes */
71    ModuleType getModuleTypeValue();
72  
73    /** @return a custom identifier (e.g. CA Id) */
74    String getCustomId();
75  
76    /** @return searchable detail1 (e.g. certificate serialnumber) */
77    String getSearchDetail1();
78  
79    /** @return searchable detail2 (e.g. username) */
80    String getSearchDetail2();
81  
82    /** @return map of additional (non-searchable) details. */
83    Map<String, Object> getMapAdditionalDetails();
84  
85    /** @return log sequence number. */
86    Long getSequenceNumber();
87  
88    /** @return node identifier to be used together with sequenceNumber. */
89    String getNodeId();
90  }
