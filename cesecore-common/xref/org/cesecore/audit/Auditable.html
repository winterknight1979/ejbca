13  package org.cesecore.audit;
14  
15  import java.util.Date;
16  import java.util.List;
17  import java.util.Map;
18  import java.util.Properties;
19  import org.cesecore.audit.audit.AuditExporter;
20  import org.cesecore.audit.audit.AuditLogExportReport;
21  import org.cesecore.audit.audit.AuditLogExporterException;
22  import org.cesecore.audit.audit.AuditLogValidationReport;
23  import org.cesecore.audit.audit.AuditLogValidatorException;
24  import org.cesecore.authentication.tokens.AuthenticationToken;
25  import org.cesecore.keys.token.CryptoToken;
26  import org.cesecore.util.query.QueryCriteria;
27  
28  /**
29   * Interface for querying, validating and exporting log.
30   *
31   * @version $Id: Auditable.java 17625 2013-09-20 07:12:06Z netmackan $
32   */
33  public interface Auditable {
34  
35    /**
36     * Select a set of events to be audited.
37     *
38     * @param token user performing the task.
39     * @param startIndex Index where select will start.
40     * @param max maximum number of results to be returned.
41     * @param criteria Criteria defining the subset of logs to be selected.
42     * @param properties properties to be passed on the device
43     * @return The audit logs to the given criteria
44     */
45    List<? extends AuditLogEntry> selectAuditLogs(
46        AuthenticationToken token,
47        int startIndex,
48        int max,
49        QueryCriteria criteria,
50        Properties properties);
51  
52    /**
53     * This operation is used to export a set logs.
54     *
55     * @param token user performing the task.
56     * @param cryptoToken Crypto Token to be used.
57     * @param timestamp Timestamp till which the logs will be exported.
58     * @param deleteAfterExport Deletes the exported results if true.
59     * @param signatureDetails Map containing signature details. (@see
60     *     AuditLogBasicExporter public static varaibles).
61     * @param exporter the exporter implementation to be used
62     * @param properties properties to be passed on the device
63     * @return A extended validation report with the path to the exported file.
64     * @throws AuditLogExporterException if export fails
65     */
66    AuditLogExportReport exportAuditLogs(
67        AuthenticationToken token,
68        CryptoToken cryptoToken,
69        Date timestamp,
70        boolean deleteAfterExport,
71        Map<String, Object> signatureDetails,
72        Properties properties,
73        Class<? extends AuditExporter> exporter)
74        throws AuditLogExporterException;
75  
76    /**
77     * This operation is used to verify integrity of log to detect potential
78     * modifications.
79     *
80     * @param token Auth Token
81     * @param date Date from which to start verifying logs.
82     * @param properties properties to be passed on the device
83     * @return validation report.
84     * @throws AuditLogValidatorException If validation fails
85     */
86    AuditLogValidationReport verifyLogsIntegrity(
87        AuthenticationToken token, Date date, Properties properties)
88        throws AuditLogValidatorException;
89  }
