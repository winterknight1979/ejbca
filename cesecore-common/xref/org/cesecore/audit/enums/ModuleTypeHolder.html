13  package org.cesecore.audit.enums;
14  
15  /**
16   * Simple implementation of ModuleType that holds the identifier.
17   * 
18   * @version $Id: ModuleTypeHolder.java 26796 2017-10-12 04:29:58Z anatom $
19   */
20  public class ModuleTypeHolder implements ModuleType {
21  
22  	private static final long serialVersionUID = 1L;
23  
24  	private final String value;
25  	
26  	public ModuleTypeHolder(final String value) {
27  		this.value = value;
28  	}
29  	
30  	@Override
31  	public String toString() {
32  		return value;
33  	}
34  	
35  	@Override
36  	public boolean equals(final ModuleType value) {
37  		if (value == null) {
38  			return false;
39  		}
40  		return this.value.equals(value.toString());
41  	}
42  }
