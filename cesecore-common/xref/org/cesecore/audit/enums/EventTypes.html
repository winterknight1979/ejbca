13  package org.cesecore.audit.enums;
14  
15  /**
16   * Contains all event types that CESeCore core needs to log to secure audit log. 
17   *
18   * When doing secure audit log it is necessary to identify the event being logged.
19   *
20   * @see org.cesecore.audit.enums.ModuleTypes
21   * @see org.cesecore.audit.enums.ServiceTypes
22   * @version $Id: EventTypes.java 26591 2017-09-20 14:19:01Z anatom $
23   */
24  public enum EventTypes implements EventType {
25  
26      /** Authorization check to resource of authenticated entity. */
27      ACCESS_CONTROL,
28      /** Authentication check of an entity. */
29      AUTHENTICATION,
30      /** Creation of a Certificate Authority. */
31      CA_CREATION,
32      /** Removal of a Certificate Authority. */
33      CA_DELETION,
34      /** Internal application name change of a Certificate Authority. Unrelated to Certificate Authority's Subject Distinguisher Name. */
35      CA_RENAMING,
36      /** Modification of a Certificate Authority. */
37      CA_EDITING,
38      /** Certificate Authority starts using a different key pair. */
39      CA_KEYACTIVATE,
40      /** Generation of a new key pair that can be used by the Certificate Authority during renewal or update. */
41      CA_KEYGEN,
42      /** Certificate Authority state change to start serving requests. Unrelated to CA private key availability. */
43      CA_SERVICEACTIVATE,
44      /** Certificate Authority state change to stop serving requests. Unrelated to CA private key availability. */
45      CA_SERVICEDEACTIVATE,
46      /** Persistence of a certificate to the database. */
47      CERT_STORED,
48      /** Change of a certificate's status to revoked or active. */
49      CERT_REVOKED,
50      /** Change of a certificate's status to unassigned, inactive, active, notified about expiration, revoked or archived. */
51      CERT_CHANGEDSTATUS,
52      /** A request for certificate issuance from a Certificate Authority is submitted. */
53      CERT_REQUEST,
54      /** Issuance of a certificate by a Certificate Authority. */
55      CERT_CREATION,
56      /** Certificate Transparency log servers responds to a pre-certificate submission from a Certificate Authority. */
57      CERT_CTPRECERT_SUBMISSION,
58      /** <i>Event type is currently not used in EJBCA.</i> */
59      CERTIFICATE_KEY_BIND,
60      /** <i>Event type is currently not used in EJBCA.</i> */
61      CERTIFICATE_KEY_UNBIND,
62      /** Creation of a certificate profile. */
63      CERTPROFILE_CREATION,
64      /** Removal of a certificate profile. */
65      CERTPROFILE_DELETION,
66      /** Name change of a certificate profile. */
67      CERTPROFILE_RENAMING,
68      /** Modification of a certificate profile. */
69      CERTPROFILE_EDITING,
70      /** Persistence of a Certificate Revocation List to the database. */
71      CRL_STORED,
72      /** <i>Event type is currently not used in EJBCA.</i> */
73      CRL_DELETED,
74      /** Issuance of a Certificate Revocation List by a Certificate Authority. */
75      CRL_CREATION,
76      /** Creation of a Crypto Token. */
77      CRYPTOTOKEN_CREATE,
78      /** Modification of a Crypto Token. */
79      CRYPTOTOKEN_EDIT,
80      /** Removal of a Crypto Token. */
81      CRYPTOTOKEN_DELETION,
82      /** Activation of a Crypto Token, making the key material available for use by the application. */
83      CRYPTOTOKEN_ACTIVATION,
84      /** Deactivation of a Crypto Token, making the key material unavailable for use by the application. */
85      CRYPTOTOKEN_DEACTIVATION,
86      /** Attempted reactivation of a Crypto Token. Since this occurs automatically, it may fail. */
87      CRYPTOTOKEN_REACTIVATION,
88      /** Removal of a key pair from the Crypto Token key material or key pair place-holder from the Crypto Token object. */
89      CRYPTOTOKEN_DELETE_ENTRY,
90      /** Generation of a new key pair in the Crypto Token. */
91      CRYPTOTOKEN_GEN_KEYPAIR,
92      /** <i>Event type is currently not used in EJBCA.</i> */
93      CRYPTOTOKEN_GEN_KEY,
94      /** <i>Event type is currently not used in EJBCA.</i> */
95      CRYPTOTOKEN_GEN_EXTRACT_KEYPAIR,
96      /** Modification of the Crypto Token's auto-activation PIN. For soft key stores, this also implies changes of the protection of the key material. */
97      CRYPTOTOKEN_UPDATEPIN,
98      /** Modification of an existing validator. */
99      VALIDATOR_CHANGE,
100     /** Creation of a new validator. */
101     VALIDATOR_CREATION,
102     /** Removal of an existing validator. */
103     VALIDATOR_REMOVAL,
104     /** Name change of an existing validator. */
105     VALIDATOR_RENAME,
106     /** Validation failed. */
107     VALIDATOR_VALIDATION_FAILED,
108     /** Validation succeeded. */
109     VALIDATOR_VALIDATION_SUCCESS,
110     /** Removal of persisted audit log records. */
111     LOG_DELETE,
112     /** Export of audit log records. */
113     LOG_EXPORT,
114     /** Change of protection settings for audit log records. */
115     LOG_MANAGEMENT_CHANGE,
116     /** <i>Event type is currently only used by EJBCA development tests.</i> */
117     LOG_SIGN,
118     /** Verification of existing audit log records. */
119     LOG_VERIFY,
120     /** Creation of an administrative role. */
121     ROLE_CREATION,
122     /** Removal of an administrative role. */
123     ROLE_DELETION,
124     /** Name change of an administrative role. */
125     ROLE_RENAMING,
126     /** New access rules added to administrative role. */
127     @Deprecated // Also msg key authorization.accessrulesadded
128     ROLE_ACCESS_RULE_ADDITION,
129     /** Modifications of existing access rules in an administrative role. */
130     ROLE_ACCESS_RULE_CHANGE,
131     /** Removal of existing access rules from administrative role. */
132     @Deprecated // Also msg key authorization.accessrulesremoved
133     ROLE_ACCESS_RULE_DELETION,
134     /** New administrator added to administrative role. */
135     ROLE_ACCESS_USER_ADDITION,
136     /** Change of existing administrator in an administrative role. */
137     ROLE_ACCESS_USER_CHANGE,
138     /** Removal of existing administrator from administrative role. */
139     ROLE_ACCESS_USER_DELETION,
140     /** Creation of new system settings stored in the database. */
141     SYSTEMCONF_CREATE,
142     /** Modification of existing system settings stored in the database. */
143     SYSTEMCONF_EDIT,
144     /** <i>Event type is currently not used in EJBCA.</i> */
145     BACKUP,
146     /** <i>Event type is currently not used in EJBCA.</i> */
147     RESTORE,
148     /** <i>Event type is currently not used in EJBCA.</i> */
149     TIME_SYNC_ACQUIRE,
150     /** <i>Event type is currently not used in EJBCA.</i> */
151     TIME_SYNC_LOST,
152     /** Creations of a new Internal Key Binding. */
153     INTERNALKEYBINDING_CREATE,
154     /** Modification of an existing Internal Key Binding. */
155     INTERNALKEYBINDING_EDIT,
156     /** Removal of an existing Internal Key Binding. */
157     INTERNALKEYBINDING_DELETE,
158     ;
159 
160     @Override
161     public boolean equals(EventType value) {
162         if(value == null) {
163             return false;
164         }
165         return this.toString().equals(value.toString());
166     }
167 }
