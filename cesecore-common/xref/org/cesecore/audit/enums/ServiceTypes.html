13  package org.cesecore.audit.enums;
14  
15  /**
16   * Represents the basic service types supported.
17   *
18   * <p>When doing secure audit log ServiceType is used to indicate if the log was
19   * executed from the core itself or by an external application.
20   *
21   * @see org.cesecore.audit.enums.EventTypes
22   * @see org.cesecore.audit.enums.ModuleTypes
23   * @version $Id: ServiceTypes.java 20951 2015-03-20 13:15:30Z jeklund $
24   */
25  public enum ServiceTypes implements ServiceType {
26    /** CE Security Core. */
27    CORE;
28  
29    @Override
30    public boolean equals(final ServiceType value) {
31      if (value == null) {
32        return false;
33      }
34      return this.toString().equals(value.toString());
35    }
36  }
