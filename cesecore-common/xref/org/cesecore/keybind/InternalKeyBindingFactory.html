29  /**
30   * Factory class with an internal registry of known implementations.
31   *
32   * @version $Id: InternalKeyBindingFactory.java 23648 2016-06-09 15:19:55Z
33   *     mikekushner $
34   */
35  public enum InternalKeyBindingFactory {
36    /** Singleton. */
37      INSTANCE;
38  
39    /** Logger. */
40     private final Logger log = Logger.getLogger(InternalKeyBindingFactory.class);
41  
42    /** Ilpl. */
43    private final Map<String, String> aliasToImplementationMap =
44        new HashMap<String, String>();
45    /** Alias. */
46    private final Map<String, String> implementationToAliasMap =
47        new HashMap<String, String>();
48    /** Properties. */
49    private final Map<
50            String, Map<String, DynamicUiProperty<? extends Serializable>>>
51        implementationPropertiesMap =
52            new HashMap<
53                String, Map<String, DynamicUiProperty<? extends Serializable>>>();
54  
55    InternalKeyBindingFactory() {
56      addImplementation(OcspKeyBinding.class);
57      addImplementation(AuthenticationKeyBinding.class);
58      // Use ServiceLoader framework to find additional available implementations
59      // We add these after the built in ones, so the built in implementations can
60      // be overridden
61      // TODO: the above when we need to
62    }
63  
64    /**
65     * @param alias alias
66     * @return bool
67     */
68    public boolean existsTypeAlias(final String alias) {
69      return aliasToImplementationMap.containsKey(alias);
70    }
71  
72    /**
73     * Creates a new InternalKeyBinding instance.
74     *
75     * @param type is the alias of the registered InternalKeyBinding's type
76     * @param id is the unique identifier of this InternalKeyBinding
77     * @param name is the unique name that this InternalKeyBinding will be given
78     * @param status the initial status to give the InternalKeyBinding
79     * @param certificateId is the certificate fingerprint (lower case) matching
80     *     the mapped key pair or null
81     * @param cryptoTokenId is the CryptoToken id of the container where the
82     *     mapped key pair is stored
83     * @param keyPairAlias is the alias of the mapped key pair in the specified
84     *     CryptoToken (may not be null)
85     * @param dataMap is a Map of implementation specific properties for this type
86     *     of IntenalKeyBinding
87     * @return a new InternalKeyBinding instance
88     */
89    public InternalKeyBinding create(
90        final String type,
91        final int id,
92        final String name,
93        final InternalKeyBindingStatus status,
94        final String certificateId,
95        final int cryptoTokenId,
96        final String keyPairAlias,
97        final LinkedHashMap<Object, Object> dataMap) {
98      final String implementationClassName = aliasToImplementationMap.get(type);
99      InternalKeyBinding internalKeyBinding = null;
100     if (implementationClassName == null) {
101       log.error(
102           "Unable to create Signer. Implementation for type '"
103               + type
104               + "' not found.");
105     } else {
106       try {
107         internalKeyBinding =
108             (InternalKeyBinding)
109                 Class.forName(implementationClassName)
110                     .getConstructor()
111                     .newInstance();
112         // Ensure that fingerprint is lower case, to match items in the database
113         final String certFp;
114         if (certificateId != null) {
115           certFp = certificateId.toLowerCase(Locale.ENGLISH);
116         } else {
117           certFp = null;
118         }
119         internalKeyBinding.init(
120             id, name, status, certFp, cryptoTokenId, keyPairAlias, dataMap);
121       } catch (InstantiationException e) {
122         log.error(
123             "Unable to create InternalKeyBinding. Could not be instantiate"
124                 + " implementation '"
125                 + implementationClassName
126                 + "'.",
127             e);
128       } catch (IllegalAccessException e) {
129         log.error(
130             "Unable to create InternalKeyBinding. Not allowed to instantiate"
131                 + " implementation '"
132                 + implementationClassName
133                 + "'.",
134             e);
135       } catch (ClassNotFoundException e) {
136         log.error(
137             "Unable to create InternalKeyBinding. Could not find"
138                 + " implementation '"
139                 + implementationClassName
140                 + "'.",
141             e);
142       } catch (InvocationTargetException e) {
143         log.error(
144             "Unable to create InternalKeyBinding. Could not be instantiate"
145                 + " implementation '"
146                 + implementationClassName
147                 + "'.",
148             e);
149       } catch (NoSuchMethodException e) {
150         log.error(
151             "Unable to create InternalKeyBinding. Could not find"
152                 + " implementation '"
153                 + implementationClassName
154                 + "'.",
155             e);
156       }
157     }
158     return internalKeyBinding;
159   }

161   /**
162    * @param internalKeyBinding binding
163    * @return the registered alias for the provided Signer or "null" if this is
164    *     an unknown implementation.
165    */
166   public String getTypeFromImplementation(
167       final InternalKeyBinding internalKeyBinding) {
168     return String.valueOf(
169         implementationToAliasMap.get(internalKeyBinding.getClass().getName()));
170   }
171 
172   private void addImplementation(final Class<? extends InternalKeyBinding> c) {
173     String alias = null;
174     List<String> implementationPropertyKeys = null;
175     Map<String, DynamicUiProperty<? extends Serializable>>
176         implementationProperties = null;
177     try {
178       final InternalKeyBinding temporaryInstance =
179           c.getConstructor().newInstance();
180       alias = temporaryInstance.getImplementationAlias();
181       implementationProperties = temporaryInstance.getCopyOfProperties();
182       implementationPropertyKeys = new ArrayList<String>();
183       for (String name : implementationProperties.keySet()) {
184         implementationPropertyKeys.add(name);
185       }
186     } catch (InstantiationException | InvocationTargetException e) {
187       log.error(
188           "Unable to create InternalKeyBinding. Could not be instantiate"
189               + " implementation '"
190               + c.getName()
191               + "'.",
192           e);
193     } catch (IllegalAccessException e) {
194       log.error(
195           "Unable to create InternalKeyBinding. Not allowed to instantiate"
196               + " implementation '"
197               + c.getName()
198               + "'.",
199           e);
200     } catch (NoSuchMethodException e) {
201       log.error(
202           "Unable to create InternalKeyBinding. Could not find implementation '"
203               + c.getName()
204               + "'.",
205           e);
206     }
207     if (alias != null) {
208       aliasToImplementationMap.put(alias, c.getName());
209       implementationToAliasMap.put(c.getName(), alias);
210       implementationPropertiesMap.put(
211           alias, Collections.unmodifiableMap((implementationProperties)));
212     }
213   }
214 
215   /**
216    * @return Properties
217    */
218   public Map<String, Map<String, DynamicUiProperty<? extends Serializable>>>
219       getAvailableTypesAndProperties() {
220     return Collections.unmodifiableMap(implementationPropertiesMap);
221   }

223   /**
224    * Method to be used from an external environment (such as the CLI) where
225    * contextual information about a key binding implementation is not available.
226    * Will presume that all key binding properties were entered as strings, will
227    * return a data wrapper that contains a map of properly casted values, and
228    * information about any values which were either unknown or of an incorrect
229    * format.
230    *
231    * @param alias the alias of the key binding type to check for
232    * @param propertiesMap the inputed properties, as strings
233    * @return a wrapper class that contains the correctly casted values, and
234    *     information about any values that were invalid.
235    */
236   public InternalKeyBindingPropertyValidationWrapper validateProperties(
237       final String alias, final Map<String, String> propertiesMap) {
238     InternalKeyBindingPropertyValidationWrapper result =
239         new InternalKeyBindingPropertyValidationWrapper();
240     Map<String, DynamicUiProperty<? extends Serializable>>
241         implementationProperties = implementationPropertiesMap.get(alias);
242     for (String key : propertiesMap.keySet()) {
243       if (!implementationProperties.containsKey(key)) {
244         result.addUnknownProperty(key);
245         continue;
246       }
247       DynamicUiProperty<? extends Serializable> property =
248           implementationProperties.get(key);
249       String value = propertiesMap.get(key);
250       Serializable recastValue = property.valueOf(value);
251       if (recastValue == null) {
252         result.addInvalidValue(key, property.getType());
253         continue;
254       }
255       result.addProperty(key, recastValue);
256     }
257     return result;
258   }
