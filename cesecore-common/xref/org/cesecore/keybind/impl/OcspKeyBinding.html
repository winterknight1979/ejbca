40      public enum ResponderIdType {
41          KEYHASH(2, "KeyHash"), NAME(1, "Name");
42          
43          private final int numericValue;
44          private final String label;
45          private static Map<Integer, ResponderIdType> numericValueLookupMap;
46          
47          static {
48              numericValueLookupMap = new HashMap<>();
49              for(ResponderIdType responderIdType : ResponderIdType.values()) {
50                  numericValueLookupMap.put(responderIdType.getNumericValue(), responderIdType);
51              }
52          }
53          
54          private ResponderIdType(int numericValue, String label) {
55              this.numericValue = numericValue;
56              this.label = label;
57          }
58          
59          public int getNumericValue() {
60              return numericValue;
61          }
62          
63          public String getLabel() {
64              return label;
65          }
66          
67          public static ResponderIdType getFromNumericValue(int numericValue) {
68              return numericValueLookupMap.get(Integer.valueOf(numericValue));
69          }
70  
71      }

198     public static boolean isOcspSigningCertificate(final Certificate certificate, AvailableExtendedKeyUsagesConfiguration ekuConfig) {
199         try {
200             assertCertificateCompatabilityInternal(certificate, ekuConfig);
201         } catch (CertificateImportException e) {
202             return false;
203         }
204         return true;
205     }
206 
207     private static void assertCertificateCompatabilityInternal(final Certificate certificate, AvailableExtendedKeyUsagesConfiguration ekuConfig) throws CertificateImportException {
208         if (certificate == null) {
209             throw new CertificateImportException("No certificate provided.");
210         }
211         if (!(certificate instanceof X509Certificate)) {
212             throw new CertificateImportException("Only X509 certificates are supported for OCSP.");
213         }
214         try {
215             final X509Certificate x509Certificate = (X509Certificate) certificate;
216             if (log.isDebugEnabled()) {
217                 log.debug("SubjectDN: " + CertTools.getSubjectDN(x509Certificate) + " IssuerDN: " + CertTools.getIssuerDN(x509Certificate));
218                 final boolean[] ku = x509Certificate.getKeyUsage();
219                 log.debug("Key usages: " + Arrays.toString(ku));
220                 if (ku != null) {
221                     log.debug("Key usage (digitalSignature): " + x509Certificate.getKeyUsage()[0]);
222                     log.debug("Key usage (nonRepudiation):   " + x509Certificate.getKeyUsage()[1]);
223                     log.debug("Key usage (keyEncipherment):  " + x509Certificate.getKeyUsage()[2]);
224                 }
225             }
226             if (x509Certificate.getExtendedKeyUsage() == null) {
227                 throw new CertificateImportException("No Extended Key Usage present in certificate.");
228             }
229             for (String extendedKeyUsage : x509Certificate.getExtendedKeyUsage()) {
230                 log.debug("EKU: " + extendedKeyUsage + " (" +
231                         ekuConfig.getAllEKUOidsAndNames().get(extendedKeyUsage) + ")");
232             }
233             if (!x509Certificate.getExtendedKeyUsage().contains(KeyPurposeId.id_kp_OCSPSigning.getId())) {
234                 throw new CertificateImportException("Extended Key Usage 1.3.6.1.5.5.7.3.9 (EKU_PKIX_OCSPSIGNING) is required.");
235             }
236             if (!x509Certificate.getKeyUsage()[0] && !x509Certificate.getKeyUsage()[1] ) {
237                 throw new CertificateImportException("Key Usage digitalSignature is required (nonRepudiation would also be accepted).");
238             }
239         } catch (CertificateParsingException e) {
240             throw new CertificateImportException(e.getMessage(), e);
241         }
