29  /**
30   * Used when this EJBCA instance authenticates to other instances.
31   * 
32   * @version $Id: AuthenticationKeyBinding.java 26192 2017-08-02 07:59:40Z anatom $
33   */
34  public class AuthenticationKeyBinding extends InternalKeyBindingBase {
35  
36      private static final long serialVersionUID = 1L;
37      private static final Logger log = Logger.getLogger(AuthenticationKeyBinding.class);
38  
39      public static final String IMPLEMENTATION_ALIAS = "AuthenticationKeyBinding"; // This should not change, even if we rename the class in EJBCA 5.3+..
40      public static final String PROPERTY_PROTOCOL_AND_CIPHER_SUITE = "protocolAndCipherSuite";
41  
42      {
43          final String[] CIPHER_SUITES_SUBSET = CesecoreConfiguration.getAvailableCipherSuites();
44          addProperty(new DynamicUiProperty<String>(PROPERTY_PROTOCOL_AND_CIPHER_SUITE, CIPHER_SUITES_SUBSET[0], Arrays.asList(CIPHER_SUITES_SUBSET)));
45      }
46  
47      /** @return an array of supported protocols named according to JSSE */
48      public String[] getSupportedProtocols() {
49          return getSelectedProtocolOrSuite(0);
50      }
51  
52      /** @return an array of supported cipher suites named according to JSSE */
53      public String[] getSupportedCipherTextSuites() {
54          return getSelectedProtocolOrSuite(1);
55      }
56  
57      private String[] getSelectedProtocolOrSuite(final int pos) {
58          final String value = (String) getProperty(PROPERTY_PROTOCOL_AND_CIPHER_SUITE).getValue();
59          final String[] values = value.split(CesecoreConfiguration.AVAILABLE_CIPHER_SUITES_SPLIT_CHAR);
60          if (log.isDebugEnabled() && pos==0) {
61              log.debug("Configured cipher suite for this AuthenticationKeyBinding: " + value);
62          }
63          if (values.length==2) {
64              return new String[] { values[pos] };
65          }
66          return new String[0];
67      }
68  
69      @Override
70      public String getImplementationAlias() {
71          return IMPLEMENTATION_ALIAS;
72      }
73  
74      @Override
75      public float getLatestVersion() {
76          return serialVersionUID;
77      }
78  
79      @Override
80      public void assertCertificateCompatability(Certificate certificate, final AvailableExtendedKeyUsagesConfiguration ekuConfig) throws CertificateImportException {
81          if (!isClientSSLCertificate(certificate, ekuConfig)) {
82              throw new CertificateImportException("Not a valid Client SSL authentication certificate.");
83          }
84      }
85  
86      @Override
87      protected void upgrade(float latestVersion, float currentVersion) {
88          // Nothing to do   
89      }
90  
91      public static boolean isClientSSLCertificate(Certificate certificate, final AvailableExtendedKeyUsagesConfiguration ekuConfig) {
92          if (certificate == null) {
93              log.debug("No certificate provided.");
94              return false;
95          }
96          if (!(certificate instanceof X509Certificate)) {
97              log.debug("Only X509 supported.");
98              return false;
99          }
100         try {
101             final X509Certificate x509Certificate = (X509Certificate) certificate;
102             if (log.isDebugEnabled()) {
103                 log.debug("SubjectDN: " + CertTools.getSubjectDN(x509Certificate) + " IssuerDN: " + CertTools.getIssuerDN(x509Certificate));
104             }
105             final boolean[] ku = x509Certificate.getKeyUsage();
106             if (ku != null) {
107                 if (log.isDebugEnabled()) {
108                     log.debug("Key usages: " + Arrays.toString(ku));
109                     log.debug("Key usage (digitalSignature): " + ku[0]);
110                     log.debug("Key usage (keyEncipherment): " + ku[2]);
111                 }
112             } else {
113                 log.debug("No Key Usage to verify.");
114                 return false;            	
115             }
116             if (x509Certificate.getExtendedKeyUsage() == null) {
117                 log.debug("No EKU to verify.");
118                 return false;
119             }
120             for (String extendedKeyUsage : x509Certificate.getExtendedKeyUsage()) {
121                 log.debug("EKU: " + extendedKeyUsage + " (" +
122                         ekuConfig.getAllEKUOidsAndNames().get(extendedKeyUsage) + ")");
123             }
124             if (!x509Certificate.getExtendedKeyUsage().contains(KeyPurposeId.id_kp_clientAuth.getId())) {
125                 log.debug("Extended Key Usage 1.3.6.1.5.5.7.3.2 (EKU_PKIX_CLIENTAUTH) is required.");
126                 return false;
127             }
128             // For TLS _client_ certificates you can actually be without KU completely, but we take the safe route here and require digitalSignature
129             // for TLS _server_ certificates also keyEncipherment is required, but not for client (it doesn't hurt it it's there for clients as well though)
130             if (!ku[0]) {
131                 log.debug("Key usage digitalSignature is required.");
132                 return false;
133             }
134         } catch (CertificateParsingException e) {
135             log.debug(e.getMessage());
136             return false;
137         }
138         return true;
139     }
