28  /**
29   * Used when this EJBCA instance authenticates to other instances.
30   *
31   * @version $Id: AuthenticationKeyBinding.java 26192 2017-08-02 07:59:40Z anatom
32   *     $
33   */
34  public class AuthenticationKeyBinding extends InternalKeyBindingBase {
35  
36    private static final long serialVersionUID = 1L;
37    /** Logger. */
38    private static final Logger LOG =
39        Logger.getLogger(AuthenticationKeyBinding.class);
40  
41    /** Alias. */
42    public static final String IMPLEMENTATION_ALIAS =
43        "AuthenticationKeyBinding"; // This should not change, even if we rename
44                                    // the class in EJBCA 5.3+..
45    /** Property name. */
46    public static final String PROPERTY_PROTOCOL_AND_CIPHER_SUITE =
47        "protocolAndCipherSuite";
48  
49    {
50      final String[] cipherSuitesSubset =
51          CesecoreConfiguration.getAvailableCipherSuites();
52      addProperty(
53          new DynamicUiProperty<String>(
54              PROPERTY_PROTOCOL_AND_CIPHER_SUITE,
55              cipherSuitesSubset[0],
56              Arrays.asList(cipherSuitesSubset)));
57    }
58  
59    /** @return an array of supported protocols named according to JSSE */
60    public String[] getSupportedProtocols() {
61      return getSelectedProtocolOrSuite(0);
62    }
63  
64    /** @return an array of supported cipher suites named according to JSSE */
65    public String[] getSupportedCipherTextSuites() {
66      return getSelectedProtocolOrSuite(1);
67    }
68  
69    private String[] getSelectedProtocolOrSuite(final int pos) {
70      final String value =
71          (String) getProperty(PROPERTY_PROTOCOL_AND_CIPHER_SUITE).getValue();
72      final String[] values =
73          value.split(CesecoreConfiguration.AVAILABLE_CIPHER_SUITES_SPLIT_CHAR);
74      if (LOG.isDebugEnabled() && pos == 0) {
75        LOG.debug(
76            "Configured cipher suite for this AuthenticationKeyBinding: "
77                + value);
78      }
79      if (values.length == 2) {
80        return new String[] {values[pos]};
81      }
82      return new String[0];
83    }
84  
85    @Override
86    public String getImplementationAlias() {
87      return IMPLEMENTATION_ALIAS;
88    }
89  
90    @Override
91    public float getLatestVersion() {
92      return serialVersionUID;
93    }
94  
95    @Override
96    public void assertCertificateCompatability(
97        final Certificate certificate,
98        final AvailableExtendedKeyUsagesConfiguration ekuConfig)
99        throws CertificateImportException {
100     if (!isClientSSLCertificate(certificate, ekuConfig)) {
101       throw new CertificateImportException(
102           "Not a valid Client SSL authentication certificate.");
103     }
104   }
105 
106   @Override
107   protected void upgrade(
108           final float latestVersion, final float currentVersion) {
109     // Nothing to do
110   }

112   /**
113    * @param certificate Cert
114    * @param ekuConfig Config
115    * @return Bool
116    */
117   public static boolean isClientSSLCertificate(
118       final Certificate certificate,
119       final AvailableExtendedKeyUsagesConfiguration ekuConfig) {
120     if (certificate == null) {
121       LOG.debug("No certificate provided.");
122       return false;
123     }
124     if (!(certificate instanceof X509Certificate)) {
125       LOG.debug("Only X509 supported.");
126       return false;
127     }
128     try {
129       final X509Certificate x509Certificate = (X509Certificate) certificate;
130       if (LOG.isDebugEnabled()) {
131         LOG.debug(
132             "SubjectDN: "
133                 + CertTools.getSubjectDN(x509Certificate)
134                 + " IssuerDN: "
135                 + CertTools.getIssuerDN(x509Certificate));
136       }
137       final boolean[] ku = x509Certificate.getKeyUsage();
138       if (ku != null) {
139         if (LOG.isDebugEnabled()) {
140           LOG.debug("Key usages: " + Arrays.toString(ku));
141           LOG.debug("Key usage (digitalSignature): " + ku[0]);
142           LOG.debug("Key usage (keyEncipherment): " + ku[2]);
143         }
144       } else {
145         LOG.debug("No Key Usage to verify.");
146         return false;
147       }
148       if (x509Certificate.getExtendedKeyUsage() == null) {
149         LOG.debug("No EKU to verify.");
150         return false;
151       }
152       for (String extendedKeyUsage : x509Certificate.getExtendedKeyUsage()) {
153         LOG.debug(
154             "EKU: "
155                 + extendedKeyUsage
156                 + " ("
157                 + ekuConfig.getAllEKUOidsAndNames().get(extendedKeyUsage)
158                 + ")");
159       }
160       if (!x509Certificate
161           .getExtendedKeyUsage()
162           .contains(KeyPurposeId.id_kp_clientAuth.getId())) {
163         LOG.debug(
164             "Extended Key Usage 1.3.6.1.5.5.7.3.2 (EKU_PKIX_CLIENTAUTH) is"
165                 + " required.");
166         return false;
167       }
168       // For TLS _client_ certificates you can actually be without KU
169       // completely, but we take the safe route here and require
170       // digitalSignature
171       // for TLS _server_ certificates also keyEncipherment is required, but not
172       // for client (it doesn't hurt it it's there for clients as well though)
173       if (!ku[0]) {
174         LOG.debug("Key usage digitalSignature is required.");
175         return false;
176       }
177     } catch (CertificateParsingException e) {
178       LOG.debug(e.getMessage());
179       return false;
180     }
181     return true;
182   }
