13  package org.cesecore.keybind.impl;
14  
15  import java.security.cert.CertificateException;
16  import java.security.cert.X509Certificate;
17  import java.util.ArrayList;
18  import java.util.Arrays;
19  import java.util.Collection;
20  import java.util.Iterator;
21  import java.util.List;
22  import javax.net.ssl.X509TrustManager;
23  import org.bouncycastle.asn1.x509.KeyPurposeId;
24  import org.cesecore.util.CertTools;
25  import org.cesecore.util.provider.EkuPKIXCertPathChecker;
26  
27  /**
28   * @version $Id: ClientX509TrustManager.java 27582 2017-12-19 10:43:30Z anatom $
29   */
30  public class ClientX509TrustManager implements X509TrustManager {
31  
32      /** Trusted chains. */
33    private List<Collection<X509Certificate>> trustedCertificatesChains = null;
34    /** Chains. */
35    private List<X509Certificate> encounteredServerCertificateChain = null;
36  
37    /**
38     * @param trustedCertificates Certs.
39     */
40    public ClientX509TrustManager(
41        final List<Collection<X509Certificate>> trustedCertificates) {
42      if (trustedCertificates != null) {
43        trustedCertificatesChains = new ArrayList<>(trustedCertificates);
44      }
45    }
46  
47    @Override
48    public void checkClientTrusted(
49            final X509Certificate[] chain, final String authType)
50        throws CertificateException {
51      X509Certificate cert = chain[0];
52      // Validate the certificate and require a critical EKU extensions (if
53      // present) to contain the purpose "clientAuth"
54      if (!CertTools.verifyWithTrustedCertificates(
55          cert,
56          trustedCertificatesChains,
57          new EkuPKIXCertPathChecker(KeyPurposeId.id_kp_clientAuth.getId()))) {
58        String subjectdn = CertTools.getSubjectDN(cert);
59        String issuerdn = CertTools.getIssuerDN(cert);
60        String sn = CertTools.getSerialNumberAsString(cert);
61        String errmsg =
62            "Client certificate with SubjectDN '"
63                + subjectdn
64                + "', IssuerDN '"
65                + issuerdn
66                + "' and serialnumber '"
67                + sn
68                + "' is NOT trusted.";
69        throw new CertificateException(errmsg);
70      }
71    }
72  
73    @Override
74    public void checkServerTrusted(
75            final X509Certificate[] chain, final String authType)
76        throws CertificateException {
77      X509Certificate cert = chain[0];
78      encounteredServerCertificateChain = new ArrayList<>(Arrays.asList(chain));
79      // Validate the certificate and require a critical EKU extensions (if
80      // present) to contain the purpose "serverAuth"
81      if (!CertTools.verifyWithTrustedCertificates(
82          cert,
83          trustedCertificatesChains,
84          new EkuPKIXCertPathChecker(KeyPurposeId.id_kp_serverAuth.getId()))) {
85        String subjectdn = CertTools.getSubjectDN(cert);
86        String issuerdn = CertTools.getIssuerDN(cert);
87        String sn = CertTools.getSerialNumberAsString(cert);
88        String errmsg =
89            "Server certificate with SubjectDN '"
90                + subjectdn
91                + "', IssuerDN '"
92                + issuerdn
93                + "' and serialnumber '"
94                + sn
95                + "' is NOT trusted.";
96        throw new CertificateException(errmsg);
97      }
98    }
99  
100   @Override
101   public X509Certificate[] getAcceptedIssuers() {
102     if (trustedCertificatesChains == null) {
103       return new X509Certificate[0];
104     }
105 
106     ArrayList<X509Certificate> acceptedIssuers =
107         new ArrayList<X509Certificate>();
108     for (Collection<X509Certificate> certChain : trustedCertificatesChains) {
109       Iterator<X509Certificate> itr = certChain.iterator();
110       X509Certificate cert = itr.next();
111       if (CertTools.isCA(cert)) {
112         acceptedIssuers.add(cert);
113       } else {
114         acceptedIssuers.add(itr.next());
115       }
116     }
117     return acceptedIssuers.toArray(new X509Certificate[0]);
118   }
119 
120   /**
121    * @return the encountered server side certificate chain that this class has
122    *     been asked to verify or null if none has been encountered yet.
123    */
124   public List<X509Certificate> getEncounteredServerCertificateChain() {
125     return encounteredServerCertificateChain;
126   }
127 }
