33  /**
34   * This is an implementation of the AuthenticationToken concept, based on using
35   * an {@link X509Certificate} as it's single credential, and that certificate's
36   * {@link X500Principal} as its principle, but as the X500Principle is contained
37   * in the X509Certificate, this remains little more than a formality. This
38   * AuthenticationToken is the default used in EJBCA.
39   *
40   * <p>The implementation of the <code>matches(...)</code> method is based on
41   * <code>AdminEntity.java 10832 2010-12-13 13:54:25Z
42   *     anatom</code> from EJBCA.
43   *
44   * @version $Id: X509CertificateAuthenticationToken.java 31393 2019-02-05
45   *     11:07:23Z samuellb $
46   */
47  public class X509CertificateAuthenticationToken
48      extends NestableAuthenticationToken {
49    /** Metadata. */
50    public static final X509CertificateAuthenticationTokenMetaData METADATA =
51        new X509CertificateAuthenticationTokenMetaData();
52    /** Logger. */
53    private static final Logger LOG =
54        Logger.getLogger(X509CertificateAuthenticationToken.class);
55  
56    private static final long serialVersionUID = 1097165653913865515L;
57    /** Pattern to extract SN from certificate. */
58    private static final Pattern SERIAL_PATTERN =
59        Pattern.compile("\\bSERIALNUMBER=", Pattern.CASE_INSENSITIVE);
60    /** Certificate. */
61    private final X509Certificate certificate;
62    // get the subjectDN from the certificate and keep it for caching (speed
63    // optimization)
64    /** Subject DN. */
65    private transient String adminSubjectDN;
66    /** CA ID. */
67    private final int adminCaId;
68    /** Extractor for D-name. */
69    private final DNFieldExtractor dnExtractor;
70    /** Extractor for A-Name. */
71    private final DNFieldExtractor anExtractor;

73    /**
74     * Standard constructor for X509CertificateAuthenticationToken.
75     *
76     * @param principals A set of X500Principals. Should contain one and only one
77     *     value.
78     * @param credentials A set of X509Certificates. As with the principals, this
79     *     set should contain one and only one value, anything else will result in
80     *     a {@link InvalidAuthenticationTokenException} being thrown.
81     */
82    public X509CertificateAuthenticationToken(
83        final Set<X500Principal> principals,
84        final Set<X509Certificate> credentials) {
85      super(principals, credentials);
86      /*
87       * In order to save having to verify the credentials set every time the
88       *<code>matches(...)</code> method is called, it's checked here, and the
89       * resulting credential is stored locally.
90       */
91      final X509Certificate[] certificateArray =
92          getCredentials().toArray(new X509Certificate[0]);
93      if (certificateArray.length != 1) {
94        throw new InvalidAuthenticationTokenException(
95            "X509CertificateAuthenticationToken was containing "
96                + certificateArray.length
97                + " credentials instead of 1.");
98      } else {
99        // Speed optimization, make it into a BC class, since
100       // we will want that many times later on
101       final String clazz = certificateArray[0].getClass().getName();
102       if (clazz.contains("org.bouncycastle")) {
103         certificate = certificateArray[0];
104       } else {
105         final CertificateFactory cf = CertTools.getCertificateFactory();
106         X509Certificate cert;
107         try {
108           cert =
109               (X509Certificate)
110                   cf.generateCertificate(
111                       new ByteArrayInputStream(
112                           certificateArray[0].getEncoded()));
113         } catch (CertificateException e) {
114           LOG.warn(
115               "Error encoding/decoding client TLS certificate "
116                   + "in BC, just passing instead of optimizing: ",
117               e);
118           cert = certificateArray[0];
119         }
120         certificate = cert;
121       }
122     }
123     String certstring = CertTools.getSubjectDN(certificate).toString();
124     adminCaId = CertTools.getIssuerDN(certificate).hashCode();
125     adminSubjectDN = CertTools.getSubjectDN(certificate);
126     certstring = SERIAL_PATTERN.matcher(certstring).replaceAll("SN=");
127     final String altNameString =
128         CertTools.getSubjectAlternativeName(certificate);
129     dnExtractor =
130         new DNFieldExtractor(certstring, DNFieldExtractor.TYPE_SUBJECTDN);
131     anExtractor =
132         new DNFieldExtractor(
133             altNameString, DNFieldExtractor.TYPE_SUBJECTALTNAME);
134   }

136   /**
137    * Standard simplified constructor for X509CertificateAuthenticationToken.
138    *
139    * @param aCertificate A X509Certificate that will be used as principal and
140    *     credential.
141    * @throws NullPointerException if the provided certificate is null
142    */
143   public X509CertificateAuthenticationToken(
144       final X509Certificate aCertificate) {
145     this(
146         new HashSet<>(Arrays.asList(aCertificate.getSubjectX500Principal())),
147         new HashSet<>(Arrays.asList(aCertificate)));
148   }

150   @SuppressWarnings("deprecation")
151   @Override
152   public boolean matches(final AccessUserAspect accessUser) {
153     // Protect against spoofing by checking if this
154     // token was created locally
155     if (!super.isCreatedInThisJvm()) {
156       return false;
157     }
158     boolean returnvalue = false;
159     int parameter;
160     int size = 0;
161     String[] clientstrings = null;
162     if (StringUtils.equals(
163         getMetaData().getTokenType(), accessUser.getTokenType())) {
164       // First check that issuers match.
165       if (accessUser.getCaId() == adminCaId) {
166         // Check if we actually have some value to match against,
167         // null is not an allowed match value
168         if (accessUser.getMatchValue() != null) {
169           // Determine part of certificate to match with.
170           DNFieldExtractor usedExtractor = dnExtractor;
171           X500PrincipalAccessMatchValue matchValue =
172               (X500PrincipalAccessMatchValue)
173                   getMatchValueFromDatabaseValue(accessUser.getMatchWith());
174           if (matchValue == X500PrincipalAccessMatchValue.WITH_SERIALNUMBER) {
175             try {
176               BigInteger matchValueAsBigInteger =
177                   new BigInteger(accessUser.getMatchValue(), 16);
178               switch (accessUser.getMatchTypeAsType()) {
179                 case TYPE_EQUALCASE:
180                 case TYPE_EQUALCASEINS:
181                   returnvalue =
182                       matchValueAsBigInteger.equals(
183                           certificate.getSerialNumber());
184                   break;
185                 case TYPE_NOT_EQUALCASE:
186                 case TYPE_NOT_EQUALCASEINS:
187                   returnvalue =
188                       !matchValueAsBigInteger.equals(
189                           certificate.getSerialNumber());
190                   break;
191                 default:
192               }
193             } catch (NumberFormatException nfe) {
194               LOG.info(
195                   "Invalid matchValue for accessUser "
196                       + "when expecting a hex serialNumber: "
197                       + accessUser.getMatchValue());
198             }
199           } else if (matchValue == X500PrincipalAccessMatchValue.WITH_FULLDN) {
200             String value = accessUser.getMatchValue();
201             switch (accessUser.getMatchTypeAsType()) {
202               case TYPE_EQUALCASE:
203                 returnvalue = value.equals(CertTools.getSubjectDN(certificate));
204                 break;
205               case TYPE_EQUALCASEINS:
206                 returnvalue =
207                     value.equalsIgnoreCase(CertTools.getSubjectDN(certificate));
208                 break;
209               case TYPE_NOT_EQUALCASE:
210                 returnvalue =
211                     !value.equals(CertTools.getSubjectDN(certificate));
212               case TYPE_NOT_EQUALCASEINS:
213                 returnvalue =
214                     !value.equalsIgnoreCase(
215                         CertTools.getSubjectDN(certificate));
216                 break;
217               default:
218             }
219           } else {
220             parameter = DNFieldExtractor.CN;
221             switch (matchValue) {
222               case WITH_COUNTRY:
223                 parameter = DNFieldExtractor.C;
224                 break;
225               case WITH_DOMAINCOMPONENT:
226                 parameter = DNFieldExtractor.DC;
227                 break;
228               case WITH_STATEORPROVINCE:
229                 parameter = DNFieldExtractor.ST;
230                 break;
231               case WITH_LOCALITY:
232                 parameter = DNFieldExtractor.L;
233                 break;
234               case WITH_ORGANIZATION:
235                 parameter = DNFieldExtractor.O;
236                 break;
237               case WITH_ORGANIZATIONALUNIT:
238                 parameter = DNFieldExtractor.OU;
239                 break;
240               case WITH_TITLE:
241                 parameter = DNFieldExtractor.T;
242                 break;
243               case WITH_DNSERIALNUMBER:
244                 parameter = DNFieldExtractor.SN;
245                 break;
246               case WITH_COMMONNAME:
247                 parameter = DNFieldExtractor.CN;
248                 break;
249               case WITH_UID:
250                 parameter = DNFieldExtractor.UID;
251                 break;
252               case WITH_DNEMAILADDRESS:
253                 parameter = DNFieldExtractor.E;
254                 break;
255               case WITH_RFC822NAME:
256                 parameter = DNFieldExtractor.RFC822NAME;
257                 usedExtractor = anExtractor;
258                 break;
259               case WITH_UPN:
260                 parameter = DNFieldExtractor.UPN;
261                 usedExtractor = anExtractor;
262                 break;
263               default:
264             }
265             size = usedExtractor.getNumberOfFields(parameter);
266             clientstrings = new String[size];
267             for (int i = 0; i < size; i++) {
268               clientstrings[i] = usedExtractor.getField(parameter, i);
269             }
270 
271             // Determine how to match.
272             if (clientstrings != null) {
273               switch (accessUser.getMatchTypeAsType()) {
274                 case TYPE_EQUALCASE:
275                   String accessUserMatchValue = accessUser.getMatchValue();
276                   for (int i = 0; i < size; i++) {
277                     returnvalue = clientstrings[i].equals(accessUserMatchValue);
278                     if (returnvalue) {
279                       break;
280                     }
281                   }
282                   break;
283                 case TYPE_EQUALCASEINS:
284                   for (int i = 0; i < size; i++) {
285                     returnvalue =
286                         clientstrings[i].equalsIgnoreCase(
287                             accessUser.getMatchValue());
288                     if (returnvalue) {
289                       break;
290                     }
291                   }
292                   break;
293                 case TYPE_NOT_EQUALCASE:
294                   for (int i = 0; i < size; i++) {
295                     returnvalue =
296                         !clientstrings[i].equals(accessUser.getMatchValue());
297                     if (returnvalue) {
298                       break;
299                     }
300                   }
301                   break;
302                 case TYPE_NOT_EQUALCASEINS:
303                   for (int i = 0; i < size; i++) {
304                     returnvalue =
305                         !clientstrings[i].equalsIgnoreCase(
306                             accessUser.getMatchValue());
307                     if (returnvalue) {
308                       break;
309                     }
310                   }
311                   break;
312                 default:
313               }
314             }
315           }
316         } else {
317           if (LOG.isTraceEnabled()) {
318             LOG.trace(
319                 "Match value is null and could not "
320                     + "be matched. A value is required.");
321           }
322         }
323       } else {
324         if (LOG.isTraceEnabled()) {
325           LOG.trace(
326               "Caid does not match. Required="
327                   + adminCaId
328                   + ", actual was "
329                   + accessUser.getCaId());
330         }
331       }
332     } else {
333       if (LOG.isTraceEnabled()) {
334         LOG.trace(
335             "Token type does not match. Required="
336                 + getMetaData().getTokenType()
337                 + ", actual was "
338                 + accessUser.getTokenType());
339       }
340     }
341 
342     return returnvalue;
343   }

345   @Override
346   public int getPreferredMatchKey() {
347     return X500PrincipalAccessMatchValue.WITH_SERIALNUMBER.getNumericValue();
348   }

350   /** Returns the serial number as a decimal string. */
351   @Override
352   public String getPreferredMatchValue() {
353     return CertTools.getSerialNumberAsString(certificate);
354   }

364   /**
365    * Override the default X500Principal.getName() when doing toString on this
366    * object.
367    */
368   @Override
369   protected String toStringOverride() {
370     // Return cached value to optimize, because this can be
371     // called multiple times during the tokens lifetime
372     if (adminSubjectDN == null) {
373       adminSubjectDN = CertTools.getSubjectDN(certificate);
374     }
375     return adminSubjectDN;
376   }

378   @Override
379   public int hashCode() {
380     final int prime = 4711;
381     int result = 1;
382     result =
383         prime * result + ((certificate == null) ? 0 : certificate.hashCode());
384     return result;
385   }

387   @Override
388   public boolean equals(final Object obj) {
389     if (this == obj) {
390       return true;
391     }
392     if (obj == null) {
393       return false;
394     }
395     if (getClass() != obj.getClass()) {
396       return false;
397     }
398     X509CertificateAuthenticationToken other =
399         (X509CertificateAuthenticationToken) obj;
400     if (certificate == null) {
401       if (other.certificate != null) {
402         return false;
403       }
404     } else if (!certificate.equals(other.certificate)) {
405       return false;
406     }
407     return true;
408   }

410   /** @return the certificate */
411   public X509Certificate getCertificate() {
412     return certificate;
413   }

415   @Override
416   protected String generateUniqueId() {
417     byte[] encodedCertificate = null;
418     try {
419       encodedCertificate = certificate.getEncoded();
420     } catch (CertificateEncodingException e) {
421       throw new IllegalStateException(e);
422     }
423     return generateUniqueId(super.isCreatedInThisJvm(), encodedCertificate)
424         + ";"
425         + super.generateUniqueId();
426   }
