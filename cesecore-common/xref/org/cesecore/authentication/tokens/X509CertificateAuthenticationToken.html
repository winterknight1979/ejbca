63      /**
64       * Standard constructor for X509CertificateAuthenticationToken
65       * 
66       * @param principals
67       *            A set of X500Principals. Should contain one and only one value.
68       * @param credentials
69       *            A set of X509Certificates. As with the principals, this set should contain one and only one value, anything else will result in a
70       *            {@link InvalidAuthenticationTokenException} being thrown.
71       */
72      public X509CertificateAuthenticationToken(final Set<X500Principal> principals, final Set<X509Certificate> credentials) {
73          super(principals, credentials);
74          /*
75           * In order to save having to verify the credentials set every time the <code>matches(...)</code> method is called, it's checked here, and the
76           * resulting credential is stored locally.
77           */
78          final X509Certificate[] certificateArray = getCredentials().toArray(new X509Certificate[0]);
79          if (certificateArray.length != 1) {
80              throw new InvalidAuthenticationTokenException("X509CertificateAuthenticationToken was containing " + certificateArray.length
81                      + " credentials instead of 1.");
82          } else {
83              // Speed optimization, make it into a BC class, since we will want that many times later on
84              final String clazz = certificateArray[0].getClass().getName();
85              if (clazz.contains("org.bouncycastle")) {
86                  certificate = certificateArray[0];
87              } else {
88                  final CertificateFactory cf = CertTools.getCertificateFactory();
89                  X509Certificate cert;
90                  try {
91                      cert = (X509Certificate) cf.generateCertificate(new ByteArrayInputStream(certificateArray[0].getEncoded()));
92                  } catch (CertificateException e) {
93                      log.warn("Error encoding/decoding client TLS certificate in BC, just passing instead of optimizing: ", e);
94                      cert = certificateArray[0];
95                  }
96                  certificate = cert;
97              } 
98          }
99          String certstring = CertTools.getSubjectDN(certificate).toString();
100         adminCaId = CertTools.getIssuerDN(certificate).hashCode();
101         adminSubjectDN = CertTools.getSubjectDN(certificate);
102         certstring = serialPattern.matcher(certstring).replaceAll("SN=");
103         final String altNameString = CertTools.getSubjectAlternativeName(certificate);
104         dnExtractor = new DNFieldExtractor(certstring, DNFieldExtractor.TYPE_SUBJECTDN);
105         anExtractor = new DNFieldExtractor(altNameString, DNFieldExtractor.TYPE_SUBJECTALTNAME);
106     }
