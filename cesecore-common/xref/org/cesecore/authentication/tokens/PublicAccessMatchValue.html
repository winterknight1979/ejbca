13  package org.cesecore.authentication.tokens;
14  
15  import java.util.Arrays;
16  import java.util.List;
17  
18  import org.cesecore.authorization.user.AccessMatchType;
19  import org.cesecore.authorization.user.matchvalues.AccessMatchValue;
20  
21  /**
22   * @version $Id: PublicAccessMatchValue.java 26057 2017-06-22 08:08:34Z anatom $
23   */
24  public enum PublicAccessMatchValue implements AccessMatchValue {
25      TRANSPORT_ANY(0),
26      TRANSPORT_PLAIN(1),
27      TRANSPORT_CONFIDENTIAL(2);
28  
29      private int numericValue;
30  
31      private PublicAccessMatchValue(int numericValue) {
32          this.numericValue = numericValue;
33      }
34  
35      @Override
36      public int getNumericValue() {
37          return numericValue;
38      }
39  
40      @Override
41      public boolean isDefaultValue() {
42          return numericValue == TRANSPORT_ANY.numericValue;
43      }
44  
45      @Override
46      public String getTokenType() {
47          return PublicAccessAuthenticationTokenMetaData.TOKEN_TYPE;
48      }
49  
50      @Override
51      public boolean isIssuedByCa() {
52          return false;
53      }
54  
55      @Override
56      public List<AccessMatchType> getAvailableAccessMatchTypes() {
57          return Arrays.asList();
58      }
59      
60      @Override
61      public String normalizeMatchValue(final String value) {
62          return null; // does not have a value
63      }
64  }
