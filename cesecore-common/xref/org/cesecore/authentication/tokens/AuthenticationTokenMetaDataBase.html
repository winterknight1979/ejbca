13  package org.cesecore.authentication.tokens;
14  
15  import java.util.HashMap;
16  import java.util.List;
17  import java.util.Map;
18  import org.cesecore.authorization.user.matchvalues.AccessMatchValue;
19  
20  /**
21   * Common functions for meta data definitions for AuthenticationTokens that
22   * should be auto-detected via ServiceLoader of AuthenticationTokenMetaData.
23   *
24   * @version $Id: AuthenticationTokenMetaDataBase.java 25431 2017-03-09 16:52:15Z
25   *     mikekushner $
26   */
27  public abstract class AuthenticationTokenMetaDataBase
28      implements AuthenticationTokenMetaData {
29  
30    /** Type of token. */
31    private final String tokenType;
32    /** List of matching values. */
33    private final List<? extends AccessMatchValue> accessMatchValues;
34    /** Allow configuration. */
35    private final boolean userConfigurable;
36    /** Map of match IDs. */
37    private final Map<Integer, AccessMatchValue> accessMatchValueIdMap =
38        new HashMap<>();
39    /** Map of matching values. */
40    private final Map<String, AccessMatchValue> accessMatchValueNameMap =
41        new HashMap<>();
42    /** Default match value. */
43    private final AccessMatchValue defaultAccessMatchValue;
44  
45    /**
46     * Constructor.
47     *
48     * @param theTokenType Type of token
49     * @param theAccessMatchValues Values to match
50     * @param isUserConfigurable Allow config
51     */
52    protected AuthenticationTokenMetaDataBase(
53        final String theTokenType,
54        final List<? extends AccessMatchValue> theAccessMatchValues,
55        final boolean isUserConfigurable) {
56      this.tokenType = theTokenType;
57      this.accessMatchValues = theAccessMatchValues;
58      this.userConfigurable = isUserConfigurable;
59      AccessMatchValue theDefaultAccessMatchValue = null;
60      for (final AccessMatchValue accessMatchValue : getAccessMatchValues()) {
61        accessMatchValueIdMap.put(
62            accessMatchValue.getNumericValue(), accessMatchValue);
63        accessMatchValueNameMap.put(accessMatchValue.name(), accessMatchValue);
64        if (theDefaultAccessMatchValue == null
65            || accessMatchValue.isDefaultValue()) {
66          theDefaultAccessMatchValue = accessMatchValue;
67        }
68      }
69      this.defaultAccessMatchValue = theDefaultAccessMatchValue;
70    }
71  
72    @Override
73    public String getTokenType() {
74      return tokenType;
75    }
76  
77    @Override
78    public boolean isUserConfigurable() {
79      return userConfigurable;
80    }
81  
82    @Override
83    public List<? extends AccessMatchValue> getAccessMatchValues() {
84      return accessMatchValues;
85    }
86  
87    @Override
88    public Map<Integer, AccessMatchValue> getAccessMatchValueIdMap() {
89      return accessMatchValueIdMap;
90    }
91  
92    @Override
93    public Map<String, AccessMatchValue> getAccessMatchValueNameMap() {
94      return accessMatchValueNameMap;
95    }
96  
97    @Override
98    public AccessMatchValue getAccessMatchValueDefault() {
99      return defaultAccessMatchValue;
100   }
101 
102   @Override
103   public boolean isSuperToken() {
104     // Legacy pattern: When default value is assigned number
105     // Integer.MAX_VALUE, the AuthenticationToken will grant
106     // any access rule...
107     return getAccessMatchValueDefault().getNumericValue() == Integer.MAX_VALUE;
108   }
109 }
