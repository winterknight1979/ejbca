13  package org.cesecore.authentication.tokens;
14  
15  import java.security.Principal;
16  import java.security.SecureRandom;
17  import java.util.Set;
18  import org.apache.commons.lang.ArrayUtils;
19  import org.apache.log4j.Logger;
20  
21  /**
22   * Common base class for tokens that are only valid in the JVM they are created
23   * and could otherwise be spoofed. E.g. X509 client certificate validation
24   * AuthenticationToken could otherwise be created and sent to a remote EJB
25   * interface.
26   *
27   * @version $Id: LocalJvmOnlyAuthenticationToken.java 17625 2013-09-20 07:12:06Z
28   *     netmackan $
29   */
30  public abstract class LocalJvmOnlyAuthenticationToken
31      extends AuthenticationToken {
32  
33    private static final long serialVersionUID = -6830176240864231535L;
34  
35    /** Logger. */
36    private static final Logger LOG =
37        Logger.getLogger(LocalJvmOnlyAuthenticationToken.class);
38  
39    /**
40     * A random token that is unique to this JVM (e.g. the application server JVM
41     * and a CLI JVM does not have the same token).
42     */
43    private static final byte[] RANDOM_TOKEN = createRandomToken();
44    /** Size of token. */
45    private static final int TOKEN_SIZE = 32;
46  
47    /** transient authToken should NOT be serialized. * */
48    private transient byte[] authToken;
49  
50    /**
51     * @param principals Principals
52     * @param credentials Credentials
53     * @see
54     *     org.cesecore.authentication.tokens.AuthenticationToken#AuthenticationToken
55     */
56    protected LocalJvmOnlyAuthenticationToken(
57        final Set<? extends Principal> principals, final Set<?> credentials) {
58      super(principals, credentials);
59      authToken = RANDOM_TOKEN;
60    }
61  
62    /** @return true if this */
63    protected final boolean isCreatedInThisJvm() {
64      boolean isCreatedInThisJvm = ArrayUtils.isEquals(authToken, RANDOM_TOKEN);
65      if (LOG.isTraceEnabled()) {
66        LOG.trace("isCreatedInThisJvm: " + isCreatedInThisJvm);
67      }
68      return isCreatedInThisJvm;
69    }
70  
71    /** Initialize rhe random token. */
72    public void initRandomToken() {
73      authToken = RANDOM_TOKEN;
74    }
75  
76    /** @return a 32-byte random token. */
77    private static byte[] createRandomToken() {
78      final byte[] token = new byte[TOKEN_SIZE];
79      new SecureRandom().nextBytes(token);
80      return token;
81    }
82  }
