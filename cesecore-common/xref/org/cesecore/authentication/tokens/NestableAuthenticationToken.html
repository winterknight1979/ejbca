13  package org.cesecore.authentication.tokens;
14  
15  import java.security.Principal;
16  import java.util.ArrayList;
17  import java.util.List;
18  import java.util.Set;
19  
20  /**
21   * A NestableAuthenticationToken represents an AuthenticationToken where the
22   * original credentials has passed through multiple step of authentication.
23   *
24   * @version $Id: NestableAuthenticationToken.java 25797 2017-05-04 15:52:00Z
25   *     jeklund $
26   */
27  public abstract class NestableAuthenticationToken
28      extends LocalJvmOnlyAuthenticationToken {
29  
30    private static final long serialVersionUID = 1L;
31    /** Maximum nesting depth. */
32    private static final int MAX_NESTING = 10;
33  
34    /** Internal token. */
35    private NestableAuthenticationToken nestedAuthenticationToken = null;
36  
37    protected NestableAuthenticationToken(
38        final Set<? extends Principal> principals, final Set<?> credentials) {
39      super(principals, credentials);
40    }
41  
42    /**
43     * @return a List of all nested AuthenticationTokens (excluding the object
44     *     itself) or an empty List, never null.
45     */
46    public List<NestableAuthenticationToken> getNestedAuthenticationTokens() {
47      final List<NestableAuthenticationToken> theNestedAuthenticationTokens =
48          new ArrayList<>();
49      NestableAuthenticationToken current = this.nestedAuthenticationToken;
50      while (current != null) {
51        theNestedAuthenticationTokens.add(current);
52        // Perform a small sanity check to protect against loops or
53        // massive nesting in order to exhaust server mem
54        if (theNestedAuthenticationTokens.size() > MAX_NESTING) {
55          throw new IllegalStateException(
56              "Hard coded limit of number "
57                  + "of nested AuthenticationTokens reached.");
58        }
59        current = current.nestedAuthenticationToken;
60      }
61      return theNestedAuthenticationTokens;
62    }
63  
64    /**
65     * Append a token.
66     *
67     * @param aNestedAuthenticationToken Token to append.
68     */
69    public void appendNestedAuthenticationToken(
70        final NestableAuthenticationToken aNestedAuthenticationToken) {
71      NestableAuthenticationToken current = this;
72      int count = 0;
73      while (true) {
74        // If the nested token we are appending was created locally,
75        // we consider all to be created locally as well
76        if (aNestedAuthenticationToken.isCreatedInThisJvm()
77            && !current.isCreatedInThisJvm()) {
78          current.initRandomToken();
79        }
80        if (current.nestedAuthenticationToken == null) {
81          current.nestedAuthenticationToken = aNestedAuthenticationToken;
82          break;
83        }
84        // Perform a small sanity check to protect against loops or
85        // massive nesting in order to exhaust server mem
86        if (count++ > MAX_NESTING) {
87          throw new IllegalStateException(
88              "Hard coded limit of number of"
89                  + " nested AuthenticationTokens reached.");
90        }
91        current = current.nestedAuthenticationToken;
92      }
93    }
94  
95    @Override
96    public void initRandomToken() {
97      super.initRandomToken();
98      if (nestedAuthenticationToken != null) {
99        nestedAuthenticationToken.initRandomToken();
100     }
101   }
102 
103   /** Returns information of the entity this authentication token belongs to. */
104   @Override
105   public String toString() {
106     String ret = toStringOverride();
107     if (ret == null) {
108       ret = super.toString();
109     }
110     if (nestedAuthenticationToken != null) {
111       ret += " [via] " + nestedAuthenticationToken.toString();
112     }
113     return ret;
114   }
115 
116   /**
117    * Override and return anything but null to use this value instead of {@link
118    * AuthenticationToken#toString()}.
119    *
120    * @return null by default
121    */
122   protected String toStringOverride() {
123     return null;
124   }
125 
126   @Override
127   protected String generateUniqueId() {
128     return nestedAuthenticationToken == null
129         ? null
130         : nestedAuthenticationToken.getMetaData().getTokenType()
131             + ";"
132             + nestedAuthenticationToken.getUniqueId();
133   }
134 }
