42  /**
43   * Wrapper of a Java KeyStore to speed up key operations by caching key references.
44   * 
45   * Aliases and certificates are read when the class is initiated from the underlying key store.
46   * PrivateKey and SecretKey objects are read and cached when the key specific protection is available (when used).
47   * 
48   * @version $Id: CachingKeyStoreWrapper.java 22480 2015-12-17 21:23:11Z primelars $
49   */
50  public class CachingKeyStoreWrapper {
51      /** Similar to Java's KeyStore.Entry */
52      private static final Logger log = Logger.getLogger(CachingKeyStoreWrapper.class);
53      private final ReentrantLock updateLock = new ReentrantLock(false);
54      private final KeyStore keyStore;
55      private final KeyStoreCache keyStoreCache;
56      private class KeyStoreMapEntry {
57          public final Key key;
58          public final Certificate[] certificateChain;
59          public final boolean isTrusted;
60          public KeyStoreMapEntry(final String alias, final KeyStore keyStore) throws KeyStoreException {
61              if (keyStore.isCertificateEntry(alias)) {
62                  // See if there is a TrustedCertificateEntry instead
63                  final Certificate certificate = keyStore.getCertificate(alias);
64                  this.certificateChain = new Certificate[] { certificate };
65                  this.key = null;
66                  this.isTrusted = true;
67                  return;
68              }
69              this.isTrusted = false;
70              this.certificateChain = keyStore.getCertificateChain(alias);
71              Key tmpKey;
72              try {
73                  tmpKey = keyStore.getKey(alias, null);
74              } catch (KeyStoreException e) {
75                  throw e;
76              } catch (Exception e) {
77                  tmpKey = null;
78              }
79              this.key = tmpKey;
80          }
81          public KeyStoreMapEntry(
82                  final String alias, final KeyStore keyStore, final char password[],
83                  final KeyStoreMapEntry oldEntry)
84                          throws UnrecoverableKeyException, KeyStoreException, NoSuchAlgorithmException {
85              assert !oldEntry.isTrusted;
86              this.isTrusted = false;
87              this.certificateChain = oldEntry.certificateChain;
88              this.key = keyStore.getKey(alias, password);
89          }
90          public KeyStoreMapEntry( final Certificate certificate ) {
91              this.key = null;
92              this.isTrusted = true;
93              this.certificateChain = new Certificate[] { certificate };
94          }
95          public KeyStoreMapEntry( final Certificate[] chain, final Key k ) {
96              this.key = k;
97              this.isTrusted = false;
98              this.certificateChain = chain;
99          }
100         public KeyStoreMapEntry(
101                 final String alias, final ProtectionParameter protection,
102                 final KeyStore keyStore )
103                         throws NoSuchAlgorithmException, UnrecoverableEntryException, KeyStoreException {
104             this(keyStore.getEntry(alias, protection));
105         }
106         public KeyStoreMapEntry( final Entry entry ) {
107             if ( entry instanceof PrivateKeyEntry ) {
108                 this.key = ((PrivateKeyEntry) entry).getPrivateKey();
109                 this.certificateChain = ((PrivateKeyEntry) entry).getCertificateChain();
110                 this.isTrusted = false;
111                 return;
112             }
113             if ( entry instanceof SecretKeyEntry ) {
114                 this.key =((SecretKeyEntry) entry).getSecretKey();
115                 this.certificateChain = null;
116                 this.isTrusted = false;
117                 return;
118             }
119             if ( entry instanceof TrustedCertificateEntry ) {
120                 this.key = null;
121                 this.certificateChain = new Certificate[] { ((TrustedCertificateEntry) entry).getTrustedCertificate() };
122                 this.isTrusted = true;
123                 return;
124             }
125             throw new Error("It should not be possible to reach this point!");
126         }
127         public Entry getEntry() {
128             if ( this.isTrusted  ) {
129                 assert this.certificateChain!=null;
130                 // No constructor puts more than one certificate in the chain when trusted.
131                 assert this.certificateChain.length==1;
132                 return new TrustedCertificateEntry(this.certificateChain[0]);
133             }
134             assert this.key!=null;
135             if ( this.certificateChain!=null ) {
136                 return new PrivateKeyEntry((PrivateKey) this.key, this.certificateChain);
137             }
138             return new SecretKeyEntry((SecretKey) this.key);
139         }
140     }

142     private class KeyStoreCache {
143         private HashMap<String, KeyStoreMapEntry> cache;
144 
145         @SuppressWarnings("synthetic-access")
146         public KeyStoreCache(final KeyStore keyStore) throws KeyStoreException {
147             this.cache = new HashMap<>();
148             // Load the whole public KeyStore content (aliases and certificate) into the cache
149             final Enumeration<String> aliases = keyStore.aliases();
150             while (aliases.hasMoreElements()) {
151                 final String alias = aliases.nextElement();
152                 this.cache.put(fixBadUTF8(alias), new KeyStoreMapEntry(alias, keyStore));
153                 if (log.isDebugEnabled()) {
154                     log.debug("KeyStore has alias: " + alias);
155                 }
156             }
157         }
158         public void addEntry(final String alias, final  KeyStoreMapEntry newEntry) {
159             final HashMap<String, KeyStoreMapEntry> clone = new HashMap<>(this.cache);
160             clone.put(alias, newEntry);
161             this.cache = clone;
162         }
163         public void removeEntry(final String alias) {
164             final HashMap<String, KeyStoreMapEntry> clone = new HashMap<>(this.cache);
165             clone.remove(alias);
166             this.cache = clone;
167         }
168         public KeyStoreMapEntry get(final String alias) {
169             return this.cache.get(alias);
170         }
171         public Enumeration<String> getAliases() {
172             return new Vector<>(this.cache.keySet()).elements();
173         }
174     }

176     private static boolean isSunP11( final KeyStore keyStore ) {
177         return keyStore.getProvider().getName().indexOf("SunPKCS11")==0;
178     }
179 
180     /**
181      * The Sun p11 implementation of the {@link KeyStore} returns aliases that
182      * are badly encoded. This method fix this encoding.
183      * @param orig a badly encoded alias.
184      * @return a correct encoded alias.
185      */
186     private String fixBadUTF8(final String orig) {
187         if ( !isSunP11(this.keyStore) ) {
188             return orig;
189         }
190         try {
191             final byte bvIn[] = orig.getBytes("UTF-16BE");
192             final byte bvOut[] = new byte[bvIn.length/2];
193             for ( int i=1; i<bvIn.length; i += 2) {
194                 bvOut[i/2] = (byte)(bvIn[i]&0xff);
195             }
196             return new String(bvOut, "UTF-8");
197         } catch (UnsupportedEncodingException e) {
198             throw new IllegalStateException("UTF-16BE and UTF-8 must be implemented for all JREs.");
199         }
200     }
201 
202     /**
203      * When deleting a key in the sun p11 implementation of {@link KeyStore} the
204      * alias {@link String} must be altered due to an implementation bug.
205      * @param orig the alias
206      * @return modified alias to suit the sun p11.
207      */
208     private String makeBadUTF8(final String orig) {
209         if ( !isSunP11(this.keyStore) ) {
210             return orig;
211         }
212         try {
213             final byte bvIn[] = orig.getBytes("UTF-8");
214             final byte bvOut[] = new byte[bvIn.length*2];
215             for ( int i=0; i<bvIn.length; i += 1) {
216                 bvOut[i*2] = 0;
217                 bvOut[i*2+1] = (byte)(bvIn[i]&0xff);
218             }
219             return new String(bvOut, "UTF-16BE");
220         } catch (UnsupportedEncodingException e) {
221             throw new IllegalStateException("UTF-16BE and UTF-8 must be implemented for all JREs.");
222         }
223     }

228     /**
229      * Wrap the key store object with optional caching of all entries.
230      * 
231      * @param keyStore the key store to wrap
232      * @param cachingEnabled true will cache a list of all aliases, certificates and lazily cache private and secret keys when accessed
233      * @throws KeyStoreException if the underlying key store cannot be accessed
234      */
235     public CachingKeyStoreWrapper(final KeyStore keyStore, final boolean cachingEnabled) throws KeyStoreException {
236         this.keyStore = keyStore;
237         if (log.isDebugEnabled()) {
238             log.debug("cachingEnabled: " + cachingEnabled);
239         }
240         if (cachingEnabled) {
241             this.keyStoreCache = new KeyStoreCache(keyStore);
242         } else {
243             this.keyStoreCache = null;
244         }
245     }

247     /** @param alias alias
248      * @return certificate
249      * @throws KeyStoreException on fail 
250      * @see java.security.KeyStore#getCertificate(String) */
251     public Certificate getCertificate(final String alias) throws KeyStoreException {
252         if (this.keyStoreCache==null) {
253             return this.keyStore.getCertificate(alias);
254         }
255         final KeyStoreMapEntry keyStoreMapEntry = this.keyStoreCache.get(alias);
256         if (keyStoreMapEntry==null) {
257             return null;
258         }
259         if (keyStoreMapEntry.certificateChain==null || keyStoreMapEntry.certificateChain.length==0) {
260             return null;
261         }
262         return keyStoreMapEntry.certificateChain[0];
263     }

265     /** @param alias alias
266      * @param certificate certificate
267      * @throws KeyStoreException on fail
268      * @see java.security.KeyStore#setCertificateEntry(String, Certificate) */
269     public void setCertificateEntry(final String alias, final Certificate certificate) throws KeyStoreException {
270         // Update the TrustedCertificateEntry in the real key store
271         this.keyStore.setCertificateEntry(alias, certificate);
272         if (this.keyStoreCache==null) {
273             return;
274         }
275         this.updateLock.lock();
276         try {
277             this.keyStoreCache.addEntry(alias, new KeyStoreMapEntry(certificate));
278         } finally {
279             this.updateLock.unlock();
280         }
281         if (log.isDebugEnabled()) {
282             log.debug("Updated certificate entry in cache for alias: " + alias);
283         }
284     }

307     /** @param alias alias
308      * @param key key
309      * @param password password 
310      * @param chain certs
311      * @throws KeyStoreException on failure 
312      * @see java.security.KeyStore#setKeyEntry(String, Key, char[], Certificate[]) */
313     public void setKeyEntry(final String alias, final Key key, final char[] password, final Certificate[] chain) throws KeyStoreException {
314         this.keyStore.setKeyEntry(alias, key, password, chain);
315         if (this.keyStoreCache==null) {
316             return;
317         }
318         this.updateLock.lock();
319         try {
320             final KeyStoreMapEntry keyStoreMapEntry = new KeyStoreMapEntry(chain,key);
321             this.keyStoreCache.addEntry(alias, keyStoreMapEntry);
322         } finally {
323             this.updateLock.unlock();
324         }
325         if (log.isDebugEnabled()) {
326             log.debug("Updated key entry in cache for alias: " + alias);
327         }
328     }

330     /** @param alias alias
331      * @throws KeyStoreException on failure
332      * @see java.security.KeyStore#deleteEntry(String) */
333     public void deleteEntry(final String alias) throws KeyStoreException {
334         this.keyStore.deleteEntry(makeBadUTF8(alias));
335         if (this.keyStoreCache==null) {
336             return;
337         }
338         this.updateLock.lock();
339         try {
340             this.keyStoreCache.removeEntry(alias);
341         } finally {
342             this.updateLock.unlock();
343         }
344         if (log.isDebugEnabled()) {
345             log.debug("Removed entry from cache for alias: " + alias);
346         }
347     }

349     /** @param alias alias
350      * @param protParam params
351      * @return entry
352      * @throws NoSuchAlgorithmException if algo not found 
353      * @throws UnrecoverableEntryException if corrupt
354      * @throws KeyStoreException on failure
355      * @see java.security.KeyStore#getEntry(String, ProtectionParameter) */
356     public Entry getEntry(final String alias, final ProtectionParameter protParam) throws NoSuchAlgorithmException, UnrecoverableEntryException, KeyStoreException {
357         if (this.keyStoreCache==null) {
358             return this.keyStore.getEntry(alias, protParam);
359         }
360         {
361             final KeyStoreMapEntry keyStoreMapEntry = this.keyStoreCache.get(alias);
362             if (keyStoreMapEntry==null) {
363                 // If the alias exists it has a KeyStoreMapEntry. No need to query the key store here.
364                 return null;
365             }
366             if ( keyStoreMapEntry.isTrusted || keyStoreMapEntry.key!=null ) {
367                 return keyStoreMapEntry.getEntry();
368             }
369         }
370         this.updateLock.lock();
371         try {
372             // Check if another thread has retrieved the key while we waited for the lock
373             final KeyStoreMapEntry afterWaitEntry = this.keyStoreCache.get(alias);
374             if ( afterWaitEntry.isTrusted || afterWaitEntry.key!=null ) {
375                 return afterWaitEntry.getEntry();
376             }
377             final KeyStoreMapEntry newEntry = new KeyStoreMapEntry(alias, protParam, this.keyStore);
378             this.keyStoreCache.addEntry(alias, newEntry);
379             return newEntry.getEntry();
380         } finally {
381             this.updateLock.unlock();
382         }
383     }

410     /** @param alias alias
411      * @param password password
412      * @return key
413      * @throws UnrecoverableKeyException if key is unrecoverable 
414      * @throws KeyStoreException on failure
415      * @throws NoSuchAlgorithmException if algo not found
416      * @see java.security.KeyStore#getKey(String, char[]) */
417     public Key getKey(final String alias, final char[] password) throws UnrecoverableKeyException, KeyStoreException, NoSuchAlgorithmException {
418         if (this.keyStoreCache==null) {
419             return this.keyStore.getKey(alias, password);
420         }
421         {
422             final KeyStoreMapEntry keyStoreMapEntry = this.keyStoreCache.get(alias);
423             if (keyStoreMapEntry==null) {
424                 // If the alias exists it has a KeyStoreMapEntry. No need to query the key store here.
425                 return null;
426             }
427             if ( keyStoreMapEntry.isTrusted || keyStoreMapEntry.key!=null ) {
428                 // There is no key if it is a trusted Entry.
429                 return keyStoreMapEntry.key;
430             }
431         }
432         this.updateLock.lock();
433         try {
434             // Check if another thread has retrieved the key while we waited for the lock
435             final KeyStoreMapEntry entryAfterWait = this.keyStoreCache.get(alias);
436             if ( entryAfterWait.isTrusted || entryAfterWait.key!=null ) {
437                 return entryAfterWait.key;
438             }
439             final KeyStoreMapEntry newEntry = new KeyStoreMapEntry(alias, this.keyStore, password, entryAfterWait);
440             this.keyStoreCache.addEntry(alias, newEntry);
441             if (log.isDebugEnabled()) {
442                 log.debug("Caching key for alias: " + alias);
443             }
444             return newEntry.key;
445         } finally {
446             this.updateLock.unlock();
447         }
448     }

450     /** @param alias alias
451      * @return boolean
452      * @throws KeyStoreException on failure
453      * @see java.security.KeyStore#isKeyEntry(String) */
454     public boolean isKeyEntry(final String alias ) throws KeyStoreException {
455         if (this.keyStoreCache==null) {
456             return this.keyStore.isKeyEntry(alias);
457         }
458         final KeyStoreMapEntry keyStoreMapEntry = this.keyStoreCache.get(alias);
459         return keyStoreMapEntry!=null && !keyStoreMapEntry.isTrusted;
460     }

462     /** @param alias alias
463      * @return certificate
464      * @throws KeyStoreException on fail
465      * @see java.security.KeyStore#getCertificateChain(String) */
466     public Certificate[] getCertificateChain(String alias) throws KeyStoreException {
467         if (this.keyStoreCache==null) {
468             return this.keyStore.getCertificateChain(alias);
469         }
470         final KeyStoreMapEntry entry = this.keyStoreCache.get(alias);
471         return entry!=null ? entry.certificateChain : null;
