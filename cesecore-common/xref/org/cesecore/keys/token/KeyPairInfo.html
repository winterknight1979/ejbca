13  package org.cesecore.keys.token;
14  
15  import java.io.Serializable;
16  
17  /**
18   * Representation of a KeyPair in a CryptoToken. Does not contain the actual keys.
19   * 
20   * @version $Id: KeyPairInfo.java 24844 2016-12-06 15:18:05Z samuellb $
21   */
22  public class KeyPairInfo implements Serializable, Comparable<KeyPairInfo> {
23  
24      private static final long serialVersionUID = 1L;
25  
26      private String alias = "";
27      private String keyAlgorithm;
28      private String keySpecification;
29      private String subjectKeyID = "";
30  
31      public KeyPairInfo(final String alias, final String keyAlgorithm, final String keySpecification, final String subjectKeyID) {
32          this.alias = alias;
33          this.keyAlgorithm = keyAlgorithm;
34          this.keySpecification = keySpecification;
35          this.subjectKeyID = subjectKeyID;
36      }
37  
38      public String getAlias() { return alias; }
39      public void setAlias(String alias) { this.alias = alias; }
40      public String getKeyAlgorithm() { return keyAlgorithm; }
41      public void setKeyAlgorithm(String keyAlgorithm) { this.keyAlgorithm = keyAlgorithm; }
42      public String getKeySpecification() { return keySpecification; }
43      public void setKeySpecification(String keySpecification) { this.keySpecification = keySpecification; }
44      public String getSubjectKeyID() { return subjectKeyID; }
45      public void setSubjectKeyID(String subjectKeyID) { this.subjectKeyID = subjectKeyID; }
46  
47      @Override
48      public int compareTo(final KeyPairInfo o) {
49          int c;
50          
51          c = alias.compareTo(o.alias);
52          if (c != 0) { return c; }
53          
54          // There shouldn't be multiple aliases with the same name, but we compare the other fields just to be sure.
55          c = keyAlgorithm.compareTo(o.keyAlgorithm);
56          if (c != 0) { return c; }
57          c = keySpecification.compareTo(o.keySpecification);
58          if (c != 0) { return c; }
59          
60          return 0;
61      }
62  }
