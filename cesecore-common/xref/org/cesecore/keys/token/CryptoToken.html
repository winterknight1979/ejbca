13  package org.cesecore.keys.token;
14  
15  import java.io.IOException;
16  import java.io.Serializable;
17  import java.security.InvalidAlgorithmParameterException;
18  import java.security.InvalidKeyException;
19  import java.security.Key;
20  import java.security.KeyStoreException;
21  import java.security.NoSuchAlgorithmException;
22  import java.security.NoSuchProviderException;
23  import java.security.PrivateKey;
24  import java.security.PublicKey;
25  import java.security.SignatureException;
26  import java.security.cert.Certificate;
27  import java.security.cert.CertificateException;
28  import java.security.spec.AlgorithmParameterSpec;
29  import java.util.List;
30  import java.util.Properties;
31  import javax.crypto.IllegalBlockSizeException;
32  import javax.crypto.NoSuchPaddingException;
33  
34  /**
35   * Handles operations of the device producing signatures and handling the
36   * private key. All Crypto Token plug-ins must implement this interface.
37   *
38   * @version $Id: CryptoToken.java 22596 2016-01-18 14:59:25Z mikekushner $
39   */
40  public interface CryptoToken extends Serializable {
41    /** Status. */
42    int STATUS_ACTIVE = 1;
43    /** Status. */
44    int STATUS_OFFLINE = 2;
45  
46    /** Auto activation property that can be
47     * defined in Crypto token properties. */
48    String AUTOACTIVATE_PIN_PROPERTY = "pin";
49    /** Boolean indicating if it should be allowed to extract private keys. */
50    String ALLOW_EXTRACTABLE_PRIVATE_KEY =
51        "allow.extractable.privatekey";
52    /**
53     * "|"-separated string of "keyalias;keyspec" pairs that act as placeholders
54     * for keyaliases that can be generated later.
55     */
56    String KEYPLACEHOLDERS_PROPERTY = "statedump.keytemplates";
57   /** sep.*/
58    String KEYPLACEHOLDERS_OUTER_SEPARATOR = "|";
59    /** sep. */
60    String KEYPLACEHOLDERS_INNER_SEPARATOR = ";";
61    /**
62     * Boolean indicating if explicit ECC parameters should be used, instead of
63     * named curves which is standard. Explicit parameters are only used by ICAO
64     * CSCA and DS certificates as defined in ICAO 9303.
65     */
66    String EXPLICIT_ECC_PUBLICKEY_PARAMETERS =
67        "explicit.ecc.publickey.parameters";
68    /** */
69    String TOKENNAME_PROPERTY = "tokenName";
70    /**
71     * Whether non-existent slots should be allowed or throw an
72     * NoSuchSlotException.
73     */
74    String ALLOW_NONEXISTING_SLOT_PROPERTY =
75        "allow.nonexisting.slot";
76  
77    /**
78     * Method called after creation of instance. Gives the object it's properties.
79     *
80     * <p>Info contains Crypto Token properties, as entered for all tokens.
81     * Properties can create properties used by extending or wrapping classes so
82     * the implementing class should not delete or remove properties that are
83     * unknown to it, but keep them transparently. Data contains Crypto Token
84     * data, can be null for tokens that don't need it.
85     *
86     * @param properties Properties info used to create token, new token or init
87     *     existing token
88     * @param data byte[] data as created internally for tokens, can be null for
89     *     tokens that don't need it
90     * @param id unique ID of the user of the token, the id is user defined and
91     *     not used internally for anything but logging.
92     * @throws Exception on fail
93     */
94    void init(Properties properties, byte[] data, int id) throws Exception;
95  
96    /**
97     * Gets the id that was passed as parameter to init.
98     *
99     * @return id
100    */
101   int getId();
102 
103   /**
104    * Method used to activate Crypto Tokens when connected after being offline.
105    *
106    * @param authenticationcode used to unlock crypto token, i.e PIN for
107    *     smartcard HSMs. This parameter might be ignored, e.g. for
108    *     auto-activated soft tokens.
109    * @throws CryptoTokenOfflineException if Crypto Token is not available or
110    *     connected.
111    * @throws CryptoTokenAuthenticationFailedException with error message if
112    *     authentication to tokens fail.
113    */
114   void activate(char[] authenticationcode)
115       throws CryptoTokenOfflineException,
116           CryptoTokenAuthenticationFailedException;
117 
118   /**
119    * Method used to deactivate tokens. Used to set a Crypto Token to offline
120    * status and to reset the HSMs authorization code.
121    */
122   void deactivate();
123 
124   /**
125    * Checks if an alias is in use by a Crypto Token.
126    *
127    * @param alias the alias that we want to check
128    * @return true if there is a private, public or symmetric entry with this
129    *     alias in the CryptoToken
130    */
131   boolean isAliasUsed(String alias);
132 
133   /**
134    * Returns the private key (if possible) of token. Note that this method does
135    * NOT check if the key is permitted to be extracted.
136    *
137    * @param alias the key alias to retrieve from the token
138    * @throws CryptoTokenOfflineException if Crypto Token is not available or
139    *     connected, or key with alias does not exist.
140    * @return PrivateKey object
141    */
142   PrivateKey getPrivateKey(String alias) throws CryptoTokenOfflineException;
143 
144   /**
145    * Returns the public key (if possible) of token.
146    *
147    * @param alias the key alias to retrieve from the token
148    * @throws CryptoTokenOfflineException if Crypto Token is not available or
149    *     connected
150    * @return the public key, or null if key with the alias does not exist.
151    */
152   PublicKey getPublicKey(String alias) throws CryptoTokenOfflineException;
153 
154   /**
155    * Returns the key (if possible) of token, used for symmetric keys. For
156    * assymmetric keys getPrivateKye and getPublic key is recommended..
157    *
158    * @param alias the key alias to retrieve from the token
159    * @throws CryptoTokenOfflineException if Crypto Token is not available or
160    *     connected, or key with alias does not exist.
161    * @return Key object
162    */
163   Key getKey(String alias) throws CryptoTokenOfflineException;
164 
165   /**
166    * Deletes an entry in the crypto token.
167    *
168    * @param alias is a reference to the entry in the token that should be
169    *     deleted.
170    * @throws KeyStoreException on fail
171    * @throws IOException on IO error
172    * @throws CertificateException on invalid cert
173    * @throws NoSuchAlgorithmException if algorithm not found
174    * @throws CryptoTokenOfflineException if offline
175    */
176   void deleteEntry(String alias)
177       throws KeyStoreException, NoSuchAlgorithmException, CertificateException,
178           IOException, CryptoTokenOfflineException;
179 
180   /**
181    * Generates a key pair (asymmetric keys) in the crypto token.
182    *
183    * @param keySpec all decimal digits RSA key length, otherwise name of ECC
184    *     curve or DSA key using syntax DSAnnnn
185    * @param alias the name of the key pair in the crypto token
186    * @throws InvalidAlgorithmParameterException if params are invalid
187    * @throws CryptoTokenOfflineException if offline
188    */
189   void generateKeyPair(String keySpec, String alias)
190       throws InvalidAlgorithmParameterException, CryptoTokenOfflineException;
191 
192   /**
193    * Generates a key pair (asymmetric keys) in the crypto token. This method is
194    * used when you have an existing PublicKey and want to generate a key of the
195    * same type. You can use KeyTools to get the AlgorithmParameterSpec from an
196    * existing PublicKey.
197    *
198    * <pre>
199    * AlgorithmParameterSpec spec = KeyTools.getKeyGenSpec(templatePublicKey);
200    * </pre>
201    *
202    * @param spec AlgorithmParameterSpec describing the key pair to be generated
203    * @param alias the name of the key pair in the crypto token
204    * @throws InvalidAlgorithmParameterException On invalid parameters
205    * @throws CertificateException On invalid certificate
206    * @throws IOException On IO error
207    * @throws CryptoTokenOfflineException if offline
208    */
209   void generateKeyPair(AlgorithmParameterSpec spec, String alias)
210       throws InvalidAlgorithmParameterException, CertificateException,
211           IOException, CryptoTokenOfflineException;
212 
213   /**
214    * Generates a symmetric key.
215    *
216    * @param algorithm symmetric algorithm specified in
217    *     http://download.oracle.com/javase/1.5.0/docs/api/index.html, suggest
218    *     AES, DESede or DES
219    * @param keysize keysize of symmetric key, suggest 128 or 256 for AES, 64 for
220    *     168 for DESede and 64 for DES
221    * @param alias the alias the key will get in the keystore
222    * @throws NoSuchProviderException If provider not found
223    * @throws NoSuchAlgorithmException of algo not found
224    * @throws KeyStoreException If store fails
225    * @throws CryptoTokenOfflineException if offline
226    * @throws InvalidKeyException if key is invalid
227    * @throws InvalidAlgorithmParameterException if parameters are invalid
228    * @throws SignatureException if signature is invalid
229    * @throws CertificateException if cert is invalid
230    * @throws IOException on IO fail
231    * @throws NoSuchPaddingException if padding is invalid
232    * @throws IllegalBlockSizeException if block size is incorrect
233    */
234   void generateKey(
235       String algorithm, int keysize, String alias)
236       throws NoSuchAlgorithmException, NoSuchProviderException,
237           KeyStoreException, CryptoTokenOfflineException, InvalidKeyException,
238           InvalidAlgorithmParameterException, SignatureException,
239           CertificateException, IOException, NoSuchPaddingException,
240           IllegalBlockSizeException;
241 
242   /**
243    * Returns the signature Provider that should be used to sign things with the
244    * PrivateKey object returned by this signing device implementation.
245    *
246    * @return String the name of the Provider
247    */
248   String getSignProviderName();
249 
250   /**
251    * Returns the crypto Provider that should be used to encrypt/decrypt things
252    * with the PrivateKey object returned by this signing device implementation.
253    * In most cases this is the same as the signature provider.
254    *
255    * @return String the name of the Provider
256    */
257   String getEncProviderName();
258 
259   /**
260    * Resets token. Might cure HW failures. It is up to each implementation to
261    * implement this or not.
262    */
263   void reset();
264 
265   /** @return user friendly identifier */
266   String getTokenName();
267   /**
268    * Set user friendly identifier.
269    *
270    * @param tokenName name
271    */
272   void setTokenName(String tokenName);
273 
274   /**
275    * Method that returns the current status of the crypto token.
276    *
277    * @return one of the CryptoToken.STATUS_.. values
278    */
279   int getTokenStatus();
280 
281   /**
282    * Return token properties, can be the same as passed to init, or an updated
283    * set.
284    *
285    * @return Properties can be empty but should never be null
286    */
287   Properties getProperties();
288 
289   /**
290    * Updates dynamic properties for the crypto token. Call this method when a
291    * new key string, autoactivation PIN has been set and the init method on the
292    * crypto token is not called. Does not update properties that is only used
293    * when token is created, for example P11 slot, this is only updated on
294    * recreation of the token.
295    *
296    * @param properties Properties containing the new key properties or other
297    *     properties, such as activation PIN
298    */
299   void setProperties(Properties properties);
300 
301   /**
302    * Stores a new key in this crypto token's keystore.
303    *
304    * @param alias The alias for the key
305    * @param key The key to be stored
306    * @param chain The associated key chain
307    * @param password Password to this slot.
308    * @throws KeyStoreException if keystore for this crypto token has not been
309    *     initialized
310    */
311   void storeKey(String alias, Key key, Certificate[] chain, char[] password)
312       throws KeyStoreException;
313 
314   /**
315    * Stores keystore data (if any) to be used when initializing a new (existing)
316    * token with the init method.
317    *
318    * @return byte[] with keystore data, can be null if not needed for
319    *     initialization
320    */
321   byte[] getTokenData();
322 
323   /**
324    * Testing a keypair to see that it is usable.
325    *
326    * @param alias the alias of the key pair to test
327    * @throws InvalidKeyException if the public key can not be used to verify a
328    *     string signed by the private key, because the key is wrong or the
329    *     signature operation fails for other reasons such as a
330    *     NoSuchAlgorithmException or SignatureException.
331    * @throws CryptoTokenOfflineException if the crypto token is offline
332    */
333   void testKeyPair(String alias)
334       throws InvalidKeyException,
335           CryptoTokenOfflineException; // NOPMD:this is not a junit test
336 
337   /**
338    * Testing a keypair to see that it is usable.
339    *
340    * @param alias the alias of the key pair to test
341    * @param publicKey the public key of the key pair
342    * @param privateKey the private key of the key pair
343    * @throws InvalidKeyException if the public key can not be used to verify a
344    *     string signed by the private key, because the key is wrong or the
345    *     signature operation fails for other reasons such as a
346    *     NoSuchAlgorithmException or SignatureException.
347    */
348   void testKeyPair(String alias, PublicKey publicKey, PrivateKey privateKey)
349       throws InvalidKeyException;
350 
351   /**
352    * Checks if generated private keys are allowed to be extracted.
353    *
354    * @return false if the private key is not extractable.
355    */
356   boolean doPermitExtractablePrivateKey();
357 
358   /**
359    * @return an enumeration of aliases from this token's key store.
360    * @throws KeyStoreException if the keystore has not been initialized
361    *     (loaded).
362    * @throws CryptoTokenOfflineException if the keystore has not been
363    *     initialized (loaded).
364    */
365   List<String> getAliases()
366       throws KeyStoreException, CryptoTokenOfflineException;
367 
368   /** @return true if there is an auto activation PIN stored in the token */
369   boolean isAutoActivationPinPresent();
370 }
