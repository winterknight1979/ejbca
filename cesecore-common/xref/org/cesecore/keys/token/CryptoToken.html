16  import java.io.Serializable;
17  import java.security.InvalidAlgorithmParameterException;
18  import java.security.InvalidKeyException;
19  import java.security.Key;
20  import java.security.KeyStoreException;
21  import java.security.NoSuchAlgorithmException;
22  import java.security.NoSuchProviderException;
23  import java.security.PrivateKey;
24  import java.security.PublicKey;
25  import java.security.SignatureException;
26  import java.security.cert.Certificate;
27  import java.security.cert.CertificateException;
28  import java.security.spec.AlgorithmParameterSpec;
29  import java.util.List;
30  import java.util.Properties;
31  
32  import javax.crypto.IllegalBlockSizeException;
33  import javax.crypto.NoSuchPaddingException;

36  /** Handles operations of the device producing signatures and handling the private key.
37   *  All Crypto Token plug-ins must implement this interface.
38   *
39   * @version $Id: CryptoToken.java 22596 2016-01-18 14:59:25Z mikekushner $
40   */
41  public interface CryptoToken extends Serializable {
42  
43      static final int STATUS_ACTIVE  = 1;
44      static final int STATUS_OFFLINE = 2;
45  
46      /** Auto activation property that can be defined in Crypto token properties */
47      static final String AUTOACTIVATE_PIN_PROPERTY = "pin";
48      /** Boolean indicating if it should be allowed to extract private keys */
49      static final String ALLOW_EXTRACTABLE_PRIVATE_KEY = "allow.extractable.privatekey";
50      /** "|"-separated string of "keyalias;keyspec" pairs that act as placeholders for keyaliases that can be generated later */
51      static final String KEYPLACEHOLDERS_PROPERTY = "statedump.keytemplates";
52      static final String KEYPLACEHOLDERS_OUTER_SEPARATOR = "|";
53      static final String KEYPLACEHOLDERS_INNER_SEPARATOR = ";";
54      /** Boolean indicating if explicit ECC parameters should be used, instead of named curves which is standard. 
55       *  Explicit parameters are only used by ICAO CSCA and DS certificates as defined in ICAO 9303.
56       */
57      static final String EXPLICIT_ECC_PUBLICKEY_PARAMETERS = "explicit.ecc.publickey.parameters";
58      /** */
59      static final String TOKENNAME_PROPERTY = "tokenName";
60      /** Whether non-existent slots should be allowed or throw an NoSuchSlotException */
61      static final String ALLOW_NONEXISTING_SLOT_PROPERTY = "allow.nonexisting.slot";
62  
63      /**
64       * Method called after creation of instance. Gives the object it's properties.
65       *
66       * Info contains Crypto Token properties, as entered for all tokens. Properties can create properties used by extending or wrapping classes so the implementing
67       * class should not delete or remove properties that are unknown to it, but keep them transparently.
68       * Data contains Crypto Token data, can be null for tokens that don't need it.
69       *
70       * @param properties Properties info used to create token, new token or init existing token
71       * @param data byte[] data as created internally for tokens, can be null for tokens that don't need it
72       * @param id unique ID of the user of the token, the id is user defined and not used internally for anything but logging.
73       * @throws Exception on fail
74       */
75      void init(Properties properties, byte[] data, int id) throws Exception;
76  
77      /** Gets the id that was passed as parameter to init
78       * @return id
79       *
80       */
81      int getId();
82  
83      /**
84       * Method used to activate Crypto Tokens when connected after being offline.
85       *
86       * @param authenticationcode used to unlock crypto token, i.e PIN for smartcard HSMs. This parameter might be ignored, e.g. for auto-activated soft tokens.
87       * @throws CryptoTokenOfflineException if Crypto Token is not available or connected.
88       * @throws CryptoTokenAuthenticationFailedException with error message if authentication to tokens fail.
89       */
90      void activate(char[] authenticationcode) throws CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException;
91  
92      /**
93       * Method used to deactivate tokens.
94       * Used to set a Crypto Token to offline status and to reset the HSMs authorization code.
95       *
96       */
97      void deactivate();
98  
99      /** Checks if an alias is in use by a Crypto Token
100      * @param alias the alias that we want to check
101      * @return true if there is a private, public or symmetric entry with this alias in the CryptoToken 
102      */
103     boolean isAliasUsed(String alias);
104 
105     /** Returns the private key (if possible) of token.
106     * Note that this method does NOT check if the key is permitted to be extracted.
107     *
108     * @param alias the key alias to retrieve from the token
109     * @throws CryptoTokenOfflineException if Crypto Token is not available or connected, or key with alias does not exist.
110     * @return PrivateKey object
111     */
112     PrivateKey getPrivateKey(String alias) throws CryptoTokenOfflineException;
113 
114     /** Returns the public key (if possible) of token.
115     *
116     * @param alias the key alias to retrieve from the token
117     * @throws CryptoTokenOfflineException if Crypto Token is not available or connected
118     * @return the public key, or null if key with the alias does not exist.
119     */
120     PublicKey getPublicKey(String alias) throws CryptoTokenOfflineException;
121 
122     /** Returns the key (if possible) of token, used for symmetric keys. For assymmetric keys getPrivateKye and getPublic key is recommended..
123     *
124     * @param alias the key alias to retrieve from the token
125     * @throws CryptoTokenOfflineException if Crypto Token is not available or connected, or key with alias does not exist.
126     * @return Key object
127     */
128     Key getKey(String alias) throws CryptoTokenOfflineException;
129 
130     /** Deletes an entry in the crypto token
131      *
132      * @param alias is a reference to the entry in the token that should be deleted.
133 
134      * @throws KeyStoreException on fail
135      * @throws IOException on IO error
136      * @throws CertificateException on invalid cert
137      * @throws NoSuchAlgorithmException if algorithm not found
138      * @throws CryptoTokenOfflineException if offline
139      */
140     void deleteEntry(String alias) throws KeyStoreException, NoSuchAlgorithmException, CertificateException, IOException, CryptoTokenOfflineException;
141 
142     /** Generates a key pair (asymmetric keys) in the crypto token.
143      *
144      * @param keySpec all decimal digits RSA key length, otherwise name of ECC curve or DSA key using syntax DSAnnnn
145      * @param alias the name of the key pair in the crypto token
146      * @throws InvalidAlgorithmParameterException if params are invalid
147      * @throws CryptoTokenOfflineException if offline
148      */
149     void generateKeyPair(final String keySpec, final String alias) throws InvalidAlgorithmParameterException,
150             CryptoTokenOfflineException;
151 
152     /** Generates a key pair (asymmetric keys) in the crypto token. This method is used when you have an existing PublicKey and
153      * want to generate a key of the same type. You can use KeyTools to get the AlgorithmParameterSpec from an existing PublicKey.
154      * <pre>
155      * AlgorithmParameterSpec spec = KeyTools.getKeyGenSpec(templatePublicKey);
156      * </pre>
157      *
158      * @param spec AlgorithmParameterSpec describing the key pair to be generated
159      * @param alias the name of the key pair in the crypto token
160      * @throws InvalidAlgorithmParameterException On invalid parameters
161      * @throws CertificateException On invalid certificate
162      * @throws IOException On IO error
163      * @throws CryptoTokenOfflineException if offline 
164      */
165     void generateKeyPair(final AlgorithmParameterSpec spec, final String alias) throws InvalidAlgorithmParameterException, CertificateException,
166             IOException, CryptoTokenOfflineException;
167 
168     /** Generates a symmetric key.
169      *
170      * @param algorithm symmetric algorithm specified in http://download.oracle.com/javase/1.5.0/docs/api/index.html, suggest AES, DESede or DES
171      * @param keysize keysize of symmetric key, suggest 128 or 256 for AES, 64 for 168 for DESede and 64 for DES
172      * @param alias the alias the key will get in the keystore
173      * @throws NoSuchProviderException If provider not found
174      * @throws NoSuchAlgorithmException of algo not found
175      * @throws KeyStoreException If store fails
176      * @throws CryptoTokenOfflineException if offline
177      * @throws InvalidKeyException if key is invalid
178      * @throws InvalidAlgorithmParameterException  if parameters are invalid
179      * @throws SignatureException if signature is invalid
180      * @throws CertificateException if cert is invalid
181      * @throws IOException on IO fail
182      * @throws NoSuchPaddingException if padding is invalid 
183      * @throws IllegalBlockSizeException if block size is incorrect
184      */
185     void generateKey(final String algorithm, final int keysize, final String alias) throws NoSuchAlgorithmException, NoSuchProviderException, KeyStoreException, CryptoTokenOfflineException,
186     InvalidKeyException, InvalidAlgorithmParameterException, SignatureException, CertificateException, IOException, NoSuchPaddingException, IllegalBlockSizeException;
187 
188     /** Returns the signature Provider that should be used to sign things with
189      *  the PrivateKey object returned by this signing device implementation.
190      * @return String the name of the Provider
191      */
192     String getSignProviderName();
193 
194     /** Returns the crypto Provider that should be used to encrypt/decrypt things with
195      *  the PrivateKey object returned by this signing device implementation.
196      *  In most cases this is the same as the signature provider.
197      * @return String the name of the Provider
198      */
199     String getEncProviderName();
200 
201     /**
202      * Resets token. Might cure HW failures. It is up to each implementation to implement this or not.
203      */
204     void reset();
205 
206     /** @return user friendly identifier */
207     String getTokenName();
208     /** Set user friendly identifier 
209      * @param tokenName name*/
210     void setTokenName(String tokenName);
211     
212     /**
213      *  Method that returns the current status of the crypto token.
214      *
215      *  @return one of the CryptoToken.STATUS_.. values
216      */
217     int getTokenStatus();
218 
219     /** Return token properties, can be the same as passed to init, or an updated set
220      *
221      * @return Properties can be empty but should never be null
222      */
223     Properties getProperties();
224 
225     /** Updates dynamic properties for the crypto token. Call this method when a new key string, autoactivation PIN has been set
226      *  and the init method on the crypto token is not called.
227      *  Does not update properties that is only used when token is created, for example P11 slot, this is only updated on recreation of the token.
228      * @param properties Properties containing the new key properties or other properties, such as activation PIN
229      */
230     void setProperties(Properties properties);
231 
232     /**
233      * Stores a new key in this crypto token's keystore. 
234      * 
235      * @param alias The alias for the key
236      * @param key The key to be stored
237      * @param chain The associated key chain
238      * @param password Password to this slot.
239      * @throws KeyStoreException if keystore for this crypto token has not been initialized
240      */
241     void storeKey(String alias, Key key, Certificate chain[], char[] password) throws KeyStoreException;
242     
243     /** Stores keystore data (if any) to be used when initializing a new (existing) token with the init method
244      *
245      * @return byte[] with keystore data, can be null if not needed for initialization
246      */
247 	byte[] getTokenData();
248 
249     /** Testing a keypair to see that it is usable
250      *
251      * @param alias the alias of the key pair to test
252      * @throws InvalidKeyException if the public key can not be used to verify a string signed by the private key, because the key is wrong or the 
253      * signature operation fails for other reasons such as a NoSuchAlgorithmException or SignatureException.
254      * @throws CryptoTokenOfflineException if the crypto token is offline
255      */
256     void testKeyPair(String alias) throws InvalidKeyException, CryptoTokenOfflineException; // NOPMD:this is not a junit test
257 
258     /** Testing a keypair to see that it is usable
259     *
260     * @param alias the alias of the key pair to test
261     * @param publicKey the public key of the key pair
262     * @param privateKey the private key of the key pair
263     * @throws InvalidKeyException if the public key can not be used to verify a string signed by the private key, because the key is wrong or the 
264     * signature operation fails for other reasons such as a NoSuchAlgorithmException or SignatureException.
265     */
266     void testKeyPair(final String alias, PublicKey publicKey, PrivateKey privateKey) throws InvalidKeyException;
267 
268     /**
269      * Checks if generated private keys are allowed to be extracted.
270      *
271      * @return false if the private key is not extractable.
272      */
273     boolean doPermitExtractablePrivateKey();
274 
275     /**
276      *
277      * @return an enumeration of aliases from this token's key store.
278      * @throws KeyStoreException if the keystore has not been initialized (loaded).
279      * @throws CryptoTokenOfflineException if the keystore has not been initialized (loaded).
280      */
281     List<String> getAliases() throws KeyStoreException, CryptoTokenOfflineException;
282 
283     /** @return true if there is an auto activation PIN stored in the token */
284     boolean isAutoActivationPinPresent();
285     
286 }
