29  /**
30   * Each instance of this class represents a slot on a P11 module.
31   * Use an instance of this class for all your access of a specific P11 slot.
32   * Use {@link P11Slot#getProvider()} to get a provider for the slot.
33   *
34   * @version $Id: P11Slot.java 30502 2018-11-14 13:44:43Z anatom $
35   */
36  public class P11Slot {

38      private static final Logger log = Logger.getLogger(P11Slot.class);
39  
40      /** Used for library key map when a sun configuration file is used to specify a token (slot). In this case only one lib could be used. */
41      private static final String ONLY_ONE = "onlyOne";
42      
43      private final static Map<String,P11Slot> slotMap = new HashMap<String, P11Slot>();
44      private final Map<Integer, P11SlotUser> p11SlotUserMap = new HashMap<Integer, P11SlotUser>();
45      private final Pkcs11SlotLabelType slotLabelType;
46      private final String slotLabel;
47      private final String sharedLibrary;
48      private final String sunP11ConfigFileName;
49      private Provider provider;
50      private final String libraryFileName;
51      
52      private P11Slot(final Pkcs11SlotLabelType slotLabelType, final String slotLabel, final String sharedLibrary, final String attributesFile,
53              final String friendlyName, boolean addProvider) throws NoSuchSlotException {
54          this.slotLabelType = slotLabelType;
55          this.slotLabel = slotLabel;
56          this.sharedLibrary = sharedLibrary;
57          this.sunP11ConfigFileName = null;
58          this.libraryFileName = new File(sharedLibrary).getName();
59          provider = Pkcs11SlotLabel.getP11Provider(slotLabel, slotLabelType, sharedLibrary, attributesFile);
60          if (provider==null) {
61              throw new NoSuchSlotException("Slot labeled " + slotLabel + " could not be located.");
62          }
63          addProviderIfNotExisting(addProvider);
64      }
65  
66      private P11Slot(final String sunP11ConfigFileName, boolean addProvider) throws NoSuchSlotException {
67          this.slotLabelType = Pkcs11SlotLabelType.SUN_FILE;
68          this.sunP11ConfigFileName = sunP11ConfigFileName;
69          this.slotLabel = null;
70          this.sharedLibrary = null;
71          this.libraryFileName = ONLY_ONE;
72          this.provider = new Pkcs11SlotLabel(Pkcs11SlotLabelType.SUN_FILE, null).getProvider(sunP11ConfigFileName, null, null);
73          if (this.provider==null) {
74              throw new NoSuchSlotException("Slot configured in " + sunP11ConfigFileName + " could not be located.");
75          }
76          addProviderIfNotExisting(addProvider);
77      }
78  
79      /** Add a PKCS11 Crypto Provider to Java Security.addProvider, if it is not already added, and add==true.
80       * The add parameter can seem redundant, but it is here in order to centralize control of what is added as a 
81       * Crypto Provider, instead if distributing it to multiple places in the code.
82       * @param add default value should be true, set to false to create a P11 slot without actually adding the P11 provider to Java Security
83       */
84      private void addProviderIfNotExisting(boolean add) {
85          // If we already have a provider installed, it means there is another Crypto Token already using this slot
86          // It can potentially be a Database Integrity Protection Crypto Token
87          // We can't remove the already existing provider as that will cause the existing one to stop working. 
88          // A classic error message when that happens is: 
89          // java.security.InvalidKeyException: Private key must be instance of RSAPrivate(Crt)Key or have PKCS#8 encoding
90          // Some HSMs (like Thales) only have one slot so in many cases it is a must to be able to use the same slot for
91          // database integrity protection and a Crypto Token for CAs
92          if (Security.getProvider(provider.getName()) != null) {
93              // Because of the above, and how Sun P11 Provider works, we can re-use the existing provider
94              // Of course, of this provider is not working (disconnected to HSM was disconnected or similar), this 
95              // P11Slot will also just not work, while a remove and re-add might have caused it to start working
96              provider = Security.getProvider(provider.getName());
97              log.info("Found an existing PKCS#11 Provider while activating Crypto Token, re-using that instead of a new one: "+provider);
98          } else {
99              // Give the possibility to create a P11 slot without actually adding the P11 provider to Java Security
100             if (add) {
101                 Security.addProvider(provider);
102                 if (log.isDebugEnabled()) {
103                     log.debug("PKCS#11 Provider successfully added: "+provider.getName());
104                 }
105             } else {
106                 log.info("Did not find an existing PKCS#11 Provider while activating Crypto Token, but was configured to not add one either: "+provider.getName());
107             }
108         }
109     }

111     @Override
112     public String toString() {
113         if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
114             return "PKCS#11, Sun configuration file name: " + sunP11ConfigFileName;
115         }
116         return "PKCS#11 slot " + slotLabel + " using library " + sharedLibrary + ".";
117     }

119     /** Reset the HSM. Could be done if it has stopped working in a try to get it working again. */
120     public void reset() {
121         synchronized (slotMap) {
122             for (final P11Slot slot : slotMap.values()) {
123                 if (slot.libraryFileName.equals(libraryFileName)) {
124                     for (final P11SlotUser p11SlotUser : slot.p11SlotUserMap.values()) {
125                         try {
126                             p11SlotUser.deactivate();
127                         } catch (Exception e) {
128                             log.error("Not possible to deactivate token.", e);
129                         }
130                     }
131                     if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
132                         break;
133                     }
134                 }
135             }
136         }
137     }

139     /** Unload if last active token on slot */
140     public void logoutFromSlotIfNoTokensActive() {
141         synchronized (slotMap) {
142             for (final P11SlotUser p11SlotUser : p11SlotUserMap.values()) {
143                 if (p11SlotUser.isActive()) {
144                     return;
145                 }
146             }
147         }
148         if (provider instanceof AuthProvider) {
149             try {
150                 ((AuthProvider)provider).logout();
151                 if (log.isDebugEnabled()) {
152                     log.debug("PKCS#11 session terminated for \"" + toString() + "\".");
153                 }
154             } catch (LoginException e) {
155                 log.warn("Not possible to logout from PKCS#11 Session. HW problems?", e);
156             }
157         } else {
158             log.warn("Not possible to logout from PKCS#11 provider '" + toString() + "'. It is not implementing '" + AuthProvider.class.getCanonicalName() + "'.");
159         }
160     }

167     /**
168      * Get P11 slot instance. Only one instance (provider) will ever be created for each slot regardless of how many times this method is called.
169      * @param slotLabel the labeling of the slot, regardless of label type. 
170      * @param sharedLibrary file path of shared
171      * @param slotLabelType The type of the label. May be a slot number [0...9]*, slot index i[0...9]* or a label.
172      * @param attributesFile Attributes file. Optional. Set to null if not used
173      * @param token Token that should use this object
174      * @param id unique ID of the user of the token. For EJBCA this is the caid. For the OCSP responder this is fixed since then there is only one user.
175      * @param addProvider default value should be true, set to false to create a P11 slot without actually adding the P11 provider to Java Security
176      * @return P11Slot
177      * @throws CryptoTokenOfflineException if token can not be activated
178      * @throws NoSuchSlotException if no slot with the label defined by slotLabel could be found
179      */
180     public static P11Slot getInstance(final String slotLabel, final String sharedLibrary, final Pkcs11SlotLabelType slotLabelType, 
181             final String attributesFile, final P11SlotUser token, final int id, boolean addProvider) throws CryptoTokenOfflineException, NoSuchSlotException {       
182         final String friendlyName = slotLabel + sharedLibrary + slotLabelType.toString();
183         return getInstance(friendlyName, slotLabel, sharedLibrary, slotLabelType, attributesFile, token, id, addProvider);
184     }

187      * Get P11 slot instance. Only one instance (provider) will ever be created for each slot regardless of how many times this method is called.
188      * @param friendlyName name to identify the instance
189      * @param slotLabel the labeling of the slot, regardless of label type. 
190      * @param sharedLibrary file path of shared
191      * @param slotLabelType The type of the label. May be a slot number [0...9]*, slot index i[0...9]* or a label. 
192      * @param attributesFile Attributes file. Optional. Set to null if not used
193      * @param p11SlotUser Token that should use this object
194      * @param id unique ID of the user of the token. For EJBCA this is the caid. For the OCSP responder this is fixed since then there is only one user.
195      * @param addProvider default value should be true, set to false to create a P11 slot without actually adding the P11 provider to Java Security
196      * @return P11Slot
197      * @throws CryptoTokenOfflineException if token can not be activated
198      * @throws NoSuchSlotException if no slot by the given label could be found
199      */
200     public static P11Slot getInstance(final String friendlyName, final String slotLabel, final String sharedLibrary, final Pkcs11SlotLabelType slotLabelType, 
201             final String attributesFile, final P11SlotUser p11SlotUser, final int id, boolean addProvider) throws NoSuchSlotException, CryptoTokenOfflineException {
202         if (log.isDebugEnabled()) {
203             log.debug("P11Slot.getInstance(): "+String.valueOf(slotLabelType)+"'"+slotLabel+"', '"+sharedLibrary+"', "+", '"+attributesFile+"', "+id);
204         }
205         return getInstance(slotLabelType, friendlyName, slotLabel, sharedLibrary, attributesFile, null, p11SlotUser, id, addProvider);
206     }
207     /**
208      * As {@link #getInstance(String, String, Pkcs11SlotLabelType, String, P11SlotUser, int, boolean)} but is using config file instead parameters. Do only use this method if the P11 shared library is ony specified in this config file.
209      * @param sunP11ConfigFileName name of config file
210      * @param p11SlotUser Token that should use this object.
211      * @param id unique ID of the user of the token. For EJBCA this is the caid. For the OCSP responder this is fixed since then there is only one user.
212      * @param addProvider default value should be true, set to false to create a P11 slot without actually adding the P11 provider to Java Security
213      * @return a new P11Slot instance
214      * @throws CryptoTokenOfflineException on fail
215      * @throws NoSuchSlotException if no slot defined by the label in configFileName could be found.
216      */
217     public static P11Slot getInstance(final String sunP11ConfigFileName, final P11SlotUser p11SlotUser, final int id, boolean addProvider) throws NoSuchSlotException, CryptoTokenOfflineException {
218         if (log.isDebugEnabled()) {
219             log.debug("P11Slot.getInstance(): '"+sunP11ConfigFileName+"', "+Pkcs11SlotLabelType.SUN_FILE.toString()+", "+id);
220         }
221         return getInstance(Pkcs11SlotLabelType.SUN_FILE, null, null, null, null, sunP11ConfigFileName, p11SlotUser, id, addProvider);
222     }

224     private static P11Slot getInstance(final Pkcs11SlotLabelType slotLabelType, final String friendlyName, final String slotLabel, final String sharedLibrary,
225             final String attributesFile, final String sunP11ConfigFileName, final P11SlotUser p11SlotUser, final int id, boolean addProvider) throws NoSuchSlotException, CryptoTokenOfflineException {
226         try {
227             final String slotMapKey;
228             if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
229                 if (sunP11ConfigFileName==null) {
230                     throw new IllegalStateException("Can't initialize PKCS#11 slot of type "+slotLabelType.name()+" without providing a config file.");
231                 }
232                 slotMapKey = new File(sunP11ConfigFileName).getName();
233             } else {
234                 if (slotLabel==null || sharedLibrary==null) {
235                     throw new IllegalStateException("Can't initialize PKCS#11 slot of type "+slotLabelType.name()+" without providing library and slot label.");
236                 }
237                 slotMapKey = friendlyName;
238             }
239             P11Slot p11Slot;
240             synchronized (slotMap) {
241                 p11Slot = slotMap.get(slotMapKey);
242                 if (p11Slot==null) {
243                     if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
244                         p11Slot = new P11Slot(sunP11ConfigFileName, addProvider);
245                     } else {
246                         p11Slot = new P11Slot(slotLabelType, slotLabel, sharedLibrary, attributesFile, friendlyName, addProvider);
247                     }
248                     slotMap.put(slotMapKey, p11Slot);
249                 }
250                 p11Slot.p11SlotUserMap.put(Integer.valueOf(id), p11SlotUser);
251             }
252             return p11Slot;
253         } catch (NoSuchSlotException e) {
254             throw e;
255         } catch (Exception e) {
256             throw new CryptoTokenOfflineException(InternalResources.getInstance().getLocalizedMessage("token.errorcreatetoken", id), e);
257         }
258     }
