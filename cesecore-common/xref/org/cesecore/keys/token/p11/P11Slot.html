27  /**
28   * Each instance of this class represents a slot on a P11 module. Use an
29   * instance of this class for all your access of a specific P11 slot. Use {@link
30   * P11Slot#getProvider()} to get a provider for the slot.

34  public final class P11Slot {

36      /** Logger. */
37    private static final Logger LOG = Logger.getLogger(P11Slot.class);

39    /**
40     * Used for library key map when a sun configuration file is used to specify a
41     * token (slot). In this case only one lib could be used.
42     */
43    private static final String ONLY_ONE = "onlyOne";
44  
45    /** Slots. */
46    private static final Map<String, P11Slot> SLOT_MAP =
47        new HashMap<String, P11Slot>();
48    /** Slots. */
49    private final Map<Integer, P11SlotUser> p11SlotUserMap =
50        new HashMap<Integer, P11SlotUser>();
51    /** Type. */
52    private final Pkcs11SlotLabelType slotLabelType;
53    /** Label. */
54    private final String slotLabel;
55    /** Library. */
56    private final String sharedLibrary;
57    /** Config file. */
58    private final String sunP11ConfigFileName;
59    /** Provider. */
60    private Provider provider;
61    /** Library. */
62    private final String libraryFileName;
63  
64    private P11Slot(
65        final Pkcs11SlotLabelType aSlotLabelType,
66        final String aSlotLabel,
67        final String aSharedLibrary,
68        final String attributesFile,
69        final String friendlyName,
70        final boolean addProvider)
71        throws NoSuchSlotException {
72      this.slotLabelType = aSlotLabelType;
73      this.slotLabel = aSlotLabel;
74      this.sharedLibrary = aSharedLibrary;
75      this.sunP11ConfigFileName = null;
76      this.libraryFileName = new File(aSharedLibrary).getName();
77      provider =
78          Pkcs11SlotLabel.getP11Provider(
79              aSlotLabel, aSlotLabelType, aSharedLibrary, attributesFile);
80      if (provider == null) {
81        throw new NoSuchSlotException(
82            "Slot labeled " + aSlotLabel + " could not be located.");
83      }
84      addProviderIfNotExisting(addProvider);
85    }
86  
87    private P11Slot(final String aSunP11ConfigFileName, final boolean addProvider)
88        throws NoSuchSlotException {
89      this.slotLabelType = Pkcs11SlotLabelType.SUN_FILE;
90      this.sunP11ConfigFileName = aSunP11ConfigFileName;
91      this.slotLabel = null;
92      this.sharedLibrary = null;
93      this.libraryFileName = ONLY_ONE;
94      this.provider =
95          new Pkcs11SlotLabel(Pkcs11SlotLabelType.SUN_FILE, null)
96              .getProvider(aSunP11ConfigFileName, null, null);
97      if (this.provider == null) {
98        throw new NoSuchSlotException(
99            "Slot configured in "
100               + aSunP11ConfigFileName
101               + " could not be located.");
102     }
103     addProviderIfNotExisting(addProvider);
104   }

106   /**
107    * Add a PKCS11 Crypto Provider to Java Security.addProvider, if it is not
108    * already added, and add==true. The add parameter can seem redundant, but it
109    * is here in order to centralize control of what is added as a Crypto
110    * Provider, instead if distributing it to multiple places in the code.
111    *
112    * @param add default value should be true, set to false to create a P11 slot
113    *     without actually adding the P11 provider to Java Security
114    */
115   private void addProviderIfNotExisting(final boolean add) {
116     // If we already have a provider installed, it means there is another Crypto
117     // Token already using this slot
118     // It can potentially be a Database Integrity Protection Crypto Token
119     // We can't remove the already existing provider as that will cause the
120     // existing one to stop working.
121     // A classic error message when that happens is:
122     // java.security.InvalidKeyException: Private key must be instance of
123     // RSAPrivate(Crt)Key or have PKCS#8 encoding
124     // Some HSMs (like Thales) only have one slot so in many cases it is a must
125     // to be able to use the same slot for
126     // database integrity protection and a Crypto Token for CAs
127     if (Security.getProvider(provider.getName()) != null) {
128       // Because of the above, and how Sun P11 Provider works, we can re-use the
129       // existing provider
130       // Of course, of this provider is not working (disconnected to HSM was
131       // disconnected or similar), this
132       // P11Slot will also just not work, while a remove and re-add might have
133       // caused it to start working
134       provider = Security.getProvider(provider.getName());
135       LOG.info(
136           "Found an existing PKCS#11 Provider while activating Crypto Token,"
137               + " re-using that instead of a new one: "
138               + provider);
139     } else {
140       // Give the possibility to create a P11 slot without actually adding the
141       // P11 provider to Java Security
142       if (add) {
143         Security.addProvider(provider);
144         if (LOG.isDebugEnabled()) {
145           LOG.debug(
146               "PKCS#11 Provider successfully added: " + provider.getName());
147         }
148       } else {
149         LOG.info(
150             "Did not find an existing PKCS#11 Provider while activating Crypto"
151                 + " Token, but was configured to not add one either: "
152                 + provider.getName());
153       }
154     }
155   }

157   @Override
158   public String toString() {
159     if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
160       return "PKCS#11, Sun configuration file name: " + sunP11ConfigFileName;
161     }
162     return "PKCS#11 slot "
163         + slotLabel
164         + " using library "
165         + sharedLibrary
166         + ".";
167   }

169   /**
170    * Reset the HSM. Could be done if it has stopped working in a try to get it
171    * working again.
172    */
173   public void reset() {
174     synchronized (SLOT_MAP) {
175       for (final P11Slot slot : SLOT_MAP.values()) {
176         if (slot.libraryFileName.equals(libraryFileName)) {
177           for (final P11SlotUser p11SlotUser : slot.p11SlotUserMap.values()) {
178             try {
179               p11SlotUser.deactivate();
180             } catch (Exception e) {
181               LOG.error("Not possible to deactivate token.", e);
182             }
183           }
184           if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
185             break;
186           }
187         }
188       }
189     }
190   }

192   /** Unload if last active token on slot. */
193   public void logoutFromSlotIfNoTokensActive() {
194     synchronized (SLOT_MAP) {
195       for (final P11SlotUser p11SlotUser : p11SlotUserMap.values()) {
196         if (p11SlotUser.isActive()) {
197           return;
198         }
199       }
200     }
201     if (provider instanceof AuthProvider) {
202       try {
203         ((AuthProvider) provider).logout();
204         if (LOG.isDebugEnabled()) {
205           LOG.debug("PKCS#11 session terminated for \"" + toString() + "\".");
206         }
207       } catch (LoginException e) {
208         LOG.warn(
209             "Not possible to logout from PKCS#11 Session. HW problems?", e);
210       }
211     } else {
212       LOG.warn(
213           "Not possible to logout from PKCS#11 provider '"
214               + toString()
215               + "'. It is not implementing '"
216               + AuthProvider.class.getCanonicalName()
217               + "'.");
218     }
219   }

226   /**
227    * Get P11 slot instance. Only one instance (provider) will ever be created
228    * for each slot regardless of how many times this method is called.
229    *
230    * @param slotLabel the labeling of the slot, regardless of label type.
231    * @param sharedLibrary file path of shared
232    * @param slotLabelType The type of the label. May be a slot number [0...9]*,
233    *     slot index i[0...9]* or a label.
234    * @param attributesFile Attributes file. Optional. Set to null if not used
235    * @param token Token that should use this object
236    * @param id unique ID of the user of the token. For EJBCA this is the caid.
237    *     For the OCSP responder this is fixed since then there is only one user.
238    * @param addProvider default value should be true, set to false to create a
239    *     P11 slot without actually adding the P11 provider to Java Security
240    * @return P11Slot
241    * @throws CryptoTokenOfflineException if token can not be activated
242    * @throws NoSuchSlotException if no slot with the label defined by slotLabel
243    *     could be found
244    */
245   public static P11Slot getInstance(
246       final String slotLabel,
247       final String sharedLibrary,
248       final Pkcs11SlotLabelType slotLabelType,
249       final String attributesFile,
250       final P11SlotUser token,
251       final int id,
252       final boolean addProvider)
253       throws CryptoTokenOfflineException, NoSuchSlotException {
254     final String friendlyName =
255         slotLabel + sharedLibrary + slotLabelType.toString();
256     return getInstance(
257         friendlyName,
258         slotLabel,
259         sharedLibrary,
260         slotLabelType,
261         attributesFile,
262         token,
263         id,
264         addProvider);
265   }

273    * @param sharedLibrary file path of shared
274    * @param slotLabelType The type of the label. May be a slot number [0...9]*,
275    *     slot index i[0...9]* or a label.
276    * @param attributesFile Attributes file. Optional. Set to null if not used
277    * @param p11SlotUser Token that should use this object
278    * @param id unique ID of the user of the token. For EJBCA this is the caid.
279    *     For the OCSP responder this is fixed since then there is only one user.
280    * @param addProvider default value should be true, set to false to create a
281    *     P11 slot without actually adding the P11 provider to Java Security
282    * @return P11Slot
283    * @throws CryptoTokenOfflineException if token can not be activated
284    * @throws NoSuchSlotException if no slot by the given label could be found
285    */
286   public static P11Slot getInstance(
287       final String friendlyName,
288       final String slotLabel,
289       final String sharedLibrary,
290       final Pkcs11SlotLabelType slotLabelType,
291       final String attributesFile,
292       final P11SlotUser p11SlotUser,
293       final int id,
294       final boolean addProvider)
295       throws NoSuchSlotException, CryptoTokenOfflineException {
296     if (LOG.isDebugEnabled()) {
297       LOG.debug(
298           "P11Slot.getInstance(): "
299               + String.valueOf(slotLabelType)
300               + "'"
301               + slotLabel
302               + "', '"
303               + sharedLibrary
304               + "', "
305               + ", '"
306               + attributesFile
307               + "', "
308               + id);
309     }
310     return getInstance(
311         slotLabelType,

313         slotLabel,
314         sharedLibrary,
315         attributesFile,
316         null,
317         p11SlotUser,
318         id,
319         addProvider);
320   }
321   /**
322    * As {@link #getInstance(String, String, Pkcs11SlotLabelType, String,
323    * P11SlotUser, int, boolean)} but is using config file instead parameters. Do
324    * only use this method if the P11 shared library is ony specified in this
325    * config file.
326    *
327    * @param sunP11ConfigFileName name of config file
328    * @param p11SlotUser Token that should use this object.
329    * @param id unique ID of the user of the token. For EJBCA this is the caid.
330    *     For the OCSP responder this is fixed since then there is only one user.
331    * @param addProvider default value should be true, set to false to create a
332    *     P11 slot without actually adding the P11 provider to Java Security
333    * @return a new P11Slot instance
334    * @throws CryptoTokenOfflineException on fail
335    * @throws NoSuchSlotException if no slot defined by the label in
336    *     configFileName could be found.
337    */
338   public static P11Slot getInstance(
339       final String sunP11ConfigFileName,
340       final P11SlotUser p11SlotUser,
341       final int id,
342       final boolean addProvider)
343       throws NoSuchSlotException, CryptoTokenOfflineException {
344     if (LOG.isDebugEnabled()) {
345       LOG.debug(
346           "P11Slot.getInstance(): '"
347               + sunP11ConfigFileName
348               + "', "
349               + Pkcs11SlotLabelType.SUN_FILE.toString()
350               + ", "
351               + id);
352     }
353     return getInstance(
354         Pkcs11SlotLabelType.SUN_FILE,
355         null,
356         null,
357         null,
358         null,
359         sunP11ConfigFileName,
360         p11SlotUser,
361         id,
362         addProvider);
363   }

365   private static P11Slot getInstance(
366       final Pkcs11SlotLabelType slotLabelType,
367       final String friendlyName,
368       final String slotLabel,
369       final String sharedLibrary,
370       final String attributesFile,
371       final String sunP11ConfigFileName,
372       final P11SlotUser p11SlotUser,
373       final int id,
374       final boolean addProvider)
375       throws NoSuchSlotException, CryptoTokenOfflineException {
376     try {
377       final String slotMapKey;
378       if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
379         if (sunP11ConfigFileName == null) {
380           throw new IllegalStateException(
381               "Can't initialize PKCS#11 slot of type "
382                   + slotLabelType.name()
383                   + " without providing a config file.");
384         }
385         slotMapKey = new File(sunP11ConfigFileName).getName();
386       } else {
387         if (slotLabel == null || sharedLibrary == null) {
388           throw new IllegalStateException(
389               "Can't initialize PKCS#11 slot of type "
390                   + slotLabelType.name()
391                   + " without providing library and slot label.");
392         }
393         slotMapKey = friendlyName;
394       }
395       P11Slot p11Slot;
396       synchronized (SLOT_MAP) {
397         p11Slot = SLOT_MAP.get(slotMapKey);
398         if (p11Slot == null) {
399           if (Pkcs11SlotLabelType.SUN_FILE.equals(slotLabelType)) {
400             p11Slot = new P11Slot(sunP11ConfigFileName, addProvider);
401           } else {
402             p11Slot =
403                 new P11Slot(
404                     slotLabelType,
405                     slotLabel,
406                     sharedLibrary,
407                     attributesFile,
408                     friendlyName,
409                     addProvider);
410           }
411           SLOT_MAP.put(slotMapKey, p11Slot);
412         }
413         p11Slot.p11SlotUserMap.put(Integer.valueOf(id), p11SlotUser);
414       }
415       return p11Slot;
416     } catch (NoSuchSlotException e) {
417       throw e;
418     } catch (Exception e) {
419       throw new CryptoTokenOfflineException(
420           InternalResources.getInstance()
421               .getLocalizedMessage("token.errorcreatetoken", id),
422           e);
423     }
424   }
