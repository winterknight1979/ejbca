13  package org.cesecore.keys.token.p11;
14  
15  /**
16   * All users of the {@link P11Slot} slot must implement this interface. The user
17   * may decide whether deactivation is allowed or not. Deactivation of a user is
18   * done when the {@link P11Slot} object wants to reset P11 session (disconnect
19   * and reconnect).
20   *
21   * <p>If deactivation is allowed and {@link #deactivate()} called the user
22   * should:<br>
23   * Deactivate itself (answer false to {@link #isActive()}) and call {@link
24   * P11Slot#logoutFromSlotIfNoTokensActive()} Then {@link P11Slot#getProvider()}
25   * must be called before using the provider again.
26   *
27   * <p>If deactivation is not allowed then the user may just continue to answer
28   * true to {@link #isActive()}.
29   *
30   * @version $Id: P11SlotUser.java 17625 2013-09-20 07:12:06Z netmackan $
31   */
32  public interface P11SlotUser {
33    /**
34     * Called by the {@link P11Slot} when resetting the slot.
35     *
36     * @throws Exception on error
37     */
38    void deactivate() throws Exception;
39  
40    /**
41     * The user should return true if not accepting a slot reset.
42     *
43     * @return true if the slot is being used.
44     */
45    boolean isActive();
46  }
