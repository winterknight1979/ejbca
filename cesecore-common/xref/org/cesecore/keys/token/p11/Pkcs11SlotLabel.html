85      /**
86       * Get provider for the slot.
87       * @param fileName path name to the P11 module so file or sun config file (only in the case of {@link #type}=={@link Pkcs11SlotLabelType#SUN_FILE})
88       * @param attributesFile Path to file with P11 attributes to be used when generating keys with the provider. If null a good default will be used.
89       * @param privateKeyLabel Label that will be set to all private keys generated by the provider. If null no label will be set.
90       * @return the provider, or null if none is available.
91       * @throws NoSuchSlotException if no slot as defined by this slot label was found
92       * 
93       */
94      public Provider getProvider(final String fileName, final String attributesFile, final String privateKeyLabel) throws NoSuchSlotException {
95          if (StringUtils.isEmpty(fileName)) {
96              throw new IllegalArgumentException("A file name must be supplied.");
97          }
98          final File libFile = new File(fileName);
99          if (!libFile.isFile() || !libFile.canRead()) {
100             throw new IllegalArgumentException("The file " + fileName + " can't be read.");
101         }
102         // We will construct the PKCS11 provider (sun.security..., or iaik...) using reflection, because
103         // the sun class does not exist on all platforms in jdk5, and we want to be able to compile everything.
104         if (log.isDebugEnabled()) {
105             log.debug("slot spec: " + toString());
106         }
107         if ( this.type==Pkcs11SlotLabelType.SUN_FILE ) {// if sun cfg file then we do not know the name of the p11 module and we must quit.
108             try {
109                 List<String> fileContent = new ArrayList<>(Files.readAllLines(Paths.get(fileName), StandardCharsets.UTF_8));
110                 replaceCkaLabel(fileContent, privateKeyLabel);
111                 final ByteArrayOutputStream baos = new ByteArrayOutputStream();
112                 try( final PrintWriter pw = new PrintWriter(baos) ) {
113                     for (String string : fileContent) {
114                         pw.println(string);
115                     }
116                 }
117                 if (log.isDebugEnabled()) {
118                     log.debug(baos.toString());
119                 }
120                 try ( final ByteArrayInputStream inputStream = new ByteArrayInputStream(baos.toByteArray()) ) {
121                     return getSunP11Provider(inputStream);
122                 }
123             } catch (FileNotFoundException e) {
124                 throw new IllegalArgumentException("File " + libFile + " was not found.");
125             } catch ( IOException e ) {
126                 throw new P11RuntimeException(String.format("The file %s can not be closed after use.", libFile), e);
127             }
128         }
129         final long slot;
130         final Pkcs11Wrapper p11 = Pkcs11Wrapper.getInstance(libFile); // must be called before any provider is created for libFile
131         switch (this.type) {
132         case SLOT_LABEL:
133             slot = getSlotID(this.value, p11);
134             if (slot < 0) {
135                 throw new IllegalStateException("Token label '" + this.value + "' not found.");
136             }
137             break;
138         case SLOT_NUMBER:
139             slot = Long.parseLong(this.value);
140             break;
141         case SLOT_INDEX:
142             //Be generous and allow numbers to act as indexes as well
143             slot = Long.parseLong((this.value.charAt(0) == 'i' ? this.value.substring(1) : this.value));
144             break;
145             //$CASES-OMITTED$
146         default:
147             throw new IllegalStateException("This should not ever happen if all type of slots are tested.");
148         }
149         {// We will first try to construct the more competent IAIK provider, if it exists in the classpath
150             final Provider prov = getIAIKP11Provider(slot, libFile, this.type);
151             if (prov != null) {
152                 return prov;
153             }
154         }
155         {// if that does not exist, we will revert back to use the SUN provider
156             final Provider prov = getSunP11Provider( getSunP11ProviderInputStream(slot, libFile, this.type, attributesFile, privateKeyLabel) );
157             if (prov != null) {
158                 return prov;
159             }
160         }
161         log.error("No provider available.");
162         return null;
163     }

165     /** @param libFile library file
166      * @return a List of "slotId;tokenLabel" in the (indexed) order we get the from the P11 */
167     public static List<String> getExtendedTokenLabels(final File libFile) {
168         final List<String> tokenLabels = new ArrayList<>();
169         final Pkcs11Wrapper p11 = Pkcs11Wrapper.getInstance(libFile);
170         final long slots[] = p11.getSlotList();
171         if (log.isDebugEnabled()) {
172             log.debug("Found numer of slots:\t" + slots.length);
173         }
174         for (int i=0; i<slots.length; i++) {
175             final long slotID = slots[i];
176             final char label[] = p11.getTokenLabel(slotID);
177             if (label == null) {
178                 continue;
179             }
180             final String tokenLabel = new String(label);
181             if (log.isDebugEnabled()) {
182                 log.debug(i+": Found token label:\t" + tokenLabel + "\tid="+slotID);
183             }
184             tokenLabels.add(slotID+";"+tokenLabel.trim());
185         }
186         return tokenLabels;
187     }

189     /**
190      * Get slot ID for a token label.
191      * @param tokenLabel the label.
192      * @param p11 object to get slot list and labels for all slots with tokens
193      * @return the slot ID.
194      * @throws NoSuchSlotException if no slot as defined by tokenLabel was found
195 
196      */
197     private static long getSlotID(final String tokenLabel, final Pkcs11Wrapper p11) throws NoSuchSlotException {
198         final long slots[] = p11.getSlotList();
199         if (log.isDebugEnabled()) {
200             log.debug("Searching for token label:\t" + tokenLabel);
201         }
202 
203         for (final long slotID : slots) {
204             final char label[] = p11.getTokenLabel(slotID);
205             if (label == null) {
206                 continue;
207             }
208             final String candidateTokenLabel = new String(label);
209             if (log.isDebugEnabled()) {
210                 log.debug("Candidate token label:\t" + candidateTokenLabel);
211             }
212             if (!tokenLabel.equals(candidateTokenLabel.trim())) {
213                 continue;
214             }
215             if (log.isDebugEnabled()) {
216                 log.debug("Label '" + tokenLabel + "' found. The slot ID is:\t" + slotID);
217             }
218             return slotID;
219         }
220         throw new NoSuchSlotException("Token label '" + tokenLabel + "' not found.");
221     }

223     /**
224      * Get the IAIK provider.
225      * @param slot Slot list index or slot ID.
226      * @param libFile P11 module so file.
227      * @param type type
228      * @return the provider
229      */
230     private static Provider getIAIKP11Provider(final long slot, final File libFile, final Pkcs11SlotLabelType type) {
231         // Properties for the IAIK PKCS#11 provider
232         final Properties prop = new Properties();
233         try {
234             prop.setProperty("PKCS11_NATIVE_MODULE", libFile.getCanonicalPath());
235         } catch (IOException e) {
236             throw new RuntimeException("Could for unknown reason not construct canonical filename.", e);
237         }
238         // If using Slot Index it is denoted by brackets in iaik
239         prop.setProperty("SLOT_ID", type.equals(Pkcs11SlotLabelType.SLOT_INDEX) ? ("[" + slot + "]") : Long.toString(slot));
240         if (log.isDebugEnabled()) {
241             log.debug(prop.toString());
242         }
243         final Provider ret;
244         try {
245             @SuppressWarnings("unchecked")
246             final Class<? extends Provider> implClass = (Class<? extends Provider>) Class.forName(IAIK_PKCS11_CLASS);
247             if (log.isDebugEnabled()) {
248                 log.debug("Using IAIK PKCS11 provider: " + IAIK_PKCS11_CLASS);
249             }
250             // iaik PKCS11 has Properties as constructor argument
251             ret = implClass.getConstructor(Properties.class).newInstance(new Object[] { prop });
252         } catch (InvocationTargetException | InstantiationException | IllegalAccessException | IllegalArgumentException |
253                 NoSuchMethodException | SecurityException | ClassNotFoundException e) {
254             return null;
255         }
256         if ( ret==null ) {
257             return null;
258         }
259         try {
260             // It's not enough just to add the p11 provider. Depending on algorithms we may have to install the IAIK JCE provider as well in order
261             // to support algorithm delegation
262             @SuppressWarnings("unchecked")
263             final Class<? extends Provider> jceImplClass = (Class<? extends Provider>) Class.forName(IAIK_JCEPROVIDER_CLASS);
264             final Provider iaikProvider = jceImplClass.getConstructor().newInstance();
265             if (Security.getProvider(iaikProvider.getName()) == null) {
266                 log.info("Adding IAIK JCE provider for Delegation: " + IAIK_JCEPROVIDER_CLASS);
267                 Security.addProvider(iaikProvider);
268             }
269         } catch (InvocationTargetException | InstantiationException | IllegalAccessException | IllegalArgumentException |
270                 NoSuchMethodException | SecurityException | ClassNotFoundException e) {
271             // NOPMD: Ignore, reflection related errors are handled elsewhere
272         }
273         return ret;
274     }

276     /**
277      * Get an InputStream to be used to create the Sun provider.
278      * @param slot Slot list index or slot ID.
279      * @param libFile P11 module so file.
280      * @param type a Pkcs11SlotLabelType if the slot is not an ID byt a label for example.
281      * @param attributesFile Path to file with P11 attributes to be used when generating keys with the provider. If null a good default will be used.
282      * @param privateKeyLabel Label that will be set to all private keys generated by the provider. If null no label will be set.
283      * @return the stream
284      */
285     private static InputStream getSunP11ProviderInputStream(final long slot, final File libFile, final Pkcs11SlotLabelType type, final String attributesFile,
286             String privateKeyLabel) {
287 
288         // Properties for the SUN PKCS#11 provider
289         final String sSlot = Long.toString(slot);
290         final String libFilePath;
291         try {
292             libFilePath = libFile.getCanonicalPath();
293         } catch (IOException e) {
294             throw new RuntimeException("Could for unknown reason not construct canonical filename.", e);
295         }
296         if (log.isDebugEnabled()) {
297             log.debug(">getSunP11ProviderInputStream: "+sSlot+", "+libFilePath+", "+type.toString()+", "+attributesFile+", "+privateKeyLabel);
298         }
299         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
300         try( final PrintWriter pw = new PrintWriter(baos) ) {
301             pw.println("name = " + libFile.getName() + "-slot" + sSlot);
302             pw.println("library = " + libFilePath);
303             if (sSlot != null) {
304                 pw.println("slot" + (type.isEqual(Pkcs11SlotLabelType.SLOT_INDEX) ? "ListIndex" : "") + " = " + sSlot);
305             }
306             if (attributesFile != null) {
307                 try {
308                     List<String> fileContent = new ArrayList<>(Files.readAllLines(Paths.get(attributesFile), StandardCharsets.UTF_8));
309                     replaceCkaLabel(fileContent, privateKeyLabel);
310                     for (String string : fileContent) {
311                         pw.println(string);
312                     }
313                 } catch (IOException e) {
314                     throw new IllegalArgumentException("File " + attributesFile + " was not found, or could not be read.", e);
315                 }
316                 //pw.println(new String(attrs));
317             } else {
318                 // setting the attributes like this should work for most HSMs.
319                 pw.println("attributes(*, CKO_PUBLIC_KEY, *) = {");
320                 pw.println("  CKA_TOKEN = false"); // Do not save public keys permanent in order to save space.
321                 pw.println("  CKA_ENCRYPT = true");
322                 pw.println("  CKA_VERIFY = true");
323                 pw.println("  CKA_WRAP = true");// no harm allowing wrapping of keys. created private keys can not be wrapped anyway since CKA_EXTRACTABLE
324                 // is false.
325                 pw.println("}");
326                 pw.println("attributes(*, CKO_PRIVATE_KEY, *) = {");
327                 pw.println("  CKA_DERIVE = false"); //Amazon CloudHSM will not accept that CKA_DERIVE is present in private key template and will give CKR_TEMPLATE_INCONSISTENT.
328                 pw.println("  CKA_TOKEN = true"); // all created private keys should be permanent. They should not only exist during the session.
329                 pw.println("  CKA_PRIVATE = true"); // always require logon with password to use the key
330                 pw.println("  CKA_SENSITIVE = true"); // not possible to read the key
331                 pw.println("  CKA_EXTRACTABLE = false"); // not possible to wrap the key with another key
332                 pw.println("  CKA_DECRYPT = true");
333                 pw.println("  CKA_SIGN = true");
334                 if (privateKeyLabel != null && privateKeyLabel.length() > 0) {
335                     final String labelStr = "  CKA_LABEL = 0h" + new String(Hex.encode(privateKeyLabel.getBytes()));
336                     if (log.isDebugEnabled()) {
337                         log.debug("Setting CKA_LABEL to '"+labelStr+"'");
338                     }
339                     pw.println(labelStr);
340                 }
341                 pw.println("  CKA_UNWRAP = true");// for unwrapping of session keys,
342                 pw.println("}");
343                 if ( CesecoreConfiguration.p11disableHashingSignMechanisms() ) {
344                     pw.println("disabledMechanisms = {");
345                     // by disabling these mechanisms the hashing will be done in the application instead of the HSM.
346                     pw.println("  CKM_SHA1_RSA_PKCS");
347                     pw.println("  CKM_SHA256_RSA_PKCS");
348                     pw.println("  CKM_SHA384_RSA_PKCS");
349                     pw.println("  CKM_SHA512_RSA_PKCS");
350                     pw.println("  CKM_MD2_RSA_PKCS");
351                     pw.println("  CKM_MD5_RSA_PKCS");
352                     pw.println("  CKM_DSA_SHA1");
353                     pw.println("  CKM_ECDSA_SHA1");
354                     pw.println("}");
355                 }
356                 pw.println("attributes(*, CKO_SECRET_KEY, *) = {");
357                 pw.println("  CKA_SENSITIVE = true"); // not possible to read the key
358                 pw.println("  CKA_EXTRACTABLE = false"); // not possible to wrap the key with another key
359                 pw.println("  CKA_ENCRYPT = true");
360                 pw.println("  CKA_DECRYPT = true");
361                 pw.println("  CKA_SIGN = true");
362                 pw.println("  CKA_VERIFY = true");
363                 pw.println("  CKA_WRAP = true");// for unwrapping of session keys,
364                 pw.println("  CKA_UNWRAP = true");// for unwrapping of session keys,
365                 pw.println("}");
366             }
367         }
368         if (log.isDebugEnabled()) {
369             log.debug(baos.toString());
370         }
371         return new ByteArrayInputStream(baos.toByteArray());
372     }

374     private static void replaceCkaLabel(List<String> fileContent, String privateKeyLabel) {
375         // If the attributes file contain a CKA_LABEL without a label, set it to our desired label
376         for (int i = 0; i < fileContent.size(); i++) {
377             if (fileContent.get(i).equals("  CKA_LABEL")) { // just the CKA_LABEL with two spaces before and nothing, not even a space after
378                 if (privateKeyLabel != null && privateKeyLabel.length() > 0) {
379                     final String labelStr = "  CKA_LABEL = 0h" + new String(Hex.encode(privateKeyLabel.getBytes()));
380                     if (log.isDebugEnabled()) {
381                         log.debug("Setting CKA_LABEL to '"+labelStr+"'");
382                     }
383                     fileContent.set(i, labelStr);
384                 } else {
385                     // If there is no privateKeyLabel, we have to remove the bogus empty CKA_LABEL
386                     log.debug("Removing placeholder CKA_LABEL");
387                     fileContent.remove(i); // .remove() is ok since we break out of the loop here, if we didn't this would be bad practice
388                 }
389                 break;
390             }
391         }
392     }

398      * @throws ClassNotFoundException If Sun provider not found
399      * @throws IllegalArgumentException If illegal args passed
400      * @throws SecurityException If security violation

406     private static Provider getSunP11ProviderNoExceptionHandling(final InputStream is) throws ClassNotFoundException, IllegalArgumentException, SecurityException, InstantiationException, IllegalAccessException, InvocationTargetException, NoSuchMethodException {

430     /**
431      * Creating dummy provider just to have C_Initialize executed with multiple
432      * thread argument. {@link Pkcs11Wrapper#getInstance(File)} should call it
433      * before making any PKCS#11 calls. If not then C_Initialize will be called
434      * with null argument causing multi threading to be disabled for the token.
435      * 
436      * There is a boolean in the sun code ensuring that C_Initialize is only
437      * called once.
438      * To check this implementation the p11 spy utils could be used. Check that
439      * it is only one C_Initialize call and that null is not passed.
440      * @param libFile a file with the path of the p11 module on which C_Finalize should
441      * be called.
442      */
443     static void doC_Initialize(final File libFile) {
444         try {
445             getSunP11ProviderNoExceptionHandling( getSunP11ProviderInputStream(-1, libFile, Pkcs11SlotLabelType.SLOT_NUMBER, null, null) );
446         } catch (InvocationTargetException e) {
447             // the p11 module don't like the bogus arguments and throws an exception but we don't bother about this since
448             // C_Initialize has already been called with multithread arguments.
449             log.debug("Get dummy sun provider throws an exception for "+libFile.getPath()+". This is OK.", e);
450         } catch (Exception e) {
451             final String msg = "Error constructing pkcs11 provider: " + e.getMessage();
452             log.error(msg);
453             throw new IllegalStateException(msg, e);
454         }
455     }
