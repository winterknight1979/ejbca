27  /**
28   *
29   * This class wraps sun.security.pkcs11.wrapper.PKCS11, so that we can access the native PKCS11 calls
30   * directly.
31   * A slot list and token labels for each slot is cached so that C_GetSlotList() only has to be called once and
32   * so that C_GetTokenInfo() only has to be called once for each slot.
33   *
34   * The {@link #getInstance(File)} method must be called before any PKCS#11 provider is created.
35   *
36   *  @version $Id: Pkcs11Wrapper.java 27614 2017-12-21 08:55:51Z bastianf $
37   *
38   */
39  public class Pkcs11Wrapper {
40      private static final Logger log = Logger.getLogger(Pkcs11Wrapper.class);
41  
42      private static volatile Map<String, Pkcs11Wrapper> instances = new HashMap<String, Pkcs11Wrapper>();
43      private static final Lock lock = new ReentrantLock();
44      private final Method getSlotListMethod;
45      private final Method getTokenInfoMethod;
46      private final Field labelField;
47      private final Object p11;
48      private final HashMap<Long, char[]> labelMap;
49      private final long slotList[];

51      private Pkcs11Wrapper(final String fileName) {
52          final Class<? extends Object> p11Class;
53          try {
54              p11Class = Class.forName("sun.security.pkcs11.wrapper.PKCS11");
55          } catch (ClassNotFoundException e) {
56              String msg = "Class sun.security.pkcs11.wrapper.PKCS11 was not found locally, could not wrap.";
57              log.error(msg, e);
58              throw new IllegalStateException(msg, e);
59          }
60  
61          try {
62              this.getSlotListMethod = p11Class.getDeclaredMethod("C_GetSlotList", new Class[] { boolean.class });
63          } catch (NoSuchMethodException e) {
64              String msg = "Method C_GetSlotList was not found in class sun.security.pkcs11.wrapper.PKCS11, this may be due to"
65                      + " a change in the underlying library.";
66              log.error(msg, e);
67              throw new IllegalStateException(msg, e);
68          } catch (SecurityException e) {
69              String msg = "Access was denied to method sun.security.pkcs11.wrapper.PKCS11.C_GetSlotList";
70              log.error(msg, e);
71              throw new IllegalStateException(msg, e);
72          }
73          try {
74              this.getTokenInfoMethod = p11Class.getDeclaredMethod("C_GetTokenInfo", new Class[] { long.class });
75          } catch (NoSuchMethodException e) {
76              String msg = "Method C_GetTokenInfo was not found in class sun.security.pkcs11.wrapper.PKCS11, this may be due to"
77                      + " a change in the underlying library.";
78              log.error(msg, e);
79              throw new IllegalStateException(msg, e);
80          } catch (SecurityException e) {
81              String msg = "Access was denied to method sun.security.pkcs11.wrapper.PKCS11.C_GetTokenInfo";
82              log.error(msg, e);
83              throw new IllegalStateException(msg, e);
84          }
85          try {
86              this.labelField = Class.forName("sun.security.pkcs11.wrapper.CK_TOKEN_INFO").getField("label");
87          } catch (NoSuchFieldException e) {
88              String msg = "Field 'label' was not found in class sun.security.pkcs11.wrapper.CK_TOKEN_INFO, this may be due to"
89                      + " a change in the underlying library.";
90              log.error(msg, e);
91              throw new IllegalStateException(msg, e);
92          } catch (SecurityException e) {
93              String msg = "Access was denied to field sun.security.pkcs11.wrapper.CK_TOKEN_INFO.label";
94              log.error(msg, e);
95              throw new IllegalStateException(msg, e);
96          } catch (ClassNotFoundException e) {
97              String msg = "Class sun.security.pkcs11.wrapper.CK_TOKEN_INFO was not found locally, could not wrap.";
98              log.error(msg, e);
99              throw new IllegalStateException(msg, e);
100         }
101         final Method getInstanceMethod;
102         try {
103             getInstanceMethod = p11Class.getDeclaredMethod("getInstance",
104                     new Class[] { String.class, String.class, Class.forName("sun.security.pkcs11.wrapper.CK_C_INITIALIZE_ARGS"), boolean.class });
105         } catch (NoSuchMethodException e) {
106             String msg = "Method getInstance was not found in class sun.security.pkcs11.wrapper.PKCS11.CK_C_INITIALIZE_ARGS, this may be due to"
107                     + " a change in the underlying library.";
108             log.error(msg, e);
109             throw new IllegalStateException(msg, e);
110         } catch (SecurityException e) {
111             String msg = "Access was denied to method sun.security.pkcs11.wrapper.CK_C_INITIALIZE_ARGS.getInstance";
112             log.error(msg, e);
113             throw new IllegalStateException(msg, e);
114         } catch (ClassNotFoundException e) {
115             String msg = "Class sun.security.pkcs11.wrapper.CK_C_INITIALIZE_ARGS was not found locally, could not wrap.";
116             log.error(msg, e);
117             throw new IllegalStateException(msg, e);
118         }
119         try {
120             this.p11 = getInstanceMethod.invoke(null, new Object[] { fileName, "C_GetFunctionList", null, Boolean.FALSE });
121         } catch (IllegalAccessException e) {
122             String msg = "Method sun.security.pkcs11.wrapper.PKCS11.CK_C_INITIALIZE_ARGS.getInstance was not accessible, this may be due to"
123                     + " a change in the underlying library.";
124             log.error(msg, e);
125             throw new IllegalStateException(msg, e);
126         } catch (IllegalArgumentException e) {
127             String msg = "Wrong arguments were passed to sun.security.pkcs11.wrapper.PKCS11.CK_C_INITIALIZE_ARGS.getInstance. This may be due to"
128                     + " a change in the underlying library.";
129             log.error(msg, e);
130             throw new IllegalStateException(msg, e);
131         } catch (InvocationTargetException e) {
132             String msg = "Wrong arguments were passed to sun.security.pkcs11.wrapper.PKCS11.CK_C_INITIALIZE_ARGS.getInstance threw an exception "
133                     + "for log.error(msg, e)";
134             log.error(msg, e);
135             throw new IllegalStateException(msg, e);
136         }
137         this.labelMap = new HashMap<Long, char[]>();
138         this.slotList = C_GetSlotList();
139         for( long id : this.slotList) {
140             this.labelMap.put(Long.valueOf(id), getTokenLabelLocal(id));
141         }
142     }

144     /**
145      * Get an instance of the class.
146      * @param file the p11 .so file.
147      * @return the instance.
148      * @throws IllegalArgumentException on fail
149      */
150     public static Pkcs11Wrapper getInstance(final File file) throws IllegalArgumentException {
151         final String canonicalFileName;
152         try {
153             canonicalFileName = file.getCanonicalPath();
154         } catch (IOException e) {
155             throw new IllegalArgumentException(file+" is not a valid filename.",e );
156         }
157         {
158             final Pkcs11Wrapper storedP11 = instances.get(canonicalFileName);
159             if (storedP11 != null) {
160                 return storedP11;// if instance exist we don't have to wait for lock just grab it.
161             }
162         }
163         try {
164             lock.lock();// wait for lock; some other tread might be creating the instance right now.
165             final Pkcs11Wrapper storedP11 = instances.get(canonicalFileName);
166             if (storedP11 != null) {
167                 return storedP11;// some other thread had already created the instance
168             }
169             // no other thread has created the instance and no other will since this thread is locking.
170             Pkcs11SlotLabel.doC_Initialize(file);// C_Initialize with multithreading args.
171             final Pkcs11Wrapper11Wrapper.html#Pkcs11Wrapper">Pkcs11Wrapper newP11 = new Pkcs11Wrapper(canonicalFileName);
172             instances.put(canonicalFileName, newP11);
173             return newP11;
174         } finally {
175             lock.unlock();// now other threads might get the instance.
176         }
177     }

187     /**
188      * Get the token label of a specific slot ID.
189      * @param slotID the ID of the slot
190      * @return the token label, or null if no matching token was found.
191      */
192     public char[] getTokenLabel(long slotID) {
193         return this.labelMap.get(Long.valueOf(slotID));
194     }

196     private long[] C_GetSlotList() {
197         try {
198             return (long[]) this.getSlotListMethod.invoke(this.p11, new Object[] { Boolean.TRUE });
199         } catch (IllegalAccessException e) {
200             String msg = "Access was denied to method sun.security.pkcs11.wrapper.PKCS11C.GetSlotList, this may be due to"
201                     + " a change in the underlying library.";
202             log.error(msg, e);
203             throw new IllegalStateException(msg, e);
204         } catch (IllegalArgumentException e) {
205             String msg = "Incorrect parameters sent to sun.security.pkcs11.wrapper.PKCS11C.GetSlotList, this may be due to"
206                     + " a change in the underlying library.";
207             log.error(msg, e);
208             throw new IllegalStateException(msg, e);
209         } catch (InvocationTargetException e) {
210             String msg = "Method sun.security.pkcs11.wrapper.PKCS11C.GetSlotList threw an unknown exception.";
211             log.error(msg, e);
212             throw new IllegalStateException(msg, e);

216     private char[] getTokenLabelLocal(long slotID)  {
217         final Object tokenInfo;
218         try {
219             tokenInfo = this.getTokenInfoMethod.invoke(this.p11, new Object[] { Long.valueOf(slotID) });
220         } catch (IllegalAccessException e) {
221             String msg = "Access was denied to method sun.security.pkcs11.wrapper.PKCS11.C_GetTokenInfo, this may be due to"
222                     + " a change in the underlying library.";
223             log.error(msg, e);
224             throw new IllegalStateException(msg, e);
225         } catch (IllegalArgumentException e) {
226             String msg = "Incorrect parameters sent to sun.security.pkcs11.wrapper.PKCS11.C_GetTokenInfo, this may be due to"
227                     + " a change in the underlying library.";
228             log.error(msg, e);
229             throw new IllegalStateException(msg, e);
230         } catch (InvocationTargetException e) {
231             String msg = "Method sun.security.pkcs11.wrapper.PKCS11.C_GetTokenInfo threw an unknown exception.";
232             log.error(msg, e);
233             return null;
234         }
235         if (tokenInfo == null) {
236             return null;
237         }
238         try {
239             String result = String.copyValueOf((char[]) this.labelField.get(tokenInfo));
240             return result.trim().toCharArray();
241         } catch (IllegalArgumentException e) {
242             String msg = "Field sun.security.pkcs11.wrapper.PKCS11.C_GetTokenInfo was not of type sun.security.pkcs11.wrapper.CK_TOKEN_INFO"
243                     + ", this may be due to a change in the underlying library.";
244             log.error(msg, e);
245             throw new IllegalStateException(msg, e);
246         } catch (IllegalAccessException e) {
247             String msg = "Access was denied to field sun.security.pkcs11.wrapper.CK_TOKEN_INFO.label, this may be due to"
248                     + " a change in the underlying library.";
249             log.error(msg, e);
250             throw new IllegalStateException(msg, e);
251         }
252     }
