13  package org.cesecore.keys.token.p11;
14  
15  import java.lang.reflect.InvocationTargetException;
16  import java.util.HashMap;
17  import java.util.Map;
18  
19  import org.apache.log4j.Logger;
20  
21  /**
22   * This enum class represents the typing of a slot label. In PKCS#11 an HSM is split up into separate partitions known 
23   * as 'slots' and each slot contains a token in a 1:1 relationship. In a fixed HSM (such as a PCI HSM or a NetHSM) the slots
24   * can be viewed as partitions on a disk containing a token, while in a SmartCard based solution each slot can be viewed as a 
25   * card reader and the physical card as the token. 
26   * This is highly relevant, because it does not guarantee for any solution that token always resides in the same slot. Some 
27   * fixed HSMs guarantee this (such as SafeNet and Utimaco) while some don't (such as Thales/nCipher). In a smartcard based HSM, naturally
28   * cards will continuously switch slots. 
29   *
30   * For this reason there are at present four different ways of referring to a slot/token, and these are:
31   *  * Slot Number   - the numeric representation of a slot. 0-indexed on certain HSMs, not on others.
32   *  * Slot Index    - The index of the slot. Nearly always the same as the slot number. 
33   *  * Slot Label    - In actuality the label of the token, so will stay constant even if the slot number shifts. 
34   *  * Sun File      - Increasingly rare. Slot/token is defined by an external configuration file
35   *  
36   *  The Pkcs11SlotLabelType is made up of three fields. The first is the key, which is the string format in which the label
37   *  type is persisted to the database and will require an upgrade instruction if it is ever changed. The second is a description
38   *  which can be used for human friendly labels, and the last is a validator used to validate inputed values (such as checking that
39   *  a slot number can be cast to an integer). 
40   * 
41   * @version $Id: Pkcs11SlotLabelType.java 19902 2014-09-30 14:32:24Z anatom $
42   *
43   */
44  
45  public enum Pkcs11SlotLabelType {
46      SLOT_LABEL("SLOT_LABEL", "Slot Label", LabelValidator.class), 
47      SLOT_INDEX("SLOT_INDEX", "Slot Index", IndexValidator.class), 
48      SLOT_NUMBER("SLOT_NUMBER", "Slot Number", NumberValidator.class), 
49      SUN_FILE("SUN_FILE", "Sun configuration file", null);
50  
51      private static final Logger log = Logger.getLogger(Pkcs11SlotLabelType.class);
52      
53      private static final Map<String, Pkcs11SlotLabelType> keyLookUpMap = new HashMap<String, Pkcs11SlotLabelType>();
54      
55      private final String description;
56      private final String key;
57      private final LabelTypeValidator validator;
58  
59      static {
60          for (Pkcs11SlotLabelType type : Pkcs11SlotLabelType.values()) {
61              keyLookUpMap.put(type.getKey(), type);
62          }
63      }
64      
65      private Pkcs11SlotLabelType(String key, String _description, Class<? extends LabelTypeValidator> validator) {
66          this.description = _description;
67          this.key = key;
68          if (validator == null) {
69              this.validator = null;
70          } else {
71              try {
72                  this.validator = validator.getConstructor().newInstance();
73              } catch (InstantiationException | NoSuchMethodException | InvocationTargetException e) {
74                  throw new RuntimeException("Could not instansiate " + validator, e);
75              } catch (IllegalAccessException e) {
76                  throw new RuntimeException("Could not instansiate " + validator, e);
77              }
78          }
79      }
80  
81      @Override
82      public String toString() {
83          return this.description;
84      }
85      
86      /**
87       * 
88       * @return the key of this slot label type
89       */
90      public String getKey() {
91          return key;
92      }
93      
94      /**
95       * 
96       * @return the human friendly description of this slot label type
97       */
98      public String getDescription() {
99          return description;
100     }

103     /**
104      * Returns a Pkcs11SlotLabelType based on a key
105      * 
106      * @param key a key, must belong to one of the predefine types.
107      * @return the Pkcs11SlotLabelType. Returns null if not found. 
108      */
109     public static Pkcs11SlotLabelType getFromKey(String key) {
110         return keyLookUpMap.get(key);
111     }
112     
113     /**
114      * Compares the keys of two slot label types. Defined because .equals for enums are final, which means that
115      * .equals isn't always applicable for serialized objects. 
116      * 
117      * @param otherType the Pkcs11SlotLabelType to compare with
118      * @return true if the two types have the same key.
119      */
120     public boolean isEqual(Pkcs11SlotLabelType otherType) {
121         return this.getKey().equals(otherType.getKey());
122     }
123     
124     /**
125      * Validates a given value, depending on what validator this enum type was instantiated with. If no validator
126      * has been defined, always return true. 
127      * 
128      * @param value the value to be validated
129      * @return true if the value can be used for this type.
130      */
131     public boolean validate(String value) {
132         if(validator != null) {
133             return validator.validate(value);
134         } else {
135             return true;
136         }
137     }

139     /**
140      * An interface defining the private validator classes.
141      * 
142      * @version $Id: Pkcs11SlotLabelType.java 19902 2014-09-30 14:32:24Z anatom $
143      *
144      */
145     private static interface LabelTypeValidator {
146         /**
147          * 
148          * @param value the value to check
149          * @return true if the value is applicable
150          */
151         boolean validate(String value);
152     }
153     
154     /**
155      * Validates true if the inputed string can be cast to long integer.
156      * 
157      * @version $Id: Pkcs11SlotLabelType.java 19902 2014-09-30 14:32:24Z anatom $
158      *
159      */
160     protected static class NumberValidator implements LabelTypeValidator {
161     	
162     	
163     	public NumberValidator() {}
164         
165         @Override
166         public boolean validate(String value) {
167             try {
168                 Long.parseLong(value);
169                 return true;
170             } catch(NumberFormatException e) {
171                 if(log.isDebugEnabled()) {
172                     log.debug(value + " was not a number.", e);
173                 }
174                 return false;
175             }
176         }   
177     }
178 
179     /**
180      * Validates true if the inputed values is an 'i' followed by a string that can be cast to a long integer. 
181      * 
182      * @version $Id: Pkcs11SlotLabelType.java 19902 2014-09-30 14:32:24Z anatom $
183      *
184      */
185     protected static class IndexValidator extends NumberValidator {
186     	
187     	public IndexValidator() {}
188         
189         @Override
190         public boolean validate(String value) {
191             if(value.charAt(0) != 'i') {
192                 if(log.isDebugEnabled()) {
193                     log.debug(value + " did not start with 'i'");
194                 }
195                 return false;
196             }
197             return super.validate(value.substring(1));
198         }
199     }
200     
201     protected static class LabelValidator implements LabelTypeValidator {
202     	
203     	public LabelValidator() {}
204 
205         @Override
206         public boolean validate(String value) {
207            //According to the PKCS#11 standard, the label field can be max 32 chars long
208             if(value.length() > 32) {
209                 if(log.isDebugEnabled()) {
210                     log.debug("Value " + value + " was longer than the permitted 32 characters.");
211                 }
212                 return false;
213             } else {
214                 return true;
215             }
216         }
217         
218     }
