26  /**
27   * Class managing available Crypto Tokens and instantiated Crypto Tokens. 
28   * Each CryptoToken plug-in should register itself by using the method register.
29   * 
30   * @version $Id: CryptoTokenFactory.java 21361 2015-05-26 11:10:24Z jeklund $
31   */
32  public class CryptoTokenFactory {
33  	
34      private static transient Logger log = Logger.getLogger(CryptoTokenFactory.class);
35  
36      /** Registry of available hard ca token classes that can be instantiated. */
37      private Map<String, AvailableCryptoToken> availabletokens = new HashMap<String, AvailableCryptoToken>(4);
38  
39      /** Implementing the Singleton pattern */
40      private static CryptoTokenFactory instance = null;
41  
42      /** Don't allow external creation of this class, implementing the Singleton pattern. */
43      private CryptoTokenFactory() {}
44      
45      /** Get the instance of this singleton 
46       * @return instance*/
47      public synchronized static CryptoTokenFactory instance() {
48          if (instance == null) {
49              instance = new CryptoTokenFactory();
50              /** Can't use class.getName() here because this class is not always available */
51              instance.addAvailableCryptoToken("se.primeKey.caToken.card.PrimeCAToken", "PrimeCAToken", false, true);
52              instance.addAvailableCryptoToken(PKCS11CryptoToken.class.getName(), "PKCS#11", false, true);
53              instance.addAvailableCryptoToken(SoftCryptoToken.class.getName(), "SOFT", true, true);
54              instance.addAvailableCryptoToken(NullCryptoToken.class.getName(), "Null", false, false);
55          }
56          return instance;
57      }

59  	/**
60  	 * Method returning to the system available CryptoToken implementations
61  	 * 
62  	 * @return a Collection (AvailableCryptoToken) of registered plug-ins.
63  	 */
64  	public Collection<AvailableCryptoToken> getAvailableCryptoTokens() {
65  	    return availabletokens.values();	
66  	}
67  
68  	/**
69  	 * Method returning to the available CryptoToken implementations with given classpath.
70  	 * @param classname name
71  	 * 
72  	 * @return the corresponding AvailableCryptoToken or null of classpath couldn't be found
73  	 */
74  	public AvailableCryptoToken getAvailableCryptoToken(final String classname) {
75          if (classname == null) {
76              return null;
77          }
78  	    return (AvailableCryptoToken)availabletokens.get(classname);
79  	}

81  	/**
82  	 * Method registering a crypto token plug-in as available to the system.
83  	 * 
84  	 * @param classname the full classname of the crypto token implementation class
85  	 * @param name the general name used in adminweb-gui.
86  	 * @param translateable indicates if the name should be translated in adminweb-gui
87  	 * @param use indicates if this plug-in should be used.
88  	 * 
89  	 * @return true if registration went successful, false if the classpath could not be found or the classpath was already registered.
90  	 */
91  	/*package*/ boolean addAvailableCryptoToken(final String classname, final String name, final boolean translateable, final boolean use) {
92  		if (log.isTraceEnabled()) {
93  			log.trace(">addAvailableCryptoToken: "+classname);
94  		}
95  		boolean retval = false;	
96  		if (!availabletokens.containsKey(classname)) {
97  		    if (log.isDebugEnabled()) {
98  	            log.debug("CryptoTokenFactory adding available crypto token " + classname);                
99  		    }
100 			if (loadClass(classname)) {
101 				// Add to the available tokens
102 				availabletokens.put(classname, new AvailableCryptoToken(classname, name, translateable, use));         
103 				retval = true;
104 	            if (log.isDebugEnabled()) {
105 	                log.debug("Registered " + classname + " successfully.");                       
106 	            }
107 			} else {
108 				// Normally not an error, since these classes are provided by HSM vendor
109 				log.info(InternalResources.getInstance().getLocalizedMessage("token.inforegisterclasspath", classname));
110 			}
111 		}			
112 		if (log.isTraceEnabled()) {
113 			log.trace("<addAvailableCryptoToken: "+classname);
114 		}
115 		return retval;
116 	}

118 	/**
119      * Method loading a class in order to test if it can be instantiated.
120      * 
121      * @param classname name
122 	 * @return boolean
123      */
124     private boolean loadClass(final String classname){
125         try {           
126         	Thread.currentThread().getContextClassLoader().loadClass(classname).getConstructor().newInstance();       
127             return true;
128         } catch (ClassNotFoundException | NoSuchMethodException e) {
129             log.info(InternalResources.getInstance().getLocalizedMessage("token.classnotfound", classname)); 
130         } catch (InstantiationException | InvocationTargetException e) {
131             log.info(InternalResources.getInstance().getLocalizedMessage("token.errorinstansiate", classname, e.getMessage()));
132         } catch (IllegalAccessException e) {
133             log.error("IllegalAccessException: "+classname, e);
134         } catch (NoClassDefFoundError e) {
135             // This happens more rarely and should be flagged as an error
136             log.error("NoClassDefFoundError: "+classname, e);
137         }
138         return false;
139     }

141     /** Creates a crypto token using reflection to construct the class from classname and initializing the CryptoToken
142      * 
143      * @param inClassname the full classname of the crypto token implementation class
144      * @param properties properties passed to the init method of the CryptoToken
145      * @param data byte data passed to the init method of the CryptoToken
146      * @param cryptoTokenId id passed to the init method of the CryptoToken, the id is user defined and not used internally for anything but logging.
147      * @param tokenName user friendly identifier
148      * @return token
149      * @throws NoSuchSlotException if no slot as defined in properties could be found.
150      */
151     public static final CryptoToken createCryptoToken(final String inClassname, final Properties properties, final byte[] data, final int cryptoTokenId,
152             String tokenName) throws NoSuchSlotException {
153         final boolean allowNonExistingSlot = Boolean.valueOf(properties.getProperty(CryptoToken.ALLOW_NONEXISTING_SLOT_PROPERTY, Boolean.FALSE.toString()));
154         return createCryptoToken(inClassname, properties, data, cryptoTokenId, tokenName, allowNonExistingSlot);
155     }
156 
157     /** Creates a crypto token using reflection to construct the class from classname and initializing the CryptoToken
158      * 
159      * @param inClassname the full classname of the crypto token implementation class
160      * @param properties properties passed to the init method of the CryptoToken
161      * @param data byte data passed to the init method of the CryptoToken
162      * @param cryptoTokenId id passed to the init method of the CryptoToken, the id is user defined and not used internally for anything but logging.
163      * @param tokenName user friendly identifier
164      * @param allowNonExistingSlot if the NoSuchSlotException should be used
165      * @return token
166      * @throws NoSuchSlotException if no slot as defined in properties could be found.
167      */
168     public static final CryptoToken createCryptoToken(final String inClassname, final Properties properties, final byte[] data, final int cryptoTokenId,
169             String tokenName, boolean allowNonExistingSlot) throws NoSuchSlotException {
170         final String classname;
171         if (inClassname != null) {
172             classname = inClassname;
173         } else {
174             classname = NullCryptoToken.class.getName();
175             log.info("This must be an imported CA that is being upgraded. Use NullCryptoToken.");
176         }
177         final CryptoToken token = createTokenFromClass(classname);
178         if (token == null) {
179             log.error("No token. Classpath=" + classname);
180             return null;
181         }
182         try {
183             token.init(properties, data, cryptoTokenId);
184         } catch (NoSuchSlotException e) {
185             final String msg = "Unable to access PKCS#11 slot for crypto token '"+tokenName+"' (" + cryptoTokenId + "). Perhaps the token was removed? " + e.getMessage();
186             if (allowNonExistingSlot) {
187                 log.warn(msg);
188                 if (log.isDebugEnabled()) {
189                     log.debug(msg, e);
190                 }
191             } else {
192                 if (log.isDebugEnabled()) {
193                     log.debug(msg, e);
194                 }
195                 throw e;
196             }
197         } catch (Exception e) {
198             log.error("Error initializing Crypto Token. Classpath=" + classname, e);
199         }
200         token.setTokenName(tokenName);
201         return token;
202     }
203     
204     private static final CryptoToken createTokenFromClass(final String classpath) {
205     	try {
206     		Class<?> implClass = Class.forName(classpath);
207     		Object obj = implClass.getConstructor().newInstance();
208     		return (CryptoToken) obj;
209     	} catch (Throwable e) {
210     		log.error("Error contructing Crypto Token (setting to null). Classpath="+classpath, e);
211     		return null;
212     	}
213     }
