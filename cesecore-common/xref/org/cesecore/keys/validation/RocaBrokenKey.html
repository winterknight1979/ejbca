104     public static boolean isAffected(BigInteger modulus) {
105         for (int i = 0; i < primes.length; i++) {
106             if (BigInteger.ONE.shiftLeft(modulus.remainder(primes[i]).intValue()).and(markers[i]).equals(BigInteger.ZERO)) {
107                 return false;
108             }
109         }
110         return true;
111     }
