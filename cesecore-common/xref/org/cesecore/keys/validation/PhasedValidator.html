17  /**
18   * Type for phased validator (a validator bound to a defined phased of the certificate issuance process, #see  ).
19   * 
20   * @version $Id: PhasedValidator.java 27527 2017-12-12 15:49:13Z mikekushner $
21   *
22   */
23  public interface PhasedValidator {
24  
25      /**
26       * Gets the list of applicable certificate issuance process phase indices ({@link IssuancePhase}).
27       * @return the list of certificate issuance process phase index.
28       */
29      List<Integer> getApplicablePhases();
30      
31      /**
32       * Gets the certificate process phase index ({@link IssuancePhase}).
33       * @return the index.
34       */
35      int getPhase();
36      
37      /**
38       * Sets the certificate process phase index ({@link IssuancePhase}).
39       * @param index the index.
40       */
41      void setPhase(int index);
42  }
