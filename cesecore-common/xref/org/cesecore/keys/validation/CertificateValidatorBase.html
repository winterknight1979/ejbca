14  package org.cesecore.keys.validation;
15  
16  import java.text.ParseException;
17  import java.text.SimpleDateFormat;
18  import java.util.ArrayList;
19  import java.util.Date;
20  import java.util.List;
21  import org.apache.commons.lang.StringUtils;
22  import org.apache.commons.lang.time.DateUtils;
23  import org.apache.log4j.Logger;
24  import org.cesecore.certificates.ca.CAInfo;
25  import org.cesecore.util.ui.DynamicUiModel;
26  
27  /**
28   * @version $Id: CertificateValidatorBase.java 28140 2018-01-30 12:40:30Z
29   *     andresjakobs $
30   */
31  public abstract class CertificateValidatorBase extends ValidatorBase
32      implements CertificateValidator {
33  
34    private static final long serialVersionUID = 1L;
35  
36    /** Class logger. */
37    private static final Logger LOG = Logger.getLogger(KeyValidatorBase.class);
38  
39    /** List of applicable CA types (see {@link #getApplicableCaTypes()}). */
40    protected static List<Integer> applicableCaTypes;
41  
42    /** List of applicable issuance phases (see {@link IssuancePhase}). */
43    protected static List<Integer> applicablePhases;
44  
45    static {
46      applicablePhases = new ArrayList<Integer>();
47      applicablePhases.add(IssuancePhase.PRE_CERTIFICATE_VALIDATION.getIndex());
48      applicablePhases.add(IssuancePhase.CERTIFICATE_VALIDATION.getIndex());
49  
50      applicableCaTypes = new ArrayList<Integer>();
51      applicableCaTypes.add(CAInfo.CATYPE_X509);
52    }
53  
54    /** Dynamic UI model extension. */
55    protected DynamicUiModel uiModel;
56  
57    /** Public constructor needed for deserialization. */
58    public CertificateValidatorBase() {
59      super();
60    }
61  
62    /**
63     * Creates a new instance.
64     *
65     * @param name name
66     */
67    public CertificateValidatorBase(final String name) {
68      super(name);
69    }
70  
71    @Override
72    public List<Integer> getApplicableCaTypes() {
73      return applicableCaTypes;
74    }
75  
76    @Override
77    public void init() {
78      super.init();
79      if (null == data.get(PHASE)) {
80        setPhase(getApplicablePhases().get(0));
81      }
82      if (null == data.get(NOT_BEFORE_CONDITION)) {
83        setNotBeforeCondition(KeyValidatorDateConditions.LESS_THAN.getIndex());
84      }
85      if (null == data.get(NOT_AFTER_CONDITION)) {
86        setNotAfterCondition(KeyValidatorDateConditions.LESS_THAN.getIndex());
87      }
88    }
89  
90    @Override
91    public List<Integer> getApplicablePhases() {
92      return applicablePhases;
93    }
94  
95    @Override
96    public Class<? extends Validator> getValidatorSubType() {
97      return CertificateValidator.class;
98    }
99  
100   @Override
101   public Date getNotBefore() {
102     return (Date) data.get(NOT_BEFORE);
103   }
104 
105   @Override
106   public void setNotBefore(final Date date) {
107     data.put(NOT_BEFORE, date);
108   }
109 
110   @Override
111   public int getNotBeforeCondition() {
112     return ((Integer) data.get(NOT_BEFORE_CONDITION)).intValue();
113   }
114 
115   @Override
116   public void setNotBeforeCondition(final int index) {
117     data.put(NOT_BEFORE_CONDITION, index);
118   }
119 
120   @Override
121   public Date getNotAfter() {
122     return (Date) data.get(NOT_AFTER);
123   }
124 
125   @Override
126   public void setNotAfter(final Date date) {
127     data.put(NOT_AFTER, date);
128   }
129 
130   @Override
131   public void setNotAfterCondition(final int index) {
132     data.put(NOT_AFTER_CONDITION, index);
133   }
134 
135   @Override
136   public int getNotAfterCondition() {
137     return ((Integer) data.get(NOT_AFTER_CONDITION)).intValue();
138   }
139 
140   @Override
141   public String getNotBeforeAsString() {
142     return formatDate(getNotBefore());
143   }
144 
145   @Override
146   public String getNotAfterAsString() {
147     return formatDate(getNotAfter());
148   }
149 
150   @Override
151   public void setNotBeforeAsString(final String formattedDate) {
152     try {
153       setNotBefore(parseDate(formattedDate));
154     } catch (ParseException e) {
155       LOG.debug("Could not parse Date: " + formattedDate);
156     }
157   }
158 
159   @Override
160   public void setNotAfterAsString(final String formattedDate) {
161     try {
162       setNotAfter(parseDate(formattedDate));
163     } catch (ParseException e) {
164       LOG.debug("Could not parse Date: " + formattedDate);
165     }
166   }
167 
168   @Override
169   public DynamicUiModel getDynamicUiModel() {
170     return uiModel;
171   }
172 
173   /**
174    * Formats a date.
175    *
176    * @param date the date
177    * @return the formatted date string.
178    */
179   private String formatDate(final Date date) {
180     String result = StringUtils.EMPTY;
181     if (null != date) {
182       result = new SimpleDateFormat(DATE_FORMAT[0]).format(date);
183     }
184     return result;
185   }
186 
187   /**
188    * Parses a date string with the date format list.
189    *
190    * @param string the formatted date string.
191    * @return the date or null, if the date could not be parsed.
192    * @throws ParseException if the date couldn't be parsed
193    */
194   public static Date parseDate(final String string) throws ParseException {
195     Date result = null;
196     if (StringUtils.isNotBlank(string)) {
197       final String dateString = string.trim();
198       result = DateUtils.parseDate(dateString, DATE_FORMAT);
199     }
200     return result;
201   }
202 }
