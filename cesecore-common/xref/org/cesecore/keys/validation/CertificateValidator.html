13  package org.cesecore.keys.validation;
14  
15  import java.security.cert.Certificate;
16  import java.security.cert.CertificateException;
17  import java.util.List;
18  import org.cesecore.certificates.ca.CA;
19  import org.cesecore.util.ui.DynamicUiModelAware;
20  
21  /**
22   * Base interface for certificate validators. All certificate validators must
23   * implement this interface.
24   *
25   * @version $Id: CertificateValidator.java 28140 2018-01-30 12:40:30Z
26   *     andresjakobs $
27   */
28  public interface CertificateValidator
29      extends Validator, ValidityAwareValidator, DynamicUiModelAware {
30  
31    /** List of accepted date formats for notBefore and notAfter filter. */
32    String[] DATE_FORMAT =
33        new String[] {
34          "yyyy-MM-dd HH:mm:ssZZ", "yyyy-MM-dd HH:mm:ss", "yyyy-MM-dd"
35        };
36  
37    /**
38     * Method that validates the public key.
39     *
40     * @param ca the issuing CA.
41     * @param certificate the certificate to validate.
42     * @param whitelist a whitelist containing all scripts permitted to be
43     *     executed
44     * @return the error messages or an empty list if the certificate was
45     *     validated successfully.
46     * @throws ValidatorNotApplicableException when this validator is not
47     *     applicable for the input, for example CVC certificate instead of X.509
48     *     or other type
49     * @throws ValidationException if the certificate could not be validated by
50     *     the external command (exit code &gt; 0).
51     * @throws CertificateException if one of the certificates could not be
52     *     parsed.
53     */
54    List<String> validate(
55        CA ca, Certificate certificate, ExternalScriptsWhitelist whitelist)
56        throws ValidatorNotApplicableException, ValidationException,
57            CertificateException;
58  }
