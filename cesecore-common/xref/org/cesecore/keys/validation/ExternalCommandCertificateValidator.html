184     @Override
185     public List<String> validate(final CA ca, final Certificate certificate, final ExternalScriptsWhitelist externalScriptsWhitelist)
186             throws ValidatorNotApplicableException, ValidationException, CertificateException {
187         final List<String> messages = new ArrayList<String>();
188         log.debug("Validating certificate with external command " + getExternalCommand());
189         if (log.isDebugEnabled()) {
190             log.debug("Validating certificate with external command (cert):" + certificate);
191         }
192         // Add CA certificate chain, that may be processed.
193         final List<Certificate> certificates = new ArrayList<Certificate>();
194         certificates.add(certificate);
195         final String cmd = getExternalCommand();
196         final List<String> out = new ArrayList<String>();
197         // Run external scripts (is used by publishers as well, writes certificate to disk!).
198         try {
199             out.addAll(runExternalCommandInternal(cmd, externalScriptsWhitelist, certificates));
200         } catch(ExternalProcessException e) {
201             throw new ValidatorNotApplicableException( "External command could not be called, because it does not exit, command can not be found, access was denied, certificate not written, or another error occured: "+e.getMessage());
202         }
203         // Validator was applicable but something bad must have happened, no exit code was returned -> validation failed.
204         boolean broken = false;
205         if (CollectionUtils.isNotEmpty(out)) {
206             try {
207                 if (isLogStandardOut()) {
208                     String stdOutput = null;
209                     for (String str : out) {
210                         if (str.startsWith(ExternalProcessTools.STDOUT_PREFIX)) {
211                             if (stdOutput == null) {
212                                 stdOutput = str;
213                             } else {
214                                 stdOutput += "\n" + str;
215                             }
216                         }
217                     }
218                     if (stdOutput != null) {
219                         log.info("External command logged to STDOUT: "+stdOutput);
220                     }
221                 }
222                 String errOutput = null;
223                 if (isLogErrorOut()) {
224                     for (String str : out) {
225                         if (str.startsWith(ExternalProcessTools.ERROUT_PREFIX)) {
226                             if (errOutput == null) {
227                                 errOutput = str;
228                             } else {
229                                 errOutput += "\n" + str;
230                             }
231                         }
232                     }
233                     if (errOutput != null) {
234                         log.info("External command logged to ERROUT: "+errOutput);
235                     }
236                 }
237                 final int exitCode = Integer.parseInt(out.get(0).replaceFirst(ExternalProcessTools.EXIT_CODE_PREFIX, StringUtils.EMPTY));
238                 if (exitCode != 0 && isFailOnErrorCode()) { // Validation failed: -1 is command could not be found or access denied.
239                     messages.add("Invalid: External command exit code was " + exitCode);
240                     if (errOutput != null) {
241                         messages.add("ERROUT was: " + errOutput);
242                     }
243                 } else if (isFailOnStandardError() && ExternalProcessTools.containsErrout(out)) {
244                     messages.add("Invalid: External command logged to ERROUT. Exit code was " + exitCode);
245                     if (errOutput != null) {
246                         messages.add("ERROUT was: " + errOutput);
247                     }
248                 }
249             } catch(Exception e2) { // In case exit code could not be parsed.
250                 broken = true;
251             }
252         } else {
253             broken = true;
254         }
255         if (broken) {
256             messages.add("Invalid: External command could not be initialized: '" + cmd + "'. Command failed.");
257         }
258         return messages;
259     }

356     /**
357      * Tests the external command with the uploaded test certificate.
358      * @return a list with size &gt; 0 and the exit code in field with index 0 and STDOUT and ERROR appended subsequently.
359      * @throws DynamicUiCallbackException if the external script path does not exist or accessible or the script call fails otherwise.
360      */
361     @SuppressWarnings("unchecked")
362     public List<String> testCommand() throws DynamicUiCallbackException {
363         log.info("Test external command certificate validator: " + getProfileName());
364         final DynamicUiProperty<byte[]> property = (DynamicUiProperty<byte[]>) uiModel.getProperties().get("testPath");
365         final List<String> out = new ArrayList<>();
366         byte[] data = null;
367         String message = null;
368         if (property != null && (data=property.getValue()) != null) {
369             final File file;
370             try {
371                 file = ExternalProcessTools.writeTemporaryFileToDisk(data, "validator_test_cert", /* use .tmp as file extension */ null);
372             } catch (ExternalProcessException e) {
373                 throw new IllegalStateException("Failed to save upload to temporary file", e);
374             }
375             try {
376                 if (!file.canRead()) {
377                     message = intres.getLocalizedMessage("validator.certificate.externalcommand.testfilenopermission", file.getAbsolutePath());
378                 }
379                 if (message == null) {
380                     try {
381                         setTestCertificates(CertTools.getCertsFromPEM(file.getAbsolutePath(), Certificate.class));
382                     } catch(IOException e) {
383                         message = intres.getLocalizedMessage("process.certificate.filenotfound", file.getAbsolutePath());
384                         log.warn(message, e);
385                     } catch(CertificateParsingException e) {
386                         message = intres.getLocalizedMessage("process.certificate.couldnotbeparsed", file.getAbsolutePath());
387                         log.warn(message, e);
388                     }
389                 }
390                 if (message == null) { // Run command.
391                     try {
392                         out.addAll(runExternalCommandInternal(getExternalCommand(), ExternalScriptsWhitelist.permitAll(), getTestCertificates()));
393                         if (log.isDebugEnabled()) {
394                             log.debug("Tested certificate with external command STOUT/ERROUT:" + System.getProperty("line.separator") + out);
395                         }
396                     } catch (CertificateEncodingException e) {
397                         message = intres.getLocalizedMessage("process.certificate.couldnotbeencoded", file.getAbsolutePath());
398                         log.info(message, e);
399                     // 1. command not found, no permission or other exception; 2. not in whitelist.
400                     } catch (ExternalProcessException | ValidatorNotApplicableException e) {
401                         message = e.getMessage();
402                         log.info(message, e);
403                     }
404                 }
405             } finally {
406                 // Delete temporary file (file is written because of file upload).
407                 if (file != null) {
408                     try {
409                         file.delete();
410                     } catch (RuntimeException e) {
411                         log.trace("Could not delete temporary file: " + file.getAbsolutePath(), e);
412                     }
413                 }
414             }
415         } else {
416             message = intres.getLocalizedMessage("validator.certificate.externalcommand.testfilemissing", getExternalCommand());
417             log.info(message);
418         }
419         if (StringUtils.isNotBlank(message)) {
420             throw new DynamicUiCallbackException(message);
421         }
422         return out;
423     }

448     /**
449      * Runs the external command
450      * @param externalCommand the external command.
451      * @param externalScriptsWhitelist whitelist
452      * @param certificates the list of certificates.
453      * @return a string list holding exit code at index 0, and the STDOUT and ERROUT appended.
454      * @throws CertificateEncodingException if the certificates could not be encoded.
455      * @throws ExternalProcessException if the command wasn't found
456      * @throws ValidatorNotApplicableException if external scripts whitelist wasn't permitted
457      */
458     private List<String> runExternalCommandInternal(final String externalCommand, final ExternalScriptsWhitelist externalScriptsWhitelist,
459             final List<Certificate> certificates) throws CertificateEncodingException, ExternalProcessException, ValidatorNotApplicableException {
460         final String cmd = extractCommand(externalCommand);
461         if (!externalScriptsWhitelist.isPermitted(cmd)) {
462              throw new ValidatorNotApplicableException(intres.getLocalizedMessage("process.whitelist.error.notlisted", cmd));
463         }
464         // Test if specified script file exists and is executable (hits files and symbolic links, but no aliases).
465         if (StringUtils.isNotBlank(cmd)) {
466             final File file = new File(cmd);
467             String message;
468             if (!file.exists()) {
469                 message = intres.getLocalizedMessage("process.commandnotfound", cmd);
470                 log.info(message);
471                 throw new ExternalProcessException(message);
472             }
473             if (!file.canExecute()) {
474                 message = intres.getLocalizedMessage("process.commandnopermission", cmd);
475                 log.info(message);
476                 throw new ExternalProcessException(message);
477             }
478         }
479         // Extract arguments and run external script.
480         final List<String> arguments = extractArguments(externalCommand);
481         final List<String> out = new ArrayList<String>();
482         try {
483             out.addAll(ExternalProcessTools.launchExternalCommand(cmd, certificates.get(0).getEncoded(),
484                     isFailOnErrorCode(), isFailOnStandardError(), isLogStandardOut(), isLogErrorOut(), arguments, ExternalCommandCertificateValidator.class.getName()));
485         } catch(ExternalProcessException e) {
486             log.info("Could not call external command '" + cmd + "' with arguments " + arguments + " sucessfully: " + e.getMessage());
487             if (log.isDebugEnabled()) {
488                 log.debug("Failed with exception: ", e);
489             }
490             if (e.getOut() != null) {
491                 out.addAll(e.getOut());
492             }
493         }
494         return out;
495     }

497     /**
498      * Extracts the script path.
499      *
500      * @param cmd the external command.
501      * @return the script path (first token in command).
502      */
503     private final String extractCommand(String cmd) {
504         cmd = cmd.trim();
505         final int index = cmd.indexOf(" ");
506         if (index > 0) {
507             cmd  = cmd.substring(0, index).trim();
508         }
509         if (log.isDebugEnabled()) {
510             log.debug("Command extracted: " + cmd);
511         }
512         return cmd;
513     }
514 
515     /**
516      * Extracts the arguments.
517      *
518      * @param cmd the external command.
519      * @return the list of arguments (second token to end).
520      */
521     private final List<String> extractArguments(String cmd) {
522         cmd = cmd.trim();
523         final List<String> arguments = new ArrayList<String>();
524         final int index = cmd.indexOf(" ");
525         if (index > 0) {
526             arguments.addAll( Arrays.asList(StringUtils.split( cmd.substring(index, cmd.length()).trim(), " ")));
527         }
528         if (log.isDebugEnabled()) {
529             log.debug("Arguments extracted: " + arguments);
530         }
531         return arguments;
532     }
