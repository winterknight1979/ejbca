99      /**
100      * Tests if the factors of the BigInteger modulus are prime.
101      * @param modulus the big integer modulus to test
102      * @return true if the modulus is power of a prime, false otherwise.
103      */
104     protected static boolean isPowerOfPrime(BigInteger modulus) {
105         // The isPowerOfPrime test is copied from org.bouncycastle.crypto.asymmetric.KeyUtils in the BC-FIPS package.
106         // If we move to use the FIPS provider we can use the methods directly instead
107         // --- Begin BC code
108         // Use the same iterations as if we were testing a candidate p or q value with error probability 2^-100
109         int bits = modulus.bitLength();
110         int iterations = bits >= 1536 ? 3
111             : bits >= 1024 ? 4
112             : bits >= 512 ? 7
113             : 50;
114         // SP 800-89 requires use of an approved DRBG.
115 //        SecureRandom testRandom = FipsDRBG.SHA256.fromEntropySource(new SecureRandom(), false)
116 //            .build(Pack.longToBigEndian(System.currentTimeMillis()), false, Strings.toByteArray(Thread.currentThread().toString()));
117         SecureRandom testRandom = new SecureRandom(); // we cheat a little and use regular SecureRandom, which is good
118         Primes.MROutput mr = Primes.enhancedMRProbablePrimeTest(modulus, testRandom, iterations);
119         if (!mr.isProvablyComposite())
120         {
121             // FSM_TRANS:5.16, "FIPS 186-3/SP 800-89 ASSURANCES CHECK", "CONDITIONAL TEST", "FIPS 186-3/SP 800-89 Assurances test failed"
122             log.debug("RSA modulus is not composite");
123             return true;
124         }
125         if (!mr.isNotPrimePower())
126         {
127             // FSM_TRANS:5.16, "FIPS 186-3/SP 800-89 ASSURANCES CHECK", "CONDITIONAL TEST", "FIPS 186-3/SP 800-89 Assurances test failed"
128             log.debug("RSA modulus is a power of a prime");
129             return true;
130         }
131         // --- end BC code
132         return false;
133     }

176     /**
177      * Gets the smallest factor of the positive natural number greater than 2.
178      * @param n the number
179      * @param intFactor factor
180      * @return the smallest factor or 2 for n=0.
181      */
182     protected static final boolean hasSmallerFactorThan(BigInteger n, int intFactor) {
183         //        BigInteger factor = BigInteger.valueOf(intFactor);
184         final BigInteger two = new BigInteger("2");
185         if (intFactor < 3) {
186             return false;
187         }
188         if (n.mod(two).equals(BigInteger.ZERO) && intFactor > 2) {
189             return true;
190         }
191         for (int i = intFactor; i > 2; i = i - 2) {
192             if (n.mod(BigInteger.valueOf(i)).equals(BigInteger.ZERO)) {
193                 return true;
194             }
195         }
196         return false;
197     }

357     /**
358      * Sets the CA/B Forum requirements chapter 6.1.6 for RSA public keys.
359      * @see <a href="https://cabforum.org/wp-content/uploads/CA-Browser-Forum-BR-1.4.2.pdf"> CA/B-Forum Baseline Requirements </a>
360      * 
361      */
362     public void setCABForumBaseLineRequirements142Settings() {
363         // Only apply most important conditions (sequence is Root-CA, Sub-CA, User-Certificate)!
364         // But this is not required at the time, because certificate validity conditions are before
365         // 2014 (now 2017). The minimal modulus size (2048 bits) is the same for all certificate types!
366         setBitLengths(getAvailableBitLengths(2048));
367         final List<Integer> ids = getCertificateProfileIds();
368         if (ids.contains(CertificateProfileConstants.CERTPROFILE_FIXED_ROOTCA)) {
369             // NOOP
370         } else if (ids.contains(CertificateProfileConstants.CERTPROFILE_FIXED_SUBCA)) {
371             // NOOP
372         } else {
373             // NOOP
374         }
375         // Reset to null before setting new values.
376         setPublicKeyExponentMin(null);
377         setPublicKeyExponentMax(null);
378 
379         setPublicKeyExponentOnlyAllowOdd(CAB_FORUM_BLR_142_PUBLIC_EXPONENT_ONLY_ALLOW_ODD);
380         setPublicKeyExponentMin(new BigInteger(CAB_FORUM_BLR_142_PUBLIC_EXPONENT_MIN));
381         setPublicKeyExponentMax(new BigInteger(CAB_FORUM_BLR_142_PUBLIC_EXPONENT_MAX));
382         setPublicKeyModulusOnlyAllowOdd(CAB_FORUM_BLR_142_PUBLIC_MODULUS_ONLY_ALLOW_ODD);
383         setPublicKeyModulusDontAllowPowerOfPrime(CAB_FORUM_BLR_142_PUBLIC_MODULUS_DONT_ALLOW_POWER_OF_PRIME);
384         setPublicKeyModulusMinFactor(CAB_FORUM_BLR_142_PUBLIC_MODULUS_SMALLEST_FACTOR);
385         setPublicKeyModulusMin(null);
386         setPublicKeyModulusMax(null);
387 
388         // Not strictly a requirement according to the Baseline Requirements, but there is no reason to allow ROCA weak keys
389         setPublicKeyModulusDontAllowRocaWeakKeys(true);
390     }

426     /**
427      * Sets the validator minimum allowed public key exponent.
428      * @param value The new minimum public key exponent as BigInteger
429      */
430     public void setPublicKeyExponentMin(BigInteger value) {
431         if (null == value) {
432             data.put(PUBLIC_KEY_EXPONENT_MIN, null);
433         } else if (!(value.compareTo(BigInteger.ZERO) == -1)){
434             if (getPublicKeyExponentMax() == null || value.compareTo(getPublicKeyExponentMax()) < 1) {
435                 data.put(PUBLIC_KEY_EXPONENT_MIN, value.toString());
436             } else {
437                 if (log.isDebugEnabled()) {
438                     final String message = intres.getLocalizedMessage("validator.error.minimum_bigger_log", value, getPublicKeyExponentMax());
439                     log.debug(message);
440                 }
441             }
442         } else {
443             if (log.isDebugEnabled()) {
444                 log.debug(intres.getLocalizedMessage("validator.error.set_key_validator_exp_min", value));
445             }
446         }
447     }

644     @Override
645     public List<String> validate(final PublicKey publicKey, final CertificateProfile certificateProfile) throws ValidatorNotApplicableException, ValidationException {
646         List<String> messages = new ArrayList<String>();
647         if (log.isDebugEnabled()) {
648             log.debug("Validating public key with algorithm " + publicKey.getAlgorithm() + ", format " + publicKey.getFormat() + ", implementation "
649                     + publicKey.getClass().getName());
650         }
651         if (!AlgorithmConstants.KEYALGORITHM_RSA.equals(publicKey.getAlgorithm()) || !(publicKey instanceof RSAPublicKey)) {
652             final String message = "Invalid: Public key algorithm is not RSA or could not be parsed: " + publicKey.getAlgorithm() + ", format "
653                     + publicKey.getFormat();
654             messages.add(message);
655             // Make sure this ends up in the server log
656             log.info(message+", "+publicKey.getClass().getName());
657             throw new ValidatorNotApplicableException(message);
658         }
659         final RSAPublicKey rsaPublicKey = (RSAPublicKey) publicKey;
660         final BigInteger publicKeyExponent = rsaPublicKey.getPublicExponent();
661         final BigInteger publicKeyModulus = rsaPublicKey.getModulus();
662         if (log.isDebugEnabled()) {
663             log.debug("Validate RSA public key with exponent " + publicKeyExponent + " and modulus " + publicKeyModulus);
664         }
665         final int settingsOption = getSettingsTemplate();
666         final int keyLength = KeyTools.getKeyLength(publicKey);
667         if (KeyValidatorSettingsTemplate.USE_CERTIFICATE_PROFILE_SETTINGS.getOption() == settingsOption) {
668             final List<Integer> bitLengths = certificateProfile.getAvailableBitLengthsAsList();
669             if (!bitLengths.contains(Integer.valueOf(keyLength))) {
670                 // Invalid Key in request: Illegal key length, not authorized by certificate profile: 2048.. Please supply a correct request.
671                 messages.add("Invalid: RSA key size/strength: Use one of the following " + bitLengths + ".");
672             }
673         } else {
674             final List<String> bitLengths = getBitLengths();
675             if (!bitLengths.contains(Integer.toString(keyLength))) {
676                 messages.add("Invalid: RSA key size/strength: Use one of the following " + bitLengths + ".");
677             }
678         }
679         if (isPublicKeyExponentOnlyAllowOdd()) {
680             if (publicKeyExponent.mod(BigInteger.valueOf(2L)).compareTo(BigInteger.ZERO) == 0) {
681                 messages.add("Invalid: RSA public key exponent is odd.");
682             } else {
683                 log.trace("isPublicKeyExponentOnlyAllowOdd passed");
684             }
685         }
686         if (null != getPublicKeyExponentMin()) {
687             if (publicKeyExponent.compareTo(getPublicKeyExponentMin()) == -1) {
688                 messages.add("Invalid: RSA public key exponent is smaller than " + getPublicKeyExponentMin());
689             } else {
690                 log.trace("getPublicKeyExponentMin passed");
691             }
692         }
693         if (null != getPublicKeyExponentMax()) {
694             if (publicKeyExponent.compareTo(getPublicKeyExponentMax()) == 1) {
695                 messages.add("Invalid: RSA public key exponent is greater than " + getPublicKeyExponentMax());
696             }
697         }
698         if (isPublicKeyModulusOnlyAllowOdd()) {
699             if (publicKeyModulus.mod(BigInteger.valueOf(2L)).compareTo(BigInteger.ZERO) == 0) {
700                 messages.add("Invalid: RSA public key modulus is odd.");
701             } else {
702                 log.trace("isPublicKeyModulusOnlyAllowOdd passed");
703             }
704         }
705         if (isPublicKeyModulusDontAllowPowerOfPrime()) {
706             if (isPowerOfPrime(publicKeyModulus)) {
707                 messages.add("Invalid: RSA public key modulus is not allowed to be the power of a prime.");
708             } else {
709                 log.trace("isPublicKeyModulusDontAllowPowerOfPrime passed");
710             }
711         }
712         if (isPublicKeyModulusDontAllowRocaWeakKeys()) {
713             if (RocaBrokenKey.isAffected(publicKeyModulus)) {
714                 messages.add("Invalid: RSA public key modulus is a weak key according to CVE-2017-15361.");
715             } else {
716                 log.trace("isPublicKeyModulusDontAllowRocaWeakKeys passed");
717             }
718         }
719         if (null != getPublicKeyModulusMinFactor()) {
720             if (hasSmallerFactorThan(publicKeyModulus, getPublicKeyModulusMinFactor() + 1)) {
721                 messages.add("Invalid: RSA public key modulus smallest factor is less than " + getPublicKeyModulusMinFactor());
722             } else {
723                 log.trace("getPublicKeyModulusMinFactor passed");
724             }
725         }
726         if (null != getPublicKeyModulusMin()) {
727             if (publicKeyModulus.compareTo(getPublicKeyModulusMin()) == -1) {
728                 messages.add("Invalid: RSA public key modulus is smaller than " + getPublicKeyModulusMin());
729             } else {
730                 log.trace("getPublicKeyModulusMin passed");
731             }
732         }
733         if (null != getPublicKeyModulusMax()) {
734             if (publicKeyModulus.compareTo(getPublicKeyModulusMax()) == 1) {
735                 messages.add("Invalid: RSA public key modulus is greater than " + getPublicKeyModulusMax());
736             } else {
737                 log.trace("getPublicKeyModulusMax passed");
738             }
739         }
740 
741         if (log.isDebugEnabled()) {
742             for (String message : messages) {
743                 log.debug(message);
744             }
745         }
746         return messages;
747     }
