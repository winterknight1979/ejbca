133   /**
134    * Tests if the factors of the BigInteger modulus are prime.
135    *
136    * @param modulus the big integer modulus to test
137    * @return true if the modulus is power of a prime, false otherwise.
138    */
139   protected static boolean isPowerOfPrime(final BigInteger modulus) {
140     // The isPowerOfPrime test is copied from
141     // org.bouncycastle.crypto.asymmetric.KeyUtils in the BC-FIPS package.
142     // If we move to use the FIPS provider we can use the methods directly
143     // instead
144     // --- Begin BC code
145     // Use the same iterations as if we were testing a candidate p or q value
146     // with error probability 2^-100
147     int bits = modulus.bitLength();
148     int iterations = 0;
149     for (int its : MILLER_RABIN_THRESHOLDS.keySet()) {
150         if (bits >= its) {
151             iterations = MILLER_RABIN_THRESHOLDS.get(its);
152         }
153     }
154     // SP 800-89 requires use of an approved DRBG.
155     //        SecureRandom testRandom = FipsDRBG.SHA256.fromEntropySource(new
156     // SecureRandom(), false)
157     //            .build(Pack.longToBigEndian(System.currentTimeMillis()),
158     // false, Strings.toByteArray(Thread.currentThread().toString()));
159     SecureRandom testRandom =
160         new SecureRandom(); // we cheat a little and use regular SecureRandom,
161                             // which is good
162     Primes.MROutput mr =
163         Primes.enhancedMRProbablePrimeTest(modulus, testRandom, iterations);
164     if (!mr.isProvablyComposite()) {
165       // FSM_TRANS:5.16, "FIPS 186-3/SP 800-89 ASSURANCES CHECK", "CONDITIONAL
166       // TEST", "FIPS 186-3/SP 800-89 Assurances test failed"
167       LOG.debug("RSA modulus is not composite");
168       return true;
169     }
170     if (!mr.isNotPrimePower()) {
171       // FSM_TRANS:5.16, "FIPS 186-3/SP 800-89 ASSURANCES CHECK", "CONDITIONAL
172       // TEST", "FIPS 186-3/SP 800-89 Assurances test failed"
173       LOG.debug("RSA modulus is a power of a prime");
174       return true;
175     }
176     // --- end BC code
177     return false;
178   }

224   /**
225    * Gets the smallest factor of the positive natural number greater than 2.
226    *
227    * @param n the number
228    * @param intFactor factor
229    * @return the smallest factor or 2 for n=0.
230    */
231   protected static final boolean hasSmallerFactorThan(
232       final BigInteger n, final int intFactor) {
233     //        BigInteger factor = BigInteger.valueOf(intFactor);
234     final BigInteger two = new BigInteger("2");
235     if (intFactor < 3) {
236       return false;
237     }
238     if (n.mod(two).equals(BigInteger.ZERO) && intFactor > 2) {
239       return true;
240     }
241     for (int i = intFactor; i > 2; i = i - 2) {
242       if (n.mod(BigInteger.valueOf(i)).equals(BigInteger.ZERO)) {
243         return true;
244       }
245     }
246     return false;
247   }

493   /**
494    * Sets the CA/B Forum requirements chapter 6.1.6 for RSA public keys.
495    *
496    * @see <a
497    *     href="https://cabforum.org/wp-content/uploads/CA-Browser-Forum-BR-1.4.2.pdf">
498    *     CA/B-Forum Baseline Requirements </a>
499    */
500   public void setCABForumBaseLineRequirements142Settings() {
501     // Only apply most important conditions (sequence is Root-CA, Sub-CA,
502     // User-Certificate)!
503     // But this is not required at the time, because certificate validity
504     // conditions are before
505     // 2014 (now 2017). The minimal modulus size (2048 bits) is the same for all
506     // certificate types!
507     setBitLengths(getAvailableBitLengths(MIN_KEY_LENGTH));
508     final List<Integer> ids = getCertificateProfileIds();
509     if (ids.contains(CertificateProfileConstants.CERTPROFILE_FIXED_ROOTCA)) {
510       // NOOP
511     } else if (ids.contains(
512         CertificateProfileConstants.CERTPROFILE_FIXED_SUBCA)) {
513       // NOOP
514     } else {
515       // NOOP
516     }
517     // Reset to null before setting new values.
518     setPublicKeyExponentMin(null);
519     setPublicKeyExponentMax(null);
520 
521     setPublicKeyExponentOnlyAllowOdd(
522         CAB_FORUM_BLR_142_PUBLIC_EXPONENT_ONLY_ALLOW_ODD);
523     setPublicKeyExponentMin(
524         new BigInteger(CAB_FORUM_BLR_142_PUBLIC_EXPONENT_MIN));
525     setPublicKeyExponentMax(
526         new BigInteger(CAB_FORUM_BLR_142_PUBLIC_EXPONENT_MAX));
527     setPublicKeyModulusOnlyAllowOdd(
528         CAB_FORUM_BLR_142_PUBLIC_MODULUS_ONLY_ALLOW_ODD);
529     setPublicKeyModulusDontAllowPowerOfPrime(
530         CAB_FORUM_BLR_142_PUBLIC_MODULUS_DONT_ALLOW_POWER_OF_PRIME);
531     setPublicKeyModulusMinFactor(
532         CAB_FORUM_BLR_142_PUBLIC_MODULUS_SMALLEST_FACTOR);
533     setPublicKeyModulusMin(null);
534     setPublicKeyModulusMax(null);
535 
536     // Not strictly a requirement according to the Baseline Requirements, but
537     // there is no reason to allow ROCA weak keys
538     setPublicKeyModulusDontAllowRocaWeakKeys(true);
539   }

602   /**
603    * Sets the validator minimum allowed public key exponent.
604    *
605    * @param value The new minimum public key exponent as BigInteger
606    */
607   public void setPublicKeyExponentMin(final BigInteger value) {
608     if (null == value) {
609       data.put(PUBLIC_KEY_EXPONENT_MIN, null);
610     } else if (!(value.compareTo(BigInteger.ZERO) == -1)) {
611       if (getPublicKeyExponentMax() == null
612           || value.compareTo(getPublicKeyExponentMax()) < 1) {
613         data.put(PUBLIC_KEY_EXPONENT_MIN, value.toString());
614       } else {
615         if (LOG.isDebugEnabled()) {
616           final String message =
617               INTRES.getLocalizedMessage(
618                   "validator.error.minimum_bigger_log",
619                   value,
620                   getPublicKeyExponentMax());
621           LOG.debug(message);
622         }
623       }
624     } else {
625       if (LOG.isDebugEnabled()) {
626         LOG.debug(
627             INTRES.getLocalizedMessage(
628                 "validator.error.set_key_validator_exp_min", value));
629       }
630     }
631   }

907   @Override
908   public List<String> validate(
909       final PublicKey publicKey, final CertificateProfile certificateProfile)
910       throws ValidatorNotApplicableException, ValidationException {
911     List<String> messages = new ArrayList<String>();
912     if (LOG.isDebugEnabled()) {
913       LOG.debug(
914           "Validating public key with algorithm "
915               + publicKey.getAlgorithm()
916               + ", format "
917               + publicKey.getFormat()
918               + ", implementation "
919               + publicKey.getClass().getName());
920     }
921     if (!AlgorithmConstants.KEYALGORITHM_RSA.equals(publicKey.getAlgorithm())
922         || !(publicKey instanceof RSAPublicKey)) {
923       final String message =
924           "Invalid: Public key algorithm is not RSA or could not be parsed: "
925               + publicKey.getAlgorithm()
926               + ", format "
927               + publicKey.getFormat();
928       messages.add(message);
929       // Make sure this ends up in the server log
930       LOG.info(message + ", " + publicKey.getClass().getName());
931       throw new ValidatorNotApplicableException(message);
932     }
933     final RSAPublicKey rsaPublicKey = (RSAPublicKey) publicKey;
934     final BigInteger publicKeyExponent = rsaPublicKey.getPublicExponent();
935     final BigInteger publicKeyModulus = rsaPublicKey.getModulus();
936     if (LOG.isDebugEnabled()) {
937       LOG.debug(
938           "Validate RSA public key with exponent "
939               + publicKeyExponent
940               + " and modulus "
941               + publicKeyModulus);
942     }
943     final int settingsOption = getSettingsTemplate();
944     final int keyLength = KeyTools.getKeyLength(publicKey);
945     if (KeyValidatorSettingsTemplate.USE_CERTIFICATE_PROFILE_SETTINGS
946             .getOption()
947         == settingsOption) {
948       final List<Integer> bitLengths =
949           certificateProfile.getAvailableBitLengthsAsList();
950       if (!bitLengths.contains(Integer.valueOf(keyLength))) {
951         // Invalid Key in request: Illegal key length, not authorized by
952         // certificate profile: 2048.. Please supply a correct request.
953         messages.add(
954             "Invalid: RSA key size/strength: Use one of the following "
955                 + bitLengths
956                 + ".");
957       }
958     } else {
959       final List<String> bitLengths = getBitLengths();
960       if (!bitLengths.contains(Integer.toString(keyLength))) {
961         messages.add(
962             "Invalid: RSA key size/strength: Use one of the following "
963                 + bitLengths
964                 + ".");
965       }
966     }
967     if (isPublicKeyExponentOnlyAllowOdd()) {
968       if (publicKeyExponent
969               .mod(BigInteger.valueOf(2L))
970               .compareTo(BigInteger.ZERO)
971           == 0) {
972         messages.add("Invalid: RSA public key exponent is odd.");
973       } else {
974         LOG.trace("isPublicKeyExponentOnlyAllowOdd passed");
975       }
976     }
977     if (null != getPublicKeyExponentMin()) {
978       if (publicKeyExponent.compareTo(getPublicKeyExponentMin()) == -1) {
979         messages.add(
980             "Invalid: RSA public key exponent is smaller than "
981                 + getPublicKeyExponentMin());
982       } else {
983         LOG.trace("getPublicKeyExponentMin passed");
984       }
985     }
986     if (null != getPublicKeyExponentMax()) {
987       if (publicKeyExponent.compareTo(getPublicKeyExponentMax()) == 1) {
988         messages.add(
989             "Invalid: RSA public key exponent is greater than "
990                 + getPublicKeyExponentMax());
991       }
992     }
993     if (isPublicKeyModulusOnlyAllowOdd()) {
994       if (publicKeyModulus
995               .mod(BigInteger.valueOf(2L))
996               .compareTo(BigInteger.ZERO)
997           == 0) {
998         messages.add("Invalid: RSA public key modulus is odd.");
999       } else {
1000         LOG.trace("isPublicKeyModulusOnlyAllowOdd passed");
1001       }
1002     }
1003     if (isPublicKeyModulusDontAllowPowerOfPrime()) {
1004       if (isPowerOfPrime(publicKeyModulus)) {
1005         messages.add(
1006             "Invalid: RSA public key modulus is not allowed to be the power of"
1007                 + " a prime.");
1008       } else {
1009         LOG.trace("isPublicKeyModulusDontAllowPowerOfPrime passed");
1010       }
1011     }
1012     if (isPublicKeyModulusDontAllowRocaWeakKeys()) {
1013       if (RocaBrokenKey.isAffected(publicKeyModulus)) {
1014         messages.add(
1015             "Invalid: RSA public key modulus is a weak key according to"
1016                 + " CVE-2017-15361.");
1017       } else {
1018         LOG.trace("isPublicKeyModulusDontAllowRocaWeakKeys passed");
1019       }
1020     }
1021     if (null != getPublicKeyModulusMinFactor()) {
1022       if (hasSmallerFactorThan(
1023           publicKeyModulus, getPublicKeyModulusMinFactor() + 1)) {
1024         messages.add(
1025             "Invalid: RSA public key modulus smallest factor is less than "
1026                 + getPublicKeyModulusMinFactor());
1027       } else {
1028         LOG.trace("getPublicKeyModulusMinFactor passed");
1029       }
1030     }
1031     if (null != getPublicKeyModulusMin()) {
1032       if (publicKeyModulus.compareTo(getPublicKeyModulusMin()) == -1) {
1033         messages.add(
1034             "Invalid: RSA public key modulus is smaller than "
1035                 + getPublicKeyModulusMin());
1036       } else {
1037         LOG.trace("getPublicKeyModulusMin passed");
1038       }
1039     }
1040     if (null != getPublicKeyModulusMax()) {
1041       if (publicKeyModulus.compareTo(getPublicKeyModulusMax()) == 1) {
1042         messages.add(
1043             "Invalid: RSA public key modulus is greater than "
1044                 + getPublicKeyModulusMax());
1045       } else {
1046         LOG.trace("getPublicKeyModulusMax passed");
1047       }
1048     }
1049 
1050     if (LOG.isDebugEnabled()) {
1051       for (String message : messages) {
1052         LOG.debug(message);
1053       }
1054     }
1055     return messages;
1056   }
