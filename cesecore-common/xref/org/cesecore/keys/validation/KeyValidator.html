14  package org.cesecore.keys.validation;
15  
16  import java.security.PublicKey;
17  import java.util.List;
18  
19  import org.cesecore.certificates.certificateprofile.CertificateProfile;
20  import org.cesecore.util.ui.DynamicUiModelAware;
21  
22  /**
23   * Base interface for key validators. All key validators must implement this interface.
24   * 
25   * @version $Id: KeyValidator.java 28140 2018-01-30 12:40:30Z andresjakobs $
26   *
27   */
28  public interface KeyValidator extends Validator, ValidityAwareValidator, DynamicUiModelAware {
29      
30      /** List of accepted date formats for notBefore and notAfter filter. */
31      static final String[] DATE_FORMAT = new String[] { "yyyy-MM-dd HH:mm:ssZZ", "yyyy-MM-dd HH:mm:ss", "yyyy-MM-dd" };
32      
33      /**
34       * Method that validates the public key.
35       * 
36       * @param publicKey the public key to validate.
37       * @param certificateProfiles the Certificate Profile as input for validation
38       * @return the error messages or an empty list if the public key was validated successfully.
39       * @throws ValidatorNotApplicableException when this validator is not applicable for the input, for example ECC keys as input to an RSA key validator
40       * @throws ValidationException if the certificate issuance MUST be aborted.
41       */
42      List<String> validate(PublicKey publicKey, CertificateProfile certificateProfiles) throws ValidatorNotApplicableException, ValidationException;    
43  }
