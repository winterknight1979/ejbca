13  package org.cesecore.keys.validation;
14  
15  import java.util.ArrayList;
16  import java.util.Collection;
17  import java.util.HashMap;
18  import java.util.List;
19  import java.util.Map;
20  import java.util.ServiceLoader;
21  
22  import org.apache.commons.collections.CollectionUtils;
23  
24  /**
25   * Reads in the implementations of the Validator interface.  
26   * 
27   * @version $Id: ValidatorFactory.java 27845 2018-01-10 15:15:37Z mikekushner $
28   *
29   */
30  public enum ValidatorFactory {
31      INSTANCE;
32  
33      private Map<String, Validator> identifierToImplementationMap = new HashMap<>();
34  
35      private ValidatorFactory() {
36          ServiceLoader<Validator> svcloader = ServiceLoader.load(Validator.class);
37          for(Validator type : svcloader) {
38              type.initialize();
39              identifierToImplementationMap.put(type.getValidatorTypeIdentifier(), type);
40          }
41      }
42      
43      public Collection<Validator> getAllImplementations() {
44          return identifierToImplementationMap.values();
45      }
46      
47      public Collection<Validator> getAllImplementations(final List<Class<?>> excludeClasses) {
48          if (CollectionUtils.isNotEmpty(excludeClasses)) {
49              final Collection<Validator> result = new ArrayList<Validator>();
50              for (Validator validator : getAllImplementations()) {
51                  if (!excludeClasses.contains(validator.getClass())) {
52                      result.add(validator);
53                  }
54              }
55              return result;
56          } else {
57              return getAllImplementations();
58          }
59      }
60      
61      public Validator getArcheType(String identifier) {
62          return identifierToImplementationMap.get(identifier).clone();
63      }
64      
65     
66  }
