13  package org.cesecore.keys.validation;
14  
15  import java.util.ArrayList;
16  import java.util.Collection;
17  import java.util.HashMap;
18  import java.util.List;
19  import java.util.Map;
20  import java.util.ServiceLoader;
21  import org.apache.commons.collections.CollectionUtils;
22  
23  /**
24   * Reads in the implementations of the Validator interface.
25   *
26   * @version $Id: ValidatorFactory.java 27845 2018-01-10 15:15:37Z mikekushner $
27   */
28  public enum ValidatorFactory {
29      /** Singleton instance. */
30    INSTANCE;
31  
32      /** Lookup table. */
33    private Map<String, Validator> identifierToImplementationMap =
34        new HashMap<>();
35  
36    ValidatorFactory() {
37      ServiceLoader<Validator> svcloader = ServiceLoader.load(Validator.class);
38      for (Validator type : svcloader) {
39        type.initialize();
40        identifierToImplementationMap.put(
41            type.getValidatorTypeIdentifier(), type);
42      }
43    }
44  
45    /**
46     * @return All implementations
47     */
48    public Collection<Validator> getAllImplementations() {
49      return identifierToImplementationMap.values();
50    }
51  
52    /**
53     * @param excludeClasses Exclusions
54     * @return Implementations
55     */
56    public Collection<Validator> getAllImplementations(
57        final List<Class<?>> excludeClasses) {
58      if (CollectionUtils.isNotEmpty(excludeClasses)) {
59        final Collection<Validator> result = new ArrayList<Validator>();
60        for (Validator validator : getAllImplementations()) {
61          if (!excludeClasses.contains(validator.getClass())) {
62            result.add(validator);
63          }
64        }
65        return result;
66      } else {
67        return getAllImplementations();
68      }
69    }
70  
71    /**
72     * @param identifier ID
73     * @return Archetype
74     */
75    public Validator getArcheType(final String identifier) {
76      return identifierToImplementationMap.get(identifier).clone();
77    }
78  }
