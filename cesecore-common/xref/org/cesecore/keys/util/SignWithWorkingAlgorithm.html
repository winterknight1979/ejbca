13  package org.cesecore.keys.util;
14  
15  import java.security.NoSuchProviderException;
16  import java.security.Provider;
17  import java.security.Security;
18  import java.util.HashMap;
19  import java.util.List;
20  import java.util.Map;
21  import java.util.concurrent.locks.Lock;
22  import java.util.concurrent.locks.ReentrantLock;
23  import org.apache.log4j.Logger;
24  
25  /**
26   * Call {@link #doSignTask(List, Provider, ISignOperation)} or {@link
27   * #doSignTask(List, String, ISignOperation)} when you want to sign with any
28   * working algorithm in the list. This is usable when working with HSMs.
29   * Different HSMs may support different algorithms. Use this class when you just
30   * want to sign with any algorithm supported by the HSM. Example of this: -
31   * Signing of certificate to be used for the p11 certificate object
32   * corresponding to the p11 key. - Signing of CSR when no particular algorithm
33   * is required by the receiver. - Test signing, just checking that the key is
34   * working.
35   *
36   * @version $Id: SignWithWorkingAlgorithm.java 24555 2016-10-21 13:42:23Z anatom
37   *     $
38   */
39  public final class SignWithWorkingAlgorithm {
40    /** Log4j instance. */
41    private static final Logger LOG =
42        Logger.getLogger(SignWithWorkingAlgorithm.class);
43  /** Map. */
44    private static final Map<Integer, SignWithWorkingAlgorithm> INSTANCE_MAP =
45        new HashMap<>();
46    /** Provider. */
47    private final Provider provider;
48    /** List of available algos. */
49    private final List<String> availableSignAlgorithms;
50    /** Alg. */
51    private String signAlgorithm;
52    /** Lock. */
53    private final Lock lock;
54  
55    /**
56     * Finds the registered provider from sProvider and calls {@link
57     * #doSignTask(List, Provider, ISignOperation)}.
58     *
59     * @param availableSignAlgorithms algorithms to choose from.
60     * @param sProvider provider name
61     * @param operation operation the performs the signing, is an instance of
62     *     ISignOperation for example KeyTools.SignDataOperation
63     * @return true if the signing was done.
64     * @throws NoSuchProviderException if the provider is not found.
65     * @throws TaskWithSigningException thrown if {@link
66     *     ISignOperation#taskWithSigning(String, Provider)} is failing.
67     */
68    public static boolean doSignTask(
69        final List<String> availableSignAlgorithms,
70        final String sProvider,
71        final ISignOperation operation)
72        throws NoSuchProviderException, TaskWithSigningException {
73      final Provider provider = Security.getProvider(sProvider);
74      if (provider == null) {
75        throw new NoSuchProviderException();
76      }
77      return doSignTask(availableSignAlgorithms, provider, operation);
78    }
79    /**
80     * First time each algorithm in availableSignAlgorithms are tried until the
81     * {@link ISignOperation#taskWithSigning(String, Provider)} is successfully
82     * completed. The working algorithm is saved after the first time. Succeeding
83     * calls with same availableSignAlgorithms and provider will directly use the
84     * algorithm that was working the first time.
85     *
86     * @param availableSignAlgorithms algorithms to choose from.
87     * @param provider provider
88     * @param operation operation that performs the signing
89     * @return true if the signing was done.
90     * @throws TaskWithSigningException thrown if {@link
91     *     ISignOperation#taskWithSigning(String, Provider)} is failing.
92     */
93    public static boolean doSignTask(
94        final List<String> availableSignAlgorithms,
95        final Provider provider,
96        final ISignOperation operation)
97        throws TaskWithSigningException {
98      final Integer mapKey =
99          Integer.valueOf(
100             availableSignAlgorithms.hashCode() ^ provider.hashCode());
101     final SignWithWorkingAlgorithm instance;
102     synchronized (INSTANCE_MAP) {
103       final SignWithWorkingAlgorithm waitInstance = INSTANCE_MAP.get(mapKey);
104       if (waitInstance == null) {
105         instance =
106             new SignWithWorkingAlgorithm(provider, availableSignAlgorithms);
107         INSTANCE_MAP.put(mapKey, instance);
108       } else {
109         instance = waitInstance;
110       }
111     }
112     return instance.tryOutWorkingAlgorithm(operation);
113   }
114 
115   /**
116    * @param aProvider Provider
117    * @param theAvailableSignAlgorithms Algs
118    */
119   private SignWithWorkingAlgorithm(
120       final Provider aProvider,
121       final List<String> theAvailableSignAlgorithms) {
122     this.provider = aProvider;
123     this.lock = new ReentrantLock();
124     this.availableSignAlgorithms = theAvailableSignAlgorithms;
125   }
126 
127   private boolean tryOutWorkingAlgorithm(final ISignOperation operation)
128       throws TaskWithSigningException {
129     if (this.signAlgorithm != null) {
130       operation.taskWithSigning(this.signAlgorithm, this.provider);
131       return true;
132     }
133     this.lock.lock();
134     try {
135       if (this.signAlgorithm != null) {
136         operation.taskWithSigning(this.signAlgorithm, this.provider);
137         return true;
138       }
139       for (final String trySignAlgorithm : this.availableSignAlgorithms) {
140         try {
141           operation.taskWithSigning(trySignAlgorithm, this.provider);
142         } catch (final Exception e) {
143           LOG.info(
144               String.format(
145                   "Signature algorithm '%s' not working for provider '%s'."
146                       + " Exception: %s",
147                   trySignAlgorithm, this.provider, e.getMessage()));
148           continue;
149         }
150         LOG.info(
151             String.format(
152                 "Signature algorithm '%s' working for provider '%s'.",
153                 trySignAlgorithm, this.provider));
154         this.signAlgorithm = trySignAlgorithm;
155         return true;
156       }
157       LOG.info(
158           String.format(
159               "No valid signing algorithm found for the provider '%s'.",
160               this.provider));
161       return false;
162     } finally {
163       this.lock.unlock();
164     }
165   }
166 }
