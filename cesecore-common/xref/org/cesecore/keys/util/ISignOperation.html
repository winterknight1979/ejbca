13  package org.cesecore.keys.util;
14  
15  import java.security.Provider;
16  import java.util.List;
17  
18  /**
19   * An object of a class implementing this interface must be constructed before
20   * using {@link SignWithWorkingAlgorithm}.
21   *
22   * @version $Id: ISignOperation.java 22408 2015-12-08 20:19:48Z primelars $
23   */
24  public interface ISignOperation {
25    /**
26     * This method must implement implement a task that is signing something. It
27     * is used when there is no requirement on which signing algorithm to use. The
28     * algorithm is chosen from a list of algorithms. The first time the method is
29     * called we have to find an algorithm that is working for the used provider
30     * (HSM). This is achieved by trying algorithms in the list until a working
31     * one is found. After this method has been successfully called succeeding
32     * calls will use the algorithm that worked. Call {@link
33     * SignWithWorkingAlgorithm#doSignTask(List, Provider, ISignOperation)} when
34     * {@link #taskWithSigning(String, Provider)} should be executed.
35     *
36     * @param signAlgorithm algorithm
37     * @param provider provider
38     * @throws TaskWithSigningException thrown if the signing can not be done.
39     */
40    void taskWithSigning(String signAlgorithm, Provider provider)
41        throws TaskWithSigningException;
42  }
