149   /**
150    * Generates a keypair.
151    *
152    * @param keySpec string specification of keys to generate, typical value is
153    *     2048 for RSA keys, 1024 for DSA keys, secp256r1 for ECDSA keys, or null
154    *     if algspec is to be used.
155    * @param algSpec AlgorithmParameterSpec of keys to generate, typically an
156    *     EXParameterSpec for EC keys, or null if keySpec is to be used.
157    * @param keyAlg algorithm of keys to generate, typical value is RSA, DSA or
158    *     ECDSA, see AlgorithmConstants.KEYALGORITHM_XX
159    * @see org.cesecore.certificates.util.AlgorithmConstants
160    * @see org.bouncycastle.asn1.x9.X962NamedCurves
161    * @see org.bouncycastle.asn1.nist.NISTNamedCurves
162    * @see org.bouncycastle.asn1.sec.SECNamedCurves
163    * @return KeyPair the generated keypair
164    * @throws InvalidAlgorithmParameterException if the given parameters are
165    *     inappropriate for this key pair generator.
166    * @see org.cesecore.certificates.util.AlgorithmConstants#KEYALGORITHM_RSA
167    */
168   public static KeyPair genKeys(
169       final String keySpec,
170       final AlgorithmParameterSpec algSpec,
171       final String keyAlg)
172       throws InvalidAlgorithmParameterException {
173     if (LOG.isTraceEnabled()) {
174       LOG.trace(">genKeys(" + keySpec + ", " + keyAlg + ")");
175     }
176 
177     final KeyPairGenerator keygen;
178     try {
179       keygen =
180           KeyPairGenerator.getInstance(
181               keyAlg, BouncyCastleProvider.PROVIDER_NAME);
182     } catch (NoSuchAlgorithmException e) {
183       throw new IllegalStateException(
184           "Algorithm " + keyAlg + " was not recognized.", e);
185     } catch (NoSuchProviderException e) {
186       throw new IllegalStateException(
187           "BouncyCastle was not found as a provider.", e);
188     }
189     if (StringUtils.equals(keyAlg, AlgorithmConstants.KEYALGORITHM_ECDSA)
190         || StringUtils.equals(keyAlg, AlgorithmConstants.KEYALGORITHM_EC)) {
191       if ((keySpec != null) && !StringUtils.equals(keySpec, "implicitlyCA")) {
192         LOG.debug("Generating named curve ECDSA key pair: " + keySpec);
193         // Check if we have an OID for this named curve
194         if (ECUtil.getNamedCurveOid(keySpec) != null) {
195           ECGenParameterSpec bcSpec = new ECGenParameterSpec(keySpec);
196           keygen.initialize(bcSpec, new SecureRandom());
197         } else {
198           if (LOG.isDebugEnabled()) {
199             LOG.debug(
200                 "Curve did not have an OID in BC, trying to pick up Parameter"
201                     + " spec: "
202                     + keySpec);
203           }
204           // This may be a new curve without OID, like curve25519 and we have to
205           // do something a bit different
206           X9ECParameters ecP = CustomNamedCurves.getByName(keySpec);
207           if (ecP == null) {
208             throw new InvalidAlgorithmParameterException(
209                 "Can not generate EC curve, no OID and no ECParameters found: "
210                     + keySpec);
211           }
212           org.bouncycastle.jce.spec.ECParameterSpec ecSpec =
213               new org.bouncycastle.jce.spec.ECParameterSpec(
214                   ecP.getCurve(),
215                   ecP.getG(),
216                   ecP.getN(),
217                   ecP.getH(),
218                   ecP.getSeed());
219           keygen.initialize(ecSpec, new SecureRandom());
220         }
221         // The old code should work in BC v1.50b6 and later, but in versions
222         // prior to that the below produces a key with explicit parameter
223         // encoding instead of named curves.
224         // There is a test for this in KeyToolsTest.testGenKeysECDSAx9
225         //                ecSpec = ECNamedCurveTable.getParameterSpec(keySpec);
226         //                if (ecSpec == null) {
227         //                    throw new
228         // InvalidAlgorithmParameterException("keySpec " + keySpec + " is
229         // invalid for ECDSA.");
230         //                }
231         //                keygen.initialize(ecSpec, new SecureRandom());
232       } else if (algSpec != null) {
233         LOG.debug(
234             "Generating ECDSA key pair from AlgorithmParameterSpec: "
235                 + algSpec);
236         keygen.initialize(algSpec, new SecureRandom());
237       } else if (StringUtils.equals(keySpec, "implicitlyCA")) {
238         LOG.debug("Generating implicitlyCA encoded ECDSA key pair");
239         // If the keySpec is null, we have "implicitlyCA" defined EC parameters
240         // The parameters were already installed when we installed the provider
241         // We just make sure that ecSpec == null here
242         keygen.initialize(null, new SecureRandom());
243       } else {
244         throw new InvalidAlgorithmParameterException(
245             "No keySpec no algSpec and no implicitlyCA specified");
246       }
247     } else if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_ECGOST3410)) {
248       final AlgorithmParameterSpec ecSpec;
249       if (keySpec != null) {
250         LOG.debug("Generating keys from given key specifications : " + keySpec);
251         ecSpec = ECGOST3410NamedCurveTable.getParameterSpec(keySpec);
252         if (ecSpec == null) {
253           throw new InvalidAlgorithmParameterException(
254               "Key specification " + keySpec + " is invalid for ECGOST3410");
255           }
256       } else if (algSpec != null) {
257         LOG.debug(
258             "Generating keys from given algorithm parameters : " + algSpec);
259         ecSpec = algSpec;
260       } else {
261         throw new InvalidAlgorithmParameterException(
262             "No key or algorithm specifications");
263       }
264       keygen.initialize(ecSpec, new SecureRandom());
265     } else if (keyAlg.equals(AlgorithmConstants.KEYALGORITHM_DSTU4145)) {
266       final AlgorithmParameterSpec ecSpec;
267       if (keySpec != null) {
268         LOG.debug("Generating keys from given key specifications : " + keySpec);
269         ecSpec = dstuOidToAlgoParams(keySpec);
270         if (ecSpec == null) {
271           throw new InvalidAlgorithmParameterException(
272               "Key specification " + keySpec + " is invalid for DSTU4145");
273           }
274       } else if (algSpec != null) {
275         LOG.debug(
276             "Generating keys from given algorithm parameters : " + algSpec);
277         ecSpec = algSpec;
278       } else {
279         throw new InvalidAlgorithmParameterException(
280             "No key or algorithm specifications");
281       }
282       keygen.initialize(ecSpec, new SecureRandom());
283     } else if (keySpec.startsWith("DSA")) {
284       // DSA key with "DSA" in keyspec
285       final int keysize = Integer.parseInt(keySpec.substring(3));
286       keygen.initialize(keysize);
287     } else {
288       // RSA or DSA key where keyspec is simply the key length
289       final int keysize = Integer.parseInt(keySpec);
290       keygen.initialize(keysize);
291     }
292 
293     final KeyPair keys = keygen.generateKeyPair();
294 
295     if (LOG.isDebugEnabled()) {
296       final PublicKey pk = keys.getPublic();
297       final int len = getKeyLength(pk);
298       LOG.debug(
299           "Generated "
300               + keys.getPublic().getAlgorithm()
301               + " keys with length "
302               + len);
303     }
304     LOG.trace("<genKeys()");
305     return keys;
306   } // genKeys

321   /**
322    * An ECDSA key can be stripped of the curve parameters so it only contains
323    * the public point, and this is not enough to use the key for verification.
324    * However, if we know the curve name we can fill in the curve parameters and
325    * get a usable EC public key
326    *
327    * @param pk PublicKey, org.ejbca.cvc.PublicKeyEC, that might miss parameters,
328    *     if parameters are there we do not touch the public key just return it
329    *     unchanged
330    * @param keySpec name of curve for example brainpoolp224r1
331    * @return PublicKey with parameters from the named curve
332    * @throws NoSuchProviderException if provider not found
333    * @throws NoSuchAlgorithmException if algorithm not found
334    * @throws InvalidKeySpecException if spec is invalid
335    */
336   public static PublicKey getECPublicKeyWithParams(
337       final PublicKey pk, final String keySpec)
338       throws NoSuchAlgorithmException, NoSuchProviderException,
339           InvalidKeySpecException {
340     PublicKey ret = pk;
341     if ((pk instanceof PublicKeyEC) && (keySpec != null)) {
342       final PublicKeyEC pkec = (PublicKeyEC) pk;
343       // The public key of IS and DV certificate do not have any parameters so
344       // we have to do some magic to get a complete EC public key
345       final ECParameterSpec spec = pkec.getParams();
346       if (spec == null) {
347         // we did not have the parameter specs, lets create them because we know
348         // which curve we are using
349         final org.bouncycastle.jce.spec.ECParameterSpec bcspec =
350             ECNamedCurveTable.getParameterSpec(keySpec);
351         final java.security.spec.ECPoint p = pkec.getW();
352         final org.bouncycastle.math.ec.ECPoint ecp =
353             EC5Util.convertPoint(bcspec.getCurve(), p, false);
354         final ECPublicKeySpec pubKey = new ECPublicKeySpec(ecp, bcspec);
355         final KeyFactory keyfact = KeyFactory.getInstance("ECDSA", "BC");
356         ret = keyfact.generatePublic(pubKey);
357       }
358     }
359     return ret;
360   }
361   /**
362    * An ECDSA key can be stripped of the curve parameters so it only contains
363    * the public point, and this is not enough to use the key for verification.
364    * However, if we know the curve name we can fill in the curve parameters and
365    * get a usable EC public key
366    *
367    * @param pk PublicKey, org.ejbca.cvc.PublicKeyEC, that might miss parameters,
368    *     if parameters are there we do not touch the public key just return it
369    *     unchanged
370    * @param pkwithparams PublicKey, org.ejbca.cvc.PublicKeyEC, that contains all
371    *     parameters.
372    * @return PublicKey with parameters from the named curve
373    * @throws InvalidKeySpecException if the key specification in pkwithparams
374    *     was invalid
375    */
376   public static PublicKey getECPublicKeyWithParams(
377       final PublicKey pk, final PublicKey pkwithparams)
378       throws InvalidKeySpecException {
379     if (!(pk instanceof PublicKeyEC)
380         || !(pkwithparams instanceof PublicKeyEC)) {
381       LOG.info(
382           "Either pk or pkwithparams is not a PublicKeyEC: "
383               + pk.toString()
384               + ", "
385               + pkwithparams.toString());
386       return pk;
387     }
388     final PublicKeyEC pkec = (PublicKeyEC) pk;
389     final ECParameterSpec spec = pkec.getParams();
390     if (spec != null) {
391       return pk; // the key allready has its parameters.
392     }
393     // The public key of IS and DV certificate do not have any parameters so we
394     // have to do some magic to get a complete EC public key
395     final PublicKeyEC pkecp = (PublicKeyEC) pkwithparams;
396     final ECParameterSpec pkspec = pkecp.getParams();
397     if (pkspec == null) {
398       LOG.info("pkwithparams does not have any params.");
399       return pk;
400     }
401     final org.bouncycastle.jce.spec.ECParameterSpec bcspec =
402         EC5Util.convertSpec(pkspec, false);
403     final java.security.spec.ECPoint p = pkec.getW();
404     final org.bouncycastle.math.ec.ECPoint ecp =
405         EC5Util.convertPoint(pkspec, p, false);
406     final ECPublicKeySpec pubKey = new ECPublicKeySpec(ecp, bcspec);
407     final KeyFactory keyfact;
408     try {
409       keyfact =
410           KeyFactory.getInstance("ECDSA", BouncyCastleProvider.PROVIDER_NAME);
411     } catch (NoSuchAlgorithmException e) {
412       throw new IllegalStateException("ECDSA was an unknown algorithm", e);
413     } catch (NoSuchProviderException e) {
414       throw new IllegalStateException(
415           "BouncyCastle was not found as a provider.", e);
416     }
417     return keyfact.generatePublic(pubKey);
418   }

420   /**
421    * Gets the key length of supported keys.
422    *
423    * @param pk PublicKey used to derive the keysize
424    * @return -1 if key is unsupported, otherwise a number &gt;= 0. 0 usually
425    *     means the length can not be calculated, for example if the key is an EC
426    *     key and the "implicitlyCA" encoding is used.
427    */
428   public static int getKeyLength(final PublicKey pk) {
429     if (pk instanceof RSAPublicKey) {
430       final RSAPublicKey rsapub = (RSAPublicKey) pk;
431       return rsapub.getModulus().bitLength();
432     }
433     if (pk instanceof JCEECPublicKey) {
434       final JCEECPublicKey ecpriv = (JCEECPublicKey) pk;
435       final org.bouncycastle.jce.spec.ECParameterSpec spec =
436           ecpriv.getParameters();
437       if (spec != null) {
438         return spec.getN().bitLength();
439       }
440       // We support the key, but we don't know the key length
441       return 0;
442     }
443     if (pk instanceof BCECPublicKey) {
444       final BCECPublicKey ecpriv = (BCECPublicKey) pk;
445       final org.bouncycastle.jce.spec.ECParameterSpec spec =
446           ecpriv.getParameters();
447       if (spec != null) {
448         return spec.getN().bitLength();
449       }
450       // We support the key, but we don't know the key length
451       return 0;
452     }
453     if (pk instanceof ECPublicKey) {
454       final ECPublicKey ecpriv = (ECPublicKey) pk;
455       final java.security.spec.ECParameterSpec spec = ecpriv.getParams();
456       if (spec != null) {
457         return spec.getOrder()
458             .bitLength(); // does this really return something we expect?
459       }
460       // We support the key, but we don't know the key length
461       return 0;
462     }
463     if (pk instanceof DSAPublicKey) {
464       final DSAPublicKey dsapub = (DSAPublicKey) pk;
465       if (dsapub.getParams() != null) {
466         return dsapub.getParams().getP().bitLength();
467       }
468       return dsapub.getY().bitLength();
469     }
470     // Unknown key type
471     return -1;
472   }

474   /**
475    * Gets the key AlgorithmParameterSpec of supported keys. Can be used to
476    * initialize a KeyPairGenerator to generate a key of equal type and size.
477    *
478    * @param pk PublicKey used to derive the AlgorithmParameterSpec
479    * @return null if key is unsupported or pk is null, otherwise a
480    *     AlgorithmParameterSpec.
481    */
482   public static AlgorithmParameterSpec getKeyGenSpec(final PublicKey pk) {
483     if (pk == null) {
484       return null;
485     }
486     if (pk instanceof RSAPublicKey) {
487       LOG.debug("getKeyGenSpec: RSA");
488       final RSAPublicKey rpk = (RSAPublicKey) pk;
489       return new RSAKeyGenParameterSpec(
490           getKeyLength(pk), rpk.getPublicExponent());
491     }
492     if (pk instanceof DSAPublicKey) {
493       LOG.debug("getKeyGenSpec: DSA");
494       final DSAPublicKey dpk = (DSAPublicKey) pk;
495       final DSAParams params = dpk.getParams();
496       return new DSAParameterSpec(params.getP(), params.getQ(), params.getG());
497     }
498     if (pk instanceof ECPublicKey) {
499       LOG.debug("getKeyGenSpec: ECPublicKey");
500       final ECPublicKey ecpub = (ECPublicKey) pk;
501       final java.security.spec.ECParameterSpec sunsp = ecpub.getParams();
502       final EllipticCurve ecurve =
503           new EllipticCurve(
504               sunsp.getCurve().getField(),
505               sunsp.getCurve().getA(),
506               sunsp.getCurve().getB());
507       // ECParameterSpec par = new ECNamedCurveSpec(null, sunsp.getCurve(),
508       // sunsp.getGenerator(), sunsp.getOrder(),
509       // BigInteger.valueOf(sunsp.getCofactor()));
510       final ECParameterSpec params =
511           new ECParameterSpec(
512               ecurve,
513               sunsp.getGenerator(),
514               sunsp.getOrder(),
515               sunsp.getCofactor());
516       if (LOG.isDebugEnabled()) {
517         LOG.debug("Fieldsize: " + params.getCurve().getField().getFieldSize());
518         final EllipticCurve curve = params.getCurve();
519         LOG.debug("CurveA: " + curve.getA().toString(16));
520         LOG.debug("CurveB: " + curve.getB().toString(16));
521         LOG.debug("CurveSeed: " + curve.getSeed());
522         final ECFieldFp field = (ECFieldFp) curve.getField();
523         LOG.debug("CurveSfield: " + field.getP().toString(16));
524         final ECPoint p = params.getGenerator();
525         LOG.debug(
526             "Generator: "
527                 + p.getAffineX().toString(16)
528                 + ", "
529                 + p.getAffineY().toString(16));
530         LOG.debug("Order: " + params.getOrder().toString(16));
531         LOG.debug("CoFactor: " + params.getCofactor());
532       }
533       return params;
534     }
535     if (pk instanceof JCEECPublicKey) {
536       LOG.debug("getKeyGenSpec: JCEECPublicKey");
537       final JCEECPublicKey ecpub = (JCEECPublicKey) pk;
538       final org.bouncycastle.jce.spec.ECParameterSpec bcsp =
539           ecpub.getParameters();
540       final ECCurve curve = bcsp.getCurve();
541       // TODO: this probably does not work for key generation with the Sun
542       // PKCS#11 provider. Maybe seed needs to be set to null as above? Or
543       // something else, the BC curve is it the same?
544       final ECParameterSpec params =
545           new ECNamedCurveSpec(
546               null, curve, bcsp.getG(), bcsp.getN(), bcsp.getH());
547       return params;
548       // EllipticCurve ecc = new EllipticCurve(curve.)
549       // ECParameterSpec sp = new ECParameterSpec(, bcsp.getG(), bcsp.getN(),
550       // bcsp.getH().intValue());
551     }
552     return null;
553   }

555   /**
556    * Creates PKCS12-file that can be imported in IE or Firefox. The alias for

564    *     case use 'cert'.
565    * @return KeyStore containing PKCS12-keystore
566    * @throws CertificateException if the certificate couldn't be parsed

653           NoSuchAlgorithmException, InvalidKeySpecException {

657               + alias
658               + ", privKey, cert="
659               + CertTools.getSubjectDN(cert)
660               + ", cachain.length="
661               + ((cachain == null) ? 0 : cachain.length));

664     if (cert == null) {
665       throw new IllegalArgumentException("Parameter cert cannot be null.");
666     }

668     if (cachain != null) {
669       len += cachain.length;
670     }

672     // To not get a ClassCastException we need to generate a real new
673     // certificate with BC

676         cf.generateCertificate(new ByteArrayInputStream(cert.getEncoded()));

681             (X509Certificate)

695           final PKCS12BagAttributeCarrier caBagAttr =
696               (PKCS12BagAttributeCarrier) chain[i];

698           // from the DN if they exist.

742           "ClassCastException setting BagAttributes, can not set friendly"

758         // in this case we just set the local key id to that of the public key
759         keyBagAttr.setBagAttribute(
760             PKCSObjectIdentifiers.pkcs_9_at_friendlyName,
761             new DERBMPString(alias));
762         keyBagAttr.setBagAttribute(
763             PKCSObjectIdentifiers.pkcs_9_at_localKeyId,
764             createSubjectKeyId(chain[0].getPublicKey()));
765       } catch (ClassCastException e) {
766         LOG.error(
767             "ClassCastException setting BagAttributes, can not set friendly"
768                 + " name: ",
769             e);
770       }
771       // store the key and the certificate chain
772       // BC PKCS12 uses 3DES for key protection and 40 bit RC2 for protecting
773       // the certificates
774       final KeyStore store =
775           KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
776       store.load(null, null);
777       store.setKeyEntry(alias, pk, null, chain);
778       if (LOG.isTraceEnabled()) {
779         LOG.trace(
780             "<createP12: alias="
781                 + alias
782                 + ", privKey, cert="
783                 + CertTools.getSubjectDN(cert)
784                 + ", cachain.length="
785                 + ((cachain == null) ? 0 : cachain.length));
786       }
787       return store;
788     } catch (NoSuchProviderException e) {
789       throw new IllegalStateException(
790           "BouncyCastle provider was not found.", e);
791     } catch (KeyStoreException e) {
792       throw new IllegalStateException(
793           "PKCS12 keystore type could not be instanced.", e);
794     } catch (IOException e) {
795       throw new IllegalStateException(
796           "IOException should not be thrown when instancing an empty keystore.",
797           e);

1534    * @param keyAlg Algorithm
1535    * @param len Key length
1536    * @throws InvalidKeyException fail
1537    */
1538   public static void checkValidKeyLength(final String keyAlg, final int len)
1539       throws InvalidKeyException {
1540     final int ecMinSize = 224;
1541     final int dsaMinSize = 1024;
1542     final boolean isEcdsa =
1543         AlgorithmConstants.KEYALGORITHM_ECDSA.equals(keyAlg);
1544     final boolean isGost3410 =
1545         AlgorithmTools.isGost3410Enabled()
1546             && AlgorithmConstants.KEYALGORITHM_ECGOST3410.equals(keyAlg);
1547     final boolean isDstu4145 =
1548         AlgorithmTools.isDstu4145Enabled()
1549             && keyAlg.startsWith(CesecoreConfiguration.getOidDstu4145() + ".");
1550     if (isEcdsa || isGost3410 || isDstu4145) {
1551       // We allow key lengths of 0, because that means that implicitlyCA is
1552       // used.
1553       // for ImplicitlyCA we have no idea what the key length is, on the other
1554       // hand only real professionals
1555       // will ever use that to we will allow it.
1556       if ((len > 0) && (len < ecMinSize)) {
1557         final String msg =
1558             INTRES.getLocalizedMessage(
1559                 "catoken.invalidkeylength",
1560                 "ECDSA",
1561                 Integer.valueOf(ecMinSize).toString(),
1562                 Integer.valueOf(len));
1563         throw new InvalidKeyException(msg);
1564       }
1565     } else if (AlgorithmConstants.KEYALGORITHM_RSA.equals(keyAlg)
1566         || AlgorithmConstants.KEYALGORITHM_DSA.equals(keyAlg)) {
1567       if (len < dsaMinSize) {
1568         final String msg =
1569             INTRES.getLocalizedMessage(
1570                 "catoken.invalidkeylength",
1571                 "RSA/DSA",
1572                 Integer.valueOf(dsaMinSize).toString(),
1573                 Integer.valueOf(len));
1574         throw new InvalidKeyException(msg);
1575       }
1576     }
1577   }
