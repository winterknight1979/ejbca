54   * Servlet for download of CA certificates and chains.
55   *
56   * @version $Id: RaCertDistServlet.java 28821 2018-05-02 15:30:26Z samuellb $
57   */
58  @WebServlet("/cert")
59  public class RaCertDistServlet extends HttpServlet {
60  
61      private static final long serialVersionUID = 1L;
62      private static final Logger log = Logger.getLogger(RaCertDistServlet.class);
63      private static final String PARAMETER_FINGERPRINTSHEET = "fpsheet";
64      private static final String PARAMETER_CERT_BUNDLE = "certbundle";
65      private static final String PARAMETER_CAID = "caid";
66      private static final String PARAMETER_FINGERPRINT = "fp";
67      private static final String PARAMETER_FORMAT = "format";
68      private static final String PARAMETER_FORMAT_OPTION_FIREFOX = "ns";
69      private static final String PARAMETER_FORMAT_OPTION_PEM = "pem";    // Applies to both certificate chain and individual certificates download
70      private static final String PARAMETER_FORMAT_OPTION_DER = "der";
71      private static final String PARAMETER_FORMAT_OPTION_JKS = "jks";    // Applies only to certificate chain download
72      private static final String PARAMETER_FORMAT_OPTION_P7C = "pkcs7";  // Applies to both certificate chain and individual certificates download
73      private static final String PARAMETER_CHAIN = "chain";
74  
75      @EJB
76      private RaMasterApiProxyBeanLocal raMasterApi;
77      @EJB
78      private WebAuthenticationProviderSessionLocal webAuthenticationProviderSession;
79  
80      private RaAuthenticationHelper raAuthenticationHelper = null;
81  
82      @Override
83      protected void service(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse) throws ServletException, IOException {
84          if (raAuthenticationHelper==null) {
85              // Initialize the authentication helper function
86              raAuthenticationHelper = new RaAuthenticationHelper(webAuthenticationProviderSession);
87          }
88          if (httpServletRequest.getParameter(PARAMETER_FINGERPRINTSHEET) != null) {
89              downloadFingerprintSheet(httpServletRequest, httpServletResponse);
90              return;
91          }
92          if (httpServletRequest.getParameter(PARAMETER_CERT_BUNDLE) != null) {
93              downloadCertificateBundle(httpServletRequest, httpServletResponse);
94              return;
95          }
96          final boolean fullChain = Boolean.valueOf(httpServletRequest.getParameter(PARAMETER_CHAIN));
97          if (httpServletRequest.getParameter(PARAMETER_CAID) != null) {
98              List<Certificate> chain = null;
99              try {
100                 final int caId = Integer.valueOf(httpServletRequest.getParameter(PARAMETER_CAID));
101                 final AuthenticationToken authenticationToken = raAuthenticationHelper.getAuthenticationToken(httpServletRequest, httpServletResponse);
102                 final List<CAInfo> caInfos = raMasterApi.getAuthorizedCas(authenticationToken);
103                 if (log.isDebugEnabled()) {
104                     log.debug(authenticationToken.toString() + " was authorized to " + caInfos.size() + " CAs.");
105                 }
106                 for (final CAInfo caInfo : caInfos) {
107                     if (caId == caInfo.getCAId()) {
108                         chain = new ArrayList<>(caInfo.getCertificateChain());
109                         break;
110                     }
111                 }
112             } catch (NumberFormatException e) {
113                 log.debug("Unable to parse " + PARAMETER_CAID + " request parameter: " + e.getMessage());
114             }
115             if (chain==null) {
116                 httpServletResponse.sendError(HttpServletResponse.SC_BAD_REQUEST, "Unable to parse " + PARAMETER_CAID + " request parameter.");
117                 return;
118             } else {
119                 try {
120                     final Certificate caCertificate = chain.get(0);
121                     String filename = RequestHelper.getFileNameFromCertNoEnding(caCertificate, "ca");
122                     String contentType = "application/octet-stream";
123                     byte[] response = null;
124                     if (fullChain) {
125                         switch (httpServletRequest.getParameter(PARAMETER_FORMAT)) {
126                         case PARAMETER_FORMAT_OPTION_JKS: {
127                             // Create a JKS truststore with the CA certificates in
128                             final KeyStore keyStore = KeyStore.getInstance("JKS");
129                             keyStore.load(null, null);
130                             for (int i=0; i<chain.size(); i++) {
131                                 final String subjectDn = CertTools.getSubjectDN(chain.get(i));
132                                 String alias = CertTools.getPartFromDN(subjectDn, "CN");
133                                 if (alias == null) {
134                                     alias = CertTools.getPartFromDN(subjectDn, "O");
135                                 }
136                                 if (alias == null) {
137                                     alias = "cacert"+i;
138                                 }
139                                 alias.replaceAll(" ", "_").substring(0, Math.min(15, alias.length()));
140                                 keyStore.setCertificateEntry(alias, chain.get(i));
141                             }
142                             try (ByteArrayOutputStream out = new ByteArrayOutputStream();) {
143                                 keyStore.store(out, "changeit".toCharArray());
144                                 response = out.toByteArray();
145                             }
146                             filename += "-chain.jks";
147                             break;
148                         }
149                         case PARAMETER_FORMAT_OPTION_P7C: {
150                             response = CertTools.createCertsOnlyCMS(CertTools.convertCertificateChainToX509Chain(chain));
151                             filename += "-chain.p7c";
152                             break;
153                         }
154                         case PARAMETER_FORMAT_OPTION_PEM:
155                         default: {
156                             response = CertTools.getPemFromCertificateChain(chain);
157                             filename += "-chain.pem";
158                             break;
159                         }
160                         }
161                     } else {
162                         response = caCertificate.getEncoded();
163                         switch (httpServletRequest.getParameter(PARAMETER_FORMAT)) {
164                         case PARAMETER_FORMAT_OPTION_FIREFOX: {
165                             filename = null;
166                             contentType = "application/x-x509-ca-cert";
167                             break;
168                         }
169                         case PARAMETER_FORMAT_OPTION_DER: {
170                             filename += (caCertificate instanceof CardVerifiableCertificate) ? ".cvcert" : ".crt";
171                             break;
172                         }
173                         case PARAMETER_FORMAT_OPTION_P7C: {
174                             response = CertTools.createCertsOnlyCMS(CertTools.convertCertificateChainToX509Chain(Arrays.asList(new Certificate[]{ caCertificate })));
175                             filename += ".p7c";
176                             break;
177                         }
178                         case PARAMETER_FORMAT_OPTION_PEM:
179                         default: {
180                             filename += ".pem";
181                             response = CertTools.getPemFromCertificateChain(Arrays.asList(new Certificate[]{ caCertificate }));
182                             break;
183                         }
184                         }
185                     }
186                     writeResponseBytes(httpServletResponse, filename, contentType, response);
187                 } catch (NoSuchFieldException | KeyStoreException | NoSuchAlgorithmException | CertificateException | ClassCastException | CMSException e) {
188                     httpServletResponse.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Unable to serve request due to internal error.");
189                     if (log.isDebugEnabled()) {
190                         log.debug("Failed to provide certificate download to client. " + e.getMessage());
191                     }
192                     return;
193                 }
194             }
195         } else if (httpServletRequest.getParameter(PARAMETER_FINGERPRINT) != null) {
196             final String fingerprint = httpServletRequest.getParameter(PARAMETER_FINGERPRINT);
197             // Serving regular leaf certificate (optionally with full chain)
198             final AuthenticationToken authenticationToken = raAuthenticationHelper.getAuthenticationToken(httpServletRequest, httpServletResponse);
199             final List<CAInfo> caInfos = raMasterApi.getAuthorizedCas(authenticationToken);
200             // Only process request if there is a chance the client is authorized to the CA that issued it
201             if (!caInfos.isEmpty()) {
202                 final CertificateDataWrapper cdw = raMasterApi.searchForCertificate(authenticationToken, fingerprint);
203                 if (cdw!=null) {
204                     for (final CAInfo caInfo : caInfos) {
205                         if (caInfo.getSubjectDN().equals(cdw.getCertificateData().getIssuerDN())) {
206                             List<Certificate> chain = new ArrayList<>();
207                             chain.add(cdw.getCertificate());
208                             if (fullChain) {
209                                 chain.addAll(caInfo.getCertificateChain());
210                             }
211                             try {
212                                 byte[] response = null;
213                                 String filename = "cert"+fingerprint;
214                                 switch (httpServletRequest.getParameter(PARAMETER_FORMAT)) {
215                                 case PARAMETER_FORMAT_OPTION_DER: {
216                                     response = chain.get(0).getEncoded();
217                                     filename += (chain.get(0) instanceof CardVerifiableCertificate) ? ".cvcert" : ".crt";
218                                     break;
219                                 }
220                                 case PARAMETER_FORMAT_OPTION_P7C: {
221                                     response = CertTools.createCertsOnlyCMS(CertTools.convertCertificateChainToX509Chain(chain));
222                                     if (fullChain) {
223                                         filename += "-chain";
224                                     }
225                                     filename += ".p7c";
226                                     break;
227                                 }
228                                 case PARAMETER_FORMAT_OPTION_PEM:
229                                 default: {
230                                     response = CertTools.getPemFromCertificateChain(chain);
231                                     if (fullChain) {
232                                         filename += "-chain";
233                                     }
234                                     filename += ".pem";
235                                     break;
236                                 }
237                                 }
238                                 writeResponseBytes(httpServletResponse, filename, "application/octet-stream", response);
239                                 return;
240                             } catch (CertificateEncodingException | ClassCastException | CMSException e) {
241                                 log.warn("Failed to provide download of certificate with fingerprint " + fingerprint);
242                             }
243                         }
244                     }
245                 }
246             }
247             httpServletResponse.sendError(HttpServletResponse.SC_BAD_REQUEST, "Unable to parse " + PARAMETER_FINGERPRINT + " request parameter.");
248         } else {
249             httpServletResponse.sendError(HttpServletResponse.SC_BAD_REQUEST, "Unable to request parameters.");
250         }
251     }

253     private static String removeSpecial(String str) {
254 		return str.replaceAll("[^a-zA-Z ]", "");
255 	}
256 
257     private void writeResponseBytes(final HttpServletResponse httpServletResponse, final String filename, final String contentType, final byte[] response) throws IOException {
258         ServletUtils.removeCacheHeaders(httpServletResponse);
259         if (filename!=null) {
260         	String name = removeSpecial(filename);
261             httpServletResponse.setHeader("Content-Disposition", "attachment; filename=\"" + name + "\"");
262         }
263         httpServletResponse.setContentType(contentType);
264         httpServletResponse.setContentLength(response.length);
265         httpServletResponse.getOutputStream().write(response);
266     }

268     /**
269      * Creates and outputs a YAML document containing the CA certificate fingerprints of all active Certification Authorities
270      * on this system to which the current user has access. The CA certificate fingerprints are computed using SHA-256.
271      * <p>Certification Authorities without a certificate, i.e. CAs with status {@link org.cesecore.certificates.ca.CAConstants#CA_UNINITIALIZED} or
272      * {@link org.cesecore.certificates.ca.CAConstants#CA_WAITING_CERTIFICATE_RESPONSE} are excluded.
273      * @param httpServletRequest the HTTP request for CA certificate fingerprints
274      * @param httpServletResponse the HTTP response to which the fingerprint sheet should be written
275      * @throws IOException if an error occurred when creating the response
276      */
277     public void downloadFingerprintSheet(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse)
278             throws IOException {
279         final Map<String, Object> entries = new LinkedHashMap<>();
280         final AuthenticationToken authenticationToken = raAuthenticationHelper.getAuthenticationToken(httpServletRequest, httpServletResponse);
281         for (final CAInfo caInfo : raMasterApi.getAuthorizedCas(authenticationToken)) {
282             if (caInfo.getCertificateChain() == null || caInfo.getCertificateChain().size() == 0) {
283                 if (log.isDebugEnabled()) {
284                     log.debug("Not computing CA certificate fingerprint for CA " + caInfo.getName()
285                             + " because no CA certificate is available. Status of this CA is " + caInfo.getStatus());
286                 }
287                 continue;
288             }
289             try {
290                 final Certificate caCertificate = caInfo.getCertificateChain().get(0);
291                 final String caFingerprint = CertTools.getFingerprintAsString(CertTools.generateSHA256Fingerprint(caCertificate.getEncoded()));
292                 final Map<String, String> caEntry = new LinkedHashMap<>();
293                 caEntry.put("Subject DN", caInfo.getSubjectDN());
294                 caEntry.put("CA Certificate Fingerprint", caFingerprint);
295                 entries.put(caInfo.getName(), caEntry);
296                 if (log.isDebugEnabled()) {
297                     log.debug("Computed CA certificate fingerprint for CA " + caInfo.getName() + ".");
298                 }
299             } catch (final CertificateEncodingException e) {
300                 log.warn("Cannot compute CA certificate fingerprint for CA " + caInfo.getName()
301                         + " because the CA certificate could not be encoded. The error was: " + e.getMessage());
302                 continue;
303             }
304         }
305         final DumperOptions dumperOptions = new DumperOptions();
306         dumperOptions.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
307         dumperOptions.setPrettyFlow(true);
308         final Yaml yaml = new Yaml(dumperOptions);
309         log.info("User " + authenticationToken.toString() + " requested a CA certificate fingerprint file.");
310         writeResponseBytes(httpServletResponse, "fingerprints.yaml", "text/plain; charset=utf-8",
311                 yaml.dumpAsMap(entries).getBytes(Charset.forName("UTF-8")));
312     }

314     /**
315      * Creates and outputs a compressed certificate bundle containing the CA certificates of all active Certification Authorities
316      * on this system to which the current user has access. The certificate bundle is provided as a zip file of DER-encoded certificates.
317      * <p>Certification Authorities without a certificate, i.e. CAs with status {@link org.cesecore.certificates.ca.CAConstants#CA_UNINITIALIZED} or
318      * {@link org.cesecore.certificates.ca.CAConstants#CA_WAITING_CERTIFICATE_RESPONSE} are excluded.
319      * @param httpServletRequest the HTTP request for a certificate bundle
320      * @param httpServletResponse the HTTP response to which the certificate bundle should be written
321      * @throws IOException if an error occurred when creating the response
322      */
323     private void downloadCertificateBundle(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse)
324             throws IOException {
325         try (final ByteArrayOutputStream zipContent = new ByteArrayOutputStream()) {
326             try (final ZipOutputStream certificateBundle = new ZipOutputStream(zipContent)) {
327                 final AuthenticationToken authenticationToken = raAuthenticationHelper.getAuthenticationToken(httpServletRequest, httpServletResponse);
328                 for (final CAInfo caInfo : raMasterApi.getAuthorizedCas(authenticationToken)) {
329                     if (caInfo.getCertificateChain() == null || caInfo.getCertificateChain().size() == 0) {
330                         if (log.isDebugEnabled()) {
331                             log.debug("Not adding CA certificate for CA " + caInfo.getName()
332                                     + " to certificate bundle because no CA certificate is available. Status of this CA is " + caInfo.getStatus());
333                         }
334                         continue;
335                     }
336                     try {
337                         final byte[] encodedCertificate = caInfo.getCertificateChain().get(0).getEncoded();
338                         final String filename = caInfo.getName() + ".crt";
339                         certificateBundle.putNextEntry(new ZipEntry(filename));
340                         certificateBundle.write(encodedCertificate);
341                         certificateBundle.closeEntry();
342                         if (log.isDebugEnabled()) {
343                             log.debug("Added CA certificate for CA " + caInfo.getName() + " to certificate bundle.");
344                         }
345                     } catch (final CertificateEncodingException e) {
346                         log.warn("Cannot add CA certificate for CA " + caInfo.getName()
347                                 + " to certificate bundle because the CA certificate could not be encoded. The error was: " + e.getMessage());
348                         continue;
349                     }
350                 }
351                 log.info("User " + authenticationToken.toString() + " requested a CA certificate bundle.");
352             }
353             writeResponseBytes(httpServletResponse, "certbundle.zip", "application/octet-stream", zipContent.toByteArray());
354         }
