13  package org.ejbca.ra.jsfext;
14  
15  import java.util.Map;
16  
17  import javax.faces.component.UIComponent;
18  import javax.faces.context.FacesContext;
19  import javax.faces.validator.FacesValidator;
20  import javax.faces.validator.RequiredValidator;
21  import javax.faces.validator.ValidatorException;
22  
23  import org.apache.log4j.Logger;
24  
25  /**
26   * Extended variant of RequiredValidator that requires request parameter "validationRequiredFromRequest" set to true to
27   * be sent alongside with request to process validation. Use for component that validation is NOT needed
28   * for ALL POST requests. Beside the request parameter "validationRequiredFromRequest" input component has to have
29   * attribute "_required" set to true (or expression that evaluates to true) to perform validation.
30   * 
31   * DO NOT try to rename the "_required" into "required". It will be treated as &lt;h:inputText required="..."&gt; thus breaking
32   * the extendedRequiredValidator.
33   * 
34   * Example usage:
35  &lt;h:inputText value="#{someBean.componentForWhichValidationIsNotAllwaysRequired}"&gt;
36      &lt;f:validator validatorId="extendedRequiredValidator" /&gt;
37      &lt;f:attribute name="_required" value="true" /&gt;
38      &lt;f:ajax event="change" execute="@this" listener="#{someBean.someMethodThatDoesntValidateRequired}"
39      render="..."/&gt;
40  &lt;/h:inputText&gt;
41  &lt;h:commandButton 
42      action="#{someBean.someMethodThatDoesntValidateRequired}"&gt;
43  &lt;/h:commandButton&gt;
44  &lt;h:commandButton 
45      action="#{someBean.someMethodThatValidatesRequired}"&gt;
46      &lt;f:param name="validationRequiredFromRequest" value="true" /&gt;
47  &lt;/h:commandButton&gt;
48   * 
49   * @version $Id: ExtendedRequiredValidator.java 24496 2016-10-10 15:25:13Z marko $
50   */
51  @FacesValidator("extendedRequiredValidator")
52  public class ExtendedRequiredValidator extends RequiredValidator {
53      
54      private static final Logger log = Logger.getLogger(ExtendedRequiredValidator.class);
55  
56      @Override
57      public void validate(FacesContext context, UIComponent component, Object value) throws ValidatorException {
58          Object _required = component.getAttributes().get("_required");
59          boolean required = false;
60          if(_required != null){
61              required = Boolean.parseBoolean(_required.toString()); //_required can be of String or Boolean type
62          }
63          Map<String, String> params = FacesContext.getCurrentInstance().getExternalContext().getRequestParameterMap();
64          String validationRequiredFromRequest = params.get("validationRequiredFromRequest");
65          if (log.isTraceEnabled()) {
66              log.trace("validationRequiredFromRequest=" + validationRequiredFromRequest + ", _required="+ required + ", clientId=" + component.getClientId() + " value=" + value);
67          }
68          
69          if(!required || validationRequiredFromRequest == null || validationRequiredFromRequest.equalsIgnoreCase("false")){
70              if(log.isTraceEnabled()){
71                  log.trace("Ignoring extendedRequiredValidator for component " + component.getClientId());
72              }
73              return;
74          }
75          
76          super.validate(context, component, value);
77          
78      }
79  }
