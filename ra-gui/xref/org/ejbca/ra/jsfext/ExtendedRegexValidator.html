13  package org.ejbca.ra.jsfext;
14  
15  import java.util.Map;
16  
17  import javax.faces.component.UIComponent;
18  import javax.faces.context.FacesContext;
19  import javax.faces.validator.FacesValidator;
20  import javax.faces.validator.RegexValidator;
21  import javax.faces.validator.ValidatorException;
22  
23  import org.apache.commons.lang.StringUtils;
24  import org.apache.log4j.Logger;
25  
26  /**
27   * Extended variant of RegexValidator handy to be used when regex pattern value is going to be evaluated during
28   * view render time, which is the use-case for many UI components like h:dataTable and ui:repeat. Also the validator
29   * requires request parameter "validationRequiredFromRequest" set to true to
30   * be sent alongside with request to perform validation. Use for component that validation is NOT needed
31   * for ALL POST requests. 
32   * 
33   * Example usage:
34  &lt;ui:repeat value="#{someBean.instances}" var="instance"&gt;
35      &lt;h:inputText id="id" value="#{instance.value}"&gt;
36          &lt;f:validator validatorId="extendedRegexValidator" /&gt;
37          &lt;f:attribute name="pattern" value="#{instance.required ? instance.regexPattern : ''}" /&gt;
38          &lt;f:ajax event="change" execute="@this" listener="#{someBean.someMethodThatDoesntInvokeExtendedRegexValidator}"
39          render="..."/&gt;
40      &lt;/h:inputText&gt;
41  &lt;/ui:repeat&gt;
42  &lt;h:commandButton 
43      action="#{someBean.someMethodThatDoesntInvokeExtendedRegexValidator}"&gt;
44  &lt;/h:commandButton&gt;
45  &lt;h:commandButton 
46      action="#{someBean.someMethodThatInvokesExtendedRegexValidator}"&gt;
47      &lt;f:param name="validationRequiredFromRequest" value="true" /&gt;
48  &lt;/h:commandButton&gt;
49   * 
50   * @version $Id: ExtendedRegexValidator.java 24512 2016-10-13 14:28:20Z marko $
51   */
52  @FacesValidator("extendedRegexValidator")
53  public class ExtendedRegexValidator extends RegexValidator {
54      
55      private static final Logger log = Logger.getLogger(ExtendedRegexValidator.class);
56  
57      @Override
58      public void validate(FacesContext context, UIComponent component, Object value) throws ValidatorException {
59          String pattern = (String) component.getAttributes().get("pattern");
60          Map<String, String> params = FacesContext.getCurrentInstance().getExternalContext().getRequestParameterMap();
61          String validationRequiredFromRequest = params.get("validationRequiredFromRequest");
62          if (log.isTraceEnabled()) {
63              log.trace("validationRequiredFromRequest=" + validationRequiredFromRequest + ", pattern=" + pattern + ", clientId="
64                      + component.getClientId() + ", value=" + value);
65          }
66       
67          if(validationRequiredFromRequest == null || validationRequiredFromRequest.equalsIgnoreCase("false")){
68              if(log.isTraceEnabled()){
69                  log.trace("Ignoring extendedRegexValidator for component " + component.getClientId());
70              }
71              return;
72          }
73          
74          //Applying regex pattern
75          if (pattern != null && StringUtils.isNotBlank(pattern) && StringUtils.isNotBlank(value.toString())){
76              setPattern(pattern);
77              super.validate(context, component, value);
78          }
79      }
80  }
