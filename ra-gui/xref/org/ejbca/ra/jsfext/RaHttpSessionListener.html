15  import javax.servlet.annotation.WebListener;
16  import javax.servlet.http.HttpSessionEvent;
17  import javax.servlet.http.HttpSessionListener;
18  
19  import org.apache.log4j.Logger;
20  import org.ejbca.ra.RaAuthenticationBean;

22  /**
23   * HttpSessionListener to be able to detect/log when a user has been logged out for example due to the session has expired. 
24   * 
25   * @version $Id: RaHttpSessionListener.java 26057 2017-06-22 08:08:34Z anatom $
26   */
27  @WebListener
28  public class RaHttpSessionListener implements HttpSessionListener {
29  
30      private static final Logger log = Logger.getLogger(RaHttpSessionListener.class);
31  
32      @Override
33      public void sessionCreated(final HttpSessionEvent httpSessionEvent) {
34          if (log.isDebugEnabled()) {
35              log.debug("HTTP session from client started. jsessionid=" + httpSessionEvent.getSession().getId());
36          }
37      }
38  
39      @Override
40      public void sessionDestroyed(final HttpSessionEvent httpSessionEvent) {
41          final RaAuthenticationBeanuthenticationBean.html#RaAuthenticationBean">RaAuthenticationBean raAuthenticationBean = (RaAuthenticationBean) httpSessionEvent.getSession().getAttribute("raAuthenticationBean");
42          if (raAuthenticationBean==null) {
43              log.debug("Failed to clean up after client session with jsessionid=" + httpSessionEvent.getSession().getId());
44          } else {
45              raAuthenticationBean.onSessionDestroyed(httpSessionEvent);
46          }
47      }
48  }
