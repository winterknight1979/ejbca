43  /**
44   * 
45   * Filter used to intercept the resource requests while loading RA-web. If the requesting administrator
46   * belongs to a role which has a custom CSS or logo set, it will be injected instead of the default one.
47   * 
48   * The modified response will be browser cached as any other resource, and the request for 
49   * it will not pass this filter until the browser invalidates the cache. Hence requests for the
50   * modified resources will not be requested via Peers for every request.
51   * 
52   * This filter is mapped in web.xml to only process CSS / Image files in the RA-web.
53   * 
54   * @version $Id: RaStyleRequestFilter.java 26914 2017-10-27 10:10:32Z henriks $
55   *
56   */
57  public class RaStyleRequestFilter implements Filter {
58      private final String RA_LOGO_PATH = "/ejbca/ra/img/pk_logo.png";
59      private static Logger log = Logger.getLogger(RaStyleRequestFilter.class);
60  
61      @EJB
62      private AdminPreferenceSessionLocal adminPreferenceSessionLocal;
63      @EJB
64      private WebAuthenticationProviderSessionLocal webAuthenticationProviderSession;
65      
66      private RaAuthenticationHelper raAuthenticationHelper = null;
67      
68      @Override
69      public void destroy() {
70          //NOOP
71      }
72  
73      @Override
74      public void init(FilterConfig arg0) throws ServletException {
75          if (log.isDebugEnabled()) {
76              log.debug(this.getClass().getName() + " initialized");        
77          }
78      }
79      
80      /** Called once for every requested resource on a RA page load. If modified resources are available, the response will be intercept */
81      @Override
82      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
83          HttpServletRequest httpRequest = (HttpServletRequest) request;
84          HttpServletResponse httpResponse = (HttpServletResponse) response;
85          String requestPath = httpRequest.getRequestURI();
86          String resource = requestPath.substring(requestPath.lastIndexOf('/') + 1, requestPath.length());
87          RaStyleInfo customResponse;
88          AuthenticationToken authenticationToken = null;
89          authenticationToken = getAuthenticationToken(httpRequest, httpResponse);
90          try {
91              customResponse = adminPreferenceSessionLocal.getPreferedRaStyleInfo(authenticationToken);
92          } catch (Exception e) {
93              // In any case of error loading the styles, display default style rather than no styles at all.
94              e.printStackTrace();
95              chain.doFilter(httpRequest, httpResponse);
96              return;
97          }
98          
99          // TODO pure/base.css is currently unsupported for injection since it can't be differentiated from /css/base.css by name
100         if (customResponse == null || requestPath.equals("/ejbca/ra/css/pure/base.css")) {
101             chain.doFilter(httpRequest, httpResponse);
102             return;
103         }
104         
105         // When logo is requested and a custom logo is applied to the administrators role, the response is intercept with
106         // the replaced logo.
107         if (requestPath.equals(RA_LOGO_PATH) && customResponse.getLogoBytes() != null) {
108             OutputStream clientPrintWriter = response.getOutputStream();
109             try {
110                 ResponseWrapper responseWrapper = new ResponseWrapper((HttpServletResponse) response);
111                 chain.doFilter(httpRequest, responseWrapper);
112                 
113                 byte[] newLogoContent = customResponse.getLogoBytes();
114                 httpResponse.setContentType(customResponse.getLogoContentType());
115                 httpResponse.setContentLength(newLogoContent.length);
116                 clientPrintWriter.write(newLogoContent);
117             } finally {
118                 clientPrintWriter.close();
119             }
120             return;
121         }
122         
123         // When a CSS resource is requested, the response is intercept with a modified CSS if the administrators role
124         // has one applied
125         RaCssInfo cssResponse = customResponse.getRaCssInfos().get(resource);
126         if (cssResponse != null) {
127             PrintWriter clientPrintWriter = response.getWriter();
128             try {
129                 ResponseWrapper responseWrapper = new ResponseWrapper((HttpServletResponse) response);
130                 chain.doFilter(httpRequest, responseWrapper);
131                 String newCssContent = new String(cssResponse.getCssBytes());
132                 httpResponse.setContentType("text/css");
133                 httpResponse.setContentLength(newCssContent.length());
134                 clientPrintWriter.write(newCssContent);
135             } finally {
136                 clientPrintWriter.close();
137                 
138             }
139             return;
140         }
141 
142         // No match. Pass on request
143         chain.doFilter(httpRequest, httpResponse);
144     }
145 
146     /** @param httpRequest Req
147      * @param httpResponse Resp
148      * @return the X509CertificateAuthenticationToken if the client has provided a certificate or a PublicAccessAuthenticationToken otherwise. */
149     private AuthenticationToken getAuthenticationToken(HttpServletRequest httpRequest, HttpServletResponse httpResponse) {
150         raAuthenticationHelper = new RaAuthenticationHelper(webAuthenticationProviderSession);
151         AuthenticationToken authenticationToken = raAuthenticationHelper.getAuthenticationToken(httpRequest, httpResponse);
152         return authenticationToken;
153     }

156     private class ResponseOutputStream extends ServletOutputStream {
157         private ByteArrayOutputStream outStream = new ByteArrayOutputStream();
158 
159         @Override
160         public void write(int writeByte) throws IOException {
161             outStream.write(writeByte);
162         }
163 
164 		@Override
165 		public boolean isReady() {
166 			return true;
167 		}
168 
169 		@Override
170 		public void setWriteListener(WriteListener writeListener) {
171 			// no-op
172 			
173 		}
174     }
175     
176     private class ResponseWrapper extends HttpServletResponseWrapper {
177         private final CharArrayWriter writer;
178         private final ResponseOutputStream imageOutputStream;
179 
180         public ResponseWrapper(HttpServletResponse response) {
181             super(response);
182             writer = new CharArrayWriter();
183             imageOutputStream = new ResponseOutputStream();
184         }
185         
186         @Override
187         public ServletOutputStream getOutputStream() throws IOException {
188             return imageOutputStream;
189         }
190         
191         @Override
192         public PrintWriter getWriter() throws IOException {
193             return new PrintWriter(writer);
194         }
195 
196         @Override
197         public String toString() {
198             return writer.toString();
199         }
200     }
