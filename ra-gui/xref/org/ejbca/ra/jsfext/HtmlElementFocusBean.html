6   /**
7    * Request scoped managed bean that can be used to focus an html element after form submit
8    * Usage:
9    * 1) For the element want to focus, add a new boolean property (etc. HtmlElementFocusBean.booleanForElementX)
10   * 2) Set styleClass to that element: styleClass="#{htmlElementFocusBean.booleanForElementX?'jsAutoFocusJsf':''}
11   * 3) Add action listener to the element that submits form &lt;f:actionListener binding="#{htmlElementFocusBean.setBooleanForElementX(true)}"/&gt;
12   * Note: Preferably bean property name should be the same as xhtml id.
13   * 
14   * @version $Id: HtmlElementFocusBean.java 23983 2016-07-22 13:06:01Z marko $
15   * TODO: Use CDI beans
16   */
17  @SuppressWarnings("deprecation")
18  @ManagedBean
19  @RequestScoped
20  public class HtmlElementFocusBean {
21  
22      private boolean requestPreviewMoreDetails = false;
23  
24      public boolean isRequestPreviewMoreDetails() {
25          return requestPreviewMoreDetails;
26      }
27  
28      public void setRequestPreviewMoreDetails(boolean requestPreviewMoreDetails) {
29          this.requestPreviewMoreDetails = requestPreviewMoreDetails;
30      }
31  
32  }
