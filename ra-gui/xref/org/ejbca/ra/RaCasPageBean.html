52      /** Representation of a CA in a chain with links to CRL download locations. */
53      public class CaAndCrl {
54          private final String name;
55          private final String subjectDn;
56          private final int caId;
57          private String crlLink;
58          private String deltaCrlLink;
59          private final int position;
60          private final List<String> chainNames;
61          private boolean x509 = false;
62  
63          CaAndCrl(final String name, final String subjectDn, final int caId, final int position, final List<String> chainNames) {
64              this.name = name;
65              this.subjectDn = subjectDn;
66              this.caId = caId;
67              this.position = position;
68              this.chainNames = chainNames;
69          }
70  
71          public String getName() { return name; }
72          public String getSubjectDn() { return subjectDn; }
73          public int getCaId() { return caId; }
74          public String getCrlLink() { return crlLink; }
75          public String getDeltaCrlLink() { return deltaCrlLink; }
76          public int getPosition() { return position; }
77          public boolean isX509() { return x509; }
78  
79          @Override
80          public int hashCode() { return subjectDn.hashCode(); }
81          @Override
82          public boolean equals(final Object obj) { return obj instanceof CaAndCrl && subjectDn.equals(((CaAndCrl)obj).subjectDn); }
83  
84          /** @return the Subject DN string of the current certificate in unescaped RDN format */
85          public final String getSubjectDnUnescapedRndValue() {
86              if (StringUtils.isNotEmpty(subjectDn)) {
87                  return org.ietf.ldap.LDAPDN.unescapeRDN(subjectDn);
88              } else {
89                  return subjectDn;
90              }
91          }
92      }

112     private List<CaAndCrl> casAndCrlItems = null;
113     private boolean atLeastOneCrlLinkPresent = false;
114 
115     /** @return true if at least one of the CAs available via #getCasAndCrlItems() has CRLs present on this system. */
116     public boolean isAtLeastOneCrlLinkPresent() {
117         getCasAndCrlItems();
118         return atLeastOneCrlLinkPresent && WebConfiguration.isCrlStoreEnabled();
119     }
120 
121     /** @return a list of all known authorized CAs with links to CRLs (if present) */
122     public List<CaAndCrl> getCasAndCrlItems() {
123         if (casAndCrlItems==null) {
124             final List<CAInfo> caInfos = new ArrayList<>(raMasterApiProxyBean.getAuthorizedCas(raAuthenticationBean.getAuthenticationToken()));
125             // First build a mapping of all subjects and their short names
126             final Map<String,String> caSubjectToNameMap = new HashMap<>();
127             for (final CAInfo caInfo : caInfos) {
128                 caSubjectToNameMap.put(caInfo.getSubjectDN(), caInfo.getName());
129             }
130             // Convert all CA's chains into CA objects (use a Set to avoid duplicates)
131             final Set<CaAndCrl> cas = new HashSet<>();
132             for (final CAInfo caInfo : caInfos) {
133                 final List<Certificate> chain = new ArrayList<>(caInfo.getCertificateChain());
134                 Collections.reverse(chain);
135                 final List<String> chainNames = new ArrayList<>();
136                 final int caId = caInfo.getCAId();
137                 for (final Certificate caCertificate : chain) {
138                     final String subjectDn = CertTools.getSubjectDN(caCertificate);
139                     String name = caSubjectToNameMap.get(subjectDn);
140                     if (name==null) {
141                         name = subjectDn;
142                     }
143                     chainNames.add(name);
144                     final CaAndCrl caAndCrl = new CaAndCrl(name, subjectDn, chainNames.size()==chain.size()?caId:NO_CAID_AVAILABLE, chainNames.size()-1, new ArrayList<>(chainNames));
145                     // Construct links to RFC4387 CRL Download Servlet
146                     if (caCertificate instanceof X509Certificate) {
147                         caAndCrl.x509 = true;
148                         final CRLInfo crlInfoFull = crlSession.getLastCRLInfo(subjectDn, false);
149                         if (crlInfoFull!=null) {
150                             atLeastOneCrlLinkPresent = true;
151                             caAndCrl.crlLink = RFC4387_DEFAULT_EJBCA_URL + "?iHash=" + getSubjectPrincipalHashAsUnpaddedBase64((X509Certificate)caCertificate);
152                             final CRLInfo crlInfoDelta = crlSession.getLastCRLInfo(subjectDn, true);
153                             if (crlInfoDelta!=null) {
154                                 caAndCrl.deltaCrlLink = RFC4387_DEFAULT_EJBCA_URL + "?iHash=" + getSubjectPrincipalHashAsUnpaddedBase64((X509Certificate)caCertificate) + "&delta=";
155                             }
156                         }
157                     }
158                     // Add missing items and replace items when we know the CAId
159                     if (caAndCrl.getCaId()!=NO_CAID_AVAILABLE) {
160                         cas.remove(caAndCrl);
161                     }
162                     if (!cas.contains(caAndCrl)) {
163                         cas.add(caAndCrl);
164                     }
165                 }
166             }
167             casAndCrlItems = new ArrayList<>(cas);
168             // Sort by higher level CAs
169             Collections.sort(casAndCrlItems, new Comparator<CaAndCrl>() {
170                 @Override
171                 public int compare(final CaAndCrl caAndCrl1, final CaAndCrl caAndCrl) {
172                     final int size1 = caAndCrl1.chainNames.size();
173                     final int size2 = caAndCrl.chainNames.size();
174                     // Avoid checking if chain length is the same
175                     for (int i=0; i<Math.min(size1, size2); i++) {
176                         final String name1 = caAndCrl1.chainNames.get(i);
177                         final String name2 = caAndCrl.chainNames.get(i);
178                         final int compareTo = name1.compareTo(name2);
179                         if (compareTo!=0) {
180                             return compareTo;
181                         }
182                     }
183                     return size1-size2;
184                 }
185             });
186         }
187         return casAndCrlItems;
188     }
189 
190     /** @param x509Certificate Cert
191      * @return the issuer hash in base64 encoding without padding which is the way RFC4387 search function expects the iHash parameter. */
192     private String getSubjectPrincipalHashAsUnpaddedBase64(final X509Certificate x509Certificate) {
193         final byte[] hashSubjectX500Principal = CertTools.generateSHA1Fingerprint(x509Certificate.getSubjectX500Principal().getEncoded());
194         return new String(Base64.encode(hashSubjectX500Principal)).substring(0, 27).replaceAll("\\+", "%2B");
195     }
