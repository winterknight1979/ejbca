13  package org.ejbca.ra;
14  
15  import java.io.IOException;
16  
17  import javax.servlet.ServletException;
18  import javax.servlet.annotation.WebServlet;
19  import javax.servlet.http.HttpServlet;
20  import javax.servlet.http.HttpServletRequest;
21  import javax.servlet.http.HttpServletResponse;
22  import javax.servlet.http.HttpSession;
23  
24  import org.apache.log4j.Logger;
25  
26  /**
27   * Servlet for receiving AJAX requests that keeps the HTTP session alive.
28   * 
29   * @version $Id: RaSessionKeepAliveServlet.java 26057 2017-06-22 08:08:34Z anatom $
30   */
31  @WebServlet("/sessionKeepAlive")
32  public class RaSessionKeepAliveServlet extends HttpServlet {
33  
34      private static final long serialVersionUID = 1L;
35      private static final Logger log = Logger.getLogger(RaSessionKeepAliveServlet.class);
36      
37      private static final int MAX_INACTIVE_INTERVAL_JSCLIENT_SECONDS = 60;
38  
39      @Override
40      protected void service(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse) throws ServletException, IOException {
41          if (httpServletRequest.getRequestedSessionId()==null || !httpServletRequest.isRequestedSessionIdValid()) {
42              // No session, so nothing to keep alive...
43              httpServletResponse.sendError(500);
44              return;
45          }
46          // Calling httpServletRequest.getSession() will re-validate the session
47          final HttpSession httpSession = httpServletRequest.getSession();
48          // Get and save the original web.xml session-timeout
49          int maxInactiveIntervalSeconds = httpSession.getMaxInactiveInterval();
50          if (maxInactiveIntervalSeconds>MAX_INACTIVE_INTERVAL_JSCLIENT_SECONDS) {
51              // Since the client has JavaScript enabled (pretty safe to assume since this is accessed via AJAX), we can set the session timeout more aggressively
52              maxInactiveIntervalSeconds = MAX_INACTIVE_INTERVAL_JSCLIENT_SECONDS;
53              httpSession.setMaxInactiveInterval(maxInactiveIntervalSeconds);
54              if (log.isDebugEnabled()) {
55                  log.debug("Client with jsessionid=" + httpSession.getId() + " called keep alive servlet. Adjusted MaxInactiveInterval to " + maxInactiveIntervalSeconds);
56              }
57          }
58          // Use a fail safe of 10000 milliseconds when calculating the time to the next check
59          final int timeToNextCheckInMs = Math.max(10000, (maxInactiveIntervalSeconds-10)*1000);
60          if (log.isDebugEnabled()) {
61              log.debug("maxInactiveIntervalSeconds: " + maxInactiveIntervalSeconds + " jsessionid=" + httpSession.getId());
62          }
63          httpServletResponse.setContentType("text/plain");
64          httpServletResponse.getWriter().write(String.valueOf(timeToNextCheckInMs));
65      }
66  }
