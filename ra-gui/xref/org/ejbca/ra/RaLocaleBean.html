67      /** @return this sessions Locale */
68      public Locale getLocale() {
69  
70          Locale localeFromDB = adminPreferenceSession.getCurrentRaLocale(raAuthenticationBean.getAuthenticationToken());
71  
72          if (localeFromDB != null) {
73              locale = localeFromDB;
74              directionLeftToRight = isDirectionLeftToRight(locale);
75          } else {
76              if (locale == null) {
77                  final FacesContext facesContext = FacesContext.getCurrentInstance();
78                  final Locale requestLocale = facesContext.getExternalContext().getRequestLocale();
79                  if (getSupportedLocales().contains(requestLocale)) {
80                      locale = requestLocale;
81                  } else {
82                      locale = facesContext.getApplication().getDefaultLocale();
83                  }
84                  directionLeftToRight = isDirectionLeftToRight(locale);
85              }
86          }
87          return locale;
88      }
89      /** Set this sessions Locale 
90       * @param locale Locale*/
91      public void setLocale(final Locale locale) {
92  
93          this.locale = locale;
94          directionLeftToRight = isDirectionLeftToRight(locale);
95      }

97      /** @return a list of all locales as defined in faces-config.xml */
98      public List<Locale> getSupportedLocales() {
99          final Application application = FacesContext.getCurrentInstance().getApplication();
100         final Iterator<Locale> iterator = application.getSupportedLocales();
101         final List<Locale> ret = new ArrayList<Locale>();
102         while (iterator.hasNext()) {
103             ret.add(iterator.next());
104         }
105         final Locale defaultLocale = application.getDefaultLocale();
106         if (!ret.contains(defaultLocale)) {
107             ret.add(defaultLocale);
108         }
109         Collections.sort(ret, new Comparator<Locale>() {
110             @Override
111             public int compare(Locale o1, Locale o2) {
112                 return o1.getLanguage().compareTo(o2.getLanguage());
113             }
114         });
115         return ret;
116     }

118     /** @param locale Locale
119      * @return true if the language direction is left to right */
120     private boolean isDirectionLeftToRight(final Locale locale) {
121         final int directionality = Character.getDirectionality(locale.getDisplayName(locale).charAt(0));
122         log.debug("directionality is " + directionality + " for " + locale.getLanguage() + " (" + locale.getDisplayName(locale) + ").");
123         return directionality != Character.DIRECTIONALITY_RIGHT_TO_LEFT && directionality != Character.DIRECTIONALITY_RIGHT_TO_LEFT_ARABIC &&
124                 directionality != Character.DIRECTIONALITY_RIGHT_TO_LEFT_EMBEDDING && directionality != Character.DIRECTIONALITY_RIGHT_TO_LEFT_OVERRIDE;
125     }
126 
127     /** @return true if the language direction is left to right */
128     public String getDirection() {
129         return directionLeftToRight ? LEFT_TO_RIGHT : RIGHT_TO_LEFT;
130     }
131 
132     /** @return true if the language direction is left to right */
133     public String getIndentionDirection() {
134         return directionLeftToRight ? "left" : "right";
135     }
136     
137     /** @return 
138      *     the reverse of the standard value, for cases when text needs to be aligned to the other side. */
139     public String getReverseIndentationDirection() {
140         return !directionLeftToRight ? "left" : "right";
141     }

170     /**
171      * Find localized message template and replace the {number} place holders with the provided parameters.
172      * In the case where the localized language template is not available, the template from the default language will be tried.
173      * 
174      * @param messageKey the message key
175      * @param params to replace place holders with. Evaluated with String.valueOf() (null-safe).
176      * @return the localized message or "???messageKey???" if no key way found.
177      */
178     public String getMessage(final String messageKey, final Object...params) {
179         if (messageKey==null) {
180             return "???null???";
181         }
182         final FacesContext facesContext = FacesContext.getCurrentInstance();
183         String messageTemplate = null;
184         try {
185             final ResourceBundle resourceBundle = facesContext.getApplication().getResourceBundle(facesContext, "msg");
186             messageTemplate = resourceBundle.getString(messageKey);
187         } catch (MissingResourceException e) {
188             // Fall-back to trying the default locale
189             facesContext.getViewRoot().setLocale(facesContext.getApplication().getDefaultLocale());
190             try {
191                 final ResourceBundle resourceBundle = facesContext.getApplication().getResourceBundle(facesContext, "msg");
192                 messageTemplate = resourceBundle.getString(messageKey);
193             } catch (MissingResourceException e2) {
194                 return "???" + messageKey + "???";
195             } finally {
196                 FacesContext.getCurrentInstance().getViewRoot().setLocale(getLocale());
197             }
198         }
199         final StringBuilder sb = new StringBuilder(messageTemplate);
200         // Go backwards so if the value was the same a placeholder tag, we wont be affected
201         if (params.length>0) {
202             for (int i=params.length-1; i>=0; i--) {
203                 final String placeHolder = "{"+i+"}";
204                 final int currentIndex = sb.indexOf(placeHolder);
205                 if (currentIndex==-1) {
206                     if (log.isDebugEnabled()) {
207                         log.debug("messageKey '" + messageKey + "' was referenced using parameter '" + params[i] + "', but no " + placeHolder + " exists.");
208                     }
209                     continue;
210                 }
211                 sb.replace(currentIndex, currentIndex+placeHolder.length(), String.valueOf(params[i]));
212             }
213         }
214         return sb.toString();
215     }

217     /**
218      * Get localized error code.
219      * @param errorCode code
220      * @return localized error code
221      */
222     public String getErrorCodeMessage(final ErrorCode errorCode){
223         if(errorCode == null){
224             return "???errorCodeNull???";
225         }
226         return getMessage("errorcode_" + errorCode.getInternalErrorCode());
227     }

231      * @param messageKey the message key
232      * @param params to replace place holders with. Evaluated with String.valueOf() (null-safe).

234      * @see RaLocaleBean#getMessage 
235      */
236     public FacesMessage getFacesMessage(final String messageKey, final Object...params){
237         return new FacesMessage(getMessage(messageKey, params));
