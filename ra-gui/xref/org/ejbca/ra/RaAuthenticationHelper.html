17  import java.nio.charset.StandardCharsets;

30  /**
31   * Web session authentication helper.
32   * 
33   * @version $Id: RaAuthenticationHelper.java 29180 2018-06-08 18:56:00Z bastianf $
34   */
35  public class RaAuthenticationHelper implements Serializable {
36  
37      private static final long serialVersionUID = 1L;
38  
39      private static final Logger log = Logger.getLogger(RaAuthenticationHelper.class);
40      private static final String HTTP_HEADER_SET_COOKIE = "Set-Cookie";
41      private static final String HTTP_HEADER_X_POWERED_BY = "X-Powered-By";
42  
43      private final WebAuthenticationProviderSessionLocal webAuthenticationProviderSession;
44      private AuthenticationToken authenticationToken = null;
45      private String authenticationTokenTlsSessionId = null;
46      private String x509AuthenticationTokenFingerprint = null;
47  
48      public RaAuthenticationHelper(final WebAuthenticationProviderSessionLocal webAuthenticationProviderSession) {
49          this.webAuthenticationProviderSession = webAuthenticationProviderSession;
50      }
51  
52      /** @param httpServletRequest req
53       * @param httpServletResponse resp
54       * @return the X509CertificateAuthenticationToken if the client has provided a certificate or a PublicAccessAuthenticationToken otherwise. */
55      public AuthenticationToken getAuthenticationToken(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse) {
56          final String currentTlsSessionId = getTlsSessionId(httpServletRequest);
57          if (authenticationToken==null || !StringUtils.equals(authenticationTokenTlsSessionId, currentTlsSessionId)) {
58              if (log.isTraceEnabled()) {
59                  log.trace("New TLS session IDs or authenticationToken: currentClientTlsSessionID: "+currentTlsSessionId+", authenticationTokenTlsSessionId: "+authenticationTokenTlsSessionId);
60              }
61              // Set the current TLS session 
62              authenticationTokenTlsSessionId = currentTlsSessionId;
63              final X509Certificate x509Certificate = getClientX509Certificate(httpServletRequest);
64              if (x509Certificate == null && x509AuthenticationTokenFingerprint!=null) {
65                  log.warn("Suspected session hijacking attempt from " + httpServletRequest.getRemoteAddr() +
66                          ". RA client presented no TLS certificate in HTTP session previously authenticated with client certificate.");
67                  authenticationToken = null;
68                  x509AuthenticationTokenFingerprint = null;
69              }
70              if (x509Certificate != null) {
71                  final String fingerprint = CertTools.getFingerprintAsString(x509Certificate);
72                  if (log.isTraceEnabled()) {
73                      log.trace("currentRequestFingerprint: "+fingerprint+", x509AuthenticationTokenFingerprint: "+x509AuthenticationTokenFingerprint);
74                  }
75                  if (x509AuthenticationTokenFingerprint!=null) {
76                      if (!StringUtils.equals(fingerprint, x509AuthenticationTokenFingerprint)) {
77                          log.warn("Suspected session hijacking attempt from " + httpServletRequest.getRemoteAddr() +
78                                  ". RA client presented a different TLS certificate in the same HTTP session." +
79                                  " new certificate had subject '" + CertTools.getSubjectDN(x509Certificate) + "'.");
80                          authenticationToken = null;
81                          x509AuthenticationTokenFingerprint = null;
82                      }
83                  }
84                  if (log.isDebugEnabled()) {
85                      log.debug("RA client presented client TLS certificate with subject DN '" + CertTools.getSubjectDN(x509Certificate) + "'.");
86                  }
87                  // No need to perform re-authentication if the client certificate was the same
88                  if (authenticationToken==null) {
89                      authenticationToken = webAuthenticationProviderSession.authenticateUsingClientCertificate(x509Certificate);
90                  }
91                  x509AuthenticationTokenFingerprint = authenticationToken==null ? null : fingerprint;
92              }
93              if (authenticationToken == null) {
94                  // Instead of checking httpServletRequest.isSecure() (connection deemed secure by container), we check if a TLS session is present
95                  final boolean confidentialTransport = currentTlsSessionId!=null;
96                  authenticationToken = webAuthenticationProviderSession.authenticateUsingNothing(httpServletRequest.getRemoteAddr(), confidentialTransport);
97              }
98          }
99          resetUnwantedHttpHeaders(httpServletRequest, httpServletResponse);
100         return authenticationToken;
101     }
102     
103     /** Invoke once the session is started to prevent security leak via HTTP headers related. 
104      * @param httpServletRequest req
105      * @param httpServletResponse resp */
106     private void resetUnwantedHttpHeaders(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse) {
107         // Ensure that we never send the JSESSIONID over an insecure (HTTP) connection
108         // By default JBoss will send the JSESSIONID cookie over HTTP with the "Secure;" option. Since this is sent in clear from the server to the broswer
109         // it does not really help security much that it is only sent over HTTPS from client to server.
110         if (!httpServletRequest.isSecure() && !StringUtils.isEmpty(httpServletResponse.getHeader(HTTP_HEADER_SET_COOKIE))) {
111             if (log.isDebugEnabled()) {
112                 log.debug("Preventing '"+HTTP_HEADER_SET_COOKIE+"' HTTP header on insecure connection with value: " + httpServletResponse.getHeader(HTTP_HEADER_SET_COOKIE));
113             }
114             httpServletResponse.setHeader(HTTP_HEADER_SET_COOKIE, "");
115         }
116         // Prevent sending the the X-Powered-By header e.g. "JSF/2.0"
117         if (!StringUtils.isEmpty(httpServletResponse.getHeader(HTTP_HEADER_X_POWERED_BY))) {
118             if (log.isDebugEnabled()) {
119                 log.debug("Preventing '"+HTTP_HEADER_X_POWERED_BY+"' HTTP header with value: " + httpServletResponse.getHeader(HTTP_HEADER_X_POWERED_BY));
120             }
121             httpServletResponse.setHeader(HTTP_HEADER_X_POWERED_BY, "");
122         }
123     }
124     
125     private X509Certificate getClientX509Certificate(final HttpServletRequest httpServletRequest) {
126         final X509Certificate[] certificates = (X509Certificate[]) httpServletRequest.getAttribute("javax.servlet.request.X509Certificate");
127         return certificates == null || certificates.length==0 ? null : certificates[0];
128     }
129     
130     private String getTlsSessionId(final HttpServletRequest httpServletRequest) {
131         final String sslSessionIdServletsStandard;
132         final Object sslSessionIdServletsStandardObject = httpServletRequest.getAttribute("javax.servlet.request.ssl_session_id");
133         if (sslSessionIdServletsStandardObject!=null && sslSessionIdServletsStandardObject instanceof byte[]) {
134             // Wildfly 9 stores the TLS sessions as a raw byte array. Convert it to a hex String.
135             sslSessionIdServletsStandard = new String(Hex.encode((byte[]) sslSessionIdServletsStandardObject), StandardCharsets.UTF_8);
136         } else {
137             sslSessionIdServletsStandard = (String) sslSessionIdServletsStandardObject; 
138         }
139         final String sslSessionIdJBoss7 = (String)httpServletRequest.getAttribute("javax.servlet.request.ssl_session");
140         return sslSessionIdJBoss7==null ? sslSessionIdServletsStandard : sslSessionIdJBoss7;
141     }
