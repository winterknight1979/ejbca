152     /** Determine if we need to query back end or just filter and execute the required action. */
153     private void searchAndFilterCommon() {
154         // First make sure we have all CA and Role names
155         getAvailableCas();
156         getAvailableRoles();
157         
158         // Make search request
159         final RaRoleMemberSearchRequest searchRequest = new RaRoleMemberSearchRequest();
160         if (criteriaCaId != null && criteriaCaId.intValue() != 0) { // JBoss EAP 6.4 sets the parameters to 0 instead of null
161             searchRequest.setCaIds(new ArrayList<>(Arrays.asList(criteriaCaId)));
162         }
163         if (criteriaRoleId != null && criteriaRoleId.intValue() != 0) {
164             searchRequest.setRoleIds(new ArrayList<>(Arrays.asList(criteriaRoleId)));
165         }
166         if (!StringUtils.isEmpty(criteriaTokenType)) {
167             searchRequest.setTokenTypes(new ArrayList<>(Arrays.asList(criteriaTokenType)));
168         }
169         searchRequest.setGenericSearchString(genericSearchString);
170         lastExecutedResponse = raMasterApiProxyBean.searchForRoleMembers(raAuthenticationBean.getAuthenticationToken(), searchRequest);
171         
172         // Add names of CAs and roles
173         resultsFiltered = new ArrayList<>();
174         for (final RoleMember member : lastExecutedResponse.getRoleMembers()) {
175             final String caName = StringUtils.defaultString(caIdToNameMap.get(member.getTokenIssuerId()), raLocaleBean.getMessage("role_members_page_info_unknownca"));
176             final String roleName = StringUtils.defaultString(roleIdToNameMap.get(member.getRoleId()));
177             final String namespace = roleIdToNamespaceMap.get(member.getRoleId());
178             final String tokenTypeText = raLocaleBean.getMessage("role_member_token_type_" + member.getTokenType());
179             resultsFiltered.add(new RaRoleMemberGUIInfo(member, caName, roleName, StringUtils.defaultString(namespace), tokenTypeText));
180         }
181         
182         sort();
183     }

193     // Sorting
194     private void sort() {
195         Collections.sort(resultsFiltered, new Comparator<RaRoleMemberGUIInfo>() {
196             @Override
197             public int compare(RaRoleMemberGUIInfo./org/ejbca/ra/RaRoleMemberGUIInfo.html#RaRoleMemberGUIInfo">RaRoleMemberGUIInfo o1, RaRoleMemberGUIInfo o2) {
198                 int sortDir = (isSortAscending() ? 1 : -1);
199                 final RoleMember rm1 = o1.getRoleMember();
200                 final RoleMember rm2 = o2.getRoleMember();
201                 switch (sortBy) {
202                 // TODO locale-aware sorting
203                 case ROLE: {
204                     int diff = o1.getRoleName().compareTo(o2.getRoleName()) * sortDir;
205                     if (diff != 0) {
206                         return diff;
207                     } else {
208                         return o1.getRoleNamespace().compareTo(o2.getRoleNamespace()) * sortDir;
209                     }
210                 }
211                 case ROLENAMESPACE: {
212                     int diff = o1.getRoleNamespace().compareTo(o2.getRoleNamespace()) * sortDir;
213                     if (diff != 0) {
214                         return diff;
215                     } else {
216                         return o1.getRoleName().compareTo(o2.getRoleName()) * sortDir;
217                     }
218                 }
219                 case CA: return o1.getCaName().compareTo(o2.getCaName()) * sortDir;
220                 case TOKENTYPE: return StringUtils.defaultString(rm1.getTokenType()).compareTo(StringUtils.defaultString(rm2.getTokenType())) * sortDir;
221                 case TOKENMATCHVALUE: return StringUtils.defaultString(rm1.getTokenMatchValue()).compareTo(StringUtils.defaultString(rm2.getTokenMatchValue())) * sortDir;
222                 case DESCRIPTION: return StringUtils.defaultString(rm1.getDescription()).compareTo(StringUtils.defaultString(rm2.getDescription())) * sortDir;
223                 default:
224                     throw new IllegalStateException("Invalid sortBy value");
225                 }
226             }
227         });
228     }

258     private String getSortedBy(final SortBy sortBy) {
259         if (this.sortBy.equals(sortBy)) {
260             return isSortAscending() ? "\u25bc" : "\u25b2";
261         }
262         return "";
263     }
264     
265     /** Set current sort column. Flip the order if the column was already selected. 
266      * @param sortBy Column
267      * @param defaultAscending Order */
268     private void sortBy(final SortBy sortBy, final boolean defaultAscending) {
269         if (this.sortBy.equals(sortBy)) {
270             sortAscending = !isSortAscending();
271         } else {
272             sortAscending = defaultAscending;
273         }
274         this.sortBy = sortBy;
275         sort();
276     }

290     public boolean isOnlyOneRoleAvailable() { return getAvailableRoles().size()==2; } // two including the "any role" choice
291     public List<SelectItem> getAvailableRoles() {
292         if (availableRoles == null) {
293             availableRoles = new ArrayList<>();
294             final List<Role> roles = new ArrayList<>(raMasterApiProxyBean.getAuthorizedRoles(raAuthenticationBean.getAuthenticationToken()));
295             Collections.sort(roles);
296             roleIdToNameMap = new HashMap<>();
297             roleIdToNamespaceMap = new HashMap<>();
298             String lastNamespace = null;
299             hasMultipleNamespaces = false;
300             for (final Role role : roles) {
301                 roleIdToNameMap.put(role.getRoleId(), role.getRoleName());
302                 if (!StringUtils.isEmpty(role.getNameSpace())) {
303                     roleIdToNamespaceMap.put(role.getRoleId(), role.getNameSpace());
304                 }
305                 // Check if there's more than one namespace. If so the namespaces are shown in the GUI
306                 if (lastNamespace != null && !lastNamespace.equals(role.getNameSpace())) {
307                     hasMultipleNamespaces = true;
308                 }
309                 lastNamespace = role.getNameSpace();
310             }
311             availableRoles.add(new SelectItem(null, raLocaleBean.getMessage("role_members_page_criteria_role_optionany")));
312             for (final Role role : roles) {
313                 final String label = hasMultipleNamespaces ? role.getRoleNameFull() : role.getRoleName();
314                 availableRoles.add(new SelectItem(role.getRoleId(), label));
315             }
316         }
317         return availableRoles;
318     }

320     public boolean isOnlyOneCaAvailable() { return getAvailableCas().size()==2; } // two including the "any CA" choice
321     public List<SelectItem> getAvailableCas() {
322         if (availableCas == null) {
323             availableCas = new ArrayList<>();
324             final List<CAInfo> caInfos = new ArrayList<>(raMasterApiProxyBean.getAuthorizedCas(raAuthenticationBean.getAuthenticationToken()));
325             Collections.sort(caInfos, new Comparator<CAInfo>() {
326                 @Override
327                 public int compare(final CAInfo caInfo1, final CAInfo caInfo2) {
328                     return caInfo1.getName().compareTo(caInfo2.getName());
329                 }
330             });
331             caIdToNameMap = new HashMap<>();
332             for (final CAInfo caInfo : caInfos) {
333                 caIdToNameMap.put(caInfo.getCAId(), caInfo.getName());
334             }
335             availableCas.add(new SelectItem(null, raLocaleBean.getMessage("role_members_page_criteria_ca_optionany")));
336             for (final CAInfo caInfo : caInfos) {
337                 availableCas.add(new SelectItem(caInfo.getCAId(), caInfo.getName()));
338             }
339         }
340         return availableCas;
341     }

345     public List<SelectItem> getAvailableTokenTypes() {
346         if (availableTokenTypes == null) {
347             if (tokenTypeInfos == null) {
348                 tokenTypeInfos = raMasterApiProxyBean.getAvailableRoleMemberTokenTypes(raAuthenticationBean.getAuthenticationToken());
349             }
350             final List<String> tokenTypes = new ArrayList<>(tokenTypeInfos.keySet());
351             Collections.sort(tokenTypes);
352             availableTokenTypes = new ArrayList<>();
353             availableTokenTypes.add(new SelectItem(null, raLocaleBean.getMessage("role_members_page_criteria_tokentype_optionany")));
354             for (final String tokenType : tokenTypes) {
355                 availableTokenTypes.add(new SelectItem(tokenType, raLocaleBean.getMessage("role_member_token_type_" + tokenType)));
356             }
357         }
358         return availableTokenTypes;
359     }
