146     public void reInitialize(final CertificateDataWrapper cdw,
147             final Map<Integer, String> cpIdToNameMap, final Map<Integer, String> eepIdToNameMap, final Map<String,String> caSubjectToNameMap) {
148         this.cdw = cdw;
149         final CertificateData certificateData = cdw.getCertificateData();
150         this.cpId = certificateData.getCertificateProfileId();
151         if (cpId != null && cpIdToNameMap != null) {
152             this.cpName = cpIdToNameMap.get(cpId);
153         } else {
154             this.cpName = null;
155         }
156         this.eepId = certificateData.getEndEntityProfileIdOrZero();
157         if (eepIdToNameMap != null) {
158             this.eepName = eepIdToNameMap.get(Integer.valueOf(eepId));
159         } else {
160             this.eepName = null;
161         }
162         this.issuerDn = certificateData.getIssuerDN();
163         if (caSubjectToNameMap != null) {
164             this.caName = getCaNameFromIssuerDn(caSubjectToNameMap, issuerDn);
165         } else {
166             this.caName = null;
167         }
168         this.status = certificateData.getStatus();
169         this.revocationReason = certificateData.getRevocationReason();
170         this.fingerprint = certificateData.getFingerprint();
171         this.serialnumberRaw = certificateData.getSerialNumber();
172         try {
173             this.serialnumber = new BigInteger(this.serialnumberRaw).toString(16);
174         } catch (NumberFormatException e) {
175             if (log.isDebugEnabled()) {
176                 log.debug("Failed to format serial number as hex. Probably a CVC certificate. Message: " + e.getMessage());
177             }
178         }
179         this.username = certificateData.getUsername()==null ? "" : certificateData.getUsername();
180         this.subjectDn = certificateData.getSubjectDnNeverNull();
181         final Certificate certificate = cdw.getCertificate();
182         byte[] certificateEncoded = null;
183         if (certificate!=null) {
184             try {
185                 certificateEncoded = certificate.getEncoded();
186             } catch (CertificateEncodingException e) {
187                 if (log.isDebugEnabled()) {
188                     log.debug("Failed to encode the certificate as a byte array: " + e.getMessage());
189                 }
190             }
191         }
192         if (certificate!=null || certificateEncoded!=null) {
193             this.type = certificate.getType();
194             this.fingerprintSha256 = new String(Hex.encode(CertTools.generateSHA256Fingerprint(certificateEncoded)));
195             final PublicKey publicKey = certificate.getPublicKey();
196             this.publicKeyAlgorithm = AlgorithmTools.getKeyAlgorithm(publicKey);
197             this.publicKeySpecification = AlgorithmTools.getKeySpecification(publicKey);
198             if (publicKey instanceof RSAPublicKey) {
199                 this.publicKeyParameter = ((RSAPublicKey)publicKey).getModulus().toString(16);
200             } else if(certificate.getPublicKey() instanceof DSAPublicKey) {
201                 this.publicKeyParameter = ((DSAPublicKey)publicKey).getY().toString(16);
202             } else if(certificate.getPublicKey() instanceof ECPublicKey) {
203                 this.publicKeyParameter = ((ECPublicKey)publicKey).getW().getAffineX().toString(16) + " " + ((ECPublicKey)publicKey).getW().getAffineY().toString(16);
204             }
205             this.created = ValidityDate.formatAsISO8601ServerTZ(CertTools.getNotBefore(certificate).getTime(), TimeZone.getDefault());
206             this.signatureAlgorithm = AlgorithmTools.getCertSignatureAlgorithmNameAsString(certificate);
207             if (certificate instanceof X509Certificate) {
208                 final X509Certificate x509Certificate = (X509Certificate)certificate;
209                 this.typeVersion = Integer.toString(x509Certificate.getVersion());
210                 this.subjectAn = CertTools.getSubjectAlternativeName(certificate);
211                 try {
212                     this.subjectDa = SubjectDirAttrExtension.getSubjectDirectoryAttributes(certificate);
213                 } catch (ParseException e) {
214                     if (log.isDebugEnabled()) {
215                         log.debug("Failed to parse Subject Directory Attributes extension: " + e.getMessage());
216                     }
217                 }
218                 final int basicConstraints = x509Certificate.getBasicConstraints();
219                 if (basicConstraints==Integer.MAX_VALUE) {
220                     this.basicConstraints = "";
221                 } else if (basicConstraints == -1) {
222                     this.basicConstraints = callbacks.getRaLocaleBean().getMessage("component_certdetails_info_basicconstraints_ee");
223                 } else {
224                     this.basicConstraints = callbacks.getRaLocaleBean().getMessage("component_certdetails_info_basicconstraints_ca", basicConstraints);
225                 }
226                 keyUsages.clear();
227                 final boolean[] keyUsageArray = x509Certificate.getKeyUsage();
228                 if (keyUsageArray != null) {
229                     for (int i=0; i<keyUsageArray.length; i++) {
230                         if (keyUsageArray[i]) {
231                             keyUsages.add(String.valueOf(i));
232                         }
233                     }
234                 }
235                 extendedKeyUsages.clear();
236                 try {
237                     final List<String> extendedKeyUsages = x509Certificate.getExtendedKeyUsage();
238                     if (extendedKeyUsages != null) {
239                         this.extendedKeyUsages.addAll(extendedKeyUsages);
240                     }
241                 } catch (CertificateParsingException e) {
242                     if (log.isDebugEnabled()) {
243                         log.debug("Failed to parse Extended Key Usage extension: " + e.getMessage());
244                     }
245                 }
246                 this.hasNameConstraints = x509Certificate.getExtensionValue(Extension.nameConstraints.getId())!=null;
247                 final CertificateTransparency ct = CertificateTransparencyFactory.getInstance();
248                 this.hasCertificateTransparencyScts = ct != null ? ct.hasSCTs(certificate) : false;
249                 this.hasQcStatements = QCStatementExtension.hasQcStatement(certificate);
250             } else if (certificate instanceof CardVerifiableCertificate) {
251                 final CardVerifiableCertificate cardVerifiableCertificate = (CardVerifiableCertificate)certificate;
252                 this.typeVersion = String.valueOf(CVCertificateBody.CVC_VERSION);
253                 // Role and access rights
254                 try {
255                     final AuthorizationField authorizationField = cardVerifiableCertificate.getCVCertificate().getCertificateBody().getAuthorizationTemplate().getAuthorizationField();
256                     if (authorizationField!=null) {
257                         this.cvcAuthorizationRole = String.valueOf(authorizationField.getAuthRole());
258                         this.cvcAuthorizationAccessRights = String.valueOf(authorizationField.getAccessRights());
259                     }
260                 } catch (NoSuchFieldException e) {
261                     if (log.isDebugEnabled()) {
262                         log.debug("Failed to parse CVC AuthorizationTemplate's AuthorizationField: " + e.getMessage());
263                     }
264                 }
265             }
266         }
267         this.expireDate = certificateData.getExpireDate();
268         this.expires = ValidityDate.formatAsISO8601ServerTZ(expireDate, TimeZone.getDefault());
269         if (status==CertificateConstants.CERT_ARCHIVED || status==CertificateConstants.CERT_REVOKED) {
270             this.updated = ValidityDate.formatAsISO8601ServerTZ(certificateData.getRevocationDate(), TimeZone.getDefault());
271             this.revocationDate = ValidityDate.formatAsISO8601ServerTZ(certificateData.getRevocationDate(), TimeZone.getDefault());
272         } else {
273             this.updated = ValidityDate.formatAsISO8601ServerTZ(certificateData.getUpdateTime(), TimeZone.getDefault());
274         }
275         final String subjectKeyIdB64 = certificateData.getSubjectKeyId();
276         if (subjectKeyIdB64!=null) {
277             this.subjectKeyId = new String(Hex.encode(Base64.decode(subjectKeyIdB64.getBytes())));
278         }
279         styleRowCallCounter = 0;    // Reset
280     }

365     public String getDump() {
366         final Certificate certificate = cdw.getCertificate();
367         if (certificate!=null) {
368             try {
369                 return CertTools.dumpCertificateAsString(certificate);
370             } catch (RuntimeException e) {
371                 try {
372                     return ASN1Dump.dumpAsString(ASN1Primitive.fromByteArray(certificate.getEncoded()));
373                 } catch (CertificateEncodingException | IOException e2) {
374                     if (log.isDebugEnabled()) {
375                         log.debug("Failed to parse certificate ASN.1: " + e2.getMessage());
376                     }
377                 }
378             }
379         }
380         return "";
381     }

425         ret.add(new SelectItem(Integer.valueOf(RevokedCertInfo.REVOCATION_REASON_CACOMPROMISE), callbacks.getRaLocaleBean().getMessage("component_certdetails_status_revoked_reason_2")));
426         ret.add(new SelectItem(Integer.valueOf(RevokedCertInfo.REVOCATION_REASON_AFFILIATIONCHANGED), callbacks.getRaLocaleBean().getMessage("component_certdetails_status_revoked_reason_3")));
427         ret.add(new SelectItem(Integer.valueOf(RevokedCertInfo.REVOCATION_REASON_SUPERSEDED), callbacks.getRaLocaleBean().getMessage("component_certdetails_status_revoked_reason_4")));

441     public void actionRevoke() {
442         try {
443             if (callbacks.changeStatus(this, CertificateConstants.CERT_REVOKED, newRevocationReason)) {
444                 callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_info_revocation_successful");
445             } else {
446                 callbacks.getRaLocaleBean().addMessageError("component_certdetails_error_revocation_failed");
447             }
448         } catch (ApprovalException e) {
449             callbacks.getRaLocaleBean().addMessageError("component_certdetails_error_revocation_approvalrequest");
450         } catch (WaitingForApprovalException e) {
451             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_info_revocation_approvalrequest", e.getRequestId());
452         }
453         styleRowCallCounter = 0;    // Reset
454     }

456     public void actionReactivate() {
457         try {

459                 callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_info_reactivation_successful");
460             } else {
461                 callbacks.getRaLocaleBean().addMessageError("component_certdetails_error_reactivation_failed");

464             callbacks.getRaLocaleBean().addMessageError("component_certdetails_error_reactivation_approvalrequest");
465         } catch (WaitingForApprovalException e) {
466             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_info_reactivation_approvalrequest", e.getRequestId());
467         }
468         styleRowCallCounter = 0;    // Reset
469     }

471     public void actionRecovery() {
472         try {
473             if (callbacks.recoverKey(this)) {
474                 callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_successful");
475             } else {
476                 callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_unknown_error");
477                 log.info("Failed to perform key recovery for user: " + subjectDn);
478             }
479         } catch (ApprovalException e) {
480             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_pending");
481             if (log.isDebugEnabled()) {
482                 log.debug("Request is still waiting for approval", e);
483             }
484         } catch (WaitingForApprovalException e) {
485             // Setting requestId will render link to 'enroll with request id' page
486             requestId = e.getRequestId();
487             log.info("Request with Id: " + e.getRequestId() + " has been sent for approval");
488         } catch (CADoesntExistsException e) {
489             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_unknown_error");
490                 log.debug("CA does not exist", e);
491         } catch (AuthorizationDeniedException e) {
492             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_unauthorized");
493                 log.debug("Not authorized to perform key recovery", e);
494         } catch (NoSuchEndEntityException e) {
495             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_no_such_end_entity", username);
496             if (log.isDebugEnabled()) {
497                 log.debug("End entity with username: " + username + " does not exist", e);
498             }
499         } catch (EndEntityProfileValidationException e) {
500             callbacks.getRaLocaleBean().addMessageInfo("component_certdetails_keyrecovery_unknown_error");
501             if (log.isDebugEnabled()) {
502                 log.debug("End entity with username: " + username + " does not match end entity profile");
503             }
504         }
505         styleRowCallCounter = 0;    // Reset
506         renderConfirmRecoveryToggle();

509     /** Validate that password and password confirm entries match and render error messages otherwise. 
510      * @param event event */
511     public final void validatePassword(ComponentSystemEvent event) {
512         if (renderConfirmRecovery){
513             FacesContext fc = FacesContext.getCurrentInstance();
514             UIComponent components = event.getComponent();
515             UIInput uiInputPassword = (UIInput) components.findComponent("passwordField");
516             String password = uiInputPassword.getLocalValue() == null ? "" : uiInputPassword.getLocalValue().toString();
517             UIInput uiInputConfirmPassword = (UIInput) components.findComponent("passwordConfirmField");
518             String confirmPassword = uiInputConfirmPassword.getLocalValue() == null ? "" : uiInputConfirmPassword.getLocalValue().toString();
519             if (password.isEmpty()){
520                 fc.addMessage(callbacks.getConfirmPasswordComponent().getClientId(fc), callbacks.getRaLocaleBean().getFacesMessage("enroll_password_can_not_be_empty"));
521                 fc.renderResponse();
522             }
523             if (!password.equals(confirmPassword)) {
524                 fc.addMessage(callbacks.getConfirmPasswordComponent().getClientId(fc), callbacks.getRaLocaleBean().getFacesMessage("enroll_passwords_are_not_equal"));
525                 fc.renderResponse();
526             }
527         }
528     }
