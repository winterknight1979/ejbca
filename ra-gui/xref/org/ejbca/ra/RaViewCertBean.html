42  /**
43   * Backing bean for certificate details view.
44   *  
45   * @version $Id: RaViewCertBean.java 26138 2017-07-06 14:39:48Z henriks $
46   * TODO: Use CDI beans
47   */
48  @SuppressWarnings("deprecation")
49  @ManagedBean
50  @ViewScoped
51  public class RaViewCertBean implements Serializable {
52  
53      private static final long serialVersionUID = 1L;
54  
55      @EJB
56      private RaMasterApiProxyBeanLocal raMasterApiProxyBean;
57  
58      @ManagedProperty(value="#{raAuthenticationBean}")
59      private RaAuthenticationBean raAuthenticationBean;
60      public void setRaAuthenticationBean(final RaAuthenticationBean raAuthenticationBean) { this.raAuthenticationBean = raAuthenticationBean; }
61  
62      @ManagedProperty(value="#{raLocaleBean}")
63      private RaLocaleBean raLocaleBean;
64      public void setRaLocaleBean(final RaLocaleBean raLocaleBean) { this.raLocaleBean = raLocaleBean; }
65  
66      private String fingerprint = null;
67      private RaCertificateDetails raCertificateDetails = null;
68      private Map<Integer, String> eepIdToNameMap = null;
69      private Map<Integer, String> cpIdToNameMap = null;
70      private Map<String,String> caSubjectToNameMap = new HashMap<>();
71  
72      private final Callbacks raCertificateDetailsCallbacks = new RaCertificateDetails.Callbacks() {
73          @Override
74          public RaLocaleBean getRaLocaleBean() {
75              return raLocaleBean;
76          }
77          @Override
78          public UIComponent getConfirmPasswordComponent() {
79              return null;
80          }
81          @Override
82          public boolean changeStatus(RaCertificateDetails raCertificateDetails, int newStatus, int newRevocationReason) throws ApprovalException, WaitingForApprovalException {
83              final boolean ret = raMasterApiProxyBean.changeCertificateStatus(raAuthenticationBean.getAuthenticationToken(), raCertificateDetails.getFingerprint(),
84                      newStatus, newRevocationReason);
85              if (ret) {
86                  // Re-initialize object if status has changed
87                  final CertificateDataWrapper cdw = raMasterApiProxyBean.searchForCertificate(raAuthenticationBean.getAuthenticationToken(), raCertificateDetails.getFingerprint());
88                  raCertificateDetails.reInitialize(cdw, cpIdToNameMap, eepIdToNameMap, caSubjectToNameMap);
89              }
90              return ret;
91          }
92          @Override
93          public boolean recoverKey(RaCertificateDetails raCertificateDetails) throws ApprovalException, CADoesntExistsException, AuthorizationDeniedException, WaitingForApprovalException, 
94                                      NoSuchEndEntityException, EndEntityProfileValidationException {
95              final boolean ret = raMasterApiProxyBean.markForRecovery(raAuthenticationBean.getAuthenticationToken(), raCertificateDetails.getUsername(), raCertificateDetails.getPassword(), 
96                                      EJBTools.wrap(raCertificateDetails.getCertificate()), false);
97              return ret;
98          }
99          @Override
100         public boolean keyRecoveryPossible(RaCertificateDetails raCertificateDetails) {
101             final boolean ret = raMasterApiProxyBean.keyRecoveryPossible(raAuthenticationBean.getAuthenticationToken(), raCertificateDetails.getCertificate(), raCertificateDetails.getUsername());
102             return ret;
103         }
104     };
105 
106     @PostConstruct
107     public void postConstruct() {
108         fingerprint = ((HttpServletRequest) FacesContext.getCurrentInstance().getExternalContext().getRequest()).getParameter("fp");
109         if (fingerprint!=null) {
110             final CertificateDataWrapper cdw = raMasterApiProxyBean.searchForCertificate(raAuthenticationBean.getAuthenticationToken(), fingerprint);
111             if (cdw!=null) {
112                 cpIdToNameMap = raMasterApiProxyBean.getAuthorizedCertificateProfileIdsToNameMap(raAuthenticationBean.getAuthenticationToken());
113                 eepIdToNameMap = raMasterApiProxyBean.getAuthorizedEndEntityProfileIdsToNameMap(raAuthenticationBean.getAuthenticationToken());
114                 final List<CAInfo> caInfos = new ArrayList<>(raMasterApiProxyBean.getAuthorizedCas(raAuthenticationBean.getAuthenticationToken()));
115                 for (final CAInfo caInfo : caInfos) {
116                     caSubjectToNameMap.put(caInfo.getSubjectDN(), caInfo.getName());
117                 }
118                 raCertificateDetails = new RaCertificateDetails(cdw, raCertificateDetailsCallbacks, cpIdToNameMap, eepIdToNameMap, caSubjectToNameMap);
119             }
120         }
121     }
122 
123     public String getFingerprint() { return fingerprint; }
124     public RaCertificateDetails getCertificate() { return raCertificateDetails; }
125 }
