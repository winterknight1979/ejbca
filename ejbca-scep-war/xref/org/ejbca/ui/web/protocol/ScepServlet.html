55  /**
56   * Servlet implementing server side of the Simple Certificate Enrollment Protocol (SCEP) 
57   * ----- 
58   * This processes does the following: 
59   * 1. decode a PKCS#7 signed data message from the standard input 
60   * 2. extract the signed attributes from the the message, which indicate the type of request 
61   * 3. decrypt the enveloped data PKCS#7 inside 
62   * 4. branch to different actions depending on the type of the message: 
63   * - PKCSReq 
64   * - GetCertInitial 
65   * - GetCert 
66   * - GetCRL 
67   * - v2PKCSReq or Proxy request 
68   * - GetCACaps
69   * 5. envelop (PKCS#7) the reply data from the previous step 
70   * 6. sign the reply data (PKCS#7) from the previous step 
71   * 7. output the result as a der encoded block on stdout 
72   * -----
73   *
74   * @version $Id: ScepServlet.java 28857 2018-05-07 08:35:30Z samuellb $
75   */
76  public class ScepServlet extends HttpServlet {
77      private static final long serialVersionUID = -6776853218419335240L;
78      private static final Logger log = Logger.getLogger(ScepServlet.class);
79      /** Internal localization of logs and errors */
80      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
81      /** Only intended to check if Peer connected instance is authorized to SCEP at all. This will not affect user authorization */
82      private final AuthenticationToken raScepAuthCheckToken = new AlwaysAllowLocalAuthenticationToken(new UsernamePrincipal("scepProtocolAuthCheck"));
83      
84      @EJB
85      private RaMasterApiProxyBeanLocal raMasterApiProxyBean;
86      
87      private static final String DEFAULT_SCEP_ALIAS = "scep";

107     /**
108      * Handles HTTP post
109      *
110      * @param request java standard arg
111      * @param response java standard arg
112      *
113      * @throws IOException input/output error
114      * @throws ServletException if the post could not be handled
115      */
116     @Override
117     public void doPost(HttpServletRequest request, HttpServletResponse response)
118             throws IOException, ServletException {
119         log.trace(">SCEP doPost()");
120         final boolean isProtocolAuthorized = raMasterApiProxyBean.isAuthorizedNoLogging(raScepAuthCheckToken,
121                 AccessRulesConstants.REGULAR_PEERPROTOCOL_SCEP);
122         if (!isProtocolAuthorized) {
123             log.info("SCEP Protocol not authorized for this Peer");
124             response.sendError(HttpServletResponse.SC_FORBIDDEN, "SCEP Protocol not authorized for this Peer");
125             return;
126         }
127         
128         /* 
129          If the remote CA supports it, any of the PKCS#7-encoded SCEP messages
130          may be sent via HTTP POST instead of HTTP GET.   This is allowed for
131          any SCEP message except GetCACert, GetCACertChain, GetNextCACert,
132          or GetCACaps.  In this form of the message, Base 64 encoding is not
133          used.
134          
135          POST /cgi-bin/pkiclient.exe?operation=PKIOperation
136          <binary PKCS7 data>
137          */
138         String operation = "PKIOperation";
139         ServletInputStream sin = request.getInputStream();
140         // This small code snippet is inspired/copied by apache IO utils to Tomas Gustavsson...
141         ByteArrayOutputStream output = new ByteArrayOutputStream();
142         byte[] buf = new byte[1024];
143         int n = 0;
144         while (-1 != (n = sin.read(buf))) {
145             output.write(buf, 0, n);
146         }
147         String message = new String(Base64.encode(output.toByteArray()));
148         service(operation, message, request.getRemoteAddr(), response, request.getPathInfo());
149         log.trace("<SCEP doPost()");
150     }

152     /**
153      * Handles HTTP get
154      *
155      * @param request java standard arg
156      * @param response java standard arg
157      *
158      * @throws IOException input/output error
159      * @throws ServletException if the post could not be handled
160      */
161     @Override
162     public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
163         log.trace(">SCEP doGet()");
164         if (log.isDebugEnabled()) {
165             log.debug("query string=" + request.getQueryString());
166         }
167         final boolean isProtocolAuthorized = raMasterApiProxyBean.isAuthorizedNoLogging(raScepAuthCheckToken,
168                 AccessRulesConstants.REGULAR_PEERPROTOCOL_SCEP);
169         if (!isProtocolAuthorized) {
170             log.info("SCEP Protocol not authorized for this Peer");
171             response.sendError(HttpServletResponse.SC_FORBIDDEN, "SCEP Protocol not authorized for this Peer");
172             return;
173         }
174         
175         // These are mandatory in SCEP GET
176         /*
177          GET /cgi-bin/pkiclient.exe?operation=PKIOperation&message=MIAGCSqGSIb3D
178          QEHA6CAMIACAQAxgDCBzAIBADB2MGIxETAPBgNVBAcTCE ......AAAAAA== 
179          */
180         String operation = request.getParameter("operation");
181         String message = request.getParameter("message");
182         // Some clients don't url encode the + sign in the request. Message is only used to PKIOperations
183         if (message != null && operation != null && operation.equals("PKIOperation")) {
184             message = message.replace(' ', '+');
185         }
186 
187         service(operation, message, request.getRemoteAddr(), response, request.getPathInfo());
188 
189         log.trace("<SCEP doGet()");
190     }

192     private void service(String operation, String message, String remoteAddr, HttpServletResponse response, String pathInfo) throws IOException {
193         String alias = getAlias(pathInfo);
194         String caname = getCAName(message);
195         if(alias == null) {
196             log.info("Wrong URL format. The SCEP URL should look like: " +
197             		"'http://HOST:PORT/ejbca/publicweb/apply/scep/ALIAS/pkiclient.exe' " +
198             		"but was 'http://HOST:PORT/ejbca/publicweb/apply/scep" + pathInfo + "'");
199             response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Wrong URL. No alias found.");
200             return;
201         }
202         if(alias.length() > 32) {
203             log.info("Unaccepted alias more than 32 characters.");
204             response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Unaccepted alias more than 32 characters.");
205             return;
206         }
207         
208         try {
209             if (operation == null) {
210         		String errMsg = intres.getLocalizedMessage("scep.errormissingparam", remoteAddr);
211                 log.error(errMsg);
212                 response.sendError(HttpServletResponse.SC_BAD_REQUEST,errMsg);
213                 return;
214             }
215             if (operation.equals("PKIOperation") && message == null) {
216                 if (message == null) {
217                     String errMsg = intres.getLocalizedMessage("scep.errormissingparam", remoteAddr);
218                     log.error(errMsg);
219                     response.sendError(HttpServletResponse.SC_BAD_REQUEST,errMsg);
220                     return;
221                 }
222             }
223             final AuthenticationToken administrator = new AlwaysAllowLocalAuthenticationToken(new WebPrincipal("ScepServlet", remoteAddr));
224             if (log.isDebugEnabled()) {
225                 log.debug("Got request '" + operation + "'");
226                 log.debug("Message: " + message);
227             }
228     		String iMsg = intres.getLocalizedMessage("scep.receivedmsg", remoteAddr);
229 			log.info(iMsg);	
230 			byte[] dispatchResponse = raMasterApiProxyBean.scepDispatch(administrator, operation, message, alias);
231 			
232             if (operation.equals("PKIOperation")) {
233                 if (dispatchResponse == null) {
234                     response.sendError(HttpServletResponse.SC_NOT_IMPLEMENTED, "Can not handle request");
235                     return;
236                 }
237                 // Send back Scep response, PKCS#7 which contains the end entity's certificate (or failure)
238                 RequestHelper.sendBinaryBytes(dispatchResponse, response, "application/x-pki-message", null);
239         		iMsg = intres.getLocalizedMessage("scep.sentresponsemsg", "PKIOperation", remoteAddr);
240     			log.info(iMsg);
241             } else if (operation.equals("GetCACert")) {
242                 // The response has the content type tagged as application/x-x509-ca-cert. 
243                 // The body of the response is a DER encoded binary X.509 certificate. 
244                 // For example: "Content-Type:application/x-x509-ca-cert\n\n"<BER-encoded X509>
245                 if (dispatchResponse != null) {
246                     log.debug("Sent CA certificate to SCEP client.");
247                     RequestHelper.sendNewX509CaCert(dispatchResponse, response);
248             		iMsg = intres.getLocalizedMessage("scep.sentresponsemsg", "GetCACert", remoteAddr);
249         			log.info(iMsg);
250                 } else {
251             		String errMsg = intres.getLocalizedMessage("scep.errorunknownca", "cert");
252                     log.error(errMsg);
253                     response.sendError(HttpServletResponse.SC_NOT_FOUND, "No CA certificates found.");
254                 }
255             } else if (operation.equals("GetCACertChain")) {
256                 // GetCACertChain was included in SCEP draft 18, "5.6.  Get Certificate Authority Certificate Chain"
257                 // This dissapeared on SCEP draft 19 however, so we should not expect any clients to use this method.
258                 
259                 // The response for GetCACertChain is a certificates-only PKCS#7 
260                 // SignedData to carry the certificates to the end entity, with a 
261                 // Content-Type of application/x-x509-ca-ra-cert-chain.
262                 if (dispatchResponse != null) {
263                     log.debug("Sent PKCS7 for CA to SCEP client.");
264                     RequestHelper.sendBinaryBytes(dispatchResponse, response, "application/x-x509-ca-ra-cert-chain", null);
265             		iMsg = intres.getLocalizedMessage("scep.sentresponsemsg", "GetCACertChain", remoteAddr);
266         			log.info(iMsg);
267                 } else {
268             		String errMsg = intres.getLocalizedMessage("scep.errorunknownca", "pkcs7");
269                     log.error(errMsg);
270                     response.sendError(HttpServletResponse.SC_NOT_FOUND,"No CA certificates found.");
271                 }
272             } else if (operation.equals("GetNextCACert")) {
273                 // Like GetCACert, but returns the next certificate during certificate rollover
274                     if (dispatchResponse != null) {
275                         RequestHelper.sendBinaryBytes(dispatchResponse, response, "application/x-x509-next-ca-cert", null);
276                         iMsg = intres.getLocalizedMessage("scep.sentresponsemsg", "GetNextCACert", remoteAddr);
277                         log.info(iMsg);
278                     } else {
279                         String errMsg = intres.getLocalizedMessage("scep.errornorollovercert", caname);
280                         log.info(errMsg);
281                         response.sendError(HttpServletResponse.SC_FORBIDDEN, "No rollover certificate found for this CA.");
282                     }
283             } else if (operation.equals("GetCACaps")) {
284                 // The response for GetCACaps is a <lf> separated list of capabilities
285 
286                 /*
287                  "GetNextCACert"       CA Supports the GetNextCACert message.
288                  "POSTPKIOperation"    PKIOPeration messages may be sent via HTTP POST.
289                  "SHA-1"               CA Supports the SHA-1 hashing algorithm in 
290                                        signatures and fingerprints.  If present, the
291                                        client SHOULD use SHA-1.  If absent, the client
292                                        MUST use MD5 to maintain backward compatability.
293                  "Renewal"             Clients may use current certificate and key to
294                                        authenticate an enrollment request for a new
295                                        certificate.  
296                  */
297                 log.debug("Got SCEP GetCACaps request");
298                 response.setContentType("text/plain");
299                 response.getOutputStream().print(new String(dispatchResponse));
300             } else {
301                 log.error("Invalid parameter '" + operation);
302                 // Send back proper Failure Response
303                 response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Invalid parameter: " + HTMLTools.htmlescape(operation));
304             }
305         } catch (CADoesntExistsException cae) {
306     		String errMsg = intres.getLocalizedMessage("scep.errorunknownca", "cert");
307             log.info(errMsg, cae);
308             // TODO: Send back proper Failure Response
309             response.sendError(HttpServletResponse.SC_NOT_FOUND, errMsg);
310         } catch (DecoderException de) {
311     		String errMsg = intres.getLocalizedMessage("scep.errorinvalidreq");
312             log.info(errMsg, de);
313             // TODO: Send back proper Failure Response
314             response.sendError(HttpServletResponse.SC_BAD_REQUEST,errMsg);
315         } catch (AuthorizationDeniedException ae) {
316     		String errMsg = intres.getLocalizedMessage("scep.errorauth");
317             log.info(errMsg, ae);
318             // TODO: Send back proper Failure Response
319             response.sendError(HttpServletResponse.SC_UNAUTHORIZED, errMsg);
320         } catch (AuthLoginException ae) {
321             final String errMsg = intres.getLocalizedMessage("scep.errorauth");
322             if (log.isDebugEnabled()) {
323                 // AuthLogin is logged as a security event already by inner layers, not need to log the exception at info level
324                 // this is seens clearly in the info log already, more details is inly needed in debug level
325                 log.debug(errMsg, ae);
326             }
327             // TODO: Send back proper Failure Response
328             response.sendError(HttpServletResponse.SC_UNAUTHORIZED, errMsg);
329         } catch (AuthStatusException ae) {
330     		String errMsg = intres.getLocalizedMessage("scep.errorclientstatus");
331             log.info(errMsg, ae);
332             // TODO: Send back proper Failure Response
333             response.sendError(HttpServletResponse.SC_UNAUTHORIZED, errMsg);
334         } catch (CryptoTokenOfflineException ee) {
335     		String errMsg = intres.getLocalizedMessage("scep.errorgeneral");
336             log.info(errMsg, ee);
337             // TODO: Send back proper Failure Response
338             response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, errMsg);
339         } catch (NoSuchEndEntityException ee) {
340             String errMsg = intres.getLocalizedMessage("scep.errorgeneral");
341             errMsg += " Registering new EndEntities is only allowed in RA mode.";
342             log.info(errMsg, ee);
343             response.sendError(HttpServletResponse.SC_FORBIDDEN, errMsg);    
344         } catch(IllegalKeyException e) {
345             String errMsg = intres.getLocalizedMessage("scep.errorclientcertificaterenewal");
346             errMsg += " Reusing the old keys was attempted, but this action is prohibited by configuration.";
347             log.info(errMsg, e);
348             response.sendError(HttpServletResponse.SC_BAD_REQUEST, errMsg);    
349         } catch(SignatureException e) {
350             String errMsg = intres.getLocalizedMessage("scep.errorclientcertificaterenewal");
351             errMsg += " Request was not signed with previous certificate's public key.";
352             log.info(errMsg, e);
353             response.sendError(HttpServletResponse.SC_BAD_REQUEST, errMsg);
354         } catch(CertificateRenewalException e) {
355             String errMsg = intres.getLocalizedMessage("scep.errorclientcertificaterenewal");
356             log.info(errMsg, e);
357             response.sendError(HttpServletResponse.SC_BAD_REQUEST, errMsg);
358         } catch (NoSuchAliasException e) {
359             String msg = intres.getLocalizedMessage("protocol.nosuchalias", "SCEP", alias);
360             log.info(msg);
361             response.sendError(HttpServletResponse.SC_BAD_REQUEST, msg);
362         } catch (Exception e) {             
363     		String errMsg = intres.getLocalizedMessage("scep.errorgeneral");
364             log.info(errMsg, e);
365             response.sendError(HttpServletResponse.SC_BAD_REQUEST, errMsg);
366         }
367     }

369     /** Later SCEP draft say that for GetCACert message is optional. If message is there, it is the CA name
370      * but if message is not provided by the client, some default CA should be used.
371      * @param message the message part for the SCEP get request, can be null or empty string
372      * @return the message parameter or the default CA from ALIAS.defaultca property if message is null or empty.
373      */
374     private String getCAName(final String message) {
375         // If message is a string, return it, but if message is empty return default CA
376         if (StringUtils.isEmpty(message)) {
377             return EjbcaConfiguration.getScepDefaultCA();
378         }
379         return message;
380     }

382     public static String getAlias(String pathInfo) {
383         // PathInfo contains the alias used for SCEP configuration. 
384         // The SCEP URL for custom configuration looks like: http://HOST:PORT/ejbca/publicweb/apply/scep/*
385         // pathInfo contains what * is and should have the form "/<SOME IDENTIFYING TEXT>/pkiclient.exe". We extract the "SOME IDENTIFYING 
386         // TEXT" and that will be the SCEP configuration alias.
387         
388         String alias = null;
389         Pattern pattern = Pattern.compile("/?([A-Za-z0-9]*)/pkiclient.exe");
390         Matcher matcher = pattern.matcher(pathInfo);
391         
392         if(matcher.find()) {
393             alias = matcher.group(1);
394             if(alias.length() == 0) {
395                 log.info("No SCEP alias specified in the URL. Using the default alias: " + DEFAULT_SCEP_ALIAS);
396                 alias = DEFAULT_SCEP_ALIAS;
397             } else {
398                 if(log.isDebugEnabled()) {
399                     log.debug("Found SCEP configuration alias: " + alias);
400                 }
401             }
402         }
403         return alias;
404     }
