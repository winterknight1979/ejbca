44    /**
45     * When caching is disabled (cachetime=-1) we expect that nothing sticks and
46     * that we always are told to fetch the object from the original source
47     * instead of the cache.
48     */
49    @Test
50    public void disabledCacheBehavior() {
51      LOG.trace(">" + Thread.currentThread().getStackTrace()[1].getMethodName());
52      ConfigurationHolder.updateConfiguration("validator.cachetime", "-1");
53      final String name1 = KeyValidatorCacheTest.class.getSimpleName() + " Publ1";
54      final String name2 = KeyValidatorCacheTest.class.getSimpleName() + " Publ2";
55      final String name3 = KeyValidatorCacheTest.class.getSimpleName() + " Publ3";
56      final ValidatorBase keyValidator1 = getNewKeyValidator(name1);
57      final ValidatorBase keyValidator2 = getNewKeyValidator(name2);
58      final ValidatorBase keyValidator3 = getNewKeyValidator(name3);
59      // Start out with an empty cache and verify state
60      ValidatorCache.INSTANCE.flush();
61      assertEquals(
62          "Expected empty map after flush.",
63          0,
64          ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
65      assertNull(
66          "Empty cache returned an entry anyway.",
67          ValidatorCache.INSTANCE.getEntry(1));
68      assertNull(
69          "Empty cache returned an entry anyway.",
70          ValidatorCache.INSTANCE.getEntry(1));
71      assertNull(
72          "Empty cache returned an entry anyway.",
73          ValidatorCache.INSTANCE.getEntry(1));
74      assertTrue(
75          "Cache indicated that non-existing object is valid.",
76          ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
77      assertTrue(
78          "Cache indicated that non-existing object is valid.",
79          ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
80      assertTrue(
81          "Cache indicated that non-existing object is valid.",
82          ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
83      assertNull(
84          "Empty cache returned a name anyway",
85          ValidatorCache.INSTANCE.getName(1));
86      assertNull(
87          "Empty cache returned a name anyway",
88          ValidatorCache.INSTANCE.getName(2));
89      assertNull(
90          "Empty cache returned a name anyway",
91          ValidatorCache.INSTANCE.getName(3));
92      // Add entries to the (disabled) cache
93      // ECA-4219 Fix. Disable cache does not work.
94      ValidatorCache.INSTANCE.updateWith(1, 1, name1, keyValidator1);
95      ValidatorCache.INSTANCE.updateWith(2, 1, name2, keyValidator2);
96      ValidatorCache.INSTANCE.updateWith(3, 1, name3, keyValidator3);
97      assertEquals(
98          "Expected empty map for disabled cache.",
99          0,
100         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
101     assertNull(
102         "Empty cache returned an entry anyway.",
103         ValidatorCache.INSTANCE.getEntry(1));
104     assertNull(
105         "Empty cache returned an entry anyway.",
106         ValidatorCache.INSTANCE.getEntry(2));
107     assertNull(
108         "Empty cache returned an entry anyway.",
109         ValidatorCache.INSTANCE.getEntry(3));
110     assertTrue(
111         "Cache indicated that non-existing object is valid.",
112         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
113     assertTrue(
114         "Cache indicated that non-existing object is valid.",
115         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
116     assertTrue(
117         "Cache indicated that non-existing object is valid.",
118         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
119     assertNull(
120         "Empty cache returned a name anyway",
121         ValidatorCache.INSTANCE.getName(1));
122     assertNull(
123         "Empty cache returned a name anyway",
124         ValidatorCache.INSTANCE.getName(2));
125     assertNull(
126         "Empty cache returned a name anyway",
127         ValidatorCache.INSTANCE.getName(3));
128     // Notify cache that the objects should be removed from the cache
129     ValidatorCache.INSTANCE.updateWith(1, 1, null, null);
130     ValidatorCache.INSTANCE.updateWith(2, 1, null, keyValidator2);
131     ValidatorCache.INSTANCE.updateWith(3, 1, name3, null);
132     assertEquals(
133         "Expected empty map for after removing objects.",
134         0,
135         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
136     assertNull(
137         "Empty cache returned an entry anyway.",
138         ValidatorCache.INSTANCE.getEntry(1));
139     assertNull(
140         "Empty cache returned an entry anyway.",
141         ValidatorCache.INSTANCE.getEntry(2));
142     assertNull(
143         "Empty cache returned an entry anyway.",
144         ValidatorCache.INSTANCE.getEntry(3));
145     assertTrue(
146         "Cache indicated that non-existing object is valid.",
147         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
148     assertTrue(
149         "Cache indicated that non-existing object is valid.",
150         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
151     assertTrue(
152         "Cache indicated that non-existing object is valid.",
153         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
154     assertNull(
155         "Empty cache returned a name anyway",
156         ValidatorCache.INSTANCE.getName(1));
157     assertNull(
158         "Empty cache returned a name anyway",
159         ValidatorCache.INSTANCE.getName(2));
160     assertNull(
161         "Empty cache returned a name anyway",
162         ValidatorCache.INSTANCE.getName(3));
163     LOG.trace("<" + Thread.currentThread().getStackTrace()[1].getMethodName());
164   }

166   /**
167    * Verify that objects put in the cache are available from it and that the
168    * name to id mapping works. Also verify that a change in the provided digest
169    * is required for any cache update to take place.
170    */
171   @Test
172   public void enabledCacheBehavior() {
173     LOG.trace(">" + Thread.currentThread().getStackTrace()[1].getMethodName());
174     ConfigurationHolder.updateConfiguration("validator.cachetime", "3000");
175     final String name1 = KeyValidatorCacheTest.class.getSimpleName() + " Publ1";
176     final String name2 = KeyValidatorCacheTest.class.getSimpleName() + " Publ2";
177     final String name3 = KeyValidatorCacheTest.class.getSimpleName() + " Publ3";
178     final ValidatorBase keyValidator1 = getNewKeyValidator(name1);
179     final ValidatorBase keyValidator2 = getNewKeyValidator(name2);
180     final ValidatorBase keyValidator3 = getNewKeyValidator(name3);
181     // Start out with an empty cache and verify state
182     ValidatorCache.INSTANCE.flush();
183     assertEquals(
184         "Expected empty map after flush.",
185         0,
186         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
187     assertNull(
188         "Empty cache returned an entry anyway.",
189         ValidatorCache.INSTANCE.getEntry(1));
190     assertNull(
191         "Empty cache returned an entry anyway.",
192         ValidatorCache.INSTANCE.getEntry(2));
193     assertNull(
194         "Empty cache returned an entry anyway.",
195         ValidatorCache.INSTANCE.getEntry(3));
196     assertTrue(
197         "Cache indicated that non-existing object is valid.",
198         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
199     assertTrue(
200         "Cache indicated that non-existing object is valid.",
201         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
202     assertTrue(
203         "Cache indicated that non-existing object is valid.",
204         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
205     assertNull(
206         "Empty cache returned a name anyway",
207         ValidatorCache.INSTANCE.getName(1));
208     assertNull(
209         "Empty cache returned a name anyway",
210         ValidatorCache.INSTANCE.getName(2));
211     assertNull(
212         "Empty cache returned a name anyway",
213         ValidatorCache.INSTANCE.getName(3));
214     // Add entries to the (enabled) cache
215     ValidatorCache.INSTANCE.updateWith(1, 1, name1, keyValidator1);
216     ValidatorCache.INSTANCE.updateWith(2, 1, name2, keyValidator2);
217     ValidatorCache.INSTANCE.updateWith(3, 1, name3, keyValidator3);
218     assertEquals(
219         "Expected non-empty map for enabled cache.",
220         3,
221         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
222     assertEquals(
223         "Wrong mapping for name.",
224         1,
225         ValidatorCache.INSTANCE.getNameToIdMap().get(name1).intValue());
226     assertEquals(
227         "Wrong mapping for name.",
228         2,
229         ValidatorCache.INSTANCE.getNameToIdMap().get(name2).intValue());
230     assertEquals(
231         "Wrong mapping for name.",
232         3,
233         ValidatorCache.INSTANCE.getNameToIdMap().get(name3).intValue());
234     assertEquals(
235         "Expected cache to return same instance of object 1.",
236         keyValidator1,
237         ValidatorCache.INSTANCE.getEntry(1));
238     assertEquals(
239         "Expected cache to return same instance of object 2.",
240         keyValidator2,
241         ValidatorCache.INSTANCE.getEntry(2));
242     assertEquals(
243         "Expected cache to return same instance of object 3.",
244         keyValidator3,
245         ValidatorCache.INSTANCE.getEntry(3));
246     assertFalse(
247         "Cache indicated that existing object is invalid.",
248         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
249     assertFalse(
250         "Cache indicated that existing object is invalid.",
251         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
252     assertFalse(
253         "Cache indicated that existing object is invalid.",
254         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
255     assertEquals(
256         "Wrong mapping for name", name1, ValidatorCache.INSTANCE.getName(1));
257     assertEquals(
258         "Wrong mapping for name", name2, ValidatorCache.INSTANCE.getName(2));
259     assertEquals(
260         "Wrong mapping for name", name3, ValidatorCache.INSTANCE.getName(3));
261     // Update test with and without changing the digest
262     ValidatorCache.INSTANCE.updateWith(1, 1, name2, keyValidator2);
263     ValidatorCache.INSTANCE.updateWith(2, 2, name3, keyValidator3);
264     ValidatorCache.INSTANCE.updateWith(3, 2, name2, keyValidator2);
265     assertEquals(
266         "Wrong mapping for name.",
267         1,
268         ValidatorCache.INSTANCE.getNameToIdMap().get(name1).intValue());
269     assertEquals(
270         "Wrong mapping for name.",
271         2,
272         ValidatorCache.INSTANCE.getNameToIdMap().get(name3).intValue());
273     assertEquals(
274         "Wrong mapping for name.",
275         3,
276         ValidatorCache.INSTANCE.getNameToIdMap().get(name2).intValue());
277     assertEquals(
278         "Cache was updated even though digest didn't change.",
279         keyValidator1,
280         ValidatorCache.INSTANCE.getEntry(1));
281     assertEquals(
282         "Cache wasn't updated even though digest did change.",
283         keyValidator3,
284         ValidatorCache.INSTANCE.getEntry(2));
285     assertEquals(
286         "Cache wasn't updated even though digest did change.",
287         keyValidator2,
288         ValidatorCache.INSTANCE.getEntry(3));
289     assertEquals(
290         "Wrong mapping for name", name1, ValidatorCache.INSTANCE.getName(1));
291     assertEquals(
292         "Wrong mapping for name", name3, ValidatorCache.INSTANCE.getName(2));
293     assertEquals(
294         "Wrong mapping for name", name2, ValidatorCache.INSTANCE.getName(3));
295     // Notify cache that the objects should be removed from the cache
296     ValidatorCache.INSTANCE.updateWith(1, 1, null, null);
297     ValidatorCache.INSTANCE.updateWith(2, 1, null, keyValidator2);
298     ValidatorCache.INSTANCE.updateWith(3, 1, name3, null);
299     assertEquals(
300         "Expected empty map for after removing objects.",
301         0,
302         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
303     assertNull(
304         "Empty cache returned an entry anyway.",
305         ValidatorCache.INSTANCE.getEntry(1));
306     assertNull(
307         "Empty cache returned an entry anyway.",
308         ValidatorCache.INSTANCE.getEntry(2));
309     assertNull(
310         "Empty cache returned an entry anyway.",
311         ValidatorCache.INSTANCE.getEntry(3));
312     assertTrue(
313         "Cache indicated that non-existing object is valid.",
314         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
315     assertTrue(
316         "Cache indicated that non-existing object is valid.",
317         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
318     assertTrue(
319         "Cache indicated that non-existing object is valid.",
320         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
321     assertNull(
322         "Empty cache returned a name anyway",
323         ValidatorCache.INSTANCE.getName(1));
324     assertNull(
325         "Empty cache returned a name anyway",
326         ValidatorCache.INSTANCE.getName(2));
327     assertNull(
328         "Empty cache returned a name anyway",
329         ValidatorCache.INSTANCE.getName(3));
330     LOG.trace("<" + Thread.currentThread().getStackTrace()[1].getMethodName());
331   }

333   /**
334    * Verify that the objects we add to the cache is available even after the
335    * cache has expired, but the caller is asked to fetch the data from the
336    * original source instead.
337    *
338    * @throws InterruptedException Fail
339    */
340   @Test
341   public void cacheExpiration() throws InterruptedException {
342     LOG.trace(">" + Thread.currentThread().getStackTrace()[1].getMethodName());
343     ConfigurationHolder.updateConfiguration("validator.cachetime", "2000");
344     final String name1 = KeyValidatorCacheTest.class.getSimpleName() + " CA1";
345     final String name2 = KeyValidatorCacheTest.class.getSimpleName() + " CA2";
346     final String name3 = KeyValidatorCacheTest.class.getSimpleName() + " CA3";
347     final ValidatorBase keyValidator1 = getNewKeyValidator(name1);
348     final ValidatorBase keyValidator2 = getNewKeyValidator(name2);
349     final ValidatorBase keyValidator3 = getNewKeyValidator(name3);
350     // Start out with an empty cache and verify state
351     ValidatorCache.INSTANCE.flush();
352     assertEquals(
353         "Expected empty map after flush.",
354         0,
355         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
356     assertNull(
357         "Empty cache returned an entry anyway.",
358         ValidatorCache.INSTANCE.getEntry(1));
359     assertNull(
360         "Empty cache returned an entry anyway.",
361         ValidatorCache.INSTANCE.getEntry(2));
362     assertNull(
363         "Empty cache returned an entry anyway.",
364         ValidatorCache.INSTANCE.getEntry(3));
365     assertTrue(
366         "Cache indicated that non-existing object is valid.",
367         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
368     assertTrue(
369         "Cache indicated that non-existing object is valid.",
370         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
371     assertTrue(
372         "Cache indicated that non-existing object is valid.",
373         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
374     assertNull(
375         "Empty cache returned a name anyway",
376         ValidatorCache.INSTANCE.getName(1));
377     assertNull(
378         "Empty cache returned a name anyway",
379         ValidatorCache.INSTANCE.getName(2));
380     assertNull(
381         "Empty cache returned a name anyway",
382         ValidatorCache.INSTANCE.getName(3));
383     // Add entries to the (enabled) cache
384     ValidatorCache.INSTANCE.updateWith(1, 1, name1, keyValidator1);
385     ValidatorCache.INSTANCE.updateWith(2, 1, name2, keyValidator2);
386     ValidatorCache.INSTANCE.updateWith(3, 1, name3, keyValidator3);
387     assertEquals(
388         "Expected non-empty map for enabled cache.",
389         3,
390         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
391     assertEquals(
392         "Wrong mapping for name.",
393         1,
394         ValidatorCache.INSTANCE.getNameToIdMap().get(name1).intValue());
395     assertEquals(
396         "Wrong mapping for name.",
397         2,
398         ValidatorCache.INSTANCE.getNameToIdMap().get(name2).intValue());
399     assertEquals(
400         "Wrong mapping for name.",
401         3,
402         ValidatorCache.INSTANCE.getNameToIdMap().get(name3).intValue());
403     assertEquals(
404         "Expected cache to return same instance of object 1.",
405         keyValidator1,
406         ValidatorCache.INSTANCE.getEntry(1));
407     assertEquals(
408         "Expected cache to return same instance of object 2.",
409         keyValidator2,
410         ValidatorCache.INSTANCE.getEntry(2));
411     assertEquals(
412         "Expected cache to return same instance of object 3.",
413         keyValidator3,
414         ValidatorCache.INSTANCE.getEntry(3));
415     assertFalse(
416         "Cache indicated that existing object is invalid.",
417         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
418     assertFalse(
419         "Cache indicated that existing object is invalid.",
420         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
421     assertFalse(
422         "Cache indicated that existing object is invalid.",
423         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
424     assertEquals(
425         "Wrong mapping for name", name1, ValidatorCache.INSTANCE.getName(1));
426     assertEquals(
427         "Wrong mapping for name", name2, ValidatorCache.INSTANCE.getName(2));
428     assertEquals(
429         "Wrong mapping for name", name3, ValidatorCache.INSTANCE.getName(3));
430     // Wait until we are sure that the cache has expired
431     Thread.sleep(2500L);
432     assertEquals(
433         "Expected non-empty map for enabled cache.",
434         3,
435         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
436     assertEquals(
437         "Wrong mapping for name.",
438         1,
439         ValidatorCache.INSTANCE.getNameToIdMap().get(name1).intValue());
440     assertEquals(
441         "Wrong mapping for name.",
442         2,
443         ValidatorCache.INSTANCE.getNameToIdMap().get(name2).intValue());
444     assertEquals(
445         "Wrong mapping for name.",
446         3,
447         ValidatorCache.INSTANCE.getNameToIdMap().get(name3).intValue());
448     assertEquals(
449         "Expected cache to return same instance of object 1.",
450         keyValidator1,
451         ValidatorCache.INSTANCE.getEntry(1));
452     assertEquals(
453         "Expected cache to return same instance of object 2.",
454         keyValidator2,
455         ValidatorCache.INSTANCE.getEntry(2));
456     assertEquals(
457         "Expected cache to return same instance of object 3.",
458         keyValidator3,
459         ValidatorCache.INSTANCE.getEntry(3));
460     assertTrue(
461         "Cache indicated that stale object is valid.",
462         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
463     assertTrue(
464         "Cache indicated that stale object is valid.",
465         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
466     assertTrue(
467         "Cache indicated that stale object is valid.",
468         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
469     assertEquals(
470         "Wrong mapping for name", name1, ValidatorCache.INSTANCE.getName(1));
471     assertEquals(
472         "Wrong mapping for name", name2, ValidatorCache.INSTANCE.getName(2));
473     assertEquals(
474         "Wrong mapping for name", name3, ValidatorCache.INSTANCE.getName(3));
475     // Notify cache that the objects should be removed from the cache
476     ValidatorCache.INSTANCE.updateWith(1, 1, null, null);
477     ValidatorCache.INSTANCE.updateWith(2, 1, null, keyValidator2);
478     ValidatorCache.INSTANCE.updateWith(3, 1, name3, null);
479     assertEquals(
480         "Expected empty map for after removing objects.",
481         0,
482         ValidatorCache.INSTANCE.getNameToIdMap().entrySet().size());
483     assertNull(
484         "Empty cache returned an entry anyway.",
485         ValidatorCache.INSTANCE.getEntry(1));
486     assertNull(
487         "Empty cache returned an entry anyway.",
488         ValidatorCache.INSTANCE.getEntry(2));
489     assertNull(
490         "Empty cache returned an entry anyway.",
491         ValidatorCache.INSTANCE.getEntry(3));
492     assertTrue(
493         "Cache indicated that non-existing object is valid.",
494         ValidatorCache.INSTANCE.shouldCheckForUpdates(1));
495     assertTrue(
496         "Cache indicated that non-existing object is valid.",
497         ValidatorCache.INSTANCE.shouldCheckForUpdates(2));
498     assertTrue(
499         "Cache indicated that non-existing object is valid.",
500         ValidatorCache.INSTANCE.shouldCheckForUpdates(3));
501     assertNull(
502         "Empty cache returned a name anyway",
503         ValidatorCache.INSTANCE.getName(1));
504     assertNull(
505         "Empty cache returned a name anyway",
506         ValidatorCache.INSTANCE.getName(2));
507     assertNull(
508         "Empty cache returned a name anyway",
509         ValidatorCache.INSTANCE.getName(3));
510     LOG.trace("<" + Thread.currentThread().getStackTrace()[1].getMethodName());
511   }
