139     @Override
140     public List<String> isCryptoTokenSlotUsed(final AuthenticationToken authenticationToken, final String tokenName,
141             final String className, final Properties properties)
142             throws AuthorizationDeniedException, CryptoTokenNameInUseException, CryptoTokenOfflineException,
143             CryptoTokenAuthenticationFailedException, NoSuchSlotException {
144         if (log.isTraceEnabled()) {
145             log.trace(">isCryptoTokenUsed: " + tokenName + ", " + className);
146         }
147         if (CryptoTokenFactory.instance().getAvailableCryptoToken(className) == null) {
148             throw new CryptoTokenClassNotFoundException("Invalid token class name: " + className);
149         }
150         // Creating a duplicate P11 crypto token can be destructive, ideally we would check all crypto tokens, 
151         // but we only check the ones the admin has access to in order to not leak information 
152         List<CryptoTokenInfo> infos = getCryptoTokenInfos(authenticationToken);
153         List<String> providers = new ArrayList<>();
154         String tokenP11Lib = properties.getProperty(PKCS11CryptoToken.SHLIB_LABEL_KEY);
155         final String providerNameToCheck = createProviderName(tokenName, className, properties);
156         if (log.isDebugEnabled()) {
157             log.debug("isCryptoTokenUsed: Provider name to check for: "+providerNameToCheck);
158         }
159         // Return list of Crypto Token name
160         List<String> ret = new ArrayList<>();
161         // Only do this check for P11 tokens
162         if (StringUtils.isNotEmpty(tokenP11Lib)) {
163             for (CryptoTokenInfo cti : infos) {
164                 // We are concerned about PKCS#11 usage
165                 String ctiP11lib = cti.getP11Library();
166                 CryptoToken token = cryptoTokenSession.getCryptoToken(cti.getCryptoTokenId());
167                 final String ctiProviderName = token.getSignProviderName();
168                 providers.add(ctiProviderName);
169                 if (token != null) {
170                     if (isP11SlotSame(tokenP11Lib, providerNameToCheck, ctiP11lib, ctiProviderName, cti.getName())) {
171                         ret.add(ctiProviderName);
172                     }
173                 }
174             }
175             // Check for database protection crypto tokens as well, these are not stored as crypto tokens in the database, but only 
176             // as parameters in databaseprotection.properties, and thus requires special handling
177             Provider[] installedProviders = Security.getProviders();
178             if (installedProviders != null) {
179                 for (int i = 0; i < installedProviders.length; i++) {
180                     final String installedProviderName = installedProviders[i].getName();
181                     if (log.isDebugEnabled()) {
182                         log.debug("isCryptoTokenUsed: Checking installed provider: "+installedProviderName);
183                     }
184                     if (StringUtils.equals(providerNameToCheck, installedProviderName)) {
185                         // We found a match, but don't add duplicates
186                         if (!providers.contains(installedProviderName)) {
187                             log.debug("isCryptoTokenUsed: Found a match between "+providerNameToCheck+" and installed provider "+installedProviderName+", which was not already listed, must be a database protection token.");
188                             ret.add(installedProviderName+" (database protection?)");
189                         }
190                     }
191                 }                
192             }
193         }
194         if (log.isTraceEnabled()) {
195             log.trace("<isCryptoTokenUsed: " + tokenName + ", " + className + ", " + ret.size());
196         }
197         return ret; 
198     }

200     private boolean isP11SlotSame(String tokenP11Lib, String providerNameToCheck, String ctiP11lib, String ctiProviderName, String ctiName) throws NoSuchSlotException {
201         boolean ret = false;
202         if (StringUtils.isNotEmpty(ctiP11lib)) {
203             if (StringUtils.equalsIgnoreCase(tokenP11Lib, ctiP11lib)) {
204                 // We have a match on the library, watch out...check if we are using the same slot as well
205                 // Now it gets exciting, since you can address the slot through different things (slotID, slotName, p11Config)
206                 // it is really hard to check easily, we need to create the provider and see if the provider name is the same
207                 if (log.isDebugEnabled()) {
208                     log.debug("isCryptoTokenUsed: Provider for token we check for: "+providerNameToCheck);
209                     log.debug("isCryptoTokenUsed Provider for existing token: "+ctiProviderName);
210                 }
211                 if (StringUtils.equals(providerNameToCheck, ctiProviderName)) {
212                     // We had a match, the caller knows the crypto token name
213                     ret = true;
214                 }
215             }
216         }
217         return ret;
218     }

220     private String createProviderName(final String tokenName, final String className, final Properties tokenprops) throws NoSuchSlotException {
221         // Make a complete clone of the original properties, as we modify it to be non-addable-provider token properties below
222         // if we did that on the original tokenprops, this method would have a side effect in modifying caller parameters 
223         // (which causes things to break since the provider is not installed)
224         Properties properties = new Properties();
225         properties.putAll(tokenprops);
226         properties.setProperty(PKCS11CryptoToken.DO_NOT_ADD_P11_PROVIDER, "true");                       
227         final CryptoToken cryptoToken = CryptoTokenFactory.createCryptoToken(className, properties, null, -1, tokenName, false);
228         final String providerName = cryptoToken.getSignProviderName();
229         if (log.isDebugEnabled()) {
230             log.debug("isCryptoTokenUsed: Created provider (without installing) to check for collisions: "+providerName);
231         }
232         return providerName;
233     }

235     @Override
236     public void createCryptoToken(final AuthenticationToken authenticationToken, final String tokenName, final Integer cryptoTokenId,
237             final String className, final Properties properties, final byte[] data, final char[] authenticationCode)
238             throws AuthorizationDeniedException, CryptoTokenNameInUseException, CryptoTokenOfflineException,
239             CryptoTokenAuthenticationFailedException, NoSuchSlotException {
240         if (log.isTraceEnabled()) {
241             log.trace(">createCryptoToken: " + tokenName + ", " + className);
242         }
243         assertAuthorizedToModifyCryptoTokens(authenticationToken);
244         if (CryptoTokenFactory.instance().getAvailableCryptoToken(className) == null) {
245             throw new CryptoTokenClassNotFoundException("Invalid token class name: " + className);
246         }
247 
248         // Note: if data is null, a new empty keystore will be created
249         final CryptoToken cryptoToken = CryptoTokenFactory.createCryptoToken(className, properties, data, cryptoTokenId.intValue(), tokenName, false);
250         if (authenticationCode != null) {
251             if (log.isDebugEnabled()) {
252                 log.debug("Activating new crypto token using supplied authentication code.");
253             }
254             cryptoToken.activate(authenticationCode);
255         }
256 
257         // This property is used only once during crypto token creation
258         properties.remove(CryptoToken.ALLOW_NONEXISTING_SLOT_PROPERTY);
259 
260         final Map<String, Object> details = new LinkedHashMap<String, Object>();
261         details.put("msg", "Created CryptoToken with id " + cryptoTokenId);
262         details.put("name", cryptoToken.getTokenName());
263         details.put("encProviderName", cryptoToken.getEncProviderName());
264         details.put("signProviderName", cryptoToken.getSignProviderName());
265         putDelta(new Properties(), cryptoToken.getProperties(), details);
266         cryptoTokenSession.mergeCryptoToken(cryptoToken);
267         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_CREATE, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
268                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
269         if (log.isTraceEnabled()) {
270             log.trace("<createCryptoToken: " + tokenName + ", " + className);
271         }
272     }

274     @Override
275     public int createCryptoToken(final AuthenticationToken authenticationToken, final String tokenName, final String className,
276             final Properties properties, final byte[] data, final char[] authenticationCode) throws AuthorizationDeniedException,
277             CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException, CryptoTokenNameInUseException, NoSuchSlotException,
278             AuditRecordStorageException {
279         final List<Integer> allCryptoTokenIds = cryptoTokenSession.getCryptoTokenIds();
280         Integer cryptoTokenId = null;
281         for (int i = 0; i < 100; i++) {
282             final int current = Integer.valueOf(rnd.nextInt());
283             if (!allCryptoTokenIds.contains(current)) {
284                 cryptoTokenId = current;
285                 break;
286             }
287         }
288         if (cryptoTokenId == null) {
289             throw new IllegalStateException("Failed to allocate a new cryptoTokenId.");
290         }
291         createCryptoToken(authenticationToken, tokenName, cryptoTokenId, className, properties, data, authenticationCode);
292         return cryptoTokenId.intValue();
293     }

309     @Override
310     public void saveCryptoToken(AuthenticationToken authenticationToken, int cryptoTokenId, String tokenName, Properties properties,
311             char[] authenticationCode) throws AuthorizationDeniedException, CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException,
312             CryptoTokenNameInUseException, NoSuchSlotException {
313         if (log.isTraceEnabled()) {
314             log.trace(">saveCryptoToken: " + tokenName + ", " + cryptoTokenId);
315         }
316         // Note that an admin that is authorized to modify a token could gain access to another HSM slot etc..
317         if (!authorizationSession.isAuthorized(authenticationToken, CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource())) {
318             final String msg = INTRES.getLocalizedMessage("authorization.notauthorizedtoresource", CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource(),
319                     authenticationToken.toString());
320             throw new AuthorizationDeniedException(msg);
321         }
322         final CryptoToken currentCryptoToken = cryptoTokenSession.getCryptoToken(cryptoTokenId);
323         final String className = currentCryptoToken.getClass().getName();
324         final byte[] tokendata = currentCryptoToken.getTokenData();
325         // Handle presence of auto-activation indicators
326         boolean keepAutoActivateIfPresent = Boolean.valueOf(String.valueOf(properties.get(CryptoTokenManagementSession.KEEP_AUTO_ACTIVATION_PIN)));
327         properties.remove(CryptoTokenManagementSession.KEEP_AUTO_ACTIVATION_PIN);
328         final String newPin = BaseCryptoToken.getAutoActivatePin(properties);
329         if (newPin != null) {
330             BaseCryptoToken.setAutoActivatePin(properties, newPin, true);
331             authenticationCode = newPin.toCharArray();
332         } else if (keepAutoActivateIfPresent) {
333             final String currentPin = BaseCryptoToken.getAutoActivatePin(currentCryptoToken.getProperties());
334             if (currentPin != null) {
335                 BaseCryptoToken.setAutoActivatePin(properties, currentPin, true);
336                 authenticationCode = null; // We have an auto-activation pin and it didn't change;
337             }
338         } else if (authenticationCode == null || authenticationCode.length == 0) {
339             // Check if the token was auto-activated before. it is now manually activated, so use the auto-activation code one last time
340             final String currentPin = BaseCryptoToken.getAutoActivatePin(currentCryptoToken.getProperties());
341             if (currentPin != null) {
342                 authenticationCode = currentPin.toCharArray();
343             }
344         }
345         // TODO: If the current token is active we would like to dig out the code used to activate it and activate the new one as well..
346         // For SoftCryptoTokens, a new secret means that we should change it and it can only be done if the token is active
347         CryptoToken newCryptoToken;
348         try {
349             newCryptoToken = CryptoTokenFactory.createCryptoToken(className, properties, tokendata, cryptoTokenId, tokenName);
350             // If a new authenticationCode is provided we should verify it before we go ahead and merge
351             if (authenticationCode != null && authenticationCode.length > 0) {
352                 newCryptoToken.deactivate();
353                 newCryptoToken.activate(authenticationCode);
354             }
355         } catch (CryptoTokenOfflineException e) {
356             // If the crypto token can not be initialized, we have a problem and can not even disable auto-activation.
357             // Go ahead and ignore this
358             log.info("CryptoTokenOfflineException getting new crypto token for saving, ignoring this error and saving anyway: ", e);
359             newCryptoToken = currentCryptoToken;
360             newCryptoToken.setProperties(properties);
361             newCryptoToken.setTokenName(tokenName);
362         }
363         final Map<String, Object> details = new LinkedHashMap<String, Object>();
364         details.put("msg", "Modified CryptoToken with id " + cryptoTokenId);
365         putDelta("name", currentCryptoToken.getTokenName(), newCryptoToken.getTokenName(), details);
366         putDelta("encProviderName", currentCryptoToken.getEncProviderName(), newCryptoToken.getEncProviderName(), details);
367         putDelta("signProviderName", currentCryptoToken.getSignProviderName(), newCryptoToken.getSignProviderName(), details);
368         putDelta(currentCryptoToken.getProperties(), newCryptoToken.getProperties(), details);
369         cryptoTokenSession.mergeCryptoToken(newCryptoToken);
370         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_EDIT, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
371                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
372         if (log.isTraceEnabled()) {
373             log.trace("<saveCryptoToken: " + tokenName + ", " + cryptoTokenId);
374         }
375     }

377     @Override
378     public void saveCryptoToken(final AuthenticationToken authenticationToken, final int cryptoTokenId,
379             final String newName, final String newPlaceholders) throws AuthorizationDeniedException, CryptoTokenNameInUseException {
380         if (log.isTraceEnabled()) {
381             log.trace(">saveCryptoToken: cryptoTokenId=" + cryptoTokenId + ", newName=" + newName);
382         }
383         if (!authorizationSession.isAuthorized(authenticationToken, CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource())) {
384             final String msg = INTRES.getLocalizedMessage("authorization.notauthorizedtoresource", CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource(),
385                     authenticationToken.toString());
386             throw new AuthorizationDeniedException(msg);
387         }
388         final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(cryptoTokenId);
389         final String oldName = cryptoToken.getTokenName();
390         cryptoToken.setTokenName(newName);
391         final Properties properties = cryptoToken.getProperties();
392         final String oldPlaceholders = cryptoToken.getProperties().getProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY);
393         properties.setProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY, newPlaceholders);
394         cryptoToken.setProperties(properties);
395 
396         final Map<String, Object> details = new LinkedHashMap<String, Object>();
397         details.put("msg", "Modified name/placeholders of CryptoToken with id " + cryptoTokenId);
398         putDelta("name", oldName, newName, details);
399         putDelta("keyPlaceholders", oldPlaceholders, newPlaceholders, details);
400 
401         cryptoTokenSession.mergeCryptoToken(cryptoToken);
402         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_EDIT, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
403                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
404         if (log.isTraceEnabled()) {
405             log.trace("<saveCryptoToken: cryptoTokenId=" + cryptoTokenId + ", newName=" + newName);
406         }
407     }

474     @TransactionAttribute(TransactionAttributeType.REQUIRED)
475     @Override
476     public boolean updatePin(AuthenticationToken authenticationToken, Integer cryptoTokenId, char[] currentAuthenticationCode,
477             char[] newAuthenticationCode, boolean updateOnly) throws AuthorizationDeniedException, CryptoTokenAuthenticationFailedException,
478             CryptoTokenOfflineException {
479         final String[] requiredAuthorization = new String[] { CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource() + "/" + cryptoTokenId,
480                 CryptoTokenRules.ACTIVATE.resource() + "/" + cryptoTokenId, CryptoTokenRules.DEACTIVATE.resource() + "/" + cryptoTokenId };
481         if (!authorizationSession.isAuthorized(authenticationToken, requiredAuthorization)) {
482             final String msg = INTRES.getLocalizedMessage("authorization.notauthorizedtoresource", Arrays.toString(requiredAuthorization),
483                     authenticationToken.toString());
484             throw new AuthorizationDeniedException(msg);
485         }
486         CryptoToken cryptoToken = getCryptoToken(cryptoTokenId);
487         final Properties cryptoTokenProperties = cryptoToken.getProperties();
488         // Get current auto-activation pin (if any)
489         final String oldAutoActivationPin = BaseCryptoToken.getAutoActivatePin(cryptoTokenProperties);
490         if (oldAutoActivationPin == null && (updateOnly || newAuthenticationCode == null)) {
491             // This is a NOOP call that will not lead to any change
492             return false;
493         }
494         if (SoftCryptoToken.class.getName().equals(cryptoToken.getClass().getName())) {
495             CryptoProviderTools.installBCProviderIfNotAvailable();
496             final KeyStore keystore;
497             try {
498                 keystore = KeyStore.getInstance("PKCS12", "BC");
499                 keystore.load(new ByteArrayInputStream(cryptoToken.getTokenData()), currentAuthenticationCode);
500             } catch (Exception e) {
501                 final String msg = "Failed to use supplied current PIN." + " " + e;
502                 log.info(msg);
503                 throw new CryptoTokenAuthenticationFailedException(msg);
504             }
505             if (newAuthenticationCode == null) {
506                 // When no new pin is supplied, we will not modify the key-store and just remove the current auto-activation pin
507                 cryptoTokenProperties.remove(CryptoToken.AUTOACTIVATE_PIN_PROPERTY);
508                 // We'll also remove the default password option
509                 cryptoTokenProperties.put(SoftCryptoToken.NODEFAULTPWD, Boolean.TRUE.toString());
510                 cryptoToken.setProperties(cryptoTokenProperties);
511             } else {
512                 try {
513                     final ByteArrayOutputStream baos = new ByteArrayOutputStream();
514                     keystore.store(baos, newAuthenticationCode);
515                     baos.close();
516                     if (oldAutoActivationPin != null || !updateOnly) {
517                         BaseCryptoToken.setAutoActivatePin(cryptoTokenProperties, new String(newAuthenticationCode), true);
518                     } else {
519                         log.debug("Auto-activation will not be used. Only changing pin for soft CryptoToken keystore.");
520                     }
521                     cryptoToken = CryptoTokenFactory.createCryptoToken(SoftCryptoToken.class.getName(), cryptoTokenProperties, baos.toByteArray(),
522                             cryptoTokenId, cryptoToken.getTokenName());
523                 } catch (Exception e) {
524                     log.info("Unable to store soft keystore with new PIN: " + e);
525                     throw new CryptoTokenAuthenticationFailedException("Unable to store soft keystore with new PIN");
526                 }
527             }
528         } else {
529             if (oldAutoActivationPin != null) {
530                 // If we have an old auto-activation pin we will compare the "current" with this value to avoid deactivating the token
531                 if (!oldAutoActivationPin.equals(new String(currentAuthenticationCode))) {
532                     final String msg = "Supplied PIN did not match auto-activation PIN.";
533                     log.info(msg);
534                     throw new CryptoTokenAuthenticationFailedException(msg);
535                 } else {
536                     log.debug("Successfully verified the PIN for non-soft CryptoToken by comparing supplied PIN to auto-activation PIN.");
537                 }
538             } else {
539                 // If we don't have an auto-activation pin to compare the supplied PIN to, we need to verify the supplied
540                 // PIN can be used in a de-activation/activation cycle.
541                 final boolean wasInactive = !isCryptoTokenStatusActive(authenticationToken, cryptoTokenId);
542                 cryptoToken.deactivate();
543                 cryptoToken.activate(currentAuthenticationCode);
544                 if (wasInactive) {
545                     // Note that there is a small glitch here where the token was active, but we have no other options to verify the pin
546                     cryptoToken.deactivate();
547                 }
548             }
549             if (newAuthenticationCode == null) {
550                 cryptoTokenProperties.remove(CryptoToken.AUTOACTIVATE_PIN_PROPERTY);
551             } else {
552                 BaseCryptoToken.setAutoActivatePin(cryptoTokenProperties, new String(newAuthenticationCode), true);
553             }
554             cryptoToken.setProperties(cryptoTokenProperties);
555         }
556         // Save the modified CryptoToken
557         try {
558             cryptoTokenSession.mergeCryptoToken(cryptoToken);
559         } catch (CryptoTokenNameInUseException e) {
560             // This should not happen here since we use the same name and id
561             throw new RuntimeException(e);
562         }
563         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_UPDATEPIN, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
564                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null,
565                 "Updated PIN of CryptoToken '" + cryptoToken.getTokenName() + "' with id " + cryptoTokenId);
566         // Return the current auto-activation state
567         return BaseCryptoToken.getAutoActivatePin(cryptoTokenProperties) != null;
568     }

609     public Integer getIdFromName(final String cryptoTokenName) {
610         if (cryptoTokenName == null) {
611             return null;
612         }
613         final Map<String, Integer> cachedNameToIdMap = cryptoTokenSession.getCachedNameToIdMap();
614         Integer cryptoTokenId = cachedNameToIdMap.get(cryptoTokenName);
615         if (cryptoTokenId == null) {
616             // Ok.. so it's not in the cache.. look for it the hard way..
617             for (final Integer currentCryptoTokenId : cryptoTokenSession.getCryptoTokenIds()) {
618                 // Don't lookup CryptoTokens we already have in the id to name cache
619                 if (!cachedNameToIdMap.values().contains(currentCryptoTokenId)) {
620                     final CryptoToken currentCryptoToken = cryptoTokenSession.getCryptoToken(currentCryptoTokenId.intValue());
621                     final String currentCryptoTokenName = currentCryptoToken == null ? null : currentCryptoToken.getTokenName();
622                     if (cryptoTokenName.equals(currentCryptoTokenName)) {
623                         cryptoTokenId = currentCryptoTokenId;
624                         break;
625                     }
626                 }
627             }
628         }
629         return cryptoTokenId;
630     }

640     private List<String> getKeyPairAliasesInternal(final CryptoToken cryptoToken) throws CryptoTokenOfflineException {
641         try {
642             final List<String> aliasEnumeration = cryptoToken.getAliases();
643             final List<String> keyPairAliases = new ArrayList<String>();
644             for(final String currentAlias : aliasEnumeration) {
645                 try {
646                     if (cryptoToken.getPublicKey(currentAlias) != null && cryptoToken.getPrivateKey(currentAlias) != null) {
647                         // A key pair exists for this alias, so add it
648                         keyPairAliases.add(currentAlias);
649                     }
650                 } catch (CryptoTokenOfflineException ignored) {
651                     if (log.isDebugEnabled()) {
652                         log.debug("Ignord key alias '"+currentAlias+"' in crypto token '"+cryptoToken.getTokenName()+"' since it is missing a public and/or private key. Perhaps it is a symmetric key?");
653                     }
654                 }
655             }
656             return keyPairAliases;
657         } catch (KeyStoreException e) {
658             throw new CryptoTokenOfflineException(e);
659         }

664     public void createKeyPair(final AuthenticationToken authenticationToken, final int cryptoTokenId, final String alias,
665             final String keySpecificationParam) throws AuthorizationDeniedException, CryptoTokenOfflineException, InvalidKeyException,
666             InvalidAlgorithmParameterException {
667         assertAuthorization(authenticationToken, cryptoTokenId, CryptoTokenRules.GENERATE_KEYS.resource() + "/" + cryptoTokenId);
668         final CryptoToken cryptoToken = getCryptoTokenAndAssertExistence(cryptoTokenId);
669         // Check if alias is already in use
670         assertAliasNotInUse(cryptoToken, alias);
671 
672         // Support "RSAnnnn" and convert it to the legacy format "nnnn"
673         final String keySpecification;
674         if (keySpecificationParam.startsWith(AlgorithmConstants.KEYALGORITHM_RSA)) {
675             keySpecification = keySpecificationParam.substring(AlgorithmConstants.KEYALGORITHM_RSA.length());
676         } else {
677             keySpecification = keySpecificationParam;
678         }
679         // Check if keySpec is valid
680         KeyTools.checkValidKeyLength(keySpecification);
681         // Audit log before generation. If the token is an HSM the merge will not make a difference.
682         final Map<String, Object> details = new LinkedHashMap<String, Object>();
683         details.put("msg", "Generated new keypair in CryptoToken " + cryptoTokenId);
684         details.put("keyAlias", alias);
685         details.put("keySpecification", keySpecification);
686         cryptoToken.generateKeyPair(keySpecification, alias);
687         cryptoToken.testKeyPair(alias);
688         // Merge is important for soft tokens where the data is persisted in the database, but will also update lastUpdate
689         try {
690             cryptoTokenSession.mergeCryptoToken(cryptoToken);
691         } catch (CryptoTokenNameInUseException e) {
692             throw new RuntimeException(e); // We have not changed the name of the CrytpoToken here, so this should never happen
693         }
694         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_GEN_KEYPAIR, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
695                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
696     }

754             throws AuthorizationDeniedException, CryptoTokenOfflineException, InvalidKeyException {
755         assertAuthorization(authenticationToken, cryptoTokenId, CryptoTokenRules.REMOVE_KEYS.resource() + "/" + cryptoTokenId);
756         final CryptoToken cryptoToken = getCryptoTokenAndAssertExistence(cryptoTokenId);
757         // Check if alias is in use
758         if (!cryptoToken.isAliasUsed(alias)) {
759             throw new InvalidKeyException("Alias " + alias + " is not in use");
760         }
761         final Map<String, Object> details = new LinkedHashMap<String, Object>();
762         details.put("msg", "Deleted key pair from CryptoToken " + cryptoTokenId);
763         details.put("keyAlias", alias);
764         try {
765             cryptoToken.deleteEntry(alias);
766         } catch (KeyStoreException e) {
767             throw new InvalidKeyException(e);
768         } catch (NoSuchAlgorithmException e) {
769             throw new InvalidKeyException(e);
770         } catch (CertificateException e) {
771             throw new InvalidKeyException(e);
772         } catch (IOException e) {
773             throw new InvalidKeyException(e);
774         }
775         assertAliasNotInUse(cryptoToken, alias);
776         log.debug("cryptoTokenSession.mergeCryptoToken");
777         // Merge is important for soft tokens where the data is persisted in the database, but will also update lastUpdate
778         try {
779             cryptoTokenSession.mergeCryptoToken(cryptoToken);
780         } catch (CryptoTokenNameInUseException e) {
781             throw new IllegalStateException(e); // We have not changed the name of the CrytpoToken here, so this should never happen
782         }
783         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_DELETE_ENTRY, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
784                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
785     }

787     @TransactionAttribute(TransactionAttributeType.REQUIRED)
788     @Override
789     public void removeKeyPairPlaceholder(final AuthenticationToken authenticationToken, final int cryptoTokenId, final String alias)
790             throws AuthorizationDeniedException, InvalidKeyException {
791         assertAuthorization(authenticationToken, cryptoTokenId, CryptoTokenRules.REMOVE_KEYS.resource() + "/" + cryptoTokenId);
792         final CryptoToken cryptoToken = getCryptoTokenAndAssertExistence(cryptoTokenId);
793 
794         boolean removed = false;
795         final Properties props = new Properties();
796         props.putAll(cryptoToken.getProperties());
797         final String placeholdersString = props.getProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY, "");
798         final List<String> entries = new ArrayList<String>(Arrays.asList(placeholdersString.split("["+CryptoToken.KEYPLACEHOLDERS_OUTER_SEPARATOR+"]")));
799         final Iterator<String> iter = entries.iterator();
800         while (iter.hasNext()) {
801             final String entry = iter.next();
802             if (entry.startsWith(alias + CryptoToken.KEYPLACEHOLDERS_INNER_SEPARATOR)) {
803                 iter.remove();
804                 removed = true;
805             }
806         }
807 
808         if (removed) {
809             final String newValue = StringUtils.join(entries, CryptoToken.KEYPLACEHOLDERS_OUTER_SEPARATOR);
810             props.setProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY, newValue);
811             cryptoToken.setProperties(props);
812         }
813 
814         // Check if alias is in use
815         if (!removed) {
816             throw new InvalidKeyException("Alias " + alias + " is not in use");
817         }
818 
819         try {
820             cryptoTokenSession.mergeCryptoToken(cryptoToken);
821         } catch (CryptoTokenNameInUseException e) {
822             throw new IllegalStateException(e); // We have not changed the name of the CrytpoToken here, so this should never happen
823         }
824 
825         final Map<String, Object> details = new LinkedHashMap<String, Object>();
826         details.put("msg", "Deleted key pair placeholder from CryptoToken " + cryptoTokenId);
827         details.put("keyAlias", alias);
828         securityEventsLoggerSession.log(EventTypes.CRYPTOTOKEN_DELETE_ENTRY, EventStatus.SUCCESS, ModuleTypes.CRYPTOTOKEN, ServiceTypes.CORE,
829                 authenticationToken.toString(), String.valueOf(cryptoTokenId), null, null, details);
830     }

866     /** Helper method for audit logging changes 
867      * @param oldProperties Old Props
868      * @param newProperties New Props
869      * @param details Details */
870     private void putDelta(Properties oldProperties, Properties newProperties, Map<String, Object> details) {
871         // Find out what has happended to all the old properties
872         for (final Object key : oldProperties.keySet()) {
873             final String oldValue = oldProperties.getProperty(String.valueOf(key));
874             final String newValue = newProperties.getProperty(String.valueOf(key));
875             putDelta(String.valueOf(key), oldValue, newValue, details);
876         }
877         // Find out which new properties that did not exist in the old
878         for (final Object key : newProperties.keySet()) {
879             final String oldValue = oldProperties.getProperty(String.valueOf(key));
880             if (oldValue == null) {
881                 final String newValue = newProperties.getProperty(String.valueOf(key));
882                 putDelta(String.valueOf(key), oldValue, newValue, details);
883             }
884         }
885     }

887     /** Helper method for audit logging changes 
888      * @param key Key
889      * @param oldValue Old value 
890      * @param newValue New value 
891      * @param details Details */
892     private void putDelta(String key, String oldValue, String newValue, Map<String, Object> details) {
893         // Treat the auto-activation pin with care
894         if (BaseCryptoToken.AUTOACTIVATE_PIN_PROPERTY.equals(key)) {
895             if (oldValue == null && newValue == null) {
896                 // NOP
897             } else if (oldValue == null && newValue != null) {
898                 details.put("autoActivation", "added");
899                 details.put("autoActivationPinProtection", StringTools.getEncryptVersionFromString(newValue));
900             } else if (oldValue != null && newValue == null) {
901                 details.put("autoActivation", "removed");
902             } else if (!oldValue.equals(newValue)) {
903                 details.put("autoActivation", "pin changed");
904                 details.put("autoActivationPinProtection", StringTools.getEncryptVersionFromString(newValue));
905             }
906         } else {
907             if (oldValue == null && newValue == null) {
908                 // NOP
909             } else if (oldValue == null && newValue != null) {
910                 details.put("added:" + key, newValue);
911             } else if (oldValue != null && newValue == null) {
912                 details.put("removed:" + key, oldValue);
913             } else if (!oldValue.equals(newValue)) {
914                 details.put("changed:" + key, newValue);
915             } else {
916                 details.put(key, newValue);
917             }
918         }
919     }
