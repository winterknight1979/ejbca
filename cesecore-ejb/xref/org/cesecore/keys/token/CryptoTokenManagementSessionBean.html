167   @Override
168   public List<String> isCryptoTokenSlotUsed(
169       final AuthenticationToken authenticationToken,
170       final String tokenName,
171       final String className,
172       final Properties properties)
173       throws AuthorizationDeniedException, CryptoTokenNameInUseException,
174           CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException,
175           NoSuchSlotException {
176     if (LOG.isTraceEnabled()) {
177       LOG.trace(">isCryptoTokenUsed: " + tokenName + ", " + className);
178     }
179     if (CryptoTokenFactory.instance().getAvailableCryptoToken(className)
180         == null) {
181       throw new CryptoTokenClassNotFoundException(
182           "Invalid token class name: " + className);
183     }
184     // Creating a duplicate P11 crypto token can be destructive, ideally we
185     // would check all crypto tokens,
186     // but we only check the ones the admin has access to in order to not leak
187     // information
188     List<CryptoTokenInfo> infos = getCryptoTokenInfos(authenticationToken);
189     List<String> providers = new ArrayList<>();
190     String tokenP11Lib =
191         properties.getProperty(PKCS11CryptoToken.SHLIB_LABEL_KEY);
192     final String providerNameToCheck =
193         createProviderName(tokenName, className, properties);
194     if (LOG.isDebugEnabled()) {
195       LOG.debug(
196           "isCryptoTokenUsed: Provider name to check for: "
197               + providerNameToCheck);
198     }
199     // Return list of Crypto Token name
200     List<String> ret = new ArrayList<>();
201     // Only do this check for P11 tokens
202     if (StringUtils.isNotEmpty(tokenP11Lib)) {
203       for (CryptoTokenInfo cti : infos) {
204         // We are concerned about PKCS#11 usage
205         String ctiP11lib = cti.getP11Library();
206         CryptoToken token =
207             cryptoTokenSession.getCryptoToken(cti.getCryptoTokenId());
208         final String ctiProviderName = token.getSignProviderName();
209         providers.add(ctiProviderName);
210         if (token != null) {
211           if (isP11SlotSame(
212               tokenP11Lib,
213               providerNameToCheck,
214               ctiP11lib,
215               ctiProviderName,
216               cti.getName())) {
217             ret.add(ctiProviderName);
218           }
219         }
220       }
221       // Check for database protection crypto tokens as well, these are not
222       // stored as crypto tokens in the database, but only
223       // as parameters in databaseprotection.properties, and thus requires
224       // special handling
225       Provider[] installedProviders = Security.getProviders();
226       if (installedProviders != null) {
227         for (int i = 0; i < installedProviders.length; i++) {
228           final String installedProviderName = installedProviders[i].getName();
229           if (LOG.isDebugEnabled()) {
230             LOG.debug(
231                 "isCryptoTokenUsed: Checking installed provider: "
232                     + installedProviderName);
233           }
234           if (StringUtils.equals(providerNameToCheck, installedProviderName)) {
235             // We found a match, but don't add duplicates
236             if (!providers.contains(installedProviderName)) {
237               LOG.debug(
238                   "isCryptoTokenUsed: Found a match between "
239                       + providerNameToCheck
240                       + " and installed provider "
241                       + installedProviderName
242                       + ", which was not already listed, must be a database"
243                       + " protection token.");
244               ret.add(installedProviderName + " (database protection?)");
245             }
246           }
247         }
248       }
249     }
250     if (LOG.isTraceEnabled()) {
251       LOG.trace(
252           "<isCryptoTokenUsed: "
253               + tokenName
254               + ", "
255               + className
256               + ", "
257               + ret.size());
258     }
259     return ret;
260   }

262   private boolean isP11SlotSame(
263       final String tokenP11Lib,
264       final String providerNameToCheck,
265       final String ctiP11lib,
266       final String ctiProviderName,
267       final String ctiName)
268       throws NoSuchSlotException {
269     boolean ret = false;
270     if (StringUtils.isNotEmpty(ctiP11lib)) {
271       if (StringUtils.equalsIgnoreCase(tokenP11Lib, ctiP11lib)) {
272         // We have a match on the library, watch out...check if we are using the
273         // same slot as well
274         // Now it gets exciting, since you can address the slot through
275         // different things (slotID, slotName, p11Config)
276         // it is really hard to check easily, we need to create the provider and
277         // see if the provider name is the same
278         if (LOG.isDebugEnabled()) {
279           LOG.debug(
280               "isCryptoTokenUsed: Provider for token we check for: "
281                   + providerNameToCheck);
282           LOG.debug(
283               "isCryptoTokenUsed Provider for existing token: "
284                   + ctiProviderName);
285         }
286         if (StringUtils.equals(providerNameToCheck, ctiProviderName)) {
287           // We had a match, the caller knows the crypto token name
288           ret = true;
289         }
290       }
291     }
292     return ret;
293   }

295   private String createProviderName(
296       final String tokenName,
297       final String className,
298       final Properties tokenprops)
299       throws NoSuchSlotException {
300     // Make a complete clone of the original properties, as we modify it to be
301     // non-addable-provider token properties below
302     // if we did that on the original tokenprops, this method would have a side
303     // effect in modifying caller parameters
304     // (which causes things to break since the provider is not installed)
305     Properties properties = new Properties();
306     properties.putAll(tokenprops);
307     properties.setProperty(PKCS11CryptoToken.DO_NOT_ADD_P11_PROVIDER, "true");
308     final CryptoToken cryptoToken =
309         CryptoTokenFactory.createCryptoToken(
310             className, properties, null, -1, tokenName, false);
311     final String providerName = cryptoToken.getSignProviderName();
312     if (LOG.isDebugEnabled()) {
313       LOG.debug(
314           "isCryptoTokenUsed: Created provider (without installing) to check"
315               + " for collisions: "
316               + providerName);
317     }
318     return providerName;
319   }

321   @Override
322   public void createCryptoToken(
323       final AuthenticationToken authenticationToken,
324       final String tokenName,
325       final Integer cryptoTokenId,
326       final String className,
327       final Properties properties,
328       final byte[] data,
329       final char[] authenticationCode)
330       throws AuthorizationDeniedException, CryptoTokenNameInUseException,
331           CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException,
332           NoSuchSlotException {
333     if (LOG.isTraceEnabled()) {
334       LOG.trace(">createCryptoToken: " + tokenName + ", " + className);
335     }
336     assertAuthorizedToModifyCryptoTokens(authenticationToken);
337     if (CryptoTokenFactory.instance().getAvailableCryptoToken(className)
338         == null) {
339       throw new CryptoTokenClassNotFoundException(
340           "Invalid token class name: " + className);
341     }
342 
343     // Note: if data is null, a new empty keystore will be created
344     final CryptoToken cryptoToken =
345         CryptoTokenFactory.createCryptoToken(
346             className,
347             properties,
348             data,
349             cryptoTokenId.intValue(),
350             tokenName,
351             false);
352     if (authenticationCode != null) {
353       if (LOG.isDebugEnabled()) {
354         LOG.debug(
355             "Activating new crypto token using supplied authentication code.");
356       }
357       cryptoToken.activate(authenticationCode);
358     }
359 
360     // This property is used only once during crypto token creation
361     properties.remove(CryptoToken.ALLOW_NONEXISTING_SLOT_PROPERTY);
362 
363     final Map<String, Object> details = new LinkedHashMap<String, Object>();
364     details.put("msg", "Created CryptoToken with id " + cryptoTokenId);
365     details.put("name", cryptoToken.getTokenName());
366     details.put("encProviderName", cryptoToken.getEncProviderName());
367     details.put("signProviderName", cryptoToken.getSignProviderName());
368     putDelta(new Properties(), cryptoToken.getProperties(), details);
369     cryptoTokenSession.mergeCryptoToken(cryptoToken);
370     securityEventsLoggerSession.log(
371         EventTypes.CRYPTOTOKEN_CREATE,
372         EventStatus.SUCCESS,
373         ModuleTypes.CRYPTOTOKEN,
374         ServiceTypes.CORE,
375         authenticationToken.toString(),
376         String.valueOf(cryptoTokenId),
377         null,
378         null,
379         details);
380     if (LOG.isTraceEnabled()) {
381       LOG.trace("<createCryptoToken: " + tokenName + ", " + className);
382     }
383   }

385   @Override
386   public int createCryptoToken(
387       final AuthenticationToken authenticationToken,
388       final String tokenName,
389       final String className,
390       final Properties properties,
391       final byte[] data,
392       final char[] authenticationCode)
393       throws AuthorizationDeniedException, CryptoTokenOfflineException,
394           CryptoTokenAuthenticationFailedException,
395           CryptoTokenNameInUseException, NoSuchSlotException,
396           AuditRecordStorageException {
397     final List<Integer> allCryptoTokenIds =
398         cryptoTokenSession.getCryptoTokenIds();
399     Integer cryptoTokenId = null;
400     final int max = 100;
401     for (int i = 0; i < max; i++) {
402       final int current = Integer.valueOf(RND.nextInt());
403       if (!allCryptoTokenIds.contains(current)) {
404         cryptoTokenId = current;
405         break;
406       }
407     }
408     if (cryptoTokenId == null) {
409       throw new IllegalStateException(
410           "Failed to allocate a new cryptoTokenId.");
411     }
412     createCryptoToken(
413         authenticationToken,
414         tokenName,
415         cryptoTokenId,
416         className,
417         properties,
418         data,
419         authenticationCode);
420     return cryptoTokenId.intValue();
421   }

443   @Override
444   public void saveCryptoToken(
445       final AuthenticationToken authenticationToken,
446       final int cryptoTokenId,
447       final String tokenName,
448       final Properties properties,
449       final char[] oauthenticationCode)
450       throws AuthorizationDeniedException, CryptoTokenOfflineException,
451           CryptoTokenAuthenticationFailedException,
452           CryptoTokenNameInUseException, NoSuchSlotException {
453       char[] authenticationCode = oauthenticationCode;
454     if (LOG.isTraceEnabled()) {
455       LOG.trace(">saveCryptoToken: " + tokenName + ", " + cryptoTokenId);
456     }
457     // Note that an admin that is authorized to modify a token could gain access
458     // to another HSM slot etc..
459     if (!authorizationSession.isAuthorized(
460         authenticationToken, CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource())) {
461       final String msg =
462           INTRES.getLocalizedMessage(
463               "authorization.notauthorizedtoresource",
464               CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource(),
465               authenticationToken.toString());
466       throw new AuthorizationDeniedException(msg);
467     }
468     final CryptoToken currentCryptoToken =
469         cryptoTokenSession.getCryptoToken(cryptoTokenId);
470     final String className = currentCryptoToken.getClass().getName();
471     final byte[] tokendata = currentCryptoToken.getTokenData();
472     // Handle presence of auto-activation indicators
473     boolean keepAutoActivateIfPresent =
474         Boolean.valueOf(
475             String.valueOf(
476                 properties.get(
477                     CryptoTokenManagementSession.KEEP_AUTO_ACTIVATION_PIN)));
478     properties.remove(CryptoTokenManagementSession.KEEP_AUTO_ACTIVATION_PIN);
479     final String newPin = BaseCryptoToken.getAutoActivatePin(properties);
480     if (newPin != null) {
481       BaseCryptoToken.setAutoActivatePin(properties, newPin, true);
482       authenticationCode = newPin.toCharArray();
483     } else if (keepAutoActivateIfPresent) {
484       final String currentPin =
485           BaseCryptoToken.getAutoActivatePin(
486               currentCryptoToken.getProperties());
487       if (currentPin != null) {
488         BaseCryptoToken.setAutoActivatePin(properties, currentPin, true);
489         authenticationCode =
490             null; // We have an auto-activation pin and it didn't change;
491       }
492     } else if (authenticationCode == null || authenticationCode.length == 0) {
493       // Check if the token was auto-activated before. it is now manually
494       // activated, so use the auto-activation code one last time
495       final String currentPin =
496           BaseCryptoToken.getAutoActivatePin(
497               currentCryptoToken.getProperties());
498       if (currentPin != null) {
499         authenticationCode = currentPin.toCharArray();
500       }
501     }
502     // TODO: If the current token is active we would like to dig out the code
503     // used to activate it and activate the new one as well..
504     // For SoftCryptoTokens, a new secret means that we should change it and it
505     // can only be done if the token is active
506     CryptoToken newCryptoToken;
507     try {
508       newCryptoToken =
509           CryptoTokenFactory.createCryptoToken(
510               className, properties, tokendata, cryptoTokenId, tokenName);
511       // If a new authenticationCode is provided we should verify it before we
512       // go ahead and merge
513       if (authenticationCode != null && authenticationCode.length > 0) {
514         newCryptoToken.deactivate();
515         newCryptoToken.activate(authenticationCode);
516       }
517     } catch (CryptoTokenOfflineException e) {
518       // If the crypto token can not be initialized, we have a problem and can
519       // not even disable auto-activation.
520       // Go ahead and ignore this
521       LOG.info(
522           "CryptoTokenOfflineException getting new crypto token for saving,"
523               + " ignoring this error and saving anyway: ",
524           e);
525       newCryptoToken = currentCryptoToken;
526       newCryptoToken.setProperties(properties);
527       newCryptoToken.setTokenName(tokenName);
528     }
529     final Map<String, Object> details = new LinkedHashMap<String, Object>();
530     details.put("msg", "Modified CryptoToken with id " + cryptoTokenId);
531     putDelta(
532         "name",
533         currentCryptoToken.getTokenName(),
534         newCryptoToken.getTokenName(),
535         details);
536     putDelta(
537         "encProviderName",
538         currentCryptoToken.getEncProviderName(),
539         newCryptoToken.getEncProviderName(),
540         details);
541     putDelta(
542         "signProviderName",
543         currentCryptoToken.getSignProviderName(),
544         newCryptoToken.getSignProviderName(),
545         details);
546     putDelta(
547         currentCryptoToken.getProperties(),
548         newCryptoToken.getProperties(),
549         details);
550     cryptoTokenSession.mergeCryptoToken(newCryptoToken);
551     securityEventsLoggerSession.log(
552         EventTypes.CRYPTOTOKEN_EDIT,
553         EventStatus.SUCCESS,
554         ModuleTypes.CRYPTOTOKEN,
555         ServiceTypes.CORE,
556         authenticationToken.toString(),
557         String.valueOf(cryptoTokenId),
558         null,
559         null,
560         details);
561     if (LOG.isTraceEnabled()) {
562       LOG.trace("<saveCryptoToken: " + tokenName + ", " + cryptoTokenId);
563     }
564   }

566   @Override
567   public void saveCryptoToken(
568       final AuthenticationToken authenticationToken,
569       final int cryptoTokenId,
570       final String newName,
571       final String newPlaceholders)
572       throws AuthorizationDeniedException, CryptoTokenNameInUseException {
573     if (LOG.isTraceEnabled()) {
574       LOG.trace(
575           ">saveCryptoToken: cryptoTokenId="
576               + cryptoTokenId
577               + ", newName="
578               + newName);
579     }
580     if (!authorizationSession.isAuthorized(
581         authenticationToken, CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource())) {
582       final String msg =
583           INTRES.getLocalizedMessage(
584               "authorization.notauthorizedtoresource",
585               CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource(),
586               authenticationToken.toString());
587       throw new AuthorizationDeniedException(msg);
588     }
589     final CryptoToken cryptoToken =
590         cryptoTokenSession.getCryptoToken(cryptoTokenId);
591     final String oldName = cryptoToken.getTokenName();
592     cryptoToken.setTokenName(newName);
593     final Properties properties = cryptoToken.getProperties();
594     final String oldPlaceholders =
595         cryptoToken
596             .getProperties()
597             .getProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY);
598     properties.setProperty(
599         CryptoToken.KEYPLACEHOLDERS_PROPERTY, newPlaceholders);
600     cryptoToken.setProperties(properties);
601 
602     final Map<String, Object> details = new LinkedHashMap<String, Object>();
603     details.put(
604         "msg",
605         "Modified name/placeholders of CryptoToken with id " + cryptoTokenId);
606     putDelta("name", oldName, newName, details);
607     putDelta("keyPlaceholders", oldPlaceholders, newPlaceholders, details);
608 
609     cryptoTokenSession.mergeCryptoToken(cryptoToken);
610     securityEventsLoggerSession.log(
611         EventTypes.CRYPTOTOKEN_EDIT,
612         EventStatus.SUCCESS,
613         ModuleTypes.CRYPTOTOKEN,
614         ServiceTypes.CORE,
615         authenticationToken.toString(),
616         String.valueOf(cryptoTokenId),
617         null,
618         null,
619         details);
620     if (LOG.isTraceEnabled()) {
621       LOG.trace(
622           "<saveCryptoToken: cryptoTokenId="
623               + cryptoTokenId
624               + ", newName="
625               + newName);
626     }
627   }

647           null,
648           null,
649           "Deleted CryptoToken with id " + cryptoTokenId);
650     } else if (LOG.isDebugEnabled()) {
651       LOG.debug(
652           "Crypto token with id "
653               + cryptoTokenId
654               + " does not exist and can not be deleted.");
655     }

764   @TransactionAttribute(TransactionAttributeType.REQUIRED)
765   @Override
766   public boolean updatePin(
767       final AuthenticationToken authenticationToken,
768       final Integer cryptoTokenId,
769       final char[] currentAuthenticationCode,
770       final char[] newAuthenticationCode,
771       final boolean updateOnly)
772       throws AuthorizationDeniedException,
773           CryptoTokenAuthenticationFailedException,
774           CryptoTokenOfflineException {
775     final String[] requiredAuthorization =
776         new String[] {
777           CryptoTokenRules.MODIFY_CRYPTOTOKEN.resource() + "/" + cryptoTokenId,
778           CryptoTokenRules.ACTIVATE.resource() + "/" + cryptoTokenId,
779           CryptoTokenRules.DEACTIVATE.resource() + "/" + cryptoTokenId
780         };
781     if (!authorizationSession.isAuthorized(
782         authenticationToken, requiredAuthorization)) {
783       final String msg =
784           INTRES.getLocalizedMessage(
785               "authorization.notauthorizedtoresource",
786               Arrays.toString(requiredAuthorization),
787               authenticationToken.toString());
788       throw new AuthorizationDeniedException(msg);
789     }
790     CryptoToken cryptoToken = getCryptoToken(cryptoTokenId);
791     final Properties cryptoTokenProperties = cryptoToken.getProperties();
792     // Get current auto-activation pin (if any)
793     final String oldAutoActivationPin =
794         BaseCryptoToken.getAutoActivatePin(cryptoTokenProperties);
795     if (oldAutoActivationPin == null
796         && (updateOnly || newAuthenticationCode == null)) {
797       // This is a NOOP call that will not lead to any change
798       return false;
799     }
800     if (SoftCryptoToken.class
801         .getName()
802         .equals(cryptoToken.getClass().getName())) {
803       CryptoProviderTools.installBCProviderIfNotAvailable();
804       final KeyStore keystore;
805       try {
806         keystore = KeyStore.getInstance("PKCS12", "BC");
807         keystore.load(
808             new ByteArrayInputStream(cryptoToken.getTokenData()),
809             currentAuthenticationCode);
810       } catch (Exception e) {
811         final String msg = "Failed to use supplied current PIN." + " " + e;
812         LOG.info(msg);
813         throw new CryptoTokenAuthenticationFailedException(msg);
814       }
815       if (newAuthenticationCode == null) {
816         // When no new pin is supplied, we will not modify the key-store and
817         // just remove the current auto-activation pin
818         cryptoTokenProperties.remove(CryptoToken.AUTOACTIVATE_PIN_PROPERTY);
819         // We'll also remove the default password option
820         cryptoTokenProperties.put(
821             SoftCryptoToken.NODEFAULTPWD, Boolean.TRUE.toString());
822         cryptoToken.setProperties(cryptoTokenProperties);
823       } else {
824         try {
825           final ByteArrayOutputStream baos = new ByteArrayOutputStream();
826           keystore.store(baos, newAuthenticationCode);
827           baos.close();
828           if (oldAutoActivationPin != null || !updateOnly) {
829             BaseCryptoToken.setAutoActivatePin(
830                 cryptoTokenProperties, new String(newAuthenticationCode), true);
831           } else {
832             LOG.debug(
833                 "Auto-activation will not be used. Only changing pin for soft"
834                     + " CryptoToken keystore.");
835           }
836           cryptoToken =
837               CryptoTokenFactory.createCryptoToken(
838                   SoftCryptoToken.class.getName(),
839                   cryptoTokenProperties,
840                   baos.toByteArray(),
841                   cryptoTokenId,
842                   cryptoToken.getTokenName());
843         } catch (Exception e) {
844           LOG.info("Unable to store soft keystore with new PIN: " + e);
845           throw new CryptoTokenAuthenticationFailedException(
846               "Unable to store soft keystore with new PIN");
847         }
848       }
849     } else {
850       if (oldAutoActivationPin != null) {
851         // If we have an old auto-activation pin we will compare the "current"
852         // with this value to avoid deactivating the token
853         if (!oldAutoActivationPin.equals(
854             new String(currentAuthenticationCode))) {
855           final String msg = "Supplied PIN did not match auto-activation PIN.";
856           LOG.info(msg);
857           throw new CryptoTokenAuthenticationFailedException(msg);
858         } else {
859           LOG.debug(
860               "Successfully verified the PIN for non-soft CryptoToken by"
861                   + " comparing supplied PIN to auto-activation PIN.");
862         }
863       } else {
864         // If we don't have an auto-activation pin to compare the supplied PIN
865         // to, we need to verify the supplied
866         // PIN can be used in a de-activation/activation cycle.
867         final boolean wasInactive =
868             !isCryptoTokenStatusActive(authenticationToken, cryptoTokenId);
869         cryptoToken.deactivate();
870         cryptoToken.activate(currentAuthenticationCode);
871         if (wasInactive) {
872           // Note that there is a small glitch here where the token was active,
873           // but we have no other options to verify the pin
874           cryptoToken.deactivate();
875         }
876       }
877       if (newAuthenticationCode == null) {
878         cryptoTokenProperties.remove(CryptoToken.AUTOACTIVATE_PIN_PROPERTY);
879       } else {
880         BaseCryptoToken.setAutoActivatePin(
881             cryptoTokenProperties, new String(newAuthenticationCode), true);
882       }
883       cryptoToken.setProperties(cryptoTokenProperties);
884     }
885     // Save the modified CryptoToken
886     try {
887       cryptoTokenSession.mergeCryptoToken(cryptoToken);
888     } catch (CryptoTokenNameInUseException e) {
889       // This should not happen here since we use the same name and id
890       throw new RuntimeException(e);
891     }
892     securityEventsLoggerSession.log(
893         EventTypes.CRYPTOTOKEN_UPDATEPIN,
894         EventStatus.SUCCESS,
895         ModuleTypes.CRYPTOTOKEN,
896         ServiceTypes.CORE,
897         authenticationToken.toString(),
898         String.valueOf(cryptoTokenId),
899         null,
900         null,
901         "Updated PIN of CryptoToken '"
902             + cryptoToken.getTokenName()
903             + "' with id "
904             + cryptoTokenId);
905     // Return the current auto-activation state
906     return BaseCryptoToken.getAutoActivatePin(cryptoTokenProperties) != null;
907   }

975   @Override
976   public Integer getIdFromName(final String cryptoTokenName) {
977     if (cryptoTokenName == null) {
978       return null;
979     }
980     final Map<String, Integer> cachedNameToIdMap =
981         cryptoTokenSession.getCachedNameToIdMap();
982     Integer cryptoTokenId = cachedNameToIdMap.get(cryptoTokenName);
983     if (cryptoTokenId == null) {
984       // Ok.. so it's not in the cache.. look for it the hard way..
985       for (final Integer currentCryptoTokenId
986           : cryptoTokenSession.getCryptoTokenIds()) {
987         // Don't lookup CryptoTokens we already have in the id to name cache
988         if (!cachedNameToIdMap.values().contains(currentCryptoTokenId)) {
989           final CryptoToken currentCryptoToken =
990               cryptoTokenSession.getCryptoToken(
991                   currentCryptoTokenId.intValue());
992           final String currentCryptoTokenName =
993               currentCryptoToken == null
994                   ? null
995                   : currentCryptoToken.getTokenName();
996           if (cryptoTokenName.equals(currentCryptoTokenName)) {
997             cryptoTokenId = currentCryptoTokenId;
998             break;
999           }
1000         }
1001       }
1002     }
1003     return cryptoTokenId;
1004   }

1019   private List<String> getKeyPairAliasesInternal(final CryptoToken cryptoToken)
1020       throws CryptoTokenOfflineException {
1021     try {
1022       final List<String> aliasEnumeration = cryptoToken.getAliases();
1023       final List<String> keyPairAliases = new ArrayList<String>();
1024       for (final String currentAlias : aliasEnumeration) {
1025         try {
1026           if (cryptoToken.getPublicKey(currentAlias) != null
1027               && cryptoToken.getPrivateKey(currentAlias) != null) {
1028             // A key pair exists for this alias, so add it
1029             keyPairAliases.add(currentAlias);
1030           }
1031         } catch (CryptoTokenOfflineException ignored) {
1032           if (LOG.isDebugEnabled()) {
1033             LOG.debug(
1034                 "Ignord key alias '"
1035                     + currentAlias
1036                     + "' in crypto token '"
1037                     + cryptoToken.getTokenName()
1038                     + "' since it is missing a public and/or private key."
1039                     + " Perhaps it is a symmetric key?");
1040           }
1041         }
1042       }
1043       return keyPairAliases;
1044     } catch (KeyStoreException e) {
1045       throw new CryptoTokenOfflineException(e);
1046     }
1047   }

1050   @Override
1051   public void createKeyPair(
1052       final AuthenticationToken authenticationToken,
1053       final int cryptoTokenId,
1054       final String alias,
1055       final String keySpecificationParam)
1056       throws AuthorizationDeniedException, CryptoTokenOfflineException,
1057           InvalidKeyException, InvalidAlgorithmParameterException {
1058     assertAuthorization(
1059         authenticationToken,
1060         cryptoTokenId,
1061         CryptoTokenRules.GENERATE_KEYS.resource() + "/" + cryptoTokenId);
1062     final CryptoToken cryptoToken =
1063         getCryptoTokenAndAssertExistence(cryptoTokenId);
1064     // Check if alias is already in use
1065     assertAliasNotInUse(cryptoToken, alias);
1066 
1067     // Support "RSAnnnn" and convert it to the legacy format "nnnn"
1068     final String keySpecification;
1069     if (keySpecificationParam.startsWith(AlgorithmConstants.KEYALGORITHM_RSA)) {
1070       keySpecification =
1071           keySpecificationParam.substring(
1072               AlgorithmConstants.KEYALGORITHM_RSA.length());
1073     } else {
1074       keySpecification = keySpecificationParam;
1075     }
1076     // Check if keySpec is valid
1077     KeyTools.checkValidKeyLength(keySpecification);
1078     // Audit log before generation. If the token is an HSM the merge will not
1079     // make a difference.
1080     final Map<String, Object> details = new LinkedHashMap<String, Object>();
1081     details.put("msg", "Generated new keypair in CryptoToken " + cryptoTokenId);
1082     details.put("keyAlias", alias);
1083     details.put("keySpecification", keySpecification);
1084     cryptoToken.generateKeyPair(keySpecification, alias);
1085     cryptoToken.testKeyPair(alias);
1086     // Merge is important for soft tokens where the data is persisted in the
1087     // database, but will also update lastUpdate
1088     try {
1089       cryptoTokenSession.mergeCryptoToken(cryptoToken);
1090     } catch (CryptoTokenNameInUseException e) {
1091       throw new RuntimeException(
1092           e); // We have not changed the name of the CrytpoToken here, so this
1093               // should never happen
1094     }
1095     securityEventsLoggerSession.log(
1096         EventTypes.CRYPTOTOKEN_GEN_KEYPAIR,
1097         EventStatus.SUCCESS,
1098         ModuleTypes.CRYPTOTOKEN,
1099         ServiceTypes.CORE,
1100         authenticationToken.toString(),
1101         String.valueOf(cryptoTokenId),
1102         null,
1103         null,
1104         details);
1105   }

1246   @TransactionAttribute(TransactionAttributeType.REQUIRED)
1247   @Override
1248   public void removeKeyPairPlaceholder(
1249       final AuthenticationToken authenticationToken,
1250       final int cryptoTokenId,
1251       final String alias)
1252       throws AuthorizationDeniedException, InvalidKeyException {
1253     assertAuthorization(
1254         authenticationToken,
1255         cryptoTokenId,
1256         CryptoTokenRules.REMOVE_KEYS.resource() + "/" + cryptoTokenId);
1257     final CryptoToken cryptoToken =
1258         getCryptoTokenAndAssertExistence(cryptoTokenId);
1259 
1260     boolean removed = false;
1261     final Properties props = new Properties();
1262     props.putAll(cryptoToken.getProperties());
1263     final String placeholdersString =
1264         props.getProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY, "");
1265     final List<String> entries =
1266         new ArrayList<String>(
1267             Arrays.asList(
1268                 placeholdersString.split(
1269                     "[" + CryptoToken.KEYPLACEHOLDERS_OUTER_SEPARATOR + "]")));
1270     final Iterator<String> iter = entries.iterator();
1271     while (iter.hasNext()) {
1272       final String entry = iter.next();
1273       if (entry.startsWith(
1274           alias + CryptoToken.KEYPLACEHOLDERS_INNER_SEPARATOR)) {
1275         iter.remove();
1276         removed = true;
1277       }
1278     }
1279 
1280     if (removed) {
1281       final String newValue =
1282           StringUtils.join(
1283               entries, CryptoToken.KEYPLACEHOLDERS_OUTER_SEPARATOR);
1284       props.setProperty(CryptoToken.KEYPLACEHOLDERS_PROPERTY, newValue);
1285       cryptoToken.setProperties(props);
1286     }
1287 
1288     // Check if alias is in use
1289     if (!removed) {
1290       throw new InvalidKeyException("Alias " + alias + " is not in use");
1291     }
1292 
1293     try {
1294       cryptoTokenSession.mergeCryptoToken(cryptoToken);
1295     } catch (CryptoTokenNameInUseException e) {
1296       throw new IllegalStateException(
1297           e); // We have not changed the name of the CrytpoToken here, so this
1298               // should never happen
1299     }
1300 
1301     final Map<String, Object> details = new LinkedHashMap<String, Object>();
1302     details.put(
1303         "msg",
1304         "Deleted key pair placeholder from CryptoToken " + cryptoTokenId);
1305     details.put("keyAlias", alias);
1306     securityEventsLoggerSession.log(
1307         EventTypes.CRYPTOTOKEN_DELETE_ENTRY,
1308         EventStatus.SUCCESS,
1309         ModuleTypes.CRYPTOTOKEN,
1310         ServiceTypes.CORE,
1311         authenticationToken.toString(),
1312         String.valueOf(cryptoTokenId),
1313         null,
1314         null,
1315         details);
1316   }

1379   /**
1380    * Helper method for audit logging changes.
1381    *
1382    * @param oldProperties Old Props
1383    * @param newProperties New Props
1384    * @param details Details
1385    */
1386   private void putDelta(
1387       final Properties oldProperties,
1388       final Properties newProperties,
1389       final Map<String, Object> details) {
1390     // Find out what has happended to all the old properties
1391     for (final Object key : oldProperties.keySet()) {
1392       final String oldValue = oldProperties.getProperty(String.valueOf(key));
1393       final String newValue = newProperties.getProperty(String.valueOf(key));
1394       putDelta(String.valueOf(key), oldValue, newValue, details);
1395     }
1396     // Find out which new properties that did not exist in the old
1397     for (final Object key : newProperties.keySet()) {
1398       final String oldValue = oldProperties.getProperty(String.valueOf(key));
1399       if (oldValue == null) {
1400         final String newValue = newProperties.getProperty(String.valueOf(key));
1401         putDelta(String.valueOf(key), oldValue, newValue, details);
1402       }
1403     }
1404   }

1406   /**
1407    * Helper method for audit logging changes.
1408    *
1409    * @param key Key
1410    * @param oldValue Old value
1411    * @param newValue New value
1412    * @param details Details
1413    */
1414   private void putDelta(
1415       final String key,
1416       final String oldValue,
1417       final String newValue,
1418       final Map<String, Object> details) {
1419     // Treat the auto-activation pin with care
1420     if (BaseCryptoToken.AUTOACTIVATE_PIN_PROPERTY.equals(key)) {
1421       if (oldValue == null && newValue == null) {
1422         // NOP
1423       } else if (oldValue == null && newValue != null) {
1424         details.put("autoActivation", "added");
1425         details.put(
1426             "autoActivationPinProtection",
1427             StringTools.getEncryptVersionFromString(newValue));
1428       } else if (oldValue != null && newValue == null) {
1429         details.put("autoActivation", "removed");
1430       } else if (!oldValue.equals(newValue)) {
1431         details.put("autoActivation", "pin changed");
1432         details.put(
1433             "autoActivationPinProtection",
1434             StringTools.getEncryptVersionFromString(newValue));
1435       }
1436     } else {
1437       if (oldValue == null && newValue == null) {
1438         // NOP
1439       } else if (oldValue == null && newValue != null) {
1440         details.put("added:" + key, newValue);
1441       } else if (oldValue != null && newValue == null) {
1442         details.put("removed:" + key, oldValue);
1443       } else if (!oldValue.equals(newValue)) {
1444         details.put("changed:" + key, newValue);
1445       } else {
1446         details.put(key, newValue);
1447       }
1448     }
1449   }
