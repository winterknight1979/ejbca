74      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
75      @Override
76      public CryptoToken getCryptoToken(final int cryptoTokenId) {
77          // 1. Check (new) CryptoTokenCache if it is time to sync-up with database
78          if (CryptoTokenCache.INSTANCE.shouldCheckForUpdates(cryptoTokenId)) {
79              if (log.isDebugEnabled()) {
80                  log.debug("CryptoToken with ID " + cryptoTokenId + " will be checked for updates.");
81              }
82              // 2. If cache is expired or missing, first thread to discover this reloads item from database and sends it to the cache
83              final CryptoTokenData cryptoTokenData = readCryptoTokenData(cryptoTokenId);
84              if (cryptoTokenData==null) {
85                  if (log.isDebugEnabled()) {
86                      log.debug("Requested cryptoTokenId did not exist in database and will be purged from cache if present: " + cryptoTokenId);
87                  }
88                  // Ensure that it is removed from cache
89                  CryptoTokenCache.INSTANCE.removeEntry(cryptoTokenId);
90              } else {
91                  final int digest = cryptoTokenData.getProtectString(0).hashCode();
92                  // 3. The cache compares the database data with what is in the cache
93                  if (CryptoTokenCache.INSTANCE.willUpdate(cryptoTokenId, digest)) {
94                      final String tokenType = cryptoTokenData.getTokenType();
95                      final Properties properties = cryptoTokenData.getTokenProperties();
96                      final byte[] data = cryptoTokenData.getTokenDataAsBytes();
97                      final String tokenName = cryptoTokenData.getTokenName();
98                      // Create new token and store it in the cache.
99                      String inClassname = getClassNameForType(tokenType);
100                     CryptoToken cryptoToken;
101                     // 4. If database is different from cache, create the crypto token and replace it in the cache (while trying to keep activation)
102                     //    (Invokes org.cesecore.keys.token.CryptoTokenFactory.createCryptoToken)
103                     try {
104                         cryptoToken = CryptoTokenFactory.createCryptoToken(inClassname, properties, data, cryptoTokenId, tokenName, true);
105                     } catch (NoSuchSlotException e) {
106                         // This should never happen now, since the specify allowNonExistingSlot in the createCryptoToken call
107                         throw new IllegalStateException("Attempted to find a slot for a PKCS#11 crypto token, but it did not exists. Perhaps the token was removed?");
108                     }
109                     CryptoTokenCache.INSTANCE.updateWith(cryptoTokenId, digest, tokenName, cryptoToken);                    
110                 }
111             }
112         }
113         // 5. Get CryptoToken from cache (or null) and be merry
114         return CryptoTokenCache.INSTANCE.getEntry(cryptoTokenId);
115     }

117     @Override
118     public String getClassNameForType(String tokenType) {
119         String inClassname = null;
120         for (final AvailableCryptoToken act : CryptoTokenFactory.instance().getAvailableCryptoTokens()) {
121             if (act.getClassPath().endsWith(tokenType)) {
122                 // We found a available token with the same Class.getSimpleName() as the CryptoToken's type, so use it!
123                 // (By only storing the "simple" classname we can switch implementation package without care)
124                 inClassname = act.getClassPath();
125             }
126         }
127         return inClassname;
128     }

130     @Override
131     public int mergeCryptoToken(final CryptoToken cryptoToken) throws CryptoTokenNameInUseException {
132         if (log.isTraceEnabled()) {
133             log.trace(">addCryptoToken " + cryptoToken.getTokenName() + " " + cryptoToken.getClass().getName());
134         }
135         final int cryptoTokenId = cryptoToken.getId();
136         final String tokenName = cryptoToken.getTokenName();
137         String tokenType = "null";
138         for (final AvailableCryptoToken act : CryptoTokenFactory.instance().getAvailableCryptoTokens()) {
139             if (cryptoToken.getClass().getName().equals(act.getClassPath())) {
140                 tokenType = cryptoToken.getClass().getSimpleName();
141                 break;
142             }
143         }
144         final long lastUpdate = System.currentTimeMillis();
145         final Properties tokenProperties = cryptoToken.getProperties();
146         final byte[] tokenDataAsBytes = cryptoToken.getTokenData();
147         CryptoTokenData cryptoTokenData = entityManager.find(CryptoTokenData.class, cryptoTokenId);
148         if (cryptoTokenData == null) {
149             // The cryptoToken does not exist in the database, before we add it we want to check that the name is not in use
150             if (isCryptoTokenNameUsed(tokenName)) {
151                 throw new CryptoTokenNameInUseException(intres.getLocalizedMessage("token.nameisinuse", tokenName));
152             }
153             cryptoTokenData = new CryptoTokenData(cryptoTokenId, tokenName, tokenType, lastUpdate, tokenProperties, tokenDataAsBytes);
154         } else {
155             if (!isCryptoTokenNameUsedByIdOnly(tokenName, cryptoTokenId)) {
156                 throw new CryptoTokenNameInUseException(intres.getLocalizedMessage("token.nameisinuse", tokenName));
157             }
158             // It might be the case that the calling transaction has already loaded a reference to this token
159             // and hence we need to get the same one and perform updates on this object instead of trying to
160             // merge a new object.
161             cryptoTokenData.setTokenName(tokenName);
162             cryptoTokenData.setTokenType(tokenType);
163             cryptoTokenData.setLastUpdate(lastUpdate);
164             cryptoTokenData.setTokenProperties(tokenProperties);
165             cryptoTokenData.setTokenDataAsBytes(tokenDataAsBytes);
166         }
167         cryptoTokenData = createOrUpdateCryptoTokenData(cryptoTokenData);
168         // Update cache with provided token (it might be active and we like keeping things active)
169         CryptoTokenCache.INSTANCE.updateWith(cryptoTokenId, cryptoTokenData.getProtectString(0).hashCode(), tokenName, cryptoToken);
170         if (log.isTraceEnabled()) {
171             log.trace("<addCryptoToken " + cryptoToken.getTokenName());
172         }
173         return cryptoTokenId;   // tokenId
174     }

176     @Override
177     public boolean removeCryptoToken(final int cryptoTokenId) {
178         final boolean ret = deleteCryptoTokenData(cryptoTokenId);
179         CryptoTokenCache.INSTANCE.updateWith(cryptoTokenId, 0, null, null);
180         return ret;
181     }

193     @Override
194     public boolean isCryptoTokenNameUsed(final String cryptoTokenName) {
195         final Query query = entityManager.createQuery("SELECT a FROM CryptoTokenData a WHERE a.tokenName=:tokenName");
196         query.setParameter("tokenName", cryptoTokenName);
197         return !query.getResultList().isEmpty();
198     }
199 
200     @Override
201     public boolean isCryptoTokenNameUsedByIdOnly(final String cryptoTokenName, final int cryptoTokenId) {
202         final Query query = entityManager.createQuery("SELECT a FROM CryptoTokenData a WHERE a.tokenName=:tokenName");
203         query.setParameter("tokenName", cryptoTokenName);
204         @SuppressWarnings("unchecked")
205         final List<CryptoTokenData> cryptoTokenDatas = query.getResultList();
206         for (final CryptoTokenData cryptoTokenData: cryptoTokenDatas) {
207             if (cryptoTokenData.getId() != cryptoTokenId) {
208                 return false;
209             }
210         }
211         return true;
212     }

218     private CryptoTokenData readCryptoTokenData(final int cryptoTokenId) {
219         final Query query = entityManager.createQuery("SELECT a FROM CryptoTokenData a WHERE a.id=:id");
220         query.setParameter("id", cryptoTokenId);
221         return QueryResultWrapper.getSingleResult(query);
222     }
223 
224     private CryptoTokenData createOrUpdateCryptoTokenData(final CryptoTokenData data) {
225         return entityManager.merge(data);
226     }
227 
228     private boolean deleteCryptoTokenData(final int cryptoTokenId) {
229         final Query query = entityManager.createQuery("DELETE FROM CryptoTokenData a WHERE a.id=:id");
230         query.setParameter("id", cryptoTokenId);
231         return query.executeUpdate() == 1;
232     }

235     @Override
236     public List<Integer> getCryptoTokenIds() {
237         return entityManager.createQuery("SELECT a.id FROM CryptoTokenData a").getResultList();
238     }

240     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
241     @Override
242     public Map<Integer,String> getCryptoTokenIdToNameMap() {
243         final Map<Integer,String> ret = new HashMap<>();
244         for (final CryptoTokenData cryptoTokenData : entityManager.createQuery("SELECT a FROM CryptoTokenData a", CryptoTokenData.class).getResultList()) {
245             ret.put(cryptoTokenData.getId(), cryptoTokenData.getTokenName());
246         }
247         return ret;
