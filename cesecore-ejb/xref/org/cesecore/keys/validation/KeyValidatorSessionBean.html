157   @Override
158   public void changeKeyValidator(
159       final AuthenticationToken admin, final Validator validator)
160       throws AuthorizationDeniedException, KeyValidatorDoesntExistsException {
161     assertIsAuthorizedToEditValidators(admin);
162     ProfileData data = profileSession.findById(validator.getProfileId());
163     final String message;
164     final String name = validator.getProfileName();
165     if (data != null) {
166       profileSession.changeProfile(validator);
167       // Since loading a KeyValidator is quite complex, we simple purge the
168       // cache here.
169       ValidatorCache.INSTANCE.removeEntry(data.getId());
170       message = INTRES.getLocalizedMessage("validator.changed_validator", name);
171       final Map<String, Object> details = new LinkedHashMap<String, Object>();
172       details.put("msg", message);
173       // TODO: Include a diff in the changelog
174       // (profileData.getProfile().diff(profile);), but make sure to resolve all
175       // steps so that we don't
176       //      output a ton of serialized garbage (see ECA-5276)
177       auditSession.log(
178           EventTypes.VALIDATOR_CHANGE,
179           EventStatus.SUCCESS,
180           ModuleTypes.VALIDATOR,
181           ServiceTypes.CORE,
182           admin.toString(),
183           null,
184           null,
185           null,
186           details);
187     } else {
188       message =
189           INTRES.getLocalizedMessage("validator.error.change_validator", name);
190       LOG.info(message);
191       throw new KeyValidatorDoesntExistsException(
192           "Validator by ID "
193               + validator.getProfileId()
194               + " does not exist in database.");
195     }
196   }

198   @Override
199   public void removeKeyValidator(
200       final AuthenticationToken admin, final int validatorId)
201       throws AuthorizationDeniedException, CouldNotRemoveKeyValidatorException {
202     if (LOG.isTraceEnabled()) {
203       LOG.trace(">removeKeyValidator(id: " + validatorId + ")");
204     }
205     assertIsAuthorizedToEditValidators(admin);
206     String message;
207 
208     ProfileData data = profileSession.findById(validatorId);
209     if (data == null) {
210       if (LOG.isDebugEnabled()) {
211         LOG.debug(
212             "Trying to remove a key validator that does not exist with ID: "
213                 + validatorId);
214       }
215       return;
216     }
217 
218     if (caSession.existsKeyValidatorInCAs(data.getId())) {
219       throw new CouldNotRemoveKeyValidatorException();
220     }
221 
222     profileSession.removeProfile(data);
223     // Purge the cache here.
224     ValidatorCache.INSTANCE.removeEntry(data.getId());
225     message =
226         INTRES.getLocalizedMessage(
227             "validator.removed_validator", data.getProfileName());
228     final Map<String, Object> details = new LinkedHashMap<String, Object>();
229     details.put("msg", message);
230     auditSession.log(
231         EventTypes.VALIDATOR_REMOVAL,
232         EventStatus.SUCCESS,
233         ModuleTypes.VALIDATOR,
234         ServiceTypes.CORE,
235         admin.toString(),
236         null,
237         null,
238         null,
239         details);
240     if (LOG.isTraceEnabled()) {
241       LOG.trace("<removeKeyValidator()");
242     }
243   }

245   @Override
246   public void removeKeyValidator(
247       final AuthenticationToken admin, final String validatorName)
248       throws AuthorizationDeniedException, CouldNotRemoveKeyValidatorException {
249     if (LOG.isTraceEnabled()) {
250       LOG.trace(">removeKeyValidator(id: " + validatorName + ")");
251     }
252     assertIsAuthorizedToEditValidators(admin);
253     String message;
254 
255     List<ProfileData> datas =
256         profileSession.findByNameAndType(validatorName, Validator.TYPE_NAME);
257     if (datas == null) {
258       if (LOG.isDebugEnabled()) {
259         LOG.debug(
260             "Trying to remove a key validator that does not exist with name: "
261                 + validatorName);
262       }
263       return;
264     } else {
265       for (ProfileData data : datas) {
266         if (caSession.existsKeyValidatorInCAs(data.getId())) {
267           throw new CouldNotRemoveKeyValidatorException();
268         }
269         profileSession.removeProfile(data);
270         // Purge the cache here.
271         ValidatorCache.INSTANCE.removeEntry(data.getId());
272         message =
273             INTRES.getLocalizedMessage(
274                 "validator.removed_validator", data.getProfileName());
275         final Map<String, Object> details = new LinkedHashMap<String, Object>();
276         details.put("msg", message);
277         auditSession.log(
278             EventTypes.VALIDATOR_REMOVAL,
279             EventStatus.SUCCESS,
280             ModuleTypes.VALIDATOR,
281             ServiceTypes.CORE,
282             admin.toString(),
283             null,
284             null,
285             null,
286             details);
287       }
288     }
289 
290     if (LOG.isTraceEnabled()) {
291       LOG.trace("<removeKeyValidator()");
292     }

336   @Override
337   public void cloneKeyValidator(
338       final AuthenticationToken admin,
339       final Validator validator,
340       final String newName)
341       throws AuthorizationDeniedException, KeyValidatorDoesntExistsException,
342           KeyValidatorExistsException {
343     Validator validatorClone = null;
344     final Integer origProfileId = validator.getProfileId();
345     if (origProfileId == null) {
346       throw new KeyValidatorDoesntExistsException(
347           "Could not find key validator " + validator.getProfileName());
348     }
349     validatorClone = getValidator(origProfileId).clone();
350     validatorClone.setProfileName(newName);
351     try {
352       addValidatorInternal(admin, validatorClone);
353       final String message =
354           INTRES.getLocalizedMessage(
355               "validator.cloned_validator",
356               newName,
357               validator.getProfileName());
358       final Map<String, Object> details = new LinkedHashMap<String, Object>();
359       details.put("msg", message);
360       auditSession.log(
361           EventTypes.VALIDATOR_CREATION,
362           EventStatus.SUCCESS,
363           ModuleTypes.VALIDATOR,
364           ServiceTypes.CORE,
365           admin.toString(),
366           null,
367           null,
368           null,
369           details);
370     } catch (KeyValidatorExistsException e) {
371       final String message =
372           INTRES.getLocalizedMessage(
373               "validator.error.clone_validator",
374               newName,
375               validator.getProfileName());
376       LOG.info(message);
377       throw e;
378     }
379   }

418       }

420     if (success) {
421       final String message =
422           INTRES.getLocalizedMessage(
423               "validator.renamed_validator",
424               validator.getProfileName(),
425               newName);

434           null,
435           null,

452   @Override
453   public Map<Integer, Validator> getAllKeyValidators() {
454     final List<ProfileData> keyValidators =
455         findAllProfiles(Validator.TYPE_NAME);
456     final Map<Integer, Validator> result = new HashMap<>();
457     final boolean enabled =
458         ((ExternalScriptsConfiguration)
459                 globalConfigurationSession.getCachedConfiguration("0"))
460             .getEnableExternalScripts();
461     for (ProfileData data : keyValidators) {
462       if (!enabled
463           && data.getProfile() instanceof ExternalCommandCertificateValidator) {
464         if (LOG.isTraceEnabled()) {
465           LOG.trace(
466               "Skip "
467                   + data.getProfileType()
468                   + " with name "
469                   + data.getProfileName()
470                   + " because calls for external command certificate"
471                   + " validators are disabled.");
472         }
473         continue;
474       }
475       // Cast is safe since we know we retrieved the correct implementation
476       try {
477         result.put(data.getId(), (Validator) data.getProfile());
478       } catch (IllegalStateException e) {
479         // NOPMD: Implementation not available in this version if EJBCA
480       }
481     }
482     if (LOG.isDebugEnabled()) {
483       for (Integer id : result.keySet()) {
484         LOG.debug(
485             "Key validators found in datastore: "
486                 + id
487                 + ":"
488                 + result.get(id).getProfileName());
489       }
490     }
491     return result;
492   }

544   @Override
545   public void validateDnsNames(
546       final AuthenticationToken authenticationToken,
547       final CA ca,
548       final EndEntityInformation endEntityInformation,
549       final RequestMessage requestMessage)
550       throws ValidationException {
551     if (!CollectionUtils.isEmpty(ca.getValidators())) {
552       Validator baseValidator;
553       DnsNameValidator validator;
554       for (Integer id : ca.getValidators()) {
555         baseValidator = getValidatorInternal(id, true);
556         if (baseValidator != null
557             && baseValidator
558                 .getValidatorSubType()
559                 .equals(DnsNameValidator.class)) {
560           validator = (DnsNameValidator) baseValidator;
561           // Filter for validator criteria.
562           if (baseValidator instanceof CertificateProfileAwareValidator
563               && !filterCertificateProfileAwareValidator(
564                   validator, endEntityInformation.getCertificateProfileId())) {
565             continue;
566           }
567           CertificateProfile certificateProfile =
568               certificateProfileSession.getCertificateProfile(
569                   endEntityInformation.getCertificateProfileId());
570           final String subjectAltName =
571               endEntityInformation.getSubjectAltName();
572           final List<String> dnsNames = new ArrayList<>();
573           for (String split : subjectAltName.split(",")) {
574             if (split
575                 .trim()
576                 .toLowerCase()
577                 .startsWith(CertTools.DNS.toLowerCase())) {
578               dnsNames.add(split.trim().substring(CertTools.DNS.length() + 1));
579             }
580           }
581           // If the certificate profile allows extension override, there may be
582           // SANs mixed in among the extensions in the request message
583           if (certificateProfile.getAllowExtensionOverride()
584               && requestMessage != null) {
585             Extensions extensions = requestMessage.getRequestExtensions();
586             if (extensions != null) {
587               Extension extension =
588                   extensions.getExtension(Extension.subjectAlternativeName);
589               if (extension != null) {
590                 String extendedSubjectAltName =
591                     CertTools.getAltNameStringFromExtension(extension);
592                 for (String split : extendedSubjectAltName.split(",")) {
593                   if (split
594                       .trim()
595                       .toLowerCase()
596                       .startsWith(CertTools.DNS.toLowerCase())) {
597                     dnsNames.add(
598                         split.trim().substring(CertTools.DNS.length() + 1));
599                   }
600                 }
601               }
602             }
603           }
604 
605           Entry<Boolean, List<String>> result =
606               validator.validate(
607                   EXECUTOR_SERVICE,
608                   dnsNames.toArray(new String[dnsNames.size()]));
609 
610           final String validatorName = validator.getProfileName();
611           final List<String> messages = result.getValue();
612           if (!result.getKey()) {
613             // Validation has failed. Not security event as such, since it will
614             // break issuance and not cause anything important to happen.
615             // We want thorough logging in order to trouble shoot though
616             final String message =
617                 INTRES.getLocalizedMessage(
618                     "validator.caa.validation_failed",
619                     validatorName,
620                     validator.getIssuer(),
621                     messages);
622             LOG.info(
623                 EventTypes.VALIDATOR_VALIDATION_FAILED
624                     + ";"
625                     + EventStatus.FAILURE
626                     + ";"
627                     + ModuleTypes.VALIDATOR
628                     + ";"
629                     + ServiceTypes.CORE
630                     + ";msg="
631                     + message);
632             final int index = validator.getFailedAction();
633             performValidationFailedActions(index, message);
634           } else {
635             // Validation succeeded, this can be considered a security audit
636             // event because CAs may be asked to present this as evidence to an
637             // auditor
638             final String message =
639                 INTRES.getLocalizedMessage(
640                     "validator.caa.validation_successful",
641                     validatorName,
642                     validator.getIssuer(),
643                     messages);
644             final Map<String, Object> details =
645                 new LinkedHashMap<String, Object>();
646             details.put("msg", message);
647             auditSession.log(
648                 EventTypes.VALIDATOR_VALIDATION_SUCCESS,
649                 EventStatus.SUCCESS,
650                 ModuleTypes.VALIDATOR,
651                 ServiceTypes.CORE,
652                 authenticationToken.toString(),
653                 String.valueOf(ca.getCAId()),
654                 null,
655                 endEntityInformation.getUsername(),
656                 details);
657           }
658         }
659       }
660     } else {
661       if (LOG.isDebugEnabled()) {
662         LOG.debug(
663             "No validators configured for CA "
664                 + ca.getName()
665                 + " (ID="
666                 + ca.getCAId()
667                 + ").");
668       }
669     }
670   }

672   @Override
673   public boolean validatePublicKey(
674       final AuthenticationToken admin,
675       final CA ca,
676       final EndEntityInformation endEntityInformation,
677       final CertificateProfile certificateProfile,
678       final Date notBefore,
679       final Date notAfter,
680       final PublicKey publicKey)
681       throws ValidationException, IllegalValidityException {
682     boolean result = true;
683     if (ca != null
684         && !CollectionUtils.isEmpty(
685             ca
686                 .getValidators())) { // || certificateProfile.isTypeRootCA() ||
687                                      // certificateProfile.isTypeSubCA()
688       final CertificateValidity certificateValidity =
689           new CertificateValidity(
690               endEntityInformation,
691               certificateProfile,
692               notBefore,
693               notAfter,
694               ca.getCACertificate(),
695               false,
696               false);
697       if (LOG.isDebugEnabled()) {
698         LOG.debug(
699             "Validate "
700                 + publicKey.getAlgorithm()
701                 + " public key with "
702                 + publicKey.getFormat()
703                 + " format.");
704         LOG.debug(
705             "Certificate 'notBefore' " + certificateValidity.getNotBefore());
706         LOG.debug(
707             "Certificate 'notAfter' " + certificateValidity.getNotAfter());
708       }
709       Validator baseValidator;
710       KeyValidator validator;
711       String name;
712       for (Integer id : ca.getValidators()) {
713         baseValidator = getValidatorInternal(id, true);
714         if (baseValidator != null
715             && baseValidator.getValidatorSubType().equals(KeyValidator.class)) {
716           validator = (KeyValidator) baseValidator;
717           name = validator.getProfileName();
718           if (LOG.isTraceEnabled()) {
719             LOG.trace(
720                 "Try to apply key validator: " + validator.toDisplayString());
721           }
722           try {
723             // Filter for validator criteria.
724             if (validator instanceof CertificateProfileAwareValidator
725                 && !filterCertificateProfileAwareValidator(
726                     validator,
727                     endEntityInformation.getCertificateProfileId())) {
728               continue;
729             }
730             if (validator instanceof ValidityAwareValidator
731                 && !filterValidityAwareValidator(
732                     validator,
733                     certificateValidity.getNotBefore(),
734                     certificateValidity.getNotAfter())) {
735               continue;
736             }
737             final String fingerprint =
738                 CertTools.createPublicKeyFingerprint(publicKey, "SHA-256");
739             LOG.info(
740                 INTRES.getLocalizedMessage(
741                     "validator.key.isbeingprocessed",
742                     name,
743                     endEntityInformation.getUsername(),
744                     fingerprint));
745             List<String> messages =
746                 validator.validate(publicKey, certificateProfile);
747             if (messages.size() > 0) { // Validation has failed.
748               result = false;
749               final int index = validator.getFailedAction();
750               final String message =
751                   INTRES.getLocalizedMessage(
752                       "validator.key.validation_failed", name, messages);
753               final Map<String, Object> details =
754                   new LinkedHashMap<String, Object>();
755               details.put("msg", message);
756               auditSession.log(
757                   EventTypes.VALIDATOR_VALIDATION_FAILED,
758                   EventStatus.FAILURE,
759                   ModuleTypes.VALIDATOR,
760                   ServiceTypes.CORE,
761                   admin.toString(),
762                   String.valueOf(ca.getCAId()),
763                   fingerprint,
764                   endEntityInformation.getUsername(),
765                   details);
766               performValidationFailedActions(index, message);
767             } else {
768               final byte[] keyBytes = publicKey.getEncoded();
769               final String publicKeyEncoded =
770                   (keyBytes != null
771                       ? new String(Base64.encode(keyBytes))
772                       : "null");
773               final String message =
774                   INTRES.getLocalizedMessage(
775                       "validator.key.validation_successful",
776                       name,
777                       publicKeyEncoded);
778               LOG.info(message);
779               final Map<String, Object> details =
780                   new LinkedHashMap<String, Object>();
781               details.put("msg", message);
782               auditSession.log(
783                   EventTypes.VALIDATOR_VALIDATION_SUCCESS,
784                   EventStatus.SUCCESS,
785                   ModuleTypes.VALIDATOR,
786                   ServiceTypes.CORE,
787                   admin.toString(),
788                   String.valueOf(ca.getCAId()),
789                   null,
790                   endEntityInformation.getUsername(),
791                   details);
792             }
793           } catch (ValidatorNotApplicableException e) {
794             // This methods either throws a KeyValidationException, or just logs
795             // a message and validation should be considered successful
796             // use method performValidationFailedActions because it's the same
797             // actions
798             performValidationFailedActions(
799                 validator.getNotApplicableAction(), e.getMessage());
800           } catch (ValidationException e) {
801             throw e;
802           }
803         }
804       }
805     } else {
806       if (LOG.isDebugEnabled()) {
807         LOG.debug(
808             "No key validator configured for CA "
809                 + ca.getName()
810                 + " (ID="
811                 + ca.getCAId()
812                 + ").");
813       }
814     }
815     return result;
816   }

818   @Override
819   public void validateCertificate(
820       final AuthenticationToken authenticationToken,
821       final IssuancePhase phase,
822       final CA ca,
823       final EndEntityInformation endEntityInformation,
824       final X509Certificate certificate)
825       throws ValidationException {
826     if (LOG.isDebugEnabled()) {
827       LOG.debug("Validate certificate for phase " + phase);
828     }
829     if (ca != null && !CollectionUtils.isEmpty(ca.getValidators())) {
830       Validator baseValidator;
831       CertificateValidator validator;
832       String name;
833       for (Integer id : ca.getValidators()) {
834         baseValidator = getValidatorInternal(id, true);
835         if (baseValidator != null
836             && baseValidator
837                 .getValidatorSubType()
838                 .equals(CertificateValidator.class)) {
839           validator = (CertificateValidator) baseValidator;
840           name = validator.getProfileName();
841           if (phase.getIndex() != validator.getPhase()) {
842             continue;
843           }
844           if (validator instanceof CertificateProfileAwareValidator
845               && !filterCertificateProfileAwareValidator(
846                   validator, endEntityInformation.getCertificateProfileId())) {
847             continue;
848           }
849           if (validator instanceof ValidityAwareValidator
850               && !filterValidityAwareValidator(
851                   validator,
852                   certificate.getNotBefore(),
853                   certificate.getNotAfter())) {
854             continue;
855           }
856           try {
857             final String fingerprint =
858                 CertTools.createPublicKeyFingerprint(
859                     certificate.getPublicKey(), "SHA-256");
860             LOG.info(
861                 INTRES.getLocalizedMessage(
862                     "validator.certificate.isbeingprocessed",
863                     name,
864                     phase,
865                     endEntityInformation.getUsername(),
866                     fingerprint));
867             final ExternalScriptsConfiguration externalScriptsConfiguration =
868                 (ExternalScriptsConfiguration)
869                     globalConfigurationSession.getCachedConfiguration("0");
870             final ExternalScriptsWhitelist externalScriptsWhitelist =
871                 ExternalScriptsWhitelist.fromText(
872                     externalScriptsConfiguration.getExternalScriptsWhitelist(),
873                     externalScriptsConfiguration
874                         .getIsExternalScriptsWhitelistEnabled());
875             final List<String> messages =
876                 validator.validate(ca, certificate, externalScriptsWhitelist);
877             if (messages.size() > 0) { // Evaluation has failed.
878               final String message =
879                   INTRES.getLocalizedMessage(
880                       "validator.certificate.validation_failed",
881                       name,
882                       messages);
883               final Map<String, Object> details =
884                   new LinkedHashMap<String, Object>();
885               details.put("msg", message);
886               auditSession.log(
887                   EventTypes.VALIDATOR_VALIDATION_FAILED,
888                   EventStatus.FAILURE,
889                   ModuleTypes.VALIDATOR,
890                   ServiceTypes.CORE,
891                   authenticationToken.toString(),
892                   String.valueOf(ca.getCAId()),
893                   fingerprint,
894                   endEntityInformation.getUsername(),
895                   details);
896               performValidationFailedActions(
897                   validator.getFailedAction(), message);
898             } else {
899               final String message =
900                   INTRES.getLocalizedMessage(
901                       "validator.certificate.validation_successful",
902                       name,
903                       fingerprint);
904               LOG.info(message);
905               final Map<String, Object> details =
906                   new LinkedHashMap<String, Object>();
907               details.put("msg", message);
908               auditSession.log(
909                   EventTypes.VALIDATOR_VALIDATION_SUCCESS,
910                   EventStatus.SUCCESS,
911                   ModuleTypes.VALIDATOR,
912                   ServiceTypes.CORE,
913                   authenticationToken.toString(),
914                   String.valueOf(ca.getCAId()),
915                   null,
916                   endEntityInformation.getUsername(),
917                   details);
918             }
919           } catch (ValidatorNotApplicableException e) {
920             // This methods either throws a KeyValidationException, or just logs
921             // a message and validation should be considered successful
922             // use method performValidationFailedActions because it's the same
923             // actions
924             performValidationFailedActions(
925                 validator.getNotApplicableAction(), e.getMessage());
926           } catch (CertificateException e) {
927             throw new ValidationException(
928                 "Certificate to validate could not be parsed or decoded: "
929                     + e.getMessage(),
930                 e);
931           }
932         }
933       }
934     }

970   @Override
971   public Collection<Integer> getAuthorizedKeyValidatorIds(
972       final AuthenticationToken admin, final String keyValidatorAccessRule) {
973     final ArrayList<Integer> result = new ArrayList<Integer>();
974     final Map<Integer, String> map = getKeyValidatorIdToNameMap();
975     if (authorizationSession.isAuthorizedNoLogging(
976         admin, keyValidatorAccessRule)) {
977       final boolean rootAccess =
978           authorizationSession.isAuthorizedNoLogging(
979               admin, StandardRules.ROLE_ROOT.resource());
980       final List<Integer> authorizedCPIDs =
981           certificateProfileSession.getAuthorizedCertificateProfileIds(
982               admin, 0);
983       for (final Entry<Integer, String> entry : map.entrySet()) {
984         // Check that administrator have access to all certificate profiles
985         // referenced by the validator
986         Validator val = getValidator(entry.getKey());
987         boolean allexists = true;
988         for (final Integer nextcpid : val.getCertificateProfileIds()) {
989           if (LOG.isTraceEnabled()) {
990             LOG.trace(
991                 "Validator '"
992                     + val.getProfileName()
993                     + "' has "
994                     + val.getCertificateProfileIds().size()
995                     + " no of CPs selected");
996           }
997           // If any CP is selected, it's access to all (only authorized will be
998           // displayed)
999           if (nextcpid.intValue() == -1) {
1000             if (LOG.isDebugEnabled()) {
1001               LOG.debug(
1002                   "Validator is applicable to all certificate profiles, not"
1003                       + " limiting access based on CPs");
1004             }
1005             allexists = true;
1006             break;
1007           }
1008           // superadmin should be able to access profiles with missing CA Ids
1009           if (!authorizedCPIDs.contains(nextcpid) && (!rootAccess)) {
1010             if (LOG.isDebugEnabled()) {
1011               LOG.debug(
1012                   "Validator have certificate profile "
1013                       + nextcpid
1014                       + " selected which admin is not authorized to:"
1015                       + admin.toString());
1016             }
1017             allexists = false;
1018             break;
1019           }
1020         }
1021         if (allexists) {
1022           result.add(entry.getKey());
1023         }
1024       }
1025     }
1026     return result;
1027   }

1074   /**
1075    * Gets a validator by cache or database, can return null. Puts it into the
1076    * cache, if not already present.
1077    *
1078    * @param id the validators id.
1079    * @param fromCache true if the validator can be taken from cache.
1080    * @return a cloned validator that can be used at will without affecting the
1081    *     cache contents, edits have to be saved with replaceKeyValidator
1082    */
1083   private Validator getValidatorInternal(
1084       final int id, final boolean fromCache) {
1085     Validator result = null;
1086     // If we should read from cache, and we have an id to use in the cache, and
1087     // the cache does not need to be updated
1088     if (fromCache && !ValidatorCache.INSTANCE.shouldCheckForUpdates(id)) {
1089       // Get from cache (or null)
1090       result = ValidatorCache.INSTANCE.getEntry(id);
1091     }
1092 
1093     // if we selected to not read from cache, or if the cache did not contain
1094     // this entry
1095     if (result == null) {
1096 
1097       // We need to read from database because we specified to not get from
1098       // cache or we don't have anything in the cache
1099       final ProfileData data = profileSession.findById(id);
1100       if (data != null) {
1101         result = (Validator) data.getProfile();
1102         if (LOG.isDebugEnabled()) {
1103           LOG.debug("Load validator: " + result.getDataMap());
1104         }
1105         final int digest = data.getProtectString(0).hashCode();
1106         // The cache compares the database data with what is in the cache
1107         // If database is different from cache, replace it in the cache
1108         ValidatorCache.INSTANCE.updateWith(
1109             data.getId(), digest, data.getProfileName(), result);
1110       } else {
1111         // Ensure that it is removed from cache if it exists
1112         ValidatorCache.INSTANCE.removeEntry(id);
1113       }
1114     }
1115     if (result == null) {
1116       LOG.warn("Validator with id " + id + " didn't return any validator");
1117     }
1118     // We need to clone the validator, otherwise the cache contents will be
1119     // modifiable from the outside
1120     if (result != null) {
1121       return result.clone();
1122     }
1123     return null;
1124   }

1175    * @param certificateNotBefore the certificates not before validity.

1179   private boolean filterValidityAwareValidator(
1180       final ValidityAwareValidator validator,
1181       final Date certificateNotBefore,

1184       LOG.trace("Try to apply validity aware validator: " + validator);

1187     boolean result = true;
1188     if (!KeyValidatorDateConditions.evaluate(
1189         validator.getNotBefore(),
1190         certificateNotBefore,
1191         validator.getNotBeforeCondition())) {
1192       if (LOG.isDebugEnabled()) {
1193         LOG.debug(
1194             INTRES.getLocalizedMessage(
1195                 "validator.filterconditiondoesnotmatch", name, "notBefore"));
1196       }
1197       result = false;
1198     }
1199     if (!KeyValidatorDateConditions.evaluate(
1200         validator.getNotAfter(),
1201         certificateNotAfter,
1202         validator.getNotAfterCondition())) {
1203       if (LOG.isDebugEnabled()) {
1204         LOG.debug(
1205             INTRES.getLocalizedMessage(
1206                 "validator.filterconditiondoesnotmatch", name, "notAfter"));
1207       }
1208       result = false;
1209     }
1210     return result;

1256   /**
1257    * Post processes every validation depending on its failed action.
1258    *
1259    * @param failedAction the failed action index (see {@link
1260    *     KeyValidationFailedActions}).
1261    * @param message the message to log.
1262    * @param shortMessage the error message to EJBCA Certificate Enrollment Error
1263    *     page
1264    * @throws ValidationException if a failed validation has to be abort the
1265    *     certificate issuance.
1266    */
1267   private void performValidationFailedActions(
1268       final int failedAction, final String message, final String shortMessage)
1269       throws ValidationException {
1270     if (LOG.isDebugEnabled()) {
1271       LOG.debug("Perform post action " + failedAction + " - " + message);
1272     }
1273     if (KeyValidationFailedActions.LOG_INFO.getIndex() == failedAction) {
1274       LOG.info(message);
1275     } else if (KeyValidationFailedActions.LOG_WARN.getIndex() == failedAction) {
1276       LOG.warn(message);
1277     } else if (KeyValidationFailedActions.LOG_ERROR.getIndex()
1278         == failedAction) {
1279       LOG.error(message);
1280     } else if (KeyValidationFailedActions.ABORT_CERTIFICATE_ISSUANCE.getIndex()
1281         == failedAction) {
1282       throw new ValidationException(shortMessage);
1283     } else {
1284       // NOOP
1285       LOG.debug(message);
1286     }
1287   }
