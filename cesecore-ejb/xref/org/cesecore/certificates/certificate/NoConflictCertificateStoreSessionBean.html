56  /**
57   * These methods call CertificateStoreSession for certificates that are plain CertificateData entities.
58   * See {@link CertificateStoreSession} for method descriptions.
59   * 
60   * <p>For NoConflictCertificateData the methods perform additional logic to check that it gets the most recent
61   * entry if there's more than one (taking permanent revocations into account), and for updates it
62   * appends new entries instead of updating existing ones. 
63   * 
64   * @version $Id: NoConflictCertificateStoreSessionBean.java 29433 2018-07-02 16:55:26Z mikekushner $
65   */
66  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "NoConflictCertificateStoreSessionRemote")
67  @TransactionAttribute(TransactionAttributeType.SUPPORTS)
68  public class NoConflictCertificateStoreSessionBean implements NoConflictCertificateStoreSessionRemote, NoConflictCertificateStoreSessionLocal {

91      private void authorizedToCA(final AuthenticationToken admin, final int caid) throws AuthorizationDeniedException {
92          if (!authorizationSession.isAuthorized(admin, StandardRules.CAACCESS.resource() + caid)) {
93              final String msg = intres.getLocalizedMessage("caadmin.notauthorizedtoca", admin.toString(), caid);
94              throw new AuthorizationDeniedException(msg);
95          }
96      }

98      /**
99       * Returns true if the CA allows revocation of non-existing certificates.
100      * @param issuerDN Subject DN of CA.
101      */
102     @Override
103     public boolean canRevokeNonExisting(final String issuerDN) {
104         final int caid = CertTools.stringToBCDNString(StringTools.strip(issuerDN)).hashCode();
105         final CAInfo cainfo = caSession.getCAInfoInternal(caid);
106         return canRevokeNonExisting(cainfo, issuerDN);
107     }
108 
109     /**
110      * @return true if the CA allows revocation of non-existing certificates.
111      * @param cainfo CA
112      * @param issuerDN Subject DN of CA, for safety check against CAId collisions.
113      * 
114      */
115     private boolean canRevokeNonExisting(final CAInfo cainfo, final String issuerDN) {
116         String dn = CertTools.stringToBCDNString(StringTools.strip(issuerDN));
117         if (cainfo == null || !cainfo.getSubjectDN().equals(dn) || !cainfo.isAcceptRevocationNonExistingEntry()) {
118             return false;
119         }
120         // XXX this option can be set in the certificate profile as well! does it make sense to have mixed locations? it would make CRL generation more complex!
121         if (cainfo.isUseCertificateStorage()) {
122             if (log.isDebugEnabled()) {
123                 log.debug("CA '" + cainfo.getName() + "' is misconfigured. Revocation of non-existing certificates is currently only supported for 'throw away CAs'.");
124             }
125             return false;
126         }
127         return true;
128     }

130     @Override
131     public CertificateDataWrapper getCertificateDataByIssuerAndSerno(final String issuerdn, final BigInteger certserno) {
132         CertificateDataWrapper cdw = certificateStoreSession.getCertificateDataByIssuerAndSerno(issuerdn, certserno);
133         if (cdw != null) {
134             // Full certificate is available, return it
135             return cdw;
136         }
137 
138         // Throw away CA or missing certificate
139         final int caid = CertTools.stringToBCDNString(StringTools.strip(issuerdn)).hashCode();
140         final CAInfo cainfo = caSession.getCAInfoInternal(caid);
141         if (!canRevokeNonExisting(cainfo, issuerdn)) {
142             if (cainfo == null && log.isDebugEnabled()) {
143                 log.debug("Tried to look up certificate " + certserno.toString(16) +", but neither certificate nor CA was found. CA Id: " + caid + ". Issuer DN: '" + issuerdn + "'");
144             }
145             return null;
146         }
147         if (cainfo.isUseNoConflictCertificateData()) {
148             final NoConflictCertificateData certificateData = getLimitedNoConflictCertDataRow(cainfo, certserno);
149             return new CertificateDataWrapper(certificateData);
150         } else {
151             final CertificateData certificateData = new CertificateData();
152             fillInLimitedCertificateData(certificateData, cainfo, certserno);
153             certificateData.setUpdateTime(System.currentTimeMillis());
154             return new CertificateDataWrapper(certificateData, null);
155         }
156     }

158     @Override
159     public CertificateStatus getStatus(final String issuerDN, final BigInteger serno) {
160         if (log.isTraceEnabled()) {
161             log.trace(">getStatus(), dn:" + issuerDN + ", serno=" + serno.toString(16));
162         }
163         // First, try to look up in CertificateData
164         final String dn = CertTools.stringToBCDNString(issuerDN);
165         CertificateStatus status = certificateStoreSession.getStatus(issuerDN, serno);
166         if (!canRevokeNonExisting(issuerDN) || status != CertificateStatus.NOT_AVAILABLE) {
167             log.trace("<getStatus()");
168             return status;
169         }
170         // If not found, take most recent certificate from NoConflictCertificateData
171         final NoConflictCertificateData noConflictCert = findMostRecentCertData(dn, serno); 
172         if (noConflictCert == null) {
173             if (log.isTraceEnabled()) {
174                 log.trace("<getStatus() did not find certificate with dn " + dn + " and serno " + serno.toString(16));
175             }
176             // For throw-away CAs that allow revocation of non-existing certificates, we pretend that non-existing is OK
177             return CertificateStatus.OK;
178         }
179         status = CertificateStatusHelper.getCertificateStatus(noConflictCert);
180         if (log.isTraceEnabled()) {
181             log.trace("<getStatus() returned " + status + " for cert number " + serno.toString(16));
182         }
183         return status;
184         
185     }

187     @Override
188     @TransactionAttribute(TransactionAttributeType.REQUIRED)
189     public CertificateDataWrapper getCertificateData(final String fingerprint) {
190         CertificateDataWrapper cdw = certificateStoreSession.getCertificateData(fingerprint);
191         if (cdw != null) {
192             return cdw;
193         }
194         // If not found, take most recent certificate from NoConflictCertificateData
195         final Collection<NoConflictCertificateData> certDatas = noConflictCertificateDataSession.findByFingerprint(fingerprint);
196         return new CertificateDataWrapper(filterMostRecentCertData(certDatas));
197     }

199     @Override
200     public Collection<RevokedCertInfo> listRevokedCertInfo(String issuerdn, long lastbasecrldate) {
201         if (log.isTraceEnabled()) {
202             log.trace(">listRevokedCertInfo('" + issuerdn + "', " + lastbasecrldate + ")");
203         }
204         final Collection<RevokedCertInfo> revokedInCertData = certificateStoreSession.listRevokedCertInfo(issuerdn, lastbasecrldate);
205         final Collection<RevokedCertInfo> revokedInNoConflictData = noConflictCertificateDataSession.getRevokedCertInfosWithDuplicates(issuerdn, lastbasecrldate);
206         if (log.isDebugEnabled()) {
207             log.debug("listRevokedCertInfo: Got " + revokedInCertData.size() + " entries from CertificateData and " + revokedInNoConflictData.size() + " entries from NoConflictCertificateData");
208         }
209         return RevokedCertInfo.mergeByDateAndStatus(revokedInCertData, revokedInNoConflictData, lastbasecrldate);
210     }

212     /**
213      * Locates the most recent entry in NoConflictCertificateData for a given issuerdn/serial number combination.
214      * @param issuerdn Issuer DN
215      * @param serno Certificate serial number
216      * @return NoConflictCertificateData entry, or null if not found. Entity is append-only, so do not modify it.
217      */
218     private NoConflictCertificateData findMostRecentCertData(final String issuerdn, final BigInteger serno) {
219         final Collection<NoConflictCertificateData> certDatas = noConflictCertificateDataSession.findByIssuerDNSerialNumber(issuerdn, serno.toString());
220         return filterMostRecentCertData(certDatas);
221     }
222     
223     /**
224      * Filters out the most recent entry in NoConflictCertificateData for a given issuerDN/serial number combination.
225      * Permanent revocations always take precedence over other updates, the first one wins.
226      * Otherwise, the most recent update wins.
227      * @param certDatas Collection of NoConflictCertificateData to filter.
228      * @return NoConflictCertificateData entry, or null if not found. Entity is append-only, so do not modify it.
229      */
230     private NoConflictCertificateData filterMostRecentCertData(final Collection<NoConflictCertificateData> certDatas) {
231         if (CollectionUtils.isEmpty(certDatas)) {
232             log.trace("<findMostRecentCertData(): no certificates found");
233             return null;
234         }
235         NoConflictCertificateData mostRecentData = null;
236         for (final NoConflictCertificateData data : certDatas) {
237             if (mostRecentData == null) {
238                 mostRecentData = data;
239                 continue;
240             }
241             long timestampThis = data.getUpdateTime() != null ? data.getUpdateTime() : 0;
242             long timestampRecent = mostRecentData.getUpdateTime() != null ? mostRecentData.getUpdateTime() : 0;
243             if (RevokedCertInfo.isPermanentlyRevoked(data.getRevocationReason())) {
244                 // Permanently revoked certificate always takes precedence over non-permanently revoked one.
245                 // Older permanent revocations take precedence over newer ones.
246                 if (!RevokedCertInfo.isPermanentlyRevoked(mostRecentData.getRevocationReason()) || timestampRecent > timestampThis) {
247                     mostRecentData = data;
248                     continue;
249                 }
250             }
251             // Permanent revocations take precedence over temporary ones
252             if (RevokedCertInfo.isPermanentlyRevoked(mostRecentData.getRevocationReason())) {
253                 continue;
254             }
255             // Otherwise, most recent status takes precedence
256             if (timestampThis > timestampRecent) {
257                 mostRecentData = data;
258             }
259         }
260         return mostRecentData;
261     }

263     @Override
264     @TransactionAttribute(TransactionAttributeType.REQUIRED)
265     public boolean setRevokeStatus(final AuthenticationToken admin, final CertificateDataWrapper cdw, final Date revokedDate, final int reason)
266             throws CertificateRevokeException, AuthorizationDeniedException {
267         if (cdw.getBaseCertificateData() instanceof NoConflictCertificateData) {
268             if (entityManager.contains(cdw.getBaseCertificateData())) {
269                 throw new IllegalStateException("Cannot update existing row in NoConflictCertificateData. It is append-only.");
270             }
271         }
272         return certificateStoreSession.setRevokeStatus(admin, cdw, revokedDate, reason);
273     }

275     @Override
276     public boolean setStatus(final AuthenticationToken admin, final String fingerprint, final int status) throws AuthorizationDeniedException {
277         if (!certificateStoreSession.setStatus(admin, fingerprint, status)) {
278             // Perhaps stored in NoConflictCertificateData
279             final List<NoConflictCertificateData> certDatas = noConflictCertificateDataSession.findByFingerprint(fingerprint);
280             NoConflictCertificateData certData = filterMostRecentCertData(certDatas);
281             if (certData != null) {
282                 changeStatus(admin, certData, status);
283             }
284         }
285         return false;
286     }

288     private void changeStatus(final AuthenticationToken admin, final NoConflictCertificateData certificateData, final int status) throws AuthorizationDeniedException {
289         if (log.isDebugEnabled()) {
290             log.debug("Set status " + status + " for certificate with serial: " + certificateData.getSerialNumberHex());
291         }
292 
293         // Must be authorized to CA in order to change status is certificates issued by the CA
294         String bcdn = CertTools.stringToBCDNString(certificateData.getIssuerDN());
295         int caid = bcdn.hashCode();
296         authorizedToCA(admin, caid);
297         
298         final NoConflictCertificateData newCertData = new NoConflictCertificateData(certificateData);
299         newCertData.setStatus(status);
300         setUniqueIdAndUpdateTime(newCertData);
301         entityManager.persist(newCertData);
302         
303         final String serialNo = certificateData.getSerialNumberHex();
304         final String msg = intres.getLocalizedMessage("store.setstatus", certificateData.getUsername(), certificateData.getFingerprint(), status, certificateData.getSubjectDnNeverNull(), certificateData.getIssuerDN(), serialNo);
305         Map<String, Object> details = new LinkedHashMap<>();
306         details.put("msg", msg);
307         logSession.log(EventTypes.CERT_CHANGEDSTATUS, EventStatus.SUCCESS, ModuleTypes.CERTIFICATE, ServiceTypes.CORE, admin.toString(), String.valueOf(caid), serialNo, certificateData.getUsername(), details);
308     }

310     /**
311      * Returns a row in the append-only NoConflictCertificateData table, or a new row that can be added.
312      * The row is initialized with the data from the most recent entry in the table,
313      * or as a new unrevoked entry if non-existent.
314      * @param cainfo Issuer.
315      * @param certserno Certificate serial number.
316      * @return New row, or copy of an existing row. Always has a fresh UUID and timestamp, so it can be appended directly.
317      */
318     private NoConflictCertificateData getLimitedNoConflictCertDataRow(final CAInfo cainfo, final BigInteger certserno) {
319         NoConflictCertificateData certificateData = findMostRecentCertData(cainfo.getSubjectDN(), certserno);
320         if (certificateData != null) {
321             // Make a copy, to prevent overwrites
322             certificateData = new NoConflictCertificateData(certificateData);
323         } else {
324             certificateData = new NoConflictCertificateData();
325             fillInLimitedCertificateData(certificateData, cainfo, certserno);
326         }
327         setUniqueIdAndUpdateTime(certificateData);
328         return certificateData;
329     }
330     
331     private void setUniqueIdAndUpdateTime(final NoConflictCertificateData certificateData) {
332         // Always generate new UUID and timestamp, so updates are stored as a new row
333         certificateData.setId(UUID.randomUUID().toString());
334         certificateData.setUpdateTime(System.currentTimeMillis());
335     }

337     /** @param certificateData Data
338      * @param cainfo CA
339      * @param certserno Serial 
340      * @see org.cesecore.certificates.certificate.CertificateStoreSessionBean#updateLimitedCertificateDataStatus(AuthenticationToken, int, String, String, String, BigInteger, int, Date, int, String) */
341     private void fillInLimitedCertificateData(final BaseCertificateData certificateData, final CAInfo cainfo, final BigInteger certserno) {
342         final int certProfId = cainfo.getDefaultCertificateProfileId();
343         certificateData.setSerialNumber(certserno.toString());
344         // A fingerprint is needed by the publisher session, so we put a dummy fingerprint here
345         certificateData.setFingerprint(generateDummyFingerprint(cainfo.getSubjectDN(), certserno));
346         certificateData.setIssuerDN(cainfo.getSubjectDN());
347         certificateData.setSubject("CN=limited");
348         certificateData.setUsername(null);
349         certificateData.setCertificateProfileId(certProfId);
350         certificateData.setStatus(CertificateConstants.CERT_ACTIVE);
351         certificateData.setRevocationReason(RevocationReasons.NOT_REVOKED.getDatabaseValue());
352         certificateData.setRevocationDate(-1L);
353         certificateData.setCaFingerprint(CertTools.getFingerprintAsString(cainfo.getCertificateChain().get(0)));
354         certificateData.setEndEntityProfileId(-1);
355         // Set expire date to the maximum possible expire date this certificate could have (now + cert profile validity) 
356         final CertificateProfile certProf = certificateProfileSession.getCertificateProfile(certProfId);
357         if (certProf == null) {
358             log.info("Missing certificate profile ID: " + certProfId);
359         } else {
360             final String encodedValidity = certProf.getEncodedValidity();
361             final Date expireDate = ValidityDate.getDate(encodedValidity, new Date());
362             certificateData.setExpireDate(expireDate);
363         }
364     }
365     
366     @Override
367     public String generateDummyFingerprint(final String issuerdn, final BigInteger certserno) {
368         final byte[] fingerprintBytes = CertTools.generateSHA1Fingerprint((certserno.toString()+';'+issuerdn).getBytes(StandardCharsets.UTF_8));
369         return new String(Hex.encode(fingerprintBytes));
370     }
