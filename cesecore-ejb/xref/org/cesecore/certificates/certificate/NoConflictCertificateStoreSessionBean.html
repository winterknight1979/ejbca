54  /**
55   * These methods call CertificateStoreSession for certificates that are plain
56   * CertificateData entities. See {@link CertificateStoreSession} for method
57   * descriptions.
58   *
59   * <p>For NoConflictCertificateData the methods perform additional logic to
60   * check that it gets the most recent entry if there's more than one (taking
61   * permanent revocations into account), and for updates it appends new entries
62   * instead of updating existing ones.
63   *
64   * @version $Id: NoConflictCertificateStoreSessionBean.java 29433 2018-07-02
65   *     16:55:26Z mikekushner $
66   */

70              + "NoConflictCertificateStoreSessionRemote")

104   private void authorizedToCA(final AuthenticationToken admin, final int caid)
105       throws AuthorizationDeniedException {
106     if (!authorizationSession.isAuthorized(
107         admin, StandardRules.CAACCESS.resource() + caid)) {
108       final String msg =
109           INTRES.getLocalizedMessage(
110               "caadmin.notauthorizedtoca", admin.toString(), caid);
111       throw new AuthorizationDeniedException(msg);
112     }
113   }

115   /**
116    * Returns true if the CA allows revocation of non-existing certificates.
117    *
118    * @param issuerDN Subject DN of CA.
119    */
120   @Override
121   public boolean canRevokeNonExisting(final String issuerDN) {
122     final int caid =
123         CertTools.stringToBCDNString(StringTools.strip(issuerDN)).hashCode();
124     final CAInfo cainfo = caSession.getCAInfoInternal(caid);
125     return canRevokeNonExisting(cainfo, issuerDN);
126   }
127 
128   /**
129    * @return true if the CA allows revocation of non-existing certificates.
130    * @param cainfo CA
131    * @param issuerDN Subject DN of CA, for safety check against CAId collisions.
132    */
133   private boolean canRevokeNonExisting(
134       final CAInfo cainfo, final String issuerDN) {
135     String dn = CertTools.stringToBCDNString(StringTools.strip(issuerDN));
136     if (cainfo == null
137         || !cainfo.getSubjectDN().equals(dn)
138         || !cainfo.isAcceptRevocationNonExistingEntry()) {
139       return false;
140     }
141     // XXX this option can be set in the certificate profile as well! does it
142     // make sense to have mixed locations? it would make CRL generation more
143     // complex!
144     if (cainfo.isUseCertificateStorage()) {
145       if (LOG.isDebugEnabled()) {
146         LOG.debug(
147             "CA '"
148                 + cainfo.getName()
149                 + "' is misconfigured. Revocation of non-existing certificates"
150                 + " is currently only supported for 'throw away CAs'.");
151       }
152       return false;
153     }
154     return true;
155   }

157   @Override
158   public CertificateDataWrapper getCertificateDataByIssuerAndSerno(
159       final String issuerdn, final BigInteger certserno) {
160     CertificateDataWrapper cdw =
161         certificateStoreSession.getCertificateDataByIssuerAndSerno(
162             issuerdn, certserno);
163     if (cdw != null) {
164       // Full certificate is available, return it
165       return cdw;
166     }
167 
168     // Throw away CA or missing certificate
169     final int caid =
170         CertTools.stringToBCDNString(StringTools.strip(issuerdn)).hashCode();
171     final CAInfo cainfo = caSession.getCAInfoInternal(caid);
172     if (!canRevokeNonExisting(cainfo, issuerdn)) {
173       if (cainfo == null && LOG.isDebugEnabled()) {
174         LOG.debug(
175             "Tried to look up certificate "
176                 + certserno.toString(16)
177                 + ", but neither certificate nor CA was found. CA Id: "
178                 + caid
179                 + ". Issuer DN: '"
180                 + issuerdn
181                 + "'");
182       }
183       return null;
184     }
185     if (cainfo.isUseNoConflictCertificateData()) {
186       final NoConflictCertificateData certificateData =
187           getLimitedNoConflictCertDataRow(cainfo, certserno);
188       return new CertificateDataWrapper(certificateData);
189     } else {
190       final CertificateData certificateData = new CertificateData();
191       fillInLimitedCertificateData(certificateData, cainfo, certserno);
192       certificateData.setUpdateTime(System.currentTimeMillis());
193       return new CertificateDataWrapper(certificateData, null);
194     }
195   }

197   @Override
198   public CertificateStatus getStatus(
199       final String issuerDN, final BigInteger serno) {
200     if (LOG.isTraceEnabled()) {
201       LOG.trace(
202           ">getStatus(), dn:" + issuerDN + ", serno=" + serno.toString(16));
203     }
204     // First, try to look up in CertificateData
205     final String dn = CertTools.stringToBCDNString(issuerDN);
206     CertificateStatus status =
207         certificateStoreSession.getStatus(issuerDN, serno);
208     if (!canRevokeNonExisting(issuerDN)
209         || status != CertificateStatus.NOT_AVAILABLE) {
210       LOG.trace("<getStatus()");
211       return status;
212     }
213     // If not found, take most recent certificate from NoConflictCertificateData
214     final NoConflictCertificateData noConflictCert =
215         findMostRecentCertData(dn, serno);
216     if (noConflictCert == null) {
217       if (LOG.isTraceEnabled()) {
218         LOG.trace(
219             "<getStatus() did not find certificate with dn "
220                 + dn
221                 + " and serno "
222                 + serno.toString(16));
223       }
224       // For throw-away CAs that allow revocation of non-existing certificates,
225       // we pretend that non-existing is OK
226       return CertificateStatus.OK;
227     }
228     status = CertificateStatusHelper.getCertificateStatus(noConflictCert);
229     if (LOG.isTraceEnabled()) {
230       LOG.trace(
231           "<getStatus() returned "
232               + status
233               + " for cert number "
234               + serno.toString(16));
235     }
236     return status;
237   }

239   @Override
240   @TransactionAttribute(TransactionAttributeType.REQUIRED)
241   public CertificateDataWrapper getCertificateData(final String fingerprint) {
242     CertificateDataWrapper cdw =
243         certificateStoreSession.getCertificateData(fingerprint);
244     if (cdw != null) {
245       return cdw;
246     }
247     // If not found, take most recent certificate from NoConflictCertificateData
248     final Collection<NoConflictCertificateData> certDatas =
249         noConflictCertificateDataSession.findByFingerprint(fingerprint);
250     return new CertificateDataWrapper(filterMostRecentCertData(certDatas));
251   }

253   @Override
254   public Collection<RevokedCertInfo> listRevokedCertInfo(
255       final String issuerdn, final long lastbasecrldate) {
256     if (LOG.isTraceEnabled()) {
257       LOG.trace(
258           ">listRevokedCertInfo('" + issuerdn + "', " + lastbasecrldate + ")");
259     }
260     final Collection<RevokedCertInfo> revokedInCertData =
261         certificateStoreSession.listRevokedCertInfo(issuerdn, lastbasecrldate);
262     final Collection<RevokedCertInfo> revokedInNoConflictData =
263         noConflictCertificateDataSession.getRevokedCertInfosWithDuplicates(
264             issuerdn, lastbasecrldate);
265     if (LOG.isDebugEnabled()) {
266       LOG.debug(
267           "listRevokedCertInfo: Got "
268               + revokedInCertData.size()
269               + " entries from CertificateData and "
270               + revokedInNoConflictData.size()
271               + " entries from NoConflictCertificateData");
272     }
273     return RevokedCertInfo.mergeByDateAndStatus(
274         revokedInCertData, revokedInNoConflictData, lastbasecrldate);
275   }

277   /**
278    * Locates the most recent entry in NoConflictCertificateData for a given
279    * issuerdn/serial number combination.
280    *
281    * @param issuerdn Issuer DN
282    * @param serno Certificate serial number
283    * @return NoConflictCertificateData entry, or null if not found. Entity is
284    *     append-only, so do not modify it.
285    */
286   private NoConflictCertificateData findMostRecentCertData(
287       final String issuerdn, final BigInteger serno) {
288     final Collection<NoConflictCertificateData> certDatas =
289         noConflictCertificateDataSession.findByIssuerDNSerialNumber(
290             issuerdn, serno.toString());
291     return filterMostRecentCertData(certDatas);
292   }

294   /**
295    * Filters out the most recent entry in NoConflictCertificateData for a given
296    * issuerDN/serial number combination. Permanent revocations always take
297    * precedence over other updates, the first one wins. Otherwise, the most
298    * recent update wins.
299    *
300    * @param certDatas Collection of NoConflictCertificateData to filter.
301    * @return NoConflictCertificateData entry, or null if not found. Entity is
302    *     append-only, so do not modify it.
303    */
304   private NoConflictCertificateData filterMostRecentCertData(
305       final Collection<NoConflictCertificateData> certDatas) {
306     if (CollectionUtils.isEmpty(certDatas)) {
307       LOG.trace("<findMostRecentCertData(): no certificates found");
308       return null;
309     }
310     NoConflictCertificateData mostRecentData = null;
311     for (final NoConflictCertificateData data : certDatas) {
312       if (mostRecentData == null) {
313         mostRecentData = data;
314         continue;
315       }
316       long timestampThis =
317           data.getUpdateTime() != null ? data.getUpdateTime() : 0;
318       long timestampRecent =
319           mostRecentData.getUpdateTime() != null
320               ? mostRecentData.getUpdateTime()
321               : 0;
322       if (RevokedCertInfo.isPermanentlyRevoked(data.getRevocationReason())) {
323         // Permanently revoked certificate always takes precedence over
324         // non-permanently revoked one.
325         // Older permanent revocations take precedence over newer ones.
326         if (!RevokedCertInfo.isPermanentlyRevoked(
327                 mostRecentData.getRevocationReason())
328             || timestampRecent > timestampThis) {
329           mostRecentData = data;
330           continue;
331         }
332       }
333       // Permanent revocations take precedence over temporary ones
334       if (RevokedCertInfo.isPermanentlyRevoked(
335           mostRecentData.getRevocationReason())) {
336         continue;
337       }
338       // Otherwise, most recent status takes precedence
339       if (timestampThis > timestampRecent) {
340         mostRecentData = data;
341       }
342     }
343     return mostRecentData;
344   }

346   @Override
347   @TransactionAttribute(TransactionAttributeType.REQUIRED)
348   public boolean setRevokeStatus(
349       final AuthenticationToken admin,
350       final CertificateDataWrapper cdw,
351       final Date revokedDate,
352       final int reason)
353       throws CertificateRevokeException, AuthorizationDeniedException {
354     if (cdw.getBaseCertificateData() instanceof NoConflictCertificateData) {
355       if (entityManager.contains(cdw.getBaseCertificateData())) {
356         throw new IllegalStateException(
357             "Cannot update existing row in NoConflictCertificateData. It is"
358                 + " append-only.");
359       }
360     }
361     return certificateStoreSession.setRevokeStatus(
362         admin, cdw, revokedDate, reason);
363   }

365   @Override
366   public boolean setStatus(
367       final AuthenticationToken admin,
368       final String fingerprint,
369       final int status)
370       throws AuthorizationDeniedException {
371     if (!certificateStoreSession.setStatus(admin, fingerprint, status)) {
372       // Perhaps stored in NoConflictCertificateData
373       final List<NoConflictCertificateData> certDatas =
374           noConflictCertificateDataSession.findByFingerprint(fingerprint);
375       NoConflictCertificateData certData = filterMostRecentCertData(certDatas);
376       if (certData != null) {
377         changeStatus(admin, certData, status);
378       }
379     }
380     return false;
381   }

383   private void changeStatus(
384       final AuthenticationToken admin,
385       final NoConflictCertificateData certificateData,
386       final int status)
387       throws AuthorizationDeniedException {
388     if (LOG.isDebugEnabled()) {
389       LOG.debug(
390           "Set status "
391               + status
392               + " for certificate with serial: "
393               + certificateData.getSerialNumberHex());
394     }
395 
396     // Must be authorized to CA in order to change status is certificates issued
397     // by the CA
398     String bcdn = CertTools.stringToBCDNString(certificateData.getIssuerDN());
399     int caid = bcdn.hashCode();
400     authorizedToCA(admin, caid);
401 
402     final NoConflictCertificateData newCertData =
403         new NoConflictCertificateData(certificateData);
404     newCertData.setStatus(status);
405     setUniqueIdAndUpdateTime(newCertData);
406     entityManager.persist(newCertData);
407 
408     final String serialNo = certificateData.getSerialNumberHex();
409     final String msg =
410         INTRES.getLocalizedMessage(
411             "store.setstatus",
412             certificateData.getUsername(),
413             certificateData.getFingerprint(),
414             status,
415             certificateData.getSubjectDnNeverNull(),
416             certificateData.getIssuerDN(),
417             serialNo);
418     Map<String, Object> details = new LinkedHashMap<>();
419     details.put("msg", msg);
420     logSession.log(
421         EventTypes.CERT_CHANGEDSTATUS,
422         EventStatus.SUCCESS,
423         ModuleTypes.CERTIFICATE,
424         ServiceTypes.CORE,
425         admin.toString(),
426         String.valueOf(caid),
427         serialNo,
428         certificateData.getUsername(),
429         details);
430   }

432   /**
433    * Returns a row in the append-only NoConflictCertificateData table, or a new
434    * row that can be added. The row is initialized with the data from the most
435    * recent entry in the table, or as a new unrevoked entry if non-existent.
436    *
437    * @param cainfo Issuer.
438    * @param certserno Certificate serial number.
439    * @return New row, or copy of an existing row. Always has a fresh UUID and
440    *     timestamp, so it can be appended directly.
441    */
442   private NoConflictCertificateData getLimitedNoConflictCertDataRow(
443       final CAInfo cainfo, final BigInteger certserno) {
444     NoConflictCertificateData certificateData =
445         findMostRecentCertData(cainfo.getSubjectDN(), certserno);
446     if (certificateData != null) {
447       // Make a copy, to prevent overwrites
448       certificateData = new NoConflictCertificateData(certificateData);
449     } else {
450       certificateData = new NoConflictCertificateData();
451       fillInLimitedCertificateData(certificateData, cainfo, certserno);
452     }
453     setUniqueIdAndUpdateTime(certificateData);
454     return certificateData;
455   }
456 
457   private void setUniqueIdAndUpdateTime(
458       final NoConflictCertificateData certificateData) {
459     // Always generate new UUID and timestamp, so updates are stored as a new
460     // row
461     certificateData.setId(UUID.randomUUID().toString());
462     certificateData.setUpdateTime(System.currentTimeMillis());
463   }

465   /**
466    * @param certificateData Data
467    * @param cainfo CA
468    * @param certserno Serial
469    * @see
470    *     org.cesecore.certificates.certificate.CertificateStoreSessionBean#updateLimitedCertificateDataStatus(AuthenticationToken,
471    *     int, String, String, String, BigInteger, int, Date, int, String)
472    */
473   private void fillInLimitedCertificateData(
474       final BaseCertificateData certificateData,
475       final CAInfo cainfo,
476       final BigInteger certserno) {
477     final int certProfId = cainfo.getDefaultCertificateProfileId();
478     certificateData.setSerialNumber(certserno.toString());
479     // A fingerprint is needed by the publisher session, so we put a dummy
480     // fingerprint here
481     certificateData.setFingerprint(
482         generateDummyFingerprint(cainfo.getSubjectDN(), certserno));
483     certificateData.setIssuerDN(cainfo.getSubjectDN());
484     certificateData.setSubject("CN=limited");
485     certificateData.setUsername(null);
486     certificateData.setCertificateProfileId(certProfId);
487     certificateData.setStatus(CertificateConstants.CERT_ACTIVE);
488     certificateData.setRevocationReason(
489         RevocationReasons.NOT_REVOKED.getDatabaseValue());
490     certificateData.setRevocationDate(-1L);
491     certificateData.setCaFingerprint(
492         CertTools.getFingerprintAsString(cainfo.getCertificateChain().get(0)));
493     certificateData.setEndEntityProfileId(-1);
494     // Set expire date to the maximum possible expire date this certificate
495     // could have (now + cert profile validity)
496     final CertificateProfile certProf =
497         certificateProfileSession.getCertificateProfile(certProfId);
498     if (certProf == null) {
499       LOG.info("Missing certificate profile ID: " + certProfId);
500     } else {
501       final String encodedValidity = certProf.getEncodedValidity();
502       final Date expireDate = ValidityDate.getDate(encodedValidity, new Date());
503       certificateData.setExpireDate(expireDate);
504     }
505   }

507   @Override
508   public String generateDummyFingerprint(
509       final String issuerdn, final BigInteger certserno) {
510     final byte[] fingerprintBytes =
511         CertTools.generateSHA1Fingerprint(
512             (certserno.toString() + ';' + issuerdn)
513                 .getBytes(StandardCharsets.UTF_8));
514     return new String(Hex.encode(fingerprintBytes));
515   }
