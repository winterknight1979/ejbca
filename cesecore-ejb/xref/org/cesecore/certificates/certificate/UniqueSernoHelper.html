77    private static X509Certificate getTestCertificate(
78        final String base64EncodedCertificate) {
79      final X509Certificate ret;
80      final byte[] certEncoded1;
81      certEncoded1 =
82          org.bouncycastle.util.encoders.Base64.decode(base64EncodedCertificate);
83      try {
84        final CertificateFactory cf =
85            CertificateFactory.getInstance("X.509", "BC");
86        ret =
87            (X509Certificate)
88                cf.generateCertificate(new ByteArrayInputStream(certEncoded1));
89      } catch (CertificateException e) {
90        throw new RuntimeException(
91            "Not possible to generate predefined dummy certificate. Should never"
92                + " happen",
93            e);
94      } catch (NoSuchProviderException e) {
95        throw new RuntimeException(
96            "Not possible to generate predefined dummy certificate. Should never"
97                + " happen",
98            e);
99      }
100     return ret;
101   }
