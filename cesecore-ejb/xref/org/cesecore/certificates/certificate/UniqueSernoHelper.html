61      private static X509Certificate getTestCertificate(final String base64EncodedCertificate) {
62          final X509Certificate ret;
63          final byte certEncoded1[];
64          certEncoded1= org.bouncycastle.util.encoders.Base64.decode(base64EncodedCertificate);
65          try {
66              final CertificateFactory cf = CertificateFactory.getInstance("X.509", "BC");
67              ret = (X509Certificate)cf.generateCertificate(new ByteArrayInputStream(certEncoded1));
68          } catch (CertificateException e) {
69              throw new RuntimeException( "Not possible to generate predefined dummy certificate. Should never happen", e );
70          } catch (NoSuchProviderException e) {
71              throw new RuntimeException( "Not possible to generate predefined dummy certificate. Should never happen", e );
72          }
73          return ret;
74      }
