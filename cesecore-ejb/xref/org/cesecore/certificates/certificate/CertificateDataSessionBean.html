129   @Override
130   public CertificateInfo findFirstCertificateInfo(
131       final String issuerDN, final String serialNumber) {
132     CertificateInfo ret = null;
133     final Query query =
134         entityManager.createNativeQuery(
135             "SELECT a.fingerprint, a.subjectDN, a.cAFingerprint, a.status,"
136                 + " a.type, a.serialNumber, a.notBefore, a.expireDate,"
137                 + " a.revocationDate, a.revocationReason, a.username, a.tag,"
138                 + " a.certificateProfileId, a.endEntityProfileId,"
139                 + " a.updateTime, a.subjectKeyId, a.subjectAltName FROM"
140                 + " CertificateData a WHERE a.issuerDN=:issuerDN AND"
141                 + " a.serialNumber=:serialNumber",
142             "CertificateInfoSubset2");
143     query.setParameter("issuerDN", issuerDN);
144     query.setParameter("serialNumber", serialNumber);
145     query.setMaxResults(1);
146     @SuppressWarnings("unchecked")
147     final List<Object[]> resultList = query.getResultList();
148     if (!resultList.isEmpty()) {
149       final Object[] fields = resultList.get(0);
150       // The order of the results are defined by the SqlResultSetMapping
151       // annotation
152       final String fingerprint = (String) fields[0];
153       final String subjectDN = (String) fields[1];
154       final String cafp = (String) fields[2];
155       final int status = ValueExtractor.extractIntValue(fields[3]);
156       final int type = ValueExtractor.extractIntValue(fields[4]);
157       final Long notBefore;
158       if (fields[5] == null) {
159         notBefore = null;
160       } else {
161         notBefore = ValueExtractor.extractLongValue(fields[5]);
162       }
163       final long expireDate = ValueExtractor.extractLongValue(fields[6]);
164       final long revocationDate = ValueExtractor.extractLongValue(fields[7]);
165       final int revocationReason = ValueExtractor.extractIntValue(fields[8]);
166       final String username = (String) fields[9];
167       final String tag = (String) fields[10];
168       final int certificateProfileId =
169           ValueExtractor.extractIntValue(fields[11]);
170       final Integer endEntityProfileId;
171       if (fields[12] == null) {
172         endEntityProfileId = null;
173       } else {
174         endEntityProfileId = ValueExtractor.extractIntValue(fields[12]);
175       }
176       final long updateTime;
177       if (fields[13] == null) {
178         updateTime = 0; // Might be null in an upgraded installation
179       } else {
180         updateTime = ValueExtractor.extractLongValue(fields[13]);
181       }
182       final String subjectKeyId = (String) fields[14];
183       final String subjectAltName = (String) fields[15];
184       ret =
185           new CertificateInfo(
186               fingerprint,
187               cafp,
188               serialNumber,
189               issuerDN,
190               subjectDN,
191               status,
192               type,
193               notBefore,
194               expireDate,
195               revocationDate,
196               revocationReason,
197               username,
198               tag,
199               certificateProfileId,
200               endEntityProfileId,
201               updateTime,
202               subjectKeyId,
203               subjectAltName);
204     }
205     return ret;
206   }

448   @Override
449   public List<Certificate> getCertificateList(final List<CertificateData> cdl) {
450     final List<Certificate> cl = new LinkedList<>();
451     for (final CertificateData cd : cdl) {
452       final Certificate cert = cd.getCertificate(entityManager);
453       if (cert == null) {
454         continue;
455       }
456       cl.add(cert);
457     }
458     return cl;
459   }

461   @Override
462   public List<Certificate> findCertificatesByIssuerDnAndSerialNumbers(
463       final String issuerDN, final Collection<BigInteger> serialNumbers) {
464     final StringBuilder sb = new StringBuilder();
465     for (final BigInteger serno : serialNumbers) {
466       sb.append(", '");
467       sb.append(serno.toString());
468       sb.append("'");
469     }
470     // to save the repeating if-statement in the above closure not to add ', '
471     // as the first characters in the StringBuilder we remove the two chars
472     // here :)
473     sb.delete(0, ", ".length());
474     // Derby: Columns of type 'LONG VARCHAR' may not be used in CREATE INDEX,
475     // ORDER BY, GROUP BY, UNION, INTERSECT, EXCEPT or DISTINCT statements
476     // because comparisons are not supported for that type.
477     // Since two certificates in the database should never be the same, "SELECT
478     // DISTINCT ..." was changed to "SELECT ..." here.
479     final TypedQuery<CertificateData> query =
480         entityManager.createQuery(
481             "SELECT a FROM CertificateData a WHERE a.issuerDN=:issuerDN AND"
482                 + " a.serialNumber IN (:sb)",
483             CertificateData.class);
484     query.setParameter("issuerDN", issuerDN);
485     query.setParameter("sb", sb.toString());
486     return getCertificateList(query.getResultList());
487   }

612   @Override
613   @SuppressWarnings("unchecked")
614   public List<Object[]> findExpirationInfo(
615       final Collection<String> cas,
616       final Collection<Integer> certificateProfiles,
617       final long activeNotifiedExpireDateMin,
618       final long activeNotifiedExpireDateMax,
619       final long activeExpireDateMin) {
620     // We don't select the base64 certificate data here, because it may be a
621     // LONG data type which we can't simply select, or we don't want to read all
622     // the data.
623     final Query query =
624         entityManager.createNativeQuery(
625             "SELECT DISTINCT fingerprint as fingerprint, username as username"
626                 + " FROM CertificateData WHERE "
627                 + "issuerDN IN (:cas) AND "
628                 // If the list of certificate profiles is empty, ignore it as a
629                 // parameter
630                 + (!certificateProfiles.isEmpty()
631                     ? "certificateProfileId IN (:certificateProfiles) AND"
632                     : "")
633                 + "(expireDate>:activeNotifiedExpireDateMin) AND"
634                 + " (expireDate<:activeNotifiedExpireDateMax) AND"
635                 + " (status=:status1 OR status=:status2) AND"
636                 + " (expireDate>=:activeExpireDateMin OR status=:status3)",
637             "FingerprintUsernameSubset");
638     query.setParameter("cas", cas);
639     if (!certificateProfiles.isEmpty()) {
640       query.setParameter("certificateProfiles", certificateProfiles);
641     }
642     query.setParameter(
643         "activeNotifiedExpireDateMin", activeNotifiedExpireDateMin);
644     query.setParameter(
645         "activeNotifiedExpireDateMax", activeNotifiedExpireDateMax);
646     query.setParameter("status1", CertificateConstants.CERT_ACTIVE);
647     query.setParameter(
648         "status2", CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION);
649     query.setParameter("activeExpireDateMin", activeExpireDateMin);
650     query.setParameter("status3", CertificateConstants.CERT_ACTIVE);
651     // How to debug log the SQL query:
652     // log.debug("findExpirationInfo:
653     // "+query.unwrap(org.hibernate.Query.class).getQueryString());
654     return query.getResultList();
655   }
