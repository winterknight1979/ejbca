264     @Override
265     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
266     public void reloadOcspSigningCache() {
267     	if (log.isTraceEnabled()) {
268     		log.trace(">reloadOcspSigningCache");
269     	}
270         // Cancel any waiting timers of this type
271         cancelTimers(TIMERID_OCSPSIGNINGCACHE);
272         try {      
273          // Verify card key holder
274             if (log.isDebugEnabled() && (CardKeyHolder.getInstance().getCardKeys() == null)) {
275                 log.debug(intres.getLocalizedMessage("ocsp.classnotfound", hardTokenClassName));
276             }
277             GlobalOcspConfiguration ocspConfiguration = (GlobalOcspConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID);
278             OcspSigningCache.INSTANCE.stagingStart();
279             try {
280                 // Populate OcspSigningCache
281                 // Add all potential CA's as OCSP responders to the staging area
282                 for (final Integer caId : caSession.getAllCaIds()) {
283                     final List<X509Certificate> caCertificateChain = new ArrayList<X509Certificate>();
284 
285                     final CAInfo caInfo = caSession.getCAInfoInternal(caId.intValue());
286                     if (caInfo == null || caInfo.getCAType() == CAInfo.CATYPE_CVC) {
287                         // Bravely ignore OCSP for CVC CAs
288                         continue;
289                     }
290                     if (caInfo.getStatus() == CAConstants.CA_ACTIVE) {
291                         //Cache active CAs as signers
292                         if (log.isDebugEnabled()) {
293                             log.debug("Processing X509 CA " + caInfo.getName() + " (" + caInfo.getCAId() + ").");
294                         }
295                         final CAToken caToken = caInfo.getCAToken();
296                         final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(caToken.getCryptoTokenId());
297                         if (cryptoToken == null) {
298                             log.info("Excluding CA with id " + caId + " for OCSP signing consideration due to missing CryptoToken.");
299                             continue;
300                         }
301                         for (final Certificate certificate : caInfo.getCertificateChain()) {
302                             caCertificateChain.add((X509Certificate) certificate);
303                         }
304                         final String keyPairAlias;
305                         try {
306                             keyPairAlias = caToken.getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN);
307                         } catch (CryptoTokenOfflineException e) {
308                             log.warn("Referenced private key with purpose " + CATokenConstants.CAKEYPURPOSE_CERTSIGN
309                                     + " could not be used. CryptoToken is off-line for CA with id " + caId + ": " + e.getMessage());
310                             continue;
311                         }
312                         final PrivateKey privateKey;
313                         try {
314                             privateKey = cryptoToken.getPrivateKey(keyPairAlias);
315                         } catch (CryptoTokenOfflineException e) {
316                             log.warn("Referenced private key with alias " + keyPairAlias
317                                     + " could not be used. CryptoToken is off-line for CA with id " + caId + ": " + e.getMessage());
318                             continue;
319                         }
320                         if (privateKey == null) {
321                             log.warn("Referenced private key with alias " + keyPairAlias + " does not exist. Ignoring CA with id " + caId);
322                             continue;
323                         }
324                         final String signatureProviderName = cryptoToken.getSignProviderName();
325                         if (caCertificateChain.size() > 0) {
326                             X509Certificate caCertificate = caCertificateChain.get(0);
327                             final CertificateStatus caCertificateStatus = getRevocationStatusWhenCasPrivateKeyIsCompromised(caCertificate, false);
328                             OcspSigningCache.INSTANCE.stagingAdd(new OcspSigningCacheEntry(caCertificate, caCertificateStatus, caCertificateChain,
329                                     null, privateKey, signatureProviderName, null, ocspConfiguration.getOcspResponderIdType()));
330                             // Check if CA cert has been revoked (only key compromise as returned above). Always make this check, even if this CA has an OCSP signing certificate, because
331                             // signing will still fail even if the signing cert is valid. Shouldn't happen, but log it just in case.
332                             if (caCertificateStatus.equals(CertificateStatus.REVOKED)) {
333                                 log.warn("Active CA with subject DN '" + CertTools.getSubjectDN(caCertificate) + "' and serial number "
334                                         + CertTools.getSerialNumber(caCertificate) + " has a revoked certificate with reason "
335                                         + caCertificateStatus.revocationReason + ".");
336                             }
337                             //Check if CA cert is expired
338                             if (!CertTools.isCertificateValid(caCertificate, true)) {
339                                 log.warn("Active CA with subject DN '" + CertTools.getSubjectDN(caCertificate) + "' and serial number "
340                                         + CertTools.getSerialNumber(caCertificate) + " has an expired certificate with expiration date "
341                                         + CertTools.getNotAfter(caCertificate) + ".");
342                             }
343                         } else {
344                             log.warn("CA with ID " + caId
345                                     + " appears to lack a certificate in the database. This may be a serious error if not in a test environment.");
346                         }
347                     } else if (caInfo.getStatus() == CAConstants.CA_EXTERNAL) {
348                         // If set, all external CA's without a keybinding (set below) will be responded to by the default responder. 
349                         for (final Certificate certificate : caInfo.getCertificateChain()) {
350                             caCertificateChain.add((X509Certificate) certificate);
351                         }
352                         final CertificateStatus caCertificateStatus = getRevocationStatusWhenCasPrivateKeyIsCompromised(caCertificateChain.get(0),
353                                 false);
354                         // Check if CA cert has been revoked (only key compromise as returned above). Always make this check, even if this CA has an OCSP signing certificate, because
355                         // signing will still fail even if the signing cert is valid. 
356                         if (caCertificateStatus.equals(CertificateStatus.REVOKED)) {
357                             log.info("External CA with subject DN '" + CertTools.getSubjectDN(caCertificateChain.get(0)) + "' and serial number "
358                                     + CertTools.getSerialNumber(caCertificateChain.get(0)) + " has a revoked certificate with reason "
359                                     + caCertificateStatus.revocationReason + ".");
360                         }
361                         //Check if CA cert is expired
362                         if (!CertTools.isCertificateValid(caCertificateChain.get(0), false)) {
363                             log.info("External CA with subject DN '" + CertTools.getSubjectDN(caCertificateChain.get(0)) + "' and serial number "
364                                     + CertTools.getSerialNumber(caCertificateChain.get(0)) + " has an expired certificate with expiration date "
365                                     + CertTools.getNotAfter(caCertificateChain.get(0)) + ".");
366                         }
367                         //Add an entry with just a chain and nothing else
368                         OcspSigningCache.INSTANCE.stagingAdd(new OcspSigningCacheEntry(caCertificateChain.get(0), caCertificateStatus, null, null,
369                                 null, null, null, ocspConfiguration.getOcspResponderIdType()));
370 
371                     }
372                 }
373                 // Add all potential InternalKeyBindings as OCSP responders to the staging area, overwriting CA entries from before
374                 for (final int internalKeyBindingId : internalKeyBindingDataSession.getIds(OcspKeyBinding.IMPLEMENTATION_ALIAS)) {
375                     final OcspKeyBinding ocspKeyBinding = (OcspKeyBinding) internalKeyBindingDataSession.getInternalKeyBinding(internalKeyBindingId);  
376                     if (log.isDebugEnabled()) {
377                         log.debug("Processing " + ocspKeyBinding.getName() + " (" + ocspKeyBinding.getId() + ")");
378                     }
379                     if (!ocspKeyBinding.getStatus().equals(InternalKeyBindingStatus.ACTIVE)) {
380                         if (log.isDebugEnabled()) {
381                             log.debug("Ignoring OcspKeyBinding since it is not active.");
382                         }
383                         continue;
384                     }
385                     final X509Certificate ocspSigningCertificate = (X509Certificate) certificateStoreSession.findCertificateByFingerprint(ocspKeyBinding.getCertificateId());
386                     if (ocspSigningCertificate == null) {
387                         log.warn("OCSP signing certificate with referenced fingerprint " + ocspKeyBinding.getCertificateId() +
388                                 " does not exist. Ignoring internalKeyBinding with id " + ocspKeyBinding.getId());
389                         continue;
390                     }
391                     //Make the same check as above 
392                     if (certificateStoreSession.getStatus(CertTools.getIssuerDN(ocspSigningCertificate), CertTools.getSerialNumber(ocspSigningCertificate))
393                             .equals(CertificateStatus.REVOKED)) {
394                         log.warn("OCSP Responder certificate with subject DN '" + CertTools.getSubjectDN(ocspSigningCertificate) + "' and serial number "
395                                 + CertTools.getSerialNumber(ocspSigningCertificate) + " is revoked.");
396                     }
397                     //Check if signing cert is expired
398                     if (!CertTools.isCertificateValid(ocspSigningCertificate, true)) {
399                         log.warn("OCSP Responder certificate with subject DN '" + CertTools.getSubjectDN(ocspSigningCertificate) + "' and serial number "
400                                 + CertTools.getSerialNumber(ocspSigningCertificate) + " is expired.");
401                     }
402                     
403                     OcspSigningCacheEntry ocspSigningCacheEntry = makeOcspSigningCacheEntry(ocspSigningCertificate, ocspKeyBinding);
404                     if (ocspSigningCacheEntry == null) {
405                         continue;
406                     } else {
407                         OcspSigningCache.INSTANCE.stagingAdd(ocspSigningCacheEntry);
408                     }
409                 }
410                 OcspSigningCache.INSTANCE.stagingCommit(ocspConfiguration.getOcspDefaultResponderReference());
411             } finally {
412                 OcspSigningCache.INSTANCE.stagingRelease();
413             }
414         } finally {
415             // Schedule a new timer of this type
416             addTimer(OcspConfiguration.getSigningCertsValidTimeInMilliseconds(), TIMERID_OCSPSIGNINGCACHE);
417         }
418     }

420     /**
421      * Constructs an OcspSigningCacheEntry from the given parameters.
422      * 
423      * @param ocspSigningCertificate The signing certificate associated with the key binding. May be found separately, so given as a separate parameter
424      * @param ocspKeyBinding the Key Binding to base the cache entry off of. 
425      * @return an OcspSigningCacheEntry, or null if any error was encountered.
426      */
427     private OcspSigningCacheEntry makeOcspSigningCacheEntry(X509Certificate ocspSigningCertificate, OcspKeyBinding ocspKeyBinding) {
428         final List<X509Certificate> caCertificateChain = getCaCertificateChain(ocspSigningCertificate);
429         if (caCertificateChain == null) {
430             log.warn("OcspKeyBinding " + ocspKeyBinding.getName() + " ( " + ocspKeyBinding.getId() + ") has a signing certificate, but no chain and will be ignored.");
431             return null;
432         }
433         final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(ocspKeyBinding.getCryptoTokenId());
434         if (cryptoToken == null) {
435             log.warn("Referenced CryptoToken with id " + ocspKeyBinding.getCryptoTokenId() + " does not exist. Ignoring OcspKeyBinding with id "
436                     + ocspKeyBinding.getId());
437             return null;
438         }
439         final PrivateKey privateKey;
440         try {
441             privateKey = cryptoToken.getPrivateKey(ocspKeyBinding.getKeyPairAlias());
442         } catch (CryptoTokenOfflineException e) {
443             log.warn("Referenced private key with alias " + ocspKeyBinding.getKeyPairAlias() + " could not be used. CryptoToken is off-line for OcspKeyBinding with id "+ocspKeyBinding.getId()+": " + e.getMessage());
444             return null;
445         }
446         if (privateKey == null) {
447             log.warn("Referenced private key with alias " + ocspKeyBinding.getKeyPairAlias() + " does not exist. Ignoring OcspKeyBinding with id "+ ocspKeyBinding.getId());
448             return null;
449         }
450         final String signatureProviderName = cryptoToken.getSignProviderName();
451         if (log.isDebugEnabled()) {
452             log.debug("Adding OcspKeyBinding "+ocspKeyBinding.getId()+", "+ocspKeyBinding.getName());
453         }
454         final CertificateStatus certificateStatus = getRevocationStatusWhenCasPrivateKeyIsCompromised(caCertificateChain.get(0), true);
455         OcspKeyBinding.ResponderIdType respIdType;
456         if (ResponderIdType.NAME.equals(ocspKeyBinding.getResponderIdType())) {
457             respIdType = OcspKeyBinding.ResponderIdType.NAME;
458         } else {
459             respIdType = OcspKeyBinding.ResponderIdType.KEYHASH;
460         }
461         return new OcspSigningCacheEntry(caCertificateChain.get(0), certificateStatus, caCertificateChain, ocspSigningCertificate, privateKey,
462                 signatureProviderName, ocspKeyBinding, respIdType);
463     }

465     /** 
466      * RFC 6960 Section 2.7 states that if it is known CA's private key has been compromised, it MAY return the "revoked"
467      * state for all certificates issued by that CA.
468      * 
469      * We interpret this as if the revocation reasons is one of "keyCompromise", "cACompromise" or "aACompromise" we know this.
470      * Additionally, if the "unspecified" reason is used we will consider this as a known private key compromise. (Safety first!)
471      * 
472      * @param caCertificate the X.509 CA certificate to check
473      * @param suppressInfo set to true to only do debug logging instead of info logging
474      * @return OK or the revocation status that we will use if the CA is revoked (same revocation date, but with reasonCode "cACompromise")
475      */
476     private CertificateStatus getRevocationStatusWhenCasPrivateKeyIsCompromised(final X509Certificate caCertificate, final boolean suppressInfo) {
477         final String issuerDn = CertTools.getIssuerDN(caCertificate);
478         final BigInteger serialNumber = CertTools.getSerialNumber(caCertificate);
479         final CertificateStatus certificateStatus = certificateStoreSession.getStatus(issuerDn, serialNumber);
480         if (certificateStatus.isRevoked()) {
481             final String subjectDn = CertTools.getSubjectDN(caCertificate);
482             if (certificateStatus.revocationReason == RevokedCertInfo.REVOCATION_REASON_UNSPECIFIED ||
483                     certificateStatus.revocationReason == RevokedCertInfo.REVOCATION_REASON_AACOMPROMISE ||
484                     certificateStatus.revocationReason == RevokedCertInfo.REVOCATION_REASON_CACOMPROMISE ||
485                     certificateStatus.revocationReason == RevokedCertInfo.REVOCATION_REASON_KEYCOMPROMISE) {
486                 final String msg = "CA certificate Subject DN '" + subjectDn + "', Issuer DN '" + issuerDn + "' and serial number " +
487                     serialNumber.toString() + " (0x" + serialNumber.toString(16) +
488                     ") is revoked with reason code " +certificateStatus.revocationReason + ". " +
489                     "The cACompromise revocation reason will be used for all certs issued by this CA.";
490                 if (suppressInfo) {
491                     log.debug(msg);
492                 } else {
493                     log.info(msg);
494                 }
495                 return new CertificateStatus(certificateStatus.toString(), certificateStatus.revocationDate.getTime(),
496                         RevokedCertInfo.REVOCATION_REASON_CACOMPROMISE, certificateStatus.certificateProfileId);
497             }
498             final String msg = "CA certificate Subject DN '" + subjectDn + "', Issuer DN '" + issuerDn + "' and serial number " +
499                 serialNumber.toString() + " (0x" + serialNumber.toString(16) +
500                 ") is revoked with reason code " +certificateStatus.revocationReason + ". " +
501                 "Status of individual leaf certificate will still be checked.";
502             if (suppressInfo) {
503                 log.debug(msg);
504             } else {
505                 log.info(msg);
506             }
507         }
508         return CertificateStatus.OK;
509     }

511     private List<X509Certificate> getCaCertificateChain(final X509Certificate leafCertificate) {
512         final List<X509Certificate> caCertificateChain = new ArrayList<X509Certificate>();
513         X509Certificate currentLevelCertificate = leafCertificate;
514         while (!CertTools.getIssuerDN(currentLevelCertificate).equals(CertTools.getSubjectDN(currentLevelCertificate))) {
515             final String issuerDn = CertTools.getIssuerDN(currentLevelCertificate);
516             currentLevelCertificate = certificateStoreSession.findLatestX509CertificateBySubject(issuerDn);
517             if (currentLevelCertificate == null) {
518                 log.warn("Unable to build certificate chain for OCSP signing certificate with Subject DN '" +
519                         CertTools.getSubjectDN(leafCertificate) + "'. CA with Subject DN '" + issuerDn + "' is missing in the database.");
520                 return null;
521             }
522             caCertificateChain.add(currentLevelCertificate);
523         }
524         try {
525             CertTools.verify(leafCertificate, caCertificateChain, new Date(), new EkuPKIXCertPathChecker(KeyPurposeId.id_kp_OCSPSigning.getId()));
526         } catch (CertPathValidatorException e) {
527             // Apparently the built chain could not be used to validate the leaf certificate
528             // this could happen if the CA keys were renewed, but the subject DN did not change
529             log.info("Unable to build a valid certificate chain for OCSP signing certificate with Subject DN '" +
530                     CertTools.getSubjectDN(leafCertificate)  + "' and Issuer DN " + CertTools.getIssuerDN(leafCertificate) +
531                     "' using the latest CA certificate(s) in the database. Trying to recover from exception: " + e.getMessage());
532             final CertificateInfo certificateInfo = certificateStoreSession.getCertificateInfo(CertTools.getFingerprintAsString(leafCertificate));
533             if(certificateInfo == null) {
534                 return null;
535             }
536             final List<Certificate> chainByFingerPrints = certificateStoreSession.getCertificateChain(certificateInfo);
537             if (chainByFingerPrints.size()>0) {
538                 // Remove the leaf certificate itself
539                 chainByFingerPrints.remove(0);
540             }
541             caCertificateChain.clear();
542             for (final Certificate current : chainByFingerPrints) {
543                 if (current instanceof X509Certificate) {
544                     caCertificateChain.add((X509Certificate) current);
545                 } else {
546                     log.warn("Unable to build certificate chain for OCSP signing certificate with Subject DN '" +
547                             CertTools.getSubjectDN(leafCertificate) + "' and Issuer DN '" + CertTools.getIssuerDN(leafCertificate) +
548                             "'. CA certificate chain contains non-X509 certificates.");
549                     return null;
550                 }
551             }
552             if (caCertificateChain.isEmpty()) {
553                 log.warn("Unable to build certificate chain for OCSP signing certificate with Subject DN '" +
554                         CertTools.getSubjectDN(leafCertificate) + "' and Issuer DN '" + CertTools.getIssuerDN(leafCertificate) +
555                         "''. CA certificate(s) are missing in the database.");
556                 return null;
557             }
558             try {
559                 CertTools.verify(leafCertificate, caCertificateChain, new Date(), new EkuPKIXCertPathChecker(KeyPurposeId.id_kp_OCSPSigning.getId()));
560             } catch (Exception e2) {
561                 log.warn("Unable to build certificate chain for OCSP signing certificate with Subject DN '" +
562                         CertTools.getSubjectDN(leafCertificate) + "' and Issuer DN '" + CertTools.getIssuerDN(leafCertificate) +
563                         "''. Found CA certificate(s) cannot be used for validation: " + e2.getMessage());
564                 return null;
565             }
566             log.info("Recovered and managed to build a valid certificate chain for OCSP signing certificate with Subject DN '" +
567                     CertTools.getSubjectDN(leafCertificate) + "' and Issuer DN '" + CertTools.getIssuerDN(leafCertificate) +
568                     "'.");
569         }
570         return caCertificateChain;
571     }

609     /**
610      * Select the preferred OCSP response sigAlg according to RFC6960 Section 4.4.7 in the following order:
611      * 
612      *    1. Select an algorithm specified as a preferred signature algorithm in the client request if it is 
613      *       an acceptable algorithm by EJBCA.
614      *    2. Select the signature algorithm used to sign a certificate revocation list (CRL) issued by the 
615      *       certificate issuer providing status information for the certificate specified by CertID.
616      *       (NOT APPLIED)
617      *    3. Select the signature algorithm used to sign the OCSPRequest if it is an acceptable algorithm in EJBCA.
618      *    4. Select a signature algorithm that has been advertised as being the default signature algorithm for 
619      *       the signing service using an out-of-band mechanism.
620      *    5. Select a mandatory or recommended signature algorithm specified for the version of OCSP in use, aka. 
621      *       specified in the properties file.
622      * 
623      *    The acceptable algorithm by EJBCA are the algorithms specified in ocsp.properties file in 'ocsp.signaturealgorithm'
624      * 
625      * @param req request
626      * @param ocspSigningCacheEntry Signing cache
627      * @param signerCert Certificate
628      * @return Algorithm
629      */
630     private String getSigAlg(OCSPReq req, final OcspSigningCacheEntry ocspSigningCacheEntry, final X509Certificate signerCert) {
631         String sigAlg = null;
632         PublicKey pk = signerCert.getPublicKey();
633         // Start with the preferred signature algorithm in the OCSP request
634         final Extension preferredSigAlgExtension = req.getExtension(new ASN1ObjectIdentifier(OCSPObjectIdentifiers.id_pkix_ocsp + ".8"));
635         if (preferredSigAlgExtension != null) {
636             final ASN1Sequence preferredSignatureAlgorithms = ASN1Sequence.getInstance(preferredSigAlgExtension.getParsedValue());
637             for (int i=0; i<preferredSignatureAlgorithms.size(); i++) {
638                 final ASN1Encodable asn1Encodable = preferredSignatureAlgorithms.getObjectAt(i);
639                 final ASN1ObjectIdentifier algorithmOid;
640                 if (asn1Encodable instanceof ASN1ObjectIdentifier) {
641                     // Handle client requests that were adapted to EJBCA 6.1.0's implementation
642                     log.info("OCSP request's PreferredSignatureAlgorithms did not contain an PreferredSignatureAlgorithm, but instead an algorithm OID."
643                             + " This will not be supported in a future versions of EJBCA.");
644                     algorithmOid = (ASN1ObjectIdentifier) asn1Encodable;
645                 } else {
646                     // Handle client requests that provide a proper AlgorithmIdentifier as specified in RFC 6960 + RFC 5280
647                     final ASN1Sequence preferredSignatureAlgorithm = ASN1Sequence.getInstance(asn1Encodable);
648                     final AlgorithmIdentifier algorithmIdentifier = AlgorithmIdentifier.getInstance(preferredSignatureAlgorithm.getObjectAt(0));
649                     algorithmOid = algorithmIdentifier.getAlgorithm();
650                 }
651                 if (algorithmOid != null) {
652                     sigAlg = AlgorithmTools.getAlgorithmNameFromOID(algorithmOid);
653                     if (sigAlg!=null && OcspConfiguration.isAcceptedSignatureAlgorithm(sigAlg) && AlgorithmTools.isCompatibleSigAlg(pk, sigAlg)) {
654                         if (log.isDebugEnabled()) {
655                             log.debug("Using OCSP response signature algorithm extracted from OCSP request extension. " + algorithmOid);
656                         }
657                         return sigAlg;
658                     }
659                 }
660             }
661         }
662         // the signature algorithm used to sign the OCSPRequest
663         if(req.getSignatureAlgOID() != null) {
664             sigAlg = AlgorithmTools.getAlgorithmNameFromOID(req.getSignatureAlgOID());
665             if(OcspConfiguration.isAcceptedSignatureAlgorithm(sigAlg) && AlgorithmTools.isCompatibleSigAlg(pk, sigAlg)) {
666                 if (log.isDebugEnabled()) {
667                     log.debug("OCSP response signature algorithm: the signature algorithm used to sign the OCSPRequest. " + sigAlg);
668                 }
669                 return sigAlg;
670             }
671         }
672         // The signature algorithm that has been advertised as being the default signature algorithm for the signing service using an
673         // out-of-band mechanism.
674         if (ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
675             // If we have an OcspKeyBinding we use this configuration to override the default
676             sigAlg = ocspSigningCacheEntry.getOcspKeyBinding().getSignatureAlgorithm();
677             if (log.isDebugEnabled()) {
678                 log.debug("OCSP response signature algorithm: the signature algorithm that has been advertised as being the default signature algorithm " +
679                         "for the signing service using an out-of-band mechanism. " + sigAlg);
680             }
681             return sigAlg;
682         }   
683         // The signature algorithm specified for the version of OCSP in use.
684         String sigAlgs = OcspConfiguration.getSignatureAlgorithm();
685         sigAlg = getSigningAlgFromAlgSelection(sigAlgs, pk);
686         if (log.isDebugEnabled()) {
687             log.debug("Using configured signature algorithm to sign OCSP response. " + sigAlg);
688         }
689         return sigAlg;
690     }

692     /**
693      * This method takes byte array and translates it onto a OCSPReq class.
694      * 
695      * @param request the byte array in question.
696      * @param remoteAddress The remote address of the HttpRequest associated with this array.
697      * @param transactionLogger A transaction logger.
698      * @return OCSP Request
699      * @throws MalformedRequestException if malformed
700      * @throws SignRequestException thrown if an unsigned request was processed when system configuration requires that all requests be signed.
701      * @throws CertificateException if cert is invalid
702      * @throws NoSuchAlgorithmException if algo not found
703      * @throws SignRequestSignatureException if sigs fails
704      */
705     private OCSPReq translateRequestFromByteArray(byte[] request, String remoteAddress, TransactionLogger transactionLogger)
706             throws MalformedRequestException, SignRequestException, SignRequestSignatureException, CertificateException, NoSuchAlgorithmException {
707         final OCSPReq ocspRequest;
708         try {
709             ocspRequest = new OCSPReq(request);
710         } catch (IOException e) {
711             throw new MalformedRequestException("Could not form OCSP request", e);
712         }
713         if (ocspRequest.getRequestorName() == null) {
714             if (log.isDebugEnabled()) {
715                 log.debug("Requestor name is null");
716             }
717         } else {
718             if (transactionLogger.isEnabled() || log.isDebugEnabled()) {
719                 final X500Name requestorDirectoryName = (X500Name) ocspRequest.getRequestorName().getName();
720                 final String requestor = CertTools.stringToBCDNString(requestorDirectoryName.toString());
721                 final String requestorRaw = GeneralName.directoryName + ": " + X500Name.getInstance(CeSecoreNameStyle.INSTANCE, requestorDirectoryName).toString();
722                 if (transactionLogger.isEnabled()) {
723                     transactionLogger.paramPut(TransactionLogger.REQ_NAME, requestor);
724                     transactionLogger.paramPut(TransactionLogger.REQ_NAME_RAW, requestorRaw);
725                 }
726                 if (log.isDebugEnabled()) {
727                     log.debug("Requestor name is: '" + requestor + "' Raw: '" + requestorRaw + "'");
728                 }
729             }
730         }
731         /**
732          * check the signature if contained in request. if the request does not contain a signature and the servlet is configured in the way the a
733          * signature is required we send back 'sigRequired' response.
734          */
735         if (log.isDebugEnabled()) {
736             log.debug("Incoming OCSP request is signed : " + ocspRequest.isSigned());
737         }
738         if (ocspRequest.isSigned()) {
739             final X509Certificate signercert = checkRequestSignature(remoteAddress, ocspRequest);
740             final String signercertIssuerName = CertTools.getIssuerDN(signercert);
741             final BigInteger signercertSerNo = CertTools.getSerialNumber(signercert);
742             final String signercertSubjectName = CertTools.getSubjectDN(signercert);
743             if (transactionLogger.isEnabled()) {
744                 transactionLogger.paramPut(TransactionLogger.SIGN_ISSUER_NAME_DN, signercertIssuerName);
745                 transactionLogger.paramPut(TransactionLogger.SIGN_SERIAL_NO, signercert.getSerialNumber().toByteArray());
746                 transactionLogger.paramPut(TransactionLogger.SIGN_SUBJECT_NAME, signercertSubjectName);
747                 transactionLogger.paramPut(PatternLogger.REPLY_TIME, TransactionLogger.REPLY_TIME);
748             }
749             // Check if we have configured request verification using the old property file way..
750             boolean enforceRequestSigning = OcspConfiguration.getEnforceRequestSigning();
751             // Next, check if there is an OcspKeyBinding where signing is required and configured for this request
752             // In the case where multiple requests are bundled together they all must be trusting the signer
753             for (final Req req : ocspRequest.getRequestList()) {
754                 OcspSigningCacheEntry ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getEntry(req.getCertID());
755                 if (ocspSigningCacheEntry==null) {
756                     if (log.isTraceEnabled()) {
757                         log.trace("Using default responder to check signature.");
758                     }
759                     ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getDefaultEntry();
760                 }   
761                 if (ocspSigningCacheEntry!=null && ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
762                     if (log.isTraceEnabled()) {
763                         log.trace("ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate: " + ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate());
764                     }
765                     final OcspKeyBinding ocspKeyBinding = ocspSigningCacheEntry.getOcspKeyBinding();
766                     if (log.isTraceEnabled()) {
767                         log.trace("OcspKeyBinding " + ocspKeyBinding.getId() + ", RequireTrustedSignature: " + ocspKeyBinding.getRequireTrustedSignature());
768                     }
769                     if (ocspKeyBinding.getRequireTrustedSignature()) {
770                         enforceRequestSigning = true;
771                         boolean isTrusted = false;
772                         final List<InternalKeyBindingTrustEntry> trustedCertificateReferences = ocspKeyBinding.getTrustedCertificateReferences();
773                         if (trustedCertificateReferences.isEmpty()) {
774                             // We trust ANY cert from a known CA
775                             isTrusted = true;
776                         } else {
777                             for (final InternalKeyBindingTrustEntry trustEntry : trustedCertificateReferences) {
778                                 final int trustedCaId = trustEntry.getCaId();
779                                 final BigInteger trustedSerialNumber = trustEntry.fetchCertificateSerialNumber();
780                                 if (log.isTraceEnabled()) {
781                                     log.trace("Processing trustedCaId="+trustedCaId + " trustedSerialNumber="+trustedSerialNumber + " signercertIssuerName.hashCode()="+
782                                             signercertIssuerName.hashCode()+" signercertSerNo="+signercertSerNo);
783                                 }
784                                 if (trustedCaId == signercertIssuerName.hashCode()) {
785                                     if (trustedSerialNumber == null) {
786                                         // We trust any certificate from this CA
787                                         isTrusted = true;
788                                         if (log.isTraceEnabled()) {
789                                             log.trace("Trusting request signature since ANY certificate from issuer "+trustedCaId+" is trusted.");
790                                         }
791                                         break;
792                                     } else if (signercertSerNo.equals(trustedSerialNumber)) {
793                                         // We trust this particular certificate from this CA
794                                         isTrusted = true;
795                                         if (log.isTraceEnabled()) {
796                                             log.trace("Trusting request signature since certificate with serialnumber " + trustedSerialNumber + " from issuer "+trustedCaId+" is trusted.");
797                                         }
798                                         break;
799                                     }
800                                 }
801                             }
802                         }
803                         if (!isTrusted) {
804                             final String infoMsg = intres.getLocalizedMessage("ocsp.infosigner.notallowed", signercertSubjectName, signercertIssuerName,
805                                     signercertSerNo.toString(16));
806                             log.info(infoMsg);
807                             throw new SignRequestSignatureException(infoMsg);
808                         }
809                     }
810                 }
811             }
812             if (enforceRequestSigning) {
813                 // If it verifies OK, check if it is revoked
814                 final String cacheLookupKey = OcspRequestSignerStatusCache.INSTANCE.createCacheLookupKey(signercertIssuerName, signercertSerNo);
815                 CertificateStatus status = OcspRequestSignerStatusCache.INSTANCE.getCachedCertificateStatus(cacheLookupKey);
816                 if (status==null) {
817                     status = certificateStoreSession.getStatus(signercertIssuerName, signercertSerNo);
818                     OcspRequestSignerStatusCache.INSTANCE.updateCachedCertificateStatus(cacheLookupKey, status);
819                 }
820                 /*
821                  * CertificateStatus.NOT_AVAILABLE means that the certificate does not exist in database. We treat this as ok, because it may be so that only revoked
822                  * certificates is in the (external) OCSP database.
823                  */
824                 if (status.equals(CertificateStatus.REVOKED)) {
825                     String serno = signercertSerNo.toString(16);
826                     String infoMsg = intres.getLocalizedMessage("ocsp.infosigner.revoked", signercertSubjectName, signercertIssuerName, serno);
827                     log.info(infoMsg);
828                     throw new SignRequestSignatureException(infoMsg);
829                 }
830             }
831         } else {
832             if (OcspConfiguration.getEnforceRequestSigning()) {
833                 // Signature required
834                 throw new SignRequestException("Signature required");
835             }
836             // Next, check if there is an OcspKeyBinding where signing is required and configured for this request
837             // In the case where multiple requests are bundled together they all must be trusting the signer
838             for (final Req req : ocspRequest.getRequestList()) {
839                 OcspSigningCacheEntry ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getEntry(req.getCertID());
840                 if (ocspSigningCacheEntry==null) {
841                     ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getDefaultEntry();
842                 }
843                 if (ocspSigningCacheEntry != null && ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
844                     final OcspKeyBinding ocspKeyBinding = ocspSigningCacheEntry.getOcspKeyBinding();
845                     if (ocspKeyBinding.getRequireTrustedSignature()) {
846                         throw new SignRequestException("Signature required");
847                     }
848                 }
849             }
850         }
851         return ocspRequest;
852     }

854     /**
855      * Checks the signature on an OCSP request. Does not check for revocation of the signer certificate
856      * 
857      * @param clientRemoteAddr The IP address or host name of the remote client that sent the request, can be null.
858      * @param req The signed OCSPReq
859      * @return X509Certificate which is the certificate that signed the OCSP request
860      * @throws SignRequestSignatureException if signature verification fail, or if the signing certificate is not authorized
861      * @throws SignRequestException if there is no signature on the OCSPReq
862      * @throws CertificateException if the certificate can not be parsed
863      * @throws NoSuchAlgorithmException if the certificate contains an unsupported algorithm
864      */
865     private X509Certificate checkRequestSignature(String clientRemoteAddr, OCSPReq req) throws SignRequestException, SignRequestSignatureException,
866             CertificateException, NoSuchAlgorithmException {
867         X509Certificate signercert = null;
868         // Get all certificates embedded in the request (probably a certificate chain)
869         try {
870             final X509CertificateHolder[] certs = req.getCerts();
871             String signerSubjectDn = null;
872             // We must find a certificate to verify the signature with...
873             boolean verifyOK = false;
874             for (int i=0; i<certs.length; i++) {
875                 final X509Certificate certificate = certificateConverter.getCertificate(certs[i]);
876                 try {
877                     if (req.isSignatureValid(CertTools.genContentVerifierProvider(certificate.getPublicKey()))) {
878                         signercert = certificate; // if the request signature verifies by this certificate, this is the signer cert 
879                         signerSubjectDn = CertTools.getSubjectDN(signercert);
880                         log.info(intres.getLocalizedMessage("ocsp.infosigner", signerSubjectDn));
881                         verifyOK = true;
882                         // Check that the signer certificate can be verified by one of the CA-certificates that we answer for
883                         final X509Certificate signerca = CaCertificateCache.INSTANCE.findLatestBySubjectDN(HashID.getFromIssuerDN(signercert));
884                         if (signerca != null) {
885                             try {
886                                 signercert.verify(signerca.getPublicKey());
887                                 final Date now = new Date();
888                                 if (log.isDebugEnabled()) {
889                                     log.debug("Checking validity. Now: " + now + ", signerNotAfter: " + signercert.getNotAfter());
890                                 }
891                                 try {
892                                     // Check validity of the request signing certificate
893                                     CertTools.checkValidity(signercert, now);
894                                 } catch (CertificateNotYetValidException e) {
895                                     log.info(intres.getLocalizedMessage("ocsp.infosigner.certnotyetvalid", signerSubjectDn, CertTools.getIssuerDN(signercert), e.getMessage()));
896                                     verifyOK = false;
897                                 } catch (CertificateExpiredException e) {
898                                     log.info(intres.getLocalizedMessage("ocsp.infosigner.certexpired", signerSubjectDn, CertTools.getIssuerDN(signercert), e.getMessage()));
899                                     verifyOK = false;
900                                 }
901                                 try {
902                                     // Check validity of the CA certificate
903                                     CertTools.checkValidity(signerca, now);
904                                 } catch (CertificateNotYetValidException e) {
905                                     log.info(intres.getLocalizedMessage("ocsp.infosigner.certnotyetvalid", CertTools.getSubjectDN(signerca), CertTools.getIssuerDN(signerca), e.getMessage()));
906                                     verifyOK = false;
907                                 } catch (CertificateExpiredException e) {
908                                     log.info(intres.getLocalizedMessage("ocsp.infosigner.certexpired", CertTools.getSubjectDN(signerca), CertTools.getIssuerDN(signerca), e.getMessage()));
909                                     verifyOK = false;
910                                 }
911                             } catch (SignatureException e) {
912                                 log.info(intres.getLocalizedMessage("ocsp.infosigner.invalidcertsignature", signerSubjectDn, CertTools.getIssuerDN(signercert), e.getMessage()));
913                                 verifyOK = false;
914                             } catch (InvalidKeyException e) {
915                                 log.info(intres.getLocalizedMessage("ocsp.infosigner.invalidcertsignature", signerSubjectDn, CertTools.getIssuerDN(signercert), e.getMessage()));
916                                 verifyOK = false;
917                             }
918                         } else {
919                             log.info(intres.getLocalizedMessage("ocsp.infosigner.nocacert", signerSubjectDn, CertTools.getIssuerDN(signercert)));
920                             verifyOK = false;
921                         }
922                         break;
923                     }
924                 } catch (OperatorCreationException e) {
925                     // Very fatal error
926                     throw new EJBException("Can not create Jca content signer: ", e);
927                 }
928             }
929             if (!verifyOK) {
930                 if (signerSubjectDn == null && certs.length > 0) {
931                     signerSubjectDn = CertTools.getSubjectDN(certificateConverter.getCertificate(certs[0]));
932                 }
933                 String errMsg = intres.getLocalizedMessage("ocsp.errorinvalidsignature", signerSubjectDn);
934                 log.info(errMsg);
935                 throw new SignRequestSignatureException(errMsg);
936             }
937         } catch (OCSPException e) {
938             throw new CryptoProviderException("BouncyCastle was not initialized properly.", e);
939         } catch (NoSuchProviderException e) {
940             throw new CryptoProviderException("BouncyCastle was not found as a provider.", e);
941         }
942         return signercert;
943     }

945     private void assertAcceptableResponseExtension(OCSPReq req) throws OcspFailureException {
946         if (null == req) {
947             throw new IllegalArgumentException();
948         }
949         if (req.hasExtensions()) {
950             final Extension acceptableResponsesExtension = req.getExtension(OCSPObjectIdentifiers.id_pkix_ocsp_response);
951             if (acceptableResponsesExtension != null) {
952                 // RFC 6960 4.4.3 AcceptableResponses ::= SEQUENCE OF OBJECT IDENTIFIER
953                 final ASN1Sequence sequence = ASN1Sequence.getInstance(acceptableResponsesExtension.getExtnValue().getOctets());
954                 @SuppressWarnings("unchecked")
955                 final Enumeration<ASN1ObjectIdentifier> oids = sequence.getObjects();
956                 boolean supportsResponseType = false;
957                 while (oids.hasMoreElements()) {
958                     final ASN1ObjectIdentifier oid = oids.nextElement();
959                     if (oid.equals(OCSPObjectIdentifiers.id_pkix_ocsp_basic)) {
960                         // This is the response type we support, so we are happy! Break the loop.

972                 }
973             }

1066     @Override
1067     public OcspResponseInformation getOcspResponse(final byte[] request, final X509Certificate[] requestCertificates, String remoteAddress,
1068             String xForwardedFor, StringBuffer requestUrl, final AuditLogger auditLogger, final TransactionLogger transactionLogger)
1069             throws MalformedRequestException, OCSPException {
1070         //Check parameters
1071         if (auditLogger == null) {
1072             throw new InvalidParameterException("Illegal to pass a null audit logger to OcspResponseSession.getOcspResponse");
1073         }
1074         if (transactionLogger == null) {
1075             throw new InvalidParameterException("Illegal to pass a null transaction logger to OcspResponseSession.getOcspResponse");
1076         }
1077         // Validate byte array.
1078         if (request.length > MAX_REQUEST_SIZE) {
1079             final String msg = intres.getLocalizedMessage("request.toolarge", MAX_REQUEST_SIZE, request.length);
1080             throw new MalformedRequestException(msg);
1081         }
1082         byte[] respBytes = null;
1083         final Date startTime = new Date();
1084         OCSPResp ocspResponse = null;
1085         // Start logging process time after we have received the request
1086         if (transactionLogger.isEnabled()) {
1087             transactionLogger.paramPut(PatternLogger.PROCESS_TIME, PatternLogger.PROCESS_TIME);
1088         }
1089         if (auditLogger.isEnabled()) {
1090             auditLogger.paramPut(PatternLogger.PROCESS_TIME, PatternLogger.PROCESS_TIME);
1091             auditLogger.paramPut(AuditLogger.OCSPREQUEST, new String(Hex.encode(request)));
1092         }
1093         OCSPReq req;
1094         long maxAge = OcspConfiguration.getMaxAge(CertificateProfileConstants.CERTPROFILE_NO_PROFILE);
1095         OCSPRespBuilder responseGenerator = new OCSPRespBuilder();
1096         X509Certificate signerCert = null;
1097         try {
1098             req = translateRequestFromByteArray(request, remoteAddress, transactionLogger);
1099             // Get the certificate status requests that are inside this OCSP req
1100             Req[] ocspRequests = req.getRequestList();
1101             if (ocspRequests.length <= 0) {
1102                 String infoMsg = intres.getLocalizedMessage("ocsp.errornoreqentities");
1103                 log.info(infoMsg);
1104                 throw new MalformedRequestException(infoMsg);
1105             }
1106             final int maxRequests = 100;
1107             if (ocspRequests.length > maxRequests) {
1108                 String infoMsg = intres.getLocalizedMessage("ocsp.errortoomanyreqentities", maxRequests);
1109                 log.info(infoMsg);
1110                 throw new MalformedRequestException(infoMsg);
1111             }
1112             if (log.isDebugEnabled()) {
1113                 log.debug("The OCSP request contains " + ocspRequests.length + " simpleRequests.");
1114             }
1115             if (transactionLogger.isEnabled()) {
1116                 transactionLogger.paramPut(TransactionLogger.NUM_CERT_ID, ocspRequests.length);
1117                 transactionLogger.paramPut(TransactionLogger.STATUS, OCSPRespBuilder.SUCCESSFUL);
1118             }
1119             if (auditLogger.isEnabled()) {
1120                 auditLogger.paramPut(AuditLogger.STATUS, OCSPRespBuilder.SUCCESSFUL);
1121             }
1122             OcspSigningCacheEntry ocspSigningCacheEntry = null;
1123             long nextUpdate = OcspConfiguration.getUntilNextUpdate(CertificateProfileConstants.CERTPROFILE_NO_PROFILE);
1124             Map<ASN1ObjectIdentifier, Extension> responseExtensions = new HashMap<>();
1125             
1126             // Look over the status requests
1127             List<OCSPResponseItem> responseList = new ArrayList<OCSPResponseItem>();
1128             boolean addExtendedRevokedExtension = false;
1129             Date producedAt = null;
1130             for (Req ocspRequest : ocspRequests) {
1131                 CertificateID certId = ocspRequest.getCertID();
1132                 ASN1ObjectIdentifier certIdhash = certId.getHashAlgOID();
1133                 if (!OIWObjectIdentifiers.idSHA1.equals(certIdhash) && !NISTObjectIdentifiers.id_sha256.equals(certIdhash)) {
1134                     throw new InvalidAlgorithmException("CertID with SHA1 and SHA256 are supported, not: "+certIdhash.getId());
1135                 }
1136                 if (transactionLogger.isEnabled()) {
1137                     transactionLogger.paramPut(TransactionLogger.SERIAL_NOHEX, certId.getSerialNumber().toByteArray());
1138                     transactionLogger.paramPut(TransactionLogger.DIGEST_ALGOR, certId.getHashAlgOID().toString());
1139                     transactionLogger.paramPut(TransactionLogger.ISSUER_NAME_HASH, certId.getIssuerNameHash());
1140                     transactionLogger.paramPut(TransactionLogger.ISSUER_KEY, certId.getIssuerKeyHash());
1141                 }
1142                 if (auditLogger.isEnabled()) {
1143                     auditLogger.paramPut(AuditLogger.ISSUER_KEY, certId.getIssuerKeyHash());
1144                     auditLogger.paramPut(AuditLogger.SERIAL_NOHEX, certId.getSerialNumber().toByteArray());
1145                     auditLogger.paramPut(AuditLogger.ISSUER_NAME_HASH, certId.getIssuerNameHash());
1146                 }
1147                 byte[] hashbytes = certId.getIssuerNameHash();
1148                 String hash = null;
1149                 if (hashbytes != null) {
1150                     hash = new String(Hex.encode(hashbytes));
1151                 }
1152                 if (xForwardedFor==null) {
1153                     log.info(intres.getLocalizedMessage("ocsp.inforeceivedrequest", certId.getSerialNumber().toString(16), hash, remoteAddress));
1154                 } else {
1155                     log.info(intres.getLocalizedMessage("ocsp.inforeceivedrequestwxff", certId.getSerialNumber().toString(16), hash, remoteAddress, xForwardedFor));
1156                 }
1157                 // Locate the CA which gave out the certificate
1158                 ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getEntry(certId);
1159                 if(ocspSigningCacheEntry == null) {
1160                   //Could it be that we haven't updated the OCSP Signing Cache?
1161                     ocspSigningCacheEntry = findAndAddMissingCacheEntry(certId);
1162                 }         
1163                 if (ocspSigningCacheEntry != null) {
1164                     if (transactionLogger.isEnabled()) {
1165                         // This will be the issuer DN of the signing certificate, whether an OCSP responder or an internal CA  
1166                         transactionLogger.paramPut(TransactionLogger.ISSUER_NAME_DN, ocspSigningCacheEntry.getSigningCertificateIssuerDn());
1167                         transactionLogger.paramPut(TransactionLogger.ISSUER_NAME_DN_RAW, ocspSigningCacheEntry.getSigningCertificateIssuerDnRaw());
1168                     }
1169                 } else {
1170                     /*
1171                      * if the certId was issued by an unknown CA 
1172                      * 
1173                      * The algorithm here: 
1174                      * We will sign the response with the CA that issued the last certificate(certId) in the request. If the issuing CA is not available on 
1175                      * this server, we sign the response with the default responderId (from params in web.xml). We have to look up the ca-certificate for 
1176                      * each certId in the request though, as we will check for revocation on the ca-cert as well when checking for revocation on the certId.
1177                      */                
1178                     // We could not find certificate for this request so get certificate for default responder
1179                     ocspSigningCacheEntry = OcspSigningCache.INSTANCE.getDefaultEntry();
1180                     if (ocspSigningCacheEntry != null) {
1181                         String errMsg = intres.getLocalizedMessage("ocsp.errorfindcacertusedefault",
1182                                 new String(Hex.encode(certId.getIssuerNameHash())));
1183                         log.info(errMsg);
1184                         // If we can not find the CA, answer UnknowStatus
1185                         responseList.add(new OCSPResponseItem(certId, new UnknownStatus(), nextUpdate));
1186                         if (transactionLogger.isEnabled()) {
1187                             transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_UNKNOWN);
1188                             transactionLogger.writeln();
1189                         }
1190                         continue;
1191                     } else {
1192                         GlobalOcspConfiguration ocspConfiguration = (GlobalOcspConfiguration) globalConfigurationSession
1193                                 .getCachedConfiguration(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID);
1194                         String defaultResponder = ocspConfiguration.getOcspDefaultResponderReference();
1195                         String errMsg = intres.getLocalizedMessage("ocsp.errorfindcacert", new String(Hex.encode(certId.getIssuerNameHash())),
1196                                 defaultResponder);
1197                         log.error(errMsg);
1198                         // If we are responding to multiple requests, the last found ocspSigningCacheEntry will be used in the end
1199                         // so even if there are not any one now, it might be later when it is time to sign the responses.
1200                         // Since we only will sign the entire response once if there is at least one valid ocspSigningCacheEntry
1201                         // we might as well include the unknown requests.
1202                         responseList.add(new OCSPResponseItem(certId, new UnknownStatus(), nextUpdate));
1203                         continue;
1204                     }
1205                 }
1206 
1207                 Collection<String> extensionOids = new ArrayList<>();
1208                 if (ocspSigningCacheEntry.getOcspKeyBinding() != null) {
1209                     extensionOids = ocspSigningCacheEntry.getOcspKeyBinding().getOcspExtensions();
1210                 }
1211                 
1212                 // Intended for debugging. Will usually be null
1213                 String alwaysUseOid = OcspConfiguration.getAlwaysSendCustomOCSPExtension();
1214                 if (alwaysUseOid != null && !extensionOids.contains(alwaysUseOid)) {
1215                     extensionOids.add(alwaysUseOid);
1216                 }
1217                 
1218                 final org.bouncycastle.cert.ocsp.CertificateStatus certStatus;
1219                 // Check if the cacert (or the default responderid) is revoked
1220                 X509Certificate caCertificate = ocspSigningCacheEntry.getIssuerCaCertificate();
1221                 final CertificateStatus signerIssuerCertStatus = ocspSigningCacheEntry.getIssuerCaCertificateStatus();
1222                 final String caCertificateSubjectDn = CertTools.getSubjectDN(caCertificate);
1223                 CertificateStatusHolder certificateStatusHolder = null;
1224                 OCSPResponseItem respItem;
1225                 if (signerIssuerCertStatus.equals(CertificateStatus.REVOKED)) {
1226                     /*
1227                      * According to chapter 2.7 in RFC2560:
1228                      * 
1229                      * 2.7 CA Key Compromise If an OCSP responder knows that a particular CA's private key has been compromised, it MAY return the revoked
1230                      * state for all certificates issued by that CA.
1231                      */
1232                     // If we've ended up here it's because the signer issuer certificate was revoked. 
1233                     certStatus = new RevokedStatus(new RevokedInfo(new ASN1GeneralizedTime(signerIssuerCertStatus.revocationDate),
1234                             CRLReason.lookup(signerIssuerCertStatus.revocationReason)));
1235                     log.info(intres.getLocalizedMessage("ocsp.signcertissuerrevoked", CertTools.getSerialNumberAsString(caCertificate),
1236                             CertTools.getSubjectDN(caCertificate)));
1237                     respItem = new OCSPResponseItem(certId, certStatus, nextUpdate);
1238                     if (transactionLogger.isEnabled()) {
1239                         transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_REVOKED);
1240                         transactionLogger.paramPut(TransactionLogger.REV_REASON, signerIssuerCertStatus.revocationReason);
1241                         transactionLogger.writeln();
1242                     }
1243                 } else {
1244                     /**
1245                      * Here is the actual check for the status of the sought certificate (easy to miss). Here we grab just the status if there aren't
1246                      * any OIDs defined (default case), but if there are we'll probably need the certificate as well. If that's the case, we'll grab
1247                      * the certificate in the same transaction.
1248                      */
1249                     final CertificateStatus status;
1250                     if (extensionOids.isEmpty()) {
1251                         status = certificateStoreSession.getStatus(caCertificateSubjectDn, certId.getSerialNumber());
1252                     } else {
1253                         certificateStatusHolder = certificateStoreSession.getCertificateAndStatus(caCertificateSubjectDn, certId.getSerialNumber());
1254                         status = certificateStatusHolder.getCertificateStatus();
1255                     }
1256                     if (transactionLogger.isEnabled()) {
1257                         transactionLogger.paramPut(TransactionLogger.CERT_PROFILE_ID, String.valueOf(status.certificateProfileId));
1258                     }
1259                     // If we have an OcspKeyBinding configured for this request, we override the default value
1260                     if (ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
1261                         nextUpdate = ocspSigningCacheEntry.getOcspKeyBinding().getUntilNextUpdate()*1000L;
1262                     }
1263                     // If we have an explicit value configured for this certificate profile, we override the the current value with this value
1264                     if (status.certificateProfileId != CertificateProfileConstants.CERTPROFILE_NO_PROFILE &&
1265                             OcspConfiguration.isUntilNextUpdateConfigured(status.certificateProfileId)) {
1266                         nextUpdate = OcspConfiguration.getUntilNextUpdate(status.certificateProfileId);
1267                     }
1268                     // If we have an OcspKeyBinding configured for this request, we override the default value
1269                     if (ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
1270                         maxAge = ocspSigningCacheEntry.getOcspKeyBinding().getMaxAge()*1000L;
1271                     }
1272                     // If we have an explicit value configured for this certificate profile, we override the the current value with this value
1273                     if (status.certificateProfileId != CertificateProfileConstants.CERTPROFILE_NO_PROFILE &&
1274                             OcspConfiguration.isMaxAgeConfigured(status.certificateProfileId)) {
1275                         maxAge = OcspConfiguration.getMaxAge(status.certificateProfileId);
1276                     }
1277 
1278                     final String sStatus;
1279                     boolean addArchiveCutoff = false;
1280                     if (status.equals(CertificateStatus.NOT_AVAILABLE)) {
1281                         // No revocation info available for this cert, handle it
1282                         if (log.isDebugEnabled()) {
1283                             log.debug("Unable to find revocation information for certificate with serial '" + certId.getSerialNumber().toString(16)
1284                                     + "'" + " from issuer '" + caCertificateSubjectDn + "'");
1285                         }
1286                         /* 
1287                          * If we do not treat non existing certificates as good or revoked
1288                          * OR
1289                          * we don't actually handle requests for the CA issuing the certificate asked about
1290                          * then we return unknown 
1291                          * */
1292                         if (OcspConfigurationCache.INSTANCE.isNonExistingGood(requestUrl, ocspSigningCacheEntry.getOcspKeyBinding()) &&
1293                                 OcspSigningCache.INSTANCE.getEntry(certId) != null) {
1294                             sStatus = "good";
1295                             certStatus = null; // null means "good" in OCSP
1296                             if (transactionLogger.isEnabled()) {
1297                                 transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_GOOD);
1298                                 transactionLogger.paramPut(TransactionLogger.REV_REASON, CRLReason.certificateHold);
1299                             }
1300                         } else if (OcspConfigurationCache.INSTANCE.isNonExistingRevoked(requestUrl, ocspSigningCacheEntry.getOcspKeyBinding()) &&
1301                                 OcspSigningCache.INSTANCE.getEntry(certId) != null) {
1302                             sStatus = "revoked";
1303                             certStatus = new RevokedStatus(new RevokedInfo(new ASN1GeneralizedTime(new Date(0)),
1304                                     CRLReason.lookup(CRLReason.certificateHold)));
1305                             if (transactionLogger.isEnabled()) {
1306                                 transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_REVOKED); 
1307                                 transactionLogger.paramPut(TransactionLogger.REV_REASON, CRLReason.certificateHold);
1308                             }
1309                             addExtendedRevokedExtension = true;
1310                         } else if (OcspConfigurationCache.INSTANCE.isNonExistingUnauthorized(ocspSigningCacheEntry.getOcspKeyBinding())
1311                                 && OcspSigningCache.INSTANCE.getEntry(certId) != null) {
1312                             // In order to save on cycles and mitigate the chances of a DOS attack, we'll return a unsigned unauthorized reply. 
1313                             ocspResponse = responseGenerator.build(OCSPRespBuilder.UNAUTHORIZED, null);
1314                             if (auditLogger.isEnabled()) {
1315                                 auditLogger.paramPut(AuditLogger.STATUS, OCSPRespBuilder.UNAUTHORIZED);
1316                             }
1317                             if (transactionLogger.isEnabled()) {
1318                                 transactionLogger.paramPut(TransactionLogger.STATUS, OCSPRespBuilder.UNAUTHORIZED);
1319                             }
1320                             log.info(intres.getLocalizedMessage("ocsp.errorfindcert", certId.getSerialNumber().toString(16), caCertificateSubjectDn));
1321                             //Return early here
1322                             return new OcspResponseInformation(ocspResponse, maxAge, null);
1323                         } else {
1324                             sStatus = "unknown";
1325                             certStatus = new UnknownStatus();
1326                             if (transactionLogger.isEnabled()) {
1327                                 transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_UNKNOWN);
1328                                 transactionLogger.paramPut(TransactionLogger.REV_REASON, CRLReason.certificateHold);
1329                             }
1330                         }
1331                     } else if (status.equals(CertificateStatus.REVOKED)) {
1332                         // Revocation info available for this cert, handle it
1333                         sStatus = "revoked";
1334                         certStatus = new RevokedStatus(new RevokedInfo(new ASN1GeneralizedTime(status.revocationDate),
1335                                 CRLReason.lookup(status.revocationReason)));
1336                         if (transactionLogger.isEnabled()) {
1337                             transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_REVOKED);
1338                             transactionLogger.paramPut(TransactionLogger.REV_REASON, status.revocationReason);
1339                         }
1340                         // If we have an explicit value configured for this certificate profile, we override the the current value with this value
1341                         if (status.certificateProfileId != CertificateProfileConstants.CERTPROFILE_NO_PROFILE &&
1342                                 OcspConfiguration.isRevokedUntilNextUpdateConfigured(status.certificateProfileId)) {
1343                             nextUpdate = OcspConfiguration.getRevokedUntilNextUpdate(status.certificateProfileId);
1344                         }
1345                         // If we have an explicit value configured for this certificate profile, we override the the current value with this value
1346                         if (status.certificateProfileId != CertificateProfileConstants.CERTPROFILE_NO_PROFILE &&
1347                                 OcspConfiguration.isRevokedMaxAgeConfigured(status.certificateProfileId)) {
1348                             maxAge = OcspConfiguration.getRevokedMaxAge(status.certificateProfileId);
1349                         }
1350                     } else {
1351                         sStatus = "good";
1352                         certStatus = null;
1353                         if (transactionLogger.isEnabled()) {
1354                             transactionLogger.paramPut(TransactionLogger.CERT_STATUS, OCSPResponseItem.OCSP_GOOD);
1355                         }
1356                         addArchiveCutoff = checkAddArchiveCuttoff(caCertificateSubjectDn, certId);
1357                     }
1358                     if (log.isDebugEnabled()) {
1359                         log.debug("Set nextUpdate=" + nextUpdate + ", and maxAge=" + maxAge + " for certificateProfileId="
1360                                 + status.certificateProfileId);
1361                     }
1362                     log.info(intres.getLocalizedMessage("ocsp.infoaddedstatusinfo", sStatus, certId.getSerialNumber().toString(16), caCertificateSubjectDn));
1363                     respItem = new OCSPResponseItem(certId, certStatus, nextUpdate);
1364                     if (addArchiveCutoff) {
1365                         addArchiveCutoff(respItem);
1366                         producedAt = new Date();
1367                     }
1368                     if (transactionLogger.isEnabled()) {
1369                         transactionLogger.writeln();
1370                     }
1371                 }
1372  
1373                 for (String oidstr : extensionOids) {
1374                     boolean useAlways = false;
1375                     if (oidstr.equals(alwaysUseOid)) {
1376                         useAlways = true;
1377                     }
1378                     ASN1ObjectIdentifier oid = new ASN1ObjectIdentifier(oidstr);
1379                     Extension extension = null;
1380                     if (!useAlways) {
1381                         // Only check if extension exists if we are not already bound to use it
1382                         if (req.hasExtensions()) {
1383                             extension = req.getExtension(oid);
1384                         }
1385                     }
1386                     //If found, or if it should be used anyway
1387                     if (useAlways || extension!=null) {
1388                         // We found an extension, call the extension class
1389                         if (log.isDebugEnabled()) {
1390                             log.debug("Found OCSP extension oid: " + oidstr);
1391                         }
1392                         OCSPExtension extObj = OcspExtensionsCache.INSTANCE.getExtensions().get(oidstr);
1393                         if (extObj != null) {
1394                             // Find the certificate from the certId
1395                             if(certificateStatusHolder != null && certificateStatusHolder.getCertificate() != null) {
1396                                 X509Certificate cert = (X509Certificate) certificateStatusHolder.getCertificate();
1397                                 // From EJBCA 6.2.10 and 6.3.2 the extension must perform the reverse DNS lookup by itself if needed.
1398                                 final String remoteHost = remoteAddress;
1399                                 // Call the OCSP extension
1400                                 Map<ASN1ObjectIdentifier, Extension> retext = null;
1401                                     retext = extObj.process(requestCertificates, remoteAddress, remoteHost, cert, certStatus, ocspSigningCacheEntry.getOcspKeyBinding());
1402                                 if (retext != null) {
1403                                     // Add the returned X509Extensions to the responseExtension we will add to the basic OCSP response
1404                                     if (extObj.getExtensionType().contains(OCSPExtensionType.RESPONSE)) {
1405                                         responseExtensions.putAll(retext);
1406                                     }
1407                                     if (extObj.getExtensionType().contains(OCSPExtensionType.SINGLE_RESPONSE)) {
1408                                         respItem.addExtensions(retext);
1409                                     }
1410                                 } else {
1411                                         log.error(intres.getLocalizedMessage("ocsp.errorprocessextension", extObj.getClass().getName(),
1412                                                 Integer.valueOf(extObj.getLastErrorCode())));
1413                                 }
1414                             }
1415                         }
1416                     }
1417                 }
1418                 responseList.add(respItem);
1419             }
1420             if (addExtendedRevokedExtension) { 
1421                 // id-pkix-ocsp-extended-revoke OBJECT IDENTIFIER ::= {id-pkix-ocsp 9}
1422                 final ASN1ObjectIdentifier extendedRevokedOID = OCSPObjectIdentifiers.id_pkix_ocsp_extended_revoke;
1423                 try {
1424                     responseExtensions.put(extendedRevokedOID, new Extension(extendedRevokedOID, false, DERNull.INSTANCE.getEncoded() ));
1425                 } catch (IOException e) {
1426                     throw new IllegalStateException("Could not get encoding from DERNull.", e);
1427                 }
1428             }
1429             if (ocspSigningCacheEntry != null) {
1430                 // Add standard response extensions
1431                 responseExtensions.putAll(getStandardResponseExtensions(req, ocspSigningCacheEntry));
1432                 
1433                 // Add responseExtensions
1434                 Extensions exts = new Extensions(responseExtensions.values().toArray(new Extension[0]));
1435                 // generate the signed response object
1436                 BasicOCSPResp basicresp = signOcspResponse(req, responseList, exts, ocspSigningCacheEntry, producedAt);

1587      * returns a Map of responseExtensions to be added to the BacisOCSPResponseGenerator with <code>
1588      * X509Extensions exts = new X509Extensions(table);
1589      * basicRes.setResponseExtensions(responseExtensions);
1590      * </code>
1591      * 
1592      * @param req the OCSP request
1593      * @param ocspSigningCacheEntry the OCSP signing cache entry used 
1594      * @return a HashMap, can be empty but not null
1595      * @throws IllegalNonceException if Nonce is larger than 32 bytes
1596      */
1597     private Map<ASN1ObjectIdentifier, Extension> getStandardResponseExtensions(final OCSPReq req, final OcspSigningCacheEntry ocspSigningCacheEntry)
1598             throws IllegalNonceException {
1599         HashMap<ASN1ObjectIdentifier, Extension> result = new HashMap<>();
1600         if (req.hasExtensions()) {
1601             // Table of extensions to include in the response
1602             // OCSP Nonce, if included in the request, the response must include the same according to RFC6960
1603             Extension ext = req.getExtension(OCSPObjectIdentifiers.id_pkix_ocsp_nonce);
1604             //Check the keybinding firsthand if nonce's are enabled, if there is no keybinding (because a CA is replying), check the global value. 
1605             boolean nonceEnable = (ocspSigningCacheEntry.getOcspKeyBinding() != null ? ocspSigningCacheEntry.getOcspKeyBinding().isNonceEnabled() :
1606                 ((GlobalOcspConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID)).getNonceEnabled());
1607             if (null != ext && nonceEnable) {
1608                 ASN1OctetString noncestr = ext.getExtnValue();
1609                 // Limit Nonce to 32 bytes to avoid chosen-prefix attack on hash collisions.
1610                 // See https://groups.google.com/forum/#!topic/mozilla.dev.security.policy/x3TOIJL7MGw
1611                 if ( (noncestr != null) && (noncestr.getOctets() != null) && (noncestr.getOctets().length > 32) ) {
1612                     log.info("Received OCSP request with Nonce larger than 32 bytes, rejecting.");
1613                     throw new IllegalNonceException("Nonce too large");
1614                 }
1615                 result.put(OCSPObjectIdentifiers.id_pkix_ocsp_nonce, ext);
1616             }
1617         }
1618         return result;

1669     private BasicOCSPResp signOcspResponse(OCSPReq req, List<OCSPResponseItem> responseList, Extensions exts, 
1670             final OcspSigningCacheEntry ocspSigningCacheEntry, Date producedAt) throws CryptoTokenOfflineException {
1671         assertAcceptableResponseExtension(req);
1672         if (!ocspSigningCacheEntry.isSigningCertificateForOcspSigning()) {
1673             log.warn("Signing with non OCSP certificate (no 'OCSP Signing' Extended Key Usage) bound by OcspKeyBinding '" + ocspSigningCacheEntry.getOcspKeyBinding().getName() + "'.");
1674         }
1675         final X509Certificate signerCert = ocspSigningCacheEntry.getSigningCertificate();
1676         final String sigAlg = getSigAlg(req, ocspSigningCacheEntry, signerCert);
1677         if (log.isDebugEnabled()) {
1678             log.debug("Signing algorithm: " + sigAlg);
1679         }
1680         try {
1681             // Now we can use the returned OCSPServiceResponse to get private key and certificate chain to sign the ocsp response
1682             final BasicOCSPResp ocspresp = generateBasicOcspResp(exts, responseList, sigAlg, signerCert, ocspSigningCacheEntry, producedAt);
1683             if (CertTools.isCertificateValid(signerCert, false)) { // Don't warn about signer validity for each OCSP response...
1684                 return ocspresp;
1685             } else {
1686                 throw new OcspFailureException("Response was not validly signed.");
1687             }
1688         } catch (OCSPException ocspe) {
1689             throw new OcspFailureException(ocspe);
1690         } catch (NoSuchProviderException nspe) {
1691             throw new OcspFailureException(nspe);
1692         } catch (IllegalArgumentException e) {
1693             log.error("IllegalArgumentException: ", e);
1694             throw new OcspFailureException(e);

1698     private BasicOCSPResp generateBasicOcspResp(Extensions exts, List<OCSPResponseItem> responses, String sigAlg,
1699                         X509Certificate signerCert, OcspSigningCacheEntry ocspSigningCacheEntry, Date producedAt)
1700                                 throws OCSPException, NoSuchProviderException, CryptoTokenOfflineException {
1701         final PrivateKey signerKey = ocspSigningCacheEntry.getPrivateKey();
1702         final String provider = ocspSigningCacheEntry.getSignatureProviderName();
1703         BasicOCSPResp returnval = null;
1704         BasicOCSPRespBuilder basicRes = new BasicOCSPRespBuilder(ocspSigningCacheEntry.getRespId());
1705         if (responses != null) {
1706             for (OCSPResponseItem item : responses) {
1707                 Date nextUpdate = item.getNextUpdate();
1708                 // Adjust nextUpdate so that it can never exceed the OCSP responder signing certificate validity
1709                 if (signerCert != null && nextUpdate != null && signerCert.getNotAfter().before(nextUpdate)) {
1710                     nextUpdate = signerCert.getNotAfter();
1711                 }
1712                 basicRes.addResponse(item.getCertID(), item.getCertStatus(), item.getThisUpdate(), nextUpdate, item.buildExtensions());
1713             }
1714         }
1715         if (exts != null) {
1716             @SuppressWarnings("rawtypes")
1717             Enumeration oids = exts.oids();
1718             if (oids.hasMoreElements()) {
1719                 basicRes.setResponseExtensions(exts);
1720             }
1721         }
1722         final X509Certificate[] chain = ocspSigningCacheEntry.getResponseCertChain();
1723         if (log.isDebugEnabled()) {
1724             log.debug("The response certificate chain contains " + chain.length + " certificates");
1725         }
1726         /*
1727          * The below code breaks the EJB standard by creating its own thread pool and creating a single thread (of the HsmResponseThread 
1728          * type). The reason for this is that the HSM may deadlock when requesting an OCSP response, which we need to guard against. Since 
1729          * there is no way of performing this action within the EJB3.0 standard, we are consciously creating threads here. 
1730          * 
1731          * Note that this does in no way break the spirit of the EJB standard, which is to not interrupt EJB's transaction handling by 
1732          * competing with its own thread pool, since these operations have no database impact.
1733          */
1734         final Future<BasicOCSPResp> task = service.submit(new HsmResponseThread(basicRes, sigAlg, signerKey, chain, provider, producedAt));
1735         try {
1736             returnval = task.get(HsmResponseThread.HSM_TIMEOUT_SECONDS, TimeUnit.SECONDS);
1737         } catch (InterruptedException e) {
1738             task.cancel(true);
1739             throw new Error("OCSP response retrieval was interrupted while running. This should not happen", e);
1740         } catch (ExecutionException e) {
1741             task.cancel(true);
1742             throw new OcspFailureException("Failure encountered while retrieving OCSP response.", e);
1743         } catch (TimeoutException e) {
1744             task.cancel(true);
1745             throw new CryptoTokenOfflineException("HSM timed out while trying to get OCSP response", e);
1746         }
1747         if (log.isDebugEnabled()) {
1748             log.debug("Signing OCSP response with OCSP signer cert: " + signerCert.getSubjectDN().getName());
1749         }
1750         if (!returnval.getResponderId().equals(ocspSigningCacheEntry.getRespId())) {
1751             log.error("Response responderId does not match signer certificate responderId!");
1752             throw new OcspFailureException("Response responderId does not match signer certificate responderId!");
1753         }
1754         if (!ocspSigningCacheEntry.checkResponseSignatureVerified()) {
1755             // We only check the response signature the first time for each OcspSigningCacheEntry to detect a misbehaving HSM.
1756             // The client is still responsible for validating the signature, see RFC 6960 Section 3.2.2
1757             boolean verify;
1758             try {
1759                 verify = returnval.isSignatureValid(CertTools.genContentVerifierProvider(signerCert.getPublicKey()));
1760             } catch (OperatorCreationException e) {
1761                 // Very fatal error
1762                 throw new EJBException("Can not create Jca content signer: ", e);
1763             }
1764             if (verify) {
1765                 if (log.isDebugEnabled()) {
1766                     log.debug("The OCSP response is verifying.");
1767                 }
1768             } else {
1769                 log.error("The response is NOT verifying! Attempted to sign using " + CertTools.getSubjectDN(signerCert) + " but signature was not valid.");
1770                 throw new OcspFailureException("Attempted to sign using " + CertTools.getSubjectDN(signerCert) + " but signature was not valid.");
1771             }
1772         }
1773         return returnval;
1774     }

1968             }

1998         } catch (Exception e) {

2000         }

2004      * @param keyStore key store
2005      * @param password  password
2006      * @return new key store
2007      * @throws KeyStoreException on key store error 
2008      * @throws NoSuchAlgorithmException if algo not found
2009      * @throws UnrecoverableEntryException if key store is corrupt
2010      * @throws NoSuchProviderException if algo not found
2011      * @throws CertificateException if cert is corrupt
2012      * @throws IOException On IO fail */
2013     @Deprecated  //Remove this method as soon as upgrading from 5->6 is dropped
2014     private KeyStore makeKeysOnlyP12(KeyStore keyStore, char[] password) throws KeyStoreException, NoSuchAlgorithmException, UnrecoverableEntryException, NoSuchProviderException, CertificateException, IOException {
2015         final KeyStore p12 = KeyStore.getInstance("PKCS12", "BC");
2016         final KeyStore.ProtectionParameter protParam =
2017             (password != null ? new KeyStore.PasswordProtection(password) : null);
2018         p12.load(null, password); // initialize
2019         
2020         final Enumeration<String> en = keyStore.aliases();
2021         while (en.hasMoreElements()) {
2022             final String alias = en.nextElement();
2023             if (!keyStore.isKeyEntry(alias)) continue;
2024             try {
2025                 KeyStore.PrivateKeyEntry entry = (KeyStore.PrivateKeyEntry)keyStore.getEntry(alias, protParam);
2026                 Certificate[] chain = new Certificate[] { entry.getCertificate() };
2027                 p12.setKeyEntry(alias, entry.getPrivateKey(), password, chain);
2028             } catch (UnsupportedOperationException uoe) {
2029                 KeyStore.PrivateKeyEntry entry = (KeyStore.PrivateKeyEntry)keyStore.getEntry(alias, null);
2030                 Certificate[] chain = new Certificate[] { entry.getCertificate() };
2031                 p12.setKeyEntry(alias, entry.getPrivateKey(), null, chain);
2032             }
2033         }
2034         return p12;
2035     }
