14  package org.cesecore.certificates.ocsp;
15  
16  import java.io.Serializable;
17  import java.util.Collection;
18  import java.util.Date;
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  import org.bouncycastle.asn1.ASN1ObjectIdentifier;
23  import org.bouncycastle.asn1.x509.Extension;
24  import org.bouncycastle.asn1.x509.Extensions;
25  import org.bouncycastle.cert.ocsp.CertificateID;
26  import org.bouncycastle.cert.ocsp.CertificateStatus;
27  
28  /**
29   * Class used to encapsulate the data that goes into a OCSP response
30   * 
31   * @version $Id: OCSPResponseItem.java 28881 2018-05-08 18:18:11Z anatom $
32   */
33  public class OCSPResponseItem implements Serializable {
34  
35      /** Constants capturing the OCSP response status. 
36       * These are the return codes defined in the RFC. 
37       * The codes are just used for simple access to the OCSP return value. 
38       */
39      public static final int OCSP_GOOD = 0;
40      public static final int OCSP_REVOKED = 1;
41      public static final int OCSP_UNKNOWN = 2;
42  
43      
44      private static final long serialVersionUID = 8520379864183774863L;
45      
46      private CertificateID certID;
47      private CertificateStatus certStatus;
48      /* RFC 2560 2.4: The time at which the status being indicated is known to be correct. */
49      private Date thisUpdate;
50      /*
51       * RFC 2560 2.4: The time at or before which newer information will be available about the status of the certificate. If nextUpdate is not set,
52       * the responder is indicating that newer revocation information is available all the time.
53       */
54      private Date nextUpdate = null;
55      
56      private Map<ASN1ObjectIdentifier, Extension> singleExtensions = new HashMap<ASN1ObjectIdentifier, Extension>();
57      
58      public OCSPResponseItem(CertificateID certID, CertificateStatus certStatus, long untilNextUpdate) {
59          this.certID = certID;
60          this.certStatus = certStatus;
61          this.thisUpdate = new Date();
62          if (untilNextUpdate > 0) {
63              this.nextUpdate = new Date(this.thisUpdate.getTime() + untilNextUpdate);
64          }
65      }
66  
67      public CertificateID getCertID() {
68          return certID;
69      }
70  
71      public CertificateStatus getCertStatus() {
72          return certStatus;
73      }
74  
75      public Date getThisUpdate() {
76          return thisUpdate;
77      }
78  
79      public Date getNextUpdate() {
80          return nextUpdate;
81      }
82      
83      public void addExtensions(Map<ASN1ObjectIdentifier, Extension> extensions) {
84          singleExtensions.putAll(extensions);
85      }
86  
87      public Extensions buildExtensions() {
88          Collection<Extension> extensionValues = singleExtensions.values();
89          Extension[] extensions = extensionValues.toArray(new Extension[singleExtensions.size()]);
90          return new Extensions(extensions);
91      }
92  }
