68    @Override
69    public void storeCRL(
70        final AuthenticationToken admin,
71        final byte[] incrl,
72        final String cafp,
73        final int number,
74        final String issuerDN,
75        final Date thisUpdate,
76        final Date nextUpdate,
77        final int deltaCRLIndicator)
78        throws CrlStoreException, AuthorizationDeniedException {
79      if (LOG.isTraceEnabled()) {
80        LOG.trace(">storeCRL(" + cafp + ", " + number + ")");
81      }
82      // Check that user is authorized to the CA that issued this CRL
83      String bcdn = CertTools.stringToBCDNString(issuerDN);
84      int caid = bcdn.hashCode();
85      authorizedToCA(admin, caid);
86  
87      try {
88        boolean deltaCRL = deltaCRLIndicator > 0;
89        int lastNo = getLastCRLNumber(issuerDN, deltaCRL);
90        if (number <= lastNo) {
91          // There is already a CRL with this number, or a later one stored. Don't
92          // create duplicates
93          final String msg =
94              INTRES.getLocalizedMessage(
95                  "store.errorstorecrlwrongnumber",
96                  Integer.valueOf(number),
97                  Integer.valueOf(lastNo),
98                  issuerDN);
99          throw new CrlStoreException(msg);
100       }
101       CRLData data =
102           new CRLData(
103               incrl,
104               number,
105               issuerDN,
106               thisUpdate,
107               nextUpdate,
108               cafp,
109               deltaCRLIndicator);
110       this.entityManager.persist(data);
111       String msg =
112           INTRES.getLocalizedMessage(
113               "store.storecrl",
114               Integer.valueOf(number),
115               data.getFingerprint(),
116               data.getIssuerDN());
117       Map<String, Object> details = new LinkedHashMap<String, Object>();
118       details.put("msg", msg);
119       logSession.log(
120           EventTypes.CRL_STORED,
121           EventStatus.SUCCESS,
122           ModuleTypes.CRL,
123           ServiceTypes.CORE,
124           admin.toString(),
125           String.valueOf(caid),
126           null,
127           null,
128           details);
129     } catch (Exception e) {
130       String msg =
131           INTRES.getLocalizedMessage(
132               "store.errorstorecrl", Integer.valueOf(number), issuerDN);
133       LOG.error(msg, e);
134       throw new CrlStoreException(e); // will rollback etc
135     }
136     if (LOG.isTraceEnabled()) {
137       LOG.trace("<storeCRL()");
138     }
139   }

141   @Override
142   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
143   public byte[] getLastCRL(final String issuerdn, final boolean deltaCRL) {
144     if (LOG.isTraceEnabled()) {
145       LOG.trace(">getLastCRL(" + issuerdn + ", " + deltaCRL + ")");
146     }
147     int maxnumber = 0;
148     try {
149       maxnumber = getLastCRLNumber(issuerdn, deltaCRL);
150       byte[] crlbytes = null;
151       final CRLData data =
152           CRLData.findByIssuerDNAndCRLNumber(
153               entityManager, issuerdn, maxnumber);
154       if (data != null) {
155         crlbytes = data.getCRLBytes();
156         if (crlbytes != null) {
157           final String msg =
158               INTRES.getLocalizedMessage(
159                   "store.getcrl", issuerdn, Integer.valueOf(maxnumber));
160           LOG.info(msg);
161           return crlbytes;
162         }
163       }
164     } catch (Exception e) {
165       final String msg =
166           INTRES.getLocalizedMessage("store.errorgetcrl", issuerdn);
167       LOG.info(msg);
168       throw new EJBException(e);
169     }
170     final String msg =
171         INTRES.getLocalizedMessage(
172             "store.errorgetcrl", issuerdn, Integer.valueOf(maxnumber));
173     LOG.info(msg);
174     if (LOG.isTraceEnabled()) {
175       LOG.trace("<getLastCRL()");
176     }
177     return null;
178   }

209   @Override
210   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
211   public CRLInfo getLastCRLInfo(final String issuerdn, final boolean deltaCRL) {
212     if (LOG.isTraceEnabled()) {
213       LOG.trace(">getLastCRLInfo(" + issuerdn + ", " + deltaCRL + ")");
214     }
215     int crlnumber = 0;
216     try {
217       crlnumber = getLastCRLNumber(issuerdn, deltaCRL);
218       CRLInfo crlinfo = null;
219       final CRLData data =
220           CRLData.findByIssuerDNAndCRLNumber(
221               entityManager, issuerdn, crlnumber);
222       if (data != null) {
223         crlinfo =
224             new CRLInfo(
225                 data.getIssuerDN(),
226                 crlnumber,
227                 data.getThisUpdate(),
228                 data.getNextUpdate());
229       } else {
230         if (deltaCRL && (crlnumber == 0)) {
231           if (LOG.isDebugEnabled()) {
232             LOG.debug("No delta CRL exists for CA with dn '" + issuerdn + "'");
233           }
234         } else if (crlnumber == 0) {
235           if (LOG.isDebugEnabled()) {
236             LOG.debug("No CRL exists for CA with dn '" + issuerdn + "'");
237           }
238         } else {
239           final String msg =
240               INTRES.getLocalizedMessage(
241                   "store.errorgetcrl", issuerdn, Integer.valueOf(crlnumber));
242           LOG.error(msg);
243         }
244       }
245       if (LOG.isTraceEnabled()) {
246         LOG.trace("<getLastCRLInfo()");
247       }
248       return crlinfo;
249     } catch (Exception e) {
250       final String msg =
251           INTRES.getLocalizedMessage("store.errorgetcrlinfo", issuerdn);
252       LOG.info(msg);
253       throw new EJBException(e);
254     }
255   }

257   @Override
258   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
259   public CRLInfo getCRLInfo(final String fingerprint) {
260     if (LOG.isTraceEnabled()) {
261       LOG.trace(">getCRLInfo(" + fingerprint + ")");
262     }
263     try {
264       CRLInfo crlinfo = null;
265       final CRLData data =
266           CRLData.findByFingerprint(entityManager, fingerprint);
267       if (data != null) {
268         crlinfo =
269             new CRLInfo(
270                 data.getIssuerDN(),
271                 data.getCrlNumber(),
272                 data.getThisUpdate(),
273                 data.getNextUpdate());
274       } else {
275         if (LOG.isDebugEnabled()) {
276           LOG.debug("No CRL exists with fingerprint '" + fingerprint + "'");
277         }
278         final String msg =
279             INTRES.getLocalizedMessage(
280                 "store.errorgetcrl", fingerprint, Integer.valueOf(0));
281         LOG.info(msg);
282       }
283       if (LOG.isTraceEnabled()) {
284         LOG.trace("<getCRLInfo()");
285       }
286       return crlinfo;
287     } catch (Exception e) {
288       String msg =
289           INTRES.getLocalizedMessage("store.errorgetcrlinfo", fingerprint);
290       LOG.info(msg);
291       throw new EJBException(e);
292     }
293   }

295   @Override
296   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
297   public int getLastCRLNumber(final String issuerdn, final boolean deltaCRL) {
298     if (LOG.isTraceEnabled()) {
299       LOG.trace(">getLastCRLNumber(" + issuerdn + ", " + deltaCRL + ")");
300     }
301     int maxnumber = 0;
302     Integer result =
303         CRLData.findHighestCRLNumber(entityManager, issuerdn, deltaCRL);
304     if (result != null) {
305       maxnumber = result.intValue();
306     }
307     if (LOG.isTraceEnabled()) {
308       LOG.trace("<getLastCRLNumber(" + maxnumber + ")");
309     }
310     return maxnumber;
311   }

313   private void authorizedToCA(final AuthenticationToken admin, final int caid)
314       throws AuthorizationDeniedException {
315     if (!authorizationSession.isAuthorized(
316         admin, StandardRules.CAACCESS.resource() + caid)) {
317       final String msg =
318           INTRES.getLocalizedMessage(
319               "caadmin.notauthorizedtoca", admin.toString(), caid);
320       throw new AuthorizationDeniedException(msg);
321     }
322   }
