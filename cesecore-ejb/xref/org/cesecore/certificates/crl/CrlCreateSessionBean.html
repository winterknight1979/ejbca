45  /**
46   * Business class for CRL actions, i.e. running CRLs.
47   *
48   * @version $Id: CrlCreateSessionBean.java 25498 2017-03-15 17:01:37Z jeklund $
49   */
50  @Stateless(
51      mappedName = JndiConstants.APP_JNDI_PREFIX + "CrlCreateSessionRemote")
52  @TransactionAttribute(TransactionAttributeType.REQUIRED)
53  public class CrlCreateSessionBean
54      implements CrlCreateSessionLocal, CrlCreateSessionRemote {
55  
56      /** Logger. */
57    private static final Logger LOG =
58        Logger.getLogger(CrlCreateSessionBean.class);
59    /** Internal localization of logs and errors. */
60    private static final InternalResources INTRES =
61        InternalResources.getInstance();
62  
63    /** Auth. */
64    @EJB private AuthorizationSessionLocal authorizationSession;
65    /** CRL. */
66    @EJB private CrlStoreSessionLocal crlSession;
67    /** Crypto. */
68    @EJB private CryptoTokenManagementSessionLocal cryptoTokenManagementSession;
69    /** Log. */
70    @EJB private SecurityEventsLoggerSessionLocal logSession;
71  
72    /** Setup. */
73    @PostConstruct
74    public void postConstruct() {
75      // Install BouncyCastle provider if not available
76      CryptoProviderTools.installBCProviderIfNotAvailable();
77    }
78  
79    @Override
80    public byte[] generateAndStoreCRL(
81        final AuthenticationToken admin,
82        final CA ca,
83        final Collection<RevokedCertInfo> certs,
84        final int basecrlnumber,
85        final int onextCrlNumber)
86        throws CryptoTokenOfflineException, AuthorizationDeniedException {
87        int nextCrlNumber = onextCrlNumber;
88      if (LOG.isTraceEnabled()) {
89        LOG.trace(">createCRL(Collection)");
90      }
91      byte[] crlBytes = null; // return value
92  
93      // Check that we are allowed to create CRLs
94      // Authorization for other things, that we have access to the CA has already
95      // been done
96      final int caid = ca.getCAId();
97      authorizedToCreateCRL(admin, caid);
98  
99      try {
100       if ((ca.getStatus() != CAConstants.CA_ACTIVE)
101           && (ca.getStatus() != CAConstants.CA_WAITING_CERTIFICATE_RESPONSE)) {
102         String msg =
103             INTRES.getLocalizedMessage(
104                 "createcert.canotactive", ca.getSubjectDN());
105         throw new CryptoTokenOfflineException(msg);
106       }
107       final X509CRLHolder crl;
108 
109       boolean deltaCRL = (basecrlnumber > -1);
110       final CryptoToken cryptoToken =
111           cryptoTokenManagementSession.getCryptoToken(
112               ca.getCAToken().getCryptoTokenId());
113       if (cryptoToken == null) {
114         throw new CryptoTokenOfflineException(
115             "Could not find CryptoToken with id "
116                 + ca.getCAToken().getCryptoTokenId());
117       }
118       if (deltaCRL) {
119         // Workaround if transaction handling fails so that crlNumber for
120         // deltaCRL would happen to be the same
121         if (nextCrlNumber == basecrlnumber) {
122           nextCrlNumber++;
123         }
124         crl =
125             ca.generateDeltaCRL(
126                 cryptoToken, certs, nextCrlNumber, basecrlnumber);
127       } else {
128         crl = ca.generateCRL(cryptoToken, certs, nextCrlNumber);
129       }
130       if (crl != null) {
131         // Store CRL in the database, this can still fail so the whole thing is
132         // rolled back
133         String cafp = CertTools.getFingerprintAsString(ca.getCACertificate());
134         if (LOG.isDebugEnabled()) {
135           LOG.debug(
136               "Encoding CRL to byte array. Free memory="
137                   + Runtime.getRuntime().freeMemory());
138         }
139         byte[] tmpcrlBytes = crl.getEncoded();
140         if (LOG.isDebugEnabled()) {
141           LOG.debug(
142               "Finished encoding CRL to byte array. Free memory="
143                   + Runtime.getRuntime().freeMemory());
144           LOG.debug("Storing CRL in certificate store.");
145         }
146         crlSession.storeCRL(
147             admin,
148             tmpcrlBytes,
149             cafp,
150             nextCrlNumber,
151             crl.getIssuer().toString(),
152             crl.toASN1Structure().getThisUpdate().getDate(),
153             crl.toASN1Structure().getNextUpdate().getDate(),
154             (deltaCRL ? 1 : -1));
155         String msg =
156             INTRES.getLocalizedMessage(
157                 "createcrl.createdcrl",
158                 Integer.valueOf(nextCrlNumber),
159                 ca.getName(),
160                 ca.getSubjectDN());
161         Map<String, Object> details = new LinkedHashMap<String, Object>();
162         details.put("msg", msg);
163         logSession.log(
164             EventTypes.CRL_CREATION,
165             EventStatus.SUCCESS,
166             ModuleTypes.CRL,
167             ServiceTypes.CORE,
168             admin.toString(),
169             String.valueOf(caid),
170             null,
171             null,
172             details);
173         // Now all is finished and audit logged, now we are ready to "really"
174         // set the return value
175         crlBytes = tmpcrlBytes;
176       }
177     } catch (CryptoTokenOfflineException ctoe) {
178       String msg =
179           INTRES.getLocalizedMessage("error.catokenoffline", ca.getSubjectDN());
180       LOG.info(msg, ctoe);
181       String auditmsg =
182           INTRES.getLocalizedMessage(
183               "createcrl.errorcreate", ca.getName(), ctoe.getMessage());
184       Map<String, Object> details = new LinkedHashMap<String, Object>();
185       details.put("msg", auditmsg);
186       logSession.log(
187           EventTypes.CRL_CREATION,
188           EventStatus.FAILURE,
189           ModuleTypes.CRL,
190           ServiceTypes.CORE,
191           admin.toString(),
192           String.valueOf(caid),
193           null,
194           null,
195           details);
196       throw ctoe;
197     } catch (Exception e) {
198       LOG.info("Error generating CRL: ", e);
199       String msg =
200           INTRES.getLocalizedMessage(
201               "createcrl.errorcreate", ca.getName(), e.getMessage());
202       Map<String, Object> details = new LinkedHashMap<String, Object>();
203       details.put("msg", msg);
204       logSession.log(
205           EventTypes.CRL_CREATION,
206           EventStatus.FAILURE,
207           ModuleTypes.CRL,
208           ServiceTypes.CORE,
209           admin.toString(),
210           String.valueOf(caid),
211           null,
212           null,
213           details);
214       if (e instanceof EJBException) {
215         throw (EJBException) e;
216       }
217       throw new EJBException(msg, e);
218     }
219     if (LOG.isTraceEnabled()) {
220       LOG.trace("<createCRL(Collection)");
221     }
222     return crlBytes;
223   }
224 
225   private void authorizedToCreateCRL(
226       final AuthenticationToken admin, final int caid)
227       throws AuthorizationDeniedException {
228     if (!authorizationSession.isAuthorized(
229         admin, StandardRules.CREATECRL.resource())) {
230       final String msg =
231           INTRES.getLocalizedMessage(
232               "createcrl.notauthorized", admin.toString(), caid);
233       throw new AuthorizationDeniedException(msg);
234     }
235   }
