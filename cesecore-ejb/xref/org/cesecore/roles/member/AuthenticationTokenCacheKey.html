19  /**
20   * POJO that provides a search key for recently used authentication tokens.

25  public class AuthenticationTokenCacheKey {
26      /** Type. */
27    private final String tokenType;
28    /** Key. */
29    private final int preferredMatchKey;
30    /** Op. */
31    private final int preferredOperator;
32    /** Value. */
33    private final String preferredTokenMatchValue;
34  /**
35   * @param authenticationToken token.
36   */
37    public AuthenticationTokenCacheKey(
38        final AuthenticationToken authenticationToken) {
39      tokenType = authenticationToken.getMetaData().getTokenType();
40      preferredMatchKey = authenticationToken.getPreferredMatchKey();
41      if (preferredMatchKey != AuthenticationToken.NO_PREFERRED_MATCH_KEY) {
42        List<AccessMatchType> accessMatchType =
43            authenticationToken
44                .getMetaData()
45                .getAccessMatchValueIdMap()
46                .get(preferredMatchKey)
47                .getAvailableAccessMatchTypes();
48        preferredOperator =
49            accessMatchType.isEmpty()
50                ? AccessMatchType.TYPE_UNUSED.getNumericValue()
51                : accessMatchType.get(0).getNumericValue();
52      } else {
53        preferredOperator = AccessMatchType.TYPE_UNUSED.getNumericValue();
54      }
55      preferredTokenMatchValue = authenticationToken.getPreferredMatchValue();
56    }
57  
58    /**
59     * Copy constructor.
60     *
61     * @param authenticationTokenCacheKey on fail
62     */
63    public AuthenticationTokenCacheKey(
64        final AuthenticationTokenCacheKey authenticationTokenCacheKey) {
65      tokenType = authenticationTokenCacheKey.tokenType;
66      preferredMatchKey = authenticationTokenCacheKey.preferredMatchKey;
67      preferredOperator = authenticationTokenCacheKey.preferredOperator;
68      preferredTokenMatchValue =
69          authenticationTokenCacheKey.preferredTokenMatchValue;
70    }

72    @Override
73    public int hashCode() {
74      final int prime = 31;
75      int result = 1;
76      result = prime * result + preferredMatchKey;
77      result = prime * result + preferredOperator;
78      result =
79          prime * result
80              + ((preferredTokenMatchValue == null)
81                  ? 0
82                  : preferredTokenMatchValue.hashCode());
83      result = prime * result + ((tokenType == null) ? 0 : tokenType.hashCode());
84      return result;
85    }
