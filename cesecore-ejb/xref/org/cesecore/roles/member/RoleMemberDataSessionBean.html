60    @Override
61    public RoleMember persistRoleMember(final RoleMember roleMember) {
62      if (roleMember == null) {
63        // Successfully did nothing
64        return null;
65      }
66      if (roleMember.getId() == RoleMember.ROLE_MEMBER_ID_UNASSIGNED) {
67        roleMember.setId(findFreePrimaryKey());
68        entityManager.persist(new RoleMemberData(roleMember));
69      } else {
70        final RoleMemberData roleMemberData = find(roleMember.getId());
71        if (roleMemberData == null) {
72          // Must have been removed by another process, but caller wants to
73          // persist it, so we proceed
74          entityManager.persist(new RoleMemberData(roleMember));
75        } else {
76          // Since the entity is managed, we just update its values
77          roleMemberData.updateValuesFromValueObject(roleMember);
78        }
79      }
80      accessTreeUpdateSession.signalForAccessTreeUpdate();
81      RoleMemberCache.INSTANCE.updateWith(
82          roleMember.getId(), roleMember.hashCode(), null, roleMember);
83      AuthenticationTokenCache.INSTANCE.flush();
84      return roleMember;
85    }

87    private int findFreePrimaryKey() {
88      final ProfileID.DB db =
89          new ProfileID.DB() {
90            @Override
91            public boolean isFree(final int i) {
92              // 0 is a protected ID for RoleMemberData. Use only positive values,
93              // since negatives are seen as "erronous" by some customers.
94              return find(i) == null && i > 0;
95            }
96          };
97      return ProfileID.getNotUsedID(db);
98    }

106   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
107   @Override
108   public RoleMember findRoleMember(final int primaryKey) {
109     // 1. Check cache if it is time to sync-up with database
110     if (RoleMemberCache.INSTANCE.shouldCheckForUpdates(primaryKey)) {
111       // 2. If cache is expired or missing, first thread to discover this
112       // reloads item from database and sends it to the cache
113       final RoleMemberData roleMemberData = find(primaryKey);
114       if (roleMemberData == null) {
115         // Ensure that it is removed from cache when the object is no longer
116         // present in the database
117         RoleMemberCache.INSTANCE.removeEntry(primaryKey);
118       } else {
119         final RoleMember roleMember =
120             roleMemberData == null ? null : roleMemberData.asValueObject();
121         // 3. The cache compares the database data with what is in the cache
122         // 4. If database is different from cache, replace it in the cache
123         RoleMemberCache.INSTANCE.updateWith(
124             primaryKey, roleMember.hashCode(), null, roleMember);
125         // Return role member, working even if the cache is disabled
126         return roleMember;
127       }
128     }
129     // 5. Get object from cache now (or null) and be merry
130     return RoleMemberCache.INSTANCE.getEntry(primaryKey);
131   }

157   @Override
158   public boolean remove(final int primaryKey) {
159     RoleMemberData roleMember = find(primaryKey);
160     if (roleMember != null) {
161       entityManager.remove(roleMember);
162       accessTreeUpdateSession.signalForAccessTreeUpdate();
163       AuthenticationTokenCache.INSTANCE.flush();
164       RoleMemberCache.INSTANCE.removeEntry(primaryKey);
165       return true;
166     } else {
167       return false;
168     }
169   }

183   /**
184    * This method will retrieves all role members that match a certain
185    * authentication token. It utilizes the AuthenticationTokenCache to do so, a
186    * cache which saves previous authorization token searches. If the cache entry
187    * hasn't expired the cache will return a list of role member IDs, which will
188    * be further pruned if any of the listed role members are missing (due to
189    * having been removed by another node in the cluster).
190    *
191    * @param authenticationToken an authentication token
192    * @return a list of role members that the authentication token matches
193    */
194   private List<RoleMember> getRoleMembersForAuthenticationToken(
195       final AuthenticationToken authenticationToken) {
196     final String tokenType = authenticationToken.getMetaData().getTokenType();
197     final TypedQuery<RoleMemberData> query;
198     final int preferredMatchKey = authenticationToken.getPreferredMatchKey();
199     List<RoleMember> result = new ArrayList<>();
200     // First check the cache if this authentication token has been checked
201     // recently
202     List<Integer> cachedRoleMemberIds =
203         AuthenticationTokenCache.INSTANCE
204             .getCachedRoleMembersForAuthenticationToken(authenticationToken);
205     if (cachedRoleMemberIds != null) {
206       // We found a cached (and valid) entry
207       for (Integer roleMemberId : cachedRoleMemberIds) {
208         // Role Members may have been removed in the background, so be prepared
209         // for this as well. If that happens, we should update
210         // the cache entry as well.
211         RoleMember roleMember = findRoleMember(roleMemberId);
212         if (roleMember != null) {
213           result.add(roleMember);
214         }
215       }
216       // If the number of found members differs from the number of ID's in the
217       // cache, there must be members missing, and we should update
218       // the authorization cache
219       if (result.size() != cachedRoleMemberIds.size()) {
220         AuthenticationTokenCache.INSTANCE
221             .cacheRoleMembersForAuthenticationToken(
222                 authenticationToken, result);
223       }
224     } else {
225       // We found nothing...
226       if (preferredMatchKey != AuthenticationToken.NO_PREFERRED_MATCH_KEY) {
227         final List<AccessMatchType> accessMatchType =
228             authenticationToken
229                 .getMetaData()
230                 .getAccessMatchValueIdMap()
231                 .get(preferredMatchKey)
232                 .getAvailableAccessMatchTypes();
233         final int preferredOperator =
234             accessMatchType.isEmpty()
235                 ? AccessMatchType.TYPE_UNUSED.getNumericValue()
236                 : accessMatchType.get(0).getNumericValue();
237         // Optimized search for preferred match values (e.g. serial number match
238         // key) amongst members with that match key.
239         // For members with other match keys, we include everything in the
240         // search
241         query =
242             entityManager
243                 .createQuery(
244                     "SELECT a FROM RoleMemberData a WHERE"
245                         + " a.tokenType=:tokenType AND a.roleId<>0 AND"
246                         + " ((a.tokenMatchKey=:preferredTokenMatchKey AND"
247                         + " a.tokenMatchOperator=:operator AND"
248                         + " a.tokenMatchValueColumn=:preferredTokenMatchValue)"
249                         + " OR NOT (a.tokenMatchKey=:preferredTokenMatchKey"
250                         + " AND a.tokenMatchOperator=:operator))",
251                     RoleMemberData.class)
252                 .setParameter("tokenType", tokenType)
253                 .setParameter("preferredTokenMatchKey", preferredMatchKey)
254                 .setParameter("operator", preferredOperator)
255                 .setParameter(
256                     "preferredTokenMatchValue",
257                     authenticationToken.getPreferredMatchValue());
258       } else {
259         // Search for all members with the same token type
260         query =
261             entityManager
262                 .createQuery(
263                     "SELECT a FROM RoleMemberData a WHERE"
264                         + " a.tokenType=:tokenType AND a.roleId<>0",
265                     RoleMemberData.class)
266                 .setParameter("tokenType", tokenType);
267       }
268       for (RoleMemberData roleMemberData : query.getResultList()) {
269         result.add(roleMemberData.asValueObject());
270       }
271       if (!result.isEmpty()) {
272         AuthenticationTokenCache.INSTANCE
273             .cacheRoleMembersForAuthenticationToken(
274                 authenticationToken, result);
275       }
276     }
277 
278     return result;
279   }

281   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
282   @Override
283   public Set<Integer> getRoleIdsMatchingAuthenticationTokenOrFail(
284       final AuthenticationToken authenticationToken)
285       throws AuthenticationFailedException {
286     final Set<Integer> ret = new HashSet<>();
287     if (authenticationToken != null) {
288       for (final RoleMember roleMemberData
289           : getRoleMembersForAuthenticationToken(authenticationToken)) {
290         if (authenticationToken.matches(
291             convertToAccessUserAspect(roleMemberData))) {
292           ret.add(roleMemberData.getRoleId());
293         }
294       }
295     }
296     return ret;
297   }

299   @TransactionAttribute(TransactionAttributeType.SUPPORTS)
300   @Override
301   public Set<RoleMember> getRoleMembersMatchingAuthenticationToken(
302       final AuthenticationToken authenticationToken) {
303     final Set<RoleMember> ret = new HashSet<>();
304     if (authenticationToken != null) {
305       for (final RoleMember roleMember
306           : getRoleMembersForAuthenticationToken(authenticationToken)) {
307         try {
308           if (authenticationToken.matches(
309               convertToAccessUserAspect(roleMember))) {
310             ret.add(roleMember);
311           }
312         } catch (AuthenticationFailedException e) {
313           LOG.debug(e.getMessage(), e);
314         }
315       }
316     }
317     return ret;
318   }

372

374       public void setMatchValue(final String matchValue) { }

376       @Override

379       }
380 
381       @Override
382       public void setCaId(final Integer caId) { }

384       @Override
385       public String getTokenType() {
