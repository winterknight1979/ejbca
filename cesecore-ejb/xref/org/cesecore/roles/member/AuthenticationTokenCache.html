13  package org.cesecore.roles.member;
14  
15  import java.util.ArrayList;
16  import java.util.HashMap;
17  import java.util.List;
18  import java.util.Map;
19  import java.util.Set;
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.config.CesecoreConfiguration;
22  import org.cesecore.internal.CommonCache;
23  import org.cesecore.internal.CommonCacheBase;
24  
25  /**
26   * This cache saves authentication token checks.
27   *
28   * @version $Id: AuthenticationTokenCache.java 28332 2018-02-20 14:40:52Z anatom
29   *     $
30   */
31  public enum AuthenticationTokenCache
32      implements CommonCache<AuthenticationTokenCacheKey> {
33    /** Singleton. */
34      INSTANCE;
35  
36      /** Cache. */
37    private final CommonCacheBase<AuthenticationTokenCacheKey> cache =
38        new CommonCacheBase<AuthenticationTokenCacheKey>() {
39          @Override
40          protected long getCacheTime() {
41            return CesecoreConfiguration.getCacheAuthorizationTime();
42          }
43  
44          @Override
45          protected long getMaxCacheLifeTime() {
46            // We never purge Role unless a database select discovers a missing
47            // object.
48            return 0;
49          }
50  
51        };
52  
53    /**
54     * This map contains a list of RoleMember ids matched to the authentication
55     * token they match against. The rationale behind saving just the IDs is
56     * because the list might grow rather large. Any new role members created,
57     * edited or removed will invalidate its contents.
58     */
59    private final Map<AuthenticationTokenCacheKey, List<Integer>>
60        authenticationTokenToKeyMap = new HashMap<>();
61  
62    @Override
63    public AuthenticationTokenCacheKey getEntry(final Integer id) {
64      if (id == null) {
65        return null;
66      }
67      return cache.getEntry(id);
68    }
69  
70    @Override
71    public AuthenticationTokenCacheKey getEntry(final int id) {
72      return cache.getEntry(id);
73    }
74  
75    @Override
76    public boolean shouldCheckForUpdates(final int id) {
77      return cache.shouldCheckForUpdates(id);
78    }
79  
80    @Override
81    public void updateWith(
82        final int id,
83        final int digest,
84        final String name,
85        final AuthenticationTokenCacheKey authenticationTokenCacheKey) {
86      // Insert a cloned instance into the cache
87      cache.updateWith(
88          id,
89          digest,
90          String.valueOf(digest),
91          new AuthenticationTokenCacheKey(authenticationTokenCacheKey));
92      authenticationTokenToKeyMap.clear();
93    }
94  
95    @Override
96    public void removeEntry(final int id) {
97      cache.removeEntry(id);
98    }
99  
100   @Override
101   public String getName(final int id) {
102     throw new UnsupportedOperationException(
103         "Role members can't be referenced by name.");
104   }
105 
106   @Override
107   public Map<String, Integer> getNameToIdMap() {
108     return cache.getNameToIdMap();
109   }
110 
111   @Override
112   public void flush() {
113     cache.flush();
114     authenticationTokenToKeyMap.clear();
115   }
116 
117   @Override
118   public void replaceCacheWith(final List<Integer> keys) {
119     throw new UnsupportedOperationException(
120         "Operation not applicable for this subtype of cache");
121   }
122 
123   /**
124    * @return values
125    */
126   public Set<AuthenticationTokenCacheKey> getAllValues() {
127     return cache.getAllEntries();
128   }
129 
130   /**
131    * Delivers a list of cached role members, which have been previously mapped
132    * against the provided authentication token. An empty list does *not* imply
133    * that the authentication token lacks matching members, just that such
134    * members may not have been cached yet. Should not be used outside of
135    * RoleMemberDataSession.
136    *
137    * @param authenticationToken an authentication token
138    * @return a list IDs of cached role members, or null if not found/entry has
139    *     expired.
140    */
141   public List<Integer> getCachedRoleMembersForAuthenticationToken(
142       final AuthenticationToken authenticationToken) {
143     AuthenticationTokenCacheKey key =
144         new AuthenticationTokenCacheKey(authenticationToken);
145     // If there exists an entry
146     if (authenticationTokenToKeyMap.containsKey(key)) {
147       // And that entry is still valid
148       if (!shouldCheckForUpdates(key.hashCode())) {
149         return authenticationTokenToKeyMap.get(key);
150       } else {
151         removeEntry(key.hashCode());
152         authenticationTokenToKeyMap.remove(key);
153         return null;
154       }
155     } else {
156       return null;
157     }
158   }
159 
160   /**
161    * This method adds an authentication token and its known role members to the
162    * cache.
163    *
164    * @param authenticationToken an authentication token.
165    * @param roleMembers a list of role members that the authentication token has
166    *     identified for.
167    */
168   public void cacheRoleMembersForAuthenticationToken(
169       final AuthenticationToken authenticationToken,
170       final List<RoleMember> roleMembers) {
171     List<Integer> identifiers = new ArrayList<>();
172     AuthenticationTokenCacheKey authenticationTokenCacheKey =
173         new AuthenticationTokenCacheKey(authenticationToken);
174     cache.updateWith(
175         authenticationTokenCacheKey.hashCode(),
176         authenticationTokenCacheKey.hashCode(),
177         String.valueOf(authenticationTokenCacheKey.hashCode()),
178         authenticationTokenCacheKey);
179     for (RoleMember roleMember : roleMembers) {
180       int roleMemberId = roleMember.getId();
181       identifiers.add(roleMemberId);
182     }
183     authenticationTokenToKeyMap.put(
184         new AuthenticationTokenCacheKey(authenticationToken), identifiers);
185   }
186 
187   @Override
188   public boolean willUpdate(final int id, final int digest) {
189     return cache.willUpdate(id, digest);
190   }
191 }
