138     @Override
139     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
140     public List<Role> getAuthorizedRoles(final AuthenticationToken authenticationToken) {
141         final List<Role> roles = new ArrayList<>();
142         final Set<Integer> roleIdsCallerBelongsTo = roleMemberDataSession.getRoleIdsMatchingAuthenticationToken(authenticationToken);
143         for (final Role role : roleDataSession.getAllRoles()) {
144             // Verify that the caller is authorized to role's namespace
145             if (!isAuthorizedToNameSpace(authenticationToken, role, roleIdsCallerBelongsTo)) {
146                 if (log.isDebugEnabled()) {
147                     log.debug("'" + authenticationToken.toString() + "' is not authorized to the namespace '"+role.getNameSpace()+"'.");
148                 }
149                 continue;
150             }
151             // Verify that the caller is authorized to all access rules in this role
152             if (!isAuthorizedToAllAccessRules(authenticationToken, role, roleIdsCallerBelongsTo)) {
153                 if (log.isDebugEnabled()) {
154                     log.debug("'" + authenticationToken.toString() + "' is not authorized to all access rules in role '"+role.getRoleNameFull()+"'.");
155                 }
156                 continue;
157             }
158             // Verify that the caller is authorized to all CAs that are issuers of members in this role
159             if (!isAuthorizedToAllRoleMembersIssuers(authenticationToken, role.getRoleId())) {
160                 if (log.isDebugEnabled()) {
161                     log.debug("'" + authenticationToken.toString() + "' is not authorized to all members in role '"+role.getRoleNameFull()+"'.");
162                 }
163                 continue;
164             }
165             roles.add(role);
166         }
167         return roles;
168     }

170     @Override
171     public boolean deleteRoleIdempotent(final AuthenticationToken authenticationToken, final int roleId) throws AuthorizationDeniedException {
172         assertAuthorizedToEditRoles(authenticationToken);
173         final Role role = roleDataSession.getRole(roleId);
174         if (role==null) {
175             return false;
176         }
177         // Check that authenticationToken is allowed to remove the role with all its rights
178         final Set<Integer> roleIdsCallerBelongsTo = roleMemberDataSession.getRoleIdsMatchingAuthenticationToken(authenticationToken);
179         assertAuthorizedToAllAccessRules(authenticationToken, role, roleIdsCallerBelongsTo);
180         assertAuthorizedToNameSpace(authenticationToken, role, roleIdsCallerBelongsTo);
181         assertNonImportantRoleMembership(authenticationToken, role, roleIdsCallerBelongsTo);
182         boolean ret = roleDataSession.deleteRoleNoAuthorizationCheck(role.getRoleId());
183         RoleCache.INSTANCE.updateWith(role.getRoleId(), 0, null, null);
184         final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.roleremoved", role.getRoleNameFull());
185         final Map<String, Object> details = new LinkedHashMap<>();
186         details.put("msg", msg);
187         details.put("roleId", role.getRoleId());
188         details.put("roleName", role.getRoleName());
189         details.put("nameSpace", role.getNameSpace());
190         securityEventsLoggerSession.log(EventTypes.ROLE_DELETION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
191                 authenticationToken.toString(), null, null, null, details);
192         final List<RoleMember> roleMembers = roleMemberDataSession.findRoleMemberByRoleId(role.getRoleId());
193         for (final RoleMember roleMember : roleMembers) {
194             ret |= roleMemberDataSession.remove(roleMember.getId());
195         }
196         return ret;
197     }

201     public void assertAuthorizedToRoleMembers(final AuthenticationToken authenticationToken, final int roleId, final boolean requireEditAccess) throws AuthorizationDeniedException {
202         // Check if the caller is authorized to edit roles in general
203         if (requireEditAccess) {

207         }
208         // Is the authToken authorized to the role found by id in the database?
209         final Role roleById = roleId==Role.ROLE_ID_UNASSIGNED ? null : roleDataSession.getRole(roleId);
210         if (roleById!=null) {
211             final Set<Integer> roleIdsCallerBelongsTo = roleMemberDataSession.getRoleIdsMatchingAuthenticationToken(authenticationToken);
212             assertAuthorizedToAllAccessRules(authenticationToken, roleById, roleIdsCallerBelongsTo);
213             assertAuthorizedToNameSpace(authenticationToken, roleById, roleIdsCallerBelongsTo);
214         }
215     }

217     private Role getOriginalRoleAndAssertAuthorizedToEdit(final AuthenticationToken authenticationToken, final Role role, final boolean requireNonImportantRoleMembership)
218             throws AuthorizationDeniedException {
219         // Check if the caller is authorized to edit roles in general
220         assertAuthorizedToEditRoles(authenticationToken);
221         // Is the authToken authorized to the role as provided as an argument?
222         final Set<Integer> roleIdsCallerBelongsTo = roleMemberDataSession.getRoleIdsMatchingAuthenticationToken(authenticationToken);
223         assertAuthorizedToAllAccessRules(authenticationToken, role, roleIdsCallerBelongsTo);
224         if (requireNonImportantRoleMembership) {
225             assertNonImportantRoleMembership(authenticationToken, role, roleIdsCallerBelongsTo);
226         }
227         assertAuthorizedToNameSpace(authenticationToken, role, roleIdsCallerBelongsTo);
228         // Is the authToken authorized to the role found by id in the database?
229         final Role roleById = role.getRoleId()==Role.ROLE_ID_UNASSIGNED ? null : roleDataSession.getRole(role.getRoleId());
230         if (roleById!=null) {
231             assertAuthorizedToAllAccessRules(authenticationToken, roleById, roleIdsCallerBelongsTo);
232             assertAuthorizedToNameSpace(authenticationToken, roleById, roleIdsCallerBelongsTo);
233         }
234         return roleById;
235     }

243     @Override
244     public Role persistRole(final AuthenticationToken authenticationToken, final Role role, final boolean requireNonImportantRoleMembership)
245             throws RoleExistsException, AuthorizationDeniedException {
246         // Normalize and minimize access rules before checking authorization
247         role.normalizeAccessRules();
248         role.minimizeAccessRules();
249         final Role roleById = getOriginalRoleAndAssertAuthorizedToEdit(authenticationToken, role, requireNonImportantRoleMembership);
250         // Sort access rules to make raw xml editing (e.g. statedump) easier
251         role.sortAccessRules();
252         final Role roleByName = roleDataSession.getRole(role.getNameSpace(), role.getRoleName());
253         if (roleById == null) {
254             if (roleByName!=null) {
255                 throw new RoleExistsException(InternalResources.getInstance().getLocalizedMessage("authorization.erroraddroleexists", role.getRoleNameFull()));
256             }
257             // Enforce a non-empty role name
258             if (StringUtils.isEmpty(role.getRoleName())) {
259                 throw new IllegalArgumentException("Role name cannot be empty.");
260             }
261             // Persist new role
262             role.setRoleId(roleDataSession.persistRole(role).getRoleId());
263             final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.roleadded", role.getRoleName());
264             final Map<String, Object> details = new LinkedHashMap<>();
265             details.put("msg", msg);
266             details.put("roleId", role.getRoleId());
267             details.put("roleName", role.getRoleName());
268             details.put("nameSpace", role.getNameSpace());
269             securityEventsLoggerSession.log(EventTypes.ROLE_CREATION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
270                     authenticationToken.toString(), null, null, null, details);
271         } else {
272             // Save to existing role
273             if (roleByName==null) {
274                 // Audit log that the role will be renamed when persisted
275                 final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.rolerenamed", roleById.getRoleNameFull(),
276                         role.getRoleNameFull());
277                 Map<String, Object> details = new LinkedHashMap<>();
278                 details.put("msg", msg);
279                 details.put("roleId", role.getRoleId());
280                 details.put("roleNameOld", roleById.getRoleName());
281                 details.put("roleNameNew", role.getRoleName());
282                 details.put("nameSpaceOld", roleById.getNameSpace());
283                 details.put("nameSpaceNew", role.getNameSpace());
284                 securityEventsLoggerSession.log(EventTypes.ROLE_RENAMING, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
285                         authenticationToken.toString(), null, null, null, details);
286             } else {
287                 if (roleByName.getRoleId() != role.getRoleId()) {
288                     throw new RoleExistsException("A role with the same name exists.");
289                 }
290             }
291             // Persist data changes
292             roleDataSession.persistRole(role);
293         }
294         // Audit log access rule changes (also for new roles)
295         final HashMap<String, Boolean> newAccessRules = role.getAccessRules();
296         final HashMap<String, Boolean> oldAccessRules = roleById==null ? new HashMap<String, Boolean>() : roleById.getAccessRules();
297         final Map<Object,Object> oldAuditMap = new HashMap<>();
298         for (final Entry<String,Boolean> entry : oldAccessRules.entrySet()) {
299             oldAuditMap.put(entry.getKey(), entry.getValue().booleanValue() ? "allow" : "deny");
300         }
301         final Map<Object,Object> newAuditMap = new HashMap<>();
302         for (final Entry<String,Boolean> entry : newAccessRules.entrySet()) {
303             newAuditMap.put(entry.getKey(), entry.getValue().booleanValue() ? "allow" : "deny");
304         }
305         final Map<Object, Object> auditLogDiffMap = UpgradeableDataHashMap.diffMaps(oldAuditMap, newAuditMap);
306         if (!auditLogDiffMap.isEmpty()) {
307             final StringBuilder rulesMsg = new StringBuilder();
308             for (Map.Entry<Object, Object> entry : auditLogDiffMap.entrySet()) {
309                 rulesMsg.append("[" + entry.getKey().toString() + ":"+entry.getValue().toString()+"]");
310             }
311             final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.accessruleschanged", role.getRoleNameFull(), rulesMsg.toString());
312             final Map<String, Object> details = new LinkedHashMap<>();
313             details.put("msg", msg);
314             details.put("roleId", role.getRoleId());
315             details.put("roleName", role.getRoleName());
316             details.put("nameSpace", role.getNameSpace());
317             securityEventsLoggerSession.log(EventTypes.ROLE_ACCESS_RULE_CHANGE, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
318                     authenticationToken.toString(), null, null, null, details);
319         }
320         return role;
321     }

346     /** @param authenticationToken Token
347      * @param role Role
348      * @param roleIdsCallerBelongsTo Caller role 
349      * @throws AuthorizationDeniedException if the caller is not authorized to one of the rules granted access to (even implied) by this role */
350     private void assertAuthorizedToAllAccessRules(final AuthenticationToken authenticationToken, final Role role, final Set<Integer> roleIdsCallerBelongsTo) throws AuthorizationDeniedException {
351         if (!isAuthorizedToAllAccessRules(authenticationToken, role, roleIdsCallerBelongsTo)) {
352             throw new AuthorizationDeniedException("Not authorized to all access rules in role.");
353         }
354     }

356     /** @param authenticationToken Token
357      * @param role Role
358      * @param roleIdsCallerBelongsTo Caller roles 
359      * @return booleam
360       */
361     private boolean isAuthorizedToAllAccessRules(final AuthenticationToken authenticationToken, final Role role, final Set<Integer> roleIdsCallerBelongsTo) {
362         // Verify that authenticationToken has access to every single added allow access rule
363         for (final Entry<String, Boolean> entry : role.getAccessRules().entrySet()) {
364             if (entry.getValue().booleanValue()) {
365                 if (!authorizationSession.isAuthorizedNoLogging(authenticationToken, entry.getKey())) {
366                     // Role would allow what is is not granted to current authenticationToken
367                     return false;
368                 }
369             }
370         }
371         // Verify that role does not have access to any rule that is denied to this authenticationToken
372         HashMap<String, Boolean> totalAccessRules = new HashMap<>();
373         for (final int roleId : roleIdsCallerBelongsTo) {
374             totalAccessRules = AccessRulesHelper.getAccessRulesUnion(totalAccessRules, roleDataSession.getRole(roleId).getAccessRules());
375         }
376         for (final Entry<String, Boolean> entry : totalAccessRules.entrySet()) {
377             if (!entry.getValue().booleanValue()) {
378                 if (role.hasAccessToResource(entry.getKey())) {
379                     // Role would allow what is denied to current authenticationToken
380                     return false;
381                 }
382             }
383         }
384         return true;
385     }

388      * @param role Role
389      * @param roleIdsCallerBelongsTo Caller role
390      * @throws AuthorizationDeniedException if the nameSpace is not "owned" by the caller. */
391     private void assertAuthorizedToNameSpace(final AuthenticationToken authenticationToken, final Role role, final Set<Integer> roleIdsCallerBelongsTo) throws AuthorizationDeniedException {
392         // Assert that AuthenticationToken is allowed to mess with the role's nameSpace
393         if (!isAuthorizedToNameSpace(authenticationToken, role, roleIdsCallerBelongsTo)) {
394             throw new AuthorizationDeniedException("Current AuthenticationToken is not authorized to the namespace '"+role.getNameSpace()+"'.");
395         }

398     /** @param authenticationToken Token
399      * @param roleId Role
400      * @return true if the authenticationToken is authorized to all CAs that are issuers of RoleMembers in this Role */
401     private boolean isAuthorizedToAllRoleMembersIssuers(final AuthenticationToken authenticationToken, final int roleId) {
402         // Verify that the caller is authorized to all CAs that are issuers of members in this role
403         final Set<String> tokenIssuerAccessRules = new HashSet<>();
404         for (final RoleMemberData roleMemberData : roleMemberDataSession.findByRoleId(roleId)) {
405             final AuthenticationTokenMetaData metaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(roleMemberData.getTokenType());
406             final AccessMatchValue accessMatchValue = metaData.getAccessMatchValueIdMap().get(roleMemberData.getTokenMatchKey());
407             if (accessMatchValue.isIssuedByCa()) {
408                 tokenIssuerAccessRules.add(StandardRules.CAACCESS.resource() + roleMemberData.getTokenIssuerId());
409             }
410         }
411         return authorizationSession.isAuthorizedNoLogging(authenticationToken, tokenIssuerAccessRules.toArray(new String[tokenIssuerAccessRules.size()]));
412     }

425     private void assertNonImportantRoleMembership(final AuthenticationToken authenticationToken, final Role role, final Set<Integer> roleIdsCallerBelongsTo) throws AuthorizationDeniedException {
426         if (role.getRoleId()!=Role.ROLE_ID_UNASSIGNED) {
427             // Check that authenticationToken is not about to lock itself out by modifying its own role
428             if (roleIdsCallerBelongsTo.contains(role.getRoleId())) {
429                 if (log.isDebugEnabled()) {
430                     log.debug("'"+authenticationToken+"' relies on match from Role with id " + role.getRoleId() + " for access.");
431                 }
432                 // As long as the admin does not lower its own privileges we are ok with
433                 HashMap<String, Boolean> accessRulesBefore = new HashMap<>();
434                 HashMap<String, Boolean> accessRulesAfter = new HashMap<>();
435                 final Set<String> accessToNamespacesBefore = new HashSet<>();
436                 final Set<String> accessToNamespacesAfter = new HashSet<>();
437                 for (final int roleId : roleIdsCallerBelongsTo) {
438                     final Role existingRole = roleDataSession.getRole(roleId);
439                     final HashMap<String, Boolean> accessRulesFromRole = existingRole.getAccessRules();
440                     accessRulesBefore = AccessRulesHelper.getAccessRulesUnion(accessRulesBefore, accessRulesFromRole);
441                     accessToNamespacesBefore.add(existingRole.getNameSpace());
442                     if (roleId!=role.getRoleId()) {
443                         accessRulesAfter = AccessRulesHelper.getAccessRulesUnion(accessRulesAfter, accessRulesFromRole);
444                         accessToNamespacesAfter.add(existingRole.getNameSpace());
445                     } else {
446                         accessToNamespacesAfter.add(role.getNameSpace());
447                     }
448                 }
449                 if (!accessRulesBefore.equals(accessRulesAfter)) {
450                     throw new AuthorizationDeniedException("Granted access of the current administrator might be affected by this change.");
451                 }
452                 if (!accessToNamespacesBefore.equals(accessToNamespacesAfter) &&
453                         !(accessToNamespacesBefore.contains("") && accessToNamespacesAfter.contains(""))) {
454                     throw new AuthorizationDeniedException("Granted namespace access of the current administrator would be affected by this change.");
455                 }
456                 if (log.isDebugEnabled()) {
457                     log.debug("Access granted to '"+authenticationToken+"' would not be affected by not being a member of Role with id " + role.getRoleId() + ".");
458                 }
459             } else {
460                 if (log.isDebugEnabled()) {
461                     log.debug("'"+authenticationToken+"' does not rely on match from Role with id " + role.getRoleId() + ".");
462                 }
463             }
464         }
465     }

467     /** @param authenticationToken Token
468      * @param role Role
469      * @param roleIdsCallerBelongsTo Caller roles
470      * @return boolean */
471     private boolean isAuthorizedToNameSpace(final AuthenticationToken authenticationToken, final Role role, final Set<Integer> roleIdsCallerBelongsTo) {
472         if (authenticationToken instanceof AlwaysAllowLocalAuthenticationToken) {
473             return true; // AlwaysAllowLocalAuthenticationToken cannot belong to any roles, so the code below will not work
474         }
475         // Assert that AuthenticationToken is allowed to mess with the role's nameSpace
476         final Set<String> ownedNameSpaces = new HashSet<>();
477         for (final int current : roleIdsCallerBelongsTo) {
478             ownedNameSpaces.add(roleDataSession.getRole(current).getNameSpace());
479         }
480         return ownedNameSpaces.contains("") || ownedNameSpaces.contains(role.getNameSpace());
481     }

483     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
484     @Override
485     public List<String> getAuthorizedNamespaces(final AuthenticationToken authenticationToken) {
486         final Set<String> namespaces = new HashSet<>();
487         for (final int current : roleMemberDataSession.getRoleIdsMatchingAuthenticationToken(authenticationToken)) {
488             namespaces.add(roleDataSession.getRole(current).getNameSpace());
489         }
490         if (namespaces.contains("") || authenticationToken instanceof AlwaysAllowLocalAuthenticationToken) {
491             // Add all namespaces from authorized roles
492             for (final Role role : getAuthorizedRoles(authenticationToken)) {
493                 namespaces.add(role.getNameSpace());
494             }
495         }
496         return new ArrayList<>(namespaces);
497     }

499     @Override
500     public boolean updateCaId(final int caIdOld, final int caIdNew, final boolean keepOldAccessRule, final boolean updateRoleMembers) {
501         final String resourceOld = AccessRulesHelper.normalizeResource(StandardRules.CAACCESS.resource() + caIdOld);
502         final String resourceNew = AccessRulesHelper.normalizeResource(StandardRules.CAACCESS.resource() + caIdNew);
503         boolean hasChangedAnything = false;
504         for (final Role role : roleDataSession.getAllRoles()) {
505             final Boolean state = role.getAccessRules().get(resourceOld);
506             if (state!=null) {
507                 if (!keepOldAccessRule) {
508                     role.getAccessRules().remove(resourceOld);
509                 }
510                 role.getAccessRules().put(resourceNew, state);
511                 roleDataSession.persistRole(role);
512                 hasChangedAnything = true;
513             }
514             if (updateRoleMembers) {
515                 for (final RoleMember roleMember : roleMemberDataSession.findRoleMemberByRoleId(role.getRoleId())) {
516                     if (roleMember.getTokenIssuerId()==caIdOld) {
517                         // Do more expensive checks if it is a potential match
518                         final AccessMatchValue accessMatchValue = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(
519                                 roleMember.getTokenType()).getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey());
520                         if (accessMatchValue.isIssuedByCa()) {
521                             roleMember.setTokenIssuerId(caIdNew);
522                             roleMemberDataSession.persistRoleMember(roleMember);
523                             hasChangedAnything = true;
524                         }
525                     }
526                 }
527             }
528         }
529         return hasChangedAnything;
530     }
