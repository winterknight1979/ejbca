129     @Override
130     public Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> getAvailableTypesAndProperties() {
131         // Perform deep cloning (this will work since we know that the property types extend Serializable)
132         final Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> clone;
133         try {
134             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
135             final ObjectOutputStream oos = new ObjectOutputStream(baos);
136             oos.writeObject(InternalKeyBindingFactory.INSTANCE.getAvailableTypesAndProperties());
137             oos.close();
138             final ObjectInputStream ois = new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
139             clone = (Map<String, Map<String, DynamicUiProperty<? extends Serializable>>>) ois.readObject();
140             ois.close();
141         } catch (IOException e) {
142             throw new RuntimeException(e);
143         } catch (ClassNotFoundException e) {
144             throw new RuntimeException(e);
145         }
146         return clone;
147     }

258     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
259     @Override
260     public Integer getIdFromName(String internalKeyBindingName) {
261         if (internalKeyBindingName == null) {
262             return null;
263         }
264         final Map<String, Integer> cachedNameToIdMap = internalKeyBindingDataSession.getCachedNameToIdMap();
265         Integer internalKeyBindingId = cachedNameToIdMap.get(internalKeyBindingName);
266         if (internalKeyBindingId == null) {
267             // Ok.. so it's not in the cache.. look for it the hard way..
268             for (final Integer currentId : internalKeyBindingDataSession.getIds(null)) {
269                 // Don't lookup CryptoTokens we already have in the id to name cache
270                 if (!cachedNameToIdMap.values().contains(currentId)) {
271                     final InternalKeyBinding current = internalKeyBindingDataSession.getInternalKeyBinding(currentId.intValue());
272                     final String currentName = current == null ? null : current.getName();
273                     if (internalKeyBindingName.equals(currentName)) {
274                         internalKeyBindingId = currentId;
275                         break;
276                     }
277                 }
278             }
279         }
280         return internalKeyBindingId;
281     }

283     @Override
284     public List<Collection<X509Certificate>> getListOfTrustedCertificates(InternalKeyBinding internalKeyBinding) throws CADoesntExistsException {
285         
286         List<InternalKeyBindingTrustEntry> trustedReferences = internalKeyBinding.getTrustedCertificateReferences();
287         if(trustedReferences == null) {
288             return null;
289         }
290         
291         List<Collection<X509Certificate> > trustedCerts = new ArrayList<>();
292         if(trustedReferences.size()==0) {
293             // If no trusted certificates are referenced, trust ANY certificates issued by ANY CA known to this EJBCA instance.
294             // This is done by adding all CAs' certificate chains to trustedCerts
295             List<Integer> allCAs = caSession.getAllCaIds();
296             for(int caid : allCAs) {
297                 final CAInfo caInfo = caSession.getCAInfoInternal(caid);
298                 if (caInfo instanceof X509CAInfo) { // ignore CVC CAs
299                     // Quick and dirty cast
300                     Collection<Certificate> certificateChain = caInfo.getCertificateChain();
301                     Collection<X509Certificate> x509CertificateChain = new ArrayList<>(Arrays.asList(certificateChain.toArray(new X509Certificate[certificateChain.size()])));
302                     if (!x509CertificateChain.isEmpty()) {
303                         trustedCerts.add(x509CertificateChain);
304                     }
305                 }
306             }
307             
308             if(log.isDebugEnabled()) {
309                 log.debug("Trusted Certificates list is empty. Trust ANY certificates issued by ANY CA known to this EJBCA instance");
310             }
311         } else {
312             for (final InternalKeyBindingTrustEntry trustedReference : trustedReferences) {
313                 final CAInfo caInfo = caSession.getCAInfoInternal(trustedReference.getCaId());
314                 if (trustedReference.getCertificateSerialNumberDecimal()==null) {
315                     // If no cert serialnumber is specified, then we trust all certificates issued by this CA. We add the entire 
316                     // CA certificate chain to be used for issuer verification
317                     Collection<Certificate> certificateChain = caInfo.getCertificateChain();
318                     Collection<X509Certificate> x509CertificateChain = new ArrayList<>(Arrays.asList(certificateChain.toArray(new X509Certificate[certificateChain.size()])));    
319                     trustedCerts.add(x509CertificateChain);
320                 } else {
321                     // If a cert serialnumber is specified, then we trust only this certificate. We create a certificate collection 
322                     // containing this certificate and it's issuer's certificate chain to be used for issuer verification
323                     X509Certificate cert = (X509Certificate) certificateStoreSession.findCertificateByIssuerAndSerno(
324                                 caInfo.getSubjectDN(), trustedReference.fetchCertificateSerialNumber());
325                     if(cert!=null) {
326                         ArrayList<X509Certificate> leafCertChain = new ArrayList<>();
327                         leafCertChain.add(cert);
328                         String issuer = CertTools.getIssuerDN(cert);
329                         CAInfo issuerInfo = caSession.getCAInfoInternal(issuer.hashCode());
330                         Collection<Certificate> certificateChain = issuerInfo.getCertificateChain();             
331                         leafCertChain.addAll(Arrays.asList(certificateChain.toArray(new X509Certificate[certificateChain.size()])));
332                         trustedCerts.add(leafCertChain);                
333                     } else {
334                         log.info("No (trusted) certificate with issuer '"+caInfo.getSubjectDN()+"' and serialNo "+trustedReference.fetchCertificateSerialNumber().toString(16)
335                                 +" could be found for authentication key binding "+internalKeyBinding.getName()+"."); 
336                     }
337                 }
338             }
339         }
340         
341         if(trustedCerts.size() == 0) {
342             // If the trusted certificates list is empty it mean that the only trusted reference was to a non-existing specific certificate. 
343             // In this case, EJBCA should not trust anything
344             return null;
345         }
346         return trustedCerts;
347     }

360     @Override
361     public int createInternalKeyBinding(AuthenticationToken authenticationToken, String type, int id, String name, InternalKeyBindingStatus status,
362             String certificateId, int cryptoTokenId, String keyPairAlias, boolean allowMissingKeyPair, String signatureAlgorithm, Map<String, Serializable> dataMap,
363             List<InternalKeyBindingTrustEntry> trustedCertificateReferences)
364             throws AuthorizationDeniedException, CryptoTokenOfflineException, InternalKeyBindingNameInUseException, InvalidAlgorithmException {
365         if (!authorizationSession.isAuthorized(authenticationToken, InternalKeyBindingRules.MODIFY.resource(), CryptoTokenRules.USE.resource()
366                 + "/" + cryptoTokenId)) {
367             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.MODIFY.resource(),
368                     authenticationToken.toString());
369             throw new AuthorizationDeniedException(msg);
370         }
371         if (!AlgorithmTools.isSigAlgEnabled(signatureAlgorithm)) {
372             throw new InvalidAlgorithmException("Signature algorithm " + signatureAlgorithm + " is not available.");
373         }
374         // Convert supplied properties using a prefix to ensure that the caller can't mess with internal ones
375         final LinkedHashMap<Object, Object> initDataMap = new LinkedHashMap<Object, Object>();
376         if (dataMap != null) {
377             for (final Entry<String, Serializable> entry : dataMap.entrySet()) {
378                 String key = entry.getKey();
379                 if (key.startsWith(InternalKeyBindingBase.SUBCLASS_PREFIX)) {
380                     initDataMap.put(key, entry.getValue());
381                 } else {
382                     initDataMap.put(InternalKeyBindingBase.SUBCLASS_PREFIX + key, entry.getValue());
383                 }
384             }
385         }
386         if (!allowMissingKeyPair) {
387             // Check that CryptoToken and alias exists (and that the user is authorized to see it)
388             final KeyPairInfo keyPairInfo = cryptoTokenManagementSession.getKeyPairInfo(authenticationToken, cryptoTokenId, keyPairAlias);
389             if (keyPairInfo == null) {
390                 if (cryptoTokenManagementSession.getCryptoTokenInfo(authenticationToken, cryptoTokenId) == null) {
391                     throw new CryptoTokenOfflineException("Unable to access keyPair with alias " + keyPairAlias + " in CryptoToken with id " + cryptoTokenId);
392                 }
393             }
394         }
395         if (certificateId != null) {
396             certificateId = certificateId.toLowerCase(Locale.ENGLISH);
397         }
398         // If we created a new InternalKeyBinding without a certificate reference (e.g. will be uploaded later) we will not allow the status to be anything but "DISABLED"
399         if (certificateId == null || certificateStoreSession.findCertificateByFingerprint(certificateId) == null) {
400             status = InternalKeyBindingStatus.DISABLED;
401         }
402         // Finally, try to create an instance of this type and persist it
403         final InternalKeyBinding internalKeyBinding = InternalKeyBindingFactory.INSTANCE.create(type, id, name, status, certificateId, cryptoTokenId,
404                 keyPairAlias, initDataMap);
405         internalKeyBinding.setSignatureAlgorithm(signatureAlgorithm);
406         if (trustedCertificateReferences!=null) {
407             internalKeyBinding.setTrustedCertificateReferences(trustedCertificateReferences);
408         }
409         final int allocatedId = internalKeyBindingDataSession.mergeInternalKeyBinding(internalKeyBinding);
410         // Audit log the result after persistence (since the id generated during)
411         final Map<String, Object> details = new LinkedHashMap<String, Object>();
412         details.put("msg", "Created InternalKeyBinding with id " + allocatedId);
413         details.put("name", internalKeyBinding.getName());
414         if (internalKeyBinding.getCertificateId()!=null) {
415             details.put("certificateId", internalKeyBinding.getCertificateId());
416         }
417         details.put("keyPairAlias", internalKeyBinding.getKeyPairAlias());
418         if (internalKeyBinding.getNextKeyPairAlias()!=null) {
419             details.put("nextKeyPairAlias", internalKeyBinding.getNextKeyPairAlias());
420         }
421         details.put("signatureAlgorithm", internalKeyBinding.getSignatureAlgorithm());
422         details.put("cryptoTokenId", String.valueOf(internalKeyBinding.getCryptoTokenId()));
423         details.put("status", internalKeyBinding.getStatus().name());
424         details.put("trustedCertificateReferences", Arrays.toString(internalKeyBinding.getTrustedCertificateReferences().toArray()));
425         putDelta(new HashMap<String, DynamicUiProperty<? extends Serializable>>(), internalKeyBinding.getCopyOfProperties(), details);
426         securityEventsLoggerSession.log(EventTypes.INTERNALKEYBINDING_CREATE, EventStatus.SUCCESS, ModuleTypes.INTERNALKEYBINDING, ServiceTypes.CORE,
427                 authenticationToken.toString(), String.valueOf(allocatedId), null, null, details);
428         return allocatedId;
429     }

440     @Override
441     public int persistInternalKeyBinding(AuthenticationToken authenticationToken, InternalKeyBinding internalKeyBinding)
442             throws AuthorizationDeniedException, InternalKeyBindingNameInUseException {
443         if (!authorizationSession.isAuthorized(authenticationToken,
444                 InternalKeyBindingRules.MODIFY.resource() + "/" + internalKeyBinding.getId(), CryptoTokenRules.USE.resource() + "/"
445                         + internalKeyBinding.getCryptoTokenId())) {
446             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.MODIFY.resource(),
447                     authenticationToken.toString());
448             throw new AuthorizationDeniedException(msg);
449         }
450         // Never allow activation of an InternalKeyBinding that has no certificate reference yet
451         if (internalKeyBinding.getStatus().equals(InternalKeyBindingStatus.ACTIVE) && (internalKeyBinding.getCertificateId()==null
452                 || internalKeyBinding.getCertificateId().length()==0)) {
453             internalKeyBinding.setStatus(InternalKeyBindingStatus.DISABLED);
454             log.info("Preventing activation of Internal Key Binding " + internalKeyBinding.getId() + " since there is no certificate referenced.");
455         }
456         // Audit log the result before persistence
457         final InternalKeyBinding originalInternalKeyBinding = internalKeyBindingDataSession.getInternalKeyBinding(internalKeyBinding.getId());
458         final Map<String, Object> details = new LinkedHashMap<String, Object>();
459         details.put("msg", "Edited InternalKeyBinding with id " + internalKeyBinding.getId());
460         if (originalInternalKeyBinding.getName().equals(internalKeyBinding.getName())) {
461             details.put("name", internalKeyBinding.getName());
462         } else {
463             putDelta("name", originalInternalKeyBinding.getName(), internalKeyBinding.getName(), details);
464         }
465         putDelta("certificateId", originalInternalKeyBinding.getCertificateId(), internalKeyBinding.getCertificateId(), details);
466         putDelta("keyPairAlias", originalInternalKeyBinding.getKeyPairAlias(), internalKeyBinding.getKeyPairAlias(), details);
467         putDelta("nextKeyPairAlias", originalInternalKeyBinding.getNextKeyPairAlias(), internalKeyBinding.getNextKeyPairAlias(), details);
468         putDelta("signatureAlgorithm", originalInternalKeyBinding.getSignatureAlgorithm(), internalKeyBinding.getSignatureAlgorithm(), details);
469         putDelta("cryptoTokenId", String.valueOf(originalInternalKeyBinding.getCryptoTokenId()), String.valueOf(internalKeyBinding.getCryptoTokenId()), details);
470         putDelta("status", originalInternalKeyBinding.getStatus().name(), internalKeyBinding.getStatus().name(), details);
471         putDelta("trustedCertificateReferences", Arrays.toString(originalInternalKeyBinding.getTrustedCertificateReferences().toArray()), Arrays.toString(internalKeyBinding.getTrustedCertificateReferences().toArray()), details);
472         putDelta(originalInternalKeyBinding.getCopyOfProperties(), internalKeyBinding.getCopyOfProperties(), details);
473         securityEventsLoggerSession.log(EventTypes.INTERNALKEYBINDING_EDIT, EventStatus.SUCCESS, ModuleTypes.INTERNALKEYBINDING, ServiceTypes.CORE,
474                 authenticationToken.toString(), String.valueOf(internalKeyBinding.getId()), null, null, details);
475         return internalKeyBindingDataSession.mergeInternalKeyBinding(internalKeyBinding);
476     }

478     @Override
479     public boolean deleteInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException {
480         if (!authorizationSession.isAuthorized(authenticationToken, InternalKeyBindingRules.DELETE.resource() + "/" + internalKeyBindingId)) {
481             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.DELETE.resource(),
482                     authenticationToken.toString());
483             throw new AuthorizationDeniedException(msg);
484         }
485         // Audit log the result before persistence
486         final InternalKeyBinding internalKeyBinding = internalKeyBindingDataSession.getInternalKeyBinding(internalKeyBindingId);
487         if (internalKeyBinding != null) {
488             final Map<String, Object> details = new LinkedHashMap<String, Object>();
489             details.put("msg", "Deleted InternalKeyBinding with id " + internalKeyBinding.getId());
490             details.put("name", internalKeyBinding.getName());
491             securityEventsLoggerSession.log(EventTypes.INTERNALKEYBINDING_DELETE, EventStatus.SUCCESS, ModuleTypes.INTERNALKEYBINDING, ServiceTypes.CORE,
492                     authenticationToken.toString(), String.valueOf(internalKeyBinding.getId()), null, null, details);
493             return internalKeyBindingDataSession.removeInternalKeyBinding(internalKeyBindingId);
494         } else {
495             // Didn't exist
496             return false;

560     @Override
561     public byte[] generateCsrForNextKey(final AuthenticationToken authenticationToken, final int internalKeyBindingId, final byte[] name) throws AuthorizationDeniedException,
562             CryptoTokenOfflineException {
563         if (!authorizationSession.isAuthorized(authenticationToken, InternalKeyBindingRules.VIEW.resource() + "/" + internalKeyBindingId)) {
564             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.VIEW.resource(),
565                     authenticationToken.toString());
566             throw new AuthorizationDeniedException(msg);
567         }
568         final InternalKeyBinding internalKeyBinding = internalKeyBindingDataSession.getInternalKeyBinding(internalKeyBindingId);
569         final int cryptoTokenId = internalKeyBinding.getCryptoTokenId();
570         final String nextKeyPairAlias = internalKeyBinding.getNextKeyPairAlias();
571         final String keyPairAlias;
572         if (nextKeyPairAlias == null) {
573             keyPairAlias = internalKeyBinding.getKeyPairAlias();
574         } else {
575             keyPairAlias = nextKeyPairAlias;
576         }
577         final PublicKey publicKey = cryptoTokenManagementSession.getPublicKey(authenticationToken, cryptoTokenId, keyPairAlias).getPublicKey();
578         final String signatureAlgorithm = internalKeyBinding.getSignatureAlgorithm();
579         final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(cryptoTokenId);
580         final PrivateKey privateKey = cryptoToken.getPrivateKey(keyPairAlias);
581         final X500Name x500Name;
582         if (name != null) {
583         	// If there was a parameter, use that
584             x500Name = X500Name.getInstance(name);
585         } else {
586             // Try to look up the currently mapped certificates SubjectDN and use that
587             final String fingerprint = internalKeyBinding.getCertificateId();
588             if (fingerprint != null) {
589                 final Certificate certificate = certificateStoreSession.findCertificateByFingerprint(fingerprint);
590                 if (certificate != null) {
591                     if (certificate instanceof X509Certificate) {
592                         x500Name = X500Name.getInstance(((X509Certificate)certificate).getSubjectX500Principal().getEncoded());
593                         if (log.isDebugEnabled()) {
594                         	log.debug("Using subject DN from active X.509 certificate '" + x500Name.toString() + "'");
595                         }
596                     } else {
597                         x500Name = CertTools.stringToBcX500Name(CertTools.getSubjectDN(certificate));
598                         if (log.isDebugEnabled()) {
599                             log.debug("Using subject DN from active " + certificate.getType() +" certificate '" + x500Name.toString() + "'");
600                         }
601                     }
602                 } else {
603                 	// No mapped certificate exists, and no parameter, use default value
604                     x500Name = CertTools.stringToBcX500Name("CN="+internalKeyBinding.getName());                
605                 }
606             } else {
607                 // No mapped certificate exists, and no parameter, use default value
608                 x500Name = CertTools.stringToBcX500Name("CN="+internalKeyBinding.getName());                
609             }
610         }
611         final String providerName = cryptoToken.getSignProviderName();
612         try {
613             return CertTools.genPKCS10CertificationRequest(signatureAlgorithm, x500Name, publicKey, new DERSet(), privateKey, providerName)
614                     .getEncoded();
615         } catch (OperatorCreationException|IOException e) {
616             log.info("CSR generation failed. internalKeyBindingId=" + internalKeyBindingId + ", cryptoTokenId=" + cryptoTokenId + ", keyPairAlias="
617                     + keyPairAlias + ". " + e.getMessage());
618         }
619         return null;
620     }

622     @Override
623     public String updateCertificateForInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId)
624             throws AuthorizationDeniedException, CertificateImportException {
625         if (!authorizationSession.isAuthorized(authenticationToken, InternalKeyBindingRules.MODIFY.resource() + "/" + internalKeyBindingId)) {
626             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.MODIFY.resource(),
627                     authenticationToken.toString());
628             throw new AuthorizationDeniedException(msg);
629         }
630         final InternalKeyBinding internalKeyBinding = internalKeyBindingDataSession.getInternalKeyBindingForEdit(internalKeyBindingId);
631         final int cryptoTokenId = internalKeyBinding.getCryptoTokenId();
632         final String originalNextKeyPairAlias = internalKeyBinding.getNextKeyPairAlias();
633         final String originalCertificateId = internalKeyBinding.getCertificateId();
634         final String originalKeyPairAlias = internalKeyBinding.getKeyPairAlias();
635         if (log.isDebugEnabled()) {
636             log.debug("nextKeyPairAlias: " + originalNextKeyPairAlias);
637         }
638         boolean updated = false;
639         if (originalNextKeyPairAlias != null) {
640             // If a nextKeyPairAlias is present we assume that this is the one we want to find a certificate for
641             PublicKey nextPublicKey;
642             try {
643                 nextPublicKey = cryptoTokenManagementSession.getPublicKey(authenticationToken, cryptoTokenId, originalNextKeyPairAlias).getPublicKey();
644             } catch (CryptoTokenOfflineException e) {
645                 throw new CertificateImportException("Operation is not available when CryptoToken is offline.", e);
646             }
647             if (nextPublicKey != null) {
648                 final byte[] subjectKeyId = KeyTools.createSubjectKeyId(nextPublicKey).getKeyIdentifier();
649                 final Certificate certificate = certificateStoreSession.findMostRecentlyUpdatedActiveCertificate(subjectKeyId);
650                 if (certificate == null) {
651                     if (log.isDebugEnabled()) {
652                         log.debug("No certificate found for " + originalNextKeyPairAlias);
653                     }
654                 } else {
655                     if (log.isDebugEnabled()) {
656                         log.debug("Certificate found for " + originalNextKeyPairAlias);
657                     }
658                     // Verify that this is an accepted type of certificate to import for the current implementation
659                     assertCertificateIsOkToImport(certificate, internalKeyBinding);
660                     // If current key matches next public key -> import and update nextKey + certificateId
661                     String fingerprint = CertTools.getFingerprintAsString(certificate);
662                     if (!fingerprint.equals(originalCertificateId)) {
663                         internalKeyBinding.updateCertificateIdAndCurrentKeyAlias(fingerprint);
664                         updated = true;
665                         if (log.isDebugEnabled()) {
666                             log.debug("New certificate with fingerprint " + fingerprint + " matching " + originalNextKeyPairAlias + " will be used.");
667                         }
668                     } else {
669                         if (log.isDebugEnabled()) {
670                             log.debug("The latest available certificate was already in use.");
671                         }
672                     }
673                 }
674             } else {
675                 if (log.isDebugEnabled()) {
676                     log.debug("There was no public key for the referenced alias " + originalNextKeyPairAlias);
677                 }
678             }
679         }
680         if (!updated) {
681             // We failed to find a matching certificate for the next key, so we instead try to do the same for the current key pair
682             final String currentKeyPairAlias = internalKeyBinding.getKeyPairAlias();
683             log.debug("currentKeyPairAlias: " + currentKeyPairAlias);
684             PublicKey currentPublicKey;
685             try {
686                 currentPublicKey = cryptoTokenManagementSession.getPublicKey(authenticationToken, cryptoTokenId, currentKeyPairAlias).getPublicKey();
687             } catch (CryptoTokenOfflineException e) {
688                 throw new CertificateImportException("Operation is not available when CryptoToken is offline.", e);
689             }
690             if (currentPublicKey != null) {
691                 final byte[] subjectKeyId = KeyTools.createSubjectKeyId(currentPublicKey).getKeyIdentifier();
692                 final Certificate certificate = certificateStoreSession.findMostRecentlyUpdatedActiveCertificate(subjectKeyId);
693                 if (certificate == null) {
694                     if (log.isDebugEnabled()) {
695                         log.debug("No certificate found for " + currentKeyPairAlias);
696                     }
697                 } else {
698                     if (log.isDebugEnabled()) {
699                         log.debug("Certificate found for " + currentKeyPairAlias);
700                     }
701                     // Verify that this is an accepted type of certificate to import for the current implementation
702                     assertCertificateIsOkToImport(certificate, internalKeyBinding);
703                     final String fingerprint = CertTools.getFingerprintAsString(certificate);
704                     if (!fingerprint.equals(internalKeyBinding.getCertificateId())) {
705                         internalKeyBinding.setCertificateId(fingerprint);
706                         updated = true;
707                         if (log.isDebugEnabled()) {
708                             log.debug("Certificate with fingerprint " + fingerprint + " matching " + currentKeyPairAlias + " will be used.");
709                         }
710                     } else {
711                         if (log.isDebugEnabled()) {
712                             log.debug("The latest available certificate was already in use.");
713                         }
714                     }
715                 }
716             } else {
717                 if (log.isDebugEnabled()) {
718                     log.debug("There was no public key for the referenced alias " + currentKeyPairAlias);
719                 }
720             }
721         }
722         if (updated) {
723             // Audit log the result before persistence
724             final Map<String, Object> details = new LinkedHashMap<String, Object>();
725             details.put("msg", "Updated certificate for InternalKeyBinding with id " + internalKeyBinding.getId());
726             details.put("name", internalKeyBinding.getName());
727             details.put("cryptoTokenId", String.valueOf(internalKeyBinding.getCryptoTokenId()));
728             putDelta("certificateId", originalCertificateId, internalKeyBinding.getCertificateId(), details);
729             putDelta("keyPairAlias", originalKeyPairAlias, internalKeyBinding.getKeyPairAlias(), details);
730             putDelta("nextKeyPairAlias", originalNextKeyPairAlias, internalKeyBinding.getNextKeyPairAlias(), details);
731             securityEventsLoggerSession.log(EventTypes.INTERNALKEYBINDING_EDIT, EventStatus.SUCCESS, ModuleTypes.INTERNALKEYBINDING, ServiceTypes.CORE,
732                     authenticationToken.toString(), String.valueOf(internalKeyBinding.getId()), null, null, details);
733             try {
734                 internalKeyBindingDataSession.mergeInternalKeyBinding(internalKeyBinding);
735             } catch (InternalKeyBindingNameInUseException e) {
736                 // This would be very strange if it happened, since we use the same name and id as for the existing one
737                 throw new CertificateImportException(e);
738             }
739             if (log.isDebugEnabled()) {
740                 log.debug("No certificate found for " + originalNextKeyPairAlias);
741             }
742             return internalKeyBinding.getCertificateId();
743         }
744         return null;
745     }

764     @Override
765     public void importCertificateForInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId, byte[] derEncodedCertificate)
766             throws AuthorizationDeniedException, CertificateImportException {
767         if (!authorizationSession.isAuthorized(authenticationToken, InternalKeyBindingRules.MODIFY.resource() + "/" + internalKeyBindingId)) {
768             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", InternalKeyBindingRules.MODIFY.resource(),
769                     authenticationToken.toString());
770             throw new AuthorizationDeniedException(msg);
771         }
772         // UnDERify
773         final Certificate certificate;
774         try {
775             certificate = CertTools.getCertfromByteArray(derEncodedCertificate, Certificate.class);
776         } catch (CertificateException e) {
777             throw new CertificateImportException(e);
778         }
779         final InternalKeyBinding internalKeyBinding = internalKeyBindingDataSession.getInternalKeyBindingForEdit(internalKeyBindingId);
780         final String originalNextKeyPairAlias = internalKeyBinding.getNextKeyPairAlias();
781         final String originalCertificateId = internalKeyBinding.getCertificateId();
782         final String originalKeyPairAlias = internalKeyBinding.getKeyPairAlias();
783         // Verify that this is an accepted type of certificate to import for the current implementation
784         assertCertificateIsOkToImport(certificate, internalKeyBinding);
785         final int cryptoTokenId = internalKeyBinding.getCryptoTokenId();
786         if (log.isDebugEnabled()) {
787             log.debug("certificate.getPublicKey(): "
788                     + new String(Hex.encode(KeyTools.createSubjectKeyId(certificate.getPublicKey()).getKeyIdentifier())));
789             log.debug("originalKeyPairAlias: " + originalKeyPairAlias);
790         }
791         boolean updated = false;
792         try {
793             String certificateId = CertTools.getFingerprintAsString(certificate);
794             final PublicKey currentPublicKey = cryptoTokenManagementSession.getPublicKey(authenticationToken, cryptoTokenId, originalKeyPairAlias).getPublicKey();
795             if (log.isDebugEnabled()) {
796                 log.debug("currentPublicKey: "
797                         + (currentPublicKey != null ? new String(Hex.encode(KeyTools.createSubjectKeyId(currentPublicKey).getKeyIdentifier()))
798                                 : "null"));
799             }
800             if (currentPublicKey != null
801                     && KeyTools.createSubjectKeyId(currentPublicKey).equals(KeyTools.createSubjectKeyId(certificate.getPublicKey()))) {
802                 // If current key matches current public key -> import + update certificateId
803                 if (isCertificateAlreadyInDatabase(certificateId)) {
804                     log.info("Certificate with fingerprint " + certificateId
805                             + " was already present in the database. Only InternalKeyBinding reference will be updated.");
806                 } else {
807                     storeCertificate(authenticationToken, internalKeyBinding, certificate);
808                 }
809                 internalKeyBinding.setCertificateId(certificateId);
810                 updated = true;
811             } else {
812                 if (log.isDebugEnabled()) {
813                     log.debug("originalNextKeyPairAlias: " + originalNextKeyPairAlias);
814                 }
815                 if (originalNextKeyPairAlias != null) {
816                     final PublicKey nextPublicKey = cryptoTokenManagementSession.getPublicKey(authenticationToken, cryptoTokenId, originalNextKeyPairAlias).getPublicKey();
817                     if (log.isDebugEnabled()) {
818                         log.debug("nextPublicKey: "
819                                 + (nextPublicKey != null ? new String(Hex.encode(KeyTools.createSubjectKeyId(nextPublicKey).getKeyIdentifier()))
820                                         : "null"));
821                     }
822                     if (nextPublicKey != null
823                             && KeyTools.createSubjectKeyId(nextPublicKey).equals(KeyTools.createSubjectKeyId(certificate.getPublicKey()))) {
824                         // If current key matches next public key -> import and update nextKey + certificateId
825                         if (isCertificateAlreadyInDatabase(certificateId)) {
826                             log.info("Certificate with fingerprint " + certificateId
827                                     + " was already present in the database. Only InternalKeyBinding reference will be updated.");
828                         } else {
829                             storeCertificate(authenticationToken, internalKeyBinding, certificate);
830                         }
831                         internalKeyBinding.updateCertificateIdAndCurrentKeyAlias(certificateId);
832                         updated = true;
833                     }
834                 }
835             }
836         } catch (CryptoTokenOfflineException e) {
837             throw new CertificateImportException(e);
838         }
839         if (updated) {
840             // Audit log the result before persistence
841             final Map<String, Object> details = new LinkedHashMap<String, Object>();
842             details.put("msg", "Edited InternalKeyBinding with id " + internalKeyBinding.getId());
843             details.put("name", internalKeyBinding.getName());
844             details.put("cryptoTokenId", String.valueOf(internalKeyBinding.getCryptoTokenId()));
845             putDelta("certificateId", originalCertificateId, internalKeyBinding.getCertificateId(), details);
846             putDelta("keyPairAlias", originalKeyPairAlias, internalKeyBinding.getKeyPairAlias(), details);
847             putDelta("nextKeyPairAlias", originalNextKeyPairAlias, internalKeyBinding.getNextKeyPairAlias(), details);
848             securityEventsLoggerSession.log(EventTypes.INTERNALKEYBINDING_EDIT, EventStatus.SUCCESS, ModuleTypes.INTERNALKEYBINDING, ServiceTypes.CORE,
849                     authenticationToken.toString(), String.valueOf(internalKeyBinding.getId()), null, null, details);
850             try {
851                 internalKeyBindingDataSession.mergeInternalKeyBinding(internalKeyBinding);
852             } catch (InternalKeyBindingNameInUseException e) {
853                 // This would be very strange if it happened, since we use the same name and id as for the existing one
854                 throw new CertificateImportException(e);
855             }
856         } else {
857             throw new CertificateImportException("No keys matching the certificate were found.");
858         }

861     @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
862     @Override
863     public String renewInternallyIssuedCertificate(AuthenticationToken authenticationToken, int internalKeyBindingId,
864             EndEntityInformation endEntityInformation) throws AuthorizationDeniedException, CryptoTokenOfflineException, CertificateImportException {
865         // Assert authorization
866         assertAuthorization(authenticationToken, InternalKeyBindingRules.MODIFY.resource() + "/" + internalKeyBindingId);
867         if (endEntityInformation == null) {
868             throw new CertificateImportException("Cannot renew certificate without an existing end entity.");
869         }
870         final InternalKeyBinding internalKeyBinding = internalKeyBindingDataSession.getInternalKeyBinding(internalKeyBindingId);
871         final String endEntityId = certificateStoreSession.findUsernameByFingerprint(internalKeyBinding.getCertificateId());
872         if (endEntityId == null || !endEntityId.equals(endEntityInformation.getUsername())) {
873             // We expect renewals to re-use the same end entity template as before
874             throw new CertificateImportException("Not allowed to switch end entity during renewal.");
875         }
876         final PublicKey publicKey = getNextPublicKeyForInternalKeyBinding(authenticationToken, internalKeyBinding);
877         final RequestMessage req = new SimpleRequestMessage(publicKey, endEntityInformation.getUsername(), endEntityInformation.getPassword());
878         final CertificateResponseMessage response;
879         final long updateTime = System.currentTimeMillis();
880         try {
881             response = certificateCreateSession.createCertificate(authenticationToken, endEntityInformation, req, X509ResponseMessage.class, new CertificateGenerationParams(), updateTime);
882         } catch (CustomCertificateSerialNumberException e) {
883             throw new CertificateImportException(e);
884         } catch (IllegalKeyException e) {
885             throw new CertificateImportException(e);
886         } catch (CADoesntExistsException e) {
887             throw new CertificateImportException(e);
888         } catch (CertificateCreateException e) {
889             throw new CertificateImportException(e);
890         } catch (CesecoreException e) {
891             throw new CertificateImportException(e);
892         } catch (CertificateExtensionException e) {
893             throw new CertificateImportException(e);
894         }
895         final String newCertificateId = updateCertificateForInternalKeyBinding(authenticationToken, internalKeyBindingId);
896         if (newCertificateId == null) {
897             throw new CertificateImportException("New certificate was never issued.");
898         }
899         // Sanity check that the certificate we issued is the one that is in use
900         final X509Certificate keyBindingCertificate = (X509Certificate) response.getCertificate();
901         if (!newCertificateId.equals(CertTools.getFingerprintAsString(keyBindingCertificate))) {
902             throw new CertificateImportException(
903                     "Issued certificate was not found in database. Throw-away setting for issuing CA is not allowed for InternalKeyBindings.");
904         }
905         return newCertificateId;
906     }

941     /** Imports the certificate to the database 
942      * @param authenticationToken Auth token
943      * @param internalKeyBinding Key binding
944      * @param certificate Cert
945      * @throws AuthorizationDeniedException If access denied 
946      * @throws CertificateImportException If cert cannot be imported*/
947     private void storeCertificate(AuthenticationToken authenticationToken, InternalKeyBinding internalKeyBinding, Certificate certificate)
948             throws AuthorizationDeniedException, CertificateImportException {
949         // First check if the certificate row has been published without the actual certificate
950         // if so, we only need to import the actual certificate
951         if (certificateStoreSession.updateCertificateOnly(authenticationToken, certificate)) {
952             return;
953         }
954         // Set some values for things we cannot know
955         final String username = "IMPORTED_InternalKeyBinding_" + internalKeyBinding.getId();
956         // Find caFingerprint through ca(Admin?)Session
957         final List<Integer> availableCaIds = caSession.getAuthorizedCaIds(authenticationToken);
958         final String issuerDn = CertTools.getIssuerDN(certificate);
959         /*
960          * Algorithm:
961          * - Find current CA with a certificate with a SubjectDN matching the IssuerDN of the leaf cert.
962          * - If found, check if the current CA certificate can verify the leaf cert's signature.
963          * - If this key didn't sign the leaf certificate, find the latest issued CA certificate with the CA's SubjectDN that can verify the leaf cert.
964          */
965         String caFingerprint = null;
966         for (final Integer caId : availableCaIds) {
967             try {
968                 final Certificate caCert = caSession.getCAInfo(authenticationToken, caId).getCertificateChain().iterator().next();
969                 final String subjectDn = CertTools.getSubjectDN(caCert);
970                 if (subjectDn.equals(issuerDn)) {
971                     try {
972                         certificate.verify(caCert.getPublicKey(), BouncyCastleProvider.PROVIDER_NAME);
973                         caFingerprint = CertTools.getFingerprintAsString(caCert);
974                     } catch (InvalidKeyException | CertificateException | NoSuchAlgorithmException | NoSuchProviderException | SignatureException e) {
975                         if (log.isDebugEnabled()) {
976                             log.debug("Unable to verify IKB certificate to import using current CA certificate with serialnumber " +
977                                     CertTools.getSerialNumber(caCert) + " (0x"+CertTools.getSerialNumberAsString(caCert)+"): " + e.getMessage());
978                         }
979                         // Since there exists a CA on this system with the correct issuer, this might be for a previous version of this CA using a different key
980                         final List<Certificate> potentialCaCerts = certificateStoreSession.findCertificatesBySubject(issuerDn);
981                         potentialCaCerts.remove(caCert);
982                         for (final Certificate potentialCaCert : new ArrayList<>(potentialCaCerts)) {
983                             if (!CertTools.isCA(potentialCaCert)) {
984                                 potentialCaCerts.remove(potentialCaCert);
985                                 continue;
986                             }
987                             try {
988                                 certificate.verify(potentialCaCert.getPublicKey(), BouncyCastleProvider.PROVIDER_NAME);
989                             } catch (InvalidKeyException | CertificateException | NoSuchAlgorithmException | NoSuchProviderException | SignatureException e1) {
990                                 if (log.isDebugEnabled()) {
991                                     log.debug("Unable to verify IKB certificate to import using CA certificate with serialnumber " +
992                                             CertTools.getSerialNumber(potentialCaCert) + " (0x"+CertTools.getSerialNumberAsString(potentialCaCert)+"): " + e1.getMessage());
993                                 }
994                                 potentialCaCerts.remove(potentialCaCert);
995                                 continue;
996                             }
997                         }
998                         if (!potentialCaCerts.isEmpty()) {
999                             // Sort by notBefore descending and to use the latest issued (newest notBefore)
1000                             Collections.sort(potentialCaCerts, new Comparator<Certificate>() {
1001                                 @Override
1002                                 public int compare(final Certificate o1, final Certificate o2) {
1003                                     return CertTools.getNotBefore(o2).compareTo(CertTools.getNotBefore(o1));
1004                                 }
1005                             });
1006                             final Certificate choosenCaCert = potentialCaCerts.get(0);
1007                             log.info("Able to verify IKB certificate to import using CA certificate with serialnumber " +
1008                                     CertTools.getSerialNumber(choosenCaCert) + " (0x"+CertTools.getSerialNumberAsString(choosenCaCert)+").");
1009                             caFingerprint = CertTools.getFingerprintAsString(choosenCaCert);
1010                         }
1011                     }
1012                     // No need to check other CAs even if the right CA cert was not found
1013                     break;
1014                 }
1015             } catch (NoSuchElementException e) {
1016                 log.debug("CA with caId " + caId + " has no certificate chain.");
1017             }
1018         }
1019         if (caFingerprint == null) {
1020             throw new CertificateImportException("No CA certificate for " + issuerDn + " was found on the system.");
1021         }
1022         certificateStoreSession.storeCertificate(authenticationToken, certificate, username, caFingerprint, CertificateConstants.CERT_ACTIVE,
1023                 CertificateConstants.CERTTYPE_ENDENTITY, CertificateProfileConstants.NO_CERTIFICATE_PROFILE, EndEntityConstants.NO_END_ENTITY_PROFILE, null, System.currentTimeMillis());
1024     }

1026     /** Helper method for audit logging changes 
1027      * @param oldProperties Old props
1028      * @param newProperties New props
1029      * @param details Details */
1030     private void putDelta(final Map<String, DynamicUiProperty<? extends Serializable>> oldProperties, final Map<String, DynamicUiProperty<? extends Serializable>> newProperties,
1031             final Map<String, Object> details) {
1032         // Find out what has happended to all the old properties
1033         for (final String key : oldProperties.keySet()) {
1034             final DynamicUiProperty<? extends Serializable> oldValue = oldProperties.get(key);
1035             final DynamicUiProperty<? extends Serializable> newValue = newProperties.get(key);
1036             putDelta(key, getAsStringValue(oldValue), getAsStringValue(newValue), details);
1037         }
1038         // Find out which new properties that did not exist in the old
1039         for (final String key : newProperties.keySet()) {
1040             final DynamicUiProperty<? extends Serializable> oldValue = oldProperties.get(key);
1041             if (oldValue==null) {
1042                 final DynamicUiProperty<? extends Serializable> newValue = newProperties.get(key);
1043                 putDelta(key, getAsStringValue(oldValue), getAsStringValue(newValue), details);
1044             }
1045         }
1046     }

1048     /** Helper method for audit logging changes 
1049      * @param key Key
1050      * @param oldValue Old value
1051      * @param newValue New value  
1052      * @param details details */
1053     private void putDelta(final String key, final String oldValue, final String newValue, Map<String, Object> details) {
1054         if (oldValue == null && newValue == null) {
1055             // NOP
1056         } else if (oldValue == null && newValue != null) {
1057             details.put("added:"+key, newValue);
1058         } else if (oldValue != null && newValue == null) {
1059             details.put("removed:"+key, oldValue);
1060         } else if (!oldValue.equals(newValue)) {
1061             details.put("changed:"+key, newValue);
1062         }
1063     }
