56      @PostConstruct
57      public void postConstruct() {
58          // We lookup the reference to our-self in PostConstruct, since we cannot inject this.
59          // We can not inject ourself, JBoss will not start then therefore we use this to get a reference to this session bean
60          // to call readData we want to do it on the real bean in order to get
61          // the transaction setting (REQUIRED) which creates a new transaction if one was not running (required to read LOBs in PostgreSQL)
62          keyBindSession = sessionContext.getBusinessObject(InternalKeyBindingDataSessionLocal.class);
63      }

67      public void flushCache() {
68          InternalKeyBindingCache.INSTANCE.flush();
69          if (log.isDebugEnabled()) {
70              log.debug("Flushed "+InternalKeyBindingCache.class.getSimpleName());
71          }

74      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
75      @Override
76      public InternalKeyBinding getInternalKeyBinding(final int id) {
77          // 1. Check (new) InternalKeyBindingCache if it is time to sync-up with database
78          if (InternalKeyBindingCache.INSTANCE.shouldCheckForUpdates(id)) {
79              if (log.isDebugEnabled()) {
80                  log.debug("Object with ID " + id + " will be checked for updates.");
81              }
82              // 2. If cache is expired or missing, first thread to discover this reloads item from database and sends it to the cache
83              final InternalKeyBindingData internalKeyBindingData = keyBindSession.readData(id);
84              if (internalKeyBindingData==null) {
85                  if (log.isDebugEnabled()) {
86                      log.debug("Requested object did not exist in database and will be purged from cache if present: " + id);
87                  }
88                  // Ensure that it is removed from cache
89                  InternalKeyBindingCache.INSTANCE.removeEntry(id);
90              } else {
91                  final int digest = internalKeyBindingData.getProtectString(0).hashCode();
92                  final String type = internalKeyBindingData.getKeyBindingType();
93                  final String name = internalKeyBindingData.getName();
94                  final InternalKeyBindingStatus status = internalKeyBindingData.getStatusEnum();
95                  final String certificateId = internalKeyBindingData.getCertificateId();
96                  final int cryptoTokenId = internalKeyBindingData.getCryptoTokenId();
97                  final String keyPairAlias = internalKeyBindingData.getKeyPairAlias();
98                  final LinkedHashMap<Object,Object> dataMapToLoad = internalKeyBindingData.getDataMap();
99                  // Create new token and store it in the cache.
100                 final InternalKeyBinding internalKeyBinding = InternalKeyBindingFactory.INSTANCE.create(type, id, name, status, certificateId, cryptoTokenId, keyPairAlias, dataMapToLoad);
101                 InternalKeyBindingCache.INSTANCE.updateWith(id, digest, name, internalKeyBinding);
102             }
103             // 3. The cache compares the database data with what is in the cache
104             // 4. If database is different from cache, replace it in the cache (while trying to keep activation)
105         }
106         // 5. Get InternalKeyBinding from cache (or null) and be merry
107         return  InternalKeyBindingCache.INSTANCE.getEntry(id);
108     }

110     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
111     @Override
112     public InternalKeyBinding getInternalKeyBindingForEdit(final int id) {
113         final InternalKeyBinding internalKeyBinding = getInternalKeyBinding(id);
114         if (internalKeyBinding == null) {
115             return null;
116         } else {
117             // 6. Create and return a clone (so clients don't mess with the cached object)
118             final InternalKeyBinding internalKeyBindingClone = InternalKeyBindingFactory.INSTANCE.create(internalKeyBinding.getImplementationAlias(),
119                     id, internalKeyBinding.getName(), internalKeyBinding.getStatus(), internalKeyBinding.getCertificateId(),
120                     internalKeyBinding.getCryptoTokenId(), internalKeyBinding.getKeyPairAlias(), internalKeyBinding.getDataMapToPersist());
121             return internalKeyBindingClone;
122         }
123     }

125     @Override
126     public int mergeInternalKeyBinding(InternalKeyBinding internalKeyBinding) throws InternalKeyBindingNameInUseException {
127         if (log.isDebugEnabled()) {
128             log.debug(">mergeInternalKeyBinding " + internalKeyBinding.getName() + " " + internalKeyBinding.getClass().getName());
129         }
130         int internalKeyBindingId = internalKeyBinding.getId();
131         final String name = internalKeyBinding.getName();
132         final InternalKeyBindingStatus status = internalKeyBinding.getStatus();
133         final String type = InternalKeyBindingFactory.INSTANCE.getTypeFromImplementation(internalKeyBinding);
134         final String certificateId = internalKeyBinding.getCertificateId();
135         final int cryptoTokenId = internalKeyBinding.getCryptoTokenId();
136         final String keyPairAlias = internalKeyBinding.getKeyPairAlias();
137         final LinkedHashMap<Object,Object> dataMap = internalKeyBinding.getDataMapToPersist();
138         // Allocate a new internalKeyBindingId if we create the InternalKeyBinding
139         InternalKeyBindingData internalKeyBindingData = null;
140         if (internalKeyBindingId == 0) {
141             final List<Integer> allUsedIds = getIds(null);
142             Integer allocatedId = null;
143             for (int i=0; i<100; i++) {
144                 final int current = Integer.valueOf(rnd.nextInt());
145                 if (!allUsedIds.contains(current)) {
146                     allocatedId = current;
147                     break;
148                 }
149             }
150             if (allocatedId == null) {
151                 throw new RuntimeException("Failed to allocate a new internalKeyBindingId.");
152             }
153             internalKeyBindingId = allocatedId.intValue();
154             // We need to replace this object with an object that has the correct ID if we are going to cache it later
155             internalKeyBinding = InternalKeyBindingFactory.INSTANCE.create(type, internalKeyBindingId, name,
156                     status, certificateId, cryptoTokenId, keyPairAlias, dataMap);
157             if (log.isDebugEnabled()) {
158                 log.debug("Allocated a new internalKeyBindingId: " + internalKeyBindingId);
159             }
160         } else {
161             // The one invoking the method has specified an id and expects the object to exist
162             internalKeyBindingData = entityManager.find(InternalKeyBindingData.class, internalKeyBindingId);
163         }
164         if (internalKeyBindingData == null) {
165             // The InternalKeyBinding does not exist in the database, before we add it we want to check that the name is not in use
166             if (isNameUsed(name)) {
167                 if (log.isDebugEnabled()) {
168                     log.debug("isNameUsed("+name+")");
169                 }
170                 throw new InternalKeyBindingNameInUseException(intres.getLocalizedMessage("internalkeybinding.nameisinuse", name));
171             }
172             internalKeyBindingData = new InternalKeyBindingData(internalKeyBindingId, name, status, type, certificateId, cryptoTokenId, keyPairAlias, dataMap);
173         } else {
174             if (!isNameUsedByIdOnly(internalKeyBindingData.getName(), internalKeyBindingId)) {
175                 if (log.isDebugEnabled()) {
176                     log.debug("!isNameUsedByIdOnly("+name+", "+internalKeyBindingId+")");
177                 }
178                 throw new InternalKeyBindingNameInUseException(intres.getLocalizedMessage("internalkeybinding.nameisinuse", name));
179             }
180             // It might be the case that the calling transaction has already loaded a reference to this token
181             // and hence we need to get the same one and perform updates on this object instead of trying to
182             // merge a new object.
183             internalKeyBindingData.setName(name);
184             internalKeyBindingData.setStatusEnum(status);
185             internalKeyBindingData.setKeyBindingType(type);
186             internalKeyBindingData.setCertificateId(certificateId);
187             internalKeyBindingData.setCryptoTokenId(cryptoTokenId);
188             internalKeyBindingData.setKeyPairAlias(keyPairAlias);
189             internalKeyBindingData.setDataMap(dataMap);
190             internalKeyBindingData.setLastUpdate(System.currentTimeMillis());
191         }
192         internalKeyBindingData = createOrUpdateData(internalKeyBindingData);
193         // Update cache with provided token (it might be active and we like keeping things active)
194         InternalKeyBindingCache.INSTANCE.updateWith(internalKeyBindingId, internalKeyBindingData.getProtectString(0).hashCode(), name, internalKeyBinding);
195         if (log.isDebugEnabled()) {
196             log.debug("<mergeInternalKeyBinding " + internalKeyBinding.getName());
197         }
198         return internalKeyBindingId;   // tokenId
199     }

201     @Override
202     public boolean removeInternalKeyBinding(final int id) {
203         final boolean ret = deleteData(id);
204         InternalKeyBindingCache.INSTANCE.updateWith(id, 0, null, null);
205         return ret;
206     }

214     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
215     @Override
216     public boolean isNameUsed(final String name) {
217         final Query query = entityManager.createQuery("SELECT a FROM InternalKeyBindingData a WHERE a.name=:name");
218         query.setParameter("name", name);
219         return !query.getResultList().isEmpty();
220     }
221 
222     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
223     @Override
224     public boolean isNameUsedByIdOnly(final String name, final int id) {
225         final Query query = entityManager.createQuery("SELECT a FROM InternalKeyBindingData a WHERE a.name=:name");
226         query.setParameter("name", name);
227         @SuppressWarnings("unchecked")
228         final List<InternalKeyBindingData> internalKeyBindingDatas = query.getResultList();
229         for (final InternalKeyBindingData internalKeyBindingData: internalKeyBindingDatas) {
230             if (internalKeyBindingData.getId() != id) {
231                 return false;
232             }
233         }
234         return true;
235     }

241     @Override
242     public InternalKeyBindingData readData(final int id) {
243         final Query query = entityManager.createQuery("SELECT a FROM InternalKeyBindingData a WHERE a.id=:id");
244         query.setParameter("id", id);
245         return QueryResultWrapper.getSingleResult(query);
246     }
247 
248     private InternalKeyBindingData createOrUpdateData(final InternalKeyBindingData data) {
249         return entityManager.merge(data);
250     }
251 
252     private boolean deleteData(final int id) {
253         final Query query = entityManager.createQuery("DELETE FROM InternalKeyBindingData a WHERE a.id=:id");
254         query.setParameter("id", id);
255         return query.executeUpdate() == 1;
256     }
257 
258     @SuppressWarnings("unchecked")
259     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
260     @Override
261     public List<Integer> getIds(String keyBindingType) {
262         if (keyBindingType == null) {
263             return entityManager.createQuery("SELECT a.id FROM InternalKeyBindingData a").getResultList();
264         } else {
265             final Query query = entityManager.createQuery("SELECT a.id FROM InternalKeyBindingData a WHERE a.keyBindingType=:keyBindingType");
266             query.setParameter("keyBindingType", keyBindingType);
267             return query.getResultList();
268         }
