93    @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
94    @Override
95    public ConfigurationBase getCachedConfiguration(final String configID) {
96      ConfigurationBase result;
97      try {
98        if (LOG.isTraceEnabled()) {
99          LOG.trace(">getCachedConfiguration(" + configID + ")");
100       }
101       // Only do the actual SQL query if we might update the configuration due
102       // to cache time anyhow
103       if (!GlobalConfigurationCacheHolder.INSTANCE.needsUpdate(configID)) {
104         result =
105             GlobalConfigurationCacheHolder.INSTANCE.getConfiguration(configID);
106       } else {
107         if (LOG.isDebugEnabled()) {
108           LOG.debug("Reading Configuration: " + configID);
109         }
110         final GlobalConfigurationData globalConfigurationData =
111             findByConfigurationId(configID);
112         if (globalConfigurationData == null) {
113           if (LOG.isDebugEnabled()) {
114             LOG.debug(
115                 "No default GlobalConfiguration exists. Creating a new one.");
116           }
117           // We create a new instance of the configuration, but we don't persist
118           // it (the first one to modify it will take care of it)
119           result =
120               GlobalConfigurationCacheHolder.INSTANCE.getNewConfiguration(
121                   configID);
122         } else {
123           result =
124               GlobalConfigurationCacheHolder.INSTANCE.getConfiguration(
125                   globalConfigurationData.getData(), configID);
126         }
127         // Always cache result
128         GlobalConfigurationCacheHolder.INSTANCE.updateConfiguration(
129             result, configID);
130       }
131       return result;
132     } finally {
133       if (LOG.isTraceEnabled()) {
134         LOG.trace("<getCachedConfiguration(" + configID + ")");
135       }
136     }
137   }

139   @Override
140   public void saveConfiguration(
141       final AuthenticationToken admin, final ConfigurationBase conf)
142       throws AuthorizationDeniedException {
143     if (LOG.isTraceEnabled()) {
144       LOG.trace(">saveConfiguration()");
145     }
146     String configID = conf.getConfigurationId();
147     assertAuthorization(admin, configID, "Could not save configuration");
148 
149     final GlobalConfigurationData gcdata = findByConfigurationId(configID);
150     if (gcdata != null) {
151       // Save object and create a diff over what has changed
152       @SuppressWarnings("unchecked")
153       final Map<Object, Object> orgmap =
154           (Map<Object, Object>)
155               GlobalConfigurationCacheHolder.INSTANCE
156                   .getConfiguration(gcdata.getData(), configID)
157                   .saveData();
158       gcdata.setConfiguration(conf);
159       GlobalConfigurationCacheHolder.INSTANCE.updateConfiguration(
160           conf, configID);
161       @SuppressWarnings("unchecked")
162       final Map<Object, Object> newmap = (Map<Object, Object>) conf.saveData();
163       // Get the diff of what changed
164       final Map<Object, Object> diff =
165           UpgradeableDataHashMap.diffMaps(orgmap, newmap);
166       // Make security audit log record
167       final String msg =
168           INTRES.getLocalizedMessage(
169               "globalconfig.savedconf", gcdata.getConfigurationId());
170       final Map<String, Object> details = new LinkedHashMap<String, Object>();
171       details.put("msg", msg);
172       for (Map.Entry<Object, Object> entry : diff.entrySet()) {
173         details.put(entry.getKey().toString(), entry.getValue().toString());
174       }
175       auditSession.log(
176           EventTypes.SYSTEMCONF_EDIT,
177           EventStatus.SUCCESS,
178           ModuleTypes.GLOBALCONF,
179           ServiceTypes.CORE,
180           admin.toString(),
181           null,
182           null,
183           null,
184           details);
185     } else {
186       // Global configuration doesn't yet exist, so persist a new one.
187       try {
188         GlobalConfigurationData gcd =
189             new GlobalConfigurationData(configID, conf);
190         entityManager.persist(gcd);
191         GlobalConfigurationCacheHolder.INSTANCE.updateConfiguration(
192             conf, configID);
193         final String msg =
194             INTRES.getLocalizedMessage("globalconfig.createdconf", configID);
195         auditSession.log(
196             EventTypes.SYSTEMCONF_CREATE,
197             EventStatus.SUCCESS,
198             ModuleTypes.GLOBALCONF,
199             ServiceTypes.CORE,
200             admin.toString(),
201             null,
202             null,
203             null,
204             msg);
205       } catch (Exception e) {
206         final String msg =
207             INTRES.getLocalizedMessage("globalconfig.errorcreateconf");
208         LOG.info(msg, e);
209         final Map<String, Object> details = new LinkedHashMap<String, Object>();
210         details.put("msg", msg);
211         details.put("error", e.getMessage());
212         auditSession.log(
213             EventTypes.SYSTEMCONF_CREATE,
214             EventStatus.FAILURE,
215             ModuleTypes.GLOBALCONF,
216             ServiceTypes.CORE,
217             admin.toString(),
218             null,
219             null,
220             null,
221             details);
222       }
223     }
224     if (LOG.isTraceEnabled()) {
225       LOG.trace("<saveGlobalConfiguration()");
226     }
227   }

229   private void assertAuthorization(
230       final AuthenticationToken authenticationToken,
231       final String configID,
232       final String errorMsg)
233       throws AuthorizationDeniedException {
234     final String accessRule = getAccessRuleFromConfigId(configID);
235     if (!authorizationSession.isAuthorized(authenticationToken, accessRule)) {
236       final String msg =
237           INTRES.getLocalizedMessage(
238               "authorization.notauthorizedtoresource", accessRule, errorMsg);
239       throw new AuthorizationDeniedException(msg);
240     }
241   }
242 
243   /**
244    * @param configID ID
245    * @return the access rule required to read the specified configuration type
246    */
247   private String getAccessRuleFromConfigId(final String configID) {
248     switch (configID) {
249       case AvailableExtendedKeyUsagesConfiguration.CONFIGURATION_ID:
250         return StandardRules.EKUCONFIGURATION_EDIT.resource();
251       case AvailableCustomCertificateExtensionsConfiguration.CONFIGURATION_ID:
252         return StandardRules.CUSTOMCERTEXTENSIONCONFIGURATION_EDIT.resource();
253       default:
254     }
255     return StandardRules.SYSTEMCONFIGURATION_EDIT.resource();
256   }

258   @Override
259   public void flushConfigurationCache(final String configID) {
260     if (LOG.isTraceEnabled()) {
261       LOG.trace(">flushConfigurationCache(" + configID + ")");
262     }
263     GlobalConfigurationCacheHolder.INSTANCE.clearCache(configID);
264     // Force cache to be loaded from the database unless another thread has
265     // already started to do it
266     getCachedConfiguration(configID);
267     if (LOG.isTraceEnabled()) {
268       LOG.trace("<flushConfigurationCache(" + configID + ")");
269     }
270   }

280   private enum GlobalConfigurationCacheHolder {
281       /** Singleton. */
282     INSTANCE;
283 
284       /** Cache. */
285     private final Map<String, ConfigurationCache> caches =
286         new ConcurrentHashMap<String, ConfigurationCache>();
287 
288     GlobalConfigurationCacheHolder() {
289       ServiceLoader<? extends ConfigurationCache> serviceLoader =
290           ServiceLoader.load(ConfigurationCache.class);
291       // Extract all the caches from the plugin list
292       for (ConfigurationCache cache : serviceLoader) {
293         if (caches.containsKey(cache.getConfigId())) {
294           throw new IllegalStateException(
295               "Two caches loaded with the same config ID: "
296                   + cache.getConfigId()
297                   + ". This is an invalid state.");
298         } else {
299           caches.put(cache.getConfigId(), cache);
300         }
301       }
302     }
303 
304     /** @return all registered configuration IDs. */
305     public Set<String> getIds() {
306       return new HashSet<String>(caches.keySet());
307     }
308 
309     public void updateConfiguration(
310         final ConfigurationBase conf, final String configId) {
311       caches.get(configId).updateConfiguration(conf);
312     }
313 
314     public void clearCache(final String configId) {
315       caches.get(configId).clearCache();
316     }
317 
318     public boolean needsUpdate(final String configId) {
319       if (caches.get(configId) == null) {
320         return true;
321       }
322       return caches.get(configId).needsUpdate();
323     }
324 
325     public ConfigurationBase getConfiguration(final String configId) {
326       ConfigurationCache cache = caches.get(configId);
327       if (cache == null) {
328         return null;
329       } else {
330         return cache.getConfiguration();
331       }
332     }
333 
334     @SuppressWarnings("rawtypes")
335     public ConfigurationBase getConfiguration(
336         final HashMap data, final String configId) {
337       ConfigurationCache cache = caches.get(configId);
338       if (cache == null) {
339         return null;
340       } else {
341         return cache.getConfiguration(data);
342       }
343     }
344 
345     public ConfigurationBase getNewConfiguration(final String configId) {
346       ConfigurationCache cache = caches.get(configId);
347       if (cache == null) {
348         return null;
349       } else {
350         return cache.getNewConfiguration();
351       }
352     }
353 
354     public Properties getAllProperties(final String configId) {
355       ConfigurationCache cache = caches.get(configId);
356       if (cache == null) {
357         return null;
358       } else {
359         return caches.get(configId).getAllProperties();
360       }
361     }
362   }
