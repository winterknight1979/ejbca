43  /**
44   * This class handles secure logs auditing. This class is responsible for checking
45   * authorization and delegating a request to the right implementation.
46   * 
47   * @version $Id: SecurityEventsAuditorSessionBean.java 25498 2017-03-15 17:01:37Z jeklund $
48   */
49  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "SecurityEventsAuditorSessionRemote")
50  @TransactionAttribute(TransactionAttributeType.REQUIRED)
51  public class SecurityEventsAuditorSessionBean implements SecurityEventsAuditorSessionLocal, SecurityEventsAuditorSessionRemote {
52  	
53  	private static final Logger LOG = Logger.getLogger(SecurityEventsAuditorSessionBean.class);
54  
55      @EJB
56      private AuthorizationSessionLocal authorizationSession;
57      @EJB
58      private IntegrityProtectedAuditorSessionLocal integrityProtectedAuditorSession;
59      @EJB
60      private QueuedAuditorSessionLocal queuedAuditorSession;
61      
62      @Override
63      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
64      public Set<String> getQuerySupportingLogDevices() {
65      	return AuditDevicesConfig.getQuerySupportingDeviceIds();
66      }
67  
68      @Override
69      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
70      public List<? extends AuditLogEntry> selectAuditLogs(final AuthenticationToken token, final int startIndex, final int max, final QueryCriteria criteria, final String logDeviceId) throws AuthorizationDeniedException {
71          assertAuthorization(token, AuditLogRules.VIEW.resource());
72      	return AuditDevicesConfig.getDevice(getEjbs(), logDeviceId).selectAuditLogs(token, startIndex, max, criteria, AuditDevicesConfig.getProperties(logDeviceId));
73      }
74  
75      @Override
76      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
77      public AuditLogExportReport exportAuditLogs(final AuthenticationToken token, final CryptoToken cryptoToken, final Date timestamp,
78              final boolean deleteAfterExport, final String keyAlias, final String algorithm, final String logDeviceId) throws AuditLogExporterException, AuthorizationDeniedException {
79          final HashMap<String, Object> details = new LinkedHashMap<String, Object>();
80          details.put(SigningFileOutputStream.EXPORT_SIGN_KEYALIAS, keyAlias);
81          details.put(SigningFileOutputStream.EXPORT_SIGN_ALG, algorithm);
82          return exportAuditLogs(token, cryptoToken, timestamp, deleteAfterExport, details, logDeviceId);
83      }
84  
85      @Override
86      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
87      public AuditLogExportReport exportAuditLogs(final AuthenticationToken token, final CryptoToken cryptoToken, final Date timestamp,
88              final boolean deleteAfterExport, final String keyAlias, final String algorithm, final Certificate certificate, final String logDeviceId)
89              throws AuditLogExporterException, AuthorizationDeniedException {
90          final HashMap<String, Object> details = new LinkedHashMap<String, Object>();
91          details.put(SigningFileOutputStream.EXPORT_SIGN_KEYALIAS, keyAlias);
92          details.put(SigningFileOutputStream.EXPORT_SIGN_ALG, algorithm);
93          details.put(SigningFileOutputStream.EXPORT_SIGN_CERT, certificate);
94          return exportAuditLogs(token, cryptoToken, timestamp, deleteAfterExport, details, logDeviceId);
95      }
96  
97      @Override
98      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
99      public AuditLogExportReport exportAuditLogs(final AuthenticationToken token, final CryptoToken cryptoToken, final Date timestamp,
100             final boolean deleteAfterExport, final Map<String, Object> signatureDetails, final String logDeviceId) throws AuditLogExporterException, AuthorizationDeniedException {
101     	// StandardRules.AUDITLOGEXPORT export implies StandardRules.AUDITLOGVERIFY.
102         assertAuthorization(token, AuditLogRules.EXPORT_LOGS.resource());
103         LOG.info("Export of audit logs from device " + logDeviceId + " requested.");
104         final Class<? extends AuditExporter> exporter = AuditDevicesConfig.getExporter(logDeviceId);
105         final Properties properties = AuditDevicesConfig.getProperties(logDeviceId);
106     	return AuditDevicesConfig.getDevice(getEjbs(), logDeviceId).exportAuditLogs(token, cryptoToken, timestamp, deleteAfterExport, signatureDetails, properties, exporter);
107     }
108 
109     @Override
110     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
111     public AuditLogValidationReport verifyLogsIntegrity(final AuthenticationToken token, final Date timestamp, final String logDeviceId) throws AuditLogValidatorException, AuthorizationDeniedException {
112     	assertAuthorization(token, AuditLogRules.VERIFY.resource());
113         LOG.info("Validation of audit logs in device " + logDeviceId + " requested.");
114     	return AuditDevicesConfig.getDevice(getEjbs(), logDeviceId).verifyLogsIntegrity(token, timestamp, AuditDevicesConfig.getProperties(logDeviceId));
115     }
116 
117 
118 	@Override
119 	public void prepareReset() throws AuditLogResetException {
120 		LogServiceState.INSTANCE.disable();
121         for (final String loggerId : AuditDevicesConfig.getAllDeviceIds()) {
122     		AuditDevicesConfig.getDevice(getEjbs(), loggerId).prepareReset();
123         }
124 	}
125 
126 	@Override
127 	public void reset() throws AuditLogResetException {
128         for (final String loggerId : AuditDevicesConfig.getAllDeviceIds()) {
129     		AuditDevicesConfig.getDevice(getEjbs(), loggerId).reset();
130         }
131         //should be called after the reset because if we enable log before might happen that a log call runs before reset
132         LogServiceState.INSTANCE.enable();
133 	}
134 
135 	/** Assert that we are authorized to the requested resource. 
136 	 * @param token Token
137 	 * @param accessRule accessRule
138 	 * @throws AuthorizationDeniedException if access denied*/
139     private void assertAuthorization(final AuthenticationToken token, final String accessRule) throws AuthorizationDeniedException {
140         if (!authorizationSession.isAuthorized(token, accessRule)) {
141 			throw new AuthorizationDeniedException("not authorized to: "+ token.toString());				
142 		} 
143     }
144 
145     /**
146      * Propagate the injected SSBs, since we can't use application server agnostic EJB lookup in EJB 3.0.
147      * With EJB 3.1 this should be delegated to each implementation where the implementation specific SSBs
148      * can be looked up.
149      * @return Map
150      */
151     private Map<Class<?>, Object> getEjbs() {
152     	final Map<Class<?>, Object> ejbs = new HashMap<Class<? extends Object>, Object>();
153     	ejbs.put(IntegrityProtectedAuditorSessionLocal.class, integrityProtectedAuditorSession);
154     	ejbs.put(QueuedAuditorSessionLocal.class, queuedAuditorSession);
155     	return ejbs;
156     }
