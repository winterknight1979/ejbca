61   * This class handles secure logs auditing.
62   * 
63   * This was created to evaluate the performance of using database integrity protection
64   * instead of custom code for log singing.
65   * 
66   * The index
67   *  "CREATE UNIQUE INDEX auditrecorddata_idx1 ON AuditRecordData (nodeId,timeStamp,sequenceNumber);"
68   * should be present for proper validation and export performance.

92  	@Override
93  	@TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
94  	public AuditLogExportReport exportAuditLogs(AuthenticationToken token, CryptoToken cryptoToken, Date timestamp, boolean deleteAfterExport,
95  			Map<String, Object> signatureDetails, final Properties properties, final Class<? extends AuditExporter> c) throws AuditLogExporterException {
96          final AuditLogExportReport report = new AuditLogExportReport();
97          try {
98              final File exportFile = AuditDevicesConfig.getExportFile(properties, timestamp);
99              try (final SigningFileOutputStream signingFileOutputStream = new SigningFileOutputStream(exportFile, cryptoToken, signatureDetails)) {
100                 final AuditExporter auditExporter = c.getConstructor().newInstance();
101                 auditExporter.setOutputStream(signingFileOutputStream);
102                 verifyAndOptionalExport(auditExporter, report, timestamp, AuditDevicesConfig.getAuditLogExportFetchSize(properties));
103                 report.setExportedFile(exportFile.getCanonicalPath());
104                 if (log.isDebugEnabled()) {
105                     log.debug("Exported " + report.getExportCount() + " rows.");
106                 }
107                 logVerificationResult(report.errors().size(), timestamp, token);
108                 // Sign the exported file ... it will write the signature on the side
109                 final String signatureFilename = signingFileOutputStream.writeSignature();
110                 report.setSignatureFile(signatureFilename);
111                 // Log export success
112                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
113                 details.put("deleteAfterExport", deleteAfterExport);
114                 details.put("timestamp", ValidityDate.formatAsISO8601(new Date(), ValidityDate.TIMEZONE_UTC));
115                 securityEventsLogger.log(EventTypes.LOG_EXPORT, EventStatus.SUCCESS, ModuleTypes.SECURITY_AUDIT, ServiceTypes.CORE, token.toString(), null, null, null, details);
116                 // Delete the exported log entries if requested
117                 if (deleteAfterExport) {
118                     if (log.isDebugEnabled()) {
119                         log.debug("deleting exported logs");
120                     }
121                     final int deletedRowCount = integrityProtectedAuditorSession.deleteRows(token, timestamp, properties);
122                     if (log.isDebugEnabled()) {
123                         log.debug("Deleted " + deletedRowCount + " rows from audit log after export.");
124                     }
125                 }
126                 auditExporter.close();
127             }
128         } catch (final Exception e) {
129             throw new AuditLogExporterException(e.getMessage(), e);
130         }
131         return report;
132 	}

136 	@TransactionAttribute(TransactionAttributeType.REQUIRED)
137 	public int deleteRows(final AuthenticationToken token, final Date timestamp, final Properties properties) {
138         final Map<String, Object> detailsDelete = new LinkedHashMap<String, Object>();
139         detailsDelete.put("timestamp", FastDateFormat.getInstance(ValidityDate.ISO8601_DATE_FORMAT, TimeZone.getTimeZone("GMT")).format(timestamp));
140         securityEventsLogger.log(EventTypes.LOG_DELETE, EventStatus.VOID, ModuleTypes.SECURITY_AUDIT, ServiceTypes.CORE, token.toString(), null, null, null, detailsDelete);
141         // Delete all the exported logs (from all nodes)
142 		final QueryCriteria queryCriteria = QueryCriteria.create().add(Criteria.leq(AuditLogEntry.FIELD_TIMESTAMP, timestamp.getTime()));
143 		return buildConditionalQuery(entityManager, "DELETE FROM AuditRecordData a", queryCriteria, 0, 0).executeUpdate();
144 	}

154 	public AuditLogValidationReport verifyLogsIntegrity(final AuthenticationToken token, final Date timestamp, final Properties properties) throws AuditLogValidatorException {
155         final AuditLogValidationReport report = new AuditLogValidationReport();
156         try {
157             verifyAndOptionalExport(null, report, timestamp, AuditDevicesConfig.getAuditLogExportFetchSize(properties));
158         	// Log the success or failure depending on if verification returns error or not
159         	logVerificationResult(report.errors().size(), timestamp, token);
160         } catch (final Exception e) {
161         	throw new AuditLogValidatorException(e.getMessage(), e);
162         }
163         return report;
164 	}

166 	/**
167 	 * Read batches of logs from the database. If the database integrity check fails, the batch will be processed row by row.
168 	 * Results are added to the report.
169 	 * @param auditExporter can be null if no export should take place
170 	 * @param report is a AuditLogValidationReport or AuditLogExportReport
171 	 * @param timestamp process all entries up until this time (should be epoch GMT)
172 	 * @param fetchSize size
173 	 * @throws IOException if io fails
174 	 */
175 	private void verifyAndOptionalExport(AuditExporter auditExporter, AuditLogValidationReport report, Date timestamp, final int fetchSize) throws IOException {
176     	// Get a list of the nodes that have data in the database
177     	for (final String nodeId : getNodeIds()) {
178     		if (log.isDebugEnabled()) {
179     			log.debug("exportAuditLogs for nodeId " + nodeId);
180     		}
181     		// Assuming timeStamp is in UTC
182     		final QueryCriteria queryCriteria = QueryCriteria.create().add((Criteria.and(Criteria.eq(AuditLogEntry.FIELD_NODEID, nodeId), Criteria.leq(AuditLogEntry.FIELD_TIMESTAMP, timestamp.getTime())))).add(Criteria.orderAsc(AuditLogEntry.FIELD_SEQUENCENUMBER));
183     		int startIndex = 1;
184     		final Holder<Long> lastSeqNumber = new Holder<Long>(Long.valueOf(-1L));
185     		while (true) {
186     			try {
187         			final List<AuditRecordData> queryResult = verifyLogsIntegritySubset(startIndex, fetchSize, queryCriteria, report, lastSeqNumber, nodeId);
188     				final int results = queryResult.size();
189     				if (results == 0) {
190     					break;	// No more data for this node
191     				}
192     				startIndex += results;
193 					if (auditExporter!=null) {
194 						for (final AuditRecordData auditRecordData : queryResult) {
195 							writeToExport(auditExporter, auditRecordData);
196         					((AuditLogExportReport) report).incExportCount();
197 						}
198 					}
199     			} catch (DatabaseProtectionException e) {
200     				// One of the FETCH_SIZE entries failed.. we have to go through line by line to find out witch one..
201     				for (int i=0; i<fetchSize; i++) {
202         				try {
203                 			final List<AuditRecordData> queryResult = verifyLogsIntegritySubset(startIndex, 1, queryCriteria, report, lastSeqNumber, nodeId);
204             				final int results = queryResult.size();
205             				if (results != 1) {
206             					break;	// No more data for this node
207             				}
208         					if (auditExporter!=null) {
209         						writeToExport(auditExporter, queryResult.get(0));
210             					((AuditLogExportReport) report).incExportCount();
211         					}
212         				} catch (DatabaseProtectionException e2) {
213         					final AuditRecordData auditRecordData = (AuditRecordData) e2.getEntity();
214         					// Add to report
215         					report.warn(new AuditLogReportElem(lastSeqNumber.get().longValue(), auditRecordData.getSequenceNumber(), "log with sequence number after " + lastSeqNumber + " on nodeId " + nodeId + " could not be verified"));
216         					lastSeqNumber.set(auditRecordData.getSequenceNumber());
217         					// We still export it
218         					// TODO: It might make sense to make it configurable to export when verification fails..
219         					if (auditExporter!=null) {
220             					writeToExport(auditExporter, auditRecordData);
221             					((AuditLogExportReport) report).incExportCount();
222         					}
223         				}
224         				startIndex += 1;
225     				}
226     			}
227     		}
228     	}
229 	}

231 	/** We want to export exactly like it was stored in the database, to comply with requirements on logging systems where no altering of the original log data is allowed. 
232 	 * @param auditExporter Exporter
233 	 * @param auditRecordData Data to export
234 	 * @throws IOException if write fails */
235     private void writeToExport(final AuditExporter auditExporter, final AuditRecordData auditRecordData) throws IOException {
236         auditExporter.writeStartObject();
237         auditExporter.writeField("pk", auditRecordData.getPk());
238         auditExporter.writeField("nodeId", auditRecordData.getNodeId());
239         auditExporter.writeField("sequenceNumber", auditRecordData.getSequenceNumber());
240         auditExporter.writeField("timestamp", auditRecordData.getTimeStamp());
241         auditExporter.writeField("eventType", auditRecordData.getEventTypeValue().toString());
242         auditExporter.writeField("eventStatus", auditRecordData.getEventStatusValue().toString());
243         auditExporter.writeField("authToken", auditRecordData.getAuthToken());
244         auditExporter.writeField("service", auditRecordData.getServiceTypeValue().toString());
245         auditExporter.writeField("module", auditRecordData.getModuleTypeValue().toString());
246         auditExporter.writeField("customId", auditRecordData.getCustomId());
247         auditExporter.writeField("searchDetail1", auditRecordData.getSearchDetail1());
248         auditExporter.writeField("searchDetail2", auditRecordData.getSearchDetail2());
249         auditExporter.writeField("additionalDetails", auditRecordData.getAdditionalDetails());
250         auditExporter.writeField("rowProtection", auditRecordData.getRowProtection());
251         auditExporter.writeEndObject();
252     }

254     /**
255      * Fetch a batch of log rows from the database (implying database integrity check) and verifies
256      * that all sequence numbers are present.
257      * @param startIndex start batch from this position 
258      * @param max entries per batch
259      * @param queryCriteria where clause
260      * @param report will be updated when a problem is found
261      * @param lastSeqNumber will be updated to the last sequence number processed in this subset
262      * @param nodeId identifier of which node that claims to have written this data
263      * @return the log entries we fetched from the database so the caller may export these
264      * @throws DatabaseProtectionException if the intregrity verification fails for one of the entries in the batch during fetch
265      */
266 	private List<AuditRecordData> verifyLogsIntegritySubset(final int startIndex, final int max, final QueryCriteria queryCriteria, final AuditLogValidationReport report, final Holder<Long> lastSeqNumber, final String nodeId) throws DatabaseProtectionException {
267 		final List<AuditRecordData> queryResult = internalSelectAuditLogs(startIndex, max, queryCriteria);	// Might throw DatabaseProtectionException
268 		// Loop through results and verify that the sequence order is correct
269 		for (int i=0; i<queryResult.size(); i++) {
270 			final long currentSeqNumber = queryResult.get(i).getSequenceNumber().longValue();
271 			if (currentSeqNumber != lastSeqNumber.get().longValue() + 1) {
272 				if (log.isDebugEnabled()) {
273 					log.debug("Log verification failure for log on node " + nodeId + ". Missing entry. Last sequenceNumber was " + lastSeqNumber + " and current is " + currentSeqNumber);
274 				}
275 				// Add to report
276 				report.warn(new AuditLogReportElem(lastSeqNumber.get(), Long.valueOf(currentSeqNumber), "missing log with sequence number " + (lastSeqNumber.get().longValue() + 1) + " on nodeId " + nodeId));
277 			}
278 			lastSeqNumber.set(Long.valueOf(currentSeqNumber));
279 		}
280 		return queryResult;
281 	}

283 	/** Log the outcome of the verification to the secure audit log based on the supplied number of errors. 
284 	 * @param errors Errors
285 	 * @param timestamp  time
286 	 * @param token Authorization */
287 	private void logVerificationResult(final int errors, final Date timestamp, final AuthenticationToken token) {
288     	final Map<String, Object> details = new LinkedHashMap<String, Object>();
289     	details.put("timestamp", FastDateFormat.getInstance(ValidityDate.ISO8601_DATE_FORMAT, TimeZone.getTimeZone("GMT")).format(timestamp));
290     	EventStatus status = EventStatus.SUCCESS;
291     	if (errors > 0) {
292     		status = EventStatus.FAILURE;
293     		details.put("errors", errors);
294     	}
295     	securityEventsLogger.log(EventTypes.LOG_VERIFY, status, ModuleTypes.SECURITY_AUDIT, ServiceTypes.CORE, token.toString(), null, null, null, details);
296 	}

318 	 * Build a JPA Query from the supplied queryStr and criteria.
319 	 * Optionally using startIndex and resultLimit (used if >0).
320 	 * @param entityManager EM
321 	 * @param queryStr Query
322 	 * @param criteria Criteria
323 	 * @param startIndex Index
324 	 * @param resultLimit Max results
325 	 * @return JPA query
326 	 */
327     private Query buildConditionalQuery(final EntityManager entityManager, final String queryStr, final QueryCriteria criteria, final int startIndex, final int resultLimit) {
328         Query query = null;
329         if (criteria == null) {
330             query = entityManager.createQuery(queryStr);
331         } else {
332             QueryGenerator generator = QueryGenerator.generator(AuditRecordData.class, criteria, "a");
333             final String conditions = generator.generate();
334             query = entityManager.createQuery(queryStr + conditions);
335             for (final String key : generator.getParameterKeys()) {
336                 final Object param = generator.getParameterValue(key);
337                 query.setParameter(key, param);
338             }
339         }
340         if (resultLimit > 0) {
341             query.setMaxResults(resultLimit);
342         }
343         if (startIndex > 0) {
344         	query.setFirstResult(startIndex-1);
345         }
346         return query;
347     }
