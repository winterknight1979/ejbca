35  /**
36   * An alternative implementation of the SecurityEventsLogger interface. It
37   * handles the creation of a signed log for an event.
38   *
39   * <p>This was created to evaluate the performance of using database integrity
40   * protection instead of custom code for log singing.
41   *
42   * @version $Id: IntegrityProtectedLoggerSessionBean.java 24600 2016-10-31
43   *     12:01:55Z jeklund $
44   */
45  @Stateless
46  @TransactionAttribute(TransactionAttributeType.REQUIRED)
47  public class IntegrityProtectedLoggerSessionBean
48      implements IntegrityProtectedLoggerSessionLocal {
49  
50      /** Logger. */
51    private static final Logger LOG =
52        Logger.getLogger(IntegrityProtectedLoggerSessionBean.class);
53  
54    /** EM. */
55    @PersistenceContext(unitName = CesecoreConfiguration.PERSISTENCE_UNIT)
56    private EntityManager entityManager;
57  
58    /** Setup. */
59    @PostConstruct
60    public void postConstruct() {
61      CryptoProviderTools.installBCProviderIfNotAvailable();
62    }
63  
64    /**
65     * Initialization of the log sequence number in combination with nodeId should
66     * be performed exactly once.
67     *
68     * <p>This callback will be invoked on the first call to
69     * NodeSequenceHolder.getNext(...) to perform this initialization.
70     *
71     * <p>In this callback implementation, the nodeId is first read from the
72     * configuration (which may default to reading the current hostname from the
73     * system). This hostname is then passed to the next method to figure out what
74     * the highest present sequenceNumber for this nodeId is in the database (e.g.
75     * last write before shutting down).
76     */
77    private final NodeSequenceHolder.OnInitCallBack sequenceHolderInitialization =
78        new NodeSequenceHolder.OnInitCallBack() {
79          @Override
80          public String getNodeId() {
81            return CesecoreConfiguration.getNodeIdentifier();
82          }
83  
84          @Override
85          public long getMaxSequenceNumberForNode(final String nodeId) {
86            // Get the latest sequenceNumber from last run from the database..
87            final Query query =
88                entityManager.createQuery(
89                    "SELECT MAX(a.sequenceNumber) FROM AuditRecordData a WHERE"
90                        + " a.nodeId=:nodeId");
91            query.setParameter("nodeId", nodeId);
92            return QueryResultWrapper.getSingleResult(query, Long.valueOf(-1))
93                .longValue();
94          }
95        };
96  
97    @Override
98    @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
99    // Always persist audit log
100   public void log(
101       final TrustedTime trustedTime,
102       final EventType eventType,
103       final EventStatus eventStatus,
104       final ModuleType module,
105       final ServiceType service,
106       final String authToken,
107       final String customId,
108       final String searchDetail1,
109       final String searchDetail2,
110       final Map<String, Object> additionalDetails,
111       final Properties properties)
112       throws AuditRecordStorageException {
113     if (LOG.isTraceEnabled()) {
114       LOG.trace(
115           String.format(
116               ">log:%s:%s:%s:%s:%s:%s",
117               eventType,
118               eventStatus,
119               module,
120               service,
121               authToken,
122               additionalDetails));
123     }
124     try {
125       final Long sequenceNumber =
126           NodeSequenceHolder.INSTANCE.getNext(sequenceHolderInitialization);
127       // Make sure to use the Node Identifier that this log sequence was
128       // initialized with (for example hostnames reported by the system could
129       // change)
130       final String nodeId = NodeSequenceHolder.INSTANCE.getNodeId();
131       final Long timeStamp = Long.valueOf(trustedTime.getTime().getTime());
132       final AuditRecordData auditRecordData =
133           new AuditRecordData(
134               nodeId,
135               sequenceNumber,
136               timeStamp,
137               eventType,
138               eventStatus,
139               authToken,
140               service,
141               module,
142               customId,
143               searchDetail1,
144               searchDetail2,
145               additionalDetails);
146       entityManager.persist(auditRecordData);
147     } catch (Exception e) {
148       LOG.error(e.getMessage(), e);
149       throw new AuditRecordStorageException(e.getMessage(), e);
150     } finally {
151       if (LOG.isTraceEnabled()) {
152         LOG.trace("<log");
153       }
154     }
155   }
156 }
