34  	/**
35       * Return the DN of the requested user.
36       * 
37  	 *@param globalConfiguration contains Active Directory configuration for autenrollment used in this method
38  	 *@param usernameShort is the SAMUserAccountName e.g. "Administrator" 
39  	 * @return String
40  	 */
41  	public static String getUserDNFromActiveDirectory(GlobalConfiguration globalConfiguration, String usernameShort) {
42  		String activeDirectoryServer = globalConfiguration.getAutoEnrollADServer();
43  		int activeDirectoryPort = globalConfiguration.getAutoEnrollADPort();
44  		boolean useSSLConnection = globalConfiguration.getAutoEnrollSSLConnection();	
45  		String activeDirectoryUserDN = globalConfiguration.getAutoEnrollConnectionDN();
46  		String activeDirectoryUserPassword = globalConfiguration.getAutoEnrollConnectionPwd();
47  		String userSearchBaseDN = globalConfiguration.getAutoEnrollBaseDNUser();
48  		return ActiveDirectoryTools.getSubjectDNFromUserInAD(activeDirectoryServer, activeDirectoryPort, useSSLConnection, activeDirectoryUserDN, activeDirectoryUserPassword,
49  				userSearchBaseDN, usernameShort);
50  	}
51  	
52  	/**
53       * Return the DN of the requested user.
54       * 
55       * @param serverAddress host of the Active Directory server
56  	 * @param port can be set to 0 to use the default port (389 or 636 for SSL)
57       * @param useSSL if the AD can handle this
58       * @param connectionDN is the connectors DN. E.g. "cn=Administrator,cn=Users,dc=org,dc=local"
59       * @param connectionPassword is the connectors password
60       * @param baseDN is base DN of where to look for the user. E.g. "cn=Users,dc=org,dc=local"
61       * @param username is the SAMAccountName for the user to look for
62       * @return the full DN of the user
63       */
64  	static public String getSubjectDNFromUserInAD(String serverAddress, int port, boolean useSSL,
65  			String connectionDN, String connectionPassword, String baseDN, String username) {
66  		String requestedDN = null;
67  		LDAPConnection lc = getNewConnection(serverAddress, port, useSSL, connectionDN, connectionPassword);
68  		if (lc == null) {
69  			return null;
70  		}
71  		LDAPEntry entry = null;
72  		try {
73  			String searchFilter = "(&(objectClass=person)(sAMAccountName=" + username + "))";
74  	        String[] attrs = { LDAPConnection.NO_ATTRS };
75  	        // Search recursively, but don't return any attributes for found objects
76  			LDAPSearchResults searchResults = lc.search(baseDN, LDAPConnection.SCOPE_SUB, searchFilter, attrs, true);
77  			if (searchResults.hasMore()) {
78  				// Re-read the object to get the attributes now
79  				requestedDN = searchResults.next().getDN();
80  				// List all props just for fun.. (TODO: Remove this..)
81  				entry = lc.read(requestedDN);
82  				if (entry != null) {
83  					@SuppressWarnings("unchecked")
84                      Iterator<LDAPAttribute> iter = entry.getAttributeSet().iterator();
85  					while (iter.hasNext()) {
86  						log.info(".. " + iter.next().toString().replaceAll("[^A-Za-z]", ""));
87  					}
88  				}
89  			} else {
90  				log.info("No matches found using filter: '" +searchFilter + "'.");
91  			}
92  		} catch (LDAPException e) {
93  			if (e.getResultCode() == LDAPException.NO_SUCH_OBJECT) {
94  				log.info("No such entry exist.");
95  			} else {
96  				log.error("Unknown AD error", e);
97  			}
98  		} finally {
99  			// disconnect with the server
100 			disconnect(lc);
101 		}
102 		return requestedDN;
103 	}
104 
105 	/**
106 	 * Create new LDAP connection to Active Directory server.
107 	 * @param serverAddress Server
108 	 * @param port can be set to 0 to use the default port (389 or 636 for SSL)
109 	 * @param useSSL bool
110 	 * @param connectionDN DN
111 	 * @param connectionPassword PWD
112 	 * @return connection
113 	 */
114 	private static LDAPConnection getNewConnection(String serverAddress, int port, boolean useSSL, String connectionDN, String connectionPassword) {
115 		LDAPConnection lc = null;
116 		int ldapPort = port;
117 		if (useSSL) {
118 			lc = new LDAPConnection(new LDAPJSSESecureSocketFactory());
119 			if (ldapPort == 0) {
120 				ldapPort = LDAPConnection.DEFAULT_SSL_PORT;	// Port 636
121 			}
122 		} else {
123 			lc = new LDAPConnection();
124 			if (ldapPort == 0) {
125 				ldapPort = LDAPConnection.DEFAULT_PORT;	// Port 389
126 			}
127 		}
128 		try {
129 			// connect to the server
130 			lc.connect(serverAddress, ldapPort);
131 			// authenticate to the server
132 			lc.bind(LDAPConnection.LDAP_V3, connectionDN, connectionPassword.getBytes());
133 		} catch (LDAPException e) {
134 			log.error("Error during AD bind", e);
135 			disconnect(lc);
136 			lc = null;
137 		}
138 		return lc;
139 	}
140 
141 	/**
142 	 * Clsoe and clean up existing connection.
143 	 * @param lc connection
144 	 */
145 	private static void disconnect(LDAPConnection lc) {
146 		try {
147 			lc.disconnect();
148 		} catch (LDAPException e) {
149 			log.error("Error during AD disconnect", e);
150 		}
151 	}
