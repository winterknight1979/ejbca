234     public void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
235 		ServletDebug debug = new ServletDebug(request, response);
236 		boolean usekeyrecovery = false;
237 
238 		RequestHelper.setDefaultCharacterEncoding(request);
239 		String iErrorMessage = null;
240 		try {
241 			setParameters(request);
242 
243             /*********************************************************************
244              ** If parameters are not set by Set... they must be
245              ** provided by request
246              *********************************************************************/
247             if (username==null) {
248                 username = StringUtils.strip(getParameter("user"));
249             }
250             if (password==null) {
251                 password = getParameter("password");
252             }
253             if (openvpn==null) {
254                 openvpn = getParameter("openvpn");
255             }
256             if (certprofile==null) {
257                 certprofile = getParameter("certprofile");
258             }
259             if (keyalgstring==null && keylengthstring==null) {
260             	// For token generation the option comes in the form "KEYALGORITHM_KEYSPEC"
261                 final String tokenKeySpec = getParameter("tokenKeySpec");
262                 if (tokenKeySpec!=null) {
263                     final String[] tokenKeySpecSplit = tokenKeySpec.split("_");
264                     if (tokenKeySpecSplit.length==2) {
265                         keyalgstring = tokenKeySpecSplit[0];
266                         keylengthstring = tokenKeySpecSplit[1];
267                     }
268                 }
269             }
270             if (keylengthstring==null) {
271                 keylengthstring = getParameter("keylength");
272             }
273             if (keyalgstring==null) {
274                 keyalgstring = getParameter("keyalg");
275             }
276             // If nothing has been set by setKeyLengthString and nothing is received by request parameters use default value of keylength
277             if (keylengthstring != null) {
278                 keylength = keylengthstring;
279             }
280             // If nothing has been set by setKeyAlgString and nothing is received by request parameters use default value of keyalg
281             if (keyalgstring != null) {
282                 keyalg = keyalgstring;
283             }
284 
285 			CertificateResponseType resulttype = CertificateResponseType.UNSPECIFIED;
286 			if(getParameter("resulttype") != null) {
287 				resulttype = CertificateResponseType.fromNumber(getParameter("resulttype")); // Indicates if certificate or PKCS7 should be returned on manual PKCS10 request.
288 			}
289 
290 			String classid = "clsid:127698e4-e730-4e5c-a2b1-21490a70c8a1\" CODEBASE=\"/CertControl/xenroll.cab#Version=5,131,3659,0";
291 
292 			if ((getParameter("classid") != null) &&
293 					!getParameter("classid").equals("")) {
294 				classid = getParameter("classid");
295 			}
296 
297             final AuthenticationToken administrator = new AlwaysAllowLocalAuthenticationToken(new PublicWebPrincipal("RequestInstance", request.getRemoteAddr()));
298 
299 			RequestHelper helper = new RequestHelper(administrator, debug);
300 
301 			log.info(intres.getLocalizedMessage("certreq.receivedcertreq", username, request.getRemoteAddr()));
302 			debug.print("Username: " + HTMLTools.htmlescape(username));
303 
304 			// Check user
305 			int tokentype = SecConst.TOKEN_SOFT_BROWSERGEN;
306 
307 			usekeyrecovery = ((GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID)).getEnableKeyRecovery();
308 
309 			EndEntityInformation data = endEntityAccessSession.findUser(administrator, username);
310 
311 			if (data == null) {
312 				throw new ObjectNotFoundException();
313 			}
314 
315 			boolean savekeys = data.getKeyRecoverable() && usekeyrecovery &&  (data.getStatus() != EndEntityConstants.STATUS_KEYRECOVERY);
316 			boolean loadkeys = (data.getStatus() == EndEntityConstants.STATUS_KEYRECOVERY) && usekeyrecovery;
317 
318 			int endEntityProfileId = data.getEndEntityProfileId();
319 			int certificateProfileId = data.getCertificateProfileId();
320 			EndEntityProfile endEntityProfile = endEntityProfileSession.getEndEntityProfile(endEntityProfileId);
321 			boolean reusecertificate = endEntityProfile.getReUseKeyRecoveredCertificate();
322 			// Set a new certificate profile, if we have requested one specific
323 			if (StringUtils.isNotEmpty(certprofile)) {
324 				boolean clearpwd = StringUtils.isNotEmpty(data.getPassword());
325 				int id = certificateProfileSession.getCertificateProfileId(certprofile);
326 				// Change the value if there exists a certprofile with the requested name, and it is not the same as
327 				// the one already registered to be used by default
328                 if (id > 0) {
329 					if (id != certificateProfileId) {
330 						// Check if it is in allowed profiles in the entity profile
331 						final Collection<Integer> ids = endEntityProfile.getAvailableCertificateProfileIds();
332 						if (ids.contains(id)) {
333 							data.setCertificateProfileId(id);
334 							// This admin can be the public web user, which may not be allowed to change status,
335 							// this is a bit ugly, but what can a man do...
336 							AuthenticationToken tempadmin = new AlwaysAllowLocalAuthenticationToken(new UsernamePrincipal("RequestInstance"+request.getRemoteAddr()));
337 							endEntityManagementSession.changeUser(tempadmin, data, clearpwd);
338 						} else {
339 							String defaultCertificateProfileName = certificateProfileSession.getCertificateProfileName(certificateProfileId);
340 							log.info(intres.getLocalizedMessage("certreq.badcertprofile", certprofile, defaultCertificateProfileName));
341 						}
342 					}
343 				} else {
344 					String defaultCertificateProfileName = certificateProfileSession.getCertificateProfileName(certificateProfileId);
345 					log.info(intres.getLocalizedMessage("certreq.nosuchcertprofile", certprofile, defaultCertificateProfileName));
346 				}
347 			}
348 
349 			// get users Token Type.
350 			tokentype = data.getTokenType();
351 			if(tokentype == SecConst.TOKEN_SOFT_P12){
352 			    // If the user is configured for a server generated token, but submitted a CSR, it is most likely an administrative error.
353 			    // The RA admin should probably have set token type usergenerated instead.
354 			    if (hasCSRInRequest()) {
355 			        throw new IncomatibleTokenTypeException();
356 			    }
357                 KeyStore ks = keyStoreCreateSession.generateOrKeyRecoverToken(administrator, username, password, data.getCAId(), keylength, keyalg,
358                         null, null, false, loadkeys, savekeys, reusecertificate, endEntityProfileId);
359 				if (StringUtils.equals(openvpn, "on")) {
360 					sendOpenVPNToken(ks, username, password, response);
361 				} else {
362 					sendP12Token(ks, username, password, response);
363 				}
364 			}
365 			if(tokentype == SecConst.TOKEN_SOFT_JKS){
366                 // If the user is configured for a server generated token, but submitted a CSR, it is most likely an administrative error.
367                 // The RA admin should probably have set token type usergenerated instead.
368                 if (hasCSRInRequest()) {
369                     throw new IncomatibleTokenTypeException();
370                 }
371                 KeyStore ks = keyStoreCreateSession.generateOrKeyRecoverToken(administrator, username, password, data.getCAId(), keylength, keyalg,
372                         null, null, true, loadkeys, savekeys, reusecertificate, endEntityProfileId);
373 				sendJKSToken(ks, username, password, response);
374 			}
375 			if(tokentype == SecConst.TOKEN_SOFT_PEM){
376                 // If the user is configured for a server generated token, but submitted a CSR, it is most likely an administrative error.
377                 // The RA admin should probably have set token type usergenerated instead.
378                 if (hasCSRInRequest()) {
379                     throw new IncomatibleTokenTypeException();
380                 }
381                 KeyStore ks = keyStoreCreateSession.generateOrKeyRecoverToken(administrator, username, password, data.getCAId(), keylength, keyalg,
382                         null, null, false, loadkeys, savekeys, reusecertificate, endEntityProfileId);
383 				sendPEMTokens(ks, username, password, response);
384 			}
385 			if(tokentype == SecConst.TOKEN_SOFT_BROWSERGEN){
386 
387 				// first check if it is a Firefox request,
388 				if (getParameter("keygen") != null) {
389 					byte[] reqBytes=getParameter("keygen").getBytes();
390 					if ((reqBytes != null) && (reqBytes.length>0)) {
391 					    if (log.isDebugEnabled()) {
392 					        log.debug("Received NS request: "+new String(reqBytes));
393 					    }
394 						byte[] certs = helper.nsCertRequest(signSession, reqBytes, username, password);
395 						if (Boolean.valueOf(getParameter("showResultPage")) && !isCertIssuerThrowAwayCA(certs, username)) {
396 						  // Send info page that redirects to download URL that
397 						  // retrieves the new certificate from the database
398 						  RequestHelper.sendResultPage(certs, response, "true".equals(getParameter("hidemenu")), "netscape");
399 						} else {
400 						  // The certificate will not be stored in the database,
401 						  // so we must send the certificate while we still have it
402 						  RequestHelper.sendNewCertToNSClient(certs, response);
403 						}
404 					} else {
405 						throw new SignRequestException("No request bytes received.");
406 					}
407 				} else if ( (getParameter("pkcs10") != null) || (getParameter("PKCS10") != null) ) {
408 					// if not firefox, check if it's IE
409 					byte[] reqBytes = getParameter("pkcs10").getBytes();
410 					if (reqBytes == null) {
411 						reqBytes=getParameter("PKCS10").getBytes();
412 					}
413 					if ((reqBytes != null) && (reqBytes.length>0)) {
414                         if (log.isDebugEnabled()) {
415                             log.debug("Received IE request: "+new String(reqBytes));
416                         }
417 						byte[] b64cert=helper.pkcs10CertRequest(signSession, caSession, reqBytes, username, password, CertificateResponseType.ENCODED_PKCS7).getEncoded();
418 						debug.ieCertFix(b64cert);
419 						response.setContentType("text/html");
420 						RequestHelper.sendNewCertToIEClient(b64cert, response.getOutputStream(), servletContext, servletConfig.getInitParameter("responseTemplate"),classid);
421 					} else {
422 						throw new SignRequestException("No request bytes received.");
423 					}
424 				} else if ( ((getParameter("pkcs10req") != null) || (getParameter("pkcs10file") != null)) && resulttype != CertificateResponseType.UNSPECIFIED) {
425 					byte[] reqBytes = null;
426 					String pkcs10req = getParameter("pkcs10req");
427 					if (StringUtils.isEmpty(pkcs10req)) {
428 						// did we upload a file instead?
429                         if (log.isDebugEnabled()) {
430                             log.debug("No pasted request received, checking for uploaded file.");
431                         }
432 						pkcs10req = getParameter("pkcs10file");
433 						if (StringUtils.isNotEmpty(pkcs10req)) {
434 							// The uploaded file has been converted to a base64 encoded string
435 							reqBytes = Base64.decode(pkcs10req.getBytes());
436 
437 						}
438 					} else {
439 						reqBytes=pkcs10req.getBytes(); // The pasted request
440 					}
441 
442 					if ((reqBytes != null) && (reqBytes.length>0)) {
443 					    try {
444 					        pkcs10Req(request, response, username, password, resulttype, helper, reqBytes);
445 					    } catch(Exception exp) {
446 					        if(exp.getCause() instanceof NullPointerException) {
447 					            iErrorMessage = "Failed to parse request";
448 					        } else {
449 					            iErrorMessage = intres.getLocalizedMessage("certreq.failed", exp.getLocalizedMessage());
450 					        }
451                         }
452 					} else {
453 						throw new SignRequestException("No request bytes received.");
454 					}
455 				} else if ( ((getParameter("cvcreq") != null) || (getParameter("cvcreqfile") != null)) && resulttype != CertificateResponseType.UNSPECIFIED) {
456 					// It's a CVC certificate request (EAC ePassports)
457 					byte[] reqBytes = null;
458 					String req = getParameter("cvcreq");
459 					if (StringUtils.isEmpty(req)) {
460 						// did we upload a file instead?
461                         if (log.isDebugEnabled()) {
462                             log.debug("No pasted request received, checking for uploaded file.");
463                         }
464 						req = getParameter("cvcreqfile");
465 						if (StringUtils.isNotEmpty(req)) {
466 							// The uploaded file has been converted to a base64 encoded string
467 							reqBytes = Base64.decode(req.getBytes());
468 
469 						}
470 					} else {
471 						reqBytes=req.getBytes(); // The pasted request
472 					}
473 
474 					if ((reqBytes != null) && (reqBytes.length>0)) {
475                         if (log.isDebugEnabled()) {
476                             log.debug("Received CVC request: "+new String(reqBytes));
477                         }
478 						byte[] b64cert=helper.cvcCertRequest(signSession, reqBytes, username, password);
479 						CVCertificate cvccert = (CVCertificate) CertificateParser.parseCVCObject(Base64.decode(b64cert));
480 						String filename = "";
481 						CAReferenceField carf = cvccert.getCertificateBody().getAuthorityReference();
482 						if (carf != null) {
483 							String car = carf.getConcatenated();
484 							filename += car;
485 						}
486 						HolderReferenceField chrf = cvccert.getCertificateBody().getHolderReference();
487 						if (chrf != null) {
488 							String chr = chrf.getConcatenated();
489 							if (filename.length() > 0) {
490 								filename += "_";
491 							}
492 							filename +=chr;
493 						}
494 						if (filename.length() == 0) {
495 							filename = username;
496 						}
497                         if (log.isDebugEnabled()) {
498                             log.debug("Filename: "+filename);
499                         }
500 						if(resulttype == CertificateResponseType.BINARY_CERTIFICATE) {
501 							RequestHelper.sendBinaryBytes(Base64.decode(b64cert), response, "application/octet-stream", filename+".cvcert");
502 						}
503 						if(resulttype == CertificateResponseType.ENCODED_CERTIFICATE) {
504 							RequestHelper.sendNewB64File(b64cert, response, filename+".pem", CertTools.BEGIN_CERTIFICATE_WITH_NL, CertTools.END_CERTIFICATE_WITH_NL);
505 						}
506 					} else {
507 						throw new SignRequestException("No request bytes received.");
508 					}
509 				} else {
510 				    // throw general exception, will be caught below and all parameters printed.
511                     throw new Exception("No known request type received.");
512 				}
513 			}
514 		} catch (AuthStatusException ase) {
515 			iErrorMessage = intres.getLocalizedMessage("certreq.wrongstatus");
516         } catch (ObjectNotFoundException oe) {
517             // Same error message for user not found and wrong password
518             iErrorMessage = intres.getLocalizedMessage("ra.wrongusernameorpassword");
519             // But debug log the real issue if needed
520             if (log.isDebugEnabled()) {
521                 log.debug(intres.getLocalizedMessage("ra.errorentitynotexist", username));
522             }
523 		} catch (AuthLoginException ale) {
524 			iErrorMessage = intres.getLocalizedMessage("ra.wrongusernameorpassword");
525 			log.info(iErrorMessage + " - username: " + username);
526         } catch (IncomatibleTokenTypeException re) {
527             iErrorMessage = intres.getLocalizedMessage("certreq.csrreceivedforservergentoken");
528 		} catch (SignRequestException re) {
529 		    log.info(re.getMessage(), re);
530 			iErrorMessage = intres.getLocalizedMessage("certreq.invalidreq", re.getMessage());
531 		} catch (SignRequestSignatureException se) {
532 			String iMsg = intres.getLocalizedMessage("certreq.invalidsign");
533 			log.info(iMsg, se);
534 			debug.printMessage(iMsg);
535 			debug.printDebugInfo();
536 			return;
537 		} catch (ArrayIndexOutOfBoundsException | DecoderException ae) {
538 			iErrorMessage = intres.getLocalizedMessage("certreq.invalidreq");
539 		} catch (IllegalKeyException e) {
540 			iErrorMessage = intres.getLocalizedMessage("certreq.invalidkey", e.getMessage());
541 		} catch (CryptoTokenOfflineException ctoe) {
542 		    String ctoeMsg = ctoe.getMessage();
543 		    for (Throwable e = ctoe; e != null; e = e.getCause()) {
544 		        //We can not do "if (e instanceof sun.security.pkcs11.wrapper.PKCS11Exception)" here because these classes are not guaranteed to exist on all platforms
545                 if (e.getClass().getName().contains("PKCS11")) {
546 		            ctoeMsg = "PKCS11 error "+e.getMessage();
547 		            break;
548 		        }
549 		    }
550             iErrorMessage = intres.getLocalizedMessage("certreq.catokenoffline", ctoeMsg);
551 		} catch (AuthorizationDeniedException e) {
552 		    iErrorMessage = intres.getLocalizedMessage("certreq.authorizationdenied") + e.getLocalizedMessage();
553 		} catch(CertificateCreateException e) {
554 		    if (e.getErrorCode()!=null && e.getErrorCode().equals(ErrorCode.CERTIFICATE_WITH_THIS_SUBJECTDN_ALREADY_EXISTS_FOR_ANOTHER_USER)) {
555 		        iErrorMessage = e.getLocalizedMessage();
556 		    } else {
557 		        if (e.getErrorCode()!=null && e.getErrorCode().equals(ErrorCode.NOT_AUTHORIZED)) {
558 		            debug.print(e.getLocalizedMessage());
559 		            debug.printDebugInfo();
560 		        } else {
561 		            debug.takeCareOfException(e);
562 		            debug.printDebugInfo();
563 		        }
564 		    }
565 		} catch (Exception e) {
566 			Throwable e1 = e.getCause();
567 			if (e1 instanceof CryptoTokenOfflineException) {
568 				String iMsg = intres.getLocalizedMessage("certreq.catokenoffline", e1.getMessage());
569 				// this is already logged as an error, so no need to log it one more time
570 				debug.printMessage(iMsg);
571 				debug.printDebugInfo();
572 				return;
573 			} else {
574 				if (e1 == null) { e1 = e; }
575                 String iMsg = intres.getLocalizedMessage("certreq.errorgeneral", e1.getMessage());
576                 if (log.isDebugEnabled()) {
577                     log.debug(iMsg, e);
578                 }
579 				iMsg = intres.getLocalizedMessage("certreq.parameters", e1.getMessage());
580 				debug.print(iMsg + ":\n");
581 				@SuppressWarnings("unchecked")
582                 Set<String> paramNames = params.keySet();
583 				for(String name : paramNames) {
584 					String parameter = getParameter(name);
585                     if (!StringUtils.equals(name, "password")) {
586                         debug.print(HTMLTools.htmlescape(name) + ": '" + HTMLTools.htmlescape(parameter) + "'\n");
587                     } else {
588                         debug.print(HTMLTools.htmlescape(name) + ": <hidden>\n");
589                     }
590 				}
591 				// Don't print back details to the client, this is considered information leak
592 				//debug.takeCareOfException(e);
593 				debug.printDebugInfo();
594 			}
595 		}
596 		if (iErrorMessage != null) {
597             if (log.isDebugEnabled()) {
598                 log.debug(iErrorMessage);
599             }
600 			debug.printMessage(iErrorMessage);
601 			debug.printDebugInfo();
602 			return;
603 		}
604 	}

621     /**
622      * Determines whether the issuer of a certificate is a "throw away" CA, i.e.
623      * if it does not store certificates it issues in the database.
624      *
625      * @param certbytes DER encoded certificate.
626      * @param username user
627      * @return vool
628      * @throws CADoesntExistsException fail
629      * @throws CertificateException fail
630      */
631     private boolean isCertIssuerThrowAwayCA(final byte[] certbytes, final String username) throws CADoesntExistsException, CertificateException {
632         Certificate cert = CertTools.getCertfromByteArray(certbytes, Certificate.class);
633         String issuerDN = CertTools.getIssuerDN(cert);
634         int caid = issuerDN.hashCode();
635         CAInfo caInfo = caSession.getCAInfoInternal(caid);
636         EndEntityInformation endEntityInformation = endEntityAccessSession.findUser(username);
637         CertificateProfile certificateProfile = certificateProfileSession.getCertificateProfile(endEntityInformation.getCertificateProfileId());
638         return (!caInfo.isUseCertificateStorage() && !certificateProfile.getUseCertificateStorage()) || !certificateProfile.getStoreCertificateData();
639     }

641 	/**
642 	 * Method creating a Map of request values, designed to handle both
643 	 * regular x-encoded forms and multipart encoded upload forms.
644 	 *
645 	 * @param request
646 	 *            HttpServletRequest
647 	 * @throws FileUploadException
648 	 *             if multipart request can not be parsed
649 	 * @throws IOException
650 	 *             If input stream of uploaded object can not be read
651 	 */
652 	@SuppressWarnings("unchecked")
653     private void setParameters(HttpServletRequest request) throws FileUploadException, IOException {
654 		if (ServletFileUpload.isMultipartContent(request)) {
655 			params = new HashMap<String, String>();
656             final DiskFileItemFactory diskFileItemFactory = new DiskFileItemFactory();
657             diskFileItemFactory.setSizeThreshold(9999);
658             ServletFileUpload upload = new ServletFileUpload(diskFileItemFactory);
659 			upload.setSizeMax(REQUEST_MAX_SIZE);
660             List<FileItem> items = upload.parseRequest(request);
661 			Iterator<FileItem> iter = items.iterator();
662 			while (iter.hasNext()) {
663                 FileItem item = iter.next();
664 				if (item.isFormField()) {
665 					params.put(item.getFieldName(), item.getString());
666 				} else {
667 					InputStream is = item.getInputStream();
668 					byte[] bytes = FileTools.readInputStreamtoBuffer(is);
669 					params.put(item.getFieldName(), new String(Base64.encode(bytes)));
670 				}
671 			}
672 		} else {
673 			params = request.getParameterMap();
674 		}
675 	}

677 	private String getParameter(String param) {
678 		String ret = null;
679 		Object o = params.get(param);
680 		if (o != null) {
681 			if (o instanceof String) {
682 			    if ( ((String)o).length() > REQUEST_MAX_SIZE ) {
683 			        if (log.isDebugEnabled()) {
684 			            log.debug("Parameter '"+param+"' exceed size limit of "+REQUEST_MAX_SIZE);
685 			        }
686 			    } else {
687 			        ret = (String) o;
688 			    }
689 			} else if (o instanceof String[]) { // keygen is of this type
690 			    // for some reason...
691 			    String[] str = (String[]) o;
692 			    if ((str != null) && (str.length > 0)) {
693 			        if ( str[0].length() > REQUEST_MAX_SIZE ) {
694 			            if (log.isDebugEnabled()) {
695 			                log.debug("Parameter (first in list) '"+param+"' exceed size limit of "+REQUEST_MAX_SIZE);
696 			            }
697 			        } else {
698 			            ret = str[0];
699 			        }
700 			    }
701 			} else {
702 				log.debug("Can not cast object of type: " + o.getClass().getName());
703 			}
704 		}
705 		return ret;
706 	}

708 	private void pkcs10Req(HttpServletRequest request, HttpServletResponse response, String username, String password, CertificateResponseType resulttype,
709 			RequestHelper helper, byte[] reqBytes) throws Exception, IOException {
710         if (log.isDebugEnabled()) {
711             log.debug("Received PKCS10 request: " + new String(reqBytes));
712         }
713 		CertificateRequestResponse result = helper.pkcs10CertRequest(signSession, caSession, reqBytes, username, password, resulttype);
714 		byte[] b64data = result.getEncoded(); // PEM cert, cert-chain or PKCS7
715 		byte[] b64subject = result.getCertificate().getEncoded(); // always a PEM cert of the subject
716 		if (Boolean.valueOf(getParameter("showResultPage")) && !isCertIssuerThrowAwayCA(b64subject, username)) {
717             RequestHelper.sendResultPage(b64subject, response, "true".equals(getParameter("hidemenu")), resulttype);
718         } else {
719             switch (resulttype) {
720             case ENCODED_PKCS7:
721                 RequestHelper.sendNewB64File(b64data, response, username + ".pem", RequestHelper.BEGIN_PKCS7_WITH_NL, RequestHelper.END_PKCS7_WITH_NL);
722                 break;
723             case ENCODED_CERTIFICATE:
724                 RequestHelper.sendNewB64File(b64data, response, username + ".pem", CertTools.BEGIN_CERTIFICATE_WITH_NL,
725                         CertTools.END_CERTIFICATE_WITH_NL);
726                 break;
727             case ENCODED_CERTIFICATE_CHAIN:
728                 //Begin/end keys have already been set in the serialized object
729                 RequestHelper.sendNewB64File(b64data, response, username + ".pem", "", "");
730                 break;
731             default:
732                 log.warn("Unknown resulttype requested from pkcs10 request.");
733                 break;
734             }
735         }
736 	}

738 	/**
739 	 * method to create an install package for OpenVPN including keys and
740 	 * send to user. Contributed by: Jon Bendtsen,
741 	 * jon.bendtsen(at)laerdal.dk
742 	 * @param ks KS
743 	 * @param username User 
744 	 * @param kspassword KS
745 	 * @param out Out
746 	 * @throws Exception Fail 
747 	 */
748 	private void sendOpenVPNToken(KeyStore ks, String username, String kspassword, HttpServletResponse out) throws Exception {
749 		ByteArrayOutputStream buffer = new ByteArrayOutputStream();
750 		ks.store(buffer, kspassword.toCharArray());
751 
752 		String tempDirectory = System.getProperty("java.io.tmpdir");
753 		File fout = new File(tempDirectory + System.getProperty("file.separator") + username + ".p12");
754 		String IssuerDN = null;
755 		String SubjectDN = null;
756 		try (FileOutputStream certfile = new FileOutputStream(fout)) {
757 		    Enumeration<String> en = ks.aliases();
758 		    String alias = en.nextElement();
759 		    // Then get the certificates
760 		    Certificate[] certs = KeyTools.getCertChain(ks, alias);
761 		    // The first one (certs[0]) is the users cert and the last
762 		    // one (certs [certs.lenght-1]) is the CA-cert
763 		    X509Certificate x509cert = (X509Certificate) certs[0];
764 		    IssuerDN = x509cert.getIssuerDN().toString();
765 		    SubjectDN = x509cert.getSubjectDN().toString();
766 
767 		    // export the users certificate to file
768 		    buffer.writeTo(certfile);
769 		    buffer.flush();
770 		    buffer.close();
771 		    certfile.close();
772 		}
773 		// run shell script, which will also remove the created files
774 		// parameters are the username, IssuerDN and SubjectDN
775 		// IssuerDN and SubjectDN will be used to select the right
776 		// openvpn configuration file
777 		// they have to be written to stdin of the script to support
778 		// spaces in the username, IssuerDN or SubjectDN
779 		Runtime rt = Runtime.getRuntime();
780 		if (rt == null) {
781 			log.error(intres.getLocalizedMessage("certreq.ovpntnoruntime"));
782 		} else {
783 			final String script = WebConfiguration.getOpenVPNCreateInstallerScript();
784 			Process p = rt.exec(script);
785 			if (p == null) {
786 				log.error(intres.getLocalizedMessage("certreq.ovpntfailedexec", script));
787 			} else {
788 				OutputStream pstdin = p.getOutputStream();
789 				PrintStream stdoutp = new PrintStream(pstdin);
790 				stdoutp.println(username);
791 				stdoutp.println(IssuerDN);
792 				stdoutp.println(SubjectDN);
793 				stdoutp.flush();
794 				stdoutp.close();
795 				pstdin.close();
796 				int exitVal = p.waitFor();
797 				if (exitVal != 0) {
798 					log.error(intres.getLocalizedMessage("certreq.ovpntexiterror", exitVal));
799 				} else {
800                     if (log.isDebugEnabled()) {
801                         log.debug(intres.getLocalizedMessage("certreq.ovpntexiterror", exitVal));
802                     }
803 				}
804 			}
805 		}
806 
807 		// we ought to check if the script was okay or not, but in a little
808 		// while we will look for the openvpn-gui-install-$username.exe
809 		// and fail there if the script failed. Also, one could question
810 		// what to do if it did fail, serve the user the certificate?
811 
812 		// sending the OpenVPN windows installer
813 		String filename = "openvpn-gui-install-" + username + ".exe";
814 		File fin = new File(tempDirectory + System.getProperty("file.separator") + filename);
815 		FileInputStream vpnfile = new FileInputStream(fin);
816 		out.setContentType("application/x-msdos-program");
817 		out.setHeader("Content-disposition", "filename=\"" + StringTools.stripFilename(filename) + "\"");
818 		out.setContentLength((int)fin.length());
819 		OutputStream os = out.getOutputStream();
820 		byte[] buf = new byte[4096];
821 		int bytes = 0;
822 		while ((bytes = vpnfile.read(buf)) != -1) {
823 			os.write(buf, 0, bytes);
824 		}
825 		vpnfile.close();
826 		// delete OpenVPN windows installer, the script will delete cert.
827 		fin.delete();
828 		out.flushBuffer();
829 	} // sendOpenVPNToken
