13  package org.ejbca.ui.web.pub;
14  
15  import java.io.IOException;
16  
17  import javax.servlet.Filter;
18  import javax.servlet.FilterChain;
19  import javax.servlet.FilterConfig;
20  import javax.servlet.ServletException;
21  import javax.servlet.ServletRequest;
22  import javax.servlet.ServletResponse;
23  import javax.servlet.annotation.WebFilter;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.apache.log4j.Logger;
28  
29  /**
30   * Filter that prevents a browser to cache everything except resources like java script, css and images.
31   * It's part of JSF best practices.
32   * 
33   * @version $Id: NoCacheFilter.java 27844 2018-01-10 15:14:34Z henriks $
34   */
35  @WebFilter(filterName = "NoCacheFilter", urlPatterns = {"/*"})
36  public class NoCacheFilter implements Filter {
37  
38      private static final Logger log = Logger.getLogger(NoCacheFilter.class);
39  
40      @Override
41      public void destroy() {
42      }
43  
44      /**
45       * @param request Http request
46       * @return true for the request URIs that should be cached, false otherwise.
47       */
48      private boolean cacheURI(HttpServletRequest request) {
49          String requestUri = request.getRequestURI();
50          return requestUri.startsWith(request.getContextPath() + "/styles") || 
51                 requestUri.startsWith(request.getContextPath() + "/scripts") ||
52                 requestUri.startsWith(request.getContextPath() + "/images");
53      }
54  
55      @Override
56      public void doFilter(ServletRequest req, ServletResponse res, FilterChain chain) throws IOException, ServletException {
57          HttpServletRequest request = (HttpServletRequest) req;
58          HttpServletResponse response = (HttpServletResponse) res;
59  
60          if (!cacheURI(request)) {
61              response.setHeader("Cache-Control", "no-cache, no-store, must-revalidate"); // HTTP 1.1.
62              response.setHeader("Pragma", "no-cache"); // HTTP 1.0.
63              response.setDateHeader("Expires", 0); // Proxies.
64          }
65  
66          chain.doFilter(req, res);
67      }
68  
69      @Override
70      public void init(FilterConfig arg0) throws ServletException {
71          log.debug("NoCacheFilter: init");
72      }
73  
74  }
