97      /**
98       * Finds all properties matching web.selfreg.certtypes.KEY.description=VALUE
99       * and returns a map with these keys and values.
100      * @return map
101      */
102     public Map<String,String> getCertificateTypes() {
103         Map<String,String> certtypes = new HashMap<>();
104         for (Entry<Object,Object> entry : EjbcaConfigurationHolder.getAsProperties().entrySet()) {
105             final Object k = entry.getKey();
106             final Object v = entry.getValue();
107             if (k instanceof String && v instanceof String) {
108                 String key = (String)k;
109                 if (key.matches("web\\.selfreg\\.certtypes\\.([^.]+)\\.description")) {
110                     final String name = key.split("\\.")[3];
111                     // Check if the certificate exists
112                     final String eeprofname = getCertTypeInfoOptional(name, "eeprofile", null);
113                     final String certprofname = getCertTypeInfoOptional(name, "certprofile", null);
114                     try {
115                         if (eeprofname == null || certprofname == null) {
116                             if (log.isDebugEnabled()) {
117                                 log.debug("Ignoring certificate type "+name+" due to missing eeprofile/certprofile properties");
118                             }
119                             continue;
120                         }
121                         endEntityProfileSession.getEndEntityProfileId(eeprofname);
122                         if (certificateProfileSession.getCertificateProfileId(certprofname) == 0) {
123                             if (log.isDebugEnabled()) {
124                                 log.debug("Ignoring certificate type "+name+" due to missing certificate profile '"+certprofname+"'");
125                             }
126                             continue; // Ignore this certificate
127                         }
128                     } catch (EndEntityProfileNotFoundException e) {
129                         if (log.isDebugEnabled()) {
130                             log.debug("Ignoring certificate type "+name+" due to missing end-entity profile '"+eeprofname+"'");
131                         }
132                         continue; // Ignore this certificate
133                     }
134                     
135                     certtypes.put(name, (String)v);
136                 }
137             }
138         }
139         return certtypes;
140     }

307     /**
308      * Checks if the Certificate Profile and End Entity Profile fulfills the self-registration requirements.
309      * 
310      * @return true if CP and EEP fulfills requirements
311      */
312     public boolean validCpAndEep() {
313         boolean valid = true;
314         CertificateProfile certificateProfile = certificateProfileSession.getCertificateProfile(getCertificateProfileId());
315         EndEntityProfile endEntityProfile = getEndEntityProfile();
316         
317         // Check that the Certificate Profile or the Default CA has an Approval Profile set for Add/Edit End Entity
318         final AuthenticationToken admin = new AlwaysAllowLocalAuthenticationToken(new PublicWebPrincipal(remoteAddress));
319         ApprovalProfile approvalProfile = null;
320         try {
321             CAInfo caInfo = ejbLocalHelper.getCaSession().getCAInfo(admin, endEntityProfile.getDefaultCA());
322             approvalProfile = ejbLocalHelper.getApprovalProfileSession().getApprovalProfileForAction(ApprovalRequestType.ADDEDITENDENTITY, caInfo, certificateProfile);
323         } catch (AuthorizationDeniedException e) {
324             log.info("Self Registration: Could not get Approval Profile", e);
325         }
326         if (approvalProfile == null) {
327             log.info("Self Registration: An Approval Profile must be set for 'Add/Edit End Entity' in the Certificate Profile/Default CA for Certificate type: " + getCertType());
328             valid = false;
329         }
330         // Check that the End Entity Profile has End Entity E-mail enabled
331         if (!endEntityProfile.getUse(EndEntityProfile.EMAIL, 0)) {
332             log.info("Self Registration: End Entity E-mail must be enabled in the End Entity Profile for Certificate type: " + getCertType());
333             valid = false;
334         }
335         // Check that the End Entity Profile has Auto-generated password enabled
336         if (!endEntityProfile.useAutoGeneratedPasswd()) {
337             log.info("Self Registration: Password must be Auto-generated in the End Entity Profile for Certificate type: " + getCertType());
338             valid = false;
339         }
340         return valid;
341     }

347     private void checkCertEEProfilesExist() {
348         String eeprofName = getCertTypeInfo(certType, "eeprofile");
349         if (eeprofName != null && endEntityProfileSession.getEndEntityProfile(eeprofName) == null) {
350             internalError("End entity profile "+eeprofName+" does not exist. Please ask the administrator to check the web.selfreg.certtypes."+certType+".eeprofile configuration");
351         }
352         
353         String certprofName = getCertTypeInfo(certType, "certprofile");
354         if (certprofName != null && certificateProfileSession.getCertificateProfile(certprofName) == null) {
355             internalError("Certificate profile "+certprofName+" does not exist. Please ask the administrator to check the web.selfreg.certtypes."+certType+".certprofile configuration");
356         }
357     }

370     /** Appends a field to a Subject DN, Subject Alternative Name, or Subject Directory Attributes. 
371      * @param dn DN
372      * @param dnName Name
373      * @param value Val
374      * @return DN  */
375     private static String appendToDN(final String dn, final String dnName, final String value) {
376         final String field = LDAPDN.escapeRDN(dnName.toUpperCase(Locale.ROOT) + "=" + value);
377         if (dn.isEmpty()) {
378             return field;
379         } else {
380             return dn + "," + field;
381         }
382     }

384     /**
385      * Reads all parameters from the request. Used to receive the parameters from both step 1 and step 2.
386      * @param request req
387      */
388     public void initialize(final HttpServletRequest request) {
389         if (!"POST".equalsIgnoreCase(request.getMethod())) {
390             internalError("Internal error: Invalid request method.");
391         }
392         
393         certType = request.getParameter("certType");
394         
395         checkConfig();
396         checkCertEEProfilesExist();
397         eeprofile = getEndEntityProfile();
398         if (eeprofile == null && !errors.isEmpty()) {
399             return;
400         }
401         String usernameMapping = getUsernameMapping();
402 
403         // Get all fields
404         final Map<String,String> allFields = new HashMap<>();
405         @SuppressWarnings("rawtypes")
406         Enumeration en = request.getParameterNames();
407         while (en.hasMoreElements()) {
408             String key = (String)en.nextElement();
409             String value = request.getParameter(key).trim();
410             
411             String id = key.replaceFirst("^[a-z]+_", ""); // format is e.g. dnfield_cn, altnamefield_123 or dirattrfield_123 
412             if (key.startsWith("dnfield_")) {
413                 if (!value.isEmpty()) {
414                     String dnName = DNFieldDescriber.extractSubjectDnNameFromId(eeprofile, id);
415                     subjectDN = appendToDN(subjectDN, dnName, value);
416                     allFields.put(dnName.toUpperCase(Locale.ROOT), value);
417                 }
418             }
419             
420             if (key.startsWith("altnamefield_")) {
421                 if (!value.isEmpty()) {
422                     boolean valid = true;
423                     String domain = request.getParameter("domain_" + key);
424                     if (domain != null) {
425                         // The field is RFC 822 or MS UPN
426                         domain = domain.trim();
427                         if (!domain.isEmpty()) {
428                             value = value += "@" + domain;
429                         } else {
430                             // Got value but empty domain name
431                             valid = false;
432                             internalError("Incomplete field: " + request.getParameter("type_" + key));
433                         }
434                     }
435                     if (valid) {
436                         String altName = DNFieldDescriber.extractSubjectAltNameFromId(eeprofile, id);
437                         subjectAltName = appendToDN(subjectAltName, altName, value);
438                         allFields.put(altName.toUpperCase(Locale.ROOT), value);
439                     }
440                 }
441             }
442             
443             if (key.startsWith("dirattrfield_")) {
444                 if (!value.isEmpty()) {
445                     String dirAttr = DNFieldDescriber.extractSubjectDirAttrFromId(eeprofile, id);
446                     subjectDirAttrs = appendToDN(subjectDirAttrs, dirAttr, value);
447                     allFields.put(dirAttr.toUpperCase(Locale.ROOT), value);
448                 }
449             }
450         }
451         
452         // User account
453         email = request.getParameter("email");
454         String domain = request.getParameter("emaildomain");
455         if (domain != null && !email.isEmpty()) email += "@" + domain;
456         captcha = request.getParameter("code");
457         
458         String tokenStr = request.getParameter("tokenType");
459         tokenType = Integer.parseInt(tokenStr != null ? tokenStr : getDefaultTokenType());
460         if ("1".equals(request.getParameter("emailindn"))) {
461             subjectDN = appendToDN(subjectDN, "E", email);
462             allFields.put("E", email);
463         }
464         
465         if (request.getParameter("emailinaltname") != null) {
466             String id = request.getParameter("emailinaltname");
467             String altName = DNFieldDescriber.extractSubjectAltNameFromId(eeprofile, id);
468             subjectAltName = appendToDN(subjectAltName, altName, email);
469             allFields.put(altName, email);
470         }
471         
472         if (isUsernameVisible()) {
473             username = request.getParameter("username");
474         } else {
475             username = allFields.get(usernameMapping);
476             if (!allFields.isEmpty() && username == null) {
477                 internalError("DN field of usernamemapping doesn't exist: "+usernameMapping);
478             }
479         }
480         
481         remoteAddress = request.getRemoteAddr();
482         initialized = true;
483     }

485     private void checkFormFields() {
486         boolean cantDoCaptcha = false;
487         
488         if (certType == null || certType.isEmpty()) {
489             errors.add("Certificate type is not specified.");
490         }
491         
492         // User account
493         if (username == null || username.isEmpty()) {
494             errors.add("Username is not specified.");
495             if (isUsernameVisible()) cantDoCaptcha = true;
496         }
497         
498         if (email == null || !email.matches("[^@]+@.+")) {
499             errors.add("E-mail is not specified.");
500             if (!isUsernameVisible()) cantDoCaptcha = true;
501         }
502         
503         // Token type
504         if (!ArrayUtils.contains(getAvailableTokenTypes(), String.valueOf(tokenType))) {
505             errors.add("Token type is not allowed by end-entity profile.");
506         }
507 
508         String captchaField = isUsernameVisible() ? username : email;
509         
510         // The captcha simply is the last character of the name
511         if (!cantDoCaptcha && (captcha == null || !captcha.equalsIgnoreCase(captchaField.substring(captchaField.length()-1)))) {
512             errors.add("Captcha code is incorrect.");
513         }
514     }

542     /**
543      * Creates a approval request from the given information in the form.
544      * initialize() must have been called before this method is called.  
545      */
546     public void submit() {
547         if (!initialized) {
548             throw new IllegalStateException("initialize not called before submit");
549         }
550         
551         // Set up config for admingui (e.g. for e-mails to admins with links to it)
552         // This should be OK to do here and to do per request, since it just
553         // sets up some hard-coded config strings, etc.
554         globalConfiguration.initializeAdminWeb();
555         
556         checkFormFields();
557         final String usernamePrefix = getCertTypeInfoOptional(certType, "usernameprefix", null);
558         if (usernamePrefix != null) {
559             username = usernamePrefix + username;
560         }
561         
562         if (!errors.isEmpty()) {
563             return;
564         }
565         
566         int eeProfileId;
567         try {
568             eeProfileId = getEndEntityProfileId();
569         } catch (EndEntityProfileNotFoundException e) {
570             errors.add("Validation error: "+e.getMessage());
571             return;
572         }
573         final EndEntityProfile eeprofile = endEntityProfileSession.getEndEntityProfile(eeProfileId);
574         final int caid = eeprofile.getDefaultCA();
575         if (caid == -1) {
576             internalError("The end-entity profile "+getCertTypeInfo(certType, "eeprofile")+" for cert type "+certType+" does not have any default CA.");
577         }
578         
579         final int certProfileId = getCertificateProfileId();
580         
581         if (endEntityManagementSession.existsUser(username)) {
582             errors.add("A user with that name exists already");
583         }
584         
585         if (!errors.isEmpty()) {
586             return;
587         }
588         
589         final AuthenticationToken admin = new AlwaysAllowLocalAuthenticationToken(new PublicWebPrincipal(remoteAddress));
590         
591         EndEntityInformation endEntity = new EndEntityInformation(username, subjectDN, caid, subjectAltName, 
592                 null, EndEntityConstants.STATUS_NEW, new EndEntityType(EndEntityTypes.ENDUSER), eeProfileId, certProfileId,
593                 null,null, tokenType, 0, null);
594         if (eeprofile.getUse(EndEntityProfile.SENDNOTIFICATION, 0)) {
595             endEntity.setSendNotification(true);
596         }
597         assignDirAttrs(endEntity);
598         if (email != null) {
599             endEntity.setEmail(email);
600         }
601         
602         try {
603             endEntity = endEntityManagementSession.canonicalizeUser(endEntity);
604             if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
605                 eeprofile.doesUserFulfillEndEntityProfile(endEntity, false);
606                 
607             }
608         } catch (EjbcaException e) {
609             errors.add("Validation error: "+e.getMessage());
610             return;
611         } catch (EndEntityProfileValidationException e) {
612             errors.add("User information does not fulfill requirements: "+e.getMessage());
613             return;
614         }
615         
616         final CaSessionLocal casession = ejbLocalHelper.getCaSession();
617         CAInfo cainfo = casession.getCAInfoInternal(caid);
618         if(cainfo == null) {
619             errors.add("CA with ID " + caid + " does not exist.");
620             return;
621         }
622         final ApprovalProfile approvalProfile = getApprovalProfile(cainfo, certProfileId);        
623         // Add approval request
624         final AddEndEntityApprovalRequest approvalReq = new AddEndEntityApprovalRequest(endEntity, false, admin, null, caid,
625                 eeProfileId, approvalProfile);
626         try {
627             approvalSession.addApprovalRequest(admin, approvalReq);
628         } catch (EjbcaException e) {
629             errors.add("Could not submit the information for approval: "+e.getMessage());
630             log.info("Approval request could not be added", e);
631         }
632     }
633     
634     private ApprovalProfile getApprovalProfile(final CAInfo cainfo, final int certProfileId) {
635         //FIXME: We shouldn't have to pluck out a session bean for this... 
636         final ApprovalProfileSessionLocal approvalProfileSession = ejbLocalHelper.getApprovalProfileSession();
637         final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(certProfileId);
638         if (certProfile != null) {
639             Integer approvalProfileId = certProfile.getApprovals().get(ApprovalRequestType.ADDEDITENDENTITY);
640             if (approvalProfileId != null) {
641                 return approvalProfileSession.getApprovalProfile(approvalProfileId);
642             }
643         }
644         Integer approvalProfileId = cainfo.getApprovals().get(ApprovalRequestType.ADDEDITENDENTITY);
645         if (approvalProfileId != null) {
646             return approvalProfileSession.getApprovalProfile(approvalProfileId);
647         }
648         log.error("Could not find a suitable approval profile for certificate profile "+certProfileId+" or CA "+cainfo.getCAId());
649         return null;
650 
651     }
