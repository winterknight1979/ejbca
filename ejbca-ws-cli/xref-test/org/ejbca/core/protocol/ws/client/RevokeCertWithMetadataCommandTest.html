14  package org.ejbca.core.protocol.ws.client;
15  
16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertNotNull;
18  
19  import java.util.List;
20  
21  import org.ejbca.core.protocol.ws.client.gen.KeyValuePair;
22  import org.junit.Test;
23  
24  /**
25   * Tests RevokeCertWithMetadataCommand parameter handling.
26   * 
27   * @version $Id: RevokeCertWithMetadataCommandTest.java 22930 2016-03-04 14:02:35Z tarmo_r_helmes $
28   */
29  public class RevokeCertWithMetadataCommandTest {
30      
31      @Test()
32      public void testRevokeCertWithNoMetadataParameters() throws Exception {
33          String[] args = new String[3];
34          args[0] = "revokecertwithmetadata";
35          args[1] = "CN=CA1";
36          args[2] = "63706289784032807";
37  
38          RevokeCertWithMetadataCommand command = new RevokeCertWithMetadataCommand(args);
39          List<KeyValuePair> parsedResult = command.parseInputArgs();
40          assertNotNull(parsedResult);
41          assertEquals(0, parsedResult.size());
42      }
43  
44      @Test()
45      public void testRevokeCertWithMetadataParameters() throws Exception {
46          String[] args = new String[7];
47          args[0] = "revokecertwithmetadata";
48          args[1] = "CN=CA1";
49          args[2] = "63706289784032807";
50  
51          args[3] = "reason=REV_SUPERSEDED";
52          args[4] = "revocationdate=2012-06-07T23:55:59+02:00";
53          args[5] = "certificateProfileId=12";
54          args[6] = "something=123";
55  
56          RevokeCertWithMetadataCommand command = new RevokeCertWithMetadataCommand(args);
57          List<KeyValuePair> parsedResult = command.parseInputArgs();
58          assertEquals(4, parsedResult.size());
59  
60          assertEquals("reason", parsedResult.get(0).getKey());
61          assertEquals("4", parsedResult.get(0).getValue());
62  
63          assertEquals("revocationdate", parsedResult.get(1).getKey());
64          assertEquals("2012-06-07T23:55:59+02:00", parsedResult.get(1).getValue());
65  
66          assertEquals("certificateProfileId", parsedResult.get(2).getKey());
67          assertEquals("12", parsedResult.get(2).getValue());
68  
69          assertEquals("something", parsedResult.get(3).getKey());
70          assertEquals("123", parsedResult.get(3).getValue());
71      }
72  }
