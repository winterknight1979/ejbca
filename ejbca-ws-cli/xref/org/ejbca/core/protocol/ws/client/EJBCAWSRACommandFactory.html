14  package org.ejbca.core.protocol.ws.client;
15  
16  import org.ejbca.ui.cli.IAdminCommand;
17  
18  /**
19   * Factory for EJBCA WS RA Admin Commands.
20   *
21   * @version $Id: EJBCAWSRACommandFactory.java 28786 2018-04-27 10:06:36Z tarmo_r_helmes $
22   */
23  public class EJBCAWSRACommandFactory {
24      /**
25       * Cannot create an instance of this class, only use static methods.
26       */
27      private EJBCAWSRACommandFactory() {
28      }
29  
30      /**
31       * Returns an Admin Command object based on contents in args[0].
32       *
33       * @param args array of arguments typically passed from main().
34       *
35       * @return Command object or null if args[0] does not specify a valid command.
36       */
37      public static IAdminCommand getCommand(String[] args) {
38          if (args.length < 1) {
39              return null;
40          }        
41          
42          if (args[0].equals("edituser")) {
43              return new EditUserCommand(args);
44          }else if (args[0].equals("finduser")) {
45              return new FindUserCommand(args);
46          }else if (args[0].equals("findcerts")) {
47              return new FindCertsCommand(args);
48          }else if (args[0].equals("getprofile")) {
49              return new GetProfileCommand(args);
50          }else if (args[0].equals("pkcs10req")) {
51              return new PKCS10ReqCommand(args);
52          } else if (args[0].equals("pkcs12req")) {
53              return new PKCS12ReqCommand(args);
54          } else if (args[0].equals("certreq")) {
55              return new CertificateRequestCommand(args);
56          } else if (args[0].equals("revokecert")) {
57              return new RevokeCertCommand(args);
58          } else if (args[0].equals("revokecertwithmetadata")) {
59              return new RevokeCertWithMetadataCommand(args);
60          } else if (args[0].equals("getpublisherqueuelength")) {
61              return new GetPublisherQueueLength(args);
62          } else if (args[0].equals("revoketoken")) {
63              return new RevokeTokenCommand(args);
64          } else if (args[0].equals("revokeuser")) {
65              return new RevokeUserCommand(args);
66          } else if (args[0].equals("checkrevokationstatus") || args[0].equals("checkrevocationstatus")) {
67              return new CheckRevokeStatusCommand(args);
68          }else if (args[0].equals("generatenewuser")) {
69              return new GenerateNewUserCommand(args);        
70          }else if (args[0].equals("createcrl")) {
71              return new CreateCRLCommand(args);        
72          } else if (args[0].equals("cacertrequest")) {
73              return new CaCertRequestCommand(args);
74          } else if (args[0].equals("cacertresponse")) {
75              return new CaCertResponseCommand(args);
76          } else if (args[0].equals("customlog")) {
77              return new CustomLogCommand(args);
78          } else if (args[0].equals("stress")) {
79              return new StressTestCommand(args);
80          } else if (args[0].equals("cvcgetchain")) {
81              return new CvcGetChainCommand(args);
82          } else if (args[0].equals("cvcrequest")) {
83              return new CvcRequestCommand(args);
84          } else if (args[0].equals("cvcprint")) {
85              return new CvcPrintCommand(args);
86          } else if (args[0].equals("cvcpem")) {
87              return new CvcPemCommand(args);
88          } else if (args[0].equals("createcryptotoken")) {
89              return new CreateCryptoTokenCommand(args);
90          } else if (args[0].equals("generatectkeys")) {
91              return new GenerateCryptoTokenKeysCommand(args);
92          } else if (args[0].equals("createca")) {
93              return new CreateCACommand(args);
94          } else if (args[0].equals("importcacert")) {
95             return new ImportCaCertCommand(args);
96          }  else if (args[0].equals("updatecacert")) {
97              return new UpdateCaCertCommand(args);
98          } else if (args[0].equals("addadmintorole")) {
99              return new AddSubjectToRoleCommand(args);
100         } else if (args[0].equals("removeadminfromrole")) {
101             return new RemoveSubjectFromRoleCommand(args);
102         } else if (args[0].equals("getexpiredcerts")) {
103             return new GetExpiredCertificatesCommand(args);
104         } else if (args[0].equals("getexpiredcertsbyissuer")) {
105             return new GetExpiredCertificatesByIssuerCommand(args);
106         } else if (args[0].equals("getexpiredcertsbytype")) {
107             return new GetExpiredCertificatesByTypeCommand(args);
108         } else if (args[0].equals("keyrecover")) {
109             return new KeyRecoverCommand(args);
110         } else if (args[0].equals("keyrecoverenroll")) {
111             return new KeyRecoverEnrollCommand(args);
112         }
113         
114         else {
115             return null;
116         }
117     }
118 
119     // getCommand
120 }
