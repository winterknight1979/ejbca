46  /**
47   * Creates or edits a user and sends a CVC request. Writes the issues CV Certificate to file
48   *
49   * @version $Id: CvcRequestCommand.java 24940 2016-12-21 09:43:55Z mikekushner $
50   */
51  public class CvcRequestCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
52  
53  
54  	private static final int ARG_USERNAME           = 1;
55  	private static final int ARG_PASSWORD           = 2;
56  	private static final int ARG_SUBJECTDN          = 3;
57  	private static final int ARG_SEQUENCE           = 4;
58  	private static final int ARG_SIGNALG            = 5;
59  	private static final int ARG_KEYSPEC            = 6;
60  	private static final int ARG_GENREQ             = 7;
61  	private static final int ARG_BASEFILENAME       = 8;
62  	private static final int ARG_AUTHSIGNKEY        = 9;
63  	private static final int ARG_AUTHSIGNCERT       = 10;
64  
65  	/**
66  	 * Creates a new instance of CvcRequestCommand
67  	 *
68  	 * @param args command line arguments
69  	 */
70  	public CvcRequestCommand(String[] args) {
71  		super(args);
72  	}
73  
74  	/**
75  	 * Runs the command
76  	 *
77  	 * @throws IllegalAdminCommandException Error in command args
78  	 * @throws ErrorAdminCommandException Error running command
79  	 */
80  	public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
81  
82  		try {   
83  			if(args.length < 9 || args.length > 11){
84  				getPrintStream().println("Number of arguments: "+args.length);
85  				usage();
86  				System.exit(-1); // NOPMD, this is not a JEE app
87  			}
88  
89  			String username = args[ARG_USERNAME];
90  			String userpassword = args[ARG_PASSWORD];
91  			String dn = args[ARG_SUBJECTDN];
92  			String sequence = args[ARG_SEQUENCE];
93  			String signatureAlg = args[ARG_SIGNALG];
94  			String keySpec = args[ARG_KEYSPEC];
95  			boolean genrequest = args[ARG_GENREQ].equalsIgnoreCase("true");
96  			String basefilename = args[ARG_BASEFILENAME];
97  			String authSignKeyFile = null;
98  			if (args.length > (ARG_AUTHSIGNKEY)) {
99  				authSignKeyFile = args[ARG_AUTHSIGNKEY];				
100 			}
101 			String authSignCertFile = null;
102 			if (args.length > (ARG_AUTHSIGNCERT)) {
103 				authSignCertFile = args[ARG_AUTHSIGNCERT];				
104 			}
105 
106 			getPrintStream().println("Enrolling user:");
107 			getPrintStream().println("Username: "+username);
108 			getPrintStream().println("Subject name: "+dn);
109 			getPrintStream().println("Sequence: "+sequence);
110 			getPrintStream().println("Signature algorithm: "+signatureAlg);                        
111 			getPrintStream().println("Key spec: "+keySpec);                        
112 
113 			try{
114 				CryptoProviderTools.installBCProvider();
115 				
116 				String cvcreq = null;
117 				if (genrequest) {
118 					getPrintStream().println("Generating a new request with base filename: "+basefilename);
119 					// Generate keys for the request
120 					String keytype = "RSA";
121 					if (signatureAlg.contains("ECDSA")) {
122 						keytype = "ECDSA";
123 					}
124 					KeyPair keyPair = KeyTools.genKeys(keySpec, keytype);
125 					String country = CertTools.getPartFromDN(dn, "C");
126 					String mnemonic = CertTools.getPartFromDN(dn, "CN");
127 					if (sequence.equalsIgnoreCase("null")) {
128 						sequence = RandomStringUtils.randomNumeric(5);
129 						getPrintStream().println("No sequence given, using random 5 number sequence: "+sequence);
130 					}
131 					//CAReferenceField caRef = new CAReferenceField(country,mnemonic,sequence);
132 					CAReferenceField caRef = null; // Don't create a caRef in the self signed request
133 					// We are making a self signed request, so holder ref is same as ca ref
134 					HolderReferenceField holderRef = new HolderReferenceField(country,mnemonic,sequence);
135 					CVCertificate request = CertificateGenerator.createRequest(keyPair, signatureAlg, caRef, holderRef);
136 					byte[] der = request.getDEREncoded();
137 					if (authSignKeyFile != null) {
138 						getPrintStream().println("Reading private key from pkcs8 file "+authSignKeyFile+" to create an authenticated request");
139 						byte[] keybytes = FileTools.readFiletoBuffer(authSignKeyFile);
140 				        KeyFactory keyfact = KeyFactory.getInstance(keytype, "BC");
141 				        PrivateKey privKey = keyfact.generatePrivate(new PKCS8EncodedKeySpec(keybytes));
142 				        KeyPair authKeyPair = new KeyPair(null, privKey); // We don't need the public key
143 				        // Default caRef if we do not pass in a certificate to get caRef from
144 						CAReferenceField authCaRef = new CAReferenceField(country,mnemonic,sequence);
145 						CVCertificate authCert = null;
146 						if (authSignCertFile != null) {
147 							getPrintStream().println("Reading cert from cvcert file "+authSignCertFile+" to create an authenticated request");							
148 							CVCObject parsedObject = CvcPrintCommand.getCVCObject(authSignCertFile);
149 							authCert = (CVCertificate)parsedObject;
150 							String c = authCert.getCertificateBody().getHolderReference().getCountry();
151 							String m = authCert.getCertificateBody().getHolderReference().getMnemonic();
152 							String s = authCert.getCertificateBody().getHolderReference().getSequence();
153 							authCaRef = new CAReferenceField(c, m, s);
154 						}
155 						CVCAuthenticatedRequest authRequest = CertificateGenerator.createAuthenticatedRequest(request, authKeyPair, signatureAlg, authCaRef);
156 						// Test to verify it yourself first
157 						if (authCert != null) {
158 							getPrintStream().println("Verifying the request before sending it...");
159 							PublicKey pk = KeyTools.getECPublicKeyWithParams(authCert.getCertificateBody().getPublicKey(), keySpec);
160 							authRequest.verify(pk);							
161 						}
162 						der = authRequest.getDEREncoded();						
163 					}
164 					cvcreq = new String(Base64.encode(der));
165 					// Print the generated request to file
166 					FileOutputStream fos = new FileOutputStream(basefilename+".cvreq");
167 					fos.write(der);
168 					fos.close();					
169 					getPrintStream().println("Wrote binary request to: "+basefilename+".cvreq");
170 					fos = new FileOutputStream(basefilename+".pkcs8");
171 					fos.write(keyPair.getPrivate().getEncoded());
172 					fos.close();					
173 					getPrintStream().println("Wrote private key in "+keyPair.getPrivate().getFormat()+" format to to: "+basefilename+".pkcs8");
174 				} else {
175 					// Read request from file
176 					getPrintStream().println("Reading request from filename: "+basefilename+".cvreq");
177 					byte[] der = FileTools.readFiletoBuffer(basefilename+".cvreq");
178 					cvcreq = new String(Base64.encode(der));
179 				}
180 				
181 				// Edit a user, creating it if it does not exist
182 				// Actually don't do that, leverage the existing commands and force to use the editUser command instead.
183 				// This also makes this CLI exactly represent the actual WS-API call 
184 				// getEjbcaRAWS().editUser(userdata);
185 				
186 				getPrintStream().println("Submitting CVC request for user '"+username+"'.");
187 				getPrintStream().println();              
188 				// Use the request and request a certificate
189 				List<Certificate> resp = getEjbcaRAWS().cvcRequest(username, userpassword, cvcreq);
190 
191 				// Handle the response
192 				Certificate cert = resp.get(0);
193 				byte[] b64cert = cert.getCertificateData();
194 				CVCObject parsedObject = CertificateParser.parseCertificate(Base64.decode(b64cert));
195 				CVCertificate cvcert = (CVCertificate)parsedObject;
196 				FileOutputStream fos = new FileOutputStream(basefilename+".cvcert");
197 				fos.write(cvcert.getDEREncoded());
198 				fos.close();
199 				getPrintStream().println("Wrote binary certificate to: "+basefilename+".cvcert");
200 				getPrintStream().println("You can look at the certificate with the command cvcwscli.sh cvcprint "+basefilename+".cvcert");
201 			}catch(AuthorizationDeniedException_Exception e){
202 				getPrintStream().println("Error : " + e.getMessage());
203 			}catch(UserDoesntFullfillEndEntityProfile_Exception e){
204 				getPrintStream().println("Error : Given userdata doesn't fulfill end entity profile. : " +  e.getMessage());
205 			}
206 
207 		} catch (Exception e) {
208 			if (e instanceof EjbcaException_Exception) {
209 				EjbcaException_Exception e1 = (EjbcaException_Exception)e;
210 				getPrintStream().println("Error code is: "+e1.getFaultInfo().getErrorCode().getInternalErrorCode());
211 			}
212 			throw new ErrorAdminCommandException(e);
213 		}
214 	}
215 
216 	protected void usage() {
217 		getPrintStream().println("Command used to make a CVC request. A user must exist, add one with 'ejbcawsracli.sh edituser'.");
218 		getPrintStream().println("Usage : cvcrequest <username> <password> <subjectdn> <sequence> <signatureAlg> <keyspec (1024|1536|2048|curve)><genreq=true|false> <basefilename> [<auth-sign-key>] [<auth-sign-cert>]\n\n");
219 		getPrintStream().println("SignatureAlg is used when generating a request and can be SHA1WithRSA, SHA256WithRSA, SHA256WithRSAAndMGF1, SHA1WithECDSA, SHA224WithECDSA, SHA256WithECDSA");
220 		getPrintStream().println("Keyspec is used when generating a request and is 1024, 1536, 2048, etc. for RSA keys and the name of a named curve for ECDSA, see User Guide for supported curves.");
221 		getPrintStream().println("DN is used when generating a request and is of form \"C=SE, CN=ISTEST2\", where SE is the country and ISTEST2 the mnemonic.");
222 		getPrintStream().println("Sequence is used when generating a request and is a sequence number for the public key, recomended form 00001 etc. If 'null' a random 5 number sequence will be generated.");
223 		getPrintStream().println("If genreq is true a new request is generated and the generated request is written to <basefilename>.cvreq, and the private key to <basefilename>.pkcs8.");
224 		getPrintStream().println("If genreq is false a request is read from <reqfilename>.cvreq and sent to the CA, the sequence from the command line is ignored.");
225 		getPrintStream().println("The issued certificate is written to <basefilename>.cvcert\n");
226 		getPrintStream().println("auth-sign-key is optional and if given the CVC request is signed by this key to create an authenticated CVC request.");
227 		getPrintStream().println("auth-sign-cert is optional and if given the caRef of the authenticated CVC request is taken from this CVC certificate.");
228 	}
