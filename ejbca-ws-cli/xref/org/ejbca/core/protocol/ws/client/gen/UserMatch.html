61      /**
62       * Constuctor to use to create a UserMatch.
63       * 
64       * @param matchwith  one of MATCH_WITH_ constants.
65       * @param matchtype  one of MATCH_TYPE_ constants.
66       * @param matchvalue a string to search for.

68      public UserMatch(int matchwith, int matchtype, String matchvalue){
