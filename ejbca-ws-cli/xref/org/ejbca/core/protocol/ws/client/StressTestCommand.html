102 									 new EditUserCommand(ejbcaWS, this.caName, this.endEntityProfileName, this.certificateProfileName, jobData, true, this.maxCertificateSN),

170 	 * @param certificateResponse Response
171 	 * @param jobData Data
172 	 * @return bool
173 	 * @throws CertificateException Fail
174 	 */
175 	private boolean checkAndLogCertificateResponse(
176 			final CertificateResponse certificateResponse, final JobData jobData)
177 					throws CertificateException {
178 		X509Certificate cert = null;
179 		for ( final java.security.cert.Certificate tmp : CertificateFactory.getInstance("X.509").generateCertificates(new ByteArrayInputStream(Base64.decode(certificateResponse.getData()))) ) {
180 			cert = (X509Certificate)tmp;
181 		}
182 		if ( cert==null ) {
183 			StressTestCommand.this.performanceTest.getLog().error("no certificate generated for user "+jobData.userName);
184 			return false;
185 		}
186 		final String commonName = CertTools.getPartFromDN(cert.getSubjectDN().getName(), "CN");
187 		if ( !commonName.equals(jobData.userName) ) {
188 			StressTestCommand.this.performanceTest.getLog().error("Cert not created for right user. Username: \""+jobData.userName+"\" Subject DN: \""+cert.getSubjectDN()+"\".");
189 			return false;
190 		}
191 		jobData.userCertsGenerated.add(cert);
192 		StressTestCommand.this.performanceTest.getLog().info("Cert created. Subject DN: \""+cert.getSubjectDN()+"\".");
193 		StressTestCommand.this.performanceTest.getLog().result(CertTools.getSerialNumber(cert));
194 		return true;
195 	}

235 		final private EjbcaWS ejbcaWS;
236 		FindUserCommand(EjbcaWS _ejbcaWS, JobData _jobData) throws Exception {
237 			super(_jobData);
238 			this.ejbcaWS = _ejbcaWS;
239 		}
240 		@Override
241 		public boolean doIt() throws Exception {
242 			final org.ejbca.core.protocol.ws.client.gen.UserMatch match = new org.ejbca.core.protocol.ws.client.gen.UserMatch();
243 			match.setMatchtype(BasicMatch.MATCH_TYPE_EQUALS);
244 			match.setMatchvalue(this.jobData.getDN());
245 			match.setMatchwith(org.ejbca.util.query.UserMatch.MATCH_WITH_DN);
246 			final List<UserDataVOWS> result = this.ejbcaWS.findUser(match);
247 			if (result.size()<1) {
248 				StressTestCommand.this.performanceTest.getLog().error("No users found for DN \""+this.jobData.getDN()+"\"");
249 				return false;
250 			}
251 			final Iterator<UserDataVOWS> i = result.iterator();
252 			while ( i.hasNext() ) {
253 				final String userName = i.next().getUsername();
254 				if( !userName.equals(this.jobData.userName) ) {
255 					StressTestCommand.this.performanceTest.getLog().error("wrong user name \""+userName+"\" for certificate with DN \""+this.jobData.getDN()+"\"");
256 					return false;
257 				}
258 			}
259 			return true;
260 		}
261 		@Override
262 		public String getJobTimeDescription() {
263 			return "Relative time spent looking for user";
264 		}

267 		final private EjbcaWS ejbcaWS;
268 		ListCertsCommand(EjbcaWS _ejbcaWS, JobData _jobData) throws Exception {
269 			super(_jobData);
270 			this.ejbcaWS = _ejbcaWS;
271 		}
272 		@Override
273 		public boolean doIt() throws Exception {
274 			final List<Certificate> result = this.ejbcaWS.findCerts(this.jobData.userName, true);
275 			final Iterator<Certificate> i = result.iterator();
276 			this.jobData.userCertsToBeRevoked = new X509Certificate[result.size()];
277 			for( int j=0; i.hasNext(); j++ ) {
278 				this.jobData.userCertsToBeRevoked[j] = (X509Certificate)CertificateFactory.getInstance("X.509").generateCertificate(new ByteArrayInputStream(Base64.decode(i.next().getCertificateData())));
279 			}
280 			if ( this.jobData.userCertsToBeRevoked.length < 1 ) {
281 				this.jobData.userCertsToBeRevoked = this.jobData.userCertsGenerated.toArray(new X509Certificate[this.jobData.userCertsGenerated.size()]);
282 			}
283 			this.jobData.userCertsGenerated.clear();
284 			if ( this.jobData.userCertsToBeRevoked.length < 1 ) {
285 				StressTestCommand.this.performanceTest.getLog().error("no cert found for user "+this.jobData.userName);
286 				return false;
287 			}
288 
289 			return true;
290 		}
291 		@Override
292 		public String getJobTimeDescription() {
293 			return "Relative time spent finding certs for user.";
294 		}

318 		final private EjbcaWS ejbcaWS;
319 		final String revoceTime;
320 		RevokeCertBackdatedCommand(EjbcaWS _ejbcaWS, JobData _jobData) throws Exception {
321 			super(_jobData);
322 			this.ejbcaWS = _ejbcaWS;
323 			final Calendar c = Calendar.getInstance();
324 			c.setTime(new Date(new Date().getTime()-1000*60*60*24));
325 			this.revoceTime = DatatypeConverter.printDateTime(c);
326 			StressTestCommand.this.performanceTest.getLog().info("Revoke time: "+this.revoceTime);
327 		}
328 		private void revokeBackdated( int i ) throws Exception {
329 			this.ejbcaWS.revokeCertBackdated(
330 					this.jobData.userCertsToBeRevoked[i].getIssuerDN().getName(),
331 					this.jobData.userCertsToBeRevoked[i].getSerialNumber().toString(16),
332 					REVOKATION_REASON_UNSPECIFIED,
333 					this.revoceTime);
334 		}
335 		private void revoke( int i ) throws Exception {
336 			this.ejbcaWS.revokeCert(
337 					this.jobData.userCertsToBeRevoked[i].getIssuerDN().getName(),
338 					this.jobData.userCertsToBeRevoked[i].getSerialNumber().toString(16),
339 					REVOKATION_REASON_UNSPECIFIED);
340 		}
341 		@Override
342 		public boolean doIt() throws Exception {
343 			for (int i=0; i<this.jobData.userCertsToBeRevoked.length; i++) {
344 				try {
345 					revokeBackdated(i);
346 				} catch (RevokeBackDateNotAllowedForProfileException_Exception e) {
347 					revoke(i);
348 					StressTestCommand.this.performanceTest.getLog().info("No back dating since not allowed for the profile.");
349 					continue;
350 				}
351 			}
352 			return true;
353 		}
354 		@Override
355 		public String getJobTimeDescription() {
356 			return "Relative time spent revoking certificates.";
