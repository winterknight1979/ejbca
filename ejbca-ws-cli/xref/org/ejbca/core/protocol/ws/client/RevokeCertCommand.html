14  package org.ejbca.core.protocol.ws.client;
15  
16  import java.math.BigInteger;
17  
18  import org.cesecore.util.CertTools;
19  import org.ejbca.core.protocol.ws.client.gen.AlreadyRevokedException_Exception;
20  import org.ejbca.core.protocol.ws.client.gen.ApprovalException_Exception;
21  import org.ejbca.core.protocol.ws.client.gen.AuthorizationDeniedException_Exception;
22  import org.ejbca.core.protocol.ws.client.gen.DateNotValidException_Exception;
23  import org.ejbca.core.protocol.ws.client.gen.RevokeBackDateNotAllowedForProfileException_Exception;
24  import org.ejbca.core.protocol.ws.client.gen.RevokeStatus;
25  import org.ejbca.core.protocol.ws.client.gen.WaitingForApprovalException_Exception;
26  import org.ejbca.ui.cli.ErrorAdminCommandException;
27  import org.ejbca.ui.cli.IAdminCommand;
28  import org.ejbca.ui.cli.IllegalAdminCommandException;

30  /**
31   * Revokes a given certificate
32   *
33   * @version $Id: RevokeCertCommand.java 28395 2018-02-27 14:19:00Z anatom $
34   */
35  public class RevokeCertCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
36  
37  
38  	private static final int ARG_ISSUERDN			= 1;
39  	private static final int ARG_CERTSN				= 2;
40  	private static final int ARG_REASON				= 3;
41  	private static final int ARG_TIME				= 4;
42  
43  
44  	/**
45  	 * Creates a new instance of RevokeCertCommand
46  	 *
47  	 * @param args command line arguments
48  	 */
49  	public RevokeCertCommand(String[] args) {
50  		super(args);
51  	}
52  
53  	@Override
54  	public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
55  
56  		try {
57  
58  			if(this.args.length < 4){
59  				usage();
60  				System.exit(-1); // NOPMD, it's not a JEE app
61  			}
62  
63  			final String issuerdn = CertTools.stringToBCDNString(this.args[ARG_ISSUERDN]);
64  			final String certsn = getCertSN(this.args[ARG_CERTSN]);
65  			final int reason = getRevokeReason(this.args[ARG_REASON]);
66  			final String time = this.args.length>ARG_TIME ? this.args[ARG_TIME] : null;
67  			final String justRevoke = "To revoke the certificate with the current time remove the last argument (revocation time).";
68  
69  			try{
70  				final RevokeStatus status =  getEjbcaRAWS().checkRevokationStatus(issuerdn,certsn);
71  				if (status != null) {
72  					getEjbcaRAWS().revokeCertBackdated(issuerdn,certsn,reason,time);
73  					getPrintStream().println("Certificate revoked (or unrevoked) successfully.");
74  				} else {
75  					getPrintStream().println("Certificate does not exist.");
76  				}
77  			} catch (AuthorizationDeniedException_Exception e) {
78  				getPrintStream().println("Error : " + e.getMessage());
79  			} catch (AlreadyRevokedException_Exception e) {
80  				getPrintStream().println("The certificate was already revoked, or you tried to unrevoke a permanently revoked certificate.");
81  			} catch (WaitingForApprovalException_Exception e) {
82  				getPrintStream().println("The revocation request has been sent for approval.");
83  			} catch (ApprovalException_Exception e) {
84  				getPrintStream().println("This revocation has already been requested.");
85  			} catch (DateNotValidException_Exception e) {
86  				getPrintStream().println(e.getMessage());
87  				getPrintStream().println(justRevoke);
88  			} catch (RevokeBackDateNotAllowedForProfileException_Exception e) {
89  				getPrintStream().println(e.getMessage());
90  				getPrintStream().println(justRevoke);
91  			}
92  		} catch (Exception e) {
93  			throw new ErrorAdminCommandException(e);
94  		}
95  	}
96  
97  
98  	private String getCertSN(String certsn) {
99  		try{
100 			new BigInteger(certsn,16);
101 		}catch(NumberFormatException e){
102 			getPrintStream().println("Error in Certificate SN");
103 			usage();
104 			System.exit(-1); // NOPMD, it's not a JEE app
105 		}
106 		return certsn;
107 	}
108 
109 	@Override
110 	protected void usage() {
111 		getPrintStream().println("Command used to revoke or unrevoke a certificate.");
112 		getPrintStream().println("Unrevocation is done using the reason REV_REMOVEFROMCRL, and can only be done if the certificate is revoked with reason REV_CERTIFICATEHOLD.");
113 		getPrintStream().println("Usage : revokecert <issuerdn> <certificatesn (HEX)> <reason> [<revocation date>]");
114 		getPrintStream().println();
115 		getPrintStream().println();
116 		getPrintStream().println("Reason should be one of : ");
117 		for(int i=1; i< REASON_TEXTS.length-1;i++){
118 			getPrintStream().print(REASON_TEXTS[i] + ", ");
119 		}
120 		getPrintStream().print(REASON_TEXTS[REASON_TEXTS.length-1]);
121 		getPrintStream().println();
122 		getPrintStream().println("Revocation date is optional. If specified it must be in the past and must be a valid ISO8601 string");
123 		getPrintStream().println("Example: 2012-06-07T23:55:59+02:00");
124 	}
