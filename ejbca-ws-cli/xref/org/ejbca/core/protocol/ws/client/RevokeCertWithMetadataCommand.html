34  /**
35   * Revokes a given certificate. Difference with normal RevokeCertCommand is that 
36   * this one here allows to include a list of keyvaluepair input parameters
37   * for example "reason=REV_SUPERSEDED" "certificateProfileId=12"
38   *
39   * @version $Id: RevokeCertWithMetadataCommand.java 28395 2018-02-27 14:19:00Z tarmo_r_helmes $
40   */
41  public class RevokeCertWithMetadataCommand extends EJBCAWSRABaseCommand implements IAdminCommand {
42  
43  
44  	private static final int ARG_ISSUERDN			= 1;
45  	private static final int ARG_CERTSN				= 2;
46  
47  	private static final String REASON_KEY = "reason";
48  
49  	/**
50  	 * Creates a new instance of RevokeCertCommand
51  	 *
52  	 * @param args command line arguments
53  	 */
54  	public RevokeCertWithMetadataCommand(String[] args) {
55  		super(args);
56  	}
57  	
58  	@Override
59  	public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
60  		try {
61  			if(this.args.length < 3) {
62  				usage();
63  				System.exit(-1); // NOPMD, it's not a JEE app
64  			}
65  
66  			List<KeyValuePair> metadata = parseInputArgs();
67  
68  			final String issuerdn = CertTools.stringToBCDNString(this.args[ARG_ISSUERDN]);
69  			final String certsn = getCertSN(this.args[ARG_CERTSN]);
70  
71  			final String justRevoke = "To revoke the certificate with the current time remove the last argument (revocation time).";
72  
73  			try {
74  				final RevokeStatus status = getEjbcaRAWS().checkRevokationStatus(issuerdn, certsn);
75  				if (status != null) {
76  					getEjbcaRAWS().revokeCertWithMetadata(issuerdn, certsn, metadata);
77  					getPrintStream().println("Certificate revoked (or unrevoked) successfully.");
78  				} else {
79  					getPrintStream().println("Certificate does not exist.");
80  				}
81  			} catch (AuthorizationDeniedException_Exception e) {
82  				getPrintStream().println("Error : " + e.getMessage());
83  			} catch (AlreadyRevokedException_Exception e) {
84  				getPrintStream().println("The certificate was already revoked, or you tried to unrevoke a permanently revoked certificate.");
85  			} catch (WaitingForApprovalException_Exception e) {
86  				getPrintStream().println("The revocation request has been sent for approval.");
87  			} catch (ApprovalException_Exception e) {
88  				getPrintStream().println("This revocation has already been requested.");
89  			} catch (DateNotValidException_Exception e) {
90  				getPrintStream().println(e.getMessage());
91  				getPrintStream().println(justRevoke);
92  			} catch (RevokeBackDateNotAllowedForProfileException_Exception e) {
93  				getPrintStream().println(e.getMessage());
94  				getPrintStream().println(justRevoke);
95  			}
96  		} catch (Exception e) {
97  			throw new ErrorAdminCommandException(e);
98  		}
99  	}
100 
101 	protected List<KeyValuePair> parseInputArgs() throws Exception {
102 		List<KeyValuePair> metadata = new ArrayList<>();
103 		if (this.args.length > 3) {
104             for (int i=3; i<this.args.length; i++) {
105                 String arg = this.args[i];
106                 String[] parts = arg.split("=", 2);
107 
108                 if (parts.length != 2) {
109                     String msg = "Problem with parameter: " + arg + ". Invalid format, please use \"key=value\"";
110                     getPrintStream().println(msg);
111                     throw new IllegalArgumentException(msg);
112                 }
113                 KeyValuePair keyValuePair = new KeyValuePair();
114                 String key = parts[0];
115                 keyValuePair.setKey(key);
116 
117                 if (key.equalsIgnoreCase(REASON_KEY)) {
118                     final int reason = getRevokeReason(parts[1]);
119                     keyValuePair.setValue(Integer.valueOf(reason).toString());
120                 } else {
121                     keyValuePair.setValue(parts[1]);
122                 }
123                 metadata.add(keyValuePair);
124             }
125         }
126 		return metadata;
127 	}
128 
129 
130 	private String getCertSN(String certsn) {
131 		try {
132 			new BigInteger(certsn, 16);
133 		} catch(NumberFormatException e) {
134 			getPrintStream().println("Error in Certificate SN");
135 			usage();
136 			System.exit(-1); // NOPMD, it's not a JEE app
137 		}
138 		return certsn;
139 	}
140 
141 	@Override
142 	protected void usage() {
143 		getPrintStream().println("Command used to revoke or unrevoke a certificate.");
144 		getPrintStream().println("Unrevocation is done using the reason REV_REMOVEFROMCRL, and can only be done if the certificate is revoked with reason REV_CERTIFICATEHOLD.");
145 		getPrintStream().println("Usage : revokecert <issuerdn> <certificatesn(HEX)> reason=<reason> [revocationdate=<revocation date>] [certificateProfileId=<certificateProfileId>]");
146 		getPrintStream().println();
147 		getPrintStream().println();
148 		getPrintStream().println("Reason should be one of : ");
149 		for(int i=1; i< REASON_TEXTS.length-1;i++){
150 			getPrintStream().print(REASON_TEXTS[i] + ", ");
151 		}
152 		getPrintStream().print(REASON_TEXTS[REASON_TEXTS.length-1]);
153 		getPrintStream().println();
154 		getPrintStream().println("Revocation date is optional. If specified it must be in the past and must be a valid ISO8601 string");
155 		getPrintStream().println("Example: 2012-06-07T23:55:59+02:00");
156 		getPrintStream().println("Certificate profile id is optional.");
157 	}
