13  package org.ejbca.core.protocol.ws.common;
14  
15  import java.io.ByteArrayInputStream;
16  import java.io.IOException;
17  import java.security.KeyStoreException;
18  import java.security.NoSuchAlgorithmException;
19  import java.security.NoSuchProviderException;
20  import java.security.cert.CertificateException;
21  
22  import org.cesecore.util.Base64;
23  
24  
25  /**
26   * Class used to generate a java.security.KeyStore from a 
27   * org.ejbca.core.protocol.ws.common.KeyStore
28   * 
29   * @author Philip Vendil
30   *
31   * @version $Id: KeyStoreHelper.java 19902 2014-09-30 14:32:24Z anatom $
32   */
33  public class KeyStoreHelper {
34  
35  	/**
36  	 * Retrieves the keystore from the encoded data.
37  	 * @param keystoreData Data
38  	 * @param type "PKCS12" or "JKS"
39  	 * @param password to lock the keystore
40  	 * @return the loaded and unlocked keystore.
41  	 * @throws CertificateException Fail
42  	 * @throws IOException Fail
43  	 * @throws NoSuchAlgorithmException Fail 
44  	 * @throws NoSuchProviderException Fail
45  	 * @throws KeyStoreException Fail
46  	 */
47  	public static java.security.KeyStore getKeyStore(byte[] keystoreData, String type, String password) throws CertificateException, NoSuchAlgorithmException, IOException, KeyStoreException, NoSuchProviderException{
48  		java.security.KeyStore ks = type.equalsIgnoreCase("JKS") ? java.security.KeyStore.getInstance("JKS") : java.security.KeyStore.getInstance(type, "BC");
49  		ByteArrayInputStream bais = new ByteArrayInputStream(Base64.decode(keystoreData));
50  		ks.load(bais, password.toCharArray());
51          return ks; 
52  	}
53  }
