13  package org.ejbca.configdump.ejb;
14  
15  import java.io.IOException;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.ejbca.configdump.ConfigDumpSetting;
20  import org.ejbca.configdump.ConfigdumpException;
21  import org.ejbca.configdump.ConfigdumpExportResult;
22  
23  /**
24   * Interface for exporting (and the future, importing) data with Configdump.
25   * 
26   * @version $Id: ConfigdumpSession.java 28387 2018-02-27 09:12:37Z aminkh $
27   */
28  public interface ConfigdumpSession {
29      
30      static final String CONFIGDUMP_MODULE = "configdump-ejb";
31      
32      /**
33       * Exports EJBCA data, except for keys and certificates, to a set of YAML files.
34       * One YAML file is created per "item" in the database (e.g. a certificate profile).
35       * @param admin Authentication token of the requesting admin (for the Configdump CLI, this is a CLI authentication token).
36       * @param setting Settings such as export directory, and items to include in the export
37       * @return Result object. In case errors are set to be ignored, then this contains a list of errors.
38       * @throws ConfigdumpException If an error occurs, which was not ignored.
39       * @throws IOException On failure to create the YAML files etc.
40       * @throws AuthorizationDeniedException fail 
41       */
42      ConfigdumpExportResult performExport(final AuthenticationToken admin, final ConfigDumpSetting setting) throws ConfigdumpException, IOException, AuthorizationDeniedException;
43  
44  }
