13  package org.ejbca.configdump;
14  
15  import java.io.Serializable;
16  import java.util.regex.Pattern;
17  
18  /**
19   * A simple pattern that supports "*" only (unlike Pattern which allows full regexes)
20   * and that keeps track of the number of matches.
21   * 
22   * @version $Id: ConfigdumpPattern.java 28148 2018-01-31 07:03:10Z samuellb $
23   */
24  public class ConfigdumpPattern implements Serializable {
25      
26      private static final long serialVersionUID = 1L;
27      private final String patternString;
28      private final Pattern pattern;
29      private int numMatches = 0;
30      
31      public ConfigdumpPattern(final String simplePattern) {
32          // Convert "basic" pattern with only "*" wildcards into a regex
33          // The string is "quoted" (avoids parsing as regex) except for * which are translated into \E.*\Q
34          // \E and \Q are the escape sequences to stop- and start quoting respectively
35          pattern = Pattern.compile('^'+Pattern.quote(simplePattern).replace("*", "\\E.*\\Q").replace("\\Q\\E", "")+'$');
36          patternString = simplePattern;
37      }
38      
39      public boolean matches(final CharSequence input) {
40          if (pattern.matcher(input).find()) {
41              numMatches++;
42              return true;
43          }
44          return false;
45      }
46      
47      public int getNumMatches() {
48          return numMatches;
49      }
50      
51      @Override
52      public String toString() {
53          return patternString;
54      }
55  }
