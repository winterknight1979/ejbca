49  public interface KeyStoreCreateSession {
50  
51      /**
52       * Creates a server-generated keystore.
53       *
54       * The method must be preceded by
55       * a editUser call, either to set the userstatus to 'new' or to add non-existing users and
56       * the user's token must be set to org.ejbca.core.protocol.ws.client.gen.UserDataVOWS.TOKEN_TYPE_P12.<br>
57       *
58       * Authorization requirements: <pre>
59       * - /administrator
60       * - /ca/&lt;ca of user&gt;
61       * - /ca_functionality/create_certificate
62       * - /endentityprofilesrules/&lt;end entity profile&gt;/view_end_entity
63       * - /ra_functionality/view_end_entity
64       * </pre>
65       *
66       * Additional authorization requirements for (non key recovery) clearing of password: <pre>
67       * - /endentityprofilesrules/&lt;end entity profile&gt;/edit_end_entity
68       * - /ra_functionality/edit_end_entity
69       * </pre>
70       *
71       * Additional authorization requirements for key recovery: <pre>
72       * - /endentityprofilesrules/&lt;end entity profile&gt;/keyrecovery
73       * - /ra_functionality/keyrecovery
74       * </pre>
75       *
76       * @param authenticationToken administrator performing the action.
77       * @param username the unique username.
78       * @param password the password sent with editUser call.
79       * @param hardTokenSN If the certificate should be connected with a hard token, it is
80       * possible to map it by give the hardTokenSN here, this will simplify revocation of a tokens
81       * certificates. Use null if no hardtokenSN should be associated with the certificate.
82       * @param keySpecification that the generated key should have, examples are 2048 for RSA or secp256r1 for ECDSA.
83       * @param keyAlgorithm that the generated key should have, RSA, ECDSA. Use one of the constants in {@link org.cesecore.certificates.util.AlgorithmConstants}.KEYALGORITHM_...
84       * @return the generated or reused key store as byte array.
85       * @throws CADoesntExistsException if a referenced CA does not exist.
86       * @throws AuthorizationDeniedException if client isn't authorized to request.
87       * @throws NotFoundException if the user could not be found.
88       * @throws EjbcaException any EjbcaException.
89       */
90      byte[] generateOrKeyRecoverTokenAsByteArray(AuthenticationToken authenticationToken, String username, String password, String hardTokenSN, String keySpecification, String keyAlgorithm)
91              throws CADoesntExistsException, AuthorizationDeniedException, NotFoundException, EjbcaException;
92      
93      /**
94       * This method generates a new pkcs12 or jks token for a user, and key recovers the token, if the user is configured for that in EJBCA.
95       *
96       * @param administrator administrator performing the action.
97       * @param username username in ejbca
98       * @param password password for user
99       * @param caid caid of the CA the user is registered for
100      * @param keyspec name of ECDSA key or length of RSA and DSA keys (endEntityInformation.extendedInformation.keyStoreAlgorithmSubType has priority over this value)
101      * @param keyalg AlgorithmConstants.KEYALGORITHM_RSA, AlgorithmConstants.KEYALGORITHM_DSA or AlgorithmConstants.KEYALGORITHM_ECDSA (endEntityInformation.extendedInformation.keyStoreAlgorithmType has priority over this value)
102      * @param notBefore The date at which the private key starts to be valid, or null
103      * @param notAfter The date at which the private key should be considered expired, or null
104      * @param createJKS true to create a JKS, false to create a PKCS12
105      * @param loadkeys true if keys should be recovered
106      * @param savekeys true if generated keys should be stored for keyrecovery
107      * @param reusecertificate true if the old certificate should be reused for a recovered key
108      * @param endEntityProfileId the end entity profile the user is registered for
109      *
110      * @return a keystore
111      *
112      * @throws AuthorizationDeniedException if the authentication token was not allowed access to the EEP or CA of the end entity, to recover keys,
113      * to issue certificates
114      * @throws KeyStoreException if keys were set to be recovered, but no key recovery data was found
115      * @throws InvalidAlgorithmParameterException  if the given parameters (keyspec, keyalg) are inappropriate for this key pair generator.
116      * @throws CADoesntExistsException if the CA defined by caid does not exist
117      * @throws AuthLoginException If the password was incorrect.
118      * @throws AuthStatusException If the end entity's status is incorrect.
119      * @throws CustomCertificateSerialNumberException (no rollback) if custom serial number is registered for user, but it is not allowed to be used (either
120      *             missing unique index in database, or certificate profile does not allow it
121      * @throws InvalidAlgorithmException if the signing algorithm in the certificate profile (or the CA Token if not found) was invalid.
122      * @throws CAOfflineException if the CA was offline
123      * @throws IllegalValidityException if the validity defined by notBefore and notAfter was invalid
124      * @throws CryptoTokenOfflineException if the crypto token for the CA wasn't found
125      * @throws CertificateSerialNumberException if certificate with same subject DN or key already exists for a user, if these limitations are enabled in CA.
126      * @throws CertificateRevokeException (rollback) if certificate was meant to be issued revoked, but could not.
127      * @throws IllegalNameException if the certificate request contained an illegal name
128      * @throws CertificateCreateException (rollback) if certificate couldn't be created.
129      * @throws IllegalKeyException if the public key didn't conform to the constrains of the CA's certificate profile.
130      * @throws NoSuchEndEntityException if the end entity was not found
131      * @throws EndEntityProfileValidationException if the password doesn't fulfill the demands set by the EE profile
132      * @throws CertificateSignatureException if verification of the CA certificate failed
133      * @throws InvalidKeySpecException if the key specification defined in keys couldn't be found
134      * @throws NoSuchAlgorithmException if the algorithm defined in the keys couldn't be found
135      * @throws CertificateException if there was a problem with the certificate
136      * @throws CertificateEncodingException if there was a problem with the certificate
137      */
138     KeyStore generateOrKeyRecoverToken(AuthenticationToken administrator, String username, String password, int caid, String keyspec, String keyalg,
139             Date notAfter, Date notBefore,
140             boolean createJKS, boolean loadkeys, boolean savekeys, boolean reusecertificate, int endEntityProfileId)
141             throws AuthorizationDeniedException, KeyStoreException, InvalidAlgorithmParameterException, CADoesntExistsException, IllegalKeyException,
142             CertificateCreateException, IllegalNameException, CertificateRevokeException, CertificateSerialNumberException,
143             CryptoTokenOfflineException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException,
144             CustomCertificateSerialNumberException, AuthStatusException, AuthLoginException, EndEntityProfileValidationException,
145             NoSuchEndEntityException, CertificateSignatureException, CertificateEncodingException, CertificateException, NoSuchAlgorithmException,
146             InvalidKeySpecException;
147 }
