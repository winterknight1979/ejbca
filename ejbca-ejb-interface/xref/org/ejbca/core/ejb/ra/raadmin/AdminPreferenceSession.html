13  package org.ejbca.core.ejb.ra.raadmin;
14  
15  import org.cesecore.authentication.tokens.AuthenticationToken;
16  import org.cesecore.authentication.tokens.X509CertificateAuthenticationToken;
17  import org.cesecore.authorization.AuthorizationDeniedException;
18  import org.ejbca.core.model.ra.raadmin.AdminPreference;
19  
20  /** Session bean to handle admin preference administration
21   * 
22   * @version $Id: AdminPreferenceSession.java 23202 2016-04-14 08:53:01Z mikekushner $
23   */
24  public interface AdminPreferenceSession {
25      
26      /**
27       * Finds the admin preference belonging to a certificate serial number.
28       * 
29       * @param certificatefingerprint The certificate finger print of the certificate the admin uses as a credential.
30       * 
31       * @return null if the admin does not exist.
32       */
33      AdminPreference getAdminPreference(String certificatefingerprint);
34  
35      /**
36       * Adds a admin preference to the database.
37       * 
38       * @param admin An {@link X509CertificateAuthenticationToken} representing the admin the preference should 
39       *              cover.
40       * @param adminpreference the admin preference to add. 
41       * 
42       *  @return  false if admin already exists.
43       */
44      boolean addAdminPreference(X509CertificateAuthenticationToken admin, AdminPreference adminpreference);
45  
46      /**
47       * Changes the admin preference in the database.
48       * 
49       * @param admin An {@link X509CertificateAuthenticationToken} representing the admin the preference should 
50       *              cover.
51       * @param adminpreference the admin preference to add. 
52       * 
53       * @return false if admin does not exist.
54       */
55      boolean changeAdminPreference(X509CertificateAuthenticationToken admin, AdminPreference adminpreference);
56  
57      /**
58       * Changes the admin preference in the database without logging. 
59       * @param admin An {@link X509CertificateAuthenticationToken} representing the admin the preference should 
60       *              cover.
61       * @param adminpreference the admin preference to add. 
62       * 
63       * @return false if admin does not exist.
64       */
65      boolean changeAdminPreferenceNoLog(X509CertificateAuthenticationToken admin, AdminPreference adminpreference);
66  
67      /** 
68       * Checks if a admin preference exists in the database. 
69       * 
70       * @param certificatefingerprint The certificate finger print of the certificate the admin uses as a credential.
71       * 
72       * @return true if it exists
73       */
74      boolean existsAdminPreference(String certificatefingerprint);
75  
76      /** Function that returns the default admin preference. 
77       * 
78       * @return the default admin preference. 
79       */
80      AdminPreference getDefaultAdminPreference();
81  
82      /** Function that saves the default admin preference. 
83       * @param admin An {@link AuthenticationToken} for authorization.
84       * @param defaultadminpreference The {@link AdminPreference} to save as default.
85       * @throws AuthorizationDeniedException if the local {@link AuthenticationToken} wasn't authorized to /system_functionality/edit_systemconfiguration
86       */
87      void saveDefaultAdminPreference(AuthenticationToken admin, AdminPreference defaultadminpreference) throws AuthorizationDeniedException;
88  
89  }
