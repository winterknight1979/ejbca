13  package org.ejbca.core.ejb.ra.raadmin;
14  
15  import java.util.Collection;
16  import java.util.List;
17  import java.util.Map;
18  
19  import org.cesecore.authentication.tokens.AuthenticationToken;
20  import org.cesecore.authorization.AuthorizationDeniedException;
21  import org.ejbca.core.model.ra.raadmin.EndEntityProfile;
22  import org.ejbca.core.model.ra.raadmin.EndEntityProfileExistsException;
23  import org.ejbca.core.model.ra.raadmin.EndEntityProfileNotFoundException;
24  
25  /** Session bean for managing End Entity Profiles.
26   * 
27   * @version $Id: EndEntityProfileSession.java 29589 2018-08-08 12:02:53Z bastianf $
28   */
29  public interface EndEntityProfileSession {
30  
31      /**
32       * Adds a profile to the database. Will automatically set a valid ID. 
33       *
34       * 
35       * @param admin administrator performing task
36       * @param profilename readable profile name
37       * @param profile profile to be added
38       * @return the ID of the end entity profile added. 
39       * @throws AuthorizationDeniedException if admin was not authorized to add end entity profiles
40       * @throws EndEntityProfileExistsException if a profile of the given name already exists
41       * 
42       */
43      int addEndEntityProfile(AuthenticationToken admin, String profilename, EndEntityProfile profile) throws EndEntityProfileExistsException,
44              AuthorizationDeniedException;
45  
46      /**
47       * Adds a profile to the database.
48       * 
49       * @param admin administrator performing task
50       * @param profileid internal ID of new profile, use only if you know it's
51       *            right.
52       * @param profilename readable profile name
53       * @param profile profile to be added
54       * @throws EndEntityProfileExistsException fail
55       * @throws AuthorizationDeniedException fail
56       */
57      void addEndEntityProfile(AuthenticationToken admin, int profileid, String profilename, EndEntityProfile profile)
58              throws EndEntityProfileExistsException, AuthorizationDeniedException;
59  
60      /**
61       * Adds a end entity profile to a group with the same content as the
62       * original profile.
63       * @param admin admin
64       * @param originalprofilename name 
65       * @param newprofilename name
66       * @throws EndEntityProfileExistsException fail 
67       * @throws AuthorizationDeniedException fail
68       */
69      void cloneEndEntityProfile(AuthenticationToken admin, String originalprofilename, String newprofilename) throws EndEntityProfileExistsException,
70              AuthorizationDeniedException;
71  
72      /** @return an unused end id number. */
73      int findFreeEndEntityProfileId();
74  
75      /**
76       * Removes an end entity profile from the database, does not throw any
77       * errors if the profile does not exist.
78       * @param admin admin
79       * @param profilename name 
80       * @throws AuthorizationDeniedException fail
81       */
82      void removeEndEntityProfile(AuthenticationToken admin, String profilename) throws AuthorizationDeniedException;
83  
84      /** Renames a end entity profile. 
85       * @param admin admin
86       * @param oldprofilename name 
87       * @param newprofilename name
88       * @throws EndEntityProfileExistsException fail 
89       * @throws AuthorizationDeniedException fail */
90      void renameEndEntityProfile(AuthenticationToken admin, String oldprofilename, String newprofilename) throws EndEntityProfileExistsException,
91              AuthorizationDeniedException;
92  
93      /** Updates profile data. 
94       * @param admin admin
95       * @param profilename name 
96       * @param profile profile
97       * @throws AuthorizationDeniedException fail 
98       * @throws EndEntityProfileNotFoundException if sought end entity profile was not found.
99       */
100     void changeEndEntityProfile(AuthenticationToken admin, String profilename, EndEntityProfile profile) throws AuthorizationDeniedException, EndEntityProfileNotFoundException;
101 
102     /**
103      * Do NOT use, use changeEndEntityProfile instead. Used internally for
104      * testing only. Updates a profile without flushing caches.
105      * 
106      * TODO: Move this method out of production code.
107      * @param admin admin
108      * @param profilename profila 
109      * @param profile profile
110      * 
111      * @throws AuthorizationDeniedException fail 
112      * @throws EndEntityProfileNotFoundException if sought end entity profile was not found.
113      */
114     void internalChangeEndEntityProfileNoFlushCache(AuthenticationToken admin, String profilename, EndEntityProfile profile)
115             throws AuthorizationDeniedException, EndEntityProfileNotFoundException;
116 
117     /** Retrieves a Collection of id:s (Integer) to authorized profiles. 
118      * @param admin the administrator for whom to get the profile Ids he/she has access to
119      * @param endentityAccessRule an access rule which is required on the profile in order for it to be returned, for example AccessRulesConstants.CREATE_END_ENTITY to only return profiles for which the admin have create rights
120      * @return Collection of end entity profile id:s (Integer)
121      */
122     Collection<Integer> getAuthorizedEndEntityProfileIds(AuthenticationToken admin, String endentityAccessRule);
123 
124     /**
125      * Retrives a list of ids to profiles with non-existent CA ids, if the admin has root rule access.
126      * Otherwise an empty list is returned.
127      * @param admin admin
128      * @return list
129      */
130     List<Integer> getAuthorizedEndEntityProfileIdsWithMissingCAs(final AuthenticationToken admin);
131 
132     /** @return mapping of profile id (Integer) to profile name (String). */
133     Map<Integer, String> getEndEntityProfileIdToNameMap();
134 
135     /** Clear and reload end entity profile caches. */
136     void flushProfileCache();
137 
138     /**
139      * Finds a end entity profile by id.
140      * @param id ID
141      * @return EndEntityProfile (cloned) or null if it does not exist
142      */
143     EndEntityProfile getEndEntityProfile(int id);
144 
145     /**
146      * Get a copy of an EndEntityProfile.
147      * @param profilename profile
148      * @return EndEntityProfile (cloned) or null if it does not exist
149      */
150     EndEntityProfile getEndEntityProfile(String profilename);
151 
152     /**
153      * Returns a end entity profile's ID, given its name
154      * @param profilename profile
155      * @return the id
156      * @throws EndEntityProfileNotFoundException if profile wasn't found
157      */
158     int getEndEntityProfileId(String profilename) throws EndEntityProfileNotFoundException;
159 
160     /**
161      * Returns a end entity profiles name given it's id.
162      * @param id ID
163      * @return profile's name or null if profile id does not exist.
164      */
165     String getEndEntityProfileName(int id);
166 
167     /**
168      * Method to check if a CertificateProfile exists in any of the end entity
169      * profiles. Used to avoid desynchronization of certificate profile data.
170      * 
171      * @param certificateprofileid the CertificateProfile's id to search for.
172      * @return  a collection of EndEntityProfile names using the sought CertificateProfile
173      */
174     List<String> getEndEntityProfilesUsingCertificateProfile(int certificateprofileid);
175 
176     /**
177      * Fetches available certificate profiles associated with an end entity profile.
178      *
179      * No authorization required.
180      *
181      * @param admin the authentication of the caller.
182      * @param entityProfileId id of the end entity profile.
183      * @return a map of available certificate profiles names and IDs or an empty map.
184      * @throws EndEntityProfileNotFoundException if a profile with the ID entityProfileId could not be found
185      */
186     Map<String, Integer> getAvailableCertificateProfiles(AuthenticationToken admin, int entityProfileId) throws EndEntityProfileNotFoundException;
187     
188     /**
189      * Fetches the available CAs associated with an end entity profile.
190      *
191      * Authorization requirements:<pre>
192      * - /administrator
193      * - /endentityprofilesrules/&lt;end entity profile&gt;
194      * </pre>
195      *
196      * @param admin the authentication of the caller.
197      * @param entityProfileId id of the end entity profile.
198      * @return a map of available CA names and IDs or an empty map.
199      * @throws AuthorizationDeniedException if the authorization was denied (still not thrown).
200      * @throws EndEntityProfileNotFoundException if the end entity could not be found.
201      */
202     Map<String,Integer> getAvailableCasInProfile(AuthenticationToken admin, final int entityProfileId) throws AuthorizationDeniedException, EndEntityProfileNotFoundException;
203     
204     /**
205      * Method to check if a CA exists in any of the end entity profiles. Used to
206      * avoid desynchronization of CA data.
207      * @param caid the caid to search for.
208      * 
209      * @return true if CA exists in any of the end entity profiles.
210      */
211     boolean existsCAInEndEntityProfiles(int caid);
212 
213     /**
214      * Fetches the profile specified by profileId and profileType in XML format.
215      *
216      * Authorization requirements:<pre>
217      * - /administrator
218      * - /endentityprofilesrules/&lt;end entity profile&gt;
219      * - /ca/&lt;ca name&gt;
220      * </pre>
221      *
222      * For detailed documentation for how to parse an End Entity Profile XML, see the org.ejbca.core.model.ra.raadmin.EndEntity class.
223      * @param authenticationToken token
224      *
225      * @param profileId ID of the profile we want to retrieve.
226      * @return a byte array containing the specified profile in XML format.
227      * @throws AuthorizationDeniedException if the requesting admin wasn't authorized to the profile or any of the CA's therein
228      * @throws EndEntityProfileNotFoundException if the profile was not found
229      */
230     byte[] getProfileAsXml(AuthenticationToken authenticationToken, int profileId) throws AuthorizationDeniedException, EndEntityProfileNotFoundException;
231 }
