13  package org.ejbca.core.ejb.ra.raadmin;
14  
15  import javax.ejb.Local;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.ejbca.core.model.ra.raadmin.EndEntityProfile;
20  
21  /**
22   * @version $Id: EndEntityProfileSessionLocal.java 29308 2018-06-22 09:16:58Z andresjakobs $
23   */
24  @Local
25  public interface EndEntityProfileSessionLocal extends EndEntityProfileSession {
26  
27      /**
28       * A method designed to be called at startup time to (possibly) upgrade end
29       * entity profiles. This method will read all End Entity Profiles and as a
30       * side-effect upgrade them if the version if changed for upgrade. Can have
31       * a side-effect of upgrading a profile, therefore the Required transaction
32       * setting.
33       */
34      void initializeAndUpgradeProfiles();
35  
36      /** Helper method that checks if an administrator is authorized to all CAs present in the profiles "available CAs"
37       * 
38       * @param admin administrator to check.
39       * @param profile the profile to check
40       * @throws AuthorizationDeniedException if admin is not authorized to one of the available CAs in the profile
41       */
42      void authorizedToProfileCas(AuthenticationToken admin, EndEntityProfile profile) throws AuthorizationDeniedException;
43      
44      /** WARNING: This method must only be used when doing read_only operation on the profile. Otherwise
45       * any changes to the profile will affect the profile in the cache and thus affect all other threads.
46       * 
47       * Use the normal getEndEntityProfile for all edit operations, followed by a changeEndEntityProfile if you want to persist your changes and make 
48       * them visible for other threads.
49       * 
50       * This method exists only for speed purposes since a clone() done by the proper getEndEntityProfile method is slightly expensive.
51       * 
52       * Finds a end entity profile by id.
53       * @param id ID
54       * @return EndEntityProfile (shared in cache) or null if it does not exist
55       */
56      EndEntityProfile getEndEntityProfileNoClone(int id);
57  
58      /** WARNING: This method must only be used when doing read_only operation on the profile. Otherwise
59       * any changes to the profile will affect the profile in the cache and thus affect all other threads.
60       * 
61       * Use the normal getEndEntityProfile for all edit operations, followed by a changeEndEntityProfile if you want to persist your changes and make 
62       * them visible for other threads.
63       * 
64       * This method exists only for speed purposes since a clone() done by the proper getEndEntityProfile method is slightly expensive.
65       * 
66       * Finds an EndEntityProfile by name.
67       * @param profilename provile
68       * @return EndEntityProfile (shared in cache) or null if it does not exist
69       */
70      EndEntityProfile getEndEntityProfileNoClone(java.lang.String profilename);
71  }
