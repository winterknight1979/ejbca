13  package org.ejbca.core.ejb.crl;
14  
15  import java.security.cert.CRLException;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.cesecore.certificates.ca.CAInfo;
20  import org.cesecore.certificates.crl.CrlImportException;
21  import org.cesecore.certificates.crl.CrlStoreException;
22  
23  public interface ImportCrlSession {
24  
25      /**
26       * Method used to import a CRL to the database if it is newer than any CRL it already has
27       * 
28       * @param authenticationToken The administrator performing the operation
29       * @param cainfo of the CA that issued the CRL
30       * @param crlbytes the CRL in bytes
31       * @throws CrlImportException If a problem occurs when processing the imported CRL
32       * @throws CrlStoreException If a problem occurs when adding the imported CRL to the database
33       * @throws CRLException If a problem occurs when parsing the CRL
34       * @throws AuthorizationDeniedException If the administrator is not authorized to perform the required operations
35       */
36      void importCrl(final AuthenticationToken authenticationToken, final CAInfo cainfo, final byte[] crlbytes) 
37              throws CrlImportException, CrlStoreException, CRLException, AuthorizationDeniedException;
38  }
