13  package org.ejbca.core.ejb.audit;
14  
15  import java.util.List;
16  
17  import org.cesecore.audit.AuditLogEntry;
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  
21  /**
22   * Workaround for the very complex CESeCore query criteria API.
23   *
24   * Allow fetching of logs in a more standard JPA fashion.
25   *
26   * @version $Id: EjbcaAuditorSession.java 19902 2014-09-30 14:32:24Z anatom $
27   */
28  public interface EjbcaAuditorSession {
29      /**
30       * Selects audit log records from the IntegrityProtectedLogDevice.
31       * 
32       * User input should never be supplied in the whereClause or orderClause, since this will result in SQL
33       * injection attacks where more data than intended is returned.
34       * 
35       * @param token The auditors authentication token
36       * @param device Must be "IntegrityProtectedLogDevice" for now.
37       * @param firstResult first entry in result set
38       * @param maxResults number of results to return. '0' means no limit.
39       * @param whereClause corresponds to JPQL where statement. E.g. " a.columnSomething = ?0 AND a.columnSomethingElse &gt; ?1 ..."
40       * @param orderClause corresponds to JPQL order statement. E.g. " a.columnSomething ASC"
41       * @param parameters Array of parameters in the order they appear in the whereClause
42       * @return result of the executed query
43       * @throws AuthorizationDeniedException if token is not authorized to StandardRules.AUDITLOGSELECT
44       */
45      List<? extends AuditLogEntry> selectAuditLog(AuthenticationToken token, String device, int firstResult, int maxResults, String whereClause, String orderClause, List<Object> parameters) throws AuthorizationDeniedException;
46  }
