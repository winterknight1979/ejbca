13  package org.ejbca.core.ejb.audit.enums;
14  
15  import org.cesecore.audit.enums.ModuleType;
16  
17  /**
18   * EJBCA specific security audit event module types, for audit using CESecore's audit log.
19   * 
20   * A modules is a group of related functionality.
21   * These module types are part of EJBCA (not the core itself) and extend the list of already existing module types of CESeCore.
22   * 
23   * @see org.cesecore.audit.enums.ModuleTypes
24   * @see org.ejbca.core.ejb.audit.enums.EjbcaEventTypes
25   * @see org.ejbca.core.ejb.audit.enums.EjbcaServiceTypes
26   * @version $Id: EjbcaModuleTypes.java 26323 2017-08-15 17:24:53Z anatom $
27   */
28  public enum EjbcaModuleTypes implements ModuleType {
29      /** Registration Authority module. */
30      RA,
31      /** (Client) hardware token management module. */
32      HARDTOKEN,
33      /** Key recovery module. */
34      KEYRECOVERY,
35      /** Approval module. */
36      APPROVAL,
37      /** Approval Profiles module. */
38      APPROVAL_PROFILE,
39      /** Publisher module. */
40      PUBLISHER,
41      /** EJBCA background service module. */
42      SERVICE,
43      /** External logging module. */
44      CUSTOM,
45      /** Administrative web GUI module. */
46      ADMINWEB,
47      /** Blacklist module. */
48      BLACKLIST;    
49  
50      @Override
51      public boolean equals(ModuleType value) {
52          if (value == null) {
53              return false;
54          }
55          return this.toString().equals(value.toString());
56      }
57  }
