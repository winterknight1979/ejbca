13  package org.ejbca.core.ejb.authorization;
14  
15  import java.util.Map;
16  
17  import javax.ejb.Local;
18  
19  /**
20   * @see AuthorizationSystemSession
21   * @version $Id: AuthorizationSystemSessionLocal.java 25480 2017-03-14 17:28:27Z jeklund $
22   */
23  @Local
24  public interface AuthorizationSystemSessionLocal extends AuthorizationSystemSession {
25  
26      /** @return a Map&lt;category name, Map&lt;resource,resourceName&gt;&gt; */
27      Map<String, Map<String, String>> getAllResourceAndResourceNamesByCategory();
28  
29      /** @param ignoreLimitations limit
30       * @return a Map of all &lt;resource,resourceName&gt;, on this installation (optionally ignoring if certain resources is not in use) */
31      Map<String,String> getAllResources(boolean ignoreLimitations);
32  
33      /**
34       * Setup the initial Role with the a single EJBCA CLI RoleMember under the condition that this is system is
35       * connected to a database that has not been used for an installation so far.
36       * (Actual check for a "fresh" system is to confirm that there exists no Roles or CAs.)
37       * 
38       * @return true if this was a fresh system and the authorization module has now been initialized.
39       */
40      boolean initializeAuthorizationModule();
41  }
