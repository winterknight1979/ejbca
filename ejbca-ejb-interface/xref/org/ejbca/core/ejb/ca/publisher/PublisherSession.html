14  package org.ejbca.core.ejb.ca.publisher;
15  
16  import java.util.Collection;
17  import java.util.HashMap;
18  import java.util.Map;
19  
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.certificate.CertificateDataWrapper;
23  import org.cesecore.certificates.endentity.ExtendedInformation;
24  import org.cesecore.common.exception.ReferencesToItemExistException;
25  import org.ejbca.core.model.ca.publisher.BasePublisher;
26  import org.ejbca.core.model.ca.publisher.PublisherDoesntExistsException;
27  import org.ejbca.core.model.ca.publisher.PublisherExistsException;
28  
29  
30  /**
31   * Interface for publisher operations
32   *
33   * @version $Id: PublisherSession.java 30470 2018-11-12 11:02:11Z samuellb $
34   */
35  public interface PublisherSession {
36  
37      /**
38       * @param id id
39       * @return a BasePublisher or null if a publisher with the given id does not
40       *         exist. Uses cache to get the object as quickly as possible.
41       *         
42       */
43      BasePublisher getPublisher(int id);
44      
45      /**
46       * @param name name
47       * @return a BasePublisher or null if a publisher with the given name does
48       *         not exist. Uses cache to get the object as quickly as possible.
49       */
50      BasePublisher getPublisher(String name);
51      
52      /**
53       * @param id id
54       * @return the name of the publisher with the given id, null if none was found.
55       */
56      String getPublisherName(int id);
57      
58      /**
59       * @param id id
60       * @return the data hashmap of the publisher with the given id.
61       * @throws PublisherDoesntExistsException if there's no publisher with the given id.
62       */
63      Map<?, ?> getPublisherData(int id) throws PublisherDoesntExistsException;
64  
65      /** @return mapping of publisher id (Integer) to publisher name (String). */
66      HashMap<Integer,String> getPublisherIdToNameMap();
67  
68      /** @return mapping of publisher name (String) to publisher id (Integer). */
69      HashMap<String, Integer> getPublisherNameToIdMap();
70  
71      /**
72       * Adds a publisher to the database. Used for importing and exporting
73       * profiles from xml-files.
74       *
75       * @param admin AuthenticationToken of admin.
76       * @param id the publisher is.
77       * @param name the name of the publisher to add.
78       * @param publisher the publisher to add.
79       *
80       * @throws PublisherExistsException if publisher already exists.
81       * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
82       */
83      void addPublisher(AuthenticationToken admin, int id, String name, BasePublisher publisher) throws PublisherExistsException, AuthorizationDeniedException;
84  
85      /**
86       * Adds a publisher to the database. Used where it's not possible to pass a BasePublisher object,
87       * such as in the CLI tools (e.g. statedump).
88       * @param admin admin
89       * @param id id
90       * @param name name
91       * @param data data
92       * 
93       * @throws PublisherExistsException if publisher already exists.
94       * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
95       */
96      void addPublisherFromData(AuthenticationToken admin, int id, String name, Map<?, ?> data) throws PublisherExistsException, AuthorizationDeniedException;
97  
98      /**
99       * Adds a publisher to the database.
100      * 
101      * @param admin AuthenticationToken of admin
102      * @param name the name of the publisher to add.
103      * @param publisher the publisher to add
104      * @return the publisher ID as added
105      * 
106      * @throws PublisherExistsException if publisher already exists.
107      * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
108      */
109     int addPublisher(AuthenticationToken admin, String name, BasePublisher publisher) throws PublisherExistsException, AuthorizationDeniedException;
110 
111     /**
112      * Adds a publisher with the same content as the original.
113      * @param admin admin
114      * @param oldname name
115      * @param newname name
116      * @throws PublisherDoesntExistsException if publisher does not exist
117      * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
118      * @throws PublisherExistsException if publisher already exists.
119      */
120     void clonePublisher(AuthenticationToken admin, String oldname, String newname) throws PublisherDoesntExistsException, AuthorizationDeniedException, PublisherExistsException;
121 
122     /**
123      * Renames a publisher.
124      * @param admin admin
125      * @param oldname name
126      * @param newname name
127      * @throws PublisherExistsException if publisher already exists.
128      * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
129      */
130     void renamePublisher(AuthenticationToken admin, String oldname, String newname) throws PublisherExistsException, AuthorizationDeniedException;
131     
132     /** Updates publisher data.
133      *  
134      * @param admin AuthenticationToken of admin.
135      * @param name the name of the publisher to change.
136      * @param publisher the publisher to be added.
137      * 
138      * @throws AuthorizationDeniedException fail */
139     void changePublisher(AuthenticationToken admin, String name, BasePublisher publisher) throws AuthorizationDeniedException;
140 
141     /**
142      * Removes a publisher. References to the publisher from CA, certificate profiles and Multi Group Publishers
143      * are checked.
144      * 
145      * @param admin AuthenticationToken of admin.
146      * @param name the name of the publisher to remove.
147      * 
148      * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publisher
149      * @throws ReferencesToItemExistException if references exist from other items in the database.
150      */
151     void removePublisher(AuthenticationToken admin, String name) throws AuthorizationDeniedException, ReferencesToItemExistException;
152 
153     /**
154      * Stores the certificate to the given collection of publishers. See
155      * BasePublisher class for further documentation about function
156      * @param admin admin
157      * 
158      * @param publisherids
159      *            a Collection (Integer) of publisher IDs.
160      * @param certWrapper cart 
161      * @param password pwd
162      * @param userDN dn
163      * @param extendedinformation info
164      * @return true if successful result on all given publishers, if the publisher is configured to not publish the certificate 
165      * (for example publishing an active certificate when the publisher only publishes revoked), true is still returned because 
166      * the publishing operation succeeded even though the publisher did not publish the certificate.
167      * @throws AuthorizationDeniedException if access is denied to the CA issuing incert
168      * @see org.ejbca.core.model.ca.publisher.BasePublisher
169      */
170     boolean storeCertificate(AuthenticationToken admin, Collection<Integer> publisherids, CertificateDataWrapper certWrapper,
171             String password, String userDN, ExtendedInformation extendedinformation) throws AuthorizationDeniedException;
172 
173     /**
174      * Performs the same operation as the other storeCertificate method in this class, but performs a lookup for a CertificateData and Base64CertData object.
175      * 
176      * To avoid unnecessary database lookups, only use this method where the CertificateData object isn't immediately available. 
177      * @param admin admin
178      * @param publisherids ID 
179      * @param fingerprint fp
180      * @param password pwd
181      * @param userDN DN
182      * @param extendedinformation info 
183      * @return bool
184      * @throws AuthorizationDeniedException fail
185      */
186     boolean storeCertificate(AuthenticationToken admin, Collection<Integer> publisherids, String fingerprint,
187             String password, String userDN, ExtendedInformation extendedinformation) throws AuthorizationDeniedException; 
188 
189     /**
190      * Stores the CRL to the given collection of publishers. See BasePublisher
191      * class for further documentation about function
192      * @param admin admin
193      * 
194      * @param publisherids a Collection (Integer) of publisherids.
195      * @param incrl crl
196      * @param cafp fp
197      * @param number num 
198      * @param issuerDn the issuer of this CRL
199      * @return true if successful result on all given publishers
200      * @throws AuthorizationDeniedException if access was denied to the CA matching userDN
201      * @see org.ejbca.core.model.ca.publisher.BasePublisher
202      */
203     boolean storeCRL(AuthenticationToken admin, Collection<Integer> publisherids, byte[] incrl, String cafp,
204             int number, String issuerDn) throws AuthorizationDeniedException;
205 }
