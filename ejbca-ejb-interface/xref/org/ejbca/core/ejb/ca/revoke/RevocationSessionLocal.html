14  package org.ejbca.core.ejb.ca.revoke;
15  
16  import java.util.Collection;
17  import java.util.Date;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.authorization.AuthorizationDeniedException;
23  import org.cesecore.certificates.certificate.CertificateDataWrapper;
24  import org.cesecore.certificates.certificate.CertificateRevokeException;
25  
26  /**
27   * @version $Id: RevocationSessionLocal.java 24976 2017-01-05 14:14:46Z mikekushner $
28   */
29  @Local
30  public interface RevocationSessionLocal extends RevocationSession {
31  
32      /**
33       * Revokes a certificate, in the database and in publishers. Also handles re-activation of suspended certificates.
34       *
35       * Re-activating (unrevoking) a certificate have two limitations.
36       * 1. A password (for for example AD) will not be restored if deleted, only the certificate and certificate status and associated info will be restored
37       * 2. ExtendedInformation, if used by a publisher will not be used when re-activating a certificate 
38       * 
39       * The method leaves up to the caller to find the correct publishers and userDataDN.
40       *
41       * @param admin      Administrator performing the operation
42       * @param cdw        The certificate data
43       * @param publishers and array of publisher ids (Integer) of publishers to revoke the certificate in.
44       * @param revocationDate from when the certificates has been revoked
45       * @param reason     the reason of the revocation. (One of the RevokedCertInfo.REVOCATION_REASON constants.)
46       * @param userDataDN if an DN object is not found in the certificate use object from user data instead.
47       * @throws CertificateRevokeException (rollback) if certificate does not exist
48       * @throws AuthorizationDeniedException (rollback)
49       */
50      void revokeCertificate(AuthenticationToken admin, CertificateDataWrapper cdw, Collection<Integer> publishers, Date revocationDate, int reason,
51              String userDataDN) throws CertificateRevokeException, AuthorizationDeniedException;
52  
53      /**
54       * Revokes a certificate, in the database and in publishers, but does so in a new transaction. Also handles re-activation of suspended certificates.
55       * Will do so in a new transaction. 
56       * 
57       * This method only needs to be used in cases where further actions rely on a certificate already being revoked, such as when producing a CRL as 
58       * a result of CA revocation. 
59       *
60       * Re-activating (unrevoking) a certificate have two limitations.
61       * 1. A password (for for example AD) will not be restored if deleted, only the certificate and certificate status and associated info will be restored
62       * 2. ExtendedInformation, if used by a publisher will not be used when re-activating a certificate 
63       * 
64       * The method leaves up to the caller to find the correct publishers and userDataDN.
65       *
66       * @param admin      Administrator performing the operation
67       * @param cdw        The certificate data
68       * @param publishers and array of publisher ids (Integer) of publishers to revoke the certificate in.
69       * @param revocationDate from when the certificates has been revoked
70       * @param reason     the reason of the revocation. (One of the RevokedCertInfo.REVOCATION_REASON constants.)
71       * @param userDataDN if an DN object is not found in the certificate use object from user data instead.
72       * @throws CertificateRevokeException (rollback) if certificate does not exist
73       * @throws AuthorizationDeniedException (rollback)
74       */
75      void revokeCertificateInNewTransaction(final AuthenticationToken admin, final CertificateDataWrapper cdw, final Collection<Integer> publishers,
76              Date revocationDate, final int reason, final String userDataDN) throws CertificateRevokeException, AuthorizationDeniedException;        
77  
78  }
