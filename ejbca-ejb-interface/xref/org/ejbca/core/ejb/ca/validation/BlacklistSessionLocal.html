14  package org.ejbca.core.ejb.ca.validation;
15  
16  import javax.ejb.Local;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  import org.ejbca.core.model.validation.BlacklistEntry;
21  
22  /**
23   * Local interface for public key blacklist operations.
24   * 
25   * @version $Id: BlacklistSessionLocal.java 26315 2017-08-15 08:16:04Z anatom $
26   */
27  @Local
28  public interface BlacklistSessionLocal extends BlacklistSession {
29  
30      /**
31       * Gets a public key blacklist entry by cache or database.
32       * @param id ID
33       * @return a Blacklist or null if a public key blacklist with the given id does not exist. Uses cache to get the object as quickly as possible.
34       *         
35       */
36      BlacklistEntry getBlacklistEntry(int id);
37  
38      /**
39       * Returns the id of the public key blacklist entry with the given fingerprint.
40       * @param type type
41       * @param value value
42       * @return the id or 0 if the public key blacklist cannot be found.
43       */
44      int getBlacklistEntryId(String type, String value);
45  
46      /**
47       * Gets the fingerprint of the public key blacklist entry with the given id.
48       * @param id ID
49       * 
50       * @return the fingerprint of the public key blacklist with the given id or null if none was found.
51       */
52      String getBlacklistEntryFingerprint(int id);
53  
54      /**
55       * Adds a public key blacklist entry to the database. Used for importing and exporting
56       * entries from xml-files.
57       *
58       * @param admin AuthenticationToken of administrator.
59       * @param id the public key blacklist is.
60       * @param entry the public key blacklist to add.
61       *
62       * @throws AuthorizationDeniedException required access rights are ca_functionality/edit_publickeyblacklist
63       * @throws BlacklistExistsException if public key blacklist already exists.
64       */
65      void addBlacklistEntry(AuthenticationToken admin, int id, BlacklistEntry entry)
66              throws AuthorizationDeniedException, BlacklistExistsException;
67  
68      /**
69       * Flushes the public key blacklist entry cache to ensure that next time they are read from database.
70       */
71      void flushBlacklistEntryCache();
72  }
