14  package org.ejbca.core.ejb.ca.caadmin;
15  
16  import java.util.Set;
17  
18  import javax.ejb.Local;
19  
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.ca.CADoesntExistsException;
23  import org.cesecore.certificates.ca.CAOfflineException;
24  import org.cesecore.certificates.certificate.CertificateRevokeException;
25  
26  @Local
27  public interface CAAdminSessionLocal extends CAAdminSession {
28  
29  	/**
30       * A method designed to be called at startup time to speed up the (next)
31       * first request to a CA. This method will initialize the CA-cache with all
32       * CAs, if they are not already in the cache. Can have a side-effect of
33       * upgrading a CA, therefore the Required transaction setting.
34       */
35      void initializeAndUpgradeCAs();
36  
37      /** Method that loads a CA in order to possibly upgrade it, in a separate transaction. 
38       * This method is called from initializeAndUpgradeCAs in order to limit the transaction scope of CA upgrades.
39       * @param caid The CA to load/upgrade
40       * @throws CADoesntExistsException is the CA does not exist
41       */
42      void initializeAndUpgradeCA(Integer caid) throws CADoesntExistsException;
43  
44      /**
45       * Used by health-check. Validate that CAs are online and optionally performs
46       * a signature test.
47       * 
48       * @return an error message or an empty String if all are ok.
49       */
50      String healthCheck();
51  
52      /**
53       * Regenerates the CMS certificate for a CA.
54       * @param admin admin
55       * @param caid ca
56       * @throws AuthorizationDeniedException fail
57       * @throws CADoesntExistsException fail
58       * @throws CAOfflineException fail
59       * @throws CertificateRevokeException fail 
60       */
61      void renewAndRevokeCmsCertificate(AuthenticationToken admin, int caid) throws AuthorizationDeniedException, CADoesntExistsException,
62              CAOfflineException, CertificateRevokeException;
63      
64  //    /**
65  //     * Checks if at least one CA references a key validator.
66  //     * @param keyValidatorId
67  //     * @return true if there are no references.
68  //     * 
69  //     * @throws AuthorizationDeniedException if not authorized.
70  //     */
71  //    boolean existsKeyValidatorInCAs(int keyValidatorId) throws AuthorizationDeniedException;
72  //    
73      /** 
74       * This method returns a set containing IDs of all authorized key validators. This set will be the sum of the following:
75       * 
76       * * Unassigned key validators
77       * * Key validators assigned to CAs that the administrator has access to.
78       * @param admin admin
79       * 
80       * @return a Set of IDs of authorized key validators. 
81       */
82      Set<Integer> getAuthorizedKeyValidatorIds(AuthenticationToken admin);
83  }
