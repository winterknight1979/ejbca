13  package org.ejbca.core.ejb.keyrecovery;
14  
15  import javax.ejb.Local;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.certificates.certificate.CertificateWrapper;
19  import org.cesecore.keys.util.KeyPairWrapper;
20  import org.ejbca.core.model.keyrecovery.KeyRecoveryInformation;
21  
22  /**
23   * Local interface for KeyRecoverySession.
24   */
25  @Local
26  public interface KeyRecoverySessionLocal extends KeyRecoverySession {
27  
28      /**
29       * Adds a certificates keyrecovery data to the database. This method allows a crypto token to be specified,
30       * and does not require the CA to be present on the system.
31       * 
32       * <p>Additionally, this method expects the caller to do the authorization checks, using the authorization
33       * HashMaps available on the RA.
34       *
35       * @param admin the administrator calling the function (used for audit logging)
36       * @param certificate the certificate used with the keypair.
37       * @param username of the administrator
38       * @param keypair the actual keypair to save.
39       * @param cryptoTokenId ID of crypto token to use to encrypt key.
40       * @param keyAlias key alias in crypto token to use to encrypt key.
41       *
42       * @return false if the certificates keyrecovery data already exists, or if the crypto token was offline.
43       * @see KeyRecoverySession#addKeyRecoveryData
44       */
45      boolean addKeyRecoveryDataInternal(AuthenticationToken admin, CertificateWrapper certificate, String username, KeyPairWrapper keypair, int cryptoTokenId,
46              String keyAlias);
47  
48      /**
49       * Returns the keyrecovery data for a user. Observe only one certificates
50       * key can be recovered for every user at the time.
51       * 
52       * <p>This method expects the caller to do the authorization checks, using the authorization
53       * HashMaps available on the RA.
54       * 
55       * <p><b>Note:</b>The returned KeyRecoveryObject does not contain a certificate. The caller
56       * is responsible to search for the certificate based on the Issuer DN and Certificate Serial Number.
57       * 
58       * @param admin the administrator calling the function (used for audit logging)
59       * @param username Username of the end entity
60       * @param cryptoTokenId ID of crypto token to use to encrypt key.
61       * @param keyAlias key alias in crypto token to use to encrypt key.
62       * @return the marked keyrecovery data or null if none can be found. Note that the certificate property will be null.
63       * 
64       * @see KeyRecoverySession#recoverKeys
65       */
66      KeyRecoveryInformation recoverKeysInternal(AuthenticationToken admin, String username, int cryptoTokenId, String keyAlias);
67      
68      /**
69       * Marks a users certificate for key recovery. This method performs no authorization checks and should only be called
70       * when local key pair generation is used and checks has already been performed by the instance holding UserData in its
71       * database
72       * 
73       * @param admin requesting administrator
74       * @param certificate to be marked for recovery
75       * @param username of the end entity related to certificate
76       * @return true if flag was set in database successfully
77       */
78      boolean markAsRecoverableInternal(AuthenticationToken admin, CertificateWrapper certificate, String username);
79  }
