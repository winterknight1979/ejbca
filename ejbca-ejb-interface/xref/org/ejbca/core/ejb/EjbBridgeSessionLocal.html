69  /**
70   * Due to the lack of standardization in JEE5 there is no way to lookup local interfaces.
71   * 
72   * This Stateless Session Bean (SSB) act as a bridge between calling classes in the same JVM,
73   * and the real ejb references.
74   * 
75   * This will allow us to define a single (this) local EJB in all web.xml and ejb-jar.xml files
76   * and are then free to change and move around SSBs and their interfaces without XML changes.
77   * 
78   * @version $Id: EjbBridgeSessionLocal.java 29151 2018-06-07 15:11:05Z jeklund $
79   */
80  @Local
81  public interface EjbBridgeSessionLocal {
