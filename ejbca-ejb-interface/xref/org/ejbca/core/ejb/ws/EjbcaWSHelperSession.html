13  package org.ejbca.core.ejb.ws;
14  
15  import org.cesecore.authentication.tokens.AuthenticationToken;
16  import org.cesecore.certificates.ca.CADoesntExistsException;
17  import org.cesecore.certificates.endentity.EndEntityInformation;
18  import org.ejbca.core.EjbcaException;
19  import org.ejbca.core.protocol.ws.objects.UserDataVOWS;
20  
21  /**
22   * @version $Id: EjbcaWSHelperSession.java 27112 2017-11-10 11:28:44Z anatom $
23   */
24  public interface EjbcaWSHelperSession {
25  
26      /**
27       * Low-level method to convert between a UserDataVOWS data structure from the web service to an EndEntityInformation.
28       * This method does not perform any verification and the method {@link #convertUserDataVOWS(AuthenticationToken, UserDataVOWS)}is preferred 
29       * as it performs sanity checks that profiles and CAs exist.
30       * 
31       * @param userdata UserDataVOWS from the WS
32       * @param caid CA ID
33       * @param endentityprofileid End Entity Profile ID
34       * @param certificateprofileid Certificate Profile ID
35       * @param hardtokenissuerid Hard Token Issuer ID, or 0 to not use.
36       * @param tokenid Token type, TOKEN_TYPE_* constant
37       * @param useRawSubjectDN if true ExtendedInformation in the returned EndEntityInformation will be populated with the Raw subject DN from the UserDataVOWS
38       * @return New EndEntityInformation object
39       * @throws EjbcaException if there are errors in the UserDataVOWS, such as incorrectly formatted validity dates.
40       */
41      EndEntityInformation convertUserDataVOWSInternal(final UserDataVOWS userdata, final int caid, final int endentityprofileid, final int certificateprofileid, final int hardtokenissuerid, final int tokenid, final boolean useRawSubjectDN) throws EjbcaException;
42      
43      /**
44       * Method to convert between a UserDataVOWS data structure from the web service to an EndEntityInformation.
45       * @param admin Authentication token. Used when looking up CA and profile IDs.
46       * @param userdata UserDataVOWS from the WS
47       * @return New EndEntityInformation object
48       * @throws CADoesntExistsException If the CA referenced by the UserDataVOWS does not exist
49       * @throws EjbcaException if any of the referenced profiles does not exist, or there are other errors in the UserDataVOWS object, such as incorrectly formatted validity dates.
50       */
51      EndEntityInformation convertUserDataVOWS(final AuthenticationToken admin, final UserDataVOWS userdata) throws CADoesntExistsException, EjbcaException;
52      
53      /**
54       * Low-level method that converts an EndEntityInformation object to a UserDataVOWS.
55       * Used in tests.
56       * @param endEntityInformation EndEntityInformation object to convert to a UserDataVOWS
57       * @param caname Name of CA. Will be used as is to the UserDataVOWS object.
58       * @param endentityprofilename Name of end entity profile. Will be used as is to the UserDataVOWS object.
59       * @param certificateprofilename Name of certificate profile. Will be used as is to the UserDataVOWS object.
60       * @param hardtokenissuername Name of hard token issuer. Will be used as is to the UserDataVOWS object.
61       * @param tokenname Token type name. Will be used as is to the UserDataVOWS object.
62       * @return New UserDataVOWS object
63       */
64      UserDataVOWS convertEndEntityInformation(final EndEntityInformation endEntityInformation, final String caname, final String endentityprofilename, 
65              final String certificateprofilename, final String hardtokenissuername, final String tokenname);
66      
67      /**
68       * Method that converts an EndEntityInformation object to a UserDataVOWS.
69       * Used in the findUser and findUserData calls.
70       * @param endEntityInformation EndEntityInformation object to convert to a UserDataVOWS
71       * @return New UserDataVOWS object
72       * @throws EjbcaException if any of the profiles or the hard token issuer referenced by the EndEntityInformation does not exist
73       * @throws CADoesntExistsException if the CA referenced by the EndEntityInformation does not exist
74       */
75      UserDataVOWS convertEndEntityInformation(final EndEntityInformation endEntityInformation) throws EjbcaException, CADoesntExistsException;
76      
77  }
