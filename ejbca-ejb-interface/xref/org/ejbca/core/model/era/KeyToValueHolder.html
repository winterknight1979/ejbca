13  package org.ejbca.core.model.era;
14  
15  import java.io.Serializable;
16  
17  /**
18   * Generic implementation which will hold any serializable object, as well as its ID and name.
19   * 
20   * @version $Id: KeyToValueHolder.java 24056 2016-07-29 10:10:23Z mikekushner $
21   * @param <T> Type
22   */
23  public class KeyToValueHolder<T extends Serializable> implements Serializable {
24  
25      private static final long serialVersionUID = 1L;
26  
27      private final int id;
28  	private final String name;
29  	private final T value;
30  
31  	public KeyToValueHolder(Integer id, String name, T value) {
32  		this.id = id;
33  		this.name = name;
34  		this.value = value;
35  	}
36  
37  	public int getId() { return id; }
38  	public String getName() { return name; }
39  	public T getValue() { return value; }
40  }
