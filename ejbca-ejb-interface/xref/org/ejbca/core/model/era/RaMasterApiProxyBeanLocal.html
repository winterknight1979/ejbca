13  package org.ejbca.core.model.era;
14  
15  import javax.ejb.Local;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  
19  /**
20   * Interface for EJB access to the RaMasterApi proxy singleton
21   *
22   * @version $Id: RaMasterApiProxyBeanLocal.java 29504 2018-07-17 13:16:50Z bastianf $
23   */
24  @Local
25  public interface RaMasterApiProxyBeanLocal extends RaMasterApi {
26  
27      /**
28       *
29       * @param apiType the implementation of RaMasterApi to check for
30       * @return returns true if an API of a certain type is available
31       */
32      boolean isBackendAvailable(Class<? extends RaMasterApi> apiType);
33  
34      /**
35       * De-prioritizes the local RA Master API implementation, causing it to not be called if a remote connection is available.
36       * Used in tests, to test "remote" peer connections to localhost.
37       */
38      void deferLocalForTest();
39  
40      /** @param authenticationToken token
41       * @param username user
42       * @return a RaCertificateSearchResponse from a search with a given username */
43      RaCertificateSearchResponse searchForCertificatesByUsername(final AuthenticationToken authenticationToken, final String username);
44  }
