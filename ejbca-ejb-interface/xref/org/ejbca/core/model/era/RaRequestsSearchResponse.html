13  package org.ejbca.core.model.era;
14  
15  import java.io.Serializable;
16  import java.util.ArrayList;
17  import java.util.HashMap;
18  import java.util.List;
19  import java.util.Map;
20  
21  /**
22   * Response of approval listing from RA GUI.
23   * 
24   * @version $Id: RaRequestsSearchResponse.java 23385 2016-05-10 14:16:47Z samuellb $
25   */
26  public class RaRequestsSearchResponse implements Serializable {
27  
28      // TODO: Make Externalizable instead to handle for future versioning. And consider creating a common base class for this one and RaCertificateSearchResponse
29      
30      private static final long serialVersionUID = 1L;
31  
32      private List<RaApprovalRequestInfo> approvalRequests = new ArrayList<>();
33      private boolean mightHaveMoreResults = false;
34  
35      public List<RaApprovalRequestInfo> getApprovalRequests() { return approvalRequests; }
36      public void getApprovalRequests(final List<RaApprovalRequestInfo> approvalRequests) { this.approvalRequests = approvalRequests; }
37  
38      public boolean isMightHaveMoreResults() { return mightHaveMoreResults; }
39      public void setMightHaveMoreResults(boolean mightHaveMoreResults) { this.mightHaveMoreResults = mightHaveMoreResults; }
40      
41      public void merge(final RaRequestsSearchResponse other) {
42          final Map<Integer,RaApprovalRequestInfo> cdwMap = new HashMap<>();
43          for (final RaApprovalRequestInfo approvalRequest : approvalRequests) {
44              cdwMap.put(approvalRequest.getId(), approvalRequest);
45          }
46          for (final RaApprovalRequestInfo approvalRequest : other.approvalRequests) {
47              cdwMap.put(approvalRequest.getId(), approvalRequest);
48          }
49          this.approvalRequests.clear();
50          this.approvalRequests.addAll(cdwMap.values());
51          if (other.isMightHaveMoreResults()) {
52              setMightHaveMoreResults(true);
53          }
54      }
55      
56  }
