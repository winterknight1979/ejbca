86      private static final Logger log = Logger.getLogger(EjbLocalHelper.class);
87  	private static Context initialContext = null;
88  	private static ReentrantLock initialContextLock = new ReentrantLock(true);
89  	// Static is more performant, but a failed JEE5 lookup from one module would block all other JEE5 lookups
90  	private /*static*/ boolean useEjb31GlobalJndiName = false;
91  	
92  	public static final String DEFAULT_MODULE = "ejbca-ejb";
93  
94  	private static Context getInitialContext() throws NamingException {
95  		try {
96  			initialContextLock.lock();
97  			if (initialContext == null) {
98  				initialContext = new InitialContext();
99  			}
100 			return initialContext;
101 		} finally {
102 			initialContextLock.unlock();
103 		}
104 	}
105 
106 	/**
107 	 * Requires a "ejb-local-ref" definition in web.xml and ejb-jar.xml from all accessing components
108 	 * or an application server that support global JNDI names (introduced in EJB 3.1).
109 	 * @return a reference to the bridge SSB
110 	 * 
111 	 * @throws LocalLookupException if local lookup couldn't be made.
112 	 */
113 	private EjbBridgeSessionLocal getEjbLocal() {
114 		EjbBridgeSessionLocal ret = null;
115 		try {
116 			if (!useEjb31GlobalJndiName) {
117 				ret = (EjbBridgeSessionLocal) getInitialContext().lookup("java:comp/env/EjbBridgeSession");
118 			}
119 		} catch (NamingException e) {
120 			// Let's try to use the EJB 3.1 syntax for a lookup. For example, JBoss 6.0.0.FINAL supports this from our CMP TCP threads, but ignores the ejb-ref from web.xml..
121 			// java:global[/<app-name>]/<module-name>/<bean-name>[!<fully-qualified-interface-name>]
122 			useEjb31GlobalJndiName = true;	// So let's not try what we now know is a failing method ever again..
123 			if (log.isDebugEnabled()) {
124 	            log.debug("Failed JEE5 version of EjbBridgeSessionLocal JNDI lookup. All future lookups will JEE6 version lookups.");
125 			}
126 		}
127 		try {
128 			if (useEjb31GlobalJndiName) {
129 				ret = (EjbBridgeSessionLocal) getInitialContext().lookup("java:global/ejbca/"+DEFAULT_MODULE+"/EjbBridgeSessionBean!org.ejbca.core.ejb.EjbBridgeSessionLocal");
130 			}
131 		} catch (NamingException e) {
132 			throw new LocalLookupException("Cannot lookup EjbBridgeSessionLocal.", e);
133 		}
134 		return ret;
135 	}

188     /** 
189      * Dynamically loads the StatedumpSession with JNDI. It's usually not available in the EJBCA source tree,
190      * and in this case this is properly handled  by returning null.
191      * 
192      * @return A statedump session object, or null if not available.
193      */
194     public StatedumpSessionLocal getStatedumpSession() {
195         try {
196             if (!useEjb31GlobalJndiName) {
197                 return (StatedumpSessionLocal) getInitialContext().lookup("java:comp/env/StatedumpSession");
198             }
199         } catch (NamingException e) {
200             // NOPMD ignore and continue
201         }
202         
203         // Try using EJB 3.1 name
204         try {
205             return (StatedumpSessionLocal) getInitialContext().lookup("java:global/ejbca/"+StatedumpSession.STATEDUMP_MODULE+"/StatedumpSessionBean!org.ejbca.statedump.ejb.StatedumpSessionLocal");
206         } catch (NamingException e) {
207             return null; // this is the common case, since statedump is an internal tool and is not included with EJBCA
208         }
209     }
