39  /** 
40   * Servlet implementing server side of the Certificate Store.
41   * For a detailed description see RFC 4387.
42   * 
43   * @version  $Id: CertStoreServlet.java 24974 2017-01-04 14:48:17Z anatom $
44   */
45  public class CertStoreServlet extends StoreServletBase {
46  	
47  	private static final long serialVersionUID = 1L;
48  
49      @EJB
50      private CertificateStoreSessionLocal certificateStoreSession;
51  
52  	private final static Logger log = Logger.getLogger(CertStoreServlet.class);
53  
54  	@Override
55  	public void init(ServletConfig config) throws ServletException {
56  		super.init(config);
57  	}
58  
59  	@Override
60  	public void iHash(String iHash, HttpServletResponse resp, HttpServletRequest req) throws IOException, ServletException {
61  	    returnCerts( this.certCache.findLatestByIssuerDN(HashID.getFromB64(iHash)), resp, iHash );
62  		return;
63  	}
64  
65  
66  	@Override
67  	public void sKIDHash(String sKIDHash, HttpServletResponse resp, HttpServletRequest req, String name) throws IOException, ServletException {
68  	    returnCert( this.certCache.findBySubjectKeyIdentifier(HashID.getFromB64(sKIDHash)), resp, name );
69  	}
70  
71  	@Override
72  	public void sKIDHash(String sKIDHash, HttpServletResponse resp, HttpServletRequest req) throws IOException, ServletException {
73  		sKIDHash( sKIDHash, resp, req, sKIDHash );
74  	}
75  
76  	@Override
77  	public void sHash(String sHash, HttpServletResponse resp, HttpServletRequest req) throws IOException, ServletException {
78  	    final X509Certificate cert = this.certCache.findLatestBySubjectDN(HashID.getFromB64(sHash));
79  		returnCert( cert, resp, sHash);
80  	}
81  
82  	@Override
83  	public void printInfo(X509Certificate cert, String indent, PrintWriter pw, String url) {
84  	    // Important to escape output that have an even small chance of coming from untrusted source
85  		pw.println(indent+HTMLTools.htmlescape(cert.getSubjectX500Principal().toString()));
86  		pw.println(indent+" "+RFC4387URL.sHash.getRef(url, HashID.getFromSubjectDN(cert)));
87  		pw.println(indent+" "+RFC4387URL.iHash.getRef(url, HashID.getFromSubjectDN(cert)));
88  		pw.println(indent+" "+RFC4387URL.sKIDHash.getRef(url, HashID.getFromKeyID(cert)));
89  	}
90  
91  	@Override
92  	public String getTitle() {
93  		return "CA certificates";
94  	}

96  	private void returnCert(X509Certificate cert, HttpServletResponse resp, String name) throws IOException, ServletException {
97  		if (cert==null) {
98  			resp.sendError(HttpServletResponse.SC_NO_CONTENT, "No certificate with hash: "+HTMLTools.htmlescape(name));
99  			return;
100 		}
101 		final byte encoded[];
102 		try {
103 			encoded = cert.getEncoded();
104 		} catch (CertificateEncodingException e) {
105 			throw new ServletException(e);
106 		}
107 		resp.setContentType("application/pkix-cert");
108 		resp.setHeader("Content-disposition", "attachment; filename=\"" + StringTools.stripFilename(name+".der") + "\"");
109 		resp.setContentLength(encoded.length);
110 		resp.getOutputStream().write(encoded);
111 	}
112 	
113 	private void returnCerts(X509Certificate certs[], HttpServletResponse resp, String name) throws IOException, ServletException {
114 		if (certs==null) {
115 			resp.sendError(HttpServletResponse.SC_NO_CONTENT, "No certificates with issuer hash DN: "+HTMLTools.htmlescape(name));
116 			return;
117 		}
118 		final Multipart mp = new MimeMultipart();// mixed is default
119 		try {
120 			resp.setContentType(mp.getContentType());
121 			for( int i=0; i<certs.length; i++ ) {
122 				final String filename = "cert" + name + '-' + i + ".der";
123 				if (log.isDebugEnabled()) {
124 					log.debug("Returning certificate with issuerDN '"+CertTools.getIssuerDN(certs[i])+"' and subjectDN '"+CertTools.getSubjectDN(certs[i])+"'. Filename="+filename);
125 				}
126 				final InternetHeaders headers = new InternetHeaders();
127 				headers.addHeader("Content-type", "application/pkix-cert");
128 				headers.addHeader("Content-disposition", "attachment; filename=\""+StringTools.stripFilename(filename)+"\"");
129 				mp.addBodyPart(new MimeBodyPart(headers,certs[i].getEncoded()));
130 			}
131 			if (log.isTraceEnabled()) {
132 				log.trace("content type: "+mp.getContentType());				
133 			}
134 			mp.writeTo(resp.getOutputStream());
135 			resp.flushBuffer();
136 		} catch (CertificateEncodingException e) {
137 			throw new ServletException(e);
138 		} catch (MessagingException e) {
139 			throw new ServletException(e);
140 		}
141 	}
