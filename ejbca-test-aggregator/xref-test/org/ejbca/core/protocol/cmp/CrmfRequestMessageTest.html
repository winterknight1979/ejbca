154     private PKIMessage createPKIMessage(final String issuerDN, final String subjectDN) throws InvalidAlgorithmParameterException, IOException {
155 		KeyPair keys = KeyTools.genKeys("1024", "RSA");
156 		ASN1EncodableVector optionalValidityV = new ASN1EncodableVector();
157 		org.bouncycastle.asn1.x509.Time nb = new org.bouncycastle.asn1.x509.Time(new DERGeneralizedTime("20030211002120Z"));
158 		org.bouncycastle.asn1.x509.Time na = new org.bouncycastle.asn1.x509.Time(new Date()); 
159 		optionalValidityV.add(new DERTaggedObject(true, 0, nb));
160 		optionalValidityV.add(new DERTaggedObject(true, 1, na));
161 		OptionalValidity myOptionalValidity = OptionalValidity.getInstance(new DERSequence(optionalValidityV));
162 		
163 		CertTemplateBuilder myCertTemplate = new CertTemplateBuilder();
164 		myCertTemplate.setValidity( myOptionalValidity );
165 		myCertTemplate.setIssuer(new X500Name(issuerDN));
166         myCertTemplate.setSubject(new X500Name(subjectDN));
167         SubjectPublicKeyInfo keyInfo = SubjectPublicKeyInfo.getInstance(keys.getPublic().getEncoded());
168         myCertTemplate.setPublicKey(keyInfo);    
169 		ByteArrayOutputStream   bOut = new ByteArrayOutputStream();
170 		DEROutputStream         dOut = new DEROutputStream(bOut);
171 		ExtensionsGenerator extgen = new ExtensionsGenerator();
172 		int bcku = X509KeyUsage.digitalSignature | X509KeyUsage.keyEncipherment | X509KeyUsage.nonRepudiation;
173 		X509KeyUsage ku = new X509KeyUsage(bcku);
174 		bOut = new ByteArrayOutputStream();
175 		dOut = new DEROutputStream(bOut);
176 		dOut.writeObject(ku);
177 		byte[] value = bOut.toByteArray();
178 		extgen.addExtension(Extension.keyUsage, false, new DEROctetString(value));
179 		myCertTemplate.setExtensions(extgen.generate());
180 		        
181 		CertRequest myCertRequest = new CertRequest(4, myCertTemplate.build(), null);
182 		        
183 		ProofOfPossession myProofOfPossession = new ProofOfPossession();
184 		AttributeTypeAndValue av = new AttributeTypeAndValue(CRMFObjectIdentifiers.id_regCtrl_regToken, new DERUTF8String("foo123"));
185 		AttributeTypeAndValue[] avs = {av};
186 		CertReqMsg myCertReqMsg = new CertReqMsg(myCertRequest, myProofOfPossession, avs);
187         CertReqMessages myCertReqMessages = new CertReqMessages(myCertReqMsg);
188         
189         PKIHeaderBuilder myPKIHeader = new PKIHeaderBuilder(2, new GeneralName(new X500Name("CN=bogusSubject")), new GeneralName(new X500Name("CN=bogusIssuer")));
190         myPKIHeader.setMessageTime(new DERGeneralizedTime(new Date()));
191         myPKIHeader.setSenderNonce(new DEROctetString(CmpMessageHelper.createSenderNonce()));
192         myPKIHeader.setTransactionID(new DEROctetString(CmpMessageHelper.createSenderNonce()));
193 
194         PKIBody myPKIBody = new PKIBody(0, myCertReqMessages);
195         PKIMessage myPKIMessage = new PKIMessage(myPKIHeader.build(), myPKIBody);
196         return myPKIMessage;
197     }

242     private void doNovosecClientRequest(final String sigAlg, final String digestAlg, final String expectedAlgOid) throws IOException, InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException, InvalidAlgorithmParameterException, SignatureException, IllegalStateException, OperatorCreationException, CertificateException {
243     	// Check that we can parse a request from  Novosec (patched by EJBCA).
244     	// Read an initialization request with a signature POP and signature protection to see that we can process it
245     	{
246     		ASN1InputStream in = new ASN1InputStream(novosecsigpopir);
247     		try {
248     		ASN1Primitive derObject = in.readObject();
249     		PKIMessage req = PKIMessage.getInstance(derObject);
250     		//log.info(req.toString());
251     		// Verify should be ok if we do not allow RA verify POP here
252     		CrmfRequestMessage msg = new CrmfRequestMessage(req, "CN=AdminCA1", false, "CN");
253     		assertTrue(msg.verify());
254     		// Since we don't have RA POP we can't test for that...
255     		assertEquals("CN=AdminCA1,O=EJBCA Sample,C=SE", msg.getIssuerDN());
256     		assertEquals("CN=abc123rry2942812801980668853,O=PrimeKey Solutions AB,C=SE", msg.getRequestDN());
257     		assertEquals("abc123rry2942812801980668853", msg.getUsername());
258     		assertEquals("foo123", msg.getPassword());
259     		// Verify signature protection
260     		AlgorithmIdentifier algId = msg.getMessage().getHeader().getProtectionAlg();
261     		String oid = algId.getAlgorithm().getId();
262     		assertEquals(PKCSObjectIdentifiers.sha1WithRSAEncryption.getId(), oid);
263     		// Check that this is an old message, created before ECA-2104, using null instead of DERNull as algorithm parameters.
264     		ASN1Encodable pp = algId.getParameters();
265     		assertNull(pp);
266     		// Try to verify, it should work good even though the small bug in ECA-2104, since we don't use algorithm parameters for RSA-PKCS signatures
267     		PublicKey pubKey = msg.getRequestPublicKey();
268     		assertTrue(CmpMessageHelper.verifyCertBasedPKIProtection(msg.getMessage(), pubKey));
269     		// Verify that our verification routine does not give positive result for any other keys
270     		KeyPair keys = KeyTools.genKeys("512", "RSA");
271     		assertFalse(CmpMessageHelper.verifyCertBasedPKIProtection(msg.getMessage(), keys.getPublic()));
272     		} finally {
273     		    in.close();
274     		}
275     	}
276     	// Re-protect the message, now fixed by ECA-2104
277     	{
278     		ASN1InputStream in = new ASN1InputStream(novosecsigpopir);
279     		try {
280     		ASN1Primitive derObject = in.readObject();
281     		PKIMessage myPKIMessage = PKIMessage.getInstance(derObject);
282     		KeyPair keys = KeyTools.genKeys("512", "RSA");
283     		X509Certificate signCert = CertTools.genSelfCert("CN=CMP Sign Test", 3650, null, keys.getPrivate(), keys.getPublic(), sigAlg, false);
284     		// Re-sign the message
285     		Collection<Certificate> signCertChain = new ArrayList<Certificate>();
286     		signCertChain.add(signCert);
287     		byte[] newmsg = CmpMessageHelper.signPKIMessage(myPKIMessage, signCertChain, keys.getPrivate(), digestAlg, "BC");
288     		in.close();
289     		in = new ASN1InputStream(newmsg);		
290     		derObject = in.readObject();
291     		PKIMessage pkimsg = PKIMessage.getInstance(derObject);
292     		// We have to do this twice, because Novosec caches ProtectedBytes in the PKIMessage object, so we need to 
293     		// encode it and re-decode it again to get the changes from ECA-2104 encoded correctly.
294     		// Not needed when simply signing a new message that you create, only when re-signing 
295     		newmsg = CmpMessageHelper.signPKIMessage(pkimsg, signCertChain, keys.getPrivate(), digestAlg, "BC");
296     		in.close();
297     		in = new ASN1InputStream(newmsg);
298     		derObject = in.readObject();
299     		pkimsg = PKIMessage.getInstance(derObject);
300     		AlgorithmIdentifier algId = pkimsg.getHeader().getProtectionAlg();
301     		String oid = algId.getAlgorithm().getId();
302     		assertEquals(expectedAlgOid, oid);
303     		// Check that we have DERNull and not plain java null as algorithm parameters.
304     		ASN1Encodable pp = algId.getParameters();
305     		assertNotNull(pp);
306     		assertEquals(DERNull.class.getName(), pp.getClass().getName());
307     		// Try to verify, also verify at the same time that encoding decoding of the signature works
308     		assertTrue(CmpMessageHelper.verifyCertBasedPKIProtection(pkimsg, keys.getPublic()));
309     		// Verify that our verification routine does not give positive result for any other keys
310     		CrmfRequestMessage msg = new CrmfRequestMessage(pkimsg, "CN=AdminCA1", false, "CN");
311     		assertTrue(msg.verify());
312     		PublicKey pubKey = msg.getRequestPublicKey();
313     		assertFalse(CmpMessageHelper.verifyCertBasedPKIProtection(pkimsg, pubKey));
314     		} finally {
315     		    in.close();
316     		}
317     	}
318     }
