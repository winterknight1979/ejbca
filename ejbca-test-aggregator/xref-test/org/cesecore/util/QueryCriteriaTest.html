50      @Test
51      public void testBadParameterQuery() {
52          try {
53              QueryCriteria criteria = QueryCriteria.create().add(Criteria.eq(FakeEntity.FIELDNAME1, "value"));
54              QueryGenerator.generator(FakeEntity.class, criteria, "a").generate();
55          } catch (QueryParameterException e) {
56              log.error("QueryCriteria unable to detect existing field.", e);
57              Assert.fail("QueryCriteria unable to detect existing field.");
58          }
59          try {
60              QueryCriteria criteria = QueryCriteria.create().add(Criteria.eq("invalidParameterName", "value"));
61              QueryGenerator.generator(FakeEntity.class, criteria, "a").generate();
62              Assert.fail("QueryCriteria unable to detect non-existing field.");
63          } catch (QueryParameterException e) {
64              // Expected
65          }
66      }

77      @Test
78      public void testBasicQueryEqual() {
79          final QueryCriteria qc1 = QueryCriteria.create().add(Criteria.eq(FakeEntity.FIELDNAME1, DUMMY_VALUE_STR1));
80          QueryGenerator generator1 = QueryGenerator.generator(FakeEntity.class, qc1, "a");
81          Assert.assertEquals(BAD_QUERY_GENERATED, " WHERE a.field1 = :field10", generator1.generate());
82          Assert.assertEquals(BAD_QUERY_PARAMTERS, 1, generator1.getParameterKeys().size());
83          Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_STR1, generator1.getParameterValue(generator1.getParameterKeys().iterator().next()));
84          final QueryCriteria qc2 = QueryCriteria.create().add(Criteria.eq(FakeEntity.FIELDNAME2, DUMMY_VALUE_INT1));
85          QueryGenerator generator2 = QueryGenerator.generator(FakeEntity.class, qc2, "a");
86          Assert.assertEquals(BAD_QUERY_GENERATED, " WHERE a.field2 = :field20", generator2.generate());
87          Assert.assertEquals(BAD_QUERY_PARAMTERS, 1, generator2.getParameterKeys().size());
88          Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_INT1, generator2.getParameterValue(generator2.getParameterKeys().iterator().next()));
89      }

192     @Test
193     public void testCombinedQueryEqual() {
194         // Or
195         final QueryCriteria qc1 = QueryCriteria.create().add(Criteria.or(Criteria.eq(FakeEntity.FIELDNAME1, DUMMY_VALUE_STR1),Criteria.eq(FakeEntity.FIELDNAME2, DUMMY_VALUE_INT1)));
196         QueryGenerator generator1 = QueryGenerator.generator(FakeEntity.class, qc1, "a");
197         Assert.assertEquals(BAD_QUERY_GENERATED, " WHERE a.field1 = :field10 OR a.field2 = :field20", generator1.generate());
198         Assert.assertEquals(BAD_QUERY_PARAMTERS, 2, generator1.getParameterKeys().size());
199         final Iterator<String> i1 = generator1.getParameterKeys().iterator();
200         Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_STR1, generator1.getParameterValue(i1.next()));
201         Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_INT1, generator1.getParameterValue(i1.next()));
202         // And
203         final QueryCriteria qc2 = QueryCriteria.create().add(Criteria.and(Criteria.eq(FakeEntity.FIELDNAME1, DUMMY_VALUE_STR1), Criteria.eq(FakeEntity.FIELDNAME2, DUMMY_VALUE_INT1)));
204         QueryGenerator generator2 = QueryGenerator.generator(FakeEntity.class, qc2, "a");
205         Assert.assertEquals(BAD_QUERY_GENERATED, " WHERE a.field1 = :field10 AND a.field2 = :field20", generator2.generate());
206         Assert.assertEquals(BAD_QUERY_PARAMTERS, 2, generator2.getParameterKeys().size());
207         final Iterator<String> i2 = generator2.getParameterKeys().iterator();
208         Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_STR1, generator2.getParameterValue(i2.next()));
209         Assert.assertEquals(BAD_QUERY_PARAMTERS, DUMMY_VALUE_INT1, generator2.getParameterValue(i2.next()));
210     }

224     @Test
225     public void testAuditorQuery() {
226         final Date timestamp = new Date();
227         final String nodeId = "localhost";
228         final QueryCriteria queryCriteria = QueryCriteria.create().add(Criteria.eq("nodeId", nodeId)).add(Criteria.leq("timeStamp", timestamp.getTime())).add(Criteria.orderAsc("sequenceNumber"));
229         final QueryGenerator generator = QueryGenerator.generator(AuditRecordData.class, queryCriteria, "a");
230         final String query = generator.generate();
231         Assert.assertEquals("Wrong query generated", " WHERE a.nodeId = :nodeId0 AND a.timeStamp <= :timeStamp0 ORDER BY a.sequenceNumber ASC", query);
232         // Make sure we get the same query if we generate again using the same criteria
233         final QueryGenerator generator1 = QueryGenerator.generator(AuditRecordData.class, queryCriteria, "a");
234         final String query1 = generator1.generate();
235         Assert.assertEquals("Wrong query generated", " WHERE a.nodeId = :nodeId0 AND a.timeStamp <= :timeStamp0 ORDER BY a.sequenceNumber ASC", query1);
236         
237         final QueryCriteria queryCriteria2 = QueryCriteria.create().add(Criteria.leq("timeStamp", timestamp.getTime())).add(Criteria.orderAsc("sequenceNumber"));
238         final QueryGenerator generator2 = QueryGenerator.generator(AuditRecordData.class, queryCriteria2, "a");
239         final String query2 = generator2.generate();
240         Assert.assertEquals("Wrong query generated", " WHERE a.timeStamp <= :timeStamp0 ORDER BY a.sequenceNumber ASC", query2);        
241         // Make sure we get the same query if we generate again using the same criteria
242         final QueryGenerator generator3 = QueryGenerator.generator(AuditRecordData.class, queryCriteria2, "a");
243         final String query3 = generator3.generate();
244         Assert.assertEquals("Wrong query generated", " WHERE a.timeStamp <= :timeStamp0 ORDER BY a.sequenceNumber ASC", query3);        
245     }
