163     private void doTestX509CABasicOperations(String algName) throws Exception {
164 	    final CryptoToken cryptoToken = getNewCryptoToken();
165         final X509CA x509ca = createTestCA(cryptoToken, CADN, algName, null, null);
166         X509Certificate cacert = (X509Certificate) x509ca.getCACertificate();
167         CertificateProfile cp = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
168 
169         // Start by creating a PKCS7
170         byte[] p7 = x509ca.createPKCS7(cryptoToken, cacert, true);
171         assertNotNull(p7);
172         CMSSignedData s = new CMSSignedData(p7);
173         Store<X509CertificateHolder> certstore = s.getCertificates();
174         Collection<X509CertificateHolder> certs = certstore.getMatches(null);
175         assertEquals(2, certs.size());
176         p7 = x509ca.createPKCS7(cryptoToken, cacert, false);
177         assertNotNull(p7);
178         s = new CMSSignedData(p7);
179         certstore = s.getCertificates();
180         certs = certstore.getMatches(null);
181         assertEquals(1, certs.size());
182 
183 		// Create a certificate request (will be pkcs10)
184         byte[] req = x509ca.createRequest(cryptoToken, null, algName, cacert, CATokenConstants.CAKEYPURPOSE_CERTSIGN, cp, cceConfig);
185         PKCS10CertificationRequest p10 = new PKCS10CertificationRequest(req);
186         assertNotNull(p10);
187         String dn = p10.getSubject().toString();
188         assertEquals(CADN, dn);
189 
190         // Make a request with some pkcs10 attributes as well
191 		Collection<ASN1Encodable> attributes = new ArrayList<>();
192 		// Add a subject alternative name
193 		ASN1EncodableVector altnameattr = new ASN1EncodableVector();
194 		altnameattr.add(PKCSObjectIdentifiers.pkcs_9_at_extensionRequest);
195 		GeneralNames san = CertTools.getGeneralNamesFromAltName("dNSName=foobar.bar.com");
196 		ExtensionsGenerator extgen = new ExtensionsGenerator();
197 		extgen.addExtension(Extension.subjectAlternativeName, false, san);
198 		Extensions exts = extgen.generate();
199 		altnameattr.add(new DERSet(exts));
200         // Add a challenge password as well
201         ASN1EncodableVector pwdattr = new ASN1EncodableVector();
202         pwdattr.add(PKCSObjectIdentifiers.pkcs_9_at_challengePassword);
203         ASN1EncodableVector pwdvalues = new ASN1EncodableVector();
204         pwdvalues.add(new DERUTF8String("foobar123"));
205         pwdattr.add(new DERSet(pwdvalues));
206         attributes.add(new DERSequence(altnameattr));
207         attributes.add(new DERSequence(pwdattr));
208         // create the p10
209         req = x509ca.createRequest(cryptoToken, attributes, algName, cacert, CATokenConstants.CAKEYPURPOSE_CERTSIGN, cp, cceConfig);
210         p10 = new PKCS10CertificationRequest(req);
211         assertNotNull(p10);
212         dn = p10.getSubject().toString();
213         assertEquals(CADN, dn);
214         Attribute[] attrs = p10.getAttributes();
215         assertEquals(2, attrs.length);
216         PKCS10RequestMessage p10msg = new PKCS10RequestMessage(new JcaPKCS10CertificationRequest(p10));
217         assertEquals("foobar123", p10msg.getPassword());
218         assertEquals("dNSName=foobar.bar.com", p10msg.getRequestAltNames());
219 
220         try {
221             x509ca.createAuthCertSignRequest(cryptoToken, p10.getEncoded());
222         } catch (UnsupportedOperationException e) {
223             // Expected for a X509 CA
224         }
225 
226         // Generate a client certificate and check that it was generated correctly
227         EndEntityInformation user = new EndEntityInformation("username", "CN=User", 666, "rfc822Name=user@user.com,dnsName=foo.bar.com,directoryName=CN=Tomas\\,O=PrimeKey\\,C=SE", "user@user.com", new EndEntityType(EndEntityTypes.ENDUSER), 0, 0, EndEntityConstants.TOKEN_USERGEN, 0, null);
228         KeyPair keypair = genTestKeyPair(algName);
229         cp.addCertificatePolicy(new CertificatePolicy("1.1.1.2", null, null));
230         cp.setUseCertificatePolicies(true);
231         Certificate usercert = x509ca.generateCertificate(cryptoToken, user, keypair.getPublic(), 0, null, "10d", cp, "00000", cceConfig);
232         assertNotNull(usercert);
233         assertEquals("CN=User", CertTools.getSubjectDN(usercert));
234         assertEquals(CADN, CertTools.getIssuerDN(usercert));
235         assertEquals(getTestKeyPairAlgName(algName).toUpperCase(), AlgorithmTools.getCertSignatureAlgorithmNameAsString(usercert).toUpperCase());
236         assertEquals(new String(CertTools.getSubjectKeyId(cacert)), new String(CertTools.getAuthorityKeyId(usercert)));
237         assertEquals("user@user.com", CertTools.getEMailAddress(usercert));
238         // directoryName is turned around, but it's just for string reasons in cert objects because it is gotten (internally in BC) getRFC2253Name().
239         assertEquals("rfc822name=user@user.com, dNSName=foo.bar.com, directoryName=c=SE\\,o=PrimeKey\\,cn=Tomas", CertTools.getSubjectAlternativeName(usercert));
240         assertNull(CertTools.getUPNAltName(usercert));
241         assertFalse(CertTools.isSelfSigned(usercert));
242         usercert.verify(cryptoToken.getPublicKey(x509ca.getCAToken().getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN)));
243         usercert.verify(x509ca.getCACertificate().getPublicKey());
244         assertTrue(CertTools.isCA(x509ca.getCACertificate()));
245         assertFalse(CertTools.isCA(usercert));
246         assertEquals("1.1.1.2", CertTools.getCertificatePolicyId(usercert, 0));
247         X509Certificate cert = (X509Certificate)usercert;
248         boolean[] ku = cert.getKeyUsage();
249         assertTrue(ku[0]);
250         assertTrue(ku[1]);
251         assertTrue(ku[2]);
252         assertFalse(ku[3]);
253         assertFalse(ku[4]);
254         assertFalse(ku[5]);
255         assertFalse(ku[6]);
256         assertFalse(ku[7]);
257         int bcku = CertTools.sunKeyUsageToBC(ku);
258         assertEquals(X509KeyUsage.digitalSignature|X509KeyUsage.nonRepudiation|X509KeyUsage.keyEncipherment, bcku);
259 
260         // Create a CRL
261         Collection<RevokedCertInfo> revcerts = new ArrayList<>();
262         X509CRLHolder crl = x509ca.generateCRL(cryptoToken, revcerts, 1);
263         assertNotNull(crl);
264         X509CRL xcrl = CertTools.getCRLfromByteArray(crl.getEncoded());
265         assertEquals(CADN, CertTools.getIssuerDN(xcrl));
266         Set<?> set = xcrl.getRevokedCertificates();
267         assertNull(set);
268         BigInteger num = CrlExtensions.getCrlNumber(xcrl);
269         assertEquals(1, num.intValue());
270         BigInteger deltanum = CrlExtensions.getDeltaCRLIndicator(xcrl);
271         assertEquals(-1, deltanum.intValue());
272         // Revoke some cert
273         Date revDate = new Date();
274         revcerts.add(new RevokedCertInfo(CertTools.getFingerprintAsString(usercert).getBytes(), CertTools.getSerialNumber(usercert).toByteArray(), revDate.getTime(), RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD, CertTools.getNotAfter(usercert).getTime()));
275         crl = x509ca.generateCRL(cryptoToken, revcerts, 2);
276         assertNotNull(crl);
277         xcrl = CertTools.getCRLfromByteArray(crl.getEncoded());
278         set = xcrl.getRevokedCertificates();
279         assertEquals(1, set.size());
280         num = CrlExtensions.getCrlNumber(xcrl);
281         assertEquals(2, num.intValue());
282         X509CRLEntry entry = (X509CRLEntry)set.iterator().next();
283         assertEquals(CertTools.getSerialNumber(usercert).toString(), entry.getSerialNumber().toString());
284         assertEquals(revDate.toString(), entry.getRevocationDate().toString());
285         // Getting the revocation reason is a pita...
286         byte[] extval = entry.getExtensionValue(Extension.reasonCode.getId());
287         ASN1InputStream aIn = new ASN1InputStream(new ByteArrayInputStream(extval));
288         ASN1OctetString octs = (ASN1OctetString) aIn.readObject();
289         aIn = new ASN1InputStream(new ByteArrayInputStream(octs.getOctets()));
290         ASN1Primitive obj = aIn.readObject();
291         CRLReason reason = CRLReason.getInstance(obj);
292         assertEquals("CRLReason: certificateHold", reason.toString());
293         //DEROctetString ostr = (DEROctetString)obj;
294 
295         // Create a delta CRL
296         revcerts = new ArrayList<RevokedCertInfo>();
297         crl = x509ca.generateDeltaCRL(cryptoToken, revcerts, 3, 2);
298         assertNotNull(crl);
299         xcrl = CertTools.getCRLfromByteArray(crl.getEncoded());
300         assertEquals(CADN, CertTools.getIssuerDN(xcrl));
301         set = xcrl.getRevokedCertificates();
302         assertNull(set);
303         num = CrlExtensions.getCrlNumber(xcrl);
304         assertEquals(3, num.intValue());
305         deltanum = CrlExtensions.getDeltaCRLIndicator(xcrl);
306         assertEquals(2, deltanum.intValue());
307         revcerts.add(new RevokedCertInfo(CertTools.getFingerprintAsString(usercert).getBytes(), CertTools.getSerialNumber(usercert).toByteArray(), revDate.getTime(), RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD, CertTools.getNotAfter(usercert).getTime()));
308         crl = x509ca.generateDeltaCRL(cryptoToken, revcerts, 4, 3);
309         assertNotNull(crl);
310         xcrl = CertTools.getCRLfromByteArray(crl.getEncoded());
311         deltanum = CrlExtensions.getDeltaCRLIndicator(xcrl);
312         assertEquals(3, deltanum.intValue());
313         set = xcrl.getRevokedCertificates();
314         assertEquals(1, set.size());
315         entry = (X509CRLEntry)set.iterator().next();
316         assertEquals(CertTools.getSerialNumber(usercert).toString(), entry.getSerialNumber().toString());
317         assertEquals(revDate.toString(), entry.getRevocationDate().toString());
318         // Getting the revocation reason is a pita...
319         extval = entry.getExtensionValue(Extension.reasonCode.getId());
320         aIn = new ASN1InputStream(new ByteArrayInputStream(extval));
321         octs = (ASN1OctetString) aIn.readObject();
322         aIn = new ASN1InputStream(new ByteArrayInputStream(octs.getOctets()));
323         obj = aIn.readObject();
324         reason = CRLReason.getInstance(obj);
325         assertEquals("CRLReason: certificateHold", reason.toString());
326 	}
