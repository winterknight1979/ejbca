38  /**
39   * Removes an admin
40   * @version $Id: RemoveAdminCommand.java 29199 2018-06-11 11:13:01Z jeklund $
41   */
42  public class RemoveAdminCommand extends BaseRolesCommand {
43  
44      private static final Logger log = Logger.getLogger(RemoveAdminCommand.class);
45  
46      private static final String ROLE_NAME_KEY = "--role";
47      private static final String CA_NAME_KEY = "--caname";
48      private static final String MATCH_WITH_KEY = "--with";
49      private static final String MATCH_TYPE_KEY = "--type";
50      private static final String MATCH_VALUE_KEY = "--value";
51      private static final String ROLE_NAMESPACE_KEY = "--namespace";
52  
53      {
54          registerParameter(new Parameter(ROLE_NAME_KEY, "Role Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
55                  "Role to list rules of."));
56          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
57                  "Name of the issuing CA"));
58          registerParameter(new Parameter(MATCH_WITH_KEY, "Value", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
59                  "The MatchWith Value"));
60          registerParameter(new Parameter(MATCH_TYPE_KEY, "Type", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
61                  "(Deprected) Match operator type. Kept to prevent legacy scripts from breaking. Currently implied by " + MATCH_WITH_KEY +" switch."));
62          registerParameter(new Parameter(MATCH_VALUE_KEY, "Value", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
63                  "The value to match against."));
64          registerParameter(new Parameter(ROLE_NAMESPACE_KEY, "Role Namespace", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
65                  "The namespace the role belongs to."));
66      }
67  
68      @Override
69      public String getMainCommand() {
70          return "removeadmin";
71      }
72  
73      @Override
74      public CommandResult execute(ParameterContainer parameters) {
75          final String roleName = parameters.get(ROLE_NAME_KEY);
76          final String namespace = parameters.get(ROLE_NAMESPACE_KEY);
77          final Role role;
78          try {
79              role = EjbRemoteHelper.INSTANCE.getRemoteSession(RoleSessionRemote.class).getRole(getAuthenticationToken(), namespace, roleName);
80          } catch (AuthorizationDeniedException e) {
81              getLogger().error("Not authorized to role " + super.getFullRoleName(namespace, roleName) + ".");
82              return CommandResult.FUNCTIONAL_FAILURE;
83          }
84          if (role == null) {
85              getLogger().error("No such role " + super.getFullRoleName(namespace, roleName) + ".");
86              return CommandResult.FUNCTIONAL_FAILURE;
87          }
88          final String matchWithKeyParam = parameters.get(MATCH_WITH_KEY);
89          if (StringUtils.isEmpty(matchWithKeyParam)) {
90              getLogger().error("No such thing to match with as '" + parameters.get(MATCH_WITH_KEY) + "'.");
91              return CommandResult.FUNCTIONAL_FAILURE;
92          }
93          final String tokenType;
94          final String matchWithKey;
95          final String[] matchWithKeySplit = matchWithKeyParam.split(":");
96          if (matchWithKeySplit.length == 1) {
97              tokenType = X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE;
98              matchWithKey = matchWithKeySplit[0];
99              getLogger().info("Match TokenType is assumed to be '" + tokenType + "'.");
100         } else {
101             tokenType = matchWithKeySplit[0];
102             matchWithKey = matchWithKeySplit[1];
103             final AuthenticationTokenMetaData authenticationTokenMetaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(tokenType);
104             if (authenticationTokenMetaData == null || !authenticationTokenMetaData.isUserConfigurable()) {
105                 getLogger().error("TokenType '" + tokenType + "' is not configurable.");
106                 return CommandResult.FUNCTIONAL_FAILURE;
107             }
108         }
109         final int tokenIssuerId;
110         final String caName;
111         if (X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE.equals(tokenType)) {
112             final String caNameParam = parameters.get(CA_NAME_KEY);
113             final CAInfo caInfo;
114             try {
115                 caInfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caNameParam);
116             }  catch (AuthorizationDeniedException e) {
117                 getLogger().error("CLI user not authorized to CA");
118                 return CommandResult.AUTHORIZATION_FAILURE;
119             }
120             if (caInfo == null) {
121                 getLogger().error("No such CA '" + caNameParam + "'.");
122                 return CommandResult.FUNCTIONAL_FAILURE;
123             }
124             tokenIssuerId = caInfo.getCAId();
125             caName = caInfo.getName();
126         } else {
127             // To be backwards compatible with existing scripts in the wild, we will still have to require parameter but we ignore the value if is unused
128             //getLogger().info("Ignoring mandatory '" + CA_NAME_KEY + "' parameter value for the selected Token Type '" + tokenType + "'.");
129             tokenIssuerId = RoleMember.NO_ISSUER;
130             caName = "-";
131         }
132         final AccessMatchValue accessMatchValue = AccessMatchValueReverseLookupRegistry.INSTANCE.lookupMatchValueFromTokenTypeAndName(tokenType, matchWithKey);
133         if (accessMatchValue == null) {
134             getLogger().error("No such thing to match with as '" + matchWithKey + "'.");
135             return CommandResult.FUNCTIONAL_FAILURE;
136         }
137         final String accessMatchTypeParam = parameters.get(MATCH_TYPE_KEY);
138         final AccessMatchType accessMatchType;
139         if (accessMatchTypeParam==null) {
140             if (accessMatchValue.getAvailableAccessMatchTypes().isEmpty()) {
141                 accessMatchType = AccessMatchType.TYPE_UNUSED;
142             } else {
143                 accessMatchType = accessMatchValue.getAvailableAccessMatchTypes().get(0);
144             }
145             getLogger().info("Using '"+accessMatchType+"' implied by '" + accessMatchValue + "'.");
146         } else {
147             accessMatchType = AccessMatchType.matchFromName(parameters.get(MATCH_TYPE_KEY));
148             if (accessMatchType == null) {
149                 getLogger().error("No such type to match with as '" + parameters.get(MATCH_TYPE_KEY) + "'.");
150                 return CommandResult.FUNCTIONAL_FAILURE;
151             }
152         }
153         final String tokenMatchValue = parameters.get(MATCH_VALUE_KEY);
154         final RoleMemberSessionRemote roleMemberSession = EjbRemoteHelper.INSTANCE.getRemoteSession(RoleMemberSessionRemote.class);
155         try {
156             boolean foundMatch = false;
157             for (final RoleMember roleMember : roleMemberSession.getRoleMembersByRoleId(getAuthenticationToken(), role.getRoleId())) {
158                 if (tokenType.equals(roleMember.getTokenType()) &&
159                         tokenIssuerId == roleMember.getTokenIssuerId() &&
160                         accessMatchValue.getNumericValue()==roleMember.getTokenMatchKey() &&
161                         accessMatchType.getNumericValue()==roleMember.getTokenMatchOperator() &&
162                         tokenMatchValue.equals(roleMember.getTokenMatchValue())) {
163                     roleMemberSession.remove(getAuthenticationToken(), roleMember.getId());
164                     foundMatch = true;
165                     getLogger().info("Removed role member: " + "'" + caName + "' " + accessMatchValue + " " + accessMatchType + " '" +
166                             tokenMatchValue + "' from role " + super.getFullRoleName(namespace, roleName));
167                 }
168             }
169             if (!foundMatch) {
170                 getLogger().info("Could not find any matching admin in role " + super.getFullRoleName(namespace, roleName) + ".");
171             }
172         } catch (AuthorizationDeniedException e) {
173             getLogger().info("Not authorized to members of role " + super.getFullRoleName(namespace, roleName) + ".");
174             return CommandResult.AUTHORIZATION_FAILURE;
175         }
176         return CommandResult.SUCCESS;
177     }

181         return "Removes an admin";
182     }
