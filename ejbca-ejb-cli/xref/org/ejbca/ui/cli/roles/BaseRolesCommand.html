14  package org.ejbca.ui.cli.roles;
15  
16  import java.util.HashMap;
17  import java.util.HashSet;
18  import java.util.Map;
19  import java.util.Map.Entry;
20  import java.util.Set;
21  
22  import org.apache.commons.lang.StringUtils;
23  import org.apache.log4j.Logger;
24  import org.cesecore.authorization.user.matchvalues.X500PrincipalAccessMatchValue;
25  import org.cesecore.util.EjbRemoteHelper;
26  import org.ejbca.core.ejb.authentication.cli.CliUserAccessMatchValue;
27  import org.ejbca.core.ejb.authorization.AuthorizationSystemSessionRemote;
28  import org.ejbca.ui.cli.infrastructure.command.EjbcaCliUserCommandBase;
29  
30  /**
31   * Base for Roles commands, contains common functions for Roles operations
32   * @version $Id: BaseRolesCommand.java 29175 2018-06-08 13:38:23Z jeklund $
33   */
34  public abstract class BaseRolesCommand extends EjbcaCliUserCommandBase {
35  
36      private static final Logger log = Logger.getLogger(BaseRolesCommand.class);
37  
38      private Map<String,String> resourceNameToResourceMap = null;
39      private Map<String,String> resourceToResourceNameMap = null;
40  
41      private static Set<String[]> commandAliases = new HashSet<String[]>();
42      static {
43          commandAliases.add(new String[] { "admins" });
44          try {
45              Class.forName(X500PrincipalAccessMatchValue.class.getName());
46              Class.forName(CliUserAccessMatchValue.class.getName());
47          } catch (ClassNotFoundException e) {
48              log.error("Failure during match value initialization", e);
49          }
50      }
51  
52      @Override
53      public String[] getCommandPath() {
54          return new String[] { "roles" };
55      }
56  
57      @Override
58      public Set<String[]> getCommandPathAliases() {
59          return commandAliases;
60      }
61  
62      @Override
63      protected abstract Logger getLogger();
64  
65      /** @return a Map&lt;resourceName,resource&gt; for authorized resources (cached in this remote JVM) */
66      public Map<String, String> getResourceNameToResourceMap() {
67          if (resourceNameToResourceMap==null) {
68              final Map<String,String> authorizedResourcesMap = EjbRemoteHelper.INSTANCE.getRemoteSession(AuthorizationSystemSessionRemote.class).
69                      getAllResources(getAuthenticationToken(), false);
70              resourceNameToResourceMap = new HashMap<>();
71              for (final Entry<String,String> entry: authorizedResourcesMap.entrySet()) {
72                  resourceNameToResourceMap.put(entry.getValue(), entry.getKey());
73              }
74          }
75          return resourceNameToResourceMap;
76      }
77  
78      /** @return a Map&lt;resource,resourceName&gt; for authorized resources (cached in this remote JVM) */
79      public Map<String, String> getResourceToResourceNameMap() {
80          if (resourceToResourceNameMap==null) {
81              resourceToResourceNameMap = EjbRemoteHelper.INSTANCE.getRemoteSession(AuthorizationSystemSessionRemote.class).
82                      getAllResources(getAuthenticationToken(), false);
83          }
84          return resourceToResourceNameMap;
85      }
86      
87      /** @param namespace Namespace
88       * @param roleName Role
89       * @return the full role name with namespace prefixed in brackets. */
90      protected String getFullRoleName(final String namespace, final String roleName) {
91          return (StringUtils.isEmpty(namespace) ? "" : "["+namespace+"] ") + "'" + roleName + "'";
92      }
93  }
