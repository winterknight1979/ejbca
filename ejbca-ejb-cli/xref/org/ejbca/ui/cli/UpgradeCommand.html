14  package org.ejbca.ui.cli;
15  
16  import org.apache.log4j.Logger;
17  import org.cesecore.util.EjbRemoteHelper;
18  import org.ejbca.core.ejb.upgrade.UpgradeSessionRemote;
19  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
20  import org.ejbca.ui.cli.infrastructure.command.EjbcaCommandBase;
21  import org.ejbca.ui.cli.infrastructure.parameter.Parameter;
22  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
23  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
24  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
25  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;
26  
27  /**
28   * Implements call to the upgrade function
29   * 
30   * @version $Id: UpgradeCommand.java 19902 2014-09-30 14:32:24Z anatom $
31   */
32  public class UpgradeCommand extends EjbcaCommandBase {
33  
34      private static final Logger log = Logger.getLogger(UpgradeCommand.class);
35  
36      private static final String DATABASE_KEY = "-d";
37      private static final String FROM_VERSION_KEY = "-v";
38      private static final String IS_POST_UPGRADE = "--post";
39  
40      {
41          registerParameter(new Parameter(DATABASE_KEY, "Database type", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
42                  "The database type."));
43          registerParameter(new Parameter(FROM_VERSION_KEY, "From version", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
44                  "From which version of EJBCA"));
45          registerParameter(new Parameter(IS_POST_UPGRADE, "Add this flag if running post upgrade", MandatoryMode.OPTIONAL, StandaloneMode.FORBID,
46                  ParameterMode.FLAG, "Set this flag when performing post upgrade."));
47      }
48  
49      @Override
50      public String getMainCommand() {
51          return "upgrade";
52      }
53  
54      @Override
55      public CommandResult execute(ParameterContainer parameters) {
56          final String database = parameters.get(DATABASE_KEY);
57          final String upgradeFromVersion = parameters.get(FROM_VERSION_KEY);
58          final boolean isPost = parameters.get(IS_POST_UPGRADE) != null;
59          log.debug(getMainCommand() + " ejbcaDB='" + database + "' ejbcaUpgradeFromVersion='" + upgradeFromVersion + "' isPost='" + isPost + "'");
60          // Upgrade the database
61          final boolean ret = EjbRemoteHelper.INSTANCE.getRemoteSession(UpgradeSessionRemote.class).upgrade(database, upgradeFromVersion, isPost);
62          if (ret) {
63              log.info("Upgrade completed.");
64          } else {
65              log.error("Upgrade not performed, see server log for details.");
66              return CommandResult.FUNCTIONAL_FAILURE;
67          }
68          return CommandResult.SUCCESS;
69  
70      }
71  
72      @Override
73      public String getCommandDescription() {
74          return "Upgrade command. Use 'ant upgrade' instead of running this directly.";
75      }
76  
77      @Override
78      public String getFullHelpText() {
79          return getCommandDescription();
80      }
81      
82      @Override
83      protected Logger getLogger() {
84          return log;
85      }
86  }
