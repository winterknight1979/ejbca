13  package org.ejbca.ui.cli.cryptotoken;
14  
15  import org.apache.log4j.Logger;
16  import org.cesecore.authorization.AuthorizationDeniedException;
17  import org.cesecore.keys.token.CryptoTokenManagementSessionRemote;
18  import org.cesecore.keys.token.CryptoTokenOfflineException;
19  import org.cesecore.util.EjbRemoteHelper;
20  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
21  import org.ejbca.ui.cli.infrastructure.parameter.Parameter;
22  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
23  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
24  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
25  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;
26  
27  /**
28   * CryptoToken EJB CLI command. 
29   * 
30   * @version $Id: CryptoTokenUpdatePinCommand.java 19902 2014-09-30 14:32:24Z anatom $
31   */
32  public class CryptoTokenUpdatePinCommand extends BaseCryptoTokenCommand {
33  
34      private static final Logger log = Logger.getLogger(CryptoTokenUpdatePinCommand.class);
35  
36      private static final String SWITCH_UPDATE_ONLY_KEY = "--update";
37      private static final String SWITCH_REMOVE_AUTO_KEY = "--remove";
38      private static final String OLD_PIN_KEY = "--oldpin";
39      private static final String NEW_PIN_KEY = "--newpin";
40  
41      {
42          registerParameter(new Parameter(OLD_PIN_KEY, "Pin", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
43                  "The old pin. Set as \"null\" to prompt"));
44          registerParameter(new Parameter(NEW_PIN_KEY, "Pin", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
45                  "The new pin. Set as \"null\" to prompt"));
46          registerParameter(Parameter.createFlag(SWITCH_UPDATE_ONLY_KEY,
47                  "Set the auto-activation pin and make the token auto-activated if it was so previously."));
48          registerParameter(Parameter.createFlag(SWITCH_REMOVE_AUTO_KEY,
49                  "Will remove any auto-activation pin if present (new pin is not required when this is used)."));
50      }
51  
52      @Override
53      public String getMainCommand() {
54          return "setpin";
55      }
56  
57      @Override
58      public CommandResult executeCommand(Integer cryptoTokenId, ParameterContainer parameters) throws AuthorizationDeniedException, CryptoTokenOfflineException {
59          final boolean updateOnly = parameters.containsKey(SWITCH_UPDATE_ONLY_KEY);
60          final boolean removeAuto = parameters.containsKey(SWITCH_REMOVE_AUTO_KEY);
61  
62          final char[] currentAuthenticationCode = getAuthenticationCode(parameters.get(OLD_PIN_KEY));
63          final char[] newAuthenticationCode = removeAuto ? null : getAuthenticationCode(parameters.get(NEW_PIN_KEY));
64          try {
65              final CryptoTokenManagementSessionRemote cryptoTokenManagementSession = EjbRemoteHelper.INSTANCE
66                      .getRemoteSession(CryptoTokenManagementSessionRemote.class);
67              boolean result = cryptoTokenManagementSession.updatePin(getAdmin(), cryptoTokenId.intValue(), currentAuthenticationCode,
68                      newAuthenticationCode, updateOnly);
69              if (result) {
70                  getLogger().info("Auto-activation is now in use for this CryptoToken.");
71              } else {
72                  getLogger().info("Auto-activation is now not in use for this CryptoToken.");
73              }
74              final boolean isActive = cryptoTokenManagementSession.isCryptoTokenStatusActive(getAdmin(), cryptoTokenId.intValue());
75              getLogger().info("CryptoToken is " + (isActive ? "active" : "deactivated") + ".");
76              return CommandResult.SUCCESS;
77          } catch (AuthorizationDeniedException e) {
78              getLogger().info(e.getMessage());
79              return CommandResult.AUTHORIZATION_FAILURE;
80          } catch (Exception e) {
81              getLogger().info("CryptoToken activation failed: " + e.getMessage());
82          }
83          return CommandResult.FUNCTIONAL_FAILURE;
84      }
85  
86      @Override
87      public String getCommandDescription() {
88          return "Modifies the current keystore and/or auto-activation pin.";
89      }
90  
91      @Override
92      public String getFullHelpText() {
93          return getCommandDescription()
94                  + " For soft CryptoTokens the underlying keystore's pin will be modified and this requires the current activation PIN."
95                  + " For PKCS#11 CryptoTokens this will only modify the auto-activation pin and requires the current (auto-activation or) activation PIN.";
96      }
97  
98      @Override
99      protected Logger getLogger() {
100         return log;
101     }
102 
103 }
