13  package org.ejbca.ui.cli.cryptotoken;
14  
15  import java.util.Collections;
16  import java.util.Comparator;
17  import java.util.List;
18  
19  import org.apache.log4j.Logger;
20  import org.cesecore.keys.token.CryptoTokenInfo;
21  import org.cesecore.keys.token.CryptoTokenManagementSessionRemote;
22  import org.cesecore.keys.token.PKCS11CryptoToken;
23  import org.cesecore.keys.token.SoftCryptoToken;
24  import org.cesecore.util.EjbRemoteHelper;
25  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
26  import org.ejbca.ui.cli.infrastructure.command.EjbcaCliUserCommandBase;
27  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
28  
29  /**
30   * CryptoToken EJB CLI command.
31   * 
32   * @version $Id: CryptoTokenListCommand.java 19902 2014-09-30 14:32:24Z anatom $
33   */
34  public class CryptoTokenListCommand extends EjbcaCliUserCommandBase {
35  
36      private static final Logger log = Logger.getLogger(CryptoTokenListCommand.class);
37      
38      @Override
39      public String[] getCommandPath() {
40          return new String[] { "cryptotoken" };
41      }
42      
43      @Override
44      public String getMainCommand() {
45          return "list";
46      }
47  
48      @Override
49      protected CommandResult execute(ParameterContainer parameters) {
50          final CryptoTokenManagementSessionRemote cryptoTokenManagementSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CryptoTokenManagementSessionRemote.class);
51          final List<CryptoTokenInfo> cryptoTokenInfos = cryptoTokenManagementSession.getCryptoTokenInfos(getAuthenticationToken());
52          // Sort by name
53          Collections.sort(cryptoTokenInfos, new Comparator<CryptoTokenInfo>(){
54              @Override
55              public int compare(CryptoTokenInfo c1, CryptoTokenInfo c2) {
56                  return c1.getName().compareTo(c2.getName());
57              }
58          });
59          getLogger().info(" \"NAME\" (ID)\t TYPE, STATUS, AUTO-ACTIVATION, TYPE-PROPERTIES...");
60          for (final CryptoTokenInfo cryptoTokenInfo : cryptoTokenInfos) {
61              final StringBuilder sb = new StringBuilder();
62              sb.append(' ').append('\"').append(cryptoTokenInfo.getName()).append('\"');
63              sb.append(" (").append(cryptoTokenInfo.getCryptoTokenId()).append(')');
64              sb.append('\t').append(cryptoTokenInfo.getType());
65              sb.append(", ").append(cryptoTokenInfo.isActive()?"active":"offline");
66              sb.append(", ").append(cryptoTokenInfo.isAutoActivation()?"auto":"manual");
67              if (SoftCryptoToken.class.getSimpleName().equals(cryptoTokenInfo.getType())) {
68                  sb.append(", ").append(cryptoTokenInfo.isAllowExportPrivateKey()?"exportable":"non-exportable");
69              }
70              if (PKCS11CryptoToken.class.getSimpleName().equals(cryptoTokenInfo.getType())) {
71                  sb.append(", library=").append(cryptoTokenInfo.getP11Library());
72                  sb.append(", Slot Label=").append(cryptoTokenInfo.getP11Slot());
73                  sb.append(", Slot Label Type=").append(cryptoTokenInfo.getP11SlotLabelTypeDescription());
74                  sb.append(", attributes=").append(cryptoTokenInfo.getP11AttributeFile());
75              }
76              getLogger().info(sb);
77          }
78          return CommandResult.SUCCESS;
79      }
80      
81      @Override
82      public String getCommandDescription() {
83          return "List all available CryptoTokens";
84      }
85  
86      @Override
87      public String getFullHelpText() {
88          return getCommandDescription();
89      }
90  
91      @Override
92      protected Logger getLogger() {
93          return log;
94      }
95  
96  }
