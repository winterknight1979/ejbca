33  /**

36   */
37  public class UploadFileCommand extends BaseCmpConfigCommand {
38  
39      private static final String ALIAS_KEY = "--alias";
40      private static final String FILE_KEY = "--file";
41  
42      private static final Logger log = Logger.getLogger(UploadFileCommand.class);
43  
44      {
45          registerParameter(new Parameter(ALIAS_KEY, "Alias", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
46                  "The alias to read to."));
47          registerParameter(new Parameter(FILE_KEY, "File", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
48                  "The file to upload."));
49      }
50  
51      @Override
52      public String getMainCommand() {
53          return "uploadfile";
54      }
55  
56      @Override
57      public CommandResult execute(ParameterContainer parameters) {
58          String filename = parameters.get(FILE_KEY);
59          String alias = parameters.get(ALIAS_KEY);
60  
61          CompositeConfiguration config = null;
62          File f = null;
63          try {
64              f = new File(filename);
65              final PropertiesConfiguration pc = new PropertiesConfiguration(f);
66              pc.setReloadingStrategy(new FileChangedReloadingStrategy());
67              config = new CompositeConfiguration();
68              config.addConfiguration(pc);
69              log.info("Reading CMP configuration from file: " + f.getAbsolutePath());
70          } catch (ConfigurationException e) {
71              log.error("Failed to load configuration from file " + f.getAbsolutePath());
72              return CommandResult.FUNCTIONAL_FAILURE;
73          }
74          readConfigurations(config, alias);
75          return CommandResult.SUCCESS;
76  
77      }
78  
79      @Override
80      public String getCommandDescription() {
81          return "Reads CMP configuration from a file.";
82      }
83  
84      @Override
85      public String getFullHelpText() {
86          StringBuilder sb = new StringBuilder();
87          sb.append(getCommandDescription() + "\n");
88          sb.append("\n");
89          sb.append("Each line that has the format 'ALIAS.key=VALUE' will be read" + "\n");
90          sb.append(
91                  "Only one alias will be read. If one file contains configurations of several aliases, you have to repeat the command with a "
92                          + "different alias each time to have all configurations of all aliases read"+ "\n");
93          sb.append(
94                  "Note that the alias 'cmp' is the default alias used when sending the CMP request through the URL 'http://HOST:PORT/ejbca/publicweb/cmp'. "
95                          + "Note also that the alias 'cmp' does not exist per default and should be created before using this URL"+ "\n");
96          sb.append("The following keys (if present) will be read from the file:" + "\n");
97          Set<String> keys = CmpConfiguration.getAllAliasKeys("<ALIAS>");
98          Iterator<String> itr = keys.iterator();
99          while (itr.hasNext()) {
100             sb.append("     " + itr.next() + "\n");
101         }
102         return sb.toString();
103     }
104     
105     
106     
107     private void readConfigurations(CompositeConfiguration config, String alias) {
108 
109         // if the alias does not already exist, create it.
110         getCmpConfiguration().addAlias(alias);
111 
112         // Reading all relevant configurations from file.
113         boolean populated = false;
114         Set<String> keys = CmpConfiguration.getAllAliasKeys(alias);
115         Iterator<String> itr = keys.iterator();
116         while (itr.hasNext()) {
117             String key = itr.next();
118             if (config.containsKey(key)) {
119                 populated = true;
120                 String value = config.getString(key);
121                 getCmpConfiguration().setValue(key, value, alias);
122                 log.info("Setting value: " + key + "=" + value);
123             }
124         }
125 
126         // Save the new configurations.
127         if (populated) {
128             try {
129                 getGlobalConfigurationSession().saveConfiguration(getAuthenticationToken(), getCmpConfiguration());
130                 log.info("\nNew configurations saved successfully.");
131                 log.info("If there are any issues with the configurations, check them in the AdminGUI and click 'Save'");
132                 getGlobalConfigurationSession().flushConfigurationCache(CmpConfiguration.CMP_CONFIGURATION_ID);
133             } catch (AuthorizationDeniedException e) {
134                 log.error("Failed to save configuration from file: " + e.getLocalizedMessage());
135             }
136         } else {
137             getCmpConfiguration().removeAlias(alias);
138             log.info("No relevent CMP configurations found with alias '" + alias + "' in the file.");
139         }
140     }
141     
142     @Override
143     protected Logger getLogger() {
144         return log;
145     }
146 }
