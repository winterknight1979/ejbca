39  public class UpdateCommand extends BaseScepConfigCommand {
40  
41      private static final String ALIAS_KEY = "--alias";
42      private static final String KEY_KEY = "--key"; //Hue hue hue
43      private static final String VALUE_KEY = "--value";
44  
45      private static final Logger log = Logger.getLogger(UpdateCommand.class);
46  
47      {
48          registerParameter(new Parameter(ALIAS_KEY, "Alias", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
49                  "The alias of the configuration to update."));
50          registerParameter(new Parameter(KEY_KEY, "Key", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
51                  "The alias of the configuration to update."));
52          registerParameter(new Parameter(VALUE_KEY, "Value", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
53                  "The alias of the configuration to update."));
54      }
55  
56      @Override
57      public String getMainCommand() {
58          return "updatealias";
59      }
60  
61      @Override
62      public CommandResult execute(ParameterContainer parameters) {
63  
64          String alias = parameters.get(ALIAS_KEY);
65          String key = parameters.get(KEY_KEY);
66          String value = parameters.get(VALUE_KEY);
67          ScepConfiguration scepConfig = getScepConfiguration();
68          List<String> bkeys = ScepConfiguration.SCEP_BOOLEAN_KEYS;
69          
70          if(StringUtils.equals(key, ScepConfiguration.SCEP_RAMODE_OLD) || StringUtils.equals(key, ScepConfiguration.SCEP_OPERATIONMODE)) {
71              key = alias + "." + ScepConfiguration.SCEP_OPERATIONMODE;
72              value = !StringUtils.equalsIgnoreCase(value, "ra") ? "ca" : "ra";
73          } else {
74              if (bkeys.contains(key)) {
75                  value = Boolean.toString(StringUtils.equalsIgnoreCase(value, "true"));
76              }
77              key = alias + "." + key;
78          }
79          
80          log.info("Configuration was: " + key + "=" + scepConfig.getValue(key, alias));
81          scepConfig.setValue(key, value, alias);
82          try {
83              getGlobalConfigurationSession().saveConfiguration(getAuthenticationToken(), scepConfig);
84              log.info("Configuration updated: " + key + "=" + getScepConfiguration().getValue(key, alias));
85              getGlobalConfigurationSession().flushConfigurationCache(ScepConfiguration.SCEP_CONFIGURATION_ID);
86              return CommandResult.SUCCESS;
87          } catch (AuthorizationDeniedException e) {
88              log.info("Failed to update configuration: " + e.getLocalizedMessage());
89              return CommandResult.AUTHORIZATION_FAILURE;
90          }
91  
92      }
93  
94      @Override
95      public String getCommandDescription() {
96          return "Updates one configuration value.";
97      }
98  
99      @Override
100     public String getFullHelpText() {
101         final String divider = " | ";
102         
103         StringBuilder sb = new StringBuilder();
104         sb.append(getCommandDescription() + "\n\n");
105         sb.append("The key could be any of the following:\n");
106         
107         sb.append("    " + ScepConfiguration.SCEP_OPERATIONMODE + " - possible values: ca " + divider + " ra" + "\n");
108         sb.append("    " + ScepConfiguration.SCEP_INCLUDE_CA + " - possible values: true " + divider + " false" + "\n");
109         
110 
111         Map<Integer, String> endentityprofileidtonamemap = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class)
112                 .getEndEntityProfileIdToNameMap();
113         StringBuilder existingEeps = new StringBuilder();
114         for (Integer profileId : EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getAuthorizedEndEntityProfileIds(
115                 getAuthenticationToken(), AccessRulesConstants.CREATE_END_ENTITY)) {
116             existingEeps.append((existingEeps.length() == 0 ? "" : divider) + endentityprofileidtonamemap.get(profileId));
117         }
118         sb.append("    " + ScepConfiguration.SCEP_RA_ENTITYPROFILE + " - possible values: " + existingEeps + "\n");
119         
120         Map<Integer, String> certificateprofileidtonamemap = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
121                 .getCertificateProfileIdToNameMap();
122         StringBuilder existingCps = new StringBuilder();
123         for (Integer profileId : EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getAuthorizedCertificateProfileIds(
124                 getAuthenticationToken(), CertificateConstants.CERTTYPE_ENDENTITY)) {
125             existingCps.append((existingCps.length() == 0 ? "" : divider) + certificateprofileidtonamemap.get(profileId));
126         }
127         sb.append("    " + ScepConfiguration.SCEP_RA_CERTPROFILE + " - possible values: ProfileDefault" + divider + existingCps + "\n");
128         
129         StringBuilder existingCas = new StringBuilder();
130         for (String ca : EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getActiveCANames(getAuthenticationToken())) {
131             existingCas.append((existingCas.length() == 0 ? "" : divider) + ca);
132         }
133         sb.append("    " + ScepConfiguration.SCEP_RA_DEFAULTCA + " - possible values: ProfileDefault" + divider + existingCas + "\n");        
134         sb.append("    " + ScepConfiguration.SCEP_RA_AUTHPWD + " - possible values: none " + divider + " any alphanumeric string" + "\n");
135         sb.append("    " + ScepConfiguration.SCEP_RA_NAME_GENERATION_SCHEME + " - possible values: DN " + divider + " RANDOM " + divider + " USERNAME " + divider + " FIXED" + "\n");
136         sb.append("    " + ScepConfiguration.SCEP_RA_NAME_GENERATION_PARAMETERS + " - possible values: See CMP Configurations in the Admin GUI" + "\n");
137         sb.append("    " + ScepConfiguration.SCEP_RA_NAME_GENERATION_PREFIX + " - possible values: ${RANDOM} " + divider + " any alphanumeric string" + "\n");
138         sb.append("    " + ScepConfiguration.SCEP_RA_NAME_GENERATION_POSTFIX + " - possible values: ${RANDOM} " + divider
139                 + " any alphanumeric string" + "\n");
140         sb.append("    " + ScepConfiguration.SCEP_CLIENT_CERTIFICATE_RENEWAL + " - (Enterprise Edition Only) possible values: true " + divider + " false" + "\n");
141         sb.append("    " + ScepConfiguration.SCEP_CLIENT_CERTIFICATE_RENEWAL_WITH_OLD_KEY + " - (Enterprise Edition Only) possible values: true " + divider + " false" + "\n");
142   
143               
144         return sb.toString();
145 
146     }
