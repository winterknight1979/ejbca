14  package org.ejbca.ui.cli.config.scep;
15  
16  import org.ejbca.config.ScepConfiguration;
17  import org.ejbca.ui.cli.config.ConfigBaseCommand;
18  
19  /**
20   * Shows the current server configuration
21   * 
22   * @version $Id: BaseScepConfigCommand.java 26057 2017-06-22 08:08:34Z anatom $
23   */
24  public abstract class BaseScepConfigCommand extends ConfigBaseCommand {
25       
26      private ScepConfiguration scepConfiguration = null;
27  
28      @Override
29      public String[] getCommandPath() {
30          return new String[] { super.getCommandPath()[0] , "scep" };
31      }
32      
33      protected ScepConfiguration getScepConfiguration() {
34          if (scepConfiguration == null) {
35              scepConfiguration = (ScepConfiguration) getGlobalConfigurationSession().getCachedConfiguration(ScepConfiguration.SCEP_CONFIGURATION_ID);
36          }
37          return scepConfiguration;
38      }
39  
40  }
