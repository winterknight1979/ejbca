13  package org.ejbca.ui.cli.batch;
14  
15  import java.io.File;
16  import java.io.FileInputStream;
17  import java.io.IOException;
18  import java.util.Properties;
19  
20  import org.apache.log4j.Logger;
21  import org.ejbca.core.model.InternalEjbcaResources;
22  
23  /**
24   * Class used to manage the batch tool property file.
25   *
26   *
27   * @version $Id: BatchToolProperties.java 19902 2014-09-30 14:32:24Z anatom $
28   */
29  public class BatchToolProperties {
30  
31  	private static final String PROPERTY_KEYSPEC          = "keys.spec";
32  	private static final String PROPERTY_KEYALG           = "keys.alg";
33  
34  
35  	Properties batchToolProperties = new Properties();
36  	private static final Logger log = Logger.getLogger(BatchToolProperties.class);
37  
38  	private Logger logger;
39  	
40  	public BatchToolProperties(Logger logger){
41  	    this.logger = logger;
42  		load();
43  	}
44  
45  	/**
46  	 * Returns the configured keysize
47  	 * Default is 2048
48  	 * @return spec
49  	 */
50  	public String getKeySpec(){
51  		return batchToolProperties.getProperty(PROPERTY_KEYSPEC,"2048");
52  	}
53  
54  	/**
55  	 * Returns the configured key algorithm
56  	 * Default is RSA, can be ECDSA
57  	 * @return Alg
58  	 */
59  	public String getKeyAlg(){
60  		return batchToolProperties.getProperty(PROPERTY_KEYALG,"RSA");
61  	}
62  
63  
64  	private boolean tryLoadFile(String filename) throws IOException {
65  		File file = new File(filename);
66  		if (file.exists()) {
67  			FileInputStream fis = new FileInputStream(file);
68  			batchToolProperties.load(fis);
69  			logger.info(InternalEjbcaResources.getInstance().getLocalizedMessage("batch.loadingconfig", filename));
70  			return true;
71  		} else {
72  			return false;
73  		}
74  	}
75  
76  	/**
77  	 * Method that tries to read the property file 'batchtool.properties'
78  	 * in the home directory then in the current directory and finally
79  	 * in the conf/batchtool.properties.
80  	 *
81  	 * It will also try the old location in bin/ and print a deprecation
82  	 * warning if it exists there.
83  	 */
84  	private void load(){
85          try {
86          	if (!tryLoadFile(System.getProperty("user.home") + "/batchtool.properties") &&
87          	    !tryLoadFile("batchtool.properties") &&
88          	    !tryLoadFile("conf/batchtool.properties")) {
89          	    // Not found
90  			    if (tryLoadFile("bin/batchtool.properties")) {
91  			    	log.info("The batchtool.properties file exists in bin/. It should be moved to conf/");
92  			    } else {
93  			    	log.debug("Could not find any batchtool property file, default values will be used.");
94  		            logger.info(InternalEjbcaResources.getInstance().getLocalizedMessage("batch.loadingconfig", "defaults"));
95  			    }
96  			}
97  		} catch (IOException e) {
98  			log.error("Error reading batchtool property file ");
99  			log.debug(e);
100 		}
101 	}
102 
103 
104 
105 }
