23  import java.security.cert.CertificateException;
24  import java.util.Enumeration;
25  
26  import org.apache.log4j.Logger;
27  import org.bouncycastle.jce.provider.BouncyCastleProvider;
28  import org.cesecore.util.EjbRemoteHelper;
29  import org.cesecore.util.FileTools;
30  import org.ejbca.core.ejb.ca.caadmin.CAAdminSessionRemote;

38  /**
39   * Restore a CA token keystore from a PKCS12 file.
40   * 
41   * @version $Id: CaRestoreKeyStoreCommand.java 20925 2015-03-18 19:24:38Z mikekushner $
42   */
43  public class CaRestoreKeyStoreCommand extends BaseCaAdminCommand {
44  
45      private static final Logger log = Logger.getLogger(CaInfoCommand.class);
46  
47      private static final String CA_NAME_KEY = "--caname";
48      private static final String PKCS12_FILE_KEY = "-f";
49      private static final String SIG_ALIAS_KEY = "-s";
50      private static final String ENC_ALIAS_KEY = "-e";
51      private static final String KEYSTORE_PASSWORD_KEY = "-kspassword";
52  
53      {
54          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
55                  "Name of the CA"));
56          registerParameter(new Parameter(PKCS12_FILE_KEY, "Filename", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
57                  "The PKCS#12 keystore to restore from."));
58          registerParameter(new Parameter(SIG_ALIAS_KEY, "Alias", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
59                  "Alias for the Signature key. If left out the existing alias will be used, and if multiple are available a list will be displayed."));
60          registerParameter(new Parameter(ENC_ALIAS_KEY, "Alias", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
61                  "Alias for the Encryption key. If left out the existing alias will be used, and if multiple are available a list will be displayed."));
62          registerParameter(new Parameter(KEYSTORE_PASSWORD_KEY, "Password", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
63                  "If left out, then password will be prompted for."));
64      }
65  
66      @Override
67      public String getMainCommand() {
68          return "restorekeystore";
69      }
70  
71      @Override
72      public CommandResult execute(ParameterContainer parameters) {
73          String kspwd = parameters.get(KEYSTORE_PASSWORD_KEY);
74          String caName = parameters.get(CA_NAME_KEY);
75          // Import soft keystore
76          String p12file = parameters.get(PKCS12_FILE_KEY);
77          String alias = parameters.get(SIG_ALIAS_KEY);
78          String encryptionAlias = parameters.get(ENC_ALIAS_KEY);
79          if (kspwd == null) {
80              getLogger().info("Enter keystore password: ");
81              // Read the password, but mask it so we don't display it on the console
82              kspwd = String.valueOf(System.console().readPassword());
83          } else {
84              getLogger().info("Keystore password was supplied on the command line.");
85          }
86          // Read old keystore file in the beginning so we know it's good
87          byte[] keystorebytes = null;
88          try {
89              keystorebytes = FileTools.readFiletoBuffer(p12file);
90          } catch (FileNotFoundException e) {
91              log.error("File " + p12file + " was not found.");
92              return CommandResult.FUNCTIONAL_FAILURE;
93          }
94          // Import CA from PKCS12 file
95          if (alias == null) {
96              // First we must find what aliases there is in the pkcs12-file
97              KeyStore ks;
98              try {
99                  ks = KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
100             } catch (KeyStoreException e) {
101                 throw new IllegalStateException("PKCS#12 implementation not found.", e);
102             } catch (NoSuchProviderException e) {
103                 throw new IllegalStateException("BouncyCastle provider not found.", e);
104             }
105             FileInputStream fis;
106             try {
107                 fis = new FileInputStream(p12file);
108             } catch (FileNotFoundException e) {
109                 log.error("File " + p12file + " was not found.");
110                 return CommandResult.FUNCTIONAL_FAILURE;
111             }
112             try {
113                 ks.load(fis, kspwd.toCharArray());
114                 fis.close();
115             } catch (NoSuchAlgorithmException e) {
116                 throw new IllegalStateException("Algorithm used to check the integrity of the keystore could npt be found.", e);
117             } catch (CertificateException e) {
118                 throw new IllegalStateException("Certificate could not be loaded from keystore.", e);
119             } catch (IOException e) {
120                 throw new IllegalStateException("Unknown IOException was caught.", e);
121             }
122 
123             Enumeration<String> aliases;
124             try {
125                 aliases = ks.aliases();
126             } catch (KeyStoreException e) {
127                 throw new IllegalStateException("Keystore was not initialized", e);
128             }
129             int length = 0;
130             while (aliases.hasMoreElements()) {
131                 alias = aliases.nextElement();
132                 getLogger().info("Keystore contains alias: " + alias);
133                 length++;
134             }
135             if (length > 1) {
136                 log.error("Keystore contains more than one alias, alias must be provided as argument.");
137                 return CommandResult.FUNCTIONAL_FAILURE;
138             } else if (length < 1) {
139                 log.error("Keystore does not contain any aliases. It can not be used for a CA.");
140                 return CommandResult.FUNCTIONAL_FAILURE;
141             }
142             // else alias already contains the only alias, so we can use that
143         }
144         EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class).restoreCAKeyStore(getAuthenticationToken(), caName, keystorebytes,
145                 kspwd, kspwd, alias, encryptionAlias);
146         return CommandResult.SUCCESS;
147 
148     }
149 
150     @Override
151     public String getCommandDescription() {
152         return "Restore a CA token keystore from a PKCS12 file.";
153     }
154 
155     @Override
156     public String getFullHelpText() {
157         return getCommandDescription();
158     }
159 
160     @Override
161     protected Logger getLogger() {
162         return log;
163     }
164 
165 }
