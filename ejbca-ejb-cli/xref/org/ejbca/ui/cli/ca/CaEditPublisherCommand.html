31  /**
32   * Changes fields in a Publisher.
33   *
34   * @version $Id: CaEditPublisherCommand.java 23457 2016-05-16 17:18:31Z mikekushner $
35   */
36  public class CaEditPublisherCommand extends BaseCaAdminCommand {
37  
38      private static final Logger log = Logger.getLogger(CaEditPublisherCommand.class);
39  
40      private static final String PUBLISHER_NAME_KEY = "--name";
41      private static final String FIELD_KEY = "--field";
42      private static final String VALUE_KEY = "--value";
43      private static final String LISTFIELDS_KEY = "-listFields";
44      private static final String GETVALUE_KEY = "-getValue";
45  
46      {
47          registerParameter(new Parameter(PUBLISHER_NAME_KEY, "Publisher Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
48                  "Name of the certificate profile."));
49          registerParameter(new Parameter(FIELD_KEY, "Field Name", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
50                  "The sought field."));
51          registerParameter(new Parameter(VALUE_KEY, "Value", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "The value to set, if any."));
53          registerParameter(new Parameter(LISTFIELDS_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
54                  "Set to only list available fields in the CA"));
55          registerParameter(new Parameter(GETVALUE_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
56                  "Use this to get the value of a single field"));
57      }
58  
59      @Override
60      public String getMainCommand() {
61          return "editpublisher";
62      }
63  
64      @Override
65      public CommandResult execute(ParameterContainer parameters) {
66          FieldEditorieldEditor">FieldEditor fieldEditor = new FieldEditor(log);
67          CryptoProviderTools.installBCProvider();
68  
69          boolean listOnly = parameters.get(LISTFIELDS_KEY) != null;
70          boolean getOnly = parameters.get(GETVALUE_KEY) != null;
71          final String name = parameters.get(PUBLISHER_NAME_KEY);
72          final String field = parameters.get(FIELD_KEY);
73          final String value = parameters.get(VALUE_KEY);
74  
75          BasePublisher pub = EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).getPublisher(name);
76  
77          if (pub == null) {
78              log.info("Publisher '" + name + "' does not exist.");
79              return CommandResult.FUNCTIONAL_FAILURE;
80          } else {
81              // List fields, get values or set value
82              try {
83                  if(field == null && !listOnly) {
84                      log.error("ERROR: No field value set.");
85                      return CommandResult.FUNCTIONAL_FAILURE;
86                  }  
87                  if(listOnly) {
88                      fieldEditor.listSetMethods(pub);
89                  } else if(getOnly) {
90                      fieldEditor.getBeanValue(field, pub);
91                  } else {
92                      fieldEditor.setValue(name, field, value, pub);
93                      // Store the modifies object
94                      log.info("Storing modified publisher '" + name + "'...");
95                      EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).changePublisher(getAuthenticationToken(), name, pub);
96                      // Verify our new value
97                      log.info("Reading modified value for verification...");
98                      final BasePublisher modpub = EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).getPublisher(name);
99  
100                     // Print return value
101                     fieldEditor.getBeanValue(field, modpub);
102                 }
103                 return CommandResult.SUCCESS;
104             } catch (FieldNotFoundException e) {
105                 log.error(e.getMessage());
106                 return CommandResult.FUNCTIONAL_FAILURE;
107             } catch (AuthorizationDeniedException e) {
108                 log.error("CLI User was not authorized to edit publishers.");
109                 return CommandResult.AUTHORIZATION_FAILURE;
110             }
111         }
112 
113     }
114 
115     @Override
116     public String getCommandDescription() {
117         return "Edits publisher fields of an existing publisher in the CA. ";
118     }
119 
120     @Override
121     public String getFullHelpText() {
122         return getCommandDescription()
123                 + "Fields that can be set are derived from setFieldName of the publisher java code. If there is a 'setFieldName(type)' method, the values to use in this command should be 'fieldName value'\n"
124                 + "Example: ca editpublisher PublisherName hostnames myhost.com\n"
125                 + "Example: ca editpublisher PublisherName addMultipleCertificates true\n"
126                 + "Example: ca editpublisher PublisherName connectionTimeOut 10000\n" + "\n"
127                 + "Use the option -listFields to only list available fields in the publisher.\n"
128                 + "Example: ca editpublisher PublisherName -listFields\n" + "\n"
129                 + "Use the option -getValue to only get the value of a field in the publisher.\n"
130                 + "Example: ca editpublisher PublisherName -getValue hostnames";
131     }
132     
133     @Override
134     protected Logger getLogger() {
135         return log;
136     }
