35  /**
36   * Retrieves the latest CRL from a CA.
37   *
38   * @version $Id: CaGetCrlCommand.java 25279 2017-02-15 14:30:49Z anatom $
39   */
40  public class CaGetCrlCommand extends BaseCaAdminCommand {
41  
42      private static final Logger log = Logger.getLogger(CaGetCrlCommand.class);
43  
44      private static final String CA_NAME_KEY = "--caname";
45      private static final String DELTA_KEY = "-delta";
46      private static final String PEM_KEY = "-pem";
47      private static final String FILE_KEY = "-f";
48      private static final String CRLNUMBER_KEY = "-crlnumber";
49  
50      {
51          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "The CA to get the CRL for."));
53          registerParameter(new Parameter(FILE_KEY, "File Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "The file to export to."));
55          registerParameter(new Parameter(DELTA_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
56                  "Fetch the latest delta CRL. Default is regular CRL."));
57          registerParameter(new Parameter(PEM_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
58                  "Use PEM encoding. Default is DER encoding."));
59          registerParameter(new Parameter(CRLNUMBER_KEY, "CRL Number", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
60                  "Get CRL with the specified CRL number, instead of the latest. Used to read historical CRLs."));
61      }
62  
63      @Override
64      public String getMainCommand() {
65          return "getcrl";
66      }
67  
68      @Override
69      public CommandResult execute(ParameterContainer parameters) {
70          boolean deltaSelector = parameters.get(DELTA_KEY) != null;
71          boolean pem = parameters.get(PEM_KEY) != null;
72          CryptoProviderTools.installBCProvider();
73          // Perform CRL fetch
74          String caname = parameters.get(CA_NAME_KEY);
75          String outfile = parameters.get(FILE_KEY);
76          String crlnumber = parameters.get(CRLNUMBER_KEY);
77          if (crlnumber != null) {
78              if ( !StringUtils.isNumeric(crlnumber) || (Integer.valueOf(crlnumber) < 0) ) {
79                  log.error("CRL Number must be a positive number");
80                  return CommandResult.FUNCTIONAL_FAILURE;
81              }
82          }
83          try {
84              String issuerdn = getIssuerDN(getAuthenticationToken(), caname);
85              final byte[] crl;
86              if (crlnumber != null) {
87                  crl = EjbRemoteHelper.INSTANCE.getRemoteSession(CrlStoreSessionRemote.class).getCRL(issuerdn, Integer.valueOf(crlnumber));                
88              } else {
89                  crl = EjbRemoteHelper.INSTANCE.getRemoteSession(CrlStoreSessionRemote.class).getLastCRL(issuerdn, deltaSelector);
90              }
91              if (crl != null) {
92                  FileOutputStream fos = new FileOutputStream(outfile);
93                  if (pem) {
94                      fos.write(CertTools.getPEMFromCrl(crl));
95                  } else {
96                      fos.write(crl);
97                  }
98                  fos.close();
99                  log.info("Wrote "+(crlnumber == null ? "latest" : ("crlNumber "+crlnumber)) + (deltaSelector ? "delta" : "") + " CRL to " + outfile + " using " + (pem ? "PEM" : "DER") + " format");
100             } else {
101                 log.info("No " + (deltaSelector ? "delta " : "") + "CRL " + (crlnumber == null ? "" : ("with crlNumber "+crlnumber+" ")) + "exists for CA " + caname + ".");
102             }
103             return CommandResult.SUCCESS;
104         } catch (AuthorizationDeniedException e) {
105             log.error("CLI User was not authorized to CA " + caname);
106         } catch (CADoesntExistsException e) {
107             log.info("CA '" + caname + "' does not exist.");
108         } catch (FileNotFoundException e) {
109             log.error("Could not create export file", e);
110         } catch (IOException e) {
111             throw new IllegalStateException("Could not write to file for unknown reason", e);
112         }
113         return CommandResult.FUNCTIONAL_FAILURE;
114 
115     }
116 
117     @Override
118     public String getCommandDescription() {
119         return "Retrieves a CRL from a CA. Either the latest CRL or a CRL with a specified CRL number.";
120 
121     }
122 
123     @Override
124     public String getFullHelpText() {
125         return getCommandDescription();
126     }
127     
128     @Override
129     protected Logger getLogger() {
130         return log;
131     }
132 }
