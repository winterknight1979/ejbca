14  package org.ejbca.ui.cli.ca;
15  
16  import java.security.cert.Certificate;
17  import java.util.Collection;
18  import java.util.Iterator;
19  
20  import org.apache.log4j.Logger;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.ca.CAInfo;
23  import org.cesecore.certificates.ca.CaSessionRemote;
24  import org.cesecore.util.CertTools;
25  import org.cesecore.util.CryptoProviderTools;
26  import org.cesecore.util.EjbRemoteHelper;
27  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
28  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
29  
30  /**
31   * Lists the names of all available CAs.
32   *
33   * @version $Id: CaListCAsCommand.java 27740 2018-01-05 07:24:53Z mikekushner $
34   */
35  public class CaListCAsCommand extends BaseCaAdminCommand {
36  
37      private static final Logger log = Logger.getLogger(CaListCAsCommand.class);
38  
39      @Override
40      public String getMainCommand() {
41          return "listcas";
42      }
43  
44      @Override
45      public CommandResult execute(ParameterContainer parameters) {
46  
47          CryptoProviderTools.installBCProvider();
48          Collection<Integer> caids = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getAuthorizedCaIds(getAuthenticationToken());
49          try {
50              for (int caid : caids) {
51                  CAInfo ca = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caid);       
52                  Collection<Certificate> certs = ca.getCertificateChain();
53                  Iterator<Certificate> ci = certs.iterator();
54                  Certificate cacert = null;
55                  if (ci.hasNext()) {
56                      cacert = (Certificate) ci.next();
57                  }
58                  getLogger().info("CA Name: " + ca.getName());
59                  getLogger().info(" Id: " + ca.getCAId());
60                  if (cacert != null) {
61                      getLogger().info(" Issuer DN: " + CertTools.getIssuerDN(cacert));
62                  }
63                  getLogger().info(" Subject DN: " + ca.getSubjectDN());
64                  getLogger().info(" Type: " + ca.getCAType());
65                  getLogger().info(" Expire time: " + ca.getExpireTime());
66                  getLogger().info(" Signed by: " + ca.getSignedBy());
67              }
68              return CommandResult.SUCCESS;
69          } catch (AuthorizationDeniedException e) {
70              log.error("CLI user not authorized to view CAs");
71              return CommandResult.AUTHORIZATION_FAILURE;
72          }
73  
74      }
75  
76      @Override
77      public String getCommandDescription() {
78          return "Lists the names of all available CAs";
79      }
80  
81      @Override
82      public String getFullHelpText() {
83          return getCommandDescription();
84      }
85  
86      @Override
87      protected Logger getLogger() {
88          return log;
89      }
90  }
