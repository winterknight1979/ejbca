77       * Retrieves the complete certificate chain from the CA
78       * @param authenticationToken token
79       * 
80       * @param caname human readable name of CA
81       * @return a Collection of certificates
82       */
83      protected Collection<Certificate> getCertChain(AuthenticationToken authenticationToken, String caname) {
84          log.trace(">getCertChain()");
85          Collection<Certificate> returnval = new ArrayList<Certificate>();
86          try {
87              CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(authenticationToken, caname);
88              if (cainfo != null) {
89                  returnval = cainfo.getCertificateChain();
90              }
91          } catch (Exception e) {
92              log.error("Error while getting certfificate chain from CA.", e);
93          }
94          log.trace("<getCertChain()");
95          return returnval;
96      }

98      protected void makeCertRequest(String dn, KeyPair rsaKeys, String reqfile) throws NoSuchAlgorithmException, IOException, NoSuchProviderException,
99              InvalidKeyException, SignatureException, OperatorCreationException, PKCSException {
100         log.trace(">makeCertRequest: dn='" + dn + "', reqfile='" + reqfile + "'.");
101 
102         PKCS10CertificationRequest req = CertTools.genPKCS10CertificationRequest("SHA1WithRSA", CertTools.stringToBcX500Name(dn),
103                 rsaKeys.getPublic(), new DERSet(), rsaKeys.getPrivate(), null);
104 
105         /*
106          * We don't use these unnecessary attributes DERConstructedSequence kName
107          * = new DERConstructedSequence(); DERConstructedSet kSeq = new
108          * DERConstructedSet();
109          * kName.addObject(PKCSObjectIdentifiers.pkcs_9_at_emailAddress);
110          * kSeq.addObject(new DERIA5String("foo@bar.se"));
111          * kName.addObject(kSeq); req.setAttributes(kName);
112          */
113         ByteArrayOutputStream bOut = new ByteArrayOutputStream();
114         DEROutputStream dOut = new DEROutputStream(bOut);
115         dOut.writeObject(req.toASN1Structure());
116         dOut.close();
117 
118         PKCS10CertificationRequest req2 = new PKCS10CertificationRequest(bOut.toByteArray());
119         ContentVerifierProvider contentVerifier = CertTools.genContentVerifierProvider(rsaKeys.getPublic());
120         boolean verify = req2.isSignatureValid(contentVerifier); //req2.verify();
121         log.info("Verify returned " + verify);
122 
123         if (verify == false) {
124             log.info("Aborting!");
125             return;
126         }
127 
128         FileOutputStream os1 = new FileOutputStream(reqfile);
129         os1.write("-----BEGIN CERTIFICATE REQUEST-----\n".getBytes());
130         os1.write(Base64.encode(bOut.toByteArray()));
131         os1.write("\n-----END CERTIFICATE REQUEST-----\n".getBytes());
132         os1.close();
133         log.info("CertificationRequest '" + reqfile + "' generated successfully.");
134         log.trace("<makeCertRequest: dn='" + dn + "', reqfile='" + reqfile + "'.");
135     }

137     protected void createCRL(final String issuerdn, final boolean deltaCRL) {
138         log.trace(">createCRL()");
139         try {
140             if (issuerdn != null) {
141                 CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(),
142                         issuerdn.hashCode());
143                 if (!deltaCRL) {
144                     EjbRemoteHelper.INSTANCE.getRemoteSession(PublishingCrlSessionRemote.class).forceCRL(getAuthenticationToken(), cainfo.getCAId());
145                     int number = EjbRemoteHelper.INSTANCE.getRemoteSession(CrlStoreSessionRemote.class).getLastCRLNumber(issuerdn, false);
146                     log.info("CRL with number " + number + " generated.");
147                 } else {
148                     EjbRemoteHelper.INSTANCE.getRemoteSession(PublishingCrlSessionRemote.class).forceDeltaCRL(getAuthenticationToken(),
149                             cainfo.getCAId());
150                     int number = EjbRemoteHelper.INSTANCE.getRemoteSession(CrlStoreSessionRemote.class).getLastCRLNumber(issuerdn, true);
151                     log.info("Delta CRL with number " + number + " generated.");
152                 }
153             } else {
154                 int createdcrls = EjbRemoteHelper.INSTANCE.getRemoteSession(PublishingCrlSessionRemote.class).createCRLs(getAuthenticationToken());
155                 log.info("  " + createdcrls + " CRLs have been created.");
156                 int createddeltacrls = EjbRemoteHelper.INSTANCE.getRemoteSession(PublishingCrlSessionRemote.class).createDeltaCRLs(
157                         getAuthenticationToken());
158                 log.info("  " + createddeltacrls + " delta CRLs have been created.");
159             }
160         } catch (Exception e) {
161             log.error("Error while creating CRL for CA: " + issuerdn, e);
162         }
163         log.trace(">createCRL()");
164     }

166     protected String getIssuerDN(AuthenticationToken authenticationToken, String caname) throws CADoesntExistsException, AuthorizationDeniedException {
167         CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(authenticationToken, caname);
168         return cainfo != null ? cainfo.getSubjectDN() : null;
169     }

171     protected CAInfo getCAInfo(AuthenticationToken authenticationToken, String caname) {
172         CAInfo result = null;
173         try {
174             result = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(authenticationToken, caname);
175         }  catch (AuthorizationDeniedException e) {
176             log.error("Authorization denied", e);
177         }
178         if (result == null) {
179             log.debug("CA " + caname + " not found.");
180         }
181         return result;
182     }

195     protected String getAvailableCasString() {
196         // List available CAs by name
197         final StringBuilder existingCas = new StringBuilder();
198         try {
199             for (final Integer nextId : EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getAuthorizedCaIds(getAuthenticationToken())) {
200                 final String caName = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class)
201                         .getCAInfo(getAuthenticationToken(), nextId.intValue()).getName();
202                 if (existingCas.length() > 0) {
203                     existingCas.append(", ");
204                 }
205                 existingCas.append("\"").append(caName).append("\"");
206             }
207         } catch (Exception e) {
208             existingCas.append("<unable to fetch available CA(s)>");
209         }
210         return existingCas.toString();
211     }

251     protected String getCaList() {
252         final String TAB = "    ";
253         Collection<Integer> cas = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getAuthorizedCaIds(getAuthenticationToken());
254         String casList = "Available CAs:\n";
255         for (Integer caid : cas) {
256             CAInfo info;
257             try {
258                 info = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caid);
259                 if (info.getStatus() == CAConstants.CA_EXTERNAL) {
260                     casList += TAB + info.getName() + ": External CA\n";
261                 } else {
262                     int cryptoTokenId = info.getCAToken().getCryptoTokenId();
263                     String cryptoTokenName = "Current CLI user does not have authorization to Crypto Tokens.\n";
264                     try {
265                         CryptoTokenInfo ctInfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CryptoTokenManagementSessionRemote.class).getCryptoTokenInfo(getAuthenticationToken(), cryptoTokenId);
266                         if (ctInfo != null) {
267                             cryptoTokenName = ctInfo.getName();
268                         } else {
269                             cryptoTokenName = "ID "+cryptoTokenId;
270                         }
271                     } catch (AuthorizationDeniedException e) {
272                         // NOPMD: ignore, use string above
273                     }
274                     casList += TAB + info.getName() + ":" + cryptoTokenName + ":" + info.getCAToken().getSignatureAlgorithm() + "\n";
275                 }
276             } catch (AuthorizationDeniedException e) {
277                 casList = "Current CLI user does not have authorization to any CAs.\n";
278                 break;
279             }
280         }
281         return casList;
282     }
