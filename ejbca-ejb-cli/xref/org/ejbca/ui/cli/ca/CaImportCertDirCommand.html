101     @Override
102     public CommandResult execute(final ParameterContainer parameters) {
103         log.trace(">execute()");
104 
105         CryptoProviderTools.installBCProviderIfNotAvailable();
106 
107         // Parse arguments into more coder friendly variable names and validate switches
108         final String usernameFilter = parameters.get(USERNAME_FILTER_KEY);
109         final String caName = parameters.get(CA_NAME_KEY);
110         final String active = parameters.get(ACTIVE_KEY);
111         final String certificateDir = parameters.get(DIRECTORY_KEY);
112         final String eeProfile = parameters.get(EE_PROFILE_KEY);
113         final String certificateProfile = parameters.get(CERT_PROFILE_KEY);
114         final String revocationReasonString = parameters.get(REVOCATION_REASON);
115         final String revocationTimeString = parameters.get(REVOCATION_TIME);
116         final int threadCount = parameters.get(THREAD_COUNT) == null ? 1 : Integer.valueOf(StringUtils.strip(parameters.get(THREAD_COUNT)));
117 
118         if (threadCount > 1 && !usernameFilter.equalsIgnoreCase("FILE")) {
119             log.error("If more than one thread is being used, filename must be used as filter (use the argument --filter FILE).");
120             return CommandResult.CLI_FAILURE;
121         }
122 
123         // Specifies whether the import should resume in case of errors, or stop
124         // on first one. Default is stop.
125         final boolean resumeOnError = parameters.containsKey(RESUME_ON_ERROR_KEY);
126 
127         // Thread pool for running multiple import operations simultaneously
128         final ExecutorService executorService = Executors.newFixedThreadPool(threadCount);
129 
130         try {
131             final int status;
132             RevocationReasons revocationReason = null;
133             Date revocationTime = null;
134             if ("ACTIVE".equalsIgnoreCase(active)) {
135                 status = CertificateConstants.CERT_ACTIVE;
136                 if(revocationReasonString != null) {
137                     log.warn("Revocation reason has been set in spite of certificates being imported as active. Ignoring.");
138                 }
139                 if(revocationTimeString != null) {
140                     log.warn("Revocation time has been set in spite of certificates being imported as active. Ignoring.");
141                 }
142             } else if ("REVOKED".equalsIgnoreCase(active)) {
143                 status = CertificateConstants.CERT_REVOKED;
144                 if(revocationReasonString != null) {
145                     revocationReason = RevocationReasons.getFromCliValue(revocationReasonString.toUpperCase());
146                     if(revocationReason == null) {
147                         log.error("ERROR: " + revocationReasonString + " is not a valid revocation reason.");
148                         return CommandResult.CLI_FAILURE;
149                     }
150                 } else {
151                     revocationReason = RevocationReasons.UNSPECIFIED;
152                 }
153                 if (revocationTimeString != null) {
154                     try {
155                         revocationTime = new SimpleDateFormat(DATE_FORMAT).parse(revocationTimeString);
156                     } catch (ParseException e) {
157                         log.error("ERROR: " + revocationTimeString + " was not a valid revocation time.");
158                         return CommandResult.CLI_FAILURE;
159                     }
160                 } else {
161                     revocationTime = new Date();
162                 }
163 
164             } else {
165                 log.error("Invalid certificate status.");
166                 return CommandResult.CLI_FAILURE;
167             }
168             if (!usernameFilter.equalsIgnoreCase("DN") && !usernameFilter.equalsIgnoreCase("CN") && !usernameFilter.equalsIgnoreCase("FILE")) {
169                 log.error(usernameFilter
170                         + "is not a valid option. Currently only \"DN\", \"CN\" and \"FILE\" username-source are implemented");
171                 return CommandResult.CLI_FAILURE;
172             }
173 
174             // Fetch CA info
175             final CAInfo caInfo = getCAInfo(getAuthenticationToken(), caName);
176             final X509Certificate cacert = (X509Certificate) caInfo.getCertificateChain().iterator().next();
177             final String issuer = CertTools.stringToBCDNString(cacert.getSubjectDN().toString());
178             log.info("CA: " + issuer);
179             // Fetch End Entity Profile info
180             log.debug("Searching for End Entity Profile " + eeProfile);
181             final int endEntityProfileId = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class)
182                     .getEndEntityProfileId(eeProfile);
183             // Fetch Certificate Profile info
184             log.debug("Searching for Certificate Profile " + certificateProfile);
185             final int certificateProfileId = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileId(
186                     certificateProfile);
187             if (certificateProfileId == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
188                 log.error("ERROR: Certificate Profile " + certificateProfile + " does not exist.");
189                 throw new Exception("Certificate Profile '" + certificateProfile + "' does not exist.");
190             }
191             // Get all files in the directory to import from and try to read and import each as a certificate
192             final File dir = new File(certificateDir);
193             if (!dir.isDirectory()) {
194                 log.error("'" + certificateDir + "' is not a directory.");
195                 return CommandResult.CLI_FAILURE;
196             }
197             final File files[] = dir.listFiles();
198             if (files == null || files.length < 1) {
199                 log.error("No files in directory '" + dir.getCanonicalPath() + "'. Nothing to do.");
200                 return CommandResult.CLI_FAILURE;
201             }
202 
203             final Queue<Future<CertificateImporter.Result>> futures = new LinkedList<>();
204             final List<CertificateImporter.Result> results = new LinkedList<>();
205             int redundant = 0;
206             int caMismatch = 0;
207             int readError = 0;
208             int constraintViolation = 0;
209             int generalImportError = 0;
210             int importOk = 0;
211 
212             final long startTime = System.currentTimeMillis();
213 
214 
215             for (final File file : files) {
216                 futures.add(executorService.submit(new CertificateImporter()
217                         .setAuthenticationToken(getAuthenticationToken())
218                         .setCaCertificate(cacert)
219                         .setCaInfo(caInfo)
220                         .setCertificateProfileId(certificateProfileId)
221                         .setEndEntityProfileId(endEntityProfileId)
222                         .setFileToImport(file)
223                         .setIssuer(issuer)
224                         .setResumeOnError(resumeOnError)
225                         .setRevocationReason(revocationReason)
226                         .setRevocationTime(revocationTime)
227                         .setStatus(status)
228                         .setUsernameFilter(usernameFilter)));
229                 // Process completed tasks
230                 while (futures.peek() != null && futures.peek().isDone()) {
231                     results.add(futures.remove().get());
232                 }
233             }
234 
235             for (final Future<CertificateImporter.Result> future : futures) {
236                 results.add(future.get());
237             }
238 
239             for (final CertificateImporter.Result result : results) {
240                 if (result == CertificateImporter.Result.REDUNDANT) {
241                     redundant++;
242                 } else if (result == CertificateImporter.Result.CA_MISMATCH) {
243                     caMismatch++;
244                 } else if (result == CertificateImporter.Result.READ_ERROR) {
245                     readError++;
246                 } else if (result == CertificateImporter.Result.CONSTRAINT_VIOLATION) {
247                     constraintViolation++;
248                 } else if (result == CertificateImporter.Result.GENERAL_IMPORT_ERROR) {
249                     generalImportError++;
250                 } else if (result == CertificateImporter.Result.IMPORT_OK) {
251                     importOk++;
252                 }
253             }
254 
255             final long stopTime = System.currentTimeMillis();
256             final double seconds = (stopTime - startTime) / 1000;
257 
258             // Print resulting statistics
259             log.info("\nImport summary:");
260             log.info(importOk + " certificates were imported successfully.");
261             log.info("Time: " + seconds + " seconds (" + (files.length / seconds) + " tps)");
262             if (redundant > 0) {
263                 log.info(redundant + " certificates were already present in the database.");
264             }
265             if (caMismatch > 0) {
266                 log.info(caMismatch + " certificates were not issued by the specified CA.");
267             }
268             if (readError > 0) {
269                 log.info(readError + " certificates could not be read.");
270             }
271             if (constraintViolation > 0) {
272                 log.info(constraintViolation + " certificates violated the end entity constraints.");
273             }
274             if (generalImportError > 0) {
275                 log.info(generalImportError + " certificates were not imported due to other errors.");
276             }
277         } catch (Exception e) {
278             log.error("ERROR: " + e.getMessage());
279             return CommandResult.FUNCTIONAL_FAILURE;
280         } finally {
281             executorService.shutdown();
282             log.trace("<execute()");
283         }
284         return CommandResult.SUCCESS;
285     }

289         return "Imports a directory with PEM encoded certficate file(s) to the database.";

293     @Override
294     public String getFullHelpText() {
295         StringBuilder sb = new StringBuilder();
296         sb.append(getCommandDescription() + "\n\n");
297         sb.append("The import will proceed as long as the only error encountered is a mismatch between the signer of end entity certificate "
298                 + "and the specified certification authority. If the -resumeonerror options is provided, the import will resume even in case of more "
299                 + "critical errors (such as violation of end entity constraints, malformed files etc). The offending files will be printed-out to the "
300                 + "standard output, and a summary will contain exact numbers on imported and failed certificates." + "\n\n");
301         // List available CAs by name
302         sb.append("Available CAs: " + getAvailableCasString() + "\n");
303         sb.append("Available end entity profiles: " + getAvailableEepsString(AccessRulesConstants.CREATE_END_ENTITY) + "\n");
304         sb.append("Available certificate profiles: " + getAvailableEndUserCpsString() + "\n");
305         sb.append("Valid Revocation reasons: ");
306         RevocationReasons[] values = RevocationReasons.values();
307         for(int i = 0; i < values.length; ++i) {
308             if(!values[i].equals(RevocationReasons.NOT_REVOKED)) {
309                 sb.append(values[i].getStringValue());
310                 if(i < values.length -1) {
311                     sb.append(" | ");
312                 }
313             }
314         }
315 
316         return sb.toString();
317     }
