22  /**
23   * Used for importing certificates from a MS CA dump. The dump is assumed to be produced with
24   *  certutil -view -restrict "GeneralFlags&gt;0" /out "UPN,CertificateTemplate,Disposition,RawCertificate" &gt; certdump.txt
25   *
26   *  The CA has to be imported before this code runs.
27   */
28  public class ImportMSCACertificates extends CaImportCertCommand {
29  
30      private static final Logger log = Logger.getLogger(ImportMSCACertificates.class);
31  
32      private static final String CA_NAME_KEY = "--caname";
33      private static final String INPUT_FILE = "-f";
34  
35      {
36          registerParameter(new Parameter(
37                  INPUT_FILE,
38                  "Filename",
39                  MandatoryMode.MANDATORY,
40                  StandaloneMode.ALLOW,
41                  ParameterMode.ARGUMENT,
42                  "Input file"));
43          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
44                  "The name of the CA to import."));
45      }
46  
47      @Override
48      public String getMainCommand() {
49          return "importcertsms";
50      }
51  
52      @Override
53      public CommandResult execute(ParameterContainer parameters) {
54          ArrayList<String> usedNames = new ArrayList<String>();
55          File infile = new File(parameters.get(INPUT_FILE));
56          if (!infile.exists() || !infile.isFile()) {
57              getLogger().error(infile + " is not a file.");
58              return CommandResult.FUNCTIONAL_FAILURE;
59          }
60          String caname = parameters.get(CA_NAME_KEY);
61          /*
62           * Locates next line that starts with "Row"
63           * Parses UPN
64           * Parses TempateName
65           * Parses certificate status from the Disposition-field
66           * writes PEM-certificate to temporary file
67           * runs the import-CLI
68                o Username: UPN-TempateName
69                o Password: foo123
70                o Ca name: From the command line of the script. Should be the name of the imported MS CA.
71                o status: ACTIVE if issued and REVOKED if revoked
72                o filename: the temporary file
73                o EndEntityProfile: TemplateName (this of course have to exist.. maybe easily importable from this page)
74                o CertificateProfile: TemplateName (this of course have to exist.. maybe easily importable from this page)
75           * Start over until there are no more "Row"s
76           */
77          FileInputStream fstream;
78          try {
79              fstream = new FileInputStream(infile);
80              BufferedReader br = new BufferedReader(new InputStreamReader(fstream));
81              String strLine;
82              String currentUPN = null;
83              String currentTemplate = null;
84              String currentStatus = null;
85              boolean isProcessingCertificate = false;
86              File tempFile = null;
87              Writer bw = null;
88              while ((strLine = br.readLine()) != null) {
89                  if (strLine.startsWith("  User Principal Name:")) {
90                      currentUPN = strLine.split("\"")[1];
91                  }
92                  if (strLine.startsWith("  Certificate Template:")) {
93                      currentTemplate = strLine.split("\"")[1];
94                  }
95                  if (strLine.startsWith("  Request Disposition:")) {
96                      if (strLine.endsWith("Issued")) {
97                          currentStatus = "ACTIVE";
98                      } else {
99                          currentStatus = "REVOKED";
100                     }
101                 }
102                 if (strLine.equals("-----BEGIN CERTIFICATE-----")) {
103                     isProcessingCertificate = true;
104                     tempFile = File.createTempFile("certificate-", ".pem");
105                     bw = new BufferedWriter(new FileWriter(tempFile));
106                 }
107                 if (isProcessingCertificate) {
108                     bw.write(strLine + "\n");
109                 }
110                 if (strLine.equals("-----END CERTIFICATE-----")) {
111                     bw.close();
112                     isProcessingCertificate = false;
113                     getLogger().info("Wrote certificate to " + tempFile);
114                     getLogger().info("Template : " + currentTemplate + " UPN: " + currentUPN);
115                     String username = currentUPN + "-" + currentTemplate;
116                     int i = 2;
117                     while (usedNames.contains(username)) {
118                         username = currentUPN + "-" + currentTemplate + "-" + i;
119                         i++;
120                     }
121                     usedNames.add(username);
122                     String[] newArgs = { super.getMainCommand(), super.getMainCommand(), username, "foo123", caname, currentStatus,
123                             tempFile.getCanonicalPath(), currentTemplate, currentTemplate };
124                     super.execute(newArgs);
125                     tempFile.delete();
126                 }
127             }
128             fstream.close();
129             return CommandResult.SUCCESS;
130         } catch (FileNotFoundException e) {
131             log.error("File " + infile.getName() + " not found.");
132             return CommandResult.FUNCTIONAL_FAILURE;
133         } catch (IOException e) {
134            throw new IllegalStateException("Unknown IOException was caught.", e);
135         }
136     }
