39  /**
40   * CLI command to change crypto token and CA token properties of a CA.
41   * 
42   * @version $Id: CaChangeCryptoTokenCommand.java 25801 2017-05-05 09:07:08Z anatom $
43   */
44  public class CaChangeCryptoTokenCommand extends BaseCaAdminCommand {
45  
46      private static final Logger log = Logger.getLogger(CaChangeCryptoTokenCommand.class);
47      private static final String CA_NAME_KEY = "--caname";
48      private static final String CRYPTOTOKEN_NAME_KEY = "--cryptotoken";
49      private static final String EXECUTE_KEY = "--execute";
50      private static final String CA_TOKEN_PROPERTIES_KEY = "--tokenprop";
51  
52  
53      {
54          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.FORBID, ParameterMode.ARGUMENT, "The name of the CA."));
55          registerParameter(new Parameter(CRYPTOTOKEN_NAME_KEY, "Crypto token name", MandatoryMode.MANDATORY, StandaloneMode.FORBID, ParameterMode.ARGUMENT, "The name of the new crypto token the CA should reference."));
56          registerParameter(new Parameter(CA_TOKEN_PROPERTIES_KEY, "Filename", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
57                  "CA Token properties is a file were you define key aliases for the CA, leave out to keep existing properties."));
58          registerParameter(Parameter.createFlag(EXECUTE_KEY, "Make the change instead of displaying what would change."));
59      }
60  
61      /*
62       * <pre>
63  certSignKey fooalias02
64  crlSignKey fooalias02
65  keyEncryptKey fooencalias
66  hardTokenEncrypt fooencalias
67  defaultKey defaultalias
68  testKey testalias
69  previousCertSignKey fooalias01
70  nextCertSignKey fooalias03
71       * </pre>
72       */
73  
74      @Override
75      public String getMainCommand() {
76          return "changecatoken";
77      }
78  
79      @Override
80      public CommandResult execute(ParameterContainer parameters) {
81          log.trace(">execute()");
82          CryptoProviderTools.installBCProvider(); // need this for CVC certificate
83          final String caName = parameters.get(CA_NAME_KEY);
84          final String cryptoTokenName = parameters.get(CRYPTOTOKEN_NAME_KEY);
85          final boolean force = parameters.containsKey(EXECUTE_KEY);
86          Properties caTokenProperties = new Properties();
87          String caTokenPropertiesFile = parameters.get(CA_TOKEN_PROPERTIES_KEY);
88          if (caTokenPropertiesFile != null) {
89              if ((caTokenPropertiesFile != null) && (!caTokenPropertiesFile.equalsIgnoreCase("null"))) {
90                  File file = new File(caTokenPropertiesFile);
91                  if (!file.exists()) {
92                      getLogger().error("CA Token properties file " + caTokenPropertiesFile + " does not exist.");
93                      return CommandResult.FUNCTIONAL_FAILURE;
94                  } else if (file.isDirectory()) {
95                      getLogger().error("CA Token properties file " + caTokenPropertiesFile + " is a directory.");
96                      return CommandResult.FUNCTIONAL_FAILURE;
97                  } else {
98                      try (final FileInputStream fis = new FileInputStream(caTokenPropertiesFile)) {
99                          caTokenProperties.load(fis);
100                     } catch (FileNotFoundException e) {
101                         //Can't happen
102                         throw new IllegalStateException("Newly referenced file " + caTokenPropertiesFile + " was not found.", e);
103                     } catch (IOException e) {
104                         throw new IllegalStateException("Unknown exception was caught when reading input stream", e);
105                     }
106                 }
107             }
108         }
109 
110         try {
111             final CaSessionRemote caSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class);
112             final CryptoTokenManagementSessionRemote cryptoTokenManagementSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CryptoTokenManagementSessionRemote.class);
113             final CAInfo caInfo = caSession.getCAInfo(getAuthenticationToken(), caName);
114             final int cryptoTokenId = caInfo.getCAToken().getCryptoTokenId();
115             final CryptoTokenInfo cryptoTokenInfo = cryptoTokenManagementSession.getCryptoTokenInfo(getAuthenticationToken(), cryptoTokenId);
116 
117             getLogger().info("CA '" + caInfo.getName() + "' references crypto token '" + (cryptoTokenInfo != null ? cryptoTokenInfo.getName() : (cryptoTokenId+" (does not exist)"))+ "'");
118             getLogger().info("New crypto token: " + cryptoTokenName);
119             getLogger().info("CA token properties: " + caTokenPropertiesFile);
120             getLogger().info("");
121 
122             
123             Integer newId = cryptoTokenManagementSession.getIdFromName(cryptoTokenName);
124             if (newId == null) {
125                 throw new IllegalArgumentException("Crypto Token with name " + cryptoTokenName + " does not exist.");
126             }
127             if (newId == cryptoTokenId) {
128                 getLogger().info("Current crypto token and new crypto token are the same, continuing in order to update CA token properties.");
129             }
130             final CryptoTokenInfo newCryptoTokenInfo = cryptoTokenManagementSession.getCryptoTokenInfo(getAuthenticationToken(), newId);
131 
132             getLogger().info((force?"Changing ":"Would change" )+" CA '" + caInfo.getName() + "' that currently references crypto token '" + (cryptoTokenInfo != null ? cryptoTokenInfo.getName() : (cryptoTokenId+" (does not exist)")) + "' to instead reference '" + newCryptoTokenInfo.getName() + "'.");
133             getLogger().info(" CA token properties that will be updated: " + caTokenProperties);
134 
135             
136             if (force) {
137                 final CAToken currentCaToken = caInfo.getCAToken();
138                 currentCaToken.setCryptoTokenId(newId);
139                 // Set the key options in the CA token properties
140                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING) != null) {
141                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING));                    
142                 }
143                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING) != null) {
144                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING));                    
145                 }
146                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_KEYENCRYPT_STRING) != null) {
147                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_KEYENCRYPT_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_KEYENCRYPT_STRING));                    
148                 }
149                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING) != null) {
150                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING));                    
151                 }
152                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING) != null) {
153                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING));                    
154                 }
155                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT_STRING) != null) {
156                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT_STRING, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT_STRING));                    
157                 }
158                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS) != null) {
159                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS));                    
160                 }
161                 if (caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT) != null) {
162                     currentCaToken.setProperty(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT, (String)caTokenProperties.get(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT));                    
163                 }
164                 caInfo.setCAToken(currentCaToken);
165                 caSession.editCA(getAuthenticationToken(), caInfo);
166                 getLogger().info(" Merged.");
167             }
168             getLogger().info("");
169 
170             if (force) {
171                 getLogger().info("Modified referenced Crypto Token for CA.");
172                 final CAInfo caInfoNew = caSession.getCAInfo(getAuthenticationToken(), caName);
173                 final CAToken newCaToken = caInfoNew.getCAToken();
174                 final CryptoTokenInfo ctInfo = cryptoTokenManagementSession.getCryptoTokenInfo(getAuthenticationToken(), newCaToken.getCryptoTokenId());
175 
176                 getLogger().info("CA '" + caInfoNew.getName() + "' now references crypto token '" + ctInfo.getName() + "'");
177                 getLogger().info("New CA token properties: " + caTokenProperties.toString());
178                 getLogger().info("");
179             } else {
180                 getLogger().info("Will modify referenced Crypto Token for CA if '" + EXECUTE_KEY + "' option is used.");
181             }
182             log.trace("<execute()");          
183         } catch (AuthorizationDeniedException e) {
184             getLogger().error("CLI User was not authorized to modify CA " + caName);
185             log.trace("<execute()");
186             return CommandResult.AUTHORIZATION_FAILURE;
187         } catch (CADoesntExistsException e) {
188             getLogger().error("No such CA with by name " + caName);
189             getLogger().error(getCaList());
190             return CommandResult.FUNCTIONAL_FAILURE;
191         } 
192         log.trace("<execute()");
193         return CommandResult.SUCCESS;
194     }
195 
196     @Override
197     public String getCommandDescription() {
198         return "Change Crypto Token and keys for a CA.";
199     }
200 
201     @Override
202     public String getFullHelpText() {
203         return getCommandDescription()
204                 + "\n\n"
205                 + "The specified CA's Crypto Token will be changed to one passed as parameter, and CA token properties (key labels) updated from the properties file.\n\n"
206                 + "The default behavior is to only show what would have changed since this command is potentially very dangerous.\n"
207                 + "Use the " + EXECUTE_KEY + " switch to execute modifications."
208                 + "\n\n" + getCaList();
209     }
210 
211     @Override
212     protected Logger getLogger() {
213         return log;
214     }
