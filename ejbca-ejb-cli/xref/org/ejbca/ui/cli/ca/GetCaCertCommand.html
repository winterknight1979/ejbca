16  import java.io.FileNotFoundException;
17  import java.io.FileOutputStream;
18  import java.io.IOException;
19  import java.security.cert.Certificate;
20  import java.security.cert.CertificateEncodingException;
21  import java.util.ArrayList;
22  import java.util.Arrays;
23  import java.util.HashSet;
24  import java.util.Set;
25  
26  import org.apache.log4j.Logger;
27  import org.cesecore.util.CertTools;
28  import org.cesecore.util.CryptoProviderTools;
29  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
30  import org.ejbca.ui.cli.infrastructure.parameter.Parameter;
31  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
32  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
33  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
34  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;

36  /**
37   * Export any CA certificate.
38   *
39   * @version $Id: GetCaCertCommand.java 20954 2015-03-23 10:14:06Z mikekushner $
40   */
41  public class GetCaCertCommand extends BaseCaAdminCommand {
42  
43      private static final Logger log = Logger.getLogger(GetCaCertCommand.class);
44  
45      private static final String CA_NAME_KEY = "--caname";
46      private static final String FILE_KEY = "-f";
47      private static final String DER_KEY = "-der";
48      private static final String FULL_CERT_CHAIN_KEY = "--include-full-chain";
49  
50      {
51          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "The CA to export the root certificate from"));
53          registerParameter(new Parameter(FILE_KEY, "File Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "The file to export to."));
55          registerParameter(new Parameter(DER_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
56                  "Use DER encoding. Default is PEM encoding."));
57          registerParameter(Parameter
58                  .createFlag(FULL_CERT_CHAIN_KEY, "Set this flag to get the full certificate chain. Only available for PEM format."));
59      }
60  
61      @Override
62      public String getMainCommand() {
63          return "getcacert";
64      }
65      
66      @Override
67      public Set<String> getMainCommandAliases() {    
68          return new HashSet<String>(Arrays.asList("getrootcert"));
69      }
70  
71      @Override
72      public CommandResult execute(ParameterContainer parameters) {
73          boolean pem = !parameters.containsKey(DER_KEY);
74          boolean fullChain = parameters.containsKey(FULL_CERT_CHAIN_KEY);
75  
76          String caname = parameters.get(CA_NAME_KEY);
77          String filename = parameters.get(FILE_KEY);
78  
79          CryptoProviderTools.installBCProvider();
80          ArrayList<Certificate> chain = new ArrayList<Certificate>(getCertChain(getAuthenticationToken(), caname));
81          try {
82              if (chain.size() > 0) {
83                  Certificate caCert = (Certificate) chain.get(0);
84  
85                  FileOutputStream fos;
86                  try {
87                      fos = new FileOutputStream(filename);
88                  } catch (FileNotFoundException e) {
89                      log.error("Could not create export file", e);
90                      return CommandResult.FUNCTIONAL_FAILURE;
91                  }
92                  if (pem) {
93                      if (fullChain) {
94                          fos.write(CertTools.getPemFromCertificateChain(chain));
95                      } else {
96                          fos.write(CertTools.getPemFromCertificateChain(chain.subList(0, 1)));
97                      }
98                  } else {
99                      fos.write(caCert.getEncoded());
100                     if (fullChain) {
101                         log.info("Full certificate chain not available in DER format. Writing only CA certificate.");
102                     }
103                 }
104                 fos.close();
105                 log.info("Wrote CA certificate " + (pem && fullChain ? "with full certificate chain " : "") + "to '" + filename + "' using " + (pem ? "PEM" : "DER") + " encoding.");
106                 return CommandResult.SUCCESS;
107             } else {
108                 log.error("No CA certificate found.");
109                 return CommandResult.FUNCTIONAL_FAILURE;
110             }
111         } catch (IOException e) {
112             throw new IllegalStateException("Could not write to file for unknown reason", e);
113         } catch (CertificateEncodingException e) {
114             throw new IllegalStateException("An encoding error was encountered", e);
115         }
116     }
117 
118     @Override
119     public String getCommandDescription() {
120         return "Save a CA certificate (PEM- or DER-format) to file";
121 
122     }
123 
124     @Override
125     public String getFullHelpText() {
126         return getCommandDescription();
127     }
128     
129     @Override
130     protected Logger getLogger() {
131         return log;
132     }
133 }
