49  /**
50   * Imports a PEM file and creates a new external CA representation from it.
51   */
52  public class CaImportCACertCommand extends BaseCaAdminCommand {
53  
54      private static final Logger log = Logger.getLogger(CaImportCACertCommand.class);
55  
56      private static final String CA_NAME_KEY = "--caname";
57      private static final String FILE_KEY = "-f";
58      private static final String INIT_AUTH_KEY = "-initauthorization";
59      private static final String SUPERADMIN_CN_KEY = "-superadmincn";
60  
61      {
62          registerParameter(new Parameter(
63                  CA_NAME_KEY,
64                  "CA Name",
65                  MandatoryMode.MANDATORY,
66                  StandaloneMode.ALLOW,
67                  ParameterMode.ARGUMENT,
68                  "Name of the affected CA. If the CA is present, it must either be waiting for a certificate response from an external CA or itself be "
69                          + "an external CA, in which case its certificate will be updated. If the CA is not present, a new CA will be added using the imported certificate chain."));
70          registerParameter(new Parameter(FILE_KEY, "File Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
71                  "A file containing a certificate, in either PEM or DER format. If the CA is waiting for a CSR, this certificate should be the response. "
72                          + "If the CA is an externally imported CA, then it will be updated using this certificate. "
73                          + "If the CA doesn't exist it will be imported using this certificate. "));
74          registerParameter(Parameter.createFlag(INIT_AUTH_KEY,
75                  "This flag may be used when importing an external CA and will create a super administrator (full access) issued by that CA."));
76          registerParameter(new Parameter(SUPERADMIN_CN_KEY, "CN", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
77                  "Required when using " + INIT_AUTH_KEY + ". The Common Name (CN) for the created super administrator."));
78      }
79  
80      @Override
81      public String getMainCommand() {
82          return "importcacert";
83      }
84  
85      @Override
86      public CommandResult execute(ParameterContainer parameters) {
87          String caName = parameters.get(CA_NAME_KEY);
88          String certificateFile = parameters.get(FILE_KEY);
89  
90          boolean initAuth = parameters.containsKey(INIT_AUTH_KEY);
91          String superAdminCN = parameters.get(SUPERADMIN_CN_KEY);
92          if (initAuth && StringUtils.isEmpty(superAdminCN)) {
93              log.error("Error: " + INIT_AUTH_KEY + " flag was used, but super administrator Common Name was not defined with the " + SUPERADMIN_CN_KEY
94                      + " switch.");
95              return CommandResult.FUNCTIONAL_FAILURE;
96          }
97          try {
98              CryptoProviderTools.installBCProviderIfNotAvailable();
99              Certificate certificate;
100             Collection<Certificate> certs;
101             try {
102                 try {
103                     //Try to parse as PEM
104                     certs = CertTools.getCertsFromPEM(certificateFile, Certificate.class);
105                     if (certs.size() != 1) {
106                         log.error("PEM file must only contain one CA certificate, this PEM file contains " + certs.size() + ".");
107                         return CommandResult.FUNCTIONAL_FAILURE;
108                     } else {
109                         certificate = certs.iterator().next();
110                     }
111                 } catch (CertificateParsingException e) {
112                     //Try parsing as binary instead.
113                     try {
114                         certificate = CertTools.getCertfromByteArray(IOUtils.toByteArray(new FileInputStream(certificateFile)), Certificate.class);
115                         certs = Arrays.asList(certificate);
116                     } catch (CertificateParsingException e2) {
117                         log.error("Error: " + certificateFile + " does not contain a certificate, either in PEM or in binary format.");
118                         return CommandResult.CLI_FAILURE;
119                     }
120                 }
121             } catch (FileNotFoundException e) {
122                 log.error("Error: " + certificateFile + " was not a file, not found or could otherwise not be opened.");
123                 return CommandResult.CLI_FAILURE;
124             } catch(IOException e) {
125                 log.error("Unknown IOException was caught", e);
126                 return CommandResult.FUNCTIONAL_FAILURE;
127             }
128 
129             /* 
130              * We need to check if the CA already exists to determine what to do:
131              *  - If CA already exist, it might be a sub CA that is waiting for certificate from an external CA
132              *  - If the CA does not already exist, we import the CA certificate as an "External CA" certificate in EJBCA, so we have the CA cert in EJBCA as a trust point
133              *    getCAInfo throws an exception (CADoesntExistsException) if the CA does not exists, that is how we check if the CA exists 
134              */
135             CAAdminSessionRemote caAdminSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class);
136             CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caName);
137             if (cainfo == null) {
138                 // CA does not exist, we can import the certificate
139                 if (initAuth) {
140                     String subjectdn = CertTools.getSubjectDN(certificate);
141                     Integer caid = Integer.valueOf(subjectdn.hashCode());
142                     initAuthorizationModule(getAuthenticationToken(), caid.intValue(), superAdminCN);
143                 }
144                 caAdminSession.importCACertificate(getAuthenticationToken(), caName, EJBTools.wrapCertCollection(certs));
145                 log.info("Imported CA " + caName);
146                 return CommandResult.SUCCESS;
147             }
148             
149             if (cainfo.getStatus() == CAConstants.CA_WAITING_CERTIFICATE_RESPONSE) {
150                 if (initAuth) {
151                     log.warn("Warning: " + INIT_AUTH_KEY + " was defined but was ignored when receiving a CSR.");
152                 }
153 
154                 log.info("CA '" + caName
155                         + "' is waiting for certificate response from external CA, importing certificate as certificate response to this CA.");
156                 X509ResponseMessage resp = new X509ResponseMessage();
157                 resp.setCertificate(certificate);
158                 caAdminSession.receiveResponse(getAuthenticationToken(), cainfo.getCAId(), resp, null, null);
159                 log.info("Received certificate response and activated CA " + caName);
160             } else if (cainfo.getStatus() == CAConstants.CA_EXTERNAL) {
161                 if (initAuth) {
162                     log.warn("Warning: " + INIT_AUTH_KEY + " was defined but was ignored when updating an externally imported CA.");
163                 }
164                 // CA exists and this is assumed to be an update of the imported CA certificate
165                 log.info("CA '" + caName + "' is an external CA created by CA certificate import. Trying to update the CA certificate chain.");
166                 caAdminSession.updateCACertificate(getAuthenticationToken(), cainfo.getCAId(), EJBTools.wrapCertCollection(certs));
167                 log.info("Updated certificate chain for imported external CA " + caName);
168             } else {
169                 log.error("CA '" + caName + "' already exists and is not waiting for certificate response from an external CA.");
170                 return CommandResult.FUNCTIONAL_FAILURE;
171             }
172             return CommandResult.SUCCESS;
173         } catch (CAExistsException e) {
174             log.error(e.getMessage());
175         } catch (IllegalCryptoTokenException e) {
176             log.error(e.getMessage());
177         } catch (AuthorizationDeniedException e) {
178             log.error(e.getMessage());
179         } catch (CertPathValidatorException e) {
180             log.error(e.getMessage());
181         } catch (EjbcaException e) {
182             log.error(e.getMessage());
183         } catch (CesecoreException e) {
184             log.error(e.getMessage());
185         }
186         return CommandResult.FUNCTIONAL_FAILURE;
187     }
188

190     public String getCommandDescription() {
191         return "Imports a PEM file and creates a new external CA representation from it";
192 
193     }

197         return getCommandDescription();
198     }

202         return log;
203     }
