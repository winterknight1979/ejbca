124     @Override
125     public CommandResult execute(ParameterContainer parameters) {
126         log.trace(">execute()");
127 
128         CryptoProviderTools.installBCProviderIfNotAvailable();
129         String username = parameters.get(ENDENTITY_USERNAME_KEY);
130         String password = parameters.get(ENDENTITY_PASSWORD_KEY);
131         String caname = parameters.get(CA_NAME_KEY);
132         String active = parameters.get(ACTIVE_KEY);
133         String email = parameters.get(E_MAIL_KEY);
134         String certfile = parameters.get(FILE_KEY);
135         String eeprofile = parameters.get(EE_PROFILE_KEY);
136         String certificateprofile = parameters.get(CERT_PROFILE_KEY);
137         final String revocationReasonString = parameters.get(REVOCATION_REASON);
138         final String revocationTimeString = parameters.get(REVOCATION_TIME);
139         EndEntityType endEntityType = EndEntityTypes.ENDUSER.toEndEntityType();
140         StringBuilder errorString = new StringBuilder();
141         int status;
142         RevocationReasons revocationReason = null;
143         Date revocationTime = null;
144         if (ACTIVE.equalsIgnoreCase(active)) {
145             status = CertificateConstants.CERT_ACTIVE;
146             if(revocationReasonString != null) {
147                 log.warn("Revocation reason has been set in spite of certificates being imported as active. Ignoring.");
148             }
149             if(revocationTimeString != null) {
150                 log.warn("Revocation time has been set in spite of certificates being imported as active. Ignoring.");
151             }
152         } else if (REVOKED.equalsIgnoreCase(active)) {
153             status = CertificateConstants.CERT_REVOKED;
154         
155             
156             if(revocationReasonString != null) {
157                 revocationReason = RevocationReasons.getFromCliValue(revocationReasonString.toUpperCase());
158                 if(revocationReason == null) {
159                     log.error("ERROR: " + revocationReasonString + " is not a valid revocation reason.");
160                     return CommandResult.CLI_FAILURE;
161                 }
162             } else {
163                 revocationReason = RevocationReasons.UNSPECIFIED;
164             } 
165             if (revocationTimeString != null) {
166                 try {
167                     revocationTime = new SimpleDateFormat(CaImportCertDirCommand.DATE_FORMAT).parse(revocationTimeString);
168                 } catch (ParseException e) {
169                     log.error("ERROR: " + revocationTimeString + " was not a valid revocation time.");
170                     return CommandResult.CLI_FAILURE;
171                 }
172             } else {
173                 revocationTime = new Date();
174             }
175         } else {
176             errorString.append("Invalid certificate status, must be " + ACTIVE + " or " + REVOKED + "\n");
177             return CommandResult.FUNCTIONAL_FAILURE;
178         }
179 
180         Certificate certificate;
181         try {
182             certificate = loadcert(certfile);
183         } catch (FileNotFoundException e) {
184             log.error("File " + certfile + " was not found.");
185             return CommandResult.FUNCTIONAL_FAILURE;
186         } catch (CertificateException e) {
187             log.error("PEM in file " + certfile + " could not be read.");
188             return CommandResult.FUNCTIONAL_FAILURE;
189         } catch (IOException e) {
190             log.error("File " + certfile + " does not seem to contain a PEM encoded certificate.");
191             return CommandResult.FUNCTIONAL_FAILURE;
192         }
193         final String fingerprint = CertTools.getFingerprintAsString(certificate);
194         if (EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class).findCertificateByFingerprintRemote(fingerprint) != null) {
195             errorString.append("Certificate number '" + CertTools.getSerialNumberAsString(certificate) + "' is already present.\n");
196         }
197         // Certificate has expired, but we are obviously keeping it for archival purposes
198         if (CertTools.getNotAfter(certificate).compareTo(new java.util.Date()) < 0) {
199             status = CertificateConstants.CERT_ARCHIVED;
200         }
201 
202         // Check if username already exists.
203         EndEntityInformation userdata;
204         try {
205             userdata = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityAccessSessionRemote.class).findUser(getAuthenticationToken(), username);
206         } catch (AuthorizationDeniedException e) {
207             log.error("ERROR: CLI user not authorized to manage end entities.");
208             return CommandResult.AUTHORIZATION_FAILURE;
209         }
210         if ((userdata != null) && !parameters.containsKey(OVERRIDE_EXISTING_ENDENTITY)) {
211             if (userdata.getStatus() != EndEntityConstants.STATUS_REVOKED) {
212                 errorString.append("User " + username + " already exists; only revoked user can be overwritten.\n");
213             }
214         }
215 
216         if (StringUtils.isEmpty(email) || StringUtils.equalsIgnoreCase(email, "null")) {
217             email = CertTools.getEMailAddress(certificate);
218         }
219 
220         int endentityprofileid = EndEntityConstants.EMPTY_END_ENTITY_PROFILE;
221         if (eeprofile != null) {
222             log.debug("Searching for End Entity Profile " + eeprofile);
223             try {
224                 endentityprofileid = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfileId(eeprofile);
225             } catch (EndEntityProfileNotFoundException e) {
226                 errorString.append("End Entity Profile '" + eeprofile + "' does not exist.\n");
227             }
228         }
229 
230         int certificateprofileid = CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER;
231         if (certificateprofile != null) {
232             log.debug("Searching for Certificate Profile " + certificateprofile);
233             certificateprofileid = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileId(
234                     certificateprofile);
235             if (certificateprofileid == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
236                 log.error("Certificate Profile " + certificateprofile + " does not exist.");
237                 errorString.append("Certificate Profile '" + certificateprofile + "' does not exist.\n");
238             }
239         }
240 
241         CAInfo cainfo = getCAInfo(getAuthenticationToken(), caname);
242         if (cainfo == null) {
243             log.error("CA with name " + caname + " does not exist.");
244             errorString.append("CA with name '" + caname + "' does not exist.\n");
245         }
246 
247         if (errorString.length() > 0) {
248             log.error(errorString.toString());
249             return CommandResult.FUNCTIONAL_FAILURE;
250         }
251 
252         final Certificate cacert = cainfo.getCertificateChain().iterator().next();
253         log.info("Trying to add user:");
254         log.info("Username: " + username);
255         log.info("Password (hashed only): " + password);
256         log.info("Email: " + email);
257         log.info("DN: " + CertTools.getSubjectDN(certificate));
258         log.info("CA Name: " + caname);
259         log.info("Certificate Profile: "
260                 + EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileName(certificateprofileid));
261         log.info("End Entity Profile: "
262                 + EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfileName(endentityprofileid));
263 
264         String subjectAltName = CertTools.getSubjectAlternativeName(certificate);
265         if (subjectAltName != null) {
266             log.info("SubjectAltName: " + subjectAltName);
267         }
268         log.info("Type: " + endEntityType.getHexValue());
269 
270         log.debug("Loading/updating user " + username);
271         EndEntityManagementSessionRemote endEntityManagementSession = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityManagementSessionRemote.class);
272         try {
273             if (userdata == null) {
274 
275                 try {
276                 endEntityManagementSession.addUser(getAuthenticationToken(), username,
277                         password, CertTools.getSubjectDN(certificate), subjectAltName, email, false, endentityprofileid, certificateprofileid,
278                         endEntityType, SecConst.TOKEN_SOFT_BROWSERGEN, SecConst.NO_HARDTOKENISSUER, cainfo.getCAId());
279                 } catch (EndEntityExistsException e) {
280                     log.error("End entity with username " + username + " already exists.");
281                     return CommandResult.FUNCTIONAL_FAILURE;
282                 } 
283                 try {
284                     if (status == CertificateConstants.CERT_ACTIVE) {
285                         endEntityManagementSession.setUserStatus(getAuthenticationToken(),
286                                 username, EndEntityConstants.STATUS_GENERATED);
287                     } else {
288                         endEntityManagementSession.setUserStatus(getAuthenticationToken(),
289                                 username, EndEntityConstants.STATUS_REVOKED);
290                     }
291                 } catch (NoSuchEndEntityException e) {
292                     throw new IllegalStateException("Newly added end entity could not be located", e);
293                 }
294 
295                 log.info("End Entity '" + username + "' has been added.");
296             } else {
297                 EndEntityInformation endEntityInformation = new EndEntityInformation(username, CertTools.getSubjectDN(certificate), cainfo.getCAId(),
298                         subjectAltName, email, (status == CertificateConstants.CERT_ACTIVE ? EndEntityConstants.STATUS_GENERATED
299                                 : EndEntityConstants.STATUS_REVOKED), endEntityType, endentityprofileid, certificateprofileid, null, null,
300                         SecConst.TOKEN_SOFT_BROWSERGEN, SecConst.NO_HARDTOKENISSUER, null);
301                 endEntityInformation.setPassword(password);
302                 try {
303                     endEntityManagementSession.changeUser(getAuthenticationToken(), endEntityInformation, false);
304                 } catch (NoSuchEndEntityException e) {
305                     log.error("No such end entity.");
306                     return CommandResult.FUNCTIONAL_FAILURE;
307                 }
308 
309                 log.info("User '" + username + "' has been updated.");
310             }
311         } catch (CADoesntExistsException e) {
312             log.error("No such CA " + caname);
313             return CommandResult.FUNCTIONAL_FAILURE;
314         } catch (AuthorizationDeniedException e) {
315             log.error("CLI user not authorized to create end entity.");
316             return CommandResult.FUNCTIONAL_FAILURE;
317         } catch (EndEntityProfileValidationException e) {
318             log.error("User doesn't fulfill End Entity Profile ");
319             return CommandResult.FUNCTIONAL_FAILURE;
320         } catch (WaitingForApprovalException e) {
321             log.error("Approval is required to add End Entity.");
322             return CommandResult.FUNCTIONAL_FAILURE;
323         } catch (CertificateSerialNumberException e) {
324             log.error(e.getMessage());
325             return CommandResult.FUNCTIONAL_FAILURE;
326         } catch (IllegalNameException e) {
327             log.error(e.getMessage());
328             return CommandResult.FUNCTIONAL_FAILURE;
329         } catch (ApprovalException e) {
330             log.error(e.getMessage());
331             return CommandResult.FUNCTIONAL_FAILURE;
332         } catch (CustomFieldException e) {
333             log.error(e.getMessage());
334             return CommandResult.FUNCTIONAL_FAILURE;
335         }
336 
337         int certificateType = CertificateConstants.CERTTYPE_ENDENTITY;
338         try {
339             EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class).storeCertificateRemote(getAuthenticationToken(), EJBTools.wrap(certificate),
340                     username, CertTools.getFingerprintAsString(cacert), CertificateConstants.CERT_ACTIVE, certificateType, certificateprofileid, endentityprofileid, null, new Date().getTime());
341             if (status == CertificateConstants.CERT_REVOKED) {
342                 try {
343                     endEntityManagementSession.revokeCert(getAuthenticationToken(), CertTools.getSerialNumber(certificate), revocationTime, CertTools.getIssuerDN(certificate),
344                             revocationReason.getDatabaseValue(), false);
345                 } catch (ApprovalException e) {
346                     log.error(e.getMessage());
347                     return CommandResult.FUNCTIONAL_FAILURE;
348                 } catch (AlreadyRevokedException e) {
349                     log.error(e.getMessage());
350                     return CommandResult.FUNCTIONAL_FAILURE;
351                 } catch (RevokeBackDateNotAllowedForProfileException e) {
352                     log.error(e.getMessage());
353                     return CommandResult.FUNCTIONAL_FAILURE;
354                 } catch (NoSuchEndEntityException e) {
355                     log.error(e.getMessage());
356                     return CommandResult.FUNCTIONAL_FAILURE;
357                 } catch (WaitingForApprovalException e) {
358                     log.error(e.getMessage());
359                     return CommandResult.FUNCTIONAL_FAILURE;
360                 }
361             }
362         } catch (AuthorizationDeniedException e) {
363             log.error("CLI user not authorized to import certificate.");
364             return CommandResult.FUNCTIONAL_FAILURE;
365         }
366 
367         log.info("Certificate number '" + CertTools.getSerialNumberAsString(certificate) + "' has been added.");
368         return CommandResult.SUCCESS;
369     }

371     /**
372      * 
373      * @param filename path to a file containing a PEM encoded certificate
374      * @return the certificate
375      * 
376      * @throws IOException if the file didn't contain the certificate keys.
377      * @throws CertificateException if the read PEM couldn't be decoded to a certificate
378      * @throws FileNotFoundException if file wasn't found
379      */
380     private Certificate loadcert(String filename) throws IOException, CertificateException {
381         File certfile = new File(filename);
382         if (!certfile.exists()) {
383             throw new FileNotFoundException(filename + " is not a file.");
384         }
385 
386         byte[] bytes = FileTools.getBytesFromPEM(FileTools.readFiletoBuffer(filename), "-----BEGIN CERTIFICATE-----", "-----END CERTIFICATE-----");
387         Certificate cert = CertTools.getCertfromByteArray(bytes, Certificate.class);
388         return cert;
389 
390     }

397     @Override
398     public String getFullHelpText() {
399         StringBuilder sb = new StringBuilder();
400         sb.append(getCommandDescription() + "\n\n");
401         sb.append("If E-mail isn't set (" + E_MAIL_KEY + "), the value from the certificate will be tried.\n\n");
402         String existingCas = "";
403 
404         Collection<Integer> cas = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getAuthorizedCaIds(getAuthenticationToken());
405         try {
406             for (int caid : cas) {
407                 CAInfo info = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caid);
408                 existingCas += (existingCas.length() == 0 ? "" : ", ") + "\"" + info.getName() + "\"";
409             }
410         } catch (AuthorizationDeniedException e) {
411             existingCas = "ERROR: CLI user not authorized to fetch available CAs>";
412         }
413         sb.append("Existing CAs: " + existingCas + "\n");
414         String endEntityProfiles = "";
415         Collection<Integer> eps = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getAuthorizedEndEntityProfileIds(
416                 getAuthenticationToken(), AccessRulesConstants.CREATE_END_ENTITY);
417         for (int epid : eps) {
418             endEntityProfiles += (endEntityProfiles.length() == 0 ? "" : ", ") + "\""
419                     + EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfileName(epid) + "\"";
420         }
421         sb.append("End entity profiles: " + endEntityProfiles + "\n");
422         String certificateProfiles = "";
423         Collection<Integer> cps = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
424                 .getAuthorizedCertificateProfileIds(getAuthenticationToken(), CertificateConstants.CERTTYPE_ENDENTITY);
425         for (int cpid : cps) {
426             certificateProfiles += (certificateProfiles.length() == 0 ? "" : ", ") + "\""
427                     + EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileName(cpid) + "\"";
428         }
429         sb.append("Certificate profiles: " + certificateProfiles + "\n\n");
430         sb.append("If an End entity profile is selected it must allow selected Certificate profiles.\n");
431         sb.append("Valid Revocation reasons: ");
432         RevocationReasons[] values = RevocationReasons.values();
433         for(int i = 0; i < values.length; ++i) {
434             if(!values[i].equals(RevocationReasons.NOT_REVOKED)) {
435                 sb.append(values[i].getStringValue());
436                 if(i < values.length -1) {
437                     sb.append(" | ");
438                 }
439             }
440         }
441         
442         return sb.toString();
443     }
