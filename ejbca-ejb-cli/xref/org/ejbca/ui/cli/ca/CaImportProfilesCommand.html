52  /**
53   * Import profiles from XML-files to the database.
54   *
55   * @version $Id: CaImportProfilesCommand.java 34163 2020-01-02 15:00:17Z samuellb $
56   */
57  public class CaImportProfilesCommand extends BaseCaAdminCommand {
58  
59      private static final Logger log = Logger.getLogger(CaImportProfilesCommand.class);
60  
61      private static final String DIRECTORY_KEY = "-d";
62      private static final String CA_NAME_KEY = "--caname";
63  
64      {
65          registerParameter(new Parameter(DIRECTORY_KEY, "Directory", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
66                  "Directory containing profiles."));
67          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
68                  "Name of a CA to restrict imported profiles to."));
69      }
70  
71      @Override
72      public String getMainCommand() {
73          return "importprofiles";
74      }
75  
76      @Override
77      public CommandResult execute(ParameterContainer parameters) {
78          String inpath = parameters.get(DIRECTORY_KEY);
79          Integer caid = null;
80          String caName = parameters.get(CA_NAME_KEY);
81          if (caName != null) {
82              CAInfo ca;
83              try {
84                  ca = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caName);
85                  if (ca == null) {
86                      getLogger().error("CA '" + caName + "' does not exist.");
87                      return CommandResult.FUNCTIONAL_FAILURE;
88                  }
89              } catch (AuthorizationDeniedException e) {
90                  getLogger().error("CLI user not authorized to CA '" + caName);
91                  return CommandResult.AUTHORIZATION_FAILURE;
92              }
93              caid = ca.getCAId();
94  
95          }
96          CryptoProviderTools.installBCProvider();
97          // Mapping used to translate certificate profile ids when importing end entity profiles. Used when the profile id of a cert profile changes
98          // and we need to change the mapping from the ee profile to cert profiles
99          HashMap<Integer, Integer> certificateProfileIdMapping = new HashMap<>();
100         getLogger().info("Importing certificate and end entity profiles: ");
101         File inFile = new File(inpath);
102         if (!inFile.isDirectory()) {
103             getLogger().error("'" + inpath + "' is not a directory.");
104             return CommandResult.FUNCTIONAL_FAILURE;
105         }
106         // List all filenames in the given directory, we will try to import them all
107         File[] infiles = inFile.listFiles();
108         FileTools.sortByName(infiles);
109         try {
110             for (int i = 0; i < infiles.length; i++) {
111                 getLogger().info("Filename: " + infiles[i].getName());
112                 if (infiles[i].isFile()
113                         && ((infiles[i].getName().indexOf("certprofile_") > -1) || (infiles[i].getName().indexOf("entityprofile_") > -1))) {
114                     boolean entityprofile = false;
115                     if (infiles[i].getName().indexOf("entityprofile_") > -1) {
116                         entityprofile = true;
117                     }
118                     int index1 = infiles[i].getName().indexOf("_");
119                     int index2 = infiles[i].getName().lastIndexOf("-");
120                     int index3 = infiles[i].getName().lastIndexOf(".xml");
121                     if (index1 < 0 || index2 < 0 || index3 < 0) {
122                         getLogger().error("Filename not as expected (cert/entityprofile_<name>-<id>.xml).");
123                     } else {
124                         final String filename = infiles[i].getName().substring(index1 + 1, index2);
125                         String profilename;
126                         try {
127                             profilename = URLDecoder.decode(filename, "UTF-8");
128                         } catch (UnsupportedEncodingException e) {
129                             throw new IllegalStateException("UTF-8 was not a known character encoding", e);
130                         }
131                         int profileid = Integer.parseInt(infiles[i].getName().substring(index2 + 1, index3));
132                         // We don't add the fixed profiles, EJBCA handles those automagically
133                         if (!entityprofile && CertificateProfileConstants.isFixedCertificateProfile(profileid)) {
134                             getLogger().error("Not adding fixed certificate profile '" + profilename + "'.");
135                         } else {
136                             if (entityprofile && profileid == EndEntityConstants.EMPTY_END_ENTITY_PROFILE) {
137                                 getLogger().error("Not adding fixed entity profile '" + profilename + "'.");
138                             } else {
139                                 // Check if the profiles already exist, and change the name and id if already taken
140                                 boolean error = false;
141                                 // when we need to create a new certprofile id, this will hodl the original value so we
142                                 // can insert a mapping in certificateProfileIdMapping when we have created a new id
143                                 int oldprofileid = -1;
144                                 if (entityprofile) {
145                                     if (EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfile(
146                                             profilename) != null) {
147                                         getLogger().error("Entity profile '" + profilename + "' already exist in database.");
148                                         error = true;
149                                     } else if (EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfile(
150                                             profileid) != null) {
151                                         int newprofileid = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class)
152                                                 .findFreeEndEntityProfileId();
153                                         getLogger()
154                                                 .warn("Entity profileid '" + profileid + "' already exist in database. Using " + newprofileid
155                                                         + " instead.");
156                                         profileid = newprofileid;
157                                     }
158                                 } else {
159                                     if (EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileId(
160                                             profilename) != CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
161                                         getLogger().error("Error: Certificate profile '" + profilename + "' already exist in database.");
162                                         error = true;
163                                     } else if (EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
164                                             .getCertificateProfile(profileid) != null) {
165                                         getLogger().warn(
166                                                 "Certificate profile id '" + profileid
167                                                         + "' already exist in database. Adding with a new profile id instead.");
168                                         oldprofileid = profileid;
169                                         profileid = -1; // means we should create a new id when adding the cert profile
170                                     }
171                                 }
172                                 if (!error) {
173                                     CertificateProfile cprofile = null;
174                                     EndEntityProfile eprofile = null;
175                                     final Object loadedObject;
176                                     try (SecureXMLDecoder decoder = new SecureXMLDecoder(new FileInputStream(infiles[i]))) {
177                                         loadedObject = decoder.readObject();
178                                     } catch (FileNotFoundException e) {
179                                         //Shouldn't happen, we've already vetted the file directory above
180                                         throw new IllegalStateException("An exception was encountered with an already vetted file directory", e);
181                                     } catch (IOException e) {
182                                         log.error("Failed to parse profile XML in '" + infiles[i] + "': " + e.getMessage());
183                                         return CommandResult.FUNCTIONAL_FAILURE;
184                                     }
185                                     if (entityprofile) {
186                                         // Add end entity profile
187                                         eprofile = new EndEntityProfile();
188                                         eprofile.loadData(loadedObject);
189                                         // Translate cert profile ids that have changed after import
190                                         final List<Integer> availableCertProfiles = new ArrayList<>();
191                                         Integer defaultCertProfile = eprofile.getDefaultCertificateProfile();
192                                         //getLogger().debug("Debug: Org - AVAILCERTPROFILES " + eprofile.getValue(EndEntityProfile.AVAILCERTPROFILES,0) + " DEFAULTCERTPROFILE "+defaultCertProfile);
193                                         for (int currentCertProfileId : eprofile.getAvailableCertificateProfileIds()) {
194                                             Integer replacementCertProfileId = certificateProfileIdMapping.get(currentCertProfileId);
195                                             if (replacementCertProfileId != null) {
196                                                 if (replacementCertProfileId != currentCertProfileId) {
197                                                     getLogger().warn(
198                                                             "Replacing cert profile with id " + currentCertProfileId + " with "
199                                                                     + replacementCertProfileId + ".");
200                                                 }
201                                                 availableCertProfiles.add(replacementCertProfileId);
202                                                 if (currentCertProfileId == defaultCertProfile) {
203                                                     defaultCertProfile = replacementCertProfileId;
204                                                 }
205                                             } else {
206                                                 if (EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
207                                                         .getCertificateProfile(currentCertProfileId) != null
208                                                         || CertificateProfileConstants.isFixedCertificateProfile(currentCertProfileId)) {
209                                                     availableCertProfiles.add(currentCertProfileId);
210                                                 } else {
211                                                     getLogger().warn(
212                                                             "End Entity Profile '" + profilename + "' references certificate profile "
213                                                                     + currentCertProfileId + " that does not exist.");
214                                                     if (currentCertProfileId == defaultCertProfile) {
215                                                         defaultCertProfile = null;
216                                                     }
217                                                 }
218                                             }
219                                         }
220                                         if (availableCertProfiles.isEmpty()) {
221                                             getLogger()
222                                                     .warn("End Entity Profile only references certificate profile(s) that does not exist. Using ENDUSER profile.");
223                                             availableCertProfiles.add(EndEntityConstants.EMPTY_END_ENTITY_PROFILE); // At least make sure the default profile is available
224                                         }
225                                         if (defaultCertProfile == null) {
226                                             defaultCertProfile = availableCertProfiles.get(0); // Use first available profile from list as default if original default was missing
227                                         }
228                                         eprofile.setAvailableCertificateProfileIds(availableCertProfiles);
229                                         eprofile.setDefaultCertificateProfile(defaultCertProfile);
230                                         // Remove any unknown CA and break if none is left
231                                         Integer defaultCA = eprofile.getDefaultCA();
232                                         final List<Integer> cas = eprofile.getAvailableCAs();
233                                         //getOutputStream().println("Debug: Org - AVAILCAS " + availableCAs + " DEFAULTCA "+defaultCA);
234                                         final List<Integer> availableCAs = new ArrayList<>();
235                                         for (int currentCaId : cas) {
236                                             // The constant ALLCAS will not be searched for among available CAs
237                                             if (currentCaId != SecConst.ALLCAS) {
238                                                 if (!EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).existsCa(currentCaId)) {
239                                                     getLogger().warn("CA with id " + currentCaId
240                                                             + " was not found and will not be used in end entity profile '" + profilename + "'.");
241                                                     if (defaultCA == currentCaId) {
242                                                         defaultCA = null;
243                                                     }
244                                                 } else {
245                                                     availableCAs.add(currentCaId); 
246                                                 }
247                                             } else {
248                                                 availableCAs.add(SecConst.ALLCAS);
249                                             }
250                                            
251 
252                                         }
253                                         if (availableCAs.isEmpty()) {
254                                             if (caid == null) {
255                                                 getLogger().error(
256                                                         "No CAs left in end entity profile '" + profilename
257                                                                 + "' and no CA specified on command line. Using ALLCAs.");
258                                                 availableCAs.add(SecConst.ALLCAS);
259                                             } else {
260                                                 availableCAs.add(caid);
261                                                 getLogger().warn(
262                                                         "No CAs left in end entity profile '" + profilename
263                                                                 + "'. Using CA supplied on command line with id '" + caid + "'.");
264                                             }
265                                         }
266                                         if (defaultCA == null) {
267                                             defaultCA = availableCAs.get(0); // Use first available
268                                             getLogger().warn("Changing default CA in end entity profile '" + profilename + "' to " + defaultCA + ".");
269                                         }
270                                         //getLogger().debug("New - AVAILCAS " + availableCAs + " DEFAULTCA "+defaultCA);
271                                         eprofile.setAvailableCAs(availableCAs);
272                                         eprofile.setDefaultCA(defaultCA);
273                                         try {
274                                             EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).addEndEntityProfile(
275                                                     getAuthenticationToken(), profileid, profilename, eprofile);
276                                             getLogger().info("Added entity profile '" + profilename + "' to database.");
277                                         } catch (EndEntityProfileExistsException eepee) {
278                                             getLogger().error("Error adding entity profile '" + profilename + "' to database.");
279                                         }
280                                     } else {
281                                         // Add certificate profile
282                                         cprofile = new CertificateProfile();
283                                         cprofile.loadData(loadedObject);
284                                         // Make sure CAs in profile exist
285                                         List<Integer> cas = cprofile.getAvailableCAs();
286                                         if (cas == null) {
287                                             cas = new ArrayList<>();
288                                         }
289                                         ArrayList<Integer> casToRemove = new ArrayList<>();
290                                         for (Integer currentCA : cas) {
291                                             // If the CA is not ANYCA and the CA does not exist, remove it from the profile before import
292                                             if (currentCA != CertificateProfile.ANYCA) {
293                                                 if (!EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).existsCa(currentCA)) {
294                                                     casToRemove.add(currentCA);
295                                                 }
296                                             }
297                                         }
298                                         for (Integer toRemove : casToRemove) {
299                                             getLogger().warn(
300                                                     "Warning: CA with id " + toRemove
301                                                             + " was not found and will not be used in certificate profile '" + profilename + "'.");
302                                             cas.remove(toRemove);
303                                         }
304                                         if (cas.size() == 0) {
305                                             if (caid == null) {
306                                                 getLogger().error(
307                                                         "Error: No CAs left in certificate profile '" + profilename
308                                                                 + "' and no CA specified on command line. Using ANYCA.");
309                                                 cas.add(Integer.valueOf(CertificateProfile.ANYCA));
310                                             } else {
311                                                 getLogger().warn(
312                                                         "Warning: No CAs left in certificate profile '" + profilename
313                                                                 + "'. Using CA supplied on command line with id '" + caid + "'.");
314                                                 cas.add(caid);
315                                             }
316                                         }
317                                         cprofile.setAvailableCAs(cas);
318                                         // Remove and warn about unknown publishers
319                                         List<Integer> publishers = cprofile.getPublisherList();
320                                         ArrayList<Integer> allToRemove = new ArrayList<>();
321                                         for (Integer publisher : publishers) {
322                                             BasePublisher pub = null;
323                                             try {
324                                                 pub = EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).getPublisher(publisher);
325                                             } catch (Exception e) {
326                                                 getLogger().warn(
327                                                         "Warning: There was an error loading publisher with id " + publisher
328                                                                 + ". Use debug logging to see stack trace: " + e.getMessage());
329                                                 getLogger().debug("Full stack trace: ", e);
330                                             }
331                                             if (pub == null) {
332                                                 allToRemove.add(publisher);
333                                             }
334                                         }
335                                         for (Integer toRemove : allToRemove) {
336                                             getLogger().warn(
337                                                     "Warning: Publisher with id " + toRemove
338                                                             + " was not found and will not be used in certificate profile '" + profilename + "'.");
339                                             publishers.remove(toRemove);
340                                         }
341                                         cprofile.setPublisherList(publishers);
342                                         // Add profile
343                                         try {
344                                             if (profileid == -1) {
345                                                 // id already existed, we need to create a new one
346                                                 profileid = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
347                                                         .addCertificateProfile(getAuthenticationToken(), profilename, cprofile);
348                                                 // make a mapping from the old id (that was already in use) to the new one so we can change end entity profiles
349                                                 certificateProfileIdMapping.put(oldprofileid, profileid);
350                                             } else {
351                                                 EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
352                                                         .addCertificateProfile(getAuthenticationToken(), profileid, profilename, cprofile);
353                                             }
354                                             // Make a mapping from the new to the new id, so we have a mapping if the profile id did not change at all
355                                             certificateProfileIdMapping.put(profileid,
356                                                     EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
357                                                             .getCertificateProfileId(profilename));
358                                             getLogger().info("Added certificate profile '" + profilename + "', '" + profileid + "' to database.");
359                                         } catch (CertificateProfileExistsException cpee) {
360                                             getLogger().error(
361                                                     "Error adding certificate profile '" + profilename + "', '" + profileid + "' to database.");
362                                         }
363                                     }
364                                 }
365                             }
366                         }
367                     }
368                 }
369             }
370         } catch (AuthorizationDeniedException e) {
371             log.error("Current CLI user doesn't have sufficient privileges to import profiles.");
372             return CommandResult.AUTHORIZATION_FAILURE;
373         }
374         return CommandResult.SUCCESS;
375     }

380
