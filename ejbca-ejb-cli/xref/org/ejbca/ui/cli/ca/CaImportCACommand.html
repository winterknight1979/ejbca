106     @Override
107     public String getMainCommand() {
108         return "importca";
109     }

111     @Override
112     public CommandResult execute(ParameterContainer parameters) {
113         CryptoProviderTools.installBCProvider();
114         String caName = parameters.get(CA_NAME_KEY);
115         boolean importHardToken = parameters.get(HARD_SWITCH_KEY) != null;
116         if (!importHardToken) {
117             // Import soft keystore
118             log.info("Importing soft token.");
119             String kspwd = parameters.get(KEYSTORE_PASSWORD_KEY);
120             if (kspwd == null) {
121                 log.info("Enter keystore password: ");
122                 // Read the password, but mask it so we don't display it on the console
123                 kspwd = String.valueOf(System.console().readPassword());
124             }
125             String p12file = parameters.get(P12_FILE_KEY);
126             if(p12file == null) {
127                 log.error("P12 file needs to be specified for soft keys, use "+P12_FILE_KEY+" switch.");
128                 return CommandResult.CLI_FAILURE;
129             }
130             String alias = parameters.get(SIGNATURE_ALIAS_KEY);
131             String encryptionAlias = parameters.get(ENCRYPTION_ALIAS_KEY);
132             // Read old keystore file in the beginning so we know it's good
133             byte[] keystorebytes = null;
134             try {
135                 keystorebytes = FileTools.readFiletoBuffer(p12file);
136                 // Import CA from PKCS12 file
137                 if (alias == null) {
138                     // First we must find what aliases there is in the pkcs12-file
139                     KeyStore ks;
140                     try {
141                         ks = KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
142                     } catch (KeyStoreException e) {
143                         throw new IllegalStateException("PKCS12 keystore couldn't be found in BouncyCastle provider.");
144                     } catch (NoSuchProviderException e) {
145                         throw new IllegalStateException("BouncyCastle provider couldn't be found.", e);
146                     }
147                     FileInputStream fis = new FileInputStream(p12file);
148                     try {
149                         ks.load(fis, kspwd.toCharArray());
150                     } catch (NoSuchAlgorithmException e) {
151                         log.error("Keystore were created with an unknown algorithm", e);
152                         return CommandResult.FUNCTIONAL_FAILURE;
153                     } catch (CertificateException e) {
154                         log.error("Certificates in keystore could not be loaded for unknown reason");
155                         return CommandResult.FUNCTIONAL_FAILURE;
156                     } catch (IOException e) {
157                         if (e.getCause() instanceof UnrecoverableKeyException) {
158                             log.error("Incorrect password to the PKCS#12 keystore inputed.");
159                             return CommandResult.FUNCTIONAL_FAILURE;
160                         } else {
161                             throw new IllegalStateException("Uknown IOException was caught", e);
162                         }
163                     }
164                     try {
165                         fis.close();
166                     } catch (IOException e) {
167                         throw new IllegalStateException("Uknown IOException was caught", e);
168                     }
169                     Enumeration<String> aliases;
170                     try {
171                         aliases = ks.aliases();
172                     } catch (KeyStoreException e) {
173                         throw new IllegalStateException("Keystore was not initialized", e);
174                     }
175                     int length = 0;
176                     while (aliases.hasMoreElements()) {
177                         alias = (String) aliases.nextElement();
178                         log.info("Keystore contains alias: " + alias);
179                         length++;
180                     }
181                     if (length > 1) {
182                         log.info("Keystore contains more than one alias, alias must be provided as argument.");
183                         return CommandResult.FUNCTIONAL_FAILURE;
184                     } else if (length < 1) {
185                         log.info("Keystore does not contain any aliases. It can not be used for a CA.");
186                         return CommandResult.FUNCTIONAL_FAILURE;
187                     }
188                     // else alias already contains the only alias, so we can use that
189                 }
190             } catch (FileNotFoundException e) {
191                 log.error("File " + p12file + " not found.");
192                 return CommandResult.FUNCTIONAL_FAILURE;
193             }
194             EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class).importCAFromKeyStore(getAuthenticationToken(), caName,
195                     keystorebytes, kspwd, kspwd, alias, encryptionAlias);
196             return CommandResult.SUCCESS;
197         } else {
198             // Import HSM keystore
199             // "Usage2: CA importca <CA name> <catokenclasspath> <catokenpassword> <catokenproperties> <ca-certificate-file>\n" +
200             log.info("Importing hard token.");
201             String tokenclasspath = parameters.get(CA_TOKEN_CLASSPATH_KEY);
202             String tokenpwd = parameters.get(CA_TOKEN_PASSWORD_KEY);
203             String catokenproperties;
204             try {
205                 catokenproperties = new String(FileTools.readFiletoBuffer(parameters.get(CA_TOKEN_PROPERTIES_FILE_KEY)));
206             } catch (FileNotFoundException e) {
207                 log.error("No such file: " + parameters.get(CA_TOKEN_PROPERTIES_FILE_KEY));
208                 return CommandResult.FUNCTIONAL_FAILURE;
209             }
210             Collection<Certificate> cacerts;
211             try {
212                 cacerts = CertTools.getCertsFromPEM(parameters.get(CA_CERTIFICATE_FILE_KEY), Certificate.class);
213             } catch (CertificateException e) {
214                 log.error("File " + parameters.get(CA_CERTIFICATE_FILE_KEY) + " was not a correctly formatted PEM file.");
215                 return CommandResult.FUNCTIONAL_FAILURE;
216             } catch (FileNotFoundException e) {
217                 log.error("No such file: " + parameters.get(CA_CERTIFICATE_FILE_KEY));
218                 return CommandResult.FUNCTIONAL_FAILURE;
219             }
220             Certificate[] cacertarray = cacerts.toArray(new Certificate[cacerts.size()]);
221             try {
222                 EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class).importCAFromHSM(getAuthenticationToken(), caName, cacertarray,
223                         tokenpwd, tokenclasspath, catokenproperties);
224                 return CommandResult.SUCCESS;
225             } catch (CryptoTokenOfflineException e) {
226                 log.error("Crypto Token was offline.");
227             } catch (CryptoTokenAuthenticationFailedException e) {
228                 log.error("Authentication to the crypto token failed.");
229             } catch (IllegalCryptoTokenException e) {
230                 log.error("The certificate chain was incomplete.");
231             } catch (CAExistsException e) {
232                 log.error("CA already exists in database.");
233             } catch (CAOfflineException e) {
234                 log.error("Could not set CA to online and thus unable to publish CRL.");
235             } catch (AuthorizationDeniedException e) {
236                 log.error("Imported CA was signed by a CA that current CLI user does not have authorization to.");
237             } catch (NoSuchSlotException e) {
238                 log.error("Slot defined in: " + parameters.get(CA_TOKEN_PROPERTIES_FILE_KEY) + " does not exist on HSM.");
239             }
240            
241         }
242         return CommandResult.FUNCTIONAL_FAILURE;
243 
244     }

248         return "Imports a keystore and creates a new X509 CA from it.";
249

252     @Override
253     public String getFullHelpText() {
254         return getCommandDescription()
255                 + " This command has two modes: importing a CA from a PKCS#12 keystore (default) or importing from a CA certificate."
256                 + " PKCS#12 keystore is the default option, while CA certificate can be chosen by specifying the flag " + HARD_SWITCH_KEY + "\n"
257                 + "The two usages are: \n" + "<CA name> <pkcs12 file> [" + KEYSTORE_PASSWORD_KEY
258                 + " <password>] [<signature alias>] [<encryption alias>]\n" + "    or:\n" + "<CA name> " + HARD_SWITCH_KEY
259                 + " <catokenclasspath> <catokenpassword> <catokenproperties> <ca-certificate-file>";
260     }
