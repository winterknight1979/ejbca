31  /**
32   * Changes fields in a Certificate Profile.
33   *
34   * @version $Id: CaEditCertificateProfileCommand.java 23457 2016-05-16 17:18:31Z mikekushner $
35   */
36  public class CaEditCertificateProfileCommand extends BaseCaAdminCommand {
37  
38      private static final Logger log = Logger.getLogger(CaEditCertificateProfileCommand.class);
39  
40      private static final String CERTIFICATEPROFILE_NAME_KEY = "--cpname";
41      private static final String FIELD_KEY = "--field";
42      private static final String VALUE_KEY = "--value";
43      private static final String LISTFIELDS_KEY = "-listFields";
44      private static final String GETVALUE_KEY = "-getValue";
45  
46      {
47          registerParameter(new Parameter(CERTIFICATEPROFILE_NAME_KEY, "Certificate Profile Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW,
48                  ParameterMode.ARGUMENT, "Name of the certificate profile."));
49          registerParameter(new Parameter(FIELD_KEY, "Field Name", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
50                  "The sought field."));
51          registerParameter(new Parameter(VALUE_KEY, "Value", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
52                  "The value to set, if any."));
53          registerParameter(new Parameter(LISTFIELDS_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
54                  "Set to only list available fields in the CA"));
55          registerParameter(new Parameter(GETVALUE_KEY, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG,
56                  "Use this to get the value of a single field"));
57      }
58  
59      @Override
60      public String getMainCommand() {
61          return "editcertificateprofile";
62      }
63  
64      @Override
65      public CommandResult execute(ParameterContainer parameters) {
66  
67          FieldEditorieldEditor">FieldEditor fieldEditor = new FieldEditor(log);
68  
69          CryptoProviderTools.installBCProvider();
70  
71          boolean listOnly = parameters.get(LISTFIELDS_KEY) != null;
72          boolean getOnly = parameters.get(GETVALUE_KEY) != null;
73          final String name = parameters.get(CERTIFICATEPROFILE_NAME_KEY);
74          final String field = parameters.get(FIELD_KEY);
75          final String value = parameters.get(VALUE_KEY);
76  
77          if(!listOnly && field == null) {
78              log.error("No field was specified.");
79              return CommandResult.CLI_FAILURE;
80          } else if(!listOnly && !getOnly && value == null) {
81              log.error("No value was specified.");
82              return CommandResult.CLI_FAILURE;
83          } else if(listOnly && getOnly) {
84              log.error("Cannot specify both " + LISTFIELDS_KEY + " and " + GETVALUE_KEY);
85              return CommandResult.CLI_FAILURE;
86          } else if(getOnly && value != null) {
87              log.error("Cannot submit a value with "+ GETVALUE_KEY + " set.");
88              return CommandResult.CLI_FAILURE;
89          }
90          
91          final CertificateProfile profile = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfile(
92                  name);
93          if (profile == null) {
94              log.info("Certificate profile '" + name + "' does not exist.");
95              return CommandResult.FUNCTIONAL_FAILURE;
96          } else {
97              // List fields, get values or set value
98              try {
99                  if(listOnly) {
100                     fieldEditor.listSetMethods(profile);
101                 } else if(getOnly) {
102                     fieldEditor.getBeanValue(field, profile);
103                 }
104                 else {
105                     fieldEditor.setValue(name, field, value, profile);             
106                     log.info("Storing modified profile '" + name + "'...");
107                     EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).changeCertificateProfile(
108                             getAuthenticationToken(), name, profile);
109                     // Verify our new value
110                     log.info("Reading modified value for verification...");
111                     final CertificateProfile modprof = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
112                             .getCertificateProfile(name);
113                     // Print return value
114                     fieldEditor.getBeanValue(field, modprof);
115                 }
116                 return CommandResult.SUCCESS;
117             } catch (FieldNotFoundException e) {
118                 log.error(e.getMessage());
119                 return CommandResult.FUNCTIONAL_FAILURE;
120             } catch (AuthorizationDeniedException e) {
121                 log.error("CLI User was not authorized to Certificate Profile " + name);
122                 return CommandResult.AUTHORIZATION_FAILURE;
123             }
124         }
125 
126     }
127 
128     @Override
129     public String getCommandDescription() {
130         return "Edits profile fields of an existing certificate profile in the CA. ";
131                
132     }
133 
134     @Override
135     public String getFullHelpText() {
136         return getCommandDescription()
137                 + "Fields that can be set are derived from setFieldName of the CertificateProfile java code. If there is a 'setFieldName(type)' method, the values to use in this command should be 'fieldName value'\n"
138                 + "Example: ca editcertificateprofile CertProfileName CRLDistributionPointURI http://my-crl-distp.com/my.crl\n"
139                 + "Example: ca editcertificateprofile CertProfileName caIssuers http://my-ca.issuer.com/ca\n"
140                 + "Example: ca editcertificateprofile CertProfileName useOcspNoCheck true\n"
141                 + "Example: ca editcertificateprofile CertProfileName numOfReqApprovals 1\n" + "\n"
142                 + "Use the option -listFields to only list available fields in the certificate profile.\n"
143                 + "Example: ca editcertificateprofile CertProfileName -listFields\n" + "\n"
144                 + "Use the option -getValue to only get the value of a field in the certificate profile.\n"
145                 + "Example: ca editcertificateprofile CertProfileName -getValue caIssuers";
146     }
147     
148     @Override
149     protected Logger getLogger() {
150         return log;
151     }
