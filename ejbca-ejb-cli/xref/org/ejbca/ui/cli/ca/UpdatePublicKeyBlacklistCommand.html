82      private static final int STATUS_OK = 0;
83      private static final int STATUS_READ_ERROR = 1;
84      private static final int STATUS_REDUNDANT = 2;
85      private static final int STATUS_CONSTRAINTVIOLATION = 3;
86      private static final int STATUS_GENERALIMPORTERROR = 4;

93      //@SuppressWarnings("unchecked")
94      @Override
95      public CommandResult execute(ParameterContainer parameters) {
96          log.trace(">execute()");
97  
98          CryptoProviderTools.installBCProviderIfNotAvailable();
99  
100         try {
101             final String command = parameters.get(COMMAND_KEY);
102             final String keySpecificationsString = parameters.get(KEY_SPECIFICATIONS_KEY);
103             final String importDirString = parameters.get(DIRECTORY_KEY);
104             final boolean byFingerprint = UPDATE_MODE_FINGERPINT.equals(parameters.get(UPDATE_MODE_KEY));
105             final boolean resumeOnError = parameters.containsKey(RESUME_ON_ERROR_KEY);
106 
107             List<String> keySpecifications = new ArrayList<String>(); // Allows any
108             if (StringUtils.isNotBlank(keySpecificationsString)) {
109                 keySpecifications = Arrays.asList(keySpecificationsString.split(CSV_SEPARATOR));
110             }
111 
112             // Get all files in the directory to add/remove to/from public key blacklist. 
113             final File importDir = new File(importDirString);
114             if (!importDir.isDirectory()) {
115                 log.error("'" + importDirString + "' is not a directory.");
116                 return CommandResult.CLI_FAILURE;
117             }
118             final File files[] = importDir.listFiles();
119             if (files == null || files.length < 1) {
120                 log.info("No files in directory '" + importDir.getCanonicalPath() + "'. Nothing to do.");
121                 return CommandResult.SUCCESS; // Nothing to do is OK
122             }
123 
124             // Read public key file (or lists of fingerprint) to add/remove to/from public key blacklist.
125             int redundant = 0;
126             int readError = 0;
127             int constraintViolation = 0;
128             int generalImportError = 0;
129             int importOk = 0;
130             int state;
131             String path;
132             FileReader reader;
133             List<String> lines;
134             PublicKey publicKey;
135             String fingerprint;
136             String keySpecification;
137             byte[] asn1Encodedbytes;
138 
139             for (final File file : files) {
140                 state = STATUS_GENERALIMPORTERROR;
141                 try {
142                     keySpecification = null;
143                     path = file.getAbsolutePath();
144                     log.debug("Read file " + path);
145 
146                     if (COMMAND_ADD.equals(command)) {
147                         if (!byFingerprint) {
148                             log.info("Read public key file " + path);
149                             asn1Encodedbytes = KeyTools.getBytesFromPublicKeyFile(FileTools.readFiletoBuffer(path));
150                             if (null == (publicKey = KeyTools.getPublicKeyFromBytes(asn1Encodedbytes))) {
151                                 state = STATUS_READ_ERROR;
152                             } else {
153                                 state = addPublicKeyToBlacklist(publicKey);
154                             }
155                         } else {
156                             log.info("Read public key fingerprints file " + path);
157                             reader = new FileReader(file);
158                             lines = IOUtils.readLines(reader);
159                             IOUtils.closeQuietly(reader);
160                             String[] tokens;
161                             for (String line : lines) {
162                                 tokens = line.split(CSV_SEPARATOR);
163                                 if (tokens.length > 0) {
164                                     fingerprint = tokens[0];
165                                     keySpecification = null;
166                                     if (CollectionUtils.isNotEmpty(keySpecifications)) {
167                                         keySpecification = keySpecifications.get(0);
168                                     }
169                                     if (tokens.length > 1) {
170                                         keySpecification = tokens[1];
171                                     }
172                                     state = addPublicKeyFingerprintToBlacklist(fingerprint, keySpecification);
173                                     if (STATUS_OK != state) {
174                                         log.info("Update public key blacklist failed on fingerprint: " + fingerprint); 
175                                         break;
176                                     }
177                                 }
178                             }
179                         }
180                     } else if (COMMAND_REMOVE.equals(command)) {
181                         if (!byFingerprint) {
182                             log.info("Remove public key by file " + path);
183                             asn1Encodedbytes = KeyTools.getBytesFromPublicKeyFile(FileTools.readFiletoBuffer(path));
184                             if (null == (publicKey = KeyTools.getPublicKeyFromBytes(asn1Encodedbytes))) {
185                                 state = STATUS_READ_ERROR;
186                             } else {
187                                 state = removePublicKeyToBlacklist(publicKey);
188                             }
189                         } else {
190                             log.info("Remove public keys by fingerprints listed in file " + path);
191                             reader = new FileReader(file);
192                             lines = IOUtils.readLines(reader);
193                             IOUtils.closeQuietly(reader);
194                             String[] tokens;
195                             for (String line : lines) {
196                                 tokens = line.split(CSV_SEPARATOR);
197                                 if (tokens.length > 0) {
198                                     fingerprint = tokens[0];
199                                     log.info("Try to remove public key from public key blacklist (fingerprint=" + fingerprint + ").");
200                                     try {
201                                         state = removeFromBlacklist(PublicKeyBlacklistEntry.TYPE, fingerprint);
202                                     } catch (BlacklistDoesntExistsException e) {
203                                         // Do nothing, it was already printed to info
204                                     }
205                                     if (STATUS_OK != state) {
206                                         log.info("remove public key blacklist failed on fingerprint: " + fingerprint);                                        
207                                     }
208                                 }
209                             }
210                         }
211                     }
212 
213                     switch (state) {
214                     case STATUS_OK:
215                         importOk++;
216                         break;
217                     case STATUS_READ_ERROR:
218                         readError++;
219                         break;
220                     case STATUS_REDUNDANT:
221                         redundant++;
222                         break;
223                     case STATUS_CONSTRAINTVIOLATION:
224                         constraintViolation++;
225                         break;
226                     case STATUS_GENERALIMPORTERROR:
227                         generalImportError++;
228                         break;
229                     default:
230                         generalImportError++;
231                         break;
232                     }
233                     if (!resumeOnError && STATUS_OK != state) {
234                         throw new Exception("Update public key blacklist aborted --resumeonerror=" + resumeOnError);
235                     }
236                 } catch (BlacklistExistsException e) {
237                     log.error("Update public key blacklist failed: " + e.getMessage());
238                     if (!resumeOnError) {
239                         return CommandResult.FUNCTIONAL_FAILURE;
240                     }
241                 } catch (BlacklistDoesntExistsException e) {
242                     log.info("Update public key blacklist failed: " + e.getMessage());
243                     if (!resumeOnError) {
244                         return CommandResult.FUNCTIONAL_FAILURE;
245                     }
246                 } catch (AuthorizationDeniedException e) {
247                     log.info("Not authorized to update blacklist: " + e.getMessage());
248                     if (!resumeOnError) {
249                         return CommandResult.FUNCTIONAL_FAILURE;
250                     }
251                 } catch (Exception e) {
252                     log.info("Update public key blacklist failed: " + e.getMessage(), e);
253                     if (!resumeOnError) {
254                         return CommandResult.FUNCTIONAL_FAILURE;
255                     }
256                 }
257             }
258 
259             printSummary(importOk, readError, redundant, constraintViolation, generalImportError, command);
260         } catch (Exception e) {
261             log.error("Update public key blacklist aborted: " + e.getMessage(), e);
262             return CommandResult.FUNCTIONAL_FAILURE;
263         }
264         log.trace("<execute()");
265         return CommandResult.SUCCESS;
266     }

290     /**
291      * Adds a public key to the public key blacklist.
292      * 
293      * @param publicKey the public key to add.
294      * @return {@link #STATUS_GENERALIMPORTERROR} if error, {@link #STATUS_CONSTRAINTVIOLATION} if already existing or {@link #STATUS_OK} if added.
295      * @throws Exception any exception.
296      */
297     private int addPublicKeyToBlacklist(final PublicKey publicKey) throws Exception {
298         log.trace(">addPublicKeyToBlacklist()");
299         int result = STATUS_GENERALIMPORTERROR;
300         final PublicKeyBlacklistEntry entry = new PublicKeyBlacklistEntry();
301         entry.setFingerprint(publicKey); // sets the fingerprint in proper format from the public key
302         entry.setKeyspec(AlgorithmTools.getKeySpecification(publicKey));
303         log.info("Try to add public key into public key blacklist (fingerprint=" + entry.getFingerprint() + ").");
304         result = addToBlacklist(entry);
305         log.trace("<addPublicKeyToBlacklist()");
306         return result;
307     }
308 
309     /**
310      * Adds a fingerprint to the public key blacklist.
311      * 
312      * @param fingerprint the fingerprint to add, note the special conditions for this fingerprint see {@link PublicKeyBlacklistEntry#setFingerprint(PublicKey)}
313      * @param keySpecification the key specification.
314      * @return {@link #STATUS_GENERALIMPORTERROR} if error, {@link #STATUS_CONSTRAINTVIOLATION} if already existing or {@link #STATUS_OK} if added.
315      * @throws Exception any exception.
316      */
317     private int addPublicKeyFingerprintToBlacklist(final String fingerprint, final String keySpecification) throws Exception {
318         log.trace(">addPublicKeyFingerprintToBlacklist()");
319         int result = STATUS_GENERALIMPORTERROR;
320         final PublicKeyBlacklistEntry entry = new PublicKeyBlacklistEntry();
321         entry.setFingerprint(fingerprint);
322         entry.setKeyspec(keySpecification);
323         log.info("Try to add public key into public key blacklist by fingerprint (fingerprint=" + fingerprint + ").");
324         result = addToBlacklist(entry);
325         log.trace("<addPublicKeyFingerprintToBlacklist()");
326         return result;
327     }

329     /**
330      * Removes a public key from the public key blacklist.
331      * 
332      * @param publicKey the public key to remove.
333      * @return {@link #STATUS_GENERALIMPORTERROR} if error, {@link #STATUS_CONSTRAINTVIOLATION} if already existing or {@link #STATUS_OK} if added.
334      * @throws Exception any exception.
335      */
336     private int removePublicKeyToBlacklist(final PublicKey publicKey) throws Exception {
337         log.trace(">removePublicKeyFromBlacklist()");
338         int result = STATUS_GENERALIMPORTERROR;
339         final String fingerprint = PublicKeyBlacklistEntry.createFingerprint(publicKey);
340         log.info("Try to remove public key from public key blacklist (fingerprint=" + fingerprint + ").");
341         result = removeFromBlacklist(PublicKeyBlacklistEntry.TYPE, fingerprint);
342         log.trace("<removePublicKeyFromBlacklist()");
343         return result;
344     }

346     /**
347      * Adds a public key to the public key blacklist if a public key with that fingerprint does not exists already.
348      * 
349      * @param entry the public key blacklist entry.
350      * @return {@link #STATUS_GENERALIMPORTERROR} if error, {@link #STATUS_CONSTRAINTVIOLATION} if already existing or {@link #STATUS_OK} if added.
351      * @throws Exception any exception.
352      */
353     private int addToBlacklist(final PublicKeyBlacklistEntry entry) throws Exception {
354         log.trace(">addToBlacklist()");
355         int result = STATUS_GENERALIMPORTERROR;
356         final BlacklistSessionRemote blacklistSession = EjbRemoteHelper.INSTANCE.getRemoteSession(BlacklistSessionRemote.class);
357         try {
358             blacklistSession.addBlacklistEntry(getAuthenticationToken(), entry);
359             result = STATUS_OK;
360         } catch (BlacklistExistsException e) {
361             result = STATUS_CONSTRAINTVIOLATION;
362             log.info("Public key blacklist entry with public key fingerprint " + entry.getFingerprint() + " already exists.");
363             throw e;
364         } catch (AuthorizationDeniedException e) {
365             result = STATUS_GENERALIMPORTERROR;
366             log.info("Authorization denied to add public key to blacklist.");
367             throw e;
368         } catch (Exception e) {
369             result = STATUS_GENERALIMPORTERROR;
370             log.info("Error while adding public key to blacklist: " + e.getMessage());
371             throw e;
372         }
373         log.trace("<addToBlacklist()");
374         return result;
375     }
376 
377     /**
378      * Removes a public key from the public key blacklist.
379      * @param type Type
380      * @param value Value
381      * 
382      * @return {@link #STATUS_GENERALIMPORTERROR} if error, {@link #STATUS_CONSTRAINTVIOLATION} if already existing or {@link #STATUS_OK} if added.
383      * @throws Exception any exception.
384      */
385     private int removeFromBlacklist(final String type, final String value) throws Exception {
386         log.trace(">removeFromBlacklist()");
387         int result = STATUS_GENERALIMPORTERROR;
388         final BlacklistSessionRemote blacklistSession = EjbRemoteHelper.INSTANCE.getRemoteSession(BlacklistSessionRemote.class);
389         try {
390             blacklistSession.removeBlacklistEntry(getAuthenticationToken(), type, value);
391             result = STATUS_OK;
392         } catch (BlacklistDoesntExistsException e) {
393             result = STATUS_CONSTRAINTVIOLATION;
394             log.info("Public key blacklist entry with public key fingerprint " + value + " does not exist.");
395             throw e;
396         } catch (AuthorizationDeniedException e) {
397             result = STATUS_GENERALIMPORTERROR;
398             log.info("Authorization denied to remove public key from blacklist.");
399             throw e;
400         } catch (Exception e) {
401             result = STATUS_GENERALIMPORTERROR;
402             log.info("Error while removing public key from blacklist: " + e.getMessage());
403             throw e;
404         }
405         log.trace("<removeFromBlacklist()");
406         return result;
407     }

409     /**
410      * Logs the summary to STDOUT.
411      * 
412      * @param importOk OK counter
413      * @param readError read error counter
414      * @param redundant redundant counter
415      * @param constraintViolation constraint violation counter
416      * @param generalImportError general import error counter
417      * @param command command
418      */
419     private final void printSummary(final int importOk, final int readError, final int redundant, final int constraintViolation,
420             final int generalImportError, final String command) {
421         // Print resulting statistics
422         log.info("\n"+command+" summary:");
423         log.info(importOk + " public key blacklist entries were processed with success (STATUS_OK)");
424         if (readError > 0) {
425             log.info(readError + " public key blacklist entries could not be parsed (STATUS_READERROR)");
426         }
427         if (redundant > 0) {
428             log.info(redundant + " public key blacklist entries were already present in the database (STATUS_REDUNDANT)");
429         }
430         if (constraintViolation > 0) {
431             log.info(constraintViolation + " public key blacklist entries could not be stored (STATUS_CONSTRAINTVIOLATION)");
432         }
433         if (generalImportError > 0) {
434             log.info(generalImportError + " public key blacklist entries were not imported due to other errors (STATUS_GENERALIMPORTERROR)");
435         }
436     }
