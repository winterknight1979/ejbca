35  /**
36   * Changes fields in a CA.
37   *
38   * @version $Id: CaEditCaCommand.java 23457 2016-05-16 17:18:31Z mikekushner $
39   */
40  public class CaEditCaCommand extends BaseCaAdminCommand {
41  
42      private static final Logger log = Logger.getLogger(CaEditCaCommand.class);
43  
44      private static final String CA_NAME_KEY = "--caname";
45      private static final String FIELD_KEY = "--field";
46      private static final String VALUE_KEY = "--value";
47  
48      {
49          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
50                  "The name of the CA to edit."));
51          registerParameter(new Parameter(FIELD_KEY, "Field Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "The sought field."));
53          registerParameter(new Parameter(VALUE_KEY, "Value", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "The value to set, if any."));
55      }
56  
57      @Override
58      public String getMainCommand() {
59          return "editca";
60      }
61  
62      @Override
63      public CommandResult execute(ParameterContainer parameters) {
64          FieldEditorieldEditor">FieldEditor fieldEditor = new FieldEditor(log);
65          CryptoProviderTools.installBCProviderIfNotAvailable();;
66          final String name = parameters.get(CA_NAME_KEY);
67          final String field = parameters.get(FIELD_KEY);
68          final String value = parameters.get(VALUE_KEY);
69          try {
70              final CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), name);
71              // List fields, get values or set value
72              try {
73                  if(field.equals(CA.NAME)) {
74                      // The CA name field is a bit of a special case. Since there's a CESeCore method specifically for it, we should use
75                      // it instead
76                      try {
77                          EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).renameCA(getAuthenticationToken(), cainfo.getName(), value);
78                          log.info("Renamed CA by the name  '" + cainfo.getName() + "' to '" + value + "'");
79                      } catch (CAExistsException e) {
80                          log.error("A CA by the name of " + value + " already exists.");
81                          return CommandResult.FUNCTIONAL_FAILURE;
82                      }
83                  } else {                
84                      fieldEditor.setValue(name, field, value, cainfo);                
85                      log.info("Storing modified CA info for CA '" + name + "'...");
86                      EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class).editCA(getAuthenticationToken(), cainfo);
87                      // Verify our new value.
88                      // If the CA Subject DN was changed, then the CA Id might have changed at this point,
89                      // so we have to do the lookup by name!
90                      log.info("Reading modified value for verification...");
91                      final CAInfo cainfomod = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(),
92                              name);
93                      // Print return value
94                      fieldEditor.getBeanValue(field, cainfomod);                    
95                  }
96                  return CommandResult.SUCCESS;
97              } catch (FieldNotFoundException e) {
98                  log.error(e.getMessage());
99                  return CommandResult.FUNCTIONAL_FAILURE;
100             }
101         } catch (CADoesntExistsException e) {
102             log.info("CA '" + name + "' does not exist.");
103             return CommandResult.FUNCTIONAL_FAILURE;
104         } catch (AuthorizationDeniedException e) {
105             log.error("CLI User was not authorized to CA " + name);
106             return CommandResult.AUTHORIZATION_FAILURE;
107         }
108 
109     }
110 
111     @Override
112     public String getCommandDescription() {
113         return "Edits CA fields of an existing CA.";
114                
115     }
116 
117     @Override
118     public String getFullHelpText() {
119         return getCommandDescription()
120                 + "Fields that can be set are derived from setFieldName of the CA java code. If there is a 'setFieldName(type)' method, the values to use in this command should be 'fieldName value'\n"
121                 + "Example: ca editca CAName CRLPeriod 2592000000\n"
122                 + "Example: ca editca CAName CRLIssueInterval 100000\n"
123                 + "Example: ca editca CAName includeInHealthCheck false\n";
124     }
125     
126     @Override
127     protected Logger getLogger() {
128         return log;
129     }
130 }
