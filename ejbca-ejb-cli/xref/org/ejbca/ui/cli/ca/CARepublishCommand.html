48  /**
49   * Re-publishes the certificates of all users belonging to a particular CA.
50   * 
51   * @version $Id: CARepublishCommand.java 27740 2018-01-05 07:24:53Z mikekushner $
52   */
53  public class CARepublishCommand extends BaseCaAdminCommand {
54  
55      private static final Logger log = Logger.getLogger(CARepublishCommand.class);
56  
57      private static final String CA_NAME_KEY = "--caname";
58      private static final String ALL_KEY = "-all";
59      private static final String CACERT_KEY = "-cacert";
60      private static final String CACRL_KEY = "-cacrl";
61      private static final String EECERT_KEY = "-eecert";
62  
63      {
64          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
65                  "Name of the CA"));
66          registerParameter(Parameter.createFlag(ALL_KEY,
67                  "Publish all certificates for each end entity instead of only the latest (default only publishes the latest)."));
68          registerParameter(Parameter.createFlag(CACERT_KEY, "Publish CA certificate."));
69          registerParameter(Parameter.createFlag(CACRL_KEY, "Publish CA CRL."));
70          registerParameter(Parameter.createFlag(EECERT_KEY, "Publish End Entity certificates."));
71      }
72  
73      @Override
74      public String getMainCommand() {
75          return "republish";
76      }
77  
78      @Override
79      public CommandResult execute(ParameterContainer parameters) {
80          boolean addAll = parameters.containsKey(ALL_KEY);
81          boolean cacertmode = parameters.containsKey(CACERT_KEY);
82          boolean cacrlmode = parameters.containsKey(CACRL_KEY);
83          boolean eecertmode = parameters.containsKey(EECERT_KEY);
84          String caname = parameters.get(CA_NAME_KEY);
85  
86          try {
87              // Get the CAs info and id
88              CAInfo cainfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caname);
89              if (cainfo == null) {
90                  getLogger().info("CA with name '" + caname + "' does not exist.");
91                  return CommandResult.FUNCTIONAL_FAILURE;
92              }
93              // If no mode is give we will enable all modes (backwards compatibility for when there were no modes)
94              if (!cacertmode && !cacrlmode && !eecertmode) {
95                  cacertmode = cacrlmode = eecertmode = true;
96              }
97              getLogger().info("Publishing with modes: cacert=" + cacertmode + ", cacrl=" + cacrlmode + ", eecert=" + eecertmode);
98              // Publish the CAs certificate and CRL
99              Collection<Certificate> cachain = cainfo.getCertificateChain();
100             Iterator<Certificate> caiter = cachain.iterator();
101             if (caiter.hasNext()) {
102                 final X509Certificate cacert = (X509Certificate) caiter.next();
103                 final byte[] crlbytes = EjbRemoteHelper.INSTANCE.getRemoteSession(CrlStoreSessionRemote.class).getLastCRL(cainfo.getSubjectDN(),
104                         false);
105                 // Get the CRLnumber
106                 X509CRL crl;
107                 try {
108                     crl = CertTools.getCRLfromByteArray(crlbytes);
109                 } catch (CRLException e) {
110                     throw new IllegalStateException("Couldn't deserialize CRL", e);
111                 }
112                 int crlNumber = CrlExtensions.getCrlNumber(crl).intValue();
113                 final Collection<Integer> capublishers = cainfo.getCRLPublishers();
114                 // Store cert and CRL in ca publishers.
115                 if (capublishers != null) {
116                     String fingerprint = CertTools.getFingerprintAsString(cacert);
117                     if (cacertmode) {
118                         getLogger().info("Publishing CA certificate to CA publishers.");
119                         EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).storeCertificate(getAuthenticationToken(),
120                                 capublishers, fingerprint, null, cainfo.getSubjectDN(), null);
121                         getLogger().info("Certificate published for " + caname);
122                     }
123                     if (cacrlmode) {
124                         if (crlbytes != null && crlbytes.length > 0 && crlNumber > 0) {
125                             getLogger().info("Publishing CRL to CA publishers.");
126                             EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).storeCRL(getAuthenticationToken(), capublishers,
127                                     crlbytes, fingerprint, crlNumber, cainfo.getSubjectDN());
128                             getLogger().info("CRL with number " + crlNumber + " published for " + caname);
129                         } else {
130                             getLogger().info("CRL not published, no CRL exists for CA.");
131                         }
132                     }
133                 } else {
134                     getLogger().info("No publishers configured for the CA, no CA certificate or CRL published.");
135                 }
136             } else {
137                 getLogger().info("CA does not have a certificate, no certificate or CRL published!");
138             }
139 
140             if (eecertmode) {
141                 // Get all users for this CA
142                 Collection<EndEntityInformation> coll = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityAccessSessionRemote.class)
143                         .findAllUsersByCaId(getAuthenticationToken(), cainfo.getCAId());
144                 Iterator<EndEntityInformation> iter = coll.iterator();
145                 while (iter.hasNext()) {
146                     EndEntityInformation data = iter.next();
147                     getLogger().info(
148                             "User: " + data.getUsername() + ", \"" + data.getDN() + "\", \"" + data.getSubjectAltName() + "\", " + data.getEmail()
149                                     + ", " + data.getStatus() + ", " + data.getType().getHexValue() + ", " + data.getTokenType() + ", "
150                                     + data.getHardTokenIssuerId() + ", " + data.getCertificateProfileId());
151 
152                     if (data.getCertificateProfileId() > 0) { // only if we find a
153                         // certificate profile
154                         CertificateProfile certProfile = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
155                                 .getCertificateProfile(data.getCertificateProfileId());
156                         if (certProfile == null) {
157                             getLogger().error("Can not get certificate profile with id: " + data.getCertificateProfileId());
158                             continue;
159                         }
160                         // Get an ordered list of certificates, last expire date first
161                         List<Certificate> certCol = EJBTools.unwrapCertCollection(EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class)
162                                 .findCertificatesByUsername(data.getUsername()));
163                         if ((certCol != null) && certCol.iterator().hasNext()) {
164                             if (certProfile.getPublisherList() != null) {
165                                 getLogger().info(
166                                         "Re-publishing user " + data.getUsername() + " to publishers in users certificate profile "
167                                                 + data.getCertificateProfileId());
168                                 if (addAll) {
169                                     getLogger().info("Re-publishing all certificates (" + certCol.size() + ").");
170                                     // Reverse the collection so we publish the latest certificate last
171                                     Collections.reverse(certCol); // now the latest (last expire date) certificate is last in the List
172                                     Iterator<Certificate> i = certCol.iterator();
173                                     while (i.hasNext()) {
174                                         X509Certificate c = (X509Certificate) i.next();
175                                         publishCert(getAuthenticationToken(), data, certProfile, c);
176                                     }
177                                 } else {
178                                     // Only publish the latest one (last expire date)
179                                     // The latest one is the first in the List according to findCertificatesByUsername()
180                                     publishCert(getAuthenticationToken(), data, certProfile, (X509Certificate) certCol.iterator().next());
181                                 }
182                             } else {
183                                 getLogger().info(
184                                         "Not publishing certificate for user " + data.getUsername() + ", no publisher in certificate profile.");
185                             }
186                         } else {
187                             getLogger().info("No certificate to publish for user " + data.getUsername());
188                         }
189                     } else {
190                         getLogger().info("No certificate profile is set for user " + data.getUsername());
191                     }
192                 }
193                
194             } // if (eecertmode)
195             return CommandResult.SUCCESS;
196         } catch (AuthorizationDeniedException e) {
197             log.error("CLI user was not authorized to CA " + caname);
198             return CommandResult.AUTHORIZATION_FAILURE;
199         }
200 
201     }
202 
203     private void publishCert(AuthenticationToken authenticationToken, EndEntityInformation data, CertificateProfile certProfile, X509Certificate cert) {
204         try {
205             String fingerprint = CertTools.getFingerprintAsString(cert);
206             final String userDataDN = data.getCertificateDN();
207             boolean ret = EjbRemoteHelper.INSTANCE.getRemoteSession(PublisherSessionRemote.class).storeCertificate(authenticationToken,
208                     certProfile.getPublisherList(), fingerprint, data.getPassword(), userDataDN, null);
209             if (!ret) {
210                 getLogger().error(
211                         "Failed to publish certificate for user " + data.getUsername() + ", continuing with next user. Publish returned false.");
212             }
213         } catch (Exception e) {
214             // catch failure to publish one user and continue with the rest
215             getLogger().error("Failed to publish certificate for user " + data.getUsername() + ", continuing with next user. " + e.getMessage());
216         }
217     }
218 
219     @Override
220     public String getCommandDescription() {
221         return "Re-publishes the certificates of a CA and/or all users issued by a particular CA. ";
222     }
223 
224     @Override
225     public String getFullHelpText() {
226         return getCommandDescription()
227                 + "Default if none of cacert, cacrl or eecert is specific is to publish all types, you can specify one or several on the command line. "
228                 + "For example to only publish CA certificate and CRL, no end entity certificates: ca republish ManagementCA -cacert -cacrl"
229                 + "Example to only publish CA certificate and CRL, and latest end entity certificates: ca republish ManagementCA";
230     }

232     @Override
233     protected Logger getLogger() {
234         return log;
235     }
