17  import java.io.FileOutputStream;
18  import java.io.IOException;

20  import org.apache.commons.lang.StringUtils;
21  import org.apache.log4j.Logger;
22  import org.cesecore.util.EjbRemoteHelper;
23  import org.ejbca.core.ejb.ca.caadmin.CAAdminSessionRemote;
24  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
25  import org.ejbca.ui.cli.infrastructure.parameter.Parameter;
26  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
27  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
28  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
29  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;

31  /**
32   * Exports CA as a PKCS#12 or PKCS#8 file
33   *
34   * @version $Id: CaExportCACommand.java 19902 2014-09-30 14:32:24Z anatom $
35   */
36  public class CaExportCACommand extends BaseCaAdminCommand {
37  
38      private static final Logger log = Logger.getLogger(CaExportCACommand.class);
39  
40      private static final String CA_NAME_KEY = "--caname";
41      private static final String FILE_KEY = "-f";
42      private static final String KEYSTORE_PASSWORD_KEY = "-kspassword";
43      private static final String SIGNATURE_ALIAS_KEY = "--signalias";
44      private static final String ENCRYPTION_ALIAS_KEY = "--encryptalias";
45      private static final String SIGNATURE_ALIAS_DEFAULT = "SignatureKeyAlias";
46      private static final String ENCRYPTION_ALIAS_DEFAULT = "EncryptionKeyAlias";
47  
48      {
49          registerParameter(new Parameter(CA_NAME_KEY, "CA Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
50                  "The name of the CA to export."));
51          registerParameter(new Parameter(FILE_KEY, "File name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "The destination file."));
53          registerParameter(new Parameter(KEYSTORE_PASSWORD_KEY, "Password", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
54                  "They keystore password. If not set then it will be prompted for."));
55          registerParameter(new Parameter(SIGNATURE_ALIAS_KEY, "Signature Key Alias", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW,
56                  ParameterMode.ARGUMENT, "The signature key alias. Will default to " + SIGNATURE_ALIAS_DEFAULT + " if not set."));
57          registerParameter(new Parameter(ENCRYPTION_ALIAS_KEY, "Encryption Key Alias", MandatoryMode.OPTIONAL, StandaloneMode.ALLOW,
58                  ParameterMode.ARGUMENT, "The encryption key alias. Will default to " + ENCRYPTION_ALIAS_DEFAULT + " if not set. "));
59      }
60  
61      @Override
62      public String getMainCommand() {
63          return "exportca";
64      }
65  
66      @Override
67      public CommandResult execute(ParameterContainer parameters) {
68  
69          String kspwd = parameters.get(KEYSTORE_PASSWORD_KEY);
70          String caName = parameters.get(CA_NAME_KEY);
71          String p12file = parameters.get(FILE_KEY);
72  
73          String signatureKeyAlias = parameters.get(SIGNATURE_ALIAS_KEY);
74          String encryptionKeyAlias = parameters.get(ENCRYPTION_ALIAS_KEY);
75          if ((signatureKeyAlias == null && encryptionKeyAlias != null && parameters.isStandalone(ENCRYPTION_ALIAS_KEY))
76                  || (signatureKeyAlias != null && encryptionKeyAlias == null && parameters.isStandalone(SIGNATURE_ALIAS_KEY))) {
77              //only one of the values was set and implicitly, kinda scary. Let's warn about that. 
78              log.error("Do not set only one of SignatureKeyAlias or EncryptionKeyAlias implicitely (without a switch).");
79              return CommandResult.FUNCTIONAL_FAILURE;
80          }
81          if (signatureKeyAlias == null) {
82              log.info("Setting SignatureKeyAlias to " + SIGNATURE_ALIAS_DEFAULT);
83              signatureKeyAlias = SIGNATURE_ALIAS_DEFAULT;
84          }
85          if (encryptionKeyAlias == null) {
86              log.info("Setting EncryptionKeyAlias to " + ENCRYPTION_ALIAS_DEFAULT);
87              encryptionKeyAlias = ENCRYPTION_ALIAS_DEFAULT;
88          }
89  
90          if (kspwd == null) {
91              log.info("Enter keystore password: ");
92              // Read the password, but mask it so we don't display it on the console
93              kspwd = String.valueOf(System.console().readPassword());
94          } else {
95              log.info("Keystore password was supplied on the command line.");
96          }
97          if (StringUtils.isEmpty(kspwd)) {
98              // Can not export CA keystore with empty password. 
99              log.error("Export a token without password protection is not allowed.");
100             return CommandResult.FUNCTIONAL_FAILURE;
101         }
102         byte[] keyStoreBytes = EjbRemoteHelper.INSTANCE.getRemoteSession(CAAdminSessionRemote.class).exportCAKeyStore(getAuthenticationToken(),
103                 caName, kspwd, kspwd, signatureKeyAlias, encryptionKeyAlias);
104         try {            
105             FileOutputStream fos = new FileOutputStream(p12file);
106             fos.write(keyStoreBytes);
107             fos.close();
108         } catch (FileNotFoundException e) {
109             log.error(e.getMessage());
110             return CommandResult.FUNCTIONAL_FAILURE;
111         } catch (IOException e) {
112             throw new IllegalStateException("Could not write to file for unknown reason", e);
113         } 
114         return CommandResult.SUCCESS;
115 
116     }
117 
118     @Override
119     public String getCommandDescription() {
120         return "Exports CA as a PKCS#12 or PKCS#8 file. ";
121 
122     }
123 
124     @Override
125     public String getFullHelpText() {
126         return getCommandDescription()
127                 + "X.509 CAs are exported as PKCS#12 files while for CVC CAs only the private certificate signing key is exported as a PKCS#8 key. "
128                 + "You will be prompted for keystore password to protect stored keystore, "
129                 + "but can optionally specify it on command line using the optional argument '-kspassword <password>'.\n\n"
130                 + "Do not set only one of SignatureKeyAlias or EncryptionKeyAlias implicitely (without a switch). Either set both, or if only doing one then with a switch. "
131                 + "Setting only one without a switch may lead to undefined behavior.";
132     }
133     
134     @Override
135     protected Logger getLogger() {
136         return log;
137     }
138 }
