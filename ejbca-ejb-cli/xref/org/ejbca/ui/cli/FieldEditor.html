13  package org.ejbca.ui.cli;
14  
15  import java.util.ArrayList;
16  import java.util.List;
17  
18  import org.apache.commons.beanutils.ConvertingWrapDynaBean;
19  import org.apache.commons.beanutils.DynaBean;
20  import org.apache.commons.beanutils.DynaProperty;
21  import org.apache.commons.beanutils.WrapDynaBean;
22  import org.apache.commons.lang.math.NumberUtils;
23  import org.apache.log4j.Logger;
24  
25  /**
26   *
27   * @version $Id: FieldEditor.java 23457 2016-05-16 17:18:31Z mikekushner $
28   */
29  public class FieldEditor {
30  
31      private final Logger logger;
32  
33      public FieldEditor(Logger logger) {
34          this.logger = logger;
35      }
36      
37      /** List of fields that should be excluded from bean lists. This should contain fields that should never be changed by a user */
38      private static final ArrayList<String> excluded = new ArrayList<String>();
39      static {
40          excluded.add("type");
41          excluded.add("version");
42          excluded.add("class");
43          excluded.add("latestVersion");
44          excluded.add("upgraded");
45          excluded.add("CAId");
46          excluded.add("CAType");
47          excluded.add("CAToken");
48          excluded.add("hidden");
49      }
50      /** Lists methods in a class the has "setXyz", and prints them as "Xyz".
51       * Ignores (does not list) type, version, latestVersion, upgrade and class
52       * 
53       * @param obj the Object where to look for setMethods
54       */
55      public void listSetMethods(final Object obj) {
56          DynaBean wrapper = new WrapDynaBean(obj);
57          DynaProperty[] props = wrapper.getDynaClass().getDynaProperties();
58          for (DynaProperty dynaProperty : props) {
59              if (!excluded.contains(dynaProperty.getName())) {
60                  logger.info(dynaProperty.getName()+", "+dynaProperty.getType());                
61              }
62          }
63      }
64      
65      public List<String> getSetMethodNames(final Object obj) {
66          List<String> result = new ArrayList<String>();
67          DynaBean wrapper = new WrapDynaBean(obj);
68          DynaProperty[] props = wrapper.getDynaClass().getDynaProperties();
69          for (DynaProperty dynaProperty : props) {
70              if (!excluded.contains(dynaProperty.getName())) {
71                  result.add(dynaProperty.getName());
72              }
73          }
74          return result;
75      }
76      
77      /** gets a field value from a bean
78       * 
79       * @param field the field to get
80       * @param obj the bran to get the value from
81       * @return the value
82       * @throws FieldNotFoundException if field doesn't exist.
83       */
84      public Object getBeanValue(final String field, final Object obj) throws FieldNotFoundException {
85          final DynaBean moddb = new WrapDynaBean(obj);
86          DynaProperty prop = moddb.getDynaClass().getDynaProperty(field);
87          if (prop == null) {
88              throw new FieldNotFoundException("Field '" + field + "' does not exist. Did you use correct case for every character of the field?");
89          }
90          final Object gotValue = moddb.get(field);
91          logger.info(field+" returned value '"+gotValue+"'.");
92          return gotValue;
93      }
94      
95      /** Lists, Gets or sets fields in a Bean.
96       * 
97       * @param name the name of the Bean to be modified
98       * @param field the field name to get or set
99       * @param value the value to set, of we should set a new value
100      * @param obj the Bean to list, get or set fields
101      * 
102      * @throws FieldNotFoundException if field was not found. 
103      */
104     public void setValue(final String name, final String field, final String value, final Object obj) throws FieldNotFoundException {
105 
106         Object val = value;
107         logger.info("Modifying '" + name + "'...");
108         final ConvertingWrapDynaBean db = new ConvertingWrapDynaBean(obj);
109         DynaProperty prop = db.getDynaClass().getDynaProperty(field);
110         if (prop == null) {
111             throw new FieldNotFoundException("Field '" + field + "' does not exist. Did you use correct case for every character of the field?");
112         }
113         if (prop.getType().isInterface()) {
114             logger.info("Converting value '" + value + "' to type '" + ArrayList.class + "', ");
115             // If the value can be converted into an integer, we will use an ArrayList<Integer>
116             // Our problem here is that the type of a collection (<Integer>, <String>) is only compile time, it can not be determined in runtime.
117             List<Object> arr = new ArrayList<Object>();
118 
119             if (NumberUtils.isNumber(value)) {
120                 logger.info("using Integer value.");
121                 arr.add(Integer.valueOf(value));
122             } else {
123                 // Make it into an array of String
124                 logger.info("using String value.");
125                 arr.add(value);
126             }
127             val = arr;
128         }
129         final Object gotValue = db.get(field);
130         logger.info("Current value of " + field + " is '" + gotValue + "'.");
131         db.set(field, val);
132     }
