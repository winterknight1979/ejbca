144     @Override
145     public CommandResult execute(ParameterContainer parameters) {
146         final String username = parameters.get(USERNAME_KEY);
147         final String password = getAuthenticationCode(parameters.get(PASSWORD_KEY));
148         final String dn = parameters.get(DN_KEY);
149         final String caname = parameters.get(CA_NAME_KEY);
150         final String subjectaltname = parameters.get(SUBJECT_ALT_NAME_KEY);
151         final String email = parameters.get(EMAIL_KEY);
152         final EndEntityType type;
153         final String tokenString = parameters.get(TYPE_KEY);
154         try {
155 
156             type = new EndEntityType(EndEntityTypes.getTypesFromHexCode(Integer.parseInt(tokenString)));
157         } catch (NumberFormatException e) {
158             log.error("ERROR: Invalid type: " + tokenString);
159             return CommandResult.FUNCTIONAL_FAILURE;
160         }
161         String tokenname = parameters.get(TOKEN_KEY);
162 
163         boolean error = false;
164         boolean usehardtokenissuer = false;
165 
166         int caid = 0;
167         try {
168             CAInfo caInfo = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getCAInfo(getAuthenticationToken(), caname);
169             if (caInfo != null) {
170                 // let it be 0 if not found, we will print a suitable error message below
171                 caid = caInfo.getCAId();
172             }
173         } catch (AuthorizationDeniedException e) {
174             log.error("CLI user not authorized to CA " + caname);
175             error = true;
176         }
177         int certificatetypeid = CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER;
178         final String certificateProfile = parameters.get(CERT_PROFILE_KEY);
179         if (certificateProfile != null) {
180             // Use certificate type, no end entity profile.
181             certificatetypeid = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileId(
182                     certificateProfile);
183         }
184         getLogger().info(
185                 "Using certificate profile: "
186                         + EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getCertificateProfileName(
187                                 certificatetypeid) + ", with id: " + certificatetypeid);
188 
189         final String endEntityProfile = parameters.get(EE_PROFILE_KEY);
190         int endEntityProfileId = EndEntityConstants.EMPTY_END_ENTITY_PROFILE;
191         if (endEntityProfile != null) {
192             try {
193                 endEntityProfileId = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getEndEntityProfileId(
194                         endEntityProfile);
195             } catch (EndEntityProfileNotFoundException e) {
196                 getLogger().error("ERROR: Could not find end entity profile in database.");
197                 error = true;
198             }
199             getLogger().info("Using entity profile: " + endEntityProfile + ", with id: " + endEntityProfileId);
200         }
201 
202         int hardtokenissuerid = SecConst.NO_HARDTOKENISSUER;
203         if (usehardtokens) {
204             final String hardTokenIssuer = parameters.get(HARDTOKEN_ISSUER_KEY);
205             if (hardTokenIssuer != null) {
206                 // Use certificate type, end entity profile and hardtokenissuer.
207                 hardtokenissuerid = EjbRemoteHelper.INSTANCE.getRemoteSession(HardTokenSessionRemote.class).getHardTokenIssuerId(hardTokenIssuer);
208                 usehardtokenissuer = true;
209                 getLogger().info("Using hard token issuer: " + hardTokenIssuer + ", with id: " + hardtokenissuerid);
210             }
211         }
212 
213         int tokenid = getTokenId(tokenname, usehardtokens, EjbRemoteHelper.INSTANCE.getRemoteSession(HardTokenSessionRemote.class));
214         if (tokenid == 0) {
215             getLogger().error("Invalid token id.");
216             error = true;
217         }
218 
219         if (certificatetypeid == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
220             // Certificate profile not found in database.
221             getLogger().error("Could not find certificate profile in database.");
222             error = true;
223         }
224 
225         if (caid == 0) { // CA not found i database.
226             getLogger().error("Could not find CA '" + caname + "' in database.");
227             error = true;
228         }
229 
230         if (usehardtokenissuer && hardtokenissuerid == SecConst.NO_HARDTOKENISSUER) {
231             getLogger().error("Could not find hard token issuer in database.");
232             error = true;
233         }
234 
235         if ((tokenid > SecConst.TOKEN_SOFT) && (hardtokenissuerid == SecConst.NO_HARDTOKENISSUER)) {
236             getLogger().error("HardTokenIssuer has to be choosen when user with hard tokens is added.");
237             error = true;
238         }
239 
240         if (email == null && type.contains(EndEntityTypes.SENDNOTIFICATION)) {
241             getLogger().error("Email field cannot be null when send notification type is given.");
242             error = true;
243         }
244 
245         // Check if username already exists.
246         if (EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityManagementSessionRemote.class).existsUser(username)) {
247             getLogger().error("ERROR: User '" + username + "' already exists in the database.");
248             error = true;
249         }
250 
251         if (!error) {
252             getLogger().info("Trying to add end entity:");
253             getLogger().info("Username: " + username);
254             getLogger().info("Password: <password hidden>");
255             getLogger().info("DN: " + dn);
256             getLogger().info("CA Name: " + caname);
257             getLogger().info("SubjectAltName: " + subjectaltname);
258             getLogger().info("Email: " + email);
259             getLogger().info("Type: " + type.getHexValue());
260             getLogger().info("Token: " + tokenname);
261             getLogger().info("Certificate profile: " + certificatetypeid);
262             getLogger().info("End entity profile: " + endEntityProfileId);
263             try {
264                 EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityManagementSessionRemote.class).addUser(getAuthenticationToken(), username,
265                         password, dn, subjectaltname, email, false, endEntityProfileId, certificatetypeid, type, tokenid, hardtokenissuerid, caid);
266                 getLogger().info("User '" + username + "' has been added.");
267                 getLogger().info("Note: If batch processing should be possible, also use 'ra setclearpwd " + username + " <pwd>'.");
268                 return CommandResult.SUCCESS;
269             } catch (AuthorizationDeniedException e) {
270                 getLogger().error(e.getMessage());
271                 return CommandResult.AUTHORIZATION_FAILURE;
272             } catch (EndEntityProfileValidationException e) {
273                 getLogger().error("Given userdata doesn't fulfill end entity profile. : " + e.getMessage());
274             } catch (WaitingForApprovalException e) {
275                 getLogger().error("\nOperation pending, waiting for approval: " + e.getMessage());
276             } catch (ApprovalException e) {
277                 getLogger().error("\nApproval exception: " + e.getMessage());
278             }  catch (EndEntityExistsException e) {
279                 log.error("ERROR: End entity already exists.");
280             } catch (CADoesntExistsException e) {
281                 throw new IllegalStateException("Should not happen, CA has already been checked.", e);
282             } catch (CustomFieldException | IllegalNameException | CertificateSerialNumberException e) {
283                 getLogger().error(e.getMessage());
284             }
285         }
286         return CommandResult.FUNCTIONAL_FAILURE;
287 
288     }

290     /**
291      * Returns the tokenid type of the user, returns 0 if invalid tokenname.
292      * @param tokenname Name
293      * @param usehardtokens bool
294      * @param hardtokensession Session
295      * @return ID
296      */
297     private int getTokenId(String tokenname, boolean usehardtokens, HardTokenSessionRemote hardtokensession) {
298         int returnval = 0;
299         // First check for soft token type
300         for (int i = 0; i < SOFT_TOKEN_NAMES.length; i++) {
301             if (SOFT_TOKEN_NAMES[i].equals(tokenname)) {
302                 returnval = SOFT_TOKEN_IDS[i];
303                 break;
304             }
305         }
306         if (returnval == 0 && usehardtokens) {
307             returnval = hardtokensession.getHardTokenProfileId(tokenname);
308         }
309         return returnval;
310     }

317     @Override
318     public String getFullHelpText() {
319         StringBuilder sb = new StringBuilder();
320         sb.append(getCommandDescription()).append("\n");
321         StringBuilder existingCas = new StringBuilder();
322         //The below require quite a few database operations, so shouldn't be run unless the help text has been called upon.
323         for (String caName : EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class).getActiveCANames(getAuthenticationToken())) {
324             existingCas.append((existingCas.length() == 0 ? "" : ", ") + caName);
325         }
326         sb.append("Existing cas: " + existingCas + "\n");
327 
328         Map<Integer, String> certificateprofileidtonamemap = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class)
329                 .getCertificateProfileIdToNameMap();
330         StringBuilder existingCps = new StringBuilder();
331         for (Integer id : EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateProfileSessionRemote.class).getAuthorizedCertificateProfileIds(
332                 getAuthenticationToken(), CertificateConstants.CERTTYPE_ENDENTITY)) {
333             existingCps.append((existingCps.length() == 0 ? "" : ", ") + certificateprofileidtonamemap.get(id));
334         }
335         sb.append("Existing certificate profiles: " + existingCps + "\n");
336 
337         StringBuilder hardTokenString = new StringBuilder();
338         HashMap<Integer, String> hardtokenprofileidtonamemap = EjbRemoteHelper.INSTANCE.getRemoteSession(HardTokenSessionRemote.class)
339                 .getHardTokenProfileIdToNameMap();
340         for (Integer id : EjbRemoteHelper.INSTANCE.getRemoteSession(HardTokenSessionRemote.class).getAuthorizedHardTokenProfileIds(
341                 getAuthenticationToken())) {
342             hardTokenString.append((hardTokenString.length() == 0 ? "" : ", ") + hardtokenprofileidtonamemap.get(id));
343         }
344 
345         sb.append("Existing tokens: " + USERGENERATED + ", " + P12 + ", " + JKS + ", " + PEM + ", " + hardTokenString + "\n");
346 
347         StringBuilder existingEeps = new StringBuilder();
348         Map<Integer, String> endentityprofileidtonamemap = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class)
349                 .getEndEntityProfileIdToNameMap();
350         for (Integer id : EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityProfileSessionRemote.class).getAuthorizedEndEntityProfileIds(
351                 getAuthenticationToken(), AccessRulesConstants.CREATE_END_ENTITY)) {
352             existingEeps.append((existingEeps.length() == 0 ? "" : ", ") + endentityprofileidtonamemap.get(id));
353         }
354         sb.append("Existing endentity profiles: " + existingEeps + "\n");
355 
356         if (usehardtokens) {
357             StringBuilder existingHtis = new StringBuilder();
358             for (String alias : EjbRemoteHelper.INSTANCE.getRemoteSession(HardTokenSessionRemote.class)
359                     .getHardTokenIssuerAliases(getAuthenticationToken()).toArray(new String[0])) {
360                 existingHtis.append((existingHtis.length() == 0 ? "" : ", ") + alias);
361             }
362             sb.append("Existing hard token issuers: " + existingHtis + "\n");
363         }
364 
365         return sb.toString();
366     }

373     private String getAuthenticationCode(final String commandLineArgument) {
374 
375         final String authenticationCode;
376         if (commandLineArgument == null || "null".equalsIgnoreCase(commandLineArgument)) {
377             getLogger().info("Enter password: ");
378             getLogger().info("");
379             authenticationCode = StringTools.passwordDecryption(String.valueOf(System.console().readPassword()), "End Entity Password");
380         } else {
381             authenticationCode = StringTools.passwordDecryption(commandLineArgument, "End Entity Password");
382         }
383         return authenticationCode;
384     }
