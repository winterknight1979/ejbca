40  /**
41   * Set status to key recovery for an end entity's certificate.
42   *
43   * @version $Id: KeyRecoveryCommand.java 26210 2017-08-03 10:12:32Z samuellb $
44   */
45  public class KeyRecoveryCommand extends BaseRaCommand {
46  
47      private static final Logger log = Logger.getLogger(KeyRecoveryCommand.class);
48  
49      private static final String SERIAL_NUMBER_KEY = "-s";
50      private static final String DN_KEY = "--dn";
51  
52      {
53          registerParameter(new Parameter(SERIAL_NUMBER_KEY, "Serial Number", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "The certificate serial number in HEX."));
55          registerParameter(new Parameter(DN_KEY, "Issuer DN", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT, "Issuer DN"));
56      }
57  
58      @Override
59      public String getMainCommand() {
60          return "keyrecover";
61      }
62  
63      @Override
64      public CommandResult execute(ParameterContainer parameters) {
65  
66          BigInteger certificatesn = new BigInteger(parameters.get(SERIAL_NUMBER_KEY), 16);
67          String issuerdn = parameters.get(DN_KEY);
68          boolean usekeyrecovery = ((GlobalConfiguration) EjbRemoteHelper.INSTANCE.getRemoteSession(GlobalConfigurationSessionRemote.class)
69                  .getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID)).getEnableKeyRecovery();
70          if (!usekeyrecovery) {
71              getLogger().error("Keyrecovery have to be enabled in the system configuration in order to use this command.");
72              return CommandResult.FUNCTIONAL_FAILURE;
73          }
74          X509Certificate cert = (X509Certificate) EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class)
75                  .findCertificateByIssuerAndSerno(issuerdn, certificatesn);
76          if (cert == null) {
77              getLogger().error("Certificate couldn't be found in database.");
78              return CommandResult.FUNCTIONAL_FAILURE;
79          }
80          String username = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class).findUsernameByCertSerno(certificatesn,
81                  issuerdn);
82          if (!EjbRemoteHelper.INSTANCE.getRemoteSession(KeyRecoverySessionRemote.class).existsKeys(EJBTools.wrap(cert))) {
83              getLogger().error("Specified keys don't exist in database.");
84              return CommandResult.FUNCTIONAL_FAILURE;
85          }
86          if (EjbRemoteHelper.INSTANCE.getRemoteSession(KeyRecoverySessionRemote.class).isUserMarked(username)) {
87              getLogger().error("User is already marked for recovery.");
88              return CommandResult.FUNCTIONAL_FAILURE;
89          }
90          try {
91              EndEntityInformation userdata = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityAccessSessionRemote.class).findUser(
92                      getAuthenticationToken(), username);
93              if (userdata == null) {
94                  getLogger().error("ERROR: The user " + username + " defined by certificate serial number doesn't exist.");
95                  return CommandResult.FUNCTIONAL_FAILURE;
96              }
97              if (EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityManagementSessionRemote.class).prepareForKeyRecovery(getAuthenticationToken(),
98                      userdata.getUsername(), userdata.getEndEntityProfileId(), cert)) {
99                  getLogger().info("Key pair corresponding to the specified certificate has been marked for recovery.");
100                 return CommandResult.SUCCESS;
101             } else {
102                 getLogger().error("Failed to mark keys corresponding to given certificate for recovery.");
103             }
104         } catch (AuthorizationDeniedException e) {
105             log.error("ERROR: CLI use not authorized to perform key recovery on user " + username);
106         } catch (ApprovalException e) {
107             log.error("ERROR: " + e.getMessage());
108         } catch (WaitingForApprovalException e) {
109             log.error("ERROR: " + e.getMessage());
110         } catch (CADoesntExistsException e) {
111             log.error("ERROR: " + e.getMessage());
112         }
113         return CommandResult.FUNCTIONAL_FAILURE;
114 
115     }
116 
117     @Override
118     public String getCommandDescription() {
119         return "Set status to key recovery for an end entity's certificate";
120     }
121 
122     @Override
123     public String getFullHelpText() {
124         return getCommandDescription();
125     }
126 
127     @Override
128     protected Logger getLogger() {
129         return log;
130     }
131 }
