38  /**
39   * Set the SubjectDN for an end entity.
40   *
41   * @version $Id: SetSubjectDNCommand.java 25012 2017-01-16 09:36:04Z mikekushner $
42   */
43  public class SetSubjectDNCommand extends BaseRaCommand {
44  
45      private static final Logger log = Logger.getLogger(SetSubjectDNCommand.class);
46  
47      private static final String USERNAME_KEY = "--username";
48      private static final String DN_KEY = "--dn";
49  
50      {
51          registerParameter(new Parameter(USERNAME_KEY, "Username", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
52                  "Username of the end entity to modify."));
53          registerParameter(new Parameter(DN_KEY, "Subject DN", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "The new subject DN."));
55      }
56  
57      @Override
58      public String getMainCommand() {
59          return "setsubjectdn";
60      }
61  
62      @Override
63      public CommandResult execute(ParameterContainer parameters) {
64  
65          String username = parameters.get(USERNAME_KEY);
66          String subjectDN = parameters.get(DN_KEY);
67          if (StringUtils.isEmpty(subjectDN)) {
68              getLogger().error("SubjectDN must be supplied.");
69              return CommandResult.FUNCTIONAL_FAILURE;
70          }
71          getLogger().info("Setting subjectDN '" + subjectDN + "' for end entity with username " + username);
72          try {
73              EndEntityInformation uservo = EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityAccessSessionRemote.class)
74                      .findUser(getAuthenticationToken(), username);
75              uservo.setDN(subjectDN);
76              EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityManagementSessionRemote.class).changeUser(getAuthenticationToken(), uservo, false);
77              return CommandResult.SUCCESS;
78          } catch (AuthorizationDeniedException e) {
79              getLogger().error("Not authorized to change end entity.");
80          } catch (EndEntityProfileValidationException e) {
81              getLogger().error("Given end entity doesn't fulfill end entity profile. : " + e.getMessage());
82          } catch (CADoesntExistsException | WaitingForApprovalException | ApprovalException | CertificateSerialNumberException | IllegalNameException
83                  | CustomFieldException e) {
84              getLogger().error("ERROR: " + e.getMessage());
85          } catch (NoSuchEndEntityException e) {
86              getLogger().error("No such end entity.");
87              return CommandResult.FUNCTIONAL_FAILURE;
88          }
89          return CommandResult.FUNCTIONAL_FAILURE;
90      }
91  
92      @Override
93      public String getCommandDescription() {
94          return "Set or update the SubjectDN for an end entity";
95      }
96  
97      @Override
98      public String getFullHelpText() {
99          return getCommandDescription()
100                 + "\n\nThe SubjectDN fields are: emailAddress=<string>, UID=< unique identifier>, CN=<common name>, serialNumber=<serialnumber>, "
101                 + "givenName=<string>, initials=<string>, surname=<string>, title=<string>, "
102                 + "OU=<the organizational unit>, O=<the organization>, L=<locality>, ST=<state of province>, "
103                 + "DC=<domain component>, C=<two letter ISO3166>, unstructuredAddress=<IP address>, "
104                 + "unstructuredName=<domain name>, postalCode=<string>, businessCategory=<organization type>, "
105                 + "dnQualifier=<string>, postalAddress=<the postal address>, telephoneNumber=<telephone number>, "
106                 + "pseudonym=<string>, streetAddress=<string>, name=<string>, CIF=<tax ID code for companies in Spain>, "
107                 + "NIF=<tax ID number for companied in Spain>";
108     }
109 
110     protected Logger getLogger() {
111         return log;
112     }
113 
114 }
