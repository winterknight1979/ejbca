14  package org.ejbca.ui.cli.ra;
15  
16  import java.util.HashSet;
17  import java.util.Set;
18  
19  import org.apache.log4j.Logger;
20  import org.cesecore.certificates.endentity.EndEntityConstants;
21  import org.cesecore.certificates.endentity.EndEntityInformation;
22  import org.cesecore.util.EjbRemoteHelper;
23  import org.ejbca.core.ejb.ra.EndEntityAccessSessionRemote;
24  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
25  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
26  
27  /**
28   * List end entities with status NEW in the database.
29   *
30   * @version $Id: ListNewEndEntitiesCommand.java 27715 2018-01-02 16:55:19Z mikekushner $
31   *
32   */
33  public class ListNewEndEntitiesCommand extends BaseRaCommand {
34  
35      private static final Logger log = Logger.getLogger(ListNewEndEntitiesCommand.class);
36  
37      private static final String COMMAND = "listnewendentities";
38      private static final String OLD_COMMAND = "listnewusers";
39  
40      private static final Set<String> ALIASES = new HashSet<String>();
41      static {
42          ALIASES.add(OLD_COMMAND);
43      }
44  
45      @Override
46      public String getMainCommand() {
47          return COMMAND;
48      }
49  
50      @Override
51      public CommandResult execute(ParameterContainer parameters) {
52          for (EndEntityInformation data : EjbRemoteHelper.INSTANCE.getRemoteSession(EndEntityAccessSessionRemote.class).findAllUsersByStatus(
53                  getAuthenticationToken(), EndEntityConstants.STATUS_NEW)) {
54              getLogger().info(
55                      "New end entity: " + data.getUsername() + ", \"" + data.getDN() + "\", \"" + data.getSubjectAltName() + "\", " + data.getEmail()
56                              + ", " + data.getStatus() + ", " + data.getType().getHexValue() + ", " + data.getTokenType());
57          }
58          return CommandResult.SUCCESS;
59  
60      }
61  
62      @Override
63      public String getCommandDescription() {
64          return "List end entities with status 'NEW'";
65      }
66  
67      @Override
68      public String getFullHelpText() {
69          return getCommandDescription();
70      }
71  
72      protected Logger getLogger() {
73          return log;
74      }
75  }
