19  import java.io.IOException;
20  import java.util.Date;
21  import java.util.HashSet;
22  import java.util.Properties;
23  import java.util.Set;
24  
25  import org.apache.log4j.Logger;

27  import org.cesecore.certificates.endentity.EndEntityType;
28  import org.cesecore.certificates.endentity.EndEntityTypes;

36  import org.ejbca.util.PrinterManager;
37  /**
38   *
39   *
40   * Class used as a help tool when creating hard token visual layout templates
41   * @version $Id: SVGTemplatePrinterCommand.java 19902 2014-09-30 14:32:24Z anatom $
42   */
43  public class SVGTemplatePrinterCommand extends EjbcaCommandBase {
44  
45      private static final Logger log = Logger.getLogger(SVGTemplatePrinterCommand.class);
46  
47      private static final String USERDATAFILENAME = "src/cli/svgtemplateprinttester.properties";
48      private static final String TEMPLATEFILENAME_KEYWORD = "-t";
49      private static final String PRINTERNAME_KEYWORD = "-p";
50  
51      //Register parameters
52      {
53          registerParameter(new Parameter(TEMPLATEFILENAME_KEYWORD, "Template filename", MandatoryMode.MANDATORY, StandaloneMode.ALLOW,
54                  ParameterMode.ARGUMENT, "The template file name."));
55          registerParameter(new Parameter(PRINTERNAME_KEYWORD, "Printer name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
56                  "The name of the printer."));
57      }
58  
59      @Override
60      public String[] getCommandPath() {
61          return new String[] { "svgtemplate" };
62      }
63  
64      @Override
65      public Set<String[]> getCommandPathAliases() {
66          Set<String[]> aliases = new HashSet<String[]>();
67          aliases.add(new String[] { "template" });
68          return aliases;
69      }
70  
71      @Override
72      public String getMainCommand() {
73          return "print";
74      }
75  
76      @Override
77      public CommandResult execute(ParameterContainer parameters) {
78          String templatefilename = parameters.get(TEMPLATEFILENAME_KEYWORD);
79          String printername = parameters.get(PRINTERNAME_KEYWORD);
80          Properties data = new Properties();
81          try {
82              try {
83                  data.load(new FileInputStream(USERDATAFILENAME));
84              } catch (FileNotFoundException e) {
85                  log.error("File " + USERDATAFILENAME + " does not exist, command cannot run.");
86                  return CommandResult.FUNCTIONAL_FAILURE;
87              }
88              EndEntityInformation userdata = new EndEntityInformation("", data.getProperty("DN"), 0, "", data.getProperty("EMAIL"), 0,
89                      new EndEntityType(EndEntityTypes.INVALID), 0, 0, (Date) null, (Date) null, 0, 0, null);
90              String[] pins = new String[2];
91              String[] puks = new String[2];
92              pins[0] = data.getProperty("PIN1");
93              pins[1] = data.getProperty("PIN2");
94              puks[0] = data.getProperty("PUK1");
95              puks[1] = data.getProperty("PUK2");
96              String copyofhardtokensn = data.getProperty("COPYOFHARDTOKENSN");
97              String hardtokensn = data.getProperty("HARDTOKENSN");
98              int validity = Integer.parseInt(data.getProperty("VALIDITY"));
99              String hardtokensnprefix = data.getProperty("HARDTOKENSNPREFIX");
100             FileInputStream fis;
101             try {
102                 fis = new FileInputStream(templatefilename);
103             } catch (FileNotFoundException e) {
104                 log.error("File " + templatefilename + " could not be found, command cannot run.");
105                 return CommandResult.FUNCTIONAL_FAILURE;
106             }
107             try {
108                 byte[] byteData = new byte[fis.available()];
109                 fis.read(byteData);
110                 String sVGData = new String(byteData, "UTF8");
111                 try {
112                     PrinterManager.print(printername, templatefilename, sVGData, 1, validity, userdata, pins, puks, hardtokensnprefix, hardtokensn,
113                             copyofhardtokensn);
114                 } catch (PrinterException e) {
115                     log.error("Printing failed, see associated stack trace.", e);
116                 }
117             } finally {
118                 fis.close();
119             }
120             return CommandResult.SUCCESS;
121         } catch (IOException e) {
122             throw new IllegalStateException("IOException was caught, see underlying error.", e);
123         }
124 
125     }
126 
127     @Override
128     public String getCommandDescription() {
129         return "Tool for creating hard token visual layout templates";
130     }
131 
132     @Override
133     public String getFullHelpText() {
134         return "Tool for creating hard token visual layout templates. User data is configured in  " + USERDATAFILENAME;
135     }
136     
137     @Override
138     protected Logger getLogger() {
139         return log;
140     }
141
