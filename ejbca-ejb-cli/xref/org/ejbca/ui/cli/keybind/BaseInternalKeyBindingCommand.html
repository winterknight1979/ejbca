33  /**
34   * @version $Id: BaseInternalKeyBindingCommand.java 23648 2016-06-09 15:19:55Z mikekushner $
35   *
36   */
37  public abstract class BaseInternalKeyBindingCommand extends EjbcaCliUserCommandBase {
38  
39      protected static final String KEYBINDING_NAME_KEY = "--name";
40  
41      {
42          registerParameter(new Parameter(KEYBINDING_NAME_KEY, "Internal Keybinding Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW,
43                  ParameterMode.ARGUMENT, "Name of the key binding."));
44      }
45      
46      @Override
47      public String[] getCommandPath() {
48          return new String[] { "keybind" };
49      }
50  
51      protected Map<String, Serializable> validateProperties(String type, Map<String, String> dataMap) {
52          InternalKeyBindingPropertyValidationWrapper validatedProperties = InternalKeyBindingFactory.INSTANCE.validateProperties(type, dataMap);
53          if (!validatedProperties.arePropertiesValid()) {
54              StringBuffer stringBuffer = new StringBuffer();
55              stringBuffer.append('\n');
56              stringBuffer.append("ERROR: Could not parse properties\n");
57              stringBuffer.append('\n');
58              if (validatedProperties.getUnknownProperties().size() > 0) {
59                  stringBuffer.append("The following properties were unknown for the type: " + type + "\n");
60                  for (String propertyName : validatedProperties.getUnknownProperties()) {
61                      stringBuffer.append("    * '" + propertyName + "'\n");
62                  }
63                  stringBuffer.append('\n');
64              }
65              if (validatedProperties.getInvalidValues().size() > 0) {
66                  stringBuffer.append("The following values were invalid:\n");
67                  for (Entry<String, Class<?>> entry : validatedProperties.getInvalidValues().entrySet()) {
68                      stringBuffer.append("Value '" + dataMap.get(entry.getKey()) + "' for property '" + entry.getKey() + "' was not of type "
69                              + entry.getValue().getSimpleName() + "\n");
70                  }
71                  stringBuffer.append('\n');
72              }
73              getLogger().error(stringBuffer);
74              return null;
75          }
76          return validatedProperties.getPropertiesCopy();
77      }
78      
79      /** Lists available types and their properties 
80       * @return Props */
81      protected String showTypesProperties() {
82          final InternalKeyBindingMgmtSessionRemote internalKeyBindingMgmtSession = EjbRemoteHelper.INSTANCE
83                  .getRemoteSession(InternalKeyBindingMgmtSessionRemote.class);
84          Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> typesAndProperties = internalKeyBindingMgmtSession
85                  .getAvailableTypesAndProperties();
86          final StringBuilder sb = new StringBuilder();
87          sb.append("Registered implementation types and implemention specific properties:\n");
88          for (Entry<String, Map<String, DynamicUiProperty<? extends Serializable>>> entry : typesAndProperties.entrySet()) {
89              sb.append(' ').append(entry.getKey()).append(":\n");
90              for (DynamicUiProperty<? extends Serializable> property : entry.getValue().values()) {
91                  sb.append("    " + property.getName()).append(",\n");
92              }
93              if (sb.charAt(sb.length() - 2) == ',') {
94                  sb.deleteCharAt(sb.length() - 2);
95              }
96              sb.append('\n');
97          }
98          return sb.toString();
99      }
100 
101     protected String showStatuses() {
102         final StringBuilder sb = new StringBuilder("Status is one of ");
103         for (InternalKeyBindingStatus internalKeyBindingStatus : InternalKeyBindingStatus.values()) {
104             sb.append(internalKeyBindingStatus.name()).append(',');
105         }
106         sb.deleteCharAt(sb.length() - 1);
107         return  sb.append('\n').toString();
108     }
109 
110     protected String showSigAlgs() {
111         final StringBuilder sbAlg = new StringBuilder("Signature algorithm is one of ");
112         for (final String algorithm : AlgorithmConstants.AVAILABLE_SIGALGS) {
113             if (AlgorithmTools.isSigAlgEnabled(algorithm)) {
114                 sbAlg.append(algorithm).append(',');
115             }
116         }
117         return sbAlg.deleteCharAt(sbAlg.length() - 1).append('\n').toString();
118     }
