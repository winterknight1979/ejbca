13  package org.ejbca.ui.cli.keybind;
14  
15  import org.cesecore.authentication.tokens.AuthenticationToken;
16  import org.cesecore.authorization.AuthorizationDeniedException;
17  import org.cesecore.keybind.InternalKeyBindingMgmtSessionRemote;
18  import org.cesecore.util.EjbRemoteHelper;
19  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
20  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
21  
22  /**
23   * RUD (Read, Update, Delete) class for the InternalKeyBinding API access.
24   * 
25   * @version $Id: RudInternalKeyBindingCommand.java 25678 2017-04-11 12:29:50Z mikekushner $
26   */
27  public abstract class RudInternalKeyBindingCommand extends BaseInternalKeyBindingCommand {
28  
29      /**
30       * Overridable InternalKeyBinding-specific execution methods that will parse and interpret the first parameter
31       * (when present) as the name of a InternalKeyBinding and lookup its InternalKeyBindingId.
32       * @param internalKeyBindingId ID
33       * @param parameters Params
34       * @return Result
35       * @throws AuthorizationDeniedException Fail
36       * @throws Exception Fail
37       */
38      public abstract CommandResult executeCommand(Integer internalKeyBindingId, ParameterContainer parameters) throws AuthorizationDeniedException, Exception;
39  
40      @Override
41      public CommandResult execute(ParameterContainer parameters) {
42          Integer internalKeyBindingId = null;
43          String internalKeyBindingName = parameters.get(KEYBINDING_NAME_KEY);
44          if (internalKeyBindingName!=null) {
45              internalKeyBindingName = internalKeyBindingName.trim();
46          }
47          internalKeyBindingId = EjbRemoteHelper.INSTANCE.getRemoteSession(InternalKeyBindingMgmtSessionRemote.class).getIdFromName(
48                  internalKeyBindingName);
49          if (internalKeyBindingId == null) {
50              getLogger().info("Unknown InternalKeyBinding: " + internalKeyBindingName);
51              return CommandResult.FUNCTIONAL_FAILURE;
52          }
53  
54          try {
55              return executeCommand(internalKeyBindingId, parameters);
56          } catch (AuthorizationDeniedException e) {
57              getLogger().info(e.getMessage());
58              return CommandResult.AUTHORIZATION_FAILURE;
59          } catch (RuntimeException e) {
60              getLogger().info("Operation failed: " + e.getMessage());
61              throw new IllegalStateException(e);
62          } catch(Exception e) {
63              getLogger().info("Failure during operation: " + e.getMessage());
64              return CommandResult.FUNCTIONAL_FAILURE;
65          }
66      }
67  
68      /** @return the EJB CLI admin */
69      protected AuthenticationToken getAdmin() {
70          return getAuthenticationToken();
71      }
72  
73  }
