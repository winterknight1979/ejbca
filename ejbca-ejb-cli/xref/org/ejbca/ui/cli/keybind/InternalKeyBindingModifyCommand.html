64                  MandatoryMode.OPTIONAL,

66                  ParameterMode.ARGUMENT,

71                  "TrustEntry",
72                  MandatoryMode.OPTIONAL,
73                  StandaloneMode.FORBID,
74                  ParameterMode.ARGUMENT,
75                  "Removes trust entries to the given keybinding. Trust entries can be of the form <CAName[;CertificateSerialNumber]> where the serialnumber is in hex and optional. "
76                          + "Multiple trust entries can be added by separating them with a ',' i.e. <CA1[;CertificateSerialNumber],CA2[;CertificateSerialNumber]>"));

83                      //Different properties could contain the same keyword. Simply use the same one.

88                  registerParameter(parameter);

98      @Override
99      public CommandResult executeCommand(Integer internalKeyBindingId, ParameterContainer parameters) throws AuthorizationDeniedException, Exception {
100         final InternalKeyBindingMgmtSessionRemote internalKeyBindingMgmtSession = EjbRemoteHelper.INSTANCE
101                 .getRemoteSession(InternalKeyBindingMgmtSessionRemote.class);
102         final CaSessionRemote caSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class);
103         final InternalKeyBinding internalKeyBinding = internalKeyBindingMgmtSession
104                 .getInternalKeyBinding(getAdmin(), internalKeyBindingId.intValue());
105 
106         // Extract properties      
107         final Map<String, String> propertyMap = new HashMap<String, String>();
108         //Get dynamically loaded properties
109         Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> typesAndProperties = EjbRemoteHelper.INSTANCE.getRemoteSession(
110                 InternalKeyBindingMgmtSessionRemote.class).getAvailableTypesAndProperties();
111         for (String propertyName : typesAndProperties.get(internalKeyBinding.getImplementationAlias()).keySet()) {
112             if (parameters.containsKey("-"+propertyName)) {
113                 propertyMap.put(propertyName, parameters.get("-"+propertyName));
114             }
115         }
116         // Extract remove trust entries
117         final List<InternalKeyBindingTrustEntry> removeTrustList = new ArrayList<InternalKeyBindingTrustEntry>();
118         final String removeTrustArguments = parameters.get(REMOVETRUST_KEY);
119         if (removeTrustArguments != null) {
120             for (String trustArgument : removeTrustArguments.split(TRUSTARGUMENT_SEPARATOR)) {
121                 String value;
122                 String key;
123                 int indexOfEqualsSign = trustArgument.indexOf(';');
124                 if (indexOfEqualsSign == -1) {
125                     value = null;
126                     key = trustArgument;
127                 } else {
128                     key = trustArgument.substring(0, indexOfEqualsSign);
129                     if (trustArgument.length() == indexOfEqualsSign) {
130                         value = null;
131                     } else {
132                         value = trustArgument.substring(indexOfEqualsSign + 1);
133                     }
134                 }
135                 final CAInfo caInfo = caSession.getCAInfo(getAdmin(), key);
136                 if (caInfo == null) {
137                     getLogger().info(" Ignoring trustEntry with unknown CA: " + key);
138 
139                 } else {
140                     if (value == null) {
141                         removeTrustList.add(new InternalKeyBindingTrustEntry(Integer.valueOf(caInfo.getCAId()), null));
142                     } else {
143                         try {
144                             final BigInteger serialNumber = new BigInteger(value, 16);
145                             removeTrustList.add(new InternalKeyBindingTrustEntry(Integer.valueOf(caInfo.getCAId()), serialNumber));
146                         } catch (NumberFormatException e) {
147                             getLogger().info(" Ignoring trustEntry with invalid certificate serial number: " + value);
148                         }
149                     }
150                 }
151             }
152         }
153      // Extract add trust entries
154         final List<InternalKeyBindingTrustEntry> addTrustList = new ArrayList<InternalKeyBindingTrustEntry>();
155         final String addTrustArguments = parameters.get(ADDTRUST_KEY);
156         if (addTrustArguments != null) {
157             for (String trustArgument : addTrustArguments.split(TRUSTARGUMENT_SEPARATOR)) {
158                 String value;
159                 String key;
160                 int indexOfEqualsSign = trustArgument.indexOf(';');
161                 if (indexOfEqualsSign == -1) {
162                     value = null;
163                     key = trustArgument;
164                 } else {
165                     key = trustArgument.substring(0, indexOfEqualsSign);
166                     if (trustArgument.length() == indexOfEqualsSign) {
167                         value = null;
168                     } else {
169                         value = trustArgument.substring(indexOfEqualsSign + 1);
170                     }
171                 }
172                 final CAInfo caInfo = caSession.getCAInfo(getAdmin(), key);
173                 if (caInfo == null) {
174                     getLogger().info(" Ignoring trustEntry with unknown CA: " + key);
175 
176                 } else {
177                     if (value == null) {
178                         addTrustList.add(new InternalKeyBindingTrustEntry(Integer.valueOf(caInfo.getCAId()), null));
179                     } else {
180                         try {
181                             final BigInteger serialNumber = new BigInteger(value, 16);
182                             addTrustList.add(new InternalKeyBindingTrustEntry(Integer.valueOf(caInfo.getCAId()), serialNumber));
183                         } catch (NumberFormatException e) {
184                             getLogger().info(" Ignoring trustEntry with invalid certificate serial number: " + value);
185                         }
186                     }
187                 }
188 
189             }
190         }
191         // Extract nextKeyPair
192         final String nextKeyPairAlias = parameters.get(NEXTKEYPAIR_KEY);
193         boolean modified = false;
194         // Perform nextKeyPair changes
195         if (nextKeyPairAlias != null) {
196             if ("null".equals(nextKeyPairAlias)) {
197                 if (internalKeyBinding.getNextKeyPairAlias() == null) {
198                     getLogger().info(" No nextKeyPairAlias mapping was currently set.");
199                 } else {
200                     getLogger().info(" Removing nextKeyPairAlias mapping.");
201                     internalKeyBinding.setNextKeyPairAlias(null);
202                     modified = true;
203                 }
204             } else {
205                 final List<String> availableKeyPairAliases = EjbRemoteHelper.INSTANCE.getRemoteSession(CryptoTokenManagementSessionRemote.class)
206                         .getKeyPairAliases(getAdmin(), internalKeyBinding.getCryptoTokenId());
207                 if (internalKeyBinding.getKeyPairAlias().equals(nextKeyPairAlias)) {
208                     getLogger().info(
209                             " Ignoring --nextkeypair with value " + nextKeyPairAlias + ". The value is already used as the current keyPairAlias.");
210                 } else if (!availableKeyPairAliases.contains(nextKeyPairAlias)) {
211                     getLogger().info(
212                             " Ignoring --nextkeypair with value " + nextKeyPairAlias + ". The alias is not present in the bound CryptoToken."
213                                     + " Available aliases: " + availableKeyPairAliases.toString());
214                 } else {
215                     getLogger().info(" Setting nextKeyPairAlias to \"" + nextKeyPairAlias + "\".");
216                     internalKeyBinding.setNextKeyPairAlias(nextKeyPairAlias);
217                     modified = true;
218                 }
219             }
220         }
221         // Perform trust changes
222         final List<InternalKeyBindingTrustEntry> internalKeyBindingTrustEntries = internalKeyBinding.getTrustedCertificateReferences();
223         for (final InternalKeyBindingTrustEntry internalKeyBindingTrustEntry : removeTrustList) {
224             if (!internalKeyBindingTrustEntries.remove(internalKeyBindingTrustEntry)) {
225                 getLogger().info(" Unable to remove non-existing trustEntry: " + internalKeyBindingTrustEntry.toString());
226             } else {
227                 getLogger().info(" Removed trustEntry: " + internalKeyBindingTrustEntry.toString());
228                 modified = true;
229             }
230         }
231         for (final InternalKeyBindingTrustEntry internalKeyBindingTrustEntry : addTrustList) {
232             if (internalKeyBindingTrustEntries.contains(internalKeyBindingTrustEntry)) {
233                 getLogger().info(" Unable to add existing trustEntry: " + internalKeyBindingTrustEntry.toString());
234             } else {
235                 internalKeyBindingTrustEntries.add(internalKeyBindingTrustEntry);
236                 getLogger().info(" Added trustEntry: " + internalKeyBindingTrustEntry.toString());
237                 modified = true;
238             }
239         }
240         internalKeyBinding.setTrustedCertificateReferences(internalKeyBindingTrustEntries);
241         // Perform property changes
242         Map<String, Serializable> validatedProperties = validateProperties(internalKeyBinding.getImplementationAlias(), propertyMap);
243         if (validatedProperties == null) {
244             return CommandResult.FUNCTIONAL_FAILURE;
245         }
246         for (Entry<String, Serializable> entry : validatedProperties.entrySet()) {
247             DynamicUiProperty<? extends Serializable> oldProperty = internalKeyBinding.getProperty(entry.getKey());
248             if (!oldProperty.getValue().equals(entry.getValue())) {
249                 internalKeyBinding.setProperty(entry.getKey(), entry.getValue());
250                 getLogger().info(" Setting " + entry.getKey() + " to " + String.valueOf(entry.getValue()) + "");
251                 modified = true;
252             }
253 
254         }
255         // Persist modifications
256         if (modified) {
257             internalKeyBindingMgmtSession.persistInternalKeyBinding(getAdmin(), internalKeyBinding);
258             getLogger().info("InternalKeyBinding with id " + internalKeyBindingId.intValue() + " modified successfully.");
259             return CommandResult.SUCCESS;
260         } else {
261             getLogger().error("No changes were made to InternalKeyBinding with id " + internalKeyBindingId.intValue() + ".");
262             return CommandResult.FUNCTIONAL_FAILURE;
263         }
264     }
