43  public class InternalKeyBindingCreateCommand extends BaseInternalKeyBindingCommand {
44  
45      private static final Logger log = Logger.getLogger(InternalKeyBindingCreateCommand.class);
46  
47      private static final String TYPE_KEY = "--type";
48      private static final String STATUS_KEY = "--status";
49      private static final String CERTIFICATE_FINGERPRINT_KEY = "--cert";
50      private static final String CRYPTO_TOKEN_KEY = "--token";
51      private static final String KEYPAIR_ALIAS_KEY = "--alias";
52      private static final String SIGNATURE_ALGORITHM_KEY = "--sigalg";
53  
54      {
55          registerParameter(new Parameter(TYPE_KEY, "Type Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
56                  "Keybinding type."));
57          registerParameter(new Parameter(STATUS_KEY, "Status", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
58                  "Keybinding status."));
59          registerParameter(new Parameter(CERTIFICATE_FINGERPRINT_KEY, "Certificate Fingerprint", MandatoryMode.MANDATORY, StandaloneMode.ALLOW,
60                  ParameterMode.ARGUMENT, "Fingerprint of a certificate from the bound keypair. Set as 'null' to set later."));
61          registerParameter(new Parameter(CRYPTO_TOKEN_KEY, "Token Name", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
62                  "Name of the crypto token where the mapped key pair is stored"));
63          registerParameter(new Parameter(KEYPAIR_ALIAS_KEY, "Alias", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
64                  "alias of the mapped key pair in the specified CryptoToken."));
65          registerParameter(new Parameter(SIGNATURE_ALGORITHM_KEY, "Algorithm", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
66                  "The signature algorithm that this InternalKeyBinding will use for signatures."));
67          //Register type specific properties dynamically
68          Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> typesAndProperties = EjbRemoteHelper.INSTANCE.getRemoteSession(
69                  InternalKeyBindingMgmtSessionRemote.class).getAvailableTypesAndProperties();
70          for (Entry<String, Map<String, DynamicUiProperty<? extends Serializable>>> entry : typesAndProperties.entrySet()) {
71              for (DynamicUiProperty<? extends Serializable> property : entry.getValue().values()) {
72                  if (isParameterRegistered("-"+property.getName())) {
73                      //Different properties could contain the same keyword. Simply use the same one. 
74                      continue;
75                  }
76                  Parameter parameter = new Parameter("-"+property.getName(), "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT, "");
77                  parameter.setAllowList(false);
78                  registerParameter(parameter);
79              }
80          }
81      }
82  
83      @Override
84      public String getMainCommand() {
85          return "create";
86      }
87  
88      @Override
89      public CommandResult execute(ParameterContainer parameters) {
90          final InternalKeyBindingMgmtSessionRemote internalKeyBindingMgmtSession = EjbRemoteHelper.INSTANCE
91                  .getRemoteSession(InternalKeyBindingMgmtSessionRemote.class);
92  
93          // Start by extracting any property
94          final Map<String, String> dataMap = new LinkedHashMap<String, String>();
95  
96          // Parse static arguments
97          final String name = parameters.get(KEYBINDING_NAME_KEY);
98          final String type = parameters.get(TYPE_KEY);
99          if (!InternalKeyBindingFactory.INSTANCE.existsTypeAlias(type)) {
100             getLogger().error("KeyBinding of type " + type + " does not exist.");
101             return CommandResult.FUNCTIONAL_FAILURE;
102         }
103         //Get dynamically loaded properties
104         Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> typesAndProperties = EjbRemoteHelper.INSTANCE.getRemoteSession(
105                 InternalKeyBindingMgmtSessionRemote.class).getAvailableTypesAndProperties();
106         for (String propertyName : typesAndProperties.get(type).keySet()) {
107             if (parameters.containsKey("-"+propertyName)) {
108                 dataMap.put(propertyName, parameters.get("-"+propertyName));
109             }
110         }
111         //Validate all properties
112         Map<String, Serializable> validatedProperties = validateProperties(type, dataMap);
113         if (validatedProperties == null) {
114             return CommandResult.FUNCTIONAL_FAILURE;
115         }
116         final InternalKeyBindingStatus status = InternalKeyBindingStatus.valueOf(parameters.get(STATUS_KEY).toUpperCase());
117         final String certificateId = "null".equalsIgnoreCase(parameters.get(CERTIFICATE_FINGERPRINT_KEY)) ? null : parameters
118                 .get(CERTIFICATE_FINGERPRINT_KEY);
119         final Integer cryptoTokenId = EjbRemoteHelper.INSTANCE.getRemoteSession(CryptoTokenManagementSessionRemote.class).getIdFromName(
120                 parameters.get(CRYPTO_TOKEN_KEY));
121         if(cryptoTokenId == null) {
122             log.error("ERROR: CryptoToken  " + parameters.get(CRYPTO_TOKEN_KEY) + " was not found.");
123             return CommandResult.FUNCTIONAL_FAILURE;
124         }
125         final String keyPairAlias = parameters.get(KEYPAIR_ALIAS_KEY);
126         final String signatureAlgorithm = parameters.get(SIGNATURE_ALGORITHM_KEY);
127         int internalKeyBindingIdNew;
128         try {
129             internalKeyBindingIdNew = internalKeyBindingMgmtSession.createInternalKeyBinding(getAuthenticationToken(), type, name, status,
130                     certificateId, cryptoTokenId, keyPairAlias, signatureAlgorithm, validatedProperties, null);
131         } catch (CryptoTokenOfflineException e) {
132             log.error("ERROR: CryptoToken  " + parameters.get(CRYPTO_TOKEN_KEY) + " was offline.");
133             return CommandResult.FUNCTIONAL_FAILURE;
134         } catch (InternalKeyBindingNameInUseException e) {
135             log.error("ERROR: Keybinding of name " + name + " already exists,");
136             return CommandResult.FUNCTIONAL_FAILURE;
137         } catch (InvalidAlgorithmException e) {
138             log.error("ERROR: " + signatureAlgorithm + " was not a valid algorithm.");
139             return CommandResult.FUNCTIONAL_FAILURE;
140         } catch (AuthorizationDeniedException e) {
141             return CommandResult.AUTHORIZATION_FAILURE;
142         }
143         getLogger().info("InternalKeyBinding with id " + internalKeyBindingIdNew + " created successfully.");
144         return CommandResult.SUCCESS;
145     }
