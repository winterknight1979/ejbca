41  public class InternalKeyBindingGenerateCsrCommand extends RudInternalKeyBindingCommand {
42  
43      private static final Logger log = Logger.getLogger(InternalKeyBindingGenerateCsrCommand.class);
44  
45      private static final String GENKEYPAIR_KEY = "--genkeypair";
46      private static final String CSR_FILE_KEY = "-f";
47      private static final String SUBJECTDN_KEY = "--subjectdn";
48      private static final String SUBJECTDN_ORDER_KEY = "--x500dnorder";
49  
50      {
51          registerParameter(Parameter.createFlag(GENKEYPAIR_KEY, "Set to generate a \"next\" key pair with the same key specification as the current and a new alias."
52                  + " If a nextKey reference already exists, it will be replaced with a reference to the new key"));
53          registerParameter(new Parameter(CSR_FILE_KEY, "Filename", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
54                  "Destination file for the CSR."));
55          registerParameter(new Parameter(SUBJECTDN_KEY, "Subject DN", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.ARGUMENT,
56                  "Subject DN in the generated CSR, if left out default subject DN will be used (CN=Should be ignored by CA)."));
57          registerParameter(Parameter.createFlag(SUBJECTDN_ORDER_KEY, "(when using --subjectdn) Use this flag to specify that the DN order in the CSR should be X500 (C=SE,O=org,CN=name) instead of the default (CN=name,O=org,C=SE)"));
58      }
59  
60      @Override
61      public String getMainCommand() {
62          return "gencsr";
63      }
64  
65      @Override
66      public CommandResult executeCommand(Integer internalKeyBindingId, ParameterContainer parameters) throws AuthorizationDeniedException, IOException,
67              InvalidKeyException, CryptoTokenOfflineException, InvalidAlgorithmParameterException {
68          final InternalKeyBindingMgmtSessionRemote internalKeyBindingMgmtSession = EjbRemoteHelper.INSTANCE
69                  .getRemoteSession(InternalKeyBindingMgmtSessionRemote.class);
70          
71          final boolean switchGenKeyPair = parameters.containsKey(GENKEYPAIR_KEY);
72          final String csrSubjectDN = parameters.get(SUBJECTDN_KEY);
73          // If contains key we want to boolean to be false, because LDAP DN order is "true"
74          final boolean x500dnorder = !parameters.containsKey(SUBJECTDN_ORDER_KEY);
75          final X500Name x500Name;
76          if (csrSubjectDN != null) {
77              x500Name = CertTools.stringToBcX500Name(csrSubjectDN, x500dnorder);
78              getLogger().info("Using subject DN from argument '" + x500Name.toString() + "', with order "+x500dnorder);
79          } else {
80              if (parameters.containsKey(SUBJECTDN_ORDER_KEY)) {
81                  getLogger().warn(SUBJECTDN_ORDER_KEY + " is defined, but " + SUBJECTDN_KEY + " is not.");
82              }
83              getLogger().info("Using default subject DN, from existing mapped certificate if it is present");
84              x500Name = null;
85          }
86          String nextKeyAlias;
87          if (switchGenKeyPair) {
88              nextKeyAlias = internalKeyBindingMgmtSession.generateNextKeyPair(getAdmin(), internalKeyBindingId);
89              getLogger().info("A new key pair has been generated with alias " + nextKeyAlias);
90          } else {
91              final InternalKeyBinding internalKeyBinding = internalKeyBindingMgmtSession.getInternalKeyBindingInfo(getAdmin(), internalKeyBindingId);
92              nextKeyAlias = internalKeyBinding.getNextKeyPairAlias();
93              if (nextKeyAlias == null) {
94                  nextKeyAlias = internalKeyBinding.getKeyPairAlias();
95              }
96              getLogger().info("Next key pair alias is " + nextKeyAlias);
97          }
98          final byte[] certificateRequestBytes = internalKeyBindingMgmtSession.generateCsrForNextKey(getAdmin(), internalKeyBindingId, x500Name != null ? x500Name.getEncoded():null);
99          if (certificateRequestBytes == null) {
100             getLogger().error("Unable to generate CSR for " + nextKeyAlias);
101             return CommandResult.FUNCTIONAL_FAILURE;
102         } else {
103             final byte[] pemEncodedPublicKey = CertTools.getPEMFromCertificateRequest(certificateRequestBytes);
104             final OutputStream fos = new FileOutputStream(parameters.get(CSR_FILE_KEY));
105             fos.write(pemEncodedPublicKey);
106             fos.close();
107             getLogger().info(
108                     "Stored PEM encoded PKCS#10 request for \"" + parameters.get(KEYBINDING_NAME_KEY) + "\" as " + parameters.get(CSR_FILE_KEY));
109             return CommandResult.SUCCESS;
110         }
111     }
112 
113     @Override
114     public String getCommandDescription() {
115         return "Generate a PKCS#10 CSR for the next key pair to be used.";
116     }
117 
118     @Override
119     public String getFullHelpText() {
120         return getCommandDescription()+ " Optionally generates a new \"next\" key pair and otherwise returns the current public key. \nThe subjectDN in the CSR will be set to the currently mapped certificate's SubjectDN (if present) and if not a default DN of (CN=Internal Key Binding Name). A subjectDN can also be set using the parameter --subjectdn (see --help)";    
121     }
122 
123     protected Logger getLogger() {
124         return log;
125     }
126 }
