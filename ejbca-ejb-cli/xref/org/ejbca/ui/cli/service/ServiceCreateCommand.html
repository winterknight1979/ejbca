13  package org.ejbca.ui.cli.service;
14  
15  import org.apache.log4j.Logger;
16  import org.cesecore.util.EjbRemoteHelper;
17  import org.ejbca.core.ejb.services.ServiceSessionRemote;
18  import org.ejbca.core.model.services.ServiceConfiguration;
19  import org.ejbca.core.model.services.ServiceExistsException;
20  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
21  import org.ejbca.ui.cli.infrastructure.parameter.Parameter;
22  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
23  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
24  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
25  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;
26  
27  /**
28   * CLI subcommand for creating services.
29   * 
30   * @version $Id: ServiceCreateCommand.java 19902 2014-09-30 14:32:24Z anatom $
31   */
32  public class ServiceCreateCommand extends BaseServiceModificationCommand {
33  
34      private static final Logger log = Logger.getLogger(ServiceCreateCommand.class);
35  
36      private static final String ARGS_KEY = "--properties";
37  
38      {
39          registerParameter(new Parameter(
40                  ARGS_KEY,
41                  "List of Properties",
42                  MandatoryMode.OPTIONAL,
43                  StandaloneMode.ALLOW,
44                  ParameterMode.ARGUMENT,
45                  "A list of properties, must be in the form of \"field1:property1 field2:property2\", e.g: \"interval.periodical.unit=DAYS interval.periodical.value=7\""));
46      }
47  
48      @Override
49      public String getMainCommand() {
50          return "create";
51      }
52  
53      @Override
54      public CommandResult execute(ParameterContainer parameters, int serviceId) {
55  
56          final ServiceSessionRemote serviceSession = EjbRemoteHelper.INSTANCE.getRemoteSession(ServiceSessionRemote.class);
57          final String serviceName = parameters.get(SERVICE_NAME_KEY);
58  
59          ServiceConfiguration serviceConfig = new ServiceConfiguration();
60          final boolean wasActive = false;
61          final String[] args;
62          if (parameters.containsKey(ARGS_KEY)) {
63              args = parameters.get(ARGS_KEY).split(" ");
64          } else {
65              args = new String[] {};
66          }
67  
68          if (modifyFromArgs(serviceConfig, args)) {
69              try {
70                  serviceSession.addService(getAdmin(), serviceName, serviceConfig);
71                  handleServiceActivation(serviceName, wasActive);
72                  getLogger().info("Service created.");
73                  return CommandResult.SUCCESS;
74              } catch (ServiceExistsException e) {
75                  getLogger().error("ERROR: Service exists already.");
76                  return CommandResult.FUNCTIONAL_FAILURE;
77              }
78          }
79          return CommandResult.FUNCTIONAL_FAILURE;
80      }
81  
82      @Override
83      protected boolean failIfServiceMissing() {
84          return false;
85      }
86  
87      @Override
88      public String getCommandDescription() {
89          return "Creates a new service.";
90      }
91  
92      @Override
93      public String getFullHelpText() {
94          StringBuilder sb = new StringBuilder();
95          sb.append("This example creates a service that updates the CRL on a daily basis. "
96                  + "The worker.caidstocheck value of 1 means check all CAs. "
97                  + "You can create services from the Admin Web and use the \"service info\" command "
98                  + "to learn how the CLI fields correspond to the fields in the Admin Web.\n\n");
99          sb.append("Example usage: service create DailyCRLUpdate workerClassPath=org.ejbca.core.model.services.workers.CRLUpdateWorker"
100                 + " worker.caidstocheck=1 intervalClassPath=org.ejbca.core.model.services.intervals.PeriodicalInterval interval.periodical.unit=DAYS"
101                 + " interval.periodical.value=1 actionClassPath=org.ejbca.core.model.services.actions.NoAction active=true");
102         sb.append("\n\n").append(FIELDS_HELP + "\n\n");
103         return sb.toString();
104     }
105 
106     @Override
107     protected Logger getLogger() {
108         return log;
109     }
110 }
