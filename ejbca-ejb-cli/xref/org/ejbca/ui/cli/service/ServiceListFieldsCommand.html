13  package org.ejbca.ui.cli.service;
14  
15  import org.apache.log4j.Logger;
16  import org.cesecore.util.EjbRemoteHelper;
17  import org.ejbca.core.ejb.services.ServiceSessionRemote;
18  import org.ejbca.core.model.services.ServiceConfiguration;
19  import org.ejbca.ui.cli.FieldEditor;
20  import org.ejbca.ui.cli.infrastructure.command.CommandResult;
21  import org.ejbca.ui.cli.infrastructure.parameter.ParameterContainer;
22  
23  /**
24   * CLI subcommand that lists all available fields
25   * 
26   * @version $Id: ServiceListFieldsCommand.java 21708 2015-08-24 14:48:34Z mikekushner $
27   */
28  public class ServiceListFieldsCommand extends BaseServiceCommand {
29  
30      private static final Logger log = Logger.getLogger(ServiceListFieldsCommand.class);
31  
32      @Override
33      public String getMainCommand() {
34          return "listfields";
35      }
36  
37      @Override
38      public CommandResult execute(ParameterContainer parameters, int serviceId) {
39          ServiceConfiguration serviceConfig = EjbRemoteHelper.INSTANCE.getRemoteSession(ServiceSessionRemote.class).getServiceConfiguration(serviceId);
40          final FieldEditorieldEditor">FieldEditor fieldEditor = new FieldEditor(log);
41          fieldEditor.listSetMethods(serviceConfig);
42          return CommandResult.SUCCESS;
43  
44      }
45  
46      @Override
47      public String getCommandDescription() {
48          return "Lists all available fields for a service";
49      }
50  
51      @Override
52      public String getFullHelpText() {
53          return getCommandDescription();
54      }
55  
56      @Override
57      protected Logger getLogger() {
58          return log;
59      }
60  
61  }
