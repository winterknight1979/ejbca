48   */
49  public class SetDefaultOcspResponderCommand extends EjbcaCliUserCommandBase {
50  
51      private static final Logger log = Logger.getLogger(SetDefaultOcspResponderCommand.class);
52  
53      private static final String DN_KEY = "--dn";
54  
55      {
56          registerParameter(new Parameter(DN_KEY, "DN", MandatoryMode.MANDATORY, StandaloneMode.ALLOW, ParameterMode.ARGUMENT,
57                  "DN to set as default responder."));
58      }
59  
60      @Override
61      public String[] getCommandPath() {
62          return new String[] { "ocsp" };
63      }
64  
65      @Override
66      public String getMainCommand() {
67          return "setdefaultresponder";
68      }
69  
70      @Override
71      public CommandResult execute(ParameterContainer parameters) {
72          GlobalConfigurationSessionRemote globalConfigurationSession = EjbRemoteHelper.INSTANCE
73                  .getRemoteSession(GlobalConfigurationSessionRemote.class);
74          GlobalOcspConfiguration conf = (GlobalOcspConfiguration) globalConfigurationSession
75                  .getCachedConfiguration(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID);
76          String reference = parameters.get(DN_KEY);
77          conf.setOcspDefaultResponderReference(reference);
78          try {
79              globalConfigurationSession.saveConfiguration(getAuthenticationToken(), conf);
80              log.info("Setting '" + reference + "' as the default responder.");
81          } catch (AuthorizationDeniedException e) {
82              log.error("ERROR: Current CLI user wasn't authorized to edit configuration.");
83              return CommandResult.AUTHORIZATION_FAILURE;
84          }
85          return CommandResult.SUCCESS;
86      }
87  
88      @Override
89      public String getCommandDescription() {
90          return "Sets the default OCSP responder.";
91      }
92  
93      @Override
94      public String getFullHelpText() {
95          StringBuilder stringBuilder = new StringBuilder();
96          stringBuilder.append(getCommandDescription());
97          stringBuilder
98                  .append(" The value is saved in the form of a DN, either the subject DN of the CA or the issuer DN of the Internal Keybinding chosen as default responder.");
99          stringBuilder
100                 .append(" Note that input from this command is not vetted, so be sure to be exact in setting the value. DNs inputed in reverse order will be saved in standard order.");
101         stringBuilder.append(" If no default responder is to be set, simply enter an emptry string within quotes (\"\"). Local CAs with active OCSP Keybindings will not be listed.\n\n");
102         InternalKeyBindingMgmtSessionRemote internalKeyBindingMgmtSession = EjbRemoteHelper.INSTANCE.getRemoteSession(InternalKeyBindingMgmtSessionRemote.class);
103         CertificateStoreSessionRemote certificateStoreSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CertificateStoreSessionRemote.class);
104         List<String[]> ikbContents = new ArrayList<String[]>();
105         Set<String> knownDNs = new HashSet<String>();
106         for(InternalKeyBindingInfo info : internalKeyBindingMgmtSession.getInternalKeyBindingInfos(getAuthenticationToken(), OcspKeyBinding.IMPLEMENTATION_ALIAS)) {
107             if(info.getStatus().equals(InternalKeyBindingStatus.ACTIVE)) {
108                 Certificate certificate = EJBTools.unwrap(certificateStoreSession.findCertificateByFingerprintRemote(info.getCertificateId()));
109                 ikbContents.add(new String[]{info.getName(), CertTools.getIssuerDN(certificate)});
110                 knownDNs.add(CertTools.getIssuerDN(certificate));
111             }
112         }
113         if(!ikbContents.isEmpty()) {
114             stringBuilder.append(bold("The following Internal Keybindings are available as signers:\n"));
115             stringBuilder.append(formatTable(1, new String[] { "Keybinding Name:", "Issuer DN:" }, ikbContents));
116             stringBuilder.append("\n");
117         }
118         CaSessionRemote caSession = EjbRemoteHelper.INSTANCE.getRemoteSession(CaSessionRemote.class);
119         List<String[]> caContents = new ArrayList<String[]>();     
120         for (CAInfo caInfo : caSession.getAuthorizedAndEnabledCaInfos(getAuthenticationToken())) {
121             final String caSubjectDn = CertTools.getSubjectDN(new ArrayList<Certificate>(caInfo.getCertificateChain()).get(0));
122             if (caInfo.getCAType() == CAInfo.CATYPE_X509 && !knownDNs.contains(caSubjectDn) && caInfo.getStatus() == CAConstants.CA_ACTIVE) {
123                 caContents.add(new String[] { caInfo.getName(), caSubjectDn});
124             }
125         }
126         if (!caContents.isEmpty()) {
127             stringBuilder.append(bold("The following CAs are available as signers:\n"));
128             stringBuilder.append(formatTable(1, new String[] { "CA Name:", "Subject DN:" }, caContents));
129             stringBuilder.append("\n");
130         }
131        
132         GlobalConfigurationSessionRemote globalConfigurationSession = EjbRemoteHelper.INSTANCE
133                 .getRemoteSession(GlobalConfigurationSessionRemote.class);
134         GlobalOcspConfiguration conf = (GlobalOcspConfiguration) globalConfigurationSession
135                 .getCachedConfiguration(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID);
136         stringBuilder.append("The current default responder is: " + conf.getOcspDefaultResponderReference());
137         return stringBuilder.toString();
138     }
139 
140     @Override
141     protected Logger getLogger() {
142         return log;
143     }
