14  package org.ejbca.core.protocol.acme;
15  
16  import javax.ejb.Local;
17  
18  /**
19   * Replay Protection nonce logic
20   *
21   * https://tools.ietf.org/html/draft-ietf-acme-acme-12#section-6.4
22   *
23   *   Example value from RFC draft: D8s4D2mLs8Vn-goWuPQeKA
24   *
25   * When presented with a nonce, we need to verify that:
26   * - It has been generated
27   * - It has not been used already
28   *
29   * We don't want to store every single nonce that we ever have generated since
30   * - it would slow down each request to wait for the transaction to commit
31   * - hammering the service would fill up the database with things that will be never used
32   *
33   * Instead we should
34   * - Use an HMAC embedded in the nonce to check if the nonce has been generated by one of our servers
35   * - Since we with an HMAC can ensure data integrity, we should embed a timestamp and prevent too old nonces from being
36   *   used. This will allow us to purge used nonces from the database, when they can no longer be used anyway.
37   *
38   * If we reject the nonce, the client "SHOULD" retry the request, so we expect that most clients wont roll over and die
39   * permanently if we expire nonces after a while.
40   *
41   * The format of the nonce is as follows:
42   * ---------------------32 byte message------------------------------|----------32 byte HMAC--------------
43   * --------8 byte timestamp-|-8 byte counter-|-16 byte nodeId--------|------------------------------------
44   *
45   * @version $Id: AcmeNonceSingletonBean.java 29587 2018-08-07 15:25:52Z mikekushner $
46   */
47  
48  @Local
49  public interface AcmeNonceSingletonLocal {
50  
51      /** @param nonce nonce
52       * @return true if the provided nonce is one that has been generated and has not yet been used. (Non RFC quirk: we will reject expired nonces.) */
53      boolean isNonceValid(String nonce);
54  
55      /** @return a new nonce as a base64 url-safe encoded String 
56       * @throws IllegalStateException fail*/
57      String getReplayNonce() throws IllegalStateException;
58  
59  }
