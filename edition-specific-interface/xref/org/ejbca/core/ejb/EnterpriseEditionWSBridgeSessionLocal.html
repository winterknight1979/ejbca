13  package org.ejbca.core.ejb;
14  
15  import java.security.InvalidAlgorithmParameterException;
16  import java.security.InvalidKeyException;
17  import java.util.List;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.authorization.AuthorizationDeniedException;
23  import org.cesecore.certificates.ca.CADoesntExistsException;
24  import org.cesecore.certificates.ca.CAExistsException;
25  import org.cesecore.certificates.ca.InvalidAlgorithmException;
26  import org.cesecore.certificates.ca.SignedByExternalCANotSupportedException;
27  import org.cesecore.certificates.certificateprofile.CertificateProfileDoesNotExistException;
28  import org.cesecore.keys.token.CryptoTokenAuthenticationFailedException;
29  import org.cesecore.keys.token.CryptoTokenNameInUseException;
30  import org.cesecore.keys.token.CryptoTokenOfflineException;
31  import org.cesecore.keys.token.p11.exception.NoSuchSlotException;
32  import org.cesecore.roles.RoleNotFoundException;
33  import org.ejbca.util.KeyValuePair;

35  /**
36   * JEE5 Lookup helper implementation for optional (enterprise edition) WS methods.
37   * 
38   * @version $Id: EnterpriseEditionWSBridgeSessionLocal.java 24602 2016-10-31 13:26:34Z anatom $
39   */
40  @Local
41  public interface EnterpriseEditionWSBridgeSessionLocal {
42  
43      /**
44       * Creates a new cryptotoken
45       * 
46       * @param admin An authentication token
47       * @param tokenName The name of the crypto token
48       * @param tokenType The type of the crypto token. Available types: SoftCryptoToken, PKCS11CryptoToken
49       * @param activationPin Pin code for the crypto token
50       * @param autoActivate Set to true|false to allow|disallow whether crypto token should be autoactivated or not
51       * @param cryptoTokenProperties The properties of the cryptotoken. See {@link org.ejbca.core.protocol.ws.objects.CryptoTokenConstantsWS}
52       * 
53       * @throws UnsupportedMethodException if trying to access this method in the community version
54       * @throws AuthorizationDeniedException if admin lacks access to resource /cryptotoken/modify
55       * @throws CryptoTokenOfflineException if the crypto token was unavailable
56       * @throws CryptoTokenAuthenticationFailedException if the password specified in activationPin was incorrect
57       * @throws CryptoTokenNameInUseException if a crypto token with the given name already exists
58       * @throws NoSuchSlotException if the slot as defined in cryptoTokenProperties doesn't exist
59       */
60      void createCryptoToken(AuthenticationToken admin, String tokenName, String tokenType, String activationPin, boolean autoActivate, 
61              List<KeyValuePair> cryptoTokenProperties) throws UnsupportedMethodException, AuthorizationDeniedException, 
62              CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException, CryptoTokenNameInUseException, NoSuchSlotException;
63      
64      /**
65       * Generates a keys pair in the specified cryptotoken
66       * 
67       * @param admin An authentication token
68       * @param cryptoTokenName The name of the cryptotoken
69       * @param keyPairAlias Key pair alias
70       * @param keySpecification Key specification, for example RSA2048, secp256r1, DSA1024, gost3410, dstu4145
71       * 
72       * @throws UnsupportedMethodException When trying to access this method in the community version
73       * @throws InvalidKeyException if key generation failed
74       * @throws CryptoTokenOfflineException if the crypto token was unavailable
75       * @throws InvalidAlgorithmParameterException if the keySpecification is not available for this CryptoToken.
76       * @throws AuthorizationDeniedException if admin lacks access to resource /cryptotoken/modify
77       */
78      void generateCryptoTokenKeys(AuthenticationToken admin, String cryptoTokenName, String keyPairAlias, String keySpecification) 
79              throws UnsupportedMethodException, InvalidKeyException, CryptoTokenOfflineException, InvalidAlgorithmParameterException, 
80              AuthorizationDeniedException;
81      
82      /**
83       * Creates a new CA
84       * 
85       * @param admin An authentication token
86       * @param caname The CA name
87       * @param cadn The CA subjectDN
88       * @param catype The CA type. It could be either 'x509' or 'cvc'
89       * @param encodedValidity Validity of the CA  encoded for example as "3650d".
90       * @param certprofile Makes the CA use the certificate profile 'certprofile' instead of the default ROOTCA or SUBCA.
91       * @param signAlg Signing Algorithm may be one of the following: SHA1WithRSA, SHA256WithRSA, SHA384WithRSA, SHA512WithRSA
92       *        SHA256WithRSAAndMGF1, SHA1withECDSA, SHA224withECDSA, SHA256withECDSA, SHA384withECDSA, SHA512withECDSA, SHA1WithDSA, 
93       *        GOST3411withECGOST3410, GOST3411withDSTU4145
94       * @param signedByCAId The ID of a CA that will sign this CA. Use '1' for self signed CA (i.e. a root CA). CAs created using the WS cannot be signed by external CAs.
95       * @param cryptoTokenName The name of the cryptotoken associated with the CA
96       * @param purposeKeyMapping The mapping the the cryptotoken keys and their purpose. See {@link org.ejbca.core.protocol.ws.objects.CAConstantsWS}
97       * @param caProperties Optional CA properties. See {@link org.ejbca.core.protocol.ws.objects.CAConstantsWS}
98       * 
99       * @throws UnsupportedMethodException When trying to access this method in the community version
100      * @throws SignedByExternalCANotSupportedException fail
101      * @throws CAExistsException if a CA with the given name already exists
102      * @throws AuthorizationDeniedException if admin is not authorized to create CAs
103      * @throws CertificateProfileDoesNotExistException if the certificate profile specified by certprofile doesn't exist
104      * @throws CertificateProfileTypeNotAcceptedException if the certificate profile was not of type ROOTCA or SUBCA
105      * @throws CryptoTokenOfflineException if the crypto token was unavailable
106      * @throws InvalidAlgorithmException if the CA signature algorithm was invalid
107      */
108     void createCA(AuthenticationToken admin, String caname, String cadn, String catype, String encodedValidity, String certprofile, 
109             String signAlg, int signedByCAId, String cryptoTokenName, List<KeyValuePair> purposeKeyMapping, List<KeyValuePair> caProperties) 
110             throws UnsupportedMethodException, SignedByExternalCANotSupportedException, CAExistsException, AuthorizationDeniedException, 
111             CertificateProfileDoesNotExistException, CertificateProfileTypeNotAcceptedException, CryptoTokenOfflineException, InvalidAlgorithmException;
112     
113     /**
114      * Adds an administrator to the specified role
115      * 
116      * @param admin An authentication token
117      * @param roleName The role to add the admin to
118      * @param caName Name of the CA that issued the new administrator's certificate
119      * @param matchWith Could be any of: NONE, WITH_COUNTRY, WITH_DOMAINCOMPONENT, WITH_STATEORPROVINCE, WITH_LOCALITY, WITH_ORGANIZATION,
120               WITH_ORGANIZATIONALUNIT, WITH_TITLE, WITH_COMMONNAME, WITH_UID, WITH_DNSERIALNUMBER, WITH_SERIALNUMBER,
121               WITH_DNEMAILADDRESS, WITH_RFC822NAME, WITH_UPN, WITH_FULLDN
122      * @param matchType Could be one of: TYPE_EQUALCASE, TYPE_EQUALCASEINS, TYPE_NOT_EQUALCASE, TYPE_NOT_EQUALCASEINS, TYPE_NONE
123      * @param matchValue the value to match against
124      * 
125      * @throws UnsupportedMethodException if trying to access this method in the community version
126      * @throws RoleNotFoundException if the role specified by rolename doesn't exist
127      * @throws CADoesntExistsException if the CA specified by caName doesn't exist
128      * @throws AuthorizationDeniedException if admin doesn't have access to CA or rights to manage roles.
129      */
130     void addSubjectToRole(AuthenticationToken admin, String roleName, String caName, String matchWith, String matchType, 
131             String matchValue) throws UnsupportedMethodException, RoleNotFoundException, CADoesntExistsException, AuthorizationDeniedException;
132     
133     /**
134      * Removes an administrator to the specified role
135      * 
136      * @param admin An authentication token
137      * @param roleName The role to add the admin to
138      * @param caName Name of the CA that issued the new administrator's certificate
139      * @param matchWith Could be any of: NONE, WITH_COUNTRY, WITH_DOMAINCOMPONENT, WITH_STATEORPROVINCE, WITH_LOCALITY, WITH_ORGANIZATION,
140               WITH_ORGANIZATIONALUNIT, WITH_TITLE, WITH_COMMONNAME, WITH_UID, WITH_DNSERIALNUMBER, WITH_SERIALNUMBER,
141               WITH_DNEMAILADDRESS, WITH_RFC822NAME, WITH_UPN, WITH_FULLDN
142      * @param matchType Could be one of: TYPE_EQUALCASE, TYPE_EQUALCASEINS, TYPE_NOT_EQUALCASE, TYPE_NOT_EQUALCASEINS, TYPE_NONE
143      * @param matchValue if admin doesn't have access to CA or rights to manage roles.
144      * 
145      * @throws UnsupportedMethodException if trying to access this method in the community version
146      * @throws RoleNotFoundException if the role specified by rolename doesn't exist
147      * @throws CADoesntExistsException if the CA specified by caName doesn't exist
148      * @throws AuthorizationDeniedException if admin doesn't have access to CA or rights to manage roles.
149      */
150     void removeSubjectFromRole(AuthenticationToken admin, String roleName, String caName, String matchWith, String matchType, 
151             String matchValue) throws UnsupportedMethodException, RoleNotFoundException, CADoesntExistsException, AuthorizationDeniedException;
152 }
