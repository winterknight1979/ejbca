133     @SuppressWarnings("deprecation")
134     @Test
135     public void testConvertAndMerge() {
136         log.trace(">testConvertAndMerge");
137         final AccessSet asNew = makeNewAccessSet();
138         // Merge with an old AccessSet to force downgrade
139         final AccessSet asOld = new AccessSet(Arrays.asList("/a", "/f", "/nonexistentold"));
140         final AccessSet merged = new AccessSet(asNew, asOld);
141         assertTrue(merged.isAuthorized("/a"));
142         assertFalse(merged.isAuthorized("/b"));
143         assertTrue(merged.isAuthorized("/b/c"));
144         assertTrue(merged.isAuthorized("/b/c/d"));
145         assertFalse(merged.isAuthorized("/b/c/e"));
146         assertTrue(merged.isAuthorized("/f"));
147         assertFalse(merged.isAuthorized("/nonexistent")); // the converter does not allow non-existent resources
148         assertTrue(merged.isAuthorized("/nonexistentold")); // this rule comes from the old AccessSet, so it's left untouched
149         log.trace("<testConvertAndMerge");
150     }

152     /** Creates an AccessSet with the legacy representation of access rules  
153      * @param resources resources
154      * @return access set*/
155     @SuppressWarnings("deprecation")
156     private AccessSet makeLegacyAccessSet(final String... resources) {
157         final Collection<String> col = new ArrayList<>();
158         for (final String resource : resources) {
159             col.add(resource);
160         }
161         return new AccessSet(col);
162     }
163     
164     @SuppressWarnings("deprecation")
165     private AccessSet makeNewAccessSet() {
166         final Set<String> allResources = new HashSet<>();
167         allResources.add("/a/");
168         allResources.add("/b/");
169         allResources.add("/b/c/");
170         allResources.add("/b/c/d/");
171         allResources.add("/b/c/e/");
172         allResources.add("/f/");
173         final HashMap<String,Boolean> accessRules = new HashMap<>();
174         accessRules.put("/a/", true);
175         accessRules.put("/b/c/", true);
176         accessRules.put("/b/c/e/", false);
177         accessRules.put("/nonexistent/", true);
178         return AccessSet.fromAccessRules(accessRules, allResources);
179     }
