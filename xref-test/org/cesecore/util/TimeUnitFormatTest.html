32      // duplicate code from SimpleTime start
33      public static final long MILLISECONDS_PER_YEAR = 31536000000L; // 365 days
34      public static final long MILLISECONDS_PER_MONTH = 2592000000L; // 30 days
35      public static final long MILLISECONDS_PER_DAY = 86400000L;
36      public static final long MILLISECONDS_PER_HOUR = 3600000L;
37      public static final long MILLISECONDS_PER_MINUTE = 60000L;
38      public static final long MILLISECONDS_PER_SECOND = 1000L;
39  
40      public static final String TYPE_YEARS = "y";
41      public static final String TYPE_MONTHS = "mo";
42      public static final String TYPE_DAYS = "d";
43      public static final String TYPE_HOURS = "h";
44      public static final String TYPE_MINUTES = "m";
45      public static final String TYPE_SECONDS = "s";
46      public static final String TYPE_MILLISECONDS = "ms";
47  
48      private static final Map<String, Long> MILLISECONDS_FACTOR = new LinkedHashMap<String, Long>();
49      static {
50          MILLISECONDS_FACTOR.put(TYPE_YEARS, MILLISECONDS_PER_YEAR);
51          MILLISECONDS_FACTOR.put(TYPE_MONTHS, MILLISECONDS_PER_MONTH);
52          MILLISECONDS_FACTOR.put(TYPE_DAYS, MILLISECONDS_PER_DAY);
53          MILLISECONDS_FACTOR.put(TYPE_HOURS, MILLISECONDS_PER_HOUR);
54          MILLISECONDS_FACTOR.put(TYPE_MINUTES, MILLISECONDS_PER_MINUTE);
55          MILLISECONDS_FACTOR.put(TYPE_SECONDS, MILLISECONDS_PER_SECOND);
56          MILLISECONDS_FACTOR.put(TYPE_MILLISECONDS, 1L);
57      }

63      /**
64       * Tests the default parser and formatter (with '*y *mo *d *h *m *s *ms'). 
65       * Both methods of the test object (a.parseMillis() and a.format()) are tested here, to not to double code.
66       * Values with '0' are not rendered (i.e. '1y 0mo' -&gt; '1y').
67       * @throws Exception fail
68       */
69      @Test
70      public void testParserAndFormatter() throws Exception {
71          final TimeUnitFormat parser = SimpleTime.getMilliSecondsFormat();
72          
73          // test zero type
74          String type = TYPE_DAYS;
75          String time = "0y";
76          long millis = parser.parseMillis(time);
77          assertEquals("'" + time + "' input long value.", 0, millis);
78          assertEquals("'" + time + "' input formatted.", "0" + type, parser.format(millis, MILLISECONDS_FACTOR, type));
79          
80          // test positive part string
81          time = "1y2mo3d4h7ms";
82          millis = parser.parseMillis(time);
83          assertEquals("'" + time + "' input long value.", 1 * MILLISECONDS_PER_YEAR + 2 * MILLISECONDS_PER_MONTH + 3 * MILLISECONDS_PER_DAY + 4 * MILLISECONDS_PER_HOUR + 7, millis);
84          assertEquals("'" + time + "' input formatted.", "1y 2mo 3d 4h 7ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
85          
86          // test positive full string 
87          time = "1y2mo3d4h5m6s7ms";
88          millis = parser.parseMillis(time);
89          assertEquals("'" + time + "' input long value.", 1 * MILLISECONDS_PER_YEAR + 2 * MILLISECONDS_PER_MONTH + 3 * MILLISECONDS_PER_DAY
90                  + 4 * MILLISECONDS_PER_HOUR + 5 * MILLISECONDS_PER_MINUTE + 6 * MILLISECONDS_PER_SECOND + 7, millis);
91          assertEquals("'" + time + "' input formatted.", "1y 2mo 3d 4h 5m 6s 7ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
92  
93          // test negative value
94          time = "-1y-2mo-3d-4h-5m-6s-7ms";
95          millis = parser.parseMillis(time);
96          assertEquals("'" + time + "' input long value.", -1 * MILLISECONDS_PER_YEAR - 2 * MILLISECONDS_PER_MONTH - 3 * MILLISECONDS_PER_DAY
97                  - 4 * MILLISECONDS_PER_HOUR - 5 * MILLISECONDS_PER_MINUTE - 6 * MILLISECONDS_PER_SECOND - 7, millis);
98          assertEquals("'" + time + "' input formatted.", "-1y -2mo -3d -4h -5m -6s -7ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
99          
100         // test canonical form for positive results no negative values! (i.e.: '400d' -> '1y 1mo 5d' NOT '1y 2mo -25d')
101         time = "400d";
102         millis = parser.parseMillis(time);
103         assertEquals("'" + time + "' input long value.", 400 * MILLISECONDS_PER_DAY, millis);
104         assertEquals("'" + time + "' input formatted.", "1y 1mo 5d", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
105         
106         // test negative canonical form -> only negative values! (i.e.: '-400d' -> '-1y -1mo -5d' NOT '-1y -2mo +25d')
107         time = "-400d";
108         millis = parser.parseMillis(time);
109         assertEquals("'" + time + "' input long value.", -400 * MILLISECONDS_PER_DAY, millis);
110         assertEquals("'" + time + "' input formatted.", "-1y -1mo -5d", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
111         
112         // test positive sum with optional sequence
113         time = " -1y 3d 3d -6s -14s  2y  7ms -2mo  -4h-5m";
114         millis = parser.parseMillis(time);
115         assertEquals("'" + time + "' input long value.", 1 * MILLISECONDS_PER_YEAR + 3 * MILLISECONDS_PER_DAY + 3 * MILLISECONDS_PER_DAY - 6 * MILLISECONDS_PER_SECOND - 14 * MILLISECONDS_PER_SECOND
116                 + 7 - 2 * MILLISECONDS_PER_MONTH - 4 * MILLISECONDS_PER_HOUR - 5 * MILLISECONDS_PER_MINUTE, millis);
117         assertEquals("'" + time + "' input formatted.", "10mo 10d 19h 54m 40s 7ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
118         
119         // test negative sum with optional sequence
120         time = " -1y 3d 3d -6s -14s  1y  7ms -2mo  -4h-5m";
121         millis = parser.parseMillis(time);
122         assertEquals("'" + time + "' input long value.", 3 * MILLISECONDS_PER_DAY + 3 * MILLISECONDS_PER_DAY - 6 * MILLISECONDS_PER_SECOND - 14 * MILLISECONDS_PER_SECOND
123                 + 7 - 2 * MILLISECONDS_PER_MONTH - 4 * MILLISECONDS_PER_HOUR - 5 * MILLISECONDS_PER_MINUTE, millis);
124         assertEquals("'" + time + "' input formatted.", "-1mo -24d -4h -5m -19s -993ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
125         
126         // test different formats
127         time = "   1y   -2mo    3d-4h-5m    -6s      7ms  ";
128         millis = parser.parseMillis(time);
129         assertEquals("'" + time + "' input long value.", 1 * MILLISECONDS_PER_YEAR - 2 * MILLISECONDS_PER_MONTH + 3 * MILLISECONDS_PER_DAY
130                 - 4 * MILLISECONDS_PER_HOUR - 5 * MILLISECONDS_PER_MINUTE - 6 * MILLISECONDS_PER_SECOND + 7, millis);
131         assertEquals("'" + time + "' input formatted.", "10mo 7d 19h 54m 54s 7ms", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
132         
133         // test unit not defined: z
134         try {
135             parser.parseMillis("5m  3d 7z -6s -4h 5m 10ms");
136             fail("NumberFormatException ('Illegal characters.') expected!");
137         } catch (NumberFormatException e) {
138             assertTrue("Expected NumberFormatException: " + e.toString(),
139                     e instanceof NumberFormatException && "Illegal characters.".equals(e.getMessage()));
140         }
141 
142         // test other illegal characters (start|end)
143         try {
144             parser.parseMillis("iiii5m 3d -6s -4h 5m");
145             fail("NumberFormatException ('Illegal characters.') expected!");
146         } catch (NumberFormatException e) {
147             assertTrue("Expected NumberFormatException: " + e.toString(),
148                     e instanceof NumberFormatException && "Illegal characters.".equals(e.getMessage()));
149         }
150 
151         // test other illegal characters (start|end)
152         try {
153             parser.parseMillis("5m 3d -6s -4h 5miiii");
154             fail("NumberFormatException ('Illegal characters.') expected!");
155         } catch (NumberFormatException e) {
156             assertTrue("Expected NumberFormatException: " + e.toString(),
157                     e instanceof NumberFormatException && "Illegal characters.".equals(e.getMessage()));
158         }
159 
160         // test null or empty string (=blank)
161         try {
162             parser.parseMillis(null);
163             fail("NumberFormatException ('Cannot parse a blank string.') expected!");
164         } catch (NumberFormatException e) {
165             assertTrue("Expected NumberFormatException: " + e.toString(),
166                     e instanceof NumberFormatException && "Cannot parse a blank string.".equals(e.getMessage()));
167         }
168         try {
169             parser.parseMillis("   ");
170             fail("NumberFormatException ('Cannot parse a blank string.') expected!");
171         } catch (NumberFormatException e) {
172             assertTrue("Expected NumberFormatException: " + e.toString(),
173                     e instanceof NumberFormatException && "Cannot parse a blank string.".equals(e.getMessage()));
174         }
175     }

191         assertEquals("'" + time + "' input formatted.", "1y 2mo 3d 4h 5m 6s", parser.format(millis, MILLISECONDS_FACTOR, TYPE_DAYS));
