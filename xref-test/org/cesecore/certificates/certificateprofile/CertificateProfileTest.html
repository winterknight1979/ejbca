415         CertificateProfile profile = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_NO_PROFILE);
416         CertificateProfile clone = (CertificateProfile)profile.clone();
417         @SuppressWarnings("unchecked")
418         HashMap<Object, Object> profmap = (HashMap<Object, Object>)profile.saveData();
419         @SuppressWarnings("unchecked")
420         HashMap<Object, Object> clonemap = (HashMap<Object, Object>)clone.saveData();
421         assertEquals(profmap.size(), clonemap.size());
422         clonemap.put("FOO", "BAR");
423         assertEquals(profmap.size()+1, clonemap.size());
424         profmap.put("FOO", "BAR");
425         assertEquals(profmap.size(), clonemap.size());
426         profmap.put("FOO", "FAR");
427         // Just changing value
428         assertEquals(profmap.size(), clonemap.size());
429         String profstr = (String)profmap.get("FOO");
430         String clonestr = (String)clonemap.get("FOO");
431         assertEquals("FAR", profstr);
432         assertEquals("BAR", clonestr);
433         CertificateProfile clone2 = (CertificateProfile)clone.clone();
434         @SuppressWarnings("unchecked")
435         HashMap<Object, Object> clonemap2 = (HashMap<Object, Object>)clone2.saveData();
436         // Added FOO, FAR to profmap and clonemap
437         assertEquals(clonemap2.size(), clonemap.size()-1);
438         assertEquals(clonemap2.size(), profmap.size()-1);
439     }

695         testInvalidKeySpecsInternal(true, keyPairRsa.getPublic(), new String[]{AlgorithmConstants.KEYALGORITHM_RSA}, new String[]{CertificateProfile.ANY_EC_CURVE}, new int[]{512});
696         // Test expected failure. ECDSA 256 bit key. RSA 512 allowed by certificate profile.
697         testInvalidKeySpecsInternal(false, keyPairEc.getPublic(), new String[]{AlgorithmConstants.KEYALGORITHM_RSA}, new String[]{CertificateProfile.ANY_EC_CURVE}, new int[]{512});
698         // Test expected failure. ECDSA 256 bit key. RSA 256,512 allowed by certificate profile.
699         testInvalidKeySpecsInternal(false, keyPairEc.getPublic(), new String[]{AlgorithmConstants.KEYALGORITHM_RSA}, new String[]{CertificateProfile.ANY_EC_CURVE}, new int[]{256,512});
700         // Test expected failure. ECDSA 256 bit key. ECDSA 512 allowed by certificate profile.
701         testInvalidKeySpecsInternal(false, keyPairEc.getPublic(), new String[]{AlgorithmConstants.KEYALGORITHM_ECDSA}, new String[]{CertificateProfile.ANY_EC_CURVE}, new int[]{512});

721         testInvalidKeySpecsInternal(true, keyPairEc.getPublic(), new String[]{AlgorithmConstants.KEYALGORITHM_ECDSA, AlgorithmConstants.KEYALGORITHM_RSA},
722                 new String[]{CertificateProfile.ANY_EC_CURVE, "secp256k1"}, new int[]{256,1024});

738     private void testInvalidKeySpecsInternal(final boolean expectedNoIllegalKeyException, final PublicKey publicKey, final String[] availableKeyAlgorithms, 
739             final String[] availableEcCurves, final int[] availableBitLengths) {
740         final CertificateProfile certificateProfile = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
741         certificateProfile.setAvailableKeyAlgorithms(availableKeyAlgorithms);
742         certificateProfile.setAvailableEcCurves(availableEcCurves);
743         certificateProfile.setAvailableBitLengths(availableBitLengths);

746             if (!expectedNoIllegalKeyException) {
747                 fail("Validation should not work with invalid key size and/or algoritmh,");
748             }
749         } catch (IllegalKeyException e) {
750             if (expectedNoIllegalKeyException) {
751                 fail("Key algorithm and spec should have been allowed by certificate profile.");
752             }
753         }
