50      public static String getHSMProvider() {
51          final File utimacoCSLinux = new File(UTIMACO_PKCS11_LINUX_LIB);
52          final File utimacoCSWindows = new File(UTIMACO_PKCS11_WINDOWS_LIB);
53          final File lunaSALinux64 = new File(LUNASA_PKCS11_LINUX_LIB);
54          final File lunaSALinux32 = new File(LUNASA_PKCS11_LINUX32_LIB);
55          final File protectServerLinux = new File(PROTECTSERVER_PKCS11_LINUX_LIB);
56          final File protectServerLinux64 = new File(PROTECTSERVER_PKCS11_LINUX64_LIB);
57          final File protectServerLinux32 = new File(PROTECTSERVER_PKCS11_LINUX32_LIB);
58          final File protectServerWindows = new File(PROTECTSERVER_PKCS11_WINDOWS_LIB);
59          String ret = null;
60          if (utimacoCSLinux.exists()) {
61              ret = "SunPKCS11-libcs2_pkcs11.so-slot1";
62          } else if (utimacoCSWindows.exists()) {
63              ret = "SunPKCS11-cs2_pkcs11.dll-slot1";
64          } else if (lunaSALinux64.exists()) {
65              ret = "SunPKCS11-libCryptoki2_64.so-slot1";
66          } else if (lunaSALinux32.exists()) {
67              ret = "SunPKCS11-libCryptoki2.so-slot1";
68          } else if ( protectServerLinux32.exists() || protectServerLinux64.exists() ||  protectServerLinux.exists()) {
69              ret = "SunPKCS11-libcryptoki.so-slot1";
70          } else if (protectServerWindows.exists()) {
71              ret = "SunPKCS11-cryptoki.dll-slot1";
72          }
73          // Override auto-detected properties if configuration exists
74          ret = getSystemTestsProperties().getProperty(PKCS11_SECURITY_PROVIDER, ret);
75          if (log.isDebugEnabled()) {
76              log.debug("getHSMProvider: "+ret);
77          }
78          return ret;
79      }

81      public static String getHSMLibrary() {
82          final File utimacoCSLinux = new File(UTIMACO_PKCS11_LINUX_LIB);
83          final File utimacoCSWindows = new File(UTIMACO_PKCS11_WINDOWS_LIB);
84          final File lunaSALinux64 = new File(LUNASA_PKCS11_LINUX_LIB);
85          final File lunaSALinux32 = new File(LUNASA_PKCS11_LINUX32_LIB);
86          final File protectServerLinux = new File(PROTECTSERVER_PKCS11_LINUX_LIB);
87          final File protectServerLinux64 = new File(PROTECTSERVER_PKCS11_LINUX64_LIB);
88          final File protectServerLinux32 = new File(PROTECTSERVER_PKCS11_LINUX32_LIB);
89          final File protectServerWindows = new File(PROTECTSERVER_PKCS11_WINDOWS_LIB);
90          String ret = null;
91          if (utimacoCSLinux.exists()) {
92              ret = utimacoCSLinux.getAbsolutePath();
93          } else if (utimacoCSWindows.exists()) {
94              ret = utimacoCSWindows.getAbsolutePath();
95          } else if (lunaSALinux64.exists()) {
96              ret = lunaSALinux64.getAbsolutePath();
97          } else if (lunaSALinux32.exists()) {
98              ret = lunaSALinux32.getAbsolutePath();
99          } else if (protectServerLinux64.exists()) {
100             ret = protectServerLinux64.getAbsolutePath();
101         } else if (protectServerLinux32.exists()) {
102             ret = protectServerLinux32.getAbsolutePath();
103         } else if (protectServerLinux.exists()) {
104             ret = protectServerLinux.getAbsolutePath();
105         } else if (protectServerWindows.exists()) {
106             ret = protectServerWindows.getAbsolutePath();
107         }
108         // Override auto-detected properties if configuration exists
109         ret = getSystemTestsProperties().getProperty(PKCS11_LIBRARY, ret);
110         if (log.isDebugEnabled()) {
111             log.debug("getHSMLibrary: "+ret);
112         }
113         return ret;
114     }

140     /** @return properties defined in systemtests.properties for override of non-default environments. See also org.cesecore.SystemTestsConfiguration. */
141     private static Properties getSystemTestsProperties() {
142         if (properties==null) {
143             properties = new Properties();
144             try {
145                 final InputStream is = PKCS11TestUtils.class.getResourceAsStream(PROPERTYFILE);
146                 if (is!=null) {
147                     properties.load(is);
148                     is.close();
149                 }
150             } catch (IOException e) {
151                 log.warn(e.getMessage());
152             }
153             if (properties.isEmpty()) {
154                 log.info(PROPERTYFILE + " was not detected. Defaults will be used.");
155             } else {
156                 log.info(PROPERTYFILE + " was detected.");
157             }
158         }
159         return properties;
160     }
