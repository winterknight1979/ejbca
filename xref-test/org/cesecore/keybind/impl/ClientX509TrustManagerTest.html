46  public class ClientX509TrustManagerTest {
47  
48      private static final Logger log = Logger.getLogger(ClientX509TrustManagerTest.class);
49  
50      @Rule
51      public final TestWatcher traceLogMethodsRule = new TestWatcher() {
52          @Override
53          protected void starting(final Description description) {
54              log.trace(">" + description.getMethodName());
55              super.starting(description);
56          };
57          @Override
58          protected void finished(final Description description) {
59              log.trace("<" + description.getMethodName());
60              super.finished(description);
61          }
62      };
63  
64      @BeforeClass
65      public static void beforeClass() throws Throwable {
66          CryptoProviderTools.installBCProviderIfNotAvailable();
67      }
68  
69      @Test
70      public void testFirstEncounteredServerCertificate() throws InvalidAlgorithmParameterException, OperatorCreationException, CertificateException {
71          final KeyPair keyPair = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA);
72          final X509Certificate x509Certificate1 = CertTools.genSelfCert("CN=ClientX509TrustManagerTest1", 365, null, keyPair.getPrivate(), keyPair.getPublic(), AlgorithmConstants.SIGALG_SHA256_WITH_RSA, false);
73          final X509Certificate x509Certificate2 = CertTools.genSelfCert("CN=ClientX509TrustManagerTest2", 365, null, keyPair.getPrivate(), keyPair.getPublic(), AlgorithmConstants.SIGALG_SHA256_WITH_RSA, false);
74          final Collection<X509Certificate> trust1 = Arrays.asList(new X509Certificate[]{x509Certificate1});
75          final List<Collection<X509Certificate>> trustedChains = new ArrayList<>();
76          trustedChains.add(trust1);
77          final ClientX509TrustManager clientX509TrustManager = new ClientX509TrustManager(trustedChains);
78          assertNull("Validated server TLS certificate was encountered before validation.", clientX509TrustManager.getEncounteredServerCertificateChain());
79          try {
80              clientX509TrustManager.checkServerTrusted(new X509Certificate[] { x509Certificate2 }, null);
81              fail("Untrusted certificate should not pass validation.");
82          } catch (CertificateException e) {
83              log.debug("Expected exception: " + e.getMessage());
84          }
85          // We should be able to see the remote TLS certificate even if it was not trusted
86          assertNotNull("Validated server TLS certificate was not encountered.", clientX509TrustManager.getEncounteredServerCertificateChain());
87          assertNotNull("Validated server TLS certificate was not encountered.", clientX509TrustManager.getEncounteredServerCertificateChain().get(0));
88          assertEquals("Validated server TLS certificate was not encountered.", CertTools.getFingerprintAsString(x509Certificate2),
89                  CertTools.getFingerprintAsString(clientX509TrustManager.getEncounteredServerCertificateChain().get(0)));
90          clientX509TrustManager.checkServerTrusted(new X509Certificate[] { x509Certificate1 }, null);
91          // Check trusted validation as well
92          final ClientX509TrustManager clientX509TrustManager2 = new ClientX509TrustManager(trustedChains);
93          clientX509TrustManager2.checkServerTrusted(new X509Certificate[] { x509Certificate1 }, null);
94          assertNotNull("Validated server TLS certificate was not encountered.", clientX509TrustManager2.getEncounteredServerCertificateChain());
95          assertNotNull("Validated server TLS certificate was not encountered.", clientX509TrustManager2.getEncounteredServerCertificateChain().get(0));
96          assertEquals("Validated server TLS certificate was not encountered.", CertTools.getFingerprintAsString(x509Certificate1),
97                  CertTools.getFingerprintAsString(clientX509TrustManager2.getEncounteredServerCertificateChain().get(0)));
98          // Check that the "first" encountered will change if validation would be done with another cert
99          try {
100             clientX509TrustManager2.checkServerTrusted(new X509Certificate[] { x509Certificate2 }, null);
101             fail("Untrusted certificate should not pass validation.");
102         } catch (CertificateException e) {
103             log.debug("Expected exception: " + e.getMessage());
104         }
105         assertEquals("Validated server TLS certificate was not encountered.", CertTools.getFingerprintAsString(x509Certificate2),
106                 CertTools.getFingerprintAsString(clientX509TrustManager2.getEncounteredServerCertificateChain().get(0)));
107     }
