91      @Test
92      public void testOcspSigningCertificateValidationPositives() throws IOException, InvalidAlgorithmParameterException, InvalidKeyException, NoSuchAlgorithmException, SignatureException, IllegalStateException, NoSuchProviderException, OperatorCreationException, CertificateException {
93          AvailableExtendedKeyUsagesConfiguration ekuConfig = new AvailableExtendedKeyUsagesConfiguration();
94          assertTrue("KU=digitalSignature and EKU=id_kp_OCSPSigning should be treated as a valid OCSP singing certificate.",
95                  OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.digitalSignature, ekuExtensionOnly), ekuConfig));
96          assertTrue("KU=digitalSignature and EKU=id_kp_OCSPSigning should be treated as a valid OCSP singing certificate.",
97                  OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.digitalSignature + X509KeyUsage.cRLSign, ekuExtensionOnly), ekuConfig));
98          assertTrue("KU=nonRepudiation and EKU=id_kp_OCSPSigning should be treated as a valid OCSP singing certificate.",
99                  OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.nonRepudiation + X509KeyUsage.cRLSign, ekuExtensionOnly), ekuConfig));
100         assertTrue("KU=digitalSignature+nonRepudiation and EKU=id_kp_OCSPSigning should be treated as a valid OCSP singing certificate.",
101                 OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.digitalSignature + X509KeyUsage.nonRepudiation, ekuExtensionOnly), ekuConfig));
102     }

113     @Test
114     public void testOcspSigningCertificateValidationNegatives() throws IOException, InvalidAlgorithmParameterException, InvalidKeyException, NoSuchAlgorithmException, SignatureException, IllegalStateException, NoSuchProviderException, OperatorCreationException, CertificateException {
115         AvailableExtendedKeyUsagesConfiguration ekuConfig = new AvailableExtendedKeyUsagesConfiguration();
116         assertFalse("KU!=digitalSignature|nonRepudiation and EKU=id_kp_OCSPSigning should be treated as an invalid OCSP singing certificate.",
117                 OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.keyAgreement + X509KeyUsage.cRLSign, ekuExtensionOnly), ekuConfig));
118         assertFalse("KU=digitalSignature and EKU!=id_kp_OCSPSigning should be treated as an invalid OCSP singing certificate.",
119                 OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.digitalSignature, null), ekuConfig));
120         assertFalse("KU=nonRepudiation and EKU!=id_kp_OCSPSigning should be treated as an invalid OCSP singing certificate.",
121                 OcspKeyBinding.isOcspSigningCertificate(getCertificate(X509KeyUsage.nonRepudiation, null), ekuConfig));
122  }

134     /** @param keyUsage usage
135      * @param extensions ext
136      * @return A self-signed certificate. 
137      * @throws InvalidAlgorithmParameterException fail
138      * @throws InvalidKeyException fail
139      * @throws NoSuchAlgorithmException fail 
140      * @throws SignatureException fail
141      * @throws IllegalStateException fail
142      * @throws NoSuchProviderException fail
143      * @throws OperatorCreationException fail
144      * @throws CertificateException fail
145      * @throws IOException fail */
146     private X509Certificate getCertificate(final int keyUsage, final List<Extension> extensions) throws InvalidAlgorithmParameterException, InvalidKeyException, NoSuchAlgorithmException,
147         SignatureException, IllegalStateException, NoSuchProviderException, OperatorCreationException, CertificateException, IOException {
148         final KeyPair keyPair = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA);
149         return CertTools.genSelfCertForPurpose("CN=OcspSinger", 365, null, keyPair.getPrivate(), keyPair.getPublic(), "SHA256WithRSA", false,
150                 keyUsage, null, null, BouncyCastleProvider.PROVIDER_NAME, true, extensions);
151     }
152 
153     /** @return An extended key usage extension with id_kp_OCSPSigning set. 
154      * @throws IOException fail*/
155     private static Extension getExtendedKeyUsageExtension() throws IOException {
156         final ASN1Encodable usage = KeyPurposeId.getInstance(KeyPurposeId.id_kp_OCSPSigning);
157         final ASN1Sequence seq = ASN1Sequence.getInstance(new DERSequence(usage));
158         return new Extension(Extension.extendedKeyUsage, true, seq.getEncoded());
159     }

161     @Test
162     public void testProperties() {
163         final OcspKeyBinding keybind = new OcspKeyBinding();
164         // Check defaults (please update if changed)
165         assertTrue("getIncludeSignCert", keybind.getIncludeSignCert());
166         assertFalse("getRequireTrustedSignature", keybind.getRequireTrustedSignature());
167         assertEquals("getResponderIdType", ResponderIdType.KEYHASH, keybind.getResponderIdType());
168         // Test getters and setters
169         keybind.setNonExistingGood(true);
170         keybind.setIncludeCertChain(false);
171         keybind.setIncludeSignCert(false);
172         keybind.setRequireTrustedSignature(true);
173         keybind.setResponderIdType(ResponderIdType.NAME);
174         assertTrue("setNonExistingGood", keybind.getNonExistingGood());
175         assertFalse("setIncludeSignCert", keybind.getIncludeSignCert());
176         assertFalse("setIncludeCertChain", keybind.getIncludeCertChain());
177         assertTrue("setRequireTrustedSignature", keybind.getRequireTrustedSignature());
178         assertEquals("setResponderIdType", ResponderIdType.NAME, keybind.getResponderIdType());
179     }
