134     @Test
135     public void test000Setup() throws Exception {
136         LOG.trace(">test000Setup");
137         logMemStats();
138         LOG.debug("Allocating memory..");
139         VARCHAR_80B = getClob(80);
140         VARCHAR_250B = getClob(250);
141         VARCHAR_400B = getClob(400);
142         VARCHAR_2000B = getClob(2000);
143         CLOB_10KiB = getClob(10 * 1024);
144         CLOB_100KiB = getClob(100 * 1024);
145         CLOB_1MiB = getClob(1024 * 1024);
146         LOG.debug("VARCHAR_80B   is      " + VARCHAR_80B.length() + " chars and     " + VARCHAR_80B.getBytes().length + " bytes.");
147         LOG.debug("VARCHAR_250B  is     " + VARCHAR_250B.length() + " chars and     " + VARCHAR_250B.getBytes().length + " bytes.");
148         LOG.debug("VARCHAR_400B  is     " + VARCHAR_400B.length() + " chars and     " + VARCHAR_400B.getBytes().length + " bytes.");
149         LOG.debug("VARCHAR_2000B is    " + VARCHAR_2000B.length() + " chars and    " + VARCHAR_2000B.getBytes().length + " bytes.");
150         LOG.debug("CLOB_10KiB    is   " + CLOB_10KiB.length()     + " chars and   " + CLOB_10KiB.getBytes().length + " bytes.");
151         LOG.debug("CLOB_100KiB   is  " + CLOB_100KiB.length()     + " chars and  " + CLOB_100KiB.getBytes().length + " bytes.");
152         LOG.debug("CLOB_1MiB     is " + CLOB_1MiB.length()        + " chars and " + CLOB_1MiB.getBytes().length + " bytes.");
153         LOG.debug("Filling HashMaps..");
154         // Make them mimic "real" UpgradeablaHashMap's, because that's what is usaully stored
155         Float f = Float.valueOf("9999"); // high number so not to trigger any UpgradeableHashMap.upgrade()'s
156         HASHMAP_200K.put("object", new String(getLob(196 * 1024))); // It need to be less than 200KiB in Serialized format..
157         HASHMAP_200K.put(UpgradeableDataHashMap.VERSION, f);
158         // make it look also like a UserData.ExtendedInformation
159         HASHMAP_200K.put(ExtendedInformation.TYPE, Integer.valueOf(ExtendedInformation.TYPE_BASIC));
160         HASHMAP_1M.put("object", new String(getLob(996 * 1024))); // It need to be less than 1MiB in Serialized format.. 
161         HASHMAP_1M.put(UpgradeableDataHashMap.VERSION, f);
162         HASHMAP_1M.put(ExtendedInformation.TYPE, Integer.valueOf(ExtendedInformation.TYPE_BASIC));
163         logMemStats();
164         LOG.trace("<test000Setup");
165     }

167     private byte[] getLob(int size) {
168         byte[] ret = new byte[size];
169         Arrays.fill(ret, (byte) '0');
170         return ret;
171     }
172 
173     private String getClob(int size) {
174         return new String(getLob(size));
175     }

634     /**
635      * Outputs which method it is run from.
636      * Validates that all getters on the entity that is annotated with @javax.persistence.Column is set. 
637      * Commits the entity in one transaction and then removes it in another transaction.
638      * @param entity Entity
639      */
640     private void storeAndRemoveEntity(Object entity) {
641         LOG.trace(">storeAndRemoveEntity");
642         logMemStats();
643         try {
644             Class<?> entityClass = entity.getClass();
645             LOG.info("  - verifying that all getter has an assigned value for " + entityClass.getName());
646             boolean allOk = true;
647             for (Method m : entityClass.getDeclaredMethods()) {
648                 for (Annotation a : m.getAnnotations()) {
649                     if (a.annotationType().equals(javax.persistence.Column.class) && m.getName().startsWith("get")) {
650                         try {
651                             m.setAccessible(true);
652                             if (m.invoke(entity) == null) {
653                                 LOG.warn(m.getName() + " was annotated with @Column, but value was null. Test should be updated!");
654                                 allOk = false;
655                             }
656                         } catch (Exception e) {
657                             LOG.error(m.getName() + " was annotated with @Column and could not be read. " + e.getMessage());
658                             allOk = false;
659                         }
660                     }
661                 }
662             }
663             assertTrue("There is a problem with a @Column annotated getter. Please refer to log output for further info.", allOk);
664             LOG.info("  - adding entity.");
665             EntityTransaction transaction = entityManager.getTransaction();
666             transaction.begin();
667             entityManager.persist(entity);
668             transaction.commit();
669             LOG.info("  - removing entity.");
670             transaction = entityManager.getTransaction();
671             transaction.begin();
672             entityManager.remove(entity);
673             transaction.commit();
674         } finally {
675             if (entityManager.getTransaction().isActive()) {
676                 entityManager.getTransaction().rollback();
677             }
678             logMemStats();
679         }
680         LOG.trace("<storeAndRemoveEntity");
681     }
