13  package org.ejbca.core.ejb.profiles;
14  
15  import static org.junit.Assert.assertEquals;
16  import static org.junit.Assert.fail;
17  
18  import org.cesecore.profiles.ProfileData;
19  import org.cesecore.util.ui.PropertyValidationException;
20  import org.ejbca.core.model.approval.profile.AccumulativeApprovalProfile;
21  import org.ejbca.core.model.approval.profile.ApprovalProfile;
22  import org.junit.Test;
23  
24  /**
25   * Unit tests for the ProfileData class
26   * 
27   * @version $Id: ProfileDataTest.java 26076 2017-06-26 12:49:12Z mikekushner $
28   *
29   */
30  public class ProfileDataTest {
31  
32      /**
33       * This test creates a ProfileData object using an ApprovalProfile, then extracts the profile back out in order to verify integrity. 
34       * @throws PropertyValidationException FAil
35       */
36      @Test
37      public void testApprovalProfile() throws PropertyValidationException {
38          final int numberOfApprovalsRequired = 4711;
39          final String profileName = "testApprovalProfile";
40          AccumulativeApprovalProfile approvalProfile = new AccumulativeApprovalProfile(profileName);
41          approvalProfile.initialize();
42          approvalProfile.setNumberOfApprovalsRequired(numberOfApprovalsRequired);
43          ProfileData profileData = new ProfileData(0, approvalProfile);
44          ApprovalProfile retrievedApprovalProfile = null;
45          try {
46              retrievedApprovalProfile = (ApprovalProfile) profileData.getProfile();
47          } catch (ClassCastException e) {
48              fail("Retrived Profile wasn't returned as an ApprovalProfile");
49          }
50          assertEquals("Profile name was lost during conversion", profileName, retrievedApprovalProfile.getProfileName());
51          AccumulativeApprovalProfile recastProfile = null;
52          try {
53              recastProfile = (AccumulativeApprovalProfile) retrievedApprovalProfile;
54          } catch (ClassCastException e) {
55              fail("Retrieved profile was not instansiated as an " + AccumulativeApprovalProfile.class.getCanonicalName());
56          }
57          assertEquals("The number of required approvals was not retained.", numberOfApprovalsRequired, recastProfile.getNumberOfApprovalsRequired());
58      }
