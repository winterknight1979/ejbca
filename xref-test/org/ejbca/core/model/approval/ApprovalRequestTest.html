83          AuthenticationToken token = new X509CertificateAuthenticationToken(testcert);
84  
85          AccumulativeApprovalProfile approvalProfile = new AccumulativeApprovalProfile("AccumulativeApprovalProfile");

91      	ObjectOutputStream oos = new ObjectOutputStream(baos);
92      	oos.writeObject(ar);
93      	oos.flush();
94      	String result = new String(Base64.encode(baos.toByteArray(),false));
95  
96      	
97      	ApprovalRequest readrequest = getApprovalRequest(result);
98      	assertTrue(readrequest.getApprovalType() == ApprovalDataVO.APPROVALTYPE_DUMMY);
99      	assertTrue(readrequest.getApprovalRequestType() == ApprovalRequest.REQUESTTYPE_SIMPLE);
100     	assertTrue(readrequest.getRequestSignature() == null);
101     	assertTrue(CertTools.getSerialNumber(readrequest.getRequestAdminCert()).equals(CertTools.getSerialNumber(testcert)));
102     	assertTrue(readrequest.getCAId() == 1);
103     	assertTrue(readrequest.getEndEntityProfileId() == 2);
104     	assertTrue(readrequest.getApprovalProfile()!=null);
105     	assertTrue(readrequest.getApprovalProfile().getProfileName().equals(approvalProfile.getProfileName()));
106     	assertTrue(!readrequest.isExecutable());
107 		
108 	}

110     private  ApprovalRequest getApprovalRequest(String data) throws IOException, ClassNotFoundException {
111         ApprovalRequest retval = null; 
112         ObjectInputStream ois = null;
113         try {
114             ois = new ObjectInputStream(new ByteArrayInputStream(Base64.decode(data.getBytes())));
115             retval= (ApprovalRequest) ois.readObject();
116         }finally {
117             if(ois != null) {
118                 ois.close();
119             }
120         }
121         return retval;
122     }

156     @Test
157     public void testEditedByMe() throws Exception {
158         X509Certificate testcert = CertTools.getCertfromByteArray(testcertenc, X509Certificate.class);
159         AuthenticationToken token = new X509CertificateAuthenticationToken(testcert);
160         
161         X509Certificate testcertDup = CertTools.getCertfromByteArray(testcertenc, X509Certificate.class);
162         AuthenticationToken tokenDup = new X509CertificateAuthenticationToken(testcertDup);
163         
164         X509Certificate testcert2 = CertTools.getCertfromByteArray(testcertenc2, X509Certificate.class);
165         AuthenticationToken token2 = new X509CertificateAuthenticationToken(testcert2);
166         
167         AuthenticationToken token3 = new PublicAccessAuthenticationToken("127.0.0.1", true);
168         
169         AccumulativeApprovalProfile approvalProfile = new AccumulativeApprovalProfile("AccumulativeApprovalProfile");
170         approvalProfile.initialize();
171         approvalProfile.setNumberOfApprovalsRequired(2);
172         DummyApprovalRequestApprovalRequest.html#DummyApprovalRequest">DummyApprovalRequest ar = new DummyApprovalRequest(token, null, 1, 2, false, approvalProfile);
173         
174         assertFalse("Fresh approval request should not say it has been edited.", ar.isEditedByMe(token));
175         assertFalse(ar.isEditedByMe(tokenDup));
176         assertFalse(ar.isEditedByMe(token2));
177         assertFalse(ar.isEditedByMe(token3));
178         
179         ar.addEditedByAdmin(token);
180         assertTrue(ar.isEditedByMe(token));
181         assertTrue(ar.isEditedByMe(tokenDup));
182         assertFalse(ar.isEditedByMe(token2));
183         assertFalse(ar.isEditedByMe(token3));
184         
185         ar.addEditedByAdmin(token2);
186         assertFalse(ar.isEditedByMe(token)); // no longer the last admin who edited it, so this admin may edit again
187         assertFalse(ar.isEditedByMe(tokenDup));
188         assertTrue(ar.isEditedByMe(token2));
189         assertFalse(ar.isEditedByMe(token3));
190     }
