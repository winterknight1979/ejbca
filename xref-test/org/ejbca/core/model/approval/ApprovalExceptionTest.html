14  package org.ejbca.core.model.approval;
15  
16  import static org.junit.Assert.assertEquals;
17  
18  import org.cesecore.ErrorCode;
19  import org.ejbca.core.EjbcaException;
20  import org.junit.Test;
21  
22  /**
23   * @version $Id: ApprovalExceptionTest.java 22139 2015-11-03 10:41:56Z mikekushner $
24   */
25  public class ApprovalExceptionTest {
26  
27  	/**
28  	 * Verify that nested ApprovalExceptions propagate errorCode.
29  	 */
30  	@Test
31  	public void testErrorCode() {
32  		ApprovalException approvalException = new ApprovalException(ErrorCode.APPROVAL_ALREADY_EXISTS, "JUnit test message..");
33  		EjbcaException ejbcaException = new EjbcaException(approvalException);
34  		assertEquals("EjbcaException did not inherit ErrorCode.", ErrorCode.APPROVAL_ALREADY_EXISTS, ejbcaException.getErrorCode());
35  	}
36  }
