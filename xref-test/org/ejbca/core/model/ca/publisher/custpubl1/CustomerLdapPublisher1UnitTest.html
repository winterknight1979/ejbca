119         final String expectedLoginPassword = "foo123#123";
120         final boolean expectedUseSSL = true;
121         final long expectedConnectionTimeout = 5001;
122         final long expectedReadTimeout = 30001;
123         final long expectedStoreTimeout = 60001;
124         final boolean expectedLogConnectionTests = true;
125         
126         final Properties config = new Properties();
127         config.setProperty("hostnames", expectedHostnames);
128         config.setProperty("port", expectedPort);
129         config.setProperty("basedn", expectedBaseDN);
130         config.setProperty("logindn", expectedLoginDN);
131         config.setProperty("loginpassword", expectedLoginPassword);
132         config.setProperty("usessl", String.valueOf(expectedUseSSL));
133         config.setProperty("connectiontimeout", String.valueOf(expectedConnectionTimeout));
134         config.setProperty("readtimeout", String.valueOf(expectedReadTimeout));
135         config.setProperty("storetimeout", String.valueOf(expectedStoreTimeout));
136         config.setProperty("logconnectiontests", String.valueOf(expectedLogConnectionTests));
137         
138         assertFalse("before inited", instance.isInited());
139         instance.init(config);
140         assertTrue("after inited", instance.isInited());
141         
142         assertEquals("hostnames", Arrays.asList(expectedHostnames), instance.getHostnames());
143         assertEquals("port", expectedPort, instance.getPort());
144         assertEquals("basedn", expectedBaseDN, instance.getBaseDN());
145         assertEquals("logindn", expectedLoginDN, instance.getLoginDN());
146         assertEquals("loginpassword", expectedLoginPassword, instance.getLoginPassword());
147         assertEquals("usessl", expectedUseSSL, instance.isUseSSL());
148         assertEquals("timeout", expectedConnectionTimeout, instance.getTimeout());
149         assertEquals("connectiontimeout:bind", expectedConnectionTimeout, instance.getLdapBindConstraints().getTimeLimit());
150         assertEquals("connectiontimeout:conn", expectedConnectionTimeout, instance.getLdapConnectionConstraints().getTimeLimit());
151         assertEquals("connectiontimeout:disconn", expectedConnectionTimeout, instance.getLdapDisconnectConstraints().getTimeLimit());
152         assertEquals("readtimeout", expectedReadTimeout, instance.getLdapSearchConstraints().getTimeLimit());
153         assertEquals("storetimeout", expectedStoreTimeout, instance.getLdapStoreConstraints().getTimeLimit());
154         assertEquals("logconnectiontests", expectedLogConnectionTests, instance.isLogConnectionTests());
