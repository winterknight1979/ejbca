162   @SuppressWarnings("deprecation")
163   @Test
164   public void testConvertAndMerge() {
165     LOG.trace(">testConvertAndMerge");
166     final AccessSet asNew = makeNewAccessSet();
167     // Merge with an old AccessSet to force downgrade
168     final AccessSet asOld =
169         new AccessSet(Arrays.asList("/a", "/f", "/nonexistentold"));
170     final AccessSet merged = new AccessSet(asNew, asOld);
171     assertTrue(merged.isAuthorized("/a"));
172     assertFalse(merged.isAuthorized("/b"));
173     assertTrue(merged.isAuthorized("/b/c"));
174     assertTrue(merged.isAuthorized("/b/c/d"));
175     assertFalse(merged.isAuthorized("/b/c/e"));
176     assertTrue(merged.isAuthorized("/f"));
177     assertFalse(
178         merged.isAuthorized(
179             "/nonexistent")); // the converter does not allow non-existent
180                               // resources
181     assertTrue(
182         merged.isAuthorized(
183             "/nonexistentold")); // this rule comes from the old AccessSet, so
184                                  // it's left untouched
185     LOG.trace("<testConvertAndMerge");
186   }

188   /**
189    * Creates an AccessSet with the legacy representation of access rules.
190    *
191    * @param resources resources
192    * @return access set
193    */
194   @SuppressWarnings("deprecation")
195   private AccessSet makeLegacyAccessSet(final String... resources) {
196     final Collection<String> col = new ArrayList<>();
197     for (final String resource : resources) {
198       col.add(resource);
199     }
200     return new AccessSet(col);
201   }
202 
203   @SuppressWarnings("deprecation")
204   private AccessSet makeNewAccessSet() {
205     final Set<String> allResources = new HashSet<>();
206     allResources.add("/a/");
207     allResources.add("/b/");
208     allResources.add("/b/c/");
209     allResources.add("/b/c/d/");
210     allResources.add("/b/c/e/");
211     allResources.add("/f/");
212     final HashMap<String, Boolean> accessRules = new HashMap<>();
213     accessRules.put("/a/", true);
214     accessRules.put("/b/c/", true);
215     accessRules.put("/b/c/e/", false);
216     accessRules.put("/nonexistent/", true);
217     return AccessSet.fromAccessRules(accessRules, allResources);
218   }
