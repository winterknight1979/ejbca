97      /** This test prints the serialized form of a CertificateData object, and was used to generated the data above. 
98       * @throws Exception on fail*/
99      @Test
100     public void testSerializeCurrent() throws Exception {
101         log.trace(">testSerializeCurrent");
102         final KeyPair kp = KeyTools.genKeys("1024", "RSA");
103         final Certificate cert = CertTools.genSelfCert("CN=certuser", 10*365, null, kp.getPrivate(), kp.getPublic(), "SHA256withRSA", false);
104         final CertificateData certData = new CertificateData(cert, kp.getPublic(), "certuser", "1234567812345678", CertificateConstants.CERT_ACTIVE,
105                 CertificateConstants.CERTTYPE_ENDENTITY, CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER, EndEntityConstants.NO_END_ENTITY_PROFILE,
106                 null, new Date().getTime(), false, true);
107         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
108         final Base64OutputStream b64os = new Base64OutputStream(baos);
109         final ObjectOutputStream oos = new ObjectOutputStream(b64os);
110         oos.writeObject(certData);
111         oos.close();
112         b64os.close();
113         log.info("Base 64 of serialized CertData is: " + baos.toString("US-ASCII"));
114         log.trace("<testSerializeCurrent");
115     }

117     @Test
118     public void testDeserializeOld() throws Exception {
119         log.trace(">testDeserializeOld");
120         final ByteArrayInputStream bais = new ByteArrayInputStream(OLD_DATA.getBytes("US-ASCII"));
121         final Base64InputStream b64is = new Base64InputStream(bais);
122         final ObjectInputStream ois = new ObjectInputStream(b64is);
123         final CertificateData certData = (CertificateData) ois.readObject();
124         ois.close();
125         assertEquals("certuser", certData.getUsername()); // unrelated column. should not be affected
126         assertNull("End Entity Profile Id in CertificateData with old serialization.", certData.getEndEntityProfileId());
127         assertEquals(EndEntityConstants.NO_END_ENTITY_PROFILE, certData.getEndEntityProfileIdOrZero());
128         log.trace("<testDeserializeOld");
129     }
