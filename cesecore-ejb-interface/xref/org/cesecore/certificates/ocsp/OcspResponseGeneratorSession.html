13  package org.cesecore.certificates.ocsp;
14  
15  import java.security.cert.X509Certificate;
16  import org.bouncycastle.cert.ocsp.OCSPException;
17  import org.cesecore.certificates.ocsp.exception.MalformedRequestException;
18  import org.cesecore.certificates.ocsp.logging.AuditLogger;
19  import org.cesecore.certificates.ocsp.logging.TransactionLogger;
20  
21  /**
22   * This interface is used to generate OCSP responses.
23   *
24   * @see <a
25   *     href="https://www.cesecore.eu/mediawiki/index.php/Functional_Specifications_(ADV_FSP)#OCSP_Response_Generation">OCSP
26   *     Response Generation</a>
27   * @version $Id: OcspResponseGeneratorSession.java 22476 2015-12-17 13:01:45Z
28   *     jeklund $
29   */
30  public interface OcspResponseGeneratorSession {
31  
32    /**
33     * This method delivers an OCSP response to a given request, as provided in
34     * the byte[] parameter.
35     *
36     * @param requestBytes a byte array representing an encoded OCSPRequest.
37     * @param requestCertificates An array of Certificates from the original
38     *     HttpServletRequest
39     * @param remoteAddress Remote address, most likely extracted from the
40     *     HttpServletRequest
41     * @param xForwardedFor Value of X-Forwarded-For header if it was present in
42     *     the request.
43     * @param requestUrl URL
44     * @param auditLogger The AuditLogger to use for this transaction
45     * @param transactionLogger The TransactionLogger to use for this transaction
46     * @return a signed and encoded OCSPResponse wrapped in an
47     *     OcspResponseInformation object
48     * @throws MalformedRequestException if the request byte array was invalid.
49     * @throws OCSPException if OCSP response generation fails
50     */
51    OcspResponseInformation getOcspResponse(
52        byte[] requestBytes,
53        X509Certificate[] requestCertificates,
54        String remoteAddress,
55        String xForwardedFor,
56        StringBuffer requestUrl,
57        AuditLogger auditLogger,
58        TransactionLogger transactionLogger)
59        throws MalformedRequestException, OCSPException;
60  
61    /** Reloads the cache of OCSP signers. */
62    void reloadOcspSigningCache();
63  
64    /**
65     * Reloads the cache of OCSP extensions (including extension specific caches,
66     * e.g. the CT OCSP response extensions cache).
67     */
68    void reloadOcspExtensionsCache();
69  
70    /**
71     * Clears CT fail fast cache. If CT is not supported in this build, then it
72     * does nothing.
73     */
74    void clearCTFailFastCache();
75  }
