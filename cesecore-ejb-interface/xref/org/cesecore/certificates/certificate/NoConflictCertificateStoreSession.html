13  package org.cesecore.certificates.certificate;
14  
15  import java.math.BigInteger;
16  import java.util.Collection;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  import org.cesecore.certificates.crl.RevokedCertInfo;
21  
22  /**
23   * Interface for NoConflictCertificateStoreSession.
24   * 
25   * These methods call CertificateStoreSession for certificates that are plain CertificateData entities.
26   * See {@link CertificateStoreSession} for method descriptions.
27   * 
28   * <p>For NoConflictCertificateData the methods perform additional logic to check that it gets the most recent
29   * entry if there's more than one (taking permanent revocations into account), and for updates it
30   * appends new entries instead of updating existing ones. 
31   * 
32   * @version $Id: NoConflictCertificateStoreSession.java 29433 2018-07-02 16:55:26Z mikekushner $
33   */
34  public interface NoConflictCertificateStoreSession  {
35  
36      /** @param issuerDN issuer DN of the desired certificate.
37       * @param serno  serial number of the desired certificate!
38       * @return Status
39       * @see CertificateStoreSession#getStatus */
40      CertificateStatus getStatus(String issuerDN, BigInteger serno);
41      
42      /**
43       * Gets full certificate meta data for the cert specified by issuer DN and serial number.
44       * 
45       * @param issuerdn issuer DN of the desired certificate.
46       * @param certserno serial number of the desired certificate!
47       * @return the sought certificate, or null if not found
48       */
49      CertificateDataWrapper getCertificateDataByIssuerAndSerno(String issuerdn, BigInteger certserno);
50  
51      /**
52       * EJBCA expects all certificate entities to have a fingerprint.
53       * This method generates a dummy fingerprint, to be used in NoConflictCertificateData and for associated publisher queue entries.
54       * @param issuerdn issuer DN of the desired certificate.
55       * @param certserno serial number of the desired certificate!
56       * @return Hex encoded fingerprint. It is unique per issuerdn/serial.
57       */
58      String generateDummyFingerprint(String issuerdn, BigInteger certserno);
59  
60      /** @param issuerdn issuerdn issuer DN of the desired certificate
61       * @param lastbasecrldate Date of last CRL
62       * @return List if revoked certs
63       * @see CertificateStoreSession#listRevokedCertInfo */
64      Collection<RevokedCertInfo> listRevokedCertInfo(String issuerdn, long lastbasecrldate);
65      
66      /** @param admin Auth token
67       * @param fingerprint FP
68       * @param status Status
69       * @return Success/fail
70       * @throws AuthorizationDeniedException If unauthorized 
71       * @see CertificateStoreSession#setStatus */
72      boolean setStatus(AuthenticationToken admin, String fingerprint, int status) throws AuthorizationDeniedException;
73  
74      /**
75       * @return true if the CA is a throw-away CA which allow revocation of non-existing entries.
76       * @param issuerDN Subject DN of CA to check.
77       */
78      boolean canRevokeNonExisting(String issuerDN);
79  
80  }
