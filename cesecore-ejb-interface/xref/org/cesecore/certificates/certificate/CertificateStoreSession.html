13  package org.cesecore.certificates.certificate;
14  
15  import java.math.BigInteger;
16  import java.security.cert.Certificate;
17  import java.security.cert.X509Certificate;
18  import java.util.Collection;
19  import java.util.Date;
20  import java.util.List;
21  import java.util.Set;
22  import org.cesecore.authentication.tokens.AuthenticationToken;
23  import org.cesecore.authorization.AuthorizationDeniedException;
24  import org.cesecore.certificates.crl.RevokedCertInfo;
25  
26  /**
27   * Interface for certificate store operations Stores certificate in the local
28   * database using Certificate JPA Beans.
29   *
30   * <p>IMPORTANT: No publishing to publishersession from setRevokeStatus
31   *
32   * @version $Id: CertificateStoreSession.java 29011 2018-05-23 13:41:10Z
33   *     jekaterina_b_helmes $
34   */
35  public interface CertificateStoreSession {
36  
37    /**
38     * Lists fingerprint (primary key) of ALL certificates in the database. NOTE:
39     * Caution should be taken with this method as execution may be very heavy
40     * indeed if many certificates exist in the database (imagine what happens if
41     * there are millions of certificates in the DB!). Should only be used for
42     * testing purposes.
43     *
44     * @param issuerdn the dn of the certificates issuer.
45     * @return Collection of fingerprints, i.e. Strings
46     */
47    Collection<String> listAllCertificates(String issuerdn);
48  
49    /**
50     * Lists RevokedCertInfo of ALL revoked certificates (status =
51     * CertificateConstants.CERT_REVOKED) in the database from a certain issuer.
52     * NOTE: Caution should be taken with this method as execution may be very
53     * heavy indeed if many certificates exist in the database (imagine what
54     * happens if there are millions of certificates in the DB!). Should only be
55     * used for testing purposes.
56     *
57     * @param issuerdn the dn of the certificates issuer.
58     * @param lastbasecrldate a date (Date.getTime()) of last base CRL or -1 for a
59     *     complete CRL
60     * @return Collection of RevokedCertInfo, reverse ordered by expireDate where
61     *     last expireDate is first in array.
62     */
63    Collection<RevokedCertInfo> listRevokedCertInfo(
64        String issuerdn, long lastbasecrldate);
65  
66    /**
67     * Lists certificates for a given subject signed by the given issuer.
68     *
69     * @param subjectDN the DN of the subject whose certificates will be
70     *     retrieved.
71     * @param issuerDN the DN of the certificates issuer.
72     * @return Collection of Certificates (java.security.cert.Certificate) in no
73     *     specified order or an empty Collection.
74     */
75    List<Certificate> findCertificatesBySubjectAndIssuer(
76        String subjectDN, String issuerDN);
77  
78    /**
79     * Lists certificates for a given subject signed by the given issuer.
80     *
81     * @param subjectDN the DN of the subject whose certificates will be
82     *     retrieved.
83     * @param issuerDN the DN of the certificates issuer.
84     * @param onlyActive set to true to limit the search to active (non revoked,
85     *     unexpired) certificates.
86     * @return Collection of Certificates (java.security.cert.Certificate) in no
87     *     specified order or an empty Collection.
88     */
89    List<Certificate> findCertificatesBySubjectAndIssuer(
90        String subjectDN, String issuerDN, boolean onlyActive);
91  
92    /**
93     * @param issuerDN Issuer DN
94     * @param subjectDN Subject DN
95     * @return set of users with certificates with specified subject DN issued by
96     *     specified issuer.
97     */
98    Set<String> findUsernamesByIssuerDNAndSubjectDN(
99        String issuerDN, String subjectDN);
100 
101   /**
102    * @param issuerDN Issuer DB
103    * @param subjectKeyId Key
104    * @return set of users with certificates with specified key issued by
105    *     specified issuer.
106    */
107   Set<String> findUsernamesByIssuerDNAndSubjectKeyId(
108       String issuerDN, byte[] subjectKeyId);
109 
110   /**
111    * Return the username of a certificate data object matching the given issuer
112    * DN and serial number (there may be only one).
113    *
114    * @param issuerDn The DN of the issuing CA
115    * @param serialNumber the serial number of the sought certificate.
116    * @return the matching username,
117    */
118   String findUsernameByIssuerDnAndSerialNumber(
119       String issuerDn, BigInteger serialNumber);
120 
121   /**
122    * Lists certificates for a given subject.
123    *
124    * @param subjectDN the DN of the subject whos certificates will be retrieved.
125    * @return Collection of Certificates (java.security.cert.Certificate) in no
126    *     specified order or an empty Collection.
127    */
128   List<Certificate> findCertificatesBySubject(String subjectDN);
129 
130   /**
131    * Finds certificates expiring within a specified time and that have status
132    * "active" or "notifiedaboutexpiration".
133    *
134    * @param expireTime The time by which the certificates will be expired
135    * @see org.cesecore.certificates.certificate.CertificateConstants#CERT_ACTIVE
136    * @see
137    *     org.cesecore.certificates.certificate.CertificateConstants#CERT_NOTIFIEDABOUTEXPIRATION
138    * @return Collection of maximum 500 certificates, never null
139    */
140   Collection<CertificateWrapper> findCertificatesByExpireTimeWithLimit(
141       Date expireTime);
142 
143   /**
144    * Finds certificates expiring within a specified time and that have status
145    * "active" or "notifiedaboutexpiration".
146    *
147    * @param expireTime The time by which the certificates will be expired
148    * @param maxNumberOfResults The maximum number of certificates to be returned
149    * @see org.cesecore.certificates.certificate.CertificateConstants#CERT_ACTIVE
150    * @see
151    *     org.cesecore.certificates.certificate.CertificateConstants#CERT_NOTIFIEDABOUTEXPIRATION
152    * @return List of certificates (java.security.cert.Certificate), never null
153    */
154   List<Certificate> findCertificatesByExpireTimeWithLimit(
155       Date expireTime, int maxNumberOfResults);
156 
157   /**
158    * Finds count of certificates expiring within a specified time and that have
159    * status "active" or "notifiedaboutexpiration".
160    *
161    * @param expirationDate The time by which the certificates will be expired
162    * @return return count of query results.
163    */
164   int findNumberOfExpiringCertificates(Date expirationDate);
165 
166   /**
167    * Finds certificates expiring within a specified time and that have status
168    * "active" or "notifiedaboutexpiration".
169    *
170    * @param expirationDate The time by which the certificates will be expired
171    * @param maxNumberOfResults The maximum number of certificates to be returned
172    * @param offset return results starting from offset
173    * @return List of certificates (java.security.cert.Certificate), never null
174    */
175   List<Certificate> findExpiringCertificates(
176       Date expirationDate, int maxNumberOfResults, int offset);
177 
178   /**
179    * Finds certificates expiring within a specified time, issued by a specified
180    * issuer and have status "active" or "notifiedaboutexpiration".
181    *
182    * @param expireTime The time by which the certificates will be expired
183    * @param issuerDN The SubjectDN of the CA that issued the certificates that
184    *     will expire within the given time
185    * @param maxNumberOfResults The maximum number of certificates to be returned
186    * @see org.cesecore.certificates.certificate.CertificateConstants#CERT_ACTIVE
187    * @see
188    *     org.cesecore.certificates.certificate.CertificateConstants#CERT_NOTIFIEDABOUTEXPIRATION
189    * @return List of certificates (java.security.cert.Certificate), never null
190    */
191   List<Certificate> findCertificatesByExpireTimeAndIssuerWithLimit(
192       Date expireTime, String issuerDN, int maxNumberOfResults);
193 
194   /**
195    * Finds certificates expiring within a specified time, of a specified type
196    * and have status "active" or "notifiedaboutexpiration".
197    *
198    * @param expireTime The time by which the certificates will be expired
199    * @param certificateType The type of the certificates that will expire within
200    *     the given time
201    * @param maxNumberOfResults The maximum number of certificates to be returned
202    * @see org.cesecore.certificates.certificate.CertificateConstants#CERT_ACTIVE
203    * @see
204    *     org.cesecore.certificates.certificate.CertificateConstants#CERT_NOTIFIEDABOUTEXPIRATION
205    * @return a List of certificates or en empty list, never null
206    */
207   List<Certificate> findCertificatesByExpireTimeAndTypeWithLimit(
208       Date expireTime, int certificateType, int maxNumberOfResults);
209 
210   /**
211    * Finds usernames of users having certificate(s) expiring within a specified
212    * time and that has status "active" or "notifiedaboutexpiration".
213    *
214    * @param expiretime Expiry
215    * @see org.cesecore.certificates.certificate.CertificateConstants#CERT_ACTIVE
216    * @see
217    *     org.cesecore.certificates.certificate.CertificateConstants#CERT_NOTIFIEDABOUTEXPIRATION
218    * @return a Collection of usernames or an empty list if not found, never null
219    */
220   Collection<String> findUsernamesByExpireTimeWithLimit(Date expiretime);
221 
222   /**
223    * Finds a certificate specified by issuer DN and serial number.
224    *
225    * @param issuerDN issuer DN of the desired certificate.
226    * @param serno serial number of the desired certificate!
227    * @return the sought certificate, or null if not found
228    */
229   Certificate findCertificateByIssuerAndSerno(
230       String issuerDN, BigInteger serno);
231 
232   /**
233    * Gets full certificate meta data for the cert specified by issuer DN and
234    * serial number.
235    *
236    * @param issuerDN issuer DN of the desired certificate.
237    * @param serno serial number of the desired certificate!
238    * @return the sought certificate, or null if not found
239    */
240   CertificateDataWrapper getCertificateDataByIssuerAndSerno(
241       String issuerDN, BigInteger serno);
242 
243   /**
244    * Find a certificate by its subject key ID.
245    *
246    * @param subjectKeyId subject key ID of the sought certificate
247    * @return the list Certificates if found, or an empty list (never null)
248    */
249   Collection<Certificate> findCertificatesBySubjectKeyId(byte[] subjectKeyId);
250 
251   /**
252    * The method retrieves all certificates from a specific issuer which are
253    * identified by list of serial numbers. The collection will be empty if the
254    * issuerDN is {@code null}/empty or the collection of serial numbers is
255    * empty.
256    *
257    * @param issuerDN the subjectDN of a CA certificate
258    * @param sernos a collection of certificate serialnumbers
259    * @return Collection a list of certificates; never {@code null}
260    */
261   Collection<Certificate> findCertificatesByIssuerAndSernos(
262       String issuerDN, Collection<BigInteger> sernos);
263 
264   /**
265    * Finds certificate(s) for a given serialnumber.
266    *
267    * @param serno the serialnumber of the certificate(s) that will be retrieved
268    * @return Certificate or null if none found.
269    */
270   List<CertificateDataWrapper> getCertificateDataBySerno(BigInteger serno);
271 
272   /**
273    * Find the latest published X509Certificate matching the given subject DN.
274    * Note that this method does not check if the returned certificate was issued
275    * by a rollover CA.
276    *
277    * @param subjectDN The subject DN to match.
278    * @return the sought result, or null if none exists.
279    */
280   X509Certificate findLatestX509CertificateBySubject(String subjectDN);
281 
282   /**
283    * Find the latest published X509Certificate matching the given subject DN.
284    * This method can search for normal or rollover certificates.
285    *
286    * @param subjectDN The subject DN to match.
287    * @param rolloverCA The rollover certificate of the issuing CA, or null to
288    *     not check if issued by a rollover certificate.
289    * @param findRollover Whether a rollover or normal certificate should be
290    *     returned.
291    * @return the sought result, or null if none exists.
292    */
293   X509Certificate findLatestX509CertificateBySubject(
294       String subjectDN, X509Certificate rolloverCA, boolean findRollover);
295 
296   /**
297    * Finds username for a given certificate serial number.
298    *
299    * @param serno the serialnumber of the certificate to find username for.
300    * @param issuerdn Issuer DN
301    * @return username or null if none found.
302    */
303   String findUsernameByCertSerno(BigInteger serno, String issuerdn);
304 
305   /**
306    * Finds certificate(s) for a given username.
307    *
308    * @param username the username of the certificate(s) that will be retrieved
309    * @return Collection of Certificates ordered by expire date, with last expire
310    *     date first, or null if none found.
311    */
312   Collection<CertificateWrapper> findCertificatesByUsername(String username);
313 
314   /**
315    * Finds certificate(s) with meta data for a given username.
316    *
317    * @param username the username of the certificate(s) that will be retrieved
318    * @param excludeExpired true if expired certificates should be excluded
319    * @param excludedStatuses a list of statuses to exclude in the query or null
320    *     to return all
321    * @return List of wrapped CertificateData and Base64CertData ordered by
322    *     expire date, with last expire date first, or empty list if none found.
323    */
324   List<CertificateDataWrapper> getCertificateDataByUsername(
325       String username, boolean excludeExpired, List<Integer> excludedStatuses);
326 
327   /**
328    * Finds certificate(s) for a given username and status.
329    *
330    * @param username the username of the certificate(s) that will be retrieved
331    * @param status the status from the CertificateConstants.CERT_ constants
332    * @return Collection of Certificates ordered by expire date, with last expire
333    *     date first, or empty list if user can not be found
334    */
335   Collection<Certificate> findCertificatesByUsernameAndStatus(
336       String username, int status);
337 
338   /**
339    * Finds certificate(s) for a given username and status if the expireDate is
340    * after the provided one.
341    *
342    * @param username the username of the certificate(s) that will be retrieved
343    * @param status the status from the CertificateConstants.CERT_ constants
344    * @param afterExpireDate only return entries that has an expireDate larger
345    *     than or equal to this one
346    * @return Collection of Certificates ordered by expire date, with last expire
347    *     date first, or empty list if user can not be found
348    */
349   Collection<Certificate> findCertificatesByUsernameAndStatusAfterExpireDate(
350       String username, int status, long afterExpireDate);
351 
352   /**
353    * Gets certificate info, which is basically all fields except the certificate
354    * itself. Note: this method should not be used within a transaction where the
355    * reading of this info might depend on something stored earlier in the
356    * transaction. This is because this method uses direct SQL.
357    *
358    * @param fingerprint FP
359    * @return CertificateInfo or null if certificate does not exist.
360    */
361   CertificateInfo getCertificateInfo(String fingerprint);
362 
363   /**
364    * Finds a certificate based on fingerprint. You can get fingerprint by for
365    * example "String fingerprint =
366    * CertTools.getFingerprintAsString(certificate);"
367    *
368    * @param fingerprint FP
369    * @return Certificate or null if it can not be found.
370    */
371   Certificate findCertificateByFingerprint(String fingerprint);
372 
373   /**
374    * Lists all active (status = 20) certificates of a specific type and if given
375    * from a specific issuer.
376    *
377    * @param issuerDN get all certificates issued by a specific issuer. If {@code
378    *     null} or empty return certificates regardless of the issuer.
379    * @param type CERTTYPE_* types from CertificateConstants
380    * @throws IllegalArgumentException when admin is null or type is not one or
381    *     more of of SecConst.CERTTYPE_SUBCA, SecConst.CERTTYPE_ENDENTITY,
382    *     SecConst.CERTTYPE_ROOTCA
383    * @return Collection of Certificate, never {@code null}
384    */
385   Collection<CertificateWrapper> findCertificatesByType(
386       int type, String issuerDN);
387 
388   /**
389    * Recursively finds the certificate chain for the given certificate.
390    *
391    * @param certinfo Certificate to start from, usually a leaf certificate.
392    * @return List containing certificate chain, starting with the given
393    *     certificate.
394    */
395   List<Certificate> getCertificateChain(CertificateInfo certinfo);
396 
397   /**
398    * Method revoking all certificates generated by the specified issuerdn. Sets
399    * revocationDate to current time. Should only be called by when a CA is about
400    * to be revoked.
401    *
402    * @param admin the administrator performing the event.
403    * @param issuerdn the dn of CA about to be revoked
404    * @param reason the reason of revocation.
405    * @throws AuthorizationDeniedException Access denied
406    */
407   void revokeAllCertByCA(AuthenticationToken admin, String issuerdn, int reason)
408       throws AuthorizationDeniedException;
409 
410   /**
411    * Checks if a certificate is revoked.
412    *
413    * @param issuerDN the DN of the issuer.
414    * @param serno the serialnumber of the certificate that will be checked
415    * @return true if the certificate is revoked or can not be found in the
416    *     database, false if it exists and is not revoked.
417    */
418   boolean isRevoked(String issuerDN, BigInteger serno);
419 
420   /**
421    * Get certificate status fast.
422    *
423    * @param issuerDN Issuer DN
424    * @param serno Serial No.
425    * @return CertificateStatus status of the certificate, never null,
426    *     CertificateStatus.NOT_AVAILABLE if the certificate is not found.
427    */
428   CertificateStatus getStatus(String issuerDN, BigInteger serno);
429 
430   /**
431    * Performs the same operation as getStatus, but returns a richer object which
432    * also contains the certificate, in order to save on database lookups when
433    * both objects are required. Issuer + serial number are always unique.
434    *
435    * @param issuerDN the issuer of the sought certificate
436    * @param serno the serial number of the sought certificate
437    * @return a {@link CertificateStatusHolder} object containing the status and
438    *     the sought certificate.
439    */
440   CertificateStatusHolder getCertificateAndStatus(
441       String issuerDN, BigInteger serno);
442 
443   /**
444    * Update the status of a cert in the database.
445    *
446    * @param admin Auth token
447    * @param fingerprint FP
448    * @param status one of CertificateConstants.CERT_...
449    * @return true if the status was updated, false if not, for example if the
450    *     certificate did not exist
451    * @throws AuthorizationDeniedException Access Denied
452    */
453   boolean setStatus(AuthenticationToken admin, String fingerprint, int status)
454       throws AuthorizationDeniedException;
455 }
