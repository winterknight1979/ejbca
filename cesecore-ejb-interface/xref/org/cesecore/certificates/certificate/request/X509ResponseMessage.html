27  /**
28   * A response message consisting of a single X509 or CVC Certificate. Name is
29   * nowadays slightly misleading since the class can care any type of
30   * "Certificate", for example a CV certificate.
31   *
32   * @version $Id: X509ResponseMessage.java 28201 2018-02-07 08:33:29Z
33   *     andresjakobs $
34   */
35  public class X509ResponseMessage implements CertificateResponseMessage {
36    /**
37     * Determines if a de-serialized file is compatible with this class.
38     *
39     * <p>Maintainers must change this value if and only if the new version of
40     * this class is not compatible with old versions. See Sun docs for <a
41     * href=http://java.sun.com/products/jdk/1.1/docs/guide

44    static final long serialVersionUID = -2157072605987735913L;

46    /** Logger. */
47    private static Logger log = Logger.getLogger(X509ResponseMessage.class);
48  
49    /** Certificate to be in response message. */
50    private byte[] certbytes = null;
51  
52    /** status for the response. */
53    private ResponseStatus status = ResponseStatus.SUCCESS;
54  
55    /** Possible fail information in the response. Defaults to null. */
56    private FailInfo failInfo = null;
57  
58    /**
59     * Possible clear text error information in the response. Defaults to null.
60     */
61    private String failText = null;
62    /** Cert. */
63    private transient Certificate certificate;
64    /** Data. */
65    private transient CertificateData certificateData;
66    /** Data. */
67    private transient Base64CertData base64CertData;

70    public CertificateData getCertificateData() {
71      return certificateData;

74    @Override
75    public void setCertificateData(final CertificateData theCertificateData) {
76      if (theCertificateData != null) {
77        this.certificateData = new CertificateData(theCertificateData);
78      } else {
79        this.certificateData = null;
80      }
81    }

84    public Base64CertData getBase64CertData() {
85      return base64CertData;
86    }
87  
88    @Override
89    public void setBase64CertData(final Base64CertData theBase64CertData) {
90      if (theBase64CertData != null) {
91        this.base64CertData = new Base64CertData(theBase64CertData);
92      } else {
93        this.base64CertData = null;
94      }
95    }
96  
97    /**
98     * Sets the complete certificate in the response message.
99     *
100    * @param aCertificate certificate in the response message.
101    */
102   @Override
103   public void setCertificate(final Certificate aCertificate) {
104     this.certificate = aCertificate;
105     try {
106       this.certbytes = aCertificate.getEncoded();
107     } catch (CertificateEncodingException e) {
108       throw new Error(
109           "Could not encode certificate. This should not happen", e);
110     }
111   }
112 
113   @Override
114   public void setCrl(final CRL crl) {
115     // This message type does not contain a CRL
116   }
117 
118   @Override
119   public void setIncludeCACert(final boolean incCACert) {
120     // Do nothing, not applicable
121   }
122 
123   @Override
124   public void setCACert(final Certificate cACert) { }
125 
126   @Override
127   public Certificate getCertificate() {
128     // Deserialize certificate using BC if this entire object has been
129     // serialized
130     if (certificate == null) {
131       try {
132         certificate =
133             CertTools.getCertfromByteArray(certbytes, Certificate.class);
134       } catch (CertificateException e) {
135         throw new Error(
136             "Response was created without containing valid certificate. This"
137                 + " should not happen",
138             e);
139       }
140     }
141     return certificate;
142   }
143 
144   @Override
145   public byte[] getResponseMessage() throws CertificateEncodingException {
146     return certbytes;
147   }

199   /**
200    * Create encrypts and creates signatures as needed to produce a complete
201    * response message. If needed setSignKeyInfo must be called before this
202    * method. After this is called the response message can be retrieved with
203    * getResponseMessage();
204    *
205    * @return True if signature/encryption was successful, false if it failed,
206    *     request should not be sent back i failed.
207    * @see #setSignKeyInfo
208    */
209   @Override
210   public boolean create() {
211 
212     if (status.equals(ResponseStatus.SUCCESS)) {
213       log.debug("Creating a STATUS_OK message.");
214     } else {
215       if (status.equals(ResponseStatus.FAILURE)) {
216         log.debug(
217             "Creating a STATUS_FAILED message (or throwing an exception).");
218         if (failInfo.equals(FailInfo.WRONG_AUTHORITY)) {
219           return false;
220         }
221         if (failInfo.equals(FailInfo.INCORRECT_DATA)) {
222           return false;
223         }
224       } else {
225         log.debug("Creating a STATUS_PENDING message.");
226       }
227     }
228     return true;
229   }
