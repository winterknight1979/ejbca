13  package org.cesecore.certificates.certificate;
14  
15  import java.math.BigInteger;
16  import java.security.cert.Certificate;
17  import java.util.Collection;
18  import java.util.Date;
19  import java.util.List;
20  import javax.ejb.Local;
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.authorization.AuthorizationDeniedException;

25  /**
26   * Local interface for CertificateStoreSession.
27   *
28   * @version $Id: CertificateStoreSessionLocal.java 28675 2018-04-12 08:09:35Z
29   *     samuellb $
30   */
31  @Local
32  public interface CertificateStoreSessionLocal extends CertificateStoreSession {
33  
34    /**
35     * return CertificateInfo for the certificate with issuerDN and serno. If
36     * there are multiple certificates with the same issuerDN, serno (should only
37     * be possible for CVC certificates) the first one (returned from database) is
38     * returned.
39     *
40     * @param issuerDN issuer DN of the desired certificate.
41     * @param serno serial number of the desired certificate!
42     * @return CertificateInfo if found or null
43     */
44    CertificateInfo findFirstCertificateInfo(String issuerDN, BigInteger serno);
45  
46    /**
47     * Stores a certificate.
48     *
49     * @param admin An authentication token to authorize the action
50     * @param incert The certificate to be stored.
51     * @param cafp Fingerprint (hex) of the CAs certificate.
52     * @param username username of end entity owning the certificate.
53     * @param status the status from the CertificateConstants.CERT_ constants
54     * @param type Type of certificate (CERTTYPE_ENDENTITY etc from
55     *     CertificateConstants).
56     * @param certificateProfileId the certificate profile id this cert was issued
57     *     under
58     * @param endEntityProfileId the end entity profile id this cert was issued
59     *     under
60     * @param tag a custom string tagging this certificate for some purpose
61     * @param updateTime epoch millis to use as last update time of the stored
62     *     object
63     * @return CertificateDataWrapper with the certificate just stored that can be
64     *     used for further publishing
65     * @throws AuthorizationDeniedException if admin was not authorized to store
66     *     certificate in database
67     */
68    CertificateDataWrapper storeCertificate(
69        AuthenticationToken admin,
70        Certificate incert,
71        String username,
72        String cafp,
73        int status,
74        int type,
75        int certificateProfileId,
76        int endEntityProfileId,
77        String tag,
78        long updateTime)
79        throws AuthorizationDeniedException;
80  
81    /**
82     * Stores a certificate without checking authorization. This should be used
83     * from other methods where authorization to the CA issuing the certificate
84     * has already been checked. For efficiency this method can then be used.
85     *
86     * @param admin An authentication token to authorize the action
87     * @param incert The certificate to be stored.
88     * @param cafp Fingerprint (hex) of the CAs certificate.
89     * @param username username of end entity owning the certificate.
90     * @param status the status from the CertificateConstants.CERT_ constants
91     * @param type Type of certificate (CERTTYPE_ENDENTITY etc from
92     *     CertificateConstants).
93     * @param certificateProfileId the certificate profile id this cert was issued
94     *     under
95     * @param endEntityProfileId the end entity profile id this cert was issued
96     *     under
97     * @param tag a custom string tagging this certificate for some purpose
98     * @param updateTime epoch millis to use as last update time of the stored
99     *     object
100    * @return CertificateDataWrapper with the certificate just stored that can be
101    *     used for further publishing
102    */
103   CertificateDataWrapper storeCertificateNoAuth(
104       AuthenticationToken admin,
105       Certificate incert,
106       String username,
107       String cafp,
108       int status,
109       int type,
110       int certificateProfileId,
111       int endEntityProfileId,
112       String tag,
113       long updateTime);
114 
115   /**
116    * Retrieve the full wrapped CertificateData and Base64CertData objects.
117    *
118    * @param fingerprint Fingerprint
119    * @return the sought certificate, or null if no data for the specified
120    *     fingerprint exists
121    */
122   CertificateDataWrapper getCertificateData(String fingerprint);
123 
124   /**
125    * Update the base64cert column if the database row exists, but the column is
126    * empty.
127    *
128    * @param authenticationToken Auth token
129    * @param certificate Certificate
130    * @return true if the column was empty and is now populated.
131    */
132   boolean updateCertificateOnly(
133       AuthenticationToken authenticationToken, Certificate certificate);
134 
135   /**
136    * Fetch a List of all certificate fingerprints and corresponding username.
137    *
138    * @param cas A list of CAs that the sought certificates should be issued from
139    * @param certificateProfiles A list if certificateprofiles to sort from. Will
140    *     be ignored if left empty.
141    * @param activeNotifiedExpireDateMin The minimal date for expiration
142    *     notification
143    * @param activeNotifiedExpireDateMax The maxmimal date for expiration
144    *     notification
145    * @param activeExpireDateMin the current rune timestamp + the threshold
146    * @return [0] = (String) fingerprint, [1] = (String) username
147    */
148   List<Object[]> findExpirationInfo(
149       Collection<String> cas,
150       Collection<Integer> certificateProfiles,
151       long activeNotifiedExpireDateMin,
152       long activeNotifiedExpireDateMax,
153       long activeExpireDateMin);
154 
155   /**
156    * Query if we have a clear cut case where no username or the provided
157    * username is the only user of the subjectDN and subjectKeyId.
158    *
159    * @param issuerDN The CA's subject
160    * @param subjectDN The entity's subject
161    * @param subjectKeyId The entity's subjectKeyId
162    * @param username the entity's username
163    * @return true if there is no other end entity under this issuer with the
164    *     same subjectKeyId or subjectDN
165    */
166   boolean isOnlyUsernameForSubjectKeyIdOrDnAndIssuerDN(
167       String issuerDN, byte[] subjectKeyId, String subjectDN, String username);
168 
169   /**
170    * Find the most recently issued/updated certificate for a public key.
171    *
172    * @param subjectKeyId Is the ASN.1 SubjectKeyIdentifier of the public key as
173    *     a byte array
174    * @return null or the certificate which is active, matches the argument and
175    *     has the latest updateTime
176    * @see
177    *     org.cesecore.keys.util.KeyTools#createSubjectKeyId(java.security.PublicKey)
178    */
179   Certificate findMostRecentlyUpdatedActiveCertificate(byte[] subjectKeyId);
180 
181   /**
182    * @param fingerprint FP
183    * @return the username or null if no row was found for this certificate
184    *     fingerprint
185    */
186   String findUsernameByFingerprint(String fingerprint);
187 
188   /**
189    * Tries to get an issuerDN/serialNumber pair from the request, and see if we
190    * have that CA certificate in the certificate store. If we have the CA dn, in
191    * CESeCore normalized form, is returned.
192    *
193    * @param req the request message that might contain an issued DN
194    * @return issuer DN or null if it does not exist in the
195    */
196   String getCADnFromRequest(RequestMessage req);
197 
198   /**
199    * Will test if there is a unique index/constraint for (certificate serial
200    * number,issuer DN) first time it is run.
201    *
202    * @return returns true if there is a database index for unique certificate
203    *     serial number / issuer DN.
204    */
205   boolean isUniqueCertificateSerialNumberIndex();
206 
207   /**
208    * ONLY use in Unit tests. Required by multiple entry unit tests since
209    * isUniqueCertificateSerialNumberIndex is a static variable.
210    */
211   void resetUniqueCertificateSerialNumberIndex();
212 
213   /**
214    * ONLY use in Unit tests or during controlled startup. Sets the check for
215    * present certificate serial number unique index to specified value. Can be
216    * used to override safety check that the index exists.
217    *
218    * @param value index
219    */
220   void setUniqueCertificateSerialNumberIndex(Boolean value);
221 
222   /**
223    * Checks for present certificate serial number unique index in a new
224    * transaction in order to avoid rollback, since we can expect SQL exceptions
225    * here. Should not be used externally. NOTE: does not add an event to the
226    * security audit log, that test certificates were added to the database, and
227    * removed. All normal operations that stores certificates in the database
228    * creates audit log events.
229    *
230    * @param admin An authentication token to authorize the action
231    * @param incert The certificate to be stored.
232    * @param cafp Fingerprint (hex) of the CAs certificate.
233    * @param username username of end entity owning the certificate.
234    * @param status the status from the CertificateConstants.CERT_ constants
235    * @param type Type of certificate (CERTTYPE_ENDENTITY etc from
236    *     CertificateConstants).
237    * @param certificateProfileId the certificate profile id this cert was issued
238    *     under
239    * @param endEntityProfileId the end entity profile id this cert was issued
240    *     under
241    * @param tag a custom string tagging this certificate for some purpose
242    * @param updateTime epoch millis to use as last update time of the stored
243    *     object
244    * @throws AuthorizationDeniedException If access denied
245    */
246   void checkForUniqueCertificateSerialNumberIndexInTransaction(
247       AuthenticationToken admin,
248       Certificate incert,
249       String username,
250       String cafp,
251       int status,
252       int type,
253       int certificateProfileId,
254       int endEntityProfileId,
255       String tag,
256       long updateTime)
257       throws AuthorizationDeniedException;
258 
259   /**
260    * Removed certificates created during checks for present certificate serial
261    * number unique index. Should not be used externally.
262    */
263   void removeUniqueCertificateSerialNumberTestCertificates();
264 
265   /**
266    * Creates a revoked certificate data, for use with OCSP or CRL generation.
267    *
268    * @param admin an admin that is authorized to the CA that issued the
269    *     certificate
270    * @param caId the CA identifier
271    * @param issuerDn the BC normalized version of the issuer DN
272    * @param serialNumber the certificate serial number
273    * @param revocationDate the date of revocation
274    * @param reasonCode one of RevokedCertInfo.REVOCATION_REASON_...
275    * @param caFingerprint the SHA-1 of the CA Certificate that issued this entry
276    * @throws AuthorizationDeniedException If access denied
277    * @see
278    *     CertificateStoreSessionLocal#updateLimitedCertificateDataStatus(AuthenticationToken,
279    *     int, String, String, String, BigInteger, int, Date, int, String)
280    */
281   void updateLimitedCertificateDataStatus(
282       AuthenticationToken admin,
283       int caId,
284       String issuerDn,
285       BigInteger serialNumber,
286       Date revocationDate,
287       int reasonCode,
288       String caFingerprint)
289       throws AuthorizationDeniedException;
290 
291   /**
292    * Method for populating the CertificateData table with limited information
293    * for example from a CRL, so the OCSP responder can answer if a certificate
294    * is revoked.
295    *
296    * <p>Existing entries may only be modified if they were created through this
297    * method. Updating an entry by providing
298    * RevokedCertInfo.REVOCATION_REASON_REMOVEFROMCRL as reasonCode will remove
299    * the row from the table.
300    *
301    * @param admin an admin that is authorized to the CA that issued the
302    *     certificate
303    * @param caId the CA identifier
304    * @param issuerDn the BC normalized version of the issuer DN
305    * @param subjectDn the BC normalized version of the subject DN
306    * @param username Username
307    * @param serialNumber the certificate serial number
308    * @param status Certificate status
309    * @param revocationDate the date of revocation
310    * @param reasonCode one of RevokedCertInfo.REVOCATION_REASON_...
311    * @param caFingerprint the SHA-1 of the CA Certificate that issued this entry
312    * @throws AuthorizationDeniedException If access denied
313    */
314   void updateLimitedCertificateDataStatus(
315       AuthenticationToken admin,
316       int caId,
317       String issuerDn,
318       String subjectDn,
319       String username,
320       BigInteger serialNumber,
321       int status,
322       Date revocationDate,
323       int reasonCode,
324       String caFingerprint)
325       throws AuthorizationDeniedException;
326 
327   /** Reloads the cache containing CA certificates. */
328   void reloadCaCertificateCache();
329 
330   /**
331    * Invoked from timer. Reloads the cache containing CA certificates and
332    * additionally sets a new timeout.
333    */
334   void reloadCaCertificateCacheAndSetTimeout();
335 
336   /** Initialize all timers and related operations used by this bean. */
337   void initTimers();
338 
339   /**
340    * Set the status of certificate with given serno to revoked, or unrevoked
341    * (re-activation).
342    *
343    * @param admin AuthenticationToken performing the operation
344    * @param cdw the certificate data to revoke or activate.
345    * @param revokedDate when it was revoked
346    * @param reason the reason of the revocation. (One of the
347    *     RevokedCertInfo.REVOCATION_REASON constants.)
348    * @return true if status was changed in the database, false if not, for
349    *     example if the certificate was already revoked
350    * @throws CertificateRevokeException (rollback) if certificate does not exist
351    * @throws AuthorizationDeniedException (rollback)
352    */
353   boolean setRevokeStatus(
354       AuthenticationToken admin,
355       CertificateDataWrapper cdw,
356       Date revokedDate,
357       int reason)
358       throws CertificateRevokeException, AuthorizationDeniedException;
359 
360   /**
361    * Lists certificate datas for a given subject signed by the given issuer.
362    *
363    * @param subjectDN the DN of the subject whose certificates will be
364    *     retrieved.
365    * @param issuerDN the DN of the certificates issuer.
366    * @param onlyActive set to true to limit the search to active (non revoked,
367    *     unexpired) certificates.
368    * @return List of certificate datas in no specified order or an empty List.
369    */
370   List<CertificateDataWrapper> getCertificateDatasBySubjectAndIssuer(
371       String subjectDN, String issuerDN, boolean onlyActive);
372 
373   /**
374    * Lists certificate datas for a given subject.
375    *
376    * @param subjectDN the DN of the subject whose certificates will be
377    *     retrieved.
378    * @return List of certificate datas in no specified order or an empty List.
379    */
380   List<CertificateDataWrapper> getCertificateDatasBySubject(String subjectDN);
381 
382   /**
383    * Method to set the status of certificate to revoked or active, without
384    * checking for authorization. This is why it is important that this method is
385    * _local only_.
386    *
387    * @param admin Administrator performing the operation
388    * @param certificateData the certificate to change status of. May also be a
389    *     new NoConflictCertificateData row to store in the database (generally,
390    *     a throw away certificate to revoke).
391    * @param revokeDate when it was revoked
392    * @param reason the reason of the revocation. (One of the
393    *     RevokedCertInfo.REVOCATION_REASON constants.)
394    * @return true if status was changed in the database, false if not, for
395    *     example if the certificate was already revoked
396    * @throws CertificateRevokeException (rollback) if certificate does not exist
397    */
398   boolean setRevokeStatusNoAuth(
399       AuthenticationToken admin,
400       BaseCertificateData certificateData,
401       Date revokeDate,
402       int reason)
403       throws CertificateRevokeException;
404 
405   /**
406    * Changes a certificate from CERT_ROLLOVERPENDING to CERT_ACTIVE. If the
407    * certificate status is already CERT_ACTIVE, then it does nothing.
408    *
409    * @param admin Administrator performing the operation
410    * @param fingerprint Fingerprint of the certificate
411    * @throws AuthorizationDeniedException if access denied
412    */
413   void setRolloverDoneStatus(AuthenticationToken admin, String fingerprint)
414       throws AuthorizationDeniedException;
415 
416   /**
417    * Method for speedy checks if a certificate exists for issuerDN, serno.
418    *
419    * @param issuerDN issuer DN of the desired certificate.
420    * @param serno serial number of the desired certificate!
421    * @return true if a Certificate exists, false otherwise
422    */
423   boolean existsByIssuerAndSerno(String issuerDN, BigInteger serno);
424 
425   /**
426    * Gets the status of the certificate, or -1 if the certificate does not
427    * exist. If more than one certificate exists with the issuerDN/serialNumber,
428    * the first one is returned. This query performs limited database read and
429    * thus will not verify database integrity protection.
430    *
431    * @param issuerDN issuer DN of the desired certificate.
432    * @param serno serial number of the desired certificate!
433    * @return one of the {@link CertificateConstants} statuses or -1 if the
434    *     certificate does not exist in the database.
435    */
436   int getFirstStatusByIssuerAndSerno(String issuerDN, BigInteger serno);
