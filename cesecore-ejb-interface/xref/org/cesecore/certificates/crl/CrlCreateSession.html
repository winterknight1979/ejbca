13  package org.cesecore.certificates.crl;
14  
15  import java.util.Collection;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.cesecore.certificates.ca.CA;
20  import org.cesecore.keys.token.CryptoTokenOfflineException;
21  
22  /**
23   * Interface for CrlCreateSession, a session bean for generating CRLs and deltaCRLs.
24   * Authorization for generating CRLs are handled by the fact that you need to be authorized to the CA. 
25   * Authorization checks are thus done in CaSessionBean when this session retrieves the CA for CRL generation.
26   * 
27   * @version $Id: CrlCreateSession.java 17625 2013-09-20 07:12:06Z netmackan $
28   */
29  public interface CrlCreateSession {
30  
31      /**
32       * Requests for a CRL to be created with the passed (revoked) certificates. 
33       * Generates the CRL and stores it in the database.
34       *
35       * @param admin administrator performing the task
36       * @param ca the CA this operation regards
37       * @param certs collection of RevokedCertInfo object.
38       * @param basecrlnumber the CRL number of the Base CRL to generate a deltaCRL, -1 to generate a full CRL
39       * @param nextCrlNumber the CRL number.
40       * @return The newly created CRL in DER encoded byte form or null, use CertTools.getCRLfromByteArray to convert to X509CRL.
41       * @throws AuthorizationDeniedException If access denied 
42       * @throws CryptoTokenOfflineException If offline
43       */
44      byte[] generateAndStoreCRL(AuthenticationToken admin, CA ca, Collection<RevokedCertInfo> certs, int basecrlnumber, int nextCrlNumber)
45              throws CryptoTokenOfflineException, AuthorizationDeniedException;
46  
47  }
