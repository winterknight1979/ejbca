13  package org.cesecore.keys.token;
14  
15  import java.util.List;
16  import java.util.Map;
17  import javax.ejb.Local;
18  
19  /**
20   * @version $Id: CryptoTokenSessionLocal.java 25377 2017-03-02 16:46:52Z jeklund
21   *     $
22   */
23  @Local
24  public interface CryptoTokenSessionLocal extends CryptoTokenSession {
25  
26    /**
27     * @param cryptoTokenId ID
28     * @return the specified CryptoToken or null if it does not exis. Throws
29     *     RuntimeException if allow.nonexisting.slot=false (default) and a
30     *     PKCS#11 slot does not exist.
31     */
32    CryptoToken getCryptoToken(int cryptoTokenId);
33  
34    /**
35     * Add the specified CryptoToken to the database and return the id used to
36     * store it.
37     *
38     * @param cryptoToken Token
39     * @return Merged token
40     * @throws CryptoTokenNameInUseException if nme in use
41     */
42    int mergeCryptoToken(CryptoToken cryptoToken)
43        throws CryptoTokenNameInUseException;
44  
45    /**
46     * Remove the specified CryptoToken from the database.
47     *
48     * @param cryptoTokenId the id of the crypto token that should be removed
49     * @return true if crypto token exists and is deleted, false if crypto token
50     *     with given id does not exist
51     */
52    boolean removeCryptoToken(int cryptoTokenId);
53  
54    /** @return a list of all CryptoToken identifiers in the database. */
55    List<Integer> getCryptoTokenIds();
56  
57    /**
58     * @return a list of all CryptoToken identifiers and corresponding human
59     *     readable name in the database.
60     */
61    Map<Integer, String> getCryptoTokenIdToNameMap();
62  
63    /** @return a (copy of a) name to id lookup table */
64    Map<String, Integer> getCachedNameToIdMap();
65  
66    /** Clears the CryptoToken cache. */
67    void flushCache();
68  
69    /**
70     * Clears the CryptoToken cache except for the cache entries specified in
71     * excludeIDs.
72     *
73     * @param excludeIDs IDs to exclude
74     */
75    void flushExcludingIDs(List<Integer> excludeIDs);
76  
77    /**
78     * @param cryptoTokenName Name
79     * @param cryptoTokenId ID
80     * @return true if the specified name is used by exactly one CryptoToken and
81     *     that cryptoToken has the same id (checks the database, not the cache)
82     */
83    boolean isCryptoTokenNameUsedByIdOnly(
84        String cryptoTokenName, int cryptoTokenId);
85  }
