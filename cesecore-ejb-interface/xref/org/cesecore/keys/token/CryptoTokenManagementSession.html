25  /**
26   * CryptoToken management operations that require authorization and/or security events audit logging.
27   * 
28   * @version $Id: CryptoTokenManagementSession.java 30544 2018-11-19 15:12:01Z anatom $
29   */
30  public interface CryptoTokenManagementSession {
31      
32      /** Indicate that we would like to keep the current auto-activation PIN (if present) when save a CryptoToken. */
33      public final String KEEP_AUTO_ACTIVATION_PIN = "keepAutoActivationPin";
34  
35      /** @param authenticationToken Token
36       * @return a list of IDs for CryptoTokens that the caller is authorized to view */
37      List<Integer> getCryptoTokenIds(AuthenticationToken authenticationToken);
38  
39      /** Requests activation of the referenced CryptoToken. 
40       * @param authenticationToken Auth Token
41       * @param cryptoTokenId Crypto Token
42       * @param authenticationCode Auth Code
43       * @throws AuthorizationDeniedException If access denied 
44       * @throws CryptoTokenOfflineException  If offline
45       * @throws CryptoTokenAuthenticationFailedException if auth failed. */
46      void activate(AuthenticationToken authenticationToken, int cryptoTokenId, char[] authenticationCode)
47              throws AuthorizationDeniedException, CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException;
48  
49      /** Requests reactivation of the referenced CryptoToken. 
50       * @param authenticationToken Auth token
51       * @param cryptoTokenId Crypto Token
52       * @throws AuthorizationDeniedException If access denied */
53      void deactivate(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException;
54  
55      /**
56       * Remove CryptoToken with the specified ID. If this CryptoToken is backed by an HSM only the reference to the
57       * PKCS#11 slot will be removed not the actual key material. If the crypto token with the given id does not exists, nothing happens.
58       * @param authenticationToken Auth token
59       * @param cryptoTokenId the id of the crypto token that should be removed
60       * @throws AuthorizationDeniedException If access denied
61       */
62      void deleteCryptoToken(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException;
63  
64      /** @param authenticationToken Auth token
65       * @param cryptoTokenId Crypto token
66       * @return true if the CryptoToken with the specified ID has been activated. 
67       * @throws AuthorizationDeniedException If access denied */
68      boolean isCryptoTokenStatusActive(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException;
69      
70      /**
71       * Checks if a crypto token is present. 
72       * @param authenticationToken Auth token
73       * 
74       * @param cryptoTokenId the ID of the crypto token
75       * @return true if it is exists and is present
76       * @throws AuthorizationDeniedException  If access denied
77       */
78      boolean isCryptoTokenPresent(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException;
79  
80      /** Checks if an intended Crypto Token (to be created) is already used. Primarily of use for PKCS#11 Crypto Tokens as there
81       * can be issues if creating multiple crypto tokens referencing the same PKCS#11 Slot.
82       * @param authenticationToken an authentication token defining which crypto tokens can be compared, if this does not have full privileges the check is not complete
83       * @param tokenName the name of the token we want to check
84       * @param className the classname, org.cesecore.keys.token.PKCS11CryptoToken if this check should return anything but an empty list 
85       * @param properties crypto token properties, with the full PKCS#11 properties needed to create a PKCS#11 crypto token
86       * @return List or crypto token names which are using the same slot, or an empty list if there is none, an empty list is thus a sign to "go ahead"
87       * @throws AuthorizationDeniedException If access denied
88       * @throws CryptoTokenNameInUseException If name already exists
89       * @throws CryptoTokenOfflineException If offline
90       * @throws CryptoTokenAuthenticationFailedException If auth fails
91       * @throws NoSuchSlotException If slot does not exist
92       */
93      public List<String> isCryptoTokenSlotUsed(AuthenticationToken authenticationToken, String tokenName, String className, Properties properties) 
94      		throws AuthorizationDeniedException, CryptoTokenNameInUseException, CryptoTokenOfflineException,
95              CryptoTokenAuthenticationFailedException, NoSuchSlotException;
96  
97      /**
98       * Creates a crypto token with a known ID. Note that using an already existing ID will lead to an error at a later state. 
99       * 
100      * @param authenticationToken an authentication token
101      * @param tokenName the name of the token
102      * @param cryptoTokenId a known and unused ID value. 
103      * @param className the class name of the crypto token
104      * @param properties a properties file containing implementation specific values
105      * @param data the keystore data. If null a new, empty keystore will be created
106      * @param authenticationCode the authentication code to the slot. Will not activate the slot if offline
107      *
108      * @throws AuthorizationDeniedException If access denied
109      * @throws CryptoTokenNameInUseException If name already exists
110      * @throws CryptoTokenOfflineException If offline
111      * @throws CryptoTokenAuthenticationFailedException If auth fails
112      * @throws NoSuchSlotException if no PKCS#11 slot as defined by the label in properties could be found
113      */
114      void createCryptoToken(AuthenticationToken authenticationToken, String tokenName, Integer cryptoTokenId, String className, Properties properties,
115             byte[] data, char[] authenticationCode) throws AuthorizationDeniedException, CryptoTokenNameInUseException, CryptoTokenOfflineException,
116             CryptoTokenAuthenticationFailedException, NoSuchSlotException;
117     
118     /**
119      * Create a crypto token. This method will generate its own ID. 
120      * 
121      * @param authenticationToken an authentication token
122      * @param tokenName the name of the token
123      * @param className the class name of the crypto token
124      * @param properties a properties file containing implementation specific values
125      * @param data  the keystore data. If null a new, empty keystore will be created
126      * @param authenticationCode authenticationCode the authentication code to the slot
127      * @return the ID of a newly persisted CryptoToken from the supplied parameters.
128      * @throws AuthorizationDeniedException If access denied
129      * @throws CryptoTokenNameInUseException If name already exists
130      * @throws CryptoTokenOfflineException If offline
131      * @throws CryptoTokenAuthenticationFailedException If auth fails
132      * @throws NoSuchSlotException if no PKCS#11 slot as defined by the label in properties could be found
133      */
134     int createCryptoToken(AuthenticationToken authenticationToken, String tokenName, String className, Properties properties, byte[] data,
135             char[] authenticationCode) throws AuthorizationDeniedException, CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException,
136             CryptoTokenNameInUseException, NoSuchSlotException;
137 
138    
139     /**
140      * Update the CryptoToken with the specified ID. The authentication code can be omitted (null) if auto-activation is used. 
141      * 
142      * @param authenticationToken Auth Token
143      * @param cryptoTokenId Crypto token
144      * @param tokenName Name
145      * @param properties Properties
146      * @param authenticationCode Auth code
147      * @throws AuthorizationDeniedException If access denied
148      * @throws CryptoTokenNameInUseException If name already exists
149      * @throws CryptoTokenOfflineException If offline
150      * @throws CryptoTokenAuthenticationFailedException If auth fails
151      * @throws NoSuchSlotException if no such slot as defined in properties could be found
152      */
153     void saveCryptoToken(AuthenticationToken authenticationToken, int cryptoTokenId, String tokenName, Properties properties,
154             char[] authenticationCode) throws AuthorizationDeniedException, CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException,
155             CryptoTokenNameInUseException, NoSuchSlotException;
156 
157     /**
158      * Changes the name and key placeholders of a CryptoToken. Doesn't de-activate the crypto token,
159      * and can't be used to change any other properties (e.g. PKCS#11 slot etc.) or the authentication code.
160      *
161      * @param authenticationToken Auth token
162      * @param cryptoTokenId Id of the existing crypto token
163      * @param newName New name of the crypto token.
164      * @param newPlaceholders New key placeholders, in the same format as they are stored in the crypto token properties.
165      * @throws AuthorizationDeniedException if access denied
166      * @throws CryptoTokenNameInUseException If the new name is already in use.
167      */
168     void saveCryptoToken(AuthenticationToken authenticationToken, int cryptoTokenId, String newName, String newPlaceholders) throws AuthorizationDeniedException, CryptoTokenNameInUseException;
169     
170     /** @param authenticationToken Auth token
171      * @param cryptoTokenId Crypto token
172      * @return value object with non-sensitive information about the CryptoToken for UI use or similar, or null if token does not exist. 
173      * @throws AuthorizationDeniedException if access denied */
174     CryptoTokenInfo getCryptoTokenInfo(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException;
175 
176     /** @param authenticationToken Auth token
177      * @return List value objects with non-sensitive information about authorized CryptoToken for UI use or similar. Returns empty list if no tokens exist. */
178     List<CryptoTokenInfo> getCryptoTokenInfos(AuthenticationToken authenticationToken);
179 
180     /** @param cryptoTokenName Token name
181      * @return the cryptoTokenId from the more user friendly name. Return null of there is no such CryptoToken. */
182     Integer getIdFromName(String cryptoTokenName);
183 
184     /** @param authenticationToken Auth token
185      * @param cryptoTokenId Crypto token
186      * @return a List of all the key pair aliases present in the specified CryptoToken. 
187      * @throws AuthorizationDeniedException if access denied 
188      * @throws CryptoTokenOfflineException If offline */
189     List<String> getKeyPairAliases(AuthenticationToken authenticationToken, int cryptoTokenId) throws AuthorizationDeniedException, CryptoTokenOfflineException;
190 
191     /**
192      * Generate a new key pair in the specified CryptoToken with the requested alias and key specification.
193      * @param authenticationToken Auth token
194      * @param cryptoTokenId Crypto token
195      * @param alias Alias
196      * 
197      * @param keySpecification should be in the form "RSAnnnn", "DSAnnnn" or an known EC curve name.
198      * @throws AuthorizationDeniedException If access denied
199      * 
200      * @throws CryptoTokenOfflineException if the CryptoToken is unavailable or inactive.
201      * @throws InvalidKeyException if key generation failed.
202      * @throws InvalidAlgorithmParameterException if the keySpecification is not available for this CryptoToken.
203      */
204     void createKeyPair(AuthenticationToken authenticationToken, int cryptoTokenId, String alias, String keySpecification)
205             throws AuthorizationDeniedException, CryptoTokenOfflineException, InvalidKeyException, InvalidAlgorithmParameterException;
206 
207     /**
208      * Generate a new key pair in the specified CryptoToken with the requested alias using the key specification from
209      * another key in the same CryptoToken.
210      * @param authenticationToken Auth token
211      * @param cryptoTokenId Crypto token
212      * @param currentSignKeyAlias Signing key
213      * @param nextSignKeyAlias New signing key
214      * @throws AuthorizationDeniedException If access denied
215      * @throws CryptoTokenOfflineException If offline
216      * @throws InvalidKeyException If key invalid
217      * @throws InvalidAlgorithmParameterException If params invalid 
218      * 
219      * @see #createKeyPair(AuthenticationToken, int, String, String)
220      */
221     void createKeyPairWithSameKeySpec(final AuthenticationToken authenticationToken, final int cryptoTokenId, String currentSignKeyAlias, String nextSignKeyAlias)
222             throws AuthorizationDeniedException, CryptoTokenOfflineException, InvalidKeyException, InvalidAlgorithmParameterException;
223 
224     /**
225      * Generates a new key pair and deletes the placeholder.
226      * 
227      *@param authenticationToken Auth token
228      * @param cryptoTokenId Crypto token
229      * @param alias Alias
230      * @param keySpecification Key spec 
231      * @throws AuthorizationDeniedException If access denied
232      * @throws CryptoTokenOfflineException If offline
233      * @throws InvalidKeyException If key invalid  
234      * @throws InvalidAlgorithmParameterException If params invalid 
235      * 
236      * @see #createKeyPair(AuthenticationToken, int, String, String)
237      */
238     void createKeyPairFromTemplate(AuthenticationToken authenticationToken, int cryptoTokenId, String alias, String keySpecification)
239             throws AuthorizationDeniedException, CryptoTokenOfflineException, InvalidKeyException, InvalidAlgorithmParameterException;

243      * @param alias Alias

245      * @throws CryptoTokenOfflineException If offline

282      * @param cryptoTokenId Crypto token
283      * @param alias Alias

285      * @throws AuthorizationDeniedException If offline

287     PublicKeyWrapper getPublicKey(AuthenticationToken authenticationToken, int cryptoTokenId, String alias) throws AuthorizationDeniedException, CryptoTokenOfflineException;
288 
289     /**
290      * Set the auto-activation PIN for a CryptoToken.
291      * For soft CryptoTokens this will change the pin of the underlying soft keystore as well.
292      * For PKCS#11 CryptoTokens this will only modify the auto-activation setting.
293      * @param authenticationToken must be authorized to modify the CryptoToken
294      * @param cryptoTokenId is the CryptoToken to operate on
295      * @param currentAuthenticationCode is the pin that can currently be used to auto-activate (or manually active it if no auto-activation is used) this CryptoToken
296      * @param newAuthenticationCode is the new pin to use or null to remove the current auto-activation pin
297      * @param updateOnly if true, will only modify the auto-activation setting if already present. Soft CryptoTokens will still have a password change.
298      * @return true if the CryptoToken is auto-activated after call
299      * @throws AuthorizationDeniedException If access denied
300      * @throws CryptoTokenAuthenticationFailedException If auth fails
301      * @throws CryptoTokenOfflineException If offline
302      */
303     boolean updatePin(AuthenticationToken authenticationToken, Integer cryptoTokenId, char[] currentAuthenticationCode, char[] newAuthenticationCode,
304             boolean updateOnly) throws AuthorizationDeniedException, CryptoTokenAuthenticationFailedException, CryptoTokenOfflineException;
305 
306     /**
307      * Returns true if the alias is in use by the crypto token
308      * 
309      * @param cryptoTokenId id of the crypto token
310      * @param alias the alias in question
311      * @return true if the alias is in use.
312      */
313     boolean isAliasUsedInCryptoToken(int cryptoTokenId, String alias);
