13  package org.cesecore.keys.validation;
14  
15  import java.io.Serializable;
16  import java.util.List;
17  
18  /**
19   * POJO containing the results of importing a zip file of key validators.
20   *
21   * @version $Id: ValidatorImportResult.java 26144 2017-07-10 07:15:34Z
22   *     mikekushner $
23   */
24  public class ValidatorImportResult implements Serializable {
25  
26    private static final long serialVersionUID = 1L;
27  
28    /** Validators. */
29    private final List<Validator> importedValidators;
30    /** Validators. */
31    private final List<String> ignoredValidators;
32  
33    /**
34     * @param theimportedValidators validators
35     * @param theignoredValidators validators
36     */
37    public ValidatorImportResult(
38        final List<Validator> theimportedValidators,
39        final List<String> theignoredValidators) {
40      super();
41      this.importedValidators = theimportedValidators;
42      this.ignoredValidators = theignoredValidators;
43    }
44  
45    /**
46     * @return validators
47     */
48    public List<Validator> getImportedValidators() {
49      return importedValidators;
50    }
51  
52    /**
53     * @return Validators
54     */
55    public List<String> getIgnoredValidators() {
56      return ignoredValidators;
57    }
58  }
