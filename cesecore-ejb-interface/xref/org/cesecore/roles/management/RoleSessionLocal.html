13  package org.cesecore.roles.management;
14  
15  import java.util.List;
16  import javax.ejb.Local;
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.cesecore.roles.Role;
20  
21  /**
22   * Local interface for Role management operations.
23   *
24   * @version $Id: RoleSessionLocal.java 25642 2017-04-03 16:27:48Z jeklund $
25   */
26  @Local
27  public interface RoleSessionLocal extends RoleSession {
28  
29    /**
30     * Checks if the administrator is allowed to add/edit/remove role members from
31     * role with the given ID. Note that role member objects may reference a CA
32     * also, which must be checked for access as well.
33     *
34     * @param authenticationToken Auth token
35     * @param roleId ID
36     * @param requireEditAccess bool
37     * @throws AuthorizationDeniedException if access denied
38     */
39    void assertAuthorizedToRoleMembers(
40        AuthenticationToken authenticationToken,
41        int roleId,
42        boolean requireEditAccess)
43        throws AuthorizationDeniedException;
44  
45    /**
46     * @param authenticationToken token
47     * @return a List of Roles the caller is a member of (without taking nesting
48     *     into account)
49     */
50    List<Role> getRolesAuthenticationTokenIsMemberOf(
51        AuthenticationToken authenticationToken);
52  
53    /**
54     * @param authenticationToken token
55     * @param resource resource
56     * @return a list of all Roles that have access to the resource and the caller
57     *     is allowed to see
58     */
59    List<Role> getAuthorizedRolesWithAccessToResource(
60        AuthenticationToken authenticationToken, String resource);
61  
62    /**
63     * Update Role access rules and optionally any RoleMember relating to the CA
64     * ID.
65     *
66     * <p>Authorization to perform this operation should be checked by the caller
67     * and implied by the type of operation.
68     *
69     * @param caIdOld Old ID
70     * @param caIdNew new ID
71     * @param keepOldAccessRule when true, the /ca/(caIdOld)/ rule will be kept as
72     *     well.
73     * @param updateRoleMembers update the RoleMember.tokenIssuerId for members
74     *     issued by this CA
75     * @return bool
76     */
77    boolean updateCaId(
78        int caIdOld,
79        int caIdNew,
80        boolean keepOldAccessRule,
81        boolean updateRoleMembers);
82  
83    /**
84     * @param authenticationToken Token
85     * @param roleId ID
86     * @throws AuthorizationDeniedException if changing this role would affect the
87     *     access granted to the specified authenticationToken.
88     */
89    void assertNonImportantRoleMembership(
90        AuthenticationToken authenticationToken, int roleId)
91        throws AuthorizationDeniedException;
92  }
