13  package org.cesecore.roles.member;
14  
15  import java.util.List;
16  import java.util.Map;
17  import java.util.Set;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.authentication.AuthenticationFailedException;
22  import org.cesecore.authentication.tokens.AuthenticationToken;
23  
24  /**
25   * CRUD session bean for managing RoleMemberData objects
26   * 
27   * @version $Id: RoleMemberDataSessionLocal.java 26449 2017-08-28 20:05:53Z mikekushner $
28   */
29  @Local
30  public interface RoleMemberDataSessionLocal extends RoleMemberDataSession {
31  
32      /**
33       * Saves a RoleMember object to the database, either overwriting an existing one with the same primary key or editing an existing instance.
34       * 
35       * @param roleMember the RoleMember entity to persist.
36       * 
37       * @return the RoleMember representing what was actually persisted
38       */
39      RoleMember persistRoleMember(RoleMember roleMember);
40  
41      /**
42       * Finds an RoleMemberData by its primary key.
43       * 
44       * @param primaryKey
45       *            Primary key of the sought instance.
46       * @return the sought RoleMember, otherwise null. .
47       */
48      RoleMemberData find(final int primaryKey);
49  
50      
51      /**
52       * Removes an RoleMemberData from the database.
53       * 
54       * @param primaryKey
55       *            The ID of the RoleMemberData to remove.
56       * @return true if removal was successful, false if no such role member was found
57       */
58      boolean remove(final int primaryKey);
59      
60      /**
61       * 
62       * @param roleId the ID of a role
63       * @return a list of members to the given role
64       */
65      List<RoleMemberData> findByRoleId(int roleId);
66      
67      /**
68       * Finds an RoleMember by its primary key from the cache.
69       * 
70       * @param primaryKey
71       *            Primary key of the sought instance.
72       * @return the sought RoleMember, otherwise null. .
73       */
74      RoleMember findRoleMember(final int primaryKey);
75  
76      /**
77       * Finds all role members belonging to a specific role.
78       * 
79       * 
80       * @param roleId the ID of a role
81       * @return a list of members to the given role
82       */
83      List<RoleMember> findRoleMemberByRoleId(int roleId);
84  
85      /** @param authenticationToken Auth token
86       * @return all roleId matching the specified valid AuthenticationToken or an empty list otherwise */
87      Set<Integer> getRoleIdsMatchingAuthenticationToken(AuthenticationToken authenticationToken);
88  
89      /** 
90       * @param authenticationToken Auth token
91       * @return all roleId matching the specified valid AuthenticationToken or an empty list otherwise
92       * @throws AuthenticationFailedException if there is problem matching the AuthenticationToken to a RoleMember
93       */
94      Set<Integer> getRoleIdsMatchingAuthenticationTokenOrFail(AuthenticationToken authenticationToken) throws AuthenticationFailedException;
95  
96      /** @param authenticationToken Auth token
97       * @return roleId,tokenMatchType values for legacy priority matching  
98       * @throws AuthenticationFailedException if access denied */
99      @Deprecated // Keep for as long as we need to support upgrades to 6.8.0
100     Map<Integer, Integer> getRoleIdsAndTokenMatchKeysMatchingAuthenticationToken(AuthenticationToken authenticationToken) throws AuthenticationFailedException;
101 
102     /** @param authenticationToken Auth token
103      * @return all role members matching the specified valid AuthenticationToken or an empty list otherwise */
104     Set<RoleMember> getRoleMembersMatchingAuthenticationToken(AuthenticationToken authenticationToken);
105 }
