13  package org.cesecore.keybind;
14  
15  import java.io.Serializable;
16  import java.security.InvalidAlgorithmParameterException;
17  import java.security.InvalidKeyException;
18  import java.util.List;
19  import java.util.Map;
20  
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.authorization.AuthorizationDeniedException;
23  import org.cesecore.certificates.ca.InvalidAlgorithmException;
24  import org.cesecore.certificates.endentity.EndEntityInformation;
25  import org.cesecore.keys.token.CryptoTokenOfflineException;
26  import org.cesecore.util.ui.DynamicUiProperty;
27  
28  /**
29   * Generic Management interface for InternalKeyBinding.
30   * 
31   * @version $Id: InternalKeyBindingMgmtSession.java 23648 2016-06-09 15:19:55Z mikekushner $
32   */
33  public interface InternalKeyBindingMgmtSession {
34  
35      /** 
36       * Get all registered InternalKeyBinding implementation and their properties.
37       * 
38       * @return a map where each entry is a registered implementation type and the value is a list of implementation specific properties for the type
39       */
40      Map<String, Map<String, DynamicUiProperty<? extends Serializable>>> getAvailableTypesAndProperties();
41      
42      /**
43       * Get a list of all InternalKeyBindings of the requested type, regardless of authorization.
44       * 
45       * @param internalKeyBindingType is the identifier of the type of InternalKeyBinding
46       * @return a list of IDs for the specific type and that the caller is authorized to view
47       */
48      List<Integer> getInternalKeyBindingIds(String internalKeyBindingType);
49      
50      /**
51       * Get a list of all InternalKeyBindings the caller is authorized to view of the requested type.
52       * 
53       * @param authenticationToken is the authentication token
54       * @param internalKeyBindingType is the identifier of the type of InternalKeyBinding
55       * @return a list of IDs for the specific type and that the caller is authorized to view
56       */
57      List<Integer> getInternalKeyBindingIds(AuthenticationToken authenticationToken, String internalKeyBindingType);
58  
59      /**
60       * Get a (cloned) InternalKeyBinding object. Use this method if you might change the object,
61       * if the object is handled by an untrusted piece of code or if you are using Remote invocation.
62       * 
63       * (Otherwise, use {@link InternalKeyBindingMgmtSessionLocal#getInternalKeyBindingReference(AuthenticationToken, int)}
64       * to avoid the unnecessary object creation.)
65       * 
66       * @param authenticationToken is the authentication token
67       * @param internalKeyBindingId is the identifier of the InternalKeyBinding
68       * @return the InternalKeyBinding for the requested Id or null if none was found
69       * @throws AuthorizationDeniedException if the authentication token was not authorized to fetch the requested InternalKeyBinding
70       */
71      InternalKeyBinding getInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException;
72  
73      /**
74       * Get the InternalKeyBinding's identifier from a user friendly name.
75       * 
76       * @param internalKeyBindingName is the (unique) name of the InternalKeyBinding
77       * @return the internalKeyBindingId from the more user friendly name. Return null of there is no such InternalKeyBinding.
78       */
79      Integer getIdFromName(String internalKeyBindingName);
80  
81      /** 
82       * Creates a new InternalKeyBinding using the InternalKeyBindingFactory on the server side.
83       * 
84       * @param authenticationToken is the authentication token
85       * @param type is the alias of the registered InternalKeyBinding's type
86       * @param id is the id to use for the new InternalKeyBinding, or 0 for automatic choice
87       * @param name is the unique name that this InternalKeyBinding will be given
88       * @param status the initial status to give the InternalKeyBinding
89       * @param certificateId is the certificate fingerprint matching the mapped key pair or null
90       * @param cryptoTokenId is the CryptoToken id of the container where the mapped key pair is stored
91       * @param keyPairAlias is the alias of the mapped key pair in the specified CryptoToken (may not be null)
92       * @param allowMissingKeyPair if a missing key pair or crypto token should be allowed
93       * @param signatureAlgorithm is the signature algorithm that this InternalKeyBinding will use for signatures (if applicable)
94       * @param dataMap is a Map of implementation specific properties for this type of IntenalKeyBinding
95       * @param trustedCertificateReferences Trusted Certificates
96       * @return the created InternalKeyBinding's unique identifier
97       * 
98       * @throws CryptoTokenOfflineException if the requested key pair was not accessible
99       * @throws AuthorizationDeniedException if the authentication token was not authorized to create the InternalKeyBinding
100      * @throws InternalKeyBindingNameInUseException if the requested name was already in use by another InternalKeyBinding
101      * @throws InvalidAlgorithmException if the requested signature algorithm is not available
102      */
103     int createInternalKeyBinding(AuthenticationToken authenticationToken, String type, int id, String name, InternalKeyBindingStatus status, String certificateId,
104             int cryptoTokenId, String keyPairAlias, boolean allowMissingKeyPair, String signatureAlgorithm, Map<String, Serializable> dataMap,
105             List<InternalKeyBindingTrustEntry> trustedCertificateReferences)
106                     throws AuthorizationDeniedException, CryptoTokenOfflineException, InternalKeyBindingNameInUseException, InvalidAlgorithmException;
107     
108     /**
109      * A createInternalKeyBinding() with allowMissingKeyPair=false
110      * @param authenticationToken is the authentication token
111      * @param type is the alias of the registered InternalKeyBinding's type
112      * @param id is the id to use for the new InternalKeyBinding, or 0 for automatic choice
113      * @param name is the unique name that this InternalKeyBinding will be given
114      * @param status the initial status to give the InternalKeyBinding
115      * @param certificateId is the certificate fingerprint matching the mapped key pair or null
116      * @param cryptoTokenId is the CryptoToken id of the container where the mapped key pair is stored
117      * @param keyPairAlias is the alias of the mapped key pair in the specified CryptoToken (may not be null)
118      * @param signatureAlgorithm is the signature algorithm that this InternalKeyBinding will use for signatures (if applicable)
119      * @param dataMap is a Map of implementation specific properties for this type of IntenalKeyBinding
120      * @param trustedCertificateReferences Trusted Certificates
121      * @return the created InternalKeyBinding's unique identifier
122      *  @throws CryptoTokenOfflineException if the requested key pair was not accessible
123      * @throws AuthorizationDeniedException if the authentication token was not authorized to create the InternalKeyBinding
124      * @throws InternalKeyBindingNameInUseException if the requested name was already in use by another InternalKeyBinding
125      * @throws InvalidAlgorithmException if the requested signature algorithm is not available
126      * @see #createInternalKeyBinding
127      */
128     int createInternalKeyBinding(AuthenticationToken authenticationToken, String type, int id, String name, InternalKeyBindingStatus status, String certificateId,
129             int cryptoTokenId, String keyPairAlias, String signatureAlgorithm, Map<String, Serializable> dataMap,
130             List<InternalKeyBindingTrustEntry> trustedCertificateReferences)
131                     throws AuthorizationDeniedException, CryptoTokenOfflineException, InternalKeyBindingNameInUseException, InvalidAlgorithmException;
132 
133     /**
134      * A createInternalKeyBinding() that chooses the id of the new object automatically.
135      * @param authenticationToken is the authentication token
136      * @param type is the alias of the registered InternalKeyBinding's type
137      * @param name is the unique name that this InternalKeyBinding will be given
138      * @param status the initial status to give the InternalKeyBinding
139      * @param certificateId is the certificate fingerprint matching the mapped key pair or null
140      * @param cryptoTokenId is the CryptoToken id of the container where the mapped key pair is stored
141      * @param keyPairAlias is the alias of the mapped key pair in the specified CryptoToken (may not be null)
142      * @param signatureAlgorithm is the signature algorithm that this InternalKeyBinding will use for signatures (if applicable)
143      * @param dataMap is a Map of implementation specific properties for this type of IntenalKeyBinding
144      * @param trustedCertificateReferences Trusted Certificates
145      * @return the created InternalKeyBinding's unique identifier
146      *  @throws CryptoTokenOfflineException if the requested key pair was not accessible
147      * @throws AuthorizationDeniedException if the authentication token was not authorized to create the InternalKeyBinding
148      * @throws InternalKeyBindingNameInUseException if the requested name was already in use by another InternalKeyBinding
149      * @throws InvalidAlgorithmException if the requested signature algorithm is not available
150      * @see #createInternalKeyBinding
151      */
152     int createInternalKeyBinding(AuthenticationToken authenticationToken, String type, String name, InternalKeyBindingStatus status, String certificateId,
153             int cryptoTokenId, String keyPairAlias, String signatureAlgorithm, Map<String, Serializable> dataMap,
154             List<InternalKeyBindingTrustEntry> trustedCertificateReferences) throws AuthorizationDeniedException, CryptoTokenOfflineException,
155             InternalKeyBindingNameInUseException, InvalidAlgorithmException;
156 
157     /**
158      * Create new (when the provided InternalKeyBinding has id 0) or merge existing InternalKeyBinding.
159      * The caller must be authorized the modify the InternalKeyBinding and to CryptoTokenRules.USE the referenced CryptoToken.
160      * 
161      * @param authenticationToken is the authentication token
162      * @param internalKeyBinding is the InternalKeyBinding to persist
163      * @return the internalKeyBindingId that can be used for future reference if the object was created
164      * @throws AuthorizationDeniedException if the authentication token is not authorized to modify the InternalKeyBinding or use the CryptoToken
165      * @throws InternalKeyBindingNameInUseException if the name is already in use by an InternalKeyBinding with a different id
166      */
167     int persistInternalKeyBinding(AuthenticationToken authenticationToken, InternalKeyBinding internalKeyBinding) throws AuthorizationDeniedException, InternalKeyBindingNameInUseException;
168 
169     /**
170      * Delete an InternalKeyBinding (this will not remove any referenced object like keys or certificates).
171      * @param authenticationToken is the authentication token
172      * @param internalKeyBindingId is the unique InternalKeyBinding identifier
173      * @return true if the InternalKeyBinding existed before deletion
174      * @throws AuthorizationDeniedException if the authentication token is not authorized to delete this InternalKeyBinding
175      */
176     boolean deleteInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException;
177     
178     /**
179      * Get the next key pair alias to use for renewals. If the next key field of the InternalKeyBinding is not set, the
180      * current key pair alias will be returned instead.
181      * 
182      * @param authenticationToken is the authentication token
183      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
184      * @return the public key in DER format 
185      * @throws AuthorizationDeniedException if the authentication token is not authorized to get this information
186      * @throws CryptoTokenOfflineException if the public key could not be retrieved from the referenced CryptoToken
187      */
188     byte[] getNextPublicKeyForInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException, CryptoTokenOfflineException;
189 
190     /**
191      * Generate a Certificate Signing Request (CSR) for the next key to be mapped by this InternalKeyBinging. If
192      * no explicit "next" key exists a CSR for the current key will be returned (e.g. only renew cert, but not keys).
193      * 
194      * @param authenticationToken is the authentication token
195      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
196      * @param name the subjectDN to be used in the CSR as binary encoded X500Name (X500Name.getEncoded), or null. If null the currently mapped certificate's SubjectDN will be used (if present) and if not a default (CN=Internal Key Binding Name), 
197      * @return the a new PKCS#10 request for the InternalKeyBinding
198      * @throws AuthorizationDeniedException is the authentication token is not authorized to this operation
199      * @throws CryptoTokenOfflineException if the key pair is not available
200      */
201     byte[] generateCsrForNextKey(AuthenticationToken authenticationToken, int internalKeyBindingId, byte[] name) throws AuthorizationDeniedException, CryptoTokenOfflineException;
202 
203     /** 
204      * Update the key mapping if there is a newer certificate in the database or a certificate matching the nextKey.
205      * This could normally be used in a setup where the certificate is published or made available be other means
206      * in the database for this EJBCA instance.
207      * 
208      * @param authenticationToken is the authentication token
209      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
210      * @return the certificate's fingerprint if a change was made or null otherwise
211      * @throws AuthorizationDeniedException if the authentication token was not authorized to this operation
212      * @throws CertificateImportException if the certificate could not be mapped (keys unavailable or wrong type of certificate)
213      */
214     String updateCertificateForInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException,
215             CertificateImportException;
216 
217     /**
218      * Imports the certificate provided in DER format to the database and updates the InternalKeyBinding reference.
219      * If the the certificates public key matches the current instance's keyPairAlias, the keyPairAlias will not be updated.
220      * If the nextKey property is set and the certificates public key matches, the instance's keyPairAlias will also be updated.
221      * 
222      * @param authenticationToken is the authentication token
223      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
224      * @param certificate DER encoded certificate to import
225      * @throws AuthorizationDeniedException if the authentication token was not authorized to this operation
226      * @throws CertificateImportException if the provided certificate's public key does not match current or next key. This is also
227      * thrown if the implementation cannot validate that the certificate is of the right type.
228      */
229     void importCertificateForInternalKeyBinding(AuthenticationToken authenticationToken, int internalKeyBindingId, byte[] certificate)
230         throws AuthorizationDeniedException, CertificateImportException;
231 
232     /**
233      * Creates a new key pair with the same key specification as the current and a new alias.
234      * If a nextKey reference already exists, it will be replaced with a reference to the new key.
235      * 
236      * @param authenticationToken is the authentication token
237      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
238      * @return the new key pair alias
239      * @throws AuthorizationDeniedException if the authentication token was not authorized to this operation
240      * @throws CryptoTokenOfflineException if the CryptoToken mapped by this InternalKeyBinding is not available
241      * @throws InvalidKeyException if the current key spec could not be read and used properly
242      * @throws InvalidAlgorithmParameterException if the current key algo could not be read and used properly
243      */
244     String generateNextKeyPair(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException,
245             CryptoTokenOfflineException, InvalidKeyException, InvalidAlgorithmParameterException;
246 
247     /**
248      * Issue a new certificate using the same end entity as the current certificate.
249      * Note that this will only work as long as the CA is internal and the user and profiles still exist.
250      * Since the user's current CA and profiles will be used, they could be changed before this call.
251      * 
252      * @param authenticationToken is the authentication token
253      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
254      * @param endEntityInformation is the template to use for the renewal
255      * @return the newly issued certificate's fingerprint
256      * @throws AuthorizationDeniedException if the authentication token is not authorized to this operation
257      * @throws CryptoTokenOfflineException if the mapped CryptoToken or keys are not available
258      * @throws CertificateImportException if the newly issued certificate cannot be used with this InternalKeyBinding (e.g. the EEP might have changed)
259      */
260     String renewInternallyIssuedCertificate(AuthenticationToken authenticationToken, int internalKeyBindingId, EndEntityInformation endEntityInformation)
261             throws AuthorizationDeniedException, CryptoTokenOfflineException, CertificateImportException;
262     
263     /**
264      * Suitable for remote invocation where the implementation might not be available.
265      * 
266      * @param authenticationToken is the authentication token
267      * @param internalKeyBindingType is the alias for a registered type of InternalKeyBindings
268      * @return a list of authorized InternalKeyBindings that extend a non-mutable general class.
269      */
270     List<InternalKeyBindingInfo> getInternalKeyBindingInfos(AuthenticationToken authenticationToken, String internalKeyBindingType);
271 
272     /**
273      * Suitable for remote invocation where the implementation might not be available.
274      * 
275      * @param authenticationToken is the authentication token
276      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
277      * @return a InternalKeyBindingInfo, or null if the key binding does not exist
278      * @throws AuthorizationDeniedException of the authentication token is not authorized to this operation
279      */
280     InternalKeyBindingInfo getInternalKeyBindingInfo(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException;
281 
282     /**
283      * Update the status of the InternalKeyBinding. Note that it is not possible to enable an InternalKeyBinding that
284      * has no certificate mapping.
285      * 
286      * @param authenticationToken is the authentication token
287      * @param internalKeyBindingId is the unique identifier of the InternalKeyBinding
288      * @param status is one of org.cesecore.keybind.InternalKeyBindingStatus enum
289      * @return true if the status was modified
290      * @throws AuthorizationDeniedException of the authentication token is not authorized to this operation
291      */
292     boolean setStatus(AuthenticationToken authenticationToken, int internalKeyBindingId, InternalKeyBindingStatus status) throws AuthorizationDeniedException;
293 }
