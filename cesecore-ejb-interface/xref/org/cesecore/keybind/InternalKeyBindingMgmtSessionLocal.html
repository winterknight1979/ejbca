13  package org.cesecore.keybind;
14  
15  import java.security.cert.X509Certificate;
16  import java.util.Collection;
17  import java.util.List;
18  import java.util.Map;
19  import javax.ejb.Local;
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.ca.CADoesntExistsException;
23  
24  /**
25   * @see InternalKeyBindingMgmtSession
26   * @version $Id: InternalKeyBindingMgmtSessionLocal.java 28216 2018-02-08
27   *     12:19:01Z henriks $
28   */
29  @Local
30  public interface InternalKeyBindingMgmtSessionLocal
31      extends InternalKeyBindingMgmtSession {
32  
33    /**
34     * Returns a list of all internal key bindings of a certain type, as {@link
35     * InternalKeyBindingInfo}s.
36     *
37     * @param internalKeyBindingType the key binding type
38     * @return a list of all internal key bindings of that type, as {@link
39     *     InternalKeyBindingInfo}s
40     */
41    List<InternalKeyBindingInfo> getAllInternalKeyBindingInfos(
42        String internalKeyBindingType);
43  
44    /**
45     * Internal (local only) method to get keybinding info without logging the
46     * authorization check (the auth check is performed though).
47     *
48     * @param authenticationToken Token
49     * @param internalKeyBindingId Binding
50     * @return Binding Info
51     * @throws AuthorizationDeniedException if access denied
52     * @see #getInternalKeyBindingInfo
53     */
54    InternalKeyBindingInfo getInternalKeyBindingInfoNoLog(
55        AuthenticationToken authenticationToken, int internalKeyBindingId)
56        throws AuthorizationDeniedException;
57  
58    /**
59     * Get a reference to a cached InternalKeyBinding object that MAY NOT be
60     * modified.
61     *
62     * @param authenticationToken is the authentication token
63     * @param internalKeyBindingId is the identifier of the InternalKeyBinding
64     * @return the InternalKeyBinding for the requested Id or null if none was
65     *     found
66     * @throws AuthorizationDeniedException if the authentication token was not
67     *     authorized to fetch the requested InternalKeyBinding
68     */
69    InternalKeyBinding getInternalKeyBindingReference(
70        AuthenticationToken authenticationToken, int internalKeyBindingId)
71        throws AuthorizationDeniedException;
72  
73    /**
74     * Returns a collection of the trusted certificates defined in
75     * internalKeyBinding along with their issuers' certificate chains.
76     *
77     * <p>- If the list of trusted certificates in internalKeyBinding contains
78     * certificate serial number(s), only the certificates with these specific
79     * serial numbers are trusted - If only trusted CAs are specified in
80     * internalKeyBinding, all certificates issued by the specified CAs will be
81     * trusted. - If the list of trusted certificates in internalKeyBinding is
82     * empty, all certificates issued by all CAs known to this instance of EJBCA
83     * are trusted - If the list of trusted certificates in internalKeyBinding is
84     * null, no certificates will be trusted.
85     *
86     * @param internalKeyBinding Binding
87     * @return a collection of the trusted certificates along with their issuers'
88     *     certificate chains or null if no trusted certificates or CAs are
89     *     specified
90     * @throws CADoesntExistsException IF CA not found
91     */
92    List<Collection<X509Certificate>> getListOfTrustedCertificates(
93        InternalKeyBinding internalKeyBinding) throws CADoesntExistsException;
94  
95    /**
96     * Returns a id-to-name map of all internal key bindings of the given type.
97     *
98     * @param internalKeyBindingType null for all types
99     * @return id-name map of internal key bindings
100    */
101   Map<Integer, String> getAllInternalKeyBindingIdNameMap(
102       String internalKeyBindingType);
103 }
