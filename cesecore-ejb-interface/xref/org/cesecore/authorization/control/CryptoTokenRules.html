13  package org.cesecore.authorization.control;
14  
15  /**
16   * CryptoToken related access rules.
17   *
18   * @version $Id: CryptoTokenRules.java 17625 2013-09-20 07:12:06Z netmackan $
19   */
20  public enum CryptoTokenRules {
21      /** Base. */
22    BASE("/cryptotoken"),
23    /** Modify. */
24    MODIFY_CRYPTOTOKEN(BASE.resource() + "/modify"),
25    /** Delete. */
26    DELETE_CRYPTOTOKEN(BASE.resource() + "/delete"),
27    /** View. */
28    VIEW(BASE.resource() + "/view"),
29    /** Use. */
30    USE(BASE.resource() + "/use"),
31    /** Activate. */
32    ACTIVATE(BASE.resource() + "/activate"),
33    /** Deactivate. */
34    DEACTIVATE(BASE.resource() + "/deactivate"),
35    /** Keys. */
36    GENERATE_KEYS(BASE.resource() + "/keys/generate"),
37    /** Keys. */
38    REMOVE_KEYS(BASE.resource() + "/keys/remove"),
39    /** Keys. */
40    TEST_KEYS(BASE.resource() + "/keys/test");
41  
42    /** Tesource. */
43    private final String resource;
44  
45    /**
46     * @param aResource resource
47     */
48    CryptoTokenRules(final String aResource) {
49      this.resource = aResource;
50    }
51  
52    /**
53     * @return resource
54     */
55    public String resource() {
56      return this.resource;
57    }
58  
59    @Override
60    public String toString() {
61      return this.resource;
62    }
63  }
