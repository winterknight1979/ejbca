13  package org.cesecore.authorization;
14  
15  import java.util.HashMap;
16  
17  import javax.ejb.Local;
18  import javax.ejb.Timer;
19  
20  import org.cesecore.authentication.AuthenticationFailedException;
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  
23  /**
24   * 
25   * @version $Id: AuthorizationSessionLocal.java 25407 2017-03-07 15:08:07Z jeklund $
26   */
27  @Local
28  public interface AuthorizationSessionLocal extends AuthorizationSession {
29  
30      /** Invoked when authorization cache should be checked for updates. */
31      void refreshAuthorizationCache();
32  
33      /** Invoked by background cache refresh timeouts 
34       * @param timer timer*/
35      void timeOut(Timer timer);
36  
37      /** Initialize background cache refresh timeouts */
38      void scheduleBackgroundRefresh();
39  
40      /** @param authenticationToken token
41       * @return the access rules available to the AuthenticationToken and its nested tokens, taking each such tokens role membership into account 
42       * @throws AuthenticationFailedException access denied */
43      HashMap<String, Boolean> getAccessAvailableToAuthenticationToken(AuthenticationToken authenticationToken) throws AuthenticationFailedException;
44  }
