13  package org.cesecore.audit.audit;

15  import java.io.File;
16  import java.io.FileNotFoundException;
17  import java.io.FileOutputStream;
18  import java.io.IOException;
19  import java.nio.charset.Charset;
20  import java.security.InvalidKeyException;
21  import java.security.NoSuchAlgorithmException;
22  import java.security.NoSuchProviderException;
23  import java.security.PrivateKey;
24  import java.security.PublicKey;
25  import java.security.Signature;
26  import java.security.SignatureException;
27  import java.security.cert.Certificate;
28  import java.util.Map;
29  
30  import org.apache.commons.io.FileUtils;
31  import org.apache.commons.io.FilenameUtils;
32  import org.cesecore.keys.token.CryptoToken;
33  import org.cesecore.keys.token.CryptoTokenOfflineException;
34  import org.cesecore.util.Base64;
35  
36  /**
37   * Wrapper of a FileOutputStream that also produces a signature for the same data
38   * that was written to disk. This way we know that we are not signing manipulated
39   * data read back from the disk.
40   * 
41   * The signature is written as raw data to a separate file with the extension ".sig".
42   * 
43   * @version $Id: SigningFileOutputStream.java 27947 2018-01-15 14:20:08Z mikekushner $
44   */
45  /* TODO: Rename/move to org.cesecore.audit.impl.BasicSigningFileOutputStream
46   * TODO: Extract reasonable interface to allow different implementations.
47   * TODO: Perhaps pass properties from config?
48   */
49  public class SigningFileOutputStream extends FileOutputStream {
50  
51  	public static final String EXPORT_SIGN_CERT     = "cert";
52      public static final String EXPORT_SIGN_ALG      = "alg";
53      public static final String EXPORT_SIGN_KEYALIAS = "key";
54  
55  	final Signature signature;
56  	final Signature signValidate;
57  	final String signatureFilename;
58  	
59      /**
60       * Generates a signature file with the same name as the export file but with .sig extension.
61       * 
62       * @param file the exported file.
63       * @param cryptoToken the crypto token that will be used to fetch the necessary keys.
64       * @param signatureDetails
65       *            Set properties containing signature details like
66       *            keyAlias(EXPORT_SIGN_KEYALIAS), algorithm(EXPORT_SIGN_ALG) and
67       *            certificate( EXPORT_SIGN_CERT ).
68       * @throws FileNotFoundException If signature file not found
69       * @throws CryptoTokenOfflineException If offline
70       * @throws NoSuchAlgorithmException If algorithm not found
71       * @throws NoSuchProviderException If provider not found
72       * @throws InvalidKeyException If key is invalid
73       */
74      public SigningFileOutputStream(final File file, final CryptoToken cryptoToken, final Map<String, Object> signatureDetails) throws FileNotFoundException, CryptoTokenOfflineException, NoSuchAlgorithmException, NoSuchProviderException, InvalidKeyException {
75  		super(file);
76      	signatureFilename = String.format("%s.sig", FilenameUtils.removeExtension(file.getAbsolutePath()));
77      	final String keyAlias = (String) signatureDetails.get(SigningFileOutputStream.EXPORT_SIGN_KEYALIAS);
78      	final PrivateKey privateKey = cryptoToken.getPrivateKey(keyAlias);
79      	final PublicKey publicKey = cryptoToken.getPublicKey(keyAlias);
80      	final String algorithm = (String) signatureDetails.get(SigningFileOutputStream.EXPORT_SIGN_ALG);
81      	signature = Signature.getInstance(algorithm, cryptoToken.getSignProviderName());
82      	signature.initSign(privateKey);
83      	signValidate = Signature.getInstance(algorithm, cryptoToken.getSignProviderName());
84      	final Certificate cert = (Certificate) signatureDetails.get(SigningFileOutputStream.EXPORT_SIGN_CERT);
85      	if (cert != null) {
86      		signValidate.initVerify(cert);
87      	} else {
88      		signValidate.initVerify(publicKey);
89      	}
90  	}
91      
92      @Override
93      public void write(final byte[] b) throws IOException {
94      	super.write(b);
95      	try {
96  			signature.update(b);
97  			signValidate.update(b);
98  		} catch (SignatureException e) {
99  			throw new IOException(e);
100 		}
101     }
102     
103     @Override
104     public void write(final int b) throws IOException {
105     	super.write(b);
106     	try {
107 			signature.update((byte) b);
108 			signValidate.update((byte) b);
109 		} catch (SignatureException e) {
110 			throw new IOException(e);
111 		}
112     }
113 
114     @Override
115     public void write(final byte[] b, final int off, final int len) throws IOException {
116     	super.write(b, off, len);
117     	try {
118 			signature.update(b, off, len);
119 			signValidate.update(b, off, len);
120 		} catch (SignatureException e) {
121 			throw new IOException(e);
122 		}
123     }
124     
125     public String writeSignature() throws SignatureException, AuditLogExporterException, IOException {
126     	byte[] signedData = signature.sign();
127     	if (!signValidate.verify(signedData)) {
128     		throw new AuditLogExporterException("export file signature mismatch");
129     	}
130     	FileUtils.writeStringToFile(new File(signatureFilename), new String(Base64.encode(signedData)), Charset.defaultCharset());
131     	return signatureFilename;
132     }
133 }
