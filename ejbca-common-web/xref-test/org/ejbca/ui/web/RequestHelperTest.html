95      @SuppressWarnings("unchecked")
96      @Test
97      public void testPkcs10CertRequestWithCertificateChain() throws Exception {
98          RequestHelper requestHelper = new RequestHelper(null, null);
99  
100         //Generate a self signed certificate to act as a CA cert, and a signed certificate.
101         KeyPair caKeys = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA);
102         Certificate caCert = CertTools.genSelfCert("CN=foo", 365, null, caKeys.getPrivate(), caKeys.getPublic(),
103                 AlgorithmConstants.SIGALG_SHA1_WITH_RSA, false);
104         KeyPair replyKeys = KeyTools.genKeys("512", AlgorithmConstants.KEYALGORITHM_RSA);
105         final SubjectPublicKeyInfo pkinfo = SubjectPublicKeyInfo.getInstance(replyKeys.getPublic().getEncoded());
106         String signedCertDn = "CN=signedcert";
107         byte[] serno = new byte[8];
108         SecureRandom random = SecureRandom.getInstance("SHA1PRNG");
109         random.setSeed(new Date().getTime());
110         random.nextBytes(serno);
111         Date firstDate = new Date();
112         // Set back startdate ten minutes to avoid some problems with wrongly set clocks.
113         firstDate.setTime(firstDate.getTime() - (10 * 60 * 1000));
114         Date lastDate = new Date();
115         // validity in days = validity*24*60*60*1000 milliseconds
116         lastDate.setTime(lastDate.getTime() + (24 * 60 * 60 * 1000));
117         X509v3CertificateBuilder certbuilder = new X509v3CertificateBuilder(CertTools.stringToBcX500Name("CN=foo"),
118                 new java.math.BigInteger(serno).abs(), firstDate, lastDate, CertTools.stringToBcX500Name(signedCertDn), pkinfo);
119         final ContentSigner signer = new BufferingContentSigner(new JcaContentSignerBuilder(AlgorithmConstants.SIGALG_SHA1_WITH_RSA).setProvider(BouncyCastleProvider.PROVIDER_NAME).build(caKeys
120                 .getPrivate()), 20480);
121         final X509CertificateHolder certHolder = certbuilder.build(signer);
122         final X509Certificate signedCert = CertTools.getCertfromByteArray(certHolder.getEncoded(), X509Certificate.class);
123 
124         //Setup mocks
125         SignSessionLocal signsession = EasyMock.createMock(SignSessionLocal.class);
126         ResponseMessage responseMessage = EasyMock.createMock(X509ResponseMessage.class);
127         //EasyMock.expect(signsession.createCertificate(authenticationToken, EasyMock.anyObject(RequestMessage.class), X509ResponseMessage.class, null)).andReturn(responseMessage);
128         EasyMock.expect(
129                 signsession.createCertificate(EasyMock.anyObject(AuthenticationToken.class), EasyMock.anyObject(RequestMessage.class),
130                         EasyMock.anyObject(X509ResponseMessage.class.getClass()), EasyMock.anyObject(EndEntityInformation.class))).andReturn(
131                 responseMessage);
132         EasyMock.expect(responseMessage.getResponseMessage()).andReturn(signedCert.getEncoded());
133         CaSessionLocal caSession = EasyMock.createMock(CaSessionLocal.class);
134         CA ca = EasyMock.createMock(CA.class);
135         EasyMock.expect(signsession.getCAFromRequest(EasyMock.anyObject(AuthenticationToken.class), EasyMock.anyObject(RequestMessage.class),
136                 EasyMock.anyBoolean())).andReturn(ca);
137         CAInfo caInfo = EasyMock.createMock(CAInfo.class);
138         EasyMock.expect(ca.getCAInfo()).andReturn(caInfo);
139         EasyMock.expect(caInfo.getCertificateChain()).andReturn(Arrays.asList(caCert));
140         EasyMock.replay(caInfo, ca, responseMessage, signsession, caSession);
141 
142         //Perform test
143         byte[] result = requestHelper.pkcs10CertRequest(signsession, caSession, PRE_GENERATED_CSR, "foo", "foo123", CertificateResponseType.ENCODED_CERTIFICATE_CHAIN).getEncoded();
144         List<Certificate> certChain =  CertTools.getCertsFromPEM(new ByteArrayInputStream(result), Certificate.class);
145         assertEquals(signedCert, certChain.get(0));
146         assertEquals(caCert, certChain.get(1));
147         
148         //Verify that mocks have behaved as planned
149         EasyMock.verify(caInfo, ca, responseMessage, signsession, caSession);
150     }

152     /** Tests RequestHelper.getRequestServerName that it handles different protocols, with and without port, and that it does HTML encoding on the result
153      */
154     @Test
155     public void testGetRequestServerName() {
156         assertNull("getRequestServerName did not handle null", RequestHelper.getRequestServerName(null));
157         assertEquals("getRequestServerName did not handle https without port correctly", "servername", RequestHelper.getRequestServerName("https://servername/foo/bar.html"));
158         assertEquals("getRequestServerName did not handle https with port correctly", "servername:8443", RequestHelper.getRequestServerName("https://servername:8443/foo/bar.html"));
159         assertEquals("getRequestServerName did not handle http without port correctly", "servername", RequestHelper.getRequestServerName("https://servername/foo/bar.html"));
160         assertEquals("getRequestServerName did not handle http with port correctly", "servername:11111", RequestHelper.getRequestServerName("https://servername:11111/foo/bar.html"));
161         assertEquals("getRequestServerName did not handle ajp without port correctly", "servername", RequestHelper.getRequestServerName("ajp://servername/foo/bar.html"));
162         assertEquals("getRequestServerName did not handle ajp with port correctly", "servername:3", RequestHelper.getRequestServerName("ajp://servername:3/foo/bar.html"));
163         assertEquals("getRequestServerName did not handle bogusprotocol without port correctly", "servername", RequestHelper.getRequestServerName("bogusprotocol://servername/foo/bar.html"));
164         assertEquals("getRequestServerName did not handle bogusprotocol with port correctly", "servername:8080", RequestHelper.getRequestServerName("bogusprotocol://servername:8080/foo/bar.html"));
165         assertEquals("getRequestServerName did not do HTML encoding ", "&gt;serve&amp;rname&lt;:8443", RequestHelper.getRequestServerName("https://>serve&rname<:8443/foo/bar.html"));
166     }
