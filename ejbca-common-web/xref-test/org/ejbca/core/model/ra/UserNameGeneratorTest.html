35  		UsernameGeneratorParams usernameGeneratorParams = new UsernameGeneratorParams();
36  		usernameGeneratorParams.setMode("DN");
37  		usernameGeneratorParams.setDNGeneratorComponent("SN;CN");
38  		usernameGeneratorParams.setPrefix(null);
39  		usernameGeneratorParams.setPostfix(null);
40  		UsernameGenerator usernameGenerator = UsernameGenerator.getInstance(usernameGeneratorParams);
41  
42  		final String errorMessage = "Did not generate an expected username.";
43  		assertEquals(errorMessage, "test", usernameGenerator.generateUsername(new X500Name("CN=test").toString()));
44  		assertEquals(errorMessage, null, usernameGenerator.generateUsername("".toString()));
45  		assertEquals(errorMessage, null, usernameGenerator.generateUsername(" ".toString()));
46  		assertEquals(errorMessage, "test", usernameGenerator.generateUsername(new X500Name("CN=test, serialNumber=1234").toString()));
47  		assertEquals(errorMessage, null, usernameGenerator.generateUsername(new X500Name("O=org").toString()));
48  		assertEquals(errorMessage, "12345", usernameGenerator.generateUsername("CN=test, SN=12345"));
49  		assertEquals(errorMessage, "1234", usernameGenerator.generateUsername("SN=1234"));

51  		// These wont work since new X509Name converts SN to SERIALNUMBER in toString()

60  	@Test
61  	public void testUsernameGeneratorRandom() {
62  		UsernameGenerator gen = UsernameGenerator.getInstance(UsernameGeneratorParams.RANDOM);
63  		String u = gen.generateUsername();
64  		assertEquals(u.length(), 12);
65  		
66  		UsernameGeneratorParams params = gen.getParams();
67  		params.setRandomNameLength(5);
68  		gen.setParams(params);
69  		u = gen.generateUsername();
70  		assertEquals(u.length(), 5);
71  		
72  		params.setPrefix("foo-");
73  		gen.setParams(params);
74  		u = gen.generateUsername();
75  		assertEquals(u.length(), 9);
76  		assertTrue(u.startsWith("foo-"));
77  
78  		params.setPostfix("-foo");
79  		params.setPrefix(null);
80  		gen.setParams(params);
81  		u = gen.generateUsername();
82  		assertEquals(u.length(), 9);
83  		assertTrue(u.endsWith("-foo"));
84  		
85  		params.setPrefix("foo-${RANDOM}");
86  		params.setPostfix(null);
87  		params.setRandomPrefixLength(6);
88  		gen.setParams(params);
89  		u = gen.generateUsername();
90  		assertEquals(u.length(), 15);
91  		assertTrue(u.startsWith("foo-"));
92  
93  		params.setPostfix("${RANDOM}-foo");
94  		params.setPrefix(null);
95  		params.setRandomPrefixLength(5);
96  		gen.setParams(params);
97  		u = gen.generateUsername();
98  		assertEquals(u.length(), 14);
99  		assertTrue(u.endsWith("-foo"));
100 		
101 		params.setPrefix("foo-");
102 		gen.setParams(params);
103 		u = gen.generateUsername();
104 		assertEquals(u.length(), 18);
105 		assertTrue(u.endsWith("-foo"));
106 		assertTrue(u.startsWith("foo-"));
107 	}

109 	@Test
110 	public void testUsernameGeneratorDN() {
111 		String dn = "C=SE, O=FooO, UID=foo, CN=bar";
112         String dnReverse = "CN=bar,UID=foo,O=FooO,C=SE";
113 		UsernameGenerator gen = UsernameGenerator.getInstance(UsernameGeneratorParams.DN);
114 		String u = gen.generateUsername(dn);
115 		assertEquals(u, "bar");
116         u = gen.generateUsername(dnReverse);
117         assertEquals(u, "bar");
118 
119 		UsernameGeneratorParams params = gen.getParams();
120         params.setDNGeneratorComponent("");
121         gen = UsernameGenerator.getInstance(params);
122         u = gen.generateUsername(dn);
123         assertEquals("Username was not constructed properly from DN", "C=SE, O=FooO, UID=foo, CN=bar", u);
124         u = gen.generateUsername(dnReverse);
125         assertEquals("Username was not constructed properly from DN", "CN=bar,UID=foo,O=FooO,C=SE", u);
126 
127 		params.setDNGeneratorComponent("UID");
128 		gen.setParams(params);
129 		u = gen.generateUsername(dn);
130 		assertEquals(u, "foo");
131 		
132 		params.setPrefix("pre-");
133 		gen.setParams(params);
134 		u = gen.generateUsername(dn);
135 		assertEquals(u, "pre-foo");
136 
137 		params.setPostfix("-post");
138 		params.setPrefix(null);
139 		gen.setParams(params);
140 		u = gen.generateUsername(dn);
141 		assertEquals(u, "foo-post");
142 		
143 		params.setPrefix("pre-${RANDOM}-");
144 		params.setPostfix(null);
145 		params.setRandomPrefixLength(6);
146 		gen.setParams(params);
147 		u = gen.generateUsername(dn);
148 		assertEquals(u.length(), 14);
149 		assertTrue(u.startsWith("pre-"));
150 		assertTrue(u.endsWith("-foo"));
151 
152 		params.setPostfix("-${RANDOM}-post");
153 		params.setPrefix(null);
154 		params.setRandomPrefixLength(5);
155 		gen.setParams(params);
156 		u = gen.generateUsername(dn);
157 		assertEquals(u.length(), 14);
158 		assertTrue(u.startsWith("foo-"));
159 		assertTrue(u.endsWith("-post"));		
160 	}
