14  package org.ejbca.util;
15  
16  import java.io.PrintWriter;
17  import java.io.StringWriter;
18  import java.util.Date;
19  import java.util.HashMap;
20  import java.util.Map;
21  import java.util.TimeZone;
22  import java.util.regex.Matcher;
23  
24  import org.apache.commons.lang.StringUtils;
25  import org.apache.commons.lang.time.FastDateFormat;
26  import org.apache.log4j.Logger;
27  import org.bouncycastle.util.encoders.Hex;
28  
29  /**
30   * This class can be extended to create highly configurable log classes.
31   * Values that are to be logged are stored in a HashMap and the output is configured using a Java.util.regex.Matcher and a sortString.
32   * The extending classes also need to supply a Logger and a String specifying how to log Dates.
33   * 
34   * Use paramPut(String key, String value) to add values,
35   * Use writeln() to log all the stored values and then use flush() to store them to file.
36   * 
37   * @author thamwickenberg
38   * @version $Id: PatternLogger.java 25870 2017-05-18 13:39:03Z samuellb $
39   */
40  public class PatternLogger implements IPatternLogger {
41  
42  	final private Map<String,String> valuepairs = new HashMap<>();
43      final private StringWriter sw = new StringWriter();
44      final private PrintWriter pw = new PrintWriter(this.sw);
45  	final private Matcher m;
46  	final private String orderString;
47  	final private Logger logger;
48      final private Date startTime;
49      private Date startProcessTime = null;
50  	
51  	/**
52       * @param m A matcher that is used together with orderstring to determine how output is formatted
53       * @param orderString A string that matches the pattern in m and specifies the order in which values are logged by the logger
54       * @param logger A log4j Logger that is used for output
55       * @param logDateFormat A string that specifies how the log-time is formatted
56  	 * @param timeZone TZ
57  	 */
58  	public PatternLogger(Matcher m, String orderString, Logger logger, String logDateFormat, String timeZone) {
59  		this.m = m;
60  		this.orderString=orderString;
61  		this.logger = logger;
62          this.startTime = new Date();
63          final FastDateFormat dateformat;
64          if (timeZone == null) {
65          	dateformat = FastDateFormat.getInstance(logDateFormat);
66          } else {
67          	dateformat = FastDateFormat.getInstance(logDateFormat, TimeZone.getTimeZone(timeZone));
68          }
69          paramPut(LOG_TIME, dateformat.format(new Date()));
70          this.paramPut(REPLY_TIME,REPLY_TIME);
71          this.paramPut(LOG_ID, "0");
72  	}
73  	
74  	/**
75  	 * 
76  	 * @return output to be logged
77  	 */
78  	private  String interpolate() {
79  		final StringBuffer sb = new StringBuffer(this.orderString.length());
80  		this.m.reset();
81  		while (this.m.find()) {
82  			// when the pattern is ${identifier}, group 0 is 'identifier'
83  			final String key = this.m.group(1);
84  			final String value = this.valuepairs.get(key);
85  
86  			// if the pattern does exists, replace it by its value
87  			// otherwise keep the pattern ( it is group(0) )
88  			if (value != null) {
89  				this.m.appendReplacement(sb, value);
90  			} else {
91  				// I'm doing this to avoid the backreference problem as there will be a $
92  				// if I replace directly with the group 0 (which is also a pattern)
93  				this.m.appendReplacement(sb, "");
94  				final String unknown = this.m.group(0);
95  				sb.append(unknown);
96  			}
97  		}
98  		this.m.appendTail(sb);
99  		return sb.toString();
100 	}
101 
102 	/**
103 	 * @see IPatternLogger#paramPut(String, byte[])
104 	 */
105 	@Override
106 	public void paramPut(String key, byte[] value){
107 		paramPut(key, new String (Hex.encode(value)));
108 	}
109 
110 	/**
111 	 * @see IPatternLogger#paramPut(String, String)
112 	 */
113 	@Override
114 	public void paramPut(String key, String value){
115 		//logger.debug("paramput: "+ key+ ";" +value +";" +valuepairs.toString());
116 		if(value == null){
117 			this.valuepairs.put(key, "");
118 		}else{
119 			this.valuepairs.put(key, value);
120 		}	  
121 		if (StringUtils.equals(key, IPatternLogger.PROCESS_TIME)) {
122 			startProcessTime = new Date();
123 		}
124 	}
125 
126 	/**
127 	 * @see IPatternLogger#paramPut(String, Integer)
128 	 */
129 	@Override
130 	public void paramPut(String key, Integer value){
131 		if(value == null){
132 			this.valuepairs.put(key, "");
133 		}else{
134 			this.valuepairs.put(key, value.toString());
135 		}
136 	}
137 
138 	/**
139 	 * @see IPatternLogger#writeln()
140 	 */
141 	@Override
142 	public void writeln() {
143 		this.pw.println(interpolate());
144 	}
145 	
146     /**
147      * @see org.ejbca.util.IPatternLogger#flush()
148      */
149 	@Override
150     public void flush() {
151         this.pw.flush();
152         String output = this.sw.toString();
153         output = output.replaceAll(IPatternLogger.REPLY_TIME, String.valueOf( new Date().getTime()-this.startTime.getTime()));
154         if (startProcessTime != null) {
155             output = output.replaceAll(IPatternLogger.PROCESS_TIME, String.valueOf(new Date().getTime()-this.startProcessTime.getTime()));
156         }
157         this.logger.debug(output); // Finally output the log row to the logging device
158     }
159 }
