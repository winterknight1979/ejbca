26  import javax.mail.internet.MimeMessage;
27  import javax.mail.internet.MimeMultipart;
28  
29  import org.apache.log4j.Logger;
30  import org.ejbca.config.MailConfiguration;
31  import org.ejbca.core.ejb.ServiceLocator;

33  /**
34   * Simple wrapper for JavaMail.
35   * 
36   * @version $Id: MailSender.java 26387 2017-08-22 14:14:36Z mikekushner $
37   */
38  public class MailSender {
39  	
40  	private static final Logger log = Logger.getLogger(MailSender.class);
41  	
42  	// Some constants to make it easier to read the client code
43  	public final static List<String> NO_TO = null;	//List<String>
44  	public final static List<String> NO_CC = null;	//List<String>
45  	public final static List<MailAttachment> NO_ATTACHMENTS = null;	//List<MailAttachment>
46  
47  	/**
48  	 * Helper method for sending mail using the mail service configured in mail.properties.
49  	 * 
50  	 * @param fromAddress The "From" address
51  	 * @param toList List&lt;String&gt; of addresses that will end up in the "To"-field or null to disable
52  	 * @param ccList List&lt;String&gt; of addresses that will end up in the "Cc"-field or null to disable
53  	 * @param subject The email subject
54  	 * @param content The text message body
55  	 * @param attachments List&lt;MailAttachment&gt; of files and objects to attach to the email or null to disable multipart messages
56  	 * @throws MailException if the message could not be successfully handed over to JavaMail
57  	 */
58  	public static void sendMailOrThrow(String fromAddress, List<String> toList, List<String> ccList, String subject, String content, List<MailAttachment> attachments) throws MailException {
59  		if (!sendMail(fromAddress, toList, ccList, subject, content, attachments)) {
60  			throw new MailException("Failed to hand over email to JavaMail.");
61  		}
62  	}
63  
64  	/**
65  	 * Helper method for sending mail using the mail service configured in mail.properties.
66  	 * 
67  	 * @param fromAddress The "From" address
68  	 * @param toList List&lt;String&gt; of addresses that will end up in the "To"-field or null to disable
69  	 * @param ccList List&lt;String&gt; of addresses that will end up in the "Cc"-field or null to disable
70  	 * @param subject The email subject
71  	 * @param content The text message body
72  	 * @param attachments List&lt;MailAttachment&gt; of files and objects to attach to the email or null to disable multipart messages
73  	 * @return true if the message was successfully handed over to JavaMail
74  	 */
75  	public static boolean sendMail(String fromAddress, List<String> toList, List<String> ccList, String subject, String content, List<MailAttachment> attachments) {
76          Session mailSession = ServiceLocator.getInstance().getMailSession(MailConfiguration.getMailJndiName());
77          // It would be good if we could set mail session properties, but it seems not possible
78          // See https://javamail.java.net/nonav/docs/api/com/sun/mail/smtp/package-summary.html
79          // mail.smtp.timeout
80          // mail.smtp.connectiontimeout
81          // mail.smtp.writetimeout
82          Message msg = new MimeMessage(mailSession);
83          try {
84          	if (log.isDebugEnabled()) {
85          		log.debug("from: " + fromAddress);
86          	}
87  			msg.setFrom(new InternetAddress(fromAddress));
88  			boolean atLeastOneRecipient = false;
89  			if (toList != null) {
90  				for (int i=0; i<toList.size(); i++) {
91  					String to = toList.get(i);
92  					msg.addRecipients(javax.mail.Message.RecipientType.TO, InternetAddress.parse(to, false));
93  		        	if (log.isDebugEnabled()) {
94  		        		log.debug("to: " + to);
95  		        	}
96  					atLeastOneRecipient = true;
97  				}
98  			}
99  			if (ccList != null) {
100 				for (int i=0; i<ccList.size(); i++) {
101 					String cc = ccList.get(i);
102 					msg.addRecipients(javax.mail.Message.RecipientType.CC, InternetAddress.parse(cc, false));
103 		        	if (log.isDebugEnabled()) {
104 		        		log.debug("cc: " + cc);
105 		        	}
106 					atLeastOneRecipient = true;
107 				}
108 			}
109 			if (!atLeastOneRecipient) {
110 				return false;	// We need at least one recipient.. either TO or CC
111 			}
112 	        msg.setSubject(subject);
113         	if (log.isDebugEnabled()) {
114         		log.debug("subject: " + subject);
115         	}
116 	        if (attachments == null || attachments.size() == 0) {
117 		        msg.setContent(content, MailConfiguration.getMailMimeType());
118 	        	if (log.isDebugEnabled()) {
119 	        		log.debug("content: " + content);
120 	        	}
121 	        } else {
122 		        Multipart multipart = new MimeMultipart();
123 		        // Add the text message first
124 		        MimeBodyPart msgBody = new MimeBodyPart();
125 		        msgBody.setContent(content, MailConfiguration.getMailMimeType());
126 		        multipart.addBodyPart(msgBody);
127 		        // Attach all the requested files
128                 for (MailAttachment mailAttachment : attachments) {
129                     MimeBodyPart msgAttachment = new MimeBodyPart();
130                     msgAttachment.setDataHandler(mailAttachment.getDataHandler());
131                     msgAttachment.setFileName(mailAttachment.getName());
132                     multipart.addBodyPart(msgAttachment);
133                 }
134 		        msg.setContent(multipart);
135 	        }
136 	        msg.setHeader("X-Mailer", "JavaMailer");
137 	        msg.setSentDate(new Date());
138 	        Transport.send(msg);
139 		} catch (MessagingException e) {
140 			log.error("Unable to send email: ", e);
141 			return false;
142 		}
143         return true;
144 	}
145 }
