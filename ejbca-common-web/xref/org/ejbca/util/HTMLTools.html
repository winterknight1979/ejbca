134     static Map<String,Integer> e2i = new HashMap<String,Integer>();
135     static Map<Integer,String> i2e = new HashMap<Integer,String>();
136     static {
137         for (int i=0; i<entities.length; ++i) {
138             e2i.put((String)entities[i][0], (Integer)entities[i][1]);
139             i2e.put((Integer)entities[i][1], (String)entities[i][0]);
140         }

143     /** 
144      * Turns funky characters into HTML entity equivalents<p>
145      * e.g. {@literal "bread" &amp; "butter"} =&lt; {@literal &amp;quot;bread&amp;quot; &amp;amp; &amp;quot;butter&amp;quot;}
146      * Update: supports nearly all HTML entities, including funky accents. See the source code for more detail.
147      * @param s1 Sq
148      * @return escape
149      * @see #htmlunescape(String)
150      **/
151     public static String htmlescape(String s1) {
152     	if (s1 == null) {
153     		return null;
154     	}
155     	final StringBuilder buf = new StringBuilder();
156         int i;
157         for (i=0; i<s1.length(); ++i) {
158             char ch = s1.charAt(i);
159             String entity = i2e.get( Integer.valueOf(ch) );
160             if (entity == null) {
161                 if ((ch) > 128) {
162                     buf.append("&#" + ((int)ch) + ";");
163                 }
164                 else {
165                     buf.append(ch);
166                 }
167             }
168             else {
169                 buf.append("&" + entity + ";");
170             }
171         }
172         return buf.toString();
173     }

175     /**
176      * Given a string containing entity escapes, returns a string
177      * containing the actual Unicode characters corresponding to the
178      * escapes.
179      *
180      * Note: nasty bug fixed by Helge Tesgaard (and, in parallel, by
181      * Alex, but Helge deserves major props for emailing me the fix).
182      * 15-Feb-2002 Another bug fixed by Sean Brown &lt;sean@boohai.com&gt;
183      * @param s1 String
184      * @return html
185      *
186      * @see #htmlescape(String)
187      **/
188     public static String htmlunescape(String s1) {
189     	if (s1 == null) {
190     		return null;
191     	}
192     	final StringBuilder buf = new StringBuilder();
193         int i;
194         for (i=0; i<s1.length(); ++i) {
195             char ch = s1.charAt(i);
196             if (ch == '&') {
197                 int semi = s1.indexOf(';', i+1);
198                 if (semi == -1) {
199                     buf.append(ch);
200                     continue;
201                 }
202                 String entity = s1.substring(i+1, semi);
203                 Integer iso;
204                 if (entity.charAt(0) == '#') {
205                     iso = Integer.valueOf(entity.substring(1));
206                 }
207                 else {
208                     iso = e2i.get(entity);
209                 }
210                 if (iso == null) {
211                     buf.append("&" + entity + ";");
212                 }
213                 else {
214                     buf.append((char)(iso.intValue()));
215                 }
216                 i = semi;
217             }
218             else {
219                 buf.append(ch);
220             }
221         }
222         return buf.toString();
223     }

225     public static String javascriptEscape(String str) {
226     	String ret = str;
227     	// In Javascript strings we need to escape the backslash and apostrophe/quote characters
228     	ret = ret.replaceAll("\\\\", "\\\\\\\\").replaceAll("'", "\\\\'").replaceAll("\"", "\\\\\"").replaceAll("<", "\\\\x3c").replaceAll(">", "\\\\x3e");
229     	return ret;
230     }
