17  /**
18   * 
19   * @version $Id: IPatternLogger.java 25870 2017-05-18 13:39:03Z samuellb $
20   *
21   */
22  public interface IPatternLogger {
23  
24  	public static final String LOG_TIME = "LOG_TIME";// The Date and time the request.
25  	public static final String LOG_ID = "LOG_ID"; //An integer identifying a log entry for a request
26  	public static final String SESSION_ID = "SESSION_ID"; //A random 32 bit number identifying a log entry for a request
27  	/** REPLY_TIME is a marker that is used to record the total time a request takes to process, including reading the request. It is replaced with 
28  	 * the correct value when the log entry is written.
29  	 * @see org.ejbca.util.PatternLogger#flush()
30  	 */
31      public static final String REPLY_TIME = "REPLY_TIME";
32  	/** PROCESS_TIME is a marker that is used to record the total time a request takes to process, excluding reading the request. It is replaced with 
33  	 * the correct value when the log entry is written. the time measurement start when this param is set in the logger with:
34  	 * <pre>
35  	 * patternLogger.paramPut(IPatternLogger.PROCESS_TIME, IPatternLogger.PROCESS_TIME);
36  	 * </pre> 
37  	 * This means that this variable can be used to measure any time you want to measure in your code.
38  	 * @see org.ejbca.util.PatternLogger#flush()
39  	 */
40      public static final String PROCESS_TIME = "PROCESS_TIME";	
41  
42  	/**
43  	 * Hex-encodes the bytes.
44  	 * method that makes sure that a "" is inserted instead of null
45  	 * @param key Key
46  	 * @param value Value
47  	 */
48  	void paramPut(String key, byte[] value);
49  
50  	/**
51  	 * method that makes sure that a "" is inserted instead of null
52  	 * @param key Key
53  	 * @param value Value
54  	 */
55  	void paramPut(String key, String value);
56  
57  	/**
58  	 * method that makes sure that a "" is inserted instead of null
59  	 * @param key Key
60  	 * @param value Value
61  	 */
62  	void paramPut(String key, Integer value);
63  	
64  	/**
65  	 * Method used for creating a log row of all added values
66  	 */
67  	void writeln();
68  	
69      /**
70       * Writes all the rows created by writeln() to the Logger
71       */
72      void flush();
73  }
