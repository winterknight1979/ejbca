39  /**
40   * Public key blacklist key validator using the Bouncy Castle BCRSAPublicKey implementation 
41   * (see org.bouncycastle.jcajce.provider.asymmetric.rsa.BCRSAPublicKey). 
42   * 
43   * The key validator is used to filter out weak debian keys or other blacklisted public keys 
44   * generated by 'openssl', 'ssh-keygen', or 'openvpn --keygen', @see "https://wiki.debian.org/SSLkeys#Identifying_Weak_Keys"
45   * 
46   * @version $Id: PublicKeyBlacklistKeyValidator.java 29467 2018-07-05 16:23:04Z mikekushner $
47   */
48  public class PublicKeyBlacklistKeyValidator extends KeyValidatorBase {

53      private static final Logger log = Logger.getLogger(PublicKeyBlacklistKeyValidator.class);

55      /** The key validator type. */
56      private static final String TYPE_IDENTIFIER = "BLACKLIST_KEY_VALIDATOR";
57  
58      protected static final String KEY_ALGORITHMS = "keyAlgorithms";
59      
60      protected static final ArrayList<String> AVAILABLE_KEY_ALGORITHMS = new ArrayList<String>();
61      
62      static {
63          AVAILABLE_KEY_ALGORITHMS.add("DSA");
64          AVAILABLE_KEY_ALGORITHMS.add("RSA");
65          AVAILABLE_KEY_ALGORITHMS.add("ECDSA");
66      }

68      /** field used for JUnit testing, avoiding lookups so we can control the cache */
69      private boolean useOnlyCache = false;

73       */

75          super();

83          super(name);

86      /**
87       * Initializes uninitialized data fields.
88       */
89      public void init() {
90          super.init();
91          if (data.get(KEY_ALGORITHMS) == null) {
92              setKeyAlgorithms(AVAILABLE_KEY_ALGORITHMS);
93          }
94      }

96      @Override
97      public void initDynamicUiModel() {
98          uiModel = new DynamicUiModel(data);
99          uiModel.add(new DynamicUiProperty<String>("settings"));
100         final LinkedHashMap<String,String> labels = new LinkedHashMap<String,String>();
101         labels.put("-1", "ALL");
102         for (final String algorithm : AlgorithmTools.getAvailableKeyAlgorithms()) {
103             labels.put(algorithm, algorithm);
104         }
105         final DynamicUiProperty<String> keyAlgorithms = new DynamicUiProperty<String>(String.class, KEY_ALGORITHMS, getKeyAlgorithmsAsString(), labels.keySet());
106         keyAlgorithms.setRenderingHint(DynamicUiProperty.RENDER_SELECT_MANY);
107         keyAlgorithms.setLabels(labels);
108         keyAlgorithms.setHasMultipleValues(true);
109         // Temp. because it is stored as String.
110         keyAlgorithms.setSaveListAsString(true);
111         keyAlgorithms.setRequired(true);
112         uiModel.add(keyAlgorithms);
113     }

115     /** Gets a list of key algorithms.
116      * @return a list.
117      */
118     public String getKeyAlgorithmsAsString() {
119         return (String) data.get(KEY_ALGORITHMS);
120     }
121     
122     /** Gets a list of key algorithms.
123      * @return a list.
124      */
125     public List<String> getKeyAlgorithms() {
126         List<String> list = Arrays.asList( StringUtils.split((String) data.get(KEY_ALGORITHMS), LIST_SEPARATOR));
127         if (list == null) {
128             list = new ArrayList<String>();
129         }
130         return list;
131     }
132 
133     /** Sets the key algorithms, RSA, ECDSA, ....
134      * 
135      * @param algorithms list of key algorithms.
136      */
137     public void setKeyAlgorithms(List<String> algorithms) {
138         data.put(KEY_ALGORITHMS, StringUtils.join(algorithms, LIST_SEPARATOR));
139     }

142     public void upgrade() {
143         super.upgrade();
144         if (log.isTraceEnabled()) {
145             log.trace(">upgrade: " + getLatestVersion() + ", " + getVersion());
146         }
147         if (Float.compare(LATEST_VERSION, getVersion()) != 0) {
148             // New version of the class, upgrade.
149             log.info(intres.getLocalizedMessage("blacklistkeyvalidator.upgrade", Float.valueOf(getVersion())));
150             init();
151         }

154     @Override
155     public List<String> validate(final PublicKey publicKey, final CertificateProfile certificateProfile) throws ValidationException {
156         List<String> messages = new ArrayList<String>();
157         final int keyLength = KeyTools.getKeyLength(publicKey);
158         final String keyAlgorithm = publicKey.getAlgorithm(); // AlgorithmTools.getKeyAlgorithm(publicKey);
159         if (log.isDebugEnabled()) {
160             log.debug("Validating public key with algorithm " + keyAlgorithm + ", length " + keyLength + ", format " + publicKey.getFormat()
161                     + ", implementation " + publicKey.getClass().getName() + " against public key blacklist.");
162         }
163         // Use the entry class to create a correct fingerprint
164         final String fingerprint = PublicKeyBlacklistEntry.createFingerprint(publicKey);
165         log.info("Matching public key with blacklist fingerprint " + fingerprint + " with public key blacklist.");
166         if (!useOnlyCache) {
167             // A bit hackish, make a call to blacklist session to ensure that blacklist cache has this entry loaded
168             // TODO: if the key is not in the cache (which it hopefully is not) this is a database lookup for each key. Huuge performance hit
169             // should better be implemented as a full in memory cache with a state so we know if it's loaded or not, with background updates. See ECA-5951
170             // We could simply add a call to load the whole map into cache, when the cache needs updating
171             // See BlacklistSessionBean.getBlacklistEntryIdToFingerprintMap (and add cache there)
172             new EjbLocalHelper().getBlacklistSession().getBlacklistEntryId(PublicKeyBlacklistEntry.TYPE, fingerprint);
173         }
174         Integer idValue = PublicKeyBlacklistEntryCache.INSTANCE.getNameToIdMap().get(fingerprint);
175         final PublicKeyBlacklistEntry entry = PublicKeyBlacklistEntryCache.INSTANCE.getEntry(idValue);
176         boolean keyAlgMatched = false;
177 
178         if (null != entry) {
179             // Filter for key specifications.
180             final Set<String> keyAlgs = new HashSet<>(getKeyAlgorithms());
181             if (keyAlgs.contains(AlgorithmConstants.KEYALGORITHM_EC) || keyAlgs.contains(AlgorithmConstants.KEYALGORITHM_ECDSA)) {
182                 keyAlgs.add(AlgorithmConstants.KEYALGORITHM_EC);
183                 keyAlgs.add(AlgorithmConstants.KEYALGORITHM_ECDSA);
184             }
185             if (getKeyAlgorithms().contains("-1") || keyAlgs.contains(getKeyAlg(publicKey, fingerprint))) {
186                 keyAlgMatched = true;
187             }
188         }
189         if (keyAlgMatched) {
190             final String message = "Public key with id " + entry.getID() + " and fingerprint " + fingerprint
191                     + " found in public key blacklist.";
192             messages.add("Invalid: " + message);
193         } else {
194             log.trace("publicKeyBlacklist passed");
195         }
196 
197         if (log.isDebugEnabled()) {
198             for (String message : messages) {
199                 log.debug(message);
200             }
201         }
202         return messages;
203     }
204 
205     private final String getKeyAlg(PublicKey publicKey, String fingerprint) {
206         String keyAlg;
207         if (publicKey instanceof BCRSAPublicKey) {
208             keyAlg = publicKey.getAlgorithm().toUpperCase();
209         } else if (publicKey instanceof BCECPublicKey) {
210             keyAlg = publicKey.getAlgorithm().toUpperCase();
211         } else {
212             keyAlg = publicKey.getAlgorithm().toUpperCase();
213         }
214         if (log.isTraceEnabled()) {
215             log.trace("Key algorithm " + keyAlg + " determined for public key with fp: " + fingerprint);
216         }
217         return keyAlg;
218     }
