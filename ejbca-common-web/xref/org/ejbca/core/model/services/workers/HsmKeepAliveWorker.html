18  import java.util.Map;

30  /**
31   * Worker that keeps HSM sessions active.
32   * 
33   * @version $Id: HsmKeepAliveWorker.java 19901 2014-09-30 14:29:38Z anatom $
34   *
35   */
36  public class HsmKeepAliveWorker extends BaseWorker {
37  
38      private static final Logger log = Logger.getLogger(HsmKeepAliveWorker.class);
39  
40      @Override
41      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
42          // Health checking will be done in three steps: 
43          if (log.isDebugEnabled()) {
44              log.debug("Performing HSM Keepalive operation.");
45          }
46          // 1. If we use Audit log integrity protection, make a test integrity protection calculation
47          InternalSecurityEventsLoggerSessionLocal logSession = (InternalSecurityEventsLoggerSessionLocal)ejbs.get(InternalSecurityEventsLoggerSessionLocal.class);
48          logSession.auditLogCryptoTest(this.getClass().getSimpleName());
49  
50          // 2. Call testKeyPair on all active crypto tokens that has an alias named testKey
51          CryptoTokenManagementSessionLocal tokenSession = (CryptoTokenManagementSessionLocal)ejbs.get(CryptoTokenManagementSessionLocal.class);
52          List<CryptoTokenInfo> infos = tokenSession.getCryptoTokenInfos(admin);
53          if (log.isDebugEnabled()) {
54              log.debug("Performing keepalive on all active crypto tokens ("+infos.size()+"), but skipping soft tokens.");
55          }
56          for (CryptoTokenInfo info : infos) {
57              // Only test active, PKCS11, crypto tokens. o need to do keepalive on soft tokens, they will not time out
58              if (info.isActive() && StringUtils.isNotEmpty(info.getP11Library())) {
59                  CryptoToken token = tokenSession.getCryptoToken(info.getCryptoTokenId());
60                  try {
61                      List<String> aliases = token.getAliases();
62                      boolean tested = false;
63                      for(final String alias : aliases) {
64                          if ("testKey".equals(alias)) {
65                              if (log.isDebugEnabled()) {
66                                  log.debug("Keepalive testing crypto token '"+info.getName()+"' with id "+info.getCryptoTokenId());
67                              }
68                              token.testKeyPair("testKey");
69                              tested = true;
70                          }
71                      }
72                      if (!tested) {
73                          if (log.isDebugEnabled()) {
74                              log.debug("No testKey on crypto token '"+info.getName()+"' with id "+info.getCryptoTokenId());
75                          }                        
76                      }
77                  } catch (InvalidKeyException e) {
78                      log.info("Error testing crypto token that suppposedly was active: ", e);
79                  } catch (CryptoTokenOfflineException e) {
80                      log.info("Error testing crypto token that suppposedly was active: ", e);
81                  } catch (KeyStoreException e) {
82                      log.info("Error testing crypto token that suppposedly was active: ", e);
83                  }
84              } else {
85                  if (log.isDebugEnabled()) {
86                      log.debug("Not testing inactive, or soft, crypto token '"+info.getName()+"' with id "+info.getCryptoTokenId());
87                  }
88              }
89          }
90          
91      }
92  }
