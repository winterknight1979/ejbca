36  /**
37   * Makes queries about which certificates that is about to expire in a given number of days and creates a 
38   * notification sent to either the end user or the administrator.
39   * 
40   * version: $Id: CertificateExpirationNotifierWorker.java 29010 2018-05-23 13:09:53Z jekaterina_b_helmes $
41   */
42  public class CertificateExpirationNotifierWorker extends EmailSendingWorker {
43  
44      private static final Logger log = Logger.getLogger(CertificateExpirationNotifierWorker.class);
45  
46      private CertificateStoreSessionLocal certificateStoreSession;
47      private transient List<Integer> certificateProfileIds;
48  
49      /**
50       * Worker that makes a query to the Certificate Store about expiring certificates.
51       * 
52       * @see org.ejbca.core.model.services.IWorker#work
53       */
54      @Override
55      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
56          log.trace(">CertificateExpirationNotifierWorker.work started");
57          final CaSessionLocal caSession = ((CaSessionLocal) ejbs.get(CaSessionLocal.class));
58          certificateStoreSession = ((CertificateStoreSessionLocal) ejbs.get(CertificateStoreSessionLocal.class));
59          final EndEntityAccessSessionLocal endEntityAccessSession = ((EndEntityAccessSessionLocal) ejbs.get(EndEntityAccessSessionLocal.class));
60  
61          ArrayList<EmailCertData> userEmailQueue = new ArrayList<EmailCertData>();
62          ArrayList<EmailCertData> adminEmailQueue = new ArrayList<EmailCertData>();
63  
64          // Build Query
65          Collection<String> cas = new ArrayList<String>();
66          Collection<Integer> caIds = getCAIdsToCheck(false);
67          Collection<Integer> certificateProfileIds = getCertificateProfileIdsToCheck();
68          if (!caIds.isEmpty()) {
69              //if caIds contains SecConst.ALLCAS, reassign caIds to contain just that.
70              if(caIds.contains(SecConst.ALLCAS)) {
71                  caIds = caSession.getAllCaIds();
72              }
73              for(Integer caid : caIds) {
74                  CAInfo caInfo;  
75                  try {
76                      caInfo = caSession.getCAInfo(getAdmin(), caid);
77                      if(caInfo == null) {
78                          log.info(InternalEjbcaResources.getInstance().getLocalizedMessage("services.errorworker.errornoca", caid, null));
79                          continue;
80                      }
81                  }  catch (AuthorizationDeniedException e) {
82                      log.info(InternalEjbcaResources.getInstance().getLocalizedMessage("authorization.notauthorizedtoresource", caid, "CAId"));
83                      continue;
84                  }
85                  String cadn = caInfo.getSubjectDN();
86                  cas.add(cadn);
87              }
88  
89              /*
90               * Algorithm:
91               * 
92               * Inputs: CertificateData.status Which either is ACTIVE or
93               * NOTIFIEDABOUTEXPIRATION in order to be candidates for
94               * notifications.
95               * 
96               * nextRunTimestamp Tells when the next service run will be
97               * 
98               * currRunTimestamp Tells when the service should run (usually "now"
99               * but there may be delayed runs as well if the app-server has been
100              * down)
101              * 
102              * thresHold The configured "threshold"
103              * 
104              * We want to accomplish two things:
105              * 
106              * 1. Notify for expirations within the service window 2. Notify
107              * _once_ for expirations that occurred before the service window
108              * like flagging certificates that have a shorter life-span than the
109              * threshold (pathologic test-case...)
110              * 
111              * The first is checked by:
112              * 
113              * notify = currRunTimestamp + thresHold <= ExpireDate <
114              * nextRunTimestamp + thresHold AND (status = ACTIVE OR status =
115              * NOTIFIEDABOUTEXPIRATION)
116              * 
117              * The second can be checked by:
118              * 
119              * notify = currRunTimestamp + thresHold > ExpireDate AND status =
120              * ACTIVE
121              * 
122              * In both case status can be set to NOTIFIEDABOUTEXPIRATION
123              * 
124              * As Tomas pointed out we do not need to flag certificates that
125              * have expired already which is a separate test.
126              */
127 
128             long now = new Date().getTime();
129             if (!cas.isEmpty()) {
130                 long thresHold = getTimeBeforeExpire();
131                 if (log.isDebugEnabled()) {
132                     log.debug("Looking for expiring certificates for CAs '"+caIds+"' and certificate profiles '"+certificateProfileIds+"', with expire treshold: "+thresHold+
133                             ". activeNotifiedExpireDateMin: "+now+", activeNotifiedExpireDateMax: "+(nextRunTimeStamp + thresHold)+", activeExpireDateMin: "+(runTimeStamp + thresHold));
134                 }
135                 try {
136                     List<Object[]> fingerprintUsernameList = certificateStoreSession.findExpirationInfo(cas, certificateProfileIds, now, (nextRunTimeStamp + thresHold),
137                             (runTimeStamp + thresHold));
138                     int count = 0;
139                     for (Object[] next : fingerprintUsernameList) {
140                         count++;
141                         // For each certificate update status.
142                         String fingerprint = (String) next[0];
143                         String username = (String) next[1];
144                         // Get the certificate through a session bean
145                         log.debug("Found a certificate we should notify. Username=" + username + ", fp=" + fingerprint);
146                         Certificate cert = certificateStoreSession.findCertificateByFingerprint(fingerprint);
147                         EndEntityInformation userData = endEntityAccessSession.findUser(getAdmin(), username);
148                         if (userData != null) {
149                             if (isSendToEndUsers()) {
150                                 if (userData.getEmail() == null || userData.getEmail().trim().equals("")) {
151                                     log.info(InternalEjbcaResources.getInstance().getLocalizedMessage("services.errorworker.errornoemail", username));
152                                 } else {
153                                     // Populate end user message
154                                     log.debug("Adding to email queue for user: " + userData.getEmail());
155                                     final UserNotificationParamGenNotificationParamGen">UserNotificationParamGen userNotificationParamGen = new UserNotificationParamGen(userData, cert);
156                                     final String message = userNotificationParamGen.interpolate(getEndUserMessage());
157                                     final String subject = userNotificationParamGen.interpolate(getEndUserSubject());
158                                     final MailActionInfoMailActionInfo.html#MailActionInfo">MailActionInfo mailActionInfo = new MailActionInfo(userData.getEmail(), subject, message);
159                                     userEmailQueue.add(new EmailCertData(fingerprint, mailActionInfo));
160                                 }
161                             }
162                         } else {
163                             log.debug("Trying to send notification to user, but no UserData can be found for user '" + username
164                                     + "', will only send to admin if admin notifications are defined.");
165                         }
166                         if (isSendToAdmins()) {
167                             // If we did not have any user for this, we will simply use empty values for substitution
168                             if (userData == null) {
169                                 userData = new EndEntityInformation();
170                                 userData.setUsername(username);
171                             }
172                             // Populate admin message
173                             log.debug("Adding to email queue for admin");
174                             final UserNotificationParamGenNotificationParamGen">UserNotificationParamGen userNotificationParamGen = new UserNotificationParamGen(userData, cert);
175                             final String message = userNotificationParamGen.interpolate(getAdminMessage());
176                             final String subject = userNotificationParamGen.interpolate(getAdminSubject());
177                             final MailActionInfoMailActionInfo.html#MailActionInfo">MailActionInfo mailActionInfo = new MailActionInfo(null, subject, message);
178                             adminEmailQueue.add(new EmailCertData(fingerprint, mailActionInfo));
179                         }
180                         if (!isSendToEndUsers() && !isSendToAdmins()) {
181                             // a little bit of a kludge to make JUnit testing feasible...
182                             log.debug("nobody to notify for cert with fp:" + fingerprint);
183                             updateStatus(fingerprint, CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION);
184                         }
185                     }
186                     if (count == 0) {
187                         log.debug("No certificates found for notification.");
188                     }
189 
190                 } catch (Exception fe) {
191                     log.error("Error running service work: ", fe);
192                     throw new ServiceExecutionFailedException(fe);
193                 }
194                 if (isSendToEndUsers()) {
195                     sendEmails(userEmailQueue, ejbs);
196                 }
197                 if (isSendToAdmins()) {
198                     sendEmails(adminEmailQueue, ejbs);
199                 }
200             } else {
201                 log.info("CAs select collection is empty, there were ids but no names?");
202             }
203         } else {
204             log.debug("No CAs to check");
205         }
206         log.trace("<CertificateExpirationNotifierWorker.work ended");
207     }

209     /**
210      * Method that must be implemented by all subclasses to EmailSendingWorker, used to update status of a certificate, user, or similar
211      * 
212      * @param pk primary key of object to update
213      * @param status status to update to
214      */
215     @Override
216     protected void updateStatus(String pk, int status) {
217         try {
218             if (!certificateStoreSession.setStatus(getAdmin(), pk, status)) {
219                 log.error("Error updating certificate status for certificate with fingerprint: " + pk);
220             }
221         } catch (AuthorizationDeniedException e) {
222             // Should not be possible...
223             log.error("Internal admin not authorized: ", e);
224             throw new RuntimeException(e);
225         }
226     }
227     
228     /**
229      * Returns the Set of Certificate Profile IDs. For performance reasons cached as a 
230      * transient class variable.
231      * 
232      * @return collection
233      */
234     private Collection<Integer>getCertificateProfileIdsToCheck() {
235         if (this.certificateProfileIds == null) {
236             this.certificateProfileIds = new ArrayList<Integer>();
237             String idString = properties.getProperty(PROP_CERTIFICATE_PROFILE_IDS_TO_CHECK);
238             if (idString != null) {
239                 for (String id : idString.split(";")) {
240                     certificateProfileIds.add(Integer.valueOf(id));
241                 }
242             }
243         }
244         return certificateProfileIds;
245     }
