33  /**
34   * Replaces CA certificate chains with a pending new rollover certificate chain once the new certificate chain becomes valid.
35   *
36   * version: $Id: RolloverWorker.java 27740 2018-01-05 07:24:53Z mikekushner $
37   */
38  public class RolloverWorker extends BaseWorker {
39  
40  	private static final Logger log = Logger.getLogger(RolloverWorker.class);
41      /** Internal localization of logs and errors */
42  
43      @Override
44      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
45          log.trace(">Worker started");
46          final CaSessionLocal caSession = ((CaSessionLocal) ejbs.get(CaSessionLocal.class));
47          // Find CAs that have a roll over certificate chain that has became valid.
48          final Date now = new Date(new Date().getTime() - 10 * 60 * 1000); // delay by 10 minutes to account for clock scew on the clients. TODO make configurable?
49  
50          // Roll over these CAs using the CAAdminSessionBean
51          Collection<Integer> caids = getCAIdsToCheck(false);
52          if (log.isDebugEnabled()) {
53              log.debug("Checking " + caids.size() + " CAs for rollover");
54          }
55          if (caids.contains(SecConst.ALLCAS)) {
56              for (CAInfo caInfo : caSession.getAuthorizedAndNonExternalCaInfos(getAdmin())) {
57                  attemptToPerformRollover(ejbs, caInfo.getCAId(), now);
58              }
59          } else {
60              for (int caid : caids) {
61                  attemptToPerformRollover(ejbs, caid, now);
62              }
63          }
64          log.trace("<Worker ended");
65      }
66  	
67  	private void attemptToPerformRollover(Map<Class<?>, Object> ejbs, int caid, Date now) {
68          final CaSessionLocal caSession = ((CaSessionLocal)ejbs.get(CaSessionLocal.class));
69          final CAAdminSessionLocal caAdminSession = ((CAAdminSessionLocal)ejbs.get(CAAdminSessionLocal.class));
70  	    try {
71              final CA ca = caSession.getCA(getAdmin(), caid);
72              final List<Certificate> rolloverChain = ca.getRolloverCertificateChain();
73              if (rolloverChain != null) {
74                  final Certificate cert = rolloverChain.get(0);
75                  if (now.after(CertTools.getNotBefore(cert))) {
76                      // Replace certificate chain with the roll over chain
77                      if (log.isDebugEnabled()) {
78                          log.debug("New certificate of CA "+caid+" is now valid, switching certificate.");
79                      }
80                      caAdminSession.rolloverCA(getAdmin(), ca.getCAId());
81                  }
82              }
83          }  catch (AuthorizationDeniedException e) {
84              log.error("Error checking CA for rollover: ", e);
85          } catch (CryptoTokenOfflineException e) {
86              log.error("Error rolling over CA: ", e);
87          }
88  	}
89  }
