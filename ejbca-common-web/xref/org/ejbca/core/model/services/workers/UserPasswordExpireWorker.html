29  /**
30   * Worker expiring users password after a configured amount of time.
31   * 
32   * Makes queries for users with status new, which was not modified in a certain amount of time, and sets
33   * the status of these users to generated. Can be used for the following scenario:
34   * - A user is generated and given username password to fetch the certificates
35   * - If the user does not fetch his certificate within a configured amount of time the user is expired and is not allowed to fetch the certificate any more
36   * 
37   * @author Tomas Gustavsson based on code by Philip Vendil
38   *
39   * version: $Id: UserPasswordExpireWorker.java 24487 2016-10-10 09:10:53Z anatom $
40   */
41  public class UserPasswordExpireWorker extends EmailSendingWorker {
42  
43      private static final Logger log = Logger.getLogger(UserPasswordExpireWorker.class);
44  
45      /**
46       * Worker that makes a query to the Certificate Store about expiring
47       * certificates.
48       * 
49       * @see org.ejbca.core.model.services.IWorker#work
50       */
51      @Override
52      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
53          log.trace(">Worker started");
54          final EndEntityManagementSessionLocal endEntityManagementSession = ((EndEntityManagementSessionLocal)ejbs.get(EndEntityManagementSessionLocal.class));
55  
56          ArrayList<EmailCertData> userEmailQueue = new ArrayList<EmailCertData>();
57          ArrayList<EmailCertData> adminEmailQueue = new ArrayList<EmailCertData>();
58         
59          long timeModified = ((new Date()).getTime() - getTimeBeforeExpire());   
60          List<EndEntityInformation> userDataList = endEntityManagementSession.findUsers(new ArrayList<Integer>(getCAIdsToCheck(false)),
61                  timeModified, EndEntityConstants.STATUS_NEW);
62  
63          for (EndEntityInformation endEntityInformation : userDataList) {
64              endEntityInformation.setStatus(EndEntityConstants.STATUS_GENERATED);
65              endEntityInformation.setPassword(null);
66              try {
67              	endEntityManagementSession.changeUser(getAdmin(), endEntityInformation, false);
68                  if (isSendToEndUsers()) {
69                  	if (endEntityInformation.getEmail() == null || endEntityInformation.getEmail().trim().equals("")) {
70                  		log.info(InternalEjbcaResources.getInstance().getLocalizedMessage("services.errorworker.errornoemail", endEntityInformation.getUsername()));
71                  	} else {
72                  		// Populate end user message
73                  		String message = new UserNotificationParamGen(endEntityInformation).interpolate(getEndUserMessage());
74                  		MailActionInfo mailActionInfo = new MailActionInfo(endEntityInformation.getEmail(), getEndUserSubject(), message);
75                  		userEmailQueue.add(new EmailCertData(endEntityInformation.getUsername(), mailActionInfo));
76                  	}
77                  }
78                  if (isSendToAdmins()) {
79                  	// Populate admin message
80                  	String message = new UserNotificationParamGen(endEntityInformation).interpolate(getAdminMessage());
81                  	MailActionInfo mailActionInfo = new MailActionInfo(null, getAdminSubject(), message);
82                  	adminEmailQueue.add(new EmailCertData(endEntityInformation.getUsername(), mailActionInfo));
83                  }
84              } catch (Exception e) {
85                  log.error("Error running service work: ", e);
86                  throw new ServiceExecutionFailedException(e);
87              }
88          }
89          // Send of the mails
90          if (isSendToEndUsers()) {
91              sendEmails(userEmailQueue, ejbs);
92          }
93          if (isSendToAdmins()) {
94              sendEmails(adminEmailQueue, ejbs);
95          }
96          log.trace("<Worker ended");
97      }
