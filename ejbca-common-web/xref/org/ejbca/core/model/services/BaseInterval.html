15  import java.util.Properties;

17  /**
18   *
19   * Help base class that manages that implements the init method of the interface
20   * and manages the properties.
21   * 
22   * @author Philip Vendil 2006 sep 27
23   *
24   * @version $Id: BaseInterval.java 22139 2015-11-03 10:41:56Z mikekushner $
25   */
26  public abstract class BaseInterval implements IInterval {
27  
28  	protected Properties properties = null;
29  	protected String serviceName = null;
30  	/**
31  	 * @see org.ejbca.core.model.services.IAction#init(Properties, String)
32  	 */
33  	public void init(Properties properties, String serviceName) {
34         this.properties = properties;	
35         this.serviceName = serviceName;
36  	}
37  	
38  }
