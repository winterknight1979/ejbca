13  package org.ejbca.core.model.approval.approvalrequests;
14  
15  import java.io.IOException;
16  import java.io.ObjectInput;
17  import java.io.ObjectOutput;
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.util.CertTools;
23  import org.ejbca.core.model.approval.ApprovalDataText;
24  import org.ejbca.core.model.approval.ApprovalDataVO;
25  import org.ejbca.core.model.approval.ApprovalRequest;
26  import org.ejbca.core.model.approval.ApprovalRequestExecutionException;
27  import org.ejbca.core.model.approval.profile.ApprovalProfile;
28  
29  /**
30   * Special Approval Request created when an administrator wants
31   * to generate a token through the Web Service interface.
32   *  
33   * It is a two step approval request were the first step is
34   * a view hard token PUK data and the second is the actual
35   * hard token generation
36   * 
37   * @version $Id: GenerateTokenApprovalRequest.java 23981 2016-07-22 12:21:55Z anatom $
38   */
39  public class GenerateTokenApprovalRequest extends ApprovalRequest {
40  
41  	private static final long serialVersionUID = -1L;
42  	
43  	public static final int STEP_0_VIEWHARDTOKENDATA = 0;
44  	public static final int STEP_1_GENERATETOKEN     = 1;
45  	
46  	private static final int LATEST_VERSION = 1;
47  		
48  	private String dn;
49  	private String username;
50  	private String tokenTypeLabel;
51  
52  	/** Constructor used in externalization only */
53  	public GenerateTokenApprovalRequest() {}
54  
55  	public GenerateTokenApprovalRequest(String username, String userDN, String tokenTypeLabel, AuthenticationToken requestAdmin, 
56  	        String requestSignature, int cAId, int endEntityProfileId, ApprovalProfile approvalProfile) {
57          // This is a 2 step approval, whatever that means...
58          super(requestAdmin, requestSignature, REQUESTTYPE_SIMPLE, cAId, endEntityProfileId, 2, approvalProfile);
59  		this.username = username;
60  		this.dn = userDN;
61  		this.tokenTypeLabel = tokenTypeLabel;
62  	}
63  	
64  	@Override
65  	public void execute() throws ApprovalRequestExecutionException {
66  		// This is a non-executable approval
67  	}
68  
69      /**
70       * Approval Id is generated of This approval type (i.e AddEndEntityApprovalRequest) and UserName
71       */
72  	public int generateApprovalId() {
73  		return new String(getApprovalType() + ";" + username + ";" + CertTools.getFingerprintAsString(getRequestAdminCert()) + 
74  		        ";" + getApprovalProfile().getProfileName()).hashCode();
75  	}
76  
77  	public int getApprovalType() {		
78  		return ApprovalDataVO.APPROVALTYPE_GENERATETOKEN;
79  	}
80  
81  	@Override
82  	public List<ApprovalDataText> getNewRequestDataAsText(AuthenticationToken admin) {
83  		ArrayList<ApprovalDataText> retval = new ArrayList<ApprovalDataText>();
84  		retval.add(new ApprovalDataText("USERNAME",username,true,false));		
85  		retval.add(new ApprovalDataText("SUBJECTDN",dn,true,false));
86  		retval.add(new ApprovalDataText("LABEL",tokenTypeLabel,true,true));
87  		return retval;
88  	}
89  	
90  	@Override
91  	public List<ApprovalDataText> getOldRequestDataAsText(AuthenticationToken admin) {
92  		return null;
93  	}
94  
95  	public boolean isExecutable() {		
96  		return false;
97  	}
98  	
99  	public void writeExternal(ObjectOutput out) throws IOException {
100 		super.writeExternal(out);
101 		out.writeInt(LATEST_VERSION);
102 		out.writeObject(username);
103 		out.writeObject(dn);
104 		out.writeObject(tokenTypeLabel);
105 	}
106 
107 	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {        
108 		super.readExternal(in);
109         int version = in.readInt();
110         if(version == 1){
111     		username = (String) in.readObject();
112             dn = (String) in.readObject();
113             tokenTypeLabel = (String) in.readObject();            
114         }
115 	}
116 
117 	/**
118 	 * @return the subject dn used in the request
119 	 */
120 	public String getDN() {
121 		return dn;
122 	}
123 }
