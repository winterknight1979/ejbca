133     public static PKIHeaderBuilder createPKIHeaderBuilder(GeneralName sender, GeneralName recipient, String senderNonce, String recipientNonce,
134             String transactionId) {
135         PKIHeaderBuilder pkiHeader = new PKIHeaderBuilder(PKIHeader.CMP_2000, sender, recipient);
136         pkiHeader.setMessageTime(new DERGeneralizedTime(new Date()));
137         if (senderNonce != null) {
138             pkiHeader.setSenderNonce(new DEROctetString(Base64.decode(senderNonce.getBytes())));
139         }
140         if (recipientNonce != null) {
141             pkiHeader.setRecipNonce(new DEROctetString(Base64.decode(recipientNonce.getBytes())));
142         }
143         if (transactionId != null) {
144             pkiHeader.setTransactionID(new DEROctetString(Base64.decode(transactionId.getBytes())));
145         }
146         return pkiHeader;
147     }

149     public static byte[] signPKIMessage(PKIMessage pkiMessage, Collection<Certificate> signCertChain, PrivateKey signKey, String digestAlg, 
150     		String provider) throws InvalidKeyException, NoSuchProviderException, NoSuchAlgorithmException, SecurityException, SignatureException,
151             CertificateEncodingException {
152         if (LOG.isTraceEnabled()) {
153             LOG.trace(">signPKIMessage()");
154         }
155         final List<CMPCertificate> extraCertsList = new ArrayList<>();
156         for (final Certificate certificate : signCertChain) {
157             extraCertsList.add(CMPCertificate.getInstance(((X509Certificate)certificate).getEncoded()));
158         }
159         final CMPCertificate[] extraCerts = extraCertsList.toArray(new CMPCertificate[signCertChain.size()]);
160         final PKIMessage signedPkiMessage = buildCertBasedPKIProtection(pkiMessage, extraCerts, signKey, digestAlg, provider);
161         if (LOG.isTraceEnabled()) {
162             LOG.trace("<signPKIMessage()");
163         }
164         // Return response as byte array 
165         return pkiMessageToByteArray(signedPkiMessage);
166     }

168     public static PKIMessage buildCertBasedPKIProtection(PKIMessage pkiMessage, CMPCertificate[] extraCerts, PrivateKey key, String digestAlg,
169             String provider) throws NoSuchProviderException, NoSuchAlgorithmException, SecurityException, SignatureException, InvalidKeyException {
170         // Select which signature algorithm we should use for the response, based on the digest algorithm and key type.
171         ASN1ObjectIdentifier oid = AlgorithmTools.getSignAlgOidFromDigestAndKey(digestAlg, key.getAlgorithm());
172         if (LOG.isDebugEnabled()) {
173             LOG.debug("Selected signature alg oid: " + oid.getId()+", key algorithm: "+key.getAlgorithm());
174         }
175         // According to PKCS#1 AlgorithmIdentifier for RSA-PKCS#1 has null Parameters, this means a DER Null (asn.1 encoding of null), not Java null.
176         // For the RSA signature algorithms specified above RFC3447 states "...the parameters MUST be present and MUST be NULL."
177         PKIHeaderBuilder headerBuilder = getHeaderBuilder(pkiMessage.getHeader());
178         AlgorithmIdentifier pAlg = null;
179         if ("RSA".equalsIgnoreCase(key.getAlgorithm())) {
180             pAlg = new AlgorithmIdentifier(oid, DERNull.INSTANCE);
181         } else {
182             pAlg = new AlgorithmIdentifier(oid);
183         }
184         headerBuilder.setProtectionAlg(pAlg);
185         // Most PKCS#11 providers don't like to be fed an OID as signature algorithm, so 
186         // we use BC classes to translate it into a signature algorithm name instead
187         PKIHeader head = headerBuilder.build();
188         String signatureAlgorithmName = AlgorithmTools.getAlgorithmNameFromOID(oid);
189         if (LOG.isDebugEnabled()) {
190             LOG.debug("Signing CMP message with signature alg: " + signatureAlgorithmName);
191         }
192         Signature sig = Signature.getInstance(signatureAlgorithmName, provider);
193         sig.initSign(key);
194         sig.update(getProtectedBytes(head, pkiMessage.getBody()));
195         final PKIMessage protectedPkiMessage;
196         if (extraCerts != null && extraCerts.length > 0) {
197             protectedPkiMessage = new PKIMessage(head, pkiMessage.getBody(), new DERBitString(sig.sign()), extraCerts);
198         } else {
199             protectedPkiMessage = new PKIMessage(head, pkiMessage.getBody(), new DERBitString(sig.sign()));
200         }
201         return protectedPkiMessage;
202     }

206         PKIHeaderBuilder builder = new PKIHeaderBuilder(head.getPvno().getValue().intValue(), head.getSender(), head.getRecipient());

208         builder.setGeneralInfo(head.getGeneralInfo());

218     /** verifies signature protection on CMP PKI messages
219      *  
220      * @param pKIMessage the CMP message to verify signature on, if protected by signature protection
221      * @param pubKey the public key used to verify the signature
222      * @return true if verification is ok or false if verification fails
223      * @throws NoSuchAlgorithmException message is signed by an unknown algorithm
224      * @throws InvalidKeyException pubKey is not valid for signature verification
225      * @throws SignatureException if the passed-in signature is improperly encoded or of the wrong type, if this signature algorithm is unable to process the input data provided, etc.
226      */
227     public static boolean verifyCertBasedPKIProtection(PKIMessage pKIMessage, PublicKey pubKey) throws NoSuchAlgorithmException,
228              InvalidKeyException, SignatureException {
229         if(pKIMessage.getProtection() == null) {
230             throw new SignatureException("Message was not signed.");
231         }
232         AlgorithmIdentifier sigAlg = pKIMessage.getHeader().getProtectionAlg();
233         if(sigAlg == null) {
234             throw new SignatureException("No signature algorithm was provided.");
235         }
236       
237         if (LOG.isDebugEnabled()) {
238             LOG.debug("Verifying signature with algorithm: " + sigAlg.getAlgorithm().getId());
239         }
240         Signature sig;
241         try {
242             sig = Signature.getInstance(sigAlg.getAlgorithm().getId(), BouncyCastleProvider.PROVIDER_NAME);
243         } catch (NoSuchProviderException e) {
244             throw new IllegalStateException("BouncyCastle provider not installed.", e);
245         }
246         sig.initVerify(pubKey);
247         sig.update(getProtectedBytes(pKIMessage));
248         boolean result = sig.verify(pKIMessage.getProtection().getBytes());
249         if (LOG.isDebugEnabled()) {
250             LOG.debug("Verification result: " + result);
251         }
252         return result;
253     }

255     public static byte[] protectPKIMessageWithPBE(PKIMessage msg, String keyId, String raSecret, String digestAlgId, String macAlgId,
256             int iterationCount) throws NoSuchAlgorithmException, NoSuchProviderException, InvalidKeyException {
257         if (LOG.isTraceEnabled()) {
258             LOG.trace(">protectPKIMessageWithPBE()");
259         }
260         // Create the PasswordBased protection of the message
261         PKIHeaderBuilder head = getHeaderBuilder(msg.getHeader());
262         if (keyId != null) {
263             // Only add senderKeyId of the client used it, it's not needed to create the actual protection
264             byte[] keyIdBytes = keyId.getBytes(StandardCharsets.UTF_8);
265             head.setSenderKID(new DEROctetString(keyIdBytes));
266         }
267         // SHA1
268         AlgorithmIdentifier owfAlg = new AlgorithmIdentifier(new ASN1ObjectIdentifier(digestAlgId));
269         // iterations, usually something like 1024
270         ASN1Integer iteration = new ASN1Integer(iterationCount);
271         // HMAC/SHA1
272         AlgorithmIdentifier macAlg = new AlgorithmIdentifier(new ASN1ObjectIdentifier(macAlgId));
273         // We need some random bytes for the nonce
274         byte[] saltbytes = createSenderNonce();
275         DEROctetString derSalt = new DEROctetString(saltbytes);
276 
277         // Create the new protected return message
278         //String objectId = "1.2.840.113533.7.66.13" = passwordBasedMac;
279         String objectId = CMPObjectIdentifiers.passwordBasedMac.getId();
280         PBMParameter pp = new PBMParameter(derSalt, owfAlg, iteration, macAlg);
281         AlgorithmIdentifier pAlg = new AlgorithmIdentifier(new ASN1ObjectIdentifier(objectId), pp);
282         head.setProtectionAlg(pAlg);
283 
284         // Calculate the protection bits
285         byte[] rasecret = raSecret.getBytes();
286         byte[] basekey = new byte[rasecret.length + saltbytes.length];
287         System.arraycopy(rasecret, 0, basekey, 0, rasecret.length);
288         System.arraycopy(saltbytes, 0, basekey, rasecret.length, saltbytes.length);
289         // Construct the base key according to rfc4210, section 5.1.3.1
290         MessageDigest dig = MessageDigest.getInstance(owfAlg.getAlgorithm().getId(), "BC");
291         for (int i = 0; i < iterationCount; i++) {
292             basekey = dig.digest(basekey);
293             dig.reset();
294         }
295 
296         PKIHeader pkiHeader = head.build();
297         // Do the mac
298         String macOid = macAlg.getAlgorithm().getId();
299         byte[] protectedBytes = getProtectedBytes(pkiHeader, msg.getBody());
300         Mac mac = Mac.getInstance(macOid, BouncyCastleProvider.PROVIDER_NAME);
301         SecretKey key = new SecretKeySpec(basekey, macOid);
302         mac.init(key);
303         mac.reset();
304         mac.update(protectedBytes, 0, protectedBytes.length);
305         byte[] out = mac.doFinal();
306         DERBitString bs = new DERBitString(out);
307 
308         if (LOG.isTraceEnabled()) {
309             LOG.trace("<protectPKIMessageWithPBE()");
310         }
311         // Return response as byte array 
312         return pkiMessageToByteArray(new PKIMessage(pkiHeader, msg.getBody(), bs, msg.getExtraCerts()));
313     }

346      * Create a standard unprotected error message with PKIStatus.rejection and PKIFailureInfo.badRequest.
347      * The generated message should be sent as a response to an unparsable CMP request only, since the sender nonce
348      * is not copied from the client's CMP request and a new transaction ID is generated.
349      * @param errorDescription error
350      * @return The byte representation of the error message
351      */
352     public static byte[] createUnprotectedErrorMessage(final String errorDescription) {
353         final PKIHeader pkiHeader = new PKIHeaderBuilder(PKIHeader.CMP_2000, PKIHeader.NULL_NAME, PKIHeader.NULL_NAME).
354                 setTransactionID(createSenderNonce()).
355                 setSenderNonce(createSenderNonce()).
356                 setMessageTime(new ASN1GeneralizedTime(new Date())).
357                 build();
358         final ErrorMsgContent errorMessage = new ErrorMsgContent(
359                 new PKIStatusInfo(PKIStatus.rejection, 
360                         new PKIFreeText(errorDescription), 
361                         new PKIFailureInfo(PKIFailureInfo.badRequest))); 
362         final PKIBody pkiBody = new PKIBody(PKIBody.TYPE_ERROR, errorMessage);
363         final PKIMessage pkiResponse = new PKIMessage(pkiHeader, pkiBody);
364         return CmpMessageHelper.pkiMessageToByteArray(pkiResponse);
365     }

368      * Create an unsigned RFC 4210 error message as described in section 5.3.21 based on a PKIHeader obtained from
369      * a previous CMP client request message.

376         final CmpErrorResponseMessageonseMessage.html#CmpErrorResponseMessage">CmpErrorResponseMessage resp = new CmpErrorResponseMessage(); 
377         try {
378             if (pkiHeader == null) {
379                 pkiHeader = new PKIHeader(PKIHeader.CMP_2000, PKIHeader.NULL_NAME, PKIHeader.NULL_NAME);

490     /**
491      * Converts the header and the body of a PKIMessage to an ASN1Encodable and 
492      * returns the as a byte array
493      *  
494      * @param header header
495      * @param body body
496      * @return the PKIMessage's header and body in byte array
497      */
498     public static byte[] getProtectedBytes(PKIHeader header, PKIBody body) {
499         byte[] res = null;
500         ASN1EncodableVector v = new ASN1EncodableVector();
501         v.add(header);
502         v.add(body);
503         ASN1Encodable protectedPart = new DERSequence(v);
504         try {
505             ByteArrayOutputStream bao = new ByteArrayOutputStream();
506             DEROutputStream out = new DEROutputStream(bao);
507             out.writeObject(protectedPart);
508             res = bao.toByteArray();
509         } catch (Exception ex) {
510             LOG.error(ex.getLocalizedMessage(), ex);
511         }
512         return res;
513     }

515     /**
516      * Parses a CRMF request created with novosec library classes and return a bouncycastle CertReqMsg object
517      * 
518      * @param messages msg
519      * @return req
520      */
521     public static CertReqMsg getNovosecCertReqMsg(CertReqMessages messages) {
522         // The encoding of the ProofOfPosession in bouncycastle and novosec is different.
523         // Novosec generator explicitly tags the PopoSigningKey while it should be implicitly tagged.
524         // Through novosec, the ProofOfPosession comes through as:
525         //         Sequence
526         //             DERSequence
527         //                 DERSequence
528         //                     ObjectIdentifier(1.2.840.113549.1.1.5)
529         //                 DERBitString[64,0]
530         //
531         // But it should be:
532         //         DERSequence
533         //             DERSequence
534         //                 ObjectIdentifier(1.2.840.113549.1.1.5)
535         //             DERBitString[64,0]
536         //
537         // The bouncycastle parser expects an implicit tag, so to it, it looks like the sequence is containing a single element.
538         //--------------------------------------
539         // A comment from bouncycastle that might not effect anything here but maybe effect something else in the future: 
540         //         What's happened is the novosec generator has explicitly tagged the PopoSigningKey structure, it should be 
541         //         implicitly tagged (this isn't true if it's a POPOPrivKey, but that's because it's a CHOICE item so the tag 
542         //         has to be preserved, but that is a different story).
543 
544         // Reconstructing the CertRequest
545         ASN1Encodable o2 = ((DERSequence) messages.toASN1Primitive()).getObjectAt(0);
546         ASN1Encodable o3 = ((DERSequence) o2).getObjectAt(0);
547         CertRequest cr = CertRequest.getInstance(o3);
548 
549         // Reconstructing the proof-of-posession
550         ASN1TaggedObject o4 = (ASN1TaggedObject) ((DERSequence) o2).getObjectAt(1);
551         ProofOfPossession pp;
552         int tagnr = o4.getTagNo();
553         ASN1Encodable o5;
554         switch (tagnr) {
555         case 0:
556             o5 = DERNull.INSTANCE;
557             pp = new ProofOfPossession();
558             break;
559         case 1:
560             o5 = POPOSigningKey.getInstance(o4.getObject());
561             pp = new ProofOfPossession((POPOSigningKey) o5);
562             break;
563         case 2:
564         case 3:
565             o5 = POPOPrivKey.getInstance(o4, false);
566             pp = new ProofOfPossession(tagnr, (POPOPrivKey) o5);
567             break;
568         default:
569             throw new IllegalArgumentException("unknown tag: " + tagnr);
570         }
571 
572         // Reconstructing the regToken
573         ASN1Sequence o6 = (ASN1Sequence) ((ASN1Sequence) o2.toASN1Primitive()).getObjectAt(2);
574         final AttributeTypeAndValue av = AttributeTypeAndValue.getInstance(o6.getObjectAt(0));
575         final AttributeTypeAndValue[] avs = { av };
576 
577         // finally, recreating the CertReqMsg object
578         return new CertReqMsg(cr, pp, avs);
579     }

581     public static RevDetails getNovosecRevDetails(RevReqContent revContent) {
582         // Novosec implements RFC2510, while bouncycastle 1.47 implements RFC4210.
583         //
584         // In RFC2510/novosec, the RevDetails structure looks like this:
585         //              RevDetails ::= SEQUENCE {
586         //                                  certDetails         CertTemplate,
587         //                                  revocationReason    ReasonFlags      OPTIONAL,
588         //                                  badSinceDate        GeneralizedTime  OPTIONAL,
589         //                                  crlEntryDetails     Extensions       OPTIONAL
590         //             }
591         //
592         // In RFC4210/bouncycastle, the REVDetails structure looks like this:
593         //                 RevDetails ::= SEQUENCE {
594         //                                  certDetails         CertTemplate,
595         //                                  crlEntryDetails     Extensions       OPTIONAL
596         //                  }
597         //
598         // This means that there is a chance that the request generated using novosec specifies the revocation reason in 'revocationReason' and not
599         // as an extension, leading to Ejbca not being able to parse the request using bouncycastle OR not setting the correct revocation reason.
600 
601         ASN1Encodable o2 = ((DERSequence) revContent.toASN1Primitive()).getObjectAt(0);
602         ASN1Encodable o3 = ((DERSequence) o2).getObjectAt(0);
603         CertTemplate ct = CertTemplate.getInstance(o3);
604 
605         ReasonFlags reasonbits = null;
606         Extensions crlEntryDetails = null;
607         int seqSize = ((DERSequence) o2).size();
608         for (int i = 1; i < seqSize; i++) {
609             ASN1Encodable o4 = ((DERSequence) o2).getObjectAt(i);
610             if (o4 instanceof DERBitString) {
611                 reasonbits = new ReasonFlags((DERBitString) o4);
612             } else if (o4 instanceof DERGeneralizedTime) {
613                 DERGeneralizedTime.getInstance(o4); // bad since time, not used in the bouncycastle class
614             } else if (o4 instanceof DERSequence) {
615                 crlEntryDetails = Extensions.getInstance(o4);
616             }
617         }
618 
619         if ((crlEntryDetails != null) && (reasonbits != null)) {
620             Extension reason = crlEntryDetails.getExtension(Extension.reasonCode);
621             if (reason == null) {
622                 reason = new Extension(Extension.reasonCode, true, ASN1OctetString.getInstance(reasonbits.getBytes()));
623             }
624         } else if ((crlEntryDetails == null) && (reasonbits != null)) {
625             ExtensionsGenerator extgen = new ExtensionsGenerator();
626             try {
627                 extgen.addExtension(Extension.reasonCode, true, ASN1OctetString.getInstance(reasonbits.getBytes()));
628                 crlEntryDetails = extgen.generate();
629             } catch (IOException e) {
630                 LOG.error(e.getLocalizedMessage(), e);
631             }
632         }
633 
634         //The constructor RevDetails(certTemplate, crlEntryDetails) only sets 'crlEntryDetails' and ignores 'certTemplate'
635         //This is a reported bug in bouncycastle. For now, the only way to have both of them set is to create a ASN1/DERSequence 
636         ASN1EncodableVector seq = new ASN1EncodableVector();
637         seq.add(ct);
638         seq.add(crlEntryDetails);
639         RevDetails res = RevDetails.getInstance(new DERSequence(seq));
640         return res;
641     }

643     /** @param octets ASN.1
644      * @return SenderKeyId of in the header or null none was found. */
645     public static String getStringFromOctets(final ASN1OctetString octets) {
646         String str = null;
647         if (octets != null) {
648             byte[] bytes = octets.getOctets();
649             if (bytes.length > 128) { // arbitrary limitation on octet string size
650                 LOG.info("Truncating octet string longer than 128 bytes");
651                 byte[] newbytes = Arrays.copyOf(bytes, 128);
652                 bytes = newbytes;
653             }
654             str = new String(bytes, StandardCharsets.UTF_8);
655             if (StringTools.isAlphaOrAsciiPrintable(str)) {
656                 if (LOG.isDebugEnabled()) {
657                     LOG.debug("Found string: " + str);
658                 }
659             } else {
660                 str = new String(Hex.encode(octets.getOctets()));
661                 if (LOG.isDebugEnabled()) {
662                     LOG.debug("DEROCtetString content is not alphaNumeric (including space), converting to hex: " + str);
663                 }
664             }
665         }
666         return str;
667     }

669     /** @param pkiMessageBytes bytes
670      * @param sanityCheckMaxLevelOfNesting bool 
671      * @return the PKIMessage if the bytes can be interpreted as a valid ASN.1 encoded CMP request message or null otherwise */
672     public static PKIMessage getPkiMessageFromBytes(final byte[] pkiMessageBytes, final boolean sanityCheckMaxLevelOfNesting) {
673         try {
674             if (pkiMessageBytes!=null) {
675                 final PKIMessage pkiMessage = PKIMessage.getInstance(pkiMessageBytes);
676                 if (sanityCheckMaxLevelOfNesting) {
677                     // Also validate nesting, if present
678                     PKIMessage nestedPkiMessage = pkiMessage;
679                     int levelOfNesting = 0;
680                     while (nestedPkiMessage!=null && nestedPkiMessage.getBody().getType()==PKIBody.TYPE_NESTED) {
681                         nestedPkiMessage = PKIMessages.getInstance(pkiMessage.getBody().getContent()).toPKIMessageArray()[0];
682                         if (levelOfNesting++ > MAX_LEVEL_OF_NESTING) {
683                             final String msg = "Rejected CMP request due to unreasonable level of nesting (>"+MAX_LEVEL_OF_NESTING+").";
684                             LOG.info(msg);
685                             throw new IllegalArgumentException(msg);
686                         }
687                     }
688                 }
689                 return pkiMessage;
690             }
691         } catch (RuntimeException e) {
692             // BC library will throw an IllegalArgumentException if the underlying ASN.1 could not be parsed. 
693             if (LOG.isDebugEnabled()) {
694                 LOG.debug(INTRES.getLocalizedMessage("cmp.errornotcmpmessage"), e);
695             }
696         }
697         return null;
698     }
