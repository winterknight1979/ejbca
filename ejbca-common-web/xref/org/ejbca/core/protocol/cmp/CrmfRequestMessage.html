166     private void init() {
167         final PKIBody pkiBody = getPKIMessage().getBody();
168         final PKIHeader pkiHeader = getPKIMessage().getHeader();
169         requestType = pkiBody.getType();
170         final CertReqMessages msgs = getCertReqFromTag(pkiBody, requestType);
171         try {
172             this.req = msgs.toCertReqMsgArray()[0];
173         } catch(Exception e) {
174             this.req = CmpMessageHelper.getNovosecCertReqMsg(msgs);
175         }
176         requestId = this.req.getCertReq().getCertReqId().getValue().intValue();
177         setTransactionId(getBase64FromAsn1OctetString(pkiHeader.getTransactionID()));
178         setSenderNonce(getBase64FromAsn1OctetString(pkiHeader.getSenderNonce()));
179         setRecipient(pkiHeader.getRecipient());
180         setSender(pkiHeader.getSender());
181     }

183     @Override
184     public PublicKey getRequestPublicKey() throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException {
185         // If we have generated a key pair by the server, we should use this one
186         if (serverGenKeyPair != null) {
187             return serverGenKeyPair.getPublic();
188         }
189         // Else, see if we can find one in the request
190         final SubjectPublicKeyInfo keyInfo = getRequestSubjectPublicKeyInfo();
191         if (keyInfo == null) {
192             // No public key, which may be OK if we are requesting server generated keys
193             return null;
194         }
195         final PublicKey pk = getPublicKey(keyInfo, BouncyCastleProvider.PROVIDER_NAME);
196         return pk;
197     }

207     @SuppressWarnings("unlikely-arg-type")
208 	private PublicKey getPublicKey(final SubjectPublicKeyInfo subjectPKInfo, final String provider) throws NoSuchAlgorithmException,
209             NoSuchProviderException, InvalidKeyException {
210         // If there is no public key here, but only an empty bit string, it means we have called for server generated keys
211         // i.e. no public key to see here...
212         if (subjectPKInfo.getPublicKeyData().equals(DERNull.INSTANCE)) {
213             return null;
214         }
215         try {
216             final X509EncodedKeySpec xspec = new X509EncodedKeySpec(new DERBitString(subjectPKInfo).getBytes());
217             final AlgorithmIdentifier keyAlg = subjectPKInfo.getAlgorithm();
218             return KeyFactory.getInstance(keyAlg.getAlgorithm().getId(), provider).generatePublic(xspec);
219         } catch (InvalidKeySpecException | IOException e) {
220             final InvalidKeyException newe = new InvalidKeyException("Error decoding public key.");
221             newe.initCause(e);
222             throw newe;
223         }
224     }

226     @Override
227     public PublicKey getProtocolEncrKey() throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException {
228         final CertRequest request = getReq().getCertReq();
229         Controls controls = request.getControls();
230         if (controls != null) {
231             AttributeTypeAndValue[] avs = controls.toAttributeTypeAndValueArray();
232             if (avs != null) {
233                 for (int i = 0; i < avs.length; i++) {
234                     if (avs[i].getType().equals(CrmfRequestMessage.id_regCtrl_protocolEncrKey)) {
235                         ASN1Encodable asn1 = avs[i].getValue();
236                         if (asn1 != null) {
237                             SubjectPublicKeyInfo spi = SubjectPublicKeyInfo.getInstance(asn1);
238                             if (spi != null) {
239                                 return getPublicKey(spi, BouncyCastleProvider.PROVIDER_NAME);
240                             }
241                         }
242                     }
243                 }                
244             }
245         }
246         return null;
247     }

289     @Override
290     public String getUsername() {
291         String ret = null;
292         if (username != null) {
293             ret = username;
294         } else {
295             // We can configure which part of the users DN should be used as username in EJBCA, for example CN or UID
296             String component = extractUsernameComponent;
297             if (StringUtils.isEmpty(component)) {
298                 component = "CN";
299             }
300             String name = CertTools.getPartFromDN(getRequestDN(), component);
301             if (name == null) {
302                 log.error("No component " + component + " in DN: " + getRequestDN());
303             } else {
304                 ret = name;
305             }
306         }
307         if (log.isDebugEnabled()) {
308             log.debug("Username is: " + ret);
309         }
310         return ret;
311     }

456     @Override
457     public boolean verify() throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException {
458         boolean ret = false;
459         final ProofOfPossession pop = getReq().getPopo();
460         if (log.isDebugEnabled()) {
461             log.debug("allowRaVerifyPopo: " + allowRaVerifyPopo);
462             if (pop != null) {
463                 log.debug("pop.getRaVerified(): " + (pop.getType() == ProofOfPossession.TYPE_RA_VERIFIED));                
464             } else {
465                 log.debug("No POP in message");
466             }
467         }
468         if (pop == null) {
469             // POP can be null only if we don't have a public key in the message, then we request
470             // server generated keys, and don't send any POP
471             // This can be either no public key info, or public key info with a algId followed by a zero length bitstring
472             // SubjectPublicKeyInfo ::= SEQUENCE {
473             //   algorithm AlgorithmIdentifier,
474             //   publicKey BIT STRING }
475             SubjectPublicKeyInfo pkinfo = getRequestSubjectPublicKeyInfo();
476             if (pkinfo == null) {
477                 if (log.isDebugEnabled()) {
478                     log.debug("POP is not present, but neither is a SubjectPublicKeyInfo, so POP is OK...for server generated keys.");
479                 }
480                 ret = true; // public key null, this is OK when there is no POP
481             } else if (pkinfo.getAlgorithm() != null && pkinfo.getPublicKeyData().intValue() == 0) {
482                 if (log.isDebugEnabled()) {
483                     log.debug("POP is not present, but SubjectPublicKeyInfo is, with an algId followed by zero length data, so POP is OK...for server generated keys.");
484                 }
485                 ret = true;
486             } else {
487                 if (log.isDebugEnabled()) {
488                     log.debug("POP is not present, but SubjectPublicKey is, but not with an algId followed by zero length data, POP is not OK...not even for server generated keys.");
489                 }
490                 ret = false;
491             }
492         } else if ( allowRaVerifyPopo && (pop.getType() == ProofOfPossession.TYPE_RA_VERIFIED) ) {
493             ret = true;
494         } else if (pop.getType() == ProofOfPossession.TYPE_SIGNING_KEY) {
495             try {
496                 final POPOSigningKey sk = (POPOSigningKey) pop.getObject();
497                 final POPOSigningKeyInput pski = sk.getPoposkInput();
498                 ASN1Encodable protObject = pski;
499                 // Use of POPOSigningKeyInput or not, as described in RFC4211, section 4.1.
500                 if (pski == null) {
501                     if (log.isDebugEnabled()) {
502                         log.debug("Using CertRequest as POPO input because POPOSigningKeyInput is missing.");
503                     }
504                     protObject = getReq().getCertReq();
505                 } else {
506                     // Assume POPOSigningKeyInput with the public key and name, MUST be the same as in the request according to RFC4211
507                     if (log.isDebugEnabled()) {
508                         log.debug("Using POPOSigningKeyInput as POPO input.");
509                     }
510                     final CertRequest req = getReq().getCertReq();
511                     // If subject is present in cert template it must be the same as in POPOSigningKeyInput
512                     final X500Name subject = req.getCertTemplate().getSubject();
513                     if (subject != null && !subject.toString().equals(pski.getSender().getName().toString())) {
514                         log.info("Subject '" + subject.toString() + "', is not equal to '" + pski.getSender().toString() + "'.");
515                         protObject = null; // pski is not a valid protection object
516                     }
517                     // If public key is present in cert template it must be the same as in POPOSigningKeyInput
518                     final SubjectPublicKeyInfo pk = req.getCertTemplate().getPublicKey();
519                     if (pk != null && !Arrays.areEqual(pk.getEncoded(), pski.getPublicKey().getEncoded())) {
520                         log.info("Subject key in cert template, is not equal to subject key in POPOSigningKeyInput.");
521                         protObject = null; // pski is not a valid protection object
522                     }
523                 }
524                 // If a protectObject is present we extract the bytes and verify it
525                 if (protObject != null) {
526                     final ByteArrayOutputStream bao = new ByteArrayOutputStream();
527                     new DEROutputStream(bao).writeObject(protObject);
528                     final byte[] protBytes = bao.toByteArray();
529                     final AlgorithmIdentifier algId = sk.getAlgorithmIdentifier();
530                     if (log.isDebugEnabled()) {
531                         log.debug("POP protection bytes length: " + (protBytes != null ? protBytes.length : "null"));
532                         log.debug("POP algorithm identifier is: " + algId.getAlgorithm().getId());
533                     }
534                     final Signature sig = Signature.getInstance(algId.getAlgorithm().getId(), "BC");
535                     sig.initVerify(getRequestPublicKey());
536                     sig.update(protBytes);
537                     final DERBitString bs = sk.getSignature();
538                     ret = sig.verify(bs.getBytes());
539                     if (log.isDebugEnabled()) {
540                         log.debug("POP verify returns: " + ret);
541                     }
542                 }
543             } catch (IOException e) {
544                 log.error("Error encoding CertReqMsg: ", e);
545             } catch (SignatureException e) {
546                 log.error("SignatureException verifying POP: ", e);
547             }
548         }
549         return ret;
550     }
