242     @SuppressWarnings("deprecation")
243 	@Override
244     public boolean create() throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException {
245         boolean ret = false;
246         // Some general stuff, common for all types of messages
247         String issuer = null;
248         String subject = null;
249         Certificate signCert = null;
250         if (CollectionUtils.isNotEmpty(signCertChain)) {
251             signCert = signCertChain.iterator().next();            
252         }
253         if (cert != null) {
254             final X509Certificate x509cert = (X509Certificate) cert;
255             issuer = x509cert.getIssuerDN().getName();
256             subject = x509cert.getSubjectDN().getName();
257         } else if (signCert != null) {
258             issuer = ((X509Certificate) signCert).getSubjectDN().getName();
259             subject = "CN=fooSubject";
260         } else {
261             issuer = "CN=fooIssuer";
262             subject = "CN=fooSubject";
263         }
264 
265 		final GeneralName issuerName = new GeneralName(new X500Name(issuer));
266 		final GeneralName subjectName = new GeneralName(new X500Name(subject));
267 		final PKIHeaderBuilder myPKIHeader = CmpMessageHelper.createPKIHeaderBuilder(issuerName, subjectName, senderNonce, recipientNonce, transactionId);
268 		PKIBody myPKIBody = null;
269 		final PKIMessage myPKIMessage;
270 
271         try {
272             if (status.equals(ResponseStatus.SUCCESS)) {
273                 if (cert != null) {
274                     if (log.isDebugEnabled()) {
275                         log.debug("Creating a CertRepMessage 'accepted'");
276                     }
277                     PKIStatusInfo myPKIStatusInfo = new PKIStatusInfo(PKIStatus.granted); // 0 = accepted
278                     ASN1InputStream certASN1InputStream = new ASN1InputStream(new ByteArrayInputStream(cert.getEncoded()));
279                     try {
280                         try {
281                             CMPCertificate cmpcert = CMPCertificate.getInstance(certASN1InputStream.readObject());
282                             CertOrEncCert retCert = new CertOrEncCert(cmpcert);
283                             CertifiedKeyPair myCertifiedKeyPair;
284                             // If the requestMessage has a server generated key pair, and the requestMessage had a public key 
285                             // "controls.protocolEncrKey" to encrypt the private key with
286                             if (reqMsg != null && reqMsg.getServerGenKeyPair() != null && reqMsg.getProtocolEncrKey() != null) {
287                                     log.debug("CMP request had a server generated key pair and controls.protocolEncrKey which we will use to encrypt the private key in the response");
288                                     final KeyPair kp = reqMsg.getServerGenKeyPair();
289                                     final PublicKey protocolEncrKey = reqMsg.getProtocolEncrKey();
290                                     if (!protocolEncrKey.getAlgorithm().equals("RSA")) {
291                                         final String msg = "CMP request had a controls.protocolEncrKey that is not an RSA key, can not create response: "+protocolEncrKey.getAlgorithm();
292                                         log.debug(msg);
293                                         throw new InvalidKeyException(msg);                                        
294                                     }
295                                     // JceAsymmetricKeyWrapper sets kp.getPublic to be the key used for wrapping
296                                     // JceCRMFEncryptorBuilder sets AES256 CBC to be the symmetric encryption algorithm used 
297                                     JcaEncryptedValueBuildereBuilder.html#JcaEncryptedValueBuilder">JcaEncryptedValueBuilder encBldr = new JcaEncryptedValueBuilder(
298                                             new JceAsymmetricKeyWrapper(protocolEncrKey).setProvider(BouncyCastleProvider.PROVIDER_NAME),
299                                             new JceCRMFEncryptorBuilder(CMSAlgorithm.AES256_CBC).setProvider(BouncyCastleProvider.PROVIDER_NAME).build());
300                                     // encBldr.build encrypts the privateKey using the wrapper above, i.e. encrypted with AES128_CBC with the symmkey wrapped with kp.getPublic
301                                     myCertifiedKeyPair = new CertifiedKeyPair(retCert, encBldr.build(kp.getPrivate()), null);                                    
302                             } else if (reqMsg != null && reqMsg.getServerGenKeyPair() != null && reqMsg.getProtocolEncrKey() == null) {
303                                 // We should actually check this in the outer CMP layers before trying to create a real certificate response, but of course we have to check in here as well
304                                 final String msg = "CMP request had a server generated key pair but no controls.protocolEncrKey, can not create response";
305                                 log.debug(msg);
306                                 throw new InvalidKeyException(msg);
307                             } else {
308                                 myCertifiedKeyPair = new CertifiedKeyPair(retCert);                                
309                             }
310                             // If we have server generated keys, add privateKey
311                             final CertResponse certResponse = new CertResponse(new ASN1Integer(requestId), myPKIStatusInfo, myCertifiedKeyPair, null);
312                             final CertResponse[] certResponses = { certResponse };
313                             
314                             // Add the user certificates signing CA certificate (at index 0) and the others by the CMP configuration to the CMP 
315                             // response 'caPubs' field (added previously to the response with CertificateResponseMessage.addAdditionalCaCertificates().
316                             final List<CMPCertificate> caPubs = new ArrayList<CMPCertificate>();
317                             for (Certificate certificate : this.cacert) {
318                                 try (ASN1InputStream stream = new ASN1InputStream(new ByteArrayInputStream(certificate.getEncoded()));) {
319                                     caPubs.add(CMPCertificate.getInstance(stream.readObject()));
320                                 }
321                             }
322 
323                             final CertRepMessage myCertRepMessage = new CertRepMessage(caPubs.toArray( new CMPCertificate[] {}), certResponses);
324                             int respType = requestType + 1; // 1 = intitialization response, 3 = certification response etc
325                             if (log.isDebugEnabled()) {
326                                 log.debug("Creating response body of type " + respType);
327                             }
328                             myPKIBody = new PKIBody(respType, myCertRepMessage);
329                             // All good, see if we should add implicitConfirm
330                             if (implicitConfirm) {
331                                 if (log.isDebugEnabled()) {
332                                     log.debug("Adding implicitConform to CMP response message with transId: "+transactionId);
333                                 }
334                                 final InfoTypeAndValue genInfo = new InfoTypeAndValue(CMPObjectIdentifiers.it_implicitConfirm);
335                                 myPKIHeader.setGeneralInfo(genInfo);                                
336                             }
337                         } finally {
338                             certASN1InputStream.close();
339                         }
340                     } catch (IOException e) {
341                         throw new IllegalStateException("Unexpected IOException caught.", e);
342                     }
343                 }
344             } else if (status.equals(ResponseStatus.FAILURE)) {
345                 if (log.isDebugEnabled()) {
346                     log.debug("Creating a CertRepMessage 'rejected'");
347                 }
348                 // Create a failure message
349                 ASN1EncodableVector statusInfoV = new ASN1EncodableVector();
350                 statusInfoV.add(ASN1Integer.getInstance(PKIStatus.rejection.toASN1Primitive()));
351                 if (failText != null) {
352                     statusInfoV.add(new PKIFreeText(new DERUTF8String(failText)));
353                 }
354                 statusInfoV.add(CmpMessageHelper.getPKIFailureInfo(failInfo.intValue()));
355                 PKIStatusInfo myPKIStatusInfo = PKIStatusInfo.getInstance(ASN1Sequence.getInstance(new DERSequence(statusInfoV)));
356                 myPKIBody = CmpMessageHelper.createCertRequestRejectBody(myPKIStatusInfo, requestId, requestType);
357                 
358             } else {
359                 if (log.isDebugEnabled()) {
360                     log.debug("Creating a 'waiting' message?");
361                 }
362                 // Not supported, lets create a PKIError failure instead
363                 // Create a failure message
364                 ASN1EncodableVector statusInfoV = new ASN1EncodableVector();
365                 statusInfoV.add(PKIStatus.rejection); // 2 = rejection
366                 if (failText != null) {
367                     statusInfoV.add(new PKIFreeText(new DERUTF8String(failText)));
368                 }
369                 statusInfoV.add(CmpMessageHelper.getPKIFailureInfo(failInfo.intValue()));
370                 PKIStatusInfo myPKIStatusInfo = PKIStatusInfo.getInstance(new DERSequence(statusInfoV));
371                 
372                 ErrorMsgContent myErrorContent = new ErrorMsgContent(myPKIStatusInfo);
373                 myPKIBody = new PKIBody(23, myErrorContent); // 23 = error                
374             }
375             
376             if ((pbeKeyId != null) && (pbeKey != null) && (pbeDigestAlg != null) && (pbeMacAlg != null)) {
377                 myPKIHeader.setProtectionAlg(new AlgorithmIdentifier(CMPObjectIdentifiers.passwordBasedMac));
378                 PKIHeader header = myPKIHeader.build();
379                 myPKIMessage = new PKIMessage(header, myPKIBody);
380                 responseMessage = CmpMessageHelper.protectPKIMessageWithPBE(myPKIMessage, pbeKeyId, pbeKey, pbeDigestAlg, pbeMacAlg,
381                         pbeIterationCount);
382             } else {
383                 myPKIHeader.setProtectionAlg(new AlgorithmIdentifier(new ASN1ObjectIdentifier(digest)));
384                 if (signCert != null) {
385                 	// set sender Key ID as well when the response is signed, so the signer (CA) can have multiple certificates out there
386                 	// with the same DN but different keys
387                     myPKIHeader.setSenderKID(CertTools.getSubjectKeyId(signCert));
388                 }
389                 PKIHeader header = myPKIHeader.build();
390                 final Collection<Certificate> extraCertsList = new ArrayList<Certificate>(signCertChain);
391                 for (Certificate extraCert : extraCerts) {
392                     if (!extraCertsList.contains(extraCert)) {
393                         extraCertsList.add(extraCert);
394                     }
395                 }
396                 myPKIMessage = new PKIMessage(header, myPKIBody);
397                 responseMessage = CmpMessageHelper.signPKIMessage(myPKIMessage, extraCertsList, signKey, digest, provider);
398             }
399             
400             ret = true;
401             
402         } catch (CertificateEncodingException e) {
403             log.error("Error creating CertRepMessage: ", e);
404         } catch (InvalidKeyException e) {
405             log.error("Error creating CertRepMessage: ", e);
406         } catch (NoSuchProviderException e) {
407             log.error("Error creating CertRepMessage: ", e);
408         } catch (NoSuchAlgorithmException e) {
409             log.error("Error creating CertRepMessage: ", e);
410         } catch (SecurityException e) {
411             log.error("Error creating CertRepMessage: ", e);
412         } catch (SignatureException e) {
413             log.error("Error creating CertRepMessage: ", e);
414         } catch (CRMFException e) {
415             log.error("Error creating CertRepMessage: ", e);
416         }
417 
418         return ret;
419     }

485     @Override
486     public void addAdditionalCaCertificates(final List<Certificate> certificates) {
487         if (CollectionUtils.isNotEmpty(certificates)) {
488             for (Certificate certificate : certificates) {
489                 if (!this.cacert.contains(certificate)) {
490                     this.cacert.add( certificate);
491                 }
492             }
493         }
