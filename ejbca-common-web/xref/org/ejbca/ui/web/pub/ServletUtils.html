3   import javax.servlet.http.HttpServletResponse;
4   
5   import org.apache.log4j.Logger;
6   
7   /** 
8    * A class containing some helpful functions used in more than one Servlet, avoiding code duplication.
9    * 
10   * @version $Id: ServletUtils.java 22139 2015-11-03 10:41:56Z mikekushner $
11   */
12  public class ServletUtils {
13  
14      private static final Logger log = Logger.getLogger(ServletUtils.class);
15  
16      /** Helper methods that removes no-cache headers from a response. No-cache headers 
17       * makes IE refuse to save a file that is sent (for example a certificate). 
18       * No-cache headers are also automatically added by Tomcat by default, so we better
19       * make sure they are set to a harmless value.
20       * 
21       * @param res HttpServletResponse parameter as taken from the doGet, doPost methods in a Servlet.
22       */
23      public static void removeCacheHeaders(final HttpServletResponse res) {
24          if (res.containsHeader("Pragma")) {
25              if (log.isDebugEnabled()) {
26                  log.debug("Removing Pragma header to avoid caching issues in IE");
27              }
28              res.setHeader("Pragma","null");
29          }
30          if (res.containsHeader("Cache-Control")) {
31              if (log.isDebugEnabled()) {
32                  log.debug("Removing Cache-Control header to avoid caching issues in IE");
33              }
34              res.setHeader("Cache-Control","null");
35          }
36      }
37  
38      /** Helper methods that adds no-cache headers to a response. 
39       * 
40       * @param res HttpServletResponse parameter as taken from the doGet, doPost methods in a Servlet.
41       */
42      public static void addCacheHeaders(final HttpServletResponse res) {
43          if (!res.containsHeader("Pragma")) {
44              if (log.isDebugEnabled()) {
45                  log.debug("Adding Pragma header");
46              }
47              res.setHeader("Pragma","no-cache");
48          }
49          if (!res.containsHeader("Cache-Control")) {
50              if (log.isDebugEnabled()) {
51                  log.debug("Adding Cache-Control header");
52              }
53              res.setHeader("Cache-Control","no-cache");
54          }
55      }
56  }
