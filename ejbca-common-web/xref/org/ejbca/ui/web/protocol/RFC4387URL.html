13  package org.ejbca.ui.web.protocol;
14  
15  import org.cesecore.certificates.certificate.HashID;
16  
17  
18  /**
19   * Create RFC4387 URLs and HTML references to such URLs. 
20   * @author Lars Silven PrimeKey
21   * @version  $Id: RFC4387URL.java 19901 2014-09-30 14:29:38Z anatom $
22   */
23  public enum RFC4387URL {
24  	sHash,
25  	iHash,
26  	sKIDHash;
27  	private String getID(HashID hash, boolean isDelta, boolean isHTML) {
28  		final String theAmp = isHTML ? "&amp;" : "&";
29  		final String deltaParam = isDelta ? theAmp+"delta=" : "";
30  		return hash.getB64url() + deltaParam;
31  	}
32  	private String appendQueryToURL(String url, String id) {
33  		return url+"?"+this.toString()+"="+id;
34  	}
35  	/**
36  	 * Append the query of the RFC hash to a URL
37  	 * @param url The URL except the query
38  	 * @param hash of the object to fetch
39  	 * @param isDelta true if it is a link to a delta CRL.
40  	 * @return URL to fetch certificate or CRL.
41  	 */
42  	public String appendQueryToURL(String url, HashID hash, boolean isDelta) {
43  		final String id = getID(hash, isDelta, false);
44  		return appendQueryToURL(url, id);
45  	}
46  	/**
47  	 * See {@link #appendQueryToURL(String, HashID, boolean)}, isDelta is false.
48  	 * @param url The URL except the query
49  	 * @param hash of the object to fetch
50  	 * @return URL to fetch certificate or CRL.
51  	 */
52  	public String appendQueryToURL(String url, HashID hash) {
53  		return appendQueryToURL(url, hash, false);
54  	}
55  	/**
56  	 * HTML string that show the reference to fetch a certificate or CRL.
57  	 * @param url The URL except the query
58  	 * @param hash of the object to fetch
59  	 * @param isDelta true if it is a link to a delta CRL.
60  	 * @return URL to fetch certificate or CRL.
61  	 */
62  	public String getRef(String url, HashID hash, boolean isDelta) {
63  		final String resURL = appendQueryToURL(url, getID(hash, isDelta, true));
64  		return this.toString()+" = "+hash.getB64()+(isDelta ? " delta":"")+" <a href=\""+resURL+"\">Download</a>";
65  	}
66  	/**
67  	 * See {@link #getRef(String, HashID, boolean)}, isDelta is false.
68  	 * @param url The URL except the query
69  	 * @param hash of the object to fetch
70  	 * @return URL to fetch certificate or CRL.
71  	 */
72  	public String getRef(String url, HashID hash) {
73  		return getRef(url, hash, false);
74  	}
75  }
