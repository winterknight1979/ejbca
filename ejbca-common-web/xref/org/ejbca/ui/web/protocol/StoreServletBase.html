106 	@Override
107     protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, java.io.IOException {
108         if (log.isTraceEnabled()) {
109             log.trace(">doGet()");
110         }
111         if (!req.getRequestURI().substring(req.getContextPath().length()).contains("search.cgi")) {
112             resp.sendRedirect(req.getRequestURI() + "search.cgi");
113             return;
114         }
115         try {
116             if (alias(req, resp)) {
117                 return;
118             }
119             if (performReload(req, resp)) {
120                 return;
121             }
122             if (fromName(req, resp)) {
123                 return;
124             }
125             rfcRequest(req, resp);
126         } finally {
127             if (log.isTraceEnabled()) {
128                 log.trace("<doGet()");
129             }
130         }
131     }

133 	private void rfcRequest(HttpServletRequest req, HttpServletResponse resp) throws IOException, ServletException {
134 		// Do actual processing of the protocol
135 		{
136 			final String sHash = req.getParameter(RFC4387URL.sHash.toString());
137 			if ( sHash!=null ) {
138 				sHash( sHash, resp, req );
139 				return;
140 			}
141 		}{
142 			final String iHash = req.getParameter(RFC4387URL.iHash.toString());
143 			if ( iHash!=null ) {
144 				iHash( iHash, resp, req );
145 				return;
146 			}
147 		}{
148 			final String sKIDHash = req.getParameter(RFC4387URL.sKIDHash.toString());
149 			if ( sKIDHash!=null ) {
150 				sKIDHash(sKIDHash, resp, req );
151 				return;
152 			}
153 		}
154 		printInfo(req, resp);
155 	}
156 	private boolean alias(HttpServletRequest req, HttpServletResponse resp) throws IOException {
157 		final String alias = req.getParameter("setAlias");
158 		if ( alias==null ) {
159 			return false;
160 		}
161 		if ( !checkIfAutorizedIP(req, resp) ) {
162 			return true;
163 		}
164 		final int ix = alias.indexOf('=');
165 		if ( ix<1 || alias.length()<=ix+2 ) {
166 			log.debug("No valid alias definition string: "+alias);
167 			return true;
168 		}
169 		final String key = alias.substring(0, ix).trim();
170 		final String hash = alias.substring(ix+1).trim();
171 		if ( !VAConfiguration.sKIDHashSetAlias(key, hash) ) {
172 			log.error("Not possible to add: "+alias);
173 			return true;
174 		}
175 		log.debug("Alias '"+key+"' defined for hash '"+hash+"'.");
176 		return true;
177 	}
178 	
179 	private boolean fromName(HttpServletRequest req, HttpServletResponse resp) throws IOException, ServletException {
180 		final String alias = req.getParameter("alias");
181 		if ( alias==null ) {
182 			return false;
183 		}
184 		final String sKIDHash = VAConfiguration.sKIDHashFromName(alias);
185 		if ( sKIDHash==null || sKIDHash.length()<1 ) {
186 			final String m = "No '"+alias+"' alias defined in va.properties .";
187 			resp.sendError(HttpServletResponse.SC_NOT_FOUND, m);
188 			log.debug(m);
189 			return true;
190 		}
191 		sKIDHash( sKIDHash, resp, req, alias );
192 		return true;
193 	}

195 	/**
196 	 * Reloads the certificate cache, if it was requested.
197 	 * 
198 	 * @param req the HttpServletRequest
199 	 * @param resp the HttpServletResponse
200 	 * @return false if reload wasn't requested, or true if it was requested (even if it turned out to be unauthorized)
201 	 */
202 	private boolean performReload(HttpServletRequest req, HttpServletResponse resp) {
203 		// We have a command to force reloading of the certificate cache that can only be run from localhost
204 		// http://localhost:8080/crls/search.cgi?reloadcache=true
205 		final boolean doReload = StringUtils.equals(req.getParameter("reloadcache"), "true");
206 		if ( !doReload ) {
207 			return false;
208 		}
209 		try {
210             if ( !checkIfAutorizedIP(req, resp) ) {
211             	return true;
212             }
213         } catch (IOException e) {
214             throw new IllegalStateException("Could not send error response", e);
215         }
216 		log.info("Reloading certificate and CRL caches due to request from "+req.getRemoteAddr());
217 		// Reload CA certificates
218 		certificateStoreSession.reloadCaCertificateCache();
219 		return true;
220 	}

222 	/**
223 	 * Checks if the request originates from localhost
224 	 * 
225      * @param req the HttpServletRequest
226      * @param resp the HttpServletResponse
227 	 * @return true if authorized, or false if not, and sends HttpServletResponse.SC_UNAUTHORIZED
228      * @throws IOException if the HttpServletResponse.SC_UNAUTHORIZED response couldn't be sent in case if a negative result
229 	 */
230 	private boolean checkIfAutorizedIP(HttpServletRequest req, HttpServletResponse resp) throws IOException {
231 		final String remote = req.getRemoteAddr();
232 		// localhost in either ipv4 and ipv6
233 		if ( StringUtils.equals(remote, "127.0.0.1") || StringUtils.equals(remote, "0:0:0:0:0:0:0:1") ) {
234 			return true;
235 		}
236 		log.info("Got reloadcache command from unauthorized ip: "+remote);
237 		resp.sendError(HttpServletResponse.SC_UNAUTHORIZED);
238 		return false;
239 	}

241 	private void printInfo(X509Certificate certs[], String indent, PrintWriter pw, String url) {
242 		for (X509Certificate cert : certs) {
243 			// Escape the URL as it might be unsafe
244 			printInfo(cert, indent, pw, HTMLTools.htmlescape(url));
245 			pw.println();
246 			final X509Certificate issuedCerts[] = this.certCache.findLatestByIssuerDN(HashID.getFromSubjectDN(cert));
247 			if (ArrayUtils.isEmpty(issuedCerts)) {
248 				continue;
249 			}
250 			printInfo(issuedCerts, SPACE + indent, pw, url);
251 		}
252 	}

272 	private void returnInfoPage(HttpServletResponse response, String info) throws IOException {
273 		response.setContentType("text/html");
274 		response.setCharacterEncoding("UTF-8");
275 		final PrintWriter writer = response.getWriter();
276 
277 		writer.println("<html>");
278 		writer.println("<head>");
279 		writer.println("<title>"+getTitle()+"</title>");
280 		writer.println("</head>");
281 		writer.println("<body>");
282 
283 		writer.println("<table border=\"0\">");
284 		writer.println("<tr>");
285 		writer.println("<td>");
286 		writer.println("<h1>"+getTitle()+"</h1>");
287 		writer.println("<p>When searching for certificates you can use iHash, sHash and sKIDHash. iHash is the ASN1 encoded DN of the issuer in a certificate, sHash of the subject and sKIDHash is the subjectKeyIdentifier. If you search with it you get all certificates that has the same issuer, except for the root certificate. You do not find a root certificate if you search with the iHash of the root. It has been assumed that sHash should be used when searching for a root.</p>");
288 		writer.println("<p>When searching for CRLs you can use iHash and sKIDHash. iHash is the ASN1 encoded DN of the issuer in a certificate and sKIDHash is the subjectKeyIdentifier.");
289         writer.println("<br/>To get the latest delta CRL you can append the parameter 'delta='.");
290         writer.println("<br/>To get a CRL with a specific CRL number you can append the parameter 'crlnumber=&lt;number&gt;'.</p>");
291 		writer.println("<hr>");
292 		writer.println(info);
293 		writer.println("</td>");
294 		writer.println("</tr>");
295 		writer.println("</table>");
296 
297 		writer.println("</body>");
298 		writer.println("</html>");
299 		writer.flush();
300 	}
301 	private void printInfo(HttpServletRequest req, HttpServletResponse resp) throws IOException {
302 		final StringWriter sw = new StringWriter();
303 		final PrintWriter pw = new HtmlPrintWriter(sw);
304 		printInfo(this.certCache.getRootCertificates(), "", pw, req.getRequestURL().toString());
305 		pw.flush();
306 		pw.close();
307 		sw.flush();
308 		returnInfoPage(resp, sw.toString());
309 		sw.close();
310 	}
