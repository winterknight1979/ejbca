22  /**
23   * Sample filter implementation to define a set of Content Security Policies, and other security headers.<br>
24   * 
25   * This implementation has a dependency on Commons Codec API.<br>
26   * 
27   * This filter set CSP policies using all HTTP headers defined into W3C specification.<br>
28   * and X-XSS-Protection, X-Content-Type-Options, X-FRAME-OPTIONS
29   * <br>
30   * This implementation is oriented to be easily understandable and easily adapted.<br>
31   * 
32   * @version $Id: ContentSecurityPolicyFilter.java 34257 2020-01-13 09:37:53Z anatom $
33   */
34  public class ContentSecurityPolicyFilter implements Filter {
35  
36      /** Configuration member to specify if web app use web fonts */
37      public static final boolean APP_USE_WEBFONTS = false;
38  
39      /** Configuration member to specify if web app use videos or audios */
40      public static final boolean APP_USE_AUDIOS_OR_VIDEOS = false;
41  
42      /** List CSP HTTP Headers */
43      private List<String> cspHeaders = new ArrayList<String>();
44  
45      /** Collection of CSP polcies that will be applied */
46      private String policies = null;
47  
48      /** which mode X-FRAME-OPTIONS should have, default DENY */
49      private String frameOptionsMode = "DENY";
50      
51      /** Used for Script Nonce */
52      //private SecureRandom prng = null;
53  
54      /**
55       * Used to prepare (one time for all) set of CSP policies that will be applied on each HTTP response.
56       * 
57       * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
58       */
59      @Override
60      public void init(FilterConfig filterConfig) throws ServletException {
61          String plugins = filterConfig.getInitParameter("plugins");
62          String objects = filterConfig.getInitParameter("objects");
63          String unsafeeval = filterConfig.getInitParameter("unsafeeval");
64  
65          // Init secure random
66          /*
67          try {
68              this.prng = SecureRandom.getInstance("SHA1PRNG");
69          } catch (NoSuchAlgorithmException e) {
70              throw new ServletException(e);
71          }
72          */
73  
74          // Define list of CSP HTTP Headers
75          // Used by all modern real browsers
76          this.cspHeaders.add("Content-Security-Policy");
77          // Used by IE10 that partially implements CSP
78          this.cspHeaders.add("X-Content-Security-Policy");
79  
80          // Define CSP policies
81          // Loading policies for Frame and Sandboxing will be dynamically defined : We need to know if context use Frame
82          List<String> cspPolicies = new ArrayList<String>();
83          String originLocationRef = "'self'";
84          // --Disable default source in order to avoid browser fallback loading using 'default-src' locations
85          cspPolicies.add("default-src 'none'");
86          // --Define loading policies for Objects (<object> tags)
87          if (StringUtils.isNotEmpty(objects)) {
88              cspPolicies.add("object-src " + originLocationRef);
89          }
90          // --Define loading policies for Styles (CSS), allow inline style elements
91          cspPolicies.add("style-src " + originLocationRef+" 'unsafe-inline'");
92          // --Define loading policies for javascript, allow inline style elements in order to use onClick attributes
93          final String evalstr;
94          if (StringUtils.isNotEmpty(unsafeeval)) {
95              evalstr = " 'unsafe-eval'";
96          } else {
97              evalstr = "";
98          }
99          cspPolicies.add("script-src " + originLocationRef+" 'unsafe-inline'"+evalstr);
100         // --Define loading policies for Images
101         cspPolicies.add("img-src " + originLocationRef);
102         // Frame + Sandbox : Here sandbox allow nothing, customize sandbox options depending on your app....
103         //policiesBuffer.append(";").append("frame-src 'self';sandbox");
104         cspPolicies.add("frame-src 'self'");
105         // --Define loading policies for Audios/Videos
106         if (APP_USE_AUDIOS_OR_VIDEOS) {
107             cspPolicies.add("media-src " + originLocationRef);
108         }
109         // --Define loading policies for Fonts
110         if (APP_USE_WEBFONTS) {
111             cspPolicies.add("font-src " + originLocationRef);
112         }
113         // --Define loading policies for Connection, which we don't use, but may use when we move to advanced JSF pages and components
114         cspPolicies.add("connect-src " + originLocationRef);
115         // --Define loading policies for Form
116         cspPolicies.add("form-action " + originLocationRef);
117         // --Define loading policies for Plugins Types, only allow PDF documents, if plug-ins are enabled in web.xml
118         if (StringUtils.isNotEmpty(plugins)) {
119             cspPolicies.add("plugin-types application/pdf");
120         }
121         // --Define browser XSS filtering feature running mode
122         cspPolicies.add("reflected-xss block");
123 
124         // Target formating
125         this.policies = cspPolicies.toString().replaceAll("(\\[|\\])", "").replaceAll(",", ";").trim();
126 
127         String mode = filterConfig.getInitParameter("frameoptionsmode");
128         if ( mode != null ) {
129             frameOptionsMode = mode;
130         }
131     }
132 
133     /**
134      * Add CSP policies on each HTTP response.
135      * 
136      * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain)
137      */
138     @Override
139     public void doFilter(ServletRequest request, ServletResponse response, FilterChain fchain) throws IOException, ServletException {
140         //HttpServletRequest httpRequest = ((HttpServletRequest) request);
141         HttpServletResponse httpResponse = ((HttpServletResponse) response);
142 
143         /* Step 1 : Detect if target resource is a Frame */
144         // Customize here according to your context...
145 
146         /* Step 2 : Add CSP policies to HTTP response */
147         StringBuilder policiesBuffer = new StringBuilder(this.policies);
148 
149         // Add Script Nonce CSP Policy
150         // With this:
151         // Content security policy script-nonce. Only javascripts that are in a self-source js file, or in-line and declare a nonce 
152         // will be allowed to execute in the browser when the content-security-policy headers are defined with 
153         // "script-src 'self' 'nonce-$RANDOM". 
154         // The RANDOM is passed in each request as an attribute CSP_SCRIPT_NONCE. See ContentSecurityPolicyFilter.java
155         // Add nonce to in-line java scripts like:
156         // <script nonce="<%= (String)request.getAttribute("CSP_SCRIPT_NONCE") %>" type="text/javascript">
157         
158         // We can not use this currently because it would require moving all <a id="item-id" ...onclick="callMethod()"/> into setting event handlers in the javascript methods instead:
159         // document.getElementById('item-id').onclick = callMethod();
160         /*
161         // Set nonce in session variable instead of on the request only
162         // This is needed because EJBCA uses frames. If/when we do not use frames anymore we can simply set the nonce in every request
163         // httpRequest.setAttribute("CSP_SCRIPT_NONCE", scriptNonce);
164         HttpSession session = httpRequest.getSession(false);
165         final String scriptNonce;
166         if ((session == null) || (session.getAttribute("CSP_SCRIPT_NONCE") == null)) {
167             // Generate a new random nonce
168             // --Generate a random number
169             String randomNum = new Integer(this.prng.nextInt()).toString();
170             // --Get its digest
171             MessageDigest sha;
172             try {
173                 sha = MessageDigest.getInstance("SHA-1");
174             }
175             catch (NoSuchAlgorithmException e) {
176                 throw new ServletException(e);
177             }
178             byte[] digest = sha.digest(randomNum.getBytes());
179             // --Encode it into HEXA
180             scriptNonce = Hex.encodeHexString(digest);
181             // Old style, pre-draft standard "script-nonce" directive"
182             //policiesBuffer.append(";").append("script-nonce ").append(scriptNonce);
183             // Standard script-src directive, including nonce
184             /// Content-Security-Policy: script-src 'self' 'nonce-$RANDOM';
185             // Allow all inline javascript with 'unsafe-inline' and 'unsafe-eval'
186             //policiesBuffer.append(";script-src 'self' 'unsafe-inline' 'unsafe-eval'");
187             // Only allow in-line javascript with nonces
188             // --Made available script nonce in view app layer, so we can include it in scripts by retrieving this parameter
189             //httpRequest.setAttribute("CSP_SCRIPT_NONCE", scriptNonce);

191                 session = httpRequest.getSession();

193             session.setAttribute("CSP_SCRIPT_NONCE", scriptNonce);

195             scriptNonce = (String)session.getAttribute("CSP_SCRIPT_NONCE");
196             
197         }
198         policiesBuffer.append("; script-src 'self' 'nonce-").append(scriptNonce).append("'");
199         */
200         
201         // Add policies to all HTTP headers
202         for (String header : this.cspHeaders) {
203             httpResponse.setHeader(header, policiesBuffer.toString());
204         }
205         // Also add X-XSS-Protection, newer header. 
206         // See https://www.owasp.org/index.php/List_of_useful_HTTP_headers
207         // https://blogs.msdn.microsoft.com/ie/2008/07/02/ie8-security-part-iv-the-xss-filter/
208         httpResponse.setHeader("X-XSS-Protection", "1; mode=block");
209         // Also X-Content-Type-Options, see https://blogs.msdn.microsoft.com/ie/2008/09/02/ie8-security-part-vi-beta-2-update/
210         httpResponse.setHeader("X-Content-Type-Options", "nosniff");
211         // Also X-FRAME-OPTIONS, see https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/X-Frame-Options
212         // Used to be in a separate filter, ClickjackFilter, but there is no point in having multiple filters adding security headers
213         httpResponse.setHeader("X-FRAME-OPTIONS", frameOptionsMode );            
214         // New header in 2018, Feature-Policy. https://scotthelme.co.uk/a-new-security-header-feature-policy/
215         // https://github.com/w3c/webappsec-feature-policy/blob/master/features.md
216         // https://w3c.github.io/webappsec-feature-policy/
217         // Add some sensible denies
218         httpResponse.setHeader("Feature-Policy", "vibrate 'none'; autoplay 'none'; camera 'none'; microphone 'none'; midi 'none'; gyroscope 'none'; accelerometer 'none'; magnetometer 'none'; payment 'none'" );            
219         // Referrer policy: https://www.w3.org/TR/referrer-policy/
220         httpResponse.setHeader("Referrer-Policy", "no-referrer-when-downgrade" );            
221 
222         /* Step 3 : Let request continue chain filter */
223         fchain.doFilter(request, response);
224     }
