90      @PostConstruct
91      public void postConstruct() {
92          try {
93              @SuppressWarnings("unchecked")
94              Class<? extends ScepOperationPlugin> extensionClass = (Class<? extends ScepOperationPlugin>) Class.forName(SCEP_RA_MODE_EXTENSION_CLASSNAME);
95              scepRaModeExtension = extensionClass.getConstructor().newInstance();
96          } catch (ClassNotFoundException | NoSuchMethodException e) {
97              scepRaModeExtension = null;
98          } catch (InstantiationException | InvocationTargetException e) {
99              scepRaModeExtension = null;
100             log.error(SCEP_RA_MODE_EXTENSION_CLASSNAME + " was found, but could not be instanced. " + e.getMessage());
101         } catch (IllegalAccessException e) {
102             scepRaModeExtension = null;
103             log.error(SCEP_RA_MODE_EXTENSION_CLASSNAME + " was found, but could not be instanced. " + e.getMessage());
104         }
105         
106         try {
107             @SuppressWarnings("unchecked")
108             Class<ScepResponsePlugin> extensionClass = (Class<ScepResponsePlugin>) Class.forName(SCEP_CLIENT_CERTIFICATE_RENEWAL_CLASSNAME);
109             scepClientCertificateRenewal = extensionClass.getConstructor().newInstance();
110         } catch (ClassNotFoundException | NoSuchMethodException e) {
111             scepClientCertificateRenewal = null;
112         } catch (InstantiationException | InvocationTargetException e) {
113             scepClientCertificateRenewal = null;
114             log.error(SCEP_CLIENT_CERTIFICATE_RENEWAL_CLASSNAME + " was found, but could not be instanced. " + e.getMessage());
115         } catch (IllegalAccessException e) {
116             scepClientCertificateRenewal = null;
117             log.error(SCEP_CLIENT_CERTIFICATE_RENEWAL_CLASSNAME + " was found, but could not be instanced. " + e.getMessage());
118         }

121     @Override
122     @TransactionAttribute(TransactionAttributeType.REQUIRED)
123     public byte[] dispatchRequest(final AuthenticationToken authenticationToken, final String operation, final String message, final String scepConfigurationAlias) 
124             throws NoSuchAliasException, CADoesntExistsException, AuthorizationDeniedException, NoSuchEndEntityException, CustomCertificateSerialNumberException, 
125             CryptoTokenOfflineException, IllegalKeyException, SignRequestException, SignRequestSignatureException, AuthStatusException, AuthLoginException, IllegalNameException, 
126             CertificateCreateException, CertificateRevokeException, CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException, 
127             SignatureException, CertificateException, CertificateExtensionException, CertificateRenewalException {
128         
129         ScepConfiguration scepConfig = (ScepConfiguration) this.globalConfigSession.getCachedConfiguration(ScepConfiguration.SCEP_CONFIGURATION_ID);
130         if(!scepConfig.aliasExists(scepConfigurationAlias)) {
131             throw new NoSuchAliasException();
132         }
133         
134         if (operation.equals("PKIOperation")) {
135             byte[] scepmsg = Base64.decode(message.getBytes());
136             // Read the message and get the certificate, this also checks authorization
137             return scepCertRequest(authenticationToken, scepmsg, scepConfigurationAlias, scepConfig);
138         } else if (operation.equals("GetCACert")) {
139             // CA_IDENT is the message for this request to indicate which CA we are talking about
140             final String caname = getCAName(message);
141             if (log.isDebugEnabled()) {
142                 log.debug("Got SCEP cert request for CA '" + caname + "'");
143             }
144             Collection<Certificate> certs = null;
145             CAInfo cainfo = caSession.getCAInfoInternal(-1, caname, true);
146             if (cainfo != null) {
147                 certs = cainfo.getCertificateChain();
148             }
149             if ((certs != null) && (certs.size() > 0)) {
150                 // CAs certificate is in the first position in the Collection
151                 X509Certificate cert = (X509Certificate) certs.iterator().next();
152                 if (log.isDebugEnabled()) {
153                     log.debug("Sent certificate for CA '" + caname + "' to SCEP client.");
154                 }
155                 return cert.getEncoded();
156             } else {
157                 return null;
158             }
159         } else if (operation.equals("GetCACertChain")) {
160             // CA_IDENT is the message for this request to indicate which CA we are talking about
161             final String caname = getCAName(message);
162             log.debug("Got SCEP pkcs7 request for CA '" + caname + "'. Old client using SCEP draft 18?");
163 
164             CAInfo cainfo = caSession.getCAInfo(authenticationToken, caname);
165             byte[] pkcs7 = signSession.createPKCS7(authenticationToken, cainfo.getCAId(), true);
166             if ((pkcs7 != null) && (pkcs7.length > 0)) {
167                 return pkcs7;
168             } else {
169                 return null;
170             }
171         } else if (operation.equals("GetNextCACert")) {
172             final String caname = getCAName(message);
173             if (log.isDebugEnabled()) {
174                 log.debug("Got SCEP next cert request for CA '" + caname + "'");
175             }
176             final CAInfo cainfo = caSession.getCAInfoInternal(-1, caname, true);
177             if (cainfo == null) {
178                 String errMsg = intres.getLocalizedMessage("scep.errorunknownca", "cert");
179                 log.error(errMsg);
180                 throw new CADoesntExistsException(errMsg);
181             } else {
182                 if (caSession.getFutureRolloverCertificate(cainfo.getCAId()) != null) {
183                     // Send full certificate chain of next CA, in SCEP-PKCS7 format 
184                     if (log.isDebugEnabled()) {
185                         log.debug("Sending next certificate chain for CA '" + caname + "' to SCEP client.");
186                     }
187                     return signSession.createPKCS7Rollover(authenticationToken, cainfo.getCAId());
188                 } else {
189                     return null;
190                 }
191             }
192         } else if (operation.equals("GetCACaps")) {
193             final String caname = getCAName(message);
194             final CAInfo cainfo = caSession.getCAInfoInternal(-1, caname, true);
195             if (cainfo != null) {
196                 final boolean hasRolloverCert = (caSession.getFutureRolloverCertificate(cainfo.getCAId()) != null);
197                 // SCEP draft 23, "4.6.1.  Get Next CA Response Message Format". 
198                 // It SHOULD also remove the GetNextCACert setting from the capabilities until it does have rollover certificates.            
199                 return  hasRolloverCert ?
200                         "POSTPKIOperation\nGetNextCACert\nRenewal\nSHA-1".getBytes() :
201                         "POSTPKIOperation\nRenewal\nSHA-1".getBytes();
202             } else {
203                 final String msg = "CA was not found: "+caname;
204                 log.debug(msg);
205                 throw new CADoesntExistsException(msg);
206             }
207         } else {
208             log.error("Invalid parameter '" + operation);
209         }
210         return null;
211     }

213     /** Later SCEP draft say that for GetCACert message is optional. If message is there, it is the CA name
214      * but if message is not provided by the client, some default CA should be used.
215      * @param message the message part for the SCEP get request, can be null or empty string
216      * @return the message parameter or the default CA from ALIAS.defaultca property if message is null or empty.
217      */
218     private String getCAName(final String message) {
219         // If message is a string, return it, but if message is empty return default CA
220         if (StringUtils.isEmpty(message)) {
221             return EjbcaConfiguration.getScepDefaultCA();
222         }
223         return message;
224     }

226     /**
227      * Handles SCEP certificate request
228      * @param administrator Admin
229      *
230      * @param msg buffer holding the SCEP-request (DER encoded).
231      * @param alias the alias of the SCEP configuration
232      * @param scepConfig The SCEP configuration
233      *
234      * @return byte[] containing response to be sent to client.
235      * @throws AuthorizationDeniedException Fail
236      * @throws CertificateExtensionException if msg specified invalid extensions
237      * @throws InvalidAlgorithmException Fail
238      * @throws CAOfflineException Fail
239      * @throws IllegalValidityException Fail
240      * @throws CertificateSerialNumberException Fail 
241      * @throws CertificateRevokeException Fail
242      * @throws CertificateCreateException Fail
243      * @throws IllegalNameException Fail
244      * @throws AuthLoginException Fail
245      * @throws AuthStatusException Fail
246      * @throws SignRequestSignatureException Fail 
247      * @throws SignRequestException Fail
248      * @throws CADoesntExistsException Fail
249      * @throws IllegalKeyException Fail
250      * @throws CryptoTokenOfflineException Fail 
251      * @throws CustomCertificateSerialNumberException Fail
252      * @throws CertificateRenewalException if an error occurs during Client Certificate Renewal
253      * @throws SignatureException if a Client Certificate Renewal request was badly signed. 
254      * @throws CertificateException Fail
255      * @throws NoSuchEndEntityException if end entity wasn't found, and RA mode isn't available. 
256      */
257     private byte[] scepCertRequest(AuthenticationToken administrator, byte[] msg, final String alias, final ScepConfiguration scepConfig)
258             throws AuthorizationDeniedException, CertificateExtensionException, NoSuchEndEntityException, CustomCertificateSerialNumberException,
259             CryptoTokenOfflineException, IllegalKeyException, CADoesntExistsException, SignRequestException, SignRequestSignatureException,
260             AuthStatusException, AuthLoginException, IllegalNameException, CertificateCreateException, CertificateRevokeException,
261             CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException,
262             CertificateRenewalException, SignatureException, CertificateException {
263       byte[] ret = null;
264         if (log.isTraceEnabled()) {
265             log.trace(">getRequestMessage(" + msg.length + " bytes)");
266         }
267         
268         try {
269             boolean includeCACert = scepConfig.getIncludeCA(alias);
270             final ScepRequestMessage reqmsg = new ScepRequestMessage(msg, includeCACert);        
271             boolean isRAModeOK = scepConfig.getRAMode(alias);
272 
273             if (reqmsg.getErrorNo() != 0) {
274                 log.error("Error '" + reqmsg.getErrorNo() + "' receiving Scep request message.");
275                 return null;
276             }
277             if (reqmsg.getMessageType() == ScepRequestMessage.SCEP_TYPE_PKCSREQ) {
278                 if (isRAModeOK && scepRaModeExtension == null) {
279                     // Fail nicely
280                     log.warn("SCEP RA mode is enabled, but not included in the community version of EJBCA. Unable to continue.");
281                     return null;
282                 } else if (isRAModeOK) {
283                     if (log.isDebugEnabled()) {
284                         log.debug("SCEP is operating in RA mode: " + isRAModeOK);
285                     }
286                     if (!scepRaModeExtension.performOperation(administrator, reqmsg, scepConfig, alias)) {
287                         String errmsg = "Error. Failed to add or edit user: " + reqmsg.getUsername();
288                         log.error(errmsg);
289                         return null;
290                     }
291                 }
292                 if (scepClientCertificateRenewal != null && scepConfig.getClientCertificateRenewal(alias)) {
293                     if (log.isDebugEnabled()) {
294                         log.debug("SCEP client certificate renewal/enrollment with alias '"+alias+"'");
295                     }
296                     ResponseMessage resp = scepClientCertificateRenewal.performOperation(administrator, reqmsg, scepConfig, alias);
297                     if (resp != null) {
298                         ret = resp.getResponseMessage();
299                     }
300                 } else {                
301                     // Get the certificate 
302                     if (log.isDebugEnabled()) {
303                         log.debug("SCEP certificate enrollment with alias '"+alias+"'");
304                     }
305                     ResponseMessage resp = signSession.createCertificate(administrator, reqmsg, ScepResponseMessage.class, null);
306                     if (resp != null) {
307                         ret = resp.getResponseMessage();
308                     }
309                 }
310             }
311             if (reqmsg.getMessageType() == ScepRequestMessage.SCEP_TYPE_GETCRL) {
312                 // create the stupid encrypted CRL message, the below can actually only be made 
313                 // at the CA, since CAs private key is needed to decrypt
314                 ResponseMessage resp = signSession.getCRL(administrator, reqmsg, ScepResponseMessage.class);
315                 if (resp != null) {
316                     ret = resp.getResponseMessage();
317                 }
318             }
319         } catch (IOException e) {
320             log.error("Error receiving ScepMessage: ", e);
321         } 
322         if (log.isTraceEnabled()) {
323             log.trace("<getRequestMessage():" + ((ret == null) ? 0 : ret.length));
324         }
325         return ret;
326     }
