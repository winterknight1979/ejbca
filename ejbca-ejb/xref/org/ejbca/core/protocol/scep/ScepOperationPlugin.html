13  package org.ejbca.core.protocol.scep;
14  
15  import org.cesecore.authentication.tokens.AuthenticationToken;
16  import org.cesecore.authorization.AuthorizationDeniedException;
17  import org.ejbca.config.ScepConfiguration;
18  
19  /**
20   * Interface for SCEP plugins which require perform operations
21   * 
22   * @version $Id: ScepOperationPlugin.java 33852 2019-11-14 13:27:03Z jekaterina_b_helmes $
23   *
24   */
25  public interface ScepOperationPlugin {
26  
27      /**
28       * Performs an operation on this extension
29       * 
30       * @param authenticationToken an authentication token for any operations that may require one
31       * @param reqmsg the requestmessage to perform operations on
32       * @param scepConfig the SCEP configuration, if required
33       * @param alias alias of the SCEP configuration
34       * @return true if the operation succeeded 
35       * @throws AuthorizationDeniedException if request was denied due to authorization error
36       */
37      boolean performOperation(AuthenticationToken authenticationToken, ScepRequestMessage reqmsg, final ScepConfiguration scepConfig, final String alias) throws AuthorizationDeniedException;
38  }
