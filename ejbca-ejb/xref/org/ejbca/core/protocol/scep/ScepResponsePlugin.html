38  /**
39   * Plugin class for SCEP plugins which expect a ResponseMessage in reply
40   * 
41   * @version $Id: ScepResponsePlugin.java 28462 2018-03-09 13:14:58Z henriks $
42   *
43   */
44  
45  public interface ScepResponsePlugin {
46  
47      /**
48       * 
49       * TODO: Generalize exception handling for this interface method when the need arises.
50       * 
51       * @param authenticationToken an authentication token with access to the relevant CA, end entities and to create certificates
52       * @param reqmsg the SCEP request message
53       * @param scepConfig a copy of the SCEP configuration 
54       * @param alias the particular SCEP configuration alias to check out
55       * @return Response
56       * @throws NoSuchEndEntityException if the username specified in the request doesn't exist for any end entity
57       * @throws CADoesntExistsException if the CA specified in the request doesn't exist
58       * @throws AuthorizationDeniedException if the authentication token lacks access
59       * @throws CryptoTokenOfflineException if the CA's crypto token is unavailable
60       * @throws IllegalKeyException if reuse of old keys is prohibited in the configuration, and this request was found to be doing so
61       * @throws SignatureException if the PKCS#10 was not signed using the previous certificate
62       * @throws CustomCertificateSerialNumberException if a custom serial number was requested for the certificate, but it was invalid
63       * @throws SignRequestException if the request message lacks username and/or password
64       * @throws SignRequestSignatureException  if the PKCS#7 request was badly signed
65       * @throws AuthStatusException if the status of the end entity was not valid for this operation
66       * @throws AuthLoginException if the password in the request was incorrect
67       * @throws IllegalNameException if the requested username was invalid
68       * @throws CertificateCreateException if certificate creation failed
69       * @throws CertificateRevokeException if the certificate was menant to be issued revoked but could not be set so
70       * @throws CertificateSerialNumberException if a certificate with the same SubjectDN and key already exist, and a limitation is set that prohibits this
71       * @throws IllegalValidityException if the validity defined in the request wasn't valid
72       * @throws CAOfflineException if the CA was offline
73       * @throws InvalidAlgorithmException f the signing algorithm in the certificate profile (or the CA Token if not found) was invalid.
74       * @throws CertificateExtensionException if there was an error with the extensions specified in the request message
75       * @throws CertificateRenewalException if the last issued certificate hasn't passed half its validity date, or if it wasn't possible to change status for the end entity
76       */
77      public abstract ResponseMessage performOperation(AuthenticationToken authenticationToken, ScepRequestMessage reqmsg,
78              ScepConfiguration scepConfig, String alias) throws NoSuchEndEntityException, CADoesntExistsException, AuthorizationDeniedException,
79              CryptoTokenOfflineException, IllegalKeyException, SignatureException, CustomCertificateSerialNumberException, SignRequestException,
80              SignRequestSignatureException, AuthStatusException, AuthLoginException, IllegalNameException, CertificateCreateException,
81              CertificateRevokeException, CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException,
82              CertificateExtensionException, CertificateRenewalException;
83  
84  }
