43  /**
44   * Message handler for certificate request confirmation message.
45   * 
46   * According to RFC 4210 4.2.2.2:
47   *  "Where verification of the cert confirmation message fails, the RA/CA
48   *   MUST revoke the newly issued certificate if it has been published or
49   *   otherwise made available."
50   * 
51   * However, EJBCA does not keep track of the transaction and always responds
52   * with a ResponseStatus.SUCCESS Certificate Confirmation ACK.
53   * 
54   * @version $Id: ConfirmationMessageHandler.java 27740 2018-01-05 07:24:53Z mikekushner $
55   */
56  public class ConfirmationMessageHandler extends BaseCmpMessageHandler implements ICmpMessageHandler {
57  	
58  	private static final Logger LOG = Logger.getLogger(ConfirmationMessageHandler.class);
59  	
60  	/** Parameter used to determine the type of protection for the response message */
61  	private String responseProtection = null;
62      private CryptoTokenSessionLocal cryptoTokenSession;
63      
64      public ConfirmationMessageHandler(final AuthenticationToken authenticationToken, final CmpConfiguration cmpConfiguration, final String configAlias,
65      		final EjbBridgeSessionLocal ejbBridgeSession, final CryptoTokenSessionLocal cryptoTokenSession) {
66          super(authenticationToken, cmpConfiguration, configAlias, ejbBridgeSession);
67          this.responseProtection = this.cmpConfiguration.getResponseProtection(this.confAlias);
68          this.cryptoTokenSession = cryptoTokenSession;
69      }
70  	
71      public ResponseMessage handleMessage(BaseCmpMessage cmpRequestMessage, boolean authenticated) {
72  		if (LOG.isTraceEnabled()) {
73  			LOG.trace(">handleMessage");
74  		}
75  		int version = cmpRequestMessage.getHeader().getPvno().getValue().intValue();
76  		CmpConfirmResponseMessage cresp = null;
77  		// if version == 1 it is cmp1999 and we should not return a message back
78          if (version == PKIHeader.CMP_1999) {
79              if (LOG.isDebugEnabled()) {
80                  LOG.debug("Cmp1999 - Not creating a PKI confirm message response");
81              }
82          } else if (version > PKIHeader.CMP_1999) {
83  			// Creating the confirm message response
84  			if (LOG.isDebugEnabled()) {
85  				LOG.debug("Creating a PKI confirm message response, responseProtection="+responseProtection);
86  			}
87  			cresp = new CmpConfirmResponseMessage();
88  			cresp.setRecipientNonce(cmpRequestMessage.getSenderNonce());
89  			cresp.setSenderNonce(new String(Base64.encode(CmpMessageHelper.createSenderNonce())));
90  			cresp.setSender(cmpRequestMessage.getRecipient());
91  			cresp.setRecipient(cmpRequestMessage.getSender());
92  			cresp.setTransactionId(cmpRequestMessage.getTransactionId());
93  			if (StringUtils.equals(responseProtection, "pbe")) {
94  			    try {
95                      setPbeParameters(cresp, cmpRequestMessage, authenticated);
96                  } catch (InvalidCmpProtectionException e) {
97                      throw new IllegalArgumentException(e);
98                  }
99  			} else if (StringUtils.equals(responseProtection, "signature")) {
100 			    signResponse(cresp, cmpRequestMessage);
101 			}
102             try {
103                 cresp.create();
104             } catch (InvalidKeyException | NoSuchAlgorithmException | NoSuchProviderException e) {
105                 LOG.error("Exception during CMP processing: ", e);
106             }						
107 		} else {
108 			if (LOG.isDebugEnabled()) {
109 				LOG.debug("Not Cmp1999 or Cmp2000 or later - Not creating a PKI confirm message response");
110 			}
111 		}
112 		return cresp;
113 	}
114 	
115 	private void setPbeParameters(final BaseCmpMessage cmpResponseMessage, final BaseCmpMessage cmpRequestMessage, final boolean authenticated) throws InvalidCmpProtectionException {
116         final String keyId = CmpMessageHelper.getStringFromOctets(cmpRequestMessage.getHeader().getSenderKID());
117         String sharedSecret = cmpConfiguration.getAuthenticationParameter(CmpConfiguration.AUTHMODULE_HMAC, confAlias);
118         if(StringUtils.equals(sharedSecret, "-")) {
119             try {
120                 final X509CAInfo cainfo = getCAInfo(cmpRequestMessage.getRecipient().getName().toString());
121                 sharedSecret = cainfo.getCmpRaAuthSecret();
122             } catch (CADoesntExistsException e) {
123                 LOG.error("Exception during CMP response protection: ", e);
124             }
125         }
126         // We don't need to check the shared secret in client mode (= the EndEntiy password) because PBE protection is only supported in RA mode
127         CmpPbeVerifyer verifyer = new CmpPbeVerifyer(cmpRequestMessage.getMessage());
128         String owfAlg = verifyer.getOwfOid();
129         String macAlg = verifyer.getMacOid();
130         int iterationCount = verifyer.getIterationCount();
131         cmpResponseMessage.setPbeParameters(keyId, sharedSecret, owfAlg, macAlg, iterationCount);
132 	}

134 	private void signResponse(CmpConfirmResponseMessage cresp, BaseCmpMessage cmpRequestMessage) {
135         try {
136             // Get the CA that should sign the response
137             X509CAInfo caInfo = getCAInfo(cmpRequestMessage.getRecipient().getName().toString());
138             if (LOG.isDebugEnabled()) {
139                 LOG.debug("Using CA '" + caInfo.getName() + "' to sign Certificate Confirm message");
140             }
141             X509Certificate cacert = (X509Certificate) caInfo.getCertificateChain().iterator().next();
142             // We use the actual asn.1 encoding from the cacert subjectDN here. This ensures that the DN is exactly as 
143             // encoded in the certificate (which it should be).
144             // If we use only the cainfo.getSubjectDN we will get "EJBCA encoding", and this may not be the same if the 
145             // CA certificate comes from an external CA that encodes thing differently than EJBCA.
146             cresp.setSender(new GeneralName(X500Name.getInstance(cacert.getSubjectX500Principal().getEncoded())));
147             final CAToken caToken = caInfo.getCAToken();
148             final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(caToken.getCryptoTokenId());
149             cresp.setSignKeyInfo(caInfo.getCertificateChain(), cryptoToken.getPrivateKey(
150                     caToken.getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN)), 
151                     cryptoToken.getSignProviderName());
152             final AlgorithmIdentifier protectionAlgorithm = cmpRequestMessage.getHeader().getProtectionAlg();
153             if (protectionAlgorithm != null) {
154                 cresp.setPreferredDigestAlg(AlgorithmTools.getDigestFromSigAlg(protectionAlgorithm.getAlgorithm().getId()));
155             }
156         } catch (CADoesntExistsException | CryptoTokenOfflineException e) {
157             LOG.error("Exception during CMP response signing: " + e.getMessage(), e);            
158         }
159     }
160     
161     private X509CAInfo getCAInfo(final String caDn) throws CADoesntExistsException {
162         CAInfo caInfo = null;
163         if (caDn == null) {
164             final String caDnDefault = CertTools.stringToBCDNString(this.cmpConfiguration.getCMPDefaultCA(this.confAlias));
165             caInfo = caSession.getCAInfoInternal(caDnDefault.hashCode(), null, true);
166         } else {
167             final String caDnNormalized = CertTools.stringToBCDNString(caDn);
168             
169                 caInfo = caSession.getCAInfoInternal(caDnNormalized.hashCode(), null, true);
170             if(caInfo == null) {
171                 final String caDnDefault = CertTools.stringToBCDNString(this.cmpConfiguration.getCMPDefaultCA(this.confAlias));
172                 LOG.info("Could not find Recipient CA with DN '" + caDnNormalized + "'." +
173                         " Trying to use CMP DefaultCA instead with DN '" + caDnDefault + "' (" + caDnDefault.hashCode() + ").");
174                 caInfo = caSession.getCAInfoInternal(caDnDefault.hashCode(), null, true);
175             }
176         }
177         if (!(caInfo instanceof X509CAInfo)) {
178             throw new CADoesntExistsException("Incorrect CA type.");
179         }
180         return (X509CAInfo) caInfo;
181     }
