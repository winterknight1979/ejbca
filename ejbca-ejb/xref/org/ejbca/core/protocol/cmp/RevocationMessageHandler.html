102 	public ResponseMessage handleMessage(final BaseCmpMessage msg, boolean authenticated) {
103 		if (LOG.isTraceEnabled()) {
104 			LOG.trace(">handleMessage");
105 		}
106 		
107         // Try to find a HMAC/SHA1 protection key
108         final String keyId = CmpMessageHelper.getStringFromOctets(msg.getHeader().getSenderKID());
109         CA ca = null;
110         try {
111             ca = getRecipientCA(msg, keyId);
112         } catch (EndEntityProfileNotFoundException e) {
113             final String errMsg = (INTRES.getLocalizedMessage(CMP_ERRORGENERAL, e.getMessage()));
114             LOG.info(errMsg, e);
115             return CmpMessageHelper.createUnprotectedErrorMessage(msg, FailInfo.INCORRECT_DATA, errMsg);
116         } catch (CADoesntExistsException e) {
117             final String errMsg = "CA with DN '" + msg.getHeader().getRecipient().getName().toString() + "' is unknown";
118             LOG.info(errMsg);
119             return CmpMessageHelper.createUnprotectedErrorMessage(msg, FailInfo.BAD_REQUEST, errMsg);
120         } catch (AuthorizationDeniedException e) {
121             LOG.info(INTRES.getLocalizedMessage(CMP_ERRORGENERAL, e.getMessage()), e);
122             return CmpMessageHelper.createUnprotectedErrorMessage(msg, FailInfo.INCORRECT_DATA, e.getMessage());
123         }
124 		
125 		// if version == 1 it is cmp1999 and we should not return a message back
126 		ResponseStatus status = ResponseStatus.FAILURE;
127 		FailInfo failInfo = FailInfo.BAD_MESSAGE_CHECK;
128 		String failText = null;
129 
130 		//Verify the authenticity of the message
131 		final VerifyPKIMessageerifyPKIMessage.html#VerifyPKIMessage">VerifyPKIMessage messageVerifyer = new VerifyPKIMessage(ca.getCAInfo(), this.confAlias, admin, caSession, endEntityAccessSession, certificateStoreSession, 
132 		        authorizationSession, endEntityProfileSession, certificateProfileSession, authenticationProviderSession, endEntityManagementSession, this.cmpConfiguration);
133 		ICMPAuthenticationModule authenticationModule = messageVerifyer.getUsedAuthenticationModule(msg.getMessage(), null, authenticated);
134 		if(authenticationModule == null) {
135 	          LOG.info(messageVerifyer.getErrorMessage());
136 	          return CmpMessageHelper.createUnprotectedErrorMessage(msg, FailInfo.BAD_REQUEST, messageVerifyer.getErrorMessage());
137 		}
138 
139 		// If authentication was correct, we will now try to find the certificate to revoke
140 		final PKIMessage pkiMessage = msg.getMessage();
141 		final PKIBody pkiBody = pkiMessage.getBody();
142 		final RevReqContent rr = (RevReqContent) pkiBody.getContent();
143 		RevDetails rd;
144 		try {
145 		    rd = rr.toRevDetailsArray()[0];
146 		} catch(Exception e) {
147 		    LOG.debug("Could not parse the revocation request. Trying to parse it as novosec generated message.");
148 		    rd = CmpMessageHelper.getNovosecRevDetails(rr);
149 		    LOG.debug("Succeeded in parsing the novosec generated request.");
150 		}
151 		final CertTemplate ct = rd.getCertDetails();
152 		final ASN1Integer serno = ct.getSerialNumber();
153 		final X500Name issuer = ct.getIssuer();
154 		// Get the revocation reason if it exists, it is optional. 
155 		int reason = RevokedCertInfo.REVOCATION_REASON_UNSPECIFIED;
156         // CRL Entry Extensions is the only thing allowed in CMPv2
157 		final Extensions crlExt = rd.getCrlEntryDetails();
158 		if (crlExt != null) {
159 		    final Extension ext = crlExt.getExtension(Extension.reasonCode);
160 		    if (ext != null) {
161 		        try {
162 		            final ASN1InputStream ai = new ASN1InputStream(ext.getExtnValue().getOctets());
163 		            final ASN1Primitive obj = ai.readObject();
164 		            final ASN1Enumerated crlreason = ASN1Enumerated.getInstance(obj);
165 		            // RevokedCertInfo.REVOCATION_REASON_AACOMPROMISE are the same integer values as the CRL reason extension code
166 		            reason = crlreason.getValue().intValue();
167 		            if (LOG.isDebugEnabled()) {
168 		                LOG.debug("CRLReason extension: "+reason);
169 		            }
170 		            ai.close();
171 		        } catch (IOException e) {
172 		            LOG.info(INTRES.getLocalizedMessage(CMP_ERRORGENERAL, e.getMessage()), e);
173 		            return CmpMessageHelper.createUnprotectedErrorMessage(msg, FailInfo.INCORRECT_DATA, e.getMessage());
174 		        }
175 		    } else {
176 		        if (LOG.isDebugEnabled()) {
177 		            LOG.debug("No CRL reason code extension present.");
178 		        }
179 		    }
180 		} else {
181 		    if (LOG.isDebugEnabled()) {
182 		        LOG.debug("No CRL entry extensions present");
183 		    }
184 		}
185 		
186 		if (serno != null && issuer != null) {
187 		    final String iMsg = INTRES.getLocalizedMessage("cmp.receivedrevreq", CertTools.stringToBCDNString(issuer.toString()), serno.getValue().toString(16));
188 		    LOG.info(iMsg);
189 		    try {
190 		        endEntityManagementSession.revokeCert(admin, serno.getValue(), CertTools.stringToBCDNString(issuer.toString()), reason);
191 		        status = ResponseStatus.SUCCESS;
192 		    } catch (AuthorizationDeniedException e) {
193 		        failInfo = FailInfo.NOT_AUTHORIZED;
194 		        final String errMsg = INTRES.getLocalizedMessage("cmp.errornotauthrevoke", CertTools.stringToBCDNString(issuer.toString()), serno.getValue().toString(16));
195 		        failText = errMsg; 
196 		        LOG.info(failText);
197 		    } catch (NoSuchEndEntityException e) {
198 		        failInfo = FailInfo.BAD_CERTIFICATE_ID;
199 		        final String errMsg = INTRES.getLocalizedMessage("cmp.errorcertnofound", CertTools.stringToBCDNString(issuer.toString()), serno.getValue().toString(16));
200 		        failText = errMsg; 
201                 // This is already info logged in endEntityManagementSession.revokeCert
202                 // LOG.info(failText);
203 		    } catch (WaitingForApprovalException e) {
204 		        status = ResponseStatus.GRANTED_WITH_MODS;
205 		    } catch (ApprovalException e) {
206 		        failInfo = FailInfo.BAD_REQUEST;
207 		        final String errMsg = INTRES.getLocalizedMessage("cmp.erroralreadyrequested");
208 		        failText = errMsg; 
209 		        LOG.info(failText);
210 		    } catch (AlreadyRevokedException e) {
211 		        failInfo = FailInfo.CERT_REVOKED;
212 		        final String errMsg = INTRES.getLocalizedMessage("cmp.erroralreadyrevoked");
213 		        failText = errMsg; 
214 		        // This is already info logged in endEntityManagementSession.revokeCert
215 		        // LOG.info(failText);
216 		    }
217 		} else {
218 		    failInfo = FailInfo.BAD_CERTIFICATE_ID;
219             final String errMsg = INTRES.getLocalizedMessage("cmp.errormissingissuerrevoke",
220                     (issuer != null ? issuer.toString() : "<no issuer in request>"),
221                     (serno != null ? serno.getValue().toString(16) : "<no serial number in request>"));
222 		    failText = errMsg; 
223 		    LOG.info(failText);
224 		}
225 		
226 		if (LOG.isDebugEnabled()) {
227 		    LOG.debug("Creating a PKI revocation message response: "+responseProtection);
228 		}
229 		final CmpRevokeResponseMessagenseMessage.html#CmpRevokeResponseMessage">CmpRevokeResponseMessage rresp = new CmpRevokeResponseMessage();
230 		rresp.setRecipientNonce(msg.getSenderNonce());
231 		rresp.setSenderNonce(new String(Base64.encode(CmpMessageHelper.createSenderNonce())));
232 		// The revocation message may have had an empty recipient, in which case we got the recipient from the CMP configuration (see above)
233 		if (StringUtils.isEmpty(msg.getRecipient().getName().toString())) {
234 		    final X509Certificate cacert = (X509Certificate)ca.getCACertificate();
235 		    final GeneralName sender = new GeneralName(X500Name.getInstance(cacert.getSubjectX500Principal().getEncoded()));
236 		    rresp.setSender(sender);
237 		} else {
238 	        rresp.setSender(msg.getRecipient());		    
239 		}
240 		rresp.setRecipient(msg.getSender());
241 		rresp.setTransactionId(msg.getTransactionId());
242 		rresp.setFailInfo(failInfo);
243 		rresp.setFailText(failText);
244 		rresp.setStatus(status);
245 
246 		if (StringUtils.equals(responseProtection, "pbe")) {
247 		    final HMACAuthenticationModuleca/core/protocol/cmp/authentication/HMACAuthenticationModule.html#HMACAuthenticationModule">HMACAuthenticationModule hmacmodule = (HMACAuthenticationModule) authenticationModule;
248 		    final String owfAlg = hmacmodule.getCmpPbeVerifyer().getOwfOid();
249 		    final String macAlg = hmacmodule.getCmpPbeVerifyer().getMacOid();
250 		    final int iterationCount = 1024;
251 		    final String cmpRaAuthSecret = hmacmodule.getAuthenticationString();
252 		    
253 		    if (owfAlg != null && macAlg != null && cmpRaAuthSecret != null) {
254 		        // Set all protection parameters
255 		        if (LOG.isDebugEnabled()) {
256 		            LOG.debug(responseProtection+", "+owfAlg+", "+macAlg+", "+keyId+", "+cmpRaAuthSecret);
257 		        }
258 		        rresp.setPbeParameters(keyId, cmpRaAuthSecret, owfAlg, macAlg, iterationCount);
259 		    }
260 		} else if(StringUtils.equals(responseProtection, "signature")) {
261 		    try {
262 		        final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(ca.getCAToken().getCryptoTokenId());
263 		        final String aliasCertSign = ca.getCAToken().getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN);
264 		        rresp.setSignKeyInfo(ca.getCertificateChain(), cryptoToken.getPrivateKey(aliasCertSign), cryptoToken.getSignProviderName());
265                 if(msg.getHeader().getProtectionAlg() != null) {
266                     rresp.setPreferredDigestAlg(AlgorithmTools.getDigestFromSigAlg(msg.getHeader().getProtectionAlg().getAlgorithm().getId()));
267                 }
268 		    } catch (CryptoTokenOfflineException e) {
269 		        LOG.error(e.getLocalizedMessage(), e);
270 		    }
271 		}
272 		try {
273 		    rresp.create();
274 		} catch (InvalidKeyException | NoSuchAlgorithmException | NoSuchProviderException e) {
275 		    LOG.error(INTRES.getLocalizedMessage("cmp.errorgeneral"), e);
276         }
277 		return rresp;
278 	}

280     private CA getRecipientCA(final BaseCmpMessage msg, final String keyId)
281             throws EndEntityProfileNotFoundException, CADoesntExistsException, AuthorizationDeniedException {
282         CA ca;
283         final String caDN = msg.getRecipient().getName().toString();
284         int caId = 0;
285         if (StringUtils.isEmpty(caDN)) {
286             LOG.debug("Empty DN in header.recipient, get CA from CMP configuration");
287             int eeProfileId = getUsedEndEntityProfileId(keyId);
288             caId = getUsedCaId(keyId, eeProfileId);
289             if (LOG.isDebugEnabled()) {
290                 LOG.debug("caId from CMP Profile is "+caId+", with eeProfileId "+eeProfileId);
291             }
292         } else {
293             if (LOG.isDebugEnabled()) {
294                 LOG.debug("Get recipient CA from recipient: "+caDN);
295             }
296             caId = CertTools.stringToBCDNString(caDN).hashCode();
297             if (LOG.isDebugEnabled()) {
298                 LOG.debug("CA DN is '"+caDN+"' and resulting caId is "+caId+", after CertTools.stringToBCDNString conversion.");
299             }
300         }
301         ca = caSession.getCA(admin, caId);
302         if(ca == null) {
303             throw new CADoesntExistsException("CA with ID " + caId + " does not exist.");
304         }
305         return ca;
306     }
