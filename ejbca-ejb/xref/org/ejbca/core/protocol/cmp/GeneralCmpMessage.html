31  /**
32   * Message class for CMP PKI confirm and CertCOnf messages
33   * 
34   * @version $Id: GeneralCmpMessage.java 29745 2018-08-25 21:35:37Z anatom $
35   */
36  public class GeneralCmpMessage extends BaseCmpMessage {
37  
38  	private static final Logger log = Logger.getLogger(GeneralCmpMessage .class);
39      /** Internal localization of logs and errors */
40      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
41  	
42      /**
43       * Determines if a de-serialized file is compatible with this class.
44       *
45       * Maintainers must change this value if and only if the new version
46       * of this class is not compatible with old versions. See Sun docs
47       * for <a href=http://java.sun.com/products/jdk/1.1/docs/guide
48       * /serialization/spec/version.doc.html> details. </a>
49       *
50       */
51      static final long serialVersionUID = 1000L;
52  
53  	public GeneralCmpMessage(final PKIMessage pkiMessage) {
54  		final PKIBody pkiBody = pkiMessage.getBody();
55  		final int tag = pkiBody.getType();
56  		if (tag == PKIBody.TYPE_CONFIRM) {
57  			if (log.isDebugEnabled()) {
58  				log.debug("Received a PKIConfirm message");
59  			}
60  			// This is a null message, so there is nothing to get here
61  			//DERNull obj = body.getConf();
62  		} else if (tag == PKIBody.TYPE_CERT_CONFIRM) {
63  			if (log.isDebugEnabled()) {
64  				log.debug("Received a Cert Confirm message");
65  			}
66  			final CertConfirmContent obj = (CertConfirmContent) pkiBody.getContent();
67  			CertStatus cs;
68  			try {
69  			    cs = CertStatus.getInstance(obj.toASN1Primitive());
70  			} catch(Exception e) {
71  			    cs = CertStatus.getInstance(((DERSequence) obj.toASN1Primitive()).getObjectAt(0));
72  			}
73  			final PKIStatusInfo status = cs.getStatusInfo();
74  			if (status != null) {
75  				final int st = status.getStatus().intValue();
76  				if (st != 0) {
77  					final String errMsg = intres.getLocalizedMessage("cmp.errorcertconfirmstatus", Integer.valueOf(st));
78  					log.error(errMsg);
79  					// TODO: if it is rejected, we should revoke the cert?
80  				}
81  			}
82  		} else if (tag == PKIBody.TYPE_REVOCATION_REQ) {
83  			// this is a RevReqContent,
84  			if (log.isDebugEnabled()) {
85  				log.debug("Received a RevReqContent");
86  			}
87  			final RevReqContent rr = (RevReqContent) pkiBody.getContent();
88  			RevDetails rd;
89  			try {
90  			    rd = rr.toRevDetailsArray()[0];
91  			} catch(Exception e) {
92  			    log.debug("Could not parse the revocation request. Trying to parse it as novosec generated message.");
93  			    rd = CmpMessageHelper.getNovosecRevDetails(rr);
94  			    log.debug("Succeeded in parsing the novosec generated request.");
95  			}
96  			final CertTemplate ct = rd.getCertDetails();
97  			final ASN1Integer serno = ct.getSerialNumber();
98  			final X500Name issuer = ct.getIssuer();
99  			if (serno != null && issuer != null) {
100 				final String errMsg = intres.getLocalizedMessage("cmp.receivedrevreq", CertTools.stringToBCDNString(issuer.toString()), serno.getValue().toString(16));
101 				log.info(errMsg);
102 			} else {
103 				final String errMsg = intres.getLocalizedMessage("cmp.receivedrevreqnoissuer");
104 				log.info(errMsg);
105 			}
106 		}
107 		setMessage(pkiMessage);
108 		final PKIHeader pkiHeader = pkiMessage.getHeader();
109         setTransactionId(getBase64FromAsn1OctetString(pkiHeader.getTransactionID()));
110         setSenderNonce(getBase64FromAsn1OctetString(pkiHeader.getSenderNonce()));
111 		setRecipient(pkiHeader.getRecipient());
112 		setSender(pkiHeader.getSender());
113 	}
114 }
