50  /**
51   * Class that receives a CMP message and passes it on to the correct message handler.
52   *
53   * -----
54   * This processes does the following:
55   * 1. receive a CMP message
56   * 2. check which message type it is
57   * 3. dispatch to the correct message handler
58   * 4. send back the response received from the handler
59   * -----
60   *
61   * Messages supported (@see <a href="https://tools.ietf.org/html/rfc4210#section-5.3">RFC4210 5.3 Operation-Specific Data Structures</a>):
62   * 
63   * Implemented:
64   * - 5.3.1 Initialization Request - will return an Initialization Response (-&gt; CertRepMessage).
65   * - 5.3.9 Revocation Request / Response (-&gt; RevRepContent)
66   * - 5.3.17 PKI Confirmation - same as certificate confirmation accept - will return a PKI Confirmation Content (-&gt; PKIConfirmContent)
67   * - 5.3.18 Certificate Confirmation - accept or reject by client - will return a PKI Confirmation Content (-&gt; PKIConfirmContent)
68   * - 5.3.3 Certificate Request / Response (-&gt; CertRepMessage)
69   * - 5.3.5 Key Update Request / Response (-&gt; CertRepMessage)
70   * 
71   *  Responses of type 'CertRepMessage' may contain additional CA certificates in its 'caPubs' field 
72   *  which can be configured in the CMP configuration ({@link CmpConfiguration#getResponseCaPubsCA(String)}.
73   *  
74   * @version $Id: CmpMessageDispatcherSessionBean.java 28471 2018-03-12 15:11:18Z henriks $
75   */
76  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "CmpMessageDispatcherSessionRemote")
77  public class CmpMessageDispatcherSessionBean implements CmpMessageDispatcherSessionLocal, CmpMessageDispatcherSessionRemote {
78

94      @Override
95      @TransactionAttribute(TransactionAttributeType.REQUIRED)
96      public byte[] dispatchRequest(final AuthenticationToken authenticationToken, final byte[] pkiMessageBytes, final String cmpConfigurationAlias)
97              throws NoSuchAliasException {
98          try {
99              final CmpConfiguration cmpConfiguration = (CmpConfiguration) this.globalConfigSession.getCachedConfiguration(CmpConfiguration.CMP_CONFIGURATION_ID);
100             if (!cmpConfiguration.aliasExists(cmpConfigurationAlias)) {
101                 final String msg = intres.getLocalizedMessage("protocol.nosuchalias", "CMP", cmpConfigurationAlias);
102                 log.info(msg);
103                 throw new NoSuchAliasException(msg);
104             }
105             final PKIMessage pkiMessage = CmpMessageHelper.getPkiMessageFromBytes(pkiMessageBytes, false);
106             if (pkiMessage == null) {
107                 // Log that we handled a bad request and respond to the client
108                 final String msg = intres.getLocalizedMessage("cmp.errornotcmpmessage");
109                 log.info(msg);
110                 return CmpMessageHelper.createUnprotectedErrorMessage(msg);
111             }
112             final ResponseMessage responseMessage = dispatch(authenticationToken, pkiMessage, pkiMessage.getHeader(), cmpConfiguration, cmpConfigurationAlias, /*levelOfNesting=*/0);
113             return responseMessage == null ? null : responseMessage.getResponseMessage();
114         } catch (CertificateEncodingException e) {
115             log.warn("Could not retrieve byte representation of from org.cesecore.certificates.certificate.request.ResponseMessage", e);
116             return null;
117         }
118     }

120     /**
121      * The message may have been received by any transport protocol, and is passed here in it's binary ASN.1 form.
122      *
123      * @param authenticationToken the authentication token.
124      * @param pkiMessage DER encoded CMP message received from the client.
125      * @param pkiHeader DER encoded PKI header of the original CMP message received from the client.
126      * @param cmpConfiguration Config
127      * @param cmpConfigurationAlias the CMP alias we want to use for this request.
128     *  @param levelOfNesting the level of nesting depth.
129      * @return A response message containing the CMP response message or null if there is no message to send back or some internal error has occurred
130      */
131     private ResponseMessage dispatch(final AuthenticationToken authenticationToken, final PKIMessage pkiMessage, final PKIHeader pkiHeader,
132             final CmpConfiguration cmpConfiguration, String cmpConfigurationAlias, final int levelOfNesting) {
133         if (levelOfNesting > CmpMessageHelper.MAX_LEVEL_OF_NESTING) {
134             return CmpMessageHelper.createUnprotectedErrorMessage(pkiHeader, FailInfo.BAD_REQUEST, "Rejected request due to unreasonable level of nesting.");
135         }
136         final boolean authenticated = levelOfNesting > 0;
137         try {
138             final PKIBody pkiBody = pkiMessage.getBody();
139             final int tagno = pkiBody.getType();
140             if (log.isDebugEnabled()) {
141                 final String message = "Received CMP message with pvno=" + pkiHeader.getPvno() + ", sender=" + pkiHeader.getSender().toString() +
142                         ", recipient=" + pkiHeader.getRecipient().toString() + System.lineSeparator() +
143                         "Cmp configuration alias: " + cmpConfigurationAlias + System.lineSeparator() +
144                         "The CMP message is already authenticated: " + authenticated + System.lineSeparator() +
145                         "Body is of type: " + tagno + System.lineSeparator() +
146                         "Transaction ID: " + pkiHeader.getTransactionID();
147                 log.debug(message);
148                 if (log.isTraceEnabled()) {
149                     log.trace(ASN1Dump.dumpAsString(pkiMessage));
150                 }
151             }
152             log.info("Dispatching message with transaction ID: " + pkiHeader.getTransactionID());
153             BaseCmpMessage cmpMessage = null;
154             ICmpMessageHandler handler = null;
155             int unknownMessageType = -1;
156             switch (tagno) {
157             case PKIBody.TYPE_INIT_REQ:
158                 // 0: ir, Initialization Request and 2 (cr, Certification Req) are both certificate requests
159             case PKIBody.TYPE_CERT_REQ:
160                 // 2:
161                 handler = new CrmfMessageHandler(authenticationToken, cmpConfiguration, cmpConfigurationAlias, ejbBridgeSession,
162                         certificateRequestSession);
163                 cmpMessage = new CrmfRequestMessage(pkiMessage, cmpConfiguration.getCMPDefaultCA(cmpConfigurationAlias),
164                         cmpConfiguration.getAllowRAVerifyPOPO(cmpConfigurationAlias),
165                         cmpConfiguration.getExtractUsernameComponent(cmpConfigurationAlias));
166                 break;
167             case PKIBody.TYPE_KEY_UPDATE_REQ:                    
168                 // 7: Key Update request (kur, Key Update Request)
169                 handler = new CrmfKeyUpdateHandler(authenticationToken, cmpConfiguration, cmpConfigurationAlias, ejbBridgeSession);
170                 cmpMessage = new CrmfRequestMessage(pkiMessage, cmpConfiguration.getCMPDefaultCA(cmpConfigurationAlias),
171                         cmpConfiguration.getAllowRAVerifyPOPO(cmpConfigurationAlias),
172                         cmpConfiguration.getExtractUsernameComponent(cmpConfigurationAlias));
173                 break;
174             case PKIBody.TYPE_CONFIRM:
175                 // 19: PKI confirm (pkiconf, Confirmation)
176             case PKIBody.TYPE_CERT_CONFIRM:
177                 // 24: Certificate confirmation (certConf, Certificate confirm)
178                 handler = new ConfirmationMessageHandler(authenticationToken, cmpConfiguration, cmpConfigurationAlias, ejbBridgeSession,
179                         cryptoTokenSession);
180                 cmpMessage = new GeneralCmpMessage(pkiMessage);
181                 break;
182             case PKIBody.TYPE_REVOCATION_REQ:
183                 // 11: Revocation request (rr, Revocation Request)
184                 handler = new RevocationMessageHandler(authenticationToken, cmpConfiguration, cmpConfigurationAlias, ejbBridgeSession,
185                         cryptoTokenSession);
186                 cmpMessage = new GeneralCmpMessage(pkiMessage);
187                 break;
188             case PKIBody.TYPE_NESTED:
189                 // 20: NestedMessageContent (nested)
190                 if (log.isDebugEnabled()) {
191                     log.debug("Received a NestedMessageContent");
192                 }
193                 final NestedMessageContentnt.html#NestedMessageContent">NestedMessageContent nestedMessage = new NestedMessageContent(pkiMessage, cmpConfiguration, cmpConfigurationAlias);
194                 if (nestedMessage.verify()) {
195                     if (log.isDebugEnabled()) {
196                         log.debug("The NestedMessageContent was verified successfully");
197                     }
198                     try {
199                         final PKIMessages nestedPkiMessages = PKIMessages.getInstance(pkiMessage.getBody().getContent());
200                         final PKIMessage nestedPkiMessage = nestedPkiMessages.toPKIMessageArray()[0];
201                         return dispatch(authenticationToken, nestedPkiMessage, pkiHeader, cmpConfiguration, cmpConfigurationAlias, levelOfNesting+1);
202                     } catch (IllegalArgumentException e) {
203                         final String errMsg = e.getMessage();
204                         log.info(errMsg, e);
205                         return CmpMessageHelper.createUnprotectedErrorMessage(pkiHeader, FailInfo.BAD_REQUEST, errMsg);
206                     }
207                 }
208                 final String errMsg = "Could not verify the RA, signature verification on NestedMessageContent failed.";
209                 log.info(errMsg);
210                 return CmpMessageHelper.createUnprotectedErrorMessage(pkiHeader, FailInfo.BAD_REQUEST, errMsg);
211             default:
212                 unknownMessageType = tagno;
213                 log.info("Received an unknown message type, tagno=" + tagno);
214                 break;
215             }
216             
217             addAdditionalResponseCaPubsCertificates(authenticationToken, cmpConfiguration, cmpConfigurationAlias, cmpMessage);
218             addAdditionalResponseExtraCertsCertificates(authenticationToken, cmpConfiguration, cmpConfigurationAlias, cmpMessage);
219             
220             if (handler == null || cmpMessage == null) {
221                 if (unknownMessageType > -1) {
222                     final String eMsg = intres.getLocalizedMessage("cmp.errortypenohandle", Integer.valueOf(unknownMessageType));
223                     log.error(eMsg);
224                     return CmpMessageHelper.createUnprotectedErrorMessage(pkiHeader, FailInfo.BAD_REQUEST, eMsg);
225                 }
226                 throw new IllegalStateException("Something is null! Handler=" + handler + ", cmpMessage=" + cmpMessage);
227             }
228             final ResponseMessage ret = handler.handleMessage(cmpMessage, authenticated);
229             if (ret == null) {
230                 log.error(intres.getLocalizedMessage("cmp.errorresponsenull"));
231             } else {
232                 if (log.isDebugEnabled()) {
233                     log.debug("Received a response message of type '" + ret.getClass().getName() + "' from CmpMessageHandler.");
234                 }
235             }
236             return ret;
237         } catch (RuntimeException e) {
238             log.error(intres.getLocalizedMessage("cmp.errorprocess"), e);
239             return null;
240         }
241     }

243     /**
244      * Adds the list of additional CA certificates to the user certificates signing CA certificate to be 
245      * returned with the CMP response 'CertRepMessage.caPubs' field.
246      * 
247      * @param admin the authentication token.
248      * @param cmpConfiguration the CMP configuration list.
249      * @param alias the CMP configuration alias.
250      * @param message the request message.
251      */
252     private void addAdditionalResponseCaPubsCertificates(final AuthenticationToken admin, final CmpConfiguration cmpConfiguration, final String alias, 
253             final BaseCmpMessage message) {
254         if (message != null) {
255             final String casToAdd = cmpConfiguration.getResponseCaPubsCA(alias);
256             if (log.isDebugEnabled()) {
257                 log.debug("Add CA certificates of CAs '" + casToAdd + "' to the CMP response message caPubs field.");
258             }
259             message.setAdditionalCaCertificates(getCaCertificates(admin, casToAdd));
260         }
261     }
262     
263     /**
264      * Adds the list of additional CA certificates to the message signing CA certificate returned
265      * with the outer PKI response message 'PKIMessage.extraCerts' field.
266      * 
267      * @param admin the authentication token.
268      * @param cmpConfiguration the CMP configuration list.
269      * @param alias the CMP configuration alias.
270      * @param message the request message.
271      */
272     private void addAdditionalResponseExtraCertsCertificates(final AuthenticationToken admin, final CmpConfiguration cmpConfiguration, final String alias, 
273             final BaseCmpMessage message) {
274         if (message != null) {
275             final String casToAdd = cmpConfiguration.getResponseExtraCertsCA(alias);
276             if (log.isDebugEnabled()) {
277                 log.debug("Add CA certificates of CAs '" + casToAdd + "' to the PKI response message extraCerts field.");
278             }
279             message.setAdditionalExtraCertsCertificates(getCaCertificates(admin, casToAdd));
280         }
281     }

283     /**
284      * Gets the CA certificates by the semicolon separated string of CA names.
285      * 
286      * @param admin the authentication token.
287      * @param caListString the semicolon separated string of CA IDs
288      * @return the list of CA certificates in the order, the CA IDs were given.
289      */
290     private List<Certificate> getCaCertificates(final AuthenticationToken admin, final String caListString) {
291         final List<Certificate> result = new ArrayList<Certificate>();
292         CAInfo cainfo = null;
293         Certificate cacert;
294         if (StringUtils.isNotBlank(caListString)) {
295             int caId;
296             for(String ca : StringUtils.split(caListString, ";")) {
297                 caId = Integer.parseInt(ca);
298                 if(log.isDebugEnabled()) {
299                     log.debug("Get CA by ID: " + caId);
300                 }
301                 try {
302                     cainfo = caSession.getCAInfo(admin, caId);
303                     if (cainfo != null && CollectionUtils.isNotEmpty(cainfo.getCertificateChain())) {
304                         cacert = (X509Certificate) cainfo.getCertificateChain().get(0);
305                         if (!result.contains(cacert)) {
306                             result.add((X509Certificate) cacert);
307                         }
308                     } else { // Should never happen.
309                         log.info("Cannot find CA: " + ca); 
310                     }
311                 } catch (AuthorizationDeniedException e) {
312                     if(log.isDebugEnabled()) {
313                         log.debug(e.getMessage());
314                     }
315                 }
316             }
317         }
318         return result;
319     }
