24  import org.ejbca.config.CmpConfiguration;
25  import org.ejbca.core.ejb.authentication.web.WebAuthenticationProviderSessionLocal;
26  import org.ejbca.core.ejb.ra.EndEntityAccessSession;

30  /**
31   * Verifies a CMP message using a suitable authentication module.
32   * 
33   * The authentication modules are specified as properties in the CmpConfiguration.
34   * 
35   * @version $Id: VerifyPKIMessage.java 25797 2017-05-04 15:52:00Z jeklund $
36   */
37  public class VerifyPKIMessage {
38      
39      private static final Logger log = Logger.getLogger(VerifyPKIMessage.class);
40      
41      private final CAInfo caInfo;
42      private String errorMessage = null;
43      private final String confAlias;
44      private final CmpConfiguration cmpConfiguration;
45      private final AuthenticationToken authenticationToken;
46      
47      private final CaSession caSession;
48      private final EndEntityAccessSession endEntityAccessSession;
49      private final CertificateStoreSession certificateStoreSession;
50      private final AuthorizationSession authorizationSession;
51      private final EndEntityProfileSession endEntityProfileSession;
52      private final CertificateProfileSession certificateProfileSession;
53      private final WebAuthenticationProviderSessionLocal authenticationProviderSession;
54      private final EndEntityManagementSession endEntityManagementSession;
55  
56      public VerifyPKIMessage(final CAInfo cainfo, final String confAlias, final AuthenticationToken admin, final CaSession caSession, final EndEntityAccessSession endEntityAccessSession, 
57              final CertificateStoreSession certificateStoreSession, final AuthorizationSession authorizationSession, final EndEntityProfileSession endEntityProfileSession, final CertificateProfileSession certificateProfileSession,  
58              final WebAuthenticationProviderSessionLocal authenticationProviderSession, final EndEntityManagementSession endEntityManagementSession, 
59              final CmpConfiguration cmpConfiguration) {
60          this.caInfo = cainfo;
61          this.confAlias = confAlias;
62          this.authenticationToken = admin;
63          this.caSession = caSession;
64          this.endEntityAccessSession = endEntityAccessSession;
65          this.certificateStoreSession = certificateStoreSession;
66          this.authorizationSession = authorizationSession;
67          this.endEntityProfileSession = endEntityProfileSession;
68          this.certificateProfileSession = certificateProfileSession;
69          this.authenticationProviderSession = authenticationProviderSession;
70          this.endEntityManagementSession = endEntityManagementSession;
71          this.cmpConfiguration = cmpConfiguration;
72      }
73      
74      /**
75       * Returns the error message resulted in failing to verify the PKIMessage. The error message  is set in the 
76       * getUsedAuthenticationModule() method.
77       * 
78       * @return the error message as String. Null if the verification succeeded.
79       */
80      public String getErrorMessage() {
81          return this.errorMessage;
82      }
83      
84      /**
85       * Verifies the authenticity of the PKIMessage
86       * 
87       * @param pkiMessage PKIMessage to verify
88       * @param username that the PKIMessage should match or null
89       * @param authenticated if the CMP message has already been authenticated in another way or not
90       * @return The authentication module that succeeded in authenticating msg. Null if message authentication failed using all 
91       * configured authentication modules.
92       */
93      public ICMPAuthenticationModule getUsedAuthenticationModule(final PKIMessage pkiMessage, final String username, boolean authenticated) {
94          final String authModules = this.cmpConfiguration.getAuthenticationModule(this.confAlias);
95          final String authparameters = this.cmpConfiguration.getAuthenticationParameters(this.confAlias);
96          final String modules[] = authModules.split(";");
97          final String params[] = authparameters.split(";");
98          if (modules.length != params.length) {
99              log.error("The number of authentication modules does not match the number of authentication parameters. " +
100                     modules.length + " modules - " + params.length + " paramters");
101             this.errorMessage = "CMP module configuration error.";
102             return null;
103         }
104         boolean raMode = this.cmpConfiguration.getRAMode(this.confAlias);
105         for (int i=0; i<modules.length; i++) {
106             final String moduleName = modules[i].trim();
107             final String moduleParameter = params[i].trim();
108             if (log.isDebugEnabled()) {
109                 log.debug("Trying to verify the message using CMP authentication module '" + moduleName + "' with parameter '" + moduleParameter + "'");
110             }
111             final ICMPAuthenticationModule module = getAuthModule(raMode, moduleName, moduleParameter, pkiMessage, authenticated);
112             if (module != null) {
113                 if (module.verifyOrExtract(pkiMessage, username)) {
114                     log.info("PKIMessage was successfully authenticated using " + module.getName());
115                     return module;
116                 } else {
117                     if (module.getErrorMessage() != null) {
118                         errorMessage = module.getErrorMessage();
119                     }
120                 }
121             }
122         }
123         if (this.errorMessage == null) {
124             this.errorMessage = "Failed to authentication PKIMessage using authentication modules: " + authModules;
125         }
126         return null;
127     }
128     
129     /** @param raMode Mode
130      * @param module Module
131      * @param parameter Param
132      * @param pkiMessage PKI
133      * @param authenticated Auth
134      * @return The requested authentication module or null if no such module is implemented. */
135     private ICMPAuthenticationModule getAuthModule(final boolean raMode, final String module, final String parameter, final PKIMessage pkiMessage, final boolean authenticated) {
136         switch (module) {
137         case CmpConfiguration.AUTHMODULE_HMAC:
138             return new HMACAuthenticationModule(authenticationToken, parameter, confAlias, cmpConfiguration, caInfo, endEntityAccessSession);
139         case CmpConfiguration.AUTHMODULE_ENDENTITY_CERTIFICATE:
140             return new EndEntityCertificateAuthenticationModule(authenticationToken, parameter, confAlias, cmpConfiguration, authenticated,
141                     caSession, certificateStoreSession, authorizationSession, endEntityProfileSession, certificateProfileSession,
142                     endEntityAccessSession, authenticationProviderSession, endEntityManagementSession);
143         case CmpConfiguration.AUTHMODULE_REG_TOKEN_PWD:
144             if (raMode) {
145                 this.errorMessage = "The authentication module '" + module + "' cannot be used in RA mode";
146                 break;
147             }
148             return new RegTokenPasswordExtractor();
149         case CmpConfiguration.AUTHMODULE_DN_PART_PWD:
150             if (raMode) {
151                 this.errorMessage = "The authentication module '" + module + "' cannot be used in RA mode";
152                 break;
153             }
154             return new DnPartPasswordExtractor(parameter);
155         default:
156             this.errorMessage = "Unrecognized authentication module: " + module;
157         }
158         log.info(this.errorMessage);
159         return null;
160     }
