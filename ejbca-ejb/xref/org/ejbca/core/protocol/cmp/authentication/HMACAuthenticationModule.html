43  /**
44   * Checks the authentication of the PKIMessage.
45   * 
46   * In RA mode, the authenticity is checked through a shared secret specified either in 
47   * the configuration file or in the CA.
48   * 
49   * In client mode, the authenticity is checked through the clear-text-password of the 
50   * pre-registered end entity from the database. 
51   * 
52   * @version $Id: HMACAuthenticationModule.java 28319 2018-02-19 09:14:59Z andresjakobs $
53   */
54  public class HMACAuthenticationModule implements ICMPAuthenticationModule {
55  
56      private static final Logger LOG = Logger.getLogger(HMACAuthenticationModule.class);
57      private static final InternalEjbcaResources INTRES = InternalEjbcaResources.getInstance();
58  
59      private final AuthenticationToken authenticationToken;
60      private final EndEntityAccessSession endEntityAccessSession;
61      
62      private final String globalSharedSecret;
63      private final CAInfo caInfo;
64      private final String confAlias;
65      private final CmpConfiguration cmpConfiguration;
66      
67      private String password = null;
68      private String errorMessage = null;
69      private CmpPbeVerifyer verifyer = null;
70          
71      public HMACAuthenticationModule(AuthenticationToken authenticationToken, String globalSharedSecret, String confAlias, CmpConfiguration cmpConfiguration, 
72              CAInfo caInfo, EndEntityAccessSession endEntityAccessSession) {
73          this.globalSharedSecret = "-".equals(globalSharedSecret) ? null : globalSharedSecret;
74          this.confAlias = confAlias;
75          this.caInfo = caInfo;
76          this.cmpConfiguration = cmpConfiguration;
77          this.authenticationToken = authenticationToken;
78          this.endEntityAccessSession = endEntityAccessSession;
79      }
80      
81      @Override
82      public String getName() {
83          return CmpConfiguration.AUTHMODULE_HMAC;
84      }
85      
86      @Override
87      public String getAuthenticationString() {
88          return this.password;
89      }
90      
91      @Override
92      public String getErrorMessage() {
93          return this.errorMessage;
94      }

96      public CmpPbeVerifyer getCmpPbeVerifyer() {
97          return this.verifyer;
98      }

100     /*
101      * Verifies that 'pkiMessage' is sent by a trusted source. 
102      * 
103      * In RA mode:
104      *      - A globally configured shared secret for all CAs will be used to authenticate the message.
105      *      - If the globally shared secret fails, the password set in the CA will be used to authenticate the message.
106      *  In client mode, the clear-text password set in the pre-registered end entity in the database will be used to 
107      *  authenticate the message. 
108      * 
109      * When successful, the authentication string will be set to the password that was successfully used in authenticating the message.
110      */
111     @Override
112     public boolean verifyOrExtract(final PKIMessage pkiMessage, final String username) {
113         if (pkiMessage == null) {
114             this.errorMessage = "No PKIMessage was found";
115             return false;
116         }
117         if (pkiMessage.getProtection() == null || pkiMessage.getHeader().getProtectionAlg() == null) {
118             this.errorMessage = "PKI Message is not authenticated properly. No HMAC protection was found.";
119             return false;
120         }
121         try {
122             verifyer = new CmpPbeVerifyer(pkiMessage);
123         } catch (InvalidCmpProtectionException e) {
124             this.errorMessage = e.getMessage();
125             return false;
126         }
127         try {
128             if (this.cmpConfiguration.getRAMode(this.confAlias)) {
129                 if (LOG.isDebugEnabled()) {
130                     LOG.debug("Verifying HMAC in RA mode");
131                 }
132                 // Check that the value of KeyId from the request is allowed 
133                 // Note that this restriction only applies to HMAC and not EndEntityCertificate because in the latter, the use of profiles can be restricted through 
134                 // Administrator privileges. Other authentication modules are not used in RA mode
135                 final boolean useKeyIdForEndEntityProfile = StringUtils.equals(cmpConfiguration.getRAEEProfile(confAlias), CmpConfiguration.PROFILE_USE_KEYID);
136                 final boolean useKeyIdForCertificateProfile = StringUtils.equals(cmpConfiguration.getRACertProfile(confAlias), CmpConfiguration.PROFILE_USE_KEYID);
137                 if (useKeyIdForEndEntityProfile || useKeyIdForCertificateProfile) {
138                     final String keyId = CmpMessageHelper.getStringFromOctets(pkiMessage.getHeader().getSenderKID());
139                     if ((useKeyIdForEndEntityProfile && StringUtils.equals(keyId, EndEntityConstants.EMPTY_ENDENTITYPROFILENAME)) ||
140                             (useKeyIdForCertificateProfile && StringUtils.equals(keyId, CertificateProfile.ENDUSERPROFILENAME))) {
141                         errorMessage = "Unaccepted KeyId '" + keyId + "' in CMP request";
142                         LOG.info(errorMessage);
143                         return false;
144                     }
145                 }
146                 // If we use a globally configured shared secret for all CAs we check it right away
147                 if (globalSharedSecret != null) {
148                     if (LOG.isDebugEnabled()) {
149                         LOG.debug("Verifying message using Global Shared secret");
150                     }
151                     if (performPbeVerification(globalSharedSecret, "cmp.errorauthmessage", "Global auth secret")) {
152                         return true;
153                     }
154                 }
155                 // We failed verification using global shared secret, try the CA secret
156                 if (caInfo instanceof X509CAInfo) {
157                     final String authSecret = ((X509CAInfo) caInfo).getCmpRaAuthSecret();
158                     if (StringUtils.isNotEmpty(authSecret)) {
159                         if (LOG.isDebugEnabled()) {
160                             LOG.debug("Verify message using 'CMP RA Authentication Secret' from CA '" + caInfo.getName() + "'.");
161                         }
162                         if (performPbeVerification(authSecret, "cmp.errorauthmessage", "Auth secret for CA=" + caInfo.getName())) {
163                             return true;
164                         }
165                     } else {
166                         if (LOG.isDebugEnabled()) {
167                             LOG.debug("CMP password is null from CA '"+caInfo.getName()+"'.");
168                         }
169                     }
170                 }
171                 // We have failed verification with CA authentication secret too.
172                 this.errorMessage = "Failed to verify message using both Global Shared Secret and CMP RA Authentication Secret";
173             } else {
174                 if (LOG.isDebugEnabled()) {
175                     LOG.debug("Verifying HMAC in Client mode");
176                 }
177                 //If client mode, we try to get the pre-registered endentity from the DB, and if there is a 
178                 //clear text password we check HMAC using this password.
179                 EndEntityInformation endEntityInformation = null;
180                 String subjectDN = null;
181                 try {
182                     if (username != null) {
183                         if (LOG.isDebugEnabled()) {
184                             LOG.debug("Searching for an end entity with username='" + username + "'.");
185                         }
186                         endEntityInformation = this.endEntityAccessSession.findUser(authenticationToken, username);
187                     } else {
188                         // No username given, so we try to find from to extract the username by the DN component specified, 
189                         // if not found, it must be found by subject/issuerDN from the certificate request.
190                         final CertTemplate certTemplate = getCertTemplate(pkiMessage);
191                         subjectDN = certTemplate.getSubject().toString();
192                         if (subjectDN != null) {
193                             final List<EndEntityInformation> endEntityInformations;
194                             final String extractedUsername = getUsernameByDnComponent(subjectDN);
195                             if (StringUtils.isNotBlank(extractedUsername)) {
196                                 endEntityInformation = this.endEntityAccessSession.findUser(authenticationToken, username);
197                             }
198                             if (endEntityInformation == null) {
199                                 final X500Name issuer = certTemplate.getIssuer();
200                                 if (issuer == null) {
201                                     if (LOG.isDebugEnabled()) {
202                                         LOG.debug("Searching for an end entity with SubjectDN='" + subjectDN + "'.");
203                                     }
204                                     endEntityInformations = this.endEntityAccessSession.findUserBySubjectDN(authenticationToken, subjectDN);
205                                     if (endEntityInformations.size() > 1) {
206                                         LOG.warn("Multiple end entities with subject DN " + subjectDN + " were found. This may lead to unexpected behavior.");
207                                     }
208                                 } else {
209                                     final String issuerDN = issuer.toString();
210                                     if (LOG.isDebugEnabled()) {
211                                         LOG.debug("Searching for an end entity with SubjectDN='" + subjectDN + "' and isserDN='" + issuerDN + "'");
212                                     }
213                                     endEntityInformations = endEntityAccessSession.findUserBySubjectAndIssuerDN(this.authenticationToken, subjectDN, issuerDN);
214                                     if (endEntityInformations.size() > 1) {
215                                         LOG.warn("Multiple end entities with subject DN " + subjectDN + " and issuer DN" + issuerDN
216                                                 + " were found. This may lead to unexpected behavior.");
217                                     }
218                                 }                    
219                                 if (!endEntityInformations.isEmpty()) {
220                                     endEntityInformation = endEntityInformations.get(0);
221                                 }
222                             }
223                         }
224                     }
225                 } catch (AuthorizationDeniedException e) {
226                     LOG.info("Not authorized to search for end entity: " + e.getMessage());
227                 }
228                 if (endEntityInformation == null) {
229                     LOG.info(INTRES.getLocalizedMessage("ra.errorentitynotexist", StringUtils.isNotEmpty(username) ? username : subjectDN));
230                     this.errorMessage = INTRES.getLocalizedMessage("ra.wrongusernameorpassword");
231                     return false;
232                 }
233                 if (LOG.isDebugEnabled()) {
234                     LOG.debug("Comparing HMAC password authentication for user '" + endEntityInformation.getUsername() + "'.");
235                 }
236                 final String eepassword = endEntityInformation.getPassword();
237                 if (StringUtils.isEmpty(eepassword)) {
238                     this.errorMessage = "No clear text password for user '" + endEntityInformation.getUsername() + "', not possible to check authentication.";
239                     if (LOG.isDebugEnabled()) {
240                         LOG.debug(this.errorMessage);
241                     }
242                     return false;
243                 }
244                 if (performPbeVerification(eepassword, "cmp.errorauthmessage", endEntityInformation.getUsername())) {
245                     return true;
246                 }
247             }
248         } catch (InvalidKeyException | NoSuchAlgorithmException e) {
249             // We don't want to explain in detail to the client why the configured key on the CA was invalid
250             this.errorMessage = INTRES.getLocalizedMessage("cmp.errorgeneral");
251             LOG.info(this.errorMessage, e);
252         } 
253         return false;
254     }

256     /**
257      * Try to validate the PKIMessage using the provided shared secret.
258      * 
259      * Saves last error message on failure to validate.
260      * Saves raAuthenticationSecret on successful validation (to be used later when creating response protection)
261      * @param raAuthenticationSecret Secret
262      * @param errorMessageKey Key
263      * @param errorMessageParameter Param 
264      * @return true if the validation was successful.
265      * @throws InvalidKeyException Fail
266      * @throws NoSuchAlgorithmException Fail
267      */
268     private boolean performPbeVerification(final String raAuthenticationSecret, final String errorMessageKey, final String errorMessageParameter)
269             throws InvalidKeyException, NoSuchAlgorithmException {
270         if (verifyer.verify(raAuthenticationSecret)) {
271             this.password = raAuthenticationSecret;
272             return true;
273         } else {
274             String errmsg = INTRES.getLocalizedMessage(errorMessageKey, errorMessageParameter);
275             LOG.info(errmsg);
276             if (verifyer.getErrMsg() != null) {
277                 errmsg = verifyer.getErrMsg();
278                 LOG.info(errmsg);
279             }
280             this.errorMessage = errmsg;
281         }
282         return false;
283     }

285     /**
286      * Returns the certificate template specified in the request impeded in msg.
287      * 
288      * @param msg message
289      * @return the certificate template embedded in msg. Null if no such template was found.
290      */
291     private CertTemplate getCertTemplate(final PKIMessage msg) {
292         final int tagnr = msg.getBody().getType();
293         if(tagnr == CmpPKIBodyConstants.INITIALIZATIONREQUEST 
294                 || tagnr==CmpPKIBodyConstants.CERTIFICATAIONREQUEST
295                 || tagnr==CmpPKIBodyConstants.KEYUPDATEREQUEST) {
296             CertReqMessages reqmsgs = (CertReqMessages) msg.getBody().getContent();
297             return reqmsgs.toCertReqMsgArray()[0].getCertReq().getCertTemplate();
298         }
299         if(tagnr==CmpPKIBodyConstants.REVOCATIONREQUEST) {
300             RevReqContent rev  =(RevReqContent) msg.getBody().getContent();
301             return rev.toRevDetailsArray()[0].getCertDetails();
302         }
303         return null;
304     }

306     /**
307      * Gets the username by the DN component specified in the CMP configuration 'extract username component' 
308      * and adds the RA name generation prefix and postfix.
309      * 
310      * @param dn the DN.
311      * @return the username with RA name generation prefix and postfix.
312      */
313     private String getUsernameByDnComponent(String dn) {
314         final String usernameComp = this.cmpConfiguration.getExtractUsernameComponent(this.confAlias);
315         if (LOG.isDebugEnabled()) {
316             LOG.debug("extractUsernameComponent: "+usernameComp);
317         }
318         if(StringUtils.isNotEmpty(usernameComp)) {
319             String username = CertTools.getPartFromDN(dn,usernameComp);
320             String fix = cmpConfiguration.getRANameGenPrefix(this.confAlias);
321             if (StringUtils.isNotBlank(fix)) {
322                 LOG.info("Preceded RA name prefix '" + fix + "' to username '" + username + "' in CMP vendor mode.");
323                 username = fix + username;
324             }
325             fix = cmpConfiguration.getRANameGenPostfix(this.confAlias);
326             if (StringUtils.isNotBlank( cmpConfiguration.getRANameGenPostfix(this.confAlias))) {
327                 LOG.info("Attached RA name postfix '" + fix + "' to username '" + username + "' in CMP vendor mode.");
328                 username += fix;
329             }
330             return username;
331         }    
332         return null;
333     }
