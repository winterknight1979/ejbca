158     /** Creates the implementation for CMP vendor mode, if it exists. If it does not exist, uses a CmpVendorModeNoop implementation that
159      * return false on the question of CMP Vendor mode is used.
160      */
161     private void createVendorModeImpl() {
162         if (implExists) {
163             try {
164                 if (implClass == null) {
165                     // We only end up here once, if the class does not exist, we will never end up here again (ClassNotFoundException)
166                     // and if the class exists we will never end up here again (it will not be null)
167                     implClass = Class.forName(implClassName);
168                     log.debug("CmpVendorModeImpl is available, and used, in this version of EJBCA.");
169                 }
170                 impl = (CmpVendorMode)implClass.getConstructor().newInstance();
171                 impl.setCaSession(caSession);
172                 impl.setCmpConfiguration(cmpConfiguration);
173             } catch (ClassNotFoundException | NoSuchMethodException e) {
174                 // We only end up here once, if the class does not exist, we will never end up here again
175                 implExists = false;
176                 log.info("CMP Vendor mode is not available in the version of EJBCA.");
177                 impl = new CmpVendorModeNoopImpl();
178             } catch (InstantiationException | InvocationTargetException e) {
179                 log.error("Error intitilizing CmpVendorMode: ", e);
180             } catch (IllegalAccessException e) {
181                 log.error("Error intitilizing CmpVendorMode: ", e);
182             }
183         } else {
184             impl = new CmpVendorModeNoopImpl();
185         }
186     }

213     /**
214      * Get the end entity certificate that was attached to the CMP request in it's extreCert filed.
215      * If the extraCerts field contains multiple certificates, these are ordered in a CertPath and the leaf certificate is returned.
216      * @param msg Message
217      *
218      * @return The end entity certificate that was attached to the CMP request in it's extraCert field, or null, as an ordered certificate path with leaf certificate in first position
219      */
220     private List<X509Certificate> getExtraCerts(final PKIMessage msg) {
221         final CMPCertificate[] extraCerts = msg.getExtraCerts();
222         if ((extraCerts == null) || (extraCerts.length == 0)) {
223             if(log.isDebugEnabled()) {
224                 log.debug("There are no certificates in the extraCert field in the PKIMessage");
225             }
226             return null;
227         } else {
228             if(log.isDebugEnabled()) {
229                 log.debug(extraCerts.length+ " certificate(s) found in the extraCert field in the CMP message");
230             }
231         }
232 
233         try {
234             //Read the extraCerts. Convert to an array of normal X509Certificates so we can later use a regular CertPath validator
235             List<X509Certificate> certlist = new ArrayList<X509Certificate>();
236             // Create CertPath
237             final JcaX509CertificateConverter jcaX509CertificateConverter = new JcaX509CertificateConverter();
238             for (int i = 0; i < extraCerts.length; i++) {
239                 certlist.add(jcaX509CertificateConverter.getCertificate(new X509CertificateHolder(extraCerts[i].getX509v3PKCert())));
240             }
241             if (!certlist.isEmpty()) {
242                 List<X509Certificate> orderedCerts = CertTools.orderX509CertificateChain(certlist);
243                 if (log.isDebugEnabled()) {
244                     log.debug("Obtaining " +certlist.size()+ " certificate(s) from extraCert field was done successfully.");
245                 }
246                 if (log.isTraceEnabled()) {
247                     log.trace("extraCerts obtained: "+orderedCerts);
248                 }
249                 return orderedCerts;
250             } else {
251                 if(log.isDebugEnabled()) {
252                     log.debug("Obtaining the certificate from extraCert field failed, the result was null.");
253                 }
254             }
255         } catch (CertificateException e) {
256             // We only log debug to prevent DOS attacks (log spamming) by sending invalid messages
257             if(log.isDebugEnabled()) {
258                 log.debug(e.getLocalizedMessage(), e);
259             }
260         } catch (CertPathValidatorException e) {
261             // We only log debug to prevent DOS attacks (log spamming) by sending invalid messages
262             if(log.isDebugEnabled()) {
263                 log.debug("extraCerts does not contain a valid certificate path: "+e.getMessage());
264             }
265         }
266         return null;
267     }

269     @Override
270     /*
271      * Verifies the signature of 'msg'. msg should be signed and the signer's certificate should be
272      * attached in msg in the extraCert field.
273      *
274      * When successful, the authentication string is set.
275      */
276     public boolean verifyOrExtract(final PKIMessage msg, final String username) {
277 
278         //Check that msg is signed
279         if(msg.getProtection() == null) {
280             this.errorMessage = "PKI Message is not authenticated properly. No PKI protection is found.";
281             return false;
282         }
283 
284         // Read the extraCert and store it in a local variable
285         List<X509Certificate> extraCertPath = getExtraCerts(msg);
286         extraCert = (extraCertPath != null ? extraCertPath.get(0) : null);
287         if(extraCert == null) {
288             this.errorMessage = "Error while reading the certificate in the extraCert field";
289             return false;
290         }
291 
292         // A CMP KeyUpdateRequest is always regarding an end entity that already exists. If that end entity
293         // does not exist, no point in looking further
294         EndEntityInformation endentity = null;
295         if(msg.getBody().getType() == CmpPKIBodyConstants.KEYUPDATEREQUEST) {
296             try {
297                 endentity = getEndEntityFromKeyUpdateRequest(msg);
298             } catch (AuthorizationDeniedException e1) {
299                 // note: this should not happen since at this point, this is an AlwaysAllowed token
300                 this.errorMessage = "Administrator " + admin.toString() + " is not authorized to retrieve end entity";
301                 return false;
302             }
303             if(endentity == null) {
304                 this.errorMessage = "Error. Received a CMP KeyUpdateRequest for a non-existing end entity";
305                 return false;
306             }
307         }
308 
309         boolean vendormode = impl.isVendorCertificateMode(msg.getBody().getType(), this.confAlias);
310         boolean omitVerifications = cmpConfiguration.getOmitVerificationsInEEC(confAlias);
311         boolean ramode = cmpConfiguration.getRAMode(confAlias);
312         if(log.isDebugEnabled()) {
313             log.debug("CMP is operating in RA mode: " + this.cmpConfiguration.getRAMode(this.confAlias));
314             log.debug("CMP is operating in Vendor mode: " + vendormode);
315             log.debug("CMP message already been authenticated: " + authenticated);
316             log.debug("Omitting some verifications: " + omitVerifications);
317             log.debug("CMP message (claimed to be) signed by (cert from extraCerts): SubjectDN '" + CertTools.getSubjectDN(extraCert)+"' IssuerDN '"+CertTools.getIssuerDN(extraCert) +"'");
318         }
319 
320         //----------------------------------------------------------------------------------------
321         // Perform the different checks depending on the configuration and previous authentication
322         //----------------------------------------------------------------------------------------
323 
324         // Not allowed combinations.
325         if(ramode && vendormode) {
326             this.errorMessage = "Vendor mode and RA mode cannot be combined";
327             return false;
328         }
329         if(omitVerifications && (!ramode || !authenticated)) {
330             this.errorMessage = "Omitting some verifications can only be accepted in RA mode and when the " +
331                                  "CMP request has already been authenticated, for example, through the use of NestedMessageContent";
332             return false;
333         }
334 
335         // Accepted combinations
336         if(omitVerifications && ramode && authenticated) {
337             // Do nothing here
338             if(log.isDebugEnabled()) {
339                 log.debug("Skipping some verification of the extraCert certificate in RA mode and an already authenticated CMP message, tex. through NestedMessageContent");
340             }
341         } else if(ramode) {
342 
343             // Get the CA to use for the authentication
344             CAInfo cainfo = getCAInfoByName(authenticationparameter);
345             if(cainfo == null) {
346                 return false;
347             }
348 
349             // Check that extraCert is in the Database
350             CertificateInfo certinfo = certSession.getCertificateInfo(CertTools.getFingerprintAsString(extraCert));
351             if(certinfo == null) {
352                 this.errorMessage = "The certificate attached to the PKIMessage in the extraCert field could not be found in the database.";
353                 return false;
354             }
355 
356             // More extraCert verifications
357             if(!isExtraCertValidAndIssuedByCA(extraCertPath, cainfo) || !isExtraCertActive(certinfo)) {
358                 return false;
359             } else {
360                 if(log.isDebugEnabled()) {
361                     log.debug("Certificate in extraCerts field is issued by " + cainfo.getName() + ", is valid and active");
362                 }
363             }
364 
365             // Check that extraCert belong to an admin with sufficient access rights
366             if(!isAuthorizedAdmin(certinfo, msg, endentity)){
367                 this.errorMessage = "'" + CertTools.getSubjectDN(extraCert) + "' is not an authorized administrator.";
368                 return false;
369             }
370 
371         } else if(!ramode) { // client mode
372 
373             String extraCertUsername = null;
374             if(vendormode) {
375 
376                 // Check that extraCert is issued  by a configured VendorCA
377                 final CAInfo cainfo = impl.isExtraCertIssuedByVendorCA(admin, this.confAlias, extraCertPath);
378                 if (cainfo == null) {
379                     this.errorMessage = "The certificate in extraCert field is not issued by any of the configured Vendor CAs: " + cmpConfiguration.getVendorCA(confAlias);
380                     return false;
381                 }
382 
383                 // Extract the username from extraCert to use for  further authentication
384                 String subjectDN = CertTools.getSubjectDN(extraCert);
385                 extraCertUsername = CertTools.getPartFromDN(subjectDN, this.cmpConfiguration.getExtractUsernameComponent(this.confAlias));
386                 if (log.isDebugEnabled()) {
387                     log.debug("Username ("+extraCertUsername+") was extracted from the '" + this.cmpConfiguration.getExtractUsernameComponent(this.confAlias) + "' part of the subjectDN of the certificate in the 'extraCerts' field.");
388                 }
389 
390                 // More extraCert verifications
391                 if (!isExtraCertValidAndIssuedByCA(extraCertPath, cainfo)) {
392                     return false;
393                 }
394             } else {
395 
396                 // Get the CA to use for the authentication
397                 CAInfo cainfo = getCAInfoByIssuer(CertTools.getIssuerDN(extraCert));
398                 // Check that extraCert is in the Database
399                 CertificateInfo certinfo = certSession.getCertificateInfo(CertTools.getFingerprintAsString(extraCert));
400                 if(certinfo == null) {
401                     this.errorMessage = "The certificate attached to the PKIMessage in the extraCert field could not be found in the database.";
402                     return false;
403                 }
404 
405                 // More extraCert verifications
406                 if(!isExtraCertValidAndIssuedByCA(extraCertPath, cainfo) || !isExtraCertActive(certinfo)) {
407                     return false;
408                 }
409 
410                 // Extract the username from extraCert to use for  further authentication
411                 extraCertUsername = certinfo.getUsername();
412             }
413 
414             // Check if this certificate belongs to the user
415             if ( (username != null) && (extraCertUsername != null) ) {
416                 if (cmpConfiguration.getVendorMode(this.confAlias)) {
417                     String fix = cmpConfiguration.getRANameGenPrefix(this.confAlias);
418                     if (StringUtils.isNotBlank(fix)) {
419                         log.info("Preceded RA name prefix '" + fix + "' to username '" + username + "' in CMP vendor mode.");
420                         extraCertUsername = fix + extraCertUsername;
421                     }
422                     fix = cmpConfiguration.getRANameGenPostfix(this.confAlias);
423                     if (StringUtils.isNotBlank( cmpConfiguration.getRANameGenPostfix(this.confAlias))) {
424                         log.info("Attached RA name postfix '" + fix + "' to username '" + username + "' in CMP vendor mode.");
425                         extraCertUsername += fix;
426                     }
427                 }
428                 if (!StringUtils.equals(username, extraCertUsername)) {
429                     this.errorMessage = "The End Entity certificate attached to the PKIMessage in the extraCert field does not belong to user '"+username+"'";
430                     if(log.isDebugEnabled()) {
431                         // Use a different debug message, as not to reveal too much information
432                         log.debug(this.errorMessage + ", but to user '"+extraCertUsername+"'");
433                     }
434                     return false;
435                 }
436 
437                 //set the password of the request to this user's password so it can later be used when issuing the certificate
438                 if (log.isDebugEnabled()) {
439                     log.debug("The End Entity certificate attached to the PKIMessage in the extraCert field belongs to user '"+username+"'.");
440                     log.debug("Extracting and setting password for user '"+username+"'.");
441                 }
442                 try {
443                     EndEntityInformation user = eeAccessSession.findUser(admin, username);
444                     password = user.getPassword();
445                     if(password == null) {
446                         password = genRandomPwd();
447                         user.setPassword(password);
448                         eeManagementSession.changeUser(admin, user, false);
449                     }
450                 } catch (AuthorizationDeniedException | IllegalNameException | CADoesntExistsException | EndEntityProfileValidationException
451                         | WaitingForApprovalException | CertificateSerialNumberException | ApprovalException | NoSuchEndEntityException | CustomFieldException e) {
452                     if (log.isDebugEnabled()) {
453                         log.debug(e.getLocalizedMessage());
454                     }
455                     this.errorMessage = e.getLocalizedMessage();
456                     return false;
457                 }
458             } else {
459                 if (log.isDebugEnabled()) {
460                     log.debug("Certificate does not belong to user. Username='"+username+"', extraCert username='"+extraCertUsername+"'.");
461                 }
462 
463             }
464         }
465 
466         //-------------------------------------------------------------
467         //Begin the signature verification process.
468         //Verify the signature of msg using the public key of extraCert
469         //-------------------------------------------------------------
470         try {
471             final Signature sig = Signature.getInstance(msg.getHeader().getProtectionAlg().getAlgorithm().getId(), BouncyCastleProvider.PROVIDER_NAME);
472             sig.initVerify(extraCert.getPublicKey());
473             sig.update(CmpMessageHelper.getProtectedBytes(msg));
474             if (sig.verify(msg.getProtection().getBytes())) {
475                 if (password == null) {
476                     // If not set earlier
477                     password = genRandomPwd();
478                 }
479             } else {
480                 this.errorMessage = "Failed to verify the signature in the PKIMessage";
481                 return false;
482             }
483         } catch (InvalidKeyException e) {
484             if(log.isDebugEnabled()) {
485                 log.debug(e.getLocalizedMessage());
486             }
487             this.errorMessage = e.getLocalizedMessage();
488             return false;
489         } catch (NoSuchAlgorithmException e) {
490             if(log.isDebugEnabled()) {
491                 log.debug(e.getLocalizedMessage());
492             }
493             this.errorMessage = e.getLocalizedMessage();
494             return false;
495         } catch (NoSuchProviderException e) {
496             if(log.isDebugEnabled()) {
497                 log.debug(e.getLocalizedMessage());
498             }
499             this.errorMessage = e.getLocalizedMessage();
500             return false;
501         } catch (SignatureException e) {
502             if(log.isDebugEnabled()) {
503                 log.debug(e.getLocalizedMessage());
504             }
505             this.errorMessage = e.getLocalizedMessage();
506             return false;
507         }
508 
509         return this.password != null;
510     }

519         return pwdgen.getNewPassword(12, 12);
520     }

522     private EndEntityInformation getEndEntityFromKeyUpdateRequest(final PKIMessage pkimessage) throws AuthorizationDeniedException {
523         String subjectDN="", issuerDN="";
524 
525         if(cmpConfiguration.getRAMode(confAlias)) {
526             CertReqMessages kur = (CertReqMessages) pkimessage.getBody().getContent();
527             CertReqMsg certmsg;
528             try {
529                 certmsg = kur.toCertReqMsgArray()[0];
530             } catch(Exception e) {
531                 log.debug("Could not parse the KeyUpdate request. Trying to parse it as novosec generated message.");
532                 certmsg = CmpMessageHelper.getNovosecCertReqMsg(kur);
533                 if(certmsg == null) {
534                     log.info("Error. Failed to parse CMP message novosec generated message. " + e.getLocalizedMessage());
535                     if(log.isDebugEnabled()) {
536                         log.debug(e);
537                     }
538                     return null;
539                 } else {
540                     if(log.isDebugEnabled()) {
541                         log.debug("Succeeded in parsing the novosec generated request.");
542                     }
543                 }
544             }
545 
546             X500Name dn = certmsg.getCertReq().getCertTemplate().getSubject();
547             if(dn != null) {
548                 subjectDN = dn.toString();
549             }
550             dn = certmsg.getCertReq().getCertTemplate().getIssuer();
551             if(dn != null) {
552                 issuerDN = dn.toString();
553             }
554         } else {
555             subjectDN = CertTools.getSubjectDN(extraCert);
556             issuerDN = CertTools.getIssuerDN(extraCert);
557         }
558         if(log.isDebugEnabled()) {
559             log.debug("Received a CMP KeyUpdateRequest for an endentity with SubjectDN '" + subjectDN + "' and issuerDN '" + issuerDN + "'");
560         }
561 
562         EndEntityInformation userdata = null;
563         if(StringUtils.isEmpty(issuerDN)) {
564             if(log.isDebugEnabled()) {
565                 log.debug("The CMP KeyUpdateRequest did not specify an issuer");
566             }
567             List<EndEntityInformation> userdataList = eeAccessSession.findUserBySubjectDN(admin, subjectDN);
568             if (userdataList.size() > 0) {
569                 userdata = userdataList.get(0);
570             }
571             if (userdataList.size() > 1) {
572                 log.warn("Multiple end entities with subject DN " + subjectDN + " were found. This may lead to unexpected behavior.");
573             }
574         } else {
575             List<EndEntityInformation> userdataList = eeAccessSession.findUserBySubjectAndIssuerDN(admin, subjectDN, issuerDN);
576             if (userdataList.size() > 0) {
577                 userdata = userdataList.get(0);
578             }
579             if (userdataList.size() > 1) {
580                 log.warn("Multiple end entities with subject DN " + subjectDN + " and issuer DN" + issuerDN
581                         + " were found. This may lead to unexpected behavior.");
582             }
583         }
584 
585         return userdata;
586     }

588     /**
589      * Checks if cert belongs to an administrator who is authorized to process the request.
590      *
591      * @param certInfo Info
592      * @param msg Message
593      * @param endentity Only used when the message received is a KeyUpdateRequest in RA mode. The administrator is authorized to handle a KeyUpdateRequest in RA mode if
594      *                  it is authorized to the EndEntityProfile, CertificateProfile and the CA specified in this end entity.
595      * @return true if the administrator is authorized to process the request and false otherwise.
596      */
597     private boolean isAuthorizedAdmin(final CertificateInfo certInfo, final PKIMessage msg, final EndEntityInformation endentity) {
598 
599         X509Certificate x509cert = (X509Certificate) extraCert;
600         Set<X509Certificate> credentials = new HashSet<X509Certificate>();
601         credentials.add(x509cert);
602 
603         AuthenticationSubject subject = new AuthenticationSubject(null, credentials);
604         AuthenticationToken reqAuthToken = authenticationProviderSession.authenticate(subject);
605 
606         final int tagnr = msg.getBody().getType();
607         if( (tagnr == CmpPKIBodyConstants.CERTIFICATAIONREQUEST) || (tagnr == CmpPKIBodyConstants.INITIALIZATIONREQUEST) ) {
608 
609             final int eeprofid;
610             final String eepname;
611             try {
612                 final String configuredId = this.cmpConfiguration.getRAEEProfile(this.confAlias);
613                 if (StringUtils.equals(CmpConfiguration.PROFILE_USE_KEYID, configuredId)) {
614                     eepname = CmpMessageHelper.getStringFromOctets(msg.getHeader().getSenderKID());
615                     eeprofid = eeProfileSession.getEndEntityProfileId(eepname);
616                 } else {
617                     eeprofid = Integer.parseInt(configuredId);
618                     eepname = eeProfileSession.getEndEntityProfileName(eeprofid);
619                 }
620                 if(eepname == null) {
621                     log.error("End Entity Profile with ID " + configuredId + " was not found");
622                     return false;
623                 }
624             } catch(NumberFormatException e) {
625                 log.error("End Entity Profile ID " + this.cmpConfiguration.getRAEEProfile(this.confAlias) +
626                         " in CMP alias " + this.confAlias + " was not an integer");
627                 return false;
628             } catch (EndEntityProfileNotFoundException e) {
629                 log.error("End Entity Profile Name specified in the senderKID field could not be mapped to an end entity profile ID: "
630                         + e.getMessage());
631                 return false;
632             }
633 
634 
635             if (!authorizedToEndEntityProfile(reqAuthToken, eeprofid, AccessRulesConstants.CREATE_END_ENTITY)) {
636                 log.info("Administrator " + reqAuthToken.toString() + " was not authorized to create end entities with EndEntityProfile " + eepname);
637                 return false;
638             } else {
639                 if(log.isDebugEnabled()) {
640                     log.debug("Administrator " + reqAuthToken.toString() + " was authorized to create end entities with EndEntityProfile " + eepname);
641                 }
642             }
643 
644             if(!authorizedToEndEntityProfile(reqAuthToken, eeprofid, AccessRulesConstants.EDIT_END_ENTITY)) {
645                 log.info("Administrator " + reqAuthToken.toString() + " was not authorized to edit end entities with EndEntityProfile " + eepname);
646                 return false;
647             } else {
648                 if(log.isDebugEnabled()) {
649                     log.debug("Administrator " + reqAuthToken.toString() + " was authorized to edit end entities with EndEntityProfile " + eepname);
650                 }
651             }
652 
653             if(!authSession.isAuthorizedNoLogging(reqAuthToken, AccessRulesConstants.REGULAR_CREATECERTIFICATE)) {
654                 log.info("Administrator " + reqAuthToken.toString() + " is not authorized to create certificates.");
655                 return false;
656             } else {
657                 if(log.isDebugEnabled()) {
658                     log.debug("Administrator " + reqAuthToken.toString() + " was authorized to create certificates");
659                 }
660             }
661 
662             final EndEntityProfile eep = eeProfileSession.getEndEntityProfile(eeprofid);
663             final int caid = getRaCaId((DEROctetString) msg.getHeader().getSenderKID(), eep);
664             if(!authSession.isAuthorizedNoLogging(reqAuthToken, StandardRules.CAACCESS.resource() + caid)) {
665                 log.info("Administrator " + reqAuthToken.toString() + " not authorized to resource " + StandardRules.CAACCESS.resource() + caid);
666                 return false;
667             } else {
668                 if(log.isDebugEnabled()) {
669                     log.debug("Administrator " + reqAuthToken.toString() + " is authorized to access CA with ID " + caid);
670                 }
671             }
672 
673             CertificateProfile cp = getCertificateProfileFromCrmf(eep, (DEROctetString) msg.getHeader().getSenderKID());
674             if(!isCertificateProfileAuthorizedToCA(cp, caid)) {
675                 log.info("CertificateProfile is not authorized to CA with ID: " + caid);
676                 return false;
677             }
678 
679 
680         } else if( tagnr == CmpPKIBodyConstants.KEYUPDATEREQUEST ) {
681             // Because we do not edit the end entity when we receive a KeyUpdateRequest (the only editing we do is reset the end entity status),
682             // the CertificateProfile and CA that will be used to issue the new certificate will be the ones already set in the end entity
683             // regardless of the content of the request or the configuration of the CMP alias.
684         	// It is, however, possible to edit the profiles and CA in the end entity (we just do not do that when receiving a CMPKeyUpdateRequest),
685         	// So if we change the implementation in CrmfKeyUpdateHandler.java, this check should be updated accordingly
686 
687             final int eeprofid = endentity.getEndEntityProfileId();
688             final String eepname = eeProfileSession.getEndEntityProfileName(eeprofid);
689             if(eepname == null) {
690                 log.error("End Entity Profile with ID " + eeprofid + " was not found");
691                 return false;
692             }
693 
694             if(!authorizedToEndEntityProfile(reqAuthToken, eeprofid, AccessRulesConstants.EDIT_END_ENTITY)) {
695                 log.info("Administrator " + reqAuthToken.toString() + " was not authorized to edit end entities with EndEntityProfile " + eepname);
696                 return false;
697             } else {
698                 if(log.isDebugEnabled()) {
699                     log.debug("Administrator " + reqAuthToken.toString() + " was authorized to edit end entities with EndEntityProfile " + eepname);
700                 }
701             }
702 
703             if(!authSession.isAuthorizedNoLogging(reqAuthToken, AccessRulesConstants.REGULAR_CREATECERTIFICATE)) {
704                 log.info("Administrator " + reqAuthToken.toString() + " is not authorized to create certificates.");
705                 return false;
706             } else {
707                 if(log.isDebugEnabled()) {
708                     log.debug("Administrator " + reqAuthToken.toString() + " is authorized to create certificates");
709                 }
710             }
711 
712             final int caid = endentity.getCAId();
713             if(!authSession.isAuthorizedNoLogging(reqAuthToken, StandardRules.CAACCESS.resource() + caid)) {
714                 log.info("Administrator " + reqAuthToken.toString() + " not authorized to resource " + StandardRules.CAACCESS.resource() + caid);
715                 return false;
716             } else {
717                 if(log.isDebugEnabled()) {
718                     log.debug("Administrator " + reqAuthToken.toString() + " is authorized to access CA with ID " + caid);
719                 }
720             }
721 
722             if(!isCertificateProfileAuthorizedToCA(certProfileSession.getCertificateProfile(endentity.getCertificateProfileId()), caid)) {
723                 log.info("CertificateProfile " + certProfileSession.getCertificateProfileName(endentity.getCertificateProfileId()) + " is not authorized to CA with ID: " + caid);
724                 return false;
725             }
726 
727 
728         } else if(tagnr == CmpPKIBodyConstants.REVOCATIONREQUEST) {
729             final String issuerdn = getIssuerDNFromRevRequest((RevReqContent) msg.getBody().getContent());
730             final int caid = CertTools.stringToBCDNString(issuerdn).hashCode();
731             if(!authSession.isAuthorizedNoLogging(reqAuthToken, StandardRules.CAACCESS.resource() + caid)) {
732                 if(log.isDebugEnabled()) {
733                     log.debug("Administrator " + reqAuthToken.toString() + " NOT authorized to revoke certificates issues by " + issuerdn);
734                 }
735                 return false;
736             } else {
737                 if(log.isDebugEnabled()) {
738                     log.debug("Administrator " + reqAuthToken.toString() + " is authorized to revoke certificates issued by " + issuerdn);
739                 }
740             }
741 
742             if(!authSession.isAuthorizedNoLogging(reqAuthToken, AccessRulesConstants.REGULAR_REVOKEENDENTITY)) {
743                 if(log.isDebugEnabled()) {
744                     log.debug("Administrator " + reqAuthToken.toString() + " is not authorized to revoke End Entities");
745                 }
746                 return false;
747             } else {
748                 if(log.isDebugEnabled()) {
749                     log.debug("Administrator " + reqAuthToken.toString() + " was authorized to revoke end entities");
750                 }
751             }
752 
753         }
754         return true;
755     }

757     /**
758      * Checks whether admin is authorized to access the EndEntityProfile with the ID profileid
759      *
760      * @param admin Admin
761      * @param profileid ID
762      * @param rights Rights
763      * @return true if admin is authorized and false otherwise.
764      */
765     private boolean authorizedToEndEntityProfile(AuthenticationToken admin, int profileid, String rights) {
766         if (profileid == EndEntityConstants.EMPTY_END_ENTITY_PROFILE
767                 && (rights.equals(AccessRulesConstants.CREATE_END_ENTITY) || rights.equals(AccessRulesConstants.EDIT_END_ENTITY))) {
768 
769             return authSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource());
770         } else {
771             return authSession.isAuthorizedNoLogging(admin, AccessRulesConstants.ENDENTITYPROFILEPREFIX + profileid + rights)
772                     && authSession.isAuthorizedNoLogging(admin, AccessRulesConstants.REGULAR_RAFUNCTIONALITY + rights);
773         }
774     }

776     /**
777      * Returns the IssuerDN specified in the CMP revocation request
778      * @param revReq Request
779      * @return the IssuerDN
780      */
781     private String getIssuerDNFromRevRequest(final RevReqContent revReq) {
782         RevDetails rd;
783         try {
784             rd = revReq.toRevDetailsArray()[0];
785         } catch(Exception e) {
786             log.debug("Could not parse the revocation request. Trying to parse it as novosec generated message.");
787             rd = CmpMessageHelper.getNovosecRevDetails(revReq);
788             log.debug("Succeeded in parsing the novosec generated request.");
789         }
790         final CertTemplate ct = rd.getCertDetails();
791         final X500Name issuer = ct.getIssuer();
792         if(issuer != null) {
793             return ct.getIssuer().toString();
794         }
795         return "";
796     }

799      * Return the ID of the CA that is used for CMP purposes.
800      * @param keyId ID
801      * @param eep Profile
802      * @return the ID of CA used for CMP purposes.
803      */
804     private int getRaCaId(final DEROctetString keyId, final EndEntityProfile eep) {
805 
806         String caname = this.cmpConfiguration.getRACAName(this.confAlias);
807         if (StringUtils.equals(caname, CmpConfiguration.PROFILE_DEFAULT)) {
808             final int caid = eep.getDefaultCA();
809             if (log.isDebugEnabled()) {
810                 log.debug("Using EndEntity profile's default CA with ID: "+caid);
811             }
812             return caid;
813         }
814 
815         if (StringUtils.equals(caname, CmpConfiguration.PROFILE_USE_KEYID) && (keyId != null)) {
816             caname = CmpMessageHelper.getStringFromOctets(keyId);
817             if (log.isDebugEnabled()) {
818                 log.debug("Using CA with same name as KeyId in request: "+caname);
819             }
820         }
821         return getCAInfoByName(caname).getCAId();
822     }

844             }
845         }

849     private boolean isCertificateProfileAuthorizedToCA(final CertificateProfile profile, final int caid) {
850         // Check that CAid is among available CAs
851         boolean caauthorized = false;
852         if(profile != null) {
853             for (final Integer availablecas : profile.getAvailableCAs()) {
854                 final int availableca = availablecas.intValue();
855                 if (availableca == caid || availableca == CertificateProfile.ANYCA) {
856                     caauthorized = true;
857                     break;
858                 }
859             }
860         } else {
861             log.info("CMP CertificateProfile not found");
862         }
863         return caauthorized;
864     }

866     /** Checks that the extracerts is a valid certificate path, that verifies up to the TrustAnchor being the Root CA certificate
867      * of the supplied CA.
868      * @param extracerts certificates from the extraCerts field from the CMP Message
869      * @param cainfo The CA we want to verify extraCerts with, can be a root CA or a sub CA, in which case the sub CA's root CA is used as trust anchor
870      * @return true if extracert(s) verifies up to the trust anchor, false otherwise
871      */
872     private boolean isExtraCertValidAndIssuedByCA(List<X509Certificate> extracerts, CAInfo cainfo) {
873         if (extracerts == null || extracerts.isEmpty()) {
874             throw new IllegalArgumentException("extracerts must contain a certificate.");
875         }
876         Certificate endentitycert = null;
877         try {
878             // What we got in extraCerts can be different things
879             // - An end entity certificate only, signed by a SubCA or a RootCA
880             // -- We need to find both SubCA and RootCA here, should be in cainfo?
881             // - An end entity certificate and a SubCA certificate
882             // -- We need to find the RootCA certificate only, should be in cainfo?
883             // - An end entity certificate a SubCA certificate and a RootCA certificate
884             // -- We need to remove the CA certificates that are not part of cainfo
885             ArrayList<Certificate> certlist = new ArrayList<>();
886             // Create CertPath
887             certlist.addAll(extracerts);
888             // Move CA certificates into cert path, except root certificate which is the trust anchor
889             X509Certificate rootcert = null;
890             Collection<Certificate> trustedCertificates = cainfo.getCertificateChain();
891             final Iterator<Certificate> itr = trustedCertificates.iterator();
892             while (itr.hasNext()) {
893                 // Trust anchor is last, so if this is the last element, don't add it
894                 Certificate crt = itr.next();
895                 if (itr.hasNext()) {
896                     if (!certlist.contains(crt)) {
897                         certlist.add(crt);
898                     } else {
899                         if (log.isDebugEnabled()) {
900                             log.debug("Certlist already contains certificate with subject "+CertTools.getSubjectDN(crt)+", not adding to list");
901                         }
902                     }
903                 } else {
904                     rootcert = (X509Certificate)crt;
905                     if (log.isDebugEnabled()) {
906                         log.debug("Using certificate with subject "+CertTools.getSubjectDN(crt)+", as trust anchor, removing from certlist if it is there");
907                     }
908                     // Don't have the trust anchor in the cert path, remove doesn't do anything if rootcert doesn't exist in certlist
909                     certlist.remove(rootcert);
910                 }
911             }
912             CertPath cp = CertificateFactory.getInstance("X.509", BouncyCastleProvider.PROVIDER_NAME).generateCertPath(certlist);
913             // The end entity cert is the first one in the CertPath according to javadoc
914             // - "By convention, X.509 CertPaths (consisting of X509Certificates), are ordered starting with the target
915             //    certificate and ending with a certificate issued by the trust anchor.
916             //    That is, the issuer of one certificate is the subject of the following one."
917             // Note: CertPath above will most likely not sort the path, at least if there is a root cert in certlist
918             // the cp will fail verification if it was not in the right order in certlist to start with
919             endentitycert = cp.getCertificates().get(0);
920             TrustAnchor anchor = new TrustAnchor(rootcert, null);
921             PKIXParameters params = new PKIXParameters(Collections.singleton(anchor));
922             params.setRevocationEnabled(false);
923             CertPathValidator cpv = CertPathValidator.getInstance("PKIX", BouncyCastleProvider.PROVIDER_NAME);
924             PKIXCertPathValidatorResult result = (PKIXCertPathValidatorResult) cpv.validate(cp, params);
925             if (log.isDebugEnabled()) {
926                 log.debug("Certificate verify result: " + result.toString());
927             }
928             // No CertPathValidatorException thrown means it passed
929             return true;
930         } catch (CertPathValidatorException e) {
931             this.errorMessage = "The certificate attached to the PKIMessage in the extraCert field is not valid - " + getCertPathValidatorExceptionMessage(e);
932             if(log.isDebugEnabled()) {
933                 log.debug(this.errorMessage + ": SubjectDN=" + CertTools.getSubjectDN(endentitycert));
934             }
935         } catch (CertificateException e) {
936             log.warn("CertificateException", e);
937         } catch (NoSuchProviderException e) {
938             // Serious error, bail out
939             log.error("NoSuchProviderException", e);
940             throw new IllegalStateException(e);
941         } catch (InvalidAlgorithmParameterException e) {
942             log.info("InvalidAlgorithmParameterException", e);
943         } catch (NoSuchAlgorithmException e) {
944             log.info("NoSuchAlgorithmException", e);
945         }
946         return false;
947     }

949     /**
950      * Returns the message from a CertPathValidatorException. For the common cases, this method is locale independent.
951      * @param e Exception
952      * @return Message
953      */
954     private String getCertPathValidatorExceptionMessage(final CertPathValidatorException e) {
955         Certificate endEntityCert = null;
956         if (e.getCertPath() != null && CollectionUtils.isNotEmpty(e.getCertPath().getCertificates())) {
957             endEntityCert = e.getCertPath().getCertificates().get(0);
958         }
959         // getReason returns BasicReason.UNSPECIFIED for expired or not yet valid certs, so we need to look at the cause.
960         final Throwable cause = e.getCause();
961         if (cause instanceof CertificateExpiredException) {
962             return "Certificate has expired. NotAfter: " + ValidityDate.formatAsUTC(CertTools.getNotAfter(endEntityCert)) + " UTC";
963         } else if (cause instanceof CertificateNotYetValidException) {
964             return "Certificate is not yet valid. NotBefore: " + ValidityDate.formatAsUTC(CertTools.getNotBefore(endEntityCert)) + " UTC";
965         } else {
966             return e.getMessage();
967         }
968     }

970     private boolean isExtraCertActive(final CertificateInfo certinfo) {
971         // CERT_NOTIFIEDABOUTEXPIRATION is also active...
972         if (certinfo.getStatus() != CertificateConstants.CERT_ACTIVE && certinfo.getStatus() != CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION) {
973             this.errorMessage = "The certificate attached to the PKIMessage in the extraCert field is not active.";
974             if (log.isDebugEnabled()) {
975                 log.debug(this.errorMessage + " Username=" + certinfo.getUsername()+", fingerprint="+certinfo.getFingerprint());
976             }
977             return false;
978         }
979         if (log.isDebugEnabled()) {
980             log.debug("The certificate in extraCert is active");
981         }
982         return true;
983     }
