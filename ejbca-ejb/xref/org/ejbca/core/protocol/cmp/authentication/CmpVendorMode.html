14  package org.ejbca.core.protocol.cmp.authentication;
15  
16  import java.security.cert.X509Certificate;
17  import java.util.List;
18  
19  import org.cesecore.authentication.tokens.AuthenticationToken;
20  import org.cesecore.certificates.ca.CAInfo;
21  import org.cesecore.certificates.ca.CaSession;
22  import org.ejbca.config.CmpConfiguration;
23  
24  /**
25   * Check the authentication of the PKIMessage by verifying the signature from a Vendor CA (3GPP mode)
26   * 
27   * @version $Id: CmpVendorMode.java 26078 2017-06-26 13:13:17Z anatom $
28   *
29   */
30  public interface CmpVendorMode {
31  
32  
33      void setCaSession(final CaSession caSession);
34  
35      void setCmpConfiguration(final CmpConfiguration cmpConfiguration);
36      
37      /** Checks if the certificate is issued by a configured Vendor CA, and that it can be verified using that Vendor CA certificate
38       * @param admin administrator making the call, must have access to get CAInfo for the Vendor CA
39       * @param confAlias the CMP alias in use
40       * @param extraCerts certificates from the extraCert field of the CMP request, a certificate path ordered with leaf certificate first (pos 0) and CA certificate(s) after
41       * @return CAInfo if the Vendor CA, in the specified CMP alias, that issues the extraCert, null if no issuing Vendor CA was found.
42       */
43      CAInfo isExtraCertIssuedByVendorCA(final AuthenticationToken admin, final String confAlias, final List<X509Certificate> extraCerts);
44      
45      /**
46       * Checks whether authentication by vendor-issued-certificate should be used. It can be used only in client mode and with initialization/certification requests.
47       *  
48       * @param reqType type
49       * @param confAlias alias
50       * @return 'True' if authentication by vendor-issued-certificate is used. 'False' otherwise
51       */
52      boolean isVendorCertificateMode(final int reqType, final String confAlias);
53  }
