67  /**
68   * Message handler for update messages using the CRMF format for the request itself.
69   * 
70   * @version $Id: CrmfKeyUpdateHandler.java 28298 2018-02-15 12:29:17Z anatom $
71   */
72  public class CrmfKeyUpdateHandler extends BaseCmpMessageHandler implements ICmpMessageHandler {
73      
74      private static final Logger LOG = Logger.getLogger(CrmfKeyUpdateHandler.class);
75      /** Internal localization of logs and errors */
76      private static final InternalEjbcaResources INTRES = InternalEjbcaResources.getInstance();
77  
78      /** strings for error messages defined in internal resources */
79      private static final String CMP_ERRORGENERAL = "cmp.errorgeneral";
80  
81      private final AuthorizationSession authorizationSession;
82      private final CertificateStoreSession certStoreSession;
83      private final EndEntityAccessSession endEntityAccessSession;
84      private final EndEntityManagementSession endEntityManagementSession;
85      private final SignSession signSession;
86      private final WebAuthenticationProviderSessionLocal authenticationProviderSession;
87  
88  
89      public CrmfKeyUpdateHandler(final AuthenticationToken authenticationToken, final CmpConfiguration cmpConfiguration, final String configAlias,
90      		final EjbBridgeSessionLocal ejbBridgeSession) {
91          super(authenticationToken, cmpConfiguration, configAlias, ejbBridgeSession);
92          this.signSession = ejbBridgeSession.getSignSession();
93          this.endEntityAccessSession = ejbBridgeSession.getEndEntityAccessSession();
94          this.certStoreSession = ejbBridgeSession.getCertificateStoreSession();
95          this.authorizationSession = ejbBridgeSession.getAuthorizationSession();
96          this.authenticationProviderSession = ejbBridgeSession.getWebAuthenticationProviderSession();
97          this.endEntityManagementSession = ejbBridgeSession.getEndEntityManagementSession();
98      }
99  
100     /*
101      * Handles the CMP message
102      * 
103      * Expects the CMP message to be a CrmfRequestMessage. The message is authenticated using 
104      * EndEntityCertificateAuthenticationModule in client mode. It used the attached certificate 
105      * to find then End Entity which this certificate belongs to and requesting for a new certificate 
106      * to be generated. 
107      * 
108      * If automatic update of the key (same as certificate renewal), the end entity's status is set to 
109      * 'NEW' before processing the request. If using the same old keys in the new certificate is not allowed, 
110      * a check is made to insure the the key specified in the request is not the same as the key of the attached 
111      * certificate.
112      * 
113      * The KeyUpdateRequest is processed only in client mode.
114      */
115     @Override
116     public ResponseMessage handleMessage(final BaseCmpMessage cmpRequestMessage, final boolean authenticated) {
117         if (LOG.isTraceEnabled()) {
118             LOG.trace(">handleMessage");
119         }
120         // NOTE. If we will issue the new certificate using another CertificateProfile and/or CA than the ones specified 
121         // in the end entity (prior to receiving the request), update the authorization check in 
122         // EndEntityCertificateAuthenticationModule.isAuthorizedAdmin() to check authorization to the right CertProfile and CA
123         final boolean raMode = this.cmpConfiguration.getRAMode(this.confAlias);
124         if (LOG.isDebugEnabled()) {
125             LOG.debug("CMP running on RA mode: " + raMode);
126         }
127         ResponseMessage resp = null;
128         try {
129             CrmfRequestMessage crmfreq = null;
130             if (cmpRequestMessage instanceof CrmfRequestMessage) {
131                 crmfreq = (CrmfRequestMessage) cmpRequestMessage;
132                 crmfreq.getMessage();
133 
134                 EndEntityCertificateAuthenticationModule eecmodule = null;
135                 X509Certificate oldCert = null;
136                         
137                 // Find the subjectDN to look for
138                 String subjectDN = null;
139                 String issuerDN = null;
140                 if(raMode) {                    
141                     // Check that EndEntityCertificate authentication module is set
142                     if(!cmpConfiguration.isInAuthModule(confAlias, CmpConfiguration.AUTHMODULE_ENDENTITY_CERTIFICATE)) {
143                         String errmsg = "EndEntityCertificate authentication module is not configured. For a KeyUpdate request to be authentication in RA mode, EndEntityCertificate " +
144                         		"authentication module has to be set and configured";
145                         LOG.info(errmsg);
146                         return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, errmsg);
147                     }
148                     // Check PKIMessage authentication
149                     String authparameter = cmpConfiguration.getAuthenticationParameter(CmpConfiguration.AUTHMODULE_ENDENTITY_CERTIFICATE, confAlias);
150                     eecmodule = new EndEntityCertificateAuthenticationModule(admin, authparameter, 
151                             confAlias, cmpConfiguration, authenticated, caSession, certStoreSession, authorizationSession, endEntityProfileSession, certificateProfileSession,
152                             endEntityAccessSession, authenticationProviderSession, endEntityManagementSession);
153                     if(!eecmodule.verifyOrExtract(crmfreq.getPKIMessage(), null)) {
154                         LOG.info(eecmodule.getErrorMessage());
155                         return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, eecmodule.getErrorMessage());
156                     } else {
157                         if(LOG.isDebugEnabled()) {
158                             LOG.debug("The CMP KeyUpdate request for SubjectDN '" + crmfreq.getSubjectDN() +"' was verified successfully");
159                         }
160                     }
161                     oldCert = certStoreSession.findLatestX509CertificateBySubject(crmfreq.getSubjectDN());
162                     
163                     CertReqMessages kur = (CertReqMessages) crmfreq.getPKIMessage().getBody().getContent();
164                     CertReqMsg certmsg;
165                     try {
166                         certmsg = kur.toCertReqMsgArray()[0];
167                     } catch(Exception e) {
168                         LOG.debug("Could not parse the revocation request. Trying to parse it as novosec generated message.");
169                         certmsg = CmpMessageHelper.getNovosecCertReqMsg(kur);
170                         LOG.debug("Succeeded in parsing the novosec generated request.");
171                     }
172                     X500Name dn = certmsg.getCertReq().getCertTemplate().getSubject();
173                     if(dn != null) {
174                         subjectDN = dn.toString();
175                     }
176                     dn = certmsg.getCertReq().getCertTemplate().getIssuer();
177                     if(dn != null) {
178                         issuerDN = dn.toString();
179                     }
180                 } else {
181                     // Client mode
182                     
183                     eecmodule = new EndEntityCertificateAuthenticationModule(admin, null, 
184                             confAlias, cmpConfiguration, authenticated, caSession, certStoreSession, authorizationSession, endEntityProfileSession, certificateProfileSession, 
185                             endEntityAccessSession, authenticationProviderSession, endEntityManagementSession);
186                     if(!eecmodule.verifyOrExtract(crmfreq.getPKIMessage(), null)) {
187                         LOG.info(eecmodule.getErrorMessage());
188                         return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, eecmodule.getErrorMessage());
189                     }
190                     oldCert = (X509Certificate) eecmodule.getExtraCert();
191                     
192                     subjectDN = oldCert.getSubjectDN().toString(); 
193                     issuerDN = oldCert.getIssuerDN().toString();
194                 }
195 
196                 if(subjectDN == null) {
197                     final String errMsg = "Cannot find a SubjectDN in the request";
198                     LOG.info(errMsg);
199                     return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, errMsg);
200                 }
201                 
202                 // Find the end entity that the certificate belongs to                
203                 if(LOG.isDebugEnabled()) {
204                     LOG.debug("Looking for an end entity with subjectDN: " + subjectDN);
205                 }
206                 EndEntityInformation endEntityInformation = null;
207                 if(issuerDN == null) {
208                     if(LOG.isDebugEnabled()) {
209                         LOG.debug("The CMP KeyUpdateRequest did not specify an issuer");
210                     }
211                     final List<EndEntityInformation> endEntityInformations = endEntityAccessSession.findUserBySubjectDN(admin, subjectDN);
212                     if (!endEntityInformations.isEmpty()) {
213                         endEntityInformation = endEntityInformations.get(0);
214                         if (endEntityInformations.size() > 1) {
215                             LOG.warn("Multiple end entities with subject DN " + subjectDN + " were found. This may lead to unexpected behavior.");
216                         }
217                     }
218                 } else {
219                     final List<EndEntityInformation> endEntityInformations = endEntityAccessSession.findUserBySubjectAndIssuerDN(admin, subjectDN, issuerDN);
220                     if (!endEntityInformations.isEmpty()) {
221                         endEntityInformation = endEntityInformations.get(0);
222                         if (endEntityInformations.size() > 1) {
223                             LOG.warn("Multiple end entities with subject DN " + subjectDN + " and issuer DN" + issuerDN
224                                     + " were found. This may lead to unexpected behavior.");
225                         }
226                     }
227                 }
228                 if(endEntityInformation == null) {
229                     final String errMsg = INTRES.getLocalizedMessage("cmp.infonouserfordn", subjectDN);
230                     LOG.info(errMsg);
231                     return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_MESSAGE_CHECK, errMsg);
232                 }
233                 if(LOG.isDebugEnabled()) {
234                     LOG.debug("Found user '" + endEntityInformation.getUsername() + "'");
235                 }
236             
237                 /*
238                  * Check the status of the certificate that should be updated.
239                  * 
240                  * RFC4210 states in ch. 5.3.5:
241                  * "[...] This message is intended to be used to request updates to existing (non-revoked and non-expired) certificates [...]" 
242                  */
243                 if(certStoreSession.isRevoked(CertTools.getIssuerDN(oldCert), CertTools.getSerialNumber(oldCert))) {
244                     String errorMessage = "Certificate for end entity with username " + endEntityInformation.getUsername() + " with subject DN " + subjectDN
245                             + " is revoked. Unable to perform key update.";
246                     if (LOG.isDebugEnabled()) {
247                         LOG.debug(errorMessage);
248                     }
249                     return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, errorMessage);
250                 }
251                 try {
252                     oldCert.checkValidity();
253                 } catch (CertificateExpiredException e) {
254                     String errorMessage = "Certificate for end entity with username " + endEntityInformation.getUsername() + " with subject DN "
255                             + subjectDN + " is expired. Unable to perform key update.";
256                     if (LOG.isDebugEnabled()) {
257                         LOG.debug(errorMessage);
258                     }
259                     return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, errorMessage);
260                 } catch(CertificateNotYetValidException e) {
261                     //A not yet valid certificate is still valid for update according to the RFC. 
262                 }
263                         
264                 // The password that should be used to obtain the new certificate
265                 String password = StringUtils.isNotEmpty(endEntityInformation.getPassword()) ? endEntityInformation.getPassword() : eecmodule.getAuthenticationString();
266                 
267                 // Set the appropriate parameters in the end entity
268                 endEntityInformation.setPassword(password);
269                 endEntityManagementSession.changeUser(admin, endEntityInformation, false);
270                 if(this.cmpConfiguration.getKurAllowAutomaticUpdate(this.confAlias)) {
271                     if(LOG.isDebugEnabled()) {
272                         LOG.debug("Setting the end entity status to 'NEW'. Username: " + endEntityInformation.getUsername());
273                     }    
274                     endEntityManagementSession.setUserStatus(admin, endEntityInformation.getUsername(), EndEntityConstants.STATUS_NEW);
275                 } else {
276                     if(LOG.isDebugEnabled()) {
277                         LOG.debug("Not setting the end entity status to 'NEW', because KUR Automatic Key Update is not allowed. Username: " + endEntityInformation.getUsername());
278                     }                    
279                 }
280                 
281                 // Set the appropriate parameters in the request
282                 crmfreq.setUsername(endEntityInformation.getUsername());
283                 crmfreq.setPassword(password);
284                 if(crmfreq.getHeader().getProtectionAlg() != null) {
285                     crmfreq.setPreferredDigestAlg(AlgorithmTools.getDigestFromSigAlg(crmfreq.getHeader().getProtectionAlg().getAlgorithm().getId()));
286                 }
287 
288                 // Check the public key, whether it is allowed to use the old keys or not.
289                 if(!this.cmpConfiguration.getKurAllowSameKey(this.confAlias)) {
290                     PublicKey certPublicKey = oldCert.getPublicKey();
291                     PublicKey requestPublicKey = crmfreq.getRequestPublicKey();
292                     if(LOG.isDebugEnabled()) {
293                         LOG.debug("Not allowing update with same key, comparing keys.");
294                         if (LOG.isTraceEnabled()) {
295                             LOG.trace("OldKey: "+certPublicKey.toString());
296                             LOG.trace("NewKey: "+requestPublicKey.toString());
297                         }
298                     }
299                     if(certPublicKey.equals(requestPublicKey)) {
300                         final String errMsg = "Invalid key. The public key in the KeyUpdateRequest is the same as the public key in the existing end entity certificate";
301                         LOG.info(errMsg);
302                         return CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_MESSAGE_CHECK, errMsg);
303                     }
304                 }
305 
306                 // Process the request
307                 resp = signSession.createCertificate(admin, crmfreq, org.ejbca.core.protocol.cmp.CmpResponseMessage.class, endEntityInformation);               
308 
309                 if (resp == null) {
310                     final String errMsg = INTRES.getLocalizedMessage("cmp.errornullresp");
311                     LOG.info(errMsg);
312                     resp = CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_MESSAGE_CHECK, errMsg);
313                 }
314             } else {
315                 final String errMsg = INTRES.getLocalizedMessage("cmp.errornocmrfreq");
316                 LOG.info(errMsg);
317                 resp = CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_MESSAGE_CHECK, errMsg);
318             }
319 
320         } catch (AuthorizationDeniedException | CADoesntExistsException | EndEntityProfileValidationException | InvalidAlgorithmException | CAOfflineException | IllegalValidityException | CertificateSerialNumberException
321                 |  CustomCertificateSerialNumberException | CryptoTokenOfflineException | IllegalKeyException
322                 | SignRequestException | SignRequestSignatureException | IllegalNameException | CertificateCreateException
323                 | CertificateRevokeException | NoSuchEndEntityException | EjbcaException | CertificateExtensionException | WaitingForApprovalException e) {
324             final String errMsg = INTRES.getLocalizedMessage(CMP_ERRORGENERAL, e.getMessage());
325             LOG.info(errMsg, e);
326             resp = CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, e.getMessage());
327         } catch (InvalidKeyException | NoSuchProviderException | NoSuchAlgorithmException e) {
328             final String errMsg = INTRES.getLocalizedMessage(CMP_ERRORGENERAL, e.getMessage());
329             LOG.info("Error while reading the public key of the extraCert attached to the CMP request");
330             LOG.info(errMsg, e);
331             resp = CmpMessageHelper.createUnprotectedErrorMessage(cmpRequestMessage, FailInfo.BAD_REQUEST, e.getMessage());
332         }
333 
334         if (LOG.isTraceEnabled()) {
335             LOG.trace("<handleMessage");
336         }
337         return resp;
338     }
