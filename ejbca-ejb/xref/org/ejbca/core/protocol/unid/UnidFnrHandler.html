32  /**
33   * Adds items to the Unid-Fnr DB.
34   * 
35   * @version $Id: UnidFnrHandler.java 28616 2018-04-03 11:51:50Z samuellb $
36   */
37  public class UnidFnrHandler implements ExtendedUserDataHandler {
38  	private static final Logger LOG = Logger.getLogger(UnidFnrHandler.class);
39  	private static final Pattern onlyDecimalDigits = Pattern.compile("^[0-9]+$");
40  	private Storage mockStorage;
41  	private UnidfnrSessionLocal unidfnrSession;
42  
43  	/**
44  	 * Used by EJBCA
45  	 */
46  	public UnidFnrHandler() {
47  		super();
48  		mockStorage = null;
49  		unidfnrSession = new EjbLocalHelper().getUnidfnrSession();
50  	}
51  	/**
52  	 * Used by unit test.
53  	 * @param mockStorage Emulates the {@link UnidfnrSessionLocal#stroreUnidFnrData(String, String)} call.
54  	 */
55  	public UnidFnrHandler(final Storage mockStorage) {
56  		super();
57  		this.mockStorage = mockStorage;
58  		unidfnrSession = null;
59  	}
60  	
61  	@Override
62  	public RequestMessage processRequestMessage(RequestMessage req, final String certificateProfileName) throws HandlerException {
63  	    final X500Name dn = req.getRequestX500Name();
64  		if (LOG.isDebugEnabled()) {
65  			LOG.debug(">processRequestMessage:'"+dn+"' and '"+certificateProfileName+"'");
66  		}
67  		final String unidPrefix = getPrefixFromCertProfileName(certificateProfileName);
68  		if ( unidPrefix==null ) {
69  			return req;
70  		}
71          final List<ASN1ObjectIdentifier> asn1ObjectIdentifiers = Arrays.asList(dn.getAttributeTypes());
72  		X500NameBuilder nameBuilder = new X500NameBuilder(new CeSecoreNameStyle());
73  		boolean changed = false;
74  		for (final ASN1ObjectIdentifier asn1ObjectIdentifier : asn1ObjectIdentifiers) {
75  			if (asn1ObjectIdentifier.equals(CeSecoreNameStyle.SERIALNUMBER) ) {
76  			    RDN[] rdns = dn.getRDNs(asn1ObjectIdentifier);
77  			    String value = rdns[0].getFirst().getValue().toString();
78  				final String newSerial = storeUnidFrnAndGetNewSerialNr(value, unidPrefix);
79  				if ( newSerial!=null ) {
80  					nameBuilder.addRDN(asn1ObjectIdentifier, newSerial);
81  					changed = true;
82  				}
83  			} else {
84  			    nameBuilder.addRDN(dn.getRDNs(asn1ObjectIdentifier)[0].getFirst());
85  			}
86  		}
87  		if(changed) {
88  		    req = new RequestMessageSubjectDnAdapter( req, nameBuilder.build());
89  		}
90  		return req;
91  	}
92  	private static boolean hasOnlyDecimalDigits(String s, int first, int last) {
93  		return hasOnlyDecimalDigits( s.substring(first, last));
94  	}
95  	private static boolean hasOnlyDecimalDigits(String s) {
96  		return onlyDecimalDigits.matcher(s).matches();
97  	}
98  	private String getPrefixFromCertProfileName(String certificateProfileName) {
99  		if ( certificateProfileName.length()<10 ) {
100 			return null;
101 		}
102 		if ( certificateProfileName.charAt(4)!='-' ) {
103 			return null;
104 		}
105 		if ( certificateProfileName.charAt(9)!='-' ) {
106 			return null;
107 		}
108 		if ( !hasOnlyDecimalDigits(certificateProfileName, 0, 4) ) {
109 			return null;
110 		}
111 		if ( !hasOnlyDecimalDigits(certificateProfileName, 5, 9) ) {
112 			return null;
113 		}
114 		return certificateProfileName.substring(0, 10);
115 	}
116 	/**
117 	 * @param inputSerialNr SN of subject DN in the incoming request
118 	 * @param unidPrefix Prefix of the unid
119 	 * @return the serial number of the subject DN of the certificate that will be created. Null if the format of the SN is not fnr-lra.
120 	 * Returning null means that the handler should not do anything (SN in DN not changed and nothing stored to DB).
121 	 * @throws HandlerException if unid-fnr can't be stored in DB. This will prevent any certificate to be created.
122 	 */
123 	private String storeUnidFrnAndGetNewSerialNr(final String inputSerialNr, final String unidPrefix) throws HandlerException {
124 	    
125         if (unidfnrSession == null && mockStorage == null) {
126             throw new HandlerException("Unidfnr session bean is null!");
127         }	    
128 	    
129 		if ( inputSerialNr.length()!=17 ) {
130 			return null;
131 		}
132 		if ( inputSerialNr.charAt(11)!='-' ) {
133 			return null;
134 		}
135 		final String fnr = inputSerialNr.substring(0, 11);
136 		if ( !hasOnlyDecimalDigits(fnr) ) {
137 			return null;
138 		}
139 		final String lra = inputSerialNr.substring(12);
140 		if ( !hasOnlyDecimalDigits(lra) ) {
141 			return null;
142 		}
143 		final String random = new LettersAndDigitsPasswordGenerator().getNewPassword(6, 6);
144 		final String unid = unidPrefix + lra + random;
145 		storeUnidFnrData(unid, fnr);
146 		return unid;
147 	}
148 	
149 	public void storeUnidFnrData(final String unid, final String fnr) throws HandlerException {
150 	    if (unidfnrSession != null) {
151 	        unidfnrSession.stroreUnidFnrData(unid, fnr);
152 	    } else if (mockStorage != null) {
153 	        mockStorage.storeIt(unid, fnr);
154 	    } else {
155 	        throw new IllegalStateException();
156 	    }
157     }
158 	
159 	
160 	/**
161 	 * To be implemented by unit test.
162 	 */
163 	public interface Storage {
164 		/**
165 		 * @param unid ID
166 		 * @param fnr FNR
167 		 * @throws HandlerException Fail
168 		 */
169 		void storeIt(String unid, String fnr) throws HandlerException;
170 	}
