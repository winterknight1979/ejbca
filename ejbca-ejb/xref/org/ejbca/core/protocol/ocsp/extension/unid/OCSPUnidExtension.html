42  /** ASN.1 OCSP extension used to map a UNID to a Fnr, OID for this extension is 2.16.578.1.16.3.2

81  	@Override
82  	public Map<ASN1ObjectIdentifier, Extension> process(X509Certificate[] requestCertificates, String remoteAddress, String remoteHost,
83              X509Certificate cert, CertificateStatus status, InternalKeyBinding internalKeyBinding) {
84  
85  	    String serialNumber = null;
86          String fnr = null;
87          
88          // Check authorization first
89          if (!checkAuthorization(requestCertificates, remoteAddress, remoteHost, internalKeyBinding.getTrustedCertificateReferences())) {
90          	errCode = UnidFnrOCSPExtensionCode.ERROR_UNAUTHORIZED.getValue();
91          	return null;
92          }
93          // If the certificate is revoked, we must not return an FNR
94          if (status != null) {
95              errCode = UnidFnrOCSPExtensionCode.ERROR_CERT_REVOKED.getValue();
96              return null;
97          }
98          
99          // The Unid is in the DN component serialNumber
100         serialNumber = CertTools.getPartFromDN(cert.getSubjectDN().getName(), "SN");
101         if (serialNumber != null) {
102             if (log.isDebugEnabled()) {
103                 log.debug("Found serialNumber: " + serialNumber);
104             }
105             String iMsg = intres.getLocalizedMessage("ocsp.receivedunidreq", remoteAddress, remoteHost, serialNumber);
106             log.info(iMsg);
107 
108             // Make sure unidfnrSession is loaded properly in all environments before using it.
109             if (unidfnrSession == null) {
110                 unidfnrSession = new EjbLocalHelper().getUnidfnrSession();
111             }
112             fnr = unidfnrSession.fetchUnidFnrData(serialNumber);
113         } else {
114             String errMsg = intres.getLocalizedMessage("ocsp.errorunidnosnindn", cert.getSubjectDN().getName());
115             log.error(errMsg);
116             errCode = UnidFnrOCSPExtensionCode.ERROR_NO_SERIAL_IN_DN.getValue();
117             return null;
118         }
119         
120         if (fnr == null) {
121 			String errMsg = intres.getLocalizedMessage("ocsp.errorunidnosnmapping", serialNumber);
122             log.error(errMsg);
123         	errCode = UnidFnrOCSPExtensionCode.ERROR_NO_FNR_MAPPING.getValue();
124         	return null;
125         }
126 
127         String successMsg = intres.getLocalizedMessage("ocsp.returnedunidresponse", remoteAddress, remoteHost, serialNumber);
128         log.info(successMsg);
129         
130         return generateUnidFnrOCSPResponce(fnr);
131 	}

141 	private boolean checkAuthorization(X509Certificate[] certificates, String remoteAddress, String remoteHost, List<InternalKeyBindingTrustEntry> bindingTrustEntries) {
142         if (certificates == null) {
143     		String errMsg = intres.getLocalizedMessage("ocsp.errornoclientauth", remoteAddress, remoteHost);
144             log.error(errMsg);
145             return false;
146         }
147         // The certificate of the entity is nr 0
148         X509Certificate cert = certificates[0];
149         if (cert == null) {
150     		String errMsg = intres.getLocalizedMessage("ocsp.errornoclientauth", remoteAddress, remoteHost);
151             log.error(errMsg);
152             return false;
153         }
154         
155         // Check if the certificate is authorized to access the Fnr
156         boolean serialExists = false;
157         final String issuerDN = CertTools.getIssuerDN(cert);
158         
159         // Make sure caSession is loaded properly in all environments before using it.
160         if (caSession == null) {
161             caSession = new EjbLocalHelper().getCaSession();
162         }
163         
164         final CAInfo caInfo = caSession.getCAInfoInternal(issuerDN.hashCode());
165         
166         for (final InternalKeyBindingTrustEntry bindingTrustEntry : bindingTrustEntries) {
167             // Match
168             final BigInteger trustEntrySerial = bindingTrustEntry.fetchCertificateSerialNumber(); 
169             if ((trustEntrySerial == null || trustEntrySerial.equals(cert.getSerialNumber())) && caInfo.getCAId() == bindingTrustEntry.getCaId()) {
170                 serialExists = true;
171             }
172         }
173         
174         if (serialExists) {
175             // If we found in the hashmap the same key with issuer and serialnumber, we know we got it. 
176             // Just verify it as well to be damn sure
177             final Certificate cacert = caInfo.getCertificateChain().get(0);
178             try {
179                 cert.verify(cacert.getPublicKey());
180             } catch (Exception e) {
181         		String errMsg = intres.getLocalizedMessage("ocsp.errorverifycert");
182                 log.error(errMsg, e);
183                 return false;
184             }
185             // If verify was successful we know if was good!
186             return true;
187         }
188         
189 		String errMsg = intres.getLocalizedMessage("ocsp.erroruntrustedclientauth", remoteAddress, remoteHost);
190         log.error(errMsg);
191 		return false;
192 	}

194     private Map<ASN1ObjectIdentifier, Extension> generateUnidFnrOCSPResponce(final String fnr) {
195         FnrFromUnidExtension ext = new FnrFromUnidExtension(fnr);
196         HashMap<ASN1ObjectIdentifier, Extension> unidOCSPResponse = new HashMap<ASN1ObjectIdentifier, Extension>();
197         try {
198             unidOCSPResponse.put(FnrFromUnidExtension.FnrFromUnidOid, new Extension(FnrFromUnidExtension.FnrFromUnidOid, false, new DEROctetString(ext)));
199         } catch (IOException e) {
200             throw new IllegalStateException("Unexpected IOException caught.", e);
201         }
202         return unidOCSPResponse;
203     }
