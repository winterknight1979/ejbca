34  /** This class gives facilities to populate user data with default values from profile.
35   *
36   * @version $Id: EndEntityInformationFiller.java 28128 2018-01-30 07:09:34Z samuellb $
37   */
38  public class EndEntityInformationFiller {
39  
40      /** For log purpose. */
41      private static final Logger log = Logger.getLogger(EndEntityInformationFiller.class.getName());
42  
43      /** This method fill user data with the default values from the specified profile.
44       * 
45       * @param userData user data.
46       * @param profile user associated profile.
47       * @return update user.
48       */
49      public static EndEntityInformation fillUserDataWithDefaultValues(final EndEntityInformation userData, final EndEntityProfile profile) {
50  
51      	
52      	if (StringUtils.isEmpty(userData.getUsername())) {
53          	userData.setUsername(profile.getUsernameDefault());
54          }
55      	if (userData.getSendNotification()==false) {
56      		if(StringUtils.isNotEmpty(profile.getValue(EndEntityProfile.SENDNOTIFICATION, 0))) {
57      			final Boolean isSendNotification = Boolean.valueOf(profile.getValue(EndEntityProfile.SENDNOTIFICATION, 0));
58      			userData.setSendNotification(isSendNotification.booleanValue());    			
59      		}
60          }
61      	if (StringUtils.isEmpty(userData.getEmail())) {
62  			final String email = profile.getValue(EndEntityProfile.EMAIL, 0);
63  			if (StringUtils.isNotEmpty(email) && email.indexOf("@") > 0) {
64  				userData.setEmail(email);
65  			}
66  		}
67          //Batch generation (clear text pwd storage) is only active when password 
68          //is not empty so is not necessary to do something here
69          if (StringUtils.isEmpty(userData.getPassword())) {
70              // check if the password is autogenerated
71          	if(!profile.useAutoGeneratedPasswd()) {
72          		userData.setPassword(profile.getValue(EndEntityProfile.PASSWORD, 0));        		
73          	}
74          }
75          
76          // Processing Subject DN values
77          String subjectDN = userData.getDN();
78          subjectDN = mergeSubjectDnWithDefaultValues(subjectDN, profile, userData.getEmail());
79          userData.setDN(subjectDN);
80          String subjectAltName = userData.getSubjectAltName();
81          // Processing Subject Altname values
82          subjectAltName = mergeSubjectAltNameWithDefaultValues(subjectAltName, profile, userData.getEmail());
83          userData.setSubjectAltName(subjectAltName);
84          if (userData.getType().getHexValue()==EndEntityTypes.INVALID.hexValue()) {
85          	if(StringUtils.isNotEmpty(profile.getValue(EndEntityProfile.FIELDTYPE, 0))){
86          	    final int type = Integer.parseInt(profile.getValue(EndEntityProfile.FIELDTYPE, 0));
87          		userData.setType(new EndEntityType(type));
88          	}
89          }
90          return userData;
91      }

93      /** This method merge subject DN with data from End entity profile.
94       * @param subjectDN user Distinguished Name.
95       * @param profile user associated profile.
96       * @param entityEmail entity email.
97       * @return updated DN.
98       */
99      private static String mergeSubjectDnWithDefaultValues(String subjectDN, EndEntityProfile profile, 
100             String entityEmail) {
101         DistinguishedName profiledn;
102         DistinguishedName userdn;
103         try {
104         	userdn = new DistinguishedName(subjectDN);
105 		} catch (InvalidNameException ine) {
106 			log.debug(subjectDN,ine);
107 			throw new RuntimeException(ine);
108 		}
109         int numberofsubjectdnfields = profile.getSubjectDNFieldOrderLength();
110         List<Rdn> rdnList = new ArrayList<Rdn>(numberofsubjectdnfields);
111         int[] fielddata = null;
112         String value;
113         //Build profile's DN
114         for (int i = 0; i < numberofsubjectdnfields; i++) {
115         	value=null;
116 			fielddata = profile.getSubjectDNFieldsInOrder(i);
117 			String parameter = DNFieldExtractor.getFieldComponent(
118 					DnComponents.profileIdToDnId(fielddata[EndEntityProfile.FIELDTYPE]),
119 					DNFieldExtractor.TYPE_SUBJECTDN);
120 			value = profile.getValue(fielddata[EndEntityProfile.FIELDTYPE], 0);
121 			if (value != null) {
122 				value = value.trim();
123 				if (!value.equals("")) {					
124 					try {
125 						parameter = StringUtils.replace(parameter, "=", "");
126 						rdnList.add(fielddata[EndEntityProfile.NUMBER],new Rdn(parameter,value));
127 					}catch(InvalidNameException ine) {
128 						log.debug("InvalidNameException while creating new Rdn with parameter "+ parameter + " and value " + value,ine);
129 						throw new RuntimeException(ine);
130 					}
131 					
132 				}
133 			}
134 		}
135         profiledn = new DistinguishedName(rdnList);
136 
137         Map<String, String> dnMap = new HashMap<String, String>();
138         if (profile.getUse(DnComponents.DNEMAILADDRESS, 0)) {
139             dnMap.put(DnComponents.DNEMAILADDRESS, entityEmail);
140         }
141         
142         return  CertTools.stringToBCDNString(profiledn.mergeDN(userdn, true, dnMap).toString());
143     }

145     /**
146      * This method merge subject Alt name with data from End entity profile.
147      * @param subjectAltName user subject alt name.
148      * @param profile user associated profile.
149      * @param entityEmail entity email field
150      * @return updated subject alt name
151      */
152     private static String mergeSubjectAltNameWithDefaultValues(String subjectAltName, EndEntityProfile profile, String entityEmail) {
153         DistinguishedName profileAltName;
154         DistinguishedName userAltName;
155         try {
156         	if(subjectAltName==null) {
157         		subjectAltName = "";
158         	}
159         	userAltName = new DistinguishedName(subjectAltName);
160 		} catch (InvalidNameException ine) {
161 			log.debug(subjectAltName,ine);
162 			throw new RuntimeException(ine);
163 		}
164         int numberofsubjectAltNamefields = profile.getSubjectAltNameFieldOrderLength();
165         List<Rdn> rdnList = new ArrayList<Rdn>(numberofsubjectAltNamefields);
166         int[] fielddata = null;
167         String value;
168         //Build profile's Alt Name
169         for (int i = 0; i < numberofsubjectAltNamefields; i++) {
170         	value=null;
171 			fielddata = profile.getSubjectAltNameFieldsInOrder(i);
172 			String parameter = DNFieldExtractor.getFieldComponent(
173 					DnComponents.profileIdToDnId(fielddata[EndEntityProfile.FIELDTYPE]),
174 					DNFieldExtractor.TYPE_SUBJECTALTNAME);
175 			value = profile.getValue(fielddata[EndEntityProfile.FIELDTYPE], fielddata[EndEntityProfile.NUMBER]);
176 			if (value != null) {
177 				value = value.trim();
178 				if (!value.equals("")) {					
179 					try {
180 						parameter = StringUtils.replace(parameter, "=", "");
181 						rdnList.add(fielddata[EndEntityProfile.NUMBER],new Rdn(parameter,value));
182 					}catch(InvalidNameException ine) {
183 						log.debug("InvalidNameException while creating new Rdn with parameter "+ parameter + " and value " + value,ine);
184 						throw new RuntimeException(ine);
185 					}
186 					
187 				}
188 			}
189 		}
190         profileAltName = new DistinguishedName(rdnList);
191 
192         Map<String, String> dnMap = new HashMap<String, String>();
193         if (profile.getUse(DnComponents.RFC822NAME, 0)) {
194             dnMap.put(DnComponents.RFC822NAME, entityEmail);
195         }
196 
197         return  profileAltName.mergeDN(userAltName, true, dnMap).toString();
198     }
