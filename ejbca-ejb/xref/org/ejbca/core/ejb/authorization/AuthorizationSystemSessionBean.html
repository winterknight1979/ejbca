106     @Override
107     public Map<String,String> getAllResources(final AuthenticationToken authenticationToken, final boolean ignoreLimitations) {
108         final Map<String,String> authorizedResources = new HashMap<>();
109         try {
110             final HashMap<String, Boolean> accessRules = authorizationSession.getAccessAvailableToAuthenticationToken(authenticationToken);
111             for (final Entry<String,String> entry : getAllResources(ignoreLimitations).entrySet()) {
112                 if (AccessRulesHelper.hasAccessToResource(accessRules, entry.getKey())) {
113                     authorizedResources.put(AccessRulesHelper.normalizeResource(entry.getKey()), AccessRulesHelper.normalizeResource(entry.getValue()));
114                 }
115             }
116         } catch (AuthenticationFailedException e) {
117             if (log.isDebugEnabled()) {
118                 log.debug("Failed to get resources due to authentication failure: " + e.getMessage());
119             }
120         }
121         return authorizedResources;
122     }

124     @Override
125     public Map<String,String> getAllResources(final boolean ignoreLimitations) {
126         final Map<Integer, String> caIdToNameMap = caSession.getCAIdToNameMap();
127         final Map<Integer, String> kvIdToNameMap = keyValidatorSession.getKeyValidatorIdToNameMap();
128         final Map<Integer, String> eepIdToNameMap = endEntityProfileSession.getEndEntityProfileIdToNameMap();
129         final Map<Integer, String> userDataSourceIdToNameMap = userDataSourceSession.getUserDataSourceIdToNameMap();
130         final Map<Integer,String> cryptoTokenIdToNameMap = cryptoTokenSession.getCryptoTokenIdToNameMap();
131         final GlobalConfiguration globalConfiguration = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
132         final boolean endEntityProfileLimitationsEnabled = ignoreLimitations || globalConfiguration.getEnableEndEntityProfileLimitations();
133         final boolean hardTokenIssuingEnabled = ignoreLimitations || globalConfiguration.getIssueHardwareTokens();
134         final boolean keyRecoveryEnabled = ignoreLimitations || globalConfiguration.getEnableKeyRecovery();
135         final Map<String, Map<String,String>> categorizedAccessRules = getAllResourceAndResourceNamesByCategory(
136                 endEntityProfileLimitationsEnabled, hardTokenIssuingEnabled, keyRecoveryEnabled,
137                 Arrays.asList(EjbcaConfiguration.getCustomAvailableAccessRules()), eepIdToNameMap, userDataSourceIdToNameMap, cryptoTokenIdToNameMap, caIdToNameMap, kvIdToNameMap);
138         final Map<String,String> ret = new HashMap<>();
139         for (final Map<String,String> acessRuleMap : categorizedAccessRules.values()) {
140             ret.putAll(acessRuleMap);
141         }
142         return ret;
143     }

146     public Map<String,Map<String,String>> getAllResourceAndResourceNamesByCategory() {
147         final Map<Integer, String> caIdToNameMap = caSession.getCAIdToNameMap();
148         final Map<Integer, String> kvIdToNameMap = keyValidatorSession.getKeyValidatorIdToNameMap();
149         final Map<Integer, String> eepIdToNameMap = endEntityProfileSession.getEndEntityProfileIdToNameMap();
150         final Map<Integer, String> userDataSourceIdToNameMap = userDataSourceSession.getUserDataSourceIdToNameMap();
151         final Map<Integer,String> cryptoTokenIdToNameMap = cryptoTokenSession.getCryptoTokenIdToNameMap();
152         final GlobalConfiguration globalConfiguration = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
153         final boolean endEntityProfileLimitationsEnabled = globalConfiguration.getEnableEndEntityProfileLimitations();
154         final boolean hardTokenIssuingEnabled = globalConfiguration.getIssueHardwareTokens();
155         final boolean keyRecoveryEnabled = globalConfiguration.getEnableKeyRecovery();
156         return getAllResourceAndResourceNamesByCategory(
157                 endEntityProfileLimitationsEnabled, hardTokenIssuingEnabled, keyRecoveryEnabled,
158                 Arrays.asList(EjbcaConfiguration.getCustomAvailableAccessRules()), eepIdToNameMap, userDataSourceIdToNameMap, cryptoTokenIdToNameMap, caIdToNameMap, kvIdToNameMap);
159     }

161     private Map<String,Map<String,String>> getAllResourceAndResourceNamesByCategory(boolean endEntityProfileLimitationsEnabled,
162             boolean hardTokenIssuingEnabled, boolean keyRecoveryEnabled,
163             Collection<String> customAccessRules, Map<Integer,String> eepIdToNameMap, Map<Integer, String> userDataSourceIdToNameMap,
164             Map<Integer,String> cryptoTokenIdToNameMap, Map<Integer,String> caIdToNameMap, Map<Integer,String> kvIdToNameMap) {
165         final Map<String,Map<String,String>> ret = new LinkedHashMap<>();
166         // Role based access rules
167         final Map<String,String> accessRulesRoleBased = new LinkedHashMap<>();
168         accessRulesRoleBased.put(AccessRulesConstants.ROLE_ADMINISTRATOR, AccessRulesConstants.ROLE_ADMINISTRATOR);
169         accessRulesRoleBased.put(StandardRules.ROLE_ROOT.resource(), StandardRules.ROLE_ROOT.resource());
170         ret.put("ROLEBASEDACCESSRULES", accessRulesRoleBased);
171         // Standard rules (including custom access rules)
172         final Map<String,String> accessRulesRegular = new LinkedHashMap<>();
173         for (final String resource : AccessRulesConstants.STANDARDREGULARACCESSRULES) {
174             accessRulesRegular.put(resource, resource);
175         }
176         if (hardTokenIssuingEnabled) {
177             for (final String resource : AccessRulesConstants.HARDTOKENACCESSRULES) {
178                 accessRulesRegular.put(resource, resource);
179             }
180             accessRulesRegular.put(AccessRulesConstants.REGULAR_VIEWHARDTOKENS, AccessRulesConstants.REGULAR_VIEWHARDTOKENS);
181             accessRulesRegular.put(AccessRulesConstants.REGULAR_VIEWPUKS, AccessRulesConstants.REGULAR_VIEWPUKS);
182         }
183         if (keyRecoveryEnabled) {
184             accessRulesRegular.put(AccessRulesConstants.REGULAR_KEYRECOVERY, AccessRulesConstants.REGULAR_KEYRECOVERY);
185         }
186         for (final String resource : customAccessRules) {
187             if (!StringUtils.isEmpty(resource.trim())) {
188                 accessRulesRegular.put(resource.trim(), resource.trim());
189             }
190         }
191         ret.put("REGULARACCESSRULES", accessRulesRegular);
192         // Insert CA access rules
193         final Map<String,String> accessRulesCaAccess = new LinkedHashMap<>();
194         accessRulesCaAccess.put(StandardRules.CAACCESSBASE.resource(), StandardRules.CAACCESSBASE.resource());
195         for (final int caId : caIdToNameMap.keySet()) {
196             String caName = caIdToNameMap.get(caId);
197             if (caName==null) {
198                 caName = String.valueOf(caId);
199             }
200             accessRulesCaAccess.put(StandardRules.CAACCESS.resource() + caId, StandardRules.CAACCESS.resource() + caName);
201         }
202         ret.put("CAACCESSRULES", accessRulesCaAccess);
203         
204         final Map<String,String> accessRulesKeyValidatorAccess = new LinkedHashMap<>();
205         accessRulesKeyValidatorAccess.put(StandardRules.VALIDATORACCESSBASE.resource(), StandardRules.VALIDATORACCESSBASE.resource());
206         String kvName;
207         for (final int kvId : kvIdToNameMap.keySet()) {
208             kvName = kvIdToNameMap.get(kvId);
209             if (kvName==null) {
210                 kvName = String.valueOf(kvId);
211             }
212             accessRulesKeyValidatorAccess.put(StandardRules.VALIDATORACCESS.resource() + kvId, StandardRules.VALIDATORACCESS.resource() + kvName);
213         }
214         ret.put("VALIDATORACCESSRULES", accessRulesKeyValidatorAccess);
215         
216         // End entity profile rules
217         if (endEntityProfileLimitationsEnabled) {
218             final Map<String,String> accessRulesEepAccess = new LinkedHashMap<>();
219             // Add most basic rule if authorized to it.
220             accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEBASE, AccessRulesConstants.ENDENTITYPROFILEBASE);
221             // Add all authorized End Entity Profiles
222             for (final int eepId : eepIdToNameMap.keySet()) {
223                 String eepName = eepIdToNameMap.get(eepId);
224                 if (eepName==null) {
225                     eepName = String.valueOf(eepId);
226                 }
227                 // Administrator is authorized to this End Entity Profile, add it.
228                 accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepId, AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepName);
229                 for (final String subResource : AccessRulesConstants.ENDENTITYPROFILE_ENDINGS) {
230                     accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepId + subResource,
231                             AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepName + subResource);
232                 }
233                 if (hardTokenIssuingEnabled) {
234                     accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepId + AccessRulesConstants.HARDTOKEN_RIGHTS,
235                             AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepName + AccessRulesConstants.HARDTOKEN_RIGHTS);
236                     accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepId + AccessRulesConstants.HARDTOKEN_PUKDATA_RIGHTS,
237                             AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepName + AccessRulesConstants.HARDTOKEN_PUKDATA_RIGHTS);
238                 }
239                 if (keyRecoveryEnabled) {
240                     accessRulesEepAccess.put(AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepId + AccessRulesConstants.KEYRECOVERY_RIGHTS,
241                             AccessRulesConstants.ENDENTITYPROFILEPREFIX + eepName + AccessRulesConstants.KEYRECOVERY_RIGHTS);
242                 }
243             }
244             ret.put("ENDENTITYPROFILEACCESSR", accessRulesEepAccess);
245         }
246         // Crypto token rules
247         final Map<String,String> accessRulesCtAccess = new LinkedHashMap<>();
248         for (final CryptoTokenRules rule : CryptoTokenRules.values()) {
249             accessRulesCtAccess.put(rule.resource(), rule.resource());
250         }
251         for (final int cryptoTokenId : cryptoTokenIdToNameMap.keySet()) {
252             String cryptoTokenName = cryptoTokenIdToNameMap.get(cryptoTokenId);
253             if (cryptoTokenName==null) {
254                 cryptoTokenName = String.valueOf(cryptoTokenId);
255             }
256             for (final CryptoTokenRules rule : CryptoTokenRules.values()) {
257                 if (!rule.equals(CryptoTokenRules.BASE) && !rule.equals(CryptoTokenRules.MODIFY_CRYPTOTOKEN) && !rule.equals(CryptoTokenRules.DELETE_CRYPTOTOKEN)) {
258                     accessRulesCtAccess.put(rule.resource() + "/" + cryptoTokenId, rule.resource() + "/" + cryptoTokenName);
259                 }
260             }
261         }
262         ret.put("CRYPTOTOKENACCESSRULES", accessRulesCtAccess);
263         // Insert User data source access rules
264         final Map<String,String> accessRulesUdsAccess = new LinkedHashMap<>();
265         accessRulesUdsAccess.put(AccessRulesConstants.USERDATASOURCEBASE, AccessRulesConstants.USERDATASOURCEBASE);
266         for (final int userDataSourceId : userDataSourceIdToNameMap.keySet()) {
267             String userDataSourceName = userDataSourceIdToNameMap.get(userDataSourceId);
268             if (userDataSourceName==null) {
269                 userDataSourceName = String.valueOf(userDataSourceId);
270             }
271             accessRulesUdsAccess.put(AccessRulesConstants.USERDATASOURCEPREFIX + userDataSourceId + AccessRulesConstants.UDS_FETCH_RIGHTS,
272                     AccessRulesConstants.USERDATASOURCEPREFIX + userDataSourceName + AccessRulesConstants.UDS_FETCH_RIGHTS);
273             accessRulesUdsAccess.put(AccessRulesConstants.USERDATASOURCEPREFIX + userDataSourceId + AccessRulesConstants.UDS_REMOVE_RIGHTS,
274                     AccessRulesConstants.USERDATASOURCEPREFIX + userDataSourceName + AccessRulesConstants.UDS_REMOVE_RIGHTS);
275         }
276         ret.put("USERDATASOURCEACCESSRULES", accessRulesUdsAccess);
277         // Insert plugin rules
278         for (final AccessRulePlugin accessRulePlugin : ServiceLoader.load(AccessRulePlugin.class)) {
279             Map<String,String> accessRulesInCategory = ret.get(accessRulePlugin.getCategory());
280             if (accessRulesInCategory==null) {
281                 accessRulesInCategory = new LinkedHashMap<>();
282                 ret.put(accessRulePlugin.getCategory(), accessRulesInCategory);
283             }
284             for (final Entry<String,String> resourceAndName : accessRulePlugin.getRules().entrySet()) {
285                 accessRulesInCategory.put(resourceAndName.getKey(), resourceAndName.getValue());
286             }
287         }
288         return ret;
289     }

291     @TransactionAttribute(TransactionAttributeType.REQUIRED)
292     @Override
293     public boolean initializeAuthorizationModule() {
294         // Since RoleData might be empty when this is invoked on a not yet upgraded system, we also check if there are any CAs present.
295         // (The CLI admin might have been reconfigured or removed in the old authorization system and we don't want to change this.)
296         if (roleDataSession.getAllRoles().isEmpty() && caSession.getAllCaIds().isEmpty()) {
297             log.info("No roles or CAs exist, intializing Super Administrator Role with default CLI user.");
298             // Create "Super Administrator Role" (with roleId "1" to ensure that upgraded installations still have the same roleId)
299             final Role roleToPersist = new Role(null, SUPERADMIN_ROLE, Arrays.asList(StandardRules.ROLE_ROOT.resource()), null);
300             roleToPersist.setRoleId(1);
301             final Role role = roleDataSession.persistRole(roleToPersist);
302             // We won't create any RoleMember for a Super Admin certificate here
303             // Add CLI user role member
304             final String username = EjbcaConfiguration.getCliDefaultUser();
305             roleMemberDataSession.persistRoleMember(new RoleMember(CliAuthenticationTokenMetaData.TOKEN_TYPE,
306                     RoleMember.NO_ISSUER, CliUserAccessMatchValue.USERNAME.getNumericValue(), AccessMatchType.TYPE_EQUALCASE.getNumericValue(),
307                     username, role.getRoleId(), null));
308             // Add CLI user end entity
309             final UserData userData = new UserData(username, EjbcaConfiguration.getCliDefaultPassword(), false, "UID=" + username, 0, null, null, null, 0,
310                     EndEntityConstants.EMPTY_END_ENTITY_PROFILE, 0, 0, 0, null);
311             userData.setStatus(EndEntityConstants.STATUS_GENERATED);
312             if (entityManager.find(UserData.class, username)==null) {
313                 entityManager.persist(userData);
314             }
315             return true;
316         }
317         log.info("Roles or CAs exist, not intializing " + SUPERADMIN_ROLE);
318         return false;
319     }

321     /*
322      * Note:
323      * This is expected to be invoked by the CLI user during "ant install" or similar operation.
324      * The StartupSingletonBean should have initialized by the SUPERADMIN_ROLE at startup on a fresh system and authorized the CLI user.
325      */
326     @TransactionAttribute(TransactionAttributeType.REQUIRED)
327     @Override
328     public boolean initializeAuthorizationModuleWithSuperAdmin(AuthenticationToken authenticationToken, int caId, String superAdminCN) throws AuthorizationDeniedException {
329         if (roleDataSession.getAllRoles().isEmpty() && caSession.getAllCaIds().isEmpty()) {
330             log.info("The Role '" + SUPERADMIN_ROLE + "' has not been initialized. Cannot add SuperAdmin '" + superAdminCN + "'.");
331             return false;
332         }
333         final Role role = roleSession.getRole(authenticationToken, null, SUPERADMIN_ROLE);
334         if (role==null) {
335             log.info("The Role '" + SUPERADMIN_ROLE + "' does not exist. Cannot add SuperAdmin '" + superAdminCN + "'.");
336             return false;
337         }
338         // We don't care if the caller has done this before. If the caller is authorized we comply.
339         roleMemberSession.persist(authenticationToken, new RoleMember(X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE,
340                 caId, X500PrincipalAccessMatchValue.WITH_COMMONNAME.getNumericValue(), AccessMatchType.TYPE_EQUALCASE.getNumericValue(),
341                 superAdminCN, role.getRoleId(), null));
342         return true;
343     }

345     @Override
346     public AccessSet getAccessSetForAuthToken(AuthenticationToken authenticationToken) throws AuthenticationFailedException {
347         final HashMap<String, Boolean> accessRules = authorizationSession.getAccessAvailableToAuthenticationToken(authenticationToken);
348         final Set<String> allResources = new HashSet<>(getAllResources(false).keySet());
349         // Since we no longer support the recursive rule in AccessSets from EJBCA 6.8.0 we also need to include non-configurable access rules
350         // ..but this is kind of theoretical since we currently don't support any of these operations from the RA
351         allResources.add(StandardRules.CAADD.resource());
352         allResources.add(StandardRules.CAREMOVE.resource());
353         return AccessSet.fromAccessRules(accessRules, allResources);
354     }
