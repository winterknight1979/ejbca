87      @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
88      @Override
89      public AuthenticationToken getAdmin(final boolean allowNonAdmins, final X509Certificate cert) throws AuthorizationDeniedException {
90          final Set<X509Certificate> credentials = new HashSet<>();
91          credentials.add(cert);
92          final AuthenticationSubject subject = new AuthenticationSubject(null, credentials);
93          final AuthenticationToken admin = authenticationSession.authenticate(subject);
94          
95          if (!raMasterApiProxyBean.isAuthorizedNoLogging(raRestAuthCheckToken, AccessRulesConstants.REGULAR_PEERPROTOCOL_REST)) {
96              throw new AuthorizationDeniedException("REST resources is not authorized for this Peer connection");
97          }
98          if ((admin != null) && (!allowNonAdmins)) {
99              if(!raMasterApiProxyBean.isAuthorizedNoLogging(admin, AccessRulesConstants.ROLE_ADMINISTRATOR)) {
100                 final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", AccessRulesConstants.ROLE_ADMINISTRATOR, null);
101                 throw new AuthorizationDeniedException(msg);
102             }
103         } else if (admin == null) {
104             final String msg = intres.getLocalizedMessage("authentication.failed", "No admin authenticated for certificate with serialNumber " +
105                     CertTools.getSerialNumber(cert) + " and issuerDN '" + CertTools.getIssuerDN(cert)+"'.");
106             throw new AuthorizationDeniedException(msg);
107         }
108         return admin;
109     }

111     @Override
112     public EndEntityInformation convertToEndEntityInformation(AuthenticationToken authenticationToken, EnrollPkcs10CertificateRequest enrollcertificateRequest)
113             throws AuthorizationDeniedException, EndEntityProfileNotFoundException, EjbcaException, CertificateProfileDoesNotExistException, CADoesntExistsException {
114 
115         EndEntityInformation endEntityInformation = new EndEntityInformation();
116         ExtendedInformation extendedInformation = new ExtendedInformation();
117 
118         endEntityInformation.setExtendedInformation(extendedInformation);
119 
120         CAInfo caInfo = getCAInfo(enrollcertificateRequest.getCertificateAuthorityName(), authenticationToken);
121         if (caInfo == null) {
122             String errorMessage = "CA with name \"" + enrollcertificateRequest.getCertificateAuthorityName() + "\" doesn't exist";
123             throw new CADoesntExistsException(errorMessage);
124         }
125         endEntityInformation.setCAId(caInfo.getCAId());
126 
127         int certificateProfileId = getCertificateProfileId(enrollcertificateRequest.getCertificateProfileName());
128         if (certificateProfileId == 0) {
129             String errorMessage = "Certificate profile with name \"" + enrollcertificateRequest.getCertificateProfileName() + "\" doesn't exist";
130             throw new CertificateProfileDoesNotExistException(errorMessage);
131         }
132         endEntityInformation.setCertificateProfileId(certificateProfileId);
133 
134         Integer endEntityProfileId = getEndEntityProfileId(enrollcertificateRequest.getEndEntityProfileName());
135         endEntityInformation.setEndEntityProfileId(endEntityProfileId);
136 
137         PKCS10CertificationRequest pkcs10CertificateRequest = CertTools.getCertificateRequestFromPem(enrollcertificateRequest.getCertificateRequest());
138         if (pkcs10CertificateRequest == null) {
139             throw new EjbcaException("Invalid certificate request");
140         }
141 
142         String altName = getSubjectAltName(pkcs10CertificateRequest);
143         endEntityInformation.setSubjectAltName(altName);
144 
145         String subjectDn = getSubjectDn(pkcs10CertificateRequest);
146         endEntityInformation.setDN(subjectDn);
147 
148         endEntityInformation.setCardNumber("");
149         endEntityInformation.setHardTokenIssuerId(0);
150         endEntityInformation.setStatus(EndEntityConstants.STATUS_NEW);
151 
152         Date timecreated = new Date();
153         endEntityInformation.setTimeCreated(timecreated);
154         endEntityInformation.setTimeModified(timecreated);
155 
156         EndEntityProfile endEntityProfile = getEndEntityProfile(endEntityProfileId);
157 
158         endEntityInformation.setType(new EndEntityType(EndEntityTypes.ENDUSER));
159         boolean isSendNotificationDefaultInProfile = EndEntityProfile.TRUE.equals(endEntityProfile.getValue(EndEntityProfile.SENDNOTIFICATION, 0));
160         endEntityInformation.setSendNotification(isSendNotificationDefaultInProfile && !endEntityInformation.getSendNotification());
161 
162         endEntityInformation.setPrintUserData(false);
163         endEntityInformation.setTokenType(EndEntityConstants.TOKEN_USERGEN);
164 
165         endEntityInformation.setUsername(enrollcertificateRequest.getUsername());
166 
167         // Fill end-entity password
168         if (endEntityProfile.useAutoGeneratedPasswd()) {
169             // If auto-generated passwords are used, this is set on the CA side when adding or changing the EE as long as the password is null
170             endEntityInformation.setPassword(null);
171         } else if (StringUtils.isEmpty(enrollcertificateRequest.getPassword())) {
172             // If not needed just use some random data
173             final byte[] randomData = new byte[16];
174             final Random random = new SecureRandom();
175             random.nextBytes(randomData);
176             endEntityInformation.setPassword(new String(Hex.encode(CertTools.generateSHA256Fingerprint(randomData))));
177         } else {
178             endEntityInformation.setPassword(enrollcertificateRequest.getPassword());
179         }
180 
181         return endEntityInformation;
182     }

184     protected String getSubjectAltName(PKCS10CertificationRequest pkcs10CertificateRequest) {
185         String altName = null;
186         final Extension subjectAlternativeNameExtension = CertTools.getExtension(pkcs10CertificateRequest, Extension.subjectAlternativeName.getId());
187         if (subjectAlternativeNameExtension != null) {
188             altName = CertTools.getAltNameStringFromExtension(subjectAlternativeNameExtension);
189         }
190         return altName;
191     }

194     protected String getSubjectDn(PKCS10CertificationRequest pkcs10CertificateRequest) {
195         String subject = "";
196         if (pkcs10CertificateRequest.getSubject() != null) {
197             subject = pkcs10CertificateRequest.getSubject().toString();
198         }
199         return subject;
200     }
