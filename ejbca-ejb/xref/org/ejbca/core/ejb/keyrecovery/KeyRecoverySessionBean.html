128     @Override
129     public void checkIfApprovalRequired(AuthenticationToken admin, CertificateWrapper certificateWrapper, String username, int endEntityProfileId, boolean checkNewest) 
130             throws ApprovalException, WaitingForApprovalException, CADoesntExistsException {
131         final Certificate certificate = EJBTools.unwrap(certificateWrapper);
132         final int caid = CertTools.getIssuerDN(certificate).hashCode();
133 		final CAInfo cainfo = caSession.getCAInfoInternal(caid);
134         final CertificateInfo certinfo = certificateStoreSession.getCertificateInfo(CertTools.getFingerprintAsString(certificate));
135 		final CertificateProfile certProfile = certProfileSession.getCertificateProfile(certinfo.getCertificateProfileId());
136 		
137         // Check if approvals is required.
138         final ApprovalProfile approvalProfile = approvalProfileSession.getApprovalProfileForAction(ApprovalRequestType.KEYRECOVER, cainfo, certProfile);
139         if (approvalProfile != null) {    
140 			KeyRecoveryApprovalRequest ar = new KeyRecoveryApprovalRequest(certificate,username,checkNewest, admin,null,caid,
141 			        endEntityProfileId, approvalProfile);
142 			if (ApprovalExecutorUtil.requireApproval(ar, NONAPPROVABLECLASSNAMES_KEYRECOVERY)){
143 			    int requestId = approvalSession.addApprovalRequest(admin, ar);
144 	            String msg = intres.getLocalizedMessage("keyrecovery.addedforapproval");            	
145 				throw new WaitingForApprovalException(msg, requestId);
146 			}
147         } 
148     }

150     private String getPublicKeyIdFromKey(final CryptoToken cryptoToken, final String keyAlias) throws CryptoTokenOfflineException {
151         return new String(Base64.encode(KeyTools.createSubjectKeyId(cryptoToken.getPublicKey(keyAlias)).getKeyIdentifier(), false), StandardCharsets.US_ASCII);
152     }

154     @Override
155     public boolean addKeyRecoveryData(AuthenticationToken admin, CertificateWrapper certificateWrapper, String username, KeyPairWrapper keyPairWrapper)
156             throws AuthorizationDeniedException {
157         if (log.isTraceEnabled()) {
158             log.trace(">addKeyRecoveryData(user: " + username + ")");
159     	}
160         if (authorizedToAdministrateKeys(admin)) {
161             final Certificate certificate = EJBTools.unwrap(certificateWrapper);
162             final KeyPair keypair = EJBTools.unwrap(keyPairWrapper);
163             final int caid = CertTools.getIssuerDN(certificate).hashCode();
164             final String certSerialNumber = CertTools.getSerialNumberAsString(certificate);
165             boolean returnval = false;
166             try {
167                 KeyRecoveryCAServiceResponse response = (KeyRecoveryCAServiceResponse) caAdminSession.extendedService(admin, caid,
168                         new KeyRecoveryCAServiceRequest(KeyRecoveryCAServiceRequest.COMMAND_ENCRYPTKEYS, keypair));
169                 entityManager.persist(new org.ejbca.core.ejb.keyrecovery.KeyRecoveryData(CertTools.getSerialNumber(certificate), CertTools
170                         .getIssuerDN(certificate), username, response.getKeyData(), response.getCryptoTokenId(), response.getKeyAlias(), response.getPublicKeyId()));
171                 // same method to make hex serno as in KeyRecoveryDataBean
172                 String msg = intres.getLocalizedMessage("keyrecovery.addeddata", CertTools.getSerialNumber(certificate).toString(16),
173                         CertTools.getIssuerDN(certificate), response.getKeyAlias(), response.getPublicKeyId(), response.getCryptoTokenId());
174                 final Map<String, Object> details = new LinkedHashMap<>();
175                 details.put("msg", msg);
176                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_ADDDATA, EventStatus.SUCCESS, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA,
177                         admin.toString(), String.valueOf(caid), certSerialNumber, username, details);
178                 returnval = true;
179             } catch (Exception e) {
180                 final String msg = intres.getLocalizedMessage("keyrecovery.erroradddata", CertTools.getSerialNumber(certificate).toString(16),
181                         CertTools.getIssuerDN(certificate));
182                 final Map<String, Object> details = new LinkedHashMap<>();
183                 details.put("msg", msg);
184                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_ADDDATA, EventStatus.FAILURE, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA,
185                         admin.toString(), String.valueOf(caid), certSerialNumber, username, details);
186                 log.error(msg, e);
187             }
188             log.trace("<addKeyRecoveryData()");
189             return returnval;
190         } else {
191             throw new AuthorizationDeniedException(admin + " not authorized to administer keys");
192         }
193         
194     }

234     @Override
235     public void removeKeyRecoveryData(AuthenticationToken admin, CertificateWrapper certificateWrapper) throws AuthorizationDeniedException {
236         if (!authorizedToAdministrateKeys(admin)) {
237             throw new AuthorizationDeniedException(admin + " not authorized to administer keys");
238         }
239         final Certificate certificate = EJBTools.unwrap(certificateWrapper);
240         final String hexSerial = CertTools.getSerialNumber(certificate).toString(16);
241     	if (log.isTraceEnabled()) {
242             log.trace(">removeKeyRecoveryData(certificate: " + CertTools.getSerialNumber(certificate).toString(16) +")");
243     	}
244         final String dn = CertTools.getIssuerDN(certificate);
245         final int caid = dn.hashCode();
246         try {
247             String username = null;
248         	org.ejbca.core.ejb.keyrecovery.KeyRecoveryData krd = org.ejbca.core.ejb.keyrecovery.KeyRecoveryData.findByPK(entityManager, new KeyRecoveryDataPK(hexSerial, dn));
249         	if (krd == null) {
250         		throw new FinderException();
251         	}
252             username = krd.getUsername();
253             entityManager.remove(krd);
254             String msg = intres.getLocalizedMessage("keyrecovery.removeddata", hexSerial, dn);            	
255             final Map<String, Object> details = new LinkedHashMap<>();
256             details.put("msg", msg);
257             auditSession.log(EjbcaEventTypes.KEYRECOVERY_REMOVEDATA, EventStatus.SUCCESS, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA, admin.toString(), String.valueOf(caid), hexSerial, username, details);
258         } catch (Exception e) {
259             final String msg = intres.getLocalizedMessage("keyrecovery.errorremovedata", hexSerial, dn);            	
260             final Map<String, Object> details = new LinkedHashMap<>();
261             details.put("msg", msg);
262             auditSession.log(EjbcaEventTypes.KEYRECOVERY_REMOVEDATA, EventStatus.FAILURE, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA, admin.toString(), String.valueOf(caid), hexSerial, null, details);

292     @Override
293     public KeyRecoveryInformation recoverKeys(AuthenticationToken admin, String username, int endEntityProfileId) throws AuthorizationDeniedException {
294     	if (log.isTraceEnabled()) {
295             log.trace(">keyRecovery(user: " + username + ")");
296     	}
297         KeyRecoveryInformation returnval = null;
298         X509Certificate certificate = null;
299         if (authorizedToKeyRecover(admin, endEntityProfileId)) { 
300         	Collection<KeyRecoveryData> result = KeyRecoveryData.findByUserMark(entityManager, username);
301         	try {
302         		String caidString = null;
303         		String certSerialNumber = null;
304         		String logMsg = null;
305         		for (final KeyRecoveryData krd : result) {
306         			if (returnval == null) {
307         				final int caid = krd.getIssuerDN().hashCode();
308         				caidString = String.valueOf(caid);
309         				final KeyRecoveryCAServiceResponse response = (KeyRecoveryCAServiceResponse) caAdminSession.extendedService(admin, caid,
310         						new KeyRecoveryCAServiceRequest(KeyRecoveryCAServiceRequest.COMMAND_DECRYPTKEYS, krd.getKeyDataAsByteArray(),
311         						        krd.getCryptoTokenId(), krd.getKeyAlias()));
312         				final KeyPair keys = response.getKeyPair();
313         				certificate = (X509Certificate) certificateStoreSession.findCertificateByIssuerAndSerno(krd.getIssuerDN(), krd.getCertificateSN());
314         				returnval = new KeyRecoveryInformation(krd.getCertificateSN(), krd.getIssuerDN(),
315         						krd.getUsername(), krd.getMarkedAsRecoverable(), keys, certificate);
316                 		certSerialNumber = CertTools.getSerialNumberAsString(certificate);
317                         logMsg = intres.getLocalizedMessage("keyrecovery.sentdata", username, response.getKeyAlias(), response.getPublicKeyId(), response.getCryptoTokenId());                
318         			}
319         		}
320         		if (logMsg == null) {
321                     logMsg = intres.getLocalizedMessage("keyrecovery.nodata", username);                        		    
322         		}
323                 final Map<String, Object> details = new LinkedHashMap<>();
324                 details.put("msg", logMsg);
325                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_SENT, EventStatus.SUCCESS, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA, admin.toString(), caidString, certSerialNumber, username, details);
326         	} catch (Exception e) {
327         		String msg = intres.getLocalizedMessage("keyrecovery.errorsenddata", username);            	
328         		log.error(msg, e);
329                 final Map<String, Object> details = new LinkedHashMap<>();
330                 details.put("msg", msg);
331                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_SENT, EventStatus.FAILURE, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, username, details);
332         	}
333         } else {
334             throw new AuthorizationDeniedException(admin + " not authorized to key recovery for end entity profile id " + endEntityProfileId);
335         }
336         if (log.isTraceEnabled()) {
337             log.trace("<keyRecovery()");
338         }
339         return returnval;
340     }

387 	@Override
388     public boolean markNewestAsRecoverable(AuthenticationToken admin, String username, int endEntityProfileId) throws AuthorizationDeniedException, 
389                         ApprovalException, WaitingForApprovalException, CADoesntExistsException {
390     	if (log.isTraceEnabled()) {
391             log.trace(">markNewestAsRecoverable(user: " + username + ")");
392     	}
393         boolean returnval = false;
394         long newesttime = 0;
395         KeyRecoveryData newest = null;
396         X509Certificate certificate = null;
397         X509Certificate newestcertificate = null;
398         if (!isUserMarked(username)) {
399             String caidString = null;
400             String certSerialNumber = null;
401         	final Collection<KeyRecoveryData> result = KeyRecoveryData.findByUsername(entityManager, username);
402     		for (final KeyRecoveryData krd : result) {
403         		caidString = String.valueOf(krd.getIssuerDN().hashCode());
404         		certificate = (X509Certificate) certificateStoreSession.findCertificateByIssuerAndSerno(krd.getIssuerDN(), krd.getCertificateSN());
405         		if (certificate != null) {
406         			if (certificate.getNotBefore().getTime() > newesttime) {
407         				newesttime = certificate.getNotBefore().getTime();
408         				newest = krd;
409         				newestcertificate = certificate;
410                 		certSerialNumber = CertTools.getSerialNumberAsString(newestcertificate);
411         			}
412         		}
413         	}
414         	if (newest != null) {
415         		// Check that the administrator is authorized to keyrecover
416                 if (authorizedToKeyRecover(admin, endEntityProfileId)) {
417                     // Check if approvals is required.            
418                     checkIfApprovalRequired(admin, EJBTools.wrap(newestcertificate), username, endEntityProfileId, true);
419                     newest.setMarkedAsRecoverable(true);
420                     returnval = true;
421                 } else {
422                     throw new AuthorizationDeniedException(admin + " not authorized to key recovery for end entity profile id " + endEntityProfileId);
423                 }
424         	}
425         	if (returnval) {
426         		String msg = intres.getLocalizedMessage("keyrecovery.markeduser", username);            	
427                 final Map<String, Object> details = new LinkedHashMap<>();
428                 details.put("msg", msg);
429                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_MARKED, EventStatus.SUCCESS, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA, admin.toString(), caidString, certSerialNumber, username, details);
430         	} else {
431         		String msg = intres.getLocalizedMessage("keyrecovery.errormarkuser", username);
432         		log.info(msg);
433         	}
434         }
435         log.trace("<markNewestAsRecoverable()");
436         return returnval;
437     }

439 	@Override
440     public boolean markAsRecoverable(AuthenticationToken admin, Certificate certificate, int endEntityProfileId) throws AuthorizationDeniedException, 
441                             WaitingForApprovalException, ApprovalException, CADoesntExistsException {
442         final String hexSerial = CertTools.getSerialNumber(certificate).toString(16); // same method to make hex as in KeyRecoveryDataBean
443         final String dn = CertTools.getIssuerDN(certificate);        
444     	if (log.isTraceEnabled()) {
445             log.trace(">markAsRecoverable(issuer: "+dn+"; certificatesn: " + hexSerial + ")");
446     	}
447         boolean returnval = false;
448     	org.ejbca.core.ejb.keyrecovery.KeyRecoveryData krd = org.ejbca.core.ejb.keyrecovery.KeyRecoveryData.findByPK(entityManager, new KeyRecoveryDataPK(hexSerial, dn));
449         if (krd != null) {
450             String username = krd.getUsername();
451             // Check that the administrator is authorized to keyrecover
452             if (authorizedToKeyRecover(admin, endEntityProfileId)) {
453                 // Check if approvals is required.
454                 checkIfApprovalRequired(admin, EJBTools.wrap(certificate), username, endEntityProfileId, false);
455                 krd.setMarkedAsRecoverable(true);
456                 int caid = krd.getIssuerDN().hashCode();
457                 String msg = intres.getLocalizedMessage("keyrecovery.markedcert", hexSerial, dn);
458                 final Map<String, Object> details = new LinkedHashMap<>();

460                 auditSession.log(EjbcaEventTypes.KEYRECOVERY_MARKED, EventStatus.SUCCESS, EjbcaModuleTypes.KEYRECOVERY, EjbcaServiceTypes.EJBCA,
