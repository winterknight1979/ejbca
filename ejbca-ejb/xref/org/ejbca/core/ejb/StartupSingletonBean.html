144     /** Checks that a class is loaded by a classloader that is local to the ejbca.ear deployment.
145      * This check is probably JBoss / WildFly specific but not certain. We want to check this so we can detect if JBoss / WildFly bundles jars that we use so we risk having a version conflict.
146      * 
147      * A classloader that is from the main/core JBoss/WildFly modules is typically printed like:
148      * <code>
149      * org.bouncycastle.jcajce.provider.asymmetric.x509.X509CertificateObject(6034186b).ClassLoader=ModuleClassLoader for Module "org.bouncycastle" from local module loader @649d209a 
150      * </code> 
151      * while a classloader from the ejbca.ear space is printed like:
152      * <code>
153      * org.bouncycastle.jcajce.provider.asymmetric.x509.X509CertificateObject(362bcba0).ClassLoader=ModuleClassLoader for Module "deployment.ejbca.ear" from Service Module Loader
154      * </code>
155      * 
156      * @param clazz the class that we want to check which classloader it comes from. For example org.bouncycastle.jcajce.provider.asymmetric.x509.X509CertificateObject.class
157      */
158     private static void checkClassLoaderIsEJBCA(Class<? extends X509Certificate> clazz) {
159         StringBuffer results = new StringBuffer();
160         ClassLoader cl = clazz.getClassLoader();
161         results.append("\n" + clazz.getName() + "(" + Integer.toHexString(clazz.hashCode()) + ").ClassLoader=" + cl);
162         // We want classes (that we check here) to be loaded from the EJBCA application classloader, and not by the 
163         // JBoss/WildFly main classloader
164         if (results.indexOf("ejbca.ear") == -1) {
165             CodeSource clazzCS = clazz.getProtectionDomain().getCodeSource();
166             if (clazzCS != null) {
167                 results.append("\n++++CodeSource: "+clazzCS);
168             } else {
169                 results.append("\n++++Null CodeSource");
170             }
171             log.error("BouncyCastle is not loaded by an EJBCA classloader, version conflict is likely: "+clazz.getName()+": "+results.toString());            
172         } else {
173             log.info("BouncyCastle provider is from our ejbca.ear classloader.");
174         }
175     }

191     @PostConstruct
192     private void startup() {
193         //
194         // Run all "safe" initializations first, 
195         // i.e. those that does not depend on other running beans, components etc
196 
197         // Log a startup message
198         String iMsg = InternalEjbcaResources.getInstance().getLocalizedMessage("startservice.startup", GlobalConfiguration.EJBCA_VERSION);
199         log.info(iMsg);
200 
201         // Reinstall BC-provider to help re-deploys to work
202         log.trace(">init re-installing BC-provider");
203         CryptoProviderTools.removeBCProvider();
204         CryptoProviderTools.installBCProvider();
205 
206         // Run java seed collector, that can take a little time the first time it is run
207         log.trace(">init initializing random seed");
208         SecureRandom rand = new SecureRandom();
209         rand.nextInt();
210         
211         //
212         // Start services that requires calling other beans or components
213         //
214         
215         // We really need BC to be installed. This is an attempt to fix a bug where the ServiceSessionBean
216         // crashes from not finding the BC-provider.
217         int waitTime = 0;
218         while (Security.getProvider(BouncyCastleProvider.PROVIDER_NAME) == null && waitTime++ < 5) {
219             log.info("Waiting for BC provider to be installed..");
220             try {
221                 Thread.sleep(2000);
222             } catch (InterruptedException e) {
223                 Thread.currentThread().interrupt();
224                 log.error("Waiting for BC provider failed.", e);
225                 break;
226             }
227         }
228         // Verify that the classloader uses "our" BC version
229         log.trace(">init checking classloader of BouncyCastle provider");
230         try {
231             final CertificateFactory cf = CertTools.getCertificateFactory();
232             final X509Certificate testcert = (X509Certificate) cf.generateCertificate(new ByteArrayInputStream(testcertbytes));
233             checkClassLoaderIsEJBCA(testcert.getClass());
234         } catch (CertificateException e1) {
235             log.error("Can not parse certificate, this should never happen: ", e1);
236             throw new IllegalStateException(e1.getMessage());
237         }
238 
239         /* 
240          * Trigger ServiceLoader for AuthenticationTokens at startup, since this is a critical function it makes
241          * debugging nicer if the available tokens are shown at startup
242          */
243         log.info("Registered AuthenticationTokens " + AccessMatchValueReverseLookupRegistry.INSTANCE.getAllTokenTypes().toString());
244         // We have to read CAs into cache (and upgrade them) early, because the log system may use CAs for signing logs
245         
246         log.trace(">init CryptoTokenFactory just to load those classes that are available");
247         CryptoTokenFactory.instance();
248         
249         authorizationSession.scheduleBackgroundRefresh();
250         // Load CAs at startup to improve impression of speed the first time a CA is accessed, it takes a little time to load it.
251         log.trace(">init loading CAs into cache");
252         try {
253             caAdminSession.initializeAndUpgradeCAs();
254         } catch (Exception e) {
255             log.error("Error creating CAAdminSession: ", e);
256         }
257 
258         // Make a log row that EJBCA is starting
259         Map<String, Object> details = new LinkedHashMap<>();
260         details.put("msg", iMsg);
261         logSession.log(EjbcaEventTypes.EJBCA_STARTING, EventStatus.SUCCESS, EjbcaModuleTypes.SERVICE, EjbcaServiceTypes.EJBCA, authenticationToken.toString(), null, getHostName(), null, details);               
262 
263         // Log the type of security audit configuration that we have enabled.
264         log.trace(">init security audit device configuration");
265         final Set<String> loggerIds = AuditDevicesConfig.getAllDeviceIds();
266         if (loggerIds.isEmpty()) {
267             final String msg = InternalEjbcaResources.getInstance().getLocalizedMessage("startservices.noauditdevices");
268             log.info(msg);
269         } else {
270             if (!checkForProtectedAudit(authenticationToken, loggerIds)) {
271                 // Make a log row that no integrity protected device is configured
272                 final String msg = InternalEjbcaResources.getInstance().getLocalizedMessage("startservices.noprotectedauditdevices");
273                 final Map<String, Object> logdetails = new LinkedHashMap<>();
274                 logdetails.put("msg", msg);
275                 logSession.log(EventTypes.LOG_MANAGEMENT_CHANGE, EventStatus.VOID, ModuleTypes.SECURITY_AUDIT, ServiceTypes.CORE, authenticationToken.toString(), null, null, null, logdetails);                
276             }
277         }
278 
279         // Initialize authorization system, if not done already
280         log.trace(">init AuthorizationSystemSession to check for initial root role");
281         final boolean isFreshInstallation = authorizationSystemSession.initializeAuthorizationModule();
282 
283         log.trace(">init calling ServiceSession.load");
284         try {
285             serviceSession.load();
286         } catch (Exception e) {
287             log.error("Error init ServiceSession: ", e);
288         }
289         
290         // Load Certificate profiles at startup to upgrade them if needed
291         log.trace(">init loading CertificateProfile to check for upgrades");
292         try {
293             certificateProfileSession.initializeAndUpgradeProfiles();
294         } catch (Exception e) {
295             log.error("Error initializing certificate profiles: ", e);
296         }
297         
298         // Load EndEntity profiles at startup to upgrade them if needed
299         // And add this node to list of nodes
300         log.trace(">init loading EndEntityProfile to check for upgrades");
301         try {
302             endEntityProfileSession.initializeAndUpgradeProfiles();         
303         } catch (Exception e) {
304             log.error("Error initializing end entity profiles: ", e);
305         }
306         
307         // Add this node's hostname to list of nodes
308         log.trace(">init checking if this node is in the list of nodes");
309         try {
310             // Requires a transaction in order to create the initial global configuration
311             final GlobalConfiguration config = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
312             final Set<String> nodes = config.getNodesInCluster();
313             final String hostname = getHostName();
314             if (hostname != null && !nodes.contains(hostname)) {
315                 log.debug("Adding this node ("+hostname+") to the list of nodes");
316                 nodes.add(hostname);
317                 config.setNodesInCluster(nodes);
318                 globalConfigurationSession.saveConfiguration(authenticationToken, config);
319             }
320         } catch (Exception e) {
321             log.error("Error adding host to node list in global configuration: ", e);
322         } 
323 
324         log.trace(">init SignSession to check for unique issuerDN,serialNumber index");
325         // Call the check for unique index, since first invocation will perform the database
326         // operation and avoid a performance hit for the first request where this is checked.
327         final Boolean unique = DatabaseIndexUtil.isIndexPresentOverColumns(JDBCUtil.getDataSourceOrNull(), "CertificateData", Arrays.asList("serialNumber", "issuerDN"), true);
328         if (unique==null) {
329             log.info("Unable to read the index meta data from the database. Will detect presence of unique index using conflicting inserts to CertificateData.");
330             // Fall-back to testing for a unique index on CertificateData using conflicting INSERTs
331             certCreateSession.isUniqueCertificateSerialNumberIndex();
332         } else {
333             certificateStoreSession.setUniqueCertificateSerialNumberIndex(unique);
334         }
335         // Perform (automatic) upgrades, if needed
336         upgradeSession.performPreUpgrade(isFreshInstallation);
337         upgradeSession.performUpgrade();
338         // Start key reload timer
339         ocspResponseGeneratorSession.initTimers();
340         // Start CA certificate cache reload
341         certificateStoreSession.initTimers();
342         // Start legacy background service for renewal of OCSP signers via EJBCA WS calls to CA
343         ocspKeyRenewalSession.startTimer();
344         // Verify that the EJB CLI user (if present) cannot be used to generate certificates
345         final String cliUsername = EjbcaConfiguration.getCliDefaultUser();
346         try {
347             final EndEntityInformation defaultCliUser = endEntityAccessSession.findUser(authenticationToken, cliUsername);
348             if (defaultCliUser!=null && defaultCliUser.getStatus()==EndEntityConstants.STATUS_NEW) {
349                 try {
350                     endEntityManagementSession.setUserStatus(authenticationToken, cliUsername, EndEntityConstants.STATUS_GENERATED);
351                 } catch (ApprovalException e) {
352                     log.warn("The EJBCA CLI user '" + cliUsername + "' could be used for certificate enrollment. Please correct the status manually. Failed with: " + e.getMessage());
353                 } catch (NoSuchEndEntityException e) {
354                     log.warn("The EJBCA CLI user '" + cliUsername + "' could be used for certificate enrollment. Please correct the status manually. Failed with: " + e.getMessage());
355                 } catch (WaitingForApprovalException e) {
356                     log.warn("The EJBCA CLI user '" + cliUsername + "' could be used for certificate enrollment. Please correct the status manually. Failed with: " + e.getMessage());
357                 }
358             }
359         } catch (AuthorizationDeniedException e) {
360             log.warn("Unable to check if the EJBCA CLI user '" + cliUsername + "' could be used for certificate enrollment. Please check and correct the status manually. Failed with: " + e.getMessage());
361         }
362     }

365     /** Method that checks if we have an integrity protected security audit device configured, and in that case logs the configuration startup 
366      * 
367      * @param admin an authentication token used to log the configuration management startup (logged as a change as audit is configured during startup from properties file) 
368      * @param loggerIds the configured loggers among which we look for the protected device
369      * @return true if there is an integrity protected audit device and is was configured during startup (and audit log of this config was made)
370      */
371     private boolean checkForProtectedAudit(AuthenticationToken admin, final Set<String> loggerIds) {
372         boolean ret = false;                                            
373         // See if we have IntegrityProtectedDevice configured, due to class loading constraints we can not use IntegrityProtectedDevice.class.getSimpleName().
374         // This is admin-gui and does not have access to that class
375         final String integrityProtectedName = "IntegrityProtectedDevice";
376         for (Iterator<String> iterator = loggerIds.iterator(); iterator.hasNext();) {
377             final String id = iterator.next();
378             if (integrityProtectedName.equals(id)) {
379                 // Make a log row that integrity protected device is configured
380                 final Map<String, Object> logdetails = new LinkedHashMap<>();
381                 try {
382                     // Use reflection to get the ProtectedDataConfiguration and make some calls to it.
383                     // This is needed since ProtectedDataConfiguration may not be available during compile time, or runtime
384                     Class<?> c = Class.forName("org.cesecore.dbprotection.ProtectedDataConfiguration");
385                     // create instance ProtectedDataConfiguration.instance()
386                     Method instance = c.getMethod("instance", (Class[])null);
387                     //Object[] args = new Object[0];
388                     Object config = instance.invoke(null);
389                     // create method ProtectedDataConfiguration.getKeyId(String)
390                     Method getKeyId = c.getMethod("getKeyId", String.class);
391                     // create method ProtectedDataConfiguration.getProtectVersion(int)
392                     Method getProtectVersion = c.getMethod("getProtectVersion", Integer.TYPE);
393                     // create method ProtectedDataConfiguration.getKeyLabel(int)
394                     Method getKeyLabel = c.getMethod("getKeyLabel", Integer.TYPE);
395                     // Call ProtectedDataConfiguration.instance().getKeyId
396                     final String auditTableName = AuditRecordData.class.getSimpleName();
397                     final Integer keyid = (Integer)getKeyId.invoke(config, auditTableName);
398                     if ((keyid != null) && (keyid > 0)) {
399                         if (CesecoreConfiguration.useDatabaseIntegrityProtection(auditTableName)) {
400                             // Call ProtectedDataConfiguration.instance().getProtectVersion
401                             final Integer protectVersion = (Integer)getProtectVersion.invoke(config, keyid);
402                             // Call ProtectedDataConfiguration.instance().getKeyLabel
403                             final String keyLabel = (String)getKeyLabel.invoke(config, keyid);
404                             logdetails.put("keyid", keyid);
405                             logdetails.put("protectVersion", protectVersion);
406                             logdetails.put("keyLabel", keyLabel);
407                             logSession.log(EventTypes.LOG_MANAGEMENT_CHANGE, EventStatus.SUCCESS, ModuleTypes.SECURITY_AUDIT, ServiceTypes.CORE, admin.toString(), null, null, null, logdetails);
408                             ret = true;                                            
409                         } else {
410                             log.debug("No database integrity protection enabled for AuditRecordData.");
411                         }
412                     } else {
413                         log.debug("No keyid configured for AuditRecordData.");
414                     }
415                 } catch (ClassNotFoundException e) {
416                     log.info("No database integrity protection available in this version of EJBCA.");
417                 } catch (IllegalAccessException e) {
418                     log.info("No database integrity protection available due to initialization error: ", e);
419                 } catch (SecurityException e) {
420                     log.info("No database integrity protection available due to initialization error: ", e);
421                 } catch (NoSuchMethodException e) {
422                     log.info("No database integrity protection available due to initialization error: ", e);
423                 } catch (IllegalArgumentException e) {
424                     log.info("No database integrity protection available due to initialization error: ", e);
425                 } catch (InvocationTargetException e) {
426                     log.info("No database integrity protection available due to initialization error: ", e);
427                 }
428             }
429         }
430         return ret;
431     }

433     /**
434      * @return The host's name or null if it could not be determined.
435      */
436     private String getHostName() {
437         String hostname = null;
438         try {
439             InetAddress addr = InetAddress.getLocalHost();    
440             // Get hostname
441             hostname = addr.getHostName();
442         } catch (UnknownHostException e) {
443             log.error("Hostname could not be determined", e);
444         }
445         return hostname;
446     }
