150     /**
151      * 
152      * 
153      * @param signerSubjectDN signerSubjectDN subject DN of the signing key to be renewed. The string "all" will result in all keys being renewed
154      * @param safetyMargin the number of seconds before actual expiration that a keystore should be renewed
155      * @throws CryptoTokenOfflineException if Crypto Token is not available or connected, or key with alias does not exist.
156      * @throws InvalidKeyException if the public key in the tokenAndChain can not be used to verify a string signed by the private key, because the key 
157      * is wrong or the signature operation fails for other reasons such as a NoSuchAlgorithmException or SignatureException.
158      */
159     private synchronized void renewKeyStores(String signerSubjectDN, long safetyMargin) throws InvalidKeyException,
160             CryptoTokenOfflineException {
161         //Cancel all running timers
162         cancelTimers();
163         try {
164             final EjbcaWS ejbcaWS = getEjbcaWS();
165             if (ejbcaWS == null) {
166                 if (log.isDebugEnabled()) {
167                     log.debug("Could not locate a suitable web service for automatic OCSP key/certificate renewal.");
168                 }
169                 return;
170             }
171             final X500Principal target;
172             try {
173                 target = signerSubjectDN.trim().equalsIgnoreCase(RENEW_ALL_KEYS) ? null : new X500Principal(signerSubjectDN);
174             } catch (IllegalArgumentException e) {
175                 log.error(intres.getLocalizedMessage("ocsp.rekey.triggered.dn.not.valid", signerSubjectDN));
176                 return;
177             }
178             final StringBuffer matched = new StringBuffer();
179             final StringBuffer unMatched = new StringBuffer();
180             final Set<Integer> processedOcspKeyBindingIds = new HashSet<Integer>();
181             for (final OcspSigningCacheEntry ocspSigningCacheEntry : OcspSigningCache.INSTANCE.getEntries()) {
182                 // Only perform renewal for non CA signing key OCSP signers
183                 if (!ocspSigningCacheEntry.isUsingSeparateOcspSigningCertificate()) {
184                     continue;
185                 }
186                 // Only perform renewal once for each OcspKeyBinding.
187                 // (the cache can map each OcspKeyBinding multiple times by SHA-1, SHA-256 and as default responder)
188                 final int ocspKeyBindingId = ocspSigningCacheEntry.getOcspKeyBinding().getId();
189                 if (!processedOcspKeyBindingIds.add(Integer.valueOf(ocspKeyBindingId))) {
190                     if (log.isDebugEnabled()) {
191                         log.debug("Skipping renewal processing of OcspKeyBinding with id " + ocspKeyBindingId + " that was already processed.");
192                     }
193                     continue;
194                 }
195                 final X509Certificate ocspSigningCertificate = ocspSigningCacheEntry.getOcspSigningCertificate();
196                 final long timeLeftBeforeRenewal = ocspSigningCertificate.getNotAfter().getTime()-new Date().getTime();
197                 if (timeLeftBeforeRenewal < (1000 * safetyMargin)) {
198                     final X500Principal src = ocspSigningCertificate.getSubjectX500Principal();
199                     if (target != null && !src.equals(target)) {
200                         unMatched.append(" '" + src.getName() + '\'');
201                         continue;
202                     }
203                     matched.append(" '" + ocspSigningCertificate.getIssuerX500Principal().getName() + '\'');
204                     try {
205                         renewKeyStore(ejbcaWS, ocspSigningCacheEntry);
206                     } catch (KeyRenewalFailedException e) {
207                         String msg = intres.getLocalizedMessage("ocsp.rekey.failed.unknown.reason", target, e.getLocalizedMessage());
208                         log.error(msg, e);
209                         continue;
210                     }
211                 }
212             }
213             if (matched.length() < 1 && target != null) {
214                 log.error(intres.getLocalizedMessage("ocsp.rekey.triggered.dn.not.existing", target.getName(), unMatched));
215                 return;
216             }
217             log.info(intres.getLocalizedMessage("ocsp.rekey.triggered", matched));
218         } finally {
219             //Set new timer to run, even if something breaks.
220             addTimer(OcspConfiguration.getRekeyingUpdateTimeInSeconds());
221         }
222     }

224     @Override
225     public synchronized void renewKeyStores(String signerSubjectDN) throws KeyStoreException, CryptoTokenOfflineException,
226             InvalidKeyException {
227         renewKeyStores(signerSubjectDN, NO_SAFETY_MARGIN);
228     }

230     /**
231      * Generate a new key pair and request a new certificate for this key pair using EJBCA WS.
232      * 
233      * @param ejbcaWS a reference to the remote EJBCA WS
234      * @param ocspSigningCacheEntry the cached OCSP signing entry backed by an OcspKeyBinding
235      * @throws InvalidKeyException if the new public key can not be used to verify a string signed by the private key, because the key is wrong or 
236      * the signature operation fails for other reasons such as a NoSuchAlgorithmException or SignatureException.
237      * @throws CryptoTokenOfflineException if Crypto Token is not available or connected, or key with alias does not exist.
238      * @throws KeyRenewalFailedException if any error occurs during signing
239      */
240     private void renewKeyStore(EjbcaWS ejbcaWS, OcspSigningCacheEntry ocspSigningCacheEntry) throws InvalidKeyException, CryptoTokenOfflineException, KeyRenewalFailedException {
241         //Generate the new key pair
242         final int internalKeyBindingId = ocspSigningCacheEntry.getOcspKeyBinding().getId();
243         try {
244             internalKeyBindingMgmtSession.generateNextKeyPair(authenticationToken, internalKeyBindingId);
245         } catch (InvalidAlgorithmParameterException e) {
246             throw new KeyRenewalFailedException(e);
247         } catch (AuthorizationDeniedException e) {
248             throw new KeyRenewalFailedException(e);
249         }
250         //Sign the new keypair
251         final X509Certificate signedCertificate = signCertificateByCa(ejbcaWS, ocspSigningCacheEntry);
252         try {
253             internalKeyBindingMgmtSession.importCertificateForInternalKeyBinding(authenticationToken, internalKeyBindingId, signedCertificate.getEncoded());
254         } catch (CertificateEncodingException e) {
255             throw new KeyRenewalFailedException(e);
256         } catch (CertificateImportException e) {
257             throw new KeyRenewalFailedException(e);
258         } catch (AuthorizationDeniedException e) {
259             throw new KeyRenewalFailedException(e);
260         }
261         /*
262          * Replace the alias and the chain at this step. If anything bad happened prior to this step the old alias and 
263          * chain are still active, and no harm done. 
264          */
265         ocspResponseGeneratorSession.reloadOcspSigningCache();
266     }

268     /**
269      * Get user data for the EJBCA user that will be used when creating the cert for the new key.
270      * @param ejbcaWS WS
271      * @param signingCertificate The OCSP signing certificate to get the end entity for
272      * @param caId the ID of the OCSP signing certificate issuing CA
273      * 
274      * @return the data
275      */
276     private UserDataVOWS getUserDataVOWS(EjbcaWS ejbcaWS, final X509Certificate signingCertificate, final int caId) {
277         final UserMatch match = new UserMatch();
278         final String subjectDN = CertTools.getSubjectDN(signingCertificate);
279         final String caName = getCAName(ejbcaWS, caId);
280         if (caName == null) {
281             throw new InvalidParameterException("No CA found for ID: " + caId);
282         }
283         match.setMatchtype(BasicMatch.MATCH_TYPE_EQUALS);
284         match.setMatchvalue(subjectDN);
285         match.setMatchwith(org.ejbca.util.query.UserMatch.MATCH_WITH_DN);
286         final List<UserDataVOWS> users;
287         try {
288             users = ejbcaWS.findUser(match);
289         } catch (Exception e) {
290             log.error("WS not working", e);
291             return null;
292         }
293         if (users == null || users.size() < 1) {
294             log.error(intres.getLocalizedMessage("ocsp.no.user.with.subject.dn", subjectDN));
295             return null;
296         }
297         log.debug("at least one user found for cert with DN: " + subjectDN + " Trying to match it with CA name: " + caName);
298         UserDataVOWS result = null;
299         for (UserDataVOWS userData : users) {
300             if (caName.equals(userData.getCaName())) {
301                 result = userData;
302                 break;
303             }
304         }
305         if (result == null) {
306             log.error("No user found for certificate '" + subjectDN + "' on CA '" + caName + "'.");
307             return null;
308         }
309         return result;
310     }

312     /**
313      * setting status of EJBCA user to new and setting password of user.
314      * @param ejbcaWS from {@link #getEjbcaWS()}
315      * @param userData from {@link #getUserDataVOWS}
316      * @return true if success
317      */
318     private boolean editUser(EjbcaWS ejbcaWS, UserDataVOWS userData) {
319         userData.setStatus(EndEntityConstants.STATUS_NEW);
320         userData.setPassword(PasswordGeneratorFactory.getInstance(PasswordGeneratorFactory.PASSWORDTYPE_LETTERSANDDIGITS).getNewPassword(12, 12));
321         userData.setTokenType(UserDataVOWS.TOKEN_TYPE_USERGENERATED);
322         try {
323             ejbcaWS.editUser(userData);
324         } catch (Exception e) {
325             log.error("Problem to edit user.", e);
326             return false;
327         }
328         return true;
329     }

334      * @param caId The ID of the sought CA
335      * 
336      * @return the name

338     private String getCAName(EjbcaWS ejbcaWS, int caId) {
339         final Map<Integer, String> mCA = new HashMap<Integer, String>();
340         try {
341             for (NameAndId nameAndId : ejbcaWS.getAvailableCAs()) {
342                 mCA.put(Integer.valueOf(nameAndId.getId()), nameAndId.getName());
343                 log.debug("CA. id: " + nameAndId.getId() + " name: " + nameAndId.getName());
344             }
345         } catch (Exception e) {
346             log.error("WS not working", e);
347             return null;
348         }
349         return mCA.get(Integer.valueOf(caId));
350     }

352     /**
353      * This method sends a keypair off to be signed by the CA that issued the original keychain.
354      * @param ejbcaWS WS
355      * @param ocspSigningCacheEntry Entry 
356      * 
357      * @return a certificate that has been signed by the CA. 
358      * @throws KeyRenewalFailedException if any error occurs during signing
359      * @throws CryptoTokenOfflineException  Fail
360      */
361     @SuppressWarnings("unchecked")
362     private X509Certificate signCertificateByCa(EjbcaWS ejbcaWS, OcspSigningCacheEntry ocspSigningCacheEntry) throws KeyRenewalFailedException,
363             CryptoTokenOfflineException {
364         /* Construct a certification request in order to have the new keystore certified by the CA. 
365          */
366         //final int caId = CertTools.stringToBCDNString(tokenAndChain.getCaCertificate().getSubjectDN().toString()).hashCode();
367         final int caId = CertTools.getSubjectDN(ocspSigningCacheEntry.getCaCertificateChain().get(0)).hashCode();
368         final X509Certificate ocspSigningCertificate = ocspSigningCacheEntry.getOcspSigningCertificate();
369         final UserDataVOWS userData = getUserDataVOWS(ejbcaWS, ocspSigningCertificate, caId);
370         if (userData == null) {
371             final String msg = "User data for certificate with subject DN '" + CertTools.getSubjectDN(ocspSigningCertificate) + "' was not found.";
372             log.error(msg);
373             throw new KeyRenewalFailedException(msg);
374         }
375         editUser(ejbcaWS, userData);
376         final int internalKeyBindingId = ocspSigningCacheEntry.getOcspKeyBinding().getId();
377         final byte[] pkcs10CertificationRequest;
378         try {
379             pkcs10CertificationRequest = internalKeyBindingMgmtSession.generateCsrForNextKey(authenticationToken, internalKeyBindingId, null);
380         } catch (AuthorizationDeniedException e) {
381             throw new KeyRenewalFailedException(e);
382         }
383         CertificateResponse certificateResponse;
384         try {
385             certificateResponse = ejbcaWS.pkcs10Request(userData.getUsername(), userData.getPassword(),
386                     new String(Base64.encode(pkcs10CertificationRequest)), null, CertificateHelper.RESPONSETYPE_CERTIFICATE);
387         } catch (Exception e) {
388             //Way too many silly exceptions to handle, wrap instead.
389             throw new KeyRenewalFailedException(e);
390         }
391         if (certificateResponse == null) {
392             throw new KeyRenewalFailedException("Certificate Response was not received");
393         }
394 
395         Collection<X509Certificate> certificates;
396         try {
397             certificates = (Collection<X509Certificate>) CertificateFactory.getInstance("X.509").generateCertificates(
398                     new ByteArrayInputStream(Base64.decode(certificateResponse.getData())));
399         } catch (CertificateException e) {
400             throw new KeyRenewalFailedException(e);
401         }
402         final byte[] publicKeyBytes;
403         try {
404             publicKeyBytes = internalKeyBindingMgmtSession.getNextPublicKeyForInternalKeyBinding(authenticationToken, internalKeyBindingId);
405         } catch (AuthorizationDeniedException e) {
406             throw new KeyRenewalFailedException(e);
407         }
408         if (log.isDebugEnabled()) {
409             log.debug("Number of certificates returned from WS: " + certificates.size());
410         }
411         X509Certificate signedCertificate = null;
412         final X509Certificate caCertificate = ocspSigningCacheEntry.getCaCertificateChain().get(0);
413         final PublicKey caCertificatePublicKey = caCertificate.getPublicKey();
414         for (X509Certificate certificate : certificates) {
415             if (log.isDebugEnabled()) {
416                 log.debug("Verifying certificate with SubjectDN : '" + CertTools.getSubjectDN(certificate) +
417                         "' using public key from CA certificate with subject '" + CertTools.getSubjectDN(caCertificate) +"'.");
418             }
419             try {
420                 certificate.verify(caCertificatePublicKey);
421             } catch (Exception e) {
422                 //Ugly, but inherited from legacy code
423                 signedCertificate = null;
424                 log.error("Exception was caught when verifying certificate", e);
425                 continue;
426             }
427             // Comparing public keys is dependent on provider used, so we must ensure same provider is used for the public keys
428             // Otherwise this will fail, even though it should work
429             // Both certPublicKey and nextPublicKey is obtained using KeyTools.getPublicKeyFromBytes, which uses the BC provider
430             final PublicKey certPublicKey = KeyTools.getPublicKeyFromBytes(certificate.getPublicKey().getEncoded());
431             final PublicKey nextPublicKey = KeyTools.getPublicKeyFromBytes(publicKeyBytes);
432             if (nextPublicKey.equals(certPublicKey)) {
433                 signedCertificate = certificate;
434                 break;
435             } else if (log.isDebugEnabled()) {
436                 log.debug("Matching public keys failed: ");
437                 log.debug("Certificate public key: "+certificate.getPublicKey());
438                 log.debug("Next public key: "+nextPublicKey);
439             }
440         }
441         if (signedCertificate == null) {
442             throw new KeyRenewalFailedException("No certificate signed by correct CA generated.");
443         }
444         return signedCertificate;
445     }

447     /** @return the EJBCA WS object. */
448     private EjbcaWS getEjbcaWS() {
449         String webUrl = OcspConfiguration.getEjbcawsracliUrl();
450         if (StringUtils.isEmpty(webUrl)) {
451             // Automatic renewal is not enabled
452             if (log.isDebugEnabled()) {
453                 log.debug("Automatic OCSP key/certificate renewal is not enabled, "+OcspConfiguration.REKEYING_WSURL+" is empty.");
454             }
455             return null;
456         }
457         final SSLSocketFactory sslSocketFactory = getSSLSocketFactory();
458         if (sslSocketFactory == null) {
459             log.warn("No AuthenticationKeyBinding is configured. Unable to authenticate to EJBCA WebService.");
460             return null;
461         }
462         HttpsURLConnection.setDefaultSSLSocketFactory(sslSocketFactory);
463         final URL ws_url;
464         try {
465             ws_url = new URL(webUrl + "?wsdl");
466         } catch (MalformedURLException e) {
467             log.warn("Problem with URL: '" + webUrl + "'", e);
468             return null;
469         }
470         final QName qname = new QName("http://ws.protocol.core.ejbca.org/", "EjbcaWSService");
471         if (log.isDebugEnabled()) {
472             log.debug("web service. URL: " + ws_url + " QName: " + qname);
473         }
474         return new EjbcaWSService(ws_url, qname).getEjbcaWSPort();
475     }

477     private SSLSocketFactory getSSLSocketFactory() {
478         final List<Integer> authenticationKeyBindingIds = internalKeyBindingMgmtSession.getInternalKeyBindingIds(authenticationToken, AuthenticationKeyBinding.IMPLEMENTATION_ALIAS);
479         AuthenticationKeyBinding authenticationKeyBinding = null;
480         for (Integer internalKeyBindingId : authenticationKeyBindingIds) {
481             try {
482                 final InternalKeyBinding internalKeyBinding = internalKeyBindingMgmtSession.getInternalKeyBindingReference(authenticationToken, internalKeyBindingId);
483                 if (internalKeyBinding.getStatus().equals(InternalKeyBindingStatus.ACTIVE)) {
484                     // Use first active one
485                     authenticationKeyBinding = (AuthenticationKeyBinding) internalKeyBinding;
486                     break;
487                 }
488             } catch (AuthorizationDeniedException e) {
489                 throw new RuntimeException(e);
490             }
491         }
492         if (authenticationKeyBinding == null) {
493             return null;
494         }
495         final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(authenticationKeyBinding.getCryptoTokenId());
496         final X509Certificate sslCertificate = (X509Certificate) certificateStoreSession.findCertificateByFingerprint(authenticationKeyBinding.getCertificateId());
497         final List<X509Certificate> chain = new ArrayList<X509Certificate>();
498         chain.add(sslCertificate);
499         chain.addAll(getCaCertificateChain(sslCertificate));
500         final List<X509Certificate> trustedCertificates = getListOfTrustedCertificates(authenticationKeyBinding.getTrustedCertificateReferences());
501         final String alias = authenticationKeyBinding.getKeyPairAlias();
502         try {
503             final TrustManager trustManagers[];
504             if (trustedCertificates == null || trustedCertificates.isEmpty()) {
505                 trustManagers = new X509TrustManager[] {new X509TrustManagerAcceptAll()};
506             } else {
507                 throw new RuntimeException("Configurable trust not yet implemented.");
508             }
509             final KeyManager keyManagers[] = new X509KeyManager[] { new ClientX509KeyManager(alias, cryptoToken.getPrivateKey(alias), chain) };
510             // Now construct a SSLContext using these (possibly wrapped) KeyManagers, and the TrustManagers.
511             // We still use a null SecureRandom, indicating that the defaults should be used.
512             final SSLContext context = SSLContext.getInstance("TLS");
513             context.init(keyManagers, trustManagers, null);
514             // Finally, we get a SocketFactory, and pass it on.
515             return context.getSocketFactory();
516         } catch (KeyManagementException e) {
517             e.printStackTrace();
518         } catch (NoSuchAlgorithmException e) {
519             e.printStackTrace();
520         } catch (CryptoTokenOfflineException e) {
521             e.printStackTrace();
522         }
523         return null;
524     }

535     // TODO: This method also exists in OcspResponseGenSSB.. merge! to method call in certificateStoreSession
536     private List<X509Certificate> getCaCertificateChain(final X509Certificate leafCertificate) {
537         final List<X509Certificate> caCertificateChain = new ArrayList<X509Certificate>();
538         X509Certificate currentLevelCertificate = leafCertificate;
539         while (!CertTools.getIssuerDN(currentLevelCertificate).equals(CertTools.getSubjectDN(currentLevelCertificate))) {
540             final String issuerDn = CertTools.getIssuerDN(currentLevelCertificate);
541             currentLevelCertificate = certificateStoreSession.findLatestX509CertificateBySubject(issuerDn);
542             if (currentLevelCertificate == null) {
543                 log.warn("Unable to build certificate chain for OCSP signing certificate with Subject DN '" +
544                         CertTools.getSubjectDN(leafCertificate) + "'. CA with Subject DN '" + issuerDn + "' is missing in the database.");
545                 return null;
546             }
547             caCertificateChain.add(currentLevelCertificate);
548         }
549         return caCertificateChain;
550     }

552     /**
553      * When the timer expires, this method will check through the cache and automatically renew keystore matching the predefined criteria, 
554      * and which expire within the designated time frame.
555      * 
556      * According to JSR 220 FR (18.2.2), this method may not throw any exceptions.
557      * 
558      * Glassfish 2.1.1:
559      * "Timeout method ....timeoutHandler(javax.ejb.Timer)must have TX attribute of TX_REQUIRES_NEW or TX_REQUIRED or TX_NOT_SUPPORTED"
560      * JBoss 5.1.0.GA: We cannot mix timer updates with our EJBCA DataSource transactions. 
561      * 
562      * @param timer The timer whose expiration caused this notification.
563      * 
564      */
565     @Timeout
566     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
567     public void timeoutHandler(Timer timer) {
568         long rekeyingUpdateTime = OcspConfiguration.getRekeyingUpdateTimeInSeconds();
569         try {
570             renewKeyStores(RENEW_ALL_KEYS, OcspConfiguration.getRekeyingSafetyMarginInSeconds());
571         } catch (InvalidKeyException e) {
572             log.error("A cached crypto token contains an invalid key pair. Stopping timers.", e);
573         } catch (CryptoTokenOfflineException e) {
574             //Rescheduling is handled in a finally clause in OcspKeyRenewalSessionBean.renewKeyStores(String, long)
575             log.error("Crypto token was offline or unavailable during automatic update. Rescheduling a new timer in " + rekeyingUpdateTime + " seconds.", e);
576         }    
577         
578     }

580     @Override
581     public void startTimer() {
582         cancelTimers();
583         addTimer(OcspConfiguration.getRekeyingUpdateTimeInSeconds());
584     }

586     /**
587      * Adds a timer to the bean
588      * 
589      * @param intervalInSeconds the time from now for the next timer to fire
590      * @return timer
591      */
592     // We don't want the appserver to persist/update the timer in the same transaction if they are stored in different non XA DataSources. This method
593     // should not be run from within a transaction.
594     private Timer addTimer(long intervalInSeconds) {
595         if (log.isDebugEnabled()) {
596             log.debug("addTimer: " + timerId+", "+intervalInSeconds);
597         }
598         return timerService.createSingleActionTimer(intervalInSeconds*1000, new TimerConfig(timerId, false));
599     }

601     /**
602      * This method cancels all timers associated with this bean.
603      */
604     private void cancelTimers() {
605         Collection<Timer> timers = timerService.getTimers();
606         for (Timer timer : timers) {
607             try {
608                 timer.cancel();
609             } catch (NoSuchObjectLocalException e) {
610             	if (log.isDebugEnabled()) {
611             	    log.debug("Timer was already expired or canceled: "+timer.getInfo());
612             	}
613             }
614         }
615     }
