199     @Override
200     public void addService(AuthenticationToken admin, int id, String name, ServiceConfiguration serviceConfiguration) throws ServiceExistsException {
201         if (log.isTraceEnabled()) {
202             log.trace(">addService(name: " + name + ", id: " + id + ")");
203         }
204         boolean success = addServiceInternal(admin, id, name, serviceConfiguration);
205         if (success) {
206             final String msg = intres.getLocalizedMessage("services.serviceadded", name);
207             final Map<String, Object> details = new LinkedHashMap<String, Object>();
208             details.put("msg", msg);
209             auditSession.log(EjbcaEventTypes.SERVICE_ADD, EventStatus.SUCCESS, EjbcaModuleTypes.SERVICE, EjbcaServiceTypes.EJBCA,
210                     admin.toString(), null, null, null, details);
211         } else {
212             final String msg = intres.getLocalizedMessage("services.erroraddingservice", name);
213             log.info(msg);
214             throw new ServiceExistsException(msg);
215         }
216         log.trace("<addService()");
217     }

219     private boolean addServiceInternal(AuthenticationToken admin, int id, String name, ServiceConfiguration serviceConfiguration) throws ServiceExistsException {
220         boolean success = false;
221         if (isAuthorizedToEditService(admin)) {
222             if (serviceDataSession.findByName(name) == null) {
223                 if (serviceDataSession.findById(Integer.valueOf(id)) == null) {
224                     serviceDataSession.addServiceData(id, name, serviceConfiguration);
225                     success = true;
226                 }
227             }
228         } else {
229             final String msg = intres.getLocalizedMessage("services.notauthorizedtoadd", name);
230             log.info(msg);
231         }
232         return success;
233     }

235     @Override
236     public void cloneService(AuthenticationToken admin, String oldname, String newname) throws ServiceExistsException {
237         if (log.isTraceEnabled()) {
238             log.trace(">cloneService(name: " + oldname + ")");
239         }
240         ServiceConfiguration servicedata = null;
241         ServiceData htp = serviceDataSession.findByName(oldname);
242         if (htp == null) {
243             String msg = "Error cloning service: No such service found.";
244             log.error(msg);
245             throw new EJBException(msg);
246         }
247         try {
248             servicedata = (ServiceConfiguration) htp.getServiceConfiguration().clone();
249             if (isAuthorizedToEditService(admin)) {
250                 addServiceInternal(admin, findFreeServiceId(), newname, servicedata);
251                 final String msg = intres.getLocalizedMessage("services.servicecloned", newname, oldname);
252                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
253                 details.put("msg", msg);
254                 auditSession.log(EjbcaEventTypes.SERVICE_ADD, EventStatus.SUCCESS, EjbcaModuleTypes.SERVICE, EjbcaServiceTypes.EJBCA,
255                         admin.toString(), null, null, null, details);
256             } else {
257                 final String msg = intres.getLocalizedMessage("services.notauthorizedtoedit", oldname);
258                 log.info(msg);
259             }
260         } catch (CloneNotSupportedException e) {
261             log.error("Error cloning service: ", e);
262             throw new EJBException(e);
263         }
264         log.trace("<cloneService()");
265     }

267     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
268     @Override
269     public boolean removeService(AuthenticationToken admin, String name) {
270         if (log.isTraceEnabled()) {
271             log.trace(">removeService(name: " + name + ")");
272         }
273         boolean retval = false;
274         try {
275             ServiceData htp = serviceDataSession.findByName(name);
276             if (htp == null) {
277                 throw new FinderException("Cannot find service " + name);
278             }
279             ServiceConfiguration serviceConfiguration = htp.getServiceConfiguration();
280             if (isAuthorizedToEditService(admin)) {
281                 IWorker worker = getWorker(serviceConfiguration, name, htp.getRunTimeStamp(), htp.getNextRunTimeStamp());
282                 if (worker != null) {
283                     serviceSession.cancelTimer(htp.getId());
284                 }
285                 serviceDataSession.removeServiceData(htp.getId());
286                 final String msg = intres.getLocalizedMessage("services.serviceremoved", name);
287                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
288                 details.put("msg", msg);
289                 auditSession.log(EjbcaEventTypes.SERVICE_REMOVE, EventStatus.SUCCESS, EjbcaModuleTypes.SERVICE, EjbcaServiceTypes.EJBCA,
290                         admin.toString(), null, null, null, details);
291                 retval = true;
292             } else {
293                 final String msg = intres.getLocalizedMessage("services.notauthorizedtoedit", name);
294                 log.info(msg);
295             }
296         } catch (Exception e) {
297             final String msg = intres.getLocalizedMessage("services.errorremovingservice", name);
298             final Map<String, Object> details = new LinkedHashMap<String, Object>();
299             details.put("msg", msg);
300             details.put("error", e.getMessage());
301             auditSession.log(EjbcaEventTypes.SERVICE_REMOVE, EventStatus.FAILURE, EjbcaModuleTypes.SERVICE, EjbcaServiceTypes.EJBCA,
302                     admin.toString(), null, null, null, details);
303         }
304         log.trace("<removeService)");
305         return retval;
306     }

382     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
383     @Override
384     public void activateServiceTimer(AuthenticationToken admin, String name) {
385         if (log.isTraceEnabled()) {
386             log.trace(">activateServiceTimer(name: " + name + ")");
387         }
388         ServiceData htp = serviceDataSession.findByName(name);
389         if (htp != null) {
390             ServiceConfiguration serviceConfiguration = htp.getServiceConfiguration();
391             if (isAuthorizedToEditService(admin)) {
392                 IWorker worker = getWorker(serviceConfiguration, name, htp.getRunTimeStamp(), htp.getNextRunTimeStamp());
393                 if (worker != null) {
394                     serviceSession.cancelTimer(htp.getId());
395                     if (serviceConfiguration.isActive() && worker.getNextInterval() != IInterval.DONT_EXECUTE) {
396                         addTimer(worker.getNextInterval() * 1000, htp.getId());
397                     }
398                 }
399             } else {
400                 final String msg = intres.getLocalizedMessage("services.notauthorizedtoedit", name);
401                 log.info(msg);
402             }
403         } else {
404             log.error("Can not find service: " + name);
405         }
406         log.trace("<activateServiceTimer()");
407     }

436     /**
437      * Method implemented from the TimerObject and is the main method of this session bean. It calls the work object for each object.
438      * 
439      * @param timer timer whose expiration caused this notification.
440      */
441     @Timeout
442     // Glassfish 2.1.1:
443     // "Timeout method ....timeoutHandler(javax.ejb.Timer)must have TX attribute of TX_REQUIRES_NEW or TX_REQUIRED or TX_NOT_SUPPORTED"
444     // JBoss 5.1.0.GA: We cannot mix timer updates with our EJBCA DataSource transactions.
445     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
446     public void timeoutHandler(Timer timer) {
447         if (log.isTraceEnabled()) {
448             log.trace(">ejbTimeout");
449         }
450         final long startOfTimeOut = System.currentTimeMillis();
451         long serviceInterval = IInterval.DONT_EXECUTE;
452         Integer timerInfo = (Integer) timer.getInfo();
453         if (timerInfo.equals(SERVICELOADER_ID)) {
454             if (log.isDebugEnabled()) {
455                 log.debug("Running the internal Service loader.");
456             }
457             load();
458         } else {
459             String serviceName = null;
460             try {
461                 serviceName = serviceDataSession.findNameById(timerInfo);
462             } catch (Throwable t) { // NOPMD: we really need to catch everything to not risk hanging somewhere in limbo
463                 log.warn("Exception finding service name: ", t); // if this throws, there is a failed database or similar
464                 // Unexpected error (probably database related). We need to reschedule the service with a default interval.
465                 addTimer(30 * 1000, timerInfo);
466             }
467             if (serviceName == null) {
468                 final String msg = intres.getLocalizedMessage("services.servicenotfound", timerInfo);
469                 log.info(msg);
470             } else {
471                 // Get interval of worker
472                 try {
473                     serviceInterval = serviceSession.getServiceInterval(timerInfo);
474                 } catch (Throwable t) { // NOPMD: we really need to catch everything to not risk hanging somewhere in limbo
475                     log.warn("Exception getting service interval: ", t); // if this throws, there is a failed database or similar
476                     // Unexpected error (probably database related). We need to reschedule the service with a default interval.
477                     addTimer(30 * 1000, timerInfo);
478                 }
479                 // Reschedule timer
480                 IWorker worker = null;
481                 if (serviceInterval != IInterval.DONT_EXECUTE) {
482                     Timer nextTrigger = addTimer(serviceInterval * 1000, timerInfo);
483                     try {
484                         // Try to acquire lock / see if this node should run
485                         worker = serviceSession.getWorkerIfItShouldRun(timerInfo, nextTrigger.getNextTimeout().getTime());
486                     } catch (Throwable t) { // NOPMD: we really need to catch everything to not risk hanging somewhere in limbo
487                         if (log.isDebugEnabled()) {
488                             log.debug("Exception: ", t); // Don't spam log with stacktraces in normal production cases
489                         }
490                     }
491                     if (worker != null) {
492                         try {
493                             serviceSession.executeServiceInNoTransaction(worker, serviceName);
494                         } catch (RuntimeException e) {
495                             /*
496                              * If the service worker fails with a RuntimeException we need to
497                              * swallow this here. If we allow it to propagate outside the
498                              * ejbTimeout method it is up to the application server config how it
499                              * should be retried, but we have already scheduled a new try
500                              * previously in this method. We still want to log this as an ERROR
501                              * since it is some kind of catastrophic failure..
502                              */
503                             log.error("Service worker execution failed.", e);
504                         }
505                     } else {
506                         if (log.isDebugEnabled()) {
507                             Object o = timerInfo;
508                             if (serviceName != null) {
509                                 o = serviceName;
510                             }
511                             final String msg = intres.getLocalizedMessage("services.servicerunonothernode", o);
512                             log.debug(msg);
513                         }
514                     }
515                     if (System.currentTimeMillis() - startOfTimeOut > serviceInterval * 1000) {
516                         log.warn("Service '" + serviceName + "' took longer than it's configured service interval ("+serviceInterval+")."
517                                 + " This can trigger simultanious service execution on several nodes in a cluster."
518                                 + " Increase interval or lower each invocations work load.");
519                     }
520                 }
521             }
522         }
523         if (log.isTraceEnabled()) {
524             log.trace("<ejbTimeout");
525         }
526     }

533     @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
534     @Override
535     public IWorker getWorkerIfItShouldRun(final Integer serviceId, final long nextTimeout, final boolean testRunOnOtherNode) {
536     	if (log.isTraceEnabled()) {
537     	    log.trace(">getWorkerIfItShouldRun: "+serviceId+", "+nextTimeout+", "+testRunOnOtherNode);
538     	}
539         IWorker worker = null;
540         ServiceData serviceData = serviceDataSession.findById(serviceId);
541         ServiceConfiguration serviceConfiguration = serviceData.getServiceConfiguration();
542         if (!serviceConfiguration.isActive()) {
543             if (log.isDebugEnabled()) {
544                 log.debug("Service " + serviceId + " is inactive.");
545             }
546             return null; // Don't return an inactive worker to run
547         }
548         String serviceName = serviceData.getName();
549         final String hostname = getHostName();
550         if (shouldRunOnThisNode(hostname, Arrays.asList(serviceConfiguration.getPinToNodes()))) {
551             long oldRunTimeStamp = serviceData.getRunTimeStamp();
552             long oldNextRunTimeStamp = serviceData.getNextRunTimeStamp();
553             worker = getWorker(serviceConfiguration, serviceName, oldRunTimeStamp, oldNextRunTimeStamp);
554             if (worker.getNextInterval() == IInterval.DONT_EXECUTE) {
555                 if (log.isDebugEnabled()) {
556                     log.debug("Service "+serviceName+" has interval IInterval.DONT_EXECUTE.");
557                 }
558                 return null; // Don't return an inactive worker to run
559             }
560             // nextRunDateCheck will typically be the same (or just a millisecond earlier) as now here
561             final Date runDateCheck = new Date(oldNextRunTimeStamp); 
562             final Date currentDate = new Date();
563             if (log.isDebugEnabled()) {
564                 final Date nextRunDate = new Date(nextTimeout);
565                 log.debug("nextRunDate is:  " + nextRunDate);
566                 log.debug("runDateCheck is: " + runDateCheck);
567                 log.debug("currentDate is:  " + currentDate);
568             }
569             // Check if this is a service that should run on all nodes, i.e. ignore if it is already running on another node in a cluster
570             // This is used for services that do (lighter) work local to each node, such as HSM keepalive service
571             if (!serviceConfiguration.isRunOnAllNodes()) {
572                 /*
573                  * Check if the current date is after when the service should run. If a
574                  * service on another cluster node has updated this timestamp already,
575                  * then it will return false and this service will not run. This is a
576                  * semaphore (not the best one admitted) so that services in a cluster
577                  * only runs on one node and don't compete with each other. If a worker
578                  * on one node for instance runs for a very long time, there is a chance
579                  * that another worker on another node will break this semaphore and run
580                  * as well.
581                  */
582                 if (currentDate.after(runDateCheck)) {
583                     /*
584                      * We only update the nextRunTimeStamp if the service is allowed to run on this node.
585                      * 
586                      * However, we need to make sure that no other node has already acquired the semaphore
587                      * if our current database allows non-repeatable reads.
588                      */
589                     final boolean updateTimestamps = serviceDataSession.updateTimestamps(serviceId, oldRunTimeStamp, oldNextRunTimeStamp, runDateCheck.getTime(), nextTimeout);
590                     if (!updateTimestamps || testRunOnOtherNode) {
591                         if (testRunOnOtherNode && updateTimestamps) {
592                             log.info("testRunOnOtherNode == true, we are returning null even though another node had not updated the database. This node will not run the service "+serviceName+".");
593                         } else {
594                             log.debug("Another node had already updated the database at this point. This node will not run the service "+serviceName+".");
595                         }
596                         worker = null; // Failed to update the database.                            
597                     } else {
598                         if (log.isTraceEnabled()) {
599                             log.trace("Timestamps updated, service "+serviceName+" will run: "+currentDate+", "+runDateCheck);
600                         }                        
601                     }
602                 } else {
603                     if (log.isTraceEnabled()) {
604                         log.trace("!currentDate.after(runDateCheck), service "+serviceName+" will not run: "+currentDate+", "+runDateCheck);
605                     }
606                     worker = null; // Don't return a worker, since this node should not run
607                 }
608             } else {
609                 if (log.isDebugEnabled()) {
610                     log.debug("Service " + serviceName + " is set to run on all nodes and will run on this node: \"" + hostname + "\", updating timeStamps");
611                 }
612                 // Always update timestamp so we have a record of running, and nextTimeout is set for service reload
613                 serviceDataSession.updateTimestamps(serviceId, oldRunTimeStamp, oldNextRunTimeStamp, runDateCheck.getTime(), nextTimeout);
614             }
615         } else {
616             worker = null;
617             if (log.isDebugEnabled()) {
618                 log.debug("Service " + serviceName + " will not run on this node: \"" + hostname + "\", Pinned to: "
619                         + Arrays.toString(serviceConfiguration.getPinToNodes()));
620             }
621         }
622         if (log.isTraceEnabled()) {
623             log.trace("<getWorkerIfItShouldRun: "+serviceName+", ret: "+ (worker != null ? worker.getClass().getName() : "null"));
624         }
625         return worker;
626     }

717     // We don't want the appserver to persist/update the timer in the same transaction if they are stored in different non XA DataSources
718     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
719     @Override
720     public void load() {
721         // Get all services
722         Collection<Timer> currentTimers = timerService.getTimers();
723         Iterator<Timer> iter = currentTimers.iterator();
724         HashSet<Serializable> existingTimers = new HashSet<Serializable>();
725         while (iter.hasNext()) {
726             Timer timer = iter.next();
727             try {
728                 Serializable info = timer.getInfo();
729                 existingTimers.add(info);
730             } catch (Throwable e) { // NOPMD: we really need to catch everything to not risk hanging somewhere in limbo
731                 // EJB 2.1 only?: We need this try because weblogic seems to ... suck ...
732                 log.debug("Error invoking timer.getInfo(): ", e);
733             }
734         }
735 
736         // Get new services and add timeouts
737         Map<Integer, Long> newTimeouts = serviceSession.getNewServiceTimeouts(existingTimers);
738         for (Integer id : newTimeouts.keySet()) {
739             addTimer(newTimeouts.get(id), id);
740         }
741 
742         if (!existingTimers.contains(SERVICELOADER_ID)) {
743             // load the service timer
744             addTimer(SERVICELOADER_PERIOD, SERVICELOADER_ID);
745         }
746     }

748     @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
749     @Override
750     public Map<Integer, Long> getNewServiceTimeouts(HashSet<Serializable> existingTimers) {
751         Map<Integer, Long> ret = new HashMap<Integer, Long>();
752         HashMap<Integer, String> idToNameMap = getServiceIdToNameMap();
753         Collection<Integer> allServices = idToNameMap.keySet();
754         Iterator<Integer> iter2 = allServices.iterator();
755         while (iter2.hasNext()) {
756             Integer id = iter2.next();
757             ServiceData htp = serviceDataSession.findById(id);
758             if (htp != null) {
759                 if (!existingTimers.contains(id)) {
760                     ServiceConfiguration serviceConfiguration = htp.getServiceConfiguration();
761                     IWorker worker = getWorker(serviceConfiguration, idToNameMap.get(id), htp.getRunTimeStamp(), htp.getNextRunTimeStamp());
762                     if (worker != null && serviceConfiguration.isActive() && worker.getNextInterval() != IInterval.DONT_EXECUTE) {
763                         ret.put(id, Long.valueOf((worker.getNextInterval()) * 1000));
764                     }
765                 }
766             } else {
767                 // Service does not exist, strange, but no panic.
768                 log.debug("Can not find service with id " + id);
769             }
770         }
771         return ret;
772     }

842     /**
843      * Method that creates a worker from the service configuration.
844      * 
845      * @param serviceConfiguration Config
846      * @param serviceName Name
847      * @param runTimeStamp the time this service runs
848      * @param nextRunTimeStamp the time this service will run next time
849      * @return a worker object or null if the worker is misconfigured.
850      */
851     private IWorker getWorker(ServiceConfiguration serviceConfiguration, String serviceName, long runTimeStamp, long nextRunTimeStamp) {
852         IWorker worker = null;
853         try {
854             String clazz = serviceConfiguration.getWorkerClassPath();
855             if (StringUtils.isNotEmpty(clazz)) {
856                 worker = (IWorker) Thread.currentThread().getContextClassLoader().loadClass(clazz).getConstructor().newInstance();
857                 worker.init(intAdmin, serviceConfiguration, serviceName, runTimeStamp, nextRunTimeStamp);
858             } else {
859                 log.info("Worker has empty classpath for service " + serviceName);
860             }
861         } catch (Exception e) {
862             // Only display a real error if it is a worker that we are actually
863             // using
864             if (serviceConfiguration.isActive()) {
865                 log.error("Worker is misconfigured, check the classpath", e);
866             } else {
867                 log.info("Worker is misconfigured, check the classpath: " + e.getMessage());
868             }
869         }
870         return worker;
871     }
