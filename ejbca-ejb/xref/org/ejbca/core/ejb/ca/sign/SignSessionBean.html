260     /**
261      * Internal helper method
262      *
263      * @param admin Information about the administrator or admin performing the event.
264      * @param caId  CA for which we want a PKCS7 certificate chain.
265      * @param cert  client certificate which we want encapsulated in a PKCS7 together with
266      *              certificate chain, or null
267      * @param includeChain nbool
268      * @return The DER-encoded PKCS7 message.
269      * @throws CADoesntExistsException if the CA does not exist or is expired, or has an invalid certificate
270      * @throws AuthorizationDeniedException if the authentication token wasn't authorized to the CA
271      * @throws SignRequestSignatureException if the certificate wasn't issued by the CA defined by caid
272      */
273     private byte[] createPKCS7(AuthenticationToken admin, int caId, X509Certificate cert, boolean includeChain)
274             throws CADoesntExistsException, SignRequestSignatureException, AuthorizationDeniedException {
275         if (log.isTraceEnabled()) {
276             log.trace(">createPKCS7(" + caId + ", " + CertTools.getIssuerDN(cert) + ")");
277         }
278         final CA ca = caSession.getCA(admin, caId);
279         final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(ca.getCAToken().getCryptoTokenId());
280         final byte[] returnval = ca.createPKCS7(cryptoToken, cert, includeChain);
281         if (returnval != null) {
282             // Audit log that we used the CA's signing key to create a CMS signature
283             final String detailsMsg = intres.getLocalizedMessage("caadmin.signedcms", ca.getName());
284             final Map<String, Object> details = new LinkedHashMap<>();
285             if (cert != null) {
286                 details.put("leafSubject", CertTools.getSubjectDN(cert));
287                 details.put("leafFingerprint", CertTools.getFingerprintAsString(cert));
288             }
289             details.put("includeChain", Boolean.toString(includeChain));
290             details.put("msg", detailsMsg);
291             securityEventsLoggerSession.log(EjbcaEventTypes.CA_SIGNCMS, EventStatus.SUCCESS, ModuleTypes.CA, ServiceTypes.CORE, admin.toString(),
292                     String.valueOf(caId), null, null, details);
293         }
294         if (log.isTraceEnabled()) {
295             log.trace("<createPKCS7()");
296         }
297         return returnval;
298     }

357     @Override
358     public Certificate createCertificate(final AuthenticationToken admin, final String username, final String password, final Certificate incert)
359             throws NoSuchEndEntityException, AuthorizationDeniedException, SignRequestSignatureException, CADoesntExistsException,
360             AuthStatusException, AuthLoginException, IllegalKeyException, CertificateCreateException, IllegalNameException,
361             CertificateRevokeException, CertificateSerialNumberException, CryptoTokenOfflineException, IllegalValidityException, CAOfflineException,
362             InvalidAlgorithmException, CustomCertificateSerialNumberException {
363 
364         // Convert the certificate to a BC certificate. SUN does not handle verifying RSASha256WithMGF1 for example 
365         Certificate bccert;
366         try {
367             bccert = CertTools.getCertfromByteArray(incert.getEncoded(), Certificate.class);
368             bccert.verify(incert.getPublicKey());
369         } catch (CertificateParsingException e) {
370             log.debug("CertificateParsingException verify POPO: ", e);
371             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
372             throw new SignRequestSignatureException(msg, e);
373         } catch (CertificateEncodingException e) {
374             log.debug("CertificateEncodingException verify POPO: ", e);
375             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
376             throw new SignRequestSignatureException(msg);
377         } catch (InvalidKeyException e) {
378             log.debug("InvalidKeyException verify POPO: ", e);
379             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
380             throw new SignRequestSignatureException(msg, e);
381         } catch (CertificateException e) {
382             log.debug("CertificateException verify POPO: ", e);
383             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
384             throw new SignRequestSignatureException(msg, e);
385         } catch (NoSuchAlgorithmException e) {
386             log.debug("NoSuchAlgorithmException verify POPO: ", e);
387             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
388             throw new SignRequestSignatureException(msg, e);
389         } catch (NoSuchProviderException e) {
390             log.debug("NoSuchProviderException verify POPO: ", e);
391             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
392             throw new SignRequestSignatureException(msg, e);
393         } catch (SignatureException e) {
394             log.debug("SignatureException verify POPO: ", e);
395             final String msg = intres.getLocalizedMessage("createcert.popverificationfailed");
396             throw new SignRequestSignatureException(msg, e);
397         }
398 
399         return createCertificate(admin, username, password, incert.getPublicKey(),
400                 CertTools.sunKeyUsageToBC(((X509Certificate) incert).getKeyUsage()), null, null);
401     }

433     @Override
434     public ResponseMessage createCertificate(final AuthenticationToken admin, final RequestMessage req,
435             Class<? extends CertificateResponseMessage> responseClass, final EndEntityInformation suppliedUserData)
436             throws AuthorizationDeniedException, CertificateExtensionException, NoSuchEndEntityException, CustomCertificateSerialNumberException,
437             CryptoTokenOfflineException, IllegalKeyException, CADoesntExistsException, SignRequestException, SignRequestSignatureException,
438             AuthStatusException, AuthLoginException, IllegalNameException, CertificateCreateException, CertificateRevokeException,
439             CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException {
440         if (log.isTraceEnabled()) {
441             log.trace(">createCertificate(RequestMessage)");
442         }
443         // Get CA that will receive request
444         EndEntityInformation endEntityInformation = null;
445         CertificateResponseMessage ret = null;
446         // Get CA object and make sure it is active
447         // Do not log access control to the CA here, that is logged later on when we use the CA to issue a certificate (if we get that far).
448         final CA ca;
449         if (suppliedUserData == null) {
450             ca = getCAFromRequest(admin, req, false);
451         } else {
452             ca = caSession.getCANoLog(admin, suppliedUserData.getCAId()); // Take the CAId from the supplied userdata, if any
453         }
454         if (ca.getStatus() != CAConstants.CA_ACTIVE) {
455             final String msg = intres.getLocalizedMessage("signsession.canotactive", ca.getSubjectDN());
456             throw new CAOfflineException(msg);
457         }
458         try {
459             // See if we need some key material to decrypt request
460             final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(ca.getCAToken().getCryptoTokenId());
461             setDecryptInfo(cryptoToken, req, ca);
462             if (ca.isUseUserStorage() && req.getUsername() == null) {
463                 String msg = intres.getLocalizedMessage("signsession.nouserinrequest", req.getRequestDN());
464                 throw new SignRequestException(msg);
465             } else if (ca.isUseUserStorage() && req.getPassword() == null) {
466                 String msg = intres.getLocalizedMessage("signsession.nopasswordinrequest");
467                 throw new SignRequestException(msg);
468             } else {
469                 try {
470                     // If we haven't done so yet, authenticate user. (Only if we store UserData for this CA.)
471                     if (ca.isUseUserStorage()) {
472                         endEntityInformation = authUser(admin, req.getUsername(), req.getPassword());
473                     } else {
474                         endEntityInformation = suppliedUserData;
475                     }
476                     // We need to make sure we use the users registered CA here
477                     if (endEntityInformation.getCAId() != ca.getCAId()) {
478                         final String failText = intres.getLocalizedMessage("signsession.wrongauthority", Integer.valueOf(ca.getCAId()),
479                                 Integer.valueOf(endEntityInformation.getCAId()));
480                         log.info(failText);
481                         ret = createRequestFailedResponse(admin, req, responseClass, FailInfo.WRONG_AUTHORITY, failText);
482                     } else {
483                         final long updateTime = System.currentTimeMillis();
484                         //Specifically check for the Single Active Certificate Constraint property, which requires that revocation happen in conjunction with renewal. 
485                         //We have to perform this check here, in addition to the true check in CertificateCreateSession, in order to be able to perform publishing. 
486                         singleActiveCertificateConstraint(admin, endEntityInformation);
487                         // Issue the certificate from the request
488                         ret = certificateCreateSession.createCertificate(admin, endEntityInformation, ca, req, responseClass, fetchCertGenParams(),
489                                 updateTime);
490                         postCreateCertificate(admin, endEntityInformation, ca,
491                                 new CertificateDataWrapper(ret.getCertificate(), ret.getCertificateData(), ret.getBase64CertData()));
492                     }
493                 } catch (NoSuchEndEntityException e) {
494                     // If we didn't find the entity return error message
495                     final String failText = intres.getLocalizedMessage("signsession.nosuchuser", req.getUsername());
496                     log.info(failText, e);
497                     throw new NoSuchEndEntityException(failText, e);
498                 }
499             }
500             ret.create();
501             // Call authentication session and tell that we are finished with this user. (Only if we store UserData for this CA.)
502             if (ca.isUseUserStorage() && endEntityInformation != null) {
503                 finishUser(ca, endEntityInformation);
504             }
505         } catch (CustomCertificateSerialNumberException e) {
506             cleanUserCertDataSN(endEntityInformation);

600     @Override
601     public Collection<CertificateWrapper> createCardVerifiableCertificateWS(final AuthenticationToken authenticationToken, final String username,
602             String password, final String cvcreq)
603             throws AuthorizationDeniedException, CADoesntExistsException, UserDoesntFullfillEndEntityProfile, NotFoundException, ApprovalException,
604             EjbcaException, WaitingForApprovalException, SignRequestException, CertificateExpiredException, CesecoreException {
605         // If password is empty we can generate a big random one to use instead.
606         if (StringUtils.isEmpty(password)) {
607             password = new AllPrintableCharPasswordGenerator().getNewPassword(15, 20);
608             log.debug("Using a long random password.");
609         }
610         // See if this user already exists.
611         // We allow renewal of certificates for IS's that are not revoked
612         // In that case look for it's last old certificate and try to authenticate the request using an outer signature.
613         // If this verification is correct, set status to NEW and continue process the request.
614         int oldUserStatus = EndEntityConstants.STATUS_GENERATED;
615         final EndEntityInformation user = endEntityAccessSession.findUser(authenticationToken, username);
616         try {
617             if (user != null) {
618                 oldUserStatus = user.getStatus();
619                 // If user is revoked, we can not proceed
620                 if ((oldUserStatus == EndEntityConstants.STATUS_REVOKED) || (oldUserStatus == EndEntityConstants.STATUS_HISTORICAL)) {
621                     throw new AuthorizationDeniedException("User '" + username + "' is revoked.");
622                 }
623                 final CVCObject parsedObject = CertificateParser.parseCVCObject(Base64.decode(cvcreq.getBytes()));
624                 if (parsedObject instanceof CVCAuthenticatedRequest) {
625                     if (log.isDebugEnabled()) {
626                         log.debug("Received an authenticated request, could be an initial DV request signed by CVCA or a renewal for DV or IS.");
627                     }
628                     final CVCAuthenticatedRequest request = (CVCAuthenticatedRequest) parsedObject;
629                     final CVCPublicKey publicKey = request.getRequest().getCertificateBody().getPublicKey();
630                     final String algorithm = AlgorithmUtil.getAlgorithmName(publicKey.getObjectIdentifier());
631                     if (log.isDebugEnabled()) {
632                         log.debug("Received request has a public key with algorithm: " + algorithm);
633                     }
634                     final HolderReferenceField holderReference = request.getRequest().getCertificateBody().getHolderReference();
635                     final CAReferenceField caReferenceField = request.getAuthorityReference();
636 
637                     // Check to see that the inner signature does not also verify using an old certificate
638                     // because that means the same keys were used, and that is not allowed according to the EU policy
639                     // This must be done whether it is signed by CVCA or a renewal request
640                     final Collection<Certificate> oldCertificates = EJBTools
641                             .unwrapCertCollection(certificateStoreSession.findCertificatesByUsername(username));
642                     if (oldCertificates != null) {
643                         if (log.isDebugEnabled()) {
644                             log.debug("Found " + oldCertificates.size() + " old certificates for user " + username);
645                         }
646                         PublicKey oldPublicKey;
647                         CVCertificate innerRequest;
648                         for (Certificate certificate : oldCertificates) {
649                             oldPublicKey = getCVPublicKey(authenticationToken, certificate);
650                             innerRequest = request.getRequest();
651                             // Throws AuthorizationDeniedException
652                             checkInnerCollision(oldPublicKey, innerRequest, holderReference.getConcatenated());
653                         }
654                     }
655                     boolean verifiedOuter = false; // So we can throw an error if we could not verify
656                     if (StringUtils.equals(holderReference.getMnemonic(), caReferenceField.getMnemonic())
657                             && StringUtils.equals(holderReference.getCountry(), caReferenceField.getCountry())) {
658                         if (log.isDebugEnabled()) {
659                             log.debug("Authenticated request is self signed, we will try to verify it using user's old certificate.");
660                         }
661                         final Collection<Certificate> userCertificates = EJBTools
662                                 .unwrapCertCollection(certificateStoreSession.findCertificatesByUsername(username));
663                         // userCertificates contains certificates ordered with last expire date first. Last expire date should be last issued cert
664                         // We have to iterate over available user certificates, because we don't know which on signed the old one
665                         // and cv certificates have very coarse grained validity periods so we can't really know which one is the latest one
666                         // if 2 certificates are issued the same day.
667                         if (userCertificates != null) {
668                             if (log.isDebugEnabled()) {
669                                 log.debug("Found " + userCertificates.size() + " old certificates for user " + username);
670                             }
671                             for (java.security.cert.Certificate certificate : userCertificates) {
672                                 try {
673                                     // Only allow renewal if the old certificate is valid
674                                     final PublicKey pk = getCVPublicKey(authenticationToken, certificate);
675                                     if (log.isDebugEnabled()) {
676                                         log.debug("Trying to verify the outer signature with an old certificate, fp: "
677                                                 + CertTools.getFingerprintAsString(certificate));
678                                     }
679                                     request.verify(pk);
680                                     if (log.isDebugEnabled()) {
681                                         log.debug("Verified outer signature.");
682                                     }
683                                     // Yes we did it, we can move on to the next step because the outer signature was actually created with some old certificate
684                                     verifiedOuter = true;
685                                     try {
686                                         // Check certificate validity and set end entity status/password.
687                                         // This will throw one of several exceptions if the certificate is invalid.
688                                         ejbcaWSHelperSession.checkValidityAndSetUserPassword(authenticationToken, certificate, username, password);
689                                         break;
690                                     } catch (EndEntityProfileValidationException e) {
691                                         throw new UserDoesntFullfillEndEntityProfile(e);
692                                     }
693                                     // If verification of outer signature fails because the signature is invalid we will break and deny the request...with a message
694                                 } catch (InvalidKeyException e) {
695                                     String msg = intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
696                                             e.getMessage());
697                                     log.warn(msg, e);
698                                 } catch (CertificateExpiredException e) { // thrown by checkValidityAndSetUserPassword
699                                     // Only log this with DEBUG since it will be a common case that happens, nothing that should cause any alerts.
700                                     if (log.isDebugEnabled()) {
701                                         log.debug(intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
702                                                 e.getMessage()));
703                                     }
704                                     // This exception we want to throw on, because we want to give this error if there was a certificate suitable for
705                                     // verification, but it had expired. This is thrown by checkValidityAndSetUserPassword after the request has already been
706                                     // verified using the public key of the certificate.
707                                     throw e;
708                                 } catch (CertificateException | NoSuchAlgorithmException | NoSuchProviderException e) {
709                                     String msg = intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
710                                             e.getMessage());
711                                     log.warn(msg, e);
712                                 } catch (SignatureException e) {
713                                     // Failing to verify the outer signature will be normal, since we must try all old certificates
714                                     if (log.isDebugEnabled()) {
715                                         log.debug(intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
716                                                 e.getMessage()));
717                                     }
718                                 }
719                             }
720                             // If verification failed because the old certificte was not yet valid, continue processing as usual, using the sent in username/password hoping the
721                             // status is NEW and password is correct. If old certificate was expired a CertificateExpiredException is thrown above.
722                         }
723                         // If there are no old certificates, continue processing as usual, using the sent in username/password hoping the
724                         // status is NEW and password is correct.
725                     } else { // if (StringUtils.equals(holderRef, caRef))
726                         // Subject and issuerDN is CN=Mnemonic,C=Country
727                         final String dn = "CN=" + caReferenceField.getMnemonic() + ",C=" + caReferenceField.getCountry();
728                         if (log.isDebugEnabled()) {
729                             log.debug("Authenticated request is not self signed, we will try to verify it using a CVCA certificate: " + dn);
730                         }
731                         final CAInfo info = caSession.getCAInfo(authenticationToken, CertTools.stringToBCDNString(dn).hashCode());
732                         if (info == null) {
733                             log.info("No CA found to authenticate request: " + dn);
734                             throw new CADoesntExistsException("CA with id " + CertTools.stringToBCDNString(dn).hashCode() + " doesn't exist.");
735                         } else {
736                             final Collection<Certificate> certificateChain = info.getCertificateChain();
737                             if (certificateChain != null) {
738                                 if (log.isDebugEnabled()) {
739                                     log.debug("Found " + certificateChain.size() + " certificates in chain for CA with DN: " + dn);
740                                 }
741                                 Iterator<Certificate> iterator = certificateChain.iterator();
742                                 if (iterator.hasNext()) {
743                                     // The CA certificate is first in chain.
744                                     final Certificate caCertificate = iterator.next();
745                                     if (log.isDebugEnabled()) {
746                                         log.debug("Trying to verify the outer signature with a CVCA certificate, fp: "
747                                                 + CertTools.getFingerprintAsString(caCertificate));
748                                     }
749                                     try {
750                                         // The CVCA certificate always contains the full key parameters, no need to du any EC curve parameter magic here
751                                         request.verify(caCertificate.getPublicKey());
752                                         if (log.isDebugEnabled()) {
753                                             log.debug("Verified outer signature");
754                                         }
755                                         verifiedOuter = true;
756                                         // Yes we did it, we can move on to the next step because the outer signature was actually created with some old certificate
757                                         try {
758                                             // Check certificate validity and set end entity status/password.
759                                             // This will throw one of several exceptions if the certificate is invalid.
760                                             ejbcaWSHelperSession.checkValidityAndSetUserPassword(authenticationToken, caCertificate, username,
761                                                     password);
762                                         } catch (EndEntityProfileValidationException e) {
763                                             throw new UserDoesntFullfillEndEntityProfile(e);
764                                         }
765                                     } catch (InvalidKeyException | CertificateException | NoSuchAlgorithmException | NoSuchProviderException
766                                             | SignatureException e) {
767                                         log.warn(intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
768                                                 e.getMessage()), e);
769                                     }
770                                 }
771                             } else {
772                                 log.info("No CA certificate found to authenticate request: " + dn);
773                             }
774                         }
775                     }
776                     // If verification failed because we could not verify the outer signature at all it is an error.
777                     if (!verifiedOuter) {
778                         final String msg = intres.getLocalizedMessage("cvc.error.outersignature", holderReference.getConcatenated(),
779                                 "No certificate found that could authenticate request");
780                         log.info(msg);
781                         throw new AuthorizationDeniedException(msg);
782                     }
783                 } // if (parsedObject instanceof CVCAuthenticatedRequest)
784                   // If it is not an authenticated request, with an outer signature, continue processing as usual,
785                   // using the sent in username/password hoping the status is NEW and password is correct.
786             } else {
787                 // If there are no old user, continue processing as usual... it will fail
788                 log.debug("No existing user with username: " + username);
789             }
790         } catch (ParseException | ConstructionException | NoSuchFieldException e) {
791             ejbcaWSHelperSession.resetUserPasswordAndStatus(authenticationToken, username, oldUserStatus);
792             throw new EjbcaException(ErrorCode.INTERNAL_ERROR, e.getMessage());
793         }
794 
795         // Finally generate the certificate (assuming user status is NEW and the password is correct.
796         try {
797             final byte[] response = createCertificateWS(authenticationToken, username, password, cvcreq, CertificateConstants.CERT_REQ_TYPE_CVC, null,
798                     CertificateHelper.RESPONSETYPE_CERTIFICATE);
799             final CertificateResponse certificateResponse = new CertificateResponse(CertificateHelper.RESPONSETYPE_CERTIFICATE, response);
800             final byte[] b64cert = certificateResponse.getData();
801             final CVCertificate certObject = CertificateParser.parseCertificate(Base64.decode(b64cert));
802             final ArrayList<Certificate> result = new ArrayList<>();
803             result.add(new CardVerifiableCertificate(certObject));
804             // Get the certificate chain.
805             if (user != null) {
806                 final int caid = user.getCAId();
807                 caSession.verifyExistenceOfCA(caid);
808                 result.addAll(getCertificateChain(caid));
809             }
810             log.trace("<cvcRequest");
811             return EJBTools.wrapCertCollection(result);
812         } catch (NoSuchEndEntityException | ParseException | ConstructionException | NoSuchFieldException
813                 | InvalidKeyException | CertificateException // | CertificateEncodingException
814                 | CertificateExtensionException | InvalidKeySpecException | NoSuchAlgorithmException | NoSuchProviderException | SignatureException
815                 | IOException e) {
816             ejbcaWSHelperSession.resetUserPasswordAndStatus(authenticationToken, username, oldUserStatus);
817             throw new EjbcaException(ErrorCode.INTERNAL_ERROR, e.getMessage());
818         }
819     }

821     /** Method that gets the public key from a CV certificate, possibly enriching it with domain parameters 
822      *  from the CVCA certificate if it is an EC public key.
823      *
824      * @param admin the authentication token.
825      * @param certificate the certificate to get the public ket from.
826      * @return the certificates public key.
827      * @throws CADoesntExistsException if the CA of the certificate does not exist.
828      * @throws AuthorizationDeniedException if authorization was denied.
829      */
830     private PublicKey getCVPublicKey(final AuthenticationToken admin, final Certificate certificate)
831             throws CADoesntExistsException, AuthorizationDeniedException {
832         PublicKey publicKey = certificate.getPublicKey();
833         if (publicKey instanceof PublicKeyEC) {
834             // The public key of IS and DV certificate do not have any EC parameters so we have to do some magic to get a complete EC public key
835             // First get to the CVCA certificate that has the parameters
836             final CAInfo caInfo = caSession.getCAInfo(admin, CertTools.getIssuerDN(certificate).hashCode());
837             if (caInfo == null) {
838                 throw new CADoesntExistsException("CA with id " + CertTools.getIssuerDN(certificate).hashCode() + " doesn't exist.");
839             }
840             final List<Certificate> caCertificates = caInfo.getCertificateChain();
841             if (caCertificates != null) {
842                 if (log.isDebugEnabled()) {
843                     log.debug("Found CA certificate chain of length: " + caCertificates.size());
844                 }
845                 // Get the last certificate in the chain, it is the CVCA certificate.
846                 if (CollectionUtils.isNotEmpty(caCertificates)) {
847                     // Do the magic adding of parameters, if they don't exist in the public key.
848                     final Certificate cvcaCertificate = caCertificates.get(caCertificates.size() - 1);
849                     try {
850                         publicKey = KeyTools.getECPublicKeyWithParams(publicKey, cvcaCertificate.getPublicKey());
851                     } catch (InvalidKeySpecException e) {
852                         String msg = intres.getLocalizedMessage("cvc.error.outersignature", CertTools.getSubjectDN(certificate), e.getMessage());
853                         log.warn(msg, e);
854                     }
855                 }
856             }
857         }
858         return publicKey;
859     }

1091         } catch (NoSuchAlgorithmException e) {

1097             log.error(msg, ctoe);

1101         }
1102         if (log.isTraceEnabled()) {

1104         }

1273     /**Specifically check for the Single Active Certificate Constraint property, which requires that revocation happen in conjunction with renewal. 
1274     * We have to perform this check here, in addition to the true check in CertificateCreateSession, in order to be able to perform publishing.
1275     * 
1276     * @param admin AuthenticationToken used for revoking the certificate
1277     * @param endEntityInformation EndEntityInformation containing username, DN and certificate profile id
1278      * @throws CertificateRevokeException Fail
1279      * @throws AuthorizationDeniedException Fail
1280     */
1281     private void singleActiveCertificateConstraint(final AuthenticationToken admin, final EndEntityInformation endEntityInformation)
1282             throws CertificateRevokeException, AuthorizationDeniedException {
1283         if (log.isTraceEnabled()) {
1284             log.trace(">singleActiveCertificateConstraint()");
1285         }
1286         final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(endEntityInformation.getCertificateProfileId());
1287         if (certProfile.isSingleActiveCertificateConstraint()) {
1288             // Only get not yet expired certificates with status CERT_ACTIVE, CERT_NOTIFIEDABOUTEXPIRATION, CERT_REVOKED
1289             final List<CertificateDataWrapper> cdws = certificateStoreSession.getCertificateDataByUsername(endEntityInformation.getUsername(), true,
1290                     Arrays.asList(CertificateConstants.CERT_ARCHIVED, CertificateConstants.CERT_INACTIVE, CertificateConstants.CERT_ROLLOVERPENDING,
1291                             CertificateConstants.CERT_UNASSIGNED));
1292             List<Integer> publishers = certProfile.getPublisherList();
1293             if (log.isDebugEnabled()) {
1294                 log.debug("SingleActiveCertificateConstraint, found " + cdws.size() + " old (non expired, active) certificates and "
1295                         + publishers.size() + " publishers.");
1296             }
1297             for (final CertificateDataWrapper cdw : cdws) {
1298                 final CertificateData certificateData = cdw.getCertificateData();
1299                 if (certificateData.getStatus() == CertificateConstants.CERT_REVOKED
1300                         && certificateData.getRevocationReason() != RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD) {
1301                     continue;
1302                 }
1303                 //Go directly to RevocationSession and not via EndEntityManagementSession because we don't care about approval checks and so forth, 
1304                 //the certificate must be revoked nonetheless. 
1305                 revocationSession.revokeCertificate(admin, cdw, publishers, new Date(), RevokedCertInfo.REVOCATION_REASON_SUPERSEDED,
1306                         endEntityInformation.getDN());
1307             }
1308         }
1309         if (log.isTraceEnabled()) {
1310             log.trace("<singleActiveCertificateConstraint()");
1311         }
1312     }

1354     public byte[] signPayload(final byte[] data, final int signingCaId)
1355             throws AuthorizationDeniedException, CryptoTokenOfflineException, CADoesntExistsException, SignRequestSignatureException {
1356         if (log.isDebugEnabled()) {
1357             log.debug("Attempting to sign payload from CA with ID " + signingCaId);
1358         }
1359         CA ca = caSession.getCA(new AlwaysAllowLocalAuthenticationToken("Called from SignSessionBean.signPayload"), signingCaId);
1360         if (ca == null) {
1361             log.debug("CA with ID " + signingCaId + " does not exist.");
1362             throw new CADoesntExistsException("CA with ID " + signingCaId + " does not exist.");
1363         }
1364         CAToken catoken = ca.getCAToken();
1365         CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(catoken.getCryptoTokenId());
1366         PrivateKey privateKey = cryptoToken.getPrivateKey(catoken.getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN));
1367         if (log.isDebugEnabled()) {
1368             if (privateKey == null) {
1369                 log.debug("Could not retrieve private key from CA with ID " + signingCaId);
1370             }
1371         }
1372         final X509Certificate signerCert;
1373         try {
1374             signerCert = (X509Certificate) ca.getCACertificate();
1375         } catch (ClassCastException e) {
1376             throw new IllegalStateException("Not possible to sign a payload using a CV CA", e);
1377         }
1378         final String provider = cryptoToken.getSignProviderName();
1379         CMSSignedDataGenerator gen = new CMSSignedDataGenerator();
1380         String signatureAlgorithmName = AlgorithmTools.getAlgorithmNameFromDigestAndKey(ca.getCAToken().getSignatureAlgorithm(),
1381                 privateKey.getAlgorithm());
1382         try {
1383             ContentSigner contentSigner = new JcaContentSignerBuilder(signatureAlgorithmName).setProvider(provider).build(privateKey);
1384             JcaDigestCalculatorProviderBuilder calculatorProviderBuilder = new JcaDigestCalculatorProviderBuilder()
1385                     .setProvider(BouncyCastleProvider.PROVIDER_NAME);
1386             JcaSignerInfoGeneratorBuilder builder = new JcaSignerInfoGeneratorBuilder(calculatorProviderBuilder.build());
1387             gen.addSignerInfoGenerator(builder.build(contentSigner, signerCert));
1388             gen.addCertificates(new CollectionStore<>(CertTools.convertToX509CertificateHolder(Arrays.asList(signerCert))));
1389             CMSSignedData sigData = gen.generate(new CMSProcessableByteArray(data), true);
1390             return sigData.getEncoded();
1391         } catch (CMSException | CertificateEncodingException | IOException | OperatorCreationException e) {
1392             log.debug("Given payload could not be signed.", e);
1393             throw new SignRequestSignatureException("Given payload could not be signed.", e);
1394         }
1395     }
