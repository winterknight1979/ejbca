51  /**
52   * Used for evoking certificates in the system, manages revocation by:
53   * - Setting revocation status in the database (using certificate store)
54   * - Publishing revocations to publishers 
55   * 
56   * @version $Id: RevocationSessionBean.java 28699 2018-04-12 19:21:46Z samuellb $
57   */
58  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "RevocationSessionRemote")
59  @TransactionAttribute(TransactionAttributeType.REQUIRED)
60  public class RevocationSessionBean implements RevocationSessionLocal, RevocationSessionRemote {
61  
62      private static final Logger log = Logger.getLogger(RevocationSessionBean.class);
63  
64      @PersistenceContext(unitName = CesecoreConfiguration.PERSISTENCE_UNIT)
65      private EntityManager entityManager;
66  
67      @EJB
68      private SecurityEventsLoggerSessionLocal auditSession;
69      @EJB
70      private CertificateStoreSessionLocal certificateStoreSession;
71      @EJB
72      private CrlStoreSessionLocal crlStoreSession;
73      @EJB
74      private NoConflictCertificateStoreSessionLocal noConflictCertificateStoreSession;
75      @EJB
76      private PublisherSessionLocal publisherSession;
77  
78      /** Internal localization of logs and errors */
79      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
80  
81      @TransactionAttribute(TransactionAttributeType.REQUIRED)
82      @Override
83      public void revokeCertificate(final AuthenticationToken admin, final Certificate cert, final Collection<Integer> publishers, final int reason, final String userDataDN) throws CertificateRevokeException, AuthorizationDeniedException {
84          final Date revokedate = new Date();
85          final CertificateDataWrapper cdw = certificateStoreSession.getCertificateData(CertTools.getFingerprintAsString(cert));
86          if (cdw != null) { 
87              revokeCertificate(admin, cdw, publishers, revokedate, reason, userDataDN);
88          } else {
89              final String msg = intres.getLocalizedMessage("store.errorfindcertserno", CertTools.getSerialNumberAsString(cert));              
90              log.info(msg);
91              throw new CertificateRevokeException(msg);
92          }
93      }
94   
95      @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
96      @Override
97      public void revokeCertificateInNewTransaction(final AuthenticationToken admin, final CertificateDataWrapper cdw,
98              final Collection<Integer> publishers, Date revocationDate, final int reason, final String userDataDN)
99              throws CertificateRevokeException, AuthorizationDeniedException {
100         revokeCertificate(admin, cdw, publishers, revocationDate, reason, userDataDN);
101     }
102     
103     @TransactionAttribute(TransactionAttributeType.REQUIRED)
104     @Override
105     public void revokeCertificate(final AuthenticationToken admin, final CertificateDataWrapper cdw, final Collection<Integer> publishers,
106             Date revocationDate, final int reason, final String userDataDN) throws CertificateRevokeException, AuthorizationDeniedException {
107     	final boolean waschanged = noConflictCertificateStoreSession.setRevokeStatus(admin, cdw, getRevocationDate(cdw, revocationDate, reason), reason);
108     	// Only publish the revocation if it was actually performed
109     	if (waschanged) {
110     	    // Since storeSession.findCertificateInfo uses a native query, it does not pick up changes made above
111     	    // that is part if the transaction in the EntityManager, so we need to get the object from the EntityManager.
112     	    final BaseCertificateData certificateData = cdw.getBaseCertificateData();
113     	    final String username = certificateData.getUsername();
114     	    final String password = null;
115     		if (!RevokedCertInfo.isRevoked(reason)) {
116     			// unrevocation, -1L as revocationDate
117     		    final boolean published = publisherSession.storeCertificate(admin, publishers, cdw, password, userDataDN, null);
118         		if (published) {
119         			log.info(intres.getLocalizedMessage("store.republishunrevokedcert", Integer.valueOf(reason)));
120         		} else {
121         		    final String serialNumber = certificateData.getSerialNumberHex();
122             		// If it is not possible, only log error but continue the operation of not revoking the certificate
123         			final String msg = "Unrevoked cert:" + serialNumber + " reason: " + reason + " Could not be republished.";
124         			final Map<String, Object> details = new LinkedHashMap<>();
125                 	details.put("msg", msg);
126                 	final int caid = certificateData.getIssuerDN().hashCode();
127                 	auditSession.log(EjbcaEventTypes.REVOKE_UNREVOKEPUBLISH, EventStatus.FAILURE, ModuleTypes.CA, ServiceTypes.CORE, admin.toString(), String.valueOf(caid), serialNumber, username, details);
128         		}    			
129     		} else {
130     			// revocation
131                 publisherSession.storeCertificate(admin, publishers, cdw, password, userDataDN, null);
132     		}
133     	}
134     }
135 
136     /** @param cdw CDW
137      * @param revocationDate Date
138      * @param reason Reason
139      * @return revocationDate as is, or null if unrevoking a certificate that's not on a base CRL in on hold state. */
140     private Date getRevocationDate(final CertificateDataWrapper cdw, final Date revocationDate, final int reason) {
141         if (revocationDate == null || (reason != RevokedCertInfo.NOT_REVOKED && reason != RevokedCertInfo.REVOCATION_REASON_REMOVEFROMCRL) ||
142                 (cdw.getBaseCertificateData().getRevocationReason() != RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD)) {
143             return revocationDate; // return unmodified
144         }
145         
146         final String issuerDN = cdw.getBaseCertificateData().getIssuerDN();
147         final CRLInfo baseCrlInfo = crlStoreSession.getLastCRLInfo(issuerDN, false);
148         if (baseCrlInfo == null || baseCrlInfo.getCreateDate().before(revocationDate)) { // if not on base CRL
149             return null;
150         } else {
151             return revocationDate;
152         }
153     }
