87      @PostConstruct
88      public void postConstruct() {
89          publisherQueueSession = sessionContext.getBusinessObject(PublisherQueueSessionLocal.class);
90          // Keep track of number of instances of this bean, so we can free the executorService thread pool when the last is destroyed
91          beanInstanceCount.incrementAndGet();
92      }

94      @PreDestroy
95      public void preDestroy() {
96          // Shut down the thread pool when the last instance of this SSB is destroyed
97          if (beanInstanceCount.decrementAndGet() == 0) {
98              executorServiceLock.lock();
99              try {
100                 if (executorService != null) {
101                     executorService.shutdown();
102                     executorService = null;
103                 }
104             } finally {
105                 executorServiceLock.unlock();
106             }
107         }
108     }
109 
110     /** @return a reference to the "CachedThreadPool" executor service (creating one if needed). */
111     private ExecutorService getExecutorService() {
112         if (executorService == null) {
113             executorServiceLock.lock();
114             try {
115                 if (executorService == null) {
116                     executorService = Executors.newCachedThreadPool();
117                 }
118             } finally {
119                 executorServiceLock.unlock();
120             }
121         }
122         return executorService;
123     }

160     @Override
161     public Collection<PublisherQueueData> getPendingEntriesForPublisher(int publisherId) {
162         if (log.isTraceEnabled()) {
163             log.trace(">getPendingEntriesForPublisher(publisherId: " + publisherId + ")");
164         }
165         Collection<org.ejbca.core.ejb.ca.publisher.PublisherQueueData> datas = org.ejbca.core.ejb.ca.publisher.PublisherQueueData
166                 .findDataByPublisherIdAndStatus(entityManager, publisherId, PublisherConst.STATUS_PENDING, 0);
167         if (datas.isEmpty()) {
168             log.debug("No publisher queue entries found for publisher " + publisherId);
169         }
170         Collection<PublisherQueueData> ret = new ArrayList<PublisherQueueData>();
171         Iterator<org.ejbca.core.ejb.ca.publisher.PublisherQueueData> iter = datas.iterator();
172         while (iter.hasNext()) {
173             org.ejbca.core.ejb.ca.publisher.PublisherQueueData d = iter.next();
174             PublisherQueueData pqd = new PublisherQueueData(d.getPk(), new Date(d.getTimeCreated()), new Date(d.getLastUpdate()),
175                     d.getPublishStatus(), d.getTryCounter(), d.getPublishType(), d.getFingerprint(), d.getPublisherId(),
176                     d.getPublisherQueueVolatileData());
177             ret.add(pqd);
178         }
179         log.trace("<getPendingEntriesForPublisher()");
180         return ret;
181     }

188     @Override
189     public int[] getPendingEntriesCountForPublisherInIntervals(int publisherId, int[] lowerBounds, int[] upperBounds) {
190         if (log.isTraceEnabled()) {
191             log.trace(">getPendingEntriesCountForPublisherInIntervals(publisherId: " + publisherId + ", lower:" + Arrays.toString(lowerBounds)
192                     + ", upper:" + Arrays.toString(upperBounds) + ")");
193         }
194         if (lowerBounds.length != upperBounds.length) {
195             throw new IllegalArgumentException("lowerBounds and upperBounds must have equal length");
196         }
197         List<Integer> entryCountList = org.ejbca.core.ejb.ca.publisher.PublisherQueueData.findCountOfPendingEntriesForPublisher(entityManager,
198                 publisherId, lowerBounds, upperBounds);
199         int[] result = new int[lowerBounds.length];
200         for (int i = 0; i < lowerBounds.length && i < result.length; i++) {
201             result[i] = entryCountList.get(i).intValue();
202         }
203         log.trace("<getPendingEntriesCountForPublisherInIntervals()");
204         return result;
205     }

255     @Override
256     public void updateData(String pk, int status, int tryCounter) {
257         if (log.isTraceEnabled()) {
258             log.trace(">updateData(pk: " + pk + ", status: " + status + ")");
259         }
260         org.ejbca.core.ejb.ca.publisher.PublisherQueueData data = org.ejbca.core.ejb.ca.publisher.PublisherQueueData.findByPk(entityManager, pk);
261         if (data != null) {
262             if (status > 0) {
263                 data.setPublishStatus(status);
264             }
265             data.setLastUpdate(new Date().getTime());
266             if (tryCounter > -1) {
267                 data.setTryCounter(tryCounter);
268             }
269         } else {
270             log.debug("Trying to set status on nonexisting data, pk: " + pk);
271         }
272         log.trace("<updateData()");
273     }

275     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
276     @Override
277     public void plainFifoTryAlwaysLimit100EntriesOrderByTimeCreated(AuthenticationToken admin, int publisherId, BasePublisher publisher) {
278         int successcount = 0;
279         // Repeat this process as long as we actually manage to publish something
280         // this is because when publishing starts to work we want to publish everything in one go, if possible.
281         // However we don't want to publish more than 20000 certificates each time, because we want to commit to the database some time as well.
282         int totalcount = 0;
283         do {
284             successcount = publisherQueueSession.doChunk(admin, publisherId, publisher);
285             totalcount += successcount;
286         } while ((successcount > 0) && (totalcount < 20000));
287     }

290     @Override
291     public int doChunk(AuthenticationToken admin, int publisherId, BasePublisher publisher) {
292         final Collection<PublisherQueueData> c = getPendingEntriesForPublisherWithLimit(publisherId, 100, 60, "order by timeCreated");
293         return doPublish(admin, publisherId, publisher, c);

296     /** @param admin Admin
297      * @param publisherId ID
298      * @param publisher Pub
299      * @param c Data
300      * @return how many publishes that succeeded */
301     private int doPublish(AuthenticationToken admin, int publisherId, BasePublisher publisher, Collection<PublisherQueueData> c) {
302         if (log.isDebugEnabled()) {
303             log.debug("Found " + c.size() + " certificates to republish for publisher " + publisherId);
304         }
305         int successcount = 0;
306         int failcount = 0;
307 
308         for (PublisherQueueData pqd : c) {
309 
310             String fingerprint = pqd.getFingerprint();
311             int publishType = pqd.getPublishType();
312             if (log.isDebugEnabled()) {
313                 log.debug("Publishing from queue to publisher: " + publisherId + ", fingerprint: " + fingerprint + ", pk: " + pqd.getPk()
314                         + ", type: " + publishType);
315             }
316             PublisherQueueVolatileInformation voldata = pqd.getVolatileData();
317             String password = null;
318             ExtendedInformation ei = null;
319             String userDataDN = null;
320             if (voldata != null) {
321                 password = voldata.getPassword();
322                 ei = voldata.getExtendedInformation();
323                 userDataDN = voldata.getUserDN();
324             }
325             boolean published = false;
326 
327             try {
328                 if (publishType == PublisherConst.PUBLISH_TYPE_CERT) {
329                     if (log.isDebugEnabled()) {
330                         log.debug("Publishing Certificate");
331                     }
332                     if (publisher != null) {
333                         // Read the actual certificate and try to publish it
334                         // again
335                         // TODO: we might need change fetch-type for all but the
336                         // actual cert or a native query w SqlResultSetMapping..
337                         final CertificateDataWrapper certificateDataWrapper = noConflictCertificateStoreSession.getCertificateData(fingerprint);
338                         if (certificateDataWrapper==null) {
339                             throw new FinderException();
340                         }
341                         try {
342                             published = publisherQueueSession.storeCertificateNonTransactional(publisher, admin, certificateDataWrapper, password, userDataDN, ei);
343                         } catch (EJBException e) {
344                             final Throwable t = e.getCause();
345                             if (t instanceof PublisherException) {
346                                 throw (PublisherException) t;
347                             } else {
348                                 throw e;
349                             }
350                         }
351                     } else {
352                         String msg = intres.getLocalizedMessage("publisher.nopublisher", publisherId);
353                         log.info(msg);
354                     }
355                 } else if (publishType == PublisherConst.PUBLISH_TYPE_CRL) {
356                     if (log.isDebugEnabled()) {
357                         log.debug("Publishing CRL");
358                     }
359 
360                     CRLData crlData = CRLData.findByFingerprint(entityManager, fingerprint);
361 
362                     if (crlData == null) {
363                         throw new FinderException();
364                     }
365                     try {
366                         published = publisherQueueSession.storeCRLNonTransactional(publisher, admin, crlData.getCRLBytes(),
367                                 crlData.getCaFingerprint(), crlData.getCrlNumber(), userDataDN);
368                     } catch (EJBException e) {
369                         final Throwable t = e.getCause();
370                         if (t instanceof PublisherException) {
371                             throw (PublisherException) t;
372                         } else {
373                             throw e;
374                         }
375                     }
376                 } else {
377                     String msg = intres.getLocalizedMessage("publisher.unknowntype", publishType);
378                     log.error(msg);
379                 }
380             } catch (FinderException e) {
381                 final String msg = intres.getLocalizedMessage("publisher.errornocert", fingerprint) + e.getMessage();
382                 log.info(msg);
383             } catch (PublisherException e) {
384                 // Publisher session have already logged this error nicely to
385                 // getLogSession().log
386                 log.debug(e.getMessage());
387                 // We failed to publish, update failcount so we can break early
388                 // if nothing succeeds but everything fails.
389                 failcount++;
390             }
391             if (published) {
392 
393                 if (publisher.getKeepPublishedInQueue()) {
394                     // Update with information that publishing was successful
395                     updateData(pqd.getPk(), PublisherConst.STATUS_SUCCESS, pqd.getTryCounter());
396                 } else {
397                     // We are done with this one.. nuke it!
398                     removeQueueData(pqd.getPk());
399                 }
400 
401                 successcount++; // jipeee update success counter
402             } else {
403                 // Update with new tryCounter, but same status as before
404                 int tryCount = pqd.getTryCounter() + 1;
405                 updateData(pqd.getPk(), pqd.getPublishStatus(), tryCount);
406             }
407             // If we don't manage to publish anything, but fails on all the
408             // first ten ones we expect that this publisher is dead for now. We
409             // don't have to try with every record.
410             if ((successcount == 0) && (failcount > 10)) {
411                 if (log.isDebugEnabled()) {
412                     log.debug("Breaking out of publisher loop because everything seems to fail (at least the first 10 entries)");
413                 }
414                 break;
415             }
416         }
417 
418         if (log.isDebugEnabled()) {
419             log.debug("Returning from publisher with " + successcount + " entries published successfully.");
420         }
421         return successcount;
422     }

424     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
425     @Override
426     public boolean storeCertificateNonTransactional(BasePublisher publisher, AuthenticationToken admin, CertificateDataWrapper certWrapper,
427             String password, String userDN, ExtendedInformation extendedinformation) throws PublisherException {
428         if (publisher.isFullEntityPublishingSupported()) {
429             return publisher.storeCertificate(admin, certWrapper.getCertificateDataOrCopy(), certWrapper.getBase64CertData(), password, userDN, extendedinformation);
430         } else {
431             final BaseCertificateData certificateData = certWrapper.getBaseCertificateData();
432             final String cAFingerprint = certificateData.getCaFingerprint();
433             final int status = certificateData.getStatus();
434             final int type = certificateData.getType();
435             final long revocationDate = certificateData.getRevocationDate();
436             final int revocationReason = certificateData.getRevocationReason();
437             final String username = certificateData.getUsername();
438             final String tag  = certificateData.getTag();
439             final Integer certificateProfileId = certificateData.getCertificateProfileId();
440             final Long updateTime = certificateData.getUpdateTime();
441             return publisher.storeCertificate(admin, certWrapper.getCertificate(), username, password, userDN, cAFingerprint, status, type, revocationDate,
442                     revocationReason, tag, certificateProfileId, updateTime, extendedinformation);
443         }
444     }

454     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
455     @Override
456     public List<Object> storeCertificateNonTransactionalInternal(final List<BasePublisher> publishers, final AuthenticationToken admin,
457             final CertificateDataWrapper certWrapper, final String password, final String userDN, final ExtendedInformation extendedinformation) {
458         final List<Object> publisherResults = new ArrayList<Object>();
459         @SuppressWarnings("deprecation")
460 		final boolean parallel = EjbcaConfiguration.isPublishParallelEnabled();
461         // Are we doing parallel publishing (only meaningful if there is more than one publisher configured)?
462         if (parallel && publishers.size() > 1) {
463             final List<Future<Boolean>> futures = new ArrayList<Future<Boolean>>();
464             BasePublisher publisherFirst = null;
465             for (final BasePublisher publisher : publishers) {
466                 if (publisherFirst == null) {
467                     // We will execute the first of the publishers in the main thread...
468                     publisherFirst = publisher;
469                 } else {
470                     // ...and the rest of the publishers will be executed in new threads
471                     final Future<Boolean> future = getExecutorService().submit(new Callable<Boolean>() {
472                         @Override
473                         public Boolean call() throws Exception {
474                             if (!storeCertificateNonTransactional(publisher, admin, certWrapper, password, userDN, extendedinformation)) {
475                                 throw new PublisherException("Return code from publisher is false.");
476                             }
477                             return Boolean.TRUE;
478                         }
479                     });
480                     futures.add(future);
481                 }
482             }
483             // Wait at most 300 seconds in total for all the publishers to complete.
484             final long deadline = System.currentTimeMillis() + 300000L;
485             // Execute the first publishing in the calling thread
486             Object publisherResultFirst;
487             try {
488                 if (!storeCertificateNonTransactional(publisherFirst, admin, certWrapper, password, userDN, extendedinformation)) {
489                     throw new PublisherException("Return code from publisher is false.");
490                 }
491                 publisherResultFirst = Boolean.TRUE;
492             } catch (Exception e) {
493                 publisherResultFirst = getAsPublisherException(e);
494             }
495             publisherResults.add(publisherResultFirst);
496             // Wait for all the background threads to finish and get the result from each invocation
497             for (final Future<Boolean> future : futures) {
498                 Object publisherResult;
499                 try {
500                     final long maxTimeToWait = Math.max(1000L, deadline - System.currentTimeMillis());
501                     publisherResult = Boolean.valueOf(future.get(maxTimeToWait, TimeUnit.MILLISECONDS));
502                 } catch (Exception e) {
503                     publisherResult = getAsPublisherException(e);
504                 }
505                 publisherResults.add(publisherResult);
506             }
507         } else {
508             // Perform publishing sequentially (old fall back behavior)
509             for (final BasePublisher publisher : publishers) {
510                 try {
511                     if (!storeCertificateNonTransactional(publisher, admin, certWrapper, password, userDN, extendedinformation)) {
512                         throw new PublisherException("Return code from publisher is false.");
513                     }
514                     publisherResults.add(Boolean.TRUE);
515                 } catch (Exception e) {
516                     publisherResults.add(getAsPublisherException(e));
517                 }
518             }
519         }
520         return publisherResults;
521     }

523     private PublisherException getAsPublisherException(final Exception e) {
524         log.debug("Publisher threw exception", e);
525         if (e instanceof PublisherException) {
526             return (PublisherException) e;
527         }
528         Throwable t = e;
529         while (t.getCause() != null) {
530             t = t.getCause();
531             if (t instanceof PublisherException) {
532                 return (PublisherException) t;
533             }
534         }
535         return new PublisherException(e.getMessage());
536     }
