40  /**
41   * Stores and manages CertReqHistory entries in the database.
42   * CertReqHistory keeps a snapshot of the user data that was used to issue a specific certificate.
43   *
44   * @version $Id: CertReqHistorySessionBean.java 28759 2018-04-20 13:50:37Z samuellb $
45   */
46  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "CertReqHistorySessionRemote")
47  @TransactionAttribute(TransactionAttributeType.SUPPORTS)
48  public class CertReqHistorySessionBean implements CertReqHistorySessionRemote, CertReqHistorySessionLocal {
49  
50      private final static Logger log = Logger.getLogger(CertReqHistorySessionBean.class);
51      /** Internal localization of logs and errors */
52      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
53      
54      @PersistenceContext(unitName="ejbca")
55      private EntityManager entityManager;
56      @EJB
57      private CertificateDataSessionLocal certificateDataSession;
58  
59      @TransactionAttribute(TransactionAttributeType.REQUIRED)
60      @Override
61      public void addCertReqHistoryData(Certificate cert, EndEntityInformation endEntityInformation){
62      	final String issuerDN = CertTools.getIssuerDN(cert);
63      	final String username = endEntityInformation.getUsername();
64      	if (log.isTraceEnabled()) {
65          	log.trace(">addCertReqHistoryData(" + CertTools.getSerialNumberAsString(cert) + ", " + issuerDN + ", " + username + ")");
66      	}
67          try {
68          	entityManager.persist(new CertReqHistoryData(cert, issuerDN, endEntityInformation));
69          	log.info(intres.getLocalizedMessage("store.storehistory", username));
70          } catch (Exception e) {
71          	log.error(intres.getLocalizedMessage("store.errorstorehistory", endEntityInformation.getUsername()));
72              throw new EJBException(e);
73          }
74      	if (log.isTraceEnabled()) {
75      		log.trace("<addCertReqHistoryData()");
76          }
77      }
78      
79      @TransactionAttribute(TransactionAttributeType.REQUIRED)
80      @Override
81      public void removeCertReqHistoryData(String certFingerprint){
82      	if (log.isTraceEnabled()) {
83          	log.trace(">removeCertReqHistData(" + certFingerprint + ")");
84      	}
85          try {          
86          	String msg = intres.getLocalizedMessage("store.removehistory", certFingerprint);
87          	log.info(msg);
88              CertReqHistoryData crh = CertReqHistoryData.findById(entityManager, certFingerprint);
89              if (crh == null) {
90              	if (log.isDebugEnabled()) {
91              		log.debug("Trying to remove CertReqHistory that does not exist: "+certFingerprint);                		
92              	}
93              } else {
94              	entityManager.remove(crh);
95              }
96          } catch (Exception e) {
97          	String msg = intres.getLocalizedMessage("store.errorremovehistory", certFingerprint);
98          	log.info(msg);
99              throw new EJBException(e);
100         }
101     	if (log.isTraceEnabled()) {
102     		log.trace("<removeCertReqHistData()");
103     	}
104     }
105     
106     // getCertReqHistory() might perform database updates, so we always need to run this in a transaction
107     @TransactionAttribute(TransactionAttributeType.REQUIRED)
108     @Override
109     public CertReqHistory retrieveCertReqHistory(BigInteger certificateSN, String issuerDN){
110     	CertReqHistory retval = null;
111     	Collection<CertReqHistoryData> result = CertReqHistoryData.findByIssuerDNSerialNumber(entityManager, issuerDN, certificateSN.toString());
112     	if(result.iterator().hasNext()) {
113     		retval = result.iterator().next().getCertReqHistory();
114     	}
115     	return retval;
116     }
117 
118     // getCertReqHistory() might perform database updates, so we always need to run this in a transaction
119     @TransactionAttribute(TransactionAttributeType.REQUIRED)
120     @Override
121     public List<CertReqHistory> retrieveCertReqHistory(String username){
122     	ArrayList<CertReqHistory> retval = new ArrayList<>();
123     	Collection<CertReqHistoryData> result = CertReqHistoryData.findByUsername(entityManager, username);
124     	Iterator<CertReqHistoryData> iter = result.iterator();
125     	while(iter.hasNext()) {
126     		retval.add(iter.next().getCertReqHistory());
127     	}
128     	return retval;
129     }
130     
131     @Override
132     public CertificateInfo findFirstCertificateInfo(final String issuerDN, final BigInteger serno) {
133     	return certificateDataSession.findFirstCertificateInfo(CertTools.stringToBCDNString(issuerDN), serno.toString());
134     }
