41  /**
42   * Workaround for the very complex CESeCore query criteria API.
43   * 
44   * @version $Id: EjbcaAuditorSessionBean.java 25494 2017-03-15 15:27:43Z jeklund $
45   */
46  @Stateless
47  @TransactionAttribute(TransactionAttributeType.SUPPORTS)
48  public class EjbcaAuditorSessionBean implements EjbcaAuditorSessionLocal {
49  
50      private static final Logger LOG = Logger.getLogger(EjbcaAuditorSessionBean.class);
51      
52      @PersistenceContext(unitName = CesecoreConfiguration.PERSISTENCE_UNIT)
53      private EntityManager entityManager;
54      @EJB
55      private AuthorizationSessionLocal authorizationSession;
56      @EJB
57      private CaSessionLocal caSession;
58  
59      @Override
60      public List<? extends AuditLogEntry> selectAuditLog(final AuthenticationToken token, final String device, final int firstResult, final int maxResults,
61              final String whereClause, final String orderClause, final List<Object> parameters) throws AuthorizationDeniedException {
62          if (!IntegrityProtectedDevice.class.getSimpleName().equals(device)) {
63              throw new UnsupportedOperationException("selectAuditLog can only be used with " + IntegrityProtectedDevice.class.getSimpleName());
64          }
65          // Require that the caller is authorized to AUDITLOGSELECT just like in org.cesecore.audit.audit.SecurityEventsAuditorSessionBean.selectAuditLogs(...)
66          assertAuthorization(token, AuditLogRules.VIEW.resource());
67          // Assert that parameter is alphanumeric or one of ". ?<>!="
68          assertLegalSqlString(whereClause, true);
69          // Assert that parameter is alphanumeric or one of ". "
70          assertLegalSqlString(orderClause, false);
71          // Start building the query
72          final StringBuilder queryBuilder = new StringBuilder("SELECT a FROM ").append(AuditRecordData.class.getSimpleName()).append(" a");
73          // Optionally add the WHERE clause
74          if (whereClause!=null && whereClause.length()>0) {
75              queryBuilder.append(" WHERE ").append(whereClause);
76          }
77          // Optionally add the ORDER clause
78          if (orderClause!=null && orderClause.length()>0) {
79              queryBuilder.append(" ORDER BY ").append(orderClause);
80          }
81          final String queryString = queryBuilder.toString();
82          if (LOG.isDebugEnabled()) {
83              LOG.debug("queryString: " + queryString);
84          }
85          final TypedQuery<AuditLogEntry> query = entityManager.createQuery(queryString, AuditLogEntry.class);
86          query.setFirstResult(firstResult);
87          if (maxResults>0) {
88              query.setMaxResults(maxResults);
89          }
90          if (parameters!=null) {
91              for (int i=0; i<parameters.size(); i++) {
92                  query.setParameter(i, parameters.get(i));
93              }
94          }
95  
96          //Prune out result pertaining to unauthorized CAs
97          Set<Integer> authorizedCaIds = new HashSet<>(caSession.getAuthorizedCaIds(token));
98          List<AuditLogEntry> resultList = new ArrayList<>();
99          for (final AuditLogEntry auditLogEntry : query.getResultList()) {
100             //The following values may leak CA Ids. 
101             if ( auditLogEntry.getModuleTypeValue().equals(ModuleTypes.CA)
102                  || auditLogEntry.getModuleTypeValue().equals(ModuleTypes.CERTIFICATE)
103                  || auditLogEntry.getModuleTypeValue().equals(ModuleTypes.CRL)) {
104                 if (!StringUtils.isEmpty(auditLogEntry.getCustomId())) {
105                     if (!authorizedCaIds.contains(Integer.valueOf(auditLogEntry.getCustomId()))) {
106                         continue;
107                     }
108                 }
109             }
110             resultList.add(auditLogEntry);
111         }
112         return resultList;
113         
114     }
115 
116     /**
117      * Make sure that the SQL query doesn't contain illegal characters.
118      * A-Z, a-z, 0-9, ' ' and '.' are always allowed.
119      * @param sql SQL
120      * @param isWhere if true we also allow '?', '<', '>', '!' and '='
121      * @throws IllegalArgumentException if an illegal character is found in the sql paramater
122      */
123     private void assertLegalSqlString(final String sql, final boolean isWhere) throws IllegalArgumentException {
124         if (sql==null) {
125             return; // no String is safe
126         }
127         for (int i=0; i<sql.length(); i++) {
128             final char c = sql.charAt(i);
129             if ((c>='0' && c<='9') || (c>='A' && c<='Z') || (c>='a' && c<='z') || c==' ' || c=='.') {
130                 continue; // ok
131             }
132             if (isWhere) {
133                 if (c=='?' || c=='<' || c=='>' || c=='!' || c=='=') {
134                     continue; // ok
135                 }
136             }
137             // Char was not in the white-list.. warn and throw an error!
138             LOG.warn("Possible SQL injection attempt: " + sql);
139             throw new IllegalArgumentException(c + " is not a legal SQL query character.");
140         }
141     }
142 
143     /** Assert that we are authorized to the requested resource. 
144      * @param token Token
145      * @param accessRule Rule
146      * @throws AuthorizationDeniedException Fail */
147     private void assertAuthorization(final AuthenticationToken token, final String accessRule) throws AuthorizationDeniedException {
148         if (!authorizationSession.isAuthorized(token, accessRule)) {
149             throw new AuthorizationDeniedException("not authorized to: "+ token.toString());                
150         } 
151     }
