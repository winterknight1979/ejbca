138     @Override
139     public int addApprovalRequest(AuthenticationToken admin, ApprovalRequest approvalRequest) throws ApprovalException {
140     	if (log.isTraceEnabled()) {
141     		log.trace(">addApprovalRequest: hash="+approvalRequest.generateApprovalId());
142     	}
143         int approvalId = approvalRequest.generateApprovalId();
144         Integer requestId = 0;
145         ApprovalDataVO data = findNonExpiredApprovalRequest(approvalId);
146         if (data != null) {
147             String msg = intres.getLocalizedMessage("approval.alreadyexists", approvalId);
148             log.info(msg);
149             throw new ApprovalException(ErrorCode.APPROVAL_ALREADY_EXISTS, msg);
150         } else {
151             // There exists no approval request with status waiting. Add a new one
152             try {
153                 requestId = findFreeApprovalId();
154                 final ApprovalData approvalData = new ApprovalData(requestId);
155                 updateApprovalData(approvalData, approvalRequest);
156                 entityManager.persist(approvalData);
157                 final ApprovalProfile approvalProfile = approvalRequest.getApprovalProfile();
158                 sendApprovalNotifications(approvalRequest, approvalProfile, approvalData, false);
159                 String msg = intres.getLocalizedMessage("approval.addedwaiting", requestId);
160                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
161                 details.put("msg", msg);
162                 List<ApprovalDataText> texts = approvalRequest.getNewRequestDataAsText(admin);
163                 for (ApprovalDataText text : texts) {
164                     details.put(text.getHeader(), text.getData());                    
165                 }
166                 auditSession.log(EjbcaEventTypes.APPROVAL_ADD, EventStatus.SUCCESS, EjbcaModuleTypes.APPROVAL, EjbcaServiceTypes.EJBCA,
167                         admin.toString(), String.valueOf(approvalRequest.getCAId()), null, null, details);
168             } catch (Exception e1) {
169                 String msg = intres.getLocalizedMessage("approval.erroradding", requestId);
170                 log.error(msg, e1);
171                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
172                 details.put("msg", msg);
173                 details.put("Error", e1.getMessage());
174                 auditSession.log(EjbcaEventTypes.APPROVAL_ADD, EventStatus.FAILURE, EjbcaModuleTypes.APPROVAL, EjbcaServiceTypes.EJBCA,
175                         admin.toString(), String.valueOf(approvalRequest.getCAId()), null, null, details);
176             }
177         }
178         if (log.isTraceEnabled()) {
179         	log.trace("<addApprovalRequest: hash="+approvalRequest.generateApprovalId()+", id="+requestId);
180         }
181         return requestId;
182     }

184     @Override
185     public void editApprovalRequest(final AuthenticationToken admin, final int id, final ApprovalRequest approvalRequest) throws ApprovalException {
186         if (log.isTraceEnabled()) {
187             log.trace(">editApprovalRequest: hash="+approvalRequest.generateApprovalId()+", id="+id);
188         }
189         final ApprovalData ad = findById(id);
190         if (ad == null) {
191             throw new ApprovalException("The approval request does not exist");
192         }
193         if (ad.getStatus() != ApprovalDataVO.STATUS_WAITINGFORAPPROVAL) {
194             throw new ApprovalException("The approval request is not in the Waiting For Approval state, and cannot be edited");
195         }
196         try {
197             approvalRequest.addEditedByAdmin(admin);
198             updateApprovalData(ad, approvalRequest);
199             String msg = intres.getLocalizedMessage("approval.edited", id);
200             final Map<String, Object> details = new LinkedHashMap<>();
201             details.put("msg", msg);
202             List<ApprovalDataText> texts = approvalRequest.getNewRequestDataAsText(admin);
203             for (ApprovalDataText text : texts) {
204                 details.put(text.getHeader(), text.getData());                    
205             }
206             auditSession.log(EjbcaEventTypes.APPROVAL_EDIT, EventStatus.SUCCESS, EjbcaModuleTypes.APPROVAL, EjbcaServiceTypes.EJBCA,
207                     admin.toString(), String.valueOf(approvalRequest.getCAId()), null, null, details);
208         } catch (Exception e) {
209             String msg = intres.getLocalizedMessage("approval.errorediting", id);
210             log.error(msg, e);
211             final Map<String, Object> details = new LinkedHashMap<>();
212             details.put("msg", msg);
213             details.put("Error", e.getMessage());
214             auditSession.log(EjbcaEventTypes.APPROVAL_EDIT, EventStatus.FAILURE, EjbcaModuleTypes.APPROVAL, EjbcaServiceTypes.EJBCA,
215                     admin.toString(), String.valueOf(approvalRequest.getCAId()), null, null, details);
216         }
217         if (log.isTraceEnabled()) {
218             log.trace(">editApprovalRequest: hash="+approvalRequest.generateApprovalId()+", id="+id);
219         }
220     }

222     /** Updates the ApprovalData from the given approval request, and initializes the list of approvals to an empty list. 
223      * @param approvalData Data
224      * @param approvalRequest Request */
225     @SuppressWarnings("deprecation")
226     private void updateApprovalData(final ApprovalData approvalData, final ApprovalRequest approvalRequest) {
227         approvalData.setApprovalid(approvalRequest.generateApprovalId());
228         approvalData.setApprovaltype(approvalRequest.getApprovalType());
229         approvalData.setEndentityprofileid(approvalRequest.getEndEntityProfileId());
230         approvalData.setCaid(approvalRequest.getCAId());
231         if (approvalRequest.getRequestAdminCert() != null) {
232             approvalData.setReqadmincertissuerdn(CertTools.getIssuerDN(approvalRequest.getRequestAdminCert()));
233             approvalData.setReqadmincertsn(CertTools.getSerialNumberAsString(approvalRequest.getRequestAdminCert()));
234         }
235         setApprovalRequest(approvalData, approvalRequest);
236         setApprovals(approvalData, new ArrayList<Approval>());
237         approvalData.setExpiredate((new Date()).getTime() + approvalRequest.getRequestValidity());
238         //Kept for legacy reasons
239         approvalData.setRemainingapprovals(approvalRequest.getNumOfRequiredApprovals());
240     }

244         if (log.isTraceEnabled()) {
245             log.trace(">removeApprovalRequest: id="+id);
246         }
247         try {
248             ApprovalData ad = findById(Integer.valueOf(id));

273     @Override
274     public int isApproved(int approvalId, int step) throws ApprovalException, ApprovalRequestExpiredException {
275         if (log.isTraceEnabled()) {
276             log.trace(">isApproved, approvalId: " + approvalId);
277         }
278         int retval = ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED;
279         Collection<ApprovalData> result = findByApprovalId(approvalId);
280         if (result.size() == 0) {
281             throw new ApprovalException(ErrorCode.APPROVAL_REQUEST_ID_NOT_EXIST, "Approval request with id : " + approvalId + " does not exist");
282         }
283         if (log.isTraceEnabled()) {
284             log.trace("Found "+result.size()+" ApprovalData with id "+approvalId);
285         }
286         for(ApprovalData adl : result) {
287             if (log.isTraceEnabled()) {
288                 log.trace("Checking if ApprovalRequest of type "+adl.getApprovaltype()+" with databaseID "+adl.getId()+" and approvalID "+adl.getApprovalid()+" is approved: "+adl.getStatus());
289             }
290             retval = isApproved(adl, step);
291             if (adl.getStatus() == ApprovalDataVO.STATUS_WAITINGFORAPPROVAL || adl.getStatus() == ApprovalDataVO.STATUS_APPROVED
292                     || adl.getStatus() == ApprovalDataVO.STATUS_REJECTED) {
293                 break;
294             }
295         }
296         if (log.isTraceEnabled()) {
297             log.trace("<isApproved, result: " + retval);
298         }
299         return retval;
300     }

405     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
406     @Override
407     public List<ApprovalDataVO> queryByStatus(final boolean includeUnfinished, final boolean includeProcessed, final boolean includeExpired,
408             final Date startDate, final Date endDate, final Date expiresBefore, int index, int numberofrows, String caAuthorizationString,
409             String endEntityProfileAuthorizationString) {
410         log.trace(">queryByStatus()");
411         
412         if (!includeUnfinished && !includeProcessed && !includeExpired) {
413             throw new IllegalArgumentException("At least one of includeUnfinished, includeProcessed or includeExpired must be true");
414         }
415         
416         final StringBuilder sb = new StringBuilder();
417         
418         String orderByString = null;
419         sb.append('(');
420         boolean first = true;
421         if (includeUnfinished || includeExpired) {
422             sb.append('(');
423             if (includeUnfinished && includeExpired) {
424                 // No additional filtering
425             } else if (!includeExpired) {
426                 // Do not include expired requests
427                 sb.append("expireDate >= ");
428                 sb.append(new Date().getTime());
429                 sb.append(" AND ");
430             } else if (includeExpired) {
431                 sb.append("expireDate < ");
432                 sb.append(new Date().getTime());
433                 sb.append(" AND ");
434             } else if (expiresBefore != null) {
435                 // Only include expired requests
436                 sb.append("expireDate < ");
437                 sb.append(new Date().getTime());
438                 sb.append(" AND ");
439             }
440             if (expiresBefore != null) {
441                 sb.append("expireDate < ");
442                 sb.append(expiresBefore.getTime());
443                 sb.append(" AND ");
444             }
445             // "STATUS_APPROVED" means that the request is still waiting to be executed by the requester
446             sb.append("status IN (" + ApprovalDataVO.STATUS_WAITINGFORAPPROVAL + ", " + ApprovalDataVO.STATUS_APPROVED + (includeExpired ? ", " + ApprovalDataVO.STATUS_EXPIRED + ", " + ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED : "") + "))");
447             orderByString = "ORDER BY requestDate ASC"; // oldest first
448             first = false;
449         }
450         if (includeProcessed) {
451             if (!first) { sb.append(" OR "); }
452             sb.append("status IN (" + ApprovalDataVO.STATUS_EXECUTED + ", " + ApprovalDataVO.STATUS_EXECUTIONDENIED + ", " +
453                     ApprovalDataVO.STATUS_EXECUTIONFAILED + ", " + ApprovalDataVO.STATUS_REJECTED + ")");
454             orderByString = "ORDER BY requestDate DESC"; // most recently created first
455             first = false;
456         }
457         sb.append(')');
458         
459         if (startDate != null) {
460             sb.append(" AND requestDate >= " + startDate.getTime());
461         }
462         if (endDate != null) {
463             sb.append(" AND requestDate < " + endDate.getTime()); 
464         }
465         final List<ApprovalDataVO> ret = queryInternal(sb.toString(), index, numberofrows,
466                 caAuthorizationString, endEntityProfileAuthorizationString,
467                 orderByString);
468         log.trace("<queryByStatus()");
469         return ret;
470     }

472     private List<ApprovalDataVO> queryInternal(final String query, int index, int numberofrows, String caAuthorizationString,
473             String endEntityProfileAuthorizationString, final String orderByString) {
474         log.trace(">queryInternal()");
475         String customQuery = "(" + query + ")";
476         if (StringUtils.isNotEmpty(caAuthorizationString)) {
477             customQuery += " AND " + caAuthorizationString;
478         }
479         if (StringUtils.isNotEmpty(endEntityProfileAuthorizationString)) {
480             customQuery += " AND " + endEntityProfileAuthorizationString;
481         }
482         if (StringUtils.isNotEmpty(orderByString)) {
483             customQuery += " " + orderByString;
484         }
485         
486         final List<ApprovalData> approvalDataList = findByCustomQuery(index, numberofrows, customQuery);
487         final List<ApprovalDataVO> returnData = new ArrayList<>(approvalDataList.size());
488         for (ApprovalData approvalData : approvalDataList) {
489             final ApprovalDataVO approvalInformation = approvalData.getApprovalDataVO();
490             //Perform a lazy upgrade of incoming approval requests produced prior to 6.5.0, which will lack a reference to an approval profile. The 
491             //upgrade procedure will have created the required approval profiles. 
492             ApprovalRequest approvalRequest = approvalInformation.getApprovalRequest();
493             if(approvalRequest.getApprovalProfile() == null) {
494                 if(log.isDebugEnabled()) {
495                     log.debug("Attempting to upgrade approval with ID " + approvalData.getApprovalid() 
496                     + " to 6.6.0+ status by retrieving an approval profile from either the certificate profile or the CA.");
497                 }
498                 ApprovalProfile approvalProfile;
499 
500                 //For the sake of upgrade, we're forced to use instanceof to find the relevant certificate profile ID, based on the behavior in 
501                 //6.5.x 
502                 CertificateProfile certificateProfile = null;
503                 if (approvalRequest instanceof ActivateCATokenApprovalRequest) {
504                     //See legacy instantiation in CAAdminSessionBean
505                     certificateProfile = certificateProfileSession
506                             .getCertificateProfile(caSession.getCAInfoInternal(approvalRequest.getCAId()).getCertificateProfileId());
507                 } else if (approvalRequest instanceof AddEndEntityApprovalRequest) {
508                     //See legacy instantiation in EndEntityManagementSessionBean
509                     certificateProfile = certificateProfileSession.getCertificateProfile(
510                             ((AddEndEntityApprovalRequest) approvalRequest).getEndEntityInformation().getCertificateProfileId());
511                 } else if (approvalRequest instanceof ChangeStatusEndEntityApprovalRequest) {
512                     //See legacy instantiation in EndEntityManagementSessionBean
513                     EndEntityInformation endEntityInformation = endEntityAccessSession
514                             .findUser(((ChangeStatusEndEntityApprovalRequest) approvalRequest).getUsername());
515                     certificateProfile = certificateProfileSession.getCertificateProfile(endEntityInformation.getCertificateProfileId());
516                 } else if (approvalRequest instanceof EditEndEntityApprovalRequest) {
517                     //See legacy instantiation in EndEntityManagementSessionBean
518                     certificateProfile = certificateProfileSession.getCertificateProfile(
519                             ((EditEndEntityApprovalRequest) approvalRequest).getNewEndEntityInformation().getCertificateProfileId());
520                 } else if (approvalRequest instanceof GenerateTokenApprovalRequest) {
521                     //TODO: Handle 100% uptime for hard token requests under ECA-5078
522                 } else if (approvalRequest instanceof KeyRecoveryApprovalRequest) {
523                     //See legacy instantiation in KeyRecoverySessionBean
524                     final CertificateInfo certificateInfor = certificateStoreSession.getCertificateInfo(
525                             CertTools.getFingerprintAsString(((KeyRecoveryApprovalRequest) approvalRequest).getRequestAdminCert()));
526                     certificateProfile = certificateProfileSession.getCertificateProfile(certificateInfor.getCertificateProfileId());
527                 } else if (approvalRequest instanceof RevocationApprovalRequest) {
528                     //See legacy instantiation in RevocationSessionBean
529                     EndEntityInformation endEntityInformation = endEntityAccessSession
530                             .findUser(((RevocationApprovalRequest) approvalRequest).getUsername());
531                     certificateProfile = certificateProfileSession.getCertificateProfile(endEntityInformation.getCertificateProfileId());
532                 } else if (approvalRequest instanceof ViewHardTokenDataApprovalRequest) {
533                     //TODO: Handle 100% uptime for hard token requests under ECA-5078
534                 }
535                 approvalProfile = approvalProfileSession.getApprovalProfileForAction(
536                         ApprovalRequestType.getFromIntegerValue(approvalRequest.getApprovalRequestType()),
537                         caSession.getCAInfoInternal(approvalRequest.getCAId()), certificateProfile);
538              
539                 approvalRequest.setApprovalProfile(approvalProfile);
540                 approvalInformation.setApprovalRequest(approvalRequest);
541                 approvalSession.updateApprovalRequest(approvalData.getId(), approvalRequest);
542                 if (log.isDebugEnabled()) {
543                     log.debug("Upgraded approval with ID " + approvalData.getApprovalid() + " to 6.6.0+ by setting approval profile with ID "
544                             + approvalProfile != null ? (approvalProfile.getProfileId() + "(" + approvalProfile.getProfileName() + ")") : "(no approval profile)" + ".");
545                 }
546             }
547             returnData.add(approvalInformation);         
548         }
549         log.trace("<queryInternal()");
550         return returnData;

553     @Override
554     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
555     public void sendApprovalNotifications(final ApprovalRequest approvalRequest, final ApprovalProfile approvalProfile,
556             final ApprovalData approvalData, final boolean expired) {
557         try {
558             final List<Approval> approvalsPerformed = approvalData.getApprovals();
559             // When adding a new approval request the list of performed approvals is empty
560             final Approval lastApproval = approvalsPerformed.isEmpty() ? null : approvalsPerformed.get(approvalsPerformed.size()-1);
561             // If all steps has been satisfied, the ApprovalStep from getStepBeingEvaluated is null
562             final ApprovalStep approvalStep = approvalProfile.getStepBeingEvaluated(approvalsPerformed);
563             if (lastApproval!=null && (!lastApproval.isApproved() || expired)) {
564                 if (log.isDebugEnabled()) {
565                     log.debug("Creating rejected or expired notification for approval profile: "+approvalProfile.getProfileName());
566                 }
567                 if (approvalStep==null || approvalStep.getStepIdentifier()==lastApproval.getStepId()) {
568                     // If the approval has been rejected or expired, we should notify all partition owners in the current step that still has not approved it
569                     final int currentStepId = lastApproval.getStepId();
570                     final ApprovalPartition currentApprovalPartition = approvalProfile.getStep(currentStepId).getPartition(lastApproval.getPartitionId());
571                     if (expired) {
572                         sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, currentApprovalPartition, ApprovalPartitionWorkflowState.EXPIRED, lastApproval);
573                     } else {
574                         sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, currentApprovalPartition, ApprovalPartitionWorkflowState.REJECTED, lastApproval);
575                     }
576                     if (approvalStep!=null) {
577                         // Check which of the remaining partitions that need to be notified
578                         for (final ApprovalPartition approvalPartition : approvalStep.getPartitions().values()) {
579                             final int remainingApprovalsInPartition = approvalProfile.getRemainingApprovalsInPartition(approvalsPerformed, lastApproval.getStepId(), approvalPartition.getPartitionIdentifier());
580                             if (remainingApprovalsInPartition>0) {
581                                 if (expired) {
582                                     sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, approvalPartition, ApprovalPartitionWorkflowState.EXPIRED, lastApproval);
583                                 } else {
584                                     sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, approvalPartition, ApprovalPartitionWorkflowState.REJECTED, lastApproval);
585                                 }
586                             }
587                         }
588                     }
589                 } else {
590                     if (log.isDebugEnabled()) {
591                         log.debug("All steps have been satisfied, so no approvals sent for approval profile: "+approvalProfile.getProfileName());
592                     }
593                 }
594             } else {
595                 if (lastApproval!=null) {
596                     if (log.isDebugEnabled()) {
597                         log.debug("Request approved, notify every partition owner who's work flow is affected by the made approval for approval profile: "+approvalProfile.getProfileName());
598                     }
599                     // Notify every partition owner who's work flow is affected by the made approval
600                     final int currentStepId = lastApproval.getStepId();
601                     final int remainingApprovalsInPartition = approvalProfile.getRemainingApprovalsInPartition(approvalsPerformed, currentStepId, lastApproval.getPartitionId());
602                     final ApprovalPartition currentApprovalPartition = approvalProfile.getStep(lastApproval.getStepId()).getPartition(lastApproval.getPartitionId());
603                     if (remainingApprovalsInPartition>0) {
604                         sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, currentApprovalPartition, ApprovalPartitionWorkflowState.APPROVED_PARTIALLY, lastApproval);
605                     } else {
606                         sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), currentStepId, currentApprovalPartition, ApprovalPartitionWorkflowState.APPROVED, lastApproval);
607                     }
608                 }
609                 // If this is a new approval request or the current approval has completed a step, we should notify all partition owners in the next step
610                 if (lastApproval==null || (approvalStep!=null && approvalStep.getStepIdentifier()!=lastApproval.getStepId())) {
611                     if (log.isDebugEnabled()) {
612                         log.debug("This is a new approval request or the current approval has completed a step, we should notify all partition owners in the next step for approval profile: "+approvalProfile.getProfileName());
613                     }
614                     for (final ApprovalPartition approvalPartition : approvalStep.getPartitions().values()) {
615                         sendApprovalNotification(approvalRequest, approvalProfile, approvalData.getId(), approvalStep.getStepIdentifier(), approvalPartition, ApprovalPartitionWorkflowState.REQUIRES_ACTION, lastApproval);
616                     }
617                 }
618             }
619         } catch (AuthenticationFailedException e) {
620             log.warn("Unexpected failure during approval notification. Already performed approval where no longer authorized to do so.");
621         }
622     }

624     /** Send approval notification to the partition owner if it has notifications enabled. 
625      * @param approvalRequest Request
626      * @param approvalProfile Profile
627      * @param requestId ID
628      * @param approvalStepId Step ID 
629      * @param approvalPartition Partition
630      * @param approvalPartitionWorkflowState State 
631      * @param lastApproval Last approval */
632     private void sendApprovalNotification(final ApprovalRequest approvalRequest, final ApprovalProfile approvalProfile, final int requestId, final int approvalStepId, final ApprovalPartition approvalPartition,
633             final ApprovalPartitionWorkflowState approvalPartitionWorkflowState, final Approval lastApproval) {
634         
635         if(!approvalProfile.isNotificationEnabled(approvalPartition) && !approvalProfile.isUserNotificationEnabled(approvalPartition)) {
636             if (log.isDebugEnabled()) {
637                 String partitionString = "";
638                 if(approvalProfile instanceof PartitionedApprovalProfile) {
639                     final DynamicUiProperty<? extends Serializable> partitionNameproperty = approvalPartition.getProperty(PartitionedApprovalProfile.PROPERTY_NAME);
640                     final String partitionName;
641                     if (partitionNameproperty != null) {
642                         partitionName = partitionNameproperty.getValueAsString();
643                     } else {
644                         partitionName = "Noname with ID "+approvalPartition.getPartitionIdentifier();
645                     }
646                     partitionString = " for partition '"+partitionName + "'";
647                 }
648                 log.debug("Neither notifications nor user notifications are enabled"+ partitionString + " in approval profile: "+approvalProfile.getProfileName());
649             }
650             return;
651         }
652         
653         final int partitionId = approvalPartition.getPartitionIdentifier();
654         // There may be no partition name if it is not a partitioned approval
655         final String partitionName;
656         DynamicUiProperty<? extends Serializable> partNameProperty = approvalPartition.getProperty(PartitionedApprovalProfile.PROPERTY_NAME);
657         if (partNameProperty != null) {
658             partitionName = partNameProperty.getValueAsString();
659         } else {
660             partitionName = null;
661         }
662         final String approvalType = intres.getLocalizedMessage(ApprovalDataVO.APPROVALTYPENAMES[approvalRequest.getApprovalType()]);
663         final String workflowState = intres.getLocalizedMessage("APPROVAL_WFSTATE_" + approvalPartitionWorkflowState.name());
664         final String requestor = approvalRequest.getRequestAdmin().toString();
665         final String lastApprovedBy;
666         if (lastApproval != null) {
667             if (lastApproval.getAdmin() != null) {
668                 lastApprovedBy = lastApproval.getAdmin().toString();
669             } else {
670                 if (log.isDebugEnabled()) {
671                     log.debug("lastApproval.getAdmin returned null for approvalId: "+requestId);
672                 }
673                 lastApprovedBy = "";
674             }
675         } else {
676             lastApprovedBy = "";            
677         }
678 
679         if(approvalProfile.isNotificationEnabled(approvalPartition)) {
680             final String recipient = (String) approvalPartition.getProperty(ApprovalProfile.PROPERTY_NOTIFICATION_EMAIL_RECIPIENT).getValue();
681             final String sender = (String) approvalPartition.getProperty(ApprovalProfile.PROPERTY_NOTIFICATION_EMAIL_SENDER).getValue();
682             final String subject = (String) approvalPartition.getProperty(ApprovalProfile.PROPERTY_NOTIFICATION_EMAIL_MESSAGE_SUBJECT).getValue();
683             final String body = ((MultiLineString)approvalPartition.getProperty(ApprovalProfile.PROPERTY_NOTIFICATION_EMAIL_MESSAGE_BODY).getValue()).getValue();
684             final ApprovalNotificationParameterGenerator parameters = new ApprovalNotificationParameterGenerator(requestId, approvalStepId, partitionId, partitionName, approvalType, workflowState, requestor, lastApprovedBy);
685             try {
686                 MailSender.sendMailOrThrow(sender, Arrays.asList(recipient.split(" ")), MailSender.NO_CC, parameters.interpolate(subject), parameters.interpolate(body), MailSender.NO_ATTACHMENTS);
687                 log.info(intres.getLocalizedMessage("approval.sentnotification", requestId));
688             } catch (MailException e) {
689                 log.info(intres.getLocalizedMessage("approval.errornotification", requestId), e);
690             }
691         } else {
692             if(log.isDebugEnabled()) {
693                 log.debug("Admin notifications are not enabled for approval profile: "+approvalProfile.getProfileName());
694             }
695         }
696         
697         if(approvalProfile.isUserNotificationEnabled(approvalPartition)) {
698             final EndEntityInformation userdata = getEndEntity(approvalRequest);
699             if ((userdata != null) && (userdata.getEmail() != null)) {
700                 final String userRecipient = userdata.getEmail();
701                 final String userSender = (String) approvalPartition.getProperty(ApprovalProfile.PROPERTY_USER_NOTIFICATION_EMAIL_SENDER).getValue();
702                 final String userSubject = (String) approvalPartition.getProperty(ApprovalProfile.PROPERTY_USER_NOTIFICATION_EMAIL_MESSAGE_SUBJECT).getValue();
703                 final String userBody = ((MultiLineString)approvalPartition.getProperty(ApprovalProfile.PROPERTY_USER_NOTIFICATION_EMAIL_MESSAGE_BODY).getValue()).getValue();
704                 final ApprovalNotificationParameterGenerator userParameters = new ApprovalNotificationParameterGenerator(requestId, approvalStepId, partitionId, partitionName, approvalType, workflowState, requestor, lastApprovedBy);
705                 try {
706                     MailSender.sendMailOrThrow(userSender, Arrays.asList(userRecipient.split(" ")), MailSender.NO_CC, userParameters.interpolate(userSubject), userParameters.interpolate(userBody), MailSender.NO_ATTACHMENTS);
707                     log.info(intres.getLocalizedMessage("approval.sentnotification", requestId));
708                 } catch (Exception e) {
709                     log.info(intres.getLocalizedMessage("approval.errornotification", requestId), e);
710                 }
711             } else {
712                 log.info(intres.getLocalizedMessage("approval.errornotification", requestId) + " No email was found in the end entity");
713             }
714         } else {
715             if(log.isDebugEnabled()) {
716                 log.debug("User notifications are not enabled for approval profile: "+approvalProfile.getProfileName());
717             }
718         }
719     }

760     private void markStepAsDone(final ApprovalData approvalData, final int step) throws ApprovalRequestExpiredException {
761         final ApprovalRequest ar = approvalData.getApprovalRequest();

769             } else {
770                 throw new ApprovalRequestExpiredException("Error step " + step + " of approval with id " + approvalData.getApprovalid()
771                         + " have alread been performed");
772             }

776     /**
777      * Method used by the requestadmin to check if an approval request have been approved
778      * @param approvalData Data
779      * @param step the number of the step to check. 
780      * @return 0 (ApprovalDataVO.STATUS_APROVED) if approved, the number of approvals left if still waiting for approval, otherwise the ApprovalDataVO.STATUS constants indicating the status.
781      * @throws ApprovalRequestExpiredException if the request or approval have expired, the status will be EXPIREDANDNOTIFIED in this case.
782      */
783     @SuppressWarnings("deprecation")
784     private int isApproved(final ApprovalData approvalData, final int step) throws ApprovalRequestExpiredException {
785         if (approvalData.getApprovalRequest().isStepDone(step)) {
786             return ApprovalDataVO.STATUS_EXPIRED;
787         }
788         if (approvalData.hasRequestOrApprovalExpired()) {
789             if (approvalData.getStatus() != ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED && approvalData.getStatus() != ApprovalDataVO.STATUS_EXECUTED
790                     && approvalData.getStatus() != ApprovalDataVO.STATUS_EXECUTIONDENIED
791                     && approvalData.getStatus() != ApprovalDataVO.STATUS_EXECUTIONFAILED) {
792                 approvalData.setStatus(ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED);
793                 throw new ApprovalRequestExpiredException();
794             }
795             return ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED;
796         }
797         if (approvalData.getStatus() == ApprovalDataVO.STATUS_WAITINGFORAPPROVAL) {
798             return approvalData.getApprovalRequest().getApprovalProfile().getRemainingApprovals(approvalData.getApprovals());
799         }
800         return approvalData.getStatus();
801     }

844     @Override
845     public void extendApprovalRequestNoAuth(final AuthenticationToken authenticationToken, final int approvalDataId, final long extendForMillisParam) {
846         if (extendForMillisParam <= 0) {
847             throw new IllegalArgumentException("Time to extend for must be a positive non-zero number: " + extendForMillisParam);
848         }
849         
850         final ApprovalData approvalData = findById(approvalDataId);
851         if (approvalData == null) {
852             throw new IllegalStateException("Approval request with ID " + approvalDataId + " does not exist");
853         }
854         
855         // Check status
856         final long status = approvalData.getStatus();
857         if (status != ApprovalDataVO.STATUS_EXPIRED &&
858                 status != ApprovalDataVO.STATUS_EXPIREDANDNOTIFIED &&
859                 status != ApprovalDataVO.STATUS_WAITINGFORAPPROVAL) {
860             throw new IllegalStateException("Can't extend approval request in this state (" + status + ")");
861         }
862         
863         // Check maximum extension time
864         long maxExtend = getMaxExtensionTime(approvalData.getApprovalDataVO());
865         if (maxExtend <= 0) {
866             throw new IllegalStateException("Approval profile (or configured default value) does not allow request extension");
867         }
868         long extendForMillis = extendForMillisParam;
869         if (extendForMillis > maxExtend) {
870             log.info("Tried to extend approval request ID " + approvalData + " for " + extendForMillisParam + " ms, " +
871                     "which is more than the maximum of the approval profile, " + maxExtend + " ms");
872             extendForMillis = maxExtend;
873         }
874         
875         approvalData.setExpiredate(new Date().getTime() + extendForMillis);
876         approvalData.setStatus(ApprovalDataVO.STATUS_WAITINGFORAPPROVAL);
877         entityManager.merge(approvalData);
878         
879         String msg = intres.getLocalizedMessage("approval.extended", approvalData.getId(), extendForMillis);
880         final Map<String, Object> details = new LinkedHashMap<>();
881         details.put("msg", msg);
882         auditSession.log(EjbcaEventTypes.APPROVAL_EXTEND, EventStatus.SUCCESS, EjbcaModuleTypes.APPROVAL, EjbcaServiceTypes.EJBCA,
883                 authenticationToken.toString(), String.valueOf(approvalData.getCaid()), null, null, details);
884     }

921     /** @param index Index
922      * @param numberofrows No. of rows to fetch
923      * @param customQuery Query
924      * @return return the query results as a List<ApprovalData>. */
925     private List<ApprovalData> findByCustomQuery(final int index, final int numberofrows, final String customQuery) {
926         final List<ApprovalData> ret = new ArrayList<ApprovalData>();
927         /* Hibernate on DB2 wont allow us to "SELECT *" in combination with setMaxResults.
928          * Ingres wont let us access a LOB in a List using a native query for all fields.
929          * -> So we will get a list of primary keys and the fetch the whole entities one by one...
930          * 
931          * As a sad little bonus, DB2 native queries returns a pair of {BigInteger, Integer}
932          * where the first value is row and the second is the value.
933          * As another sad little bonus, Oracle native queries returns a pair of {BigDecimal, BigDecimal}
934          * where the first value is the value and the second is the row.
935          */
936         final javax.persistence.Query query = entityManager.createNativeQuery("SELECT id FROM ApprovalData WHERE " + customQuery);
937         query.setFirstResult(index);
938         query.setMaxResults(numberofrows);
939         @SuppressWarnings("unchecked")
940         final List<Object> ids = query.getResultList();
941         for (Object object : ids) {
942             final int id = ValueExtractor.extractIntValue(object);
943             ret.add(entityManager.find(ApprovalData.class, id));
944         }
945         return ret;
946     }

963         ApprovalData approvalData = findById(requestId);

970             return result;
