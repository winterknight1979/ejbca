13  package org.ejbca.core.ejb.approval;
14  
15  import java.util.HashMap;
16  import java.util.Map;
17  import java.util.concurrent.locks.ReentrantLock;
18  
19  import javax.annotation.PostConstruct;
20  import javax.ejb.ConcurrencyManagement;
21  import javax.ejb.ConcurrencyManagementType;
22  import javax.ejb.EJB;
23  import javax.ejb.Singleton;
24  import javax.ejb.Startup;
25  import javax.ejb.TransactionManagement;
26  import javax.ejb.TransactionManagementType;
27  
28  import org.apache.log4j.Logger;
29  import org.cesecore.profiles.ProfileData;
30  import org.ejbca.config.EjbcaConfiguration;
31  import org.ejbca.core.model.approval.profile.ApprovalProfile;
32  
33  /**
34   * Class Holding cache variable.
35   * 
36   * This cache is designed so only one thread at the time will update the cache if it is too old. Other
37   * threads will happily return a bit too old object. If a cache update is forced, for example when
38   * a profile is edited, it will always update the cache even if the commit of the transaction fails.
39   * 
40   * Another known issue during forced updates is the race condition exists, so an update in progress
41   * might overwrite the result from forced update's database query.
42   * 
43   * The intention of this design is better throughput than fully ordered sequential updates.
44   * 
45   * @version $Id: ApprovalProfileCacheBean.java 26076 2017-06-26 12:49:12Z mikekushner $
46   */
47  @Singleton
48  @Startup
49  @ConcurrencyManagement(ConcurrencyManagementType.BEAN)
50  @TransactionManagement(TransactionManagementType.BEAN)
51  public class ApprovalProfileCacheBean {
52  
53      private final Logger LOG = Logger.getLogger(ApprovalProfileCacheBean.class);
54  
55      @EJB
56      private ApprovalProfileSessionLocal approvalProfileSession;
57  
58      /*
59       * Cache of profiles, with Id as keys. This cache may be
60       * unsynchronized between multiple instances of EJBCA, but is common to all
61       * threads in the same VM. Set volatile to make it thread friendly.
62       */
63  
64      /** Cache of mappings between profileId and profileName */
65      private volatile Map<Integer, String> idNameMapCache = null;
66      /** Cache of approval profiles, with Id as keys */
67      private volatile Map<Integer, ApprovalProfile> profileCache = null;
68  
69      private volatile long lastUpdate = 0;
70  
71      private ReentrantLock lock;
72      
73      /* Create template maps with all static constants */
74      private HashMap<Integer, String> idNameMapCacheTemplate;
75  
76      @PostConstruct
77      public void initialize() {
78          idNameMapCacheTemplate = new HashMap<>();
79          lock = new ReentrantLock(false);
80  
81          try {
82              updateProfileCache(true);
83          } catch (RuntimeException e) {
84              //We don't want to murder the entire deployment if the database happens to be unresponsive during startup, it's something we might 
85              //recover from
86              LOG.error(e);
87          }
88      }
89  
90      public ApprovalProfileCacheBean() {
91  
92      }
93  
94      /**
95       * This method sets the update time back down to zero, effectively forcing the cache to be reloaded on next read. Required due to the fact that 
96       * the cache can't reload until whatever transaction performing CRUD ops finishes.
97       */
98      public void forceCacheExpiration() {
99          lastUpdate = 0;
100     }
101     
102     /**
103      * Fetch all profiles from the database, unless cache is enabled, valid and we do not force an update.
104      * 
105      * @param force if true, this will force an update even if the cache is not yet invalid
106      */
107     public void updateProfileCache(final boolean force) {
108         if (LOG.isTraceEnabled()) {
109             LOG.trace(">updateProfileCache");
110         }
111         final long cacheApprovalProfileTime = EjbcaConfiguration.getCacheApprovalProfileTime();
112         final long now = System.currentTimeMillis();
113         // Check before acquiring lock
114         if (!force && cacheApprovalProfileTime != 0 && lastUpdate + cacheApprovalProfileTime > now) {
115             return; // We don't need to update cache
116         }
117         try {
118             lock.lock();
119             if (!force && cacheApprovalProfileTime != 0 && lastUpdate + cacheApprovalProfileTime > now) {
120                 return; // We don't need to update cache
121             }
122             lastUpdate = now; // make sure next thread does not also pass the update test
123         } finally {
124             lock.unlock();
125         }
126         final Map<Integer, String> idNameCache = new HashMap<>(idNameMapCacheTemplate);
127         final Map<Integer, ApprovalProfile> profCache = new HashMap<>();
128         for (final ProfileData current : approvalProfileSession.findAllProfiles()) {
129             final Integer id = current.getId();
130             final String approvalProfileName = current.getProfileName();
131             idNameCache.put(id, approvalProfileName);
132             //We can cast safely because we're retrieved the correct type of object
133             profCache.put(id, (ApprovalProfile) current.getProfile());
134         }
135         idNameMapCache = idNameCache;
136         profileCache = profCache;
137         if (LOG.isTraceEnabled()) {
138             LOG.trace("<updateProfileCache");
139         }
140     }
141 
142     /** @return the latest object from the cache or a current database representation if no caching is used. */
143     public Map<Integer, ApprovalProfile> getProfileCache() {
144         updateProfileCache(false);
145         return profileCache;
146     }
147 
148     /** @return the latest object from the cache or a current database representation if no caching is used. */
149     public Map<Integer, String> getIdNameMapCache() {
150         updateProfileCache(false);
151         return idNameMapCache;
152     }
153 
154 }
