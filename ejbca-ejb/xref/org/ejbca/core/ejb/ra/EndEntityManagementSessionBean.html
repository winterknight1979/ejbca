199     private boolean authorizedToCA(final AuthenticationToken admin, final int caid) {
200         boolean returnval = false;
201         returnval = authorizationSession.isAuthorizedNoLogging(admin, StandardRules.CAACCESS.resource() + caid);
202         if (!returnval) {
203             log.info("Admin " + admin.toString() + " not authorized to resource " + StandardRules.CAACCESS.resource() + caid);
204         }
205         return returnval;

208     /** Checks CA authorization and logs an official error if not and throws and AuthorizationDeniedException.
209      * Does not log access control granted if granted
210      * @param admin Admin
211      * @param caid ID
212      * @throws AuthorizationDeniedException Fail
213      */
214     private void assertAuthorizedToCA(final AuthenticationToken admin, final int caid) throws AuthorizationDeniedException {
215         if (!authorizedToCA(admin, caid)) {
216             final String msg = intres.getLocalizedMessage("ra.errorauthca", Integer.valueOf(caid), admin.toString());
217             final Map<String, Object> details = new LinkedHashMap<String, Object>();
218             details.put("msg", msg);
219             auditSession.log(EventTypes.ACCESS_CONTROL, EventStatus.FAILURE, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
220                     String.valueOf(caid), null, null, details);
221             throw new AuthorizationDeniedException(msg);
222         }
223     }

304     /**
305      * 
306      *
307      * 
308      * @param admin Admin
309      * @param endEntity EE
310      * @param clearpwd PWD
311      * @param lastApprovingAdmin Approval 
312      * @return EEI
313      * @throws ApprovalException if an approval already exists for this request.
314      * @throws AuthorizationDeniedException if the admin is not authorized to the CA, or lacks rights to add end entities. 
315      * @throws CADoesntExistsException if the CA specified does not exist
316      * @throws CertificateSerialNumberException if SubjectDN serial number already exists.
317      * @throws CustomFieldException if the end entity was not validated by a locally defined field validator
318      * @throws EndEntityExistsException if the end entity already exists
319      * @throws IllegalNameException if the Subject DN failed constraints
320      * @throws EndEntityProfileValidationException if the end entity fails constrains set by the end entity profile
321      * @throws WaitingForApprovalException to mark that a request has been created and is awaiting approval
322 
323      */
324     private EndEntityInformation addUser(final AuthenticationToken admin, EndEntityInformation endEntity, final boolean clearpwd,
325             final AuthenticationToken lastApprovingAdmin)
326             throws AuthorizationDeniedException, EndEntityExistsException, EndEntityProfileValidationException, WaitingForApprovalException,
327             CADoesntExistsException, CustomFieldException, IllegalNameException, ApprovalException, CertificateSerialNumberException {
328         final int endEntityProfileId = endEntity.getEndEntityProfileId();
329         final int caid = endEntity.getCAId();
330         // Check if administrator is authorized to add user to CA.
331         assertAuthorizedToCA(admin, caid);
332         final GlobalConfiguration globalConfiguration = getGlobalConfiguration();
333         if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
334             // Check if administrator is authorized to add user.
335             assertAuthorizedToEndEntityProfile(admin, endEntityProfileId, AccessRulesConstants.CREATE_END_ENTITY, caid);
336         }
337         
338         final String originalDN = endEntity.getDN();
339         EndEntityInformation unCanonicalized = endEntity;
340         endEntity = canonicalizeUser(endEntity);
341         if (log.isTraceEnabled()) {
342             log.trace(">addUser(" + endEntity.getUsername() + ", password, " + endEntity.getDN() + ", " + originalDN + ", " + endEntity.getSubjectAltName()
343                     + ", " + endEntity.getEmail() + ", profileId: " + endEntityProfileId + ")");
344         }
345         
346         final String endEntityProfileName = endEntityProfileSession.getEndEntityProfileName(endEntityProfileId);
347         final String dn = endEntity.getDN();
348         String altName = endEntity.getSubjectAltName();
349         try {
350             altName =  RFC4683Tools.generateSimForInternalSanFormat( altName);
351         } catch(Exception e) {
352             log.info("Could not generate SIM string for SAN: " + altName, e);
353             throw new EndEntityProfileValidationException("Could not generate SIM string for SAN: " + e.getMessage(), e);
354         }
355         if (log.isTraceEnabled()) {
356             log.trace("addUser(calculated SIM " + altName + ")");
357         }
358         final String email = endEntity.getEmail();
359         final EndEntityType type = endEntity.getType();
360         String newpassword = endEntity.getPassword();
361         EndEntityProfile profile = null; // Only look this up if we need it..
362         if (endEntity.getPassword() == null) {
363             profile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
364             if (profile.useAutoGeneratedPasswd()) {
365                 // special case used to signal regeneration of password
366                 newpassword = profile.getAutoGeneratedPasswd();
367             }
368         }
369         //Autogenerate username if it's not modifiable and it's empty
370         if(StringUtils.isBlank(endEntity.getUsername())){
371             profile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
372             if (profile.isAutoGeneratedUsername()) {
373                 final byte[] randomData = new byte[16];
374                 final Random random = new SecureRandom();
375                 random.nextBytes(randomData);
376                 String autousername = new String(Hex.encode(randomData));
377                 while (endEntityAccessSession.findUser(autousername) != null) {
378                     if(log.isDebugEnabled()){
379                         log.debug("Autogenerated username '" + autousername + "' is already reserved. Generating the new one...");
380                     }
381                     random.nextBytes(randomData);
382                     autousername = new String(Hex.encode(randomData));
383                 }
384                 if(log.isDebugEnabled()){
385                     log.debug("Unique username '" + autousername + "' has been generated");
386                 }
387                 unCanonicalized.setUsername(autousername);
388                 endEntity.setUsername(autousername);
389             }
390         }
391         final String username = endEntity.getUsername();
392         if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
393             if (profile == null) {
394                 profile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
395             }
396             // Check if user fulfills it's profile.
397             try {
398                 final String dirattrs = endEntity.getExtendedInformation() != null ? endEntity.getExtendedInformation()
399                         .getSubjectDirectoryAttributes() : null;
400                 profile.doesUserFulfillEndEntityProfile(username, endEntity.getPassword(), dn, altName, dirattrs, email,
401                         endEntity.getCertificateProfileId(), clearpwd, type.contains(EndEntityTypes.KEYRECOVERABLE),
402                         type.contains(EndEntityTypes.SENDNOTIFICATION), endEntity.getTokenType(), endEntity.getHardTokenIssuerId(), caid,
403                         endEntity.getExtendedInformation());
404             } catch (EndEntityProfileValidationException e) {
405                 final String msg = intres.getLocalizedMessage("ra.errorfulfillprofile", endEntityProfileName, dn, e.getMessage());
406                 Map<String, Object> details = new LinkedHashMap<String, Object>();
407                 details.put("msg", msg);
408                 auditSession.log(EjbcaEventTypes.RA_ADDENDENTITY, EventStatus.FAILURE, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
409                         String.valueOf(caid), null, username, details);
410                 throw e;
411             }
412         }
413         // Get CAInfo, to be able to read configuration
414         // No need to access control on the CA here just to get these flags, we have already checked above that we are authorized to the CA
415         final CAInfo caInfo = caSession.getCAInfoInternal(caid, null, true);
416         if(caInfo == null) {
417             throw new CADoesntExistsException("CA with ID " + caid + " does not exist.");
418         }       
419         
420         // Check name constraints
421         if (caInfo instanceof X509CAInfo && caInfo.getCertificateChain() != null && !caInfo.getCertificateChain().isEmpty()) {
422             final X509CAInfo x509cainfo = (X509CAInfo) caInfo;
423             final X509Certificate cacert = (X509Certificate)caInfo.getCertificateChain().iterator().next();
424             final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(endEntity.getCertificateProfileId());
425             
426             final X500NameStyle nameStyle;
427             if (x509cainfo.getUsePrintableStringSubjectDN()) {
428                 nameStyle = PrintableStringNameStyle.INSTANCE;
429             } else {
430                 nameStyle = CeSecoreNameStyle.INSTANCE;
431             }
432             
433             final boolean ldaporder;
434             if (x509cainfo.getUseLdapDnOrder() && certProfile.getUseLdapDnOrder()) {
435                 ldaporder = true; // will cause an error to be thrown later if name constraints are used
436             } else {
437                 ldaporder = false;
438             }
439             
440             X500Name subjectDNName = CertTools.stringToBcX500Name(dn, nameStyle, ldaporder);
441             GeneralNames subjectAltName = CertTools.getGeneralNamesFromAltName(altName);
442             try {
443                 CertTools.checkNameConstraints(cacert, subjectDNName, subjectAltName);
444             } catch (IllegalNameException e) {
445                 e.setErrorCode(ErrorCode.NAMECONSTRAINT_VIOLATION);
446                 throw e;
447             }
448         }
449         // Check if approvals is required. (Only do this if store users, otherwise this approval is disabled.)
450         if (caInfo.isUseUserStorage()) {
451             final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(endEntity.getCertificateProfileId());
452             final ApprovalProfile approvalProfile = approvalProfileSession.getApprovalProfileForAction(ApprovalRequestType.ADDEDITENDENTITY, caInfo, 
453                     certProfile);
454             if (approvalProfile != null) {
455                 AddEndEntityApprovalRequest ar = new AddEndEntityApprovalRequest(endEntity, clearpwd, admin, null, caid,
456                         endEntityProfileId, approvalProfile);
457                 // How come we pass through here when the request is actually approved?
458                 // When the approval request is finally executed, it is executed through AddEndEntityApprovalRequest.execute, which is
459                 // the NONAPPROVABLECLASSNAMES_ADDUSER below.
460                 if (ApprovalExecutorUtil.requireApproval(ar, NONAPPROVABLECLASSNAMES_ADDUSER)) {
461                     final int requestId = approvalSession.addApprovalRequest(admin, ar);
462                     sendNotification(admin, endEntity, EndEntityConstants.STATUS_WAITINGFORADDAPPROVAL, requestId, lastApprovingAdmin, null);
463                     throw new WaitingForApprovalException(intres.getLocalizedMessage("ra.approvalad"), requestId);
464                 }
465             }
466         }
467         // Check if the subjectDN serialnumber already exists.
468         if (caInfo.isDoEnforceUniqueSubjectDNSerialnumber()) {
469             if (caInfo.isUseUserStorage()) {
470                 if (!isSubjectDnSerialnumberUnique(caid, dn, username)) {
471                     throw new CertificateSerialNumberException("Error: SubjectDN serial number already exists.");
472                 }
473             } else {
474                 log.warn("CA configured to enforce unique SubjectDN serialnumber, but not to store any user data. Check will be ignored. Please verify your configuration.");
475             }
476         }
477         // Store a new UserData in the database, if this CA is configured to do so.
478         if (caInfo.isUseUserStorage()) {
479             try {
480                 // Create the user in one go with all parameters at once. This was important in EJB2.1 so the persistence layer only creates *one*
481                 // single
482                 // insert statement. If we do a home.create and the some setXX, it will create one insert and one update statement to the database.
483                 // Probably not important in EJB3 anymore.
484                 final UserData userData = new UserData(username, newpassword, clearpwd, dn, caid, endEntity.getCardNumber(), altName, email, type.getHexValue(),
485                         endEntityProfileId, endEntity.getCertificateProfileId(), endEntity.getTokenType(), endEntity.getHardTokenIssuerId(),
486                         endEntity.getExtendedInformation());
487                 // Since persist will not commit and fail if the user already exists, we need to check for this
488                 // Flushing the entityManager will not allow us to rollback the persisted user if this is a part of a larger transaction.
489                 if (endEntityAccessSession.findByUsername(userData.getUsername()) != null) {
490                     throw new EndEntityExistsException("User " + userData.getUsername() + " already exists.");
491                 }
492                 entityManager.persist(userData);
493                 // Although EndEntityInformation should always have a null password for
494                 // autogenerated end entities, the notification framework
495                 // expect it to exist. Since nothing else but printing is done after
496                 // this point it is safe to set the password
497                 endEntity.setPassword(newpassword);
498                 // Send notifications, if they should be sent
499                 // This is an add user request, if there was an approval involved in add user, it will have been added to extendedInformation
500                 int approvalRequestID = 0;
501                 if ( (endEntity != null) && (endEntity.getExtendedInformation() != null) && (endEntity.getExtendedInformation().getAddEndEntityApprovalRequestId() != null) ) {
502                     approvalRequestID = endEntity.getExtendedInformation().getAddEndEntityApprovalRequestId().intValue();
503                 }
504                 sendNotification(admin, endEntity, EndEntityConstants.STATUS_NEW, approvalRequestID, lastApprovingAdmin, null);
505                 if (type.contains(EndEntityTypes.PRINT)) {
506                     if (profile == null) {
507                         profile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
508                     }
509                     print(profile, endEntity);
510                 } else {
511                     if (log.isDebugEnabled()) {
512                         log.debug("Type ("+type.getHexValue()+") does not contain SecConst.USER_PRINT, no print job created.");
513                     }
514                 }
515                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
516                 details.put("msg", intres.getLocalizedMessage("ra.addedentity", username));
517                 details.putAll(endEntity.getDetailMap());
518                 auditSession.log(EjbcaEventTypes.RA_ADDENDENTITY, EventStatus.SUCCESS, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
519                         String.valueOf(caid), null, username, details);
520             } catch (EndEntityExistsException e) {
521                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
522                 details.put("msg", intres.getLocalizedMessage("ra.errorentityexist", username));
523                 details.put("error", e.getMessage());
524                 auditSession.log(EjbcaEventTypes.RA_ADDENDENTITY, EventStatus.FAILURE, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
525                         String.valueOf(caid), null, username, details);
526                 throw e;
527             } catch (Exception e) {
528                 final String msg = intres.getLocalizedMessage("ra.erroraddentity", username);
529                 log.error(msg, e);
530                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
531                 details.put("msg", msg);
532                 details.put("error", e.getMessage());
533                 auditSession.log(EjbcaEventTypes.RA_ADDENDENTITY, EventStatus.FAILURE, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
534                         String.valueOf(caid), null, username, details);
535                 throw new EJBException(e);
536             }
537         } else if (log.isDebugEnabled()) {
538             log.debug("User storage disabled on CA '"+caInfo.getName()+"', user with username '"+username+"' is not stored.");
539         }
540         if (log.isTraceEnabled()) {
541             log.trace("<addUser(" + username + ", password, " + dn + ", " + email + ")");
542         }
543         return endEntity;
544     }

547     private boolean isSubjectDnSerialnumberUnique(final int caid, final String subjectDN, final String username) {
548         final String serialnumber = CertTools.getPartFromDN(subjectDN, "SN");
549         if (log.isDebugEnabled()) {
550             log.debug("subjectDN=" + subjectDN + " extracted SN=" + serialnumber);
551         }
552         // We treat the lack of a serialnumber field as unique
553         if (serialnumber == null) {
554             return true;
555         }
556         // Without a username we cannot determine if this is the same user, if we find any in the database later
557         if (username == null) {
558             return false;
559         }
560         final List<String> subjectDNs = endEntityAccessSession.findSubjectDNsByCaIdAndNotUsername(caid, username, serialnumber);
561         // Even though we push down most of the work to the database we still have to verify the serialnumber here since
562         // for example serialnumber '1' will match both "SN=1" and "SN=10" etc
563         for (final String currentSubjectDN : subjectDNs) {
564             final String currentSn = CertTools.getPartFromDN(currentSubjectDN, "SN");
565             if (serialnumber.equals(currentSn)) {
566                 return false;
567             }
568         }
569         return true;
570     }

573     public boolean renameEndEntity(final AuthenticationToken admin, String currentUsername, String newUsername) throws AuthorizationDeniedException, EndEntityExistsException {
574         // Sanity check parameters
575         if (currentUsername==null || newUsername==null) {
576             throw new IllegalArgumentException("Cannot rename an end entity to or from null.");
577         }
578         currentUsername = StringTools.stripUsername(currentUsername).trim();
579         newUsername = StringTools.stripUsername(newUsername).trim();
580         if (currentUsername.length()==0 || newUsername.length()==0) {
581             throw new IllegalArgumentException("Cannot rename an end entity to or from empty string.");
582         }
583         // Check that end entity exists and that the target username isn't already in use
584         final UserData currentUserData = endEntityAccessSession.findByUsername(currentUsername);
585         if (currentUserData==null) {
586             return false;
587         }
588         if (endEntityAccessSession.findByUsername(newUsername) !=null) {
589             throw new EndEntityExistsException("Unable to rename end entity, since end entity with username '" + newUsername + "' already exists.");
590         }
591         // Check authorization
592         final int currentCaId = currentUserData.getCaId();
593         assertAuthorizedToCA(admin, currentCaId);
594         final GlobalConfiguration globalConfiguration = getGlobalConfiguration();
595         if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
596             // Check if administrator is authorized to edit user.
597             assertAuthorizedToEndEntityProfile(admin, currentUserData.getEndEntityProfileId(), AccessRulesConstants.EDIT_END_ENTITY, currentCaId);
598         }
599         // Rename the end entity. Username is a primary key of the UserData table and we need to use JPA for this to get rowProtection.
600         // we need to add a new end entity and remove the old one.
601         final long now = System.currentTimeMillis();
602         final UserData userDataClone = currentUserData.clone();
603         userDataClone.setUsername(newUsername);
604         userDataClone.setTimeModified(now);
605         entityManager.persist(userDataClone);
606         entityManager.remove(currentUserData);
607         // Find all entities and update the username (we cant just do UPDATE ... SET username.. WHERE username since rowProtection might be enabled)
608         final List<CertificateData> certificateDatas = entityManager.createQuery(
609                 "SELECT a FROM CertificateData a WHERE a.username=:username", CertificateData.class).setParameter("username", currentUsername).getResultList();
610         int updatedPublisherQueueDataRows = 0;
611         for (final CertificateData certificateData : certificateDatas) {
612             final String fingerprint = certificateData.getFingerprint();
613             certificateData.setUsername(newUsername);
614             certificateData.setUpdateTime(now);
615             // Find all publisher queue data where PublisherQueueData.fingerprint matches CertificateData.fingerprint for this user
616             final List<PublisherQueueData> publisherQueueDatas = PublisherQueueData.findDataByFingerprint(entityManager, fingerprint);
617             for (final PublisherQueueData publisherQueueData : publisherQueueDatas) {
618                 // Only process entries that has not yet been published/processed
619                 if (publisherQueueData.getPublishStatus()==PublisherConst.STATUS_PENDING) {
620                     final PublisherQueueVolatileInformation volatileInformation = publisherQueueData.getPublisherQueueVolatileData();
621                     if (currentUsername.equals(volatileInformation.getUsername())) {
622                         volatileInformation.setUsername(newUsername);
623                         // Invoke setter to trigger update of still managed JPA entity
624                         publisherQueueData.setPublisherQueueVolatileData(volatileInformation);
625                         updatedPublisherQueueDataRows++;
626                     }
627                 }
628             }
629         }
630         if (log.isDebugEnabled()) {
631             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + certificateDatas.size() + " rows of CertificateData.");
632             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + updatedPublisherQueueDataRows + " rows of PublisherQueueData.");
633         }
634         final List<CertReqHistoryData> certReqHistoryDatas = entityManager.createQuery(
635                 "SELECT a FROM CertReqHistoryData a WHERE a.username=:username", CertReqHistoryData.class).setParameter("username", currentUsername).getResultList();
636         for (final CertReqHistoryData current : certReqHistoryDatas) {
637             // Note: Ignore the username inside certReqHistoryData.getUserDataVO(), since this should reflect the state of the UserData at the time of certificate issuance
638             current.setUsername(newUsername);
639         }
640         if (log.isDebugEnabled()) {
641             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + certReqHistoryDatas.size() + " rows of CertReqHistoryData.");
642         }
643         final List<KeyRecoveryData> keyRecoveryDatas = entityManager.createQuery(
644                 "SELECT a FROM KeyRecoveryData a WHERE a.username=:username", KeyRecoveryData.class).setParameter("username", currentUsername).getResultList();
645         for (final KeyRecoveryData current : keyRecoveryDatas) {
646             current.setUsername(newUsername);
647         }
648         if (log.isDebugEnabled()) {
649             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + keyRecoveryDatas.size() + " rows of KeyRecoveryData.");
650         }
651         final List<HardTokenData> hardTokenDatas = entityManager.createQuery(
652                 "SELECT a FROM HardTokenData a WHERE a.username=:username", HardTokenData.class).setParameter("username", currentUsername).getResultList();
653         for (final HardTokenData current : hardTokenDatas) {
654             current.setUsername(newUsername);
655         }
656         if (log.isDebugEnabled()) {
657             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + hardTokenDatas.size() + " rows of HardTokenData.");
658         }
659         // Update CLI admins where this username is used in AdminEntityData table.
660         final List<RoleMemberData> roleMemberDatas = entityManager.createQuery(
661                 "SELECT a FROM RoleMemberData a WHERE a.tokenType=:tokenType AND a.tokenMatchKey=:tokenMatchKey AND a.tokenMatchValueColumn=:tokenMatchValue", RoleMemberData.class)
662                 .setParameter("tokenType", CliAuthenticationTokenMetaData.TOKEN_TYPE)
663                 .setParameter("tokenMatchKey", CliUserAccessMatchValue.USERNAME.getNumericValue())
664                 .setParameter("tokenMatchValue", currentUsername)
665                 .getResultList();
666         for (final RoleMemberData current : roleMemberDatas) {
667             current.setTokenMatchValue(newUsername);
668         }
669         if (log.isDebugEnabled()) {
670             log.debug("Changed username '" + currentUsername + "' to '" + newUsername + "' in " + roleMemberDatas.size() + " rows of RoleMemberData.");
671         }
672         final String msg = intres.getLocalizedMessage("ra.editedentityrename", currentUsername, newUsername);
673         auditSession.log(EjbcaEventTypes.RA_EDITENDENTITY, EventStatus.SUCCESS, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
674                 String.valueOf(currentCaId), newUsername, currentUsername, msg);
675         return true;
676     }

724     private void changeUser(final AuthenticationToken admin, final EndEntityInformation endEntityInformation, final boolean clearpwd,
725             final boolean fromWebService, final int approvalRequestId, final AuthenticationToken lastApprovingAdmin, String newUsername) 
726             throws AuthorizationDeniedException, EndEntityProfileValidationException, WaitingForApprovalException,
727             CADoesntExistsException, ApprovalException, CertificateSerialNumberException, IllegalNameException, NoSuchEndEntityException, CustomFieldException {
728         final int endEntityProfileId = endEntityInformation.getEndEntityProfileId();
729         final int caid = endEntityInformation.getCAId();
730         String username = endEntityInformation.getUsername();
731         // Check if administrator is authorized to edit user to CA.
732         assertAuthorizedToCA(admin, caid);
733         final GlobalConfiguration globalConfiguration = getGlobalConfiguration();
734         if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
735             // Check if administrator is authorized to edit user.
736             assertAuthorizedToEndEntityProfile(admin, endEntityProfileId, AccessRulesConstants.EDIT_END_ENTITY, caid);
737         }
738 
739         FieldValidator.validate(endEntityInformation, endEntityProfileId, endEntityProfileSession.getEndEntityProfileName(endEntityProfileId));
740 
741         String dn = CertTools.stringToBCDNString(StringTools.strip(endEntityInformation.getDN()));
742         String altName = endEntityInformation.getSubjectAltName();
743         if (log.isTraceEnabled()) {
744             log.trace(">changeUser(" + username + ", " + dn + ", " + endEntityInformation.getEmail() + ")");
745         }   
746         try {
747             altName =  RFC4683Tools.generateSimForInternalSanFormat( altName);
748         } catch(Exception e) {
749              log.info("Could not generate SIM string for SAN: " + altName, e);
750              throw new EndEntityProfileValidationException("Could not generate SIM string for SAN: " + e.getMessage(), e);
751         }
752         if (log.isTraceEnabled()) {
753             log.trace(">changeUser(calculated SIM " + altName + ")");
754         }
755         UserData userData = endEntityAccessSession.findByUsername(username); 
756         if (userData == null) {
757             final String msg = intres.getLocalizedMessage("ra.erroreditentity", username);
758             log.info(msg);
759             throw new NoSuchEndEntityException(msg);
760         }
761         final EndEntityInformation originalCopy = userData.toEndEntityInformation();
762         
763         final EndEntityProfile profile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
764         // if required, we merge the existing user dn into the dn provided by the web service.
765         if (fromWebService && profile.getAllowMergeDnWebServices()) {
766             if (userData != null) {
767                 final Map<String, String> sdnMap = new HashMap<String, String>();
768                 if (profile.getUse(DnComponents.DNEMAILADDRESS, 0)) {
769                     sdnMap.put(DnComponents.DNEMAILADDRESS, endEntityInformation.getEmail());
770                 }
771                 try {
772                     // SubjectDN is not mandatory so
773                     if (dn == null) {
774                         dn = "";
775                     }
776                     dn = new DistinguishedNameshedName(userData.getSubjectDnNeverNull()).mergeDN(new DistinguishedName(dn), true, sdnMap).toString();
777                 } catch (InvalidNameException e) {
778                     if (log.isDebugEnabled()) {
779                         log.debug("Invalid Subject DN when merging '"+dn+"' with '"+userData.getSubjectDnNeverNull()+"'. Setting it to empty. Exception was: " + e.getMessage());
780                     }
781                     dn = "";
782                 }
783                 final Map<String, String> sanMap = new HashMap<String, String>();
784                 if (profile.getUse(DnComponents.RFC822NAME, 0)) {
785                     sanMap.put(DnComponents.RFC822NAME, endEntityInformation.getEmail());
786                 }
787                 try {
788                     // SubjectAltName is not mandatory so
789                     if (altName == null) {
790                         altName = "";
791                     }
792                     altName = new DistinguishedNameame(userData.getSubjectAltNameNeverNull()).mergeDN(new DistinguishedName(altName), true, sanMap).toString();
793                 } catch (InvalidNameException e) {
794                     if (log.isDebugEnabled()) {
795                         log.debug("Invalid Subject AN when merging '"+altName+"' with '"+userData.getSubjectAltNameNeverNull()+"'. Setting it to empty. Exception was: " + e.getMessage());
796                     }
797                     altName = "";
798                 }
799             }
800         }
801         String newpassword = endEntityInformation.getPassword();
802         if (profile.useAutoGeneratedPasswd() && newpassword != null) {
803             // special case used to signal regeneraton of password
804             newpassword = profile.getAutoGeneratedPasswd();
805         }
806 
807         final EndEntityType type = endEntityInformation.getType();
808         final ExtendedInformation ei = endEntityInformation.getExtendedInformation();
809         // Check if user fulfills it's profile.
810         if (globalConfiguration.getEnableEndEntityProfileLimitations()) {
811             try {
812                 String dirattrs = null;
813                 if (ei != null) {
814                     dirattrs = ei.getSubjectDirectoryAttributes();
815                 }
816                 // It is only meaningful to verify the password if we change it in some way, and if we are not autogenerating it
817                 if (!profile.useAutoGeneratedPasswd() && StringUtils.isNotEmpty(newpassword)) {
818                     profile.doesUserFulfillEndEntityProfile(username, endEntityInformation.getPassword(), dn, altName, dirattrs, endEntityInformation.getEmail(),
819                             endEntityInformation.getCertificateProfileId(), clearpwd, type.contains(EndEntityTypes.KEYRECOVERABLE),
820                             type.contains(EndEntityTypes.SENDNOTIFICATION), endEntityInformation.getTokenType(), endEntityInformation.getHardTokenIssuerId(), caid, ei);
821                 } else {
822                     profile.doesUserFulfillEndEntityProfileWithoutPassword(username, dn, altName, dirattrs, endEntityInformation.getEmail(),
823                             endEntityInformation.getCertificateProfileId(), type.contains(EndEntityTypes.KEYRECOVERABLE),
824                             type.contains(EndEntityTypes.SENDNOTIFICATION), endEntityInformation.getTokenType(), endEntityInformation.getHardTokenIssuerId(), caid, ei);
825                 }
826             } catch (EndEntityProfileValidationException e) {
827                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
828                 details.put("msg", intres.getLocalizedMessage("ra.errorfulfillprofile", Integer.valueOf(endEntityProfileId), dn, e.getMessage()));
829                 auditSession.log(EjbcaEventTypes.RA_EDITENDENTITY, EventStatus.FAILURE, EjbcaModuleTypes.RA, ServiceTypes.CORE, admin.toString(),
830                         String.valueOf(caid), null, username, details);
831                 throw e;
832             }
833         }
834         // Check name constraints
835         final CAInfo cainfo = caSession.getCAInfoInternal(caid, null, true);
836         final boolean nameChanged = // only check when name is changed so existing end-entities can be changed even if they violate NCs
837                 !userData.getSubjectDnNeverNull().equals(CertTools.stringToBCDNString(dn)) ||
838                 (userData.getSubjectAltName() != null && !userData.getSubjectAltName().equals(altName));
839         if (nameChanged && cainfo instanceof X509CAInfo && !cainfo.getCertificateChain().isEmpty()) {
840             final X509CAInfo x509cainfo = (X509CAInfo) cainfo;
841             final X509Certificate cacert = (X509Certificate)cainfo.getCertificateChain().iterator().next();
842             final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(userData.getCertificateProfileId());
843             
844             final X500NameStyle nameStyle;
845             if (x509cainfo.getUsePrintableStringSubjectDN()) {
846                 nameStyle = PrintableStringNameStyle.INSTANCE;
847             } else {
848                 nameStyle = CeSecoreNameStyle.INSTANCE;
849             }
850             
851             final boolean ldaporder;
852             if (x509cainfo.getUseLdapDnOrder() && (certProfile != null && certProfile.getUseLdapDnOrder())) {
853                 ldaporder = true; // will cause an error to be thrown later if name constraints are used
854             } else {
855                 ldaporder = false;
856             }
857             
858             X500Name subjectDNName = CertTools.stringToBcX500Name(dn, nameStyle, ldaporder);
859             GeneralNames subjectAltName = CertTools.getGeneralNamesFromAltName(altName);
860             try {
861                 CertTools.checkNameConstraints(cacert, subjectDNName, subjectAltName);
862             } catch (IllegalNameException e) {
863                 e.setErrorCode(ErrorCode.NAMECONSTRAINT_VIOLATION);
864                 throw e;
865             }
866         }
867         // Check if approvals is required.
868         final CertificateProfile certificateProfile = certificateProfileSession.getCertificateProfile(endEntityInformation.getCertificateProfileId());
869         final ApprovalProfile approvalProfile = approvalProfileSession.getApprovalProfileForAction(ApprovalRequestType.ADDEDITENDENTITY, cainfo, 
870                 certificateProfile);
871         if (approvalProfile != null) {
872             final EndEntityInformation orguserdata = userData.toEndEntityInformation();
873             final EndEntityInformation requestInfo = new EndEntityInformation(endEntityInformation);
874             if (newUsername != null && !newUsername.equals(username)) {
875                 requestInfo.setUsername(newUsername);
876             }
877             final EditEndEntityApprovalRequest ar = new EditEndEntityApprovalRequest(requestInfo, clearpwd, orguserdata, admin, null,
878                      caid, endEntityProfileId, approvalProfile, true);
879             if (ApprovalExecutorUtil.requireApproval(ar, NONAPPROVABLECLASSNAMES_CHANGEUSER)) {
880                 final int requestId = approvalSession.addApprovalRequest(admin, ar);
881                 throw new WaitingForApprovalException(intres.getLocalizedMessage("ra.approvaledit"), requestId);
882             }
883         }
884         // Rename the end entity if there's a new username
885         if (newUsername != null && !newUsername.equals(username)) {
886             boolean success = false;
887             try {
888                 success = renameEndEntity(admin, username, newUsername);
889             } catch (EndEntityExistsException e) {
890                 throw new IllegalNameException("Username already taken");
891             }
892             if (!success) {
893                 throw new NoSuchEndEntityException("End entity does not exist");
894             }
895             username = newUsername;
896             userData = endEntityAccessSession.findByUsername(username);
897         }      
898         // Check if the subjectDN serialnumber already exists.
899         // No need to access control on the CA here just to get these flags, we have already checked above that we are authorized to the CA
900         if (cainfo.isDoEnforceUniqueSubjectDNSerialnumber()) {
901             if (!isSubjectDnSerialnumberUnique(caid, dn, username)) {
902                 throw new CertificateSerialNumberException("Error: SubjectDN Serialnumber already exists.");
903             }
904         }
905         
906         try {
907             userData.setDN(dn);
908             userData.setSubjectAltName(altName);
909             userData.setSubjectEmail(endEntityInformation.getEmail());
910             userData.setCaId(caid);
911             userData.setType(type.getHexValue());
912             userData.setEndEntityProfileId(endEntityProfileId);
913             userData.setCertificateProfileId(endEntityInformation.getCertificateProfileId());
914             userData.setTokenType(endEntityInformation.getTokenType());
915             userData.setHardTokenIssuerId(endEntityInformation.getHardTokenIssuerId());
916             userData.setCardNumber(endEntityInformation.getCardNumber());
917             final int newstatus = endEntityInformation.getStatus();
918             final int oldstatus = userData.getStatus();
919             if (oldstatus == EndEntityConstants.STATUS_KEYRECOVERY && newstatus != EndEntityConstants.STATUS_KEYRECOVERY
920                     && newstatus != EndEntityConstants.STATUS_INPROCESS) {
921                 keyRecoverySession.unmarkUser(admin, username);
922             }
923             if (ei != null) {
924                 final String requestCounter = ei.getCustomData(ExtendedInformationFields.CUSTOM_REQUESTCOUNTER);
925                 if (StringUtils.equals(requestCounter, "0") && newstatus == EndEntityConstants.STATUS_NEW && oldstatus != EndEntityConstants.STATUS_NEW) {
926                     // If status is set to new, we should re-set the allowed request counter to the default values
927                     // But we only do this if no value is specified already, i.e. 0 or null
928                     resetRequestCounter(admin, false, ei, username, endEntityProfileId);
929                 } else {
930                     // If status is not new, we will only remove the counter if the profile does not use it
931                     resetRequestCounter(admin, true, ei, username, endEntityProfileId);
932                 }
933                 
934                 // Make sure that information about related approval requests are carried over to the edited end entity. 
935                 // This is done to make it possible to  trace/find an approval request from the actually added/edited end entity
936                 final ExtendedInformation oldExtendedInfo = userData.getExtendedInformation();

1707             final String msg = intres.getLocalizedMessage("ra.errorfindentitycert", issuerdn, certserno.toString(16));

1709             throw new NoSuchEndEntityException(msg);

1724         final CertReqHistory certReqHistory = certreqHistorySession.retrieveCertReqHistory(certserno, issuerdn);
1725         int endEntityProfileId = certificateData.getEndEntityProfileId()==null ? -1 : certificateData.getEndEntityProfileIdOrZero();
1726         final EndEntityInformation endEntityInformation = endEntityInformationParam==null ? endEntityAccessSession.findUser(username) : endEntityInformationParam;
1727         if (certReqHistory == null) {
1728             if (endEntityInformation!=null) {
1729                 // Get the EEP that is currently used as a fallback, if we can find it
1730                 endEntityProfileId = endEntityInformation.getEndEntityProfileId();

1950     private void sendNotification(final AuthenticationToken admin, final EndEntityInformation endEntityInformation, final int newstatus, 
1951             final int approvalRequestID, final AuthenticationToken lastApprovingAdmin, CertificateDataWrapper revokedCertificate) {
1952         if (endEntityInformation == null) {
1953             if (log.isDebugEnabled()) {
1954                 log.debug("No UserData, no notification sent.");
1955             }
1956             return;
1957         }
1958         final String userEmail = endEntityInformation.getEmail();
1959         if (log.isTraceEnabled()) {
1960             log.trace(">sendNotification: user=" + endEntityInformation.getUsername() + ", email=" + userEmail);
1961         }
1962         // Make check if we should send notifications at all
1963         if (endEntityInformation.getType().contains(EndEntityTypes.SENDNOTIFICATION)) {
1964             final int endEntityProfileId = endEntityInformation.getEndEntityProfileId();
1965             final EndEntityProfile endEntityProfile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
1966             final List<UserNotification> userNotifications = endEntityProfile.getUserNotifications();
1967             if (log.isDebugEnabled()) {
1968                 log.debug("Number of user notifications: " + userNotifications.size());
1969             }
1970             String recipientEmail = userEmail; // Default value
1971             for (final UserNotification userNotification : userNotifications) {
1972                 final Collection<String> events = userNotification.getNotificationEventsCollection();
1973                 if (events.contains(String.valueOf(newstatus))) {
1974                     if (log.isDebugEnabled()) {
1975                         log.debug("Status is " + newstatus + ", notification sent for notificationevents: " + userNotification.getNotificationEvents());
1976                     }
1977                     try {
1978                         if (StringUtils.equals(userNotification.getNotificationRecipient(), UserNotification.RCPT_USER)) {
1979                             recipientEmail = userEmail;
1980                         } else if (StringUtils.contains(userNotification.getNotificationRecipient(), UserNotification.RCPT_CUSTOM)) {
1981                             // Just if this fail it will say that sending to user with email "custom" failed.
1982                             recipientEmail = "custom";
1983                             // Plug-in mechanism for retrieving custom notification email recipient addresses
1984                             if (userNotification.getNotificationRecipient().length() < 6) {
1985                                 final String msg = intres.getLocalizedMessage("ra.errorcustomrcptshort", userNotification.getNotificationRecipient());
1986                                 log.error(msg);
1987                             } else {
1988                                 final String customClassName = userNotification.getNotificationRecipient().substring(7);
1989                                 if (StringUtils.isNotEmpty(customClassName)) {
1990                                     ICustomNotificationRecipient plugin;
1991                                     try {
1992                                         plugin = (ICustomNotificationRecipient) Thread.currentThread()
1993                                                 .getContextClassLoader().loadClass(customClassName).getConstructor().newInstance();
1994                                     } catch (InstantiationException | IllegalAccessException | ClassNotFoundException | NoSuchMethodException | InvocationTargetException e) {
1995                                         throw new MailException("Custom notification class " + customClassName + " could not be instansiated.", e);
1996                                     }
1997                                     recipientEmail = plugin.getRecipientEmails(endEntityInformation);
1998                                     if (StringUtils.isEmpty(recipientEmail)) {
1999                                         final String msg = intres.getLocalizedMessage("ra.errorcustomnoemail", userNotification.getNotificationRecipient());
2000                                         log.error(msg);
2001                                     } else {
2002                                         if (log.isDebugEnabled()) {
2003                                             log.debug("Custom notification recipient plugin returned email: " + recipientEmail);
2004                                         }
2005                                     }
2006                                 } else {
2007                                     final String msg = intres.getLocalizedMessage("ra.errorcustomnoclasspath", userNotification.getNotificationRecipient());
2008                                     log.error(msg);
2009                                 }
2010                             }
2011                         } else {
2012                             // Just a plain email address specified in the recipient field
2013                             recipientEmail = userNotification.getNotificationRecipient();
2014                         }
2015                         if (StringUtils.isEmpty(recipientEmail)) {
2016                             final String msg = intres.getLocalizedMessage("ra.errornotificationnoemail", endEntityInformation.getUsername());
2017                             throw new MailException(msg);
2018                         }
2019                         // Get the administrators DN from the admin certificate, if one exists
2020                         EndEntityInformation requestAdmin = null;
2021                         if (admin instanceof X509CertificateAuthenticationToken) {
2022                             final X509CertificateAuthenticationToken xtok = (X509CertificateAuthenticationToken) admin;
2023                             final X509Certificate adminCert = xtok.getCertificate();
2024                             final String username = certificateStoreSession.findUsernameByFingerprint(CertTools.getFingerprintAsString(adminCert));
2025                             if (username!=null) {
2026                                 requestAdmin = endEntityAccessSession.findUser(username);
2027                             }
2028                         }
2029                         String lastApprovalAdminDN = null;
2030                         if (lastApprovingAdmin instanceof X509CertificateAuthenticationToken) {
2031                             final X509CertificateAuthenticationToken xtok = (X509CertificateAuthenticationToken) lastApprovingAdmin;
2032                             final X509Certificate adminCert = xtok.getCertificate();
2033                             lastApprovalAdminDN = CertTools.getSubjectDN(adminCert);
2034                         }
2035 
2036                         final UserNotificationParamGen paramGen = new UserNotificationParamGen(endEntityInformation, lastApprovalAdminDN, requestAdmin, 
2037                                 approvalRequestID, revokedCertificate);
2038                         // substitute any $ fields in the recipient and from fields
2039                         recipientEmail = paramGen.interpolate(recipientEmail);
2040                         final String fromemail = paramGen.interpolate(userNotification.getNotificationSender());
2041                         final String subject = paramGen.interpolate(userNotification.getNotificationSubject());
2042                         final String message = paramGen.interpolate(userNotification.getNotificationMessage());
2043                         MailSender.sendMailOrThrow(fromemail, Arrays.asList(recipientEmail), MailSender.NO_CC, subject, message, MailSender.NO_ATTACHMENTS);
2044                         final String logmsg = intres.getLocalizedMessage("ra.sentnotification", endEntityInformation.getUsername(), recipientEmail);
2045                         log.info(logmsg);
2046                     } catch (MailException e) {
2047                         final String msg = intres.getLocalizedMessage("ra.errorsendnotification", endEntityInformation.getUsername(), recipientEmail);
2048                         log.error(msg, e);
2049                     }
2050                 } else {
2051                     if (log.isDebugEnabled()) {
2052                         log.debug("Status is " + newstatus + ", no notification sent for notificationevents: " + userNotification.getNotificationEvents());
2053                     }
2054                 }
2055             }
2056         } else {
2057             if (log.isDebugEnabled()) {
2058                 log.debug("Type ("+endEntityInformation.getType().getHexValue()+") does not contain EndEntityTypes.USER_SENDNOTIFICATION, no notification sent.");
2059             }
2060         }
2061         if (log.isTraceEnabled()) {
2062             log.trace("<sendNotification: user=" + endEntityInformation.getUsername() + ", email=" + userEmail);
2063         }
2064     }
