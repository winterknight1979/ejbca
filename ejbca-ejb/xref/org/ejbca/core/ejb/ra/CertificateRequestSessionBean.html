119     @Override
120     public byte[] processCertReq(AuthenticationToken admin, EndEntityInformation userdata, String req, int reqType, String hardTokenSN,
121             int responseType) throws AuthorizationDeniedException, NotFoundException, InvalidKeyException, NoSuchAlgorithmException,
122             InvalidKeySpecException, NoSuchProviderException, SignatureException, IOException, CertificateException,
123             EndEntityProfileValidationException, ApprovalException, EjbcaException, CesecoreException, CertificateExtensionException {
124         byte[] retval = null;
125 
126         // Check tokentype
127         if (userdata.getTokenType() != SecConst.TOKEN_SOFT_BROWSERGEN) {
128             throw new WrongTokenTypeException("Error: Wrong Token Type of user, must be 'USERGENERATED' for PKCS10/SPKAC/CRMF/CVC requests");
129         }
130         // This is the secret sauce, do the end entity handling automagically here before we get the cert
131         addOrEditUser(admin, userdata, false, true);
132         // Process request
133         try {
134             String password = userdata.getPassword();
135             String username = userdata.getUsername();
136             RequestMessage imsg = RequestMessageUtils.getRequestMessageFromType(username, password, req, reqType);
137             if (imsg != null) {
138                 retval = getCertResponseFromPublicKey(admin, imsg, hardTokenSN, responseType, userdata);
139             }
140         } catch (NotFoundException e) {
141             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
142             throw e;
143         } catch (InvalidKeyException e) {
144             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
145             throw e;
146         } catch (NoSuchAlgorithmException e) {
147             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
148             throw e;
149         } catch (InvalidKeySpecException e) {
150             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
151             throw e;
152         } catch (NoSuchProviderException e) {
153             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
154             throw e;
155         } catch (SignatureException e) {
156             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
157             throw e;
158         } catch (IOException e) {
159             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
160             throw e;
161         } catch (CertificateException e) {
162             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
163             throw e;
164         } catch (EjbcaException e) {
165             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
166             throw e;
167         } catch (CesecoreException e) {
168             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
169             throw e;
170         } catch (ParseException e) {
171             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
172             throw new EjbcaException(ErrorCode.FIELD_VALUE_NOT_VALID, e);
173         } catch (ConstructionException e) {
174             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
175             throw new EjbcaException(ErrorCode.FIELD_VALUE_NOT_VALID, e);
176         } catch (NoSuchFieldException e) {
177             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
178             throw new EjbcaException(ErrorCode.FIELD_VALUE_NOT_VALID, e);
179         } catch (CertificateExtensionException e) {
180             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
181             throw e;
182         }
183         return retval;
184     }

186     @Override
187     public ResponseMessage processCertReq(AuthenticationToken admin, EndEntityInformation userdata, RequestMessage req, Class<? extends CertificateResponseMessage> responseClass)
188             throws EndEntityExistsException, AuthorizationDeniedException, EndEntityProfileValidationException, EjbcaException, CesecoreException, CertificateExtensionException {
189         // Check tokentype
190         if (userdata.getTokenType() != SecConst.TOKEN_SOFT_BROWSERGEN) {
191             throw new WrongTokenTypeException("Error: Wrong Token Type of user, must be 'USERGENERATED' for PKCS10/SPKAC/CRMF/CVC requests");
192         }
193         // This is the secret sauce, do the end entity handling automagically here before we get the cert
194         addOrEditUser(admin, userdata, false, true);
195         ResponseMessage retval = null;
196         try {
197             retval = signSession.createCertificate(admin, req, responseClass, userdata);
198         } catch (EjbcaException e) {
199             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
200             throw e;
201         } catch (CertificateExtensionException e) {
202             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
203             throw e;
204         }
205         return retval;
206     }

208     /**
209      * @param admin Admin
210      * @param userdata User
211      * @param clearpwd PWD
212      * @param fromwebservice bool 
213      * @throws AuthorizationDeniedException fail
214      * @throws EndEntityProfileValidationException fail 
215      * @throws ApprovalException fail
216      * @throws EndEntityExistsException fail 
217      * @throws CADoesntExistsException if userdata.caId is not a valid caid. This is checked in editUser or addUserFromWS
218      * @throws IllegalNameException  if the Subject DN failed constraints
219      * @throws CertificateSerialNumberException if SubjectDN serial number already exists.
220      * @throws CustomFieldException if the end entity was not validated by a locally defined field validator
221      * @throws NoSuchEndEntityException if the end entity was not found
222      */
223     private void addOrEditUser(AuthenticationToken admin, EndEntityInformation userdata, boolean clearpwd, boolean fromwebservice)
224             throws AuthorizationDeniedException, EndEntityProfileValidationException, ApprovalException, EndEntityExistsException,
225             CADoesntExistsException, CertificateSerialNumberException, IllegalNameException, CustomFieldException, NoSuchEndEntityException {
226 
227         int caid = userdata.getCAId();
228         if (!authorizationSession.isAuthorizedNoLogging(admin, StandardRules.CAACCESS.resource() + caid)) {
229             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", StandardRules.CAACCESS.resource() + caid, null);
230             throw new AuthorizationDeniedException(msg);
231         }
232         if (!authorizationSession.isAuthorizedNoLogging(admin, AccessRulesConstants.REGULAR_CREATECERTIFICATE)) {
233             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", AccessRulesConstants.REGULAR_CREATECERTIFICATE,
234                     null);
235             throw new AuthorizationDeniedException(msg);
236         }
237         // First we need to fetch the CA configuration to see if we save UserData, if not, we still run addUserFromWS to
238         // get all the proper authentication checks for CA and end entity profile.
239         // No need to to access control here just to fetch this flag, access control for the CA is done in EndEntityManagementSession
240         boolean useUserStorage = caSession.getCAInfoInternal(caid, null, true).isUseUserStorage();
241         // Add or edit user
242         try {
243             String username = userdata.getUsername();
244             if (useUserStorage && endEntityManagementSession.existsUser(username)) {
245                 if (log.isDebugEnabled()) {
246                     log.debug("User " + username + " exists, update the userdata. New status of user '" + userdata.getStatus() + "'.");
247                 }
248                 endEntityManagementSession.changeUser(admin, userdata, clearpwd, fromwebservice);
249             } else {
250                 if (log.isDebugEnabled()) {
251                     log.debug("New User " + username + ", adding userdata. New status of user '" + userdata.getStatus() + "'.");
252                 }
253                 // addUserfromWS also checks useUserStorage internally, so don't duplicate the check
254                 endEntityManagementSession.addUserFromWS(admin, userdata, clearpwd);
255             }
256         } catch (WaitingForApprovalException e) {
257             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
258             String msg = "Single transaction enrollment request rejected since approvals are enabled for this CA (" + caid
259                     + ") or Certificate Profile (" + userdata.getCertificateProfileId() + ").";
260             log.info(msg);
261             throw new ApprovalException(msg);
262         }
263     }

265     /**
266      * Process a request in the CA module.
267      *
268      * @param admin is the requesting administrator
269      * @param msg is the request message processed by the CA
270      * @param hardTokenSN is the hard token to associate this or null
271      * @param responseType is one of SecConst.CERT_RES_TYPE_...
272      * @param userData Data
273      * @return a encoded certificate of the type specified in responseType
274      * @throws EjbcaException fail
275      * @throws CesecoreException fail
276      * @throws CertificateEncodingException fail
277      * @throws CertificateException fail
278      * @throws IOException fail
279      * @throws AuthorizationDeniedException fail
280      * @throws CertificateExtensionException if the request message contained invalid extensions
281      */
282     private byte[] getCertResponseFromPublicKey(AuthenticationToken admin, RequestMessage msg, String hardTokenSN, int responseType,
283             EndEntityInformation userData) throws EjbcaException, CesecoreException, CertificateEncodingException, CertificateException, IOException,
284             AuthorizationDeniedException, CertificateExtensionException {
285         byte[] retval = null;
286         Class<X509ResponseMessage> respClass = X509ResponseMessage.class;
287         ResponseMessage resp = signSession.createCertificate(admin, msg, respClass, userData);
288         X509Certificate cert = CertTools.getCertfromByteArray(resp.getResponseMessage(), X509Certificate.class);
289         if (responseType == CertificateConstants.CERT_RES_TYPE_CERTIFICATE) {
290             retval = cert.getEncoded();
291         }
292         if (responseType == CertificateConstants.CERT_RES_TYPE_PKCS7) {
293             retval = signSession.createPKCS7(admin, cert, false);
294         }
295         if (responseType == CertificateConstants.CERT_RES_TYPE_PKCS7WITHCHAIN) {
296             retval = signSession.createPKCS7(admin, cert, true);
297         }
298 
299         if (hardTokenSN != null) {
300             hardTokenSession.addHardTokenCertificateMapping(admin, hardTokenSN, cert);
301         }
302         return retval;
303     }

305     @Override
306     public byte[] processSoftTokenReq(AuthenticationToken admin, EndEntityInformation userdata, String hardTokenSN, String keyspec, String keyalg,
307             boolean createJKS) throws ApprovalException, EndEntityExistsException, CADoesntExistsException, CertificateSerialNumberException,
308             IllegalNameException, CustomFieldException, AuthorizationDeniedException, EndEntityProfileValidationException, NoSuchAlgorithmException,
309             InvalidKeySpecException, CertificateException, InvalidAlgorithmParameterException, KeyStoreException, NoSuchEndEntityException {
310 
311         // This is the secret sauce, do the end entity handling automagically here before we get the cert
312         addOrEditUser(admin, userdata, false, true);
313         // Process request
314         byte[] ret = null;
315         try {
316             // Get key recovery info
317             boolean usekeyrecovery = ( (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID)).getEnableKeyRecovery();
318             if (log.isDebugEnabled()) {
319                 log.debug("usekeyrecovery: " + usekeyrecovery);
320             }
321             boolean savekeys = userdata.getKeyRecoverable() && usekeyrecovery && (userdata.getStatus() != EndEntityConstants.STATUS_KEYRECOVERY);
322             if (log.isDebugEnabled()) {
323                 log.debug("userdata.getKeyRecoverable(): " + userdata.getKeyRecoverable());
324                 log.debug("userdata.getStatus(): " + userdata.getStatus());
325                 log.debug("savekeys: " + savekeys);
326             }
327             boolean loadkeys = (userdata.getStatus() == EndEntityConstants.STATUS_KEYRECOVERY) && usekeyrecovery;
328             if (log.isDebugEnabled()) {
329                 log.debug("loadkeys: " + loadkeys);
330             }
331             int endEntityProfileId = userdata.getEndEntityProfileId();
332             EndEntityProfile endEntityProfile = endEntityProfileSession.getEndEntityProfileNoClone(endEntityProfileId);
333             boolean reusecertificate = endEntityProfile.getReUseKeyRecoveredCertificate();
334             if (log.isDebugEnabled()) {
335                 log.debug("reusecertificate: " + reusecertificate);
336             }
337             // Generate keystore
338             String password = userdata.getPassword();
339             String username = userdata.getUsername();
340             int caid = userdata.getCAId();
341             KeyStore keyStore = keyStoreCreateSession.generateOrKeyRecoverToken(admin, username, password, caid, keyspec, keyalg, null, null,
342                     createJKS, loadkeys, savekeys,
343                     reusecertificate, endEntityProfileId);
344             String alias = keyStore.aliases().nextElement();
345             X509Certificate cert = (X509Certificate) keyStore.getCertificate(alias);
346             if ((hardTokenSN != null) && (cert != null)) {
347                 hardTokenSession.addHardTokenCertificateMapping(admin, hardTokenSN, cert);
348             }
349             ByteArrayOutputStream baos = new ByteArrayOutputStream();
350             keyStore.store(baos, password.toCharArray());
351             ret = baos.toByteArray();
352         } catch (NoSuchAlgorithmException e) {
353             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
354             throw e;
355         } catch (InvalidKeySpecException e) {
356             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
357             throw e;
358         } catch (IOException e) {
359             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
360             throw new IllegalStateException(e);
361         } catch (CertificateException e) {
362             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
363             throw e;
364         } catch (InvalidAlgorithmParameterException e) {
365             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
366             throw e;
367         } catch (RuntimeException e) {
368             throw e;
369         } catch (Exception e) {
370             sessionContext.setRollbackOnly(); // This is an application exception so it wont trigger a roll-back automatically
371             throw new KeyStoreException(e);
372         }
373         return ret;
374     }
