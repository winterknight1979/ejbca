118     @Override
119     public byte[] generateOrKeyRecoverTokenAsByteArray(final AuthenticationToken authenticationToken, final String username, final String password, final String hardTokenSN, final String keySpecification, final String keyAlgorithm)
120             throws CADoesntExistsException, AuthorizationDeniedException, NotFoundException, EjbcaException {
121         // Check if user exists.
122         final EndEntityInformation endEntity = endEntityAccessSession.findUser(authenticationToken, username);
123         if(endEntity == null) {
124             log.info(intres.getLocalizedMessage("ra.errorentitynotexist", username));
125             throw new NotFoundException(intres.getLocalizedMessage("ra.wrongusernameorpassword"));
126         }
127         // Check CA ID and authorization.
128         final int caId = endEntity.getCAId();
129         caSession.verifyExistenceOfCA(caId);
130         if(!authorizationSession.isAuthorized(authenticationToken, StandardRules.CAACCESS.resource() + caId, StandardRules.CREATECERT.resource())) {
131             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", StandardRules.CAACCESS.resource() +caId +
132                     "," + StandardRules.CREATECERT.resource(), null);
133             throw new AuthorizationDeniedException(msg);
134         }
135         // Check token type.
136         if(endEntity.getTokenType() != SecConst.TOKEN_SOFT_P12) { // logger
137             throw new EjbcaException(ErrorCode.BAD_USER_TOKEN_TYPE, "Error: Wrong Token Type of user, must be 'P12' for PKCS12 requests");
138         }
139         final boolean useKeyRecovery = ((GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID)).getEnableKeyRecovery();
140         if (log.isDebugEnabled()) {
141             log.debug("usekeyrecovery: " + useKeyRecovery);
142         }
143         final boolean saveKeys = endEntity.getKeyRecoverable() && useKeyRecovery && (endEntity.getStatus() != EndEntityConstants.STATUS_KEYRECOVERY);
144         if (log.isDebugEnabled()) {
145             log.debug("userdata.getKeyRecoverable(): " + endEntity.getKeyRecoverable());
146             log.debug("userdata.getStatus(): " + endEntity.getStatus());
147             log.debug("savekeys: " + saveKeys);
148         }
149         final boolean loadKeys = (endEntity.getStatus() == EndEntityConstants.STATUS_KEYRECOVERY) && useKeyRecovery;
150         if (log.isDebugEnabled()) {
151             log.debug("loadkeys: " + loadKeys);
152         }
153         final int endEntityProfileId = endEntity.getEndEntityProfileId();
154         final EndEntityProfile endEntityProfile = endEntityProfileSession.getEndEntityProfile(endEntityProfileId);
155         final boolean reuseCertificate = endEntityProfile.getReUseKeyRecoveredCertificate();
156         if (log.isDebugEnabled()) {
157             log.debug("reusecertificate: " + reuseCertificate);
158         }
159         try {
160             final KeyStore keyStore = generateOrKeyRecoverToken(authenticationToken, username, password, caId,
161                     keySpecification, keyAlgorithm, null, null, false, loadKeys, saveKeys, reuseCertificate, endEntityProfileId);
162             final String alias = keyStore.aliases().nextElement();
163             final X509Certificate certificate = (X509Certificate) keyStore.getCertificate(alias);
164             if ((hardTokenSN != null) && (certificate != null)) {
165                 hardTokenSession.addHardTokenCertificateMapping(authenticationToken,hardTokenSN,certificate);
166             }
167             return KeyStoreTools.getAsByteArray(keyStore, password);
168         } catch (AuthLoginException e) { // Is handled as EjbcaException at caller (EjbcaWS).
169             throw e;
170         } catch (AuthStatusException e) { // Is handled as EjbcaException at caller (EjbcaWS).
171             throw e;
172         } catch (Exception e) {
173             if (log.isDebugEnabled()) {
174                 log.debug("Re-throw exception in RA master API: " + e.getMessage(), e);
175             }
176             throw new EjbcaException(ErrorCode.INTERNAL_ERROR, e.getMessage());
177         }
178     }

180     @Override
181     public byte[] generateOrKeyRecoverTokenAsByteArray(AuthenticationToken administrator, String username, String password, int caid, String keyspec,
182             String keyalg, boolean createJKS, boolean loadkeys, boolean savekeys, boolean reusecertificate, int endEntityProfileId)
183             throws AuthorizationDeniedException, KeyStoreException, InvalidAlgorithmParameterException, CADoesntExistsException, IllegalKeyException,
184             CertificateCreateException, IllegalNameException, CertificateRevokeException, CertificateSerialNumberException,
185             CryptoTokenOfflineException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException,
186             CustomCertificateSerialNumberException, AuthStatusException, AuthLoginException, EndEntityProfileValidationException, NoSuchEndEntityException,
187             CertificateSignatureException, CertificateEncodingException, CertificateException, NoSuchAlgorithmException, InvalidKeySpecException {
188         KeyStore keyStore = generateOrKeyRecoverToken(administrator, username, password, caid, keyspec, keyalg, null, null, createJKS, loadkeys,
189                 savekeys,
190                 reusecertificate, endEntityProfileId);
191         return KeyStoreTools.getAsByteArray(keyStore, password);
192     }

194     @Override
195     public KeyStore generateOrKeyRecoverToken(AuthenticationToken administrator, String username, String password, int caid,
196             String keyspec, String keyalg, Date notBefore, Date notAfter, boolean createJKS, boolean loadkeys, boolean savekeys,
197             boolean reusecertificate,
198             int endEntityProfileId)
199             throws AuthorizationDeniedException, KeyStoreException, InvalidAlgorithmParameterException, CADoesntExistsException, IllegalKeyException,
200             CertificateCreateException, IllegalNameException, CertificateRevokeException, CertificateSerialNumberException,
201             CryptoTokenOfflineException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException,
202             CustomCertificateSerialNumberException, AuthStatusException, AuthLoginException, EndEntityProfileValidationException, NoSuchEndEntityException,
203             CertificateSignatureException, CertificateEncodingException, CertificateException, NoSuchAlgorithmException, InvalidKeySpecException {
204         if (log.isTraceEnabled()) {
205             log.trace(">generateOrKeyRecoverToken");
206         }
207         boolean isNewToken = false;
208     	KeyRecoveryInformation keyData = null;
209     	KeyPair rsaKeys = null;
210     	EndEntityInformation userdata = endEntityAccessSession.findUser(administrator, username);
211     	if (userdata.getStatus() == EndEntityConstants.STATUS_NEW) {
212     	    isNewToken = true;
213     	}
214     	if (loadkeys) {
215     	    if (log.isDebugEnabled()) {
216     	        log.debug("Recovering keys for user: "+ username);
217     	    }
218             // used saved keys.
219 			keyData = keyRecoverySession.recoverKeys(administrator, username, endEntityProfileId);
220     		if (keyData == null) {
221     			throw new KeyStoreException("No key recovery data exists for user");
222     		}
223     		rsaKeys = keyData.getKeyPair();
224     		if (reusecertificate) {
225     			// This is only done if reusecertificate == true because if you don't re-use certificate
226     		    // signSession.createCertificate is called, which set status to generated, unless finishUser == false in CA config
227                 if (log.isDebugEnabled()) {
228                     log.debug("Re-using old certificate for user: "+ username);
229                 }
230     			keyRecoverySession.unmarkUser(administrator,username);
231     		}
232     		caid = keyData.getIssuerDN().hashCode(); // always use the CA of the certificate
233     	} else {
234             if (log.isDebugEnabled()) {
235                 log.debug("Generating new keys for user: "+ username);
236             }
237 
238             //KeyStore algorithm specification inside endEntityInformation has priority since its algorithm is approved
239             if (userdata.getExtendedInformation() != null) {
240                 if (userdata.getExtendedInformation().getKeyStoreAlgorithmType() != null
241                         && userdata.getExtendedInformation().getKeyStoreAlgorithmSubType() != null) {
242                     if (log.isDebugEnabled()) {
243                         log.debug("Using the key-store algorithm specification found inside the endEntityInformation ("
244                                 + userdata.getExtendedInformation().getKeyStoreAlgorithmType() + "_"
245                                 + userdata.getExtendedInformation().getKeyStoreAlgorithmSubType() + ") instead of one provided separately (" + keyalg
246                                 + "_" + keyspec + ")");
247                     }
248                     keyalg = userdata.getExtendedInformation().getKeyStoreAlgorithmType();
249                     keyspec = userdata.getExtendedInformation().getKeyStoreAlgorithmSubType();
250                 }
251             }
252             // generate new keys.
253             rsaKeys = KeyTools.genKeys(keyspec, keyalg);
254     	}
255     	X509Certificate cert = null;
256     	if ((reusecertificate) && (keyData != null)) {
257             cert = (X509Certificate) keyData.getCertificate(); // TODO Validity ignored here?
258     		boolean finishUser = true;
259 			finishUser = caSession.getCAInfo(administrator,caid).getFinishUser();
260     		if (finishUser) {
261 				authenticationSession.finishUser(userdata);
262     		}
263     	} else {
264             if (log.isDebugEnabled()) {
265                 log.debug("Generating new certificate for user: "+ username);
266             }
267             cert = (X509Certificate) signSession.createCertificate(administrator, username, password, new PublicKeyWrapper(rsaKeys.getPublic()), -1,
268                     notBefore, notAfter);
269     	}
270     	// Clear password from database
271     	userdata = endEntityAccessSession.findUser(administrator, username); //Get GENERATED end entity information
272         if (userdata.getStatus() == EndEntityConstants.STATUS_GENERATED) {
273             // If we have a successful key recovery via EJBCA WS we implicitly want to allow resetting of the password without edit_end_entity rights (ECA-4947)
274             if (loadkeys) {
275                 endEntityManagementSession.setClearTextPassword(new AlwaysAllowLocalAuthenticationToken(
276                         new UsernamePrincipal("Implicit authorization from key recovery operation to reset password.")), username, null);
277             } else if (isNewToken) {
278                 // If we generate a new token through an enrollment, we don't want to demand access to edit_end_entity
279                 endEntityManagementSession.setClearTextPassword(new AlwaysAllowLocalAuthenticationToken(
280                         new UsernamePrincipal("Implicit authorization from new enrollments")), username, null);
281             } else {
282                 endEntityManagementSession.setClearTextPassword(administrator, username, null);
283             }
284     	}
285         // Make a certificate chain from the certificate and the CA-certificate
286         Certificate[] cachain = signSession.getCertificateChain(caid).toArray(new Certificate[0]);
287         // Verify CA-certificate
288     	Certificate rootcert = cachain[cachain.length - 1];
289     	if (CertTools.isSelfSigned(rootcert)) {
290     		try {
291     			rootcert.verify(rootcert.getPublicKey());
292     		} catch (GeneralSecurityException se) {
293                 throw new CertificateSignatureException("RootCA certificate does not verify, issuerDN: " + CertTools.getIssuerDN(rootcert)
294                         + ", subjectDN: " + CertTools.getSubjectDN(rootcert), se);
295     		}
296     	} else {
297     		throw new CertificateSignatureException("RootCA certificate not self-signed, issuerDN: "+CertTools.getIssuerDN(rootcert)+", subjectDN: "+CertTools.getSubjectDN(rootcert));
298     	}
299         // Verify that the user-certificate is signed by our CA
300     	Certificate cacert = cachain[0];
301     	try {
302     		cert.verify(cacert.getPublicKey());
303     	} catch (GeneralSecurityException se) {
304     		throw new CertificateSignatureException("Generated certificate does not verify using CA-certificate, issuerDN: "+CertTools.getIssuerDN(cert)+", subjectDN: "+CertTools.getSubjectDN(cert)+
305     				", caIssuerDN: "+CertTools.getIssuerDN(cacert)+", caSubjectDN: "+CertTools.getSubjectDN(cacert), se);
306     	}
307     	if (savekeys) {
308             // Save generated keys to database.
309             if (log.isDebugEnabled()) {
310                 log.debug("Saving generated keys for recovery for user: "+ username);
311             }
312 			keyRecoverySession.addKeyRecoveryData(administrator, EJBTools.wrap(cert), username, EJBTools.wrap(rsaKeys));
313     	}
314         //  Use CN if as alias in the keystore, if CN is not present use username
315     	String alias = CertTools.getPartFromDN(CertTools.getSubjectDN(cert), "CN");
316     	if (alias == null) {
317     		alias = username;
318     	}
319         // Store keys and certificates in keystore.
320     	KeyStore ks = null;
321     	if (createJKS) {
322             if (log.isDebugEnabled()) {
323                 log.debug("Generating JKS for user: "+ username);
324             }
325     		ks = KeyTools.createJKS(alias, rsaKeys.getPrivate(), password, cert, cachain);
326     	} else {
327             if (log.isDebugEnabled()) {
328                 log.debug("Generating PKCS12 for user: "+ username);
329             }
330     		ks = KeyTools.createP12(alias, rsaKeys.getPrivate(), cert, cachain);
331     	}
332         if (log.isTraceEnabled()) {
333             log.trace("<generateOrKeyRecoverToken");
334         }
335     	return ks;
336     }
