278     /** 
279      * @param admin Admin
280      * @param data Data
281      * @param requestedUsername User 
282      * @return the userdata value object if admin is authorized. Does not leak username if auth fails. 
283      * 
284      * @throws AuthorizationDeniedException if the admin was not authorized to the end entity profile or issuing CA
285      */
286     private EndEntityInformation convertUserDataToEndEntityInformation(final AuthenticationToken admin, final UserData data,
287             final String requestedUsername) throws AuthorizationDeniedException {
288         if (data != null) {
289             if (((GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID))
290                     .getEnableEndEntityProfileLimitations()) {
291                 // Check if administrator is authorized to view user.
292                 if (!authorizedToEndEntityProfile(admin, data.getEndEntityProfileId(), AccessRulesConstants.VIEW_END_ENTITY)) {
293                     if (requestedUsername == null) {
294                         final String msg = intres.getLocalizedMessage("ra.errorauthprofile", Integer.valueOf(data.getEndEntityProfileId()),
295                                 admin.toString());
296                         throw new AuthorizationDeniedException(msg);
297                     } else {
298                         final String msg = intres.getLocalizedMessage("ra.errorauthprofileexist", Integer.valueOf(data.getEndEntityProfileId()),
299                                 requestedUsername, admin.toString());
300                         throw new AuthorizationDeniedException(msg);
301                     }
302                 }
303             }
304             if (!authorizedToCA(admin, data.getCaId())) {
305                 if (requestedUsername == null) {
306                     final String msg = intres.getLocalizedMessage("ra.errorauthca", Integer.valueOf(data.getCaId()), admin.toString());
307                     throw new AuthorizationDeniedException(msg);
308                 } else {
309                     final String msg = intres.getLocalizedMessage("ra.errorauthcaexist", Integer.valueOf(data.getCaId()), requestedUsername,
310                             admin.toString());
311                     throw new AuthorizationDeniedException(msg);
312                 }
313             }
314             return data.toEndEntityInformation();
315         }
316         return null;
317     }

319     private boolean authorizedToEndEntityProfile(AuthenticationToken admin, int profileid, String rights) {
320         boolean returnval = false;
321         if (profileid == EndEntityConstants.EMPTY_END_ENTITY_PROFILE
322                 && (rights.equals(AccessRulesConstants.CREATE_END_ENTITY) || rights.equals(AccessRulesConstants.EDIT_END_ENTITY))) {
323             if (authorizationSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource())) {
324                 returnval = true;
325             } else {
326                 log.info("Admin " + admin.toString() + " was not authorized to resource " + StandardRules.ROLE_ROOT);
327             }
328         } else {
329             returnval = authorizationSession.isAuthorizedNoLogging(admin, AccessRulesConstants.ENDENTITYPROFILEPREFIX + profileid + rights,
330                     AccessRulesConstants.REGULAR_RAFUNCTIONALITY + rights);
331         }
332         return returnval;
333     }

335     private boolean authorizedToCA(AuthenticationToken admin, int caid) {
336         boolean returnval = false;
337         returnval = authorizationSession.isAuthorizedNoLogging(admin, StandardRules.CAACCESS.resource() + caid);
338         if (!returnval) {
339             log.info("Admin " + admin.toString() + " not authorized to resource " + StandardRules.CAACCESS.resource() + caid);
340         }
341         return returnval;
342     }

466     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
467     @Override
468     public Collection<EndEntityInformation> query(final AuthenticationToken admin, final Query query, final String caauthorizationstr,
469             final String endentityprofilestr, final int numberofrows, final String endentityAccessRule) throws IllegalQueryException {
470         boolean authorizedtoanyprofile = true;
471         final String caauthorizationstring = StringTools.strip(caauthorizationstr);
472         final String endentityprofilestring = StringTools.strip(endentityprofilestr);
473         final ArrayList<EndEntityInformation> returnval = new ArrayList<EndEntityInformation>();
474         int fetchsize = getGlobalCesecoreConfiguration().getMaximumQueryCount();
475 
476         if (numberofrows != 0) {
477             fetchsize = numberofrows;
478         }
479 
480         // Check if query is legal.
481         if (query != null && !query.isLegalQuery()) {
482             throw new IllegalQueryException();
483         }
484 
485         String sqlquery = "";
486         if (query != null) {
487             sqlquery += query.getQueryString();
488         }
489 
490         final GlobalConfiguration globalconfiguration = getGlobalConfiguration();
491         String caauthstring = caauthorizationstring;
492         String endentityauth = endentityprofilestring;
493         RAAuthorization raauthorization = null;
494         if (caauthorizationstring == null || endentityprofilestring == null) {
495             raauthorization = new RAAuthorization(admin, globalConfigurationSession, authorizationSession, caSession, endEntityProfileSession);
496             caauthstring = raauthorization.getCAAuthorizationString();
497             if (globalconfiguration.getEnableEndEntityProfileLimitations()) {
498                 endentityauth = raauthorization.getEndEntityProfileAuthorizationString(true, endentityAccessRule);
499             } else {
500                 endentityauth = "";
501             }
502         }
503         if (!StringUtils.isBlank(caauthstring)) {
504             if (StringUtils.isBlank(sqlquery)) {
505                 sqlquery += caauthstring;
506             } else {
507                 sqlquery = "(" + sqlquery + ") AND " + caauthstring;
508             }
509         } 
510         if (globalconfiguration.getEnableEndEntityProfileLimitations()) {
511             if (endentityauth == null || StringUtils.isBlank(endentityauth)) {
512                 authorizedtoanyprofile = false;
513             } else {
514                 if (StringUtils.isEmpty(sqlquery)) {
515                     sqlquery += endentityauth;
516                 } else {
517                     sqlquery = "(" + sqlquery + ") AND " + endentityauth;
518                 }
519             }
520         }
521         // Finally order the return values
522         sqlquery += " ORDER BY " + USERDATA_CREATED_COL + " DESC";
523         if (log.isDebugEnabled()) {
524             log.debug("generated query: " + sqlquery);
525         }
526         if (authorizedtoanyprofile) {
527             final javax.persistence.Query dbQuery = entityManager.createQuery("SELECT a FROM UserData a WHERE " + sqlquery);
528             if (fetchsize > 0) {
529                 dbQuery.setMaxResults(fetchsize);
530             }
531             @SuppressWarnings("unchecked")
532             final List<UserData> userDataList = dbQuery.getResultList();
533             for (UserData userData : userDataList) {
534                 returnval.add(userData.toEndEntityInformation());
535             }
536         } else {
537             if (log.isDebugEnabled()) {
538                 log.debug("authorizedtoanyprofile=false");
539             }
540         }
541         if (log.isTraceEnabled()) {
542             log.trace("<query(): " + returnval.size());
543         }
544         return returnval;
545     }

628     @Override
629     public CertificateWrapper getCertificate(AuthenticationToken authenticationToken, String certSNinHex, String issuerDN)
630             throws AuthorizationDeniedException, CADoesntExistsException, EjbcaException {
631         final String bcString = CertTools.stringToBCDNString(issuerDN);
632         final int caId = bcString.hashCode();
633         caSession.verifyExistenceOfCA(caId);
634         final String[] rules = {StandardRules.CAFUNCTIONALITY.resource()+"/view_certificate", StandardRules.CAACCESS.resource() + caId};
635         if(!authorizationSession.isAuthorizedNoLogging(authenticationToken, rules)) {
636             final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", Arrays.toString(rules), null);
637             throw new AuthorizationDeniedException(msg);
638         }
639         final Certificate result = certificateStoreSession.findCertificateByIssuerAndSerno(issuerDN, new BigInteger(certSNinHex,16));
640         if (log.isDebugEnabled()) {
641             log.debug("Found certificate for issuer '" + issuerDN + "' and SN " + certSNinHex + " for admin " + authenticationToken.getUniqueId());
642         }
643         return EJBTools.wrap(result);
644     }

646     @Override
647     public Collection<CertificateWrapper> findCertificatesByUsername(final AuthenticationToken authenticationToken, final String username, final boolean onlyValid, final long now)
648             throws AuthorizationDeniedException, CertificateEncodingException {
649         if (log.isDebugEnabled()) {
650             log.debug( "Find certificates by username requested by " + authenticationToken.getUniqueId());
651         }
652         // Check authorization on current CA and profiles and view_end_entity by looking up the end entity.
653         if (findUser(authenticationToken, username) == null) {
654             if (log.isDebugEnabled()) {
655                 log.debug(intres.getLocalizedMessage("ra.errorentitynotexist", username));
656             }
657         }
658         // Even if there is no end entity, it might be the case that we don't store UserData, so we still need to check CertificateData.
659         Collection<CertificateWrapper> searchResults;
660         if (onlyValid) {
661             // We will filter out not yet valid certificates later on, but we as the database to not return any expired certificates
662             searchResults = EJBTools.wrapCertCollection(certificateStoreSession.findCertificatesByUsernameAndStatusAfterExpireDate(username, CertificateConstants.CERT_ACTIVE, now));
663         } else {
664             searchResults = certificateStoreSession.findCertificatesByUsername(username);
665         }
666         // Assume the user may have certificates from more than one CA.
667         Certificate certificate = null;
668         int caId = -1;
669         Boolean authorized = null;
670         final Map<Integer, Boolean> authorizationCache = new HashMap<>();
671         final List<CertificateWrapper> result = new ArrayList<>();
672         for (Object searchResult: searchResults) {
673             certificate = ((CertificateWrapper) searchResult).getCertificate();
674             caId = CertTools.getIssuerDN(certificate).hashCode();
675             authorized = authorizationCache.get(caId);
676             if (authorized == null) {
677                 authorized = authorizationSession.isAuthorizedNoLogging(authenticationToken, StandardRules.CAACCESS.resource() + caId);
678                 authorizationCache.put(caId, authorized);
679             }
680             if (authorized.booleanValue()) {
681                 result.add((CertificateWrapper) searchResult);
682             }
683         }
684         if (log.isDebugEnabled()) {
685             log.debug( "Found " + result.size() + " certificate(s) by username requested by " + authenticationToken.getUniqueId());
686         }
687         return result;
688     }
