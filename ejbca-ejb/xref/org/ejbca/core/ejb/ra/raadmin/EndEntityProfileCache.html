13  package org.ejbca.core.ejb.ra.raadmin;
14  
15  import java.util.HashMap;
16  import java.util.List;
17  import java.util.Map;
18  import java.util.concurrent.locks.ReentrantLock;
19  
20  import javax.persistence.EntityManager;
21  
22  import org.apache.log4j.Logger;
23  import org.cesecore.certificates.endentity.EndEntityConstants;
24  import org.ejbca.config.EjbcaConfiguration;
25  import org.ejbca.core.model.InternalEjbcaResources;
26  import org.ejbca.core.model.ra.raadmin.EndEntityProfile;
27  
28  /**
29   * Class Holding cache variable. Needed because EJB spec does not allow volatile, non-final fields
30   * in session beans.
31   * 
32   * This cache is designed so only one thread at the time will update the cache if it is too old. Other
33   * threads will happily return a bit too old object. If a cache update is forced, for example when
34   * a profile is edited, it will always update the cache even if the commit of the transaction fails.
35   * 
36   * Another known issue during forced updates is the race condition exists, so an update in progress
37   * might overwrite the result from forced update's database query.
38   * 
39   * The intention of this design is better throughput than fully ordered sequential updates.
40   * 
41   * @version $Id: EndEntityProfileCache.java 27816 2018-01-09 16:19:36Z samuellb $
42   */
43  public enum EndEntityProfileCache {
44      INSTANCE;
45  
46      private final Logger LOG = Logger.getLogger(EndEntityProfileCache.class);
47      /** Internal localization of logs and errors */
48      private final InternalEjbcaResources INTRES = InternalEjbcaResources.getInstance();
49  
50      /*
51       * Cache of profiles, with Id as keys. This cache may be
52       * unsynchronized between multiple instances of EJBCA, but is common to all
53       * threads in the same VM. Set volatile to make it thread friendly.
54       */
55  
56      /** Cache of mappings between profileId and profileName */
57      private volatile Map<Integer, String> idNameMapCache = null;
58      /** Cache of mappings between profileName and profileId */
59      private volatile Map<String, Integer> nameIdMapCache = null;
60      /** Cache of end entity profiles, with Id as keys */
61      private volatile Map<Integer, EndEntityProfile> profileCache = null;
62      
63      private volatile long lastUpdate = 0;
64  
65      /* Create template maps with all static constants */
66      private final HashMap<Integer,String> idNameMapCacheTemplate = new HashMap<Integer,String>();
67      private final HashMap<String,Integer> nameIdMapCacheTemplate = new HashMap<String,Integer>();
68  
69      private final ReentrantLock lock = new ReentrantLock(false);
70      
71      private EndEntityProfileCache() {
72      	idNameMapCacheTemplate.put(Integer.valueOf(EndEntityConstants.EMPTY_END_ENTITY_PROFILE), EndEntityConstants.EMPTY_ENDENTITYPROFILENAME);
73      	nameIdMapCacheTemplate.put(EndEntityConstants.EMPTY_ENDENTITYPROFILENAME, Integer.valueOf(EndEntityConstants.EMPTY_END_ENTITY_PROFILE));
74      }
75  
76      /**
77       * Fetch all profiles from the database, unless cache is enabled, valid and we do not force an update.
78       * @param entityManager is required for reading the profiles from the database if we need to update the cache
79       * @param force if true, this will force an update even if the cache is not yet invalid
80       */
81      public void updateProfileCache(final EntityManager entityManager, final boolean force) {
82          if (LOG.isTraceEnabled()) {
83              LOG.trace(">updateProfileCache: "+force);
84          }
85          final long cacheEndEntityProfileTime = EjbcaConfiguration.getCacheEndEntityProfileTime();
86          final long now = System.currentTimeMillis();
87          // Check before acquiring lock
88          if (!force && cacheEndEntityProfileTime!=0 && lastUpdate+cacheEndEntityProfileTime > now) {
89              if (LOG.isTraceEnabled()) {
90                  LOG.trace("<updateProfileCache returning, cache time not expired: "+cacheEndEntityProfileTime);
91              }
92              return; // We don't need to update cache
93          }
94          try {
95          	lock.lock();
96          	if (!force && cacheEndEntityProfileTime!=0 && lastUpdate+cacheEndEntityProfileTime > now) {
97          		return;	// We don't need to update cache
98          	}
99          	lastUpdate = now; // make sure next thread does not also pass the update test
100         } finally {
101         	lock.unlock();
102         }
103         final Map<Integer, String> idNameCache = new HashMap<Integer, String>(idNameMapCacheTemplate);
104         final Map<String, Integer> nameIdCache = new HashMap<String, Integer>(nameIdMapCacheTemplate);
105         final Map<Integer, EndEntityProfile> profCache = new HashMap<Integer, EndEntityProfile>();
106         try {
107         	final List<EndEntityProfileData> result = EndEntityProfileData.findAll(entityManager);
108         	for (final EndEntityProfileData next : result) {
109         		final Integer id = Integer.valueOf(next.getId());
110         		final String profileName = next.getProfileName();
111         		idNameCache.put(id, profileName);
112         		nameIdCache.put(profileName, id);
113         		profCache.put(id, next.getProfile());
114         	}
115         } catch (Exception e) {
116         	LOG.error(INTRES.getLocalizedMessage("ra.errorreadprofiles"), e);
117         }
118         idNameMapCache = idNameCache;
119         nameIdMapCache = nameIdCache;
120         profileCache = profCache;
121         if (LOG.isTraceEnabled()) {
122             final long end = System.currentTimeMillis();
123             LOG.trace("<updateProfileCache took: "+(end-now)+"ms");
124         }
125 	}
126 
127 	/** @param entityManager EM
128 	 * @return the latest object from the cache or a current database representation if no caching is used. */
129 	public Map<Integer, EndEntityProfile> getProfileCache(final EntityManager entityManager) {
130 		updateProfileCache(entityManager, false);
131 		return profileCache;
132 	}
133 
134 	/** @param entityManager EM
135 	 * @return the latest object from the cache or a current database representation if no caching is used. */
136 	public Map<Integer, String> getIdNameMapCache(final EntityManager entityManager) {
137 		updateProfileCache(entityManager, false);
138 		return idNameMapCache;
139 	}
140 
141 	/** @param entityManager EM
142 	 * @return the latest object from the cache or a current database representation if no caching is used. */
143 	public Map<String, Integer> getNameIdMapCache(final EntityManager entityManager) {
144 		updateProfileCache(entityManager, false);
145 		return nameIdMapCache;
146 	}
147 }
