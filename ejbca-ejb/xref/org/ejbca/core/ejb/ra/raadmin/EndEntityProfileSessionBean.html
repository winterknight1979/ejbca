94      @Override
95      public void addEndEntityProfile(final AuthenticationToken admin, final int profileid, final String profilename, final EndEntityProfile profile) throws AuthorizationDeniedException, EndEntityProfileExistsException {
96          if (profilename.trim().equalsIgnoreCase(EndEntityConstants.EMPTY_ENDENTITYPROFILENAME)) {
97              final String msg = INTRES.getLocalizedMessage("ra.erroraddprofilefixed", profilename, EndEntityConstants.EMPTY_ENDENTITYPROFILENAME);
98              LOG.info(msg);
99              throw new EndEntityProfileExistsException(msg);
100         } else if (!isFreeEndEntityProfileId(profileid)) {
101         	final String msg = INTRES.getLocalizedMessage("ra.erroraddprofileexists", profilename);
102             LOG.info(msg);
103             throw new EndEntityProfileExistsException(msg);
104         } else if (EndEntityProfileData.findByProfileName(entityManager, profilename) != null) {
105         	final String msg = INTRES.getLocalizedMessage("ra.erroraddprofileexists", profilename);
106         	LOG.info(msg);
107             throw new EndEntityProfileExistsException(msg);
108         } else {
109             // Check authorization before adding
110             authorizedToEditProfile(admin, profile);
111             try {
112                 entityManager.persist(new EndEntityProfileData(Integer.valueOf(profileid), profilename, profile));
113                 flushProfileCache();
114                 final String msg = INTRES.getLocalizedMessage("ra.addedprofile", profilename);
115                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
116                 details.put("msg", msg);
117                 auditSession.log(EjbcaEventTypes.RA_ADDEEPROFILE, EventStatus.SUCCESS, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
118             } catch (Exception e) {
119             	final String msg = INTRES.getLocalizedMessage("ra.erroraddprofile", profilename);
120                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
121                 details.put("msg", msg);
122                 details.put("error", e.getMessage());
123                 auditSession.log(EjbcaEventTypes.RA_ADDEEPROFILE, EventStatus.FAILURE, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
124             }
125         }
126     }

134     @Override
135     public void cloneEndEntityProfile(final AuthenticationToken admin, final String orgname, final String newname) throws AuthorizationDeniedException, EndEntityProfileExistsException {
136         if (newname.trim().equalsIgnoreCase(EndEntityConstants.EMPTY_ENDENTITYPROFILENAME)) {
137         	final String msg = INTRES.getLocalizedMessage("ra.errorcloneprofile", newname, orgname);
138         	LOG.info(msg);
139             throw new EndEntityProfileExistsException();
140         }
141         if (EndEntityProfileData.findByProfileName(entityManager, newname) == null) {
142         	final EndEntityProfileData pdl = EndEntityProfileData.findByProfileName(entityManager, orgname);
143             boolean success = false;
144             if (pdl != null) {
145                 // Check authorization before cloning
146                 final EndEntityProfile profile = pdl.getProfile();
147                 authorizedToEditProfile(admin, profile);
148             	try {
149             		final int profileid = findFreeEndEntityProfileId();
150             		entityManager.persist(new EndEntityProfileData(profileid, newname, (EndEntityProfile)profile.clone()));
151             		flushProfileCache();
152             		final String msg = INTRES.getLocalizedMessage("ra.clonedprofile", newname, orgname);
153                     final Map<String, Object> details = new LinkedHashMap<String, Object>();
154                     details.put("msg", msg);
155                     auditSession.log(EjbcaEventTypes.RA_ADDEEPROFILE, EventStatus.SUCCESS, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
156             		success = true;
157             	} catch (CloneNotSupportedException e) {
158             		LOG.error("Cloe not supported?: ", e);
159             	}
160             }
161             if (!success) {
162             	final String msg = INTRES.getLocalizedMessage("ra.errorcloneprofile", newname, orgname);
163                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
164                 details.put("msg", msg);
165                 auditSession.log(EjbcaEventTypes.RA_ADDEEPROFILE, EventStatus.FAILURE, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
166             }
167         } else {
168         	final String msg = INTRES.getLocalizedMessage("ra.errorcloneprofile", newname, orgname);
169         	LOG.info(msg);
170             throw new EndEntityProfileExistsException();
171         }
172     }

191     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
192     @Override
193     public boolean existsCAInEndEntityProfiles(final int caid) {
194         String[] availablecas = null;
195         boolean exists = false;
196         final Collection<EndEntityProfileData> result = EndEntityProfileData.findAll(entityManager);
197         final Iterator<EndEntityProfileData> i = result.iterator();
198         while (i.hasNext() && !exists) {
199         	final EndEntityProfileData ep = i.next();
200             availablecas = ep.getProfile().getValue(EndEntityProfile.AVAILCAS, 0).split(EndEntityProfile.SPLITCHAR);
201             for (int j = 0; j < availablecas.length; j++) {
202                 if (StringUtils.isNotEmpty(availablecas[j])) {
203                     if (Integer.parseInt(availablecas[j]) == caid) {
204                         exists = true;
205                         if (LOG.isDebugEnabled()) {
206                             LOG.debug("CA exists in entity profile " + ep.getProfileName());
207                         }
208                         break;
209                     }
210                 } else if (LOG.isDebugEnabled()) {
211                     LOG.debug("One of the availableCAs is empty string, fishy, but we ignore it. EE profile: "+ep.getProfileName());
212                 }
213             }
214         }
215         return exists;
216     }

283     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
284     @Override
285     public Collection<Integer> getAuthorizedEndEntityProfileIds(final AuthenticationToken admin, String endentityAccessRule) {
286     	final ArrayList<Integer> returnval = new ArrayList<Integer>();
287     	final HashSet<Integer> authorizedcaids = new HashSet<Integer>(caSession.getAuthorizedCaIds(admin));
288     	final HashSet<Integer> allcaids = new HashSet<Integer>(caSession.getAllCaIds());
289 		// If this is the special value ALLCAs we are authorized
290     	authorizedcaids.add(Integer.valueOf(SecConst.ALLCAS));
291 
292     	final boolean rootAccess = authorizationSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource());
293         // We have to manually add the EMPTY end entity profile because it is not included in the profile cache
294         if (authorizationSession.isAuthorizedNoLogging(admin, AccessRulesConstants.ENDENTITYPROFILEBASE + "/" + EndEntityConstants.EMPTY_END_ENTITY_PROFILE + endentityAccessRule)) {
295             returnval.add(EndEntityConstants.EMPTY_END_ENTITY_PROFILE);
296         }
297         for (final Entry<Integer, EndEntityProfile> entry : EndEntityProfileCache.INSTANCE.getProfileCache(entityManager).entrySet()) {
298             // Check if all profiles available CAs exists in authorizedcaids.
299             final String availableCasString = entry.getValue().getValue(EndEntityProfile.AVAILCAS, 0);
300             boolean authorizedToProfile = false;
301             // Check authorization for the endentityAccessRule here. The built in EMPTY EE profile is obviously not included in the cache, so added manually above
302             if (authorizationSession.isAuthorizedNoLogging(admin, AccessRulesConstants.ENDENTITYPROFILEBASE + "/" + entry.getKey().toString() + endentityAccessRule)) {
303                 authorizedToProfile = true;
304                 if (StringUtils.isNotBlank(availableCasString)) {
305                     for (final String caidString : availableCasString.split(EndEntityProfile.SPLITCHAR)) {
306                         try {
307                             final int caIdInt = Integer.parseInt(caidString);
308                             // with root rule access you can edit profiles with missing CA ids
309                             if (!authorizedcaids.contains(caIdInt) && (!rootAccess || allcaids.contains(caIdInt))) {
310                                 authorizedToProfile = false;
311                                 if (LOG.isDebugEnabled()) {
312                                     LOG.debug("Profile " + entry.getKey().toString() + " not authorized to CA with ID " + caIdInt);
313                                 }
314                                 break;
315                             }
316                         } catch (NumberFormatException e) {
317                             throw new IllegalStateException(
318                                     "CA ID was stored in an end entity profile as something other than a number: " + caidString, e);
319                         }
320                     }
321                     if (authorizedToProfile) {
322                         returnval.add(entry.getKey());
323                     }
324                 }
325             }
326 
327         }
328         return returnval;
329     }

331     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
332     @Override
333     public List<Integer> getAuthorizedEndEntityProfileIdsWithMissingCAs(final AuthenticationToken admin) {
334         final ArrayList<Integer> returnval = new ArrayList<Integer>();
335         final HashSet<Integer> allcaids = new HashSet<Integer>(caSession.getAllCaIds());
336         allcaids.add(Integer.valueOf(SecConst.ALLCAS));
337         if (!authorizationSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource())) {
338             // we can only see profiles with missing CA Ids if we have root rule access
339             return returnval;
340         }
341 
342         try {
343             for (final Entry<Integer, EndEntityProfile> entry : EndEntityProfileCache.INSTANCE.getProfileCache(entityManager).entrySet()) {
344                 final String availableCasString = entry.getValue().getValue(EndEntityProfile.AVAILCAS, 0);
345                 if (availableCasString != null) {
346                     boolean nonExistingCA = false;
347                     for (final String caidString : availableCasString.split(EndEntityProfile.SPLITCHAR)) {
348                         final int caIdInt = Integer.parseInt(caidString);
349                         if (!allcaids.contains(caIdInt)) {
350                             nonExistingCA = true;
351                         }
352                     }
353                     if (nonExistingCA) {
354                         returnval.add(entry.getKey());
355                     }
356                 }
357             }
358         } catch (Exception e) {
359             LOG.error(INTRES.getLocalizedMessage("ra.errorgetids"), e);
360         }
361         return returnval;
362     }

503     @Override
504     public void internalChangeEndEntityProfileNoFlushCache(final AuthenticationToken admin, final String profilename, final EndEntityProfile profile)
505             throws AuthorizationDeniedException, EndEntityProfileNotFoundException {
506 	final EndEntityProfileData pdl = EndEntityProfileData.findByProfileName(entityManager, profilename);
507         if (pdl == null) {
508         	final String msg = INTRES.getLocalizedMessage("ra.errorchangeprofile", profilename);
509         	LOG.info(msg);
510         	throw new EndEntityProfileNotFoundException("End entity profile of name \"" + profilename + "\" not found.");
511         } else {
512             // Check authorization before editing
513             authorizedToEditProfile(admin, profile);
514             // Get the diff of what changed
515             Map<Object, Object> diff = pdl.getProfile().diff(profile);
516             pdl.setProfile(profile);
517             final String msg = INTRES.getLocalizedMessage("ra.changedprofile", profilename);
518             final Map<String, Object> details = new LinkedHashMap<String, Object>();
519             details.put("msg", msg);
520         	// Log diff
521             for (Map.Entry<Object, Object> entry : diff.entrySet()) {
522                 details.put(entry.getKey().toString(), entry.getValue().toString());
523             }
524             auditSession.log(EjbcaEventTypes.RA_EDITEEPROFILE, EventStatus.SUCCESS, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(),
525                     null, null, null, details);
526         }

529     @Override
530     public void removeEndEntityProfile(final AuthenticationToken admin, final String profilename) throws AuthorizationDeniedException {
531     	final EndEntityProfileData pdl = EndEntityProfileData.findByProfileName(entityManager, profilename);
532     		if (pdl == null) {
533     			if (LOG.isDebugEnabled()) {
534     				LOG.debug("Trying to remove an end entity profile that does not exist: "+profilename);
535     			}
536     		} else {
537     		    // Check authorization before removing
538                 authorizedToEditProfile(admin, pdl.getProfile());
539     		    try {
540     		        entityManager.remove(pdl);
541     		        flushProfileCache();
542     		        final String msg = INTRES.getLocalizedMessage("ra.removedprofile", profilename);
543     		        final Map<String, Object> details = new LinkedHashMap<String, Object>();
544     		        details.put("msg", msg);
545     		        auditSession.log(EjbcaEventTypes.RA_REMOVEEEPROFILE, EventStatus.SUCCESS, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
546     		    } catch (Exception e) {
547     		        LOG.error("Error was caught when trying to remove end entity profile " + profilename, e);
548     		        final String msg = INTRES.getLocalizedMessage("ra.errorremoveprofile", profilename);
549     		        final Map<String, Object> details = new LinkedHashMap<String, Object>();
550     		        details.put("msg", msg);
551     		        details.put("error", e.getMessage());
552     		        auditSession.log(EjbcaEventTypes.RA_REMOVEEEPROFILE, EventStatus.FAILURE, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
553     		    }
554     		}

557     @Override
558     public void renameEndEntityProfile(final AuthenticationToken admin, final String oldprofilename, final String newprofilename) throws AuthorizationDeniedException, EndEntityProfileExistsException {
559         if (newprofilename.trim().equalsIgnoreCase(EndEntityConstants.EMPTY_ENDENTITYPROFILENAME) || oldprofilename.trim().equalsIgnoreCase(EndEntityConstants.EMPTY_ENDENTITYPROFILENAME)) {
560         	final String msg = INTRES.getLocalizedMessage("ra.errorrenameprofile", oldprofilename, newprofilename);
561         	LOG.info(msg);
562             throw new EndEntityProfileExistsException();
563         }
564         if (EndEntityProfileData.findByProfileName(entityManager, newprofilename) == null) {
565         	final EndEntityProfileData pdl = EndEntityProfileData.findByProfileName(entityManager, oldprofilename);
566             if (pdl == null) {
567             	final String msg = INTRES.getLocalizedMessage("ra.errorrenameprofile", oldprofilename, newprofilename);
568             	LOG.info(msg);
569             } else {
570                 // Check authorization before renaming
571                 authorizedToEditProfile(admin, pdl.getProfile());
572                 pdl.setProfileName(newprofilename);
573                 flushProfileCache();
574                 final String msg = INTRES.getLocalizedMessage("ra.renamedprofile", oldprofilename, newprofilename);
575                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
576                 details.put("msg", msg);
577                 auditSession.log(EjbcaEventTypes.RA_RENAMEEEPROFILE, EventStatus.SUCCESS, EjbcaModuleTypes.RA, EjbcaServiceTypes.EJBCA, admin.toString(), null, null, null, details);
578             }
579         } else {
580         	final String msg = INTRES.getLocalizedMessage("ra.errorrenameprofile", oldprofilename, newprofilename);
581         	LOG.info(msg);
582             throw new EndEntityProfileExistsException();
583         }
584     }

586     @Override
587     public void authorizedToProfileCas(final AuthenticationToken admin, final EndEntityProfile profile) throws AuthorizationDeniedException {
588         if (profile == null) {
589             return;
590         }
591         final HashSet<Integer> authorizedcaids = new HashSet<Integer>(caSession.getAuthorizedCaIds(admin));
592         final String availablecasstring = profile.getValue(EndEntityProfile.AVAILCAS, 0);
593         if (StringUtils.isNotEmpty(availablecasstring)) {
594             /*
595              * Go through all available CAs in the profile and check
596              * that the administrator is authorized to all CAs specified
597              * in the profile If ALLCAS is selected in the end entity
598              * profile, we must check that the administrator is
599              * authorized to all CAs in the system.
600              */
601             String[] availablecas = profile.getValue(EndEntityProfile.AVAILCAS, 0).split(EndEntityProfile.SPLITCHAR);
602             /*
603              * If availablecas contains SecConst ALLCAS, change
604              * availablecas /to be a list of all CAs
605              */
606             if (ArrayUtils.contains(availablecas, String.valueOf(SecConst.ALLCAS))) {
607                 Collection<Integer> allcaids = caSession.getAllCaIds();
608                 if (LOG.isDebugEnabled()) {
609                     LOG.debug("Available CAs in end entity profile contains ALLCAS, lising all CAs in the system instead. There are "
610                             + allcaids.size() + " CAs in the system");
611                 }
612                 availablecas = new String[allcaids.size()];
613                 int index = 0;
614                 for (Integer id : allcaids) {
615                     availablecas[index++] = id.toString();
616                 }
617             }
618             for (int j = 0; j < availablecas.length; j++) {
619                 Integer caid = Integer.valueOf(availablecas[j]);
620                 if (!authorizedcaids.contains(caid)) {
621                     // We want to allow removal of the EE profile if the CA does not exists, it can happen that a rogue CAId
622                     // sneaks in under "availableCAs" in the profile. So make a double check here, and let it pass if the CA does not exist
623                     try {
624                         caSession.verifyExistenceOfCA(caid);
625                         final String msg = INTRES.getLocalizedMessage("caadmin.notauthorizedtoca", admin.toString(), caid);
626                         throw new AuthorizationDeniedException(msg);
627                     } catch (CADoesntExistsException e) {
628                         LOG.info("Admin was not authorized to CA "+caid+", but this CA does not even exist so we allow it.");
629                     }
630                 }
631             }
632         }
633     }

635     @Override
636     public byte[] getProfileAsXml(final AuthenticationToken authenticationToken, final int profileId)
637             throws EndEntityProfileNotFoundException, AuthorizationDeniedException {
638         EndEntityProfile profile = null;
639         profile = getEndEntityProfileNoClone(profileId);
640         if (profile == null) {
641             throw new EndEntityProfileNotFoundException("Could not find end entity profile with ID '" + profileId + "' in the database.");
642         }
643         authorizedToViewProfile(authenticationToken, profile);
644         try (ByteArrayOutputStream baos = new ByteArrayOutputStream(); XMLEncoder encoder = new XMLEncoder(baos)) {
645             encoder.writeObject(profile.saveData());
646             encoder.close();
647             return baos.toByteArray();
648         } catch (IOException e) {
649             String msg = "Could not encode profile with ID " + profileId + " to XML: " + e.getMessage();
650             LOG.debug(msg, e);
651             throw new IllegalStateException(msg, e);
652         }
653     }
