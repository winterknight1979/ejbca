14  package org.ejbca.core.ejb.ra.raadmin;
15  
16  import java.util.HashMap;
17  import java.util.List;
18  import java.util.Map;
19  
20  import javax.annotation.PostConstruct;
21  import javax.ejb.ConcurrencyManagement;
22  import javax.ejb.ConcurrencyManagementType;
23  import javax.ejb.EJB;
24  import javax.ejb.Singleton;
25  import javax.ejb.Startup;
26  import javax.ejb.TransactionManagement;
27  import javax.ejb.TransactionManagementType;
28  
29  import org.apache.log4j.Logger;
30  import org.cesecore.authentication.tokens.AuthenticationToken;
31  import org.cesecore.config.RaStyleInfo;
32  import org.ejbca.core.model.era.RaMasterApiProxyBeanLocal;
33  
34  /**
35   * Global cache of RaStyleInfos fetched via Peers. Each entry is mapped to an administrator, containing its available custom RA Styles (CSS files and logo).
36   * This cache prevents multiple requests over the Peers protocol when an administrator loads the RA page and style resources are requested by the servlet filter
37   * (RaStyleRequestFilter). Since style archives may be very heavy to transfer, a check for changes is sent via Peers before reloading the styles.
38   * 
39   * @version $Id: RaStyleCacheBean.java 27422 2017-12-05 14:05:42Z bastianf $
40   */
41  @Singleton
42  @Startup
43  @ConcurrencyManagement(ConcurrencyManagementType.BEAN)
44  @TransactionManagement(TransactionManagementType.BEAN)
45  public class RaStyleCacheBean {
46  
47      @EJB
48      private RaMasterApiProxyBeanLocal raMasterApiProxyBean;
49      
50      private final Logger log = Logger.getLogger(RaStyleCacheBean.class);
51      
52      private Map<AuthenticationToken, List<RaStyleInfo>> raStyleCache;
53      private Map<AuthenticationToken, Long> lastUpdateMap;
54      
55      @PostConstruct
56      public void initialize() {
57          raStyleCache = new HashMap<>();
58          lastUpdateMap = new HashMap<>();
59          if (log.isDebugEnabled()) {
60              log.info(this.getClass().getName() + " initialized");
61          }
62      }
63      
64      public RaStyleCacheBean() {}
65      
66      public void invalidateCache() {
67          raStyleCache = new HashMap<>();
68          lastUpdateMap = new HashMap<>();
69      }
70      
71      /**
72       * Checks if administrators actual available RA Styles differentiate from the cached styles
73       * @param authenticationToken of the requesting administrator
74       * @return true if cache is invalid and requires update
75       */
76      public boolean needsUpdate(AuthenticationToken authenticationToken) {
77          List<RaStyleInfo> cachedStyle = raStyleCache.get(authenticationToken);
78          if (cachedStyle != null) {
79              return raMasterApiProxyBean.getAvailableCustomRaStyles(authenticationToken, cachedStyle.hashCode()) != null;
80          }
81          return true;
82      }
83      
84      /**
85       * Returns custom RA Styles associated with the role of the requesting administrator. If the administrator has requested
86       * styles very recently, the same content will be returned to prevent requests via Peers for for every resource request
87       * e.g. in a page load in the RA web.
88       * @param authenticationToken of the requesting administrator
89       * @return List of custom RA styles available for the requesting administrator
90       */
91      public List<RaStyleInfo> getAvailableRaStyles(AuthenticationToken authenticationToken) {
92          List<RaStyleInfo> availableRaStyles;
93          long now = System.currentTimeMillis();
94          if (raStyleCache.containsKey(authenticationToken)) {
95              List<RaStyleInfo> cachedStyles = raStyleCache.get(authenticationToken);
96              // This prevents lookup over Peers for every requested resource in a page load.
97              if (now - getLastUpdate(authenticationToken) < 1000) {
98                  return cachedStyles;
99              }
100             // Check for changes
101             availableRaStyles = raMasterApiProxyBean.getAvailableCustomRaStyles(authenticationToken, cachedStyles.hashCode());
102             if (availableRaStyles == null) {
103                 // No changes, use cached styles
104                 availableRaStyles = cachedStyles;
105             } else {
106                 raStyleCache.put(authenticationToken, availableRaStyles);
107             }
108         } else {
109             // Full reload of styles
110             availableRaStyles = raMasterApiProxyBean.getAvailableCustomRaStyles(authenticationToken, 0);
111             raStyleCache.put(authenticationToken, availableRaStyles);
112         }
113         lastUpdateMap.put(authenticationToken, now);
114         return availableRaStyles;
115     }
116     
117     /** @return returns Map of all cached RA Styles*/
118     public Map<AuthenticationToken, List<RaStyleInfo>> getCachedAdminStylesMap() {
119         return raStyleCache;
120     }
121     
122     /** @param authenticationToken Token
123      * @return last time the administrator requested styles from this cache */
124     public long getLastUpdate(AuthenticationToken authenticationToken) {
125         Long lastUpdate = lastUpdateMap.get(authenticationToken);
126         return lastUpdate == null ? -1 : lastUpdate;
127     }
128 }
