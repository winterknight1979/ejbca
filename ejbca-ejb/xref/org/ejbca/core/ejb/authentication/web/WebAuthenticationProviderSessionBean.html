50  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "WebAuthenticationProviderSessionLocal")
51  @TransactionAttribute(TransactionAttributeType.SUPPORTS)
52  public class WebAuthenticationProviderSessionBean implements WebAuthenticationProviderSessionLocal {
53  
54      private static final long serialVersionUID = 1524951666783567785L;
55  
56      private final static Logger LOG = Logger.getLogger(WebAuthenticationProviderSessionBean.class);
57      /** Internal localization of logs and errors */
58      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
59  
60      @EJB
61      private CertificateStoreSessionLocal certificateStoreSession;
62      @EJB
63      private SecurityEventsLoggerSessionLocal securityEventsLoggerSession;
64  
65      @Override
66      public X509CertificateAuthenticationToken authenticateUsingClientCertificate(final X509Certificate x509Certificate) {
67          return (X509CertificateAuthenticationToken) authenticate(new AuthenticationSubject(null, new HashSet<X509Certificate>( Arrays.asList(new X509Certificate[]{ x509Certificate }))));
68      }
69  
70      @Override
71      public PublicAccessAuthenticationToken authenticateUsingNothing(final String principal, final boolean confidentialTransport) {
72          return new PublicAccessAuthenticationToken(principal, confidentialTransport);
73      }
74  
75      /**
76       * Performs client certificate authentication for a subject. This requires:
77       * - An AuthenticationSubject containing a Set&lt;X509Certificate&gt;, where there should be only one certificate 
78       *   being the administrators client certificate.
79       * If the admin certificate is required to be in the database (properties configuration option) it is
80       * verified that the certificate is present in the database and that it is not revoked.
81       * 
82       * @param subject an AuthenticationSubject containing a Set&lt;X509Certificate&gt; of credentials, the set must contain one certificate which is the admin client certificate.
83       * @return an AuthenticationToken if the subject was authenticated, null otherwise.
84       */
85      @Override
86      public AuthenticationToken authenticate(AuthenticationSubject subject) {
87          @SuppressWarnings("unchecked")
88          final Set<X509Certificate> certs = (Set<X509Certificate>) subject.getCredentials();
89          if (certs.size() != 1) {
90              if (LOG.isDebugEnabled()) {
91                  LOG.debug("certificateArray contains "+certs.size()+" certificates, instead of 1 that is required.");
92              }
93              return null;
94          } else {
95              final X509Certificate certificate = certs.iterator().next();
96              // Check Validity
97              try {
98                  certificate.checkValidity();
99              } catch (Exception e) {
100                 final String msg = intres.getLocalizedMessage("authentication.certexpired", CertTools.getSubjectDN(certificate), CertTools.getNotAfter(certificate).toString());
101             	LOG.info(msg);
102                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
103                 details.put("msg", msg);
104                 securityEventsLoggerSession.log(EventTypes.AUTHENTICATION, EventStatus.FAILURE, EjbcaModuleTypes.ADMINWEB, EjbcaServiceTypes.EJBCA, LogConstants.NO_AUTHENTICATION_TOKEN, null, null, null, details);
105             	return null;
106             }
107             // Find out if this is a certificate present in the local database (even if we don't require a cert to be present there we still want to allow a mix)
108             // Database integrity protection verification not performed running this query
109             final int status = certificateStoreSession.getFirstStatusByIssuerAndSerno(CertTools.getIssuerDN(certificate), CertTools.getSerialNumber(certificate));
110             if (status != -1) {
111                 // The certificate is present in the database.
112                 if (!(status == CertificateConstants.CERT_ACTIVE || status == CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION)) {
113                     // The certificate is neither active, nor active (but user is notified of coming revocation)
114                     final String msg = intres.getLocalizedMessage("authentication.revokedormissing", CertTools.getSubjectDN(certificate));
115                     LOG.info(msg);
116                     final Map<String, Object> details = new LinkedHashMap<String, Object>();
117                     details.put("msg", msg);
118                     securityEventsLoggerSession.log(EventTypes.AUTHENTICATION, EventStatus.FAILURE, EjbcaModuleTypes.ADMINWEB, EjbcaServiceTypes.EJBCA, LogConstants.NO_AUTHENTICATION_TOKEN, null, null, null, details);
119                     return null;
120                 }
121             } else {
122                 // The certificate is not present in the database.
123                 if (WebConfiguration.getRequireAdminCertificateInDatabase()) {
124                     final String msg =  intres.getLocalizedMessage("authentication.revokedormissing", CertTools.getSubjectDN(certificate));
125                     LOG.info(msg);
126                     final Map<String, Object> details = new LinkedHashMap<String, Object>();
127                     details.put("msg", msg);
128                     securityEventsLoggerSession.log(EventTypes.AUTHENTICATION, EventStatus.FAILURE, EjbcaModuleTypes.ADMINWEB, EjbcaServiceTypes.EJBCA, LogConstants.NO_AUTHENTICATION_TOKEN, null, null, null, details);
129                     return null;
130                 }
131                 // TODO: We should check the certificate for CRL or OCSP tags and verify the certificate status
132             }
133             return new X509CertificateAuthenticationToken(certificate);
134         }
135     }
