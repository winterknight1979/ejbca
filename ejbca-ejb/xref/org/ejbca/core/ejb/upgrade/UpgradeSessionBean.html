307     @Override
308     public void performPreUpgrade(final boolean isFreshInstallation) {
309         try {
310             if (isFreshInstallation) {
311                 // Unlock statedump in new installations
312                 final GlobalConfiguration globalConfig = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
313                 globalConfig.setStatedumpLockedDown(false);
314                 globalConfigurationSession.saveConfiguration(authenticationToken, globalConfig);
315                 setEndEntityProfileInCertificateData(true);
316                 // Since we know that this is a brand new installation, no upgrade should be needed
317                 setLastUpgradedToVersion(InternalConfiguration.getAppVersionNumber());
318                 setLastPostUpgradedToVersion("6.10.1");
319             } else {
320                 // Ensure that we save currently known oldest installation version before any upgrade is invoked
321                 if(getLastUpgradedToVersion() != null) {
322                     setLastUpgradedToVersion(getLastUpgradedToVersion());
323                 }
324             }
325         } catch (AuthorizationDeniedException e) {
326             throw new IllegalStateException("AlwaysAllowLocalAuthenticationToken should not have been denied authorization");
327         }
328     }

330     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
331     @Override
332     public boolean performUpgrade() {
333         final String dbType = DatabaseConfiguration.getDatabaseName();
334         final String currentVersion = InternalConfiguration.getAppVersionNumber();
335         String last = getLastUpgradedToVersion();
336         if (last==null) {
337             // Start auto-detection, since no version info was present
338             // This auto-detection was added for EJBCA 6.4.0
339             if (!checkColumnExists500()) {
340                 // The CAId column was removed during post upgrade to EJBCA 5.0
341                 last = "5.0";
342                 if (globalConfigurationSession.findByConfigurationId(GlobalOcspConfiguration.OCSP_CONFIGURATION_ID)!=null) {
343                     last = "6.2.4";
344                 }
345                 setLastUpgradedToVersion(last);
346                 if (!publisherSession.isOldVaPublisherPresent()) {
347                     // For all practical purposes, this version can used as post-upgrade version
348                     setLastPostUpgradedToVersion("6.4.0");
349                 }
350             } else {
351                 // We are on EJBCA 4.0 or 3.11 or even earlier
352                 log.error(
353                         "Post-upgrade from EJBCA prior to version 5.0.0 is forbidden. It is recommended that you upgrade to the intermediate release"
354                                 + " EJBCA 6.3.2.6 first. Read the EJBCA Upgrade Guide for more information.");
355                 return false;
356             }
357         }
358         boolean ret = true;
359         if (isLesserThan(last, currentVersion)) {
360             log.info("Database content version: " + last + ", current application version: " + currentVersion + " -> Starting upgrade.");
361             ret = upgradeSession.upgrade(dbType, last, false);
362         } else {
363             log.info("Database content version: " + last + ", current application version: " + currentVersion + " -> Upgrade is not needed.");
364         }
365         return ret;
366     }

368     @TransactionAttribute(TransactionAttributeType.NOT_SUPPORTED)
369     @Asynchronous
370     @Override
371     public Future<Boolean> startPostUpgrade() {
372         log.trace(">startPostUpgrade");
373         boolean ret = false;
374         if (setPostUpgradeStartedInternal(System.currentTimeMillis())) {
375             try {
376                 upgradeStatusSingleton.logAppenderAttach(log);
377                 if (upgradeStatusSingleton.setPostUpgradeInProgressIfDifferent(true)) {
378                     try {
379                         final String dbType = DatabaseConfiguration.getDatabaseName();
380                         final String currentVersion = InternalConfiguration.getAppVersionNumber();
381                         final String last = getLastPostUpgradedToVersion();
382                         if (isLesserThan(last, currentVersion)) {
383                             log.info("Database content version: " + last + ", current application version: " + currentVersion + " -> Starting post-upgrade.");
384                             ret = upgradeSession.upgrade(dbType, last, true);
385                         } else {
386                             log.info("Database content version: " + last + ", current application version: " + currentVersion + " -> Post-upgrade is not needed.");
387                             ret = true;
388                         }
389                     } finally {
390                         upgradeStatusSingleton.resetPostUpgradeInProgress();
391                     }
392                 } else {
393                     log.info("Preventing start of post-upgrade background tasks since it has already been started on this cluster node.");
394                 }
395             } catch (RuntimeException e) {
396                 // Since this is invoked asynchronously the calling client might no longer be around to receive the "result"
397                 log.error("Unexpected error from post-upgrade: " + e.getMessage(), e);
398             } finally {
399                 setPostUpgradeStartedInternal(0L);
400                 upgradeStatusSingleton.logAppenderDetach(log);
401             }
402         } else {
403             log.info("Preventing start of post-upgrade background tasks since it has already been started by a cluster node.");
404         }
405         log.trace("<startPostUpgrade");
406         return new AsyncResult<Boolean>(ret);
407     }

411     public boolean upgrade(String dbtype, String oldVersion, boolean isPost) {
412         try {
413             log.debug("Upgrading from version=" + oldVersion);
414             if (isPost) {
415                 // TODO: We might want to check that upgrade has run ok before allowing this.
416                 // ...on the other hand... we wont allow it via the GUI so it might be good to be able to force upgrade retries
417                 return postUpgrade(oldVersion, dbtype);
418             } else {
419                 return upgrade(dbtype, oldVersion);
420             }
421         } catch (RuntimeException e) {
422         	// We want to log in server.log so we can analyze the error
423             log.error("Error thrown during upgrade: ", e);
424             throw e;
425         } finally {
426             log.trace("<upgrade()");
427         }
428     }

434                     "Upgrading from EJBCA prior to version 5.0.0 is forbidden. You must upgrade to the intermediate release EJBCA 6.3.2.6 first. Read the EJBCA Upgrade Guide for more information.");
435             return false;

439             ocspResponseGeneratorSession.adhocUpgradeFromPre60(null);

512             } catch (UpgradeFailedException e) {

553             if (!postMigrateDatabase632()) {
554                 return false;

556             setLastPostUpgradedToVersion("6.3.2");
557         }

565             if (!postMigrateDatabase6101()) {
566                 return false;
567             }
568             setLastPostUpgradedToVersion("6.10.1");
569         }
570         // NOTE: If you add additional post upgrade tasks here, also modify isPostUpgradeNeeded() and performPreUpgrade()
571         //setLastPostUpgradedToVersion(InternalConfiguration.getAppVersionNumber());
572         return true;
573     }

606             final String ekuoid = "extendedkeyusage.oid.";
607             int j=0;

609                 final String oid = conf.getString(ekuoid+i);
610                 if (oid != null) {
611                     String name = conf.getString(ekuname+i);
612                     if (name != null) {

622                 }
623                 // No eku with a certain number == continue trying next, we will try 0-255.
624             }

627             }

1437     private void importUnidFnrTrustDir() {
1438         List<X509Certificate> trustedCerts = new ArrayList<>();
1439         Certificate cacert = null;
1440         boolean isUnidFnrEnabled = OcspConfiguration.isUnidEnabled();
1441         String trustDir = OcspConfiguration.getUnidTrustDir();
1442         String cacertfile = OcspConfiguration.getUnidCaCert();
1443         if (StringUtils.isEmpty(trustDir)) {
1444             // This installation is probably not using UnidFnr at all.
1445             log.debug("No UnidFnr Trust directory found. Skipping import (expected for most installations).");
1446             if (isUnidFnrEnabled) {
1447                 log.error("No UnidFnr Trust directory found. Cannot procede import");
1448             }
1449             return;
1450         }
1451         
1452         // Read all files from trustDir, expect that they are PEM formatted certificates.
1453         CryptoProviderTools.installBCProviderIfNotAvailable();
1454         File dir = new File(trustDir);
1455         try {
1456             if (dir == null || dir.isDirectory() == false) {
1457                 log.error("Could not read UnidFnr Trust Directory: " + dir.getCanonicalPath()+ " is not a directory.\nImport interrupted");
1458                 return;                
1459             }
1460             File files[] = dir.listFiles();
1461             if (files == null || files.length == 0) {
1462                 log.info("No files found in UnidFnr Trust directory: " + dir.getCanonicalPath() + ". Skipping import");
1463                 return;
1464             }
1465             for (int i=0; i < files.length; i++) {
1466                 final String fileName = files[i].getCanonicalPath();
1467                 // Read the file, don't stop completely if one file has errors in it.
1468                 try {
1469                     final byte bytesFromFile[] = FileTools.readFiletoBuffer(fileName);
1470                     byte[] bytes;
1471                     try {
1472                         bytes = FileTools.getBytesFromPEM(bytesFromFile, CertTools.BEGIN_CERTIFICATE, CertTools.END_CERTIFICATE);
1473                     } catch (Exception e) {
1474                         bytes = bytesFromFile; // assume binary data (.der).
1475                     }
1476                     final X509Certificate  cert = CertTools.getCertfromByteArray(bytes, X509Certificate.class);
1477                     trustedCerts.add(cert);
1478                 } catch (CertificateException | IOException e) {
1479                     log.error("error reading '" + fileName + "' from trustDir: " + e.getMessage(), e);
1480                 }
1481             }
1482         } catch (IOException e) {
1483             String errMsg = "Error reading files from trustDir: " + e.getMessage();
1484             log.error(errMsg, e);
1485             // Since the file exists but we can't read it. We should stop here and warn the user
1486             throw new IllegalStateException(errMsg);
1487         }
1488         // Read the CA Certificate file
1489         if (StringUtils.isEmpty(cacertfile)) {
1490             // Since this MUST be set if UnidFnr Extension is used, we should skip import if not found
1491             log.debug("No UnidFnr CA Cert directory found. Skipping import");
1492             if (isUnidFnrEnabled) {
1493                 log.error("No UnidFnr CA Cert directory found. Cannot procede import");
1494             }
1495             return;
1496         }
1497         try {
1498             byte[] bytes = FileTools.getBytesFromPEM(FileTools
1499                     .readFiletoBuffer(cacertfile),
1500                     CertTools.BEGIN_CERTIFICATE, CertTools.END_CERTIFICATE);
1501             cacert = CertTools.getCertfromByteArray(bytes, Certificate.class);
1502         } catch (Exception e) {
1503             String errMsg = "Error reading CA Certificate from UnidFnr cacertfile";
1504             log.error(errMsg, e);
1505             // Since the file exists but we can't read it. We should stop here and warn the user
1506             throw new IllegalStateException(errMsg);
1507         }
1508         
1509         if (!CertTools.isCA(cacert)) {
1510             log.error(cacertfile + " does not point to a CA Certificate");
1511             return;
1512         }
1513         final String subjectdn = CertTools.getSubjectDN(cacert);
1514         
1515         final int caid = CertTools.stringToBCDNString(subjectdn).hashCode();
1516         try {
1517             caSession.verifyExistenceOfCA(caid);
1518         } catch (CADoesntExistsException e) {
1519             log.info("Could not add CA to OCSP Key Binding trusted certificates. " + subjectdn + " is not known by EJBCA.");
1520             return;
1521         }
1522         // Add all found certificate serial numbers to the IKB trust entries
1523         final List<Integer> ocspKbIds = internalKeyBindingDataSession.getIds(OcspKeyBinding.IMPLEMENTATION_ALIAS);
1524         for (Integer ocspKbId : ocspKbIds) {
1525             InternalKeyBinding ikbToEdit = internalKeyBindingDataSession.getInternalKeyBindingForEdit(ocspKbId);
1526             List<InternalKeyBindingTrustEntry> currentTrustEntries = ikbToEdit.getTrustedCertificateReferences();
1527             
1528             for (X509Certificate trustedCert : trustedCerts) {
1529                 final String subjectDn = trustedCert.getSubjectDN().getName();
1530                 final DNFieldExtractor dnFieldExtractor = new DNFieldExtractor(subjectDn, DNFieldExtractor.TYPE_SUBJECTDN);
1531                 final String commonName = dnFieldExtractor.getFieldString(DNFieldExtractor.CN);
1532                 currentTrustEntries.add(new InternalKeyBindingTrustEntry(caid, trustedCert.getSerialNumber(), commonName));
1533             }
1534             ikbToEdit.setTrustedCertificateReferences(currentTrustEntries);
1535             try {
1536                 internalKeyBindingDataSession.mergeInternalKeyBinding(ikbToEdit);
1537             } catch (InternalKeyBindingNameInUseException e) {
1538                 // Should not happen when merging
1539                 log.info("Could not edit key binding: " + ikbToEdit.getName() + ". Name already in use");
1540             }
1541         }
1542     }
