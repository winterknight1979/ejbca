77      @Override
78      public AdminGroupData create(AuthenticationToken authenticationToken, String roleName) throws RoleExistsException {
79          if (getRole(roleName) == null) {
80              AdminGroupData role = new AdminGroupData(findFreeRoleId(), roleName);
81              entityManager.persist(role);
82              final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.roleadded", roleName);
83              securityEventsLogger.log(EventTypes.ROLE_CREATION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
84                      authenticationToken.toString(), null, null, null, msg);
85              return role;
86          } else {
87              final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.erroraddroleexists", roleName);
88              securityEventsLogger.log(EventTypes.ROLE_CREATION, EventStatus.FAILURE, ModuleTypes.ROLES, ServiceTypes.CORE,
89                      authenticationToken.toString(), null, null, null, msg);
90              throw new RoleExistsException(msg);
91          }
92      }

94      private Integer findFreeRoleId() {
95          final ProfileID.DB db = new ProfileID.DB() {
96              @Override
97              public boolean isFree(int i) {
98                  return entityManager.find(AdminGroupData.class, i)==null;
99              }
100         };
101         return Integer.valueOf(ProfileID.getNotUsedID(db));
102     }

104     @Override
105     public void addAccessRuleDataToRolesWhenAccessIsImplied(final AuthenticationToken authenticationToken, final String skipWhenRecursiveAccessTo,
106             final List<String> requiredAccessRules, final List<String> grantedAccessRules, final boolean grantedAccessRecursive) {
107         for (final AdminGroupData role : getAllRoles()) {
108             if (role.hasAccessToRule(skipWhenRecursiveAccessTo, true)) {
109                 // No need to grant extra privileges to if the specified recursive access is granted to the current role
110                 continue;
111             }
112             boolean allGranted = true;
113             for (final String requiredAccess : requiredAccessRules) {
114                 // If a rule will be granted, we don't require access to it just as the legacy code
115                 if (!grantedAccessRules.contains(requiredAccess) && !role.hasAccessToRule(requiredAccess)) {
116                     allGranted = false;
117                     break;
118                 }
119             }
120             if (allGranted) {
121                 final List<AccessRuleData> accessRuleDatas = new ArrayList<>();
122                 for (final String grantedResource : grantedAccessRules) {
123                     accessRuleDatas.add(new AccessRuleData(role.getRoleName(), grantedResource, AccessRuleState.RULE_ACCEPT, grantedAccessRecursive));
124                 }
125                 addAccessRulesToRole(authenticationToken, role, accessRuleDatas);
126             }
127         }
128     }

130     @Override
131     public AdminGroupData addAccessRulesToRole(AuthenticationToken authenticationToken, final AdminGroupData adminGroupData, final Collection<AccessRuleData> accessRules) {
132         AdminGroupData result;
133         if (!entityManager.contains(adminGroupData)) {
134             result = entityManager.find(AdminGroupData.class, adminGroupData.getPrimaryKey());
135         } else {
136             result = adminGroupData;
137         }
138         Map<Integer, AccessRuleData> rules = result.getAccessRules();
139         Collection<AccessRuleData> rulesAdded = new ArrayList<AccessRuleData>();
140         Collection<AccessRuleData> rulesMerged = new ArrayList<AccessRuleData>();
141         for (AccessRuleData accessRule : accessRules) {
142             // If this rule isn't persisted, persist it.
143             if (entityManager.find(AccessRuleData.class, accessRule.getPrimaryKey()) == null) {
144                 entityManager.persist(accessRule);
145                 rulesAdded.add(accessRule);
146             }
147             // If the rule exists, then merely update its values.
148             if (rules.containsKey(accessRule.getPrimaryKey())) {
149                 rules.remove(accessRule.getPrimaryKey());
150                 accessRule = setAccessRuleDataState(accessRule, accessRule.getInternalState(), accessRule.getRecursive());
151                 rulesMerged.add(accessRule);
152             }
153             rules.put(accessRule.getPrimaryKey(), accessRule);
154         }
155         result.setAccessRules(rules);
156         result = entityManager.merge(result);
157         logAccessRulesAdded(authenticationToken, adminGroupData.getRoleName(), rulesAdded);
158         return result;
159     }

161     @Override
162     public AdminGroupData addSubjectsToRole(AuthenticationToken authenticationToken, final AdminGroupData adminGroupData, Collection<AccessUserAspectData> accessUserAspectDatas) {
163         final AdminGroupData role;
164         if (!entityManager.contains(adminGroupData)) {
165             role = entityManager.find(AdminGroupData.class, adminGroupData.getPrimaryKey());
166         } else {
167             role = adminGroupData;
168         }
169         Map<Integer, AccessUserAspectData> existingUsers = role.getAccessUsers();
170         final StringBuilder subjectsAdded = new StringBuilder();
171         final StringBuilder subjectsChanged = new StringBuilder();
172         for (AccessUserAspectData accessUserAspectData : accessUserAspectDatas) {
173             AccessUserAspectData legacyVersion = getAccessUserAspectData(accessUserAspectData.getLegacyPrimaryKey());
174             if (legacyVersion != null) {
175                 //If an aspect exists using the old primary key, remove it so that we can replace it with the new one.
176                 entityManager.remove(legacyVersion);
177             }
178             if (getAccessUserAspectData(accessUserAspectData.getPrimaryKey()) == null) {
179                 // if userAspect hasn't been persisted, do so.
180                 entityManager.persist(accessUserAspectData);
181             }
182             if (existingUsers.containsKey(accessUserAspectData.getPrimaryKey())) {
183                 existingUsers.remove(accessUserAspectData.getPrimaryKey());
184                 subjectsChanged.append("[" + accessUserAspectData.toString() + "]");
185             } else {
186                 subjectsAdded.append("[" + accessUserAspectData.toString() + "]");
187             }
188             existingUsers.put(accessUserAspectData.getPrimaryKey(), accessUserAspectData);
189         }
190         role.setAccessUsers(existingUsers);
191         AdminGroupData result = entityManager.merge(role);
192         if (subjectsAdded.length() > 0) {
193             final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.adminadded", subjectsAdded, role.getRoleName());
194             securityEventsLogger.log(EventTypes.ROLE_ACCESS_USER_ADDITION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
195                     authenticationToken.toString(), null, null, null, msg);
196         }
197         if (subjectsChanged.length() > 0) {
198             final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.adminchanged", subjectsChanged, role.getRoleName());
199             securityEventsLogger.log(EventTypes.ROLE_ACCESS_USER_CHANGE, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
200                     authenticationToken.toString(), null, null, null, msg);
201         }
202         return result;
203     }

212     private void logAccessRulesAdded(AuthenticationToken authenticationToken, String rolename, Collection<AccessRuleData> addedRules) {
213         if (addedRules.size() > 0) {
214             StringBuilder addedRulesMsg = new StringBuilder();
215             for(AccessRuleData addedRule : addedRules) {
216                 addedRulesMsg.append("[" + addedRule.toString() + "]");
217             }            
218             final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.accessrulesadded", rolename, addedRulesMsg);
219             Map<String, Object> details = new LinkedHashMap<String, Object>();
220             details.put("msg", msg);
221             securityEventsLogger.log(EventTypes.ROLE_ACCESS_RULE_ADDITION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
222                     authenticationToken.toString(), null, null, null, details);
223         }
224     }

226     private void removeAccessRuleDatas(Collection<AccessRuleData> accessRules) {
227         for (final AccessRuleData accessRule : accessRules) {
228             entityManager.remove(getAccessRuleDataManaged(accessRule));
229         }
230     }

232     private AccessRuleData getAccessRuleDataManaged(final AccessRuleData accessRuleData) {
233         if (entityManager.contains(accessRuleData)) {
234             // If this was already managed, assume that rowVersion is proper
235             return accessRuleData;
236         }
237         return entityManager.find(AccessRuleData.class, accessRuleData.getPrimaryKey());
238     }

241         AccessRuleData result = getAccessRuleDataManaged(rule);
242         result.setInternalState(state);

262     @TransactionAttribute(TransactionAttributeType.REQUIRED)
263     @Override
264     public void createSuperAdministrator() {
265         final String TEMPORARY_SUPERADMIN_ROLE = "Temporary Super Administrator Group";
266         final String SUPERADMIN_ROLE = AuthorizationSystemSession.SUPERADMIN_ROLE;
267         // Create the Super Admin
268         AdminGroupData role = getRole(SUPERADMIN_ROLE);
269         if (role == null) {
270             log.debug("Creating new role '" + SUPERADMIN_ROLE + "'.");
271             role = new AdminGroupData(1, SUPERADMIN_ROLE);
272             entityManager.persist(role);
273         } else {
274             log.debug("'" + SUPERADMIN_ROLE + "' already exists, not creating new.");            
275         }
276         AccessRuleData rule = new AccessRuleData(SUPERADMIN_ROLE, StandardRules.ROLE_ROOT.resource(), AccessRuleState.RULE_ACCEPT, true);
277         if (!role.getAccessRules().containsKey(rule.getPrimaryKey())) {
278             log.debug("Adding new rule '/' to " + SUPERADMIN_ROLE + ".");
279             Map<Integer, AccessRuleData> newrules = new HashMap<>();
280             newrules.put(rule.getPrimaryKey(), rule);
281             role.setAccessRules(newrules);
282         } else {
283             log.debug("rule '/' already exists in " + SUPERADMIN_ROLE + ".");
284         }
285         // Pick up the aspects from the old temp. super admin group and add them to the new one.
286         Map<Integer, AccessUserAspectData> newUsers = new HashMap<>();
287         AdminGroupData oldSuperAdminRole = getRole(TEMPORARY_SUPERADMIN_ROLE);
288         if (oldSuperAdminRole != null) {
289             Map<Integer, AccessUserAspectData> oldSuperAdminAspects = oldSuperAdminRole.getAccessUsers();
290             Map<Integer, AccessUserAspectData> existingSuperAdminAspects = role.getAccessUsers();
291             for (AccessUserAspectData aspect : oldSuperAdminAspects.values()) {
292                 AccessMatchValue matchWith = AccessMatchValueReverseLookupRegistry.INSTANCE.performReverseLookup(
293                         X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE, aspect.getMatchWith());
294                 AccessUserAspectData superAdminUserAspect = new AccessUserAspectData(SUPERADMIN_ROLE, aspect.getCaId(), matchWith,
295                         aspect.getMatchTypeAsType(), aspect.getMatchValue());
296                 if (existingSuperAdminAspects.containsKey(superAdminUserAspect.getPrimaryKey())) {
297                     log.debug(SUPERADMIN_ROLE + " already contains aspect matching " + aspect.getMatchValue() + " for CA with ID " + aspect.getCaId());
298                 } else {
299                     newUsers.put(superAdminUserAspect.getPrimaryKey(), superAdminUserAspect);
300                 }
301             }
302         }
303         // Create the CLI Default User
304         Map<Integer, AccessUserAspectData> users = role.getAccessUsers();
305         AccessUserAspectData defaultCliUserAspect = new AccessUserAspectData(SUPERADMIN_ROLE, 0, CliUserAccessMatchValue.USERNAME,
306                 AccessMatchType.TYPE_EQUALCASE, EjbcaConfiguration.getCliDefaultUser());
307         if (!users.containsKey(defaultCliUserAspect.getPrimaryKey())) {
308             log.debug("Adding new AccessUserAspect '"+EjbcaConfiguration.getCliDefaultUser()+"' to " + SUPERADMIN_ROLE + ".");
309             newUsers.put(defaultCliUserAspect.getPrimaryKey(), defaultCliUserAspect);
310             UserData defaultCliUserData = new UserData(EjbcaConfiguration.getCliDefaultUser(), EjbcaConfiguration.getCliDefaultPassword(), false, "UID="
311                     + EjbcaConfiguration.getCliDefaultUser(), 0, null, null, null, 0, EndEntityConstants.EMPTY_END_ENTITY_PROFILE, 0, 0, 0, null);
312             defaultCliUserData.setStatus(EndEntityConstants.STATUS_GENERATED);
313             if (entityManager.find(UserData.class, defaultCliUserData.getUsername())==null) {
314                 entityManager.persist(defaultCliUserData);
315             }
316         } else {
317             log.debug("AccessUserAspect '"+EjbcaConfiguration.getCliDefaultUser()+"' already exists in " + SUPERADMIN_ROLE + ".");            
318         }
319         // Add all created aspects to role
320         role.setAccessUsers(newUsers);
321     }

323     @Override
324     public void setTokenTypeWhenNull(final AuthenticationToken authenticationToken) {
325         for (final AdminGroupData role : getAllRoles()) {
326             final Collection<AccessUserAspectData> updatedUsers = new ArrayList<>();
327             for (AccessUserAspectData userAspect : role.getAccessUsers().values()) {
328                 if (userAspect.getTokenType() == null) {
329                     userAspect.setTokenType(X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE);
330                     updatedUsers.add(userAspect);
331                 }
332             }
333             addSubjectsToRole(authenticationToken, role, updatedUsers);
334         }
335     }

352     private void deleteRole(final AuthenticationToken authenticationToken, final AdminGroupData role) {
353         for (AccessUserAspectData userAspect : role.getAccessUsers().values()) {
354             entityManager.remove(userAspect);
355         }
356         removeAccessRuleDatas(role.getAccessRules().values());
357         entityManager.remove(role);
358         final String msg = INTERNAL_RESOURCES.getLocalizedMessage("authorization.roleremoved", role.getRoleName());
359         securityEventsLogger.log(EventTypes.ROLE_DELETION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
360                 authenticationToken.toString(), null, null, null, msg);
361     }
