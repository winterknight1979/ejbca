179     /**
180      * Sends a certificate request (PKCS10) to the CA and receives the reply.
181      *
182      * @param requestUrl DOCUMENT ME!
183      * @param request Base64 encoded PKCS10-request (PEM-format)
184      * @param username username
185      * @param password password
186      *
187      * @exception IllegalArgumentException if requesturl is not a vlid HTTP or HTTPS url.
188      * @exception Exception if the trusted webcert is not a correct certificate.
189      * @exception Exception if we get back a HTTP response code != 200 from the CA.
190      * @exception Exception if the reply is not a correct certificate.
191      */
192     public void sendHttpReq(String requestUrl, String request, String username, String password)
193         throws Exception {
194         log.trace(">sendHttpReq: request=" + request.toString() + ", username=" + username +
195             ", password=" + password);
196 
197         if (requestUrl == null) {
198             throw new IllegalArgumentException("requesturl can not be  null.");
199         }
200 
201         log.debug("Sending request to: " + requestUrl);
202 
203         URL url = new URL(requestUrl);
204         HttpURLConnection con = (HttpURLConnection) getUrlConnection(url);
205 
206         // we are going to do a POST
207         con.setDoOutput(true);
208         con.setRequestMethod("POST");
209 
210         // POST it
211         PrintWriter out = new PrintWriter(con.getOutputStream());
212         out.println("pkcs10req=" + URLEncoder.encode(request,"UTF-8") + "&user=" +
213             URLEncoder.encode(username,"UTF-8") + "&password=" + URLEncoder.encode(password,"UTF-8") +
214             "&submit=Submit+Query");
215         out.close();
216 
217         // Read the reqponse
218         BufferedReader in = null;
219         try {
220             in = new BufferedReader(new InputStreamReader(con.getInputStream()));
221             String inputLine;
222 
223             while ((inputLine = in.readLine()) != null) {
224                 System.out.println(inputLine);
225             }        	
226         } finally {
227         	if (in != null) {
228         		in.close();
229         	}
230         }
231 
232         if (con.getResponseCode() == 200) {
233             log.debug("Received certificate reply.");
234         } else {
235             throw new Exception("Error sending PKCS10-request.");
236         }
237 
238         // We are done, disconnect
239         con.disconnect();
240 
241         log.trace("<sendHttpReq:");
242     }
