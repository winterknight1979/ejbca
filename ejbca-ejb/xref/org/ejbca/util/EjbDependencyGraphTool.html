31  /**
32   * Simple tool that
33   * - expands all JARs in an EAR to a temporary directory
34   * - looks for all {@literal @}Stateless annotated classes
35   * - finds out what {@literal @}Local interfaces the SSBs implement
36   * - looks through all SSBs' for {@literal @}EJB annotated fields and builds a list of dependencies from that for each SSB
37   * - traverses all dependencies to remove implied dependencies (if A-&gt;B,C and B-&gt;C we can just say A-&gt;B-&gt;C)
38   * - outputs dependency tree to console and a .dot-file that can be used to generate a graph.
39   * 
40   * @version $Id: EjbDependencyGraphTool.java 22142 2015-11-03 14:15:51Z mikekushner $
41   */
42  public class EjbDependencyGraphTool {
43  
44  	/** public entry point that spawns and runs a non-static version of this class 
45  	 * @param args Args*/
46  	public static void main(String[] args) {
47  		try {
48  			new EjbDependencyGraphTool().run(args);
49  		} catch (Exception e) {
50  			e.printStackTrace();
51  		}
52  	}
53  
54      private void run(String[] args) throws IOException, ClassNotFoundException {
55          if (args.length != 2) {
56              log("Syntax: <EAR file> <output.dot>");
57          }
58          log("Extracting JARs form EAR into temp directory..");
59          final File earFile = new File(args[0]);
60          final JarFile earJarFile = new JarFile(earFile);
61          final JarInputStream earInputStream = new JarInputStream(new FileInputStream(earFile));
62          final List<URL> jarUrls = new ArrayList<URL>();
63          final List<String> interestingClasses = new ArrayList<String>();
64          JarEntry earEntry;
65          while ((earEntry = earInputStream.getNextJarEntry()) != null) {
66              if (earEntry.getName().endsWith(".jar") && !earEntry.getName().contains("systemtests")) {
67                  final File tempFile = getTempFileFromJar(earJarFile.getInputStream(earEntry));
68                  jarUrls.add(tempFile.toURI().toURL());
69                  interestingClasses.addAll(getInterestingClasses(tempFile));
70              }
71          }
72          log("Loading potentially interesting classes..");
73          @SuppressWarnings("resource") //Loader can only be closed on JDK7
74          final URLClassLoader loader = new URLClassLoader(jarUrls.toArray(new URL[0]), this.getClass().getClassLoader()); // NOPMD this is a stand-alone tools, not a part of a JEE application
75          try {
76              List<BeanInfo> ejbs = new ArrayList<BeanInfo>();
77              for (String className : interestingClasses) {
78                  Class<?> c = loader.loadClass(className);
79                  if (c.isAnnotationPresent(javax.ejb.Stateless.class)) {
80                      ejbs.add(new BeanInfo(c));
81                  }
82              }
83  
84              log("Translating local interface dependencies into bean dependencies..");
85              for (BeanInfo beanInfoToAlter : ejbs) {
86                  for (Class<?> dependingOnIface : beanInfoToAlter.interfaceDependencies) {
87                      for (BeanInfo beanInfoToMatch : ejbs) {
88                          for (Class<?> iface : beanInfoToMatch.ifaceClasses) {
89                              if (dependingOnIface.equals(iface)) {
90                                  beanInfoToAlter.beanDependencies.add(beanInfoToMatch);
91                              }
92                          }
93                      }
94                  }
95              }
96              log("Removing implied dependencies..");
97              boolean moreIterationsRequired = true;
98              while (moreIterationsRequired) {
99                  moreIterationsRequired = false;
100                 for (BeanInfo currentBean : ejbs) {
101                     //log(" processing " + currentBean.beanClass.getSimpleName());
102                     final List<BeanInfo> toRemove = new ArrayList<BeanInfo>();
103                     for (BeanInfo dependsOn : currentBean.beanDependencies) {
104                         if (!toRemove.contains(dependsOn)) {
105                             for (BeanInfo dependsOn2 : currentBean.beanDependencies) {
106                                 if (!toRemove.contains(dependsOn2)) {
107                                     if (dependsOn.dependsOnBean(dependsOn2, currentBean, 0, 10)) {
108                                         toRemove.add(dependsOn2);
109                                         moreIterationsRequired = true;
110                                     }
111                                 }
112                             }
113                         }
114                     }
115                     currentBean.beanDependencies.removeAll(toRemove);
116                 }
117             }
118             // Show what we got..
119             for (BeanInfo currentBean : ejbs) {
120                 log(currentBean.beanClass.getSimpleName());
121                 for (BeanInfo dep : currentBean.beanDependencies) {
122                     log(" -&gt; " + dep.beanClass.getSimpleName());
123                 }
124             }
125             // Create .dot-file
126             final StringBuilder sb = new StringBuilder();
127             sb.append("digraph \"" + args[0].substring(args[0].lastIndexOf("/") + 1) + "\" {\n");
128             sb.append("graph [rankdir=\"BT\",nslimit=\"5.0\",mclimit=\"5.0\"];\n");
129             sb.append("node [fontsize=\"12\",fontname=\"Helvetica-Bold\",shape=box];\n");
130             sb.append("edge [style=\"bold\"];\n");
131             for (BeanInfo currentBean : ejbs) {
132                 sb.append("\"" + currentBean.beanClass.getSimpleName()
133                         + "\" [fillcolor=\"yellow\",style=\"filled,bold\",fontname=\"Helvetica-Bold\"]\n");
134                 for (BeanInfo dep : currentBean.beanDependencies) {
135                     sb.append("\"" + currentBean.beanClass.getSimpleName() + "\" -&gt; \"" + dep.beanClass.getSimpleName() + "\" [color=\"#000068\"]\n");
136                 }
137             }
138             sb.append("}\n");
139             FileOutputStream fos = new FileOutputStream(new File(args[1]));
140             try {
141                 fos.write(sb.toString().getBytes());
142             } finally {
143                 fos.close();
144             }
145         } finally {
146             //loader.close(); // close is inly available since java 7
147             earInputStream.close();
148             earJarFile.close();
149         }
150 	}

152 	/** @param jarFile JAR
153 	 * @return a list of class-names that end with "SessionBean" (we don't want to load all classes from the EAR into the ClassLoader..) 
154 	 * @throws IOException Fail */
155     private List<String> getInterestingClasses(File jarFile) throws IOException {
156         final JarInputStream jarInputStream = new JarInputStream(new FileInputStream(jarFile));
157         try {
158             final List<String> interestingClasses = new ArrayList<String>();
159             JarEntry jarEntry;
160             //log("    Processing " + jarFile.getName() + ":" + jarFile.length());
161             while ((jarEntry = jarInputStream.getNextJarEntry()) != null) {
162                 final String jarEntryName = jarEntry.getName();
163                 if (jarEntryName.endsWith("SessionBean.class")) {
164                     final String className = jarEntryName.replaceAll(".class", "").replaceAll("/", ".");
165                     //log("  Matching filename: " + jarEntryName + " -&gt; " + className);
166                     interestingClasses.add(className);
167                 }
168             }
169             return interestingClasses;
170         } finally {
171             jarInputStream.close();
172         }
173     }

175 	/** Write JAR entry to temporary file and return a reference to this file. 
176 	 * @param jarInputStream Jar
177 	 * @return File
178 	 * @throws IOException fail */
179 	private File getTempFileFromJar(InputStream jarInputStream) throws IOException {
180 		final File tempFile = File.createTempFile("jee5deps-", ".jar");
181 		final byte[] buffer = new byte[10*1024*1024];
182 		final OutputStream os = new FileOutputStream(tempFile);
183 		while ( jarInputStream.available() > 0) {
184 			final int len = jarInputStream.read(buffer);
185 			if (len == -1) {
186 				break;
187 			}
188 			os.write(buffer, 0, len);
189 		}
190 		os.close();
191 		return tempFile;
192 	}

198 	/** Class to represent a SSB and its dependencies. */
199 	private class BeanInfo {
200 		public final List<Class<?>> interfaceDependencies = new ArrayList<Class<?>>();
201 		public final List<BeanInfo> beanDependencies = new ArrayList<BeanInfo>();
202 		public final Class<?> beanClass;
203 		public final List<Class<?>> ifaceClasses = new ArrayList<Class<?>>();
204 
205 		public BeanInfo(Class<?> beanClass) {
206 			this.beanClass = beanClass;
207 			// Find all @Local interfaces that this class implements
208 			for (Class<?> iface : beanClass.getInterfaces()) {
209 				if (iface.isAnnotationPresent(javax.ejb.Local.class)) {
210 					ifaceClasses.add(iface);
211 					//log(" @Stateless " + beanClass.getSimpleName() + " implements @Local " + iface.getSimpleName());
212 				}
213 			}
214 			// Find all @EJB annotated fields and add these classes as dependencies
215 			for (Field field : beanClass.getDeclaredFields()) {
216 				if (field.isAnnotationPresent(javax.ejb.EJB.class)) {
217 					interfaceDependencies.add(field.getType());
218 					//log("   depends on " + field.getType().getSimpleName());
219 					if (field.getType().getName().endsWith("Remote")) {
220 						log("   WARNING: depends on " + field.getType().getSimpleName() + ". @Remote interface?");
221 					}
222 				}
223 			}
224 		}
225 
226 		/** Recurses through all dependencies to find out if beanInfo is present in any dependency 
227 		 * @param beanInfo Info
228 		 * @param origReqBeanInfo Info 
229 		 * @param steps Sterps
230 		 * @param maxSteps Max
231 		 * @return bool */
232 		public boolean dependsOnBean(BeanInfo beanInfo, BeanInfo origReqBeanInfo, int steps, int maxSteps) {
233 			if (steps>maxSteps) {
234 				return false;	// Don't recurse in loops forever
235 			}
236 			if (origReqBeanInfo.beanClass.equals(beanInfo.beanClass)) {
237 				return false;	// Don't remove circular dependencies
238 			}
239 			if (beanDependencies.contains(beanInfo)) {
240 				return true;
241 			}
242 			for (BeanInfo bean : beanDependencies) {
243 				if (bean.dependsOnBean(beanInfo, origReqBeanInfo, steps++, maxSteps)) {
244 					return true;
245 				}
246 			}
247 			return false;
248 		}
249 	}
