32  /** Class to execute a file full of sql commands. Useful for running update scripts.
33   * @version $Id: SqlExecutor.java 22142 2015-11-03 14:15:51Z mikekushner $
34   */
35  public class SqlExecutor {
36      static Logger log = Logger.getLogger(SqlExecutor.class);
37  
38      private Connection con = null;
39      private int commands = 0;
40      private int errors = 0;
41      private boolean continueOnSqlError = false;
42      public Connection getConnection() {
43          return this.con;
44      }
45      public void setContinueOnSqlError(boolean cont) {
46          this.continueOnSqlError = cont;
47      }
48      public int getErrors() {
49          return this.errors;
50      }
51      /** Creates a new SqlExecutor. Caller is responsible for releasing the connection
52       * @param connection Connection
53       * @param continueOnSQLError bool
54       */
55      public SqlExecutor(Connection connection, boolean continueOnSQLError) {
56      	if (log.isTraceEnabled()) {
57              log.trace("> SqlExecutor(" + connection + "," + continueOnSQLError+ ")");
58      	}
59          con = connection;
60  //        try {
61  //            con.setAutoCommit(false);            
62  //        } catch (SQLException ignore) {}
63          this.continueOnSqlError = continueOnSQLError;
64          log.trace("< SqlExecutor()");
65      }
66      
67      /** Runs a single sql update command
68       * 
69       * @param command the sql command to execute
70       * @return the result returned from executeUpdate
71       * @throws SQLException fail
72       */
73      public int runCommand(String command) throws SQLException {
74      	if (log.isTraceEnabled()) {
75              log.trace("> runCommand: " + command);
76      	}
77          int res = executeCommand(command);
78          log.debug(++commands + " commands executed with " + errors + " errors");
79          commands = 0;
80          errors = 0;
81          log.trace("< runCommand");
82          return res;
83      }
84      public void runCommandFile(File file) throws SQLException, FileNotFoundException, IOException {
85      	if (log.isTraceEnabled()) {
86              log.trace("> runCommandFile: " + file.getPath());
87      	}
88          Reader rdr = new FileReader(file);
89          runCommands(rdr);
90          log.trace("< runCommandFile()");
91      }
92      
93      /** Reads sql statements from the Reader object aqnd executes one statement at a time.
94       * Statements can be on one or more lines and must be terminated by ';'
95       * 
96       * @param rdr Reader object to read commends from.
97       * @throws SQLException thrown on sql errors if continueOnSqlError is set to false.
98       * @throws IOException if the reader object is invalid.
99       */
100     public void runCommands(Reader rdr) throws SQLException, IOException {
101         log.trace(">runCommands");
102         BufferedReader br = new BufferedReader(rdr);
103         Timestamp start = new Timestamp(System.currentTimeMillis());
104         try {
105             String temp;
106             final StringBuilder strBuf = new StringBuilder();
107             commands = 0;
108             List<String> list = new LinkedList<String>();
109             while ((temp = br.readLine()) != null) {
110             	if (!temp.startsWith("#") && !temp.startsWith("--")) { // Don't include comments and SQL comments
111             		list.add(temp);
112             	}
113                 if (!temp.endsWith(";")) {
114                     continue;
115                 }
116             }
117             Iterator<String> it = list.iterator();
118             while (it.hasNext()) {
119                 temp = it.next();
120                 temp = temp.trim();
121                 if (temp.length() != 0) {
122                     strBuf.append(temp);
123                     if (temp.endsWith(";")) {
124                         // end of command, remove the ';' and execute
125                         char ch = ' ';
126                         strBuf.setCharAt(strBuf.length() - 1, ch);
127                         executeCommand(strBuf.toString());
128                         commands++;
129                         strBuf.setLength(0);
130                     } else {
131                         // continue to read the command
132                         strBuf.append(" ");
133                     }
134                 }
135             }            
136         } finally {
137             if (br != null) { br.close(); }            
138         }
139         Timestamp stop = new Timestamp(System.currentTimeMillis());
140         log.debug("Execution started: " + start.toString());
141         log.debug("Execution stopped: " + stop.toString());
142         log.debug(commands + " commands executed with " + errors + " errors");
143         commands = 0;
144         errors = 0;
145         log.trace("<runCommands");
146     }
147     
148     /** Executes an sql update. 
149      * SQL INSERT, UPDATE or DELETE statement; or an SQL statement that returns nothing, such as a DDL statement. 
150      * 
151      * @param sql String holding the SQL.
152      * @return number of rows updates (returned by executeUpdate)
153      * @throws SQLException fail
154      */
155     private int executeCommand(String sql) throws SQLException {
156     	if (log.isTraceEnabled()) {
157             log.trace("> executeCommand: " + sql);
158     	}
159         Statement stmt = null;
160         int res = 0;
161         try {
162             stmt = con.createStatement();
163             res = stmt.executeUpdate(sql);
164         } catch (SQLException exception) {
165             log.error("Exception: " + exception.getMessage() + ", sql: "+ sql);
166             if (!this.continueOnSqlError) {
167                 throw exception;
168             }
169             errors++;
170         } finally {
171             if (stmt != null) {
172                 stmt.close();
173             }            
174         }
175         log.trace("< executeCommand");
176         return res;
177     } // executeCommand
