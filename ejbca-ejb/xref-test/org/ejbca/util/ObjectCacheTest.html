14  package org.ejbca.util;
15  
16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertNotNull;
18  import static org.junit.Assert.assertNull;
19  
20  import org.junit.Test;
21  
22  /**
23   * Tests the simple object cache class .
24   * 
25   * @version $Id: ObjectCacheTest.java 19902 2014-09-30 14:32:24Z anatom $
26   */
27  public class ObjectCacheTest {
28  	
29      @Test
30  	public void testObjectCache() throws Exception {
31  		ObjectCache<Object,Object> cache = new ObjectCache<Object,Object>(200);
32  		cache.put("foo", "bar");
33  		cache.put("foo1", "bar1");
34  		cache.put("foo2", "bar2");
35  		Object o = cache.get("foo1");
36  		assertNotNull(o);
37  		String s = (String)o;
38  		assertEquals("bar1", s);
39  		Thread.sleep(250);
40  		o = cache.get("foo1");
41  		assertNull(o);
42  		o = cache.get("foo");
43  		assertNull(o);
44  		o = cache.get("foo2");
45  		assertNull(o);
46  	}
47  }
