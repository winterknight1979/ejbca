139     /**
140      * Calls willPublishCertificate and storeCertificate with EasyMock.
141      * @param status To be put in CertificateData
142      * @param revocationReason To be put in CertificateData
143      * @param revocationDate To be put in CertificateData
144      * @param expectedWillPublishResult Expected return value from willPublishCertificate
145      * @throws AuthorizationDeniedException Fail
146      * @throws PublisherException Fail
147      */
148     private void selectivePublishingTest(final int status, final RevocationReasons revocationReason, final Date revocationDate, final boolean expectedWillPublishResult) throws AuthorizationDeniedException, PublisherException {
149         // Set up
150         final List<TreeSet<Integer>> publisherGroups = new ArrayList<>();
151         final TreeSet<Integer> group1 = new TreeSet<>();
152         group1.add(222);
153         publisherGroups.add(group1);
154         publisher.setName(TEST_PUBLISHER_NAME);
155         publisher.setPublisherGroups(publisherGroups);
156         final BasePublisher onlyRevokedPublisher = new MockPublishRevokedOnlyPublisher();
157         onlyRevokedPublisher.setPublisherId(222);
158         onlyRevokedPublisher.setName("VA_Publisher_1");
159         final CertificateData certData = new CertificateData();
160         certData.setSerialNumber(EE_SERIAL.toString());
161         certData.setSubject(EE_DN);
162         certData.setIssuer(EE_ISSUERDN);
163         certData.setCaFingerprint("111222333444555");
164         certData.setStatus(status);
165         certData.setRevocationReason(revocationReason.getDatabaseValue());
166         certData.setRevocationDate(revocationDate);
167         final Base64CertData b64CertData = new Base64CertData();
168         final CertificateDataWrapper certWrapper = new CertificateDataWrapper(null, certData, b64CertData);
169         final ExtendedInformation extinfo = new ExtendedInformation();
170         extinfo.setMaxLoginAttempts(2);
171         // Expected flow
172         expect(publisherSession.getPublisher(222)).andReturn(onlyRevokedPublisher).times(2); // for willPublishCertificate and for storeCertificate
173         if (expectedWillPublishResult) {
174             expect(publisherSession.storeCertificate(same(admin), eq(Arrays.asList(222)), eq(certWrapper), eq(EE_PASSWORD), eq(EE_DN), same(extinfo))).andReturn(true).once();
175         }
176         replay(publisherSession);
177         // Run and verify
178         assertEquals("Wrong result from willPublishCertificate", expectedWillPublishResult, publisher.willPublishCertificate(certData.getStatus(), certData.getRevocationReason()));
179         assertEquals("Wrong result from storeCertificate", true, publisher.storeCertificate(admin, certData, b64CertData, EE_PASSWORD, EE_DN, extinfo));
180         verify(publisherSession);
181     }

183     /**
184      * Tests the storeCrl using a mock PublisherSession. It tests with non-existing publisher IDs, which should
185      * never happen, but we should not crash if it happens.
186      * @throws PublisherException Fail
187      * @throws AuthorizationDeniedException Fail
188      */
189     @Test
190     public void storeCrl() throws PublisherException, AuthorizationDeniedException {
191         // Set up
192         final List<TreeSet<Integer>> publisherGroups = new ArrayList<>();
193         final TreeSet<Integer> group1 = new TreeSet<>();
194         group1.add(123);
195         group1.add(234);
196         group1.add(345);
197         group1.add(456); // this one "exists" in the mock session bean
198         group1.add(567);
199         publisherGroups.add(group1);
200         publisher.setName(TEST_PUBLISHER_NAME);
201         publisher.setPublisherGroups(publisherGroups);
202         // Expected flow
203         final BasePublisher referencedPublisher456 = new LdapPublisher();
204         referencedPublisher456.setPublisherId(456);
205         referencedPublisher456.setName("Referenced Publisher 456");
206         final byte[] myCrl = new byte[] { 22, 33, 44, 55 }; // bogus data. publisher won't decode it
207         // getPublisher should be called until it returns the existing publisher (ID 456) 
208         expect(publisherSession.getPublisher(not(eq(456)))).andReturn(null).anyTimes();
209         expect(publisherSession.getPublisher(eq(456))).andReturn(referencedPublisher456).once();
210         expect(publisherSession.storeCRL(same(admin), eq(Arrays.asList(456)), aryEq(myCrl), eq("ABCD1234"), eq(10), eq("CN=Some CA"))).andReturn(true).once();
211         replay(publisherSession);
212         // Run and verify
213         publisher.storeCRL(admin, myCrl, "ABCD1234", 10, "CN=Some CA");
214         verify(publisherSession);
215     }

244      * Tests exception handling in testConnection.
245      * @throws PublisherConnectionException Fail
246      */
247     @Test
248     public void testConnectionFailure() throws PublisherConnectionException {
249         // Set up
250         final List<TreeSet<Integer>> publisherGroups = new ArrayList<>();
251         final TreeSet<Integer> group1 = new TreeSet<>();
252         group1.add(123);
253         publisherGroups.add(group1);
254         publisher.setName(TEST_PUBLISHER_NAME);
255         publisher.setPublisherGroups(publisherGroups);
256         // Expected flow
257         final BasePublisher referencedPublisher = EasyMock.createMock(LdapPublisher.class);
258         expect(referencedPublisher.getPublisherId()).andReturn(123).anyTimes();
259         expect(referencedPublisher.getName()).andReturn("Some Failing Publisher 456").anyTimes();
260         expect(publisherSession.getPublisher(123)).andReturn(referencedPublisher).once();
261         referencedPublisher.testConnection();
262         expectLastCall().andThrow(new PublisherConnectionException("Some error 32474")).once();
263         replay(publisherSession, referencedPublisher);
264         // Run and verify
265         try {
266             publisher.testConnection();
267             fail("testConnection should throw");
268         } catch (PublisherConnectionException e) {
269             log.debug("Exception message is: " + e);
270             assertTrue("Missing cause in exception message.", e.getMessage().contains("Some error 32474"));
271             assertTrue("Missing name of failed publisher in message.", e.getMessage().contains("Some Failing Publisher 456"));
272         }
273         verify(publisherSession, referencedPublisher);
274     }

276     @Test
277     public void testClone() throws CloneNotSupportedException {
278         final List<TreeSet<Integer>> originalPublisherGroups = new ArrayList<>();
279         final TreeSet<Integer> publisherGroup1 = new TreeSet<>();
280         publisherGroup1.add(456);
281         publisherGroup1.add(123);
282         originalPublisherGroups.add(publisherGroup1);
283 
284         final MultiGroupPublisher publisher = new MultiGroupPublisher();
285         publisher.setDescription(TEST_PUBLISHER_DESCRIPTION);
286         publisher.setName(TEST_PUBLISHER_NAME); 
287         publisher.setOnlyUseQueue(false);
288         publisher.setUseQueueForCertificates(false);
289         publisher.setUseQueueForCRLs(false);
290         publisher.setPublisherGroups(originalPublisherGroups);
291 
292         final MultiGroupPublisher clone = (MultiGroupPublisher) publisher.clone();
293         assertEquals("Description is different", TEST_PUBLISHER_DESCRIPTION, clone.getDescription());
294         // Name and id is not cloned in any of the publishers, because the clone() method is only used to copy publishers to a new name in the GUI.
295         assertFalse("getOnlyUseQueue is different", clone.getOnlyUseQueue());
296         assertFalse("getUseQueueForCertificates is different", clone.getUseQueueForCertificates());
297         assertFalse("getUseQueueForCRLs is different", clone.getUseQueueForCRLs());
298         assertNotSame("Publisher groups list was re-used", publisher.getPublisherGroups(), clone.getPublisherGroups());
299         assertEquals("Publisher groups list was different", publisher.getPublisherGroups(), clone.getPublisherGroups());
300     }
