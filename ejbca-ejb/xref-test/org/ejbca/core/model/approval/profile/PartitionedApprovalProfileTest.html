29  /**
30   * Unit tests for the PartitionedApprovalProfile class.
31   * 
32   * @version $Id: PartitionedApprovalProfileTest.java 24007 2016-07-26 10:11:52Z mikekushner $
33   *
34   */
35  public class PartitionedApprovalProfileTest {
36  
37      /**
38       * Approval is required if there are any partitions that the ANYBODY pseudo-role doesn't have access to
39       * @throws ApprovalException Fail
40       * @throws AuthenticationFailedException Fail
41       */
42      @Test
43      public void testCanApprovalExecute() throws ApprovalException, AuthenticationFailedException {
44          //Create a profile with two steps, two partitions in each. 
45          PartitionedApprovalProfile approvalProfile = new PartitionedApprovalProfile("PartitionedApprovalProfile");
46          approvalProfile.initialize();
47          //Create another step (one is default)
48          approvalProfile.addStepFirst();
49          for (ApprovalStep approvalStep : approvalProfile.getSteps().values()) {
50              approvalProfile.addPartition(approvalStep.getStepIdentifier());
51          }
52          List<Approval> approvals = new ArrayList<>();
53          for (ApprovalStep step : approvalProfile.getSteps().values()) {
54              for (ApprovalPartition partition : step.getPartitions().values()) {
55                  approvals.add(new Approval("", step.getStepIdentifier(), partition.getPartitionIdentifier()));
56              }
57          }
58          assertFalse("No approvals submitted, check should have failed.", approvalProfile.canApprovalExecute(new ArrayList<Approval>()));
59          assertFalse("Incorrect approvals submitted, check should have failed.",
60                  approvalProfile.canApprovalExecute(Arrays.asList(approvals.get(0), approvals.get(0), approvals.get(0), approvals.get(0))));
61          assertTrue("Correct set of approvals submitted, check should have passed.", approvalProfile.canApprovalExecute(approvals));
62  
63      }
