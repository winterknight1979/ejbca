72      /** Creates a new instance of CertificateView 
73       * @param cdw CDW*/
74      public CertificateView(final CertificateDataWrapper cdw) {
75          certificateData = cdw.getCertificateData();
76          revokedinfo = new RevokedInfoView(CertificateStatusHelper.getCertificateStatus(certificateData), getSerialNumberBigInt(certificate, certificateData));
77          certificate = cdw.getCertificate();
78          username = certificateData.getUsername();
79          subjectDnFieldExtractor = new DNFieldExtractor(certificateData.getSubjectDnNeverNull(), DNFieldExtractor.TYPE_SUBJECTDN);
80          issuerDnFieldExtractor = new DNFieldExtractor(certificateData.getIssuerDN(), DNFieldExtractor.TYPE_SUBJECTDN);
81      }

83      /** Creates a new instance of CertificateView for CA certificates 
84       * @param certificate Cert
85       * @param revokedinfo Info */
86      public CertificateView(Certificate certificate, RevokedInfoView revokedinfo) {
87          this.certificate=certificate;
88          this.revokedinfo= revokedinfo;
89          this.username=null;
90          subjectDnFieldExtractor = new DNFieldExtractor(CertTools.getSubjectDN(certificate), DNFieldExtractor.TYPE_SUBJECTDN);
91          subjectDnFieldExtractor = new DNFieldExtractor(CertTools.getIssuerDN(certificate), DNFieldExtractor.TYPE_SUBJECTDN);
92      }

94      /** Method that returns the version number of the X509 certificate. 
95       * @return String*/
96      public String getVersion() {
97          if (certificate==null) {
98              return UNKNOWN;
99          }
100         if (certificate instanceof X509Certificate) {
101             X509Certificate x509cert = (X509Certificate)certificate;
102             return Integer.toString(x509cert.getVersion());
103         } else {
104             return String.valueOf(CVCertificateBody.CVC_VERSION);
105         }
106     }

126     private BigInteger getSerialNumberBigInt(final Certificate certificate, final CertificateData certificateData) {
127         if (certificate==null) {
128             try {
129                 // This will work for X.509
130                 return new BigInteger(certificateData.getSerialNumber(), 10);
131             } catch (NumberFormatException e) {
132                 return BigInteger.valueOf(0);
133             }
134         }
135         return CertTools.getSerialNumber(certificate);
136     }

225     public boolean checkValidity() {
226         if (certificate==null) {
227             // We can't check not before field in this case, so make a best effort
228             return certificateData.getExpireDate()>=System.currentTimeMillis();
229         }
230         boolean valid = true;
231         try {
232             CertTools.checkValidity(certificate, new Date());
233         } catch (CertificateExpiredException e) {
234             valid = false;
235         } catch (CertificateNotYetValidException e) {
236             valid = false;
237         }
238         return valid;
239     }

241     public boolean checkValidity(final Date date)  {
242         if (certificate==null) {
243             // We can't check not before field in this case, so make a best effort
244             return certificateData.getExpireDate()>=date.getTime();
245         }
246         boolean valid = true;
247         try {
248             CertTools.checkValidity(certificate, date);
249         } catch (CertificateExpiredException e) {
250             valid = false;
251         } catch (CertificateNotYetValidException e) {
252             valid = false;
253         }
254         return valid;
255     }

283     public String getPublicKeyModulus(){
284         if (certificate==null) {
285             return UNKNOWN;
286         }
287     	String mod = null;
288     	if( certificate.getPublicKey() instanceof RSAPublicKey){
289     		mod = "" + ((RSAPublicKey)certificate.getPublicKey()).getModulus().toString(16);
290     		mod = mod.toUpperCase();
291     		mod = StringUtils.abbreviate(mod, 50);
292     	} else if( certificate.getPublicKey() instanceof DSAPublicKey){
293     		mod = "" + ((DSAPublicKey)certificate.getPublicKey()).getY().toString(16);
294     		mod = mod.toUpperCase();
295     		mod = StringUtils.abbreviate(mod, 50);
296     	} else if( certificate.getPublicKey() instanceof ECPublicKey){
297     		mod = "" + ((ECPublicKey)certificate.getPublicKey()).getW().getAffineX().toString(16);
298     		mod = mod + ((ECPublicKey)certificate.getPublicKey()).getW().getAffineY().toString(16);
299     		mod = mod.toUpperCase();
300     		mod = StringUtils.abbreviate(mod, 50);
301     	}
302     	return mod;
303     }

314     /** Method that returns if key is allowed for given usage. Usage must be one of this class key usage constants. 
315      * @param usage Usage
316      * @return bool */
317     public boolean getKeyUsage(int usage) {
318         if (certificate==null) {
319             return false;
320         }
321     	boolean returnval = false;
322     	if (certificate instanceof X509Certificate) {
323     		X509Certificate x509cert = (X509Certificate)certificate;
324     		if(x509cert.getKeyUsage() != null) {
325     			returnval= x509cert.getKeyUsage()[usage];
326     		}
327     	} else {
328     		returnval = false;
329     	}
330     	return returnval;
331     }

333     public String[] getExtendedKeyUsageAsTexts(AvailableExtendedKeyUsagesConfiguration ekuConfig){
334         if (certificate==null) {
335             return new String[0];
336         }
337         List<String> extendedkeyusage = null;
338         if (certificate instanceof X509Certificate) {
339             X509Certificate x509cert = (X509Certificate)certificate;
340             try {
341                 extendedkeyusage = x509cert.getExtendedKeyUsage();  
342             } catch (CertificateParsingException e) {}
343         }
344         if (extendedkeyusage == null) {
345             extendedkeyusage = new ArrayList<>();
346         }
347         final String[] returnval = new String[extendedkeyusage.size()];
348         for (int i=0; i<extendedkeyusage.size(); i++) {
349             returnval[i] = ekuConfig.getExtKeyUsageName(extendedkeyusage.get(i));
350         }
351         return returnval; 
352     }

362     public String getBasicConstraints(String localizedNoneText, String localizedNolimitText, String localizedEndEntityText, String localizedCaPathLengthText) {
363         if (certificate==null) {
364             return UNKNOWN;
365         }
366     	String retval = localizedNoneText;	//ejbcawebbean.getText("EXT_NONE");
367     	if (certificate instanceof X509Certificate) {
368     		X509Certificate x509cert = (X509Certificate)certificate;
369     		int bc = x509cert.getBasicConstraints();
370     		if (bc == Integer.MAX_VALUE) {
371     			retval = localizedNolimitText;	//ejbcawebbean.getText("EXT_PKIX_BC_CANOLIMIT");
372     		} else if (bc == -1) {
373     			retval = localizedEndEntityText;	//ejbcawebbean.getText("EXT_PKIX_BC_ENDENTITY");
374     		} else {
375     			retval = localizedCaPathLengthText /*ejbcawebbean.getText("EXT_PKIX_BC_CAPATHLENGTH")*/ + " : " + x509cert.getBasicConstraints();                    	     			
376     		}
377     	} else if (certificate.getType().equals("CVC")) {
378     		CardVerifiableCertificate cvccert = (CardVerifiableCertificate)certificate;
379     		try {
380     			retval = cvccert.getCVCertificate().getCertificateBody().getAuthorizationTemplate().getAuthorizationField().getAuthRole().toString();
381     		} catch (NoSuchFieldException e) {
382     	    	retval = localizedNoneText;	//ejbcawebbean.getText("EXT_NONE");
383     		}
384     	}
385     	return retval;
386     }

395     public String getSHA1Fingerprint(){
396         if (certificate==null) {
397             return certificateData.getFingerprint().toUpperCase();
398         }
399         String returnval = "";
400         try {
401             byte[] res = CertTools.generateSHA1Fingerprint(certificate.getEncoded());
402             String ret = new String(Hex.encode(res));
403             returnval = ret.toUpperCase();
404         } catch (CertificateEncodingException e) {
405         }
406         return returnval;
407     }

493     public boolean hasNameConstraints() {
494         if (certificate==null) {
495             return false;
496         }
497         if (certificate instanceof X509Certificate) {
498             X509Certificate x509cert = (X509Certificate)certificate;
499             byte[] ext = x509cert.getExtensionValue(Extension.nameConstraints.getId());
500             return ext != null;
501         }
502         return false;
503     }
