44  /**
45   * Servlet used for requesting browser certificate renewals.
46   * 
47   * @version $Id: RenewServlet.java 22811 2016-02-15 16:48:23Z samuellb $
48   */
49  public class RenewServlet extends HttpServlet {
50  
51      private static final long serialVersionUID = 1L;
52  
53      private static final Logger log = Logger.getLogger(RenewServlet.class);
54      
55      /** Submit button on the web page */
56  	public static final String BUTTONRENEW = "buttonrenew";
57  
58  	@EJB
59  	private CertificateStoreSessionLocal certificateStoreSession;
60  	@EJB
61  	private EndEntityAccessSessionLocal endEntityAccessSession;
62  	@EJB
63  	private EndEntityProfileSessionLocal endEntityProfileSession;
64  	@EJB
65  	private EndEntityManagementSessionLocal endEntityManagementSession;
66      
67      /**
68       * Servlet init
69       *
70       * @param config servlet configuration
71       * @throws ServletException on error
72       */
73      @Override
74      public void init(ServletConfig config) throws ServletException {
75          super.init(config);
76          // Install BouncyCastle provider
77          CryptoProviderTools.installBCProviderIfNotAvailable();
78      }
79  
80      public void doRequest(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
81      	
82          AuthenticationToken admin = new AlwaysAllowLocalAuthenticationToken(new WebPrincipal("RenewServlet", request.getRemoteAddr()));
83      	
84      	// SSL client authentication
85      	Object o = request.getAttribute("javax.servlet.request.X509Certificate");
86      	if (o == null || !(o instanceof X509Certificate[])) {
87      	    response.sendError(HttpServletResponse.SC_UNAUTHORIZED, "This servlet requires certificate authentication!");
88      	    return;
89      	}
90      	X509Certificate certificate = ((X509Certificate[]) o)[0];
91      	request.setAttribute("certificate", certificate);
92      	boolean isrevoked = certificateStoreSession.isRevoked(certificate.getIssuerDN().getName(), certificate.getSerialNumber());
93      	if (isrevoked) {
94      		request.setAttribute("errorMessage", "User certificate with serial number "+certificate.getSerialNumber() + " from issuer \'"+certificate.getIssuerDN()+"\' is revoked.");
95      	} else {
96  	    	String username = certificateStoreSession.findUsernameByCertSerno(certificate.getSerialNumber(), CertTools.getIssuerDN(certificate));
97  	    	if (username==null || username.length()==0) {
98  	    		throw new ServletException(new ObjectNotFoundException("Not possible to retrieve user name"));
99  	    	}
100 	    	request.setAttribute("username", username);
101 	    	if(log.isDebugEnabled()) {
102 	    		log.debug("User authenticated as '" + username + "'.");
103 	    	}
104 	    	
105 	    	// Request certificate renewal
106 	    	if(request.getParameter(BUTTONRENEW) != null) {
107 	    		if(log.isDebugEnabled()) {
108 	    			log.debug("Got renewal request for '" + username + "'.");
109 	    		}
110 	    		String statusMessage;
111 	    		try {
112 		    		EndEntityInformation userdata = endEntityAccessSession.findUser(admin, username);
113 		    		if (userdata.getType().contains(EndEntityTypes.SENDNOTIFICATION)) {
114 		    		    EndEntityProfile profile = endEntityProfileSession.getEndEntityProfile(userdata.getEndEntityProfileId());
115 		    		    userdata.setPassword(profile.getAutoGeneratedPasswd());
116 		    		    userdata.setStatus(EndEntityConstants.STATUS_NEW);
117 		    		    endEntityManagementSession.changeUser(admin, userdata, false);
118 		    		    statusMessage = "Your request for certificate renewal has been submitted. You will be notified by mail of the status change and new password.";
119 		    		} else {	    		    
120 		    		    log.warn("End entity with username " + username + " attempted to renew browser certificate, but password cannot be retrieved since notifications are not configured.");
121 		    		    statusMessage = "No notifications have been set for this end entity, so the new password can't be retrieved.";
122 		    		}
123 	    		} catch(WaitingForApprovalException ex) {
124 	    			statusMessage = "Your request for certificate renewal has been submitted and is now waiting for approval.";
125 	    		} catch(ApprovalException ex) {
126 	    			statusMessage = "Your request for certificate renewal has been submitted before and is already waiting for approval.";
127 	    		} catch(Exception ex) {
128 	    			throw new ServletException(ex);
129 	    		}
130 	    		request.setAttribute("statusMessage", statusMessage);
131 	    	}
132     	}
133     	request.setAttribute("buttonRenew", BUTTONRENEW);
134     	request.getRequestDispatcher("/renewpage.jsp").forward(request, response); 
135     }
136 
137     /**
138      * Handles HTTP POST
139      *
140      * @param request servlet request
141      * @param response servlet response
142      *
143      * @throws IOException input/output error
144      * @throws ServletException on error
145      */
146     public void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
147         doRequest(request, response);
148     }
149 
150     /**
151      * Handles HTTP GET
152      *
153      * @param request servlet request
154      * @param response servlet response
155      *
156      * @throws IOException input/output error
157      * @throws ServletException on error
158      */
159     public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
160     	doRequest(request, response);
161     }
