58  /**
59   * This is a servlet that is used for creating a user into EJBCA and retrieving her certificate. Supports only POST.

65   * <dd>
66   * A PKCS#10 request, mandatory.</dd>
67   * <dt>username</dt>

70   * <dt>password</dt>
71   * <dd>

118     /**
119      * Handles PKCS10 certificate request, these are constructed as:
120      *
121      * <pre>
122      * CertificationRequest ::= SEQUENCE {
123      * certificationRequestInfo  CertificationRequestInfo,
124      * signatureAlgorithm          AlgorithmIdentifier{{ SignatureAlgorithms }},
125      * signature                       BIT STRING
126      * }
127      * CertificationRequestInfo ::= SEQUENCE {
128      * version             INTEGER { v1(0) } (v1,...),
129      * subject             Name,
130      * subjectPKInfo   SubjectPublicKeyInfo{{ PKInfoAlgorithms }},
131      * attributes          [0] Attributes{{ CRIAttributes }}
132      * }
133      * SubjectPublicKeyInfo { ALGORITHM : IOSet} ::= SEQUENCE {
134      * algorithm           AlgorithmIdentifier {{IOSet}},
135      * subjectPublicKey    BIT STRING
136      * }
137      * </pre>
138      *
139      * PublicKey's encoded-format has to be RSA X.509.
140      */

338     /**
339      * @param admin admin
340      * @param username user
341      * @param adminsession admin
342      * @return true if the username is ok (does not already exist), false otherwise
343      * @throws ServletException fail
344      */
345     private final boolean checkUsername(AuthenticationToken admin, String username, EndEntityAccessSession adminsession) throws ServletException {
346         if (username != null) {
347             username = username.trim();
348         }
349         if (username == null || username.length() == 0) {
350             throw new ServletException("Username must not be empty.");
351         }
352 
353         EndEntityInformation tmpuser = null;
354         try {
355             tmpuser = adminsession.findUser(admin, username);
356         } catch (Exception e) {
357             throw new ServletException("Error checking username '" + username + ": ", e);
358         }
359         return (tmpuser == null) ? true : false;
360     }
