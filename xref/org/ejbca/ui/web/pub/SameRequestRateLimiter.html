13  package org.ejbca.ui.web.pub;
14  
15  import java.util.concurrent.locks.ReentrantLock;
16  
17  /**
18   * Helper class for use when multiple threads a requesting something that will end up with the same result.
19   * 
20   * Instead of allowing each request to proceed, only the first will return a ticket where isFirst() is false.
21   * Example use:
22   * 
23   * final static SameRequestRateLimiter&lt;Object&gt; srrl = new SameRequestRateLimiter&lt;Object&gt;();
24   * ...
25   * SameRequestRateLimiter&lt;Object&gt;.Result result = srrl.getResult();
26   * if (result.isFirst()) {
27   *   try {
28   *     // Perform common action
29   *     result.setValue(...);
30   *   } catch (Throwable t) {
31   *     result.setError(t);
32   *   }
33   * }
34   * Object resultValue = result.getValue();
35   * ...
36   * 
37   * @version $Id: SameRequestRateLimiter.java 20084 2014-10-27 13:00:50Z jeklund $
38   * @param <T> Type
39   */
40  public class SameRequestRateLimiter<T> {
41      
42      /** Lock for keeping all but the first thread hanging and waiting for the first thread to calculate the result. */
43      private final ReentrantLock rateLimiterLock = new ReentrantLock(false);
44      /** Lock for modifying the current shared Result object. */
45      private final ReentrantLock resultObjectLock = new ReentrantLock(false);
46      /** Shared Result object between all the threads that asks for the result while the first calling thread is calculating the result. */
47      private Result result = null;
48      
49      public class Result {
50          private boolean isFirst = true;
51          private T value;
52          private Throwable throwable = null;
53          
54          /** @return true if the setValue should be called. */
55          public boolean isFirst() {
56              return isFirst;
57          }
58          
59          /** @return the result if isFirst returns false or setValue has been called. */
60          public T getValue() {
61              if (isFirst) {
62                  // Programming/meatware problem..
63                  throw new IllegalStateException("Current thread should have called setValue first!");
64              }
65              if (throwable != null) {
66                  throw new RuntimeException(throwable);
67              }
68              return value;
69          }
70          
71          /** Store result of operation and release all pending threads 
72           * @param value Value*/
73          public void setValue(final T value) {
74              // Acquire a lock for modifying the class level result reference (we will reset it)
75              resultObjectLock.lock();
76              try {
77                  this.isFirst = false;   // Allow getValue() method to be called.
78                  this.value = value;
79                  result = null;  // First new thread entering getResult() will start working on a new Result.
80                  /*
81                   * At this point we have a result of the operation and other threads waiting in the getResult() method
82                   * are allowed to proceed and return the shared Result object.
83                   */
84                  rateLimiterLock.unlock();
85              } finally {
86                  resultObjectLock.unlock();
87              }
88          }
89          
90          /** Store resulting exception and release all pending threads 
91           * @param throwable Exception*/
92          public void setError(final Throwable throwable) {
93              // Acquire a lock for modifying the class level result reference (we will reset it)
94              resultObjectLock.lock();
95              try {
96                  this.isFirst = false;   // Allow getValue() method to be called.
97                  this.throwable = throwable;
98                  result = null;  // First new thread entering getResult() will start working on a new Result.
99                  /*
100                  * At this point we have a result of the operation and other threads waiting in the getResult() method
101                  * are allowed to proceed and return the shared Result object.
102                  */
103                 rateLimiterLock.unlock();
104             } finally {
105                 resultObjectLock.unlock();
106             }
107         }
108     }
109     
110     /** @return a result object that is shared by multiple threads. */
111     public Result getResult() {
112         /*
113          * Create a new shared Result object, if no such exist. This will only happen during:
114          * 1. First call to this method after this class has been instantiated.
115          * 2. First call to this method after the previous Result object has been assigned a value and is no longer
116          *    referenced by this instance.
117          */
118         resultObjectLock.lock();
119         // Declared as a method local reference, so we can return the value even if the class level reference has been reset.
120         final Result result;
121         try {
122             if (this.result == null) {
123                 this.result = new Result();
124             }
125             result = this.result;
126         } finally {
127             resultObjectLock.unlock();
128         }
129         /*
130          * Acquire a lock on this instance. The first thread passed this point will not release the lock, leaving all
131          * other threads waiting here.
132          * 
133          * Once the first thread has a result and set it on the Result object, the lock will be releases and the other
134          * threads will be able to return the same result object
135          */
136         rateLimiterLock.lock();
137         try {
138             return result;
139         } finally {
140             if (!result.isFirst()) {
141                 rateLimiterLock.unlock();
142             }
143         }
144     }
145 }
