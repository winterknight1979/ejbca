144 	/**
145 	 * Get revocation info for a certificate.
146 	 * This method fills in the supplied RevokedCertInfo object with data about a certificate.
147 	 * Since Java uses "call by reference" this works fine, but we can't create our own object because
148 	 * the caller doesn't read the reference when unwinding the stack after this method returns.
149 	 *
150 	 * @param issuerDN DN of the certificate's issuer
151 	 * @param serialNumber The serial number of the certificate
152 	 * @param result An allocated object. Data about the certificate is entered in the result object by this method.
153 	 *        If no info can be found (e.g., if the certificate does not exist), the revocationDate and
154 	 *        userCertificate fields of result are set to null.
155 	 */
156 	public void lookupRevokedInfo(String issuerDN, String serialNumber, RevokedCertInfo result) {
157 		serialNumber = ("0000000000000000" + serialNumber).substring(serialNumber.length());	// Pad with zeroes up to 16 chars
158 		if (log.isTraceEnabled()) {
159 			log.trace(">lookupRevokedInfo(" + issuerDN + ", " + serialNumber + ", " + result + ")");
160 		}
161 		if (result == null) {
162 			return; // There's nothing we can do here.
163 		}
164 		try {
165 			BigInteger serialBignum = new BigInteger(Hex.decode(StringUtils.trimToEmpty(serialNumber)));
166 			CertificateStatus info = mStoreSession.getStatus(StringUtils.trimToEmpty(issuerDN), serialBignum);
167 			if (info.equals(CertificateStatus.NOT_AVAILABLE)) {
168 				result.setRevocationDate(null);
169 				result.setUserCertificate(null);
170 			} else {
171 				result.setReason(info.revocationReason);
172 				result.setRevocationDate(info.revocationDate);
173 				result.setUserCertificate(serialBignum);
174 			}
175         } catch (NumberFormatException e) {
176             log.info("Invalid serial number entered (NumberFormatException): "+serialNumber+": "+e.getMessage());
177 		} catch (StringIndexOutOfBoundsException e) {
178 			log.info("Invalid serial number entered (StringIndexOutOfBoundsException): "+serialNumber+": "+e.getMessage());
179 		} catch (DecoderException e) {
180             log.info("Invalid serial number entered (DecoderException): "+serialNumber+": "+e.getMessage());
181 		}
182 	}

184 	/**
185 	 * Uses the store session to look up all certificates for a subject.
186 	 * The parameter <code>result</code> is updated so that it contains
187 	 * the certificates as CertificateView objects.
188 	 * @param subject The DN of the subject
189 	 * @param result a Collection (not null) that will be filled by CertificateView objects
190 	 */
191 	public void lookupCertificatesBySubject(String subject, Collection<CertificateView> result) {
192 		if (log.isTraceEnabled()) {
193 			log.trace(">lookupCertificatesBySubject(" + subject + ", " + result + ")");
194 		}
195 		if (result == null) {
196 			return; // There's nothing we can do here.
197 		}
198 		result.clear();
199 		if (subject == null) {
200 			return; // We can't lookup any certificates, so return with an empty result.
201 		}
202 		final List<CertificateDataWrapper> cdws = mStoreSession.getCertificateDatasBySubject(subject);
203 		Collections.sort(cdws);
204 		for (final CertificateDataWrapper cdw : cdws) {
205             // TODO: CertificateView is located in web.admin package, but this is web.pub package...
206             result.add(new CertificateView(cdw));
207 		}
208 	}

210 	/**
211 	 * Looks up a certificate information by issuer and serial number.
212 	 * The information can be accessed by getter methods in this class.
213 	 * @param issuer issuer
214 	 * @param serno SN
215 	 * @see #getIssuerDN()
216 	 * @see #getSubjectDN()
217 	 * @see #getSerialNumber()
218 	 */
219     public void lookupCertificateInfo(String issuer, String serno) {
220         BigInteger sernoBigInt = CertTools.getSerialNumberFromString(serno);
221         Certificate cert = mStoreSession.findCertificateByIssuerAndSerno(issuer, sernoBigInt);
222         if (cert != null) {
223             this.issuerDN = CertTools.getIssuerDN(cert);
224             this.subjectDN = CertTools.getSubjectDN(cert);
225             this.serialNumber = CertTools.getSerialNumberAsString(cert);
226             this.fingerprint = CertTools.getFingerprintAsString(cert);
227         }
228     }

281     /** @param param param
282      * @return the param as it's URL encoded counterpart, taking the configured encoding into account. */
283     private String getHttpParamAsUrlEncoded(final String param) {
284         final String encoding = WebConfiguration.getWebContentEncoding();
285         try {
286             return URLEncoder.encode(param, encoding);
287         } catch (UnsupportedEncodingException e) {
288             throw new IllegalStateException("The property 'web.contentencoding' is set to " + encoding + ", but this encoding is not available on this system.", e);
289         }
290     }
