44  /**
45   * <p>Methods to update a changed CA Subject DN and CA Id in various objects.</p>
46   * 
47   * <p>The methods always take an object to change, as well as the old and new CA Ids and the new Subject DN.
48   * They return true if the object was changed and should be updated in the database.</p>
49   * 
50   * <p>The reason for having the methods here and not in the respective objects is:</p>
51   * <ol>
52   * <li>Access rules are simply a list of access rule objects, so it can't have any methods directly on it anyway.</li>
53   * <li>Some objects do not know about all of their properties. The ServiceConfiguration object
54   * does not know about the different types of services it can store.</li>
55   * <li>Some objects do not have a single base class, such as InternalKeyBindings which has two.</li>
56   * </ol>
57   * 
58   * @version $Id: CAIdTools.java 28117 2018-01-29 09:58:24Z samuellb $
59   */

67      /**
68       * Updates any references to a CA's CAId and Subject DN.
69       * @param certProfile Profile object to modify.
70       * @param fromId Old CA Id to replace.
71       * @param toId New CA Id to replace with.
72       * @param toSubjectDN New CA Subject DN.
73       * @return True if the certificate profile was changed. If so it should be persisted to the database.
74       */
75      public static boolean updateCAIds(final CertificateProfile certProfile, final int fromId, final int toId, final String toSubjectDN) {
76          boolean changed = false;
77          final List<Integer> availableCAs = new ArrayList<>(certProfile.getAvailableCAs());
78          // The list is modified so we can't use an iterator
79          for (int i = 0; i < availableCAs.size(); i++) {
80              int value = availableCAs.get(i);
81              if (value == fromId) {
82                  availableCAs.set(i, toId);
83                  changed = true;
84              }
85          }
86          if (changed) {
87              certProfile.setAvailableCAs(availableCAs);
88          }
89          return changed;
90      }

92      /**
93       * Updates any references to a CA's CAId and Subject DN.
94       * @param endEntityProfile Profile object to modify.
95       * @param fromId Old CA Id to replace.
96       * @param toId New CA Id to replace with.
97       * @param toSubjectDN New CA Subject DN.
98       * @return True if the end entity profile was changed. If so it should be persisted to the database.
99       */
100     public static boolean updateCAIds(final EndEntityProfile endEntityProfile, final int fromId, final int toId, final String toSubjectDN) {
101         boolean changed = false;
102         
103         final Collection<Integer> original = endEntityProfile.getAvailableCAs();
104         final List<Integer> updated = new ArrayList<>();
105         for (int oldvalue : original) {
106             int newvalue;
107             if (oldvalue == fromId) {
108                 newvalue = toId;
109                 changed = true;
110             } else {
111                 newvalue = oldvalue;
112             }
113             updated.add(newvalue);
114         }
115         
116         if (changed) {
117             endEntityProfile.setAvailableCAs(updated);
118         }
119         
120         if (endEntityProfile.getDefaultCA() == fromId) {
121             endEntityProfile.setDefaultCA(toId);
122             changed = true;
123         }
124         
125         return changed;
126     }

153     /**
154      * Updates any references to a CA's CAId and Subject DN.
155      * @param serviceConf Service object to modify.
156      * @param fromId Old CA Id to replace.
157      * @param toId New CA Id to replace with.
158      * @param toSubjectDN New CA Subject DN.
159      * @return True if the service was changed. If so it should be persisted to the database.
160      */
161     public static boolean updateCAIds(final ServiceConfiguration serviceConf, final int fromId, final int toId, final String toSubjectDN) {
162         boolean changed = false;
163         final Properties workerProps = serviceConf.getWorkerProperties();
164         final String idsToCheckStr = workerProps.getProperty(IWorker.PROP_CAIDSTOCHECK);
165         if (!StringUtils.isEmpty(idsToCheckStr)) {
166             final String[] caIds = idsToCheckStr.split(";");
167             for (int i = 0; i < caIds.length; i++) {
168                 if (Integer.parseInt(caIds[i]) == fromId) {
169                     caIds[i] = String.valueOf(toId);
170                     changed = true;
171                 }
172             }
173             if (changed) {
174                 workerProps.setProperty(IWorker.PROP_CAIDSTOCHECK, StringUtils.join(caIds, ';'));
175                 serviceConf.setWorkerProperties(workerProps);
176             }
177         }
178         return changed;
179     }

181     /**
182      * Updates any references to a CA's CAId and Subject DN.
183      * @param keybind Internal key binding object to modify.
184      * @param fromId Old CA Id to replace.
185      * @param toId New CA Id to replace with.
186      * @param toSubjectDN New CA Subject DN.
187      * @return True if the key binding was changed. If so it should be persisted to the database.
188      */
189     public static boolean updateCAIds(final InternalKeyBinding keybind, final int fromId, final int toId, final String toSubjectDN) {
190         boolean changed = false;
191         List<InternalKeyBindingTrustEntry> trustentries = new ArrayList<>();
192         for (InternalKeyBindingTrustEntry trustentry : keybind.getTrustedCertificateReferences()) {
193             int trustCaId = trustentry.getCaId();
194             if (trustCaId == fromId) {
195                 trustCaId = toId;
196                 changed = true;
197             }
198             trustentries.add(new InternalKeyBindingTrustEntry(trustCaId, trustentry.fetchCertificateSerialNumber()));
199         }
200         if (changed) {
201             keybind.setTrustedCertificateReferences(trustentries);
202         }
203         return changed;
204     }

224      * Updates any references to a CA's CAId and Subject DN.
225      * @param cmpConfig CMP configuration object to modify.
226      * @param fromId Old CA Id to replace.
227      * @param toId New CA Id to replace with.
228      * @param toSubjectDN New CA Subject DN.
229      * @return True if the configuration changed. If so it should be persisted to the database.
230      */
231     public static boolean updateCAIds(final CmpConfiguration cmpConfig, final int fromId, final int toId, final String toSubjectDN) {
232         boolean changed = false;
233         for (String alias : cmpConfig.getAliasList()) {
234             final String defaultCaDN = cmpConfig.getCMPDefaultCA(alias);
235             if (defaultCaDN != null && defaultCaDN.hashCode() == fromId) {
236                 cmpConfig.setCMPDefaultCA(alias, toSubjectDN);
237                 changed = true;
238             }
239         }
240         return changed;
241     }

261     /**
262      * Updates any references to a CA's CAId and Subject DN in the given role and list of role members.
263      * @param role Role to modify.
264      * @param roleMembers List of the role's members. The members may be modified.
265      * @param fromId Old CA Id.
266      * @param toId New CA Id.
267      * @param toSubjectDN New CA Subject DN.
268      * @return True if there was a change.
269      */
270     public static boolean updateCAIds(final Role role, final List<RoleMember> roleMembers, final int fromId, final int toId, final String toSubjectDN) {
271         boolean changed = false;
272         // Look for references from access rules (currently only the /ca/<CA ID> rule) */
273         final String oldResource = AccessRulesHelper.normalizeResource(StandardRules.CAACCESS.resource() + String.valueOf(fromId));
274         final String newResource = AccessRulesHelper.normalizeResource(StandardRules.CAACCESS.resource() + String.valueOf(toId));
275         final Boolean state = role.getAccessRules().remove(oldResource);
276         if (state != null) { // rule for old CA ID exists
277             role.getAccessRules().put(newResource, state);
278             changed = true;
279         }
280         // Look for references from members
281         for (final RoleMember roleMember : roleMembers) {
282             if (roleMember.getTokenIssuerId() == fromId) {
283                 // Also check that tokenIssuerId refers to a CA. This check is more expensive performance-wise so it's done last
284                 final AccessMatchValue accessMatchValue = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(
285                         roleMember.getTokenType()).getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey());
286                 if (accessMatchValue.isIssuedByCa()) {
287                     roleMember.setTokenIssuerId(toId);
288                     changed = true;
289                 }
290             }
291         }
292         return changed;
293     }

295     /**
296      * Updates any references to a CA's CAId and Subject DN.
297      * @param roleName Name of the role. Used when creating roles to replace the old roles with.
298      * @param rules Access rules of the role. Updated in place.
299      * @param users Access users of the role. Updated in place.
300      * @param fromId Old CA Id.
301      * @param toId New CA Id.
302      * @param toSubjectDN New CA Subject DN.
303      * @return True if there was a change.
304      */
305     @Deprecated
306     public static boolean updateCAIds(final String roleName, final Map<Integer,AccessRuleData> rules, final Map<Integer,AccessUserAspectData> users, final int fromId, final int toId, final String toSubjectDN) {
307         final String toReplace = StandardRules.CAACCESS.resource()+String.valueOf(fromId);
308         final String toReplaceSlash = toReplace+"/";
309         boolean changed = false;
310         // Look for references from access rules
311         for (int id : new ArrayList<>(rules.keySet())) {
312             AccessRuleData rule = rules.get(id);
313             final String accessRuleName = rule.getAccessRuleName();
314             
315             if (accessRuleName.equals(toReplace) || accessRuleName.startsWith(toReplaceSlash)) {
316                 final String newName = StandardRules.CAACCESS.resource() + String.valueOf(toId) + accessRuleName.substring(toReplace.length());
317                 final int state = rule.getState();
318                 rule = new AccessRuleData(roleName, newName, rule.getInternalState(), rule.getRecursive());
319                 rule.setState(state);
320                 rules.put(id, rule);
321                 changed = true;
322             }
323         }
324         // Look for references from access users
325         for (int id : new ArrayList<>(users.keySet())) {
326             AccessUserAspectData user = users.get(id);
327             if (user.getCaId() == fromId) {
328                 user = new AccessUserAspectData(roleName, toId, user.getMatchWith(), user.getTokenType(), user.getMatchTypeAsType(), user.getMatchValue());
329                 users.put(id, user);
330                 changed = true;
331             }
332         }
333         return changed;
334     }
