14  package org.ejbca.util;
15  
16  import java.io.PrintWriter;
17  import java.io.StringWriter;
18  import java.util.logging.Handler;
19  import java.util.logging.Level;
20  import java.util.logging.LogRecord;
21  import java.util.logging.Logger;
22  
23  /**
24   * Use {@link #add()} to add a handler to {@link java.util.logging.Logger} that logs to {@link org.apache.log4j.Logger}
25   * 
26   * @version $Id: Log4jHandler.java 22311 2015-11-30 08:11:23Z mikekushner $
27   *
28   */
29  public class Log4jHandler extends Handler {
30  	static private boolean isStarted = false;
31  	/**
32  	 * Add handler to java sun logging that logs to log4j
33  	 */
34  	public static void add() {
35  		if ( isStarted ) {
36  			return;
37  		}
38  		isStarted = true;
39  		final Level logLevel = Level.FINEST;
40  		final Logger rootLogger = Logger.getLogger("");
41  		final Handler handlers[] = rootLogger.getHandlers();
42  		for ( int i=0; i<handlers.length; i++ ) {
43  			rootLogger.removeHandler(handlers[i]);
44  		}
45  		rootLogger.setLevel(logLevel);
46  		final Handler handler = new Log4jHandler();
47  		handler.setLevel(logLevel);
48  		rootLogger.addHandler( handler );
49  	}
50  
51  
52  	/* (non-Javadoc)
53  	 * @see java.util.logging.Handler#close()
54  	 */
55  	public void close() throws SecurityException {
56  		// do nothing
57  	}
58  	/* (non-Javadoc)
59  	 * @see java.util.logging.Handler#flush()
60  	 */
61  	public void flush() {
62  		// do nothing
63  	}
64  	private org.apache.log4j.Level translateLevel( Level level ) {
65  		if ( level.intValue() < Level.FINEST.intValue() ) {
66  			return org.apache.log4j.Level.ALL;
67  		} else if ( level.intValue() <= Level.FINEST.intValue() ) {
68  			return org.apache.log4j.Level.TRACE;
69  		} else if ( level.intValue() <= Level.FINER.intValue() ) {
70  			return org.apache.log4j.Level.DEBUG;
71  		} else if ( level.intValue() <= Level.FINE.intValue() ) {
72  			return org.apache.log4j.Level.DEBUG;
73  		} else if ( level.intValue() <= Level.INFO.intValue() ) {
74  			return org.apache.log4j.Level.INFO;
75  		} else if ( level.intValue() <= Level.WARNING.intValue() ) {
76  			return org.apache.log4j.Level.WARN;
77  		} else if ( level.intValue() <= Level.SEVERE.intValue() ) {
78  			return org.apache.log4j.Level.FATAL;
79  		} else {
80  			return org.apache.log4j.Level.OFF;
81  		}
82  	}
83  	/* (non-Javadoc)
84  	 * @see java.util.logging.Handler#publish(java.util.logging.LogRecord)
85  	 */
86  	public void publish(LogRecord record) {
87  		final org.apache.log4j.Logger logger = org.apache.log4j.Logger.getLogger( record.getSourceClassName() );
88  		StringWriter stringWriter = new StringWriter();
89  		PrintWriter printWriter = new PrintWriter(stringWriter);
90  		printWriter.print(record.getSourceMethodName());
91  		printWriter.print(": ");
92  		printWriter.print(record.getMessage());
93  		Object parameters[] = record.getParameters();
94  		for ( int i=0; parameters!=null && i<parameters.length; i++ ) {
95  			printWriter.println();
96  			printWriter.print("{" + i +"} = " + parameters[i] );
97  		}
98  		logger.log( Log4jHandler.class.getCanonicalName(),
99  		            translateLevel(record.getLevel()),
100 		            stringWriter.toString(),
101 		            record.getThrown() );
102 	}
103 }
