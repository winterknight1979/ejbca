13  package org.ejbca.statedump.ejb;
14  
15  import java.io.IOException;
16  import java.util.Map;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  
21  /**
22   * Statedump is an internal PrimeKey tool.
23   * 
24   * @version $Id: StatedumpSession.java 22512 2015-12-29 17:16:44Z samuellb $
25   */
26  public interface StatedumpSession {
27      
28      static final String STATEDUMP_MODULE = "statedump-ejb";
29  
30      /**
31       * Performs a dry run import (nothing is done) of a statedump, and returns an object with lists of
32       * all conflicts due to already existing items and items that need a password to be specified.  
33       * 
34       * @param admin The authentication token. Must have / access, and of course access to create each item to import.
35       * @param options Options. Must specify a location.
36       * @return Result object with lists of conflicts and items that need a password.
37       * @throws AuthorizationDeniedException fail
38       * @throws IOException If the files in the dump are malformed.
39       */
40      StatedumpImportResult performDryRun(AuthenticationToken admin, StatedumpImportOptions options) throws AuthorizationDeniedException, IOException;
41      
42      /**
43       * Performs an import of a statedump.  
44       * 
45       * @param admin The authentication token. Must have / access, and of course access to create each item to import.
46       * @param options Options. Specifies location, lists of items to overwrite / not overwrite, passwords and other options.
47       * @return Result object with lists of conflicts and items that need a password.
48       * @throws AuthorizationDeniedException fail
49       * @throws IOException If the files in the dump are malformed.
50       */
51      StatedumpImportResult performImport(AuthenticationToken admin, StatedumpImportOptions options) throws AuthorizationDeniedException, IOException;
52  
53      
54      String getTemplatesBasedir(AuthenticationToken admin) throws AuthorizationDeniedException;
55  
56      /**
57       * Lists all available statedumps in the configured statedump templates directory.
58       * @param admin admin
59       * 
60       * @return A map from the directory names (not full path) to language strings to be used as descriptions.
61       * @throws AuthorizationDeniedException fail
62       */
63      Map<String, String> getAvailableTemplates(AuthenticationToken admin) throws AuthorizationDeniedException;
64  
65  }
