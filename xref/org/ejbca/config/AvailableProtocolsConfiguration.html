14  package org.ejbca.config;
15  
16  import java.io.Serializable;
17  import java.util.HashMap;
18  import java.util.LinkedHashMap;
19  import java.util.Map;
20  
21  import org.cesecore.configuration.ConfigurationBase;
22  
23  
24  /**
25   * Handles configuration of protocols supporting enable / disable
26   *
27   * @version $Id: AvailableProtocolsConfiguration.java 29241 2018-06-15 15:08:02Z henriks $
28   *
29   */
30  public class AvailableProtocolsConfiguration extends ConfigurationBase implements Serializable {
31      private static final long serialVersionUID = 1L;
32  
33      public final static String CONFIGURATION_ID = "AVAILABLE_PROTOCOLS";
34  
35      /**
36       * Protocols currently supporting enable/disable configuration by EJBCA
37       */
38      public enum AvailableProtocols {
39          // If you add a protocol > 6.11.0 it should be disabled by default
40          ACME("ACME", "/ejbca/acme"),
41          CERT_STORE("Certstore", "/certificates"),
42          CMP("CMP", "/ejbca/publicweb/cmp"),
43          CRL_STORE("CRLstore", "/crls"),
44          EST("EST", "/.well-known/est"),
45          OCSP("OCSP", "/ejbca/publicweb/status/ocsp"),
46          PUBLIC_WEB("Public Web", "/ejbca"),
47          SCEP("SCEP", "/ejbca/publicweb/apply/scep"),
48          RA_WEB("RA Web", "/ejbca/ra"),
49          REST("REST Certificate Management", "/ejbca/ejbca-rest-api"),
50          WEB_DIST("Webdist", "/ejbca/publicweb/webdist"),
51          WS("Web Service", "/ejbca/ejbcaws");
52  
53          private final String name;
54          private final String url;
55          private static final Map<String, String> reverseLookupMap = new HashMap<>();
56          
57          static {
58              for (final AvailableProtocols protocol : AvailableProtocols.values()) {
59                  reverseLookupMap.put(protocol.getName(), protocol.getUrl());
60              }
61          }
62          
63          /**
64           * Creates a new instance of an available protocol enum.
65           * @param name the name of the enum, same as the "serviceName" from web.xml
66           * @param url the URL to the servlet
67           */
68          private AvailableProtocols(final String name, final String url) {
69              this.name = name;
70              this.url = url;
71          }
72  
73          /** @return user friendly name of protocol */
74          public String getName() {
75              return name;
76          }
77  
78          public String getUrl() {
79              return url;
80          }
81          
82          public static String getContextPathByName(String name) {
83              return reverseLookupMap.get(name);
84          }
85      };
86  
87      /** Initializes the configuration */
88      public AvailableProtocolsConfiguration() {
89          super();
90      }
91  
92      /**
93       * Checks whether protocol is enabled / disabled locally
94       * @param protocol to check status of @see {@link AvailableProtocols}
95       * @return true if protocol is enabled, false otherwise
96       */
97      public boolean getProtocolStatus(String protocol) {
98          Boolean ret = (Boolean)data.get(protocol);
99          // All protocols added > 6.11.0 should be disabled by default
100         if (ret == null && (
101                 protocol.equals(AvailableProtocols.ACME.getName())  ||
102                 protocol.equals(AvailableProtocols.EST.getName())   || 
103                 protocol.equals(AvailableProtocols.REST.getName()))) {
104             setProtocolStatus(protocol, false);
105             return false;
106         }
107         return ret == null ? true : ret;
108     }
109 
110     public void setProtocolStatus(String protocol, boolean status) {
111         data.put(protocol, status);
112     }
113 
114     /** @return map containing the current status of all configurable protocols. */
115     public LinkedHashMap<String, Boolean> getAllProtocolsAndStatus() {
116         LinkedHashMap<String, Boolean> protocolStatusMap = new LinkedHashMap<>();
117         for (AvailableProtocols protocol : AvailableProtocols.values()) {
118             protocolStatusMap.put(protocol.getName(), getProtocolStatus(protocol.getName()));
119         }
120         return protocolStatusMap;
121     }
122 
123     @Override
124     public String getConfigurationId() {
125         return CONFIGURATION_ID;
126     }
127 
128     @Override
129     public void upgrade() {
130         if(Float.compare(LATEST_VERSION, getVersion()) != 0) {
131             data.put(VERSION,  Float.valueOf(LATEST_VERSION));
132         }
133     }
134 }
