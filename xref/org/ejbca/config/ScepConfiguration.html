117      * @param alias Alias*/
118     public void initialize(String alias){
119         alias += ".";
120         if(StringUtils.isNotEmpty(alias)) {
121             data.put(alias + SCEP_OPERATIONMODE, DEFAULT_OPERATION_MODE);
122             data.put(alias + SCEP_INCLUDE_CA, DEFAULT_INCLUDE_CA);
123             data.put(alias + SCEP_RA_CERTPROFILE, DEFAULT_RA_CERTPROFILE);
124             data.put(alias + SCEP_RA_ENTITYPROFILE, DEFAULT_RA_ENTITYPROFILE);
125             data.put(alias + SCEP_RA_DEFAULTCA, DEFAULT_RA_DEFAULTCA);
126             data.put(alias + SCEP_RA_AUTHPWD, DEFAULT_RA_AUTHPWD);
127             data.put(alias + SCEP_RA_NAME_GENERATION_SCHEME, DEFAULT_RA_NAME_GENERATION_SCHEME);
128             data.put(alias + SCEP_RA_NAME_GENERATION_PARAMETERS, DEFAULT_RA_NAME_GENERATION_PARAMETERS);
129             data.put(alias + SCEP_RA_NAME_GENERATION_PREFIX, DEFAULT_RA_NAME_GENERATION_PREFIX);
130             data.put(alias + SCEP_RA_NAME_GENERATION_POSTFIX, DEFAULT_RA_NAME_GENERATION_POSTFIX);

155     /**
156      * Client Certificate Renewal is defined in the SCEP draft as the capability of a certificate enrollment request to be interpreted as a 
157      * certificate renewal request if the previous certificate has passed half its validity. 
158      * 
159      * @param alias A SCEP configuration alias
160      * @return true of SCEP Client Certificate Renewal is enabled
161      */
162     public boolean getClientCertificateRenewal(final String alias) {
163         String key = alias + "." + SCEP_CLIENT_CERTIFICATE_RENEWAL;
164         String value = getValue(key, alias);
165         //Lazy initialization for SCEP configurations older than 6.3.1
166         if(value == null) {
167             data.put(alias + "." + SCEP_CLIENT_CERTIFICATE_RENEWAL, DEFAULT_CLIENT_CERTIFICATE_RENEWAL);
168             return Boolean.getBoolean(DEFAULT_CLIENT_CERTIFICATE_RENEWAL);
169         }
170         return Boolean.valueOf(value);
171     }

178     public void setClientCertificateRenewal(String alias, boolean clientCertificateRenewal) {

184      * @see ScepConfiguration#getClientCertificateRenewal(String) for information about Client Certificate Renewal
185      * 
186      * The SCEP draft makes it optional whether or not old keys may be reused during Client Certificate Renewal
187      * 
188      * @param alias A SCEP configuration alias
189      * @return true of old keys are allowed Client Certificate Renewal
190      */
191     public boolean getAllowClientCertificateRenewalWithOldKey(final String alias) {
192         String key = alias + "." + SCEP_CLIENT_CERTIFICATE_RENEWAL_WITH_OLD_KEY;
193         String value = getValue(key, alias);
194         //Lazy initialization for SCEP configurations older than 6.3.1
195         if(value == null) {
196             data.put(alias + "." + SCEP_CLIENT_CERTIFICATE_RENEWAL_WITH_OLD_KEY, DEFAULT_ALLOW_CLIENT_CERTIFICATE_RENEWAL_WITH_OLD_KEY);
197             return Boolean.getBoolean(DEFAULT_ALLOW_CLIENT_CERTIFICATE_RENEWAL_WITH_OLD_KEY);
198         }
199         return Boolean.valueOf(value);
200     }

313     public String getValue(String key, String alias) {
314         if(aliasExists(alias)) {
315             if(data.containsKey(key)) {
316                 return (String) data.get(key);
317             } else {
318                 log.error("Could not find key '" + key + "' in the SCEP configuration data");
319             }
320         } else {
321             log.error("SCEP alias '" + alias + "' does not exist");
322         }
323         return null;
324     }
325     public void setValue(String key, String value, String alias) {
326         if(aliasExists(alias)) {
327             if(data.containsKey(key)) {
328                 data.put(key, value);
329                 if(log.isDebugEnabled()) {
330                     log.debug("Added '" + key + "=" + value + "' to the SCEP configuration data");
331                 }
332             } else {
333                 log.error("Key '" + key + "' does not exist in the SCEP configuration data");
334             }
335         } else {
336             log.error("SCEP alias '" + alias + "' does not exist");
337         }
338     }

347     public Set<String> getAliasList() {
348         @SuppressWarnings("unchecked")
349         Set<String> ret = (Set<String>) data.get(ALIAS_LIST);
350         return (ret == null ? DEFAULT_ALIAS_LIST : ret);
351     }
352     public boolean aliasExists(String alias) {
353         if(StringUtils.isNotEmpty(alias)) {
354             Set<String> aliases = getAliasList();
355             return aliases.contains(alias);
356         }
357         return false;
358     }

360     public void addAlias(String alias) {
361         if(log.isDebugEnabled()) {
362             log.debug("Adding SCEP alias: " + alias);
363         }   
364             
365         if(StringUtils.isEmpty(alias)) {
366             if(log.isDebugEnabled()) {
367                 log.debug("No alias is added because no alias was provided.");
368             }
369             return;
370         }
371             
372         Set<String> aliases = getAliasList();
373         if(aliases.contains(alias)) {
374             if(log.isDebugEnabled()) {
375                 log.debug("SCEP alias '" + alias + "' already exists.");
376             }
377             return;
378         }
379         
380         initialize(alias);
381         aliases.add(alias);
382         data.put(ALIAS_LIST, aliases);
383     }
384     public void removeAlias(String alias) {
385         if(log.isDebugEnabled()) {
386             log.debug("Removing SCEP alias: " + alias);
387         }
388         
389         if(StringUtils.isEmpty(alias)) {
390             if(log.isDebugEnabled()) {
391                 log.debug("No alias is removed because no alias was provided.");
392             }
393             return;
394         }
395         
396         Set<String> aliases = getAliasList();
397         if(!aliases.contains(alias)) {
398             if(log.isDebugEnabled()) {
399                 log.debug("SCEP alias '" + alias + "' does not exist");
400             }
401             return;
402         }
403         
404         Set<String> removeKeys = getAllAliasKeys(alias);
405         Iterator<String> itr = removeKeys.iterator();
406         while(itr.hasNext()) {
407             String key = itr.next();
408             data.remove(key);
409         }
410         aliases.remove(alias);
411         data.put(ALIAS_LIST, aliases);
412     }
413     public void renameAlias(String oldAlias, String newAlias) {
414         if(log.isDebugEnabled()) {
415             log.debug("Renaming SCEP alias '" + oldAlias + "' to '" + newAlias + "'");
416         }
417         
418         if(StringUtils.isEmpty(oldAlias) || StringUtils.isEmpty(newAlias)) {
419             log.info("No alias is renamed because one or both aliases were not provided.");
420             return;
421         }
422         
423         Set<String> aliases = getAliasList();
424         if(!aliases.contains(oldAlias)) {
425             log.info("Cannot rename. SCEP alias '" + oldAlias + "' does not exists.");
426             return;
427         }
428         
429         if(aliases.contains(newAlias)) {
430             log.info("Cannot rename. SCEP alias '" + newAlias + "' already exists.");
431             return;
432         }
433         
434         Set<String> oldKeys = getAllAliasKeys(oldAlias);
435         Iterator<String> itr = oldKeys.iterator();
436         while(itr.hasNext()) {
437             String oldkey = itr.next();
438             String newkey = oldkey;
439             newkey = StringUtils.replace(newkey, oldAlias, newAlias);
440             Object value = data.get(oldkey);
441             data.put(newkey, value);
442         }
443         removeAlias(oldAlias);
444         aliases.remove(oldAlias);
445         aliases.add(newAlias);
446         data.put(ALIAS_LIST, aliases);
447     }
448     public void cloneAlias(String originAlias, String cloneAlias) {
449         if(log.isDebugEnabled()) {
450             log.debug("Cloning SCEP alias '" + originAlias + "' to '" + cloneAlias + "'");
451         }
452         
453         if(StringUtils.isEmpty(originAlias) || StringUtils.isEmpty(cloneAlias)) {
454             log.info("No alias is cloned because one or both aliased were not provided");
455             return;
456         }
457         
458         Set<String> aliases = getAliasList();
459         if(!aliases.contains(originAlias)) {
460             log.info("Cannot clone. SCEP alias '" + originAlias + "' does not exist.");
461             return;
462         }
463         
464         if(aliases.contains(cloneAlias)) {
465             log.info("Cannot clone. SCEP alias '" + cloneAlias + "' already exists.");
466             return;
467         }
468         
469         Iterator<String> itr = getAllAliasKeys(originAlias).iterator();
470         while(itr.hasNext()) {
471             String originalKey = itr.next();
472             String cloneKey = originalKey;
473             cloneKey = StringUtils.replace(cloneKey, originAlias, cloneAlias);
474             Object value = data.get(originalKey);
475             data.put(cloneKey, value);
476         }
477         aliases.add(cloneAlias);
478         data.put(ALIAS_LIST, aliases);
479     }

483     /**
484      * @return the configuration as a regular Properties object
485      */
486     public Properties getAsProperties() {
487         final Properties properties = new Properties();
488         Set<String> aliases = getAliasList();
489         Iterator<String> itr = aliases.iterator();
490         while(itr.hasNext()) {
491             String alias = itr.next();
492             Properties aliasp = getAsProperties(alias);
493             properties.putAll(aliasp);
494         }   
495         return properties;
496     }
497     
498     public Properties getAsProperties(String alias) {
499         if(aliasExists(alias)) {
500             final Properties properties = new Properties();
501             final Iterator<String> i = getAllAliasKeys(alias).iterator();
502             while (i.hasNext()) {
503                 final String key = i.next();
504                 final Object value = data.get(key);
505                 properties.setProperty(key, value == null? "" : value.toString());
506             }
507             return properties;
508         }
509         return null;
510     }
