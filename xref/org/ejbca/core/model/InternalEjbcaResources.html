13  package org.ejbca.core.model;
14  
15  import java.io.FileInputStream;
16  import java.io.FileNotFoundException;
17  import java.io.IOException;
18  import java.io.InputStream;
19  import java.util.Locale;
20  import java.util.Properties;
21  
22  import org.apache.log4j.Logger;
23  import org.cesecore.config.CesecoreConfiguration;
24  import org.cesecore.internal.InternalResources;
25  
26  /**
27   * Class managing internal localization of texts such as notification messages
28   * and log comments.
29   * 
30   * If fetched the resource files from the src/intresources directory and is
31   * included in the file ejbca-properties.jar
32   * 
33   * @version $Id: InternalEjbcaResources.java 26158 2017-07-18 05:46:19Z mikekushner $
34   */
35  public class InternalEjbcaResources extends InternalResources {
36  
37      private static final Logger log = Logger.getLogger(InternalEjbcaResources.class);
38  
39      /**
40       * Determines if a de-serialized file is compatible with this class.
41       * 
42       * Maintainers must change this value if and only if the new version of this
43       * class is not compatible with old versions. See Sun docs for <a
44       * href=http://java.sun.com/products/jdk/1.1/docs/guide
45       * /serialization/spec/version.doc.html> details. </a>
46       * 
47       */
48      private static final long serialVersionUID = -1001L;
49  
50      public static final String PREFEREDINTERNALRESOURCES = CesecoreConfiguration.getInternalResourcesPreferredLanguage();
51      public static final String SECONDARYINTERNALRESOURCES = CesecoreConfiguration.getInternalResourcesSecondaryLanguage();
52  
53      protected static InternalEjbcaResources instance = null;
54  
55      protected Properties primaryEjbcaResource = new Properties();
56      protected Properties secondaryEjbcaResource = new Properties();
57  
58      private static final String RESOURCE_PATH = "/intresources";
59      private static final String RESOURCE_NAME = "/ejbcaresources.";
60      private static final String RESOURCE_LOCATION = RESOURCE_PATH+RESOURCE_NAME;
61  
62      /**
63       * Method used to setup the Internal Resource management.
64     
65       */
66      protected InternalEjbcaResources() {
67      	super();
68          setupResources(RESOURCE_LOCATION);
69      }
70  
71      protected InternalEjbcaResources(String resPath) {
72      	super(resPath);
73          setupResources(resPath+RESOURCE_NAME);
74      }
75  
76      private void setupResources(String resLocation) {
77          final String primaryLanguage = PREFEREDINTERNALRESOURCES.toLowerCase(Locale.ENGLISH);
78          final String secondaryLanguage = SECONDARYINTERNALRESOURCES.toLowerCase(Locale.ENGLISH);
79          // The test flag is defined when called from test code (junit)
80          InputStream primaryStream = null;
81          InputStream secondaryStream = null;
82          try {
83          	// We first check for presence of the file in the classpath, if it does not exist we also allow to have the
84          	// the file in the filesystem
85              primaryStream = InternalEjbcaResources.class.getResourceAsStream(resLocation + primaryLanguage + ".properties");
86              if (primaryStream == null) {
87              	try {
88              		primaryStream = new FileInputStream(resLocation + primaryLanguage + ".properties");
89                  } catch (FileNotFoundException e) {
90                      log.error("Localization files not found in InternalEjbcaResources: " +e.getMessage());
91                  }
92              }
93              secondaryStream = InternalEjbcaResources.class.getResourceAsStream(resLocation + secondaryLanguage + ".properties");
94              if (secondaryStream == null) {
95              	try {
96              		secondaryStream = new FileInputStream(resLocation + secondaryLanguage + ".properties");
97                  } catch (FileNotFoundException e) {
98                      log.error("Localization files not found in InternalEjbcaResources: " +  e.getMessage());
99                  }
100             }
101 
102             try {
103                 if (primaryStream != null) {
104                     primaryEjbcaResource.load(primaryStream);
105                 } else {
106                     log.error("primaryResourse == null");
107                 }
108                 if (secondaryStream != null) {
109                     secondaryEjbcaResource.load(secondaryStream);
110                 } else {
111                     log.error("secondaryResource == null");
112                 }
113             } catch (IOException e) {
114                 log.error("Error reading internal resourcefile", e);
115             }
116         } finally {
117             try {
118                 if (primaryStream != null) {
119                     primaryStream.close();
120                 }
121                 if (secondaryStream != null) {
122                     secondaryStream.close();
123                 }
124             } catch (IOException e) {
125                 log.error("Error closing internal resources language streams: ", e);
126             }
127         }
128     }
129 
130     /** @return an instance of the InternalEjbcaResources. */
131     public static synchronized InternalEjbcaResources getInstance() {
132         if (instance == null) {
133             instance = new InternalEjbcaResources();
134         }
135         return instance;
136     }
137 
138     @Override
139     public String getLocalizedMessage(final String key, final Object... params) {
140         return getLocalizedMessageCs(key, params).toString();
141     }
142 
143     @Override
144     protected CharSequence getLocalizedMessageCs(final String key, final Object... params) {
145         final StringBuilder sb = new StringBuilder();
146         if (primaryEjbcaResource.containsKey(key)) {
147             sb.append(primaryEjbcaResource.getProperty(key));
148         } else if (secondaryEjbcaResource.containsKey(key)) {
149             sb.append(secondaryEjbcaResource.getProperty(key));
150         }
151         return getLocalizedMessageInternal(sb, key, params);
152     }
153 }
