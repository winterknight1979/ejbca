52  	public HardTokenEncryptCAService(final ExtendedCAServiceInfo serviceinfo)  {
53  		super(serviceinfo);
54  		log.debug("HardTokenEncryptCAService : constructor " + serviceinfo.getStatus());
55  		CryptoProviderTools.installBCProviderIfNotAvailable();
56  		data = new LinkedHashMap<Object, Object>();
57  		data.put(ExtendedCAServiceInfo.IMPLEMENTATIONCLASS, this.getClass().getName());
58  		data.put(EXTENDEDCASERVICETYPE, Integer.valueOf(ExtendedCAServiceTypes.TYPE_HARDTOKENENCEXTENDEDSERVICE));
59  		data.put(VERSION, Float.valueOf(LATEST_VERSION));
60  		setStatus(serviceinfo.getStatus());
61  	}

63  	public HardTokenEncryptCAService(final HashMap<?, ?> data) {
64  		super(data);
65  		CryptoProviderTools.installBCProviderIfNotAvailable();
66  		loadData(data);
67  	}

80  		setStatus(serviceinfo.getStatus());
81  		setCA(ca);
82  	}

131 	@Override
132 	public void upgrade() {
133 		if (Float.compare(LATEST_VERSION, getVersion()) != 0) {
134 			String msg = intres.getLocalizedMessage("caservice.upgrade", Float.valueOf(getVersion()));
