56  	public KeyRecoveryCAService(final ExtendedCAServiceInfo serviceinfo)  {
57  		super(serviceinfo);
58  		if (log.isDebugEnabled()) {
59  		    log.debug("KeyRecoveryCAService : constructor " + serviceinfo.getStatus());
60  		}
61  		CryptoProviderTools.installBCProviderIfNotAvailable();
62  		data = new LinkedHashMap<Object, Object>();
63  		data.put(ExtendedCAServiceInfo.IMPLEMENTATIONCLASS, this.getClass().getName());
64  		data.put(EXTENDEDCASERVICETYPE, Integer.valueOf(ExtendedCAServiceTypes.TYPE_KEYRECOVERYEXTENDEDSERVICE));
65  		data.put(VERSION, Float.valueOf(LATEST_VERSION));
66  		setStatus(serviceinfo.getStatus());
67  	}

70  		super(data);

72  		loadData(data);

97  		    log.trace(">extendedService");

100 			String msg = intres.getLocalizedMessage("caservice.notactive", "KeyRecovery");

103 		}

105 			throw new IllegalExtendedCAServiceRequestException("Not a KeyRecoveryCAServiceRequest: "+request.getClass().getName());

111 			try {
112 			    final String keyAlias = getCa().getCAToken().getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_KEYENCRYPT);
113                 if (log.isDebugEnabled()) {
114                     log.debug("Encrypting using alias '"+keyAlias+"' from crypto token "+cryptoToken.getId());

148 				        if (!StringUtils.equals(keyAlias, defaultAlias)) {

150 	                            log.debug("Trying to decrypt using default alias '"+defaultAlias+"' from crypto token "+cryptoToken.getId());
151 	                        }
152 	                        keys = X509CA.decryptKeys(cryptoToken, defaultAlias, serviceReq.getKeyData());

187 	@Override
188 	public void upgrade() {
189 		if (Float.compare(LATEST_VERSION, getVersion()) != 0) {
190 			String msg = intres.getLocalizedMessage("caservice.upgrade", Float.valueOf(getVersion()));
191 			log.info(msg);
192 			data.put(VERSION, Float.valueOf(LATEST_VERSION));
193 		}
