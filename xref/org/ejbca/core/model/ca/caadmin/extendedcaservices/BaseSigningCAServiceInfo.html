14  package org.ejbca.core.model.ca.caadmin.extendedcaservices;
15  
16  import java.io.Serializable;
17  import java.security.cert.Certificate;
18  import java.util.List;
19  
20  import org.cesecore.certificates.ca.extendedservices.ExtendedCAServiceInfo;
21  import org.cesecore.certificates.util.AlgorithmConstants;
22  
23  /**
24   * Base class for CAServiceInfo used by extended services that does signing 
25   * 
26   * @version $Id: BaseSigningCAServiceInfo.java 19901 2014-09-30 14:29:38Z anatom $
27   */
28  public abstract class BaseSigningCAServiceInfo extends ExtendedCAServiceInfo implements Serializable {    
29         
30      private static final long serialVersionUID = -6607852949410303766L;
31      private String subjectdn      = null;
32      private String subjectaltname = null;   
33  	private String keyspec        = "1024"; // Default key length
34      private String keyalgorithm   = AlgorithmConstants.KEYALGORITHM_RSA; // Default key algo
35      private List<Certificate>   certchain  = null;
36      
37      private boolean renew = false;
38             
39      /** Used when creating new service. 
40       * @param status Status
41       * @param subjectdn DN
42       * @param subjectaltname Name 
43       * @param keyspec Spec
44       * @param keyalgorithm Algo */
45      public BaseSigningCAServiceInfo(int status, String subjectdn, String subjectaltname, String keyspec, String keyalgorithm) {
46      	super(status);
47      	this.subjectdn = subjectdn;
48      	this.subjectaltname = subjectaltname;
49      	this.keyspec = keyspec;
50      	this.keyalgorithm = keyalgorithm; 	 
51      }
52      
53  	/** Used when returning information from service. 
54  	 * @param status Status
55  	 * @param subjectdn DN
56  	 * @param subjectaltname Name 
57  	 * @param keyspec Spec
58  	 * @param keyalgorithm Algo 
59  	 * @param certpath Certs */
60  	public BaseSigningCAServiceInfo(int status, String subjectdn, String subjectaltname, String keyspec, String keyalgorithm, List<Certificate> certpath) {
61  		super(status);
62  		this.subjectdn = subjectdn;
63  		this.subjectaltname = subjectaltname;
64  		this.keyspec = keyspec;
65  		this.keyalgorithm = keyalgorithm;
66  		this.certchain = certpath;
67  	}    
68      
69      /* Used when updating existing services, only status is used. */
70      public BaseSigningCAServiceInfo(int status, boolean renew){
71      	super(status);
72      	this.renew = renew;
73      }
74      
75      public String getSubjectDN(){ return this.subjectdn; }
76      public String getSubjectAltName(){ return this.subjectaltname; }
77      public String getKeySpec(){ return this.keyspec; }
78      public String getKeyAlgorithm(){ return this.keyalgorithm; }
79      public boolean getRenewFlag(){ return this.renew; } 
80      public List<Certificate> getCertificatePath(){ return this.certchain;}   
81  }
