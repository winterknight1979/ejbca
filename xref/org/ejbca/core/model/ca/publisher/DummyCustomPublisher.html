14  package org.ejbca.core.model.ca.publisher;
15  
16  import java.io.Serializable;
17  import java.security.cert.Certificate;
18  import java.util.Properties;
19  
20  import org.apache.log4j.Logger;
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.certificates.certificate.CertificateWrapper;
23  import org.cesecore.certificates.endentity.ExtendedInformation;
24  import org.cesecore.util.EJBTools;
25  
26  
27  /**
28   * This is a class used for testing and example purposes.
29   * It is supposed to illustrate how to implement a custom publisher in EJBCA 3.
30   *  
31   *
32   * @version $Id: DummyCustomPublisher.java 23421 2016-05-12 13:16:22Z mikekushner $
33   */
34  public class DummyCustomPublisher implements ICustomPublisher, Serializable {
35      		
36      private static final long serialVersionUID = 1L;
37  
38      private static Logger log = Logger.getLogger(DummyCustomPublisher.class);
39  
40      private CertificateWrapper certificate = null;
41  
42      
43      /**
44       * Creates a new instance of DummyCustomPublisher
45       */
46      public DummyCustomPublisher() {}
47  
48  	/**
49  	 * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#init(java.util.Properties)
50  	 */
51  	public void init(Properties properties) {
52  	  log.debug("Initializing DummyCustomPublisher " + properties.getProperty(BasePublisher.DESCRIPTION, ""));
53  	}
54  
55  	/**
56  	 * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#storeCertificate
57  	 */
58  	public boolean storeCertificate(AuthenticationToken admin, Certificate incert, String username, String password, String userDN, String cafp, int status, int type, long revocationDate, int revocationReason, String tag, int certificateProfileId, long lastUpdate, ExtendedInformation extendedinformation) throws PublisherException {
59          log.debug("DummyCustomPublisher, Storing Certificate for user: " + username);	
60          this.certificate = EJBTools.wrap(incert);
61          return true;
62      }
63      
64      public Certificate getStoredCertificate() {
65          return EJBTools.unwrap(certificate);
66      }
67  	/**
68  	 * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#storeCRL
69  	 */
70  	public boolean storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String userDN) throws PublisherException {
71          log.debug("DummyCustomPublisher, Storing CRL");
72  		return true;
73  	}
74  
75  	/**
76  	 * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#testConnection()
77  	 */
78  	public void testConnection() throws PublisherConnectionException {
79          log.debug("DummyCustomPublisher, Testing connection");			
80  	}
81  
82      @Override
83      public boolean willPublishCertificate(int status, int revocationReason) {
84          return true;
85      }
86      
87      @Override
88      public boolean isReadOnly() {
89          return false;
90      }
91  	
92  }
