180 	/**
181 	 * Publishes certificate in LDAP, if the certificate is not revoked. If the certificate is revoked, nothing is done
182 	 * and the publishing is counted as successful (i.e. returns true).
183 	 * 
184 	 * @see org.ejbca.core.model.ca.publisher.BasePublisher#storeCertificate
185 	 */    
186 	public boolean storeCertificate(AuthenticationToken admin, Certificate incert, String username, String password, String userDN, String cafp, int status, int type, long revocationDate, int revocationReason, String tag, int certificateProfileId, long lastUpdate, ExtendedInformation extendedinformation) throws PublisherException{
187 		if (log.isTraceEnabled()) {
188 			log.trace(">storeCertificate(username="+username+")");
189 		}
190 
191 		if (status == CertificateConstants.CERT_REVOKED) {
192         	// Call separate script for revocation
193         	revokeCertificate(admin, incert, username, revocationReason, userDN);
194         } else if (status == CertificateConstants.CERT_ACTIVE) {
195             // Don't publish non-active certificates
196     		int ldapVersion = LDAPConnection.LDAP_V3;
197     		LDAPConnection lc = createLdapConnection();
198 
199     		final String dn;
200     		final String certdn;
201     		try {
202     			// Extract the users DN from the cert.
203     			certdn = CertTools.getSubjectDN(incert);
204     			if (log.isDebugEnabled()) {
205     				log.debug( "Constructing DN for: " + username);
206     			}
207     			dn = constructLDAPDN(certdn, userDN);
208     			if (log.isDebugEnabled()) {
209     				log.debug("LDAP DN for user " +username +" is '" + dn+"'");
210     			}
211     		} catch (Exception e) {
212     			String msg = intres.getLocalizedMessage("publisher.errorldapdecode", "certificate");
213     			log.error(msg, e);            
214     			throw new PublisherException(msg);            
215     		}
216 
217     		// Extract the users email from the cert.
218     		String email = CertTools.getEMailAddress(incert);
219 
220     		// Check if the entry is already present, we will update it with the new certificate.
221     		// To work well with the LdapSearchPublisher we need to pass the full certificate DN to the 
222     		// search function, and not only the LDAP DN. The regular publisher should only use the LDAP DN though, 
223     		// but the searchOldEntity function will take care of that.
224     		LDAPEntry oldEntry = searchOldEntity(username, ldapVersion, lc, certdn, userDN, email);
225 
226     		// PART 2: Create LDAP entry
227     		LDAPEntry newEntry = null;
228     		ArrayList<LDAPModification> modSet = new ArrayList<LDAPModification>();
229     		LDAPAttributeSet attributeSet = null;
230     		String attribute = null;
231     		String objectclass = null;
232 
233     		if (type == CertificateConstants.CERTTYPE_ENDENTITY) {
234     			if (log.isDebugEnabled()) {
235     				log.debug("Publishing end user certificate to first available server of " + getHostnames());
236     			}
237     			if (oldEntry != null) {
238     				modSet = getModificationSet(oldEntry, certdn, email, ADD_MODIFICATION_ATTRIBUTES, true, password, incert);
239     			} else {
240     				objectclass = getUserObjectClass(); // just used for logging
241     				attributeSet = getAttributeSet(incert, getUserObjectClass(), certdn, email, true, true, password, extendedinformation);
242     			}
243 
244     			try {
245     				attribute = getUserCertAttribute();
246     				LDAPAttribute certAttr = new LDAPAttribute(getUserCertAttribute(), incert.getEncoded());
247     				if (oldEntry != null) {
248     					String oldDn = oldEntry.getDN();
249     					if (getAddMultipleCertificates()) {
250     						modSet.add(new LDAPModification(LDAPModification.ADD, certAttr));                        
251     						if (log.isDebugEnabled()) {
252     							log.debug("Appended new certificate in user entry; " + username+": "+oldDn);
253     						}
254     					} else {
255     						modSet.add(new LDAPModification(LDAPModification.REPLACE, certAttr));                                            
256     						if (log.isDebugEnabled()) {
257     							log.debug("Replaced certificate in user entry; " + username+": "+oldDn);
258     						}
259     					}
260     				} else {
261     					attributeSet.add(certAttr);
262     					if (log.isDebugEnabled()) {
263     						log.debug("Added new certificate to user entry; " + username+": "+dn);
264     					}
265     				}
266     			} catch (CertificateEncodingException e) {
267     				String msg = intres.getLocalizedMessage("publisher.errorldapencodestore", "certificate");
268     				log.error(msg, e);
269     				throw new PublisherException(msg);                
270     			}
271     		} else if (type == CertificateConstants.CERTTYPE_SUBCA ||
272     		           type == CertificateConstants.CERTTYPE_ROOTCA) {
273     			if (log.isDebugEnabled()) {
274     				log.debug("Publishing CA certificate to first available server of " + getHostnames());
275     			}
276     			if (oldEntry != null) {
277     				modSet = getModificationSet(oldEntry, certdn, null, false, false, password, incert);
278     			} else {
279     				objectclass = getCAObjectClass(); // just used for logging
280     				attributeSet = getAttributeSet(incert, getCAObjectClass(), certdn, null, true, false, password, extendedinformation);
281     			}
282     			try {
283     				attribute = getCACertAttribute();
284     				LDAPAttribute certAttr = new LDAPAttribute(getCACertAttribute(), incert.getEncoded());
285     				if (oldEntry != null) {
286     					modSet.add(new LDAPModification(LDAPModification.REPLACE, certAttr));
287     				} else {
288     					attributeSet.add(certAttr);
289     					// Also create using the crlattribute, it may be required
290     					LDAPAttribute crlAttr = new LDAPAttribute(getCRLAttribute(), getFakeCRL());
291     					attributeSet.add(crlAttr);
292     					// Also create using the arlattribute, it may be required
293     					LDAPAttribute arlAttr = new LDAPAttribute(getARLAttribute(), getFakeCRL());
294     					attributeSet.add(arlAttr);
295     					if (log.isDebugEnabled()) {
296     						log.debug("Added (fake) attribute for CRL and ARL.");
297     					}
298     				}
299     			} catch (CertificateEncodingException e) {
300     				String msg = intres.getLocalizedMessage("publisher.errorldapencodestore", "certificate");
301     				log.error(msg, e);
302     				throw new PublisherException(msg);            
303     			}
304     		} else {
305     			String msg = intres.getLocalizedMessage("publisher.notpubltype", Integer.valueOf(type));
306     			log.info(msg);
307     			throw new PublisherException(msg);                      
308     		}
309 
310     		// PART 3: MODIFICATION AND ADDITION OF NEW USERS
311     		// Try all the listed servers
312     		Iterator<String> servers = getHostnameList().iterator();
313     		boolean connectionFailed;
314     		do {
315     			connectionFailed = false;
316     			String currentServer = servers.next();
317     			try {
318     				TCPTool.probeConnectionLDAP(currentServer, Integer.parseInt(getPort()), getConnectionTimeOut());	// Avoid waiting for halfdead-servers
319     				lc.connect(currentServer, Integer.parseInt(getPort()));
320     				// Execute a STARTTLS handshake if it was requested.
321     				if (getConnectionSecurity() == ConnectionSecurity.STARTTLS) {
322                         if (log.isDebugEnabled()) {
323                             log.debug("STARTTLS to LDAP server "+currentServer);
324                         }
325     				    lc.startTLS();
326     				}
327     				// authenticate to the server
328     				lc.bind(ldapVersion, getLoginDN(), getLoginPassword().getBytes("UTF8"), ldapBindConstraints);            
329     				// Add or modify the entry
330     				if (oldEntry != null && getModifyExistingUsers()) {
331     					LDAPModification[] mods = new LDAPModification[modSet.size()]; 
332     					mods = (LDAPModification[])modSet.toArray(mods);
333     					String oldDn = oldEntry.getDN();
334     					if (log.isDebugEnabled()) {
335     						log.debug("Writing modification to DN: "+oldDn);
336     					}
337     					lc.modify(oldDn, mods, ldapStoreConstraints);
338     					String msg = intres.getLocalizedMessage("publisher.ldapmodify", "CERT", oldDn);
339     					log.info(msg);  
340     				} else {
341     					if(this.getCreateNonExistingUsers()){     
342     						if (oldEntry == null) {           
343     							// Check if the intermediate parent node is present, and if it is not
344     							// we can create it, of allowed to do so by the publisher configuration
345     							if(getCreateIntermediateNodes()) {
346     								final String parentDN = CertTools.getParentDN(dn);
347     								try {
348     									lc.read(parentDN, ldapSearchConstraints);
349     								} catch(LDAPException e) {
350     									if(e.getResultCode() == LDAPException.NO_SUCH_OBJECT) {
351     										this.createIntermediateNodes(lc, dn);
352     										String msg = intres.getLocalizedMessage("publisher.ldapaddedintermediate", "CERT", parentDN);
353     										log.info(msg);
354     									}
355     								}
356     							}
357     							newEntry = new LDAPEntry(dn, attributeSet);
358     							if (log.isDebugEnabled()) {
359     								log.debug("Adding DN: "+dn);
360     							}
361     							lc.add(newEntry, ldapStoreConstraints);
362     							String msg = intres.getLocalizedMessage("publisher.ldapadd", "CERT", dn);
363     							log.info(msg);
364     						}
365     					}  
366     				}
367     			} catch (LDAPException e) {
368     				connectionFailed = true;
369     				// If multiple certificates are allowed per entity, and the certificate is already published, 
370     				// an exception will be thrown. Catch this type of exception and just log an informational message.
371     				if (e.getResultCode() == LDAPException.ATTRIBUTE_OR_VALUE_EXISTS) {
372                         final String msg = intres.getLocalizedMessage("publisher.certalreadyexists", CertTools.getFingerprintAsString(incert), dn, e.getMessage());
373     				    log.info(msg);
374     				} else if (servers.hasNext()) {
375     					log.warn("Failed to publish to " + currentServer + ". Trying next in list.");
376     				} else {
377     					String msg = intres.getLocalizedMessage("publisher.errorldapstore", "certificate", attribute, objectclass, dn, e.getMessage());
378     					log.error(msg, e);  
379     					throw new PublisherException(msg);            
380     				}
381     			} catch (UnsupportedEncodingException e) {
382     				String msg = intres.getLocalizedMessage("publisher.errorpassword", getLoginPassword());
383     				log.error(msg, e);
384     				throw new PublisherException(msg);            
385     			} finally {
386     				// disconnect with the server
387     				try {
388     					lc.disconnect(ldapDisconnectConstraints);
389     				} catch (LDAPException e) {
390     					String msg = intres.getLocalizedMessage("publisher.errordisconnect", getLoginPassword());
391     					log.error(msg, e);
392     				}
393     			}
394     		} while (connectionFailed && servers.hasNext()) ;
395         } else {
396 			String msg = intres.getLocalizedMessage("publisher.notpublwithstatus", Integer.valueOf(status));
397 			log.info(msg);        	
398         }
399 		if (log.isTraceEnabled()) {
400 			log.trace("<storeCertificate()");
401 		}
402 		return true;
403 	}

406 	 * Creates intermediate nodes to host an LDAP entry at <code>dn</code>.
407 	 * @param lc Active LDAP connection
408 	 * @param dn Distinguished name
409 	 * @throws PublisherException on fail
410 	 */
411 	private void createIntermediateNodes(LDAPConnection lc, String dn) throws PublisherException {
412 		LDAPAttributeSet attrSet;
413 		LDAPEntry entry;
414 	    for (String dnFragment : LdapTools.getIntermediateDNs(dn, getBaseDN())) {
415 			try {
416 				lc.read(dnFragment, ldapSearchConstraints);
417 			} catch(LDAPException e) {
418 				if(e.getResultCode() == LDAPException.NO_SUCH_OBJECT) {
419 				    final String rdn = LdapTools.getFirstDNComponent(dnFragment);
420 				    final String field = new String(rdn.substring(0, rdn.indexOf('=')));
421 				    final String value = new String(rdn.substring(rdn.indexOf('=') + 1));
422 				    
423 					attrSet = new LDAPAttributeSet();
424 					attrSet.add(getObjectClassAttribute(field));
425 					attrSet.add(new LDAPAttribute(field.toLowerCase(), value));
426 					entry = new LDAPEntry(dnFragment, attrSet);
427 
428 					try {
429 						lc.add(entry, ldapStoreConstraints);
430 						if (log.isDebugEnabled()) {
431 							log.debug("Created node " + dnFragment);
432 						}
433 					} catch(LDAPException e1) {
434 						String msg = intres.getLocalizedMessage("publisher.ldapaddedintermediate", dnFragment);
435 						log.error(msg, e1);
436 						throw new PublisherException(msg);            
437 					}
438 				}
439 			}
440 		}
441 	}

444 	 * Returns an LDAPAttribute initialized with the LDAP object class
445 	 * definition that corresponds to a DN <code>field</code>.
446 	 * <p>The only allowed fields are </code>O</code> (organization) and
447 	 * <code>OU</code> (organizationalUnit).</p>
448 	 *
449 	 * @param field A DN field (case-insensitive). Only <code>O</code> and
450 	 * <code>OU</code> are allowed. 
451 	 * @return LDAPAttribute initialized with the LDAP object class definition
452 	 * that corresponds to a DN <code>field</code>.
453 	 */
454 	private LDAPAttribute getObjectClassAttribute(String field) {
455 		final String lowCaseField = field.toLowerCase();
456 		if(lowCaseField.equals("o")) {
457 			return new LDAPAttribute("objectclass", new String[] { "top", "organization" });
458 		} else if(lowCaseField.equals("ou")) {
459 			return new LDAPAttribute("objectclass", new String[] { "top", "organizationalUnit" });
460 		} else {
461 			String msg = intres.getLocalizedMessage("publisher.ldapintermediatenotappropriate", field);
462 			log.warn(msg);
463 			return new LDAPAttribute("objectclass");
464 		}

468 	 * @see org.ejbca.core.model.ca.publisher.BasePublisher#storeCRL
469 	 */    
470 	public boolean storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String userDN) throws PublisherException{
471 		if (log.isTraceEnabled()) {
472 			log.trace(">storeCRL");
473 		}
474 		int ldapVersion = LDAPConnection.LDAP_V3;
475 
476 		final String dn;
477 		final String crldn;
478 		final boolean isDeltaCRL;
479 		try {
480 			// Extract the users DN from the crl. Use the least number of encodings...
481 			final X509CRL crl = CertTools.getCRLfromByteArray(incrl);
482 			crldn = CertTools.stringToBCDNString(crl.getIssuerDN().toString());
483 			// Is it a delta CRL?
484 			if (crl.getExtensionValue(Extension.deltaCRLIndicator.getId()) != null) {
485 				isDeltaCRL = true;
486 			} else {
487 				isDeltaCRL = false;
488 			}
489 			// Construct the DN used for the LDAP object entry
490 			dn = constructLDAPDN(crldn, userDN);
491 		} catch (Exception e) {
492 			String msg = intres.getLocalizedMessage("publisher.errorldapdecode", "CRL");
493 			log.error(msg, e);        	
494 			throw new PublisherException(msg);            
495 		}
496 
497 		LDAPConnection lc = createLdapConnection();
498 
499 		// Check if the entry is already present, we will update it with the new CRL.
500 		LDAPEntry oldEntry = searchOldEntity(null, ldapVersion, lc, crldn, userDN, null);
501 
502 		LDAPEntry newEntry = null;
503 		ArrayList<LDAPModification> modSet = new ArrayList<LDAPModification>();
504 		LDAPAttributeSet attributeSet = null;
505 
506 		if (oldEntry != null) {
507 			modSet = getModificationSet(oldEntry, crldn, null, false, false, null, null);
508 		} else {
509 			attributeSet = getAttributeSet(null, this.getCAObjectClass(), crldn, null, true, false, null,null);
510 		}
511 
512 		if(isDeltaCRL) {
513 			// It's a delta CRL.
514 			LDAPAttribute attr = new LDAPAttribute(getDeltaCRLAttribute(), incrl);
515 			if (oldEntry != null) {
516 				modSet.add(new LDAPModification(LDAPModification.REPLACE, attr));
517 			} else {
518 				attributeSet.add(attr);
519 			}
520 		} else {
521 			// It's a CRL
522 			LDAPAttribute crlAttr = new LDAPAttribute(getCRLAttribute(), incrl);
523 			LDAPAttribute arlAttr = new LDAPAttribute(getARLAttribute(), incrl);
524 			if (oldEntry != null) {
525 				modSet.add(new LDAPModification(LDAPModification.REPLACE, crlAttr));
526 				modSet.add(new LDAPModification(LDAPModification.REPLACE, arlAttr));
527 			} else {
528 				attributeSet.add(crlAttr);
529 				attributeSet.add(arlAttr);
530 			}
531 		}
532 		if (oldEntry == null) {
533 			newEntry = new LDAPEntry(dn, attributeSet);
534 		}
535 		// Try all the listed servers
536 		Iterator<String> servers = getHostnameList().iterator();
537 		boolean connectionFailed;
538 		do {
539 			connectionFailed = false;
540 			String currentServer = servers.next();
541 			try {
542 				TCPTool.probeConnectionLDAP(currentServer, Integer.parseInt(getPort()), getConnectionTimeOut());	// Avoid waiting for halfdead-servers
543 				// connect to the server
544 				lc.connect(currentServer, Integer.parseInt(getPort()));
545 				// Execute a STARTTLS handshake if it was requested.
546 				if (getConnectionSecurity() == ConnectionSecurity.STARTTLS) {
547                     if (log.isDebugEnabled()) {
548                         log.debug("STARTTLS to LDAP server "+currentServer);
549                     }
550 					lc.startTLS();
551 				}
552 				// authenticate to the server
553 				lc.bind(ldapVersion, getLoginDN(), getLoginPassword().getBytes("UTF8"), ldapBindConstraints);
554 				// Add or modify the entry
555 				if (oldEntry != null) {
556 					LDAPModification[] mods = new LDAPModification[modSet.size()]; 
557 					mods = (LDAPModification[])modSet.toArray(mods);
558 					lc.modify(dn, mods, ldapStoreConstraints);
559 					String msg = intres.getLocalizedMessage("publisher.ldapmodify", "CRL", dn);
560 					log.info(msg);  
561 				} else {
562 					lc.add(newEntry, ldapStoreConstraints);
563 					String msg = intres.getLocalizedMessage("publisher.ldapadd", "CRL", dn);
564 					log.info(msg);  
565 				}
566 			} catch (LDAPException e) {
567 				connectionFailed = true;
568 				if (servers.hasNext()) {
569 					log.warn("Failed to publish to " + currentServer + ". Trying next in list.");
570 				} else {
571 					String msg = intres.getLocalizedMessage("publisher.errorldapstore", "CRL", getCRLAttribute(), getCAObjectClass(), dn, e.getMessage());
572 					log.error(msg, e);  
573 					throw new PublisherException(msg);            
574 				}
575 			} catch (UnsupportedEncodingException e) {
576 				String msg = intres.getLocalizedMessage("publisher.errorpassword", getLoginPassword());
577 				log.error(msg, e);
578 				throw new PublisherException(msg);            
579 			} finally {
580 				// disconnect with the server
581 				try {
582 					lc.disconnect(ldapDisconnectConstraints);
583 				} catch (LDAPException e) {
584 					String msg = intres.getLocalizedMessage("publisher.errordisconnect");
585 					log.error(msg, e);
586 				}
587 			}
588 		} while (connectionFailed && servers.hasNext()) ;
589 		if (log.isTraceEnabled()) {
590 			log.trace("<storeCRL");
591 		}
592 		return true;
593 	}

596 	 * Revokes a certificate, which means for LDAP that we may remove the certificate or the whole user entry.
597 	 * @param admin Token
598 	 * 
599      * @param cert The certificate to be revoked.
600      * @param username Username of end entity owning the certificate.
601      * @param reason reason for revocation from RevokedCertInfo, RevokedCertInfo.NOT_REVOKED if not revoked.
602      * @param userDN if an DN object is not found in the certificate use object from user data instead.
603 	 * @throws PublisherException On fail
604 	 */    
605 	public void revokeCertificate(AuthenticationToken admin, Certificate cert, String username, int reason, String userDN) throws PublisherException {
606 		if (log.isTraceEnabled()) {
607 			log.trace(">revokeCertificate()");
608 		}
609 		// Check first if we should do anything then revoking
610 		boolean removecert = getRemoveRevokedCertificates();
611 		boolean removeuser = getRemoveUsersWhenCertRevoked();
612 		if ( (!removecert) && (!removeuser) ) {
613 			if (log.isDebugEnabled()) {
614 				log.debug("The configuration for the publisher '" + getDescription() + "' does not allow removing of certificates or users.");
615 			}
616 			return;
617 		}
618 		if (removecert) {
619 			if (log.isDebugEnabled()) {
620 				log.debug("Removing user certificate from ldap");
621 			}
622 		}
623 		if (removeuser) {
624 			if (log.isDebugEnabled()) {
625 				log.debug("Removing user entry from ldap");
626 			}
627 		}
628 
629 		int ldapVersion = LDAPConnection.LDAP_V3;
630 		LDAPConnection lc = createLdapConnection();
631 
632 		final String dn;
633 		final String certdn;
634 		try {
635 			// Extract the users DN from the cert.
636 			certdn = CertTools.getSubjectDN(cert);
637 			dn = constructLDAPDN(certdn, userDN);
638 		} catch (Exception e) {
639 			String msg = intres.getLocalizedMessage("publisher.errorldapdecode", "certificate");
640 			log.error(msg, e);            
641 			throw new PublisherException(msg);            
642 		}
643 
644 		// Extract the users email from the cert.
645 		String email = CertTools.getEMailAddress(cert);
646 
647 		// Check if the entry is already present, we will update it with the new certificate.
648 		final LDAPEntry oldEntry;
649 
650 		ArrayList<LDAPModification> modSet = null;
651 
652 		if (!CertTools.isCA(cert)) {
653 			oldEntry = searchOldEntity(username, ldapVersion, lc, certdn, userDN, email);
654 			if (log.isDebugEnabled()) {
655 				log.debug("Removing end user certificate from first available server of " + getHostnames());
656 			}
657 			if (oldEntry != null) {          
658 				if (removecert) {
659 					// Don't try to remove the cert if there does not exist any
660 					LDAPAttribute oldAttr = oldEntry.getAttribute(getUserCertAttribute());
661 					if (oldAttr != null) {
662 						modSet = getModificationSet(oldEntry, certdn, null, false, true, null, cert);
663 						LDAPAttribute attr = new LDAPAttribute(getUserCertAttribute());
664 						modSet.add(new LDAPModification(LDAPModification.DELETE, attr));                    
665 					} else {
666 						String msg = intres.getLocalizedMessage("publisher.inforevokenocert");
667 						log.info(msg);
668 					}            		
669 				}
670 			} else {
671 				String msg = intres.getLocalizedMessage("publisher.errorrevokenoentry");
672 				log.warn(msg);
673 			}
674 		} else  {
675 			oldEntry = null;
676 			// Removal of CA certificate isn't support because of object class restrictions
677 			if (log.isDebugEnabled()) {
678 				log.debug("Not removing CA certificate from first available server of " + getHostnames() + ", because of object class restrictions.");
679 			}
680 		}
681 
682 		// Try all the listed servers
683 		final Iterator<String> servers = getHostnameList().iterator();
684 		boolean isConnectionNotDone = true;
685         if (log.isDebugEnabled() && (oldEntry == null)) {
686             log.debug("Not modifying LDAP entry because there is no existing entry.");                      
687         }
688 		while ( oldEntry!=null && isConnectionNotDone && servers.hasNext()) {
689 			isConnectionNotDone = false;
690 			String currentServer = servers.next(); 
691 			if (log.isDebugEnabled()) {
692 				log.debug("currentServer: "+currentServer);
693 			}
694 			try {
695 				TCPTool.probeConnectionLDAP(currentServer, Integer.parseInt(getPort()), getConnectionTimeOut());	// Avoid waiting for halfdead-servers
696 				lc.connect(currentServer, Integer.parseInt(getPort()));
697 				// Execute a STARTTLS handshake if it was requested.
698 				if (getConnectionSecurity() == ConnectionSecurity.STARTTLS) {
699                     if (log.isDebugEnabled()) {
700                         log.debug("STARTTLS to LDAP server "+currentServer);
701                     }
702 					lc.startTLS();
703 				}
704 				// authenticate to the server
705 				lc.bind(ldapVersion, getLoginDN(), getLoginPassword().getBytes("UTF8"), ldapBindConstraints);            
706 				// Add or modify the entry
707 				if (modSet != null && getModifyExistingUsers()) {
708 					if (removecert) {
709 						LDAPModification[] mods = new LDAPModification[modSet.size()]; 
710 						mods = (LDAPModification[])modSet.toArray(mods);
711 						lc.modify(oldEntry.getDN(), mods, ldapStoreConstraints);            		
712 					}
713 					if (removeuser) {
714 						lc.delete(oldEntry.getDN(), ldapStoreConstraints);            		
715 					}
716 					String msg = intres.getLocalizedMessage("publisher.ldapremove", dn);
717 					log.info(msg);  
718 				} else {
719 					if (log.isDebugEnabled()) {
720 						if (modSet == null) {
721 							log.debug("Not modifying LDAP entry because we don't have anything to modify.");						
722 						}
723 						if (!getModifyExistingUsers()) {
724 							log.debug("Not modifying LDAP entry because we're not configured to do so.");						
725 						}
726 					}
727 				}
728 			} catch (LDAPException e) {
729 				isConnectionNotDone = true;
730 				if (servers.hasNext()) {
731 					log.warn("Failed to publish to " + currentServer + ". Trying next in list.");
732 				} else {
733 					String msg = intres.getLocalizedMessage("publisher.errorldapremove", dn);
734 					log.error(msg, e);  
735 					throw new PublisherException(msg);            
736 				}
737 			} catch (UnsupportedEncodingException e) {
738 				String msg = intres.getLocalizedMessage("publisher.errorpassword", getLoginPassword());
739 				log.error(msg, e);
740 				throw new PublisherException(msg);            
741 			} finally {
742 				// disconnect with the server
743 				try {
744 					lc.disconnect(ldapDisconnectConstraints);
745 				} catch (LDAPException e) {
746 					String msg = intres.getLocalizedMessage("publisher.errordisconnect");
747 					log.error(msg, e);
748 				}
749 			}
750 		}
751 		if (log.isTraceEnabled()) {
752 			log.trace("<revokeCertificate()");
753 		}
754 	}
