33  /**
34   * Publishes to multiple groups of publishers. For each group it publishes to one random publisher.
35   * <p>
36   * Useful when you have a lot of publishers, and you want to manage them in a single place.
37   * @version $Id: MultiGroupPublisher.java 34192 2020-01-07 15:10:21Z aminkh $
38   */

51          super();

68          if (cachedPublisherSession == null) {
69              cachedPublisherSession = new EjbLocalHelper().getPublisherSession();
70          }
71          return cachedPublisherSession;

74      private BasePublisher getPublisher(int publisherId) {
75          return getPublisherSession().getPublisher(publisherId);
76      }

78      /**
79       * Returns a list of publishers to use for some certificate or CRL.
80       * The result is randomized, so it will return a different result each time.
81       * This method uses the PublisherCache for better performance.
82       * @param getAll Return ALL publishers in each group, instead of only a random one.
83       * @return List of publishers to use.
84       */
85      private List<BasePublisher> getPublishersToUse(final boolean getAll) {
86          final List<BasePublisher> publishers = new ArrayList<>();
87          for (final TreeSet<Integer> group : getPublisherGroups()) {
88              if (getAll) {
89                  for (int publisherId : group) {
90                      final BasePublisher publisher = getPublisher(publisherId);
91                      if (publisher != null) {
92                          publishers.add(publisher);
93                      } else if (log.isDebugEnabled()) {
94                          log.debug("Ignoring non-existent publisher: " + publisherId);
95                      }
96                  }
97              } else {
98                  final List<Integer> ids = new ArrayList<>(group);
99                  while (!ids.isEmpty()) {
100                     // Grab a random publisher
101                     final int index = ThreadLocalRandom.current().nextInt(ids.size());
102                     final int publisherId = ids.get(index);
103                     final BasePublisher publisher = getPublisher(publisherId);
104                     if (publisher != null) {
105                         publishers.add(publisher);
106                         break;
107                     } else {
108                         // This happens when clicking "Test Connection", so it won't spam the logs
109                         log.warn("Ignoring non-existent publisher " + publisherId + " in publisher " + getName());
110                     }
111                     ids.remove(index);
112                 }
113             }
114         }
115         return publishers;
116     }

119     @Override
120     public boolean willPublishCertificate(int status, int revocationReason) {
121         log.trace(">willPublishCertificate");
122         // We don't know exactly which publishers storeCertificate will use,
123         // so we just check the "first" one in each group. ("first" means lowest ID)
124         for (final TreeSet<Integer> group : getPublisherGroups()) {
125             if (group.isEmpty()) {
126                 log.debug("An empty group was found in publisher '" + getName() + "'");
127             }
128             final int publisherId = group.first();
129             final BasePublisher publisher = getPublisher(publisherId);
130             if (publisher.willPublishCertificate(status, revocationReason)) {
131                 log.trace("<willPublishCertificate: true");
132                 return true;
133             }
134         }
135         log.trace("<willPublishCertificate: false");
136         return false;
137     }

151     @Override
152     public boolean storeCertificate(final AuthenticationToken authenticationToken, final CertificateData certificateData, final Base64CertData base64CertData, final String password, final String userDN, final ExtendedInformation extendedinformation) throws PublisherException {
153         log.trace(">storeCertificate");
154         final List<Integer> publisherIdsToUse = new ArrayList<>();
155         for (final BasePublisher publisher : getPublishersToUse(false)) {
156             final boolean willPublish = publisher.willPublishCertificate(certificateData.getStatus(), certificateData.getRevocationReason());
157             if (willPublish) {
158                 publisherIdsToUse.add(publisher.getPublisherId());
159             }
160             if (log.isDebugEnabled()) {
161                 log.debug("Will " + (willPublish ? "" : "NOT ") +"publish certificate " + certificateData.getSerialNumberHex() + " to publisher '" + publisher.getName() + "'");
162             }
163         }
164         if (!publisherIdsToUse.isEmpty()) {
165             try {
166                 getPublisherSession().storeCertificate(authenticationToken, publisherIdsToUse, new CertificateDataWrapper(null, certificateData, base64CertData), password, userDN, extendedinformation);
167             } catch (AuthorizationDeniedException e) {
168                 throw new PublisherException("Authorization was denied: " + e.getMessage());
169             }
170         }
171         log.trace("<storeCertificate");
172         return true;
173     }

180     @Override
181     public boolean storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String userDN) throws PublisherException {
182         final List<Integer> publisherIdsToUse = new ArrayList<>();
183         log.trace(">storeCRL");
184         for (final BasePublisher publisher : getPublishersToUse(false)) {
185             if (log.isDebugEnabled()) {
186                 log.debug("Will publish CRL " + number + " for CA " + cafp + " to publisher '" + publisher.getName() + "'");
187             }
188             publisherIdsToUse.add(publisher.getPublisherId());
189         }
190         if (publisherIdsToUse.isEmpty()) {
191             log.info("No publishers available in multi group publisher '" + getName() + "'. Can't publish CRL " + number + " for CA " + cafp);
192             return false;
193         }
194         try {
195             getPublisherSession().storeCRL(admin, publisherIdsToUse, incrl, cafp, number, userDN);
196         } catch (AuthorizationDeniedException e) {
197             throw new PublisherException("Authorization was denied: " + e.getMessage());
198         }
199         log.trace("<storeCRL");
200         return true;
201     }

203     @Override
204     public void testConnection() throws PublisherConnectionException {
205         Exception publisherException = null;
206         List<String> failedNames = new ArrayList<>();
207         log.debug("Testing all publishers in multi group publisher.");
208         for (final BasePublisher publisher : getPublishersToUse(true)) {
209             if (log.isDebugEnabled()) {
210                 log.debug("Testing publisher: " + publisher.getName());
211             }
212             try {
213                 publisher.testConnection();
214             } catch (PublisherConnectionException | RuntimeException e) {
215                 if (log.isDebugEnabled()) {
216                     log.debug("Publisher '" + publisher.getName() + "' failed: " + e.getMessage(), e);
217                 }
218                 failedNames.add(publisher.getName());
219                 if (publisherException == null) {
220                     publisherException = e;
221                 }
222             }
223         }
224         log.debug("Done testing publishers in multi group publisher.");
225         if (publisherException != null) {
226             final String msg = "Publishers [" + StringUtils.join(failedNames, ", ") + "] failed. First failure: " +publisherException.getMessage();
227             log.info(msg, publisherException);
228             throw new PublisherConnectionException(msg, publisherException);
229         }
230     }
