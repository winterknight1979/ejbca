35  /**
36   * A custom LDAP publisher that publishes User Entries with Certificate Serial Number. 
37   * The purpose is that you can search for certificates in the LDAP directory on certificate serial number, and each certificate is stored in one entry in the LDAP directory.
38   *
39   * The publisher inherits the standard LDAP publisher, but always makes sure that UID=certSerno (in decimal format) is set in the UserDN that is passed to publishing.
40   * If the publisher is configured to use UID as "Use Fields in LDAP DN" this will be used in the LDAP DN of the entry created.
41   * Configuration options are the same as for the standard LDAP publisher, but must be set using properties, as custom publishers are configured with a properties field.
42   * 
43   * To build using ejbca-custom:
44   * 1. Create a directory ejbca-custom/modules/ejbca-common/src/org/ejbca/core/model/ca/publisher/custpubl2 on the same level as your ejbca directory:
45   *    ejbca_6_0_0
46   *    ejbca-custom
47   * 2. Copy CertSernoCustomLdapPublisher.java to this directory
48   * 3. Build and re-deploy EJBCA with 'ant clean; ant bootstrap'
49   * 
50   * To use:
51   * 1. In EJBCA 6 and later, configure web.properties with web.manualclasspathsenabled=true
52   * 2. Create a new Custom Publisher in EJBCA
53   * 3. Specify the class path: org.ejbca.core.model.ca.publisher.custpubl2.CertSernoCustomLdapPublisher
54   * 4. Set properties (use the example normal config below)
55   * 5. Save and test connection
56   * 
57   * A normal configuration that should work in most cases (just change values to match your LDAP server) looks like:
58   * 
59  hostname localhost
60  port 1389
61  baswdn dc=example,dc=com
62  logindn cn=Directory Manager
63  loginpassword foo123
64  usefieldsinldapdn 1
65   * 
66   * usefieldsinldapdn uses decimal values from DNFieldExtractor where the most important are: 1=UID, 2=CN, 8=OU, 9=O
67   * 
68   * With the configuration above certificate entries with LDAP DN "UID=123456789,dc=example,dc=com" (where 123456789 is the certificate serial number in decimal format)
69   * will be added to the LDAP directory. The tree of the base DN (dc=example,dc=com) must exist already.
70   * 
71   * If you want to use a custom LDAP schema, such as the inetOrgPersonWithCertSerno, you can set this property, for example:
72   * 
73  userobjectclass top;person;organizationalPerson;inetOrgPerson;inetOrgPersonWithCertSerno
74   * 
75   * Possible options for the properties field are (default values in parenthesis):
76   * 
77  baswdn (misspelled, but it must be like this)
78  logindn
79  loginpassword
80  hostname
81  port (389)
82  connectionsecurity (STARTTLS)
83  timeout (5000)
84  readtimeout (30000)
85  storetimeout (60000)
86  createnonexisting (true)
87  modifyexisting (true)
88  addnonexistingattr (true)
89  modifyexistingattr (false)
90  userobjectclass (top;person;organizationalPerson;inetOrgPerson)
91  caobjectclass (top;applicationProcess;certificationAuthority-V2)
92  usercertattribute (userCertificate;binary)
93  cacertattribute (cACertificate;binary)
94  crlattribute (certificateRevocationList;binary)
95  deltacrlattribute (deltaRevocationList;binary)
96  arlattribute (authorityRevocationList;binary)
97  addmultiplecertificates (false)
98  removerevoked (true)
99  removeusersoncertrevoke (false)
100 createintermediatenodes (false)
101 setuserpasssword (false)
102 usefieldsinldapdn
103  *
104  * In most cases default values are good. 
105  *
106  * @version $Id: CertSernoCustomLdapPublisher.java 34192 2020-01-07 15:10:21Z aminkh $
107  */
108 public class CertSernoCustomLdapPublisher extends LdapPublisher implements ICustomPublisher {

114     @Override
115     public void init(Properties properties) {
116         if (log.isDebugEnabled()) {
117             log.debug(">init");
118         }
119         // Transfer Properties into data used in LdapPublisher
120         Enumeration<Object> keys = properties.keys();
121         while (keys.hasMoreElements()) {
122             final String key = (String)keys.nextElement();
123             final String value = properties.getProperty(key);
124             if (log.isDebugEnabled()) {
125                 log.debug("Setting property: "+key+","+value);
126             }
127             if (key.equals("usefieldsinldapdn")) {
128                 // Create use fieldsin ldapDN, it is a Collection that needs to be created
129                 Collection<Integer> usefieldinldapdn = new ArrayList<Integer>();
130                 String[] values = StringUtils.split(value, ',');
131                 for (int i = 0; i < values.length; i++) {
132                     usefieldinldapdn.add(Integer.valueOf(values[i]));
133                     setUseFieldInLdapDN(usefieldinldapdn);
134                 }
135             } else if (key.equals("connectionsecurity")) {
136                 if ("PLAIN".equalsIgnoreCase(value)) {
137                     setConnectionSecurity(ConnectionSecurity.PLAIN);                    
138                 } else if ("STARTTLS".equalsIgnoreCase(value)) {
139                     setConnectionSecurity(ConnectionSecurity.STARTTLS);                    
140                 } else if ("SSL".equalsIgnoreCase(value)) {
141                     setConnectionSecurity(ConnectionSecurity.SSL);                    
142                 } 
143             } else {
144                 // Booleans should be added as Booleans, not strings
145                 if ("true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value)) {
146                     data.put(key, Boolean.valueOf(value));
147                 } else {
148                     // Everything else added as String
149                     data.put(key, value);                    
150                 }
151             }
152         }
153         if (log.isDebugEnabled()) {
154             log.debug(">init");
155         }        
156     }

158     @Override
159     public boolean storeCertificate(AuthenticationToken admin, Certificate incert, String username, String password, String userDN, String cafp, int status, int type, long revocationDate, int revocationReason, String tag, int certificateProfileId, long lastUpdate, ExtendedInformation extendedinformation) throws PublisherException{
160         userDN = getUidCertSernoDN(incert, username, userDN);
161         if (log.isDebugEnabled()) {
162             log.debug("storeCertificate: "+userDN);
163         }
164         return super.storeCertificate(admin, incert, username, password, userDN, cafp, status, type, revocationDate, revocationReason, tag, certificateProfileId, lastUpdate, extendedinformation);
165     }
166 
167     private String getUidCertSernoDN(Certificate incert, String username, String userDN) {
168         // Construct the userDN with the certificate serial number as UID
169         X509Certificate xcert = (X509Certificate)incert;
170         String certSerNo = xcert.getSerialNumber().toString();
171         String snfromuser = CertTools.getPartFromDN(userDN, "UID");
172         if (StringUtils.isNotEmpty(snfromuser)) {
173             log.info("User '"+username+"' aready has a UID in DN, this will be replaced by Cert Serial No: "+snfromuser);
174             StringUtils.replace(userDN, snfromuser, certSerNo);
175         } else {
176             if (StringUtils.isEmpty(userDN)) {
177                 userDN = "UID="+certSerNo;
178             } else {
179                 userDN += ",UID="+certSerNo;                
180             }
181         }
182         return userDN;
183     }
184 
185     @Override
186     public void revokeCertificate(AuthenticationToken admin, Certificate cert, String username, int reason, String userDN) throws PublisherException {
187         userDN = getUidCertSernoDN(cert, username, userDN);
188         if (log.isDebugEnabled()) {
189             log.debug("revokeCertificate: "+userDN);
190         }
191         super.revokeCertificate(admin, cert, username, reason, userDN);
192     }

194     @Override
195     protected LDAPAttributeSet getAttributeSet(Certificate cert, String objectclass, String dn, String email, boolean extra, boolean person,
196             String password, ExtendedInformation extendedinformation) {
197         LDAPAttributeSet set = super.getAttributeSet(cert, objectclass, dn, email, extra, person, password, extendedinformation);
198         // Add SerialNumber (from DN) attribute as well, it is not included by default by LDAPPublisher
199         String serno = CertTools.getPartFromDN(dn, "SN");
200         if (serno != null) {
201             set.add(new LDAPAttribute("serialNumber", serno));
202         }
203         return set;
204     }
