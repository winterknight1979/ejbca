28  /**
29   * Variables used with userdata
30   * ${USERNAME} or ${user.USERNAME} = The users username
31   * ${PASSWORD} or ${user.PASSWORD} = The users password 
32   * ${UID} or ${user.UID}           = The user's unique identifier
33   * ${CN} or ${user.CN}             = The common name of the user.
34   * ${SN} or ${user.SN}             = The serial number (in DN) of the user.
35   * ${O} or ${user.O}               = The user's organization
36   * ${OU} or ${user.OU}             = The user's organization unit
37   * ${C} or ${user.C}               = The user's country
38   * ${user.E}                       = The user's email address from Subject DN
39   * ${user.TIMECREATED}             = The time the user was created
40   * ${user.TIMEMODIFIED}            = The time the user was modified          
41   * 
42   * ${requestAdmin.CN}              = The common name of the requesting administrator.
43   * ${requestAdmin.EE.EMAIL}        = The email address of the administrator adding the end entity according to the administrator's End Entity data.
44   * ${requestAdmin.SAN.EMAIL}       = The email address of the administrator adding the end entity according to the administrator's Subject Alternative Name, RFC822Name field.
45   *
46   * Variables used with  expiring certificates. 
47   * ${expiringCert.CERTSERIAL}      = The serial number of the certificate about to expire 
48   * ${expiringCert.EXPIREDATE}      = The date the certificate will expire
49   * ${expiringCert.CERTSUBJECTDN}   = The certificate subject dn
50   * ${expiringCert.CERTISSUERDN}    = The certificate issuer dn
51   * The variables ${CN}, ${SN}, ${O}, ${OU}, ${C} are also available.
52   * 
53   * Variable used with notification where approval is involved in the state change (approvalAdmin = last admin to approve).
54   * ${approvalAdmin.USERNAME}          = The approving administrator's username
55   * ${approvalAdmin.CN}                = The common name of the approving administrator.
56   * ${approvalAdmin.SN}                = The common name of the approving administrator.
57   * ${approvalAdmin.O}                 = The approving administrator's organization
58   * ${approvalAdmin.OU}                = The approving administrator's organization unit
59   * ${approvalAdmin.C}                 = The approving administrator's country
60   * ${approvalRequestID}               = The id of the approval request
61   * 
62   * @version $Id: UserNotificationParamGen.java 24838 2016-12-06 10:45:31Z jeklund $
63   */

107 	protected void populateWithRevokedCertificate(final CertificateDataWrapper revokedCertificate) {
108 	    if (revokedCertificate!=null) {
109 	        final CertificateData certificateData = revokedCertificate.getCertificateData();
110 	        if (certificateData != null) {
111 	            paramPut("revokedCertificate.CERTSERIAL", certificateData.getSerialNumberHex());
112 	            paramPut("revokedCertificate.EXPIREDATE", fastDateFormat(new Date(certificateData.getExpireDate())));
113 	            paramPut("revokedCertificate.CERTSUBJECTDN", certificateData.getSubjectDnNeverNull());
114 	            paramPut("revokedCertificate.CERTISSUERDN", certificateData.getIssuerDN());
115 	            String newStatus = "Unknown";
116 	            switch (certificateData.getStatus()) {
117 	            case CertificateConstants.CERT_ACTIVE:
118                 case CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION:
119                     newStatus = "Active";
120                     break;
121                 case CertificateConstants.CERT_REVOKED:
122                 case CertificateConstants.CERT_ARCHIVED:
123                     newStatus = "Revoked";
124                     break;
125 	            }
126                 String newRevocationReason = "";
127                 // Use plain RFC5280 CRLReason naming until we localize this
128                 switch (RevocationReasons.getFromDatabaseValue(certificateData.getRevocationReason())) {
129                 case UNSPECIFIED: newRevocationReason = "unspecified"; break;
130                 case KEYCOMPROMISE: newRevocationReason = "keyCompromise"; break;
131                 case CACOMPROMISE: newRevocationReason = "cACompromise"; break;
132                 case AFFILIATIONCHANGED: newRevocationReason = "affiliationChanged"; break;
133                 case SUPERSEDED: newRevocationReason = "superseded"; break;
134                 case CESSATIONOFOPERATION: newRevocationReason = "cessationOfOperation"; break;
135                 case CERTIFICATEHOLD: newRevocationReason = "certificateHold"; break;
136                 case REMOVEFROMCRL: newRevocationReason = "removeFromCRL"; break;
137                 case PRIVILEGESWITHDRAWN: newRevocationReason = "privilegeWithdrawn"; break;
138                 case AACOMPROMISE: newRevocationReason = "aACompromise"; break;
139                 default: break;
140                 }
141                 paramPut("revokedCertificate.REVOCATIONSTATUS", newStatus);
142                 paramPut("revokedCertificate.REVOCATIONREASON", newRevocationReason);
143 	        }
144 	    }
145     }

147     protected void populateWithExpiringCert(Certificate expiringCert) {
148 		if(expiringCert != null){
149 			paramPut("expiringCert.CERTSERIAL",CertTools.getSerialNumberAsString(expiringCert));
150 			paramPut("expiringCert.EXPIREDATE", fastDateFormat(CertTools.getNotAfter(expiringCert)));
151 			paramPut("expiringCert.CERTSUBJECTDN",CertTools.getSubjectDN(expiringCert));
152 			paramPut("expiringCert.CERTISSUERDN",CertTools.getIssuerDN(expiringCert));          
153 		}
154 	}

156 	protected void populateWithUserData(EndEntityInformation userData) {
157 		if (userData != null) {
158 			paramPut("USERNAME", userData.getUsername());
159 			paramPut("user.USERNAME", userData.getUsername());
160 
161 			paramPut("PASSWORD", userData.getPassword());
162 			paramPut("user.PASSWORD", userData.getPassword());
163 
164 			DNFieldExtractor dnfields = new DNFieldExtractor(userData.getDN(), DNFieldExtractor.TYPE_SUBJECTDN);
165 			paramPut("UID", dnfields.getField(DNFieldExtractor.UID, 0));
166 			paramPut("user.UID", dnfields.getField(DNFieldExtractor.UID, 0));
167 			paramPut("CN", dnfields.getField(DNFieldExtractor.CN, 0));
168 			paramPut("user.CN", dnfields.getField(DNFieldExtractor.CN, 0));
169 			paramPut("SN", dnfields.getField(DNFieldExtractor.SN, 0));
170 			paramPut("user.SN", dnfields.getField(DNFieldExtractor.SN, 0));
171 			paramPut("O", dnfields.getField(DNFieldExtractor.O, 0));
172 			paramPut("user.O", dnfields.getField(DNFieldExtractor.O, 0));
173 			paramPut("OU", dnfields.getField(DNFieldExtractor.OU, 0));
174 			paramPut("user.OU", dnfields.getField(DNFieldExtractor.OU, 0));
175 			paramPut("C", dnfields.getField(DNFieldExtractor.C, 0));
176 			paramPut("user.C", dnfields.getField(DNFieldExtractor.C, 0));
177 			paramPut("user.E", dnfields.getField(DNFieldExtractor.E, 0));
178 
179 			String time = "(time not available)";
180 			if (userData.getTimeCreated() != null) {
181 				time = fastDateFormat(userData.getTimeCreated());
182 			}
183 			paramPut("user.TIMECREATED", time);
184 			time = "(time not available)";
185 			if (userData.getTimeModified() != null) {
186 				time = fastDateFormat(userData.getTimeModified());
187 			}
188 			paramPut("user.TIMEMODIFIED", time);
189 			
190 		}
191 	}
