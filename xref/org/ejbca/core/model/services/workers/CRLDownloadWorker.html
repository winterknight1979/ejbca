41  /**
42   * Worker used for downloading external CRLs and populating the local database with limited CertificateData entries,
43   * that can be used to service OCSP responses and stores the CRL locally, so it could be served for example through
44   * the public web.
45   * 
46   * If the freshest CRL extension is present in a full CRL, the delta CRL will be downloaded and processed as well.
47   * 
48   * The worker can be configured to not respect the nextUpdate
49   * 
50   * @version $Id: CRLDownloadWorker.java 27740 2018-01-05 07:24:53Z mikekushner $
51   */
52  public class CRLDownloadWorker extends BaseWorker {
53      private static final Logger log = Logger.getLogger(CRLDownloadWorker.class);
54  
55      public static final String PROP_IGNORE_NEXT_UPDATE = "ignoreNextUpdate";
56      public static final String PROP_MAX_DOWNLOAD_SIZE = "maxDownloadSize";
57      public static final int DEFAULT_MAX_DOWNLOAD_SIZE = 1*1024*1024;
58  
59      @Override
60      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
61          // Get references to all EJB's that will be used
62          final CaSessionLocal/../../org/cesecore/certificates/ca/CaSessionLocal.html#CaSessionLocal">CaSessionLocal caSession = (CaSessionLocal) ejbs.get(CaSessionLocal.class);
63          final CrlStoreSessionLocalesecore/certificates/crl/CrlStoreSessionLocal.html#CrlStoreSessionLocal">CrlStoreSessionLocal crlStoreSession = (CrlStoreSessionLocal) ejbs.get(CrlStoreSessionLocal.class);
64          final ImportCrlSessionLocalca/core/ejb/crl/ImportCrlSessionLocal.html#ImportCrlSessionLocal">ImportCrlSessionLocal importCrlSession = (ImportCrlSessionLocal) ejbs.get(ImportCrlSessionLocal.class);
65          // Parse worker configuration
66          Collection<Integer> caIdsToCheck = getCAIdsToCheck(true);
67          if (caIdsToCheck!=null && caIdsToCheck.contains(Integer.valueOf(CAConstants.ALLCAS))) {
68              caIdsToCheck = caSession.getAllCaIds();
69          }
70          final boolean ignoreNextUpdate = Boolean.valueOf(properties.getProperty(PROP_IGNORE_NEXT_UPDATE, Boolean.FALSE.toString()));
71          final int maxDownloadSize = Integer.parseInt(properties.getProperty(PROP_MAX_DOWNLOAD_SIZE, String.valueOf(DEFAULT_MAX_DOWNLOAD_SIZE)));
72          // Process all the configured CAs
73          for (final int caId : caIdsToCheck) {
74              if (log.isTraceEnabled()) {
75                  log.trace("Processing CA with Id " + caId);
76              }
77              try {
78                  final CAInfo caInfo = caSession.getCAInfoInternal(caId);
79                  if (caInfo != null && caInfo.getCAType() == CAInfo.CATYPE_X509 && caInfo.getStatus() == CAConstants.CA_EXTERNAL) {
80                      final X509Certificate caCertificate = (X509Certificate) caInfo.getCertificateChain().iterator().next();
81                      // Parse the configured external CDP into a URL
82                      final String cdp = ((X509CAInfo)caInfo).getExternalCdp();
83                      if (cdp==null || cdp.length()==0) {
84                          log.info("No external CDP configured for CA '" + caInfo.getName() + "'. Ignoring CA.");
85                          continue;
86                      }
87                      final URL url = NetworkTools.getValidHttpUrl(cdp);
88                      if (url==null) {
89                          log.info("Invalid HTTP URL '" + cdp + "' in external CDP configured for CA '" + caInfo.getName() + "'. Ignoring CA.");
90                          continue;
91                      }
92                      final String issuerDn = CertTools.getSubjectDN(caCertificate);
93                      // Get last known CRL (if any) and check when the next update will be
94                      final Date now = new Date();
95                      final X509CRL lastFullCrl = getCRLFromBytes(crlStoreSession.getLastCRL(issuerDn, false));
96                      final X509CRL newestFullCrl;
97                      if (!ignoreNextUpdate && lastFullCrl!=null && now.before(lastFullCrl.getNextUpdate())) {
98                          log.info("Next full CRL update for CA '" + caInfo.getName() + "' will be " + ValidityDate.formatAsISO8601(lastFullCrl.getNextUpdate(), null) + ". Skipping download.");
99                          newestFullCrl = lastFullCrl;
100                     } else {
101                         final X509CRL downloadedFullCrl = getAndProcessCrl(url, maxDownloadSize, caCertificate, caInfo, importCrlSession);
102                         if (downloadedFullCrl==null) {
103                             newestFullCrl = lastFullCrl;
104                         } else {
105                             newestFullCrl = downloadedFullCrl;
106                         }
107                     }
108                     if (newestFullCrl!=null) {
109                         final List<String> freshestCdps = CrlExtensions.extractFreshestCrlDistributionPoints(newestFullCrl);
110                         if (!freshestCdps.isEmpty()) {
111                             // Delta CRLs are used and we might already have a valid one stored
112                             X509CRL lastDeltaCrl = getCRLFromBytes(crlStoreSession.getLastCRL(issuerDn, true));
113                             if (lastDeltaCrl!=null && lastDeltaCrl.getThisUpdate().before(newestFullCrl.getThisUpdate())) {
114                                 // The last known delta CRL info is already included in the latest full CRL, so treat the last delta as non-existent
115                                 lastDeltaCrl = null;
116                             }
117                             if (!ignoreNextUpdate && lastDeltaCrl!=null && now.before(lastDeltaCrl.getNextUpdate())) {
118                                 log.info("Next delta CRL update for CA '" + caInfo.getName() + "' will be " + ValidityDate.formatAsISO8601(lastDeltaCrl.getNextUpdate(), null) + ". Skipping download.");
119                             } else {
120                                 // Check for and process first delta CRL that can be reached over HTTP (if any)
121                                 for (final String freshestCdp : freshestCdps) {
122                                     final URL freshestCdpUrl = NetworkTools.getValidHttpUrl(freshestCdp);
123                                     if (freshestCdpUrl==null) {
124                                         log.info("Unusable Freshest CDP HTTP URL '" + freshestCdpUrl + "' in CRL. Skipping download.");
125                                         continue;
126                                     }
127                                     final X509CRL newDeltaCrl = getAndProcessCrl(freshestCdpUrl, maxDownloadSize, caCertificate, caInfo, importCrlSession);
128                                     if (newDeltaCrl!=null) {
129                                         break;
130                                     }
131                                 }
132                             }
133                         }
134                     }
135                 } else {
136                     log.info("'" + caInfo.getName() + "' is not an external X509 CA. Ignoring.");
137                 }
138             } catch (CRLException e) {
139                 log.error("Last known CRL read from the database for CA Id " + caId + " has encoding problems.", e);
140             } catch (CrlStoreException e) {
141                 log.error("Failed to store the downloaded CRL in the database for CA Id " + caId + ".", e);
142             } catch (CrlImportException e) {
143                 log.error("Failed to import the downloaded CRL in the database for CA Id " + caId + ".", e);
144             } catch (AuthorizationDeniedException e) {
145                 throw new ServiceExecutionFailedException("Service should always be authorized to any CA.", e);
146             }
147         }
148     }
149     
150     private X509CRL getCRLFromBytes(final byte[] crlBytes) throws CRLException {
151         if (crlBytes != null) {
152             return CertTools.getCRLfromByteArray(crlBytes);
153         }
154         return null;
155     }
156     
157     private X509CRL getAndProcessCrl(final URL cdpUrl, final int maxSize, final X509Certificate caCertificate, final CAInfo caInfo,
158             final ImportCrlSessionLocal importCrlSession) throws CrlStoreException, AuthorizationDeniedException, CrlImportException {
159         X509CRL newCrl = null;
160         final byte[] crlBytesNew = NetworkTools.downloadDataFromUrl(cdpUrl, maxSize);
161         if (crlBytesNew==null) {
162             log.warn("Unable to download CRL for " + CertTools.getSubjectDN(caCertificate));
163         } else {
164             try {
165                 newCrl = CertTools.getCRLfromByteArray(crlBytesNew);
166                 importCrlSession.importCrl(admin, caInfo, crlBytesNew);
167             } catch (CRLException e) {
168                 log.warn("Unable to decode downloaded CRL for '" + caInfo.getSubjectDN() + "'.");
169                 return null;
170             }
171         }
172         return newCrl;
173     }
