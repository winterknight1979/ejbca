38      /**
39       * Semaphore making sure not two identical services run at the same time.
40       * This must be decided by serviceName, since we can configure one of these
41       * services for every publisher.
42       */
43      private static HashMap<String, Boolean> runmap = new HashMap<String, Boolean>();

45      /**
46       * Checks if there are any publishings in the publisher queue that should be
47       * published.
48       * 
49       * @see org.ejbca.core.model.services.IWorker#work
50       */
51      @Override
52      public void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException {
53          log.trace(">work");
54          final PublisherSessionLocala/core/ejb/ca/publisher/PublisherSessionLocal.html#PublisherSessionLocal">PublisherSessionLocal publisherSession = ((PublisherSessionLocal)ejbs.get(PublisherSessionLocal.class));
55          final PublisherQueueSessionLocal/ca/publisher/PublisherQueueSessionLocal.html#PublisherQueueSessionLocal">PublisherQueueSessionLocal publisherQueueSession = ((PublisherQueueSessionLocal)ejbs.get(PublisherQueueSessionLocal.class));
56          // A semaphore used to not run parallel processing jobs
57          boolean running = false;
58          synchronized (runmap) {
59              Boolean b = runmap.get(this.serviceName);
60              if (b != null) {
61                  running = b.booleanValue();
62              }
63          }
64          if (!running) {
65              try {
66                  synchronized (runmap) {
67                      runmap.put(this.serviceName, Boolean.TRUE);
68                  }
69                  Object o = properties.get(PROP_PUBLISHER_IDS);
70                  if (o != null) {
71                      String idstr = (String) o;
72                      log.debug("Ids: " + idstr);
73                      // Loop through all handled publisher ids and process
74                      // anything in the queue
75                      String[] ids = StringUtils.split(idstr, ';');
76                      for (int i = 0; i < ids.length; i++) {
77                          int publisherId = Integer.valueOf(ids[i]);
78                          // Get everything from the queue for this publisher id
79                          BasePublisher publisher = publisherSession.getPublisher(publisherId);                  
80                          publisherQueueSession.plainFifoTryAlwaysLimit100EntriesOrderByTimeCreated(getAdmin(), publisherId, publisher);
81                      }
82                  } else {
83                      log.debug("No publisher ids configured for worker.");
84                  }
85              } finally {
86                  synchronized (runmap) {
87                      runmap.put(this.serviceName, Boolean.FALSE);
88                  }
89              }
90          } else {
91              log.info(InternalEjbcaResources.getInstance().getLocalizedMessage("services.alreadyrunninginvm", PublishQueueProcessWorker.class.getName()));
92          }
93          log.trace("<work");
94      }
