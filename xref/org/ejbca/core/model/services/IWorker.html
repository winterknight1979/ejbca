13  package org.ejbca.core.model.services;
14  
15  import java.util.Map;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  
19  /**
20   * The worker interface of a service. It is the main class of a service.
21   * A worker has one IInterval and one IAction.
22   * 
23   * The methods work() method is used to signal to this service that
24   * it is time to work.
25   *
26   * @version $Id: IWorker.java 22117 2015-10-29 10:53:42Z mikekushner $
27   */
28  public interface IWorker {
29  
30  	/** Should be a ';' separated string of CAIds. */
31  	String PROP_CAIDSTOCHECK     = "worker.caidstocheck";
32  	String PROP_CERTIFICATE_PROFILE_IDS_TO_CHECK     = "worker.certificateprofileidstocheck";
33  	
34  	/** The time in 'timeunit' that a user is allowed to have status 'new' since last modification date */
35  	String PROP_TIMEBEFOREEXPIRING = "worker.timebeforeexpiring";
36  	
37  	/** Unit in days, hours or seconds */
38  	String PROP_TIMEUNIT           = "worker.timeunit";
39  
40  	String UNIT_SECONDS = "SECONDS";
41  	String UNIT_MINUTES = "MINUTES";
42  	String UNIT_HOURS = "HOURS";
43  	String UNIT_DAYS = "DAYS";
44  	
45  	int UNITVAL_SECONDS = 1;
46  	int UNITVAL_MINUTES = 60;
47  	int UNITVAL_HOURS = 3600;
48  	int UNITVAL_DAYS = 86400;
49  
50  	String[] AVAILABLE_UNITS = {UNIT_SECONDS, UNIT_MINUTES, UNIT_HOURS, UNIT_DAYS};
51  	int[] AVAILABLE_UNITSVALUES = {UNITVAL_SECONDS, UNITVAL_MINUTES, UNITVAL_HOURS, UNITVAL_DAYS};
52  
53  	/**
54  	 * Method that configures this worker and also sets up it's action and interval.
55  	 * @param admin Admin
56  	 * 
57  	 * @param serviceConfiguration Config
58  	 * @param serviceName Name
59  	 * @param runTimeStamp Time
60  	 * @param nextRunTimeStamp Time
61  	 */
62  	void init(AuthenticationToken admin, ServiceConfiguration serviceConfiguration, String serviceName, long runTimeStamp, long nextRunTimeStamp);
63  	
64  	/**
65  	 * The main method that is called by the TimeSessionBean each time
66  	 * it is time to activate this service
67  	 * @param ejbs A map between Local EJB interface classes and their injected stub
68  	 * @throws ServiceExecutionFailedException Fail
69  	 */
70  	void work(Map<Class<?>, Object> ejbs) throws ServiceExecutionFailedException;
71  	
72  	/**
73  	 * 
74  	 * @return the time in seconds to next execution.
75  	 */
76  	long getNextInterval();
77  }
