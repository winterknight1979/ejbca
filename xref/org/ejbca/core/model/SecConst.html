14  package org.ejbca.core.model;

16  /**
17   * Constants for users and certificates. Constants for Type of user: Type is constructed as a mask
18   * since one user can be of several types. To test a user type:
19   * <pre>
20   * if (((type &amp; USER_ENDUSER) == USER_ENDUSER) &amp;&amp; ((type &amp; USER_CAADMIN) == USER_ADMINISTOR) || ...
21   *    ...
22   * </pre>
23   * Bit usage: bits 0-7   (1:st byte):  user types bits 8-15  (2:nd byte):  unused bits 16-23 (3:rd
24   * byte):  unused bits 24-30 (4:th byte):  unused Constants for certificates are simple integer
25   * types. Constants for Token Types Token type is constructed of integer constants since only one
26   * token type can be generated.
27   *
28   * @version $Id: SecConst.java 27422 2017-12-05 14:05:42Z bastianf $
29   */
30  public final class SecConst {
31  
32      // Token types.
33  
34      /** Indicates that a browser generated token should be used. */
35      public static final int TOKEN_SOFT_BROWSERGEN = 1;
36  
37      /** Indicates that a p12 token should be generated. */
38      public static final int TOKEN_SOFT_P12 = 2;
39  
40      /** Indicates that a jks token should be generated. */
41      public static final int TOKEN_SOFT_JKS = 3;
42  
43      /** Indicates that a pem token should be generated. */
44      public static final int TOKEN_SOFT_PEM = 4;
45  
46      /** All values equal or below this constant should be treated as a soft token. */
47      public static final int TOKEN_SOFT = 100;

89       */
90      public static final int[] DEFAULT_KEY_LENGTHS = new int[] {512, 1024, 2048};
