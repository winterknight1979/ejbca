69      public AddEndEntityApprovalRequest(EndEntityInformation userdata, boolean clearpwd, AuthenticationToken requestAdmin, String requestSignature,
70              int cAId, int endEntityProfileId, final ApprovalProfile approvalProfile) {
71      	super(requestAdmin, requestSignature, REQUESTTYPE_SIMPLE, cAId, endEntityProfileId, approvalProfile);
72  		this.userdata = userdata;
73  		this.clearpwd = clearpwd;
74  	}

76  	@Override
77  	public void execute() throws ApprovalRequestExecutionException {
78  		throw new IllegalStateException("This execution requires additional bean references.");
79  	}
80  	
81  	public void execute(EndEntityManagementSession endEntityManagementSession, final int approvalRequestID, 
82  	        final AuthenticationToken lastApprovingAdmin) throws ApprovalRequestExecutionException {
83  		log.debug("Executing AddEndEntity for user:" + userdata.getUsername());
84  		
85  		// Add the ID of the approval request to the end entity as extended information.
86          ExtendedInformation ext = userdata.getExtendedInformation();
87          if(ext == null) {
88              ext = new ExtendedInformation();
89          }
90          ext.setAddEndEntityApprovalRequestId(approvalRequestID);
91          userdata.setExtendedInformation(ext);
92  		
93  		try{
94  			endEntityManagementSession.addUserAfterApproval(getRequestAdmin(), userdata, clearpwd, lastApprovingAdmin);
95  		} catch (EndEntityExistsException e) {
96  			throw new ApprovalRequestExecutionException("Error, user already exist", e);		
97  		} catch (AuthorizationDeniedException e) {
98  			throw new ApprovalRequestExecutionException("Authorization denied :" + e.getMessage(), e);
99  		} catch (EndEntityProfileValidationException e) {
100 			throw new ApprovalRequestExecutionException("User doesn't fullfil end entity profile:" + e.getMessage()  + e.getMessage(), e);			
101 		} catch (ApprovalException e) {
102 			throw new EJBException("This ApprovalException should never happen", e);
103 		} catch (WaitingForApprovalException e) {
104 			throw new EJBException("This WaitingForApprovalException should never happen", e);
105 		} catch (CADoesntExistsException e) {
106 			throw new ApprovalRequestExecutionException("CA does not exist:" + e.getMessage(), e);
107 		} catch (PersistenceException e) {
108 		  throw new ApprovalRequestExecutionException("Database error", e);
109 		} catch (IllegalNameException | CertificateSerialNumberException e) {
110 		    throw new ApprovalRequestExecutionException("Failed adding user:" + e.getErrorCode() + e.getMessage(), e);
111         } catch (CustomFieldException e) {
112             throw new ApprovalRequestExecutionException("Failed adding user:" + e.getErrorCode() + e.getMessage(), e);
113         }
114 	}

119 	@Override
120 	public int generateApprovalId() {
121 		if (log.isTraceEnabled()) {
122 		    log.trace(">generateApprovalId '"+getApprovalType() + ";" + userdata.getUsername() + ";" + getApprovalProfile().getProfileName()+"'");
123 		}
124 		return new String(getApprovalType() + ";" + userdata.getUsername() + ";" + getApprovalProfile().getProfileName()).hashCode();
125 	}

136 	/** Returns a summary of the information in the request, without doing any database queries. See also the overloaded method */
137 	@Override
138 	public List<ApprovalDataText> getNewRequestDataAsText(AuthenticationToken admin) {
139 	    ArrayList<ApprovalDataText> retval = new ArrayList<>();
140         retval.add(new ApprovalDataText("USERNAME",userdata.getUsername(),true,false));
141         retval.add(new ApprovalDataText("SUBJECTDN",CertTools.stringToBCDNString(userdata.getDN()),true,false));
142         retval.add(getTextWithNoValueString("SUBJECTALTNAME",userdata.getSubjectAltName()));
143         String dirattrs = userdata.getExtendedInformation() != null ? userdata.getExtendedInformation().getSubjectDirectoryAttributes() : null;
144         retval.add(getTextWithNoValueString("SUBJECTDIRATTRIBUTES",dirattrs));
145         retval.add(getTextWithNoValueString("EMAIL",userdata.getEmail()));
146         retval.add(new ApprovalDataText("KEYRECOVERABLE",userdata.getKeyRecoverable() ? "YES" : "NO",true,true));
147         retval.add(new ApprovalDataText("SENDNOTIFICATION",userdata.getSendNotification() ? "YES" : "NO",true,true));       
148         return retval;
149 	}

151 	public List<ApprovalDataText> getNewRequestDataAsText(CaSessionLocal caSession, EndEntityProfileSession endEntityProfileSession,
152 			CertificateProfileSession certificateProfileSession, HardTokenSession hardTokenSession) {
153 		ArrayList<ApprovalDataText> retval = new ArrayList<>();
154 		retval.add(new ApprovalDataText("USERNAME",userdata.getUsername(),true,false));
155 		retval.add(new ApprovalDataText("SUBJECTDN",CertTools.stringToBCDNString(userdata.getDN()),true,false));
156 		retval.add(getTextWithNoValueString("SUBJECTALTNAME",userdata.getSubjectAltName()));
157 		String dirattrs = userdata.getExtendedInformation() != null ? userdata.getExtendedInformation().getSubjectDirectoryAttributes() : null;
158 		retval.add(getTextWithNoValueString("SUBJECTDIRATTRIBUTES",dirattrs));
159 		retval.add(getTextWithNoValueString("EMAIL",userdata.getEmail()));
160 		CAInfo caInfo = caSession.getCAInfoInternal(userdata.getCAId());
161 		final String caname;
162 		if(caInfo != null)  {
163 			caname = caInfo.getName();
164 		} else {
165 			caname = "NotExist";
166 		}
167 		retval.add(new ApprovalDataText("CA", caname, true, false));
168 		retval.add(new ApprovalDataText("ENDENTITYPROFILE", endEntityProfileSession.getEndEntityProfileName(userdata.getEndEntityProfileId()),true,false));		
169 		retval.add(new ApprovalDataText("CERTIFICATEPROFILE", certificateProfileSession.getCertificateProfileName(userdata.getCertificateProfileId()),true,false));
170 		final ExtendedInformation eei = userdata.getExtendedInformation();
171         if (eei != null && eei.getKeyStoreAlgorithmType() != null) {
172             String keyTypeString = eei.getKeyStoreAlgorithmType();
173             if (eei.getKeyStoreAlgorithmSubType() != null) {
174                 keyTypeString += " " + eei.getKeyStoreAlgorithmSubType();
175             }
176             retval.add(new ApprovalDataText("KEYALGORITHM", keyTypeString, true, false));
177         }
178 		retval.add(ApprovalRequestHelper.getTokenName(hardTokenSession, userdata.getTokenType()));
179 		retval.add(getTextWithNoValueString("HARDTOKENISSUERALIAS", hardTokenSession.getHardTokenIssuerAlias(userdata.getHardTokenIssuerId())));
180 		retval.add(new ApprovalDataText("KEYRECOVERABLE",userdata.getKeyRecoverable() ? "YES" : "NO",true,true));
181 		retval.add(new ApprovalDataText("SENDNOTIFICATION",userdata.getSendNotification() ? "YES" : "NO",true,true));		
182 		return retval;
183 	}

185 	private ApprovalDataText getTextWithNoValueString(String header, String data){
186 		if(data==null || data.equals("")){
187 			return new ApprovalDataText(header,"NOVALUE",true,true);
188 		}
189 		return new ApprovalDataText(header,data,true,false);
190 	}

212 		super.readExternal(in);

216     		clearpwd = in.readBoolean();
217         }
