96      public void execute(EndEntityManagementSession endEntityManagementSession, final int approvalRequestID, 
97              final AuthenticationToken lastApprovingAdmin) throws ApprovalRequestExecutionException {
98          log.debug("Executing ChangeEndEntity for user:" + newuserdata.getUsername());
99          
100         // Add the ID of the approval request to the end entity as extended information.
101         ExtendedInformation ext = newuserdata.getExtendedInformation();
102         if(ext == null) {
103             ext = orguserdata.getExtendedInformation();
104         }
105         if(ext == null) {
106             ext = new ExtendedInformation();
107         }
108         ext.addEditEndEntityApprovalRequestId(approvalRequestID);
109         newuserdata.setExtendedInformation(ext);
110         
111         try {
112             if (newuserdata.getUsername().equals(orguserdata.getUsername())) {
113         	    endEntityManagementSession.changeUserAfterApproval(getRequestAdmin(), newuserdata, clearpwd, approvalRequestID, lastApprovingAdmin);
114             } else {
115                 endEntityManagementSession.changeUserAfterApproval(getRequestAdmin(), newuserdata, clearpwd, approvalRequestID, lastApprovingAdmin, orguserdata.getUsername());
116             }
117         } catch (AuthorizationDeniedException e) {
118             throw new ApprovalRequestExecutionException("Authorization Denied :" + e.getMessage(), e);
119         } catch (EndEntityProfileValidationException e) {
120             throw new ApprovalRequestExecutionException("User Doesn't fullfil end entity profile :" + e.getMessage() + e.getMessage(), e);
121         } catch (ApprovalException e) {
122             throw new EJBException("This should never happen", e);
123         } catch (WaitingForApprovalException e) {
124             throw new EJBException("This should never happen", e);
125         } catch (CADoesntExistsException e) {
126             throw new ApprovalRequestExecutionException("CA does not exist:" + e.getMessage(), e);
127 		} catch (CertificateSerialNumberException e) {
128 		    throw new ApprovalRequestExecutionException("Error with the SubjectDN serialnumber :" + e.getErrorCode() + e.getMessage(), e);
129         } catch (IllegalNameException e) {
130             throw new ApprovalRequestExecutionException("The Subject DN failed constraints. " + e.getErrorCode() + e.getMessage(), e);
131         } catch (NoSuchEndEntityException e) {
132             throw new ApprovalRequestExecutionException("End entity not found.", e);
133         } catch (CustomFieldException e) {
134             throw new ApprovalRequestExecutionException("The end entity was not validated by a locally defined field validator", e);
135         }
136     }

143         if (useOrgUsername) {
144             return new String(getApprovalType() + ";" + orguserdata.getUsername() + ";" + getApprovalProfile().getProfileName()).hashCode();
145         } else {
146             return new String(getApprovalType() + ";" + newuserdata.getUsername() + ";" + getApprovalProfile().getProfileName()).hashCode();
147         }

161 	public List<ApprovalDataText> getNewRequestDataAsText(AuthenticationToken admin) {
162 	    ArrayList<ApprovalDataText> retval = new ArrayList<>();
163         retval.add(new ApprovalDataText("USERNAME",newuserdata.getUsername(),true,false));
164         String passwordtext = "NOTSHOWN";
165         if((newuserdata.getPassword() == null && !StringUtils.isEmpty(orguserdata.getPassword())) ||
166            (!StringUtils.isEmpty(newuserdata.getPassword()) && orguserdata.getPassword() == null)) {            
167             passwordtext = "NEWPASSWORD";           
168         }       
169         if(newuserdata.getPassword() != null && orguserdata.getPassword() != null){
170             if(!newuserdata.getPassword().equals(orguserdata.getPassword())){
171                 passwordtext = "NEWPASSWORD";
172             }
173         }               
174         retval.add(new ApprovalDataText("PASSWORD",passwordtext,true,true));
175         retval.add(new ApprovalDataText("SUBJECTDN",CertTools.stringToBCDNString(newuserdata.getDN()),true,false));
176         retval.add(getTextWithNoValueString("SUBJECTALTNAME",newuserdata.getSubjectAltName()));
177         String dirattrs = newuserdata.getExtendedInformation() != null ? newuserdata.getExtendedInformation().getSubjectDirectoryAttributes() : null;
178         retval.add(getTextWithNoValueString("SUBJECTDIRATTRIBUTES",dirattrs));
179         retval.add(getTextWithNoValueString("EMAIL",newuserdata.getEmail()));
180         retval.add(new ApprovalDataText("KEYRECOVERABLE",newuserdata.getKeyRecoverable() ? "YES" : "NO",true,true));
181         retval.add(new ApprovalDataText("SENDNOTIFICATION",newuserdata.getSendNotification() ? "YES" : "NO",true,true));
182         retval.add(new ApprovalDataText("STATUS",EndEntityConstants.getTranslatableStatusText(newuserdata.getStatus()),true,true));
183         return retval;
184 	}

186 	public List<ApprovalDataText> getNewRequestDataAsText(CaSessionLocal caSession, EndEntityProfileSession endEntityProfileSession,
187 			CertificateProfileSession certificateProfileSession, HardTokenSession hardTokenSession) {
188 		ArrayList<ApprovalDataText> retval = new ArrayList<>();
189 		retval.add(new ApprovalDataText("USERNAME",newuserdata.getUsername(),true,false));
190 		String passwordtext = "NOTSHOWN";
191 		if((newuserdata.getPassword() == null && !StringUtils.isEmpty(orguserdata.getPassword())) ||
192 		   (!StringUtils.isEmpty(newuserdata.getPassword()) && orguserdata.getPassword() == null)) {			
193 			passwordtext = "NEWPASSWORD";			
194 		}		
195 		if(newuserdata.getPassword() != null && orguserdata.getPassword() != null){
196 			if(!newuserdata.getPassword().equals(orguserdata.getPassword())){
197 				passwordtext = "NEWPASSWORD";
198 			}
199 		}				
200 		retval.add(new ApprovalDataText("PASSWORD",passwordtext,true,true));
201 		retval.add(new ApprovalDataText("SUBJECTDN",CertTools.stringToBCDNString(newuserdata.getDN()),true,false));
202 		retval.add(getTextWithNoValueString("SUBJECTALTNAME",newuserdata.getSubjectAltName()));
203 		String dirattrs = newuserdata.getExtendedInformation() != null ? newuserdata.getExtendedInformation().getSubjectDirectoryAttributes() : null;
204 		retval.add(getTextWithNoValueString("SUBJECTDIRATTRIBUTES",dirattrs));
205 		retval.add(getTextWithNoValueString("EMAIL",newuserdata.getEmail()));
206 		CAInfo caInfo = caSession.getCAInfoInternal(newuserdata.getCAId());
207 		String caname;
208 		if(caInfo != null) {
209 			caname = caInfo.getName();
210 		} else {
211 			caname = "NotExist";
212 		}
213 		retval.add(new ApprovalDataText("CA", caname, true, false));
214 		retval.add(new ApprovalDataText("ENDENTITYPROFILE", endEntityProfileSession.getEndEntityProfileName(newuserdata.getEndEntityProfileId()),true,false));		
215 		retval.add(new ApprovalDataText("CERTIFICATEPROFILE", certificateProfileSession.getCertificateProfileName(newuserdata.getCertificateProfileId()),true,false));
216 		final ExtendedInformation neweei = newuserdata.getExtendedInformation();
217 		if (neweei != null && neweei.getKeyStoreAlgorithmType() != null) {
218 		    String keyTypeString = neweei.getKeyStoreAlgorithmType();
219 		    if (neweei.getKeyStoreAlgorithmSubType() != null) {
220 		        keyTypeString += " " + neweei.getKeyStoreAlgorithmSubType();
221 		    }
222 		    retval.add(new ApprovalDataText("KEYALGORITHM", keyTypeString, true, false));
223 		}
224 		retval.add(ApprovalRequestHelper.getTokenName(hardTokenSession, newuserdata.getTokenType()));
225 		retval.add(getTextWithNoValueString("HARDTOKENISSUERALIAS", hardTokenSession.getHardTokenIssuerAlias(newuserdata.getHardTokenIssuerId())));
226 		retval.add(new ApprovalDataText("KEYRECOVERABLE",newuserdata.getKeyRecoverable() ? "YES" : "NO",true,true));
227 		retval.add(new ApprovalDataText("SENDNOTIFICATION",newuserdata.getSendNotification() ? "YES" : "NO",true,true));
228 		retval.add(new ApprovalDataText("STATUS",EndEntityConstants.getTranslatableStatusText(newuserdata.getStatus()),true,true));
229 		return retval;
230 	}
