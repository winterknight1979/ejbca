41  /**
42   * Approval Request created when an administrator wants
43   * to recovery a end entities keyset
44   * 
45   * @version $Id: KeyRecoveryApprovalRequest.java 26106 2017-06-30 13:39:10Z henriks $
46   */
47  public class KeyRecoveryApprovalRequest extends ApprovalRequest {
48  
49  	private static final long serialVersionUID = -1L;
50  	private static final Logger log = Logger.getLogger(KeyRecoveryApprovalRequest.class);
51  	private static final int LATEST_VERSION = 1;
52  		
53  	private String username;
54  	private Certificate cert;
55  
56  	private boolean recoverNewestCert = false; 
57  
58  	/** Constructor used in externalization only */
59  	public KeyRecoveryApprovalRequest() {}
60  
61      public KeyRecoveryApprovalRequest(Certificate cert, String username, boolean recoverNewestCert, AuthenticationToken requestAdmin,
62              String requestSignature, int cAId, int endEntityProfileId, ApprovalProfile approvalProfile) {
63          super(requestAdmin, requestSignature, REQUESTTYPE_SIMPLE, cAId, endEntityProfileId, approvalProfile);
64  		this.username = username;
65  		this.cert = cert;
66  		this.recoverNewestCert = recoverNewestCert;
67  	}
68  	
69  
70  	@Override
71  	public void execute() throws ApprovalRequestExecutionException {
72  		throw new RuntimeException("This execution requires additional bean references.");
73  	}
74  
75      public void execute(EndEntityManagementSession endEntityManagementSession) throws ApprovalRequestExecutionException {
76          log.debug("Executing mark for recovery for user:" + username);
77          try {
78              if (recoverNewestCert) {
79                  endEntityManagementSession.prepareForKeyRecovery(getRequestAdmin(), username, getEndEntityProfileId(), null);
80              } else {
81                  endEntityManagementSession.prepareForKeyRecovery(getRequestAdmin(), username, getEndEntityProfileId(), cert);
82              }
83          } catch (AuthorizationDeniedException e) {
84              throw new ApprovalRequestExecutionException("Authorization Denied :" + e.getMessage(), e);
85          } catch (ApprovalException e) {
86              throw new EJBException("This should never happen", e);
87          } catch (WaitingForApprovalException e) {
88              throw new EJBException("This should never happen", e);
89          } catch (CADoesntExistsException e) {
90              throw new EJBException("This should never happen", e);
91          }
92      }
93  
94      /**
95       * Approval Id is generated of This approval type (i.e AddEndEntityApprovalRequest) and UserName
96       */
97  	public int generateApprovalId() {		
98  		return new String(getApprovalType() + ";" + username + ";" + getApprovalProfile().getProfileName()).hashCode();
99  	}
100 
101 	public int getApprovalType() {		
102 		return ApprovalDataVO.APPROVALTYPE_KEYRECOVERY;
103 	}
104 	
105 	public String getUsername() {
106 	    return username;
107 	}
108 
109 	@Override
110 	public List<ApprovalDataText> getNewRequestDataAsText(AuthenticationToken admin) {
111 		ArrayList<ApprovalDataText> retval = new ArrayList<ApprovalDataText>();
112 		retval.add(new ApprovalDataText("USERNAME",username,true,false));
113 		retval.add(new ApprovalDataText("CERTSERIALNUMBER",CertTools.getSerialNumberAsString(cert),true,false));
114 		retval.add(new ApprovalDataText("SUBJECTDN",CertTools.getSubjectDN(cert).toString(),true,false));
115 		retval.add(new ApprovalDataText("ISSUERDN",CertTools.getIssuerDN(cert).toString(),true,false));
116 		return retval;
117 	}
118 	
119 	@Override
120 	public List<ApprovalDataText> getOldRequestDataAsText(AuthenticationToken admin) {
121 		return null;
122 	}
123 
124 	public boolean isExecutable() {		
125 		return true;
126 	}
127 	
128 	public void writeExternal(ObjectOutput out) throws IOException {
129 		super.writeExternal(out);
130 		out.writeInt(LATEST_VERSION);
131 		out.writeObject(username);
132 		out.writeBoolean(recoverNewestCert);
133 		try {
134 			String certString = new String(Base64.encode(cert.getEncoded()),"UTF8");
135 			out.writeObject(certString);
136 		} catch (CertificateEncodingException e) {
137 			log.debug("Error serializing certificate", e);
138 			throw new IOException(e.getMessage());
139 		}	
140 	}
141 
142 	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {        
143 		super.readExternal(in);
144         int version = in.readInt();
145         if(version == 1){
146     		username = (String) in.readObject();
147     		recoverNewestCert = in.readBoolean();
148     		String certString = (String) in.readObject();    		
149     		try {
150 				cert = CertTools.getCertfromByteArray(Base64.decode(certString.getBytes("UTF8")), Certificate.class);
151 			} catch (CertificateException e) {
152 				log.debug("Error deserializing certificate", e);
153 				throw new IOException(e.getMessage());
154 			}	
155         }
156 	}
157 }
