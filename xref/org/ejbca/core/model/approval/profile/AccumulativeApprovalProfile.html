32  /**
33   * This approval archetype represents the legacy method of approvals, i.e where a fixed number of administrators need to approve a request for it to 
34   * pass.
35   * 
36   * @version $Id: AccumulativeApprovalProfile.java 28907 2018-05-09 23:24:08Z mikekushner $
37   */
38  public class AccumulativeApprovalProfile extends ApprovalProfileBase {
39  
40      public static final int FIXED_STEP_ID = 0; //Contains only a single sequence
41      
42      private static final long serialVersionUID = 6432620040542676563L;
43      
44      private static final InternalResources intres = InternalResources.getInstance();
45  
46      /**
47       * Note: do not change, may cause problems in deployed installations.
48       */
49      private static final String TYPE_IDENTIFIER = "ACCUMULATIVE_APPROVAL";
50          
51      public AccumulativeApprovalProfile() {
52          //Public constructor needed deserialization 
53          super();
54      }
55  
56      public AccumulativeApprovalProfile(final String name) {
57          super(name);
58          initialize();
59      }
60      
61      /*
62       * This method only needs to be called by the factory method (and some unit tests), because it sets a ton of boilerplate stuff which isn't 
63       * required by already initialized profiles.
64       */
65      @Override
66      public void initialize() {
67          super.initialize();
68          //Workaround, since this profile normally doesn't allow adding sequences. 
69          if (getSteps().isEmpty()) {
70              getSteps().put(FIXED_STEP_ID, new ApprovalStep(FIXED_STEP_ID));
71              addPartition(FIXED_STEP_ID);
72              setFirstStep(FIXED_STEP_ID);
73          }
74      }
75  
76  
77      @Override
78      public String getApprovalProfileTypeIdentifier() {
79          return TYPE_IDENTIFIER;
80      }
81  
82      @Override
83      public String getApprovalProfileLabel() {
84          return intres.getLocalizedMessage("approval.profile.implementation.accumulative.approval.name");
85      }
86  
87      public void setNumberOfApprovalsRequired(int approvalsRequired) throws PropertyValidationException {
88          final int partitionIdentifier = getSinglePartitionIdentifier(FIXED_STEP_ID);
89          DynamicUiProperty<? extends Serializable> approvalsRequiredProperty = getSteps().get(FIXED_STEP_ID).getPartition(partitionIdentifier)
90                  .getProperty(PROPERTY_NUMBER_OF_REQUIRED_APPROVALS);
91          approvalsRequiredProperty.setValueGeneric(Integer.valueOf(approvalsRequired));
92          addPropertyToPartition(FIXED_STEP_ID, partitionIdentifier, approvalsRequiredProperty);
93          saveTransientObjects();
94      }
95  
96      public int getNumberOfApprovalsRequired() {
97          return getNumberOfApprovalsRequired(FIXED_STEP_ID, getSinglePartitionIdentifier(FIXED_STEP_ID));
98      }
99      
100     private int getSinglePartitionIdentifier(final int stepIdentifier) {
101         return getSteps().get(stepIdentifier).getPartitions().values().iterator().next().getPartitionIdentifier();
102     }
103 
104     @Override
105     public boolean isApprovalRequired() {
106         return getNumberOfApprovalsRequired() > 0;
107     }
108 
109     @Override
110     public boolean canApprovalExecute(final Collection<Approval> approvalsPerformed) throws ApprovalException {
111         //Verify that at least one of the approvals performed covers the single sequence in this implementation (Though it would be odd if they didn't)
112         boolean sequenceAndPartitionFound = false;
113         for(Approval approval : approvalsPerformed) {
114             if(approval.getStepId() == FIXED_STEP_ID) {
115                 sequenceAndPartitionFound = true;
116                 break;
117             }
118         }
119         if(!sequenceAndPartitionFound) {
120             return false;
121         } else {
122             int numberofapprovalsleft = getRemainingApprovals(approvalsPerformed);
123             if (numberofapprovalsleft < 0) {
124                 throw new ApprovalException("Approval cannot execute due to already being rejected.");
125             }
126             return numberofapprovalsleft == 0;
127         }
128 
129     }
130 
131     @Override
132     public int getRemainingApprovals(Collection<Approval> approvalsPerformed) {
133         return getRemainingApprovalsInPartition(approvalsPerformed, FIXED_STEP_ID, getSinglePartitionIdentifier(FIXED_STEP_ID));
134     }

139         return true;
140     }
