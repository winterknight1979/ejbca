13  package org.ejbca.core.model.approval;
14  
15  import org.ejbca.core.ejb.hardtoken.HardTokenSession;
16  import org.ejbca.core.model.SecConst;
17  
18  /**
19   * Helper class containing static methods for RMI lookups
20   * 
21   * @version $Id: ApprovalRequestHelper.java 22139 2015-11-03 10:41:56Z mikekushner $
22   */
23  public class ApprovalRequestHelper { 
24  
25      public static ApprovalDataText getTokenName(HardTokenSession hardTokenSession, int tokenid) {
26          ApprovalDataText retval;
27          if (tokenid <= SecConst.TOKEN_SOFT) {
28              int tokenindex = 0;
29              for (int i = 0; i < SecConst.TOKENIDS.length; i++) {
30                  if (SecConst.TOKENIDS[i] == tokenid) {
31                      tokenindex = i;
32                  }
33              }
34              retval = new ApprovalDataText("TOKEN", SecConst.TOKENTEXTS[tokenindex], true, true);
35  
36          } else {
37              String name = hardTokenSession.getHardTokenProfileName(tokenid);
38              retval = new ApprovalDataText("TOKEN", name, true, false);
39          }
40          return retval;
41      }
42  }
