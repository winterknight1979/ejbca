13  package org.ejbca.core.ejb.keyrecovery;
14  
15  import java.security.cert.Certificate;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  import org.cesecore.certificates.ca.CADoesntExistsException;
20  import org.cesecore.certificates.certificate.CertificateWrapper;
21  import org.cesecore.keys.util.KeyPairWrapper;
22  import org.ejbca.core.model.approval.ApprovalException;
23  import org.ejbca.core.model.approval.WaitingForApprovalException;
24  import org.ejbca.core.model.keyrecovery.KeyRecoveryInformation;
25  
26  /**
27   * @version $Id: KeyRecoverySession.java 26210 2017-08-03 10:12:32Z samuellb $
28   */
29  public interface KeyRecoverySession {
30  
31       /**
32       * Method checking the following authorizations:
33       * 
34       * If /superadmin -&gt; true
35       * 
36       * Other must have both
37       * AccessRulesConstants.
38       *  /ra_functionality/keyrecovery
39       *  and /endentityprofilesrules/&lt;endentityprofile&gt;/keyrecovery
40       * 
41       * @param admin authentication token of requesting administrator
42       * @param profileid end entity profile
43       * @return true if the admin is authorized to keyrecover
44       */
45      public boolean authorizedToKeyRecover(AuthenticationToken admin, int profileid);
46      
47      /**
48       * Help method to check if approval of key recovery is required
49       * @param admin authentication token of requesting administrator
50       * @param certificate to recover
51       * @param username of the end entity related to the certificate
52       * @param endEntityProfileId used by the end entity
53       * @param checkNewest bool
54       * @throws ApprovalException if approval already exists
55       * @throws WaitingForApprovalException if approval is required. Expected to be thrown in this case
56       * @throws CADoesntExistsException if the issuer of the certificate doesn't exist
57       */
58      public void checkIfApprovalRequired(AuthenticationToken admin, CertificateWrapper certificate, String username, int endEntityProfileId, boolean checkNewest) 
59              throws ApprovalException, WaitingForApprovalException, CADoesntExistsException;
60      
61      /**
62       * Adds a certificates keyrecovery data to the database.
63       *
64       * @param admin the administrator calling the function
65       * @param certificate the certificate used with the keypair.
66       * @param username of the administrator
67       * @param keypair the actual keypair to save.
68       *
69       * @return false if the certificates keyrecovery data already exists.
70       * @throws AuthorizationDeniedException if not authorized to administer keys.
71       */
72      boolean addKeyRecoveryData(AuthenticationToken admin, CertificateWrapper certificate, String username, KeyPairWrapper keypair) throws AuthorizationDeniedException;
73  
74      /**
75       * Removes a certificates keyrecovery data from the database.
76       *
77       * @param admin the administrator calling the function
78       * @param certificate the certificate used with the keys about to be removed.
79       * @throws AuthorizationDeniedException if not authorized to administer keys
80       */
81      void removeKeyRecoveryData(AuthenticationToken admin, CertificateWrapper certificate) throws AuthorizationDeniedException;
82  
83      /** Removes a all keyrecovery data saved for a user from the database. 
84       * @param admin admin
85       * @param username user */
86      void removeAllKeyRecoveryData(AuthenticationToken admin, String username);
87  
88      /**
89       * Returns the keyrecovery data for a user. Observe only one certificates
90       * key can be recovered for every user at the time.
91       * @param admin admin
92       * @param username user
93       * @param endEntityProfileId the end entity profile id the user belongs to.
94       * @return the marked keyrecovery data or null if none can be found.
95       * 
96       * @throws AuthorizationDeniedException if not authorized to recover keys
97       */
98       KeyRecoveryInformation recoverKeys(AuthenticationToken admin, String username, int endEntityProfileId)
99              throws AuthorizationDeniedException;
100 
101     /**
102      * Marks a users newest certificate for key recovery. Newest means certificate with latest not
103      * before date.
104      *
105      * @param admin the administrator calling the function
106      * @param username or the user.
107      * @param endEntityProfileId the end entity profile of the user, used for access control
108      * @return true if operation went successful or false if no certificates could be found for
109      *         user, or user already marked.
110      * @throws AuthorizationDeniedException fail
111      * @throws ApprovalException fail
112      * @throws WaitingForApprovalException fail
113      *         
114      * @throws CADoesntExistsException if the issuer of the certificate doesn't exist
115      */
116     boolean markNewestAsRecoverable(AuthenticationToken admin, String username, int endEntityProfileId)
117             throws AuthorizationDeniedException, ApprovalException, WaitingForApprovalException, CADoesntExistsException;
118 
119     /**
120      * Marks a users certificate for key recovery.
121      *
122      * @param admin the administrator calling the function
123      * @param certificate the certificate used with the keys about to be removed.
124      * @param endEntityProfileId ID
125      * @return true if operation went successful or false if  certificate couldn't be found.
126      * @throws AuthorizationDeniedException fail
127      * @throws WaitingForApprovalException fail
128      * @throws ApprovalException fail
129      * 
130      * @throws CADoesntExistsException if the issuer of the certificate doesn't exist
131      */
132     boolean markAsRecoverable(AuthenticationToken admin, Certificate certificate, int endEntityProfileId)
133             throws AuthorizationDeniedException, WaitingForApprovalException, ApprovalException, CADoesntExistsException;
134     
135     /** Resets keyrecovery mark for a user. 
136      * @param admin admin
137      * @param username user */
138     void unmarkUser(AuthenticationToken admin, String username);
139 
140     /** @param username user
141      * @return true if user is already marked for key recovery. */
142     boolean isUserMarked(String username);
143 
144     /**
145      * @param certificate the certificate used with the keys about to be removed.
146      * @return true if specified certificates keys exists in database.
147      */
148     boolean existsKeys(CertificateWrapper certificate);
149 }
