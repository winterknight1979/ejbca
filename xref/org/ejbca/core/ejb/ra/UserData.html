86      /**
87       * Entity Bean holding info about a User. Create by sending in the instance, username, password and subject DN. SubjectEmail, Status and Type are
88       * set to default values (null, STATUS_NEW, USER_INVALID). and should be set using the respective set-methods. Clear text password is not set at
89       * all and must be set using setClearPassword();
90       * 
91       * @param username the unique username used for authentication.
92       * @param password the password used for authentication. If clearpwd is false this only sets passwordhash, if clearpwd is true it also sets
93       *            cleartext password.
94       * @param clearpwd true if clear password should be set for CA generated tokens (p12, jks, pem), false otherwise for only storing hashed
95       *            passwords.
96       * @param dn the DN the subject is given in his certificate.
97       * @param caid ID
98       * @param cardnumber the number printed on the card.
99       * @param altname string of alternative names, i.e. rfc822name=foo2bar.com,dnsName=foo.bar.com, can be null
100      * @param email user email address, can be null
101      * @param type user type, i.e. EndEntityTypes.USER_ENDUSER etc
102      * @param eeprofileid end entity profile id, can be 0
103      * @param certprofileid certificate profile id, can be 0
104      * @param tokentype token type to issue to the user, i.e. SecConst.TOKEN_SOFT_BROWSERGEN
105      * @param hardtokenissuerid hard token issuer id if hard token issuing is used, 0 otherwise
106      * @param extendedInformation ExtendedInformation object
107      * 
108      */
109     public UserData(String username, String password, boolean clearpwd, String dn, int caid, String cardnumber, String altname, String email,
110             int type, int eeprofileid, int certprofileid, int tokentype, int hardtokenissuerid, ExtendedInformation extendedInformation) {
111         long time = new Date().getTime();
112         setUsername(username);
113         if (clearpwd) {
114             setOpenPassword(password);
115         } else {
116             setPasswordHash(CryptoTools.makePasswordHash(password));
117             setClearPassword(null);
118         }
119         transientPwd = password; // performance optimization within a transaction
120         setSubjectDN(CertTools.stringToBCDNString(dn));
121         setCaId(caid);
122         setSubjectAltName(altname);
123         setSubjectEmail(email);
124         setStatus(EndEntityConstants.STATUS_NEW);
125         setType(type);
126         setTimeCreated(time);
127         setTimeModified(time);
128         setEndEntityProfileId(eeprofileid);
129         setCertificateProfileId(certprofileid);
130         setTokenType(tokentype);
131         setHardTokenIssuerId(hardtokenissuerid);
132         setExtendedInformation(extendedInformation);
133         setCardNumber(cardnumber);
134         if (log.isDebugEnabled()) {
135             log.debug("Created user " + username);
136         }
137     }

507      * 
508      * @return which hashing algorithm was used for this UserData object
509      */
510     public SupportedPasswordHashAlgorithm findHashAlgorithm() {
511         final String hash = getPasswordHash();
512         if (StringUtils.startsWith(hash, "$2")) {
513             return SupportedPasswordHashAlgorithm.SHA1_BCRYPT;
514         } else {
515             return SupportedPasswordHashAlgorithm.SHA1_OLD;
516         }
517     }

519     /**
520      * Verifies password by verifying against passwordhash
521      * @param password Pass
522      * @return bool
523      * @throws NoSuchAlgorithmException Fail
524      */
525     public boolean comparePassword(final String password) throws NoSuchAlgorithmException {
526         if (log.isTraceEnabled()) {
527             log.trace(">comparePassword()");
528         }
529         boolean ret = false;
530         if (password != null) {
531             if (transientPwd != null) {
532                 // Performance optimization within a transaction, to not have to hash the password when comparing internally in the same transaction, saving one BCrypt operation
533                 ret = transientPwd.equals(password);
534             } else {
535                 final String hash = getPasswordHash();
536                 // Check if it is a new or old style hashing
537                 switch (findHashAlgorithm()) {
538                 case SHA1_BCRYPT:
539                     // new style with good salt
540                     ret = BCrypt.checkpw(password, hash);
541                     break;
542                 case SHA1_OLD:
543                 default:
544                     ret = CryptoTools.makeOldPasswordHash(password).equals(getPasswordHash());
545                     break;
546                 }
547             }
548         }
549         if (log.isTraceEnabled()) {
550             log.trace("<comparePassword()");
551         }
552         return ret;
553     }

651     /**
652      * Assumes authorization has already been checked.. Modifies the ExtendedInformation object to reset the remaining login attempts.
653      * @param ei Info
654      * @param username Iser
655      * @return true if any change was made, false otherwise
656      */
657     @Transient
658     public static boolean resetRemainingLoginAttemptsInternal(final ExtendedInformation ei, final String username) {
659         if (log.isTraceEnabled()) {
660             log.trace(">resetRemainingLoginAttemptsInternal");
661         }
662         final boolean ret;
663         if (ei != null) {
664             final int resetValue = ei.getMaxLoginAttempts();
665             if (resetValue != -1 || ei.getRemainingLoginAttempts() != -1) {
666                 ei.setRemainingLoginAttempts(resetValue);
667                 final String msg = intres.getLocalizedMessage("ra.resettedloginattemptscounter", username, resetValue);
668                 log.info(msg);
669                 ret = true;
670             } else {
671                 ret = false;
672             }            
673         } else {
674             ret = false;
675         }
676         if (log.isTraceEnabled()) {
677             log.trace("<resetRamainingLoginAttemptsInternal: "+ret);
678         }
679         return ret;
680     }

686     @Transient
687     @Override
688     protected String getProtectString(final int version) {
689         final ProtectionStringBuildergBuilder.html#ProtectionStringBuilder">ProtectionStringBuilder build = new ProtectionStringBuilder();
690         // rowVersion is automatically updated by JPA, so it's not important, it is only used for optimistic locking
691         build.append(getUsername());
692         if (version>=2) {
693             // From version 2 we always use empty String here to allow future migration between databases when this value is unset
694             build.append(getSubjectDnNeverNull());
695         } else {
696             build.append(getSubjectDN());
697         }
698         build.append(getCardNumber()).append(getCaId());
699         if (version>=2) {
700             // From version 2 we always use empty String here to allow future migration between databases when this value is unset
701             build.append(getSubjectAltNameNeverNull());
702         } else {
703             build.append(getSubjectAltName());
704         }
705         build.append(getCardNumber());
706         build.append(getSubjectEmail()).append(getStatus()).append(getType()).append(getClearPassword()).append(getPasswordHash()).append(getTimeCreated()).append(getTimeModified());
707         build.append(getEndEntityProfileId()).append(getCertificateProfileId()).append(getTokenType()).append(getHardTokenIssuerId()).append(getExtendedInformationData());
708         return build.toString();
709     }

717     @PrePersist
718     @PreUpdate
719     @Override
720     protected void protectData() {
721         // This is a speed optimization to avoid encoding the extendedInformation into XML data too often
722         // We instead use the cached object in this class, and serialize it out to XML data only when we persist the object
723         // (create or update). This means you can call getExtendedInformation as much as you want, without causing an expensive
724         // XMLEncoder/Decoder
725         setExtendedInformationPrePersist(getExtendedInformation());
726         // After setting the data we want, continue on to the normal database integrity protection
727         super.protectData();
728     }
