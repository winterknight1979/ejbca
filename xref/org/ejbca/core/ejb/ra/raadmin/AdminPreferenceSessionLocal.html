13  package org.ejbca.core.ejb.ra.raadmin;
14  
15  import java.util.List;
16  import java.util.Locale;
17  import java.util.Map;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.config.RaStyleInfo;
23  import org.ejbca.core.model.ra.raadmin.AdminPreference;
24  
25  /**
26   * Local interface for RaAdminSession.
27   */
28  @Local
29  public interface AdminPreferenceSessionLocal extends AdminPreferenceSession {
30      
31      /**
32       * Returns custom RA Styles associated with the role of the requesting administrator. If the administrator has requested
33       * styles very recently, the same content will be returned to prevent requests via Peers for for every resource request
34       * e.g. in a page load in the RA web.
35       * @param admin authenticationToken of the requesting administrator
36       * @return List of custom RA styles available for the requesting administrator
37       */
38      List<RaStyleInfo> getAvailableRaStyleInfos(AuthenticationToken admin);
39     
40      /**
41       * Returns the currently set RA style id for the requested administrator.
42       * @param admin AuthenticationToken the administrators holding the preferences.
43       * @return RA style id currently set in the RA GUI for the administrator @admin. Null if @admin has no preferences set.
44       */
45      Integer getCurrentRaStyleId(AuthenticationToken admin);
46      
47      /**
48       * Sets the current RA Style id the requesting administrator.
49       * @param currentStyleId current style id for the admin.
50       * @param admin is the admin who is saving its preferences in database.
51       */
52      void setCurrentRaStyleId(int currentStyleId, AuthenticationToken admin);
53      
54      /**
55       * Returns the currently set locale for the requesting administrator.
56       * @param admin Administrator saving its data to the database.
57       * @return Currently set Locale in the RA GUI for the administrator @admin.
58       */
59      Locale getCurrentRaLocale(AuthenticationToken admin);
60      
61      /**
62       * Sets the current locale for the requesting administrator.
63       * @param locale current locale for this admin.
64       * @param admin is the admin who is saving its preferences in database.
65       */
66      void setCurrentRaLocale(Locale locale, AuthenticationToken admin);
67      
68      /**
69       * Returns the RaStyleInfo which will be used by the requesting administrator. If there are multiple
70       * styles available and no preferences has been set by the administrator, the first one available will be
71       * returned.
72       * @param admin Authentication token of the requesting administrator
73       * @return Preferred RaStyleInfo for the requested administrator. Null if 'Default' (Id 0) is selected or 
74       * if no styles are available
75       */
76      RaStyleInfo getPreferedRaStyleInfo(AuthenticationToken admin);
77      
78      /**
79       * 
80       * @return map containing all admin preferences defined in the current ejbca instance
81       *         and their human readable name as String.
82       *         
83       */
84      Map<String, AdminPreference> getAdminPreferences();
85  
86  }
