14  package org.ejbca.core.ejb.rest;
15  
16  import java.security.cert.X509Certificate;
17  
18  import javax.ejb.Local;
19  
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.ca.CADoesntExistsException;
23  import org.cesecore.certificates.certificateprofile.CertificateProfileDoesNotExistException;
24  import org.cesecore.certificates.endentity.EndEntityInformation;
25  import org.ejbca.core.EjbcaException;
26  import org.ejbca.core.model.ra.raadmin.EndEntityProfileNotFoundException;
27  import org.ejbca.core.protocol.rest.EnrollPkcs10CertificateRequest;
28  
29  /**
30   * 
31   * @version $Id: EjbcaRestHelperSessionLocal.java 29056 2018-05-29 12:06:45Z tarmo_r_helmes $
32   *
33   */
34  @Local
35  public interface EjbcaRestHelperSessionLocal extends EjbcaRestHelperSession {
36  
37      /**
38       * @param allowNonAdmins false if we should verify that it is a real administrator, true only extracts the certificate and checks that it is not revoked.
39       * @param cert X509 certificate
40       * @return AuthenticationToken object based on the SSL client certificate
41       * @throws AuthorizationDeniedException if no client certificate or allowNonAdmins = false and the certificate does not belong to an administrator
42       */
43      AuthenticationToken getAdmin(boolean allowNonAdmins, X509Certificate cert) throws AuthorizationDeniedException;
44  
45      /**
46       * Compose EndEntityInformation object based on EnrollPkcs10CertificateRequest input
47       * @param authenticationToken of the requesting administrator
48       * @param enrollcertificateRequest input data object for enrolling a certificate
49       * @return info
50       * @throws AuthorizationDeniedException fail
51       * @throws EndEntityProfileNotFoundException fail
52       * @throws EjbcaException fail
53       * @throws CertificateProfileDoesNotExistException fail
54       * @throws CADoesntExistsException fail
55       */
56      public EndEntityInformation convertToEndEntityInformation(AuthenticationToken authenticationToken, EnrollPkcs10CertificateRequest enrollcertificateRequest)
57              throws AuthorizationDeniedException, EndEntityProfileNotFoundException, EjbcaException, CertificateProfileDoesNotExistException, CADoesntExistsException;
58  }
