13  package org.ejbca.core.ejb.upgrade;
14  
15  import java.util.List;
16  
17  import javax.ejb.Local;
18  
19  /**
20   * Local interface for UpgradeStatusSingletonBean.
21   * 
22   * @version $Id: UpgradeStatusSingletonLocal.java 25312 2017-02-21 19:09:38Z jeklund $
23   */
24  @Local
25  public interface UpgradeStatusSingletonLocal {
26  
27      /** @return true of the post-upgrade is running as a background task on this node */
28      boolean isPostUpgradeInProgress();
29  
30      /** @return Log4J logging events from UpgradeSessionBean while post-upgrade background task is running */
31      List<org.apache.log4j.spi.LoggingEvent> getLogged();
32  
33      /** @param newValue value
34       * @return true if successfully claimed the node-local post-upgrade lock */
35      boolean setPostUpgradeInProgressIfDifferent(boolean newValue);
36  
37      /** Reset the node-local post-upgrade lock */
38      void resetPostUpgradeInProgress();
39  
40      /** Start listen to Log4J log events 
41       * @param log log*/
42      void logAppenderAttach(org.apache.log4j.Logger log);
43  
44      /** Stop listen to Log4J log events 
45       * @param log log*/
46      void logAppenderDetach(org.apache.log4j.Logger log);
47  
48  }
