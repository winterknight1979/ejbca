15  import java.util.Collection;
16  import java.util.List;
17  
18  import javax.ejb.Local;
19  
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.rules.AccessRuleData;
22  import org.cesecore.authorization.user.AccessUserAspectData;
23  import org.cesecore.roles.AdminGroupData;
24  import org.cesecore.roles.RoleExistsException;

26  /**
27   * The legacy Roles Management interface manages the list of roles and which access rules applies to defined roles. The roles interface also manages the list
28   * of Subjects who are part of the roles. There are three distinct methods to this interface:
29   * <ul>
30   * <li>
31   * managing the roles, which by default are only a name not associated with anything.</li>
32   * <li>
33   * managing access rules, which makes a role into something that defines what subject can do.</li>
34   * <li>
35   * managing subjects, which makes users part of the role thus giving them the access rights defined by the access rules of the role.</li>
36   * </ul>
37   * 
38   * @see <a href="https://www.cesecore.eu/mediawiki/index.php/Functional_Specifications_(ADV_FSP)#Roles_Management">specs</a>
39   * 
40   * @version $Id: LegacyRoleManagementSessionLocal.java 25569 2017-03-21 23:28:38Z jeklund $
41   */
42  @Deprecated
43  @Local
44  public interface LegacyRoleManagementSessionLocal {
45  
46      /**
47       * Adds a legacy role.
48       * 
49       * @param authenticationToken only used for audit logging
50       * @param roleName Name of the role
51       * @throws RoleExistsException If role by that name already exists.
52       * @return the {@link AdminGroupData} that was created
53       */
54      AdminGroupData create(AuthenticationToken authenticationToken, String roleName) throws RoleExistsException;
55  
56      /**
57       * Removes a legacy role if present. Will also remove all associated access rules and user aspects.
58       * 
59       * @param authenticationToken only used for audit logging
60       * @param roleName The name of the role to remove.
61       */
62      void deleteRole(AuthenticationToken authenticationToken, String roleName);
63  
64      /**
65       * Associates a list of access rules to a role. If the given role already exists, replace it.
66       * 
67       * @param authenticationToken only used for audit logging
68       * @param role The role
69       * @param accessRules A collection of access rules. 
70       * 
71       * @return the merged {@link AdminGroupData} with the new access rules
72       */
73      AdminGroupDataroupData">AdminGroupData addAccessRulesToRole(AuthenticationToken authenticationToken, AdminGroupData role, Collection<AccessRuleData> accessRules);
74  
75      /**
76       * Gives the collection of subjects the given role. If the subject already exists, update it with the new value.
77       * 
78       * @param authenticationToken only used for audit logging
79       * @param subjects A collection of subjects
80       * @param role The role to give.
81       * @return the merged {@link AdminGroupData} with the new subjects
82       */
83      AdminGroupDatainGroupData">AdminGroupData addSubjectsToRole(AuthenticationToken authenticationToken, AdminGroupData role, Collection<AccessUserAspectData> subjects);
84  
85      /**
86       * Retrieves all roles in the database..
87       * 
88       * @return all the roles in the database.
89       */
90      List<AdminGroupData> getAllRoles();
91  
92  
93      /**
94       * Finds a specific role by name.
95       * 
96       * @param roleName Name of the sought role.
97       * @return The sought roll, null if not found
98       */
99      AdminGroupData getRole(String roleName);
100 
101 
102     /**
103      * Creates a super administrator role and a default CLI user. A role and default CLI user is needed in order
104      * to do operations with the CLI (command line interface).  
105      */
106     void createSuperAdministrator();
107 
108     /**
109      * Add the grantedAccessRules to the role when conditions are met.
110      * 
111      * @param authenticationToken only used for audit logging
112      * @param skipWhenRecursiveAccessTo ignore all roles that have recursive access to the rule
113      * @param requiredAccessRules all access rules that the role must be authorized to in order to extend the grants (if a grantedAccessRules is present here it is not required)
114      * @param grantedAccessRules the access rules to grant
115      * @param grantedAccessRecursive true if the access rules should be granted as recursive
116      */
117     void addAccessRuleDataToRolesWhenAccessIsImplied(AuthenticationToken authenticationToken, String skipWhenRecursiveAccessTo, List<String> requiredAccessRules,
118             List<String> grantedAccessRules, boolean grantedAccessRecursive);
119 
120     /**
121      * Set the tokenType to X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE if it was not set before.
122      * 
123      * @param authenticationToken only used for audit logging
124      */
125     void setTokenTypeWhenNull(AuthenticationToken authenticationToken);
126 
127     /**
128      * Delete ALL legacy AdminGroupData, AdminEntityData and AccessRuleData.
129      * 
130      * @param authenticationToken only used for audit logging
131      */
132     void deleteAllRoles(AuthenticationToken authenticationToken);
133 }
