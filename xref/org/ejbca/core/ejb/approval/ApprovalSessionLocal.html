14  package org.ejbca.core.ejb.approval;
15  
16  import java.util.Collection;
17  import java.util.Date;
18  import java.util.List;
19  
20  import javax.ejb.Local;
21  
22  import org.cesecore.authentication.tokens.AuthenticationToken;
23  import org.ejbca.core.model.approval.Approval;
24  import org.ejbca.core.model.approval.ApprovalDataVO;
25  import org.ejbca.core.model.approval.ApprovalRequest;
26  import org.ejbca.core.model.approval.profile.ApprovalProfile;
27  import org.ejbca.util.query.IllegalQueryException;
28  import org.ejbca.util.query.Query;
29  
30  /**
31   * Local interface for ApprovalSession.
32   * 
33   * @version $Id: ApprovalSessionLocal.java 26119 2017-07-03 15:02:08Z oskareriksson $
34   */
35  @Local
36  public interface ApprovalSessionLocal extends ApprovalSession {
37  
38      /**
39       * Send all work-flow related approval notifications given the last added Approval to the approvalsPerformed.
40       * 
41       * @param approvalRequest the approval request
42       * @param approvalProfile the profile determining the approval work flow
43       * @param approvalData the ApprovalData object referring to the approval request
44       * @param expired should be set to true if the notification is due to expiration of the approval request
45       */
46      void sendApprovalNotifications(ApprovalRequest approvalRequest, ApprovalProfile approvalProfile,
47              ApprovalData approvalData, boolean expired);
48  
49  	/**
50  	 * Encode a Collection of Approval and set it in ApprovalData object.
51  	 * @param approvalData data
52  	 * @param approvals cannot be null.
53  	 */
54  	void setApprovals(ApprovalData approvalData, Collection<Approval> approvals);
55  	
56  	ApprovalData findNonExpiredApprovalDataLocal(int approvalId);
57  	
58  	/**
59  	 * Returns a list of all approval requests whose status is WAITING_FOR_APPROVAL only (aka. not including status APPROVED).
60  	 * This method is called mainly when upgrading older approval requests to EJBCA 6.6.0 or later.
61  	 * 
62  	 * @return A list of all approval requests whose status is WAITING_FOR_APPROVAL
63  	 */
64  	List<ApprovalData> findWaitingForApprovalApprovalDataLocal();
65  	
66      /**
67       * Method that takes an approvalId (hash) and returns the request's ID (the value in the "Id" column in the "ApprovalData" table in the database)
68       * 
69       * @param approvalId the request hash (approvalId in the database)
70       * @return the ID of the approval request or 0 if it does not exist
71       */
72      int getIdFromApprovalId(int approvalId);
73  
74  	 /**
75        * Updates the approval request field for an approval. Doesn't do any authorization checks.
76        * 
77        * @param approvalDataId the ID of an approvalData object (not the approvalID hash)
78        * @param approvalRequest the updated approval request
79        */
80       void updateApprovalRequest(final int approvalDataId, final ApprovalRequest approvalRequest);
81       
82       /**
83        * Moves the expiration date forward and sets the status for Waiting for Approval. Doesn't do any authorization checks.
84        * 
85        * @param authenticationToken The administrator requesting extension of the request. Used for audit logging only.
86        * @param approvalDataId Id of the approval request
87        * @param extendForMillis The new expiration date will be set to current time plus this number of milliseconds
88        * @throws IllegalStateException if the request has been approved or denied already.
89        */
90       void extendApprovalRequestNoAuth(AuthenticationToken authenticationToken, int approvalDataId, long extendForMillis);
91       
92       /**
93        * Method returning a list of approvals from the give query
94        * 
95        * @param query should be a Query object containing ApprovalMatch and
96        *            TimeMatch
97        * @param index where the ResultSet should start
98        * @param numberofrows maximum number of rows 
99        * @param caAuthorizationString
100       *            a list of authorized CA Ids in the form 'cAId=... OR cAId=...'
101       * @param endEntityProfileAuthorizationString
102       *            a list of authorized end entity profile ids in the form
103       *            '(endEntityProfileId=... OR endEntityProfileId=...) objects
104       *            only
105       * @return a List of ApprovalDataVO, never null
106       * @throws IllegalQueryException fail
107       */
108      List<ApprovalDataVO> query(final Query query, int index, int numberofrows, String caAuthorizationString,
109              String endEntityProfileAuthorizationString) throws IllegalQueryException;
110      
111      /**
112       * Returns a list of non-expired approvals with the given statuses.
113       * @param includeUnfinished Includes requests that haven't been executed or rejected yet.
114       * @param includeProcessed Includes requests that have been approved and executed, or rejected.
115       * @param includeExpired Includes requests that have expired.
116       * @param startDate Include requests from this date and later, or null for no limit.
117       * @param endDate Include requests up to this date, or null for no limit.
118       * @param expiresBefore When searching unfinished requests, include only those that expire before this date. Set to null to include all.
119       * @param index where the ResultSet should start
120       * @param numberofrows maximum number of rows
121       * @param caAuthorizationString
122       *            a list of authorized CA Ids in the form 'cAId=... OR cAId=...'
123       * @param endEntityProfileAuthorizationString
124       *            a list of authorized end entity profile ids in the form
125       *            '(endEntityProfileId=... OR endEntityProfileId=...) objects
126       *            only
127       * @return a List of ApprovalDataVO, never null
128       */
129      List<ApprovalDataVO> queryByStatus(boolean includeUnfinished, boolean includeProcessed, boolean includeExpired,
130              final Date startDate, final Date endDate, final Date expiresBefore, int index, int numberofrows, String caAuthorizationString,
131              String endEntityProfileAuthorizationString);
132 }
