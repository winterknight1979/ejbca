13  package org.ejbca.core.ejb.approval;
14  
15  import java.util.List;
16  import java.util.Map;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  import org.ejbca.core.model.approval.profile.ApprovalProfile;
21  
22  /** Session bean to manage approval profiles, i.e. add, remove, find
23   * 
24   * In order to add or edit ApprovalProfiles the admin needs access to:
25   * - /ca_functionality/ (StandardRules.CAFUNCTIONALITY)
26   * 
27   * TODO: Generate a separate session beans specifically for performing CRUD operations on ProfileData objects, agnostic of their end purpose. This 
28   *       bean is sufficient for now though.
29   * 
30   * @version $Id: ApprovalProfileSession.java 24037 2016-07-27 15:00:00Z anatom $
31   */
32  public interface ApprovalProfileSession {
33  
34      /**
35       * Adds an approval profile to the database.
36       * 
37       * @param admin administrator performing the task
38       * @param profile the profile to be added
39       * @return the generated approval profile id
40       * @throws ApprovalProfileExistsException trying to add a profile with a name or id that already exists
41       * @throws AuthorizationDeniedException if current administrator is not authorized to modify profiles
42       */
43      int addApprovalProfile(final AuthenticationToken admin, final ApprovalProfile profile) throws ApprovalProfileExistsException, AuthorizationDeniedException;
44      
45      /**
46       * Updates approval profile data
47       * 
48       * @param admin Administrator performing the operation
49       * @param profile profile
50       * @throws AuthorizationDeniedException if current administrator is not authorized to modify profiles
51       */
52      void changeApprovalProfile(final AuthenticationToken admin, final ApprovalProfile profile) throws AuthorizationDeniedException;
53  
54      /**
55       * Adds an approval profile with the same content as the original approval profile.
56       * 
57       * @param admin Administrator performing the operation
58       * @param profile name of original approval profile
59       * @param newname name of new approval profile
60       * @throws ApprovalProfileExistsException if a profile with newname already exists
61       * @throws ApprovalProfileDoesNotExistException if a profile with orgname does not exist
62       * @throws AuthorizationDeniedException if current administrator is not authorized to modify profiles
63       */
64      void cloneApprovalProfile(final AuthenticationToken admin, final ApprovalProfile profile, final String newname) 
65              throws ApprovalProfileExistsException, ApprovalProfileDoesNotExistException, AuthorizationDeniedException;
66  
67      /**
68       * Retrieves a Collection of id:s (Integer) to authorized profiles. Only profiles that the specified administrator is 
69       * authorized to will be returned. 
70       * 
71       * @param admin Administrator performing the operation
72       * @return List of id:s (Integer)
73       */
74      List<Integer> getAuthorizedApprovalProfileIds(final AuthenticationToken admin);
75  
76      /**
77       * Finds an approval profile by id.
78       * 
79       * @param id approval profile id
80       * @return Approval profile (cloned) or null if it can not be found.
81       */
82      ApprovalProfile getApprovalProfile(final int id);
83  
84      /**
85       * Returns an approval profile's name given it's id.
86       * 
87       * @param id approval profile id
88       * @return approval profile name or null if approval profile id does not exist.
89       */
90      String getApprovalProfileName(final int id);
91  
92      /**
93       * Method creating a Map mapping profile id (Integer) to profile name
94       * (String).
95       * 
96       * @return a Map mapping profile id (Integer) to profile name (String) 
97       */
98      Map<Integer, String> getApprovalProfileIdToNameMap();
99      
100     /**
101      * Renames an approval profile
102      * 
103      * @param admin Administrator performing the operation
104      * @param profile the profile to rename
105      * @param newname the new name of the approval profile
106      * @throws ApprovalProfileExistsException if a profile with name newname already exists
107      * @throws ApprovalProfileDoesNotExistException if the profile to be renamed hasn't been persisted.
108      * @throws AuthorizationDeniedException if current administrator is not authorized to rename profiles
109      */
110     void renameApprovalProfile(final AuthenticationToken admin, final ApprovalProfile profile, final String newname)
111             throws ApprovalProfileExistsException, ApprovalProfileDoesNotExistException, AuthorizationDeniedException;
112 
113     /**
114      * Removes an approval profile from the database, does not throw any errors if the profile does not exist.
115      *
116      * @param admin Administrator performing the operation
117      * @param profile the approval profile to remove
118      * @throws AuthorizationDeniedException if current administrator is not authorized to modify profiles
119      */
120     void removeApprovalProfile(final AuthenticationToken admin, final ApprovalProfile profile) throws AuthorizationDeniedException;
121     
122     /**
123      * Removes an approval profile from the database, does not throw any errors if the profile does not exist.
124      *
125      * @param admin Administrator performing the operation
126      * @param id the ID of the approval profile to remove
127      * @throws AuthorizationDeniedException if current administrator is not authorized to modify profiles
128      */
129     void removeApprovalProfile(final AuthenticationToken admin, final int id) throws AuthorizationDeniedException;
130 
131     /**
132      * Forces the profile cache to rebuild. 
133      */
134     void forceProfileCacheRebuild();
135 
136 }
