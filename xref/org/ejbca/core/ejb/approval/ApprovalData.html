285 	/**
286 	 * Method that checks if the request or approval have expired
287 	 * The status is set to expired if it is
288 	 * @return true of the request or approval have expired
289 	 */
290 	public boolean hasRequestOrApprovalExpired() {
291 		final Date currentDate = new Date();
292 		boolean retval = false;
293 		if(currentDate.after(getExpireDate())){
294 			if(getStatus() == ApprovalDataVO.STATUS_WAITINGFORAPPROVAL ||
295 					getStatus() == ApprovalDataVO.STATUS_APPROVED ||
296 					getStatus() == ApprovalDataVO.STATUS_REJECTED){
297 				setStatus(ApprovalDataVO.STATUS_EXPIRED);
298 			}
299 			retval=true;
300 		}
301 		return retval;
302 	}
