13  package org.ejbca.core.ejb.approval;
14  
15  import java.util.Collection;
16  import java.util.List;
17  import java.util.Map;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.certificates.ca.ApprovalRequestType;
22  import org.cesecore.certificates.ca.CAInfo;
23  import org.cesecore.certificates.certificateprofile.CertificateProfile;
24  import org.cesecore.profiles.ProfileData;
25  import org.ejbca.core.model.approval.profile.ApprovalProfile;
26  
27  /**
28   * Session to access approval profiles locally
29   * 
30   * @version $Id: ApprovalProfileSessionLocal.java 26076 2017-06-26 12:49:12Z mikekushner $
31   */
32  @Local
33  public interface ApprovalProfileSessionLocal extends ApprovalProfileSession {
34      
35      
36      /**
37       * @param profileName profile
38       * @return the found entity instance or null if the entity does not exist
39       */
40      List<ProfileData> findByApprovalProfileName(String profileName);
41      
42      /** @return return all approval profiles  as a List. */
43      List<ProfileData> findAllProfiles();
44      
45      /**
46       * @return a map of all existing approval profiles.
47       */
48      Map<Integer, ApprovalProfile> getAllApprovalProfiles();
49      
50      /**
51       * @return a list of all existing approval profiles
52       */
53      Collection<ApprovalProfile> getApprovalProfilesList();
54      
55      /**
56       * Returns the appropriate approval profile for the given action, where any approval profile defined in the certificate profile trumps any 
57       * define in the CA
58       * 
59       * @param action an approval action
60       * @param cainfo a CA information object
61       * @param certProfile a certificate profile
62       * @return the most appropriate profile for the action, or null if none was found.
63       */
64      ApprovalProfile getApprovalProfileForAction(final ApprovalRequestType action, final CAInfo cainfo, final CertificateProfile certProfile);
65  }
