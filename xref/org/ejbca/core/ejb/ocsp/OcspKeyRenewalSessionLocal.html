13  package org.ejbca.core.ejb.ocsp;
14  
15  import java.security.InvalidKeyException;
16  import java.security.KeyStoreException;
17  
18  import javax.ejb.Local;
19  
20  import org.cesecore.keys.token.CryptoTokenOfflineException;
21  
22  /**
23   * @version $Id: OcspKeyRenewalSessionLocal.java 19902 2014-09-30 14:32:24Z anatom $
24   *
25   */
26  @Local
27  public interface OcspKeyRenewalSessionLocal {
28  
29       final String RENEW_ALL_KEYS = "all";
30      
31      /**
32       * This method causes the standalone OCSP responder to renew its keystores. 
33       * 
34       * This method will use the password stored in configuration, and is mainly used for automatic key renewal. Will throw
35       * an exception if passwords can't be set in memory.
36       * 
37       * @param signerSubjectDN subject DN of the signing key to be renewed. The string OcspKeyRenewalSessionLocal.RENEW_ALL_KEYS will result 
38       * @throws KeyStoreException if p11 key store hasn't been activated
39       * @throws CryptoTokenOfflineException if Crypto Token is not available or connected, or key with alias does not exist.
40       * @throws InvalidKeyException if the public key can not be used to verify a string signed by the private key, because the key is wrong or the 
41       * signature operation fails for other reasons such as a NoSuchAlgorithmException or SignatureException.
42       * 
43       */
44      void renewKeyStores(String signerSubjectDN) throws KeyStoreException, CryptoTokenOfflineException, InvalidKeyException; 
45      
46      /**
47       * Cancels all running timers and starts the initial timer. This method should be called from a servlet and be used to start the rekeying 
48       * timer at startup.
49       */
50      void startTimer();
51  
52  }
