13  package org.ejbca.core.ejb.services;
14  
15  import java.util.Collection;
16  import java.util.List;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.ejbca.core.model.services.ServiceConfiguration;
20  import org.ejbca.core.model.services.ServiceExistsException;
21  
22  /**
23   * @version $Id: ServiceSession.java 24490 2016-10-10 11:32:07Z anatom $
24   */
25  public interface ServiceSession {
26  
27      /**
28       * Adds a Service to the database.
29       * @param admin admin
30       * @param name name
31       * @param serviceConfiguration config 
32       * @throws ServiceExistsException if service already exists.
33       */
34      void addService(AuthenticationToken admin, String name, ServiceConfiguration serviceConfiguration) throws ServiceExistsException;
35  
36      /**
37       * Adds a service to the database. Used for importing and exporting profiles
38       * from xml-files.
39       * @param admin admin
40       * @param id id
41       * @param name name
42       * @param serviceConfiguration config 
43       * 
44       * @throws ServiceExistsException if service already exists.
45       */
46      void addService(AuthenticationToken admin, int id, String name, ServiceConfiguration serviceConfiguration) throws ServiceExistsException;
47  
48      /**
49       * Adds a service with the same content as the original.
50       * @param admin admin
51       * @param oldname name
52       * @param newname name
53       * @throws ServiceExistsException if service already exists.
54       */
55      void cloneService(AuthenticationToken admin, String oldname, String newname) throws ServiceExistsException;
56  
57      /** Removes a service from the database. 
58       * @param admin admin
59       * @param name name
60       * @return bool */
61      boolean removeService(AuthenticationToken admin, String name);
62  
63      /**
64       * Renames a service.
65       * @param admin admin
66       * @param oldname name
67       * @param newname name
68       * @throws ServiceExistsException if service already exists.
69       */
70       void renameService(AuthenticationToken admin, String oldname, String newname) throws ServiceExistsException;
71  
72      /**
73       * Retrieves a Collection of id:s (Integer) to visible services.
74       * 
75       * @return Collection of id:s (Integer)
76       */
77      Collection<Integer> getVisibleServiceIds();
78  
79      /**
80       * Retrieves a named service.
81       * @param name name
82       * @return the service configuration or null if it doesn't exist.
83       */
84      ServiceConfiguration getService(String name);
85  
86      /**
87       * Returns a service id, given it's service name
88       * @param name name
89       * @return the id or 0 if the service cannot be found.
90       */
91      int getServiceId(String name);
92  
93      /**
94       * Activates the timer for a named service. The service must already be
95       * previously added.
96       * 
97       * @param admin The administrator performing the action
98       * @param name the name of the service for which to activate the timer
99       */
100      void activateServiceTimer(AuthenticationToken admin, String name);
101 
102     /**
103      * Returns a Service name given its id.
104      * @param id id
105      * @return the name or null if id does not exist
106      */
107     String getServiceName(int id);
108 
109     /**
110      * Checks if a list of certificate profiles is used by any service.
111      * 
112      * @param certificateProfileId IDs of the certificate profile to check
113      * @return a list of ServiceData objects using the given ID, or an empty list if nothing is found
114      */
115     List<String> getServicesUsingCertificateProfile(Integer certificateProfileId);
116  
117     /** Loads and activates all the services from database that are active. */
118     void load();
119 
120     /** Cancels all existing timers a unload. */
121     void unload();
122     
123     /**
124      * Updates service configuration, but does not re-set the timer
125      * @param admin admin
126      * @param name name
127      * @param serviceConfiguration config
128      * @param noLogging if true no logging to the database will be done
129      */
130     void changeService(AuthenticationToken admin, String name, ServiceConfiguration serviceConfiguration, boolean noLogging);
131     
132     /**
133      * Finds a service configuration by id.
134      * @param id id
135      * @return  the service configuration or null if it doesn't exist.
136      */
137     ServiceConfiguration getServiceConfiguration(int id);
138 
139 }
