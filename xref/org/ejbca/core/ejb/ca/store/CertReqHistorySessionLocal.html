13  package org.ejbca.core.ejb.ca.store;
14  
15  import java.math.BigInteger;
16  import java.security.cert.Certificate;
17  import java.util.List;
18  
19  import javax.ejb.Local;
20  
21  import org.cesecore.certificates.certificate.CertificateInfo;
22  import org.cesecore.certificates.endentity.EndEntityInformation;
23  import org.ejbca.core.model.ca.store.CertReqHistory;
24  
25  /**
26   * Local interface for CertificateStoreSession.
27   * @version $Id: CertReqHistorySessionLocal.java 19902 2014-09-30 14:32:24Z anatom $
28   */
29  @Local
30  public interface CertReqHistorySessionLocal extends CertReqHistorySession {
31  
32      /**
33       * Method used to add a CertReqHistory to database
34       * 
35       * @param cert the certificate to store (Only X509Certificate used for now)
36       * @param endEntityInformation the user information used when issuing the certificate.
37       */
38      void addCertReqHistoryData(Certificate cert, EndEntityInformation endEntityInformation);
39      
40  	CertificateInfo findFirstCertificateInfo(String issuerDN, BigInteger serno);
41  	
42      /**
43       * Method to remove CertReqHistory data.
44       *
45       * @param certFingerprint the primary key.
46       */
47      void removeCertReqHistoryData(String certFingerprint);
48      
49      /**
50       * Retrieves the certificate request data belonging to given certificate serialnumber and issuerdn
51       * 
52       * NOTE: This method will try to repair broken XML and will in that case
53       * update the database. This means that this method must always run in a
54       * transaction! 
55       * 
56       * @param certificateSN serial number of the certificate
57       * @param issuerDN DN
58       * @return the CertReqHistory or null if no data is stored with the certificate.
59       */
60      CertReqHistory retrieveCertReqHistory(BigInteger certificateSN, String issuerDN);
61  
62      /**
63       * NOTE: This method will try to repair broken XML and will in that case
64       * update the database. This means that this method must always run in a
65       * transaction! 
66       * @param username User
67       * 
68       * @return all certificate request data belonging to a user.
69       */
70      List<CertReqHistory> retrieveCertReqHistory(String username);
71  }
