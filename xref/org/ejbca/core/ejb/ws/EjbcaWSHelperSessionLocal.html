46  /**
47   * Local interface for EjbcaWSHelperSession. These methods are used from EjbcaWS only
48   * @version $Id: EjbcaWSHelperSessionLocal.java 29326 2018-06-25 12:45:51Z bastianf $
49   */
50  @Local
51  public interface EjbcaWSHelperSessionLocal extends EjbcaWSHelperSession {
52  
53      /**
54       * Gets an AuthenticationToken object for a WS-API administrator authenticated with the given client certificate.
55       * - Checks (through authenticationSession.authenticate) that the certificate is valid
56       * - If (WebConfiguration.getRequireAdminCertificateInDatabase) checks (through authenticationSession.authenticate) that the admin certificate is not revoked.
57       * - If (allowNonAdmin == false), checks that the admin have access to /administrator, i.e. really is an administrator with the certificate mapped in an admin role. 
58       *   Does not check any other authorization though, other than that it is an administrator.
59       * 
60       * @param cert The X.509 client certificate.
61       * @param allowNonAdmins false if we should verify that it is a real administrator, true only extracts the certificate and checks that it is not revoked.
62       * @return AuthenticationToken object based on the SSL client certificate
63       * @throws AuthorizationDeniedException if no client certificate or allowNonAdmins == false and the cert does not belong to an admin
64       */
65      AuthenticationToken getAdmin(final boolean allowNonAdmins, final X509Certificate cert) throws AuthorizationDeniedException;
66      
67      /**
68       * Method that converts profile names etc to corresponding ID's
69       * @param admin admin
70       * @param usermatch a usermatch containing names of profiles
71       * @return a query containing id's of profiles.
72       * @throws AuthorizationDeniedException if searching by a CA which the admin is not authorized to.
73       * @throws CADoesntExistsException if searching for a non-existent CA.
74       * @throws EndEntityProfileNotFoundException if usermatch was for and end entity profile, and that profile didn't exist
75       */
76      Query convertUserMatch(AuthenticationToken admin, UserMatch usermatch) throws CADoesntExistsException,
77          AuthorizationDeniedException, EndEntityProfileNotFoundException;
78  
79      /**
80       * Checks authorization for each certificate and optionally check that it's valid. Does not check revocation status. 
81       * @param admin is the admin used for authorization
82       * @param certs is the collection of certs to verify
83       * @param validate set to true to perform validation of each certificate
84       * @param nowMillis current time
85       * @return a List of valid and authorized certificates
86       */
87      List<Certificate> returnAuthorizedCertificates(final AuthenticationToken admin, Collection<java.security.cert.Certificate> certs, boolean validate, long nowMillis);
88  
89      /**
90       * Verifies that the given certificate has a correct signature and is currently valid, and
91       * if so, sets the end entity status to NEW and sets its password. Used from  EjbcaWS.cvcRequest during renewals.
92       * @param admin authentication token
93       * @param cert certificate to check validity and signature of.
94       * @param username username of end entity to update.
95       * @param password new password to set for end entity.
96       * @throws CertificateNotYetValidException if the certificate is not yet valid
97       * @throws CertificateExpiredException if the certificate has expired
98       * @throws EndEntityProfileValidationException if the password does not satisfy the requirements in the end entity profile
99       * @throws NoSuchEndEntityException if there's no end entity with the given username
100      * @throws AuthorizationDeniedException if not authorized to the given end entity
101      * @throws ApprovalException if an approval error occurred while setting end entity status 
102      * @throws WaitingForApprovalException if approval is required for setting the end entity status
103      */
104     void checkValidityAndSetUserPassword(AuthenticationToken admin, java.security.cert.Certificate cert, String username, String password) 
105             throws CertificateNotYetValidException, CertificateExpiredException, EndEntityProfileValidationException,
106             AuthorizationDeniedException, NoSuchEndEntityException, ApprovalException, WaitingForApprovalException;
107 
108     /**
109      * Resets the password and restores the status for the given end entity.
110      * Used for restoring the end entity status if a certificate request operation failed.
111      * All exceptions are swallowed and error-logged, since this method is intended to be called from catch blocks.
112      * @param admin authentication token.
113      * @param username username of end entity
114      * @param status original end entity status to restore
115      */
116     void resetUserPasswordAndStatus(AuthenticationToken admin, String username, int status);
117     
118     /**
119      * @param admin admin
120      * @param caname name
121      * @param cachain cert
122      * @param regenerateKeys keys 
123      * @param usenextkey bool
124      * @param activatekey bool
125      * @param keystorepwd krys
126      * @return req
127      * @throws CADoesntExistsException fail 
128      * @throws AuthorizationDeniedException fai; 
129      * @throws CertPathValidatorException fail
130      * @throws CryptoTokenOfflineException fail
131      * @throws CryptoTokenAuthenticationFailedException fail 
132      * @see org.ejbca.core.protocol.ws.common.IEjbcaWS#caRenewCertRequest 
133      */
134     byte[] caRenewCertRequest(AuthenticationToken admin, String caname, List<byte[]> cachain, boolean regenerateKeys, boolean usenextkey, boolean activatekey, String keystorepwd) 
135         throws CADoesntExistsException, AuthorizationDeniedException, CertPathValidatorException, CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException;
136 
137     /**
138      * @param admin admin
139      * @param caname ca
140      * @param certbytes certs
141      * @throws AuthorizationDeniedException fail
142      * @throws CAExistsException fail
143      * @throws CertificateImportException fail
144      * @throws EjbcaException fail
145      * @throws CertificateParsingException fail
146      * @throws IllegalCryptoTokenException fail
147      * @see org.ejbca.core.protocol.ws.common.IEjbcaWS#importCaCert
148      */
149     void importCaCert(AuthenticationToken admin, String caname, byte[] certbytes) throws AuthorizationDeniedException, 
150         CAExistsException, IllegalCryptoTokenException, CertificateImportException, EjbcaException, CertificateParsingException;
151     
152     /**
153      * @param admin admin
154      * @param caname ca
155      * @param certbytes cert 
156      * @throws AuthorizationDeniedException fail
157      * @throws CADoesntExistsException fail
158      * @throws CertificateImportException fail
159      * @throws EjbcaException fail
160      * @throws CertificateParsingException fail
161      * @see org.ejbca.core.protocol.ws.common.IEjbcaWS#updateCaCert
162      */
163      void updateCaCert(AuthenticationToken admin, String caname, byte[] certbytes) throws AuthorizationDeniedException, 
164          CADoesntExistsException, CertificateImportException, EjbcaException, CertificateParsingException; 
165 
166      /** Makes a CA certificate that has been created prior to its "not before" date active (CA rollover) 
167      * @param admin admin
168      * @param caname ca
169      * @throws AuthorizationDeniedException fail 
170      * @throws CADoesntExistsException fail
171      * @throws CryptoTokenOfflineException fail*/
172      void rolloverCACert(AuthenticationToken admin, String caname) throws AuthorizationDeniedException, CADoesntExistsException, CryptoTokenOfflineException;
173      
174      /**
175       * @param admin admin
176      * @param caname ca
177      * @param cert cart
178      * @param cachain chain
179      * @param keystorepwd pwd
180      * @param futureRollover bool
181      * @throws AuthorizationDeniedException fail
182       * @throws EjbcaException fail
183       * @throws ApprovalException fail
184       * @throws WaitingForApprovalException fail
185       * @throws CertPathValidatorException fail
186       * @throws CesecoreException fail
187       * @throws CertificateParsingException fail 
188       * @see org.ejbca.core.protocol.ws.common.IEjbcaWS#caRenewCertRequest 
189       */
190      void caCertResponse(AuthenticationToken admin, String caname, byte[] cert, List<byte[]> cachain, String keystorepwd, boolean futureRollover) 
191          throws AuthorizationDeniedException, EjbcaException, ApprovalException, WaitingForApprovalException, CertPathValidatorException, CesecoreException, CertificateParsingException;
192 
193      void isAuthorizedToHardTokenData(AuthenticationToken admin, String username, boolean viewPUKData) throws AuthorizationDeniedException, EjbcaException;
194 
195      /**
196       * Help method returning a subset of certificates containing only valid certificates
197       * expiredate and revocation status is checked.
198      * @param admin admin
199      * @param certs carts
200      * @return valid certs
201       * @throws ClassCastException fasil
202       */
203      Collection<java.security.cert.Certificate> returnOnlyValidCertificates(AuthenticationToken admin, Collection<java.security.cert.Certificate> certs);
204      
205      /**
206       * Method used to convert a HardToken data to a WS version
207       * @param data data
208      * @param certificates certs
209      * @param includePUK bool
210      * @return ws
211       * @throws EjbcaException fail 
212       */
213      HardTokenDataWS convertHardTokenToWS(HardTokenInformation data, Collection<java.security.cert.Certificate> certificates, boolean includePUK) throws EjbcaException;
214      
215      void isAuthorizedToRepublish(AuthenticationToken admin, String username, int caid) throws AuthorizationDeniedException, EjbcaException;
216 }
