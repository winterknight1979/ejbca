44  @Local
45  public interface ScepMessageDispatcherSessionLocal extends ScepMessageDispatcherSession {
46  
47      /**
48       * Handles received SCEP message. 
49       * 
50       * @param authenticationToken the origin of the request
51       * @param operation desired SCEP operation to perform
52       * @param message to dispatch
53       * @param scepConfigurationAlias name of alias containing SCEP configuration
54       * @return byte array containing dispatch response. Content depends on operation
55       * @throws NoSuchAliasException fail
56       * @throws CertificateEncodingException fail
57       * @throws AuthorizationDeniedException fail
58       * @throws CADoesntExistsException fail
59       * @throws CertificateRenewalException fail
60       * @throws CertificateExtensionException fail
61       * @throws CertificateException fail
62       * @throws SignatureException fail
63       * @throws InvalidAlgorithmException fail
64       * @throws CAOfflineException fail
65       * @throws IllegalValidityException fail
66       * @throws CertificateSerialNumberException fail
67       * @throws CertificateRevokeException fail
68       * @throws CertificateCreateException fail
69       * @throws IllegalNameException fail
70       * @throws AuthLoginException fail
71       * @throws AuthStatusException fail
72       * @throws SignRequestSignatureException fail
73       * @throws SignRequestException fail
74       * @throws IllegalKeyException fail
75       * @throws CryptoTokenOfflineException fail
76       * @throws CustomCertificateSerialNumberException fail
77       * @throws NoSuchEndEntityException fail
78       */
79      byte[] dispatchRequest(AuthenticationToken authenticationToken, String operation, String message, String scepConfigurationAlias) throws NoSuchAliasException, 
80          CertificateEncodingException, CADoesntExistsException, AuthorizationDeniedException, NoSuchEndEntityException, CustomCertificateSerialNumberException, 
81          CryptoTokenOfflineException, IllegalKeyException, SignRequestException, SignRequestSignatureException, AuthStatusException, AuthLoginException, IllegalNameException, 
82          CertificateCreateException, CertificateRevokeException, CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException, 
83          SignatureException, CertificateException, CertificateExtensionException, CertificateRenewalException;
84  }
