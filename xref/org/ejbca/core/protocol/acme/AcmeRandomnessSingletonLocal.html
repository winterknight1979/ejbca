14  package org.ejbca.core.protocol.acme;
15  
16  /**
17   * Source of required randomness used in the ACME protocol.
18   * 
19   * @version $Id: AcmeRandomnessSingletonLocal.java 29657 2018-08-15 14:18:12Z tarmo_r_helmes $
20   */
21  public interface AcmeRandomnessSingletonLocal {
22  
23      /**
24       * https://tools.ietf.org/html/draft-ietf-acme-acme-12#section-8.3
25       * 
26       * "  token (required, string):  A random value that uniquely identifies
27       *       the challenge.  This value MUST have at least 128 bits of entropy.
28       *       It MUST NOT contain any characters outside the base64url alphabet,
29       *       and MUST NOT include base64 padding characters ("=")."
30       *       
31       *  128 bits = 16 bytes
32       * @return token
33       */
34      String generateAcmeChallengeToken();
35  
36      /** @return the end entity password used to finalize the ACME order */
37      String generateAcmeOrderEnrollmentCode();
38  
39      /** @param byteCount count
40       * @return a new unique identity for this local node (of the requested size) */
41      byte[] generateAcmeNodeId(int byteCount);
42  
43      String generateAcmeAccountId();
44  
45      String generateAcmeChallengeId();
46  
47      byte[] generateReplayNonceSharedSecret(int byteCount);
48  
49  }
