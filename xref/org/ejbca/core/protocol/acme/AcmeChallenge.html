13  package org.ejbca.core.protocol.acme;
14  
15  import java.util.LinkedHashMap;
16  import org.ejbca.core.protocol.acme.AcmeIdentifier.AcmeIdentifierTypes;
17  
18  /**
19   * An ACME Challenge is a proof a client needs to provide in order to be authorized to get a certificate for an identifier.
20   * 
21   * PROCESSING constant in AcmeChallengeStatus ENUM is a requirement imposed by draft-ietf-acme-acme-12 and is preserved for
22   * future use. 
23   * 
24   * @version $Id: AcmeChallenge.java 30434 2018-11-08 07:40:52Z andrey_s_helmes $
25   */
26  public interface AcmeChallenge {
27  
28      String getChallengeId();
29  
30      void setChallengeId(String challengeId);
31  
32      String getAuthorizationId();
33  
34      void setAuthorizationId(String authorizationId);
35  
36      String getType();
37  
38      void setType(String type);
39  
40      String getUrl();
41  
42      void setUrl(String url);
43  
44      AcmeChallengeStatus getStatus();
45  
46      void setStatus(AcmeChallengeStatus status);
47  
48      String getValidated();
49  
50      void setValidated(String validated);
51  
52      String getToken();
53  
54      void setToken(String token);
55  
56      String getKeyAuthorization();
57  
58      void setKeyAuthorization(String keyAuthorization);
59  
60      float getLatestVersion();
61  
62      void upgrade();
63      
64      LinkedHashMap<Object, Object> getRawData();
65  
66      public static enum AcmeChallengeType {
67          DNS_HTTP_01(AcmeIdentifierTypes.DNS, "http-01"),
68          DNS_DNS_01(AcmeIdentifierTypes.DNS, "dns-01"),
69          ;
70  
71          private final AcmeIdentifierTypes acmeIdentifierType;
72          private final String challengeType;
73  
74          private AcmeChallengeType(final AcmeIdentifierTypes acmeIdentifierType, final String challengeType) {
75              this.acmeIdentifierType = acmeIdentifierType;
76              this.challengeType = challengeType;
77          }
78  
79          public AcmeIdentifierTypes getAcmeIdentifierType() { return acmeIdentifierType; }
80          public String getChallengeType() { return challengeType; }
81      }
82  }
