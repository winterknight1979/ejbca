38      private static final int ARG_CANAME                     = 1;
39      private static final int ARG_CADN                       = 2;
40      private static final int ARG_CATYPE                     = 3;
41      private static final int ARG_VALIDITY_IN_DAYS           = 4;
42      private static final int ARG_CERT_PROFILE               = 5;
43      private static final int ARG_SIGN_ALG                   = 6;
44      private static final int ARG_SIGNED_BY_CAID             = 7;
45      private static final int ARG_CRYPTOTOKEN_NAME           = 8;
46      private static final int ARG_PURPOSE_KEY_MAPPING_PATH   = 9;
47      private static final int ARG_CA_PROPERTIES_PATH         = 10;

49      /**
50       * Creates a new instance of Command
51       *
52       * @param args command line arguments
53       */
54      public CreateCACommand(String[] args) {
55          super(args);
56      }

58      /**
59       * Runs the command
60       *
61       * @throws IllegalAdminCommandException Error in command args
62       * @throws ErrorAdminCommandException Error running command
63       */
64      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
65          try {   
66              if (args.length < 10 ) {
67                  getPrintStream().println("Error. Too few arguments: "+args.length);
68                  usage();
69                  System.exit(-1); // NOPMD, this is not a JEE app
70              }
71  
72              String caname = args[ARG_CANAME];
73              String cadn = args[ARG_CADN];
74              String catype = args[ARG_CATYPE];
75              String validityString = args[ARG_VALIDITY_IN_DAYS];
76              String certprofile = StringUtils.equalsIgnoreCase(args[ARG_CERT_PROFILE], "null")? null : args[ARG_CERT_PROFILE];
77              String signalg = args[ARG_SIGN_ALG];
78              int signedByCAId = Integer.parseInt(args[ARG_SIGNED_BY_CAID]);
79              String cryptotokenName = args[ARG_CRYPTOTOKEN_NAME];
80  
81              Properties mapping = new Properties();
82              mapping.load(new FileInputStream(new File(args[ARG_PURPOSE_KEY_MAPPING_PATH])));
83              ArrayList<KeyValuePair> purposeKeyMapping = getKeyValuePairListFromProperties(mapping);
84              
85              ArrayList<KeyValuePair> caproperties = new ArrayList<KeyValuePair>();
86              if(args.length > 10) {
87                  Properties props = new Properties();
88                  props.load(new FileInputStream(new File(args[ARG_CA_PROPERTIES_PATH])));
89                  caproperties = getKeyValuePairListFromProperties(props);
90              }
91                          
92              final long validityInDays = Long.valueOf(validityString);
93              getEjbcaRAWS().createCA(caname, cadn, catype, validityInDays, certprofile, signalg, signedByCAId, cryptotokenName, 
94                      purposeKeyMapping, caproperties);
95              getPrintStream().println("Create new CA: " + caname);
96          } catch (Exception e) {
97              if (e instanceof EjbcaException_Exception) {
98                  EjbcaException_Exception e1 = (EjbcaException_Exception)e;
99                  getPrintStream().println("Error code: " + e1.getFaultInfo().getErrorCode().getInternalErrorCode());
100             }
101             ErrorAdminCommandExceptionhtml#ErrorAdminCommandException">ErrorAdminCommandException adminexp = new ErrorAdminCommandException(e);
102             getPrintStream().println("Error message: " + adminexp.getLocalizedMessage());
103         }
104         
105     }

107     protected void usage() {
108         getPrintStream().println("Command used to create a new CA with a specific existing cryptotoken");
109         getPrintStream().println("Usage : createca <caname> <cadn> <catype> <validity> <certprofile> <signalg> <signedByCAID> " +
110         		"<cryptotokenName> <pathToKeyPurposeMappingFile> [<pathToCAPropertiesFile>]");
111         getPrintStream().println();
112         getPrintStream().println("'caname' is the name of the CA");
113         getPrintStream().println("'cadn' is the subjectDN av the CA");
114         getPrintStream().println("'catype' can be either 'x509' or 'cvc'.");
115         getPrintStream().println("'validity' is the validity of the CA in days");
116         getPrintStream().println("'certprofile' the certificate profile that should be used when issueing the CA certificate instead of the default profiles ROOTCA or SUBCA. Use 'null' for default profiles.");
117         
118         StringBuilder availableSignAlgs = new StringBuilder();
119         for (String algorithm : AlgorithmConstants.AVAILABLE_SIGALGS) {
120             availableSignAlgs.append((availableSignAlgs.length() == 0 ? "" : ", ") + algorithm);
121         }
122         getPrintStream().println("'signalg' can be any of: " + availableSignAlgs);
123         getPrintStream().println("'signedByCAID' is the ID of a CA that will sign this CA. Use '1' for self signed CA (i.e. a root CA). CAs created using the WS cannot be signed by external CAs");
124         getPrintStream().println("'cryptotokenName' is the name of the cryptotoken associated with the CA");
125         getPrintStream().println("'pathToKeyPurposeMappingFile' is the path to a .properties file containing the mapping of the cryptotoken keys and their purpose");
126         getPrintStream().println("'pathToCAPropertiesFile' is the path to a .properties file containing additional CA properties. Optional");
127         getPrintStream().println();
128         getPrintStream().println("Please note that the policy ID is specified in the properties file containing additional properties with the property key 'policyid'.");
129         getPrintStream().println("The policy ID can be 'null' if no Certificate Policy extension should be present, or\nobjectID as '2.5.29.32.0' or objectID and cpsurl "
130                 + "as \"2.5.29.32.0 http://foo.bar.com/mycps.txt\". You can add multiple policies such as "
131                 + "\"2.5.29.32.0 http://foo.bar.com/mycps.txt 1.1.1.1.1 http://foo.bar.com/111cps.txt\".");
132     }

134     private ArrayList<KeyValuePair> getKeyValuePairListFromProperties(Properties properties) {
135         ArrayList<KeyValuePair> kvlist = new ArrayList<KeyValuePair>();
136         Iterator<Object> itr = properties.keySet().iterator();
137         while(itr.hasNext()) {
138             String key = (String) itr.next();
139             KeyValuePair kvp = new KeyValuePair();
140             kvp.setKey(key);
141             kvp.setValue(properties.getProperty(key));
142             kvlist.add(kvp);
143         }
144         return kvlist;
145     }
