20  import java.util.Collection;
21  import java.util.Iterator;
22  import java.util.List;
23  
24  import org.cesecore.util.CertTools;
25  import org.cesecore.util.CryptoProviderTools;
26  import org.cesecore.util.FileTools;
27  import org.ejbca.core.protocol.ws.client.gen.EjbcaException_Exception;

29  import org.ejbca.ui.cli.IAdminCommand;

33  /**
34   * Imports a certificate issued by an external CA. Used to import certificates sent as a reply to a request created with CaCertRequestCommand.
35   *
36   * @version $Id: CaCertResponseCommand.java 22553 2016-01-11 13:06:46Z mikekushner $
37   */
38  public class CaCertResponseCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
39  
40  
41  	private static final int ARG_CANAME           = 1;
42  	private static final int ARG_CERT             = 2;
43  	private static final int ARG_CACHAIN          = 3;
44  	private static final int ARG_KEYSTOREPWD      = 4;
45  
46  	/**
47  	 * Creates a new instance of CvcRequestCommand
48  	 *
49  	 * @param args command line arguments
50  	 */
51  	public CaCertResponseCommand(String[] args) {
52  		super(args);
53  	}
54  
55  	/**
56  	 * Runs the command
57  	 *
58  	 * @throws IllegalAdminCommandException Error in command args
59  	 * @throws ErrorAdminCommandException Error running command
60  	 */
61  	public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
62  
63  		try {   
64  			if(args.length < 4 || args.length > 5){
65  				getPrintStream().println("Number of arguments: "+args.length);
66  				usage();
67  				System.exit(-1); // NOPMD, this is not a JEE app
68  			}
69  
70  			String caname = args[ARG_CANAME];
71  			String certfile = args[ARG_CERT];
72  			String cachainfile = args[ARG_CACHAIN];
73  			String keystorepwd = null;
74  			if (args.length > 5) {
75  				keystorepwd = args[ARG_KEYSTOREPWD];				
76  			}
77  			if (keystorepwd == null) {
78  				// prompt for keystore password
79  				 System.out.print("Enter CA token password: ");
80  			
81  				 keystorepwd = String.valueOf(System.console().readPassword());            	
82  			}
83  
84  			getPrintStream().println("Importing certificate for CA: "+caname);
85  			getPrintStream().println("Certificate filename: "+certfile);
86  			getPrintStream().println("CA chain filename: "+cachainfile);
87  			//getPrintStream().println("CA token password: "+keystorepwd);                        
88  
89  			CryptoProviderTools.installBCProvider();
90  
91  			Certificate incert = null;
92  			try {
93  				FileInputStream in = new FileInputStream(certfile);
94  				Collection<Certificate> certs = CertTools.getCertsFromPEM(in, Certificate.class);
95  				Iterator<Certificate> iter = certs.iterator();
96  				if (iter.hasNext()) {
97  					incert = (Certificate)iter.next();
98  				}
99  			} catch (IOException e) {
100 				// It was perhaps not a PEM chain...see if it was a single binary CVC certificate
101 				byte[] bytes = FileTools.readFiletoBuffer(certfile);
102 				incert = CertTools.getCertfromByteArray(bytes, Certificate.class); // check if it is a good cert, decode PEM if it is PEM, etc
103 			}
104 
105 			getPrintStream().println("Importing certificate with subjectDN '"+CertTools.getSubjectDN(incert)+"', and issuerDN '"+CertTools.getIssuerDN(incert)+"'.");
106 
107 			List<byte[]> cachain = new ArrayList<byte[]>();
108 			try {
109 				FileInputStream in = new FileInputStream(cachainfile);
110 				Collection<Certificate> certs = CertTools.getCertsFromPEM(in, Certificate.class);
111 				Iterator<Certificate> iter = certs.iterator();
112 				while (iter.hasNext()) {
113 					Certificate c = iter.next();
114 					cachain.add(c.getEncoded());
115 				}
116 			} catch (IOException e) {
117 				// It was perhaps not a PEM chain...see if it was a single binary CVC certificate
118 				byte[] bytes = FileTools.readFiletoBuffer(cachainfile);
119 				Certificate c = CertTools.getCertfromByteArray(bytes, Certificate.class); // check if it is a good cert, decode PEM if it is PEM, etc
120 				cachain.add(c.getEncoded());
121 			}
122 	        getEjbcaRAWS().caCertResponse(caname, incert.getEncoded(), cachain, keystorepwd);
123 			getPrintStream().println("Imported CA certificate.");	        
124 		} catch (Exception e) {
125 			if (e instanceof EjbcaException_Exception) {
126 				EjbcaException_Exception e1 = (EjbcaException_Exception)e;
127 				getPrintStream().println("Error code is: "+e1.getFaultInfo().getErrorCode().getInternalErrorCode());
128 			}
129 			throw new ErrorAdminCommandException(e);
130 		}
131 	}
132 
133 	protected void usage() {
134 		getPrintStream().println("Command used to import a certificate from an external CA. Can be X.509 or CVC. Used to receive certificate responses to request created with 'cacertrequest' command.");
135 		getPrintStream().println("Usage : cacertrequest <caname> <certfile> <cachainfile> [<CA token password>]\n\n");
136 		getPrintStream().println("Caname is the name of the CA that will generate the request.");
137 		getPrintStream().println("Certfile is a file with the certificate issued by the external CA. This is a file with a single PEM or binary certificate.");
138 		getPrintStream().println("Cachainfile is a file with the certificate chain of the external CA. This can be a file with several PEM certificates in it, or a file with a single PEM or binary Root CA certificate.");
139 		getPrintStream().println("CA token password is needed if importing this certificate means that a new CA signing key pair must be activated. If not given this command will prompt for the input. If you are certain that a new key pair will not be activated you can give any input as this password will be ignored.");
140 	}
