39      public GenerateCryptoTokenKeysCommand(String[] args) {

43      /**
44       * Runs the command
45       *
46       * @throws IllegalAdminCommandException Error in command args
47       * @throws ErrorAdminCommandException Error running command
48       */
49      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
50          try {   
51              if (args.length < 4 ) {
52                  getPrintStream().println("Error. Too few arguments: "+args.length);
53                  usage();
54                  System.exit(-1); // NOPMD, this is not a JEE app
55              }
56  
57              String cryptotokenName = args[ARG_CRYPTOTOKEN_NAME];
58              String keyAlias = args[ARG_KEY_PAIR_ALIAS];
59              String keySpec = args[ARG_KEY_SPECIFICATION];
60              
61              getEjbcaRAWS().generateCryptoTokenKeys(cryptotokenName, keyAlias, keySpec);
62              getPrintStream().println("Generate key with alias '" + keyAlias + "' in cryptotoken '" + cryptotokenName + "'");
63          } catch (Exception e) {
64              if (e instanceof EjbcaException_Exception) {
65                  EjbcaException_Exception e1 = (EjbcaException_Exception)e;
66                  getPrintStream().println("Error code: " + e1.getFaultInfo().getErrorCode().getInternalErrorCode());
67              }

69              getPrintStream().println("Error message: " + adminexp.getLocalizedMessage());
70          }
71      }
72  
73      protected void usage() {
74          getPrintStream().println("Command used to generate a keypair for a specific cryptotoken");
75          getPrintStream().println("Usage : generatectkeys <cryptotokenName> <keypairAlias> <keySpecification>");
76          getPrintStream().println();
77          getPrintStream().println("Examples of key specifications: RSA2048, secp256r1, DSA1024, gost3410, dstu4145");
78      }
