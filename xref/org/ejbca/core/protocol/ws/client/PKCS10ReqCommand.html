31  /**
32   * Request a certificate given a pkcs10
33   *
34   * @version $Id: PKCS10ReqCommand.java 24815 2016-12-01 11:25:30Z anatom $
35   */
36  public class PKCS10ReqCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
37  
38  	
39  	private static final int ARG_USERNAME                 = 1;
40  	private static final int ARG_PASSWORD                 = 2;
41  	private static final int ARG_PKCS10                   = 3;
42  	private static final int ARG_ENCODING                 = 4;
43  	private static final int ARG_HARDTOKENSN              = 5;
44  	private static final int ARG_OUTPUTPATH               = 6;
45  	
46      /**
47       * Creates a new instance of PKCS10ReqCommand
48       *
49       * @param args command line arguments
50       */
51      public PKCS10ReqCommand(String[] args) {
52          super(args);
53      }
54  
55      /**
56       * Runs the command
57       *
58       * @throws IllegalAdminCommandException Error in command args
59       * @throws ErrorAdminCommandException Error running command
60       */
61      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
62   
63          try {   
64             
65              if(args.length <  6 || args.length > 7){
66              	usage();
67              	System.exit(-1); // NOPMD, it's not a JEE app
68              }
69              
70              String username = args[ARG_USERNAME];            
71              String password = args[ARG_PASSWORD];
72              String pkcs10 = getPKCS10(args[ARG_PKCS10]);
73              String encoding = getEncoding(args[ARG_ENCODING]);
74              String hardtokensn = getHardTokenSN(args[ARG_HARDTOKENSN]);
75              
76              String outputPath = null;
77              if(args.length == 7){
78                outputPath = getOutputPath(args[ARG_OUTPUTPATH]);
79              }
80              
81              try{
82              	CertificateResponse result = getEjbcaRAWS().pkcs10Request(username,password,pkcs10,hardtokensn,CertificateHelper.RESPONSETYPE_CERTIFICATE);
83              	
84              	if(result==null){
85              		getPrintStream().println("No certificate could be generated for user, check server logs for error.");
86              	}else{
87              		String filepath = username;
88              		if(encoding.equals("DER")){
89              			filepath += ".cer";
90              		}else{
91              			filepath += ".pem";
92              		}
93              		if(outputPath != null){
94              			filepath = outputPath + "/" + filepath;
95              		}
96              		
97              		
98              		if(encoding.equals("DER")){
99              			FileOutputStream fos = new FileOutputStream(filepath);
100             			fos.write(CertificateHelper.getCertificate(result.getData()).getEncoded());
101             			fos.close();
102             		}else{
103             			FileOutputStream fos = new FileOutputStream(filepath);
104             			ArrayList<java.security.cert.Certificate> list = new ArrayList<java.security.cert.Certificate>();
105             			list.add(CertificateHelper.getCertificate(result.getData()));
106             			fos.write(CertTools.getPemFromCertificateChain(list));
107             			fos.close();            				            				
108             		}
109             		getPrintStream().println("Certificate generated, written to " + filepath);
110             	}
111             	             
112             }catch(AuthorizationDeniedException_Exception e){
113             	getPrintStream().println("Error : " + e.getMessage());            
114             }
115         } catch (Exception e) {
116             throw new ErrorAdminCommandException(e);
117         }
118     }

124 	private String getHardTokenSN(String hardtokensn) {
125 		if(hardtokensn.equalsIgnoreCase("NONE")){
126 		  return null;
127 		}
128 		
129 		return hardtokensn;
130 	}
131 
132 	private String getPKCS10(String pkcs10Path) {
133 		String retval=null;
134 		try {
135 			FileInputStream fis = new FileInputStream(pkcs10Path);
136 			byte[] contents = new byte[fis.available()];
137 			fis.read(contents);			
138 			fis.close();
139 			retval = new String(contents);
140 		} catch (FileNotFoundException e) {
141 			getPrintStream().println("Error : PKCS10 file could not be found.");
142 			System.exit(-1); // NOPMD, it's not a JEE app		
143 		} catch (IOException e) {
144 			getPrintStream().println("Error reading content of PKCS10 file.");
145 			System.exit(-1); // NOPMD, it's not a JEE app	
146 		}
147 		
148 		
149 		return retval;
150 	}
151 
152 	private String getOutputPath(String outputpath) {
153 		File dir = new File(outputpath);
154 		if(!dir.exists()){
155 			getPrintStream().println("Error : Output directory doesn't seem to exist.");
156 			System.exit(-1); // NOPMD, it's not a JEE app
157 		}
158 		if(!dir.isDirectory()){
159 			getPrintStream().println("Error : Output directory doesn't seem to be a directory.");
160 			System.exit(-1); // NOPMD, it's not a JEE app			
161 		}
162 		if(!dir.canWrite()){
163 			getPrintStream().println("Error : Output directory isn't writeable.");
164 			System.exit(-1); // NOPMD, it's not a JEE app
165 
166 		}
167 		return outputpath;
168 	}
169 
170 	private String getEncoding(String encoding) {
171 		if(!encoding.equalsIgnoreCase("PEM") && !encoding.equalsIgnoreCase("DER")){
172 			usage();
173 			System.exit(-1); // NOPMD, it's not a JEE app
174 		}
175 		
176 		return encoding.toUpperCase();
177 	}

181 	protected void usage() {
182 		getPrintStream().println("Command used to generate a users certificate");
183 		getPrintStream().println("Usage : pkcs10req <username> <password> <pkcs10path> <encoding (DER|PEM)> <hardtokensn (or NONE)> <outputpath (optional)> \n\n");       
184         getPrintStream().println("outputpath : directory where certificate is written in form username+.cer|.pem ");
185    }
