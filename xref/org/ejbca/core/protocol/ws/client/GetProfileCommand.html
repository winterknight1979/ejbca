14  package org.ejbca.core.protocol.ws.client;
15  
16  import java.io.File;
17  import java.io.FileOutputStream;
18  
19  import org.apache.commons.lang.StringUtils;
20  import org.ejbca.core.protocol.ws.client.gen.AuthorizationDeniedException_Exception;
21  import org.ejbca.core.protocol.ws.client.gen.EjbcaException_Exception;
22  import org.ejbca.ui.cli.ErrorAdminCommandException;
23  import org.ejbca.ui.cli.IAdminCommand;
24  import org.ejbca.ui.cli.IllegalAdminCommandException;
25  
26  public class GetProfileCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
27  
28     private static final int ARG_PROFILE_ID = 1;
29     private static final int ARG_PROFILE_TYPE = 2;
30     private static final int ARG_DESTINATION_DIRECTORY = 3;
31  
32     GetProfileCommand(String[] args) {
33         super(args);
34     }
35  
36     @Override
37     public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
38         try {
39             if(args.length !=  4){
40                 usage();
41                 System.exit(-1); // NOPMD, it's not a JEE app
42             }
43  
44             int profileid = Integer.parseInt(args[ARG_PROFILE_ID]);
45             
46             // Read and validate the profile type
47             String profiletype = args[ARG_PROFILE_TYPE];
48             if(!StringUtils.equalsIgnoreCase(profiletype, "eep") && !StringUtils.equalsIgnoreCase(profiletype, "cp")) {
49                 getPrintStream().println("Error: Unknown profile type. Accepted types are 'eep' for end entity profiles and 'cp' for certificate profiles.");
50                 return;
51             }
52             
53             // Read and verify that the path is to a directory
54             String directory = args[ARG_DESTINATION_DIRECTORY];
55             File dir = new File(directory);
56             if(!dir.exists() || !dir.isDirectory()) {
57                 getPrintStream().println("Error: The specified directory is either not a directory or it does not exist");
58                 return;
59             }
60  
61  
62             byte[] profile;
63             try {
64                 profile = getEjbcaRAWS().getProfile(profileid, profiletype);
65  
66                 final String outfile = directory + "/" + profiletype + "-" + profileid + ".xml";
67                 FileOutputStream out = new FileOutputStream(outfile);
68                 out.write(profile);
69                 out.close();
70                 getPrintStream().println("Profile exported to " + outfile);
71             } catch (AuthorizationDeniedException_Exception e) {
72                 getPrintStream().println("Error : " + e.getMessage());
73             } catch (EjbcaException_Exception e) {
74                 getPrintStream().println("Error : " + e.getMessage());
75             }
76         } catch (NumberFormatException e) {
77             getPrintStream().println("Error: The profileID should be a number");
78         } catch (Exception e) {
79             throw new ErrorAdminCommandException(e);
80         }
81     }
82  
83     @Override
84     protected void usage() {
85         getPrintStream().println("Command used to retrieve a specific end entity or certificate profile userdata");
86         getPrintStream().println("Usage : getprofile <profileid> <profiletype> <destdirectory>\n");
87         getPrintStream().print("Profile type can be either 'eep' for End Entity Profiles or 'cp' for Certificate Profiles");
88     }
89  }
