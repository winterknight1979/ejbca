89      /**
90       * Runs the command
91       *
92       * @throws IllegalAdminCommandException Error in command args
93       * @throws ErrorAdminCommandException Error running command
94       */
95      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
96  
97          try {   
98             
99              if(args.length !=  4){
100             	usage();
101             	System.exit(-1); // NOPMD, it's not a JEE app
102             }
103             
104             int matchwith = getMatchWith(args[ARG_MATCHWITH]);
105             int matchtype = getMatchType(args[ARG_MATCHTYPE], matchwith);
106             String matchvalue = args[ARG_MATCHVALUE];
107             
108             
109             try{
110             	UserMatch match = new UserMatch();
111             	match.setMatchtype(matchtype);
112             	match.setMatchvalue(matchvalue);
113             	match.setMatchwith(matchwith);
114             	
115             	List<UserDataVOWS> result = getEjbcaRAWS().findUser(match);
116             	
117             	Iterator<UserDataVOWS> iter = result.iterator();
118             	if(result==null || result.size() == 0){
119             		getPrintStream().println("No matching users could be found in database");
120             	}else{
121             		getPrintStream().println("The following users found in database :");
122             		for(int i=0;i<result.size();i++){
123         			    UserDataVOWS next = iter.next();
124                         getPrintStream().println("\nUser : " + (i +1));
125                         getPrintStream().println("  Username: "+next.getUsername());
126                         getPrintStream().println("  Subject DN: "+next.getSubjectDN());
127                         if(next.getSubjectAltName() == null){
128                             getPrintStream().println("  Subject Altname: NONE");	
129                         }else{
130                             getPrintStream().println("  Subject Altname: "+next.getSubjectAltName());
131                         }
132                         if(next.getEmail() == null){
133                         	getPrintStream().println("  Email: NONE");	
134                         }else{
135                         	getPrintStream().println("  Email: "+next.getEmail());
136                         }                        
137                         getPrintStream().println("  CA Name: "+next.getCaName());                        
138                         getPrintStream().println("  Type: "+getType(next).getHexValue());
139                         getPrintStream().println("  Token: "+next.getTokenType());
140                         getPrintStream().println("  Status: "+ getStatus(next.getStatus()));
141                         getPrintStream().println("  Certificate profile: "+next.getCertificateProfileName());
142                         getPrintStream().println("  End entity profile: "+next.getEndEntityProfileName());
143                         if(next.getHardTokenIssuerName() == null){
144                         	getPrintStream().println("  Hard Token Issuer Alias: NONE");
145                         }else{
146                         	getPrintStream().println("  Hard Token Issuer Alias: " + next.getHardTokenIssuerName());
147                         }
148             		}
149             	}
150             	             
151             }catch(AuthorizationDeniedException_Exception e){
152             	getPrintStream().println("Error : " + e.getMessage());
153             }           
154         } catch (Exception e) {
155             throw new ErrorAdminCommandException(e);
156         }
157     }

159 	private int getMatchType(String matchtype, int matchwith) {
160 		if(matchtype.equalsIgnoreCase("EQUALS")){
161 			if(matchwith >= org.ejbca.util.query.UserMatch.MATCH_WITH_UID){
162 				getPrintStream().println("Error: Matchtype 'EQUALS' cannot be used with DN fields, only 'BEGINSWITH' can be used.");
163 				usage();				
164 				System.exit(-1); // NOPMD, it's not a JEE app
165 			}else{
166 				return org.ejbca.util.query.UserMatch.MATCH_TYPE_EQUALS;
167 			}		
168 		}
169 		if(matchtype.equalsIgnoreCase("BEGINSWITH")){
170 			if(matchwith > org.ejbca.util.query.UserMatch.MATCH_WITH_UID || matchwith == org.ejbca.util.query.UserMatch.MATCH_WITH_DN
171 					|| matchwith == org.ejbca.util.query.UserMatch.MATCH_WITH_EMAIL){
172 			    return org.ejbca.util.query.UserMatch.MATCH_TYPE_BEGINSWITH;
173 			}else{
174 				getPrintStream().println("Error: Matchtype 'BEGINSWITH' can only be used with DN fields, full DN and EMAIL, use EQUALS");
175 				usage();				
176 				System.exit(-1); // NOPMD, it's not a JEE app
177 			}
178 		}		
179 		if(matchtype.equalsIgnoreCase("CONTAINS")){
180 			if ( (matchwith == org.ejbca.util.query.UserMatch.MATCH_WITH_DN) || (matchwith == org.ejbca.util.query.UserMatch.MATCH_WITH_USERNAME) ) {
181 			  return org.ejbca.util.query.UserMatch.MATCH_TYPE_CONTAINS;
182 			}else{
183 			  getPrintStream().println("Error: Matchtype 'CONTAINS' can only be used with matchwith 'DN' or 'USERNAME'.");
184 			  usage();				
185 			  System.exit(-1); // NOPMD, it's not a JEE app
186 			}
187 	}		
188 		usage();				
189 		System.exit(-1); // NOPMD, it's not a JEE app				
190 		return 0; // will never happen
191 	}
192 
193 	private int getMatchWith(String matchwith) {
194 		for(int i=0;i< MATCHWITHTEXTS.length; i++){
195 			if(MATCHWITHTEXTS[i].equalsIgnoreCase(matchwith)){
196 				return MATCHWITHVALUES[i];
197 			}
198 		}
199 		usage();
200 		
201 		System.exit(-1); // NOPMD, it's not a JEE app
202 		return 0; // Will never happen
203 	}
204 
205 	private String getStatus(int status){
206 		for(int i=0;i<STATUS_VALUES.length;i++){
207 			if(STATUS_VALUES[i] == status){
208 				return STATUS_TEXTS[i];
209 			}
210 		}
211 		return "ERROR : Status text not found";
212 	}
213 	
214 	private EndEntityType getType(UserDataVOWS userData) {
215 		EndEntityType type = EndEntityTypes.ENDUSER.toEndEntityType();
216 		
217     	if(userData.isSendNotification()) {
218     		type.addType(EndEntityTypes.SENDNOTIFICATION);
219     	} else {
220     		type.removeType(EndEntityTypes.SENDNOTIFICATION);
221     	}
222     	if(userData.isKeyRecoverable()) {
223     		type.addType(EndEntityTypes.KEYRECOVERABLE);
224     	} else {
225     		type.removeType(EndEntityTypes.KEYRECOVERABLE);
226     	}
227 		return type;
228 	}

230 	protected void usage() {
231 		getPrintStream().println("Command used to find userdata, maximum of 100 users will be returned ");
232 		getPrintStream().println("Usage : finduser <matchwith> <matchtype> <value> \n\n");
233         getPrintStream().print("Matchwith can be : ");
234         for(int i=0;i<MATCHWITHTEXTS.length-1;i++){
235         	getPrintStream().print(MATCHWITHTEXTS[i] +", ");
236         }
237         getPrintStream().println(MATCHWITHTEXTS[MATCHWITHTEXTS.length-1]);
238         getPrintStream().println("Matchtype can be : EQUALS (not DN fields), BEGINSWITH (DN fields), CONTAINS (matchwith complete DN or USERNAME only)");
239     }
