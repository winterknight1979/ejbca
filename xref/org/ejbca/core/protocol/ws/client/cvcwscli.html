14  package org.ejbca.core.protocol.ws.client;
15  
16  import org.ejbca.ui.cli.IAdminCommand;
17  
18  /**
19   * Implements the EJBCA WS command line interface specific for CVC requests
20   *
21   * @version $Id: cvcwscli.java 19902 2014-09-30 14:32:24Z anatom $
22   */
23  public class cvcwscli  {
24  
25  	public static void main(String[] args) {
26          try {
27              IAdminCommand cmd = EJBCAWSRACommandFactory.getCommand(args);
28  
29              if (cmd != null) {
30                  cmd.execute();
31              } else {
32                  System.out.println(
33                      "Usage: cvcrequest cvcgetchain cvcprint cvcpem");
34              }
35          } catch (Exception e) {
36              System.out.println(e.getMessage());
37              e.printStackTrace();
38              System.exit(-1); // NOPMD, this is not a JEE app
39          }
40      }
41  }
