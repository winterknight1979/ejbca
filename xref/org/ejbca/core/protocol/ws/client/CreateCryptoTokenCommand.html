31  /**
32   * Creates new cryptotoken
33   *
34   * @version $Id: CreateCryptoTokenCommand.java 26057 2017-06-22 08:08:34Z anatom $
35   */
36  public class CreateCryptoTokenCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
37  
38      private static final int ARG_CRYPTOTOKEN_NAME    = 1;
39      private static final int ARG_CRYPTOTOKEN_TYPE    = 2;
40      private static final int ARG_AUTO_ACTIVATE       = 3;
41      private static final int ARG_ACTIVATION_PIN      = 4;
42      private static final int ARG_PROPERTIES_FILE     = 5;
43      
44      /**
45       * Creates a new instance of Command
46       *
47       * @param args command line arguments
48       */
49      public CreateCryptoTokenCommand(String[] args) {
50          super(args);
51      }
52  
53      /**
54       * Runs the command
55       *
56       * @throws IllegalAdminCommandException Error in command args
57       * @throws ErrorAdminCommandException Error running command
58       */
59      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
60          try {   
61              if (args.length < 5 ) {
62                  getPrintStream().println("Error. Too few arguments: "+args.length);
63                  usage();
64                  System.exit(-1); // NOPMD, this is not a JEE app
65              }
66              
67              String tokenName = args[ARG_CRYPTOTOKEN_NAME];
68              String tokenType = args[ARG_CRYPTOTOKEN_TYPE];
69              boolean autoActivate = Boolean.parseBoolean(args[ARG_AUTO_ACTIVATE]);
70              String pin = args[ARG_ACTIVATION_PIN];
71              ArrayList<KeyValuePair> properties = new ArrayList<KeyValuePair>();
72              if(args.length > 5) {
73                  Properties props = new Properties();
74                  props.load(new FileInputStream(new File(args[ARG_PROPERTIES_FILE])));
75                  
76                  Iterator<Object> itr = props.keySet().iterator();
77                  while(itr.hasNext()) {
78                      String key = (String) itr.next();
79                      KeyValuePair kvp = new KeyValuePair();
80                      kvp.setKey(key);
81                      kvp.setValue(props.getProperty(key));
82                      properties.add(kvp);
83                  }
84                  
85              }
86  
87              //getPrintStream().println("autoactivate: " + autoActivate);
88              getEjbcaRAWS().createCryptoToken(tokenName, tokenType, pin, autoActivate, properties);
89              getPrintStream().println("Create new cryptotoken: " + tokenName);
90          } catch (Exception e) {
91              if (e instanceof EjbcaException_Exception) {
92                  EjbcaException_Exception e1 = (EjbcaException_Exception)e;
93                  getPrintStream().println("Error code: " + e1.getFaultInfo().getErrorCode().getInternalErrorCode());
94              }
95              ErrorAdminCommandExceptionhtml#ErrorAdminCommandException">ErrorAdminCommandException adminexp = new ErrorAdminCommandException(e);
96              getPrintStream().println("Error message: " + adminexp.getLocalizedMessage());
97          }
98          
99      }
100

104         		"[<pathToCryptoTokenPropertiesFile>]");
105         getPrintStream().println();
106         getPrintStream().println("cryptotokenName: The name of the new crypto token.");
107         getPrintStream().println("cryptotokenType: Can be one of: " + SoftCryptoToken.class.getSimpleName() + ", " + PKCS11CryptoToken.class.getSimpleName());
108         getPrintStream().println("autoActivate: Set to true|false to allow|disallow whether crypto token should be autoactivated or not.");
109         getPrintStream().println("activationPin: Pin code for the crypto token.");
110         getPrintStream().println("pathToCryptoTokenPropertiesFile: The path to a .properties file containing the new cryptotoken properties. Optional");
