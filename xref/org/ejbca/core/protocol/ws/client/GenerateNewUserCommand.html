72      /**
73       * Runs the command
74       *
75       * @throws IllegalAdminCommandException Error in command args
76       * @throws ErrorAdminCommandException Error running command
77       */
78      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
79  
80      	final UserDataVOWS userdata = new UserDataVOWS();
81      	final String[] myArgs = ParseUserData.getDataFromArgs(this.args, userdata, getPrintStream());
82      	if(myArgs.length < NR_OF_MANDATORY_ARGS || myArgs.length > MAX_NR_OF_ARGS){
83      		usage();
84      		System.exit(-1); // NOPMD, it's not a JEE app
85      	}
86  
87      	try {
88              userdata.setUsername(myArgs[ARG_USERNAME]);
89              userdata.setPassword(myArgs[ARG_PASSWORD]);
90              userdata.setClearPwd(myArgs[ARG_CLEARPWD].equalsIgnoreCase("true"));
91              userdata.setSubjectDN(myArgs[ARG_SUBJECTDN]);
92              if(!myArgs[ARG_SUBJECTALTNAME].equalsIgnoreCase("NULL")){                        
93              	userdata.setSubjectAltName(myArgs[ARG_SUBJECTALTNAME]);
94              }
95              if(!myArgs[ARG_EMAIL].equalsIgnoreCase("NULL")){
96              	userdata.setEmail(myArgs[ARG_EMAIL]);
97              }
98              userdata.setCaName(myArgs[ARG_CA]);
99              userdata.setTokenType(myArgs[ARG_TOKEN]);
100             userdata.setStatus(getStatus(myArgs[ARG_STATUS]));
101             userdata.setEndEntityProfileName(myArgs[ARG_ENDENTITYPROFILE]);
102             userdata.setCertificateProfileName(myArgs[ARG_CERTIFICATEPROFILE]);
103             
104             EndEntityTypendentity/EndEntityType.html#EndEntityType">EndEntityType type = new EndEntityType(EndEntityTypes.getTypesFromHexCode(Integer.parseInt(myArgs[ARG_TYPE])));
105             
106             if(type.contains(EndEntityTypes.SENDNOTIFICATION)){
107             	userdata.setSendNotification(true);
108             }
109             if(type.contains(EndEntityTypes.KEYRECOVERABLE)){
110             	userdata.setKeyRecoverable(true);
111             }
112 
113             if(!myArgs[ARG_ISSUERALIAS].equalsIgnoreCase("NONE")){
114             	userdata.setEmail(myArgs[ARG_ISSUERALIAS]);
115             }
116             
117             final String username = myArgs[ARG_USERNAME];
118             final String password = myArgs[ARG_PASSWORD];
119             final String pkcs10 = getPKCS10(myArgs[ARG_PKCS10]);
120             final String encoding = getEncoding(myArgs[ARG_ENCODING]);
121             final String hardtokensn = getHardTokenSN(myArgs[ARG_HARDTOKENSN]);
122             final String outputPath = myArgs.length>ARG_OUTPUTPATH ? getOutputPath(myArgs[ARG_OUTPUTPATH]) : null;
123             
124             getPrintStream().println("Trying to add user:");
125             getPrintStream().println("Username: "+userdata.getUsername());
126             getPrintStream().println("Subject DN: "+userdata.getSubjectDN());
127             getPrintStream().println("Subject Altname: "+userdata.getSubjectAltName());
128             getPrintStream().println("Email: "+userdata.getEmail());
129             getPrintStream().println("CA Name: "+userdata.getCaName());                        
130             getPrintStream().println("Type: "+type.getHexValue());
131             getPrintStream().println("Token: "+userdata.getTokenType());
132             getPrintStream().println("Status: "+userdata.getStatus());
133             getPrintStream().println("End entity profile: "+userdata.getEndEntityProfileName());
134             getPrintStream().println("Certificate profile: "+userdata.getCertificateProfileName());
135             getPrintStream().println("Hard Token Issuer Alias: "+(userdata.getHardTokenIssuerName()!=null ? userdata.getHardTokenIssuerName() :"NONE"));
136             {
137             	final List<ExtendedInformationWS> eil = userdata.getExtendedInformation();
138             	if ( eil!=null ) {
139             		getPrintStream().println("Extended information:");
140             		for ( ExtendedInformationWS ei : eil ) {
141             			getPrintStream().println("	'"+ei.getName()+"' = '"+ei.getValue()+"'");
142             		}
143             	}
144             }
145             {
146             	final BigInteger bi = userdata.getCertificateSerialNumber();
147             	if ( bi!=null ) {
148             		getPrintStream().println(ParseUserData.certificateSerialNumber+"=0x"+bi.toString(16));
149             	}
150             }
151             
152             
153             try{
154             	getEjbcaRAWS().editUser(userdata);            	
155             	getPrintStream().println("User '"+userdata.getUsername()+"' has been added/edited.");
156             	getPrintStream().println();       
157             	
158              	CertificateResponse result = getEjbcaRAWS().pkcs10Request(username,password,pkcs10,hardtokensn,CertificateHelper.RESPONSETYPE_CERTIFICATE);
159             	
160             	if(result==null){
161             		getPrintStream().println("No certificate could be generated for user, check server logs for error.");
162             	}else{
163             		String filepath = username;
164             		if(encoding.equals("DER")){
165             			filepath += ".cer";
166             		}else{
167             			filepath += ".pem";
168             		}
169             		if(outputPath != null){
170             			filepath = outputPath + "/" + filepath;
171             		}
172             		
173             		
174             		if(encoding.equals("DER")){
175             			FileOutputStream fos = new FileOutputStream(filepath);
176             			fos.write(CertificateHelper.getCertificate(result.getData()).getEncoded());
177             			fos.close();
178             		}else{
179             			FileOutputStream fos = new FileOutputStream(filepath);
180             			ArrayList<java.security.cert.Certificate> list = new ArrayList<java.security.cert.Certificate>();
181             			list.add(CertificateHelper.getCertificate(result.getData()));
182             			fos.write(CertTools.getPemFromCertificateChain(list));
183             			fos.close();            				            				
184             		}
185             		getPrintStream().println("Certificate generated, written to " + filepath);
186             	}
187             }catch(AuthorizationDeniedException_Exception e){
188             	getPrintStream().println("Error : " + e.getMessage());
189             }catch(UserDoesntFullfillEndEntityProfile_Exception e){
190             	getPrintStream().println("Error : Given userdata doesn't fulfill end entity profile. : " +  e.getMessage());
191             }            
192         } catch (Exception e) {
193             throw new ErrorAdminCommandException(e);
194         }
195     }

197 	private int getStatus(String status) {
198 		if(status.equalsIgnoreCase("NEW")){
199 			return EndEntityConstants.STATUS_NEW;
200 		}
201 		if(status.equalsIgnoreCase("INPROCESS")){
202 			return EndEntityConstants.STATUS_INPROCESS;
203 		}
204 		if(status.equalsIgnoreCase("FAILED")){
205 			return EndEntityConstants.STATUS_FAILED;
206 		}
207 		if(status.equalsIgnoreCase("HISTORICAL")){
208 			return EndEntityConstants.STATUS_HISTORICAL;
209 		}		
210 		
211 		getPrintStream().println("Error in status string : " + status );
212 		usage();
213 		System.exit(-1); // NOPMD, it's not a JEE app
214 		return 0;
215 	}

218 		if(hardtokensn.equalsIgnoreCase("NONE")){
219 		  return null;
220 		}
221 		
222 		return hardtokensn;
223 	}

225 	private String getPKCS10(String pkcs10Path) {
226 		String retval=null;
227 		try {
228 			FileInputStream fis = new FileInputStream(pkcs10Path);
229 			byte[] contents = new byte[fis.available()];
230 			fis.read(contents);			
231 			fis.close();
232 			retval = new String(contents);
233 		} catch (FileNotFoundException e) {
234 			getPrintStream().println("Error : PKCS10 file could not found.");
235 			System.exit(-1); // NOPMD, it's not a JEE app		
236 		} catch (IOException e) {
237 			getPrintStream().println("Error reading content of PKCS10 file.");
238 			System.exit(-1); // NOPMD, it's not a JEE app	
239 		}
240 		
241 		
242 		return retval;
243 	}

245 	private String getOutputPath(String outputpath) {
246 		File dir = new File(outputpath);
247 		if(!dir.exists()){
248 			getPrintStream().println("Error : Output directory doesn't seem to exist.");
249 			System.exit(-1); // NOPMD, it's not a JEE app
250 		}
251 		if(!dir.isDirectory()){
252 			getPrintStream().println("Error : Output directory doesn't seem to be a directory.");
253 			System.exit(-1); // NOPMD, it's not a JEE app			
254 		}
255 		if(!dir.canWrite()){
256 			getPrintStream().println("Error : Output directory isn't writeable.");
257 			System.exit(-1); // NOPMD, it's not a JEE app
258 
259 		}
260 		return outputpath;
261 	}
262 
263 	private String getEncoding(String encoding) {
264 		if(!encoding.equalsIgnoreCase("PEM") && !encoding.equalsIgnoreCase("DER")){
265 			usage();
266 			System.exit(-1); // NOPMD, it's not a JEE app
267 		}
268 		
269 		return encoding.toUpperCase();
270 	}
