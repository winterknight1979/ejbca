13  package org.ejbca.core.protocol.ws.client.gen;
14  
15  import java.io.Serializable;
16  import java.util.ArrayList;
17  import java.util.Iterator;
18  import java.util.List;
19  import java.util.Set;
20  
21  /**
22   * Class used to represent userdatasource in the WebService API.
23   * because of profilenames is used instead of id's.
24   * 
25   * @author Philip Vendil
26   * @version $Id: UserDataSourceVOWS.java 19902 2014-09-30 14:32:24Z anatom $
27   */
28  public class UserDataSourceVOWS implements Serializable{
29  
30  	private static final long serialVersionUID = 4582471233805101416L;
31      private UserDataVOWS userDataVOWS = null;
32  	private List<Integer> isModifyable = null;
33  
34  	/**
35  	 * WS Constructor
36  	 */
37  	public UserDataSourceVOWS(){}
38  
39  	public UserDataSourceVOWS(UserDataVOWS userDataVOWS, Set<Integer> isModifyableSet){
40  		this.userDataVOWS = userDataVOWS;
41  		this.isModifyable = new ArrayList<Integer>();
42  		Iterator<Integer> iter = isModifyableSet.iterator();
43  		while(iter.hasNext()){
44  			isModifyable.add(iter.next());
45  		}	  
46  	}
47  
48  	/**
49  	 * Gets a list of modifyable fields, (Should be one of the constants
50  	 * of the UserDataSourceVO.ISMODIFYABLE_ or
51  	 * DNFieldExtractor constants defined in the AVAILABLEMODIFYABLEFIELDS array.
52  	 * 
53  	 * use the contains(Object) method do find out if a field should be modifyable or not.
54  	 * @return data
55  	 */
56  	public List<Integer> getIsModifyable() {
57  		return isModifyable;
58  	}
59  
60  	/**
61  	 * Method that shouldn't be used outside the WS framework.
62  	 * @param isModifyable data
63  	 */
64  	public void setIsModifyable(List<Integer> isModifyable) {
65  		this.isModifyable = isModifyable;
66  	}
67  	
68      /**
69       * 
70       * @return user data connected with this instance of user data source vo
71       */
72  	public UserDataVOWS getUserDataVOWS() {
73  		return userDataVOWS;
74  	}
75  
76  	/**
77  	 * Method that shouldn't be used outside the WS framework.
78  	 * @param userDataVOWS data
79  	 */
80  	public void setUserDataVOWS(UserDataVOWS userDataVOWS) {
81  		this.userDataVOWS = userDataVOWS;
82  	}
83  
84  
85  
86  
87  
88  }
