14  package org.ejbca.core.protocol.ws.client;
15  
16  import org.ejbca.core.protocol.ws.client.gen.EjbcaException;
17  import org.ejbca.core.protocol.ws.client.gen.EjbcaException_Exception;
18  import org.ejbca.ui.cli.ErrorAdminCommandException;
19  import org.ejbca.ui.cli.IAdminCommand;
20  import org.ejbca.ui.cli.IllegalAdminCommandException;
21  
22  /**
23   * Implements the EJBCA RA WS command line interface
24   *
25   * @version $Id: ejbcawsracli.java 28822 2018-05-03 06:37:20Z tarmo_r_helmes $
26   */
27  public class ejbcawsracli  {
28      /**
29       * main Client
30       *
31       * @param args command line arguments
32       */
33      public static void main(String[] args) {
34          try {
35              IAdminCommand cmd = EJBCAWSRACommandFactory.getCommand(args);
36  
37              if (cmd != null) {
38                  cmd.execute();
39              } else {
40                  System.out.println(
41                      "Usage: edituser | finduser | findcerts | pkcs10req | pkcs12req | certreq | revokecert | revokecertwithmetadata | " +
42                      "getpublisherqueuelength | revoketoken | revokeuser | checkrevocationstatus | generatenewuser | " +
43                      "createcrl | cacertrequest | cacertresponse | customlog | getprofile | createcryptotoken | " + 
44                      "generatectkeys | createca | importcacert | updatecacert | addadmintorole | removeadminfromrole | getexpiredcerts | " +
45                      "getexpiredcertsbyissuer | getexpiredcertsbytype | stress | keyrecover | keyrecoverenroll");
46              }
47          } catch (ErrorAdminCommandException e) {
48              final Throwable cause = e.getCause();
49              if ( Thread.currentThread().getStackTrace().length > 12 && cause instanceof SecurityException ) {
50                  throw (SecurityException)cause; // throw it if called by clientToolBoxTest and exit exception. clientToolBox call has a length of 8. 12 gives some margin for code changes.
51              }
52              if (cause instanceof EjbcaException_Exception) {
53                  final EjbcaException_Exception ejbcaex = (EjbcaException_Exception)cause;
54                  final EjbcaException ee = ejbcaex.getFaultInfo();
55                  System.out.println("Error: "+ee.getErrorCode().getInternalErrorCode()+": "+ee.getMessage());
56              } else {
57                  System.out.println(e.getMessage());
58              }
59              e.printStackTrace(System.err);
60              System.exit(-1); // NOPMD, this is not a JEE app
61          } catch (IllegalAdminCommandException e) {
62              System.out.println(e.getMessage());
63              System.exit(-2); // NOPMD, this is not a JEE app
64          }
65      }
66  }
