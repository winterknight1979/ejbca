14  package org.ejbca.core.protocol.ws.client;
15  
16  import org.cesecore.util.CertTools;
17  import org.cesecore.util.CryptoProviderTools;
18  import org.ejbca.core.protocol.ws.client.gen.EjbcaException_Exception;
19  import org.ejbca.ui.cli.ErrorAdminCommandException;
20  import org.ejbca.ui.cli.IAdminCommand;
21  import org.ejbca.ui.cli.IllegalAdminCommandException;
22  
23  /**
24   * Imports an external CA certificate for X.509 CAs or CVC CAs and creates the CA in EJBCA with the given name.
25   *
26   * @version $Id: ImportCaCertCommand.java 26057 2017-06-22 08:08:34Z anatom $
27   */
28  public class ImportCaCertCommand extends EJBCAWSRABaseCommand implements IAdminCommand {
29  
30      private static final int ARG_CANAME = 1;
31      private static final int ARG_CACHAIN = 2;
32  
33      /**
34       * Creates a new instance of Command
35       *
36       * @param args command line arguments
37       */
38      public ImportCaCertCommand(String[] args) {
39          super(args);
40      }
41  
42      /**
43       * Runs the command
44       *
45       * @throws IllegalAdminCommandException Error in command args
46       * @throws ErrorAdminCommandException Error running command
47       */
48      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
49          try {
50              if (args.length < 3 || args.length > 3) {
51                  getPrintStream().println("Number of arguments: " + args.length);
52                  usage();
53                  System.exit(-1); // NOPMD, this is not a JEE app
54              }
55  
56              CryptoProviderTools.installBCProvider();
57  
58              final String caname = args[ARG_CANAME];
59              final String file = args[ARG_CACHAIN];
60              getPrintStream().println("Import external CA: " + caname);
61              getPrintStream().println("Import external CA chain file: " + file);
62  
63              final byte[] cachain = CertTools.readCertificateChainAsArrayOrThrow(file);
64              getEjbcaRAWS().importCaCert(caname, cachain);
65  
66              getPrintStream().println("CA imported sucessfully.");
67          } catch (Exception e) {
68              if (e instanceof EjbcaException_Exception) {
69                  EjbcaException_Exception e1 = (EjbcaException_Exception) e;
70                  getPrintStream().println("Error code is: " + e1.getFaultInfo().getErrorCode().getInternalErrorCode());
71              }
72              throw new ErrorAdminCommandException(e);
73          }
74      }
75  
76      protected void usage() {
77          getPrintStream().println("Command used to import an external CA certificate. Can be X.509 or CVC CA.");
78          getPrintStream().println("Usage : importcacert <caname> <cachainfile>\n\n");
79          getPrintStream().println("Caname is the name of the CA created.");
80          getPrintStream().println(
81                  "Cachainfile is a file with the certificate chain of the external CA. This can be a file with several PEM certificates in it, or a file with a single PEM or binary Root CA certificate.");
82      }
83  }
