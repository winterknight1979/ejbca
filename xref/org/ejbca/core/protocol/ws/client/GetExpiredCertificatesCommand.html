30  /**
31   * Finds a certificates in the database
32   *
33   * @version $Id: GetExpiredCertificatesCommand.java 26057 2017-06-22 08:08:34Z anatom $
34   */
35  public class GetExpiredCertificatesCommand extends EJBCAWSRABaseCommand implements IAdminCommand{
36  
37      
38      private static final int ARG_DAYS                  = 1;
39      private static final int ARG_MAX_NUMBER_OF_RESULTS = 2;  
40      private static final int ARG_ENCODING              = 3;
41      private static final int ARG_OUTPUTPATH            = 4;
42      
43      public GetExpiredCertificatesCommand(String[] args) {
44          super(args);
45      }
46  
47      /**
48       * Runs the command
49       *
50       * @throws IllegalAdminCommandException Error in command args
51       * @throws ErrorAdminCommandException Error running command
52       */
53      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
54  
55          try {   
56             
57              if(args.length <  5){
58                  usage();
59                  System.exit(-1); // NOPMD, it's not a JEE app
60              }
61              
62              long days = Long.parseLong(args[ARG_DAYS]);            
63              int maxNumberOfResults = Integer.parseInt(args[ARG_MAX_NUMBER_OF_RESULTS]);
64              String encoding = getEncoding(args[ARG_ENCODING]);
65              String outputPath = getOutputPath(args[ARG_OUTPUTPATH]);
66              
67              
68              try{
69                  List<Certificate> result = getEjbcaRAWS().getCertificatesByExpirationTime(days, maxNumberOfResults);
70                  
71                  if(result==null || result.size() == 0){
72                      getPrintStream().println("No certificate will expire within " + days + " days.");
73                  }else{
74                      getPrintStream().println(result.size() + " certificate found, written to " + outputPath);
75                      Iterator<Certificate> iter = result.iterator();
76                      int i=0;
77                      while(iter.hasNext()){
78                          i++;
79                          Certificate cert = iter.next();
80                          if(encoding.equals("DER")){
81                              FileOutputStream fos = new FileOutputStream(outputPath + "/" + "cert-" + i +".cer");
82                              fos.write(CertificateHelper.getCertificate(cert.getCertificateData()).getEncoded());
83                              fos.close();
84                          }else{
85                              FileOutputStream fos = new FileOutputStream(outputPath + "/" + "cert-" + i +".pem");
86                              ArrayList<java.security.cert.Certificate> list = new ArrayList<java.security.cert.Certificate>();
87                              list.add(CertificateHelper.getCertificate(cert.getCertificateData()));
88                              fos.write(CertTools.getPemFromCertificateChain(list));
89                              fos.close();                                                        
90                          }                        
91                      }
92                  }
93                               
94              }catch(AuthorizationDeniedException_Exception e){
95                  getPrintStream().println("Error : " + e.getMessage());
96              }           
97          } catch (Exception e) {
98              ErrorAdminCommandExceptionhtml#ErrorAdminCommandException">ErrorAdminCommandException adminexp = new ErrorAdminCommandException(e);
99              getPrintStream().println("Error: " + adminexp.getLocalizedMessage());;
100         }
101     }
102 
103     private String getOutputPath(String outputpath) {
104         File dir = new File(outputpath);
105         if(!dir.exists()){
106             getPrintStream().println("Error : Output directory doesn't seem to exist.");
107             System.exit(-1); // NOPMD, it's not a JEE app
108         }
109         if(!dir.isDirectory()){
110             getPrintStream().println("Error : Output directory doesn't seem to be a directory.");
111             System.exit(-1); // NOPMD, it's not a JEE app           
112         }
113         if(!dir.canWrite()){
114             getPrintStream().println("Error : Output directory isn't writeable.");
115             System.exit(-1); // NOPMD, it's not a JEE app
116 
117         }
118         return outputpath;
119     }
120 
121     private String getEncoding(String encoding) {
122         if(!encoding.equalsIgnoreCase("PEM") && !encoding.equalsIgnoreCase("DER")){
123             usage();
124             System.exit(-1); // NOPMD, it's not a JEE app
125         }
126         
127         return encoding.toUpperCase();
128     }
129 
130     protected void usage() {
131         getPrintStream().println("Command used to find certificates that will expire within a specified number of days");
132         getPrintStream().println("Usage : getexpiredcerts <numberOfDays> <maxNumberOfResults> <encoding (DER|PEM)> <outputpath>");
133         getPrintStream().println();
134         getPrintStream().println("maxNumberOfResults: the maximum number of returned certificates");
135         getPrintStream().println("outputpath : directory where certificates are written");
136         getPrintStream().println("Note that all returned certificates are of status active or notified about expiration");
137    }
