43  	private static final int ARG_USERNAME           = 1;
44  	private static final int ARG_PASSWORD           = 2;
45  	private static final int ARG_CLEARPWD           = 3;
46  	private static final int ARG_SUBJECTDN          = 4;
47  	private static final int ARG_SUBJECTALTNAME     = 5;
48  	private static final int ARG_EMAIL              = 6;
49  	private static final int ARG_CA                 = 7;
50  	private static final int ARG_TYPE               = 8;
51  	private static final int ARG_TOKEN              = 9;
52  	private static final int ARG_STATUS             = 10;
53  	private static final int ARG_ENDENTITYPROFILE   = 11;
54  	private static final int ARG_CERTIFICATEPROFILE = 12;
55  	private static final int ARG_ISSUERALIAS        = 13;
56  	private static final int ARG_STARTTIME          = 14;
57  	private static final int ARG_ENDTIME            = 15;
58  
59  	private static final int NR_OF_MANDATORY_ARGS = ARG_CERTIFICATEPROFILE+1;
60  	private static final int MAX_NR_OF_ARGS = ARG_ENDTIME+1;

62      public EditUserCommand(String[] args) {
63          super(args);
64      }

66      /**
67       * Runs the command
68       *
69       * @throws IllegalAdminCommandException Error in command args
70       * @throws ErrorAdminCommandException Error running command
71       */
72      public void execute() throws IllegalAdminCommandException, ErrorAdminCommandException {
73      	
74          final UserDataVOWS userdata = new UserDataVOWS();
75          final String[] myArgs = ParseUserData.getDataFromArgs(this.args, userdata, getPrintStream());
76      	if(myArgs.length < NR_OF_MANDATORY_ARGS || myArgs.length > MAX_NR_OF_ARGS){
77          	usage();
78          	System.exit(-1); // NOPMD, this is not a JEE app
79          }
80          try {
81              userdata.setUsername(myArgs[ARG_USERNAME]);
82              String pwd = myArgs[ARG_PASSWORD];
83              if (StringUtils.equalsIgnoreCase("null", pwd)) {
84              	pwd = null;
85              }
86              userdata.setPassword(pwd);
87              userdata.setClearPwd(myArgs[ARG_CLEARPWD].equalsIgnoreCase("true"));
88              userdata.setSubjectDN(myArgs[ARG_SUBJECTDN]);
89              if(!myArgs[ARG_SUBJECTALTNAME].equalsIgnoreCase("NULL")){                        
90              	userdata.setSubjectAltName(myArgs[ARG_SUBJECTALTNAME]);
91              }
92              if(!myArgs[ARG_EMAIL].equalsIgnoreCase("NULL")){
93              	userdata.setEmail(myArgs[ARG_EMAIL]);
94              }
95              userdata.setCaName(myArgs[ARG_CA]);
96              userdata.setTokenType(myArgs[ARG_TOKEN]);
97              userdata.setStatus(getStatus(myArgs[ARG_STATUS]));
98              userdata.setEndEntityProfileName(myArgs[ARG_ENDENTITYPROFILE]);
99              userdata.setCertificateProfileName(myArgs[ARG_CERTIFICATEPROFILE]);
100             
101             EndEntityTypendentity/EndEntityType.html#EndEntityType">EndEntityType type = new EndEntityType(EndEntityTypes.getTypesFromHexCode(Integer.parseInt(myArgs[ARG_TYPE])));
102             
103             if(type.contains(EndEntityTypes.SENDNOTIFICATION)){
104             	userdata.setSendNotification(true);
105             }
106             if(type.contains(EndEntityTypes.KEYRECOVERABLE)){
107             	userdata.setKeyRecoverable(true);
108             }
109 
110             if(myArgs.length > ARG_ISSUERALIAS){
111             	if(!myArgs[ARG_ISSUERALIAS].equalsIgnoreCase("NULL")){                        
112             		userdata.setHardTokenIssuerName(myArgs[ARG_ISSUERALIAS]);
113             	}
114             }
115             if(myArgs.length > ARG_STARTTIME){
116             	if(!myArgs[ARG_STARTTIME].equalsIgnoreCase("NULL")){                        
117             		userdata.setStartTime(myArgs[ARG_STARTTIME]);
118             	}
119             }
120             if(myArgs.length > ARG_ENDTIME){
121             	if(!myArgs[ARG_ENDTIME].equalsIgnoreCase("NULL")){                        
122             		userdata.setEndTime(myArgs[ARG_ENDTIME]);
123             	}
124             }
125    
126             getPrintStream().println("Trying to add user:");
127             getPrintStream().println("Username: "+userdata.getUsername());
128             getPrintStream().println("Subject DN: "+userdata.getSubjectDN());
129             getPrintStream().println("Subject Altname: "+userdata.getSubjectAltName());
130             getPrintStream().println("Email: "+userdata.getEmail());
131             getPrintStream().println("CA Name: "+userdata.getCaName());                        
132             getPrintStream().println("Type: "+type.getHexValue());
133             getPrintStream().println("Token: "+userdata.getTokenType());
134             getPrintStream().println("Status: "+userdata.getStatus());
135             getPrintStream().println("End entity profile: "+userdata.getEndEntityProfileName());
136             getPrintStream().println("Certificate profile: "+userdata.getCertificateProfileName());
137 
138             if(userdata.getHardTokenIssuerName() == null){
139             	getPrintStream().println("Hard Token Issuer Alias: NONE");
140             }else{
141             	getPrintStream().println("Hard Token Issuer Alias: " + userdata.getHardTokenIssuerName());
142             }
143             if(userdata.getStartTime() == null){
144                 getPrintStream().println("Start time: NONE");
145             }else{
146                 getPrintStream().println("Start time: "+userdata.getStartTime());
147             }
148             if(userdata.getEndTime() == null){
149                 getPrintStream().println("End time: NONE");
150             }else{
151                 getPrintStream().println("End time: "+userdata.getEndTime());
152             }
153             {
154             	final List<ExtendedInformationWS> eil = userdata.getExtendedInformation();
155             	if ( eil!=null ) {
156             		getPrintStream().println("Extended information:");
157             		for ( ExtendedInformationWS ei : eil ) {
158             			getPrintStream().println("	'"+ei.getName()+"' = '"+ei.getValue()+"'");
159             		}
160             	}
161             }
162             {
163             	final BigInteger bi = userdata.getCertificateSerialNumber();
164             	if ( bi!=null ) {
165                     getPrintStream().println(ParseUserData.certificateSerialNumber+"=0x"+bi.toString(16));
166             	}
167             }
168             
169            try{
170             	getEjbcaRAWS().editUser(userdata);
171 
172             	
173             	getPrintStream().println("User '"+userdata.getUsername()+"' has been added/edited.");
174             	getPrintStream().println();              
175             }catch(AuthorizationDeniedException_Exception e){
176             	getPrintStream().println("Error : " + e.getMessage());
177             }catch(UserDoesntFullfillEndEntityProfile_Exception e){
178             	getPrintStream().println("Error : Given userdata doesn't fulfill end entity profile. : " +  e.getMessage());
179             }
180                       
181         } catch (Exception e) {
182             throw new ErrorAdminCommandException(e);
183         }
184     }
185     private int getStatus(String status) {
186 		if(status.equalsIgnoreCase("NEW")){
187 			return EndEntityConstants.STATUS_NEW;
188 		}
189 		if(status.equalsIgnoreCase("INPROCESS")){
190 			return EndEntityConstants.STATUS_INPROCESS;
191 		}
192 		if(status.equalsIgnoreCase("FAILED")){
193 			return EndEntityConstants.STATUS_FAILED;
194 		}
195 		if(status.equalsIgnoreCase("HISTORICAL")){
196 			return EndEntityConstants.STATUS_HISTORICAL;
197 		}		
198 		
199 		getPrintStream().println("Error in status string : " + status );
200 		usage();
201 		System.exit(-1); // NOPMD, this is not a JEE app
202 		return 0;
203 	}

205 	protected void usage() {
206 		getPrintStream().println("Command used to add or edit userdata, if user exist will the data be overwritten.");
207 		getPrintStream().println("Usage : edituser <username> <password|null> <clearpwd (true|false)> <subjectdn> <subjectaltname or NULL> <email or NULL> <caname> <type> <token> <status> <endentityprofilename> <certificateprofilename> <issueralias or NULL (optional)> <starttime or NULL (optional)> <endtime (starttime)>\n\n");
208         getPrintStream().println("DN is of form \"C=SE, O=MyOrg, OU=MyOrgUnit, CN=MyName\" etc.");
209         getPrintStream().println("SubjectAltName is of form \"rfc822Name=<email>, dNSName=<host name>, uri=<http://host.com/>, ipaddress=<address>, guid=<globally unique id>\"");
210         getPrintStream().println("Type (mask): INVALID=0; END-USER=1; KEYRECOVERABLE=128; SENDNOTIFICATION=256");
211         getPrintStream().println("Existing tokens: USERGENERATED, P12, JKS, PEM");
212         getPrintStream().println("Existing statuses (new users will always be set as NEW) : NEW, INPROCESS, FAILED, HISTORICAL");
213         getPrintStream().println("Start time and end time is of form \"May 26, 2009 9:52 AM\" or \"days:hours:minutes\"");
214         getPrintStream().println();
215         ParseUserData.printCliHelp(getPrintStream());
216 	}
