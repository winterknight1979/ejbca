19   */
20  public class CryptoTokenConstantsWS {

22      // -----------------
23      // Soft Crypto Token
24      //------------------
25      /** If set, no default password allowed for this soft cryptotoken */
26      public static final String NODEFAULTPWD = org.cesecore.keys.token.SoftCryptoToken.NODEFAULTPWD;
27      /** Boolean indicating if it should be allowed to extract private keys */
28      public static final String ALLOW_EXTRACTABLE_PRIVATE_KEY = org.cesecore.keys.token.CryptoToken.ALLOW_EXTRACTABLE_PRIVATE_KEY;
29      //public static final String ALLOW_NONEXISTING_SLOT_PROPERTY = org.cesecore.keys.token.CryptoToken.ALLOW_NONEXISTING_SLOT_PROPERTY;
30      
31      // -------------------
32      // PKCS11 Crypto Token
33      // -------------------
34      /** Specific to PKCS#11. The slot label value */
35      public static final String SLOT_LABEL_VALUE = org.cesecore.keys.token.PKCS11CryptoToken.SLOT_LABEL_VALUE;
36      /** Specific to PKCS#11. The slot label type */
37      public static final String SLOT_LABEL_TYPE = org.cesecore.keys.token.PKCS11CryptoToken.SLOT_LABEL_TYPE;
38      /** Specific to PKCS#11. The path to the shared PKCS11 library */
39      public static final String SHLIB_LABEL_KEY = org.cesecore.keys.token.PKCS11CryptoToken.SHLIB_LABEL_KEY;
40      /** Specific to PKCS#11. The attributes label */
41      public static final String ATTRIB_LABEL_KEY = org.cesecore.keys.token.PKCS11CryptoToken.ATTRIB_LABEL_KEY;
42      /** A user defined name of the slot provider. Used in order to be able to have two different providers
43       * (with different PKCS#11 attributes) for the same slot. If this is not set (null), the default
44       * java provider name is used (SunPKCS11-pkcs11LibName-slotNr for example SunPKCS11-libcryptoki.so-slot1).
45       */
46      public final static String TOKEN_FRIENDLY_NAME = org.cesecore.keys.token.PKCS11CryptoToken.TOKEN_FRIENDLY_NAME;
47      
48  }
