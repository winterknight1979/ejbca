13  package org.ejbca.core.protocol.ws.objects;
14  
15  import org.cesecore.certificates.ca.catoken.CATokenConstants;
16  
17  /**
18   * Property keys for creation of CAs via WS.
19   * 
20   * @version $Id: CAConstantsWS.java 26057 2017-06-22 08:08:34Z anatom $
21   */
22  public class CAConstantsWS {
23  
24      /**
25       * The policy ID can be 'null' if no Certificate Policy extension should be present, or\nobjectID as '2.5.29.32.0' 
26       * or objectID and cpsurl as '2.5.29.32.0 http://foo.bar.com/mycps.txt'. You can add multiple policies such as 
27       * '2.5.29.32.0 http://foo.bar.com/mycps.txt 1.1.1.1.1 http://foo.bar.com/111cps.txt'.
28       */
29      public static final String POLICYID = "policyid";
30      
31      /** Certificate signing key alias */
32      public static final String CAKEYPURPOSE_CERTSIGN_STRING = CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING;
33      /** Certificate Revocation List (CRL) signing key alias. Must be the same as the certificate signing key.  */
34      public static final String CAKEYPURPOSE_CRLSIGN_STRING = CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING;
35      /** Used for decryption of key recovery data. Must be an RSA key. */
36      public static final String CAKEYPURPOSE_KEYENCRYPT_STRING = CATokenConstants.CAKEYPURPOSE_KEYENCRYPT_STRING;
37      /** Test signing key. Used by health-check. */
38      public static final String CAKEYPURPOSE_TESTKEY_STRING = CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING;
39      /** Default key. If any of the other aliases are not specified, this will be used in their place. Must be an RSA key if decryption key aliases are not specified.*/
40      public static final String CAKEYPURPOSE_DEFAULT_STRING = CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING;
41      /** Used for decryption of hard token data (e.g. PUK code). Must be an RSA key. */
42      public static final String CAKEYPURPOSE_HARDTOKENENCRYPT_STRING = CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT_STRING;
43  }
