14  package org.ejbca.core.protocol.ws.logger;
15  
16  import java.util.regex.Pattern;
17  
18  import org.apache.log4j.Logger;
19  import org.cesecore.util.GUIDGenerator;
20  import org.ejbca.config.WebServiceConfiguration;
21  import org.ejbca.util.DummyPatternLogger;
22  import org.ejbca.util.IPatternLogger;
23  import org.ejbca.util.PatternLogger;
24  
25  /**
26   * Transaction logger for EJBCA WS with output in DEBUG log level when enabled.
27   * 
28   * @version $Id: TransactionLogger.java 31242 2019-01-17 22:41:24Z jeklund $
29   */
30  public class TransactionLogger {
31      
32      private static TransactionLogger instance = null;
33  
34      /** regexp pattern to match ${identifier} patterns */// ${DN};${IP}
35      final private Pattern PATTERN = Pattern.compile(WebServiceConfiguration.getTransactionLogPattern());
36  
37      final private String orderString = WebServiceConfiguration.getTransactionLogOrder();
38      final private String logDateFormat = WebServiceConfiguration.getTransactionLogDateFormat(); 
39      final private String timeZone = WebServiceConfiguration.getTransactionLogTimeZone();
40      final private boolean doLog = WebServiceConfiguration.getTransactionLogEnabled();
41      final private String sessionID = GUIDGenerator.generateGUID(this);
42      final private Logger log = Logger.getLogger(TransactionLogger.class.getName());
43      private int transactionID = 0;
44  
45      private TransactionLogger() {}
46      
47      /**
48       * @return a new IPatterLogger
49       */
50      public static IPatternLogger getPatternLogger() {
51      	if (instance == null) {
52      		instance = new TransactionLogger();
53      	}
54          return instance.getNewPatternLogger();
55      }
56  
57      private IPatternLogger getNewPatternLogger() {
58          if ( !this.doLog || !this.log.isDebugEnabled()) {
59              return new DummyPatternLogger();
60          }
61          IPatternLogger pl = new PatternLogger(this.PATTERN.matcher(this.orderString), this.orderString, this.log, this.logDateFormat, this.timeZone);
62          pl.paramPut(TransactionTags.ERROR_MESSAGE.toString(), "NO_ERROR");
63          pl.paramPut(TransactionTags.METHOD.toString(), new Throwable().getStackTrace()[2].getMethodName());
64          pl.paramPut(IPatternLogger.LOG_ID, Integer.toString(this.transactionID++));
65          pl.paramPut(IPatternLogger.SESSION_ID, this.sessionID);
66          return pl;
67      }
68  }
