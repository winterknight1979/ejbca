13  package org.ejbca.core.protocol.ws.common;
14  
15  import java.security.cert.Certificate;
16  import java.security.cert.CertificateException;
17  
18  import org.cesecore.certificates.certificate.CertificateConstants;
19  import org.cesecore.util.Base64;
20  import org.cesecore.util.CertTools;
21  
22  /**
23   * Class used to generate a java.security.Certificate from a 
24   * org.ejbca.core.protocol.ws.common.Certificate
25   * 
26   *
27   * $Id: CertificateHelper.java 22553 2016-01-11 13:06:46Z mikekushner $
28   */
29  public class CertificateHelper {
30  
31  	/**
32  	 * Indicates that the requester want a BASE64 encoded certificate in the CertificateResponse object.
33  	 */
34  	public static final String RESPONSETYPE_CERTIFICATE    = "CERTIFICATE";
35  	/**
36  	 * Indicates that the requester want a BASE64 encoded pkcs7 in the CertificateResponse object.
37  	 */
38  	public static final String RESPONSETYPE_PKCS7          = "PKCS7";
39  	/**
40  	 * Indicates that the requester want a BASE64 encoded pkcs7 with the complete chain in the CertificateResponse object.
41  	 */
42  	public static final String RESPONSETYPE_PKCS7WITHCHAIN = "PKCS7WITHCHAIN";
43  	
44  	/**
45  	 * Request data types for WS
46  	 */
47      public static final int CERT_REQ_TYPE_PKCS10	= CertificateConstants.CERT_REQ_TYPE_PKCS10;
48      public static final int CERT_REQ_TYPE_CRMF		= CertificateConstants.CERT_REQ_TYPE_CRMF;
49      public static final int CERT_REQ_TYPE_SPKAC     = CertificateConstants.CERT_REQ_TYPE_SPKAC;
50      public static final int CERT_REQ_TYPE_PUBLICKEY = CertificateConstants.CERT_REQ_TYPE_PUBLICKEY;
51  
52  	
53  	/**
54  	 * Method that builds a certificate from the data in the WS response.
55  	 * @param certificateData Data
56  	 * @return Cert
57  	 * @throws CertificateException On fail
58  	 */
59  	public static Certificate getCertificate(byte[] certificateData) throws CertificateException{
60          Certificate retval = CertTools.getCertfromByteArray(Base64.decode(certificateData), Certificate.class); 
61          return retval; 
62  	}
63  	
64  	/**
65  	 * Simple method that just returns raw PKCS7 data instead of the BASE64 encoded contained in
66  	 * the WS response
67  	 * @param pkcs7Data Data
68  	 * @return PKCS7
69  	 */
70  	public static byte[] getPKCS7(byte[] pkcs7Data) {
71  		return Base64.decode(pkcs7Data);
72  	}
73  	
74  	
75  }
