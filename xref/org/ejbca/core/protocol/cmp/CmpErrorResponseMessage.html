114 	@Override
115     public boolean create() throws InvalidKeyException, NoSuchAlgorithmException, NoSuchProviderException {
116 		final PKIHeaderBuilder myPKIHeaderBuilder = CmpMessageHelper.createPKIHeaderBuilder(getSender(), getRecipient(), getSenderNonce(), getRecipientNonce(), getTransactionId());
117 		boolean pbeProtected = (getPbeDigestAlg() != null) && (getPbeMacAlg() != null) && (getPbeKeyId() != null) && (getPbeKey() != null) ;
118 		if(pbeProtected) {
119 		    myPKIHeaderBuilder.setProtectionAlg(new AlgorithmIdentifier(CMPObjectIdentifiers.passwordBasedMac));
120 		}
121 		final PKIHeader myPKIHeader = myPKIHeaderBuilder.build();
122 		
123 		PKIStatusInfo myPKIStatusInfo = new PKIStatusInfo(PKIStatus.rejection);
124 		if(failInfo != null && failText != null) {
125 		    myPKIStatusInfo = new PKIStatusInfo(PKIStatus.rejection, new PKIFreeText(new DERUTF8String(failText)), CmpMessageHelper.getPKIFailureInfo(failInfo.intValue()));
126 		} else if(failText != null) {
127 		    myPKIStatusInfo = new PKIStatusInfo(PKIStatus.rejection, new PKIFreeText(new DERUTF8String(failText)));
128 		}
129 		
130 		PKIBody myPKIBody = null;
131 		log.debug("Create error message from requestType: "+requestType);
132 		if (requestType==0 || requestType==2) {
133 			myPKIBody = CmpMessageHelper.createCertRequestRejectBody(myPKIStatusInfo, requestId, requestType);
134 		} else {
135 			ErrorMsgContent myErrorContent = new ErrorMsgContent(myPKIStatusInfo);
136 			myPKIBody = new PKIBody(23, myErrorContent); // 23 = error						
137 		}
138 		PKIMessage myPKIMessage = new PKIMessage(myPKIHeader, myPKIBody);
139 		if ( pbeProtected ) {
140 			responseMessage = CmpMessageHelper.protectPKIMessageWithPBE(myPKIMessage, getPbeKeyId(), getPbeKey(), getPbeDigestAlg(), getPbeMacAlg(), getPbeIterationCount());
141 		} else {
142 			responseMessage = CmpMessageHelper.pkiMessageToByteArray(myPKIMessage);			
143 		}
144 		return true;		
145 	}
