13  package org.cesecore.roles.management;
14  
15  import java.util.List;
16  
17  import javax.ejb.Local;
18  
19  import org.cesecore.authentication.tokens.AuthenticationToken;
20  import org.cesecore.authorization.AuthorizationDeniedException;
21  import org.cesecore.roles.Role;
22  
23  /**
24   * Local interface for Role management operations.
25   * 
26   * @version $Id: RoleSessionLocal.java 25642 2017-04-03 16:27:48Z jeklund $
27   */
28  @Local
29  public interface RoleSessionLocal extends RoleSession {
30  
31      /**
32       * Checks if the administrator is allowed to add/edit/remove role members from role with the given ID.
33       * Note that role member objects may reference a CA also, which must be checked for access as well. 
34       * @param authenticationToken Auth token
35       * @param roleId ID
36       * @param requireEditAccess bool
37       * @throws AuthorizationDeniedException if access denied 
38       */
39      void assertAuthorizedToRoleMembers(AuthenticationToken authenticationToken, int roleId, boolean requireEditAccess) throws AuthorizationDeniedException;
40  
41      /** @param authenticationToken token
42       * @return a List of Roles the caller is a member of (without taking nesting into account) */
43      List<Role> getRolesAuthenticationTokenIsMemberOf(AuthenticationToken authenticationToken);
44  
45      /** @param authenticationToken token
46       * @param resource resource
47       * @return a list of all Roles that have access to the resource and the caller is allowed to see */
48      List<Role> getAuthorizedRolesWithAccessToResource(AuthenticationToken authenticationToken, String resource);
49  
50      /** 
51       * Update Role access rules and optionally any RoleMember relating to the CA ID.
52       * 
53       * Authorization to perform this operation should be checked by the caller and implied by the type of operation.
54       * @param caIdOld Old ID
55       * @param caIdNew new ID
56       * 
57       * @param keepOldAccessRule when true, the /ca/(caIdOld)/ rule will be kept as well.
58       * @param updateRoleMembers update the RoleMember.tokenIssuerId for members issued by this CA
59       * @return bool
60       */
61      boolean updateCaId(int caIdOld, int caIdNew, boolean keepOldAccessRule, boolean updateRoleMembers);
62  
63      /** @param authenticationToken Token
64       * @param roleId ID
65       * @throws AuthorizationDeniedException if changing this role would affect the access granted to the specified authenticationToken. */
66      void assertNonImportantRoleMembership(AuthenticationToken authenticationToken, int roleId) throws AuthorizationDeniedException;
67  }
