43  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "RoleDataSessionRemote")
44  @TransactionAttribute(TransactionAttributeType.REQUIRED)
45  public class RoleDataSessionBean implements RoleDataSessionLocal, RoleDataSessionRemote {

47      private static final Logger log = Logger.getLogger(RoleDataSessionBean.class);
48  
49      @EJB
50      private AccessTreeUpdateSessionLocal accessTreeUpdateSession;
51      @EJB
52      private RoleMemberDataSessionLocal roleMemberDataSession;
53  
54      @PersistenceContext(unitName = CesecoreConfiguration.PERSISTENCE_UNIT)
55      private EntityManager entityManager;

57      @Override
58      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
59      public List<Role> getAllRoles() {
60          final TypedQuery<RoleData> query = entityManager.createQuery("SELECT a FROM RoleData a", RoleData.class);
61          final List<Role> ret = new ArrayList<>();
62          for (final RoleData roleData : query.getResultList()) {
63              ret.add(roleData.getRole());
64          }
65          return ret;
66      }
67  
68      @Override
69      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
70      public Role getRole(final String nameSpace, final String roleName) {
71          final Integer roleId = RoleCache.INSTANCE.getNameToIdMap().get(Role.getRoleNameFullAsCacheName(nameSpace, roleName));
72          if (roleId != null) {
73              return getRole(roleId.intValue());
74          }
75          final RoleData result = getRoleData(nameSpace, roleName);
76          final Role role = result==null ? null : result.getRole();
77          if (role!=null) {
78              RoleCache.INSTANCE.updateWith(role.getRoleId(), role.hashCode(), Role.getRoleNameFullAsCacheName(role.getNameSpace(), role.getRoleName()), role);
79          }
80          return role;
81      }
82  
83      private RoleData getRoleData(final String nameSpace, final String roleName) {
84          if (StringUtils.isEmpty(nameSpace)) {
85              final Query query = entityManager.createQuery("SELECT a FROM RoleData a WHERE a.roleName=:roleName AND a.nameSpaceColumn IS NULL");
86              query.setParameter("roleName", roleName);
87              return QueryResultWrapper.getSingleResult(query);
88          } else {
89              final Query query = entityManager.createQuery("SELECT a FROM RoleData a WHERE a.roleName=:roleName AND a.nameSpaceColumn=:nameSpace");
90              query.setParameter("roleName", roleName);
91              query.setParameter("nameSpace", nameSpace);
92              return QueryResultWrapper.getSingleResult(query);
93          }
94      }
95  
96      @Override
97      @TransactionAttribute(TransactionAttributeType.SUPPORTS)
98      public Role getRole(final int roleId) {
99          if (roleId==Role.ROLE_ID_UNASSIGNED) {
100             // The reserved ID will never have a database entry, so return quickly with what we know will be the result
101             return null;
102         }
103         // 1. Check cache if it is time to sync-up with database
104         if (RoleCache.INSTANCE.shouldCheckForUpdates(roleId)) {
105             if (log.isDebugEnabled()) {
106                 log.debug("Object with ID " + roleId + " will be checked for updates.");
107             }
108             // 2. If cache is expired or missing, first thread to discover this reloads item from database and sends it to the cache
109             final RoleData roleData = getRoleData(roleId);
110             if (roleData==null) {
111                 if (log.isDebugEnabled()) {
112                     log.debug("Requested object did not exist in database and will be purged from cache if present: " + roleId);
113                 }
114                 // Ensure that it is removed from cache when the object is no longer present in the database
115                 RoleCache.INSTANCE.removeEntry(roleId);
116             } else {
117                 final Role role = roleData.getRole();
118                 final int digest = role.hashCode();
119                 // 3. The cache compares the database data with what is in the cache
120                 // 4. If database is different from cache, replace it in the cache
121                 RoleCache.INSTANCE.updateWith(roleId, digest, Role.getRoleNameFullAsCacheName(role.getNameSpace(), role.getRoleName()), role);
122                 // Return role, working even if the cache is disabled
123                 return role;
124             }
125         }
126         // 5. Get object from cache now (or null) and be merry
127         return RoleCache.INSTANCE.getEntry(roleId);
128     }
129 
130     private RoleData getRoleData(final int roleId) {
131         final TypedQuery<RoleData> query = entityManager.createQuery("SELECT a FROM RoleData a WHERE a.id=:id", RoleData.class);
132         query.setParameter("id", roleId);
133         return QueryResultWrapper.getSingleResult(query);
134     }
135 
136     @Override
137     public Rolele">Role persistRole(final Role role) {
138         if (role==null) {
139             // Successfully did nothing
140             return null;
141         }
142         boolean authorizationMightHaveChanged = true;
143         if (role.getRoleId()==Role.ROLE_ID_UNASSIGNED) {
144             role.setRoleId(findFreeRoleId());
145             entityManager.persist(new RoleData(role));
146         } else {
147             final RoleData roleData = getRoleData(role.getRoleId());
148             if (roleData==null) {
149                 // Must have been removed by another process, but caller wants to persist it, so we proceed (keeping the requested Role ID)
150                 entityManager.persist(new RoleData(role));
151             } else {
152                 final Role oldRole = roleData.getRole();
153                 if (role.getAccessRules().equals(oldRole.getAccessRules())) {
154                     // We will not care if the role has any members, since there is no change to the access rules
155                     authorizationMightHaveChanged = false;
156                 }
157                 // Since the entity is managed, we just update its values
158                 roleData.setRole(role);
159             }
160         }
161         RoleCache.INSTANCE.updateWith(role.getRoleId(), role.hashCode(), Role.getRoleNameFullAsCacheName(role.getNameSpace(), role.getRoleName()), role);
162         // If we only created a new Role that has no members yet or the access rules did no change, the authorization would not have changed
163         authorizationMightHaveChanged &= isRoleMembersPresent(role.getRoleId());
164         if (authorizationMightHaveChanged) {
165             accessTreeUpdateSession.signalForAccessTreeUpdate();
166         }
167         return role;
168     }
169 
170     /** @return a integer Id that is currently unused in the database */
171     private int findFreeRoleId() {
172         final ProfileID.DB db = new ProfileID.DB() {
173             @Override
174             public boolean isFree(final int candidate) {
175                 return candidate!=Role.ROLE_ID_UNASSIGNED && getRole(candidate) == null;
176             }
177         };
178         return ProfileID.getNotUsedID(db);
179     }
180 
181     @Override
182     public boolean deleteRoleNoAuthorizationCheck(final int roleId) {
183         // Use an DELETE query instead of entityManager.remove to tolerate concurrent deletion better
184         final Query query = entityManager.createQuery("DELETE FROM RoleData a WHERE a.id=:id");
185         query.setParameter("id", roleId);
186         final boolean ret = query.executeUpdate()==1;
187         if (ret && isRoleMembersPresent(roleId)) {
188             accessTreeUpdateSession.signalForAccessTreeUpdate();
189         }
190         return ret;
191     }
192     
193     private boolean isRoleMembersPresent(final int roleId) {
194         return !roleMemberDataSession.findByRoleId(roleId).isEmpty();
195     }
196 
197     @Override
198     public void forceCacheExpire() {
199         RoleCache.INSTANCE.flush();
200     }
