71      /** @param authenticationToken Token
72       * @param roleMember Member
73       * @return the authorized role 
74       * @throws AuthorizationDeniedException if access denied */
75      private Role lookupRoleAndCheckAuthorization(final AuthenticationToken authenticationToken, final RoleMember roleMember) throws AuthorizationDeniedException {
76          // Check existence and authorization of referenced objects
77          final Role role = roleSession.getRole(authenticationToken, roleMember.getRoleId());
78          if (roleMember.getRoleId() != RoleMember.NO_ROLE && role == null) {
79              throw new IllegalStateException("Role with ID " + roleMember.getRoleId() + " was not found, or administrator is not authorized to it");
80          }
81          if (roleMember.getTokenIssuerId() != RoleMember.NO_ISSUER) {
82              // Do more expensive fully correct check if it is potentially issued by a CA
83              final AuthenticationTokenMetaData metaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(roleMember.getTokenType());
84              final AccessMatchValue accessMatchValue = metaData.getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey());
85              if (accessMatchValue.isIssuedByCa()) {
86                  // According to the meta data, this tokenIssuerId should be interpreted as a CA ID
87                  final int caId = roleMember.getTokenIssuerId();
88                  if (!authorizationSession.isAuthorizedNoLogging(authenticationToken, StandardRules.CAACCESS.resource() + caId)) {
89                      throw new AuthorizationDeniedException("CA with ID " + caId + " was not found, or administrator is not authorized to it");
90                  }
91              }
92          }
93          return role;
94      }

101     @Override
102     public RoleMember persist(fiRoleMemberg> AuthenticationToken authenticationToken, final RoleMember roleMember, final boolean requireNonImportantRoleMembership)
103             throws AuthorizationDeniedException {
104         if (roleMember==null) {
105             // Successfully did nothing
106             return null;
107         }
108         if (requireNonImportantRoleMembership) {
109             assertNonImportantRoleMembership(authenticationToken, roleMember);
110         }
111         roleSession.assertAuthorizedToRoleMembers(authenticationToken, roleMember.getRoleId(), true);
112         RoleMember oldRoleMember = null;
113         if (roleMember.getId() != RoleMember.ROLE_MEMBER_ID_UNASSIGNED) {
114             // Try to locate an existing RoleMember
115             oldRoleMember = roleMemberDataSession.findRoleMember(roleMember.getId());
116             // If the role Id will change, assert that we have access to the old value first
117             if (oldRoleMember != null && roleMember.getRoleId()!=oldRoleMember.getRoleId()) {
118                 lookupRoleAndCheckAuthorization(authenticationToken, oldRoleMember);
119             }
120         }
121         if (log.isDebugEnabled()) {
122             log.debug("Persisting a role member with ID " + roleMember.getId() + " and match value '" + roleMember.getTokenMatchValue() + "'");
123         }
124         final Role role = lookupRoleAndCheckAuthorization(authenticationToken, roleMember);
125         normalizeRoleMember(roleMember);
126         final RoleMember persistedRoleMember = roleMemberDataSession.persistRoleMember(roleMember);
127         final boolean addedRoleMember = (oldRoleMember==null);
128         final String tokenType = persistedRoleMember.getTokenType();
129         final int tokenMatchKey = persistedRoleMember.getTokenMatchKey();
130         final String tokenMatchKeyName = AccessMatchValueReverseLookupRegistry.INSTANCE.performReverseLookup(tokenType, tokenMatchKey).name();
131         final String msg;
132         if (addedRoleMember) {
133             msg = InternalResources.getInstance().getLocalizedMessage("authorization.adminadded", persistedRoleMember.getTokenMatchValue(), role.getRoleNameFull());
134         } else {
135             msg = InternalResources.getInstance().getLocalizedMessage("authorization.adminchanged", persistedRoleMember.getTokenMatchValue(), role.getRoleNameFull());
136         }
137         final Map<String, Object> details = new LinkedHashMap<>();
138         details.put("msg", msg);
139         details.put("id", persistedRoleMember.getId());
140         if (addedRoleMember || !oldRoleMember.getTokenType().equals(persistedRoleMember.getTokenType())) {
141             details.put("tokenType", persistedRoleMember.getTokenType());
142         }
143         if (addedRoleMember || oldRoleMember.getTokenIssuerId()!=persistedRoleMember.getTokenIssuerId()) {
144             details.put("tokenIssuerId", roleMember.getTokenIssuerId());
145         }
146         if (addedRoleMember || oldRoleMember.getTokenMatchKey()!=persistedRoleMember.getTokenMatchKey()) {
147             details.put("tokenMatchKey", tokenMatchKeyName + " (" + persistedRoleMember.getTokenMatchKey()+ ")");
148         }
149         if (addedRoleMember || oldRoleMember.getTokenMatchOperator()!=persistedRoleMember.getTokenMatchOperator()) {
150             details.put("tokenMatchOperator", persistedRoleMember.getAccessMatchType().name() + " (" + persistedRoleMember.getTokenMatchOperator()+ ")");
151         }
152         if (addedRoleMember || !StringUtils.equals(oldRoleMember.getTokenMatchValue(), persistedRoleMember.getTokenMatchValue())) {
153             details.put("tokenMatchValue", persistedRoleMember.getTokenMatchValue());
154         }
155         if (addedRoleMember || oldRoleMember.getRoleId()!=persistedRoleMember.getRoleId()) {
156             details.put("roleId", roleMember.getRoleId());
157             details.put("nameSpace", role.getNameSpace());
158             details.put("roleName", role.getRoleName());
159         }
160         if (addedRoleMember || !StringUtils.equals(oldRoleMember.getDescription(), persistedRoleMember.getDescription())) {
161             details.put("description", persistedRoleMember.getDescription());
162         }
163         final EventType eventType = addedRoleMember ? EventTypes.ROLE_ACCESS_USER_ADDITION : EventTypes.ROLE_ACCESS_USER_CHANGE;
164         securityEventsLoggerSession.log(eventType, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE, authenticationToken.toString(), null, null, null, details);
165         return persistedRoleMember;
166     }

168     /**
169      * Normalizes data in the role member, e.g. changing serial numbers to uppercase without leading zeros.
170      * @param roleMember Member
171      */
172     private void normalizeRoleMember(final RoleMember roleMember) {
173         final AccessMatchValue matchKey = AccessMatchValueReverseLookupRegistry.INSTANCE.performReverseLookup(roleMember.getTokenType(), roleMember.getTokenMatchKey());
174         if (matchKey != null) {
175             roleMember.setTokenMatchValue(matchKey.normalizeMatchValue(roleMember.getTokenMatchValue()));
176         }
177     }

179     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
180     @Override
181     public RoleMember getRoleMember(final AuthenticationToken authenticationToken, final int roleMemberId) throws AuthorizationDeniedException {
182         final RoleMember roleMember = roleMemberDataSession.findRoleMember(roleMemberId);
183         if (roleMember == null) {
184             return null;
185         }
186         // Authorization checks
187         roleSession.assertAuthorizedToRoleMembers(authenticationToken, roleMember.getRoleId(), false);
188         lookupRoleAndCheckAuthorization(authenticationToken, roleMember);
189         return roleMember;
190     }

192     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
193     @Override
194     public List<RoleMember> getRoleMembersByRoleId(final AuthenticationToken authenticationToken, final int roleId) throws AuthorizationDeniedException {
195         // Ensure that the role exists and that the caller is authorized to it
196         if (!authorizationSession.isAuthorizedNoLogging(authenticationToken, StandardRules.VIEWROLES.resource())) {
197             final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.notauthorizedtoviewroles", authenticationToken.toString());
198             throw new AuthorizationDeniedException(msg);
199         }
200         if (roleSession.getRole(authenticationToken, roleId)==null) {
201             return null;
202         }
203         final List<RoleMember> ret = new ArrayList<>();
204         final Set<String> requiredCaAccessResources = new HashSet<>();
205         for (final RoleMemberData roleMemberData : roleMemberDataSession.findByRoleId(roleId)) {
206             final RoleMember roleMember = roleMemberData.asValueObject();
207             final AuthenticationTokenMetaData metaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(roleMember.getTokenType());
208             if (metaData.getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey()).isIssuedByCa()) {
209                 requiredCaAccessResources.add(StandardRules.CAACCESS.resource() + roleMember.getTokenIssuerId());
210             }
211             ret.add(roleMember);
212         }
213         final String[] requiredCaAccessResourcesArray = requiredCaAccessResources.toArray(new String[requiredCaAccessResources.size()]);
214         if (!requiredCaAccessResources.isEmpty() && !authorizationSession.isAuthorizedNoLogging(authenticationToken, requiredCaAccessResourcesArray)) {
215             throw new AuthorizationDeniedException("Not authorized to all members in role.");
216         }
217         return ret;
218     }

220     /** @param authenticationToken Token
221      * @param roleMember Member
222      * @throws AuthorizationDeniedException if the provided RoleMember is the only member in the particular RoleMember's Role matching the authentication token */
223     private void assertNonImportantRoleMembership(final AuthenticationToken authenticationToken, final RoleMember roleMember) throws AuthorizationDeniedException {
224         final List<RoleMember> roleMembers = getRoleMembersByRoleId(authenticationToken, roleMember.getRoleId());
225         int count = 0;
226         for (final RoleMember current : roleMembers) {
227             if (current.getTokenType().equals(roleMember.getTokenType()) &&
228                     current.getTokenIssuerId()==roleMember.getTokenIssuerId() &&
229                     current.getTokenMatchKey()==roleMember.getTokenMatchKey() &&
230                     current.getTokenMatchOperator()==roleMember.getTokenMatchOperator() &&
231                     StringUtils.equals(current.getTokenMatchValue(), roleMember.getTokenMatchValue())) {
232                 count++;
233             }
234         }
235         // If there are no duplicate matches for this RoleMember...
236         if (count<2) {
237             if (log.isDebugEnabled()) {
238                 log.debug("No RoleMember provides the same match as the one with id " + roleMember.getId() + ". count="+count);
239             }
240             // ...and the caller relies on this match for access the access granted by this Role...
241             for (final RoleMember current : roleMemberDataSession.getRoleMembersMatchingAuthenticationToken(authenticationToken)) {
242                 if (roleMember.getId() == current.getId()) {
243                     if (log.isDebugEnabled()) {
244                         log.debug("'"+authenticationToken+"' relies on match from RoleMember with id " + roleMember.getId() + ".");
245                     }
246                     // ...also check if there are other roles that would provide the same access as a this Role
247                     roleSession.assertNonImportantRoleMembership(authenticationToken, roleMember.getRoleId());
248                 }
249             }
250         } else {
251             if (log.isDebugEnabled()) {
252                 log.debug("RoleMembers provides the same match as the one with id " + roleMember.getId() + ". count="+count);
253             }
254         }
255     }

257     @Override
258     public boolean remove(final AuthenticationToken authenticationToken, final int roleMemberId) throws AuthorizationDeniedException {
259         final RoleMember roleMember = roleMemberDataSession.findRoleMember(roleMemberId);
260         if (roleMember == null) {
261             return false;
262         }
263         assertNonImportantRoleMembership(authenticationToken, roleMember);
264         roleSession.assertAuthorizedToRoleMembers(authenticationToken, roleMember.getRoleId(), true);
265         final Role role = lookupRoleAndCheckAuthorization(authenticationToken, roleMember);
266         final boolean removed = roleMemberDataSession.remove(roleMemberId);
267         if (removed) {
268             final String tokenType = roleMember.getTokenType();
269             final int tokenMatchKey = roleMember.getTokenMatchKey();
270             final String tokenMatchKeyName = AccessMatchValueReverseLookupRegistry.INSTANCE.performReverseLookup(tokenType, tokenMatchKey).name();
271             final String msg = InternalResources.getInstance().getLocalizedMessage("authorization.adminremoved", roleMember.getTokenMatchValue(), role.getRoleNameFull());
272             final Map<String, Object> details = new LinkedHashMap<>();
273             details.put("msg", msg);
274             details.put("id", roleMember.getId());
275             details.put("tokenType", roleMember.getTokenType());
276             details.put("tokenIssuerId", roleMember.getTokenIssuerId());
277             details.put("tokenMatchKey", tokenMatchKeyName + " (" + roleMember.getTokenMatchKey()+ ")");
278             details.put("tokenMatchOperator", roleMember.getAccessMatchType().name() + " (" + roleMember.getTokenMatchOperator()+ ")");
279             details.put("tokenMatchValue", roleMember.getTokenMatchValue());
280             details.put("roleId", roleMember.getRoleId());
281             details.put("nameSpace", role.getNameSpace());
282             details.put("roleName", role.getRoleName());
283             details.put("description", roleMember.getDescription());
284             securityEventsLoggerSession.log(EventTypes.ROLE_ACCESS_USER_DELETION, EventStatus.SUCCESS, ModuleTypes.ROLES, ServiceTypes.CORE,
285                     authenticationToken.toString(), null, null, null, details);
286         }
287         return removed;
288     }
