58      @Override
59      public RoleMemberoleMember">RoleMember persistRoleMember(final RoleMember roleMember) {
60          if (roleMember==null) {
61              // Successfully did nothing
62              return null;
63          }
64          if (roleMember.getId() == RoleMember.ROLE_MEMBER_ID_UNASSIGNED) {
65              roleMember.setId(findFreePrimaryKey());
66              entityManager.persist(new RoleMemberData(roleMember));
67          } else {
68              final RoleMemberData roleMemberData = find(roleMember.getId());
69              if (roleMemberData==null) {
70                  // Must have been removed by another process, but caller wants to persist it, so we proceed
71                  entityManager.persist(new RoleMemberData(roleMember));
72              } else {
73                  // Since the entity is managed, we just update its values
74                  roleMemberData.updateValuesFromValueObject(roleMember);
75              }
76          }
77          accessTreeUpdateSession.signalForAccessTreeUpdate();
78          RoleMemberCache.INSTANCE.updateWith(roleMember.getId(), roleMember.hashCode(), null, roleMember);
79          AuthenticationTokenCache.INSTANCE.flush();
80          return roleMember;
81      }

83      private int findFreePrimaryKey() {
84          final ProfileID.DB db = new ProfileID.DB() {
85              @Override
86              public boolean isFree(int i) {
87                  //0 is a protected ID for RoleMemberData. Use only positive values, since negatives are seen as "erronous" by some customers.
88                  return find(i) == null && i > 0;
89              }
90          };
91          return ProfileID.getNotUsedID(db);
92      }

100     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
101     @Override
102     public RoleMember findRoleMember(int primaryKey) {
103         // 1. Check cache if it is time to sync-up with database
104         if (RoleMemberCache.INSTANCE.shouldCheckForUpdates(primaryKey)) {
105             // 2. If cache is expired or missing, first thread to discover this reloads item from database and sends it to the cache
106             final RoleMemberData roleMemberData = find(primaryKey);
107             if (roleMemberData == null) {
108                 // Ensure that it is removed from cache when the object is no longer present in the database
109                 RoleMemberCache.INSTANCE.removeEntry(primaryKey);
110             } else {
111                 final RoleMember roleMember = roleMemberData == null ? null : roleMemberData.asValueObject();
112                 // 3. The cache compares the database data with what is in the cache
113                 // 4. If database is different from cache, replace it in the cache
114                 RoleMemberCache.INSTANCE.updateWith(primaryKey, roleMember.hashCode(), null, roleMember);
115                 // Return role member, working even if the cache is disabled
116                 return roleMember;
117             }
118         }
119         // 5. Get object from cache now (or null) and be merry
120         return RoleMemberCache.INSTANCE.getEntry(primaryKey);
121     }

144     @Override
145     public boolean remove(final int primaryKey) {
146         RoleMemberData roleMember = find(primaryKey);
147         if (roleMember != null) {
148             entityManager.remove(roleMember);
149             accessTreeUpdateSession.signalForAccessTreeUpdate();
150             AuthenticationTokenCache.INSTANCE.flush();
151             RoleMemberCache.INSTANCE.removeEntry(primaryKey);
152             return true;
153         } else {
154             return false;
155         }
156     }

169     /**
170      * This method will retrieves all role members that match a certain authentication token. It utilizes the AuthenticationTokenCache to do so, 
171      * a cache which saves previous authorization token searches. If the cache entry hasn't expired the cache will return a list of role 
172      * member IDs, which will be further pruned if any of the listed role members are missing (due to having been removed by another node in the 
173      * cluster).
174      * 
175      * @param authenticationToken an authentication token
176      * @return a list of role members that the authentication token matches
177      */
178     private List<RoleMember> getRoleMembersForAuthenticationToken(final AuthenticationToken authenticationToken) {
179         final String tokenType = authenticationToken.getMetaData().getTokenType();
180         final TypedQuery<RoleMemberData> query;
181         final int preferredMatchKey = authenticationToken.getPreferredMatchKey();
182         List<RoleMember> result = new ArrayList<>();
183         //First check the cache if this authentication token has been checked recently
184         List<Integer> cachedRoleMemberIds = AuthenticationTokenCache.INSTANCE.getCachedRoleMembersForAuthenticationToken(authenticationToken);            
185         if (cachedRoleMemberIds != null) {
186             //We found a cached (and valid) entry
187             for(Integer roleMemberId : cachedRoleMemberIds) {
188                 //Role Members may have been removed in the background, so be prepared for this as well. If that happens, we should update 
189                 //the cache entry as well.
190                 RoleMember roleMember = findRoleMember(roleMemberId);
191                 if(roleMember != null) {
192                     result.add(roleMember);
193                 }
194             }
195             //If the number of found members differs from the number of ID's in the cache, there must be members missing, and we should update 
196             //the authorization cache
197             if(result.size() != cachedRoleMemberIds.size()) {
198                 AuthenticationTokenCache.INSTANCE.cacheRoleMembersForAuthenticationToken(authenticationToken, result);
199             }    
200         } else {
201             //We found nothing...
202             if (preferredMatchKey != AuthenticationToken.NO_PREFERRED_MATCH_KEY) {
203                 final List<AccessMatchType> accessMatchType = authenticationToken.getMetaData().getAccessMatchValueIdMap().get(preferredMatchKey)
204                         .getAvailableAccessMatchTypes();
205                 final int preferredOperator = accessMatchType.isEmpty() ? AccessMatchType.TYPE_UNUSED.getNumericValue()
206                         : accessMatchType.get(0).getNumericValue();
207                 // Optimized search for preferred match values (e.g. serial number match key) amongst members with that match key.
208                 // For members with other match keys, we include everything in the search
209                 query = entityManager
210                         .createQuery("SELECT a FROM RoleMemberData a WHERE a.tokenType=:tokenType AND a.roleId<>0 AND "
211                                 + "((a.tokenMatchKey=:preferredTokenMatchKey AND a.tokenMatchOperator=:operator AND a.tokenMatchValueColumn=:preferredTokenMatchValue) OR "
212                                 + "NOT (a.tokenMatchKey=:preferredTokenMatchKey AND a.tokenMatchOperator=:operator))", RoleMemberData.class)
213                         .setParameter("tokenType", tokenType).setParameter("preferredTokenMatchKey", preferredMatchKey)
214                         .setParameter("operator", preferredOperator)
215                         .setParameter("preferredTokenMatchValue", authenticationToken.getPreferredMatchValue());
216             } else {
217             // Search for all members with the same token type
218             query = entityManager.createQuery("SELECT a FROM RoleMemberData a WHERE a.tokenType=:tokenType AND a.roleId<>0", RoleMemberData.class)
219                     .setParameter("tokenType", tokenType);
220             }
221             for (RoleMemberData roleMemberData : query.getResultList()) {
222                 result.add(roleMemberData.asValueObject());
223             }
224             if (!result.isEmpty()) {
225                 AuthenticationTokenCache.INSTANCE.cacheRoleMembersForAuthenticationToken(authenticationToken, result);
226             }
227         }
228 
229         return result;
230 
231     }

233     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
234     @Override
235     public Set<Integer> getRoleIdsMatchingAuthenticationTokenOrFail(final AuthenticationToken authenticationToken) throws AuthenticationFailedException {
236         final Set<Integer> ret = new HashSet<>();
237         if (authenticationToken!=null) {
238             for (final RoleMember roleMemberData : getRoleMembersForAuthenticationToken(authenticationToken)) {
239                 if (authenticationToken.matches(convertToAccessUserAspect(roleMemberData))) {
240                     ret.add(roleMemberData.getRoleId());
241                 }
242             }
243         }
244         return ret;
245     }
246     
247     @TransactionAttribute(TransactionAttributeType.SUPPORTS)
248     @Override
249     public Set<RoleMember> getRoleMembersMatchingAuthenticationToken(final AuthenticationToken authenticationToken) {
250         final Set<RoleMember> ret = new HashSet<>();
251         if (authenticationToken!=null) {
252             for (final RoleMember roleMember : getRoleMembersForAuthenticationToken(authenticationToken)) {
253                 try {
254                     if (authenticationToken.matches(convertToAccessUserAspect(roleMember))) {
255                         ret.add(roleMember);
256                     }
257                 } catch (AuthenticationFailedException e) {
258                     log.debug(e.getMessage(), e);
259                 }
260             }
261         }
262         return ret;
263     }
