13  package org.cesecore.audit.impl.integrityprotected;
14  
15  import java.util.concurrent.atomic.AtomicLong;
16  import java.util.concurrent.locks.ReentrantLock;
17  
18  /**
19   * Get this node's next log row sequence number.
20   * 
21   * The sequence number is guaranteed to be unique as long as the read "node identifier"
22   * read on first access is unique among the nodes sharing the database.
23   * 
24   * @version $Id: NodeSequenceHolder.java 24598 2016-10-31 11:34:40Z jeklund $
25   */
26  public enum NodeSequenceHolder {
27      INSTANCE;
28  
29      // We only want to use this from IntegrityProtectedDevice
30      private NodeSequenceHolder() {}
31  
32      private final AtomicLong lastSequenceNumberAtomic = new AtomicLong(-1);
33      private final ReentrantLock lockDataBaseUpdate = new ReentrantLock();
34      private String nodeId = null;
35      
36      /** Interface for callback of methods that is invoked once. */
37      public interface OnInitCallBack {
38          /** @return the current node identifier */
39          String getNodeId();
40          /** @param nodeId ID
41           * @return the highest known sequence number for the node identifier returned by {@link OnInitCallBack#getNodeId()}*/
42          long getMaxSequenceNumberForNode(String nodeId);
43      }
44  
45      /** @param callBack Callback
46       * @return the node's next log row sequence number. */
47      public long getNext(final OnInitCallBack callBack) {
48          if (lastSequenceNumberAtomic.get()==-1L) {
49              try {
50                  // Lock threads during initializations of value from the database (only one thread needs to/should do the lookup)
51                  lockDataBaseUpdate.lock();
52                  if (lastSequenceNumberAtomic.get()==-1L) {
53                      // Note that it very important that the nodeId is cached to avoid gaps in the sequence if the nodeId changes
54                      nodeId = callBack.getNodeId();
55                      lastSequenceNumberAtomic.set(callBack.getMaxSequenceNumberForNode(nodeId));
56                  }
57              } finally {
58                  lockDataBaseUpdate.unlock();
59              }
60          }
61          return lastSequenceNumberAtomic.incrementAndGet();
62      }
63  
64      /** @return the Node Identifier that this sequence number applies to. */
65      public String getNodeId() {
66          return nodeId;
67      }
68  
69      /**
70       * Trigger a re-read of the nodeId and last sequence number for the node on next call to {@link NodeSequenceHolder#getNext(OnInitCallBack)}.
71       * 
72       * WARNING: This class is NOT safe to reset in an concurrent environment and this method should only be used from unit tests.
73       */
74      public void reset() {
75          lastSequenceNumberAtomic.set(-1L);
76      }
77  }
