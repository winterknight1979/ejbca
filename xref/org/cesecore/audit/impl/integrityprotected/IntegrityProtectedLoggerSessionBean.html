37  /**
38   * An alternative implementation of the SecurityEventsLogger interface. It handles the creation of a signed log for an event.
39   * 
40   * This was created to evaluate the performance of using database integrity protection instead of custom code for log singing.
41   * 
42   * @version $Id: IntegrityProtectedLoggerSessionBean.java 24600 2016-10-31 12:01:55Z jeklund $
43   */
44  @Stateless
45  @TransactionAttribute(TransactionAttributeType.REQUIRED)
46  public class IntegrityProtectedLoggerSessionBean implements IntegrityProtectedLoggerSessionLocal {
47  
48      private static final Logger log = Logger.getLogger(IntegrityProtectedLoggerSessionBean.class);
49  
50      @PersistenceContext(unitName = CesecoreConfiguration.PERSISTENCE_UNIT)
51      private EntityManager entityManager;
52  
53      @PostConstruct
54      public void postConstruct() {
55          CryptoProviderTools.installBCProviderIfNotAvailable();
56      }
57  
58      /**
59       * Initialization of the log sequence number in combination with nodeId should be performed exactly once.
60       * 
61       * This callback will be invoked on the first call to NodeSequenceHolder.getNext(...) to perform this initialization.
62       * 
63       * In this callback implementation, the nodeId is first read from the configuration (which may default to reading
64       * the current hostname from the system). This hostname is then passed to the next method to figure out what the
65       * highest present sequenceNumber for this nodeId is in the database (e.g. last write before shutting down).
66       */
67      private final NodeSequenceHolder.OnInitCallBack sequenceHolderInitialization = new NodeSequenceHolder.OnInitCallBack() {
68          @Override
69          public String getNodeId() {
70              return CesecoreConfiguration.getNodeIdentifier();
71          }
72          @Override
73          public long getMaxSequenceNumberForNode(final String nodeId) {
74              // Get the latest sequenceNumber from last run from the database..
75              final Query query = entityManager.createQuery("SELECT MAX(a.sequenceNumber) FROM AuditRecordData a WHERE a.nodeId=:nodeId");
76              query.setParameter("nodeId", nodeId);
77              return QueryResultWrapper.getSingleResult(query, Long.valueOf(-1)).longValue();
78          }
79      };
80  
81      @Override
82      @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
83      // Always persist audit log
84      public void log(final TrustedTime trustedTime, final EventType eventType, final EventStatus eventStatus, final ModuleType module,
85              final ServiceType service, final String authToken, final String customId, final String searchDetail1, final String searchDetail2,
86              final Map<String, Object> additionalDetails, final Properties properties) throws AuditRecordStorageException {
87          if (log.isTraceEnabled()) {
88              log.trace(String.format(">log:%s:%s:%s:%s:%s:%s", eventType, eventStatus, module, service, authToken, additionalDetails));
89          }
90          try {
91              final Long sequenceNumber = NodeSequenceHolder.INSTANCE.getNext(sequenceHolderInitialization);
92              // Make sure to use the Node Identifier that this log sequence was initialized with (for example hostnames reported by the system could change)
93              final String nodeId = NodeSequenceHolder.INSTANCE.getNodeId();
94              final Long timeStamp = Long.valueOf(trustedTime.getTime().getTime());
95              final AuditRecordDataAuditRecordData.html#AuditRecordData">AuditRecordData auditRecordData = new AuditRecordData(nodeId, sequenceNumber, timeStamp, eventType, eventStatus, authToken,
96                      service, module, customId, searchDetail1, searchDetail2, additionalDetails);
97              entityManager.persist(auditRecordData);
98          } catch (Exception e) {
99              log.error(e.getMessage(), e);
100             throw new AuditRecordStorageException(e.getMessage(), e);
101         } finally {
102             if (log.isTraceEnabled()) {
103                 log.trace("<log");
104             }
105         }
106     }
