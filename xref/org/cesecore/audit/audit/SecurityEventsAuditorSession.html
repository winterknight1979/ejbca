13  package org.cesecore.audit.audit;
14  
15  import java.security.cert.Certificate;
16  import java.util.Date;
17  import java.util.List;
18  import java.util.Map;
19  import java.util.Set;
20  
21  import org.cesecore.audit.AuditLogEntry;
22  import org.cesecore.authentication.tokens.AuthenticationToken;
23  import org.cesecore.authorization.AuthorizationDeniedException;
24  import org.cesecore.keys.token.CryptoToken;
25  import org.cesecore.util.query.QueryCriteria;
26  
27  /**
28   * Allows auditing of securely logged events.
29   * 
30   * @see  <a href="https://www.cesecore.eu/mediawiki/index.php/Functional_Specifications_(ADV_FSP)#Audit_Security_Events">
31   * Audit Security Events</a>
32   * 
33   * @version $Id: SecurityEventsAuditorSession.java 17625 2013-09-20 07:12:06Z netmackan $
34   */
35  public interface SecurityEventsAuditorSession {
36  
37      /**
38       * Select a set of events to be audited.
39       * 
40       * @param token identifier of the entity performing the task.
41       * @param startIndex Index where select will start. Set to 0 to start from the beginning.
42       * @param max maximum number of results to be returned. Set to 0 to use no limit.
43       * @param criteria Criteria defining the subset of logs to be selected.
44       * @param logDeviceId identifier of the AuditLogDevice
45       * 
46       * @return The audit logs to the given criteria
47       * @throws AuthorizationDeniedException If non-admin user
48       */
49      List<? extends AuditLogEntry> selectAuditLogs(AuthenticationToken token, int startIndex, int max, QueryCriteria criteria, String logDeviceId) throws AuthorizationDeniedException;
50  
51      /**
52       * This operation is used to export a set logs.
53       * 
54       * @param token identifier of the entity performing the task.
55       * @param cryptoToken Crypto Token to be used.
56       * @param timestamp Timestamp till which the logs will be exported.
57       * @param deleteAfterExport Deletes the exported results if true.
58       * @param signatureDetails
59       *            Map containing signature details used by the signer implementation.
60       *            @see SigningFileOutputStream public static variables for examples.
61       * @param logDeviceId identifier of the AuditLogDevice
62       * 
63       * @return A extended validation report with the path to the exported file.
64       * 
65       * @throws AuditLogExporterException If export fails
66       * @throws AuthorizationDeniedException If non-admin user
67       */
68      AuditLogExportReport exportAuditLogs(AuthenticationToken token, CryptoToken cryptoToken, Date timestamp, boolean deleteAfterExport,
69              Map<String, Object> signatureDetails, String logDeviceId) throws AuditLogExporterException, AuthorizationDeniedException;
70  
71      /**
72       * This operation is used to export a set logs.
73       * 
74       * @param token identifier of the entity performing the task.
75       * @param cryptoToken Crypto Token to be used.
76       * @param timestamp Timestamp till which the logs will be exported.
77       * @param deleteAfterExport Deletes the exported results if true.
78       * @param keyAlias alias identifying the key to be use in the file signature.
79       * @param algorithm signature algorithm.
80       * @param logDeviceId identifier of the AuditLogDevice
81       * 
82       * @return A extended validation report with the path to the exported file.
83       * @throws AuditLogExporterException If export fails
84       * @throws AuthorizationDeniedException If non-admin user
85       */
86      AuditLogExportReport exportAuditLogs(AuthenticationToken token, CryptoToken cryptoToken, Date timestamp, boolean deleteAfterExport,
87              String keyAlias, String algorithm, String logDeviceId) throws AuditLogExporterException, AuthorizationDeniedException;
88  
89      /**
90       * This operation is used to export a set logs.
91       * 
92       * @param token identifier of the entity performing the task.
93       * @param cryptoToken Crypto Token to be used.
94       * @param timestamp Timestamp till which the logs will be exported.
95       * @param deleteAfterExport Deletes the exported results if true.
96       * @param keyAlias alias identifying the key to be use in the file signature.
97       * @param algorithm signature algorithm.
98       * @param certificate certificate used in the signature.
99       * @param logDeviceId identifier of the AuditLogDevice
100      * 
101      * @return A extended validation report with the path to the exported file.
102      *
103      * @throws AuditLogExporterException If export fails
104      * @throws AuthorizationDeniedException If non-admin user
105      */
106     AuditLogExportReport exportAuditLogs(final AuthenticationToken token, final CryptoToken cryptoToken, final Date timestamp, final boolean deleteAfterExport,
107             final String keyAlias, final String algorithm, final Certificate certificate, String logDeviceId) throws AuditLogExporterException, AuthorizationDeniedException;
108 
109     /**
110      * This operation is used to verify integrity of log to detect potential
111      * modifications.
112      * 
113      * @param token identifier of the entity performing the task.
114      * @param date Date from which to start verifying logs.
115      * @param logDeviceId identifier of the AuditLogDevice
116      * 
117      * @return validation report.
118      * @throws AuditLogValidatorException If export fails
119      * @throws AuthorizationDeniedException If non-admin user
120      */
121     AuditLogValidationReport verifyLogsIntegrity(AuthenticationToken token, Date date, String logDeviceId) throws AuditLogValidatorException, AuthorizationDeniedException;
122 
123     /** @return a Set of ids for AuditLogDevice that supports querying. */
124 	Set<String> getQuerySupportingLogDevices();
125 }
