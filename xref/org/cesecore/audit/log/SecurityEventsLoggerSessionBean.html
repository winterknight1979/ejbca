13  package org.cesecore.audit.log;
14  
15  import java.util.LinkedHashMap;
16  import java.util.Map;
17  
18  import javax.ejb.EJB;
19  import javax.ejb.Stateless;
20  import javax.ejb.TransactionAttribute;
21  import javax.ejb.TransactionAttributeType;
22  
23  import org.apache.log4j.Logger;
24  import org.cesecore.audit.enums.EventStatus;
25  import org.cesecore.audit.enums.EventType;
26  import org.cesecore.audit.enums.ModuleType;
27  import org.cesecore.audit.enums.ServiceType;

29  import org.cesecore.authorization.AuthorizationDeniedException;
30  import org.cesecore.authorization.AuthorizationSessionLocal;
31  import org.cesecore.authorization.control.AuditLogRules;

33  import org.cesecore.jndi.JndiConstants;

36  import org.cesecore.time.providers.TrustedTimeProviderException;

38  /**
39   * This class implements the SecurityEventsLogger interface. It handles the
40   * creation of a signed log for an event.
41   * 
42   * @version $Id: SecurityEventsLoggerSessionBean.java 29010 2018-05-23 13:09:53Z jekaterina_b_helmes $
43   * 
44   */
45  @Stateless(mappedName = JndiConstants.APP_JNDI_PREFIX + "SecurityEventsLoggerSessionRemote")
46  @TransactionAttribute(TransactionAttributeType.SUPPORTS)
47  public class SecurityEventsLoggerSessionBean implements SecurityEventsLoggerSessionLocal, SecurityEventsLoggerSessionRemote {
48  
49      private static final Logger log = Logger.getLogger(SecurityEventsLoggerSessionBean.class);
50      /** Internal localization of logs and errors */
51      private static final InternalResources intres = InternalResources.getInstance();
52  
53      @EJB
54      private InternalSecurityEventsLoggerSessionLocal internalSecurityEventsLoggerSession;
55      @EJB
56      private AuthorizationSessionLocal authorizationSession;
57      @EJB
58      private TrustedTimeWatcherSessionLocal trustedTimeWatcherSession;
59  
60      @Override
61      public void log(final AuthenticationToken authToken, EventType eventType, EventStatus eventStatus, ModuleType module, ServiceType service)
62              throws AuditRecordStorageException, AuthorizationDeniedException {
63          log(authToken, eventType, eventStatus, module, service, null, null, null, null);
64      }
65  
66      @Override
67      public void log(final AuthenticationToken authToken, EventType eventType, EventStatus eventStatus, ModuleType module, ServiceType service, String customId, String searchDetail1, String searchDetail2,
68              Map<String, Object> additionalDetails) throws AuditRecordStorageException, AuthorizationDeniedException {        
69          // We need to check that admin have rights to log
70          if (!authorizationSession.isAuthorized(authToken, AuditLogRules.LOG.resource())) {
71              final String msg = intres.getLocalizedMessage("authorization.notauthorizedtoresource", AuditLogRules.LOG.resource(), null);
72              throw new AuthorizationDeniedException(msg);
73          }
74          log(eventType, eventStatus, module, service, authToken.toString(), customId, searchDetail1, searchDetail2, additionalDetails);
75      }
76      
77      @Override
78      public void log(final EventType eventType, final EventStatus eventStatus, final ModuleType module, final ServiceType service, final String authToken, final String customId, final String searchDetail1, final String searchDetail2,
79              final String additionalDetailsMsg) throws AuditRecordStorageException {
80          final Map<String, Object> additionalDetails = new LinkedHashMap<String, Object>();
81          additionalDetails.put("msg", additionalDetailsMsg);
82          log(eventType, eventStatus, module, service, authToken, customId, searchDetail1, searchDetail2, additionalDetails);
83      }
84  
85      @Override
86      public void log(final EventType eventType, final EventStatus eventStatus, final ModuleType module, final ServiceType service, final String authToken, final String customId, final String searchDetail1, final String searchDetail2,
87              final Map<String, Object> additionalDetails) throws AuditRecordStorageException {
88          if (log.isTraceEnabled()) {
89              log.trace(String.format(">log:%s:%s:%s:%s:%s:%s:%s:%s:%s", eventType, eventStatus, module, service, authToken, customId, searchDetail1, searchDetail2, additionalDetails));
90          }
91          try {
92              final TrustedTime tt = trustedTimeWatcherSession.getTrustedTime(false);
93              internalSecurityEventsLoggerSession.log(tt, eventType, eventStatus, module, service, authToken, customId, searchDetail1, searchDetail2, additionalDetails);
94          } catch (TrustedTimeProviderException e) {
95              log.error(e.getMessage(), e);
96              throw new AuditRecordStorageException(e.getMessage(), e);
97          } finally {
98              if (log.isTraceEnabled()) {
99                  log.trace("<log");
100             }
101         }
102     }
103 }
