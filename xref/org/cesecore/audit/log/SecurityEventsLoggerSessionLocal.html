17  import javax.ejb.Local;
18  
19  import org.cesecore.audit.enums.EventStatus;
20  import org.cesecore.audit.enums.EventType;
21  import org.cesecore.audit.enums.ModuleType;

24  /**
25   * Local interface for the SecurityEventsLogger
26   * 
27   * @version $Id: SecurityEventsLoggerSessionLocal.java 17625 2013-09-20 07:12:06Z netmackan $
28   */
29  @Local
30  public interface SecurityEventsLoggerSessionLocal extends SecurityEventsLoggerSession {
31  	
32      /** Gets trusted time, then calls log
33       * 
34       * @param eventType The event log type.
35       * @param eventStatus The status of the operation to log.
36       * @param module The module where the operation took place.
37       * @param service The service(application) that performed the operation.
38       * @param authToken The authentication token that invoked the operation.
39       * @param customId A custom identifier related to this event (e.g. a CA's identifier)
40       * @param searchDetail1 A detail of this event that can be queried for using QueryCriteria (database) searches (e.g. a certificate serialnumber)
41       * @param searchDetail2 A detail of this event that can be queried for using QueryCriteria (database) searches (e.g. a username)
42       * @param additionalDetails Additional details to be logged.
43       * @throws AuditRecordStorageException On fail
44       */
45  	void log(EventType eventType, EventStatus eventStatus, ModuleType module, ServiceType service, String authToken, 
46  			String customId, String searchDetail1, String searchDetail2, Map<String, Object> additionalDetails) throws AuditRecordStorageException;
47  
48      /** Gets trusted time, then calls log
49       * 
50       * @param eventType The event log type.
51       * @param eventStatus The status of the operation to log.
52       * @param module The module where the operation took place.
53       * @param service The service(application) that performed the operation.
54       * @param authToken The authentication token that invoked the operation.
55       * @param customId A custom identifier related to this event (e.g. a CA's identifier)
56       * @param searchDetail1 A detail of this event that can be queried for using QueryCriteria (database) searches (e.g. a certificate serialnumber)
57       * @param searchDetail2 A detail of this event that can be queried for using QueryCriteria (database) searches (e.g. a username)
58       * @param additionalDetailsMsg A single additional details String to be logged.
59       * @throws AuditRecordStorageException On fail
60       */
61      void log(EventType eventType, EventStatus eventStatus, ModuleType module, ServiceType service, String authToken, 
62              String customId, String searchDetail1, String searchDetail2, String additionalDetailsMsg) throws AuditRecordStorageException;
63  
64  }
