13  package org.cesecore.keybind;
14  
15  import java.security.cert.X509Certificate;
16  import java.util.Collection;
17  import java.util.List;
18  import java.util.Map;
19  
20  import javax.ejb.Local;
21  
22  import org.cesecore.authentication.tokens.AuthenticationToken;
23  import org.cesecore.authorization.AuthorizationDeniedException;
24  import org.cesecore.certificates.ca.CADoesntExistsException;
25  
26  /**
27   * @see InternalKeyBindingMgmtSession
28   * @version $Id: InternalKeyBindingMgmtSessionLocal.java 28216 2018-02-08 12:19:01Z henriks $
29   */
30  @Local
31  public interface InternalKeyBindingMgmtSessionLocal extends InternalKeyBindingMgmtSession {
32         
33      /**
34       * Returns a list of all internal key bindings of a certain type, as {@link InternalKeyBindingInfo}s
35       * 
36       * @param internalKeyBindingType the key binding type
37       * @return a list of all internal key bindings of that type, as {@link InternalKeyBindingInfo}s
38       */
39      List<InternalKeyBindingInfo> getAllInternalKeyBindingInfos(String internalKeyBindingType);
40      
41      /**
42       * Internal (local only) method to get keybinding info without logging the authorization check
43       * (the auth check is performed though).
44       * @param authenticationToken Token
45       * @param internalKeyBindingId Binding
46       * @return Binding Info
47       * @throws AuthorizationDeniedException if access denied
48       * 
49       * @see #getInternalKeyBindingInfo
50       */
51      InternalKeyBindingInfo getInternalKeyBindingInfoNoLog(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException;
52  
53      /**
54       * Get a reference to a cached InternalKeyBinding object that MAY NOT be modified.
55       * 
56       * @param authenticationToken is the authentication token
57       * @param internalKeyBindingId is the identifier of the InternalKeyBinding
58       * @return the InternalKeyBinding for the requested Id or null if none was found
59       * @throws AuthorizationDeniedException if the authentication token was not authorized to fetch the requested InternalKeyBinding
60       */
61      InternalKeyBinding getInternalKeyBindingReference(AuthenticationToken authenticationToken, int internalKeyBindingId) throws AuthorizationDeniedException;
62  
63      /**
64       * Returns a collection of the trusted certificates defined in internalKeyBinding along with their issuers' certificate chains.
65       * 
66       * - If the list of trusted certificates in internalKeyBinding contains certificate serial number(s), only the certificates with these specific serial numbers
67       *   are trusted
68       * - If only trusted CAs are specified in internalKeyBinding, all certificates issued by the specified CAs will be trusted.
69       * - If the list of trusted certificates in internalKeyBinding is empty, all certificates issued by all CAs known to this instance of EJBCA are trusted
70       * - If the list of trusted certificates in internalKeyBinding is null, no certificates will be trusted.
71       *
72       * 
73       * @param internalKeyBinding Binding
74       * @return a collection of the trusted certificates along with their issuers' certificate chains or null if no trusted certificates or CAs are specified
75       * @throws CADoesntExistsException IF CA not found
76       */
77      List< Collection<X509Certificate> > getListOfTrustedCertificates(InternalKeyBinding internalKeyBinding) throws CADoesntExistsException;
78  
79      /**
80       * Returns a id-to-name map of all internal key bindings of the given type 
81       * 
82       * @param internalKeyBindingType null for all types
83       * @return id-name map of internal key bindings
84       */
85      Map<Integer, String> getAllInternalKeyBindingIdNameMap(String internalKeyBindingType);
86  }
