147      * Imports CustomCertExtensions from certextensions.properties into the database.
148      * 
149      * TODO Remove this method when support for EJBCA 6.4.0 is dropped.
150      */
151     @Deprecated
152     private void addAvailableCustomCertExtensionsFromFile() {
153         // If the file has already been removed, no need to go further
154         InputStream is = CertificateExtensionFactory.class.getResourceAsStream("/certextensions.properties");
155         if(is == null) {
156             return;
157         }
158         
159         try{
160             Properties props = new Properties();
161             try {
162                 props.load(is);
163             } finally {
164                 is.close();
165             }
166             
167             int count = 0;
168             for(int i=1;i<255;i++){
169                 if(props.get("id" + i +".oid")!=null){
170                     if(log.isDebugEnabled()) {
171                         log.debug("found " + props.get("id" + i +".oid"));
172                     }
173                     CertificateExtension ce = getCertificateExtensionFromFile(i, props);
174                     addCustomCertExtension(ce);
175                     count++;
176                 }
177             }
178             if(log.isDebugEnabled()) {
179                 log.debug("Nr of read Custom Certificate Extensions from file: " + count);
180             }
181         }catch(IOException e){
182             log.error("Error parsing the 'certextensions.properties' file.",e);
183         } catch (CertificateExtentionConfigurationException e) {
184             log.error(e.getMessage(),e);
185         }
186     }

188     /**
189      * Used for upgrading from old certextensions.properties files.
190      * Package-internal to allow for testing.
191      * @param id ID
192      * @param propertiesInFile Properties
193      * @return Certificate extension
194      * @throws CertificateExtentionConfigurationException on fail 
195      */
196     static CertificateExtension getCertificateExtensionFromFile(int id, Properties propertiesInFile) throws CertificateExtentionConfigurationException {
197         String PROPERTY_ID           = "id";
198         String PROPERTY_OID          = ".oid";
199         String PROPERTY_CLASSPATH    = ".classpath";
200         String PROPERTY_DISPLAYNAME  = ".displayname";
201         String PROPERTY_USED         = ".used";
202         String PROPERTY_TRANSLATABLE = ".translatable";
203         String PROPERTY_CRITICAL     = ".critical";
204         
205         try{
206             String oid = StringUtils.trim(propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_OID));
207             String classPath = StringUtils.trim(propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_CLASSPATH));
208             String displayName = StringUtils.trim(propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_DISPLAYNAME));
209             log.debug(PROPERTY_ID + id + PROPERTY_USED + ":" + propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_USED));
210             boolean used = propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_USED).trim().equalsIgnoreCase("TRUE");
211             boolean translatable = propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_TRANSLATABLE).trim().equalsIgnoreCase("TRUE");
212             boolean critical = propertiesInFile.getProperty(PROPERTY_ID + id + PROPERTY_CRITICAL).trim().equalsIgnoreCase("TRUE");
213             log.debug(id + ", " + used + ", " +oid + ", " +critical+ ", " +translatable +  ", " + displayName);   
214             if(used){
215                 if(oid != null && classPath != null && displayName != null){
216                     Class<?> implClass = Class.forName(classPath);
217                     CertificateExtensioncertificates/certificate/certextensions/CertificateExtension.html#CertificateExtension">CertificateExtension certificateExtension = (CertificateExtension) implClass.getConstructor().newInstance();
218                     Properties extensionProperties = getExtensionProperties(id, propertiesInFile);
219                     if(translatable) {
220                         extensionProperties.put("translatable", true);
221                     }
222                     certificateExtension.init(id, oid.trim(), displayName, critical, /*required=*/true, extensionProperties);
223                     return certificateExtension;
224 
225                 }else{
226                     throw new CertificateExtentionConfigurationException("Certificate Extension " + Integer.valueOf(id) + " seems to be misconfigured in the certextensions.properties");
227                 }
228             }
229             
230         }catch(Exception e){
231             throw new CertificateExtentionConfigurationException("Certificate Extension " + Integer.valueOf(id) + " seems to be misconfigured in the certextensions.properties",e);
232         }
233         return null;
234     }

236     private static Properties getExtensionProperties(int id, Properties propertiesInFile) {
237         Properties extProps = new Properties();
238         Iterator<Object> keyIter = propertiesInFile.keySet().iterator();
239         String matchString = "id" + id + ".property."; 
240         while(keyIter.hasNext()){
241             String nextKey = (String) keyIter.next();
242             if(nextKey.startsWith(matchString)){
243                 if(nextKey.length() > matchString.length()){
244                   extProps.put(nextKey.substring(matchString.length()), propertiesInFile.get(nextKey));               
245                 }
246             }           
247         }
248         return extProps;
249     }
