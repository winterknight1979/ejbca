45  /**
46   * The default basic certificate extension that has two property.
47   * 
48   * 'value' : The value returned 'encoding' : How the value is encoded.
49   * 
50   * Optionally, a new property can be defined:
51   * 
52   * 'nvalues' : number of values of type 'encoding'
53   * 
54   * Thus, the extension will be of type 'SEQUENCE OF ENCODING' with a size of nvalues. The members will be: 'value1', 'value2' and so on.
55   * 
56   * Optionally, an other property can be defined:
57   * 
58   *  'dynamic' : true/false if the extension value(s) should be allowed to be 
59   *              overridden by value(s) put as extensiondata in 
60   *              ExtendedInformation. Default is 'false'.
61   *
62   *
63   * See documentation for more information.
64   * 
65   * @version $Id: BasicCertificateExtension.java 30583 2018-11-22 17:32:11Z samuellb $
66   */
67  public class BasicCertificateExtension extends CertificateExtension implements CustomCertificateExtension {
68

78      private enum Encoding {
79          ENCODING_DERBITSTRING("DERBITSTRING"),
80          ENCODING_DERINTEGER("DERINTEGER"),
81          ENCODING_DEROCTETSTRING("DEROCTETSTRING"),
82          ENCODING_DERBOOLEAN("DERBOOLEAN"),
83          ENCODING_DERPRINTABLESTRING("DERPRINTABLESTRING"),
84          ENCODING_DERUTF8STRING("DERUTF8STRING"),
85          ENCODING_DERIA5STRING("DERIA5STRING"),
86          ENCODING_DERNULL("DERNULL"),
87          ENCODING_DEROBJECT("DEROBJECT"),
88          ENCODING_DEROID("DERBOJECTIDENTIFIER");
89          
90          private static final Map<String, Encoding> lookupMap = new HashMap<String, Encoding>();
91          
92          static {
93              for(Encoding encoding : Encoding.values()) {
94                  lookupMap.put(encoding.value(), encoding);
95              }
96          }
97          
98          private final String value;
99  
100         
101         private Encoding(String value) {
102             this.value = value;
103         }
104         
105         public String value() {
106             return value;
107         }
108         
109         public boolean equals(Encoding otherValue) {
110             if(otherValue == null) {
111                 return false;
112             }
113             return value.equalsIgnoreCase(otherValue.value());
114         }
115         
116         public static final Encoding fromString(String value) {
117             return lookupMap.get(StringUtils.upperCase(value, Locale.ROOT));
118         }

128     private static String ENCODING_RAW = "RAW";
129     private static String ENCODING_DERNULL = "DERNULL";

131     // Defined Properties
132     private static String PROPERTY_VALUE = "value";
133     private static String PROPERTY_ENCODING = "encoding";
134     private static String PROPERTY_NVALUES = "nvalues";
135     private static String PROPERTY_DYNAMIC  = "dynamic";
136     
137     private static final Map<String, String[]> propertiesMap = new HashMap<String, String[]>();
138     
139     static {
140         Encoding[] encodings = Encoding.values();
141         // +1 because we need to add RAW as well in the end
142         String[] encodingValues = new String[encodings.length+1];
143         for(int i = 0; i < encodings.length; i++) {
144             encodingValues[i] = encodings[i].value;
145         }
146         // Add RAW last
147         encodingValues[encodingValues.length-1] = ENCODING_RAW;
148         
149         propertiesMap.put(PROPERTY_ENCODING, encodingValues);
150         propertiesMap.put(PROPERTY_VALUE, new String[]{});
151         propertiesMap.put(PROPERTY_DYNAMIC, CustomCertificateExtension.BOOLEAN);
152     }

155         setDisplayName(DISPLAY_NAME);

166     /**
167      * Returns the defined property 'value' in the encoding specified in 'encoding'.
168      * 
169      * This certificate extension implementations overrides this method as it 
170      * want to be able to return a byte[] with the extension value. Otherwise 
171      * the implementation could have been put in the getValue method as the 
172      * super class CertificateExtension has a default implementation for 
173      * getValueEncoded which calls getValue.
174      * 
175      * @param userData
176      *            Used to lookup extension data
177      * @param ca
178      *            not used
179      * @param certProfile
180      *            not used
181      * @see org.cesecore.certificates.certificate.certextensions.CertificateExtension#getValueEncoded(EndEntityInformation, CA, CertificateProfile, PublicKey,
182      *      PublicKey, CertificateValidity)
183      */
184     @Override
185     public byte[] getValueEncoded(EndEntityInformation userData, CA ca, CertificateProfile certProfile, PublicKey userPublicKey,
186             PublicKey caPublicKey, CertificateValidity val) throws CertificateExtensionException {
187         String[] values = getValues(userData, null);
188         return handleValues(values);
189     }

191     @Override
192     public byte[] getValueEncoded(EndEntityInformation userData, CA ca, CertificateProfile certProfile, PublicKey userPublicKey,
193             PublicKey caPublicKey, CertificateValidity val, String oid) throws CertificateExtensionException {
194         String[] values = getValues(userData, oid);
195         return handleValues(values);
196     }

198     private byte[] handleValues(String[] values) throws CertificateExtensionException {
199         final byte[] result;
200         String encoding = StringUtils.trim(getProperties().getProperty(PROPERTY_ENCODING));
201         if (values == null || values.length == 0 || (values[0] == null && !encoding.equalsIgnoreCase(ENCODING_DERNULL))) {
202             if (!isRequiredFlag()) {
203                 return null;
204             }
205             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.incorrectvalue", Integer.valueOf(getId()), getOID()));
206         }
207 
208         if (encoding.equalsIgnoreCase(ENCODING_RAW)) {
209             if (values.length > 1) {
210                 // nvalues can not be used together with encoding=RAW
211                 throw new CertificateExtensionException(intres.getLocalizedMessage("certext.certextmissconfigured", Integer.valueOf(getId())));
212             } else {
213                 result = parseRaw(values[0]);
214             }
215         } else {
216             try {
217                 if (values.length > 1) {
218                     ASN1EncodableVector ev = new ASN1EncodableVector();
219                     for (String value : values) {
220                         ASN1Encodable derval = parseValue(encoding, value);
221                         ev.add(derval);
222                     }
223                     result = new DERSequence(ev).getEncoded();
224                 } else {
225                     result = parseValue(encoding, values[0]).toASN1Primitive().getEncoded();
226                 }
227             } catch (IOException ioe) {
228                 throw new CertificateExtensionException(ioe.getMessage(), ioe);
229             }
230         }
231         return result;
232     }

234     /**
235      * Get the extension value by first looking in the ExtendedInformation (if 
236      * dynamic is enabled) and then in the static configuration.
237      * 
238      * @param userData The userdata to get the ExtendedInformation from
239      * @param oid OID
240      * @return The value(s) for the extension (usually 1) or null if no value found
241      */
242     private String[] getValues(EndEntityInformation userData, final String oid) {
243         String[] result = null;
244 
245         boolean dynamic = Boolean.parseBoolean(StringUtils.trim(getProperties().getProperty(PROPERTY_DYNAMIC, Boolean.FALSE.toString())));
246 
247         String strnvalues = getProperties().getProperty(PROPERTY_NVALUES);
248         int nvalues;
249 
250         if ( strnvalues == null || strnvalues.trim().equals("") ) {
251             nvalues = 0;
252         } else {
253             nvalues = Integer.parseInt(strnvalues);
254         }
255 
256         if (dynamic) {
257             final ExtendedInformation ei = userData.getExtendedInformation();
258             if (ei == null) {
259                 result = null;
260             } else {
261                 if (nvalues < 1 ) {
262                     String value = null;
263                     if (oid != null) {
264                         value = userData.getExtendedInformation().getExtensionData(oid);
265                         if (value == null || value.trim().isEmpty()) {
266                             value = userData.getExtendedInformation().getExtensionData(oid + "." + PROPERTY_VALUE);
267                         }
268                     } else {
269                         value = userData.getExtendedInformation().getExtensionData(getOID());
270                         if (value == null || value.trim().isEmpty()) {
271                             value = userData.getExtendedInformation().getExtensionData(getOID() + "." + PROPERTY_VALUE);
272                         }
273                     }
274                     if (value == null) {
275                         result = null;
276                     } else {
277                         result = new String[] { value };
278                     }
279                 } else {
280                     for (int i = 1; i <= nvalues; i++) {
281                         String value = null;
282                         if (oid != null) {
283                             value = userData.getExtendedInformation().getExtensionData(oid + "." + PROPERTY_VALUE + Integer.toString(i));
284                         } else {
285                             value = userData.getExtendedInformation().getExtensionData(getOID() + "." + PROPERTY_VALUE + Integer.toString(i));
286                         }
287                         if (value != null) {
288                             if (result == null) {
289                                 result = new String[nvalues];
290                             }
291                             result[i - 1] = value;
292                         }
293                     }
294                 }
295             }
296         }
297         if (result == null) {
298             if (nvalues < 1 ) {
299                 String value = getProperties().getProperty(PROPERTY_VALUE);
300                 if ( value == null || value.trim().equals("") ) {
301                     value = getProperties().getProperty(PROPERTY_VALUE+"1");
302                 }
303                 result = new String[] { value };
304             } else {
305                 result = new String[nvalues];
306                 for (int i=1; i<=nvalues; i++) {
307                     result[i - 1] = getProperties().getProperty(PROPERTY_VALUE+Integer.toString(i));
308                 }
309             }
310         }
311         return result;
312     }

314     private ASN1Encodable parseValue(String encoding, String value) throws CertificateExtensionException {
315 
316         ASN1Encodable toret = null;
317         
318         Encoding encodingType = Encoding.fromString(encoding);
319         
320         if(encodingType == null) {
321             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.incorrectenc", encoding,
322                     Integer.valueOf(getId())));
323         }
324 
325         if (!Encoding.ENCODING_DERNULL.equals(encodingType) && ((value == null || value.trim().equals("")) && isRequiredFlag())) {
326             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.incorrectvalue", Integer.valueOf(getId()), getOID()));
327         }
328 
329         switch(encodingType) { 
330         case ENCODING_DERBITSTRING:
331             toret = parseDERBitString(value);
332             break;
333         case ENCODING_DERINTEGER:
334             toret = parseDERInteger(value);
335             break;
336         case ENCODING_DEROCTETSTRING:
337             toret = parseDEROctetString(value);
338             break;
339         case ENCODING_DERBOOLEAN:
340             toret = parseDERBoolean(value);
341             break;
342         case ENCODING_DEROID:
343             toret = parseDEROID(value);
344             break;
345         case ENCODING_DERPRINTABLESTRING:
346             toret = parseDERPrintableString(value);
347             break;
348         case ENCODING_DERUTF8STRING:
349             toret = parseDERUTF8String(value);
350             break;
351         case ENCODING_DERIA5STRING:
352             toret = parseDERIA5String(value);
353             break;
354         case ENCODING_DERNULL:
355             toret = DERNull.INSTANCE;
356             break;
357         case ENCODING_DEROBJECT:
358             toret = parseHexEncodedDERObject(value);
359             break;
360         default:
361             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.incorrectenc", encoding,
362                     Integer.valueOf(getId())));
363         }
364         return toret;
365     }

367     private ASN1Encodable parseDERBitString(String value) throws CertificateExtensionException {
368         ASN1Encodable retval = null;
369         try {
370             BigInteger bigInteger = new BigInteger(value, 2);
371             int padBits = value.length() - 1 - value.lastIndexOf("1");
372             if (padBits == 8) {
373                 padBits = 0;
374             }
375             byte[] byteArray = bigInteger.toByteArray();
376             if (byteArray[0] == 0) {
377                 // Remove empty extra byte
378                 // System.arraycopy handles creating of temporary array when destinatio is the same
379                 System.arraycopy(byteArray, 1, byteArray, 0, byteArray.length-1);
380             }
381             retval = new DERBitString(byteArray, padBits);
382         } catch (NumberFormatException e) {
383             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
384                     Integer.valueOf(getId()), getOID()));
385         }
386 
387         return retval;
388     }

391         ASN1Encodable retval = null;

396                     Integer.valueOf(getId()), getOID()));

402     private ASN1Encodable parseDERInteger(String value) throws CertificateExtensionException {
403         ASN1Encodable retval = null;
404         try {
405             BigInteger intValue = new BigInteger(value, 10);
406             retval = new ASN1Integer(intValue);
407         } catch (NumberFormatException e) {
408             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
409                     Integer.valueOf(getId()), getOID()));
410         }
411 
412         return retval;
413     }

415     private ASN1Encodable parseDEROctetString(String value) throws CertificateExtensionException {
416         ASN1Encodable retval = null;
417         if (value.matches("^\\p{XDigit}*")) {
418             byte[] bytes = Hex.decode(value);
419             retval = new DEROctetString(bytes);
420         } else {
421             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
422                     Integer.valueOf(getId()), getOID()));
423         }
424         return retval;
425     }

427     /**
428      * Tries to read the hex-string as an DERObject. If it contains more than one ASN1Encodable object, return a DERSequence of the objects.
429      * @param value value
430      * @return DER sequence
431      * @throws CertificateExtensionException on fail
432      */
433     private ASN1Encodable parseHexEncodedDERObject(String value) throws CertificateExtensionException {
434         ASN1Encodable retval = null;
435         if (value.matches("^\\p{XDigit}*")) {
436             byte[] bytes = Hex.decode(value);
437             try {
438                 ASN1InputStream ais = new ASN1InputStream(bytes);
439                 ASN1Encodable firstObject = ais.readObject();
440                 if (ais.available() > 0) {
441                     ASN1EncodableVector ev = new ASN1EncodableVector();
442                     ev.add(firstObject);
443                     while (ais.available() > 0) {
444                         ev.add(ais.readObject());
445                     }
446                     retval = new DERSequence(ev);
447                 } else {
448                     retval = firstObject;
449                 }
450                 ais.close();
451             } catch (Exception e) {
452                 throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
453                         Integer.valueOf(getId()), getOID()));
454             }
455         } else {
456             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
457                     Integer.valueOf(getId()), getOID()));
458         }
459         return retval;
460     }

462     private ASN1Encodable parseDERBoolean(String value) throws CertificateExtensionException {
463         ASN1Encodable retval = null;
464         if (value.equalsIgnoreCase("TRUE")) {
465             retval = ASN1Boolean.TRUE;
466         }
467 
468         if (value.equalsIgnoreCase("FALSE")) {
469             retval = ASN1Boolean.FALSE;
470         }
471 
472         if (retval == null) {
473             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.illegalvalue", value,
474                     Integer.valueOf(getId()), getOID()));
475         }
476 
477         return retval;
478     }

502     private byte[] parseRaw(String value) throws CertificateExtensionException {
503         if(value == null) {
504             throw new CertificateExtensionException(intres.getLocalizedMessage("certext.basic.incorrectvalue", Integer.valueOf(getId()), getOID()));
505         }
506         return Hex.decode(value);
507     }
