13  package org.cesecore.certificates.certificate.certextensions.standard;
14  
15  import java.security.PublicKey;
16  import java.util.ArrayList;
17  
18  import org.apache.log4j.Logger;
19  import org.bouncycastle.asn1.ASN1Encodable;
20  import org.bouncycastle.asn1.ASN1EncodableVector;
21  import org.bouncycastle.asn1.ASN1Integer;
22  import org.bouncycastle.asn1.ASN1Object;
23  import org.bouncycastle.asn1.DERPrintableString;
24  import org.bouncycastle.asn1.DERSequence;
25  import org.bouncycastle.asn1.DERSet;
26  import org.bouncycastle.asn1.util.ASN1Dump;
27  import org.cesecore.certificates.ca.CA;
28  import org.cesecore.certificates.ca.internal.CertificateValidity;
29  import org.cesecore.certificates.certificateprofile.CertificateProfile;
30  import org.cesecore.certificates.endentity.EndEntityInformation;

32  /**
33   * Class for ICAO X509 certificate extension. 
34   * See ICAO MRTD Technical Report LDS and PKI Maintenance 1.0 or later for spec of this extension.
35   * 
36   * documentTypeList  EXTENSION  ::=  {
37   *     SYNTAX DocumentTypeListSyntax
38   *     IDENTIFIED BY id-icao-mrtd-security-extensions-documentTypeList}
39   * DocumentTypeListSyntax ::= SEQUENCE {
40   *   version DocumentTypeListVersion,
41   *   docTypeList SET OF DocumentType }
42   *
43   * DocumentTypeListVersion ::= INTEGER {v0(0)}
44   * 
45   * -- Document Type as contained in MRZ, e.g. "P" or "ID" where a
46   * -- single letter denotes all document types starting with that letter
47   * DocumentType ::= PrintableString(1..2) 
48   * 
49   * @version $Id: DocumentTypeList.java 24408 2016-09-23 06:56:00Z anatom $
50   */
51  public class DocumentTypeList extends StandardCertificateExtension {
52      private static final long serialVersionUID = 1L;
53      private static final Logger log = Logger.getLogger(DocumentTypeList.class);
54  
55      @Override
56      public void init(final CertificateProfile certProf) {
57          super.setOID("2.23.136.1.1.6.2");
58          super.setCriticalFlag(certProf.getDocumentTypeListCritical());
59      }
60      
61      @Override
62      public ASN1Encodable getValue(final EndEntityInformation subject, final CA ca, final CertificateProfile certProfile,
63              final PublicKey userPublicKey, final PublicKey caPublicKey, CertificateValidity val) {
64          
65          ArrayList<String> docTypes = certProfile.getDocumentTypeList();
66          if(docTypes.size() == 0) {
67              if (log.isDebugEnabled()) {
68                  log.debug("No DocumentTypeList to make a certificate extension");
69              }
70              return null;
71          }
72          
73          ASN1EncodableVector vec = new ASN1EncodableVector();
74  
75          // version
76          vec.add(new ASN1Integer(0));
77          
78          // Add SET OF DocumentType
79          final ASN1Encodable[] dts = new ASN1Encodable[docTypes.size()];
80          int i = 0;
81          for (final String type : docTypes) {
82              dts[i++] = new DERPrintableString(type);
83          }
84          vec.add(new DERSet(dts)); // the DERSet constructor performs the DER normalization (i.e., it sorts the set)
85          
86          ASN1Object gn = new DERSequence(vec);
87          if(log.isTraceEnabled()) {
88              log.trace("Constructed DocumentTypeList: "+ASN1Dump.dumpAsString(gn, true));
89          }
90          
91          return gn;
92      }   
93  }
