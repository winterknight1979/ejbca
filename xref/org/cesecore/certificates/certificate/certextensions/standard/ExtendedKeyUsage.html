13  package org.cesecore.certificates.certificate.certextensions.standard;
14  
15  import java.security.PublicKey;
16  import java.util.Collection;
17  
18  import org.apache.log4j.Logger;
19  import org.bouncycastle.asn1.ASN1Encodable;
20  import org.bouncycastle.asn1.ASN1ObjectIdentifier;
21  import org.bouncycastle.asn1.ASN1Sequence;
22  import org.bouncycastle.asn1.DERSequence;
23  import org.bouncycastle.asn1.x509.Extension;
24  import org.cesecore.certificates.ca.CA;
25  import org.cesecore.certificates.ca.internal.CertificateValidity;
26  import org.cesecore.certificates.certificate.certextensions.CertificateExtensionException;
27  import org.cesecore.certificates.certificateprofile.CertificateProfile;
28  import org.cesecore.certificates.endentity.EndEntityInformation;
29  
30  /**
31   * Class for standard X509 certificate extension. 
32   * See rfc3280 or later for spec of this extension.
33   * 
34   * @version $Id: ExtendedKeyUsage.java 27639 2017-12-22 08:51:09Z anatom $
35   */
36  public class ExtendedKeyUsage extends StandardCertificateExtension {
37      private static final long serialVersionUID = 1L;
38      private static final Logger log = Logger.getLogger(ExtendedKeyUsage.class);
39  
40      @Override
41  	public void init(final CertificateProfile certProf) {
42  		super.setOID(Extension.extendedKeyUsage.getId());
43          // Extended Key Usage may be either critical or non-critical
44  		super.setCriticalFlag(certProf.getExtendedKeyUsageCritical());
45  	}
46      
47      @Override
48      public ASN1Encodable getValue(final EndEntityInformation subject, final CA ca, final CertificateProfile certProfile,
49              final PublicKey userPublicKey, final PublicKey caPublicKey, CertificateValidity val) throws CertificateExtensionException {
50  		org.bouncycastle.asn1.x509.ExtendedKeyUsage ret = null;
51  		// Get extended key usage from certificate profile
52  		final Collection<String> oids = certProfile.getExtendedKeyUsageOids();
53  		
54  		// Don't add empty key usage extension
55  		if (oids.size() != 0) {
56  			final ASN1Encodable[] usage = new ASN1Encodable[oids.size()];
57  			int i = 0;
58  			for (String oid : oids) {
59  				usage[i] = org.bouncycastle.asn1.x509.KeyPurposeId.getInstance(new ASN1ObjectIdentifier(oid));
60  				i++;
61  			}
62  			
63  			ASN1Sequence seq = ASN1Sequence.getInstance(new DERSequence(usage));
64  			ret = org.bouncycastle.asn1.x509.ExtendedKeyUsage.getInstance(seq);
65  		}
66  		if (ret == null) {
67  			log.error("ExtendedKeyUsage missconfigured, ExtendedKeyUsage is configured to be used but no OIDs are defined in certificate profile");
68  		}
69  		return ret;
70  	}	
71  }
