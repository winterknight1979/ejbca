34  /**
35   * Interface for creating certificates
36   *
37   * Only one method from this bean is used.
38   * 
39   * @version $Id: CertificateCreateSession.java 25676 2017-04-10 10:56:35Z mikekushner $
40   */
41  public interface CertificateCreateSession {
42  
43      /**
44       * Requests for a certificate to be created for the passed public key wrapped in a
45       * certification request message (ex PKCS10).  The username and password used to authorize is
46       * taken from the request message. Verification of the signature (proof-of-possesion) on the
47       * request is performed, and an exception thrown if verification fails. The method queries the
48       * user database for authorization of the user.
49       *
50       * @param admin         Information about the administrator or admin performing the event.
51       * @param endEntityInformation Supplied user data, containing the issuing CAid, subject DN etc. Must contain the following information
52       * 						type, username, certificateProfileId. Optionally it contains:
53       * 						subjectDN, required if certificateProfile does not allow subject DN override
54       * 						caid, if not possible to get it from issuerDN of the request
55       * 						extendedInformation
56       * @param req           a Certification Request message, containing the public key to be put in the
57       *                      created certificate. Currently no additional parameters in requests are considered!
58       * @param responseClass The implementation class that will be used as the response message.
59       * @param certGenParams Parameters for certificate generation (e.g for the CT extension), or null.
60       * @return The newly created response or null.
61       * 
62       * @throws AuthorizationDeniedException (rollback) if admin is not authorized to issue this certificate
63  	 * @throws CustomCertificateSerialNumberException (no rollback) if custom serial number is registered for user, but it is not allowed to be used (either missing unique index in database, 
64  	 *     or certificate profile does not allow it
65       * @throws IllegalKeyException (no rollback) if the passed in PublicKey does not fulfill requirements in CertificateProfile
66       * @throws CertificateCreateException (rollback) if another error occurs
67       * @throws CADoesntExistsException (no rollback) if CA to issue certificate does not exist
68       * @throws CertificateSerialNumberException if certificate with same subject DN or key already exists for a user, if these limitations are enabled in CA.
69       * @throws CertificateRevokeException if certificate was meant to be issued revoked, but could not. Causes rollback. 
70       * @throws IllegalNameException if the certificate request contained an illegal name 
71       * @throws SignRequestSignatureException (no rollback) if POPO verification on the request fails 
72       * @throws CryptoTokenOfflineException (no rollback) if token in the CA was unavailable. 
73       * @throws CertificateExtensionException if any if the extensions were invalid
74       * @throws IllegalValidityException if the validity defined by notBefore and notAfter was invalid
75       * @throws CAOfflineException if the CA was offline
76       * @throws InvalidAlgorithmException if the signing algorithm in the certificate profile (or the CA Token if not found) was invalid.
77       * 
78       */
79      CertificateResponseMessage createCertificate(AuthenticationToken admin, EndEntityInformation endEntityInformation, RequestMessage req,
80              Class<? extends ResponseMessage> responseClass, CertificateGenerationParams certGenParams) throws AuthorizationDeniedException, CustomCertificateSerialNumberException,
81              IllegalKeyException, CADoesntExistsException, CertificateCreateException, CryptoTokenOfflineException, SignRequestSignatureException,
82              IllegalNameException, CertificateRevokeException, CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException, CertificateExtensionException;
83  
84  }
