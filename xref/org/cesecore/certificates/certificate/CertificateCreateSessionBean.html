140     @Override
141     public CertificateResponseMessage createCertificate(final AuthenticationToken admin, final EndEntityInformation endEntityInformation, final CA ca,
142             final RequestMessage requestMessage, final Class<? extends ResponseMessage> responseClass, CertificateGenerationParams certGenParams,
143             final long updateTime) throws CryptoTokenOfflineException, SignRequestSignatureException, IllegalKeyException, IllegalNameException,
144             CustomCertificateSerialNumberException, CertificateCreateException, CertificateRevokeException, CertificateSerialNumberException,
145             AuthorizationDeniedException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException, CertificateExtensionException {
146         if (log.isTraceEnabled()) {
147             log.trace(">createCertificate(IRequestMessage, CA)");
148         }
149         CertificateResponseMessage ret = null;
150         try {
151             final CAToken catoken = ca.getCAToken();
152             final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(catoken.getCryptoTokenId());
153             final String alias;
154             final Collection<Certificate> cachain;
155             final Certificate cacert;
156             if (ca.getUseNextCACert(requestMessage)) {
157                 alias = catoken.getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT);
158                 cachain = ca.getRolloverCertificateChain();
159                 cacert = cachain.iterator().next();
160             } else {
161                 alias = catoken.getAliasFromPurpose(CATokenConstants.CAKEYPURPOSE_CERTSIGN);
162                 cachain = ca.getCertificateChain();
163                 cacert = ca.getCACertificate();
164             }
165             // See if we need some key material to decrypt request
166             if (requestMessage.requireKeyInfo()) {
167                 // You go figure...scep encrypts message with the public CA-cert
168                 requestMessage.setKeyInfo(cacert, cryptoToken.getPrivateKey(alias), cryptoToken.getEncProviderName());
169             }
170             // Verify the request
171             final PublicKey reqpk;
172             try {
173                 if (!requestMessage.verify()) {
174                     throw new SignRequestSignatureException(intres.getLocalizedMessage("createcert.popverificationfailed"));
175                 }
176                 reqpk = requestMessage.getRequestPublicKey();
177                 if (reqpk == null) {
178                     final String msg = intres.getLocalizedMessage("createcert.nokeyinrequest");
179                     throw new InvalidKeyException(msg);
180                 }
181             } catch (InvalidKeyException e) {
182                 // If we get an invalid key exception here, we should throw an IllegalKeyException to the caller
183                 // The catch of InvalidKeyException in the end of this method, catches error from the CA crypto token
184                 throw new IllegalKeyException(e);
185             }
186 
187             final Date notBefore = requestMessage.getRequestValidityNotBefore(); // Optionally requested validity
188             final Date notAfter = requestMessage.getRequestValidityNotAfter(); // Optionally requested validity
189             final Extensions exts = requestMessage.getRequestExtensions(); // Optionally requested extensions
190             int keyusage = -1;
191             if (exts != null) {
192                 if (log.isDebugEnabled()) {
193                     log.debug("we have extensions, see if we can override KeyUsage by looking for a KeyUsage extension in request");
194                 }
195                 final KeyUsage keyUsage = KeyUsage.fromExtensions(exts);
196                 if (keyUsage!=null) {
197                     final DERBitString bitString = (DERBitString) keyUsage.toASN1Primitive();
198                     keyusage = bitString.intValue();
199                     if (log.isDebugEnabled()) {
200                         log.debug("We have a key usage request extension: " + keyusage);
201                     }
202                 }
203             }
204             String sequence = null;
205             byte[] ki = requestMessage.getRequestKeyInfo();
206             // CVC sequence is only 5 characters, don't fill with a lot of garbage here, it must be a readable string
207             if ((ki != null) && (ki.length > 0) && (ki.length < 10) ) {
208                 final String str = new String(ki);
209                 // A cvc sequence must be ascii printable, otherwise it's some binary data
210                 if (StringUtils.isAsciiPrintable(str)) {
211                     sequence = new String(ki);                  
212                 }
213             }
214             
215             CertificateDataWrapper certWrapper = createCertificate(admin, endEntityInformation, ca, requestMessage, reqpk, keyusage, notBefore, notAfter, exts, sequence, certGenParams, updateTime);
216             // Create the response message with all nonces and checks etc            
217             ret = ResponseMessageUtils.createResponseMessage(responseClass, requestMessage, cachain, cryptoToken.getPrivateKey(alias), cryptoToken.getEncProviderName());
218             ResponseStatus status = ResponseStatus.SUCCESS;
219             FailInfo failInfo = null;
220             String failText = null;
221             if ((certWrapper == null) && (status == ResponseStatus.SUCCESS)) {
222                 status = ResponseStatus.FAILURE;
223                 failInfo = FailInfo.BAD_REQUEST;
224             } else {
225                 ret.setCertificate(certWrapper.getCertificate());
226                 ret.setCACert(cacert);
227                 // Add in case of success after CMP message -> CmpResponseMessage.
228                 ret.addAdditionalCaCertificates(requestMessage.getAdditionalCaCertificates());
229                 ret.setBase64CertData(certWrapper.getBase64CertData());
230                 ret.setCertificateData(certWrapper.getCertificateData());
231             }
232             // Add in all cases -> PKI message.
233             ret.addAdditionalResponseExtraCertsCertificates(requestMessage.getAdditionalExtraCertsCertificates());
234             ret.setStatus(status);
235             if (failInfo != null) {
236                 ret.setFailInfo(failInfo);
237                 ret.setFailText(failText);
238             }
239             ret.create();          
240         } catch (InvalidKeyException e) {
241             throw new CertificateCreateException(ErrorCode.INVALID_KEY, e);
242         } catch (NoSuchAlgorithmException e) {
243             throw new CertificateCreateException(ErrorCode.BAD_REQUEST_SIGNATURE, e);
244         } catch (NoSuchProviderException e) {
245             throw new CertificateCreateException(ErrorCode.INTERNAL_ERROR, e);
246         } catch(CertificateEncodingException e) {
247             throw new CertificateCreateException(ErrorCode.CERT_COULD_NOT_BE_PARSED, e);
248         } catch (CRLException e) {
249             throw new CertificateCreateException(ErrorCode.CERT_COULD_NOT_BE_PARSED, e);
250         } 
251 
252         if (log.isTraceEnabled()) {
253             log.trace("<createCertificate(IRequestMessage, CA)");
254         }
255         return ret;
256     }

268     @Override
269     public CertificateResponseMessage createCertificate(final AuthenticationToken admin, final EndEntityInformation userData,
270             final RequestMessage req, final Class<? extends ResponseMessage> responseClass, CertificateGenerationParams certGenParams, final long updateTime) throws CADoesntExistsException,
271             AuthorizationDeniedException, CryptoTokenOfflineException, SignRequestSignatureException, IllegalKeyException, IllegalNameException,
272             CustomCertificateSerialNumberException, CertificateCreateException, CertificateRevokeException, CertificateSerialNumberException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException, CertificateExtensionException {
273         if (log.isTraceEnabled()) {
274             log.trace(">createCertificate(IRequestMessage)");
275         }
276         final CA ca;
277         // First find the CA, this checks authorization and that the CA exists
278         if ((userData == null) || (userData.getCAId() == 0)) {
279             // If no CAid in the supplied userdata
280             ca = getCAFromRequest(admin, req);
281         } else {
282             ca = caSession.getCA(admin, userData.getCAId());
283         }
284 
285         if (log.isTraceEnabled()) {
286             log.trace("<createCertificate(IRequestMessage)");
287         }
288         return createCertificate(admin, userData, ca, req, responseClass, certGenParams, updateTime);
289     }

291     /**
292      * Help Method that extracts the CA specified in the request.
293      * @param admin Auth token
294      * @param req Request
295      * @return CA
296      * 
297      * @throws AuthorizationDeniedException If access denied
298      * @throws CADoesntExistsException On fail
299      */
300     private CA getCAFromRequest(final AuthenticationToken admin, final RequestMessage req) throws CADoesntExistsException,
301             AuthorizationDeniedException {
302         CA ca = null;
303         // See if we can get issuerDN directly from request
304         if (req.getIssuerDN() != null) {
305             String dn = certificateStoreSession.getCADnFromRequest(req);
306             ca = caSession.getCA(admin, dn.hashCode());
307             if (log.isDebugEnabled()) {
308                 log.debug("Using CA (from issuerDN) with id: " + ca.getCAId() + " and DN: " + ca.getSubjectDN());
309             }
310         } else {
311             throw new CADoesntExistsException(intres.getLocalizedMessage("createcert.canotfoundissuerusername", req.getIssuerDN(), req.getUsername()));
312         }
313 
314         if (ca.getStatus() != CAConstants.CA_ACTIVE) {
315             final String msg = intres.getLocalizedMessage("createcert.canotactive", ca.getSubjectDN());
316             throw new EJBException(msg);
317         }
318         return ca;
319     }

321     @Override
322     public CertificateDataWrapper createCertificate(final AuthenticationToken admin, final EndEntityInformation endEntityInformation, final CA ca,
323             final RequestMessage request, final PublicKey pk, final int keyusage, final Date notBefore, final Date notAfter,
324             final Extensions extensions, final String sequence, CertificateGenerationParams certGenParams, final long updateTime)
325             throws AuthorizationDeniedException, IllegalNameException, CustomCertificateSerialNumberException, CertificateCreateException,
326             CertificateRevokeException, CertificateSerialNumberException, CryptoTokenOfflineException, IllegalKeyException,
327             CertificateExtensionException, IllegalValidityException, CAOfflineException, InvalidAlgorithmException {
328         if (log.isTraceEnabled()) {
329             log.trace(">createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)");
330         }
331         
332         // Even though CA is passed as an argument to this method, we do check authorization on that.
333         // To make sure we properly log authorization checks needed to issue a cert.
334         // We need to check that admin have rights to create certificates, and have access to the CA
335         if (!authorizationSession.isAuthorized(admin, StandardRules.CREATECERT.resource(), StandardRules.CAACCESS.resource() + ca.getCAId())) {
336             final String msg = intres.getLocalizedMessage("createcert.notauthorized", admin.toString(), ca.getCAId());
337             throw new AuthorizationDeniedException(msg);
338         }
339 
340         // Audit log that we received the request
341         final Map<String, Object> details = new LinkedHashMap<String, Object>();
342         details.put("subjectdn", endEntityInformation.getDN());
343         details.put("requestX500name", (request == null || request.getRequestX500Name() == null) ? "null" : request.getRequestX500Name().toString());
344         details.put("subjectaltname", endEntityInformation.getSubjectAltName());
345         if (null != request) {
346             details.put("requestaltname", request.getRequestAltNames());
347         }
348         details.put("certprofile", endEntityInformation.getCertificateProfileId());
349         details.put("keyusage", keyusage);
350         details.put("notbefore", notBefore);
351         details.put("notafter", notAfter);
352         details.put("sequence", sequence);
353         details.put("publickey", new String(Base64.encode(pk.getEncoded(), false)));
354         logSession.log(EventTypes.CERT_REQUEST, EventStatus.SUCCESS, ModuleTypes.CERTIFICATE, ServiceTypes.CORE, admin.toString(),
355                 String.valueOf(ca.getCAId()), null, endEntityInformation.getUsername(), details);
356         
357         // Retrieve the certificate profile this user should have, checking for authorization to the profile
358         final int certProfileId = endEntityInformation.getCertificateProfileId();
359         final CertificateProfile certProfile = getCertificateProfile(certProfileId, ca.getCAId());
360         
361         // Validate ValidatorPhase.DATA_VALIDATION
362         try {
363             // Which public key to validate follows the criteria established in RequestAndPublicKeySelector, which is the same as used in the CA.
364             final ExtendedInformation ei = endEntityInformation.getExtendedInformation();
365             final RequestAndPublicKeySelectorcKeySelector.html#RequestAndPublicKeySelector">RequestAndPublicKeySelector pkSelector = new RequestAndPublicKeySelector(request, pk, ei);
366             keyValidatorSession.validatePublicKey(admin, ca, endEntityInformation, certProfile, notBefore, notAfter,
367                     pkSelector.getPublicKey());
368         } catch(ValidationException e) {
369             throw new CertificateCreateException(ErrorCode.ILLEGAL_KEY, e);
370         }
371         try {
372             keyValidatorSession.validateDnsNames(admin, ca, endEntityInformation, request);
373         } catch (ValidationException e) {
374             // Re-factor: ErrorCode could be specified more precisely.
375             throw new CertificateCreateException(ErrorCode.NOT_AUTHORIZED, e.getLocalizedMessage());
376         }
377         
378         // Set up audit logging of CT pre-certificate
379         addCTLoggingCallback(certGenParams, admin.toString());
380 
381         try {
382             CertificateDataWrapper result = null;
383             // If the user is of type USER_INVALID, it cannot have any other type (in the mask)
384             if (endEntityInformation.getType().isType(EndEntityTypes.INVALID)) {
385                 final String msg = intres.getLocalizedMessage("createcert.usertypeinvalid", endEntityInformation.getUsername());
386                 throw new CertificateCreateException(ErrorCode.INTERNAL_ERROR, msg);
387             }
388             
389             assertSubjectEnforcements(ca.getCAInfo(), endEntityInformation);
390             assertSubjectKeyIdEnforcements(ca.getCAInfo(), endEntityInformation, pk);
391     
392             //certProfile.verifyKey(pk); Verifying the public key against certificate profile is going to be executed in *CA.generateCertificate
393 
394             // Below we have a small loop if it would happen that we generate the same serial number twice
395             // If using only 4 byte serial numbers this do happen once in a while
396             Certificate cert = null;
397             String cafingerprint = null;
398             final boolean useCustomSN;
399             {
400                 final ExtendedInformation ei = endEntityInformation.getExtendedInformation();
401                 useCustomSN = ei != null && ei.certificateSerialNumber() != null;
402             }
403             final int maxRetrys;
404             if (useCustomSN) {
405                 if (ca.isUseCertificateStorage() && !isUniqueCertificateSerialNumberIndex()) {
406                     final String msg = intres.getLocalizedMessage("createcert.not_unique_certserialnumberindex");
407                     log.error(msg);
408                     throw new CustomCertificateSerialNumberException(msg);
409                 }
410                 if (!certProfile.getAllowCertSerialNumberOverride()) {
411                     final String msg = intres
412                             .getLocalizedMessage("createcert.certprof_not_allowing_cert_sn_override", Integer.valueOf(certProfileId));
413                     log.info(msg);
414                     throw new CustomCertificateSerialNumberException(msg);
415                 }
416                 maxRetrys = 1;
417             } else {
418                 maxRetrys = 5;
419             }
420             
421             // Before storing the new certificate, check if single active certificate constraint is active, and if so let's revoke all active and unexpired certificates
422             if (certProfile.isSingleActiveCertificateConstraint()) {
423                 // Only get not yet expired certificates with status CERT_ACTIVE, CERT_NOTIFIEDABOUTEXPIRATION, CERT_REVOKED
424                 final List<CertificateDataWrapper> cdws = certificateStoreSession.getCertificateDataByUsername(endEntityInformation.getUsername(),
425                         true, Arrays.asList(CertificateConstants.CERT_ARCHIVED, CertificateConstants.CERT_INACTIVE,
426                                 CertificateConstants.CERT_ROLLOVERPENDING, CertificateConstants.CERT_UNASSIGNED));
427                 if (log.isDebugEnabled()) {
428                     log.debug("SingleActiveCertificateConstraint, found "+cdws.size()+" old (non expired, active) certificates.");
429                 }
430                 for (final CertificateDataWrapper cdw : cdws) {
431                     final CertificateData certificateData = cdw.getCertificateData();
432                     if (certificateData.getStatus() == CertificateConstants.CERT_REVOKED && certificateData.getRevocationReason() != RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD) {
433                         // It's possible that revocation may have been already called from a higher level bean (such as SignSession) which had to 
434                         // perform operations (such as publishing) which are out of scope of this method. This check is performed twice in order 
435                         // to ensure that operations entirely contained within CESeCore follow this constraint as well. 
436                         continue;
437                     }                  
438                     // Authorization to the CA was already checked at the head of this method, so no need to do so now
439                     certificateStoreSession.setRevokeStatusNoAuth(admin, certificateData, new Date(), RevokedCertInfo.REVOCATION_REASON_SUPERSEDED);
440                 }
441             }
442             
443             CertificateSerialNumberException storeEx = null; // this will not be null if stored == false after the below passage
444             String serialNo = "unknown";
445             for (int retrycounter = 0; retrycounter < maxRetrys; retrycounter++) {
446                 final CryptoToken cryptoToken = cryptoTokenManagementSession.getCryptoToken(ca.getCAToken().getCryptoTokenId());
447                 if (cryptoToken==null) {
448                     final String msg = intres.getLocalizedMessage("error.catokenoffline", ca.getCAId());
449                     log.info(msg);
450                     CryptoTokenOfflineExceptionhtml#CryptoTokenOfflineException">CryptoTokenOfflineException exception = new CryptoTokenOfflineException("CA's CryptoToken not found.");
451                     auditFailure(admin, exception, exception.getMessage(), "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
452                     throw exception;
453                 }
454                 final AvailableCustomCertificateExtensionsConfigurationate/certextensions/AvailableCustomCertificateExtensionsConfiguration.html#AvailableCustomCertificateExtensionsConfiguration">AvailableCustomCertificateExtensionsConfiguration cceConfig = (AvailableCustomCertificateExtensionsConfiguration) 
455                         globalConfigurationSession.getCachedConfiguration(AvailableCustomCertificateExtensionsConfiguration.CONFIGURATION_ID);
456                 certGenParams.setAuthenticationToken(admin);
457                 certGenParams.setCertificateValidationDomainService(keyValidatorSession);
458 
459                 // Validate ValidatorPhase.PRE_CERTIFICATE_VALIDATION (X.509 CA only)
460                 cert = ca.generateCertificate(cryptoToken, endEntityInformation, request, pk, keyusage, notBefore, notAfter, certProfile, extensions, sequence, certGenParams, cceConfig);
461                 // Set null required here?
462                 certGenParams.setCertificateValidationDomainService(null);
463                 
464                 // Validate ValidatorPhase.CERTIFICATE_VALIDATION (X.509 CA only)
465                 if (CAInfo.CATYPE_X509 == ca.getCAType()) {
466                     try {
467                         keyValidatorSession.validateCertificate(admin, IssuancePhase.CERTIFICATE_VALIDATION, ca, endEntityInformation, (X509Certificate) cert);
468                     } catch (ValidationException e) {
469                         throw new CertificateCreateException(ErrorCode.INVALID_CERTIFICATE, e);
470                     }
471                 }
472                 
473                 cafingerprint = CertTools.getFingerprintAsString(ca.getCACertificate());
474                 serialNo = CertTools.getSerialNumberAsString(cert);
475                 // Store certificate in the database, if this CA is configured to do so.
476                 if (!ca.isUseCertificateStorage() || !certProfile.getUseCertificateStorage()) {
477                     // We still need to return a CertificateData object for publishers
478                     final CertificateData throwAwayCertData = new CertificateData(cert, cert.getPublicKey(), endEntityInformation.getUsername(),
479                             cafingerprint, CertificateConstants.CERT_ACTIVE, certProfile.getType(), certProfileId,
480                             endEntityInformation.getEndEntityProfileId(), null, updateTime, false, certProfile.getStoreSubjectAlternativeName());
481                     result = new CertificateDataWrapper(cert, throwAwayCertData, null);
482                     // Always Store full certificate for OCSP signing certificates.
483                     boolean isOcspSigner = certProfile.getExtendedKeyUsageOids().contains("1.3.6.1.5.5.7.3.9");
484                     if (!isOcspSigner) {
485                         break; // We have our cert and we don't need to store it.. Move on..
486                     }
487                     log.debug("Storing certificate even though storage is disabled since OCSP signer EKU is used.");
488                 }
489                 try {
490                     // Remember for CVC serialNo can be alphanumeric, so we can't just try to decode that using normal Java means (BigInteger.valueOf)...
491                     assertSerialNumberForIssuerOk(ca, CertTools.getSerialNumber(cert));
492                     // Tag is reserved for future use, currently only null
493                     final String tag = null;
494                     // Authorization was already checked by since this is a private method, the CA parameter should
495                     // not be possible to get without authorization
496                     result = certificateStoreSession.storeCertificateNoAuth(admin, cert, endEntityInformation.getUsername(), cafingerprint, CertificateConstants.CERT_ACTIVE,
497                             certProfile.getType(), certProfileId, endEntityInformation.getEndEntityProfileId(), tag, updateTime);
498                     storeEx = null;
499                     break;
500                 } catch (CertificateSerialNumberException e) {
501                     // If we have created a unique index on (issuerDN,serialNumber) on table CertificateData we can
502                     // get a CreateException here if we would happen to generate a certificate with the same serialNumber
503                     // as one already existing certificate.
504                     if (retrycounter + 1 < maxRetrys) {
505                         log.info("Can not store certificate with serNo (" + serialNo + "), will retry (retrycounter=" + retrycounter
506                                 + ") with a new certificate with new serialNo: " + e.getMessage());
507                     }
508                     storeEx = e;
509                 }
510             }
511             if (storeEx != null) {
512                 if (useCustomSN) {
513                     final String msg = intres.getLocalizedMessage("createcert.cert_serial_number_already_in_database", serialNo);
514                     log.info(msg);
515                     throw new CustomCertificateSerialNumberException(msg);
516                 }
517                 log.error("Can not store certificate in database in 5 tries, aborting: ", storeEx);
518                 throw storeEx;
519             }
520 
521             // Finally we check if this certificate should not be issued as active, but revoked directly upon issuance
522             int revreason = RevokedCertInfo.NOT_REVOKED;
523             final ExtendedInformation ei = endEntityInformation.getExtendedInformation();
524             if (ei != null) {
525                 revreason = ei.getIssuanceRevocationReason();
526                 if (revreason != RevokedCertInfo.NOT_REVOKED) {
527                     // If we don't store the certificate in the database, we wont support revocation/reactivation so issuing revoked certificates would be
528                     // really strange.
529                     if (ca.isUseCertificateStorage() && certProfile.getUseCertificateStorage()) {
530                         certificateStoreSession.setRevokeStatus(admin, result, new Date(), revreason);
531                     } else {
532                         log.warn("CA configured to revoke issued certificates directly, but not to store issued the certificates. Revocation will be ignored. Please verify your configuration.");
533                     }
534                 }
535             }
536             if (log.isDebugEnabled()) {
537                 log.debug("Generated certificate with SerialNumber '" + serialNo + "' for user '" + endEntityInformation.getUsername() + "', with revocation reason="
538                         + revreason);
539                 log.debug(cert.toString());
540             }
541             
542             // Audit log that we issued the certificate
543             final Map<String, Object> issuedetails = new LinkedHashMap<String, Object>();
544             issuedetails.put("subjectdn", endEntityInformation.getDN());
545             issuedetails.put("certprofile", endEntityInformation.getCertificateProfileId());
546             issuedetails.put("issuancerevocationreason", revreason);
547             try {
548                 issuedetails.put("cert", new String(Base64.encode(cert.getEncoded(), false)));
549             } catch (CertificateEncodingException e) {
550                 //Should not be able to happen at this point
551                 throw new IllegalStateException();
552             }
553             logSession.log(EventTypes.CERT_CREATION, EventStatus.SUCCESS, ModuleTypes.CERTIFICATE, ServiceTypes.CORE, admin.toString(), String.valueOf(ca.getCAId()), serialNo, endEntityInformation.getUsername(),
554             		issuedetails);
555 
556             if (log.isTraceEnabled()) {
557                 log.trace("<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)");
558             }
559             return result;
560             // We need to catch and re-throw all of these exception just because we need to audit log all failures
561         } catch (CustomCertificateSerialNumberException e) {
562             log.info(e.getMessage());
563             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
564             throw e;
565         }  catch (AuthorizationDeniedException e) {
566             log.info(e.getMessage());
567             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
568             throw e;
569         } catch (CertificateCreateException e) {
570             log.info(e.getMessage());
571             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
572             // Rollback
573             throw e;
574         } catch(CryptoTokenOfflineException e) {
575             final String msg = intres.getLocalizedMessage("error.catokenoffline", ca.getCAId());
576             log.info(msg);
577             auditFailure(admin, e, e.getMessage(), "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
578             throw e;
579         } catch (CAOfflineException e) {
580             log.error("Error creating certificate", e);
581             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
582             throw e;
583         } catch (InvalidAlgorithmException e) {
584             log.error("Error creating certificate", e);
585             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
586             throw e;
587         } catch (IllegalValidityException e) {
588             log.error("Error creating certificate", e);
589             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
590             throw e;
591         } catch (OperatorCreationException e) {
592             log.error("Error creating certificate", e);
593             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
594             // Rollback
595             throw new CertificateCreateException(e);
596         } catch (SignatureException e) {
597             log.error("Error creating certificate", e);
598             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
599             // Rollback
600             throw new CertificateCreateException(e);
601         } catch (CertificateExtensionException e) {
602             log.error("Error creating certificate", e);
603             auditFailure(admin, e, null, "<createCertificate(EndEntityInformation, CA, X500Name, pk, ku, notBefore, notAfter, extesions, sequence)", ca.getCAId(), endEntityInformation.getUsername());
604             throw e;
605         }
606     }

721     private CertificateProfile getCertificateProfile(final int certProfileId, final int caid) throws AuthorizationDeniedException {
722         final CertificateProfile certProfile = certificateProfileSession.getCertificateProfile(certProfileId);
723         // What if certProfile == null?
724         if (certProfile == null) {
725             final String msg = intres.getLocalizedMessage("createcert.errorcertprofilenotfound", Integer.valueOf(certProfileId));
726             throw new AuthorizationDeniedException(msg);
727         }
728         if (log.isDebugEnabled()) {
729             log.debug("Using certificate profile with id " + certProfileId);
730         }
731 
732         // Check that CAid is among available CAs
733         boolean caauthorized = false;
734         for (final Integer nextInt : certProfile.getAvailableCAs()) {
735             final int next = nextInt.intValue();
736             if (next == caid || next == CertificateProfile.ANYCA) {
737                 caauthorized = true;
738                 break;
739             }
740         }
741         if (!caauthorized) {
742             final String msg = intres.getLocalizedMessage("createcert.errorcertprofilenotauthorized", Integer.valueOf(caid),
743                     Integer.valueOf(certProfileId));
744             throw new AuthorizationDeniedException(msg);
745         }
746         return certProfile;
747     }

773     /**
774      * Small function that makes a list of users, space separated. Used for logging. Only actually displays the first 10 records, then a notice how
775      * many records were not displayed
776      * 
777      * @param users a set of usernames to create a string of
778      * @return space separated list of usernames, i.e. "'user1' 'user2' 'user3'", max 10 users
779      */
780     private String listUsers(final Set<String> users) {
781         final StringBuilder sb = new StringBuilder();
782         int bar = 0; // limit number of displayed users
783         for (final String user : users) {
784             if (sb.length() > 0) {
785                 sb.append(' ');
786             }
787             if (bar++ > 9) {
788                 sb.append("and ").append(users.size() - bar + 1).append(" users not displayed");
789                 break;
790             }
791             sb.append('\'');
792             sb.append(user);
793             sb.append('\'');
794         }
795         return sb.toString();
796     }
