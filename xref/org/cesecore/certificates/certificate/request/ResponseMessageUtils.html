14  package org.cesecore.certificates.certificate.request;
15  
16  import java.lang.reflect.InvocationTargetException;
17  import java.security.PrivateKey;
18  import java.security.cert.Certificate;
19  import java.util.Collection;
20  import java.util.Random;
21  
22  import org.apache.log4j.Logger;
23  import org.cesecore.util.Base64;
24  
25  /**
26   * 
27   * @version $Id: ResponseMessageUtils.java 26080 2017-06-26 17:32:09Z anatom $
28   *
29   */
30  
31  public abstract class ResponseMessageUtils {
32  
33      private static final Logger log = Logger.getLogger(ResponseMessageUtils.class);
34      
35      public static CertificateResponseMessage createResponseMessage(Class<? extends ResponseMessage> responseClass, RequestMessage req,
36              Collection<Certificate> certs, PrivateKey signPriv, String provider) {
37          CertificateResponseMessage ret = null;
38          // Create the response message and set all required fields
39          try {
40              ret = (CertificateResponseMessage) responseClass.getConstructor().newInstance();
41          } catch (InstantiationException | NoSuchMethodException | InvocationTargetException e) {
42              //TODO : do something with these exceptions
43              log.error("Error creating response message", e);
44              return null;
45          } catch (IllegalAccessException e) {
46              log.error("Error creating response message", e);
47              return null;
48          }
49          if (ret.requireSignKeyInfo()) {
50              ret.setSignKeyInfo(certs, signPriv, provider);
51          }
52          if (req.getSenderNonce() != null) {
53              ret.setRecipientNonce(req.getSenderNonce());
54          }
55          if (req.getTransactionId() != null) {
56              ret.setTransactionId(req.getTransactionId());
57          }
58          // Sender nonce is a random number
59          byte[] senderNonce = new byte[16];
60          Random randomSource = new Random();
61          randomSource.nextBytes(senderNonce);
62          ret.setSenderNonce(new String(Base64.encode(senderNonce)));
63          // If we have a specified request key info, use it in the reply
64          if (req.getRequestKeyInfo() != null) {
65              ret.setRecipientKeyInfo(req.getRequestKeyInfo());
66          }
67          // Which digest algorithm to use to create the response, if applicable
68          ret.setPreferredDigestAlg(req.getPreferredDigestAlg());
69          // Include the CA cert or not in the response, if applicable for the response type
70          ret.setIncludeCACert(req.includeCACert());
71          // Hint to the response which request type it is in response to
72          ret.setRequestType(req.getRequestType());
73          ret.setRequestId(req.getRequestId());
74          // If there is some protection parameters, or other params, we need to lift over from the request message, the request and response knows about it
75          ret.setProtectionParamsFromRequest(req);
76          return ret;
77      }
78  }
