13  package org.cesecore.certificates.certificate;
14  
15  import java.math.BigInteger;
16  import java.util.Collection;
17  import java.util.List;
18  
19  import javax.persistence.EntityManager;
20  import javax.persistence.Query;
21  
22  import org.apache.log4j.Logger;
23  import org.cesecore.certificates.crl.RevokedCertInfo;
24  import org.cesecore.config.CesecoreConfiguration;
25  import org.cesecore.util.CompressedCollection;
26  import org.cesecore.util.ValueExtractor;
27  
28  /**
29   * Contains common function for CertificateDataSessionBean and NoConflictCertificateDataSessionBean.
30   * 
31   * @version $Id: BaseCertificateDataSessionBean.java 34193 2020-01-07 15:18:15Z samuellb $
32   */
33  public abstract class BaseCertificateDataSessionBean {
34  
35      private static final Logger log = Logger.getLogger(CertificateDataSessionBean.class);
36      
37      /** Returns the name of the table in the database. Either "CertificateData" or "NoConflictCertificateData" 
38       * @return name*/
39      protected abstract String getTableName();
40      
41      /** @return the entity manager to use. */
42      protected abstract EntityManager getEntityManager();
43      
44      /**
45       * @param issuerDN DN
46       * @param lastbasecrldate CRL date 
47       * @param forceGetAll Get all?
48       * @return a list with information about revoked certificates. Depending on the table, the result can
49       * either contain at most one entry per certificate, or it may contain duplicates.
50       */
51      protected Collection<RevokedCertInfo> getRevokedCertInfosInternal(final String issuerDN, final long lastbasecrldate, final boolean forceGetAll) {
52          final String tableName = getTableName();
53          final Query query;
54          if (lastbasecrldate > 0) {
55              // Delta CRL
56              query = getEntityManager().createNativeQuery(
57                      "SELECT a.fingerprint as fingerprint, a.serialNumber as serialNumber, a.expireDate as expireDate, a.revocationDate as revocationDate, a.revocationReason as revocationReason FROM " + tableName + " a WHERE "
58                              + "a.issuerDN=:issuerDN AND a.revocationDate>:revocationDate AND (a.status=:status1 OR a.status=:status2 OR a.status=:status3)",
59                      "RevokedCertInfoSubset");
60              query.setParameter("issuerDN", issuerDN);
61              query.setParameter("revocationDate", lastbasecrldate);
62              query.setParameter("status1", CertificateConstants.CERT_REVOKED);
63              query.setParameter("status2", CertificateConstants.CERT_ACTIVE); // in case the certificate has been changed from on hold, we need to include it as "removeFromCRL" in the Delta CRL
64              query.setParameter("status3", CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION); // could happen if a cert is re-activated just before expiration
65          } else if (forceGetAll) {
66              // Base CRL
67              query = getEntityManager().createNativeQuery(
68                      "SELECT a.fingerprint as fingerprint, a.serialNumber as serialNumber, a.expireDate as expireDate, a.revocationDate as revocationDate, a.revocationReason as revocationReason FROM " + tableName + " a WHERE "
69                              + "a.issuerDN=:issuerDN AND (a.status=:status1 OR a.status=:status2 OR a.status=:status3)",
70                      "RevokedCertInfoSubset");
71              query.setParameter("issuerDN", issuerDN);
72              query.setParameter("status1", CertificateConstants.CERT_REVOKED);
73              query.setParameter("status2", CertificateConstants.CERT_ACTIVE); // in case the certificate has been changed from on hold, we need to include it as "removeFromCRL" in the Delta CRL
74              query.setParameter("status3", CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION); // could happen if a cert is re-activated just before expiration
75          } else {
76              // Base CRL
77              query = getEntityManager().createNativeQuery(
78                      "SELECT a.fingerprint as fingerprint, a.serialNumber as serialNumber, a.expireDate as expireDate, a.revocationDate as revocationDate, a.revocationReason as revocationReason FROM " + tableName + " a WHERE "
79                              + "a.issuerDN=:issuerDN AND a.status=:status",
80                      "RevokedCertInfoSubset");
81              query.setParameter("issuerDN", issuerDN);
82              query.setParameter("status", CertificateConstants.CERT_REVOKED);
83          }
84          return getRevokedCertInfosInternal(query);
85      }
86      
87      private Collection<RevokedCertInfo> getRevokedCertInfosInternal(final Query query) {
88          final int maxResults = CesecoreConfiguration.getDatabaseRevokedCertInfoFetchSize();
89          query.setMaxResults(maxResults);
90          int firstResult = 0;
91          final CompressedCollection<RevokedCertInfo> revokedCertInfos = new CompressedCollection<>(RevokedCertInfo.class);
92          while (true) {
93              query.setFirstResult(firstResult);
94              @SuppressWarnings("unchecked")
95              final List<Object[]> incompleteCertificateDatas = query.getResultList();
96              if (incompleteCertificateDatas.size()==0) {
97                  break;
98              }
99              if (log.isDebugEnabled()) {
100                 log.debug("Read batch of " + incompleteCertificateDatas.size() + " RevokedCertInfo.");
101             }
102             for (final Object[] current : incompleteCertificateDatas) {
103                 // The order of the results are defined by the SqlResultSetMapping annotation
104                 final byte[] fingerprint = ((String)current[0]).getBytes();
105                 final byte[] serialNumber = new BigInteger((String)current[1]).toByteArray();
106                 final long expireDate = ValueExtractor.extractLongValue(current[2]);
107                 final long revocationDate = ValueExtractor.extractLongValue(current[3]);
108                 int revocationReason = ValueExtractor.extractIntValue(current[4]);
109                 if (revocationReason == -1) {
110                     revocationReason = RevokedCertInfo.REVOCATION_REASON_REMOVEFROMCRL;
111                 }
112                 revokedCertInfos.add(new RevokedCertInfo(fingerprint, serialNumber, revocationDate, revocationReason, expireDate));
113             }
114             firstResult += maxResults;
115         }
116         revokedCertInfos.closeForWrite();
117         return revokedCertInfos;
118     }
