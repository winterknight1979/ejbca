179     /** same as storeCertificateNoAuth but with a flag to not audit log certificate storage.
180      * The only reason to not audit log is when called from checkForUniqueCertificateSerialNumberIndexInTransaction
181      *
182      * @param adminForLogging the AuthenticationToken that will be used for audit logging of the event
183      * @param incert The certificate to be stored.
184      * @param username username of end entity owning the certificate.
185      * @param cafp Fingerprint (hex) of the CAs certificate.
186      * @param status the status from the CertificateConstants.CERT_ constants
187      * @param type Type of certificate (CERTTYPE_ENDENTITY etc from CertificateConstants).
188      * @param certificateProfileId the certificate profile id this cert was issued under
189      * @param endEntityProfileId the end entity profile id this cert was issued under
190      * @param tag a custom string tagging this certificate for some purpose
191      * @param updateTime epoch millis to use as last update time of the stored object
192      * @param doAuditLog determines if a security audit log event shall be written or not with, EventTypes.CERT_STORED, ModuleTypes.CERTIFICATE,
193      * must only be used when storing special internal certificates, such as the test certificates for checking unique database index.
194      * @return data
195      */
196     private CertificateDataWrapper storeCertificateNoAuthInternal(AuthenticationToken adminForLogging, Certificate incert, String username, String cafp, int status, int type,
197             int certificateProfileId, final int endEntityProfileId, String tag, long updateTime, boolean doAuditLog) {
198         final PublicKey pubk = enrichEcPublicKey(incert.getPublicKey(), cafp);
199         // Create the certificate in one go with all parameters at once. This used to be important in EJB2.1 so the persistence layer only creates
200         // *one* single
201         // insert statement. If we do a home.create and the some setXX, it will create one insert and one update statement to the database.
202         // Probably not important in EJB3 anymore
203         final boolean useBase64CertTable = CesecoreConfiguration.useBase64CertTable();
204         Base64CertData base64CertData = null;
205         final CertificateProfile certificateProfile = certificateProfileSession.getCertificateProfile(certificateProfileId);
206         final boolean storeCertificateData = certificateProfile==null || certificateProfile.getStoreCertificateData();
207         if (useBase64CertTable && storeCertificateData) {
208             // use special table for encoded data if told so.
209             base64CertData = new Base64CertData(incert);
210             entityManager.persist(new Base64CertData(incert));
211         }
212         final boolean storeSubjectAlternativeName = certificateProfile==null || certificateProfile.getStoreSubjectAlternativeName();
213         final CertificateData certificateData = new CertificateData(incert, pubk, username, cafp, status, type, certificateProfileId, endEntityProfileId, tag, updateTime,
214                 !useBase64CertTable && storeCertificateData, storeSubjectAlternativeName);
215         entityManager.persist(certificateData);
216         if (doAuditLog) {
217             final String serialNo = CertTools.getSerialNumberAsString(incert);
218             final String msg = INTRES.getLocalizedMessage("store.storecert", username, certificateData.getFingerprint(), certificateData.getSubjectDnNeverNull(), certificateData.getIssuerDN(), serialNo);
219             final String caId = String.valueOf(CertTools.getIssuerDN(incert).hashCode());
220             logSession.log(EventTypes.CERT_STORED, EventStatus.SUCCESS, ModuleTypes.CERTIFICATE, ServiceTypes.CORE, adminForLogging.toString(), caId,
221                     serialNo, username, msg);
222         }
223         return new CertificateDataWrapper(incert, certificateData, base64CertData);
224     }

243     /**
244      * We need special handling here of CVC certificate with EC keys, because they lack EC parameters in all certs
245      * except the Root certificate (CVCA)
246      * @param pubk key
247      * @param cafp FP
248      * @return enriched key
249      */
250     private PublicKey enrichEcPublicKey(final PublicKey pubk, final String cafp) {
251         PublicKey ret = pubk;
252         if ((pubk instanceof PublicKeyEC)) {
253             PublicKeyEC pkec = (PublicKeyEC) pubk;
254             // The public key of IS and DV certificate (CVC) do not have any parameters so we have to do some magic to get a complete EC public key
255             ECParameterSpec spec = pkec.getParams();
256             if (spec == null) {
257                 // We need to enrich this public key with parameters
258                 try {
259                     if (cafp != null) {
260                         String cafingerp = cafp;
261                         CertificateData cacert = certificateDataSession.findByFingerprint(cafp);
262                         if(cacert != null) {
263                         String nextcafp = cacert.getCaFingerprint();
264                         int bar = 0; // never go more than 5 rounds, who knows what strange things can exist in the CAFingerprint column, make sure we
265                                      // never get stuck here
266                         while ((!StringUtils.equals(cafingerp, nextcafp)) && (bar++ < 5)) {
267                             cacert = certificateDataSession.findByFingerprint(cafp);
268                             if (cacert == null) {
269                                 break;
270                             }
271                             cafingerp = nextcafp;
272                             nextcafp = cacert.getCaFingerprint();
273                         }
274                             if (cacert != null) {
275                                 // We found a root CA certificate, hopefully ?
276                                 PublicKey pkwithparams = cacert.getCertificate(this.entityManager).getPublicKey();
277                                 ret = KeyTools.getECPublicKeyWithParams(pubk, pkwithparams);
278                             }
279                         }
280                     }
281                 }  catch (InvalidKeySpecException e) {
282                     if (log.isDebugEnabled()) {
283                         log.debug("Can not enrich EC public key with missing parameters: ", e);
284                     }
285                 }
286             }
287         } // finished with ECC key special handling
288         return ret;
289     }

291     @Override
292     @TransactionAttribute(TransactionAttributeType.REQUIRED)
293     public boolean updateCertificateOnly(AuthenticationToken authenticationToken, Certificate certificate) {
294         final String fingerprint = CertTools.getFingerprintAsString(certificate);
295         final CertificateData certificateData = certificateDataSession.findByFingerprint(fingerprint);
296         if (certificateData==null || certificateData.getCertificate(entityManager) != null) {
297             return false;
298         }
299         final boolean useBase64CertTable = CesecoreConfiguration.useBase64CertTable();
300         if (useBase64CertTable) {
301             // use special table for encoded data if told so.
302             entityManager.persist(new Base64CertData(certificate));
303         } else {
304             try {
305                 certificateData.setBase64Cert(new String(Base64.encode(certificate.getEncoded())));
306             } catch (CertificateEncodingException e) {
307                 log.error("Failed to encode certificate for fingerprint " + fingerprint, e);
308                 return false;
309             }
310         }
311         final String username = certificateData.getUsername();
312         final String serialNo = CertTools.getSerialNumberAsString(certificate);
313         final String msg = INTRES.getLocalizedMessage("store.storecert", username, fingerprint, certificateData.getSubjectDnNeverNull(),
314                 certificateData.getIssuerDN(), serialNo);
315         Map<String, Object> details = new LinkedHashMap<>();
316         details.put("msg", msg);
317         final String caId = String.valueOf(CertTools.getIssuerDN(certificate).hashCode());
318         logSession.log(EventTypes.CERT_STORED, EventStatus.SUCCESS, ModuleTypes.CERTIFICATE, ServiceTypes.CORE, authenticationToken.toString(),
319                 caId, serialNo, username, details);
320         return true;
321     }

360     @Override
361     public List<CertificateDataWrapper> getCertificateDatasBySubjectAndIssuer(String subjectDN, String issuerDN, boolean onlyActive) {
362         // First make a DN in our well-known format
363         final String dn = CertTools.stringToBCDNString(StringTools.strip(subjectDN));
364         final String issuerdn = CertTools.stringToBCDNString(StringTools.strip(issuerDN));
365         if (log.isDebugEnabled()) {
366             log.debug("Looking for cert with (transformed)DN: " + dn);
367         }
368         final List<CertificateDataWrapper> ret = new ArrayList<>();
369         final Query query;
370         if (onlyActive) {
371             query = entityManager.createQuery("SELECT a FROM CertificateData a WHERE " + "a.subjectDN=:subjectDN AND a.issuerDN=:issuerDN"
372                     + " AND (a.status=:active OR a.status=:notifiedexpired OR (a.status=:revoked AND a.revocationReason=:onhold))" + "AND a.expireDate>:expireDate");
373             query.setParameter("active", CertificateConstants.CERT_ACTIVE);
374             query.setParameter("notifiedexpired", CertificateConstants.CERT_NOTIFIEDABOUTEXPIRATION);
375             query.setParameter("revoked", CertificateConstants.CERT_REVOKED);
376             query.setParameter("onhold", RevokedCertInfo.REVOCATION_REASON_CERTIFICATEHOLD);
377             query.setParameter("expireDate", System.currentTimeMillis());
378         } else {
379             query = entityManager.createQuery("SELECT a FROM CertificateData a WHERE a.subjectDN=:subjectDN AND a.issuerDN=:issuerDN");
380         }
381         query.setParameter("subjectDN", dn);
382         query.setParameter("issuerDN", issuerdn);
383         for (final Object certificateDataObject : query.getResultList()) {
384             final CertificateData certificateData = (CertificateData) certificateDataObject;
385             ret.add(new CertificateDataWrapper(certificateData, Base64CertData.findByFingerprint(entityManager, certificateData.getFingerprint())));
386         }
387         return ret;
388     }

509     @Override
510     public X509Certificate findLatestX509CertificateBySubject(String subjectDN, X509Certificate rolloverCA, boolean findRollover) {
511         final Collection<CertificateDataWrapper> certificateDatas = getCertificateDatasBySubject(subjectDN);
512         X509Certificate result = null;
513         Collection<X509Certificate> trustedChain = null;
514         if (rolloverCA != null) {
515             trustedChain = new ArrayList<>();
516             trustedChain.add(rolloverCA);
517         }
518 
519         // Find the newest certificate
520         for (CertificateDataWrapper certDataWrapper : certificateDatas) {
521             final int status = certDataWrapper.getCertificateData().getStatus();
522             // Ignore rollover CA certificates unless explicitly requested
523             if (status == CertificateConstants.CERT_ROLLOVERPENDING && !findRollover) {
524                 continue;
525             }
526             if (certDataWrapper.getCertificate() instanceof X509Certificate) {
527                 final X509Certificate x509Certificate = (X509Certificate) certDataWrapper.getCertificate();
528                 if (rolloverCA != null) {
529                     // The old and new CA certificate will generally have different keys, but we also handle the case where they don't by checking the date
530                     boolean signedByRolloverCAKey = false;
531                     try {
532                         CertTools.verify(x509Certificate, trustedChain, CertTools.getNotBefore(x509Certificate));
533                         signedByRolloverCAKey = true;
534                     } catch (CertPathValidatorException e) {
535                         // NOPMD
536                     }
537                     // Check that the EE roll-over certificate validity starts equal to or after the roll-over CA certificates validity
538                     final Date notBeforeX509Certificate = CertTools.getNotBefore(x509Certificate);
539                     final Date notBeforeRolloverCA = CertTools.getNotBefore(rolloverCA);
540                     final boolean eeCertValidUnderCaValidity = !notBeforeX509Certificate.before(notBeforeRolloverCA);
541                     final boolean isRollover = signedByRolloverCAKey && eeCertValidUnderCaValidity;
542                     if (isRollover != findRollover) {
543                         if (log.isTraceEnabled()) {
544                             final String fingerprint = CertTools.getFingerprintAsString(x509Certificate);
545                             log.trace("Certificate with fingerprint '"+fingerprint+"' is not considered a rollover certificate. signedByRolloverCAKey: " +
546                                     signedByRolloverCAKey + " leaf not before: " + notBeforeX509Certificate + " CA not before: " + notBeforeRolloverCA);
547                         }
548                         continue;
549                     }
550                 }
551                 if (result == null || CertTools.getNotBefore(x509Certificate).after(CertTools.getNotBefore(result))) {
552                     result = x509Certificate;
553                 }
554             }
555         }
556 
557         return result;
558     }

1033     @Override
1034     public List<Certificate> getCertificateChain(final CertificateInfo certinfo) {
1035         final List<Certificate> chain = new ArrayList<>();
1036         final Set<String> seenFingerprints = new HashSet<>();
1037 
1038         CertificateInfo certInChain = certinfo;
1039         do {
1040             final String fingerprint = certInChain.getFingerprint();
1041             final Certificate thecert = findCertificateByFingerprint(fingerprint);
1042             if (!seenFingerprints.add(fingerprint) || thecert == null) {
1043                 break; // detected loop or missing cert. should not happen
1044             }
1045             chain.add(thecert);
1046             // roots are self-signed
1047             if (certInChain.getCAFingerprint().equals(fingerprint)) {
1048                 break;
1049             }
1050             // proceed with issuer
1051             certInChain = getCertificateInfo(certInChain.getCAFingerprint());
1052         } while (certInChain != null); // should not happen
1053         return chain;
1054     }

1532             if (reasonCode==RevokedCertInfo.REVOCATION_REASON_REMOVEFROMCRL) {
1533                 deleteLimitedCertificateData(limitedFingerprint);
1534             } else {
1535                 // Create a limited entry
1536                 final CertificateData limitedCertificateData = new CertificateData();
1537                 limitedCertificateData.setFingerprint(limitedFingerprint);
1538                 limitedCertificateData.setSerialNumber(serialNumber.toString());
1539                 limitedCertificateData.setIssuer(issuerDn);
1540                 limitedCertificateData.setSubjectDN(subjectDn);
1541                 limitedCertificateData.setUsername(username);
1542                 limitedCertificateData.setCertificateProfileId(Integer.valueOf(CertificateProfileConstants.CERTPROFILE_NO_PROFILE));
1543                 limitedCertificateData.setStatus(status);
1544                 limitedCertificateData.setRevocationReason(reasonCode);
1545                 limitedCertificateData.setRevocationDate(revocationDate);
1546                 limitedCertificateData.setUpdateTime(Long.valueOf(System.currentTimeMillis()));
1547                 limitedCertificateData.setCaFingerprint(caFingerprint);
1548                 log.info("Adding limited CertificateData entry with fingerprint=" + limitedFingerprint + ", serialNumber=" + serialNumber.toString(16).toUpperCase()+", issuerDn='"+issuerDn+"'");
1549                 entityManager.persist(limitedCertificateData);
1550             }
1551         } else if (limitedFingerprint.equals(cdw.getCertificateData().getFingerprint())) {
1552         	if (reasonCode==RevokedCertInfo.REVOCATION_REASON_REMOVEFROMCRL) {
1553                 deleteLimitedCertificateData(limitedFingerprint);
1554         	} else {
1555         	    final CertificateData limitedCertificateData = cdw.getCertificateData();
1556         	    if (cdw.getCertificateData().getRevocationDate()!=revocationDate.getTime() || cdw.getCertificateData().getRevocationReason()!=reasonCode) {
1557                     // Update the limited entry
1558                     log.info("Updating limited CertificateData entry with fingerprint=" + limitedFingerprint + ", serialNumber=" + serialNumber.toString(16).toUpperCase()+", issuerDn='"+issuerDn+"'");
1559                     limitedCertificateData.setStatus(CertificateConstants.CERT_REVOKED);
1560                     limitedCertificateData.setRevocationReason(reasonCode);
1561                     limitedCertificateData.setRevocationDate(revocationDate);
1562                     limitedCertificateData.setUpdateTime(Long.valueOf(System.currentTimeMillis()));
1563                     entityManager.merge(limitedCertificateData);
1564         	    } else {
1565         	        if (log.isDebugEnabled()) {
1566                         log.debug("Limited CertificateData entry with fingerprint=" + limitedFingerprint + ", serialNumber=" + serialNumber.toString(16).toUpperCase()+", issuerDn='"+issuerDn+"' was already up to date.");
1567         	        }
1568         	    }
1569         	}
1570         } else {
1571             // Refuse to update a normal entry with this method
1572         	throw new UnsupportedOperationException("Only limited certificate entries can be updated using this method.");
1573         }
1574     }

1616     public void reloadCaCertificateCacheAndSetTimeout() {

1618             log.trace(">timeOutReloadCaCertificateCache");
1619         }
1620         // Cancel any waiting timers of this type
1621         final Collection<Timer> timers = timerService.getTimers();

1623             if (timer.getInfo() instanceof Integer) {
1624                 final int currentTimerId = ((Integer)timer.getInfo()).intValue();
1625                 if (currentTimerId==TIMERID_CACERTIFICATECACHE) {
1626                     timer.cancel();

1628             }

1630         try {

1633             // Schedule a new timer of this type

1635             if (interval > 0) {
