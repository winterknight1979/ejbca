13  package org.cesecore.certificates.ocsp;
14  
15  import java.io.IOException;
16  import java.io.Serializable;
17  import java.security.MessageDigest;
18  import java.security.NoSuchAlgorithmException;
19  import java.security.NoSuchProviderException;
20  import java.security.cert.X509Certificate;
21  
22  import org.apache.log4j.Logger;
23  import org.bouncycastle.asn1.ocsp.OCSPObjectIdentifiers;
24  import org.bouncycastle.cert.ocsp.BasicOCSPResp;
25  import org.bouncycastle.cert.ocsp.OCSPException;
26  import org.bouncycastle.cert.ocsp.OCSPResp;
27  import org.bouncycastle.cert.ocsp.SingleResp;
28  import org.bouncycastle.cert.ocsp.UnknownStatus;
29  import org.bouncycastle.jce.provider.BouncyCastleProvider;
30  import org.bouncycastle.util.encoders.Hex;
31  import org.cesecore.certificates.ocsp.exception.OcspFailureException;
32  
33  /**
34   * Data carrier that wraps the contents of an OCSPResp, since OCSPResp and many of its members aren't serializable. 
35   * 
36   * @version $Id: OcspResponseInformation.java 25974 2017-06-09 15:51:49Z anatom $
37   *
38   */
39  public class OcspResponseInformation implements Serializable {
40  
41      private static final long serialVersionUID = -4177593916232755218L;
42      private static final Logger log = Logger.getLogger(OcspResponseInformation.class);
43      private final byte[] ocspResponse;
44      private final long maxAge;
45      private boolean addCacheHeaders = true;
46      private boolean explicitNoCache = false;
47      private Long nextUpdate = null;
48      private Long thisUpdate = null;
49      private String responseHeader = null;
50      private X509Certificate signerCert = null;
51  
52      public OcspResponseInformation(OCSPResp ocspResponse, long maxAge, X509Certificate signerCert) throws OCSPException {
53          try {
54              this.ocspResponse = ocspResponse.getEncoded();
55          } catch (IOException e) {
56              throw new IllegalStateException("Unexpected IOException caught when encoding ocsp response.", e);
57          }
58          this.maxAge = maxAge;
59          this.signerCert = signerCert;
60          /*
61           * This may seem like a somewhat odd place to perform the below operations (instead of in the end servlet which demanded 
62           * this object), but BouncyCastle (up to 1.47) is  a bit shy about making their classes serializable. This means that 
63           * OCSPResp can't be transmitted, neither can many of the objects it contains such as SingleResp. Luckily we only need 
64           * these classes for the diagnostic operations performed below, so we can sum up the result in the boolean member 
65           * addCacheHeaders.  If BC choose to change their policy, the below code can med moved to a more logical location. 
66           *  -mikek
67           */
68          if (ocspResponse.getResponseObject() == null) {
69              if (log.isDebugEnabled()) {
70                  log.debug("Will not add cache headers for response to bad request.");
71              }
72              addCacheHeaders = false;
73          } else {
74              final BasicOCSPResp basicOCSPResp = (BasicOCSPResp)ocspResponse.getResponseObject();
75              final SingleResp[] singleRespones = basicOCSPResp.getResponses();
76              if (singleRespones.length != 1) {
77                  if (log.isDebugEnabled()) {
78                      log.debug("Will not add RFC 5019 cache headers: reponse contains multiple embedded responses.");
79                  }
80                  addCacheHeaders = false;
81              } else if (singleRespones[0].getNextUpdate() == null) {
82                  if (log.isDebugEnabled()) {
83                      log.debug("Will not add RFC 5019 cache headers: nextUpdate isn't set.");
84                  }
85                  addCacheHeaders = false;
86              } else if (basicOCSPResp.hasExtensions() && basicOCSPResp.getExtension(OCSPObjectIdentifiers.id_pkix_ocsp_nonce) != null) {
87                  if (log.isDebugEnabled()) {
88                      log.debug("Will not add RFC 5019 cache headers: response contains a nonce.");
89                  }
90                  addCacheHeaders = false;
91              } else {
92                  nextUpdate = singleRespones[0].getNextUpdate().getTime();
93                  thisUpdate = singleRespones[0].getThisUpdate().getTime();
94                  try {
95                      responseHeader = new String(Hex.encode(MessageDigest.getInstance("SHA-1", BouncyCastleProvider.PROVIDER_NAME).digest(this.ocspResponse)));
96                  } catch (NoSuchProviderException e) {
97                      throw new OcspFailureException("Bouncycastle was not available as a provider", e);
98                  } catch (NoSuchAlgorithmException e) {
99                      throw new OcspFailureException("SHA-1 was not an available algorithm for MessageDigester", e);
100                 }
101             }
102             if (addCacheHeaders && singleRespones[0].getCertStatus() instanceof UnknownStatus) {
103                 explicitNoCache = true;
104             }
105         }
106     }
107 
108     public byte[] getOcspResponse() {
109         return ocspResponse;
110     }
111 
112     /**
113      * @return duration in milliseconds how long the reponse should be cacheable
114      */
115     public long getMaxAge() {
116         return maxAge;
117     }
118 
119     public boolean shouldAddCacheHeaders() {
120         return addCacheHeaders;
121     }
122 
123     /**
124      * @return Date.getTime() long value for nextUpdate time
125      */
126     public long getNextUpdate() {
127         return nextUpdate;
128     }
129 
130     /**
131      * @return Date.getTime() long value for thisUpdate time
132      */
133     public long getThisUpdate() {
134         return thisUpdate;
135     }
136 
137     public String getResponseHeader() {
138         return responseHeader;
139     }
140 
141     /** @return true if we explicitly should state that the response should not be cached. */
142     public boolean isExplicitNoCache() {
143         return explicitNoCache;
144     }
145 
146     public X509Certificate getSignerCert() {
147         return signerCert;
148     }
149 
150 }
