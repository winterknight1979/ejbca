13  package org.cesecore.certificates.ocsp;
14  
15  import java.security.cert.X509Certificate;
16  
17  import org.bouncycastle.cert.ocsp.OCSPException;
18  import org.cesecore.certificates.ocsp.exception.MalformedRequestException;
19  import org.cesecore.certificates.ocsp.logging.AuditLogger;
20  import org.cesecore.certificates.ocsp.logging.TransactionLogger;
21  
22  /**
23   * This interface is used to generate OCSP responses.
24   * 
25   * @see <a href="https://www.cesecore.eu/mediawiki/index.php/Functional_Specifications_(ADV_FSP)#OCSP_Response_Generation">OCSP Response Generation</a>
26   * 
27   * @version $Id: OcspResponseGeneratorSession.java 22476 2015-12-17 13:01:45Z jeklund $
28   * 
29   */
30  public interface OcspResponseGeneratorSession {
31  
32      /**
33       * This method delivers an OCSP response to a given request, as provided in the byte[] parameter.
34       * 
35       * @param requestBytes a byte array representing an encoded OCSPRequest.
36       * @param requestCertificates An array of Certificates from the original HttpServletRequest
37       * @param remoteAddress Remote address, most likely extracted from the HttpServletRequest
38       * @param xForwardedFor Value of X-Forwarded-For header if it was present in the request.
39       * @param requestUrl URL
40       * @param auditLogger The AuditLogger to use for this transaction
41       * @param transactionLogger The TransactionLogger to use for this transaction
42       * 
43       * @return a signed and encoded OCSPResponse wrapped in an OcspResponseInformation object
44       * @throws MalformedRequestException if the request byte array was invalid.
45       * @throws OCSPException if OCSP response generation fails
46       */
47      OcspResponseInformation getOcspResponse(byte[] requestBytes,
48              X509Certificate[] requestCertificates, String remoteAddress, String xForwardedFor, StringBuffer requestUrl, AuditLogger auditLogger,
49              TransactionLogger transactionLogger) throws MalformedRequestException, OCSPException;
50      
51      /** Reloads the cache of OCSP signers. */
52      void reloadOcspSigningCache();
53      
54      /** Reloads the cache of OCSP extensions (including extension specific caches, e.g. the CT OCSP response extensions cache). */
55      void reloadOcspExtensionsCache();
56  
57      /** Clears CT fail fast cache. If CT is not supported in this build, then it does nothing. */
58      void clearCTFailFastCache();
59  }
