13  package org.cesecore.certificates.endentity;
14  
15  import java.util.ArrayList;
16  import java.util.HashMap;
17  import java.util.List;
18  import java.util.Map;
19  
20  /**
21   * 
22   * Constants for users and certificates. Constants for Type of user: Type is constructed as a mask
23   * since one user can be of several types. To test a user type:
24   * <pre>
25   * if (((type &amp; USER_ENDUSER) == USER_ENDUSER) &amp;&amp; ((type &amp; USER_CAADMIN) == USER_ADMINISTOR) || ...
26   *    ...
27   * </pre>
28   * Bit usage: bits 0-7   (1:st byte):  user types bits 8-15  (2:nd byte):  unused bits 16-23 (3:rd
29   * byte):  unused bits 24-30 (4:th byte):  unused Constants for certificates are simple integer
30   * types. Constants for Token Types Token type is constructed of integer constants since only one
31   * 
32   * @version $Id: EndEntityTypes.java 17625 2013-09-20 07:12:06Z netmackan $
33   *
34   */
35  public enum EndEntityTypes  { 
36      INVALID(0x0, "Dummy type."),
37      ENDUSER(0x1, "This is an end user certificate (default)."),
38      ADMINISTRATOR(0x40, "This user is an administrator."),
39      KEYRECOVERABLE(0x80, "This users keystores are key recoverable."),
40      SENDNOTIFICATION(0x100, "Notification will be sent to this users emailaddress"), 
41      PRINT(0x200, "Registration data will be printed for this user");
42      
43      private static final Map<Integer, EndEntityTypes> lookupTable;
44      
45      private int hexCode;
46      private String description;
47      
48      static {
49          lookupTable = new HashMap<Integer, EndEntityTypes>();
50          for(EndEntityTypes type : EndEntityTypes.values()) {
51              lookupTable.put(type.hexValue(), type);
52          }
53      }
54      
55      private EndEntityTypes(int hexCode, String description) {
56          this.hexCode = hexCode;
57          this.description = description;
58      }
59  
60      /**
61       * 
62       * @return the hexcode value for this end entity type.
63       */
64      public int hexValue() {
65          return hexCode;
66      }
67      
68      public String getDescription() {
69          return description;
70      }
71      
72      public boolean equals(EndEntityTypes endEntityType) {
73          return hexCode == endEntityType.hexValue();
74      }
75      
76      public static EndEntityTypes[] getTypesFromHexCode(int hexCode) {
77          List<EndEntityTypes> result = new ArrayList<EndEntityTypes>();
78          for(EndEntityTypes type : EndEntityTypes.values()) {
79              if(type.isContainedInType(hexCode)) {
80                  result.add(type);
81              }
82          }
83          return result.toArray(new EndEntityTypes[result.size()]);
84      }
85      
86      public static int combineAll(EndEntityTypes[] types) {
87          int result = 0;
88          for(EndEntityTypes type : types) {
89              result = type.addTo(result);
90          }
91          return result;
92      }
93      
94      /**
95       * Filters this end entity type to the given parameter by performing a binary AND operation on
96       * their hex values and returning whatever type which matches the result.  
97       * 
98       * @param otherHexCode hex code EndEntityType to filter with
99       * @return the corresponding EndEntityType, or null if not found.
100      */
101     public boolean isContainedInType(int otherHexCode) {
102         return (hexCode & otherHexCode) == hexCode; 
103     }
104 
105 
106     /**
107      * Joins this end entity type to the given parameter by performing a binary OR operation on
108      * their hex valyes and returning whatever type which matches the result
109      * 
110      * @param otherHexCode hex code of an EndEntityType to join with
111      * @return the resulting amalgam 
112      */
113     public int addTo(int otherHexCode) {   
114         return hexCode | otherHexCode;
115     }
116     
117     /**
118      * Removes this type from the given hexcode
119      * 
120      * @param otherHexCode the hexcode to remove this type from
121      * @return the resulting hexcode
122      */
123     public int removeFromType(int otherHexCode) {
124         return otherHexCode & ~this.hexCode;
125     }
126     
127     /**
128      * Simple utility method for creating an EndEntityType with a single type.
129      * 
130      * @return a new EndEntityType.
131      */
132     public EndEntityType toEndEntityType() {
133         return new EndEntityType(this);
134     }
135 }
