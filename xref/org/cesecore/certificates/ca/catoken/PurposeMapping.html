16  import java.util.Hashtable;
17  import java.util.Map;
18  import java.util.Properties;
19  import java.util.Set;

23  /** Class wraps keystring properties. The properties passed in to it can contain fields as the constants:
24   * 
25   * The values of the properties consists of purposeproperty and keyalias
26   * <pre>
27   *    certSignKey fooalias02
28   *    crlSignKey fooalias02
29   *    keyEncryptKey fooencalias
30   *    hardTokenEncrypt fooencalias
31   *    previousCertSignKey fooalias01
32   *    nextCertSignKey fooalias03
33   *    testKey testalias
34   *    defaultKey defaultalias
35   * </pre>
36   *  When the strings are added they are mapped to different key purposes, CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN etc. 
37   *  When the method getString is called with CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN it will return fooalias, if getString is called
38   *  with a key purpose that was not specified, for example CryptoTokenConstants.CAKEYPURPOSE_KEYENCRYPT it will return defaultalias.
39   *  
40   *   The returned values are supposed to be used to get keys for different aliases from a keystore.
41   * 
42   * @version $Id: PurposeMapping.java 22852 2016-02-23 17:37:16Z jeklund $
43   */
44  public final class PurposeMapping {
45      
46      final private Map<Integer, String> map;
47      final private Map<Integer, String> keymap;
48      final private String defaultKeyAlias;
49      
50      /** 
51       * Key string properties with entries consisting of one of the fixed key strings mapping to a key alias. The key alias is user defined.
52  	 * <pre>
53  	 *    certSignKey fooalias02
54  	 *    crlSignKey fooalias02
55  	 *    keyEncryptKey fooencalias
56  	 *    hardTokenEncrypt fooencalias
57  	 *    previousCertSignKey fooalias01
58  	 *    nextCertSignKey fooalias03
59  	 *    testKey testalias
60  	 *    defaultKey defaultalias
61  	 * </pre>
62       * 
63       * @param properties key string properties
64       */
65      public PurposeMapping(final Properties properties) {
66      	/** Map of keypurpose integer (CATokenConstants.CAKEYPURPOSE_CERTSIGN) and alias string as defined in Properties */
67      	map = new Hashtable<Integer, String>();
68      	/** Map of keypurpose integer (CATokenConstants.CAKEYPURPOSE_CERTSIGN) and key purpose string (CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING)
69      	 * for the properties defined in Properties */
70      	keymap = new Hashtable<Integer, String>();
71      	String defaultKeyAliasTmp = null;
72      	if (properties != null) {
73      		defaultKeyAliasTmp = properties.getProperty(CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING);
74      		addKey(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING,
75      				CATokenConstants.CAKEYPURPOSE_CERTSIGN,
76      				properties);
77      		addKey(CATokenConstants.CAKEYPURPOSE_CRLSIGN_STRING,
78      				CATokenConstants.CAKEYPURPOSE_CRLSIGN,
79      				properties);
80      		addKey(CATokenConstants.CAKEYPURPOSE_KEYENCRYPT_STRING,
81      				CATokenConstants.CAKEYPURPOSE_KEYENCRYPT,
82      				properties);
83      		addKey(CATokenConstants.CAKEYPURPOSE_TESTKEY_STRING,
84      				CATokenConstants.CAKEYPURPOSE_KEYTEST,
85      				properties);
86      		addKey(CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT_STRING,
87      				CATokenConstants.CAKEYPURPOSE_HARDTOKENENCRYPT,
88      				properties);    		
89      		addKey(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_PREVIOUS,
90      				CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS,
91      				properties);    		
92      		addKey(CATokenConstants.CAKEYPURPOSE_CERTSIGN_STRING_NEXT,
93      				CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT,
94      				properties);    		
95      	}
96      	defaultKeyAlias = defaultKeyAliasTmp!=null ? defaultKeyAliasTmp.trim() : null;
97      } 
98      private void addKey(final String keyPurposeString, final int purpose, final Properties properties) {
99          String alias = properties.getProperty(keyPurposeString);
100         if (alias!=null) {
101             alias = alias.trim();
102             if (!alias.isEmpty()) {
103                 map.put(Integer.valueOf(purpose), alias);
104                 keymap.put(Integer.valueOf(purpose), keyPurposeString);
105             }
106         }
107     }
108     /** Returns which key alias string is used for a certain key purpose. 
109      * For example for CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN would either a key alias as defined by the property "CAKEYPURPOSE_CERTSIGN_STRING myCertSignKey" (myCertSignKey) 
110      * or null be returned. null is returned if no CAKEYPURPOSE_CERTSIGN_STRING (certSignKey) property was specified by the user.
111      * @param purpose purpose
112      * @return alias
113      */ 
114     public String getAlias(final int purpose) {
115         String alias;
116         try {
117             alias = map.get(Integer.valueOf(purpose));
118         } catch (Exception e) {
119             alias = null;
120         }
121         if (alias!=null && !alias.isEmpty()) {
122             return alias;
123         }
124         // Special handling of these two key purposes, because if they do not exist, very strange things can happen 
125         // if we claim that our "defaultKey" is the previous or next signing key, when it in fact is not.
126         if (purpose != CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS && purpose != CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT) {
127         	return defaultKeyAlias;
128         }
129         return null;
130     }
131     /** Returns which property key is used for a certain key purpose. 
132      * For example for CryptoTokenConstants.CAKEYPURPOSE_CERTSIGN would either CAKEYPURPOSE_CERTSIGN_STRING (certSignKey) 
133      * or CAKEYPURPOSE_DEFAULT_STRING (defaultKey) be returned.
134      * Special handling is for CERTSIGN_PREVIOUS and CERTSIGN_NEXT. If they can not be found, the defaultKey is _not_ returned.
135      * @param purpose purpose
136      * @return property
137      */ 
138     public String getPurposeProperty(final int purpose) {
139         String s;
140         try {
141             s = keymap.get(Integer.valueOf(purpose));
142         } catch(Exception e) {
143             s = null;
144         }
145         if ( s!=null && s.length()>0 ) {
146             return s;
147         }
148         // Special handling of these two key purposes, because if they do not exist, very strange things can happen 
149         // if we claim that our "defaultKey" is the previous or next signing key, when it in fact is not.
150         if ((purpose != CATokenConstants.CAKEYPURPOSE_CERTSIGN_PREVIOUS) && (purpose != CATokenConstants.CAKEYPURPOSE_CERTSIGN_NEXT)) {
151         	return CATokenConstants.CAKEYPURPOSE_DEFAULT_STRING;
152         }
153         return null;
154     }
155     
156     /** Returns an array with all key aliases that have been registered in this mapping.
157      * 
158      * @return String[] with key aliases
159      */
160     public String[] getAliases() {
161         final Set<String> set = new HashSet<>(map.values());
162         if (defaultKeyAlias != null) {
163             set.add(defaultKeyAlias);
164         }
165         return set.toArray(new String[set.size()]);
166     }
167     
168     public String toString() {
169     	return map.toString();
170     }
171     
172     /** @param alias alias
173      * @return true if the provided alias is mapped exactly once */
174     public boolean isAliasMappedForSinglePurpose(final String alias) {
175         if (alias==null) {
176             return false;
177         }
178         boolean mappedOnce = false;
179         for (final int keyPupose : CATokenConstants.ALL_KEY_PURPOSES) {
180             if (alias.equals(getAlias(keyPupose))) {
181                 if (mappedOnce) {
182                     return false;
183                 }
184                 mappedOnce = true;
185             }
186         }
187         return mappedOnce;
188     }
