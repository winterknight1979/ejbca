48      /** Issuing certificates with 'notAfter' greater than this value throws an exception. */
49      private static Date TOO_LATE_EXPIRE_DATE;
50      static {
51          final String value = CesecoreConfiguration.getCaTooLateExpireDate();
52          try {
53              TOO_LATE_EXPIRE_DATE = ValidityDate.parseCaLatestValidDateTime(value);
54          } catch (Exception e) {
55              final String newValue = ValidityDate.formatAsISO8601(new Date(Long.MAX_VALUE), ValidityDate.TIMEZONE_SERVER);
56              TOO_LATE_EXPIRE_DATE = ValidityDate.parseCaLatestValidDateTime(newValue);
57              log.warn("cesecore.properties ca.toolateexpiredate '" + value + "' could not be parsed Using default value '"+newValue+"'.", e);
58          }
59      }

61      /** 
62       * Validity offset in milliseconds (offset for the 'notBefore' value)
63       * The default start date is set 10 minutes back to avoid some problems with unsynchronized clocks.
64       */
65      private static long DEFAULT_VALIDITY_OFFSET;
66      static {
67          final String value = CesecoreConfiguration.getCertificateValidityOffset();
68          try {
69              DEFAULT_VALIDITY_OFFSET = SimpleTime.getSecondsFormat().parseMillis( value);
70          } catch (Exception e) {
71              // Use old value for compatibility reasons!
72              DEFAULT_VALIDITY_OFFSET = -10L * 60 * 1000;
73              log.warn("cesecore.properties certificate.validityoffset '" + value + "' could not be parsed as relative time string. Using default value '-10m' = -60000ms", e);
74          }
75      }

113     /** Constructor that injects the reference point (now). This constructor mainly is used for unit testing. 
114      * @param now Now
115      * @param subject Subject 
116      * @param certProfile Profila
117      * @param notBefore Start date
118      * @param notAfter End date
119      * @param cacert CA Certificate
120      * @param isRootCA  bool
121      * @param isLinkCertificate bool 
122      * @throws IllegalValidityException on fail */
123 	public CertificateValidity(Date now, final EndEntityInformation subject, final CertificateProfile certProfile, 
124 			final Date notBefore, final Date notAfter, final Certificate cacert, final boolean isRootCA, final boolean isLinkCertificate) throws IllegalValidityException {
125 		if (log.isDebugEnabled()) {
126 		    log.debug("Requested notBefore: "+notBefore);
127 			log.debug("Requested notAfter: "+notAfter);
128 			if (null != subject.getExtendedInformation()) {
129 			    log.debug("End entity extended information 'notBefore': "+subject.getExtendedInformation().getCustomData(ExtendedInformation.CUSTOM_STARTTIME));
130 			}
131 			if (null != subject.getExtendedInformation()) {
132                 log.debug("End entity extended information 'notAfter': "+subject.getExtendedInformation().getCustomData(ExtendedInformation.CUSTOM_ENDTIME));
133             }
134 			log.debug("Default validty offset: "+DEFAULT_VALIDITY_OFFSET);
135 			log.debug("Certificate profile validty: "+certProfile.getEncodedValidity());
136 			log.debug("Certificate profile use validty offset: "+certProfile.getUseCertificateValidityOffset());
137 			log.debug("Certificate profile validty offset: "+certProfile.getCertificateValidityOffset());
138 			log.debug("Certificate profile use expiration restrictions for weekdays: "+certProfile.getUseExpirationRestrictionForWeekdays());
139 			log.debug("Certificate profile expiration restrictions weekdays: "+Arrays.toString(certProfile.getExpirationRestrictionWeekdays()));
140 			log.debug("Certificate profile expiration restrictions for weekdays before: "+certProfile.getExpirationRestrictionForWeekdaysExpireBefore());
141 		}
142 		if ( TOO_LATE_EXPIRE_DATE==null ) {
143 		    throw new IllegalStateException("ca.toolateexpiredate in cesecore.properties is not a valid date.");
144 		}
145         
146 		// ECA-3554 add the offset
147 		now = getNowWithOffset(now, certProfile);
148 		if (log.isDebugEnabled()) {
149 		    log.debug("Using new start time including offset: " + now);
150 		}
151          
152 		// Find out what start and end time to actually use..
153         if (certProfile.getAllowValidityOverride()) {
154             // First Priority has information supplied in Extended information object. This allows RA-users to set the time-span.
155             // Second Priority has the information supplied in the method arguments
156             firstDate = getExtendedInformationStartTime(now, subject);
157             if (firstDate == null) {
158             	firstDate = notBefore;
159             }
160             if ((lastDate = getExtendedInformationEndTime(now, subject)) == null) {
161             	lastDate = notAfter;
162             }    	
163             if (log.isDebugEnabled()) {
164                 log.debug("Allow validity override, notBefore: "+firstDate);
165                 log.debug("Allow validity override, notAfter: "+lastDate);
166             }
167         }
168         // Third priority: If nothing could be set by external information have the default  3 is default values
169         if (firstDate == null) {
170         	firstDate = now;
171         }
172         Date certProfileLastDate = new Date(getCertificateProfileValidtyEndDate(certProfile, firstDate));
173         // Limit validity: ECA-5330 Apply expiration restriction for weekdays 
174         if (certProfile.getUseExpirationRestrictionForWeekdays() && isRelativeTime(certProfile.getEncodedValidity())) {
175             log.info("Applying expiration restrictions for weekdays: " + Arrays.asList(certProfile.getExpirationRestrictionWeekdays()));
176             try {
177                 final Date newDate = ValidityDate.applyExpirationRestrictionForWeekdays(certProfileLastDate, 
178                     certProfile.getExpirationRestrictionWeekdays(), certProfile.getExpirationRestrictionForWeekdaysExpireBefore());
179                 if (!firstDate.before(newDate)) {
180                     log.warn("Expiration restriction of certificate profile could not be applied because it's before start date!");    
181                 } else if (!TOO_LATE_EXPIRE_DATE.after(newDate)) {
182                     log.warn("Expiration restriction of certificate profile could not be applied because it's after latest possible end date!");
183                 } else {
184                     certProfileLastDate = newDate;
185                 }
186             } catch(Exception e) {
187                 log.warn("Expiration restriction of certificate profile could not be applied!");
188             }
189         }
190         //If it is a link certificate that we create, we use the old ca's expire date, as requested, as link certificate expire date
191         if (isLinkCertificate) {
192             lastDate = notAfter;
193         }
194         if (lastDate == null) {
195         	lastDate = certProfileLastDate;
196         }
197         // Limit validity: Do not allow last date to be before first date
198         if (lastDate.before(firstDate)) {
199 			log.info(intres.getLocalizedMessage("createcert.errorinvalidcausality",firstDate,lastDate));
200         	Date tmp = lastDate;
201         	lastDate = firstDate;
202         	firstDate = tmp;
203         }
204 		// Limit validity: We do not allow a certificate to be valid before the current date, i.e. not back dated start dates
205         // Unless allowValidityOverride is set, then we allow everything
206         // So this check is probably completely unneeded and can never be true
207     	if (firstDate.before(now) && !certProfile.getAllowValidityOverride()) {
208 			log.error(intres.getLocalizedMessage("createcert.errorbeforecurrentdate",firstDate,subject.getUsername()));
209     		firstDate = now;
210     		// Update valid length from the profile since the starting point has changed
211 			certProfileLastDate = new Date(getCertificateProfileValidtyEndDate(certProfile, firstDate));
212     		// Update lastDate if we use maximum validity
213     	}
214 		// Limit validity: We do not allow a certificate to be valid after the the validity of the certificate profile
215     	if (lastDate.after(certProfileLastDate)) {
216     		log.info(intres.getLocalizedMessage("createcert.errorbeyondmaxvalidity",lastDate,subject.getUsername(),certProfileLastDate));
217     		lastDate = certProfileLastDate;
218     	}
219 		// Limit validity: We do not allow a certificate to be valid after the the validity of the CA (unless it's RootCA during renewal)
220     	if (cacert != null && !isRootCA) {
221     	    final Date caNotAfter = CertTools.getNotAfter(cacert);
222     	    if (lastDate.after(caNotAfter)) {
223     	        log.info(intres.getLocalizedMessage("createcert.limitingvalidity", lastDate.toString(), caNotAfter));
224     	        lastDate = caNotAfter;
225     	    }
226     	}
227     	// Limit validity: We do not allow a certificate to be valid before the the CA becomes valid (unless it's RootCA during renewal)
228     	if (cacert != null && !isRootCA) {
229     	    final Date caNotBefore = CertTools.getNotBefore(cacert);
230     	    if (firstDate.before(caNotBefore)) {
231     	        log.info(intres.getLocalizedMessage("createcert.limitingvaliditystart", firstDate.toString(), caNotBefore));
232     	        firstDate = caNotBefore;
233     	    }
234         }
235         if ( !lastDate.before(CertificateValidity.TOO_LATE_EXPIRE_DATE) ) {
236         	String msg = intres.getLocalizedMessage("createcert.errorbeyondtoolateexpiredate", lastDate.toString(), CertificateValidity.TOO_LATE_EXPIRE_DATE.toString()); 
237         	log.info(msg);
238             throw new IllegalValidityException(msg);
239         }
240 	}

259      * Gets the validity end date for the certificate using the certificate profiles encoded validity.
260      * @param profile the certificate profile
261      * @param firstDate the start time.
262      * @return the encoded validity.
263      */
264 	@SuppressWarnings("deprecation")
265     private long getCertificateProfileValidtyEndDate(CertificateProfile profile, Date firstDate) {
266         final String encodedValidity = profile.getEncodedValidity();
267         Date date = null;
268         if (StringUtils.isNotBlank(encodedValidity)) {
269             date = ValidityDate.getDate( encodedValidity, firstDate);
270         } else {
271             date = ValidityDate.getDateBeforeVersion661(profile.getValidity(),firstDate);
272         }
273         return date.getTime();
274 	}

277 	 * Offsets the certificates 'notBefore' (reference point) with the global offset or the offset of the certificate profile.
278 	 * @param now the reference point
279 	 * @param profile the certificate profile
280 	 * @return the offset reference point
281 	 */
282 	private Date getNowWithOffset(final Date now, final CertificateProfile profile) {
283         Date result = null;
284         if (profile.getUseCertificateValidityOffset()) {
285             final String offset = profile.getCertificateValidityOffset();
286             try {
287                 result = new Date(now.getTime() + SimpleTime.parseMillies(offset));
288                 if (log.isDebugEnabled()) {
289                     log.debug( "Using validity offset by certificate profile: " + offset);
290                 }
291             } catch(NumberFormatException e) {
292                 log.warn("Could not parse certificate validity offset " + offset + "; using default " + DEFAULT_VALIDITY_OFFSET);
293             }
294         } else {
295             result = new Date(now.getTime() + DEFAULT_VALIDITY_OFFSET);
296             if (log.isDebugEnabled()) {
297                 log.debug( "Using validity offset by cesecore.properties: " + SimpleTime.toString(DEFAULT_VALIDITY_OFFSET, SimpleTime.TYPE_DAYS));
298             }
299         }
300         return result;
301 	}

304 	 * Gets the start time by the extended entity information.
305 	 * @param now the reference point.
306 	 * @param subject the end entity information.
307 	 * @return Start time
308 	 */
309 	private Date getExtendedInformationStartTime(final Date now, final EndEntityInformation subject) {
310 	    Date result = null;
311         final ExtendedInformation extendedInformation = subject.getExtendedInformation();
312         if (extendedInformation != null) {
313             result = parseExtendedInformationEncodedValidity(now, extendedInformation.getCustomData(ExtendedInformation.CUSTOM_STARTTIME));
314         }
315         if (log.isDebugEnabled()) {
316             log.debug("Using ExtendedInformationStartTime: " + result);
317         }
318         return result;
319 	}

341 	 * Checks that the PrivateKeyUsagePeriod of the certificate is valid at this time
342 	 * @param cert Certificate
343 
344 	 * @throws CAOfflineException if PrivateKeyUsagePeriod either is not valid yet or has expired, exception message gives details
345 	 */
346 	public static void checkPrivateKeyUsagePeriod(final X509Certificate cert) throws CAOfflineException {
347 	    checkPrivateKeyUsagePeriod(cert, new Date());
348 	}
349 	
350     public static void checkPrivateKeyUsagePeriod(final X509Certificate cert, final Date checkDate) throws CAOfflineException {
351         if (cert != null) {
352             final PrivateKeyUsagePeriod pku = CertTools.getPrivateKeyUsagePeriod(cert);
353             if (pku != null) {
354                 final ASN1GeneralizedTime notBefore = pku.getNotBefore();
355                 final Date pkuNotBefore;
356                 final Date pkuNotAfter;
357                 try {
358                     if (notBefore == null) {
359                         pkuNotBefore = null;
360                     } else {
361                         pkuNotBefore = notBefore.getDate();
362                     }
363                     if (log.isDebugEnabled()) {
364                         log.debug("PrivateKeyUsagePeriod.notBefore is " + pkuNotBefore);
365                     }
366                     if (pkuNotBefore != null && checkDate.before(pkuNotBefore)) {
367                         final String msg = intres.getLocalizedMessage("createcert.privatekeyusagenotvalid", pkuNotBefore.toString(), cert
368                                 .getSubjectDN().toString());
369                         if (log.isDebugEnabled()) {
370                             log.debug(msg);
371                         }
372                         throw new CAOfflineException(msg);
373                     }
374                     final ASN1GeneralizedTime notAfter = pku.getNotAfter();
375 
376                     if (notAfter == null) {
377                         pkuNotAfter = null;
378                     } else {
379                         pkuNotAfter = notAfter.getDate();
380                     }
381                 } catch (ParseException e) {
382                     throw new IllegalStateException("Could not parse dates.", e);
383                 } 
384                 if (log.isDebugEnabled()) {
385                     log.debug("PrivateKeyUsagePeriod.notAfter is " + pkuNotAfter);
386                 }
387                 if (pkuNotAfter != null && checkDate.after(pkuNotAfter)) {
388                     final String msg = intres.getLocalizedMessage("createcert.privatekeyusageexpired", pkuNotAfter.toString(), cert.getSubjectDN().toString());
389                     if (log.isDebugEnabled()) {
390                         log.debug(msg);
391                     }
392                     throw new CAOfflineException(msg);
393                 }
394             } else if (log.isDebugEnabled()) {
395                 log.debug("No PrivateKeyUsagePeriod available in certificate.");
396             }
397         } else if (log.isDebugEnabled()) {
398             log.debug("No CA certificate available, not checking PrivateKeyUsagePeriod.");       
399         }
400     }

402     /**
403      * Checks if the encoded validity is an ISO8601 date or a relative time.
404      * @param encodedValidity the validity
405      * @return Boolean.TRUE if it is a relative time, Boolean.FALSE if it is an ISO8601 date, otherwise NULL.
406      * @See {@link org.cesecore.util.ValidityDate ValidityDate}
407      * @See {@link org.cesecore.util.SimpleTime SimpleTime}
408      */
409     private static final Boolean isRelativeTime(final String encodedValidity) {
410         try {
411         	// Try the most likely setting first, for fail-fast
412             SimpleTime.parseMillies(encodedValidity);
413             return Boolean.TRUE;
414         } catch(NumberFormatException nfe) {
415             // NOOP
416         }
417         try {
418             ValidityDate.parseAsIso8601(encodedValidity);
419             return Boolean.FALSE;
420         } catch(ParseException e) {
421             return null;
422         }
423     }

425     /**
426      * Parses the entity extended information start and end time format and offsets it with the reference point.
427      * @param now the reference point
428      * @param timeString the value in form of 'days:hours:minutes'
429      * @return the parse value offset with now (reference point).
430      */
431     private static final Date parseExtendedInformationEncodedValidity(final Date now, final String timeString) {
432         Date result = null;
433         if (timeString != null) {
434             if (timeString.matches("^\\d+:\\d?\\d:\\d?\\d$")) {
435                 final String[] endTimeArray = timeString.split(":");
436                 long relative = (Long.parseLong(endTimeArray[0])*24*60 
437                               + Long.parseLong(endTimeArray[1])*60 
438                               + Long.parseLong(endTimeArray[2])) * 60 * 1000;
439                 result = new Date(now.getTime() + relative);
440             } else {
441                 try {
442                     // Try parsing data as "yyyy-MM-dd HH:mm" assuming UTC
443                     result = ValidityDate.parseAsUTC(timeString);
444                 } catch (ParseException e) {
445                     log.error(intres.getLocalizedMessage("createcert.errorinvalidstarttime",timeString));
446                 }
447             }
448             if ((log.isDebugEnabled())) {
449                 log.debug("Time string by end entity extended Information: "+result);
450             }
451         }
452         return result;
453     }
