174         if (ca != null) {
175             final int cryptoTokenId = ca.getCAToken().getCryptoTokenId();
176             if (!authorizationSession.isAuthorized(admin, StandardRules.CAADD.resource(), CryptoTokenRules.USE.resource() + "/" + cryptoTokenId)) {
177                 String msg = intres.getLocalizedMessage("caadmin.notauthorizedtoaddca", admin.toString(), Integer.valueOf(ca.getCAId()));
178                 throw new AuthorizationDeniedException(msg);
179             }
180             CAInfo cainfo = ca.getCAInfo();
181             // The CA needs a name and a subject DN in order to store it

183                 throw new CAExistsException("Null CA name or SubjectDN. Name: '"+ca.getName()+"', SubjectDN: '"+ca.getSubjectDN()+"'.");
184             }
185             if (findByName(cainfo.getName()) != null) {
186                 String msg = intres.getLocalizedMessage("caadmin.caexistsname", cainfo.getName());
187                 throw new CAExistsException(msg);
188             }
189             if (findById(ca.getCAId()) != null) {
190                 String msg = intres.getLocalizedMessage("caadmin.caexistsid", Integer.valueOf(ca.getCAId()));
191                 throw new CAExistsException(msg);
192             }
193             final CAData/CAData.html#CAData">CAData caData = new CAData(cainfo.getSubjectDN(), cainfo.getName(), cainfo.getStatus(), ca);
194             entityManager.persist(caData);
195             caIDCache.forceCacheExpiration(); // Clear ID cache so this one will be reloaded as well.
196             String msg = intres.getLocalizedMessage("caadmin.addedca", ca.getCAId(), cainfo.getName(), cainfo.getStatus());
197             final Map<String, Object> details = new LinkedHashMap<String, Object>();
198             details.put("msg", msg);
199             details.put("tokenproperties", ca.getCAToken().getProperties());
200             details.put("tokensequence", ca.getCAToken().getKeySequence());
201             logSession.log(EventTypes.CA_CREATION, EventStatus.SUCCESS, ModuleTypes.CA, ServiceTypes.CORE, admin.toString(), String.valueOf(ca.getCAId()), null, null, details);

254     @Override
255     public void editCA(final AuthenticationToken admin, final CA ca, boolean auditlog) throws CADoesntExistsException, AuthorizationDeniedException {
256         if (ca != null) {
257             if (log.isTraceEnabled()) {
258                 log.trace(">editCA (CA): "+ca.getName());
259             }
260             final CA orgca = getCAInternal(ca.getCAId(), null, true);
261             // Check if we can edit the CA (also checks authorization)
262             assertAuthorizationAndTarget(admin, ca.getName(), ca.getSubjectDN(), ca.getCAToken().getCryptoTokenId(), orgca);
263             if (auditlog) {
264                 // Get the diff of what changed
265                 final Map<Object, Object> diff = orgca.diff(ca);
266                 String msg = intres.getLocalizedMessage("caadmin.editedca", ca.getCAId(), ca.getName(), ca.getStatus());
267                 // Use a LinkedHashMap because we want the details logged (in the final log string) in the order we insert them, and not randomly
268                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
269                 details.put("msg", msg);
270                 for (Map.Entry<Object,Object> entry : diff.entrySet()) {
271                     details.put(entry.getKey().toString(), entry.getValue().toString());
272                 }
273                 details.put("tokenproperties", ca.getCAToken().getProperties());
274                 details.put("tokensequence", ca.getCAToken().getKeySequence());
275                 logSession.log(EventTypes.CA_EDITING, EventStatus.SUCCESS, ModuleTypes.CA, ServiceTypes.CORE,admin.toString(), String.valueOf(ca.getCAId()), null, null, details);
276             }
277             if (log.isTraceEnabled()) {
278                 log.trace("<editCA (CA): "+ca.getName());
279             }
280             // Store it
281             mergeCa(ca);
282         } else {
283             log.debug("Trying to edit null CA, nothing done.");
284         }

323 		assertAuthorizationAndTargetWithNewSubjectDn(admin, name, subjectDN, cryptoTokenId, ca);

329 	/** Ensure that the caller is authorized to the CA we are about to edit and that the CA name matches. 
330 	 * @param admin Admin
331 	 * @param name Name
332 	 * @param subjectDN DN 
333 	 * @param cryptoTokenId ID
334 	 * @param ca CA
335 	 * @throws CADoesntExistsException If CA not found 
336 	 * @throws AuthorizationDeniedException If access denied*/
337     private void assertAuthorizationAndTargetWithNewSubjectDn(AuthenticationToken admin, final String name, final String subjectDN, final int cryptoTokenId, final CA ca)
338             throws CADoesntExistsException, AuthorizationDeniedException {
339         // Check if we are authorized to edit CA and authorization to specific CA
340         if (cryptoTokenId == ca.getCAToken().getCryptoTokenId() || cryptoTokenId==0) {
341             if (!authorizationSession.isAuthorized(admin, StandardRules.CAEDIT.resource(), StandardRules.CAACCESS.resource() + ca.getCAId())) {
342                 String msg = intres.getLocalizedMessage("caadmin.notauthorizedtoeditca", admin.toString(), Integer.valueOf(ca.getCAId()));
343                 throw new AuthorizationDeniedException(msg);
344             }
345         } else {
346             // We only need to check usage authorization if we change CryptoToken reference (and not to 0 which means "removed").
347             if (!authorizationSession.isAuthorized(admin, StandardRules.CAEDIT.resource(), StandardRules.CAACCESS.resource() + ca.getCAId(), CryptoTokenRules.USE.resource() + "/" + cryptoTokenId)) {
348                 String msg = intres.getLocalizedMessage("caadmin.notauthorizedtoeditca", admin.toString(), Integer.valueOf(ca.getCAId()));
349                 throw new AuthorizationDeniedException(msg);
350             }
351         }
352         // The CA needs the same name and subject DN in order to store it
353         if (name == null || subjectDN == null) {
354             throw new CADoesntExistsException("Null CA name or SubjectDN");
355         } else if (!StringUtils.equals(name, ca.getName())) {
356             throw new CADoesntExistsException("Not same CA name.");
357         }
358     }

486     @Override
487     public void removeCA(final AuthenticationToken admin, final int caid) throws AuthorizationDeniedException {
488         // check authorization
489         if (!authorizationSession.isAuthorized(admin, StandardRules.CAREMOVE.resource())) {
490             String msg = intres.getLocalizedMessage("caadmin.notauthorizedtoremoveca", admin.toString(), Integer.valueOf(caid));
491             throw new AuthorizationDeniedException(msg);
492         }
493         // Get CA from database if it does not exist, ignore
494         CAData cadata = findById(Integer.valueOf(caid));
495         if (cadata != null) {
496             // Remove CA
497             entityManager.remove(cadata);
498             // Invalidate CA cache to refresh information
499             CaCache.INSTANCE.removeEntry(caid);
500             caIDCache.forceCacheExpiration(); // Clear ID cache so this one will be reloaded as well.
501             final String detailsMsg = intres.getLocalizedMessage("caadmin.removedca", Integer.valueOf(caid), cadata.getName());
502             logSession.log(EventTypes.CA_DELETION, EventStatus.SUCCESS, ModuleTypes.CA, ServiceTypes.CORE,admin.toString(), String.valueOf(caid), null, null, detailsMsg);
503         }
504     }

506     @Override
507     public void renameCA(final AuthenticationToken admin, final String oldname, final String newname) throws CAExistsException,
508             CADoesntExistsException, AuthorizationDeniedException {
509         // Get CA from database
510         CAData cadata = findByNameOrThrow(oldname);
511         // Check authorization, to rename we need remove (for the old name) and add for the new name)
512         if (!authorizationSession.isAuthorized(admin, StandardRules.CAREMOVE.resource(), StandardRules.CAADD.resource())) {
513             String msg = intres.getLocalizedMessage("caadmin.notauthorizedtorenameca", admin.toString(), cadata.getCaId());
514             throw new AuthorizationDeniedException(msg);
515         }
516         if (findByName(newname) == null) {
517             // The new CA doesn't exist, it's okay to rename old one.
518             cadata.setName(newname);
519             // Invalidate CA cache to refresh information
520             int caid = cadata.getCaId().intValue();
521             CaCache.INSTANCE.removeEntry(caid);
522             caIDCache.forceCacheExpiration(); // Clear ID cache so this one will be reloaded as well.
523             final String detailsMsg = intres.getLocalizedMessage("caadmin.renamedca", oldname, cadata.getCaId(), newname);
524             logSession.log(EventTypes.CA_RENAMING, EventStatus.SUCCESS, ModuleTypes.CA, ServiceTypes.CORE,admin.toString(), String.valueOf(caid), null, null, detailsMsg);
525         } else {
526             throw new CAExistsException("CA " + newname + " already exists.");
527         }
528     }

663     /**
664      * Internal method for getting CA, to avoid code duplication. Tries to find the CA even if the CAId is wrong due to CA certificate DN not being
665      * the same as CA DN. Uses CACache directly if configured to do so in ejbca.properties.
666      *
667      * Note! No authorization checks performed in this internal method
668      *
669      * @param caid
670      *            numerical id of CA (subjectDN.hashCode()) that we search for, or -1 if a name is to be used instead
671      * @param name
672      *            human readable name of CA, used instead of caid if caid == -1, can be null if caid != -1
673      * @param fromCache if we should use the CA cache or return a new, decoupled, instance from the database, to be used when you need
674      *             a completely distinct object, for edit, and not a shared cached instance.
675      * @return CA value object, or null if it doesn't exist.
676      */
677 	private CA getCAInternal(int caid, final String name, boolean fromCache) {
678 	    if (log.isTraceEnabled()) {
679 	        log.trace(">getCAInternal: " + caid + ", " + name);
680 	    }
681 	    Integer caIdValue = Integer.valueOf(caid);
682 	    if (caid == -1) {
683 	        caIdValue = CaCache.INSTANCE.getNameToIdMap().get(name);
684 	    }
685 	    CA ca = null;
686 	    if (fromCache && caIdValue!=null) {
687 	        ca = getCa(caIdValue.intValue());
688 	        if (ca != null && hasCAExpiredNow(ca)) {
689 	            // CA has expired, re-read from database with the side affect that the status will be updated
690 	            ca = getCAData(caid, name).getCA();
691 	        }
692 	    } else {
693             CAData caData = getCAData(caid, name);
694             if (caData != null) {
695                 ca = caData.getCA();
696             }
697         }
698 	    if (log.isTraceEnabled()) {
699 	        log.trace("<getCAInternal: " + caid + ", " + name);
700 	    }
701 	    return ca;
702 	}

705      * Checks if the CA certificate has expired (or is not yet valid) since last check.
706      * Logs an info message first time that the CA certificate has expired, or every time when not yet valid.
707 	 * @param ca CA
708      *
709      * @return the true if the CA is expired
710      */
711     private boolean hasCAExpiredNow(final CA ca) {
712         boolean expired = false;
713         // Check that CA hasn't expired.
714         try {
715             CertTools.checkValidity(ca.getCACertificate(), new Date());
716         } catch (CertificateExpiredException cee) {
717             // Signers Certificate has expired, we want to make sure that the
718             // status in the database is correctly EXPIRED for this CA
719             // Don't set external CAs to expired though, because they should always be treated as external CAs
720             if (ca.getStatus()!=CAConstants.CA_EXPIRED && ca.getStatus()!=CAConstants.CA_EXTERNAL) {
721                 log.info(intres.getLocalizedMessage("caadmin.caexpired", ca.getSubjectDN()) + " " + cee.getMessage());
722                 expired = true;
723             }
724         } catch (CertificateNotYetValidException e) {
725             // Signers Certificate is not yet valid.
726             log.warn(intres.getLocalizedMessage("caadmin.canotyetvalid", ca.getSubjectDN()) + " " + e.getMessage());
727         }
728         return expired;
729     }

731     /**
732      * Internal method for getting CAData. Tries to find the CA even if the CAId is wrong due to CA certificate DN not being the same as CA DN.
733      *
734      * The returned CAData object is guaranteed to be upgraded and these upgrades merged back to the database.
735      *
736      * @param caid numerical id of CA (subjectDN.hashCode()) that we search for, or -1 of a name is to ge used instead
737      * @param name human readable name of CA, used instead of caid if caid == -1, can be null of caid != -1
738      * 
739      * @return the CA, or null if it was not found
740      */
741     private CAData getCAData(final int caid, final String name)  {
742         CAData cadata = null;
743         if (caid != -1) {
744             cadata = upgradeAndMergeToDatabase(findById(Integer.valueOf(caid)));
745             if (log.isDebugEnabled() && cadata == null) {
746                 log.debug("Unable to get CAData with ID (from SubjectDN): "+caid);
747             }
748         } else {
749             cadata = upgradeAndMergeToDatabase(findByName(name));
750             if (log.isDebugEnabled() && cadata == null) {
751                 log.debug("Unable to get CAData with name: "+name);
752             }
753         }
754         if (cadata == null) {
755             // We should never get to here if we are searching for name, in any
756             // case if the name does not exist, the CA really does not exist
757             // We don't have to try to find another mapping for the CAId
758             if (caid != -1) {
759                 // subject DN of the CA certificate might not have all objects
760                 // that is the DN of the certificate data.
761                 final Integer oRealCAId = CACacheHelper.getCaCertHash(Integer.valueOf(caid));
762                 // has the "real" CAID been mapped to the certificate subject
763                 // hash by a previous call?
764                 if (oRealCAId != null) {
765                     // yes, using cached value of real caid.
766                 	if (log.isDebugEnabled()) {
767                 		log.debug("Found a mapping from caid "+caid+" to realCaid "+oRealCAId);
768                 	}
769                     cadata = findById(oRealCAId);
770                 } else {
771                     // no, we have to search for it among all CA certs
772                     for (final CAData currentCaData : findAll()) {
773                         final CAData currentUpgradedCaData = upgradeAndMergeToDatabase(currentCaData);
774                         final Certificate caCert = currentUpgradedCaData.getCA().getCACertificate();
775                         if (caCert != null && caid == CertTools.getSubjectDN(caCert).hashCode()) {
776                             cadata = currentUpgradedCaData; // found.
777                             // Do also cache it if someone else is needing it later
778                         	if (log.isDebugEnabled()) {
779                         		log.debug("Adding a mapping from caid "+caid+" to realCaid "+cadata.getCaId());
780                         	}
781                             CACacheHelper.putCaCertHash(Integer.valueOf(caid), Integer.valueOf(cadata.getCaId()));
782                         }
783                         if (cadata != null) {
784                             break;
785                         }
786                     }
787                 }
788             }
789             if (cadata == null) {
790                 String msg;
791                 if (caid != -1) {
792                     msg = intres.getLocalizedMessage("caadmin.canotexistsid", Integer.valueOf(caid));
793                 } else {
794                     msg = intres.getLocalizedMessage("caadmin.canotexistsname", name);
795                 }
796                 log.info(msg);
797             }
798         }
799         return cadata;
800     }

822     /** @param caId ID
823      * @return the CA object, from the database (including any upgrades) is necessary */
824     private CA getCa(int caId) {
825         final Integer realCAId = CACacheHelper.getCaCertHash(Integer.valueOf(caId));
826         if (realCAId!=null) {
827             // Since we have found a cached "real" CA Id and the cache will use this one (if cached)
828             caId = realCAId.intValue();
829         }
830         // 1. Check (new) CaCache if it is time to sync-up with database
831         if (CaCache.INSTANCE.shouldCheckForUpdates(caId)) {
832             log.debug("CA with ID " + caId + " will be checked for updates.");
833             // 2. If cache is expired or missing, first thread to discover this reloads item from database and sends it to the cache         
834             CAData caData = getCAData(caId, null);
835             if (caData != null) {
836                 final int digest = caData.getProtectString(0).hashCode();
837                 // Special for splitting out the CAToken and committing it..
838                 // Since getCAData has already run upgradeAndMergeToDatabase we can just get the CA here..
839                 CA ca = caData.getCA();
840                 if (ca != null) {
841                     // Note that we store using the "real" CAId in the cache.
842                     CaCache.INSTANCE.updateWith(caData.getCaId(), digest, ca.getName(), ca);
843                 }
844                 // Since caching might be disabled, we return the value returned from the database here
845                 return ca;
846             } else {
847                 // Ensure that it is removed from cache
848                 CaCache.INSTANCE.removeEntry(caId);
849             }
850             // 3. The cache compares the database data with what is in the cache
851             // 4. If database is different from cache, replace it in the cache
852         }
853         // 5. Get CA from cache (or null) and be merry
854         return CaCache.INSTANCE.getEntry(caId);
855     }

857     @TransactionAttribute(TransactionAttributeType.REQUIRED)
858     @Override
859     public int mergeCa(final CA ca) {
860         final int caId = ca.getCAId();
861         CAData caData = entityManager.find(CAData.class, caId);
862         if (caData == null) {
863             caData = new CAData(ca.getSubjectDN(), ca.getName(), ca.getStatus(), ca);
864         } else {
865             // It might be the case that the calling transaction has already loaded a reference to this object
866             // and hence we need to get the same one and perform updates on this object instead of trying to
867             // merge a new object.
868             caData.setCA(ca);
869         }
870         caData = entityManager.merge(caData);
871         // Since loading a CA is quite complex (populating CAInfo etc), we simple purge the cache here
872         CaCache.INSTANCE.removeEntry(caId);
873         caIDCache.forceCacheExpiration();
874         return caId;
875     }

877     /** Performs upgrades on the entity if needed within a transaction. 
878      * @param cadata Data
879      * @return Upgraded data */
880     private CADatacesecore/certificates/ca/CAData.html#CAData">CAData upgradeAndMergeToDatabase(CAData cadata) {
881         if (cadata == null) {
882             return null;
883         }
884         CAData caDataReturn = cadata;
885         final LinkedHashMap<Object, Object> caDataMap = cadata.getDataMap();
886         // If CA-data is upgraded we want to save the new data, so we must get the old version before loading the data
887         // and perhaps upgrading
888         final float oldversion = ((Float) caDataMap.get(UpgradeableDataHashMap.VERSION)).floatValue();
889         // Perform "live" upgrade from 5.0.x and earlier
890         boolean adhocUpgrade = adhocUpgradeFrom50(cadata.getCaId().intValue(), caDataMap, cadata.getName());
891         if (adhocUpgrade) {
892             // Convert map into storage friendly format now since we changed it
893             cadata.setDataMap(caDataMap);
894         }
895         // Fetching the CA object will trigger UpgradableHashMap upgrades
896         CA ca = cadata.getCA();
897         if (ca != null) {
898             final boolean expired = hasCAExpiredNow(ca);
899             if (expired) {
900                 ca.setStatus(CAConstants.CA_EXPIRED);
901             }
902             final boolean upgradedExtendedService = ca.upgradeExtendedCAServices();
903             // Compare old version with current version and save the data if there has been a change
904             final boolean upgradeCA = (Float.compare(oldversion, ca.getVersion()) != 0);
905             if (adhocUpgrade || upgradedExtendedService || upgradeCA || expired) {
906                 if (log.isDebugEnabled()) {
907                     log.debug("Merging CA to database. Name: " + cadata.getName() + ", id: " + cadata.getCaId() +
908                             ", adhocUpgrade: " + adhocUpgrade+", upgradedExtendedService: " + upgradedExtendedService +
909                             ", upgradeCA: " + upgradeCA + ", expired: " + expired);
910                 }
911                 ca.getCAToken();
912                 final int caId = caSession.mergeCa(ca);
913                 caDataReturn = entityManager.find(CAData.class, caId);
914             }            
915         }
916         return caDataReturn;
917     }

920      * Extract keystore or keystore reference and store it as a CryptoToken. Add a reference to the keystore.
921      * @param caid ID
922      * @param data Data
923      * @param caName Name
924      * @return true if any changes where made
925      */
926     @SuppressWarnings("unchecked")
927     @Deprecated // Remove when we no longer need to support upgrades from 5.0.x
928     private boolean adhocUpgradeFrom50(int caid, LinkedHashMap<Object, Object> data, String caName) {
929         HashMap<String, String> tokendata = (HashMap<String, String>) data.get(CA.CATOKENDATA);
930         if (tokendata.get(CAToken.CRYPTOTOKENID) != null) {
931             // Already upgraded
932             if (!CesecoreConfiguration.isKeepInternalCAKeystores()) {
933                 // All nodes in the cluster has been upgraded so we can remove any internal CA keystore now
934                 if (tokendata.get(CAToken.KEYSTORE)!=null) {
935                     tokendata.remove(CAToken.KEYSTORE);
936                     tokendata.remove(CAToken.CLASSPATH);
937                     log.info("Removed duplicate of upgraded CA's internal keystore for CA '" + caName + "' with id: " + caid);
938                     return true;
939                 }
940             } else {
941                 if (log.isDebugEnabled()) {
942                     log.debug("CA '" + caName + "' already has cryptoTokenId and will not have it's token split of to a different db table because db.keepinternalcakeystores=true: " + caid);
943                 }
944             }
945             return false;
946         }
947         // Perform pre-upgrade of CATokenData to correct classpath changes (org.ejbca.core.model.ca.catoken.SoftCAToken)
948         tokendata = (LinkedHashMap<String, String>) new CAToken(tokendata).saveData();
949         data.put(CA.CATOKENDATA, tokendata);
950         log.info("Pulling CryptoToken out of CA '" + caName + "' with id " + caid + " into a separate database table.");
951         final String str = tokendata.get(CAToken.KEYSTORE);
952         byte[] keyStoreData = null;
953         if (StringUtils.isNotEmpty(str)) {
954             keyStoreData = Base64.decode(str.getBytes());
955         }
956         String propertyStr = tokendata.get(CAToken.PROPERTYDATA);
957         final Properties prop = new Properties();
958         if (StringUtils.isNotEmpty(propertyStr)) {
959             try {
960                 // If the input string contains \ (backslash on windows) we must convert it to \\
961                 // Otherwise properties.load will parse it as an escaped character, and that is not good
962                 propertyStr = StringUtils.replace(propertyStr, "\\", "\\\\");
963                 prop.load(new ByteArrayInputStream(propertyStr.getBytes()));
964             } catch (IOException e) {
965                 log.error("Error getting CA token properties: ", e);
966             }
967         }
968         final String classpath = tokendata.get(CAToken.CLASSPATH);
969         if (log.isDebugEnabled()) {
970             log.debug("CA token classpath: " + classpath);
971         }
972         // Upgrade the properties value
973         final Properties upgradedProperties = PKCS11CryptoToken.upgradePropertiesFileFrom5_0_x(prop);
974         // If it is an P11 we are using and the library and slot are the same as an existing CryptoToken we use that CryptoToken's id.
975         int cryptoTokenId = 0;
976         if (PKCS11CryptoToken.class.getName().equals(classpath)) {
977             if (upgradedProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_TYPE)==null) {
978                 log.error("Upgrade of CA '" + caName + "' failed due to failed upgrade of PKCS#11 CA token properties.");
979                 return false;
980             }
981             for (final Integer currentCryptoTokenId : cryptoTokenSession.getCryptoTokenIds()) {
982                 final CryptoToken cryptoToken = cryptoTokenSession.getCryptoToken(currentCryptoTokenId.intValue());
983                 final Properties cryptoTokenProperties = cryptoToken.getProperties();
984                 if (StringUtils.equals(upgradedProperties.getProperty(PKCS11CryptoToken.SHLIB_LABEL_KEY), cryptoTokenProperties.getProperty(PKCS11CryptoToken.SHLIB_LABEL_KEY))
985                         && StringUtils.equals(upgradedProperties.getProperty(PKCS11CryptoToken.ATTRIB_LABEL_KEY), cryptoTokenProperties.getProperty(PKCS11CryptoToken.ATTRIB_LABEL_KEY))
986                         && StringUtils.equals(upgradedProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_VALUE), cryptoTokenProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_VALUE))
987                         && StringUtils.equals(upgradedProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_TYPE), cryptoTokenProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_TYPE))) {
988                     // The current CryptoToken point to the same HSM slot in the same way.. re-use this id!
989                     cryptoTokenId = currentCryptoTokenId.intValue();
990                     break;
991                 }
992             }
993         }
994         if (cryptoTokenId == 0) {
995             final String cryptoTokenName = "Upgraded CA CryptoToken for " + caName;
996             try {
997                 cryptoTokenId = cryptoTokenSession.mergeCryptoToken(CryptoTokenFactory.createCryptoToken(classpath, upgradedProperties, keyStoreData, caid, cryptoTokenName, true));
998             } catch (CryptoTokenNameInUseException e) {
999                 final String msg = "Crypto token name already in use upgrading (adhocUpgradeFrom50) crypto token for CA '"+caName+"', cryptoTokenName '"+cryptoTokenName+"'.";
1000                 log.info(msg, e);
1001                 throw new RuntimeException(msg, e);  // Since we have a constraint on CA names to be unique, this should never happen
1002             } catch (NoSuchSlotException e) {
1003                 final String msg = "Slot as defined by " + upgradedProperties.getProperty(PKCS11CryptoToken.SLOT_LABEL_VALUE) + " for CA '" + caName + "' could not be found.";
1004                 log.error(msg, e);
1005                 throw new RuntimeException(msg, e);
1006             }
1007         }
1008         // Mark this CA as upgraded by setting a reference to the CryptoToken if the merge was successful
1009         tokendata.put(CAToken.CRYPTOTOKENID, String.valueOf(cryptoTokenId));
1010         // Note: We did not remove the keystore in the CA properties here, so old versions running in parallel will still work
1011         log.info("CA '" + caName + "' with id " + caid + " is now using CryptoToken with cryptoTokenId " + cryptoTokenId);
1012         return true;
1013     }
