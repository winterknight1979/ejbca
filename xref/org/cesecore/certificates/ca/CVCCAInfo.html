30  /**
31   * Holds non-sensitive information about a CVC CA (Card Verifiable Certificate).
32   *
33   * @version $Id: CVCCAInfo.java 34456 2020-02-06 18:36:20Z anatom $
34   */
35  public class CVCCAInfo extends CAInfo {
36  
37  	private static final long serialVersionUID = 2L;
38  
39      /**
40       * This constructor can be used when creating a CA.
41       * This constructor uses defaults for the fields that are not specified.
42       * @param subjectdn DN
43       * @param name name
44       * @param status status
45       * @param certificateprofileid ID
46       * @param encodedValidity Validity
47       * @param signedby signer
48       * @param certificatechain chain 
49       * @param catoken token
50       * */
51      public CVCCAInfo(String subjectdn, String name, int status, int certificateprofileid, String encodedValidity, int signedby,
52              List<Certificate> certificatechain, CAToken catoken) {
53          this(subjectdn, name, status, new Date(), certificateprofileid,
54                  0, // defaultCertprofileId
55                  encodedValidity, null, // expire time
56                  CAInfo.CATYPE_CVC, signedby,
57                  certificatechain, // Certificate chain
58                  catoken, // CA token
59                  "", // Description
60                  -1, // Revocation reason
61                  null, // Revocation date
62                  1 * SimpleTime.MILLISECONDS_PER_DAY, // CRL period
63                  0L, // CRL issue interval
64                  10 * SimpleTime.MILLISECONDS_PER_MINUTE, // CRL overlap time
65                  0L, // Delta CRL period
66                  new ArrayList<Integer>(), // CRL publishers
67                  new ArrayList<Integer>(), // Key validators
68                  true, // Finish user
69                  new ArrayList<ExtendedCAServiceInfo>(), // Extended CA services
70                  new HashMap<ApprovalRequestType, Integer>(),
71                  true, // includeInHealthCheck
72                  true, // isDoEnforceUniquePublicKeys
73                  true, // isDoEnforceUniqueDistinguishedName
74                  false, // isDoEnforceUniqueSubjectDNSerialnumber
75                  false, // useCertReqHistory
76                  true, // useUserStorage
77                  true, // useCertificateStorage
78  				false // acceptRevocationNonExistingEntry
79              );
80      }
81  
82  	/**
83  	 * Constructor that should be used when creating CA and retrieving CA info.
84  	 * Please use the shorter form if you do not need to set all of the values.
85  	 * @param subjectDn DN
86  	 * @param name Name
87  	 * @param status Status
88  	 * @param updateTime Update
89  	 * @param certificateprofileid ID 
90  	 * @param defaultCertprofileId ID
91  	 * @param encodedValidity Validity
92  	 * @param expiretime Expiry
93  	 * @param catype Type of CA
94  	 * @param signedBy Signer
95  	 * @param certificatechain Chain 
96  	 * @param catoken Token
97  	 * @param description Description
98  	 * @param revocationReason Reason for revoking
99  	 * @param revocationDate Date of revoking
100 	 * @param crlperiod CRL period
101 	 * @param crlIssueInterval CRL Issue interval
102 	 * @param crlOverlapTime CRL Overlap
103 	 * @param deltacrlperiod CRL change
104 	 * @param crlpublishers Publisher of CRL
105 	 * @param keyValidators Validators
106 	 * @param finishuser boolean
107 	 * @param extendedcaserviceinfos info 
108 	 * @param approvals approvals
109 	 * @param includeInHealthCheck bool 
110 	 * @param _doEnforceUniquePublicKeys bool 
111 	 * @param _doEnforceUniqueDistinguishedName bool 
112 	 * @param _doEnforceUniqueSubjectDNSerialnumber bool
113 	 * @param _useCertReqHistory bool
114 	 * @param _useUserStorage bool
115 	 * @param _useCertificateStorage bool 
116 	 * @param _acceptRevocationNonExistingEntry bool  
117 	 */
118 	public CVCCAInfo(String subjectDn, String name, int status, Date updateTime, int certificateprofileid, int defaultCertprofileId,
119             String encodedValidity, Date expiretime, int catype, int signedBy, List<Certificate> certificatechain,
120 			CAToken catoken, String description, int revocationReason, Date revocationDate,
121 			long crlperiod, long crlIssueInterval, long crlOverlapTime, long deltacrlperiod,
122 			Collection<Integer> crlpublishers, Collection<Integer> keyValidators,
123 			boolean finishuser,Collection<ExtendedCAServiceInfo> extendedcaserviceinfos,
124 			Map<ApprovalRequestType, Integer> approvals,
125 			boolean includeInHealthCheck, boolean _doEnforceUniquePublicKeys,
126 			boolean _doEnforceUniqueDistinguishedName, boolean _doEnforceUniqueSubjectDNSerialnumber,
127 			boolean _useCertReqHistory, boolean _useUserStorage, boolean _useCertificateStorage, boolean _acceptRevocationNonExistingEntry) {
128 		this.subjectdn = CertTools.stringToBCDNString(StringTools.strip(subjectDn));
129 		this.caid = CertTools.stringToBCDNString(this.subjectdn).hashCode();
130 		this.name = name;
131 		this.status = status;
132 		this.updatetime = updateTime;
133 		setEncodedValidity(encodedValidity);
134 		this.expiretime = expiretime;
135 		this.catype = catype;
136 		this.signedby = signedBy;
137 		setCertificateChain(certificatechain);
138 		this.catoken = catoken;
139 		this.description = description;
140 		this.revocationReason = revocationReason;
141 		this.revocationDate = revocationDate;
142 		this.crlperiod = crlperiod;
143 		this.crlIssueInterval = crlIssueInterval;
144 		this.crlOverlapTime = crlOverlapTime;
145 		this.deltacrlperiod = deltacrlperiod;
146 		this.crlpublishers = crlpublishers;
147 		this.validators = keyValidators;
148 		this.finishuser = finishuser;
149 		this.certificateprofileid = certificateprofileid;
150 		this.defaultCertificateProfileId = defaultCertprofileId;
151 		this.extendedcaserviceinfos = extendedcaserviceinfos;
152 		setApprovals(approvals);
153 		this.includeInHealthCheck = includeInHealthCheck;
154 		this.doEnforceUniquePublicKeys = _doEnforceUniquePublicKeys;
155 		this.doEnforceUniqueDistinguishedName = _doEnforceUniqueDistinguishedName;
156 		this.doEnforceUniqueSubjectDNSerialnumber = _doEnforceUniqueSubjectDNSerialnumber;
157 		this.useCertReqHistory = _useCertReqHistory;
158         this.useUserStorage = _useUserStorage;
159         this.useCertificateStorage = _useCertificateStorage;
160         this.acceptRevocationNonExistingEntry = _acceptRevocationNonExistingEntry;
161 	}
162 
163 	/**
164 	 * Constructor that should be used when updating CA data.
165      * Used by the web. Jsp and stuff like that.
166 	 * @param caid CA ID
167 	 * @param encodedValidity Validity 
168 	 * @param catoken Token
169 	 * @param description Description
170 	 * @param crlperiod CRL period
171 	 * @param crlIssueInterval CRL Issue
172 	 * @param crlOverlapTime CRL Overlap
173 	 * @param deltacrlperiod CRL Delta
174 	 * @param crlpublishers CRL Publisher
175 	 * @param keyValidators Validators
176 	 * @param finishuser bool
177 	 * @param extendedcaserviceinfos info
178 	 * @param approvals approvals
179 	 * @param includeInHealthCheck bool 
180 	 * @param _doEnforceUniquePublicKeys bool 
181 	 * @param _doEnforceUniqueDistinguishedName bool 
182 	 * @param _doEnforceUniqueSubjectDNSerialnumber bool
183 	 * @param _useCertReqHistory bool
184 	 * @param _useUserStorage bool
185 	 * @param _useCertificateStorage bool 
186 	 * @param _acceptRevocationNonExistingEntry bool 
187 	 * @param defaultCertprofileId ID
188 	 */
189 	public CVCCAInfo(int caid, String encodedValidity, CAToken catoken, String description,
190 			long crlperiod, long crlIssueInterval, long crlOverlapTime, long deltacrlperiod,
191 			Collection<Integer> crlpublishers, Collection<Integer> keyValidators,
192 			boolean finishuser, Collection<ExtendedCAServiceInfo> extendedcaserviceinfos,
193 			Map<ApprovalRequestType, Integer> approvals,
194 			boolean includeInHealthCheck, boolean _doEnforceUniquePublicKeys,
195 			boolean _doEnforceUniqueDistinguishedName, boolean _doEnforceUniqueSubjectDNSerialnumber,
196 			boolean _useCertReqHistory, boolean _useUserStorage, boolean _useCertificateStorage, boolean _acceptRevocationNonExistingEntry,
197 			int defaultCertprofileId) {
198 		this.caid = caid;
199 		setEncodedValidity(encodedValidity);
200 		this.catoken = catoken;
201 		this.description = description;
202 		this.crlperiod = crlperiod;
203 		this.crlIssueInterval = crlIssueInterval;
204 		this.crlOverlapTime = crlOverlapTime;
205 		this.deltacrlperiod = deltacrlperiod;
206 		this.crlpublishers = crlpublishers;
207 		this.validators = keyValidators;
208 		this.finishuser = finishuser;
209 		this.extendedcaserviceinfos = extendedcaserviceinfos;
210 		setApprovals(approvals);
211 		this.includeInHealthCheck = includeInHealthCheck;
212 		this.doEnforceUniquePublicKeys = _doEnforceUniquePublicKeys;
213 		this.doEnforceUniqueDistinguishedName = _doEnforceUniqueDistinguishedName;
214 		this.doEnforceUniqueSubjectDNSerialnumber = _doEnforceUniqueSubjectDNSerialnumber;
215 		this.useCertReqHistory = _useCertReqHistory;
216         this.useUserStorage = _useUserStorage;
217         this.useCertificateStorage = _useCertificateStorage;
218         this.acceptRevocationNonExistingEntry = _acceptRevocationNonExistingEntry;
219         this.defaultCertificateProfileId = defaultCertprofileId;
220 	}
