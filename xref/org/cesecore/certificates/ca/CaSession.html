14  
15  import java.security.cert.Certificate;
16  import java.util.Collection;
17  import java.util.List;
18  import java.util.TreeMap;
19  
20  import org.cesecore.authentication.tokens.AuthenticationToken;
21  import org.cesecore.authorization.AuthorizationDeniedException;
22  import org.cesecore.certificates.certificate.CertificateWrapper;

24  /*! \mainpage The CESeCore project
25  *
26  * \section section1 Heading towards Common Criteria certification
27  *
28  * In order to make the CESeCore security core publicly available for
29  * integration in numerous security based applications, the CESeCore
30  * project aims to realize a CESeCore Common Criteria EAL4+
31  * certification.
32  *
33  * \section section2 Reusable Java library
34  *
35  * Taking on the form of a common security function Java library, the
36  * CESeCore security core will provide a reusable base for implementing
37  * third-party trustworthy systems. The ready-made CESeCore library
38  * simplifies development of secure software, and allows for Common
39  * Criteria certification to be easily extended to any related product
40  * or service.
41  *
42  * \section section3 Integrator friendly
43  *
44  * CESeCore integrators will be able to correct, improve and extend their
45  * applications at any time - without the need to perform frequent system
46  * re-evaluations, nor perform continous checks of the security functions
47  * implemented by the security core (including features like digital
48  signature
49  * creation/validation, digital certificate and CRLs creation, key management
50  * and maintenance of a secure audit log).
51  */

53  /**
54   * CRUD bean for creating, removing and retrieving CAs.
55   * 
56   * @version $Id: CaSession.java 29306 2018-06-21 14:07:02Z andresjakobs $
57   */
58  public interface CaSession {
59  
60      /** Adds a CA to the database 
61       * 
62       * @param admin AuthenticationToken of admin
63       * @param ca the CA to add
64       * @throws CAExistsException if CA already exists
65       * @throws AuthorizationDeniedException if not authorized to add CA
66       */
67       void addCA(AuthenticationToken admin, CA ca) throws CAExistsException, AuthorizationDeniedException;
68      
69      /** Changes a CA in the database. Can change mostly everything except caid, caname and subject DN. When editing a CA the CA token will usually be taken off line.
70       * So you need to activate the CA token after editing, if auto-activation of the CA token is not enabled. 
71       * 
72       * There's also CAAdminSession.editCA() which handles some special cases, e.g. changing the Subject DN of
73       * uninitialized CAs.
74       * 
75       * @param admin AuthenticationToken of admin
76       * @param cainfo the CAInfo to change values of the CA
77       * @throws CADoesntExistsException If CA not found
78       * @throws AuthorizationDeniedException If access denied
79       */
80      void editCA(final AuthenticationToken admin, final CAInfo cainfo) throws CADoesntExistsException, AuthorizationDeniedException;
81  
82      /**
83       * Method returning id's of all CA's in system.
84       * 
85       * @return a List (Integer) of CA id's
86       */
87      List<Integer> getAllCaIds();
88  
89      /**
90       * Method returning id's of all CA's available to the system that the
91       * administrator is authorized to, 
92       * 
93       * Does not log access control to all CAs it checks, because this does not 
94       * give access to the CAs but only returns IDs of CAs.
95       * 
96       * @param admin AuthenticationToken of admin
97       * @return a List&lt;Integer&gt; of available CA IDs
98       */
99       List<Integer> getAuthorizedCaIds(AuthenticationToken admin);
100      
101      /**
102       * Method returning names of all CA's available to the system that the
103       * administrator is authorized to, 
104       * 
105       * Does not log access control to all CAs it checks, because this does not 
106       * give access to the CAs but only returns names of CAs.
107       * 
108       * @param admin AuthenticationToken of admin
109       * @return a Collection&lt;String&gt; of available CA names
110       */
111      Collection<String> getAuthorizedCaNames(AuthenticationToken admin);
112      
113      /**
114       * Like {@link #getAuthorizedCaNames(AuthenticationToken)}, but returns a TreeMap which includes the CA Id as well.
115       * @param admin AuthenticationToken of admin
116       * @return a Collection&lt;String&gt; of available CA names
117       * @see #getAuthorizedCaNames
118       */
119      TreeMap<String,Integer> getAuthorizedCaNamesToIds(AuthenticationToken admin);
120      
121      /**
122       * Method returning info objects for all CA's available to the system that the administrator is authorized to. 
123       * 
124       * Does not log access control to all CAs it checks, because this does not 
125       * give access to the CAs but only returns CAInfo of CAs.
126       * 
127       * @param authenticationToken AuthenticationToken of admin
128       * @return a List&lt;CAInfo&gt; of authorized CAs
129       */
130      List<CAInfo> getAuthorizedCaInfos(AuthenticationToken authenticationToken);
131 
132      /**
133       * Method returning info objects for  all active CA's available to the system, i.e. not 
134       * having status "external", uninitialized or "waiting for certificate response" and that the
135       * administrator is authorized to, 
136       * 
137       * Does not log access control to all CAs it checks, because this does not 
138       * give access to the CAs but only returns CAInfo of CAs.
139       * 
140       * @param authenticationToken AuthenticationToken of admin
141       * @return a List&lt;CAInfo&gt; of authorized and enabled CAs
142       */
143      List<CAInfo> getAuthorizedAndEnabledCaInfos(AuthenticationToken authenticationToken);
144      
145      /**
146       * Method returning info objects for all non-external CA's available to the system, i.e. not 
147       * having status "external", and that the administrator is authorized to. 
148       * 
149       * Does not log access control to all CAs it checks, because this does not 
150       * give access to the CAs but only returns CAInfo of CAs.
151       * 
152       * @param authenticationToken AuthenticationToken of admin
153       * @return a List&lt;CAInfo&gt; of authorized and non-external CAs
154       */
155      List<CAInfo> getAuthorizedAndNonExternalCaInfos(AuthenticationToken authenticationToken);
156      
157     /**
158      * Method returning names of all CA's available to the system that the
159      * administrator is authorized to i.e. not having status "external" or
160      * "waiting for certificate response"
161      * 
162      * @param admin AuthenticationToken of admin
163      * @return a List&lt;String&gt; of available CA names
164      */
165     List<String> getActiveCANames(final AuthenticationToken admin);
166 
167     /**
168      * Returns a value object containing non-sensitive information about a CA
169      * give it's name.
170      * 
171      * @param admin administrator calling the method
172      * @param name human readable name of CA
173      * @return CAInfo value object, null if it doesn't exist
174      * @throws AuthorizationDeniedException if admin not authorized to CA 
175      */
176     CAInfo getCAInfo(AuthenticationToken admin, String name) throws AuthorizationDeniedException;
177 
178     /**
179      * Returns a value object containing non-sensitive information about a CA
180      * give it's name.
181      * 
182      * @param admin administrator calling the method
183      * @param caid numerical unique id of CA
184      * @return CAInfo value object, null if it doesn't exist
185      * @throws AuthorizationDeniedException if admin not authorized to CA 
186      */
187     CAInfo getCAInfo(AuthenticationToken admin, int caid) throws  AuthorizationDeniedException;
188 
189     /**
190      * Fetches the current certificate chain for a CA.
191      *
192      * <pre>
193      * <b>Authorization requirements:</b>
194      * - /administrator
195      * - /ca/&lt;ca in question&gt;
196      * </pre>
197      * @param authenticationToken administrator calling the method
198      * @param caName the unique name of the CA whose certificate chain should be returned.
199      * @return a list of X509 Certificates or CVC Certificates with the root certificate last, or an empty list if the CA's status is "Waiting for certificate response".
200      * @throws AuthorizationDeniedException if the client does not fulfill the authorization requirements specified above.
201      * @throws CADoesntExistsException if the CA with the CA name given as input does not exist.
202      */
203     Collection<CertificateWrapper> getCaChain(AuthenticationToken authenticationToken, String caName)
204             throws AuthorizationDeniedException, CADoesntExistsException;
205     
206     /**
207      * Method used to remove a CA from the system. You should first check that
208      * the CA isn't used by any EndEntity, Profile or AccessRule before it is
209      * removed. CADataHandler for example makes this check. 
210      * Should be used with care. If any certificate has been created with the CA use revokeCA
211      * instead and don't remove it.
212      * 
213      * @param admin AuthenticationToken of admin
214      * @param caid numerical unique id of CA
215      * 
216      * If the CA does not exist, nothing happens the method return silently.
217      * @throws AuthorizationDeniedException if not authorized to remove CA
218      */
219     void removeCA(AuthenticationToken admin, int caid) throws AuthorizationDeniedException;
220 
221     /**
222      * Renames the short name of CA (used in administrators interfaces). 
223      * This name does not have to be the same as SubjectDN and is only used for reference.
224      * 
225      * @param admin AuthenticationToken of admin
226      * @param oldname the name of the CA to rename
227      * @param newname the new name of the CA
228      * 
229      * @throws CAExistsException if the CA with newname already exists
230      * @throws CADoesntExistsException if the CA with oldname does not exist
231      * @throws AuthorizationDeniedException if not authorized to rename CA
232      */
233     void renameCA(AuthenticationToken admin, String oldname, String newname) throws CAExistsException, CADoesntExistsException, AuthorizationDeniedException;
234 
235     /**
236      * Check if a CA with given ID exists
237      * 
238      * @param caId the CA ID
239      * @return true if a CA with the given ID exists
240      */
241     boolean existsCa(int caId);
242 
243     /**
244      * Check if a CA with given name exists
245      * 
246      * @param name the CA name
247      * @return true if a CA with the given name exists
248      */
249     boolean existsCa(String name);
250 
251     /**
252      * Returns the future key/certificate rollover if a rollover is pending, or otherwise null.
253      * @param caid numerical unique id of CA
254      * @return Certificate 
255      * @throws CADoesntExistsException If no CA with the given CA Id exists.
256      */
257     Certificate getFutureRolloverCertificate(int caid) throws CADoesntExistsException;
258     
259 }
