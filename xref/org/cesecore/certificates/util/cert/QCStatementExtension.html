52      /** Returns true if the certificate contains a QC-statements extension.
53       * 
54       * @param cert Certificate containing the extension
55       * @return true or false.
56       */
57      public static boolean hasQcStatement(final Certificate cert) {
58      	boolean ret = false;
59          if (cert instanceof X509Certificate) {
60          	final X509Certificate x509cert = (X509Certificate) cert;
61          	final ASN1Primitive obj = getExtensionValue(x509cert, Extension.qCStatements.getId());
62  	        if (obj != null) {
63  	            ret = true;
64  	        }
65          }
66          return ret;
67      }
68      /** Returns all the 'statementId' defined in the QCStatement extension (rfc3739).
69       * 
70       * @param cert Certificate containing the extension
71       * @return Collection of String with the oid, for example "1.1.1.2", or empty Collection if no identifier is found, never returns null.
72       * @throws IOException if there is a problem parsing the certificate
73       */
74      public static Collection<String> getQcStatementIds(final Certificate cert) throws IOException {
75      	final ArrayList<String> ret = new ArrayList<String>();
76          if (cert instanceof X509Certificate) {
77          	final X509Certificate x509cert = (X509Certificate) cert;
78          	final ASN1Primitive obj = getExtensionValue(x509cert, Extension.qCStatements.getId());
79              if (obj == null) {
80                  return ret;
81              }
82              final ASN1Sequence seq = (ASN1Sequence)obj;
83              for (int i = 0; i < seq.size(); i++) {
84              	final QCStatement qc = QCStatement.getInstance(seq.getObjectAt(i));
85              	final ASN1ObjectIdentifier oid = qc.getStatementId();
86                  if (oid != null) {
87                      ret.add(oid.getId());
88                  }        	
89              }
90          }
91          return ret;
92      }
93      /** Returns the value limit ETSI QCStatement if present.
94       * 
95       * @param cert Certificate possibly containing the QCStatement extension
96       * @return String with the value and currency (ex '50000 SEK')or null if the extension is not present
97       * @throws IOException if there is a problem parsing the certificate
98       */
99      public static String getQcStatementValueLimit(final Certificate cert) throws IOException {
100     	String ret = null;
101         if (cert instanceof X509Certificate) {
102         	final X509Certificate x509cert = (X509Certificate) cert;
103         	final ASN1Primitive obj = getExtensionValue(x509cert, Extension.qCStatements.getId());
104 	        if (obj == null) {
105 	            return null;
106 	        }
107 	        final ASN1Sequence seq = (ASN1Sequence)obj;
108 	        MonetaryValue mv = null;
109 	        // Look through all the QCStatements and see if we have a stadard ETSI LimitValue
110 	        for (int i = 0; i < seq.size(); i++) {
111 	        	final QCStatement qc = QCStatement.getInstance(seq.getObjectAt(i));
112 	        	final ASN1ObjectIdentifier oid = qc.getStatementId();
113 	        	if ((oid != null) && oid.equals(ETSIQCObjectIdentifiers.id_etsi_qcs_LimiteValue)) {
114 	        		// We MAY have a MonetaryValue object here
115 	        		final ASN1Encodable enc = qc.getStatementInfo();
116 	        		if (enc != null) {
117 	        			mv = MonetaryValue.getInstance(enc);
118 	        			// We can break the loop now, we got it!
119 	        			break;
120 	        		}
121 	        	}
122 	        }
123 	        if (mv != null) {
124 	        	final BigInteger amount = mv.getAmount();
125 	        	final BigInteger exp = mv.getExponent();
126 	        	final BigInteger ten = BigInteger.valueOf(10);
127 	        	// A possibly gotcha here if the monetary value is larger than what fits in a long...
128 	        	final long value = amount.longValue() * (ten.pow(exp.intValue())).longValue();
129 	        	if (value < 0) {
130 	        		log.error("ETSI LimitValue amount is < 0.");
131 	        	}
132 	        	final String curr = mv.getCurrency().getAlphabetic();
133 	        	if (curr == null) {
134 	        		log.error("ETSI LimitValue currency is null");
135 	        	}
136 	        	if ( (value >= 0) && (curr != null) ) {
137 	        		ret = value + " "+curr;
138 	        	}
139 	        }
140         }
141     	return ret;    	
142     }

144     /** Returns the 'NameRegistrationAuthorities' defined in the QCStatement extension (rfc3739).
145      * 
146      * @param cert Certificate containing the extension
147      * @return String with for example 'rfc822Name=foo2bar.se, rfc822Name=bar2foo.se' etc. Supports email, dns and uri name, or null of no RAs are found.
148      * @throws IOException if there is a problem parsing the certificate
149      */
150     public static String getQcStatementAuthorities(final Certificate cert) throws IOException {
151         String ret = null;
152         if (cert instanceof X509Certificate) {
153         	final X509Certificate x509cert = (X509Certificate) cert;
154         	final ASN1Primitive obj = getExtensionValue(x509cert, Extension.qCStatements.getId());
155 	        if (obj == null) {
156 	            return null;
157 	        }
158 	        final ASN1Sequence seq = (ASN1Sequence)obj;
159 	        SemanticsInformation si = null;
160 	        // Look through all the QCStatements and see if we have a standard RFC3739 pkixQCSyntax
161 	        for (int i = 0; i < seq.size(); i++) {
162 	        	final QCStatement qc = QCStatement.getInstance(seq.getObjectAt(i));
163 	        	final ASN1ObjectIdentifier oid = qc.getStatementId();
164 	        	if ((oid != null) && (oid.equals(RFC3739QCObjectIdentifiers.id_qcs_pkixQCSyntax_v1) || oid.equals(RFC3739QCObjectIdentifiers.id_qcs_pkixQCSyntax_v2))) {
165 	        		// We MAY have a SemanticsInformation object here
166 	        		final ASN1Encodable enc = qc.getStatementInfo();
167 	        		if (enc != null) {
168 	        			si = SemanticsInformation.getInstance(enc);
169 	        			// We can break the loop now, we got it!
170 	        			break;
171 	        		}
172 	        	}
173 	        }
174 	        if (si != null) {
175 	        	final GeneralName[] gns = si.getNameRegistrationAuthorities();
176 	            if (gns == null) {
177 	                return null;
178 	            }
179 	            final StringBuilder strBuf = new StringBuilder(); 
180 	            for (int i = 0; i < gns.length; i++) {
181 	            	final GeneralName gn = gns[i];
182 	                if (strBuf.length() != 0) {
183 	                    // Append comma so we get nice formatting if there are more than one authority
184 	                    strBuf.append(", ");
185 	                }
186 	                final String str = getGeneralNameString(gn.getTagNo(), gn.getName());
187 	                if (str != null) {
188 	                    strBuf.append(str);
189 	                }
190 	            }
191 	            if (strBuf.length() > 0) {
192 	                ret = strBuf.toString();
193 	            }
194 	        }
195         }
196         return ret;
197     }

199     /** Assumes that the statementoid in the QcStatements Sequence, seq, is a String and extracts that value from position, pos, of sequence 
200      * @param seq ASN.1 sequence
201      * @param statementoid OIS
202      * @param pos position
203      * @return value
204      * @throws IOException  on error */
205     public static String getStatementStringValue(final ASN1Sequence seq, final String statementoid, final int pos) throws IOException {
206         // Look through all the QCStatements and see if we have a standard ETSI PDS URL
207         for (int i = 0; i < seq.size(); i++) {
208             final QCStatement qc = QCStatement.getInstance(seq.getObjectAt(i));
209             final ASN1ObjectIdentifier oid = qc.getStatementId();
210             if ((oid != null) && oid.toString().equals(statementoid)) {
211                 // We MUST have a URL and LANG here
212                 final ASN1Encodable enc = qc.getStatementInfo();
213                 if (enc != null) {
214                     ASN1Encodable str;
215                     if (enc instanceof ASN1Sequence) {
216                         ASN1Sequence valueseq = ASN1Sequence.getInstance(enc);
217                         str = valueseq.getObjectAt(pos);
218                     } else {
219                         str = enc; 
220                     }
221                     return str.toString();
222                 }
223             }
224         }
225         return null;
226     }
