136     /**
137      * Returns the valid components for the given DN type (Subject DN, Subject Alternative Name or Subject Directory Attributes)
138      * @param dnType DNFieldExtractor.TYPE_*
139      * @return List of valid components from DnComponents.*
140      */
141     public static List<String> getValidFieldComponents(final int dnType) {
142         switch (dnType) {
143         case DNFieldExtractor.TYPE_SUBJECTDN: return DnComponents.getDnProfileFields();
144         case DNFieldExtractor.TYPE_SUBJECTALTNAME: return DnComponents.getAltNameFields();
145         case DNFieldExtractor.TYPE_SUBJECTDIRATTR: return DnComponents.getDirAttrFields();
146         default: throw new IllegalStateException("Invalid DN type");
147         }
148     }

151         final String ret;
152         if (type == DNFieldExtractor.TYPE_SUBJECTDN) {
153             ret = DnComponents.getDnExtractorFieldFromDnId(field);
154         } else if (type == DNFieldExtractor.TYPE_SUBJECTALTNAME) {
155             ret = DnComponents.getAltNameExtractorFieldFromDnId(field);
156         } else {
157             ret = DnComponents.getDirAttrExtractorFieldFromDnId(field);
158         }
159         return ret;

163      * Looks up a DN Id (for use with DnComponents functions etc.) from a DN component.
164      * @param dnComponent Component, e.g. "CN". Not case sensitive.
165      * @param dnType DN type, e.g. DNFieldExtractor.TYPE_SUBJECTDN
166      * @return DN Id, or null if no such component exists for the given DN type.

168     public static Integer getDnIdFromComponent(final String dnComponent, final int dnType) {
169         switch (dnType) {
170         case DNFieldExtractor.TYPE_SUBJECTDN: return DnComponents.getDnIdFromDnName(dnComponent);
171         case DNFieldExtractor.TYPE_SUBJECTALTNAME: return DnComponents.getDnIdFromAltName(dnComponent);
172         case DNFieldExtractor.TYPE_SUBJECTDIRATTR: DnComponents.getDnIdFromDirAttr(dnComponent);
173         default: throw new IllegalStateException("Invalid DN type");
174         }

177     /**
178      * Fills the dnfields variable with dn (or altname or subject dir attrs) numerical ids and the value of the components 
179      * (i.e. the value of CN). Also populates fieldnumbers with number of occurances in dn
180      * 
181      * @param dn
182      *            DOCUMENT ME!
183      * @param type
184      *            DOCUMENT ME!
185      */
186     public final void setDN(final String dn, final int type) {
187 
188         this.type = type;
189         final ArrayList<Integer> ids;
190         if (type == TYPE_SUBJECTDN) {
191             ids = DnComponents.getDnDnIds();
192         } else if (type == TYPE_SUBJECTALTNAME) {
193             ids = DnComponents.getAltNameDnIds();
194         } else if (type == TYPE_SUBJECTDIRATTR) {
195             ids = DnComponents.getDirAttrDnIds();
196         } else {
197             ids = new ArrayList<>();
198         }
199         fieldnumbers = new HashMap<>();
200         for(Integer id : ids) {
201             fieldnumbers.put(id, 0);
202         }
203 
204         if ((dn != null) && !dn.equalsIgnoreCase("null")) {
205             dnfields = new HashMap<>();
206             try {
207                 final String[] dnexploded = LDAPDN.explodeDN(dn, false);
208                 for (int i = 0; i < dnexploded.length; i++) {
209                     boolean exists = false;
210 
211                     for(Integer id : ids) {
212                         Integer number = fieldnumbers.get(id);
213                         String field;
214                         if (type == TYPE_SUBJECTDN) {
215                             field = DnComponents.getDnExtractorFieldFromDnId(id.intValue());
216                         } else if (type == TYPE_SUBJECTALTNAME) {
217                             field = DnComponents.getAltNameExtractorFieldFromDnId(id.intValue());
218                         } else {
219                             field = DnComponents.getDirAttrExtractorFieldFromDnId(id.intValue());
220                         }
221                         final String dnex = dnexploded[i].toUpperCase();
222                         if (id.intValue() == DNFieldExtractor.URI) {
223                             // Fix up URI, which can have several forms
224                             if (dnex.indexOf(CertTools.URI.toUpperCase(Locale.ENGLISH) + "=") > -1) {
225                                 field = CertTools.URI.toUpperCase(Locale.ENGLISH) + "=";
226                             }
227                             if (dnex.indexOf(CertTools.URI1.toUpperCase(Locale.ENGLISH) + "=") > -1) {
228                                 field = CertTools.URI1.toUpperCase(Locale.ENGLISH) + "=";
229                             }
230                         }
231                        
232                         if (dnex.startsWith(field)) {
233                            
234                             exists = true;
235                             final String rdn;
236                             final String tmp;
237                             // LDAPDN.unescapeRDN don't like fields with just a key but no contents. Example: 'OU='
238                             if (dnexploded[i].charAt(dnexploded[i].length() - 1) != '=') {
239                                 tmp = LDAPDN.unescapeRDN(dnexploded[i]);
240                             } else {
241                                 tmp = dnexploded[i];
242                             }
243                             // We don't want the CN= (or whatever) part of the RDN
244                             if (tmp.toUpperCase().startsWith(field)) {
245                                 rdn = tmp.substring(field.length(), tmp.length());
246                             } else {
247                                 rdn = tmp;
248                             }
249 
250                             // Same code for TYPE_SUBJECTDN, TYPE_SUBJECTALTNAME and TYPE_SUBJECTDIRATTR and we will never get here
251                             // if it is not one of those types
252                             dnfields.put(Integer.valueOf((id.intValue() * BOUNDRARY) + number.intValue()), rdn);
253                             
254                             number = Integer.valueOf(number.intValue() + 1);
255                             fieldnumbers.put(id, number);
256                         }
257                     }
258                     if (!exists) {
259                         existsother = true;
260                     }
261                 }
262             } catch (Exception e) {
263                 log.warn("setDN: ", e);
264                 illegal = true;
265                 if (type == TYPE_SUBJECTDN) {
266                     dnfields.put(Integer.valueOf((CN * BOUNDRARY)), "Illegal DN : " + dn);
267                 } else if (type == TYPE_SUBJECTALTNAME) {
268                     dnfields.put(Integer.valueOf((RFC822NAME * BOUNDRARY)), "Illegal Subjectaltname : " + dn);
269                 } else if (type == TYPE_SUBJECTDIRATTR) {
270                     dnfields.put(Integer.valueOf((PLACEOFBIRTH * BOUNDRARY)), "Illegal Subjectdirectory attribute : " + dn);
271                 }
272             }
273         }
274     }

276     /**
277      * Returns the value of a certain DN component.
278      * 
279      * @param field
280      *            the DN component, one of the constants DNFieldExtractor.CN, ...
281      * @param number
282      *            the number of the component if several entries for this component exists, normally 0 fir the first
283      * 
284      * @return A String for example "PrimeKey" if DNFieldExtractor.O and 0 was passed, "PrimeKey" if DNFieldExtractor.DC and 0 was passed or "com" if
285      *         DNFieldExtractor.DC and 1 was passed. Returns an empty String "", if no such field with the number exists.
286      */
287     public String getField(final int field, final int number) {
288         String returnval = dnfields.get(Integer.valueOf((field * BOUNDRARY) + number));
289 
290         if (returnval == null) {
291             returnval = "";
292         }
293 
294         return returnval;
295     }

297     /**
298      * Returns a string representation of a certain DN component
299      * 
300      * @param field
301      *            the DN component, one of the constants DNFieldExtractor.CN, ...
302      * @return A String for example "CN=Tomas Gustavsson" if DNFieldExtractor.CN was passed, "DC=PrimeKey,DC=com" if DNFieldExtractor.DC was passed.
303      *            This string is escaped so it can be used in a DN string.
304      */
305     public String getFieldString(final int field) {
306         String retval = "";
307         String fieldname = DnComponents.getDnExtractorFieldFromDnId(field);
308         if (type != TYPE_SUBJECTDN) {
309             fieldname = DnComponents.getAltNameExtractorFieldFromDnId(field);
310         }
311         final int num = getNumberOfFields(field);
312         for (int i = 0; i < num; i++) {
313             if (retval.length() == 0) {
314                 retval += LDAPDN.escapeRDN(fieldname + getField(field, i));
315             } else {
316                 retval += "," + LDAPDN.escapeRDN(fieldname + getField(field, i));
317             }
318         }
319         return retval;
320     }

331     /**
332      * Returns the number of one kind of dn field.
333      * 
334      * @param field
335      *            the DN component, one of the constants DNFieldExtractor.CN, ...
336      * 
337      * @return number of components available for a field, for example 1 if DN is "dc=primekey" and 2 if DN is "dc=primekey,dc=com"
338      */
339     public int getNumberOfFields(final int field) {
340         Integer ret = fieldnumbers.get(Integer.valueOf(field));
341         if (ret == null) {
342             log.error("Not finding fieldnumber value for " + field);
343             ret = Integer.valueOf(0);
344         }
345         return ret.intValue();
346     }
