13  package org.cesecore.certificates.crl;
14  
15  import java.util.Date;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  
20  /** Session bean storing and retrieving CRLs
21   * 
22   * @version $Id: CrlStoreSession.java 25229 2017-02-09 12:58:10Z anatom $
23   */
24  public interface CrlStoreSession {
25  
26  	/**
27  	 * Retrieves the latest CRL issued by this CA.
28  	 * 
29  	 * @param issuerdn the CRL issuers DN (CAs subject DN)
30  	 * @param deltaCRL true to get the latest deltaCRL, false to get the latest complete CRL
31  	 * @return byte[] with DER encoded X509CRL or null of no CRLs have been issued.
32  	 */
33  	byte[] getLastCRL(String issuerdn, boolean deltaCRL);
34  
35      /**
36       * Retrieves a specific CRL issued by this CA.
37       * 
38       * @param issuerdn the CRL issuers DN (CAs subject DN)
39       * @param crlNumber a crlNumber of a complete, or delta, CRL
40       * @return byte[] with DER encoded X509CRL or null of no CRLs have been issued.
41       */
42      byte[] getCRL(String issuerdn, int crlNumber);
43  
44  	/**
45  	 * Retrieves the information about the latest CRL issued by this CA.
46  	 * Retrieves less information than getLastCRL, i.e. not the actual CRL data.
47  	 * 
48  	 * @param issuerdn the CRL issuers DN (CAs subject DN)
49  	 * @param deltaCRL true to get the latest deltaCRL, false to get the latest complete CRL
50  	 * @return CRLInfo of last CRL by CA or null if no CRL exists.
51  	 */
52  	CRLInfo getLastCRLInfo(String issuerdn, boolean deltaCRL);
53  
54  	/**
55  	 * Retrieves the information about the specified CRL. Retrieves less
56  	 * information than getLastCRL, i.e. not the actual CRL data.
57  	 * 
58  	 * @param fingerprint fingerprint of the CRL
59  	 * @return CRLInfo of CRL or null if no CRL exists.
60  	 */
61  	CRLInfo getCRLInfo(String fingerprint);
62  
63  	/**
64  	 * Retrieves the highest CRLNumber issued by the CA.
65  	 * 
66  	 * @param issuerdn the subjectDN of a CA certificate
67  	 * @param deltaCRL true to get the latest deltaCRL, false to get the latest complete CRL
68  	 * @return the highest CRL number of CRLs issued by issuerdn, or 0 if no CRLs were issued.
69  	 */
70  	int getLastCRLNumber(String issuerdn, boolean deltaCRL);
71  	
72  	/**
73       * Stores a CRL
74       *
75  	 * @param admin Administrator performing the operation
76       * @param incrl  The DER coded CRL to be stored.
77       * @param cafp   Fingerprint (hex) of the CAs certificate.
78       * @param number CRL number.
79       * @param issuerDN the issuer of the CRL
80       * @param thisUpdate when this CRL was created
81       * @param nextUpdate when this CRL expires
82       * @param deltaCRLIndicator -1 for a normal CRL and 1 for a deltaCRL
83       * 
84       * @throws CrlStoreException (rollback) if an error occured storing the CRL
85       * @throws AuthorizationDeniedException (rollback) if admin was not authorized to store CRL
86       */
87      void storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String issuerDN, Date thisUpdate, Date nextUpdate, int deltaCRLIndicator)
88      	throws CrlStoreException, AuthorizationDeniedException;
89  	
90  }
