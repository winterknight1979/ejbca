13  package org.cesecore.time;
14  
15  import java.io.Serializable;
16  import java.util.Date;
17  
18  /**
19   * This class encapsulates a Date object that represents a trusted time. It also
20   * provides information related to thhe trusted time source: accuracy and
21   * stratum
22   * 
23   * @version $Id: TrustedTime.java 17625 2013-09-20 07:12:06Z netmackan $
24   */
25  public class TrustedTime implements Serializable {
26  
27      private static final long serialVersionUID = 1L;
28      private static final Integer delta = 1;
29  
30      private String source;
31      private Double accuracy;
32      private Integer stratum;
33      private Long previousUpdate; //seconds
34      private Long nextUpdate; //seconds
35      private boolean sync = false;
36  
37      public TrustedTime() { }
38  
39      public Integer getStratum() { return stratum; }
40      public void setStratum(final Integer stratum) { this.stratum = stratum; }
41  
42      public Long getPreviousUpdate() { return this.previousUpdate; }
43      public Long getNextUpdate() { return this.nextUpdate; }
44  
45      public void setNextUpdate(Integer when, Integer poll) {
46          Long nextUpdate = Long.valueOf(((poll - when) + delta)*1000);
47          if(nextUpdate.longValue() <= 0) { 
48              nextUpdate = Long.valueOf(1); 
49          }
50  
51          if(this.nextUpdate != null) {
52              this.previousUpdate = this.nextUpdate;
53          } 
54  
55          this.nextUpdate = nextUpdate;
56      }
57  
58      public boolean isSync() { return this.sync; }
59      public void setSync(boolean sync) { this.sync = sync; }
60  
61      public Date getTime() { return new Date(); }
62  
63      public String getSource() { return this.source; }
64      public void setSource(String source) { this.source = source; }
65  
66  
67      public Double getAccuracy() { return accuracy; }
68      public void setAccuracy(final Double accuracy) { this.accuracy = accuracy; }
69  
70      public TrustedTime(final Double accuracy) {
71          this.accuracy = accuracy;
72      }
73  
74      public String toString() {
75          final StringBuilder sb = new StringBuilder();
76          sb.append(this.accuracy).append(";").append(this.stratum).append(";").
77              append(this.previousUpdate).append(";").append(this.nextUpdate).
78              append(";").append(this.sync).append(this.source);
79          return sb.toString();
80      }
81  }
