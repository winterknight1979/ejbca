13  package org.cesecore.dbprotection;
14  
15  import java.lang.reflect.InvocationTargetException;
16  
17  import org.apache.log4j.Logger;
18  
19  
20  /**
21   * Used as base class for JPA data beans that has a rowProtection column. The JPA class should extend this class and implement the simple methods:
22   *
23   * <pre>
24   * &#064;Transient
25   * &#064;Override
26   * String getProtectString(int version) {
27   *   return &quot;concatenation of fields to be integrity protected. Must be deterministic and can change with different version of rowprotection.&quot;;
28   *   // Example from CertificateProfileData
29   *   // StringBuilder build = new StringBuilder();
30   *   // What is important to protect here is the data that we define, id, name and certificate profile data
31   *   // rowVersion is automatically updated by JPA, so it's not important, it is only used for optimistic locking
32   *   // build.append(getId()).append(getCertificateProfileName()).append(getData());
33   *   // return build.toString();
34   * }
35   *
36   * &#064;Transient
37   * &#064;Override
38   * int getProtectVersion() {
39   *   return 1;
40   * }
41   *
42   * &#064;PrePersist
43   * &#064;PreUpdate
44   * &#064;Transient
45   * &#064;Override
46   * void protectData() {
47   *   super.protectData();
48   * }
49   * 
50   * &#064;PostLoad
51   * &#064;Transient
52   * &#064;Override
53   * void verifyData() {
54   *   super.verifyData();
55   * }
56   * 
57   * &#064;Override
58   * &#064;Transient
59   * protected String getRowId() {
60   *   return String.valueOf(getPrimaryKey());
61   * }
62   * </pre>
63   * 
64   * @version $Id: ProtectedData.java 29874 2018-09-13 09:38:35Z samuellb $
65   */
66  public abstract class ProtectedData {
67  
68      private static final Logger log = Logger.getLogger(ProtectedData.class);
69  
70      /** Implementation for the database protection method used */
71      protected ProtectedDataImpl impl;
72  
73      /** Definition of the optional database integrity protection implementation */
74      private static final String implClassName = "org.cesecore.dbprotection.ProtectedDataIntegrityImpl";
75      /** Cache class so we don't have to do Class.forName for every entity object created */
76      private static volatile Class<?> implClass = null;
77  
78      /** Optimization variable so we don't have to check for existence of implClass for every construction of an entity object */
79      private static volatile boolean integrityExists = true;
80  
81      /** A default constructor is needed by JPA.
82       * This constructor initializes the available database integrity protection module, if any is available
83       */
84      public ProtectedData() {
85          if (integrityExists) {
86              try {
87                  if (implClass == null) {
88                      // We only end up here once, if the class does not exist, we will never end up here again (ClassNotFoundException)
89                      // and if the class exists we will never end up here again (it will not be null)
90                      implClass = Class.forName(implClassName);
91                      log.debug("ProtectedDataIntegrityImpl is available, and used, in this version of EJBCA.");
92                  }
93                  impl = (ProtectedDataImpl)implClass.getConstructor().newInstance();
94                  impl.setTableName(getTableName());
95              } catch (ClassNotFoundException | NoSuchMethodException e) {
96                  // We only end up here once, if the class does not exist, we will never end up here again
97                  integrityExists = false;
98                  log.info("No database integrity protection available in this version of EJBCA.");
99                  impl = new ProtectedDataNoopImpl();
100             } catch (InstantiationException | InvocationTargetException e ) {
101                 log.error("Error intitilizing database integrity protection: ", e);
102             } catch (IllegalAccessException e) {
103                 log.error("Error intitilizing database integrity protection: ", e);
104             }
105         } else {
106             impl = new ProtectedDataNoopImpl();
107         }
108     }
109 
110     /**
111      * asks the data class for the string to be protected. Version is -1 for a new row to be protected, and otherwise a version given earlier from the
112      * data class when storing the row.
113      *
114      * @param rowversion the version of the string that is protected, used as input when verifying data. -1 when getting protection string for data to be
115      *          inserted or updated. -1 means that the data class should use it's latest version of protect string
116      * @return String to be integrity protected, i.e. input to hmac.
117      */
118     protected abstract String getProtectString(int rowversion);
119 
120     /**
121      * asks the data class for the version of the string that is protected, used as input to getProtectString() when verifying data. This is used so
122      * that the data class can alter itself with new fields, but still be backwards compatible and verify older database data. Called when getting the
123      * version for inserts or updates.
124      *
125      * @return int version the latest version of protection string.
126      */
127     protected abstract int getProtectVersion();
128 
129     /**
130      * The extending class must have a database column "rowProtection" that can be read and set.
131      * @param rowProtection protect
132      */
133     abstract public void setRowProtection(final String rowProtection); 
134 
135     abstract public String getRowProtection();
136 
137     /**
138      * Returns id of the row in the database, in case of failure we can see in the log which row failed to verify
139      *
140      * @return id of database row, specific for implementing class.
141      */
142     protected abstract String getRowId();
143 
144     /** @return the database table name. Should be overridden by classes that does not share the same name as the database table it maps to (for example by subclassing). */
145     protected String getTableName() {
146         return this.getClass().getSimpleName();
147     }
148 
149     /** Overridden by extending class to be able to use @PrePersist, overriding class calls super.protectData().
150      * This method creates integrity protection for the specific entity in the database.
151      *
152      * @throws DatabaseProtectionException (RuntimeException) on error creating integrity protection.
153      */
154     protected void protectData() {
155         impl.protectData(this);
156     }
157 
158     /**
159      * Overridden by extending class to be able to use @PostLoad, overriding class calls super.verifyData().
160      * This method verifies integrity protection for the specific entity in the database. If the data verification
161      * failed, it invokes {@link #onDataVerificationError(DatabaseProtectionException)}.
162      */
163     protected void verifyData() {
164         try {
165             impl.verifyData(this);
166         } catch (final DatabaseProtectionException e) {
167             onDataVerificationError(e);
168         }
169     }
170 
171     /** Method that calculates integrity protection of an entity, but does not store it anywhere. Used primarily to make test protection
172      * in order to exercise the CryptoToken.
173      * @return the calculated protection string
174      */
175     public String calculateProtection() {
176         return impl.calculateProtection(this);
177     }
178 
179     /**
180      * Throws DatabaseProtectionException if erroronverifyfail is enabled in databaseprotection.properties
181      * and logs a "row protection failed" message on ERROR level.
182      * @param e exception
183      * @throws DatabaseProtectionException the exception given as parameter if erroronverifyfail is enabled
184      */
185     protected void onDataVerificationError(final DatabaseProtectionException e) {
186         impl.onDataVerificationError(e);
187     }
188 }
