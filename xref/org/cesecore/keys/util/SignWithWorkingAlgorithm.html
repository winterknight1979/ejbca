13  package org.cesecore.keys.util;
14  
15  import java.security.NoSuchProviderException;
16  import java.security.Provider;
17  import java.security.Security;
18  import java.util.HashMap;
19  import java.util.List;
20  import java.util.Map;
21  import java.util.concurrent.locks.Lock;
22  import java.util.concurrent.locks.ReentrantLock;
23  
24  import org.apache.log4j.Logger;
25  
26  /**
27   * Call {@link #doSignTask(List, Provider, ISignOperation)} or {@link #doSignTask(List, String, ISignOperation)}
28   * when you want to sign with any working algorithm in the list.
29   * This is usable when working with HSMs. Different HSMs may support different
30   * algorithms. Use this class when you just want to sign with any algorithm
31   * supported by the HSM. Example of this:
32   * - Signing of certificate to be used for the p11 certificate object corresponding
33   *   to the p11 key.
34   * - Signing of CSR when no particular algorithm is required by the receiver.
35   * - Test signing, just checking that the key is working.
36   * @version $Id: SignWithWorkingAlgorithm.java 24555 2016-10-21 13:42:23Z anatom $
37   *
38   */
39  public class SignWithWorkingAlgorithm {
40      /** Log4j instance */
41      final private static Logger log = Logger.getLogger(SignWithWorkingAlgorithm.class);
42      final private static Map<Integer, SignWithWorkingAlgorithm> instanceMap = new HashMap<>();
43      final private Provider provider;
44      final private List<String> availableSignAlgorithms;
45      private String signAlgorithm;
46      final private Lock lock;
47  
48      /**
49       * Finds the registered provider from sProvider and calls {@link #doSignTask(List, Provider, ISignOperation)}.
50       * @param availableSignAlgorithms algorithms to choose from.
51       * @param sProvider provider name
52       * @param operation operation the performs the signing, is an instance of ISignOperation for example KeyTools.SignDataOperation
53       * @return true if the signing was done.
54       * @throws NoSuchProviderException if the provider is not found.
55       * @throws TaskWithSigningException  thrown if {@link ISignOperation#taskWithSigning(String, Provider)} is failing.
56       */
57      public static boolean doSignTask(
58              final List<String> availableSignAlgorithms,
59              final String sProvider,
60              final ISignOperation operation) throws NoSuchProviderException, TaskWithSigningException {
61          final Provider provider = Security.getProvider(sProvider);
62          if ( provider==null ) {
63              throw new NoSuchProviderException();
64          }
65          return doSignTask(availableSignAlgorithms, provider, operation);
66      }
67      /**
68       * First time each algorithm in availableSignAlgorithms are tried until the 
69       * {@link ISignOperation#taskWithSigning(String, Provider)} is successfully completed.
70       * The working algorithm is saved after the first time. Succeeding calls
71       * with same availableSignAlgorithms and provider will directly use the 
72       * algorithm that was working the first time.
73       * @param availableSignAlgorithms algorithms to choose from.
74       * @param provider provider
75       * @param operation operation that performs the signing
76       * @return true if the signing was done.
77       * @throws TaskWithSigningException thrown if {@link ISignOperation#taskWithSigning(String, Provider)} is failing.
78       */
79      public static boolean doSignTask(
80              final List<String> availableSignAlgorithms,
81              final Provider provider,
82              final ISignOperation operation) throws TaskWithSigningException {
83          final Integer mapKey = Integer.valueOf(availableSignAlgorithms.hashCode()^provider.hashCode());
84          final SignWithWorkingAlgorithm instance;
85          synchronized (instanceMap) {
86              final SignWithWorkingAlgorithm waitInstance = instanceMap.get(mapKey);
87              if ( waitInstance==null ) {
88                  instance = new SignWithWorkingAlgorithm(provider, availableSignAlgorithms);
89                  instanceMap.put(mapKey, instance);
90              } else {
91                  instance = waitInstance;
92              }
93          }
94          return instance.tryOutWorkingAlgorithm(operation);
95      }
96      private SignWithWorkingAlgorithm(
97              final Provider _provider,
98              final List<String> _availableSignAlgorithms) {
99          this.provider = _provider;
100         this.lock = new ReentrantLock();
101         this.availableSignAlgorithms = _availableSignAlgorithms;
102     }
103     private boolean tryOutWorkingAlgorithm(final ISignOperation operation) throws TaskWithSigningException {
104         if ( this.signAlgorithm!=null ) {
105             operation.taskWithSigning(this.signAlgorithm, this.provider);
106             return true;
107         }
108         this.lock.lock();
109         try {
110             if ( this.signAlgorithm!=null ) {
111                 operation.taskWithSigning(this.signAlgorithm, this.provider);
112                 return true;
113             }
114             for ( final String trySignAlgorithm : this.availableSignAlgorithms ) {
115                 try {
116                     operation.taskWithSigning(trySignAlgorithm, this.provider);
117                 } catch( final Exception e ) {
118                     log.info(String.format("Signature algorithm '%s' not working for provider '%s'. Exception: %s", trySignAlgorithm, this.provider, e.getMessage()));
119                     continue;
120                 }
121                 log.info(String.format("Signature algorithm '%s' working for provider '%s'.", trySignAlgorithm, this.provider));
122                 this.signAlgorithm = trySignAlgorithm;
123                 return true;
124             }
125             log.info(String.format("No valid signing algorithm found for the provider '%s'.",  this.provider));
126             return false;
127         } finally {
128             this.lock.unlock();
129         }
130     }
131 }
