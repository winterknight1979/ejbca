14  package org.cesecore.keys.validation;
15  
16  import java.util.List;
17  
18  import org.cesecore.certificates.ca.CAInfo;
19  import org.cesecore.internal.UpgradeableDataHashMap;
20  import org.cesecore.profiles.Profile;
21  
22  /**
23   * Base interface for validators. All validators must implement this interface.
24   * 
25   * @version $Id: Validator.java 28537 2018-03-21 12:21:25Z mikekushner $
26   */
27  
28  public interface Validator extends PhasedValidator, CertificateProfileAwareValidator, Profile, Cloneable {
29  
30      static final String TYPE_NAME = "VALIDATOR";
31      
32      /**
33       * Initializes the key validator.
34       */
35      void init();
36  
37      /**
38       * Populates the sub class specific key validator values with template values based on {@link ValidatorBase#getSettingsTemplate()}. 
39       * Sub classes only need to implement this method if they support configuration templates.
40       * @param template the validator settings template.
41       */
42      void setKeyValidatorSettingsTemplate(KeyValidatorSettingsTemplate template);
43      
44      /**
45       * Gets the failed action index (@see #setFailedAction(int)).
46       * @return the index.
47       */
48      int getFailedAction();
49      
50      /**
51       * Sets the failed action index (@link KeyValidationFailedActions), defining what action should
52       * be taken when validation fails, i.e. #validate returns errors
53       * @param index the index.
54       */
55      void setFailedAction(int index);
56  
57      
58      /**
59       * Gets the not_applicable action index (@see #setNotApplicableAction(int)).
60       * @return the index.
61       */
62      int getNotApplicableAction();
63  
64      /**
65       * Sets the not_applicable action index {@link KeyValidationFailedActions}, defining what action should
66       * be taken when a Validator is not applicable for the input (for example ECC keys to an RSA key validator),
67       * i.e. #validate throws ValidatorNotApplicableException
68       * @param index the index.
69       */
70      void setNotApplicableAction(int index);
71  
72      /**
73       * Gets a list of applicable CA types (X509 or CVC see {@link CAInfo#CATYPE_X509} or {@link CAInfo#CATYPE_CVC}).
74       * @return the list of class names of the allowed CA types.
75       */
76      List<Integer> getApplicableCaTypes();
77      
78      /**
79       * @return the settings template index.
80       */
81      Integer getSettingsTemplate();
82      
83      /**
84       * Sets the settings template index.
85       * @param option the type {@link KeyValidatorSettingsTemplate}.
86       */
87      void setSettingsTemplate(Integer option);
88      
89       /**
90        * 
91        * @return a display friendly string of this validator
92        */
93       String toDisplayString();
94  
95       /**
96        * Clone has to be implemented instead of a copy constructor due to the fact that we'll be referring to implementations by this interface only. 
97        * 
98        * @return a deep copied clone of this validator
99        */
100       Validator clone();
101       
102       /**
103        * @return the description.
104        */
105       public String getDescription();
106 
107       /**
108        * @param description the description. 
109        */
110       void setDescription(String description);
111       
112       /** Implementation of UpgradableDataHashMap function getLatestVersion 
113      * @return version */    
114       float getLatestVersion();
115       
116       UpgradeableDataHashMap getUpgradableHashmap();
117       
118       /**
119        * Returns an identifier for the type of the approval profile.
120        * @return type of approval, e.g. "RSA_KEY_VALIDATOR"
121        */
122       String getValidatorTypeIdentifier();
123       
124       /**
125        * @return the type as a human readable name.
126        */
127       String getLabel();
128       
129       /**
130        * @return the subtype of this validator
131        */
132       Class<? extends Validator> getValidatorSubType();
133 }
