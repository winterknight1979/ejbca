135     @Override
136     public void changeKeyValidator(AuthenticationToken admin, Validator validator)
137             throws AuthorizationDeniedException, KeyValidatorDoesntExistsException {
138         assertIsAuthorizedToEditValidators(admin);
139         ProfileData data = profileSession.findById(validator.getProfileId());
140         final String message;
141         final String name = validator.getProfileName();
142         if (data != null) {
143             profileSession.changeProfile(validator);
144             // Since loading a KeyValidator is quite complex, we simple purge the cache here.
145             ValidatorCache.INSTANCE.removeEntry(data.getId());
146             message = intres.getLocalizedMessage("validator.changed_validator", name);
147             final Map<String, Object> details = new LinkedHashMap<String, Object>();
148             details.put("msg", message);
149             //TODO: Include a diff in the changelog (profileData.getProfile().diff(profile);), but make sure to resolve all steps so that we don't
150             //      output a ton of serialized garbage (see ECA-5276)
151             auditSession.log(EventTypes.VALIDATOR_CHANGE, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE, admin.toString(), null, null,
152                     null, details);
153         } else {
154             message = intres.getLocalizedMessage("validator.error.change_validator", name);
155             log.info(message);
156             throw new KeyValidatorDoesntExistsException("Validator by ID " + validator.getProfileId() + " does not exist in database.");

194     @Override
195     public void removeKeyValidator(AuthenticationToken admin, final String validatorName)
196             throws AuthorizationDeniedException, CouldNotRemoveKeyValidatorException {
197         if (log.isTraceEnabled()) {
198             log.trace(">removeKeyValidator(id: " + validatorName + ")");
199         }
200         assertIsAuthorizedToEditValidators(admin);
201         String message;
202 
203         List<ProfileData> datas = profileSession.findByNameAndType(validatorName, Validator.TYPE_NAME);
204         if (datas == null) {
205             if (log.isDebugEnabled()) {
206                 log.debug("Trying to remove a key validator that does not exist with name: " + validatorName);
207             }
208             return;
209         } else {
210             for (ProfileData data : datas) {
211                 if (caSession.existsKeyValidatorInCAs(data.getId())) {
212                     throw new CouldNotRemoveKeyValidatorException();
213                 }
214                 profileSession.removeProfile(data);
215                 // Purge the cache here.
216                 ValidatorCache.INSTANCE.removeEntry(data.getId());
217                 message = intres.getLocalizedMessage("validator.removed_validator", data.getProfileName());
218                 final Map<String, Object> details = new LinkedHashMap<String, Object>();
219                 details.put("msg", message);
220                 auditSession.log(EventTypes.VALIDATOR_REMOVAL, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE, admin.toString(), null,
221                         null, null, details);
222             }
223         }
224 
225         if (log.isTraceEnabled()) {
226             log.trace("<removeKeyValidator()");
227         }

255     @Override
256     public void cloneKeyValidator(final AuthenticationToken admin, final Validator validator, final String newName)
257             throws AuthorizationDeniedException, KeyValidatorDoesntExistsException, KeyValidatorExistsException {
258         Validator validatorClone = null;
259         final Integer origProfileId = validator.getProfileId();
260         if (origProfileId == null) {
261             throw new KeyValidatorDoesntExistsException("Could not find key validator " + validator.getProfileName());
262         }
263         validatorClone = getValidator(origProfileId).clone();
264         validatorClone.setProfileName(newName);
265         try {
266             addValidatorInternal(admin, validatorClone);
267             final String message = intres.getLocalizedMessage("validator.cloned_validator", newName, validator.getProfileName());
268             final Map<String, Object> details = new LinkedHashMap<String, Object>();
269             details.put("msg", message);
270             auditSession.log(EventTypes.VALIDATOR_CREATION, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE, admin.toString(), null,
271                     null, null, details);
272         } catch (KeyValidatorExistsException e) {
273             final String message = intres.getLocalizedMessage("validator.error.clone_validator", newName, validator.getProfileName());
274             log.info(message);
275             throw e;
276         }

288         if (log.isTraceEnabled()) {

293         if (profileSession.findByNameAndType(newName, Validator.TYPE_NAME).isEmpty()) {
294             ProfileData data = profileSession.findById(validator.getProfileId());
295             if (data != null) {

297                 success = true;
298                 // Since loading a key validator is quite complex, we simple purge the cache here.
299                 ValidatorCache.INSTANCE.removeEntry(data.getId());
300             }
301         }
302         if (success) {
303             final String message = intres.getLocalizedMessage("validator.renamed_validator", validator.getProfileName(), newName);
304             final Map<String, Object> details = new LinkedHashMap<String, Object>();
305             details.put("msg", message);

309             final String message = intres.getLocalizedMessage("validator.errorrenamekeyvalidator", validator.getProfileName(), newName);

311             throw new KeyValidatorExistsException();
312         }
313         if (log.isTraceEnabled()) {
314             log.trace("<renameKeyValidator()");

318     @Override
319     public Map<Integer, Validator> getAllKeyValidators() {
320         final List<ProfileData> keyValidators = findAllProfiles(Validator.TYPE_NAME);
321         final Map<Integer, Validator> result = new HashMap<>();
322         final boolean enabled = ((ExternalScriptsConfiguration) globalConfigurationSession
323                 .getCachedConfiguration("0")).getEnableExternalScripts();
324         for (ProfileData data : keyValidators) {
325             if (!enabled && data.getProfile() instanceof ExternalCommandCertificateValidator) {
326                 if (log.isTraceEnabled()) {
327                     log.trace("Skip " + data.getProfileType() + " with name " + data.getProfileName() + " because calls for external command certificate validators are disabled.");
328                 }
329                 continue;
330             }
331             //Cast is safe since we know we retrieved the correct implementation
332             try {
333                 result.put(data.getId(), (Validator) data.getProfile());
334             } catch (IllegalStateException e) {
335                 // NOPMD: Implementation not available in this version if EJBCA
336             }
337         }
338         if (log.isDebugEnabled()) {
339             for (Integer id : result.keySet()) {
340                 log.debug("Key validators found in datastore: " + id + ":" + result.get(id).getProfileName());
341             }
342         }
343         return result;
344     }

385     @Override
386     public void validateDnsNames(final AuthenticationToken authenticationToken, final CA ca, final EndEntityInformation endEntityInformation,
387             final RequestMessage requestMessage) throws ValidationException {
388         if (!CollectionUtils.isEmpty(ca.getValidators())) {
389             Validator baseValidator;
390             DnsNameValidator validator;
391             for (Integer id : ca.getValidators()) {
392                 baseValidator = getValidatorInternal(id, true);
393                 if (baseValidator != null && baseValidator.getValidatorSubType().equals(DnsNameValidator.class)) {
394                     validator = (DnsNameValidator) baseValidator;
395                     // Filter for validator criteria.
396                     if (baseValidator instanceof CertificateProfileAwareValidator
397                             && !filterCertificateProfileAwareValidator(validator, endEntityInformation.getCertificateProfileId())) {
398                         continue;
399                     }
400                     CertificateProfile certificateProfile = certificateProfileSession
401                             .getCertificateProfile(endEntityInformation.getCertificateProfileId());
402                     final String subjectAltName = endEntityInformation.getSubjectAltName();
403                     final List<String> dnsNames = new ArrayList<>();
404                     for (String split : subjectAltName.split(",")) {
405                         if (split.trim().toLowerCase().startsWith(CertTools.DNS.toLowerCase())) {
406                             dnsNames.add(split.trim().substring(CertTools.DNS.length() + 1));
407                         }
408                     }
409                     //If the certificate profile allows extension override, there may be SANs mixed in among the extensions in the request message
410                     if (certificateProfile.getAllowExtensionOverride() && requestMessage != null) {
411                         Extensions extensions = requestMessage.getRequestExtensions();
412                         if (extensions != null) {
413                             Extension extension = extensions.getExtension(Extension.subjectAlternativeName);
414                             if (extension != null) {
415                                 String extendedSubjectAltName = CertTools.getAltNameStringFromExtension(extension);
416                                 for (String split : extendedSubjectAltName.split(",")) {
417                                     if (split.trim().toLowerCase().startsWith(CertTools.DNS.toLowerCase())) {
418                                         dnsNames.add(split.trim().substring(CertTools.DNS.length() + 1));
419                                     }
420                                 }
421                             }
422                         }
423                     }
424 
425                     Entry<Boolean, List<String>> result = validator.validate(executorService, dnsNames.toArray(new String[dnsNames.size()]));
426 
427                     final String validatorName = validator.getProfileName();
428                     final List<String> messages = result.getValue();
429                     if (!result.getKey()) {
430                         // Validation has failed. Not security event as such, since it will break issuance and not cause anything important to happen.
431                         // We want thorough logging in order to trouble shoot though
432                         final String message = intres.getLocalizedMessage("validator.caa.validation_failed", validatorName,
433                                 validator.getIssuer(), messages);
434                         log.info(EventTypes.VALIDATOR_VALIDATION_FAILED + ";" + EventStatus.FAILURE + ";" + ModuleTypes.VALIDATOR + ";" + ServiceTypes.CORE + ";msg=" + message);
435                         final int index = validator.getFailedAction();
436                         performValidationFailedActions(index, message);
437                     } else {
438                         // Validation succeeded, this can be considered a security audit event because CAs may be asked to present this as evidence to an auditor
439                         final String message = intres.getLocalizedMessage("validator.caa.validation_successful", validatorName,
440                                 validator.getIssuer(),
441                                 messages);
442                         final Map<String, Object> details = new LinkedHashMap<String, Object>();
443                         details.put("msg", message);
444                         auditSession.log(EventTypes.VALIDATOR_VALIDATION_SUCCESS, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE,
445                                 authenticationToken.toString(), String.valueOf(ca.getCAId()), null, endEntityInformation.getUsername(), details);
446                     }
447 
448                 }
449             }
450         } else {
451             if (log.isDebugEnabled()) {
452                 log.debug("No validators configured for CA " + ca.getName() + " (ID=" + ca.getCAId() + ").");
453             }
454         }
455     }

457     @Override
458     public boolean validatePublicKey(final AuthenticationToken admin, final CA ca, EndEntityInformation endEntityInformation,
459             CertificateProfile certificateProfile, Date notBefore, Date notAfter, PublicKey publicKey)
460             throws ValidationException, IllegalValidityException {
461         boolean result = true;
462         if (ca != null && !CollectionUtils.isEmpty(ca.getValidators())) { // || certificateProfile.isTypeRootCA() || certificateProfile.isTypeSubCA()
463             final CertificateValidityity.html#CertificateValidity">CertificateValidity certificateValidity = new CertificateValidity(endEntityInformation, certificateProfile, notBefore, notAfter,
464                     ca.getCACertificate(), false, false);
465             if (log.isDebugEnabled()) {
466                 log.debug("Validate " + publicKey.getAlgorithm() + " public key with " + publicKey.getFormat() + " format.");
467                 log.debug("Certificate 'notBefore' " + certificateValidity.getNotBefore());
468                 log.debug("Certificate 'notAfter' " + certificateValidity.getNotAfter());
469             }
470             Validator baseValidator;
471             KeyValidator validator;
472             String name;
473             for (Integer id : ca.getValidators()) {
474                 baseValidator = getValidatorInternal(id, true);
475                 if (baseValidator != null && baseValidator.getValidatorSubType().equals(KeyValidator.class)) {
476                     validator = (KeyValidator) baseValidator;
477                     name = validator.getProfileName();
478                     if (log.isTraceEnabled()) {
479                         log.trace("Try to apply key validator: " + validator.toDisplayString());
480                     }
481                     try {
482                         // Filter for validator criteria.
483                         if (validator instanceof CertificateProfileAwareValidator
484                                 && !filterCertificateProfileAwareValidator(validator, endEntityInformation.getCertificateProfileId())) {
485                             continue;
486                         }
487                         if (validator instanceof ValidityAwareValidator
488                                 && !filterValidityAwareValidator(validator, certificateValidity.getNotBefore(), certificateValidity.getNotAfter())) {
489                             continue;
490                         }
491                         final String fingerprint = CertTools.createPublicKeyFingerprint(publicKey, "SHA-256");
492                         log.info(intres.getLocalizedMessage("validator.key.isbeingprocessed", name, endEntityInformation.getUsername(), fingerprint));
493                         List<String> messages = validator.validate(publicKey, certificateProfile);
494                         if (messages.size() > 0) { // Validation has failed.
495                             result = false;
496                             final int index = validator.getFailedAction();
497                             final String message = intres.getLocalizedMessage("validator.key.validation_failed", name, messages);
498                             final Map<String, Object> details = new LinkedHashMap<String, Object>();
499                             details.put("msg", message);
500                             auditSession.log(EventTypes.VALIDATOR_VALIDATION_FAILED, EventStatus.FAILURE, ModuleTypes.VALIDATOR, ServiceTypes.CORE,
501                                     admin.toString(), String.valueOf(ca.getCAId()), fingerprint, endEntityInformation.getUsername(), details);
502                             performValidationFailedActions(index, message);
503                         } else {
504                             final byte[] keyBytes = publicKey.getEncoded();
505                             final String publicKeyEncoded = (keyBytes != null ? new String(Base64.encode(keyBytes)) : "null");
506                             final String message = intres.getLocalizedMessage("validator.key.validation_successful", name, publicKeyEncoded);
507                             log.info(message);
508                             final Map<String, Object> details = new LinkedHashMap<String, Object>();
509                             details.put("msg", message);
510                             auditSession.log(EventTypes.VALIDATOR_VALIDATION_SUCCESS, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE,
511                                     admin.toString(), String.valueOf(ca.getCAId()), null, endEntityInformation.getUsername(), details);
512                         }
513                     } catch (ValidatorNotApplicableException e) {
514                         // This methods either throws a KeyValidationException, or just logs a message and validation should be considered successful
515                         // use method performValidationFailedActions because it's the same actions
516                         performValidationFailedActions(validator.getNotApplicableAction(), e.getMessage());
517                     } catch (ValidationException e) {
518                         throw e;
519                     }
520                 }
521             }
522         } else {
523             if (log.isDebugEnabled()) {
524                 log.debug("No key validator configured for CA " + ca.getName() + " (ID=" + ca.getCAId() + ").");
525             }
526         }
527         return result;
528     }

530     @Override
531     public void validateCertificate(final AuthenticationToken authenticationToken, final IssuancePhase phase, final CA ca,
532             final EndEntityInformation endEntityInformation, final X509Certificate certificate) throws ValidationException {
533         if (log.isDebugEnabled()) {
534             log.debug("Validate certificate for phase " + phase);
535         }
536         if (ca != null && !CollectionUtils.isEmpty(ca.getValidators())) {
537             Validator baseValidator;
538             CertificateValidator validator;
539             String name;
540             for (Integer id : ca.getValidators()) {
541                 baseValidator = getValidatorInternal(id, true);
542                 if (baseValidator != null && baseValidator.getValidatorSubType().equals(CertificateValidator.class)) {
543                     validator = (CertificateValidator) baseValidator;
544                     name = validator.getProfileName();
545                     if (phase.getIndex() != validator.getPhase()) {
546                         continue;
547                     }
548                     if (validator instanceof CertificateProfileAwareValidator
549                             && !filterCertificateProfileAwareValidator(validator, endEntityInformation.getCertificateProfileId())) {
550                         continue;
551                     }
552                     if (validator instanceof ValidityAwareValidator
553                             && !filterValidityAwareValidator(validator, certificate.getNotBefore(), certificate.getNotAfter())) {
554                         continue;
555                     }
556                     try {
557                         final String fingerprint = CertTools.createPublicKeyFingerprint(certificate.getPublicKey(), "SHA-256");
558                         log.info(intres.getLocalizedMessage("validator.certificate.isbeingprocessed", name, phase, endEntityInformation.getUsername(),
559                                 fingerprint));
560                         final ExternalScriptsConfigurationptsConfiguration.html#ExternalScriptsConfiguration">ExternalScriptsConfiguration externalScriptsConfiguration = (ExternalScriptsConfiguration) globalConfigurationSession
561                                 .getCachedConfiguration("0");
562                         final ExternalScriptsWhitelist externalScriptsWhitelist = ExternalScriptsWhitelist.fromText(
563                                 externalScriptsConfiguration.getExternalScriptsWhitelist(),
564                                 externalScriptsConfiguration.getIsExternalScriptsWhitelistEnabled());
565                         final List<String> messages = validator.validate(ca, certificate, externalScriptsWhitelist);
566                         if (messages.size() > 0) { // Evaluation has failed.
567                             final String message = intres.getLocalizedMessage("validator.certificate.validation_failed", name, messages);
568                             final Map<String, Object> details = new LinkedHashMap<String, Object>();
569                             details.put("msg", message);
570                             auditSession.log(EventTypes.VALIDATOR_VALIDATION_FAILED, EventStatus.FAILURE, ModuleTypes.VALIDATOR, ServiceTypes.CORE,
571                                     authenticationToken.toString(), String.valueOf(ca.getCAId()), fingerprint, endEntityInformation.getUsername(),
572                                     details);
573                             performValidationFailedActions(validator.getFailedAction(), message);
574                         } else {
575                             final String message = intres.getLocalizedMessage("validator.certificate.validation_successful", name, fingerprint);
576                             log.info(message);
577                             final Map<String, Object> details = new LinkedHashMap<String, Object>();
578                             details.put("msg", message);
579                             auditSession.log(EventTypes.VALIDATOR_VALIDATION_SUCCESS, EventStatus.SUCCESS, ModuleTypes.VALIDATOR, ServiceTypes.CORE,
580                                     authenticationToken.toString(), String.valueOf(ca.getCAId()), null, endEntityInformation.getUsername(), details);
581                         }
582                     } catch (ValidatorNotApplicableException e) {
583                         // This methods either throws a KeyValidationException, or just logs a message and validation should be considered successful
584                         // use method performValidationFailedActions because it's the same actions
585                         performValidationFailedActions(validator.getNotApplicableAction(), e.getMessage());
586                     } catch (CertificateException e) {
587                         throw new ValidationException("Certificate to validate could not be parsed or decoded: " + e.getMessage(), e);
588                     }
589                 }
590             }
591         }

613     public Collection<Integer> getAuthorizedKeyValidatorIds(AuthenticationToken admin, String keyValidatorAccessRule) {
614         final ArrayList<Integer> result = new ArrayList<Integer>();
615         final Map<Integer, String> map = getKeyValidatorIdToNameMap();
616         if (authorizationSession.isAuthorizedNoLogging(admin, keyValidatorAccessRule)) {
617             final boolean rootAccess = authorizationSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource());
618             final List<Integer> authorizedCPIDs = certificateProfileSession.getAuthorizedCertificateProfileIds(admin, 0);
619             for (final Entry<Integer, String> entry : map.entrySet()) {
620                 // Check that administrator have access to all certificate profiles referenced by the validator
621                 Validator val = getValidator(entry.getKey());
622                 boolean allexists = true;
623                 for (final Integer nextcpid : val.getCertificateProfileIds()) {
624                     if (log.isTraceEnabled()) {
625                         log.trace("Validator '" + val.getProfileName() + "' has " + val.getCertificateProfileIds().size() + " no of CPs selected");
626                     }
627                     // If any CP is selected, it's access to all (only authorized will be displayed)
628                     if (nextcpid.intValue() == -1) {
629                         if (log.isDebugEnabled()) {
630                             log.debug("Validator is applicable to all certificate profiles, not limiting access based on CPs");
631                         }
632                         allexists = true;
633                         break;
634                     }
635                     // superadmin should be able to access profiles with missing CA Ids
636                     if (!authorizedCPIDs.contains(nextcpid) && (!rootAccess)) {
637                         if (log.isDebugEnabled()) {
638                             log.debug("Validator have certificate profile " + nextcpid + " selected which admin is not authorized to:" + admin.toString());
639                         }
640                         allexists = false;
641                         break;
642                     }
643                 }
644                 if (allexists) {
645                     result.add(entry.getKey());
646                 }
647             }
648         }
649         return result;

681     /**
682      * Gets a validator by cache or database, can return null. Puts it into the cache, if not already present.
683      *
684      * @param id the validators id.
685      * @param fromCache true if the validator can be taken from cache.
686      * @return a cloned validator that can be used at will without affecting the cache contents, edits have to be saved with replaceKeyValidator
687      * */
688     private Validator getValidatorInternal(int id, boolean fromCache) {
689         Validator result = null;
690         // If we should read from cache, and we have an id to use in the cache, and the cache does not need to be updated
691         if (fromCache && !ValidatorCache.INSTANCE.shouldCheckForUpdates(id)) {
692             // Get from cache (or null)
693             result = ValidatorCache.INSTANCE.getEntry(id);
694         }
695 
696         // if we selected to not read from cache, or if the cache did not contain this entry
697         if (result == null) {
698 
699             // We need to read from database because we specified to not get from cache or we don't have anything in the cache
700             final ProfileData data = profileSession.findById(id);
701             if (data != null) {
702                 result = (Validator) data.getProfile();
703                 if (log.isDebugEnabled()) {
704                     log.debug("Load validator: " + result.getDataMap());
705                 }
706                 final int digest = data.getProtectString(0).hashCode();
707                 // The cache compares the database data with what is in the cache
708                 // If database is different from cache, replace it in the cache
709                 ValidatorCache.INSTANCE.updateWith(data.getId(), digest, data.getProfileName(), result);
710             } else {
711                 // Ensure that it is removed from cache if it exists
712                 ValidatorCache.INSTANCE.removeEntry(id);
713             }
714         }
715         if (result == null) {
716             log.warn("Validator with id " + id + " didn't return any validator");
717         }
718         // We need to clone the validator, otherwise the cache contents will be modifiable from the outside
719         if (result != null) {
720             return result.clone();
721         }
722         return null;
723     }

760      * Applies validity conditions (see {@link KeyValidatorDateConditions}) to the validator.
761      *
762      * @param validator the validator.
763      * @param certificateNotBefore the certificates not before validity.
764      * @param certificateNotAfter the certificates not after validity.
765      * @return false, if the conditions does not match.
766      */
767     private boolean filterValidityAwareValidator(final ValidityAwareValidator validator, final Date certificateNotBefore,
768             final Date certificateNotAfter) {
769         if (log.isTraceEnabled()) {
770             log.trace("Try to apply validity aware validator: " + validator);
771         }
772         final String name = validator.getProfileName();
773         boolean result = true;
774         if (!KeyValidatorDateConditions.evaluate(validator.getNotBefore(), certificateNotBefore, validator.getNotBeforeCondition())) {
775             if (log.isDebugEnabled()) {
776                 log.debug(intres.getLocalizedMessage("validator.filterconditiondoesnotmatch", name, "notBefore"));
777             }
778             result = false;
779         }
780         if (!KeyValidatorDateConditions.evaluate(validator.getNotAfter(), certificateNotAfter, validator.getNotAfterCondition())) {
781             if (log.isDebugEnabled()) {
782                 log.debug(intres.getLocalizedMessage("validator.filterconditiondoesnotmatch", name, "notAfter"));
783             }
784             result = false;
785         }
786         return result;
787     }

796     private boolean filterCertificateProfileAwareValidator(final CertificateProfileAwareValidator validator,

824     /**
825      * Post processes every validation depending on its failed action.
826      *
827      * @param failedAction the failed action index (see {@link KeyValidationFailedActions}).
828      * @param message the message to log.
829      * @param shortMessage the error message to EJBCA Certificate Enrollment Error page
830      * @throws ValidationException if a failed validation has to be abort the certificate issuance.
831      * */
832     private void performValidationFailedActions(final int failedAction, final String message, final String shortMessage) throws ValidationException {
833         if (log.isDebugEnabled()) {
834             log.debug("Perform post action " + failedAction + " - " + message);
835         }
836         if (KeyValidationFailedActions.LOG_INFO.getIndex() == failedAction) {
837             log.info(message);
838         } else if (KeyValidationFailedActions.LOG_WARN.getIndex() == failedAction) {
839             log.warn(message);
840         } else if (KeyValidationFailedActions.LOG_ERROR.getIndex() == failedAction) {
841             log.error(message);
842         } else if (KeyValidationFailedActions.ABORT_CERTIFICATE_ISSUANCE.getIndex() == failedAction) {
843             throw new ValidationException(shortMessage);
844         } else {
845             // NOOP
846             log.debug(message);
847         }
848     }
