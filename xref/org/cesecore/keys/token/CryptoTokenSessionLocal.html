13  package org.cesecore.keys.token;
14  
15  import java.util.List;
16  import java.util.Map;
17  
18  import javax.ejb.Local;
19  
20  /**
21   * 
22   * @version $Id: CryptoTokenSessionLocal.java 25377 2017-03-02 16:46:52Z jeklund $
23   */
24  @Local
25  public interface CryptoTokenSessionLocal extends CryptoTokenSession {
26  
27      /** @param cryptoTokenId ID
28       * @return the specified CryptoToken or null if it does not exis.
29       * Throws RuntimeException if allow.nonexisting.slot=false (default) and a PKCS#11 slot does not exist. */
30      CryptoToken getCryptoToken(int cryptoTokenId);
31  
32      /** Add the specified CryptoToken to the database and return the id used to store it 
33       * @param cryptoToken Token
34       * @return Merged token
35       * @throws CryptoTokenNameInUseException if nme in use */
36      int mergeCryptoToken(CryptoToken cryptoToken) throws CryptoTokenNameInUseException;
37  
38      /** Remove the specified CryptoToken from the database. 
39       * @param cryptoTokenId the id of the crypto token that should be removed
40       * @return true if crypto token exists and is deleted, false if crypto token with given id does not exist 
41       */
42      boolean removeCryptoToken(final int cryptoTokenId);
43  
44      /** @return a list of all CryptoToken identifiers in the database. */
45      List<Integer> getCryptoTokenIds();
46  
47      /** @return a list of all CryptoToken identifiers and corresponding human readable name in the database. */
48      Map<Integer, String> getCryptoTokenIdToNameMap();
49  
50      /** @return a (copy of a) name to id lookup table */
51      Map<String, Integer> getCachedNameToIdMap();
52  
53      /** Clears the CryptoToken cache. */
54      void flushCache();
55      
56      /** Clears the CryptoToken cache except for the cache entries specified in excludeIDs 
57       * @param excludeIDs IDs to exclude*/
58      void flushExcludingIDs(List<Integer> excludeIDs);
59  
60      /** @param cryptoTokenName Name
61       * @param cryptoTokenId ID
62       * @return true if the specified name is used by exactly one CryptoToken and that cryptoToken has the same id (checks the database, not the cache) */
63      boolean isCryptoTokenNameUsedByIdOnly(String cryptoTokenName, int cryptoTokenId);
64  }
