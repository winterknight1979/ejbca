13  package org.cesecore.keys.token.p11;
14  
15  import java.lang.reflect.Method;
16  import java.security.Key;
17  
18  import org.apache.log4j.Logger;
19  import org.cesecore.keys.token.p11.exception.P11RuntimeException;
20  
21  /**
22   * The normal way to access PKCS#11 is with a JCA/JCE provider. But there are
23   * a lot of functionality that can't be done with the provider. Most of this
24   * functionality is not needed by the CESeCore library but there are some
25   * functionality that the library needs. This class will be used to access such
26   * functions.
27   * 
28   * Functionalities of the Sun PKCS#11 implementation are used but this class is not
29   * using this implementation directly - the class sun.security.pkcs11.CESeCoreUtils
30   * is used to provide the interface to the Sun PKCS#11 implementation.
31   * 
32   * Since non public resources of the Sun PKCS#11 implementation is used the
33   * CESeCoreUtils class must be located in the sun.security.pkcs11 package.
34   * The class must also be deployed as an installed extension 
35   * (see https://docs.oracle.com/javase/tutorial/ext/basics/install.html) in order
36   * to use parts of the sun implementation.
37   * 
38   * To deploy it, copy the jar $EJBCA_HOME/dist/ext/primekey-sunP11.jar to one of
39   * the directories defined by the 'java.ext.dirs' system property. But when running
40   * clientToolBox this deployment is not needed since the start script takes care
41   * of this.
42   * 
43   * If CESeCoreUtils is not in the classpath then this class will work as a
44   * dummy and just return each call without doing anything.
45   * 
46   * When CESeCoreUtils is not in the classpath a warning will be written
47   * to {@link Logger}.
48   * 
49   * @version $Id: PKCS11Utils.java 25864 2017-05-17 13:46:10Z anatom $
50   */
51  public class PKCS11Utils {
52      private static final Logger log = Logger.getLogger(PKCS11Utils.class);
53      private static PKCS11Utils p11utils = null;
54      private final Method makeKeyUnmodifiable;
55      private final Method isKeyModifiable;
56      private final Method securityInfo;
57  
58      private PKCS11Utils( final Method _makeKeyUnmodifiable, final Method _isKeyModifiable, final Method _securityInfo ) {
59          this.makeKeyUnmodifiable = _makeKeyUnmodifiable;
60          this.isKeyModifiable = _isKeyModifiable;
61          this.securityInfo = _securityInfo;
62      }
63  
64      /**
65       * @return The instance.
66       */
67      public static synchronized PKCS11Utils getInstance() {
68          if ( p11utils!=null ) {
69              return p11utils;
70          }
71          final String className = "sun.security.pkcs11.CESeCoreUtils";
72          final Class<? extends Object> clazz;
73          if ( log.isDebugEnabled() ) {
74              final String propertyKey = "java.ext.dirs";
75              log.debug(String.format("The value of the system property '%s' is '%s'.", propertyKey, System.getProperty(propertyKey)));
76          }
77          try {
78              clazz = Class.forName(className);
79          } catch (ClassNotFoundException e) {
80              log.warn(String.format(
81                      "Class '%s' not available. The attribute of all generated keys will have 'CKA_MODIFYABLE=TRUE'. A '%s' exception was thrown with the message '%s'.",
82                      className, e.getClass().getName(), e.getMessage() ));
83              p11utils = new PKCS11Utils(null, null, null);
84              return p11utils;
85          }
86          try {
87              p11utils = new PKCS11Utils(
88                      clazz.getMethod("makeKeyUnmodifiable", new Class[]{String.class, Key.class}),
89                      clazz.getMethod("isKeyModifiable", new Class[]{String.class, Key.class}),
90                      clazz.getMethod("securityInfo", new Class[]{String.class, Key.class, StringBuilder.class}) );
91          } catch (NoSuchMethodException e) {
92              throw new Error(String.format("Not compatible version of %s. Required methods not found.", className), e);
93          }
94          return p11utils;
95      }
96  
97      /**
98       * If CESeCoreUtils is in the classpath
99       * then the CKA_MODIFIABLE attribute of the key will be set to false.
100      * @param key key
101      * @param providerName provider
102      */
103     public void makeKeyUnmodifiable( final Key key, final String providerName) {
104         final String sError = "Not possible to set the attribute CKA_MODIFIABLE to false for the key object.";
105         if ( this.makeKeyUnmodifiable==null ) {
106             log.warn(sError);
107             return;
108         }
109         try {
110             final Object oResult = this.makeKeyUnmodifiable.invoke(null, new Object[]{providerName, key});
111             assert oResult instanceof Boolean;
112             if ( log.isDebugEnabled() ) {
113                 if ( ((Boolean)oResult).booleanValue() ) {
114                     log.debug(String.format("CKA_MODIFIABLE attribute set to false for key '%s'.", key));
115                 } else {
116                     log.debug(String.format("CKA_MODIFIABLE attribute not changed for key '%s'. It was already set to false, or could not be changed", key));
117                 }
118             }
119         } catch (ReflectiveOperationException e) {
120             throw new P11RuntimeException(sError,e);
121         }
122     }
123 
124     /**
125      * Reads the CKA_MODIFIABLE attribute of the p11 key object.
126      * @param key key
127      * @param providerName provider
128      * @return the value of CKA_MODIFIABLE
129      */
130     public boolean isKeyModifiable( final Key key, final String providerName ) {
131         final String sError = "Not possible to read the attribute CKA_MODIFIABLE for the key object.";
132         if ( this.isKeyModifiable==null ) {
133             log.warn(sError);
134             return true;// we say modifiable when we can't find out.
135         }
136         try {
137             final Object oResult = this.isKeyModifiable.invoke(null, new Object[]{providerName, key});
138             assert oResult instanceof Boolean;
139             return ((Boolean)oResult).booleanValue();
140         } catch (ReflectiveOperationException e) {
141             throw new P11RuntimeException(sError, e);
142         }
143     }
144 
145     /**
146      * Writes info about security related attributes of the key. If there is
147      * no CESeCoreUtils in the classpath then a message that the class is not
148      * in the classpath will be written.
149      * @param key key
150      * @param providerName Provider
151      * @param sb the info is written to this.
152      */
153     public void securityInfo(final Key key, final String providerName, final StringBuilder sb) {
154         if ( this.securityInfo==null ) {
155             sb.append("No CESeCoreUtils in classpath.");
156             return;
157         }
158         try {
159             this.securityInfo.invoke(null, new Object[]{providerName, key, sb});
160         } catch (ReflectiveOperationException e) {
161             throw new P11RuntimeException("Not possible to read attributes from key object.", e);
162         }
163     }
