92      @Override
93      public void init(final Properties properties, final byte[] data, final int id) throws CryptoTokenOfflineException, NoSuchSlotException {
94          if (log.isDebugEnabled()) {
95              log.debug(">init: id=" + id);
96          }
97          // Don't autoactivate this right away, we must dynamically create the auth-provider with a slot
98          setProperties(properties);
99          init(properties, false, id);
100         sSlotLabel = getSlotLabel(SLOT_LABEL_VALUE, properties);
101         Pkcs11SlotLabelType type = Pkcs11SlotLabelType.getFromKey(getSlotLabel(SLOT_LABEL_TYPE, properties));
102         String sharedLibrary = properties.getProperty(PKCS11CryptoToken.SHLIB_LABEL_KEY);
103         String attributesFile = properties.getProperty(PKCS11CryptoToken.ATTRIB_LABEL_KEY);
104         Boolean addProvider = !BooleanUtils.toBoolean(properties.getProperty(PKCS11CryptoToken.DO_NOT_ADD_P11_PROVIDER));
105 
106         String friendlyName = properties.getProperty(TOKEN_FRIENDLY_NAME);
107 
108         if(friendlyName != null){
109             p11slot = P11Slot.getInstance(friendlyName, sSlotLabel, sharedLibrary, type, attributesFile, this, id, addProvider);
110         } else {
111             // getInstance will run autoActivate()
112             p11slot = P11Slot.getInstance(sSlotLabel, sharedLibrary, type, attributesFile, this, id, addProvider);
113             
114         }
115         final Provider provider = p11slot.getProvider();
116         if (addProvider) {
117             setJCAProvider(provider);
118         } else {
119             setJCAProviderName(provider.getName());
120             log.info("Configured to not add PKCS#11 Provider: "+provider.getName());
121         }
122         if (log.isDebugEnabled()) {
123             log.debug("<init: id=" + id);
124         }
125     }

133     public void activate(final char[] authCode) throws CryptoTokenOfflineException, CryptoTokenAuthenticationFailedException {
134         if (this.p11slot == null) {
135             throw new CryptoTokenOfflineException("Slot not initialized.");
136         }
137         try {
138             final KeyStore keyStore = createKeyStore(authCode);
139             setKeyStore(keyStore);
140         } catch (Throwable t) { // NOPMD: when dealing with HSMs we need to catch everything
141             log.warn("Failed to initialize PKCS11 provider slot '" + this.sSlotLabel + "'.", t);
142             CryptoTokenAuthenticationFailedExceptionon.html#CryptoTokenAuthenticationFailedException">CryptoTokenAuthenticationFailedException authfe = new CryptoTokenAuthenticationFailedException(
143                     "Failed to initialize PKCS11 provider slot '" + this.sSlotLabel + "'.");
144             authfe.initCause(t);
145             throw authfe;
146         }
147         String msg = intres.getLocalizedMessage("token.activated", getId());
148         log.info(msg);
149     }

151     private KeyStore createKeyStore(final char[] authCode) throws NoSuchAlgorithmException, CertificateException, UnsupportedEncodingException,
152             IOException, KeyStoreException {
153         final Provider provider = this.p11slot.getProvider();
154         final KeyStore keyStore = KeyStore.getInstance( "PKCS11", provider );
155         log.debug("Loading key from slot '" + this.sSlotLabel + "' using pin.");
156         // See ECA-1395 for an explanation of this special handling for the IAIK provider.
157         // If the application uses several instances of the IAIKPkcs11 provider, it has two options to get an initialized key store. First, it can get
158         // the initialized key store directly from the provider instance. This looks like this
159         // KeyStore tokenKeyStore = pkcs11Provider_.getTokenManager().getKeyStore();
160         // where pkcs11Provider_ is the instance of the IAIKPkcs11 provider. Second, the application can instantiate the key store as usual and then
161         // initialize it. For initialization, the application must provide the name of the instance that this key store shall operate with. Just
162         // instantiating the key store is not enough, and if the application calls tokenKeyStore.load(null, null), it always(!) binds the key store to
163         // the first instance of the IAIKPkcs11 provider. This is the case, because there is no means for the KeyStoreSPI class to get the instance of
164         // the provider that was used to instantiate it. This means, it does not help to provide the provider name and calling
165         // KeyStore.getInstance("PKCS11KeyStore", providerName), the call to the load(InputStream, char[]) method with appropriate arguments is
166         // required nevertheless. The correct usage will look like this
167         // KeyStore cardKeyStore = KeyStore.getInstance("PKCS11KeyStore");
168         // String providerName = pkcs11Provider_.getName();
169         // ByteArrayInputStream providerNameInpustStream =
170         // new ByteArrayInputStream(providerName.getBytes("UTF-8"));
171         // cardKeyStore.load(providerNameInpustStream, null);
172         // The password parameter of the load method (this is the second parameter, which is null here) will be used if provided (i.e. if it is not
173         // null). If it is null, the default login manager will use the configured method for prompting the PIN on demand. If the application just
174         // provides the instance number as a string instead of the complete provider name, the key store will also accept it.
175         if (provider.getClass().getName().equals(Pkcs11SlotLabel.IAIK_PKCS11_CLASS)) {
176             keyStore.load(new ByteArrayInputStream(getSignProviderName().getBytes("UTF-8")), authCode);
177         } else {
178             // For the Sun provider no provider name is used.
179             keyStore.load(null, authCode);
180         }
181         return keyStore;
182     }

184     @Override
185     public void deactivate() {
186         try {
187             setKeyStore(null);
188         } catch (KeyStoreException e) {
189             // Exception should only be thrown if loading a non-null KeyStore fails
190             throw new IllegalStateException("This should never happen.");
191         }
192         if (this.p11slot != null) {
193             this.p11slot.logoutFromSlotIfNoTokensActive();            
194         } else {
195             log.debug("p11slot was null, token was not active trying to deactivate.");
196         }
197         final String msg = intres.getLocalizedMessage("token.deactivate", getId());
198         log.info(msg);
199     }

269     /**
270      * Extracts the slotLabel that is used for many tokens in construction of the provider
271      *
272      * @param sSlotLabelKey which key in the properties that gives us the label
273      * @param properties CA token properties
274      * @return String with the slot label, trimmed from whitespace
275      */
276     private static String getSlotLabel(String sSlotLabelKey, Properties properties) {
277         String ret = null;
278         if (sSlotLabelKey != null && properties != null) {
279             ret = properties.getProperty(sSlotLabelKey);
280             if (ret != null) {
281                 ret = ret.trim();
282             }
283         }
284         return ret;
285     }

287     /**
288      * Will replace deprecated properties values with the new ones. 
289      * 
290      * @param properties a properties file of the old format.
291      * @return a defensive copy of the submitted properties
292      */
293     @Deprecated
294     //Remove when we no longer support upgrading from 5.0.x -> 6.0.x 
295     public static Properties upgradePropertiesFileFrom5_0_x(final Properties properties) {
296         Properties returnValue = new Properties();
297         
298         for (Object key : properties.keySet()) {
299             final String keyString = (String) key;
300             if (log.isDebugEnabled()) {
301                 log.debug(">upgradePropertiesFileFrom5_0_x, keyString: "+key);
302             }
303             if (keyString.equalsIgnoreCase(SLOT_LABEL_KEY)) {
304                 String keyValue = properties.getProperty(keyString);
305                 if (log.isDebugEnabled()) {
306                     log.debug(">upgradePropertiesFileFrom5_0_x, keyValue: "+keyValue);
307                 }
308                 // In 5.0.11, the "slot" value may contain just an integer, but may also encode an integer, an index
309                 // a token label or a config file. 
310                 final String oldLabelPrefix = "TOKEN_LABEL:";
311                 final String oldIndexPrefix = "SLOT_LIST_IX:";
312                 final String oldSlotNumberPrefix = "SLOT_ID:";
313                 final String oldSunFilePrefix = "SUN_FILE:";
314                 final String delimiter = ":";
315                 if(Pkcs11SlotLabelType.SLOT_NUMBER.validate(keyValue)) {
316                     //If it was a straight integer, then save as is
317                     returnValue.setProperty(SLOT_LABEL_VALUE, keyValue);
318                     returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SLOT_NUMBER.getKey());
319                 } else if(keyValue.startsWith(oldSlotNumberPrefix)) {
320                    //If not, check with the rest of the values 
321                     returnValue.setProperty(SLOT_LABEL_VALUE, keyValue.split(delimiter, 2)[1]);
322                     returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SLOT_NUMBER.getKey());
323                 } else if(keyValue.startsWith(oldIndexPrefix)) {
324                     returnValue.setProperty(SLOT_LABEL_VALUE, keyValue.split(delimiter, 2)[1]);
325                     returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SLOT_INDEX.getKey());
326                 } else if(keyValue.startsWith(oldLabelPrefix)) {
327                     returnValue.setProperty(SLOT_LABEL_VALUE, keyValue.split(delimiter, 2)[1]);
328                     returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SLOT_LABEL.getKey());
329                 } else if(keyValue.startsWith(oldSunFilePrefix)) {
330                     returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SUN_FILE.getKey());
331                 }
332                 
333             } else if (keyString.equalsIgnoreCase(SLOT_LIST_INDEX_KEY)) {
334                 String indexValue = properties.getProperty(keyString);
335                 if (indexValue.charAt(0) != 'i') {
336                     indexValue = "i" + indexValue;
337                 }
338                 if (log.isDebugEnabled()) {
339                     log.debug(">upgradePropertiesFileFrom5_0_x, indexValue: "+indexValue);
340                 }
341                 returnValue.setProperty(SLOT_LABEL_VALUE, indexValue);
342                 returnValue.setProperty(SLOT_LABEL_TYPE, Pkcs11SlotLabelType.SLOT_INDEX.getKey());
343             } else {
344                 if (log.isDebugEnabled()) {
345                     log.debug(">upgradePropertiesFileFrom5_0_x, keyString is neither "+SLOT_LABEL_KEY+" or "+SLOT_LIST_INDEX_KEY+", just setting the property without SLOT_LABEL_TYPE.");
346                 }
347                 returnValue.setProperty(keyString, properties.getProperty(keyString));
348             }
349         }
350         return returnValue;
351     }
