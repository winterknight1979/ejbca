13  package org.cesecore.configuration;
14  
15  import java.util.Properties;
16  
17  import org.cesecore.authentication.tokens.AuthenticationToken;
18  import org.cesecore.authorization.AuthorizationDeniedException;
19  /** 
20   * Session bean to handle global configuration and such.
21   * 
22   * @version $Id: GlobalConfigurationSession.java 23544 2016-05-25 11:24:19Z anatom $
23   */
24  public interface GlobalConfigurationSession {
25      
26      /**
27       * Retrieves the cached GlobalConfiguration. This cache is updated from
28       * persistence either by the time specified by
29       * MIN_TIME_BETWEEN_GLOBCONF_UPDATES or when {@link #flushConfigurationCache(String)}
30       * is executed. This method should be used in all cases where a quick
31       * response isn't necessary, otherwise use {@link #flushConfigurationCache(String)}.
32       * @param configID ID
33       * 
34       * @return the cached GlobalConfiguration value.
35       */
36      ConfigurationBase getCachedConfiguration(String configID);
37  
38      /** Clear and load global configuration cache. 
39       * @param configID ID*/
40      void flushConfigurationCache(String configID);
41      
42      /** @param admin Admin
43       * @param configID ID
44       * @return all currently used properties (configured in conf/*.properties.
45       * Required admin access to '/' to dump these properties. 
46       * @throws AuthorizationDeniedException if privileges are insufficient 
47       */
48      Properties getAllProperties(AuthenticationToken admin, String configID) throws AuthorizationDeniedException;
49      
50      /** Saves the GlobalConfiguration. 
51      *
52      * @param admin an authentication token
53      * @param conf the new Configuration
54      * 
55      * @throws AuthorizationDeniedException if admin was not authorized to edit the specific configuration
56      */
57     void saveConfiguration(AuthenticationToken admin, ConfigurationBase conf) throws AuthorizationDeniedException;
58  
59  }
