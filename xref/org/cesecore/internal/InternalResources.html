74      private void setupResources(String resLocation) {
75          final String primaryLanguage = CesecoreConfiguration.getInternalResourcesPreferredLanguage().toLowerCase(Locale.ENGLISH);
76          final String secondaryLanguage = CesecoreConfiguration.getInternalResourcesSecondaryLanguage().toLowerCase(Locale.ENGLISH);
77          // The test flag is defined when called from test code (junit)
78          InputStream primaryStream = null;
79          InputStream secondaryStream = null;
80          try {
81  
82              primaryStream = InternalResources.class.getResourceAsStream(resLocation + primaryLanguage + ".properties");
83              if (primaryStream == null) {
84              	try {
85              		primaryStream = new FileInputStream(resLocation + primaryLanguage + ".properties");
86                  } catch (FileNotFoundException e) {
87                      log.error("Localization files not found: "+e.getMessage());
88                  }
89              }
90              secondaryStream = InternalResources.class.getResourceAsStream(resLocation + secondaryLanguage + ".properties");
91              if (secondaryStream == null) {
92              	try {
93              		secondaryStream = new FileInputStream(resLocation + secondaryLanguage + ".properties");
94                  } catch (FileNotFoundException e) {
95                      log.error("Localization files not found: "+e.getMessage());
96                  }
97              }
98  
99              try {
100                 if (primaryStream != null) {
101                     primaryResource.load(primaryStream);
102                 } else {
103                     log.warn("primaryResourse == null");
104                 }
105                 if (secondaryStream != null) {
106                     secondaryResource.load(secondaryStream);
107                 } else {
108                     log.warn("secondaryResource == null");
109                 }
110             } catch (IOException e) {
111                 log.error("Error reading internal resourcefile", e);
112             }
113         } finally {
114             try {
115                 if (primaryStream != null) {
116                     primaryStream.close();
117                 }
118                 if (secondaryStream != null) {
119                     secondaryStream.close();
120                 }
121             } catch (IOException e) {
122                 log.error("Error closing internal resources language streams: ", e);
123             }
124         }
125     }

135     /**
136      * Method returning the localized message for the given resource key.
137      * 
138      * It first looks up in the primary language then in the secondary If not
139      * found in any of the resource file "no text" is returned.
140      * <br><br>
141      * NOTE: String is immutable and you will get a copy of the String instead
142      * of a reference to it. This is more memory consuming than using
143      * getLocalizedMessageCs(..) if you pass on the result to Log4J.
144      * @see #getLocalizedMessageCs(String, Object...)
145      * 
146      * @param key
147      *            is the key searched for in the resource files
148      * @param params
149      *            indicates the parameter that will be replaced by {X} in the
150      *            language resource, a maximum of 10 parameters can be given.
151      * 
152      *            Ex Calling the method with key = TEST and param0 set to "hi"
153      *            and the resource file have "TEST = messages is {0}" will
154      *            result in the string "message is hi".
155      * 
156      * @return The message as a String, not trimmed for whitespace
157      */
158     public String getLocalizedMessage(final String key, final Object... params) {
159         return getLocalizedMessageCs(key, params).toString();
160     }

162     /**
163      * Method returning the localized message for the given resource key.
164      * 
165      * It first looks up in the primary language then in the secondary If not
166      * found in any of the resource file "no text" is returned.
167      * 
168      * @param key
169      *            is the key searched for in the resource files
170      * @param params
171      *            indicates the parameter that will be replaced by {X} in the
172      *            language resource, a maximum of 10 parameters can be given.
173      * 
174      *            Ex Calling the method with key = TEST and param0 set to "hi"
175      *            and the resource file have "TEST = messages is {0}" will
176      *            result in the string "message is hi".
177      * 
178      * @return The message as a CharSequence, the return value is not trimmed for whitespace. 
179      */
180     protected CharSequence getLocalizedMessageCs(final String key, final Object... params) {
181         final StringBuilder sb = new StringBuilder();
182         return getLocalizedMessageInternal(sb, key, params);
183     }
184 
185     /** Lookup the default string if the StringBuilder is empty. Perform place holder replacement processing. 
186      * @param sb StringBuilder buffer
187      * @param key key
188      * @param params parameters
189      * @return processed message */
190     protected CharSequence getLocalizedMessageInternal(final StringBuilder sb, final String key, final Object... params) {
191         if (sb.length()==0) {
192             if (primaryResource.containsKey(key)) {
193                 sb.append(primaryResource.getProperty(key));
194             } else if (secondaryResource.containsKey(key)) {
195                 sb.append(secondaryResource.getProperty(key));
196             } else {
197                 sb.append(key);
198                 for(Object param : params) {
199                     if (param != null) {
200                         sb.append(", ").append(param.toString());
201                     }
202                 }
203             }
204         }
205         replacePlaceholders(sb, params);
206         if (log.isTraceEnabled()) {
207             log.trace(key + "=" + sb.toString());
208         }
209         return sb;
210     }
211     
212     public static void replacePlaceholders(final StringBuilder sb, final Object... params) {
213         for (int i=0; i<params.length; i++) {
214             replaceAll(sb, i, params[i]);
215         }
216         removeUnusedPlaceHolders(sb, params.length);
217     }
218     
219     /** @return a lazily instantiated array of place holders like "{number}". */
220     private static String[] getPlaceHolders() {
221         if (placeHolders==null) {
222             final String[] arr = new String[100];
223             for (int i=0; i<arr.length; i++) {
224                 arr[i] = new StringBuilder('{').append(i).append('}').toString();
225             }
226             placeHolders = arr;
227         }
228         return placeHolders;
229     }
230     
231     /** Replace any "{placeHolderIndex}" String that is present in the StringBuilder with 'replacementObject'. 
232      * @param sb StringBuilder buffer
233      * @param placeHolderIndex Index
234      * @param replacementObject Object */
235     private static void replaceAll(final StringBuilder sb, final int placeHolderIndex, final Object replacementObject) {
236         if (sb==null) {
237             log.error("No StringBuilder. Unable to create localized message.");
238             return;
239         }
240         final String[] placeHolders = getPlaceHolders();
241         if (placeHolderIndex<0 || placeHolderIndex>(placeHolders.length-1)) {
242             log.error("Place holder index out of range. Unable to create localized message.");
243             return;
244         }
245         final String placeHolder = placeHolders[placeHolderIndex];
246         final int placeHolderLength = placeHolder.length();
247         int currentIndex = -placeHolderLength;
248         int bar = 20; // never allow more than 20 placeholders to avoid recursion
249         if (replacementObject==null) {
250             while ((currentIndex=sb.indexOf(placeHolder, currentIndex+placeHolderLength))!=-1 && bar > 0) {
251                 sb.delete(currentIndex-1, currentIndex+placeHolderLength);
252                 bar--;
253             }
254         } else {
255             final String replacement = replacementObject.toString();
256             while ((currentIndex=sb.indexOf(placeHolder, currentIndex+placeHolderLength))!=-1 && bar > 0) {
257                 sb.replace(currentIndex-1, currentIndex+placeHolderLength, replacement);
258                 bar--;
259             }
260         }
261     }
262 
263     /** Remove any "{number}" string that is still present in the StringBuilder where number starts with 'startPlaceHolderIndex'. 
264      * @param sb StringBuilder buffer
265      * @param startPlaceHolderIndex  index */
266     private static void removeUnusedPlaceHolders(final StringBuilder sb, final int startPlaceHolderIndex) {
267         final String[] placeHolders = getPlaceHolders();
268         if (startPlaceHolderIndex<0 || startPlaceHolderIndex>(placeHolders.length-1)) {
269             log.error("Place holder index out of range. Unable to create localized message.");
270             return;
271         }
272         for (int i=startPlaceHolderIndex; i<placeHolders.length; i++) {
273             final String placeHolder = placeHolders[i];
274             final int placeHolderLength = placeHolder.length();
275             int currentIndex = -placeHolderLength;
276             boolean someThingRemoved = false;
277             while ((currentIndex=sb.indexOf(placeHolder, currentIndex+placeHolderLength))!=-1) {
278                 sb.delete(currentIndex-1, currentIndex+placeHolderLength);
279                 someThingRemoved = true;
280             }
281             if (!someThingRemoved) {
282                 // If a place holder with the current number didn't exist, we assume that the ones with higher doesn't either..
283                 break;
284             }
285         }
286     }
