116     @Transient
117     @SuppressWarnings({"rawtypes", "unchecked"})
118     public void setDataMap(final LinkedHashMap<Object, Object> dataMap) {
119         try {
120             // We must base64 encode string for UTF safety
121             final LinkedHashMap<?, ?> a = new Base64PutHashMap();
122             a.putAll((LinkedHashMap)dataMap);
123             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
124             final XMLEncoder encoder = new XMLEncoder(baos);
125             encoder.writeObject(a);
126             encoder.close();
127             final String data = baos.toString("UTF8");
128             setRawData(data);
129         } catch (UnsupportedEncodingException e) {
130             throw new IllegalStateException(e);
131         }
132     }
