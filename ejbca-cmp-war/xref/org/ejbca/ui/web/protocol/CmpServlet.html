38  /**
39   * Servlet implementing server side of the Certificate Management Protocols (CMP)

41   * @version $Id: CmpServlet.java 28713 2018-04-13 14:35:47Z anatom $

43  public class CmpServlet extends HttpServlet {
44  
45      private static final long serialVersionUID = 1L;
46      private static final Logger log = Logger.getLogger(CmpServlet.class);
47      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
48      
49      /** Only intended to check if Peer connected instance is authorized to CMP at all. */
50      private final AuthenticationToken raCmpAuthCheckToken = new AlwaysAllowLocalAuthenticationToken(new UsernamePrincipal("cmpProtocolAuthCheck"));
51      
52      private static final String DEFAULT_CMP_ALIAS = "cmp";
53      
54      @EJB
55      private RaMasterApiProxyBeanLocal raMasterApiProxyBean;
56      @EJB
57      private GlobalConfigurationSessionLocal globalConfigurationSession;
58      /**
59       * Handles HTTP post
60       * 
61       * @param request java standard arg
62       * @param response java standard arg
63       * 
64       * @throws IOException input/output error
65       */
66      @Override
67      public void doPost(final HttpServletRequest request, final HttpServletResponse response) throws IOException {
68          if (log.isTraceEnabled()) {
69              log.trace(">doPost()");
70          }
71          boolean isProtocolAuthorized = raMasterApiProxyBean.isAuthorizedNoLogging(raCmpAuthCheckToken, AccessRulesConstants.REGULAR_PEERPROTOCOL_CMP);
72          try {
73              if (!isProtocolAuthorized) {
74                  log.info("CMP Protocol not authorized for this Peer");
75                  response.sendError(HttpServletResponse.SC_FORBIDDEN, "CMP Protocol not authorized for this Peer");
76                  return;
77              }
78              final String alias = getAlias(request.getPathInfo());
79              if(alias.length() > 32) {
80                  log.info("Unaccepted alias more than 32 characters.");
81                  response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Unaccepted alias more than 32 characters.");
82                  return;
83              }
84              final ServletInputStream sin = request.getInputStream();
85              // This small code snippet is inspired/copied by apache IO utils to Tomas Gustavsson...
86              final ByteArrayOutputStream output = new ByteArrayOutputStream();
87              final byte[] buf = new byte[1024];
88              int n = 0;
89              int bytesRead = 0;
90              while (-1 != (n = sin.read(buf))) {
91                  bytesRead += n;
92                  if (bytesRead > LimitLengthASN1Reader.MAX_REQUEST_SIZE) {
93                      throw new IllegalArgumentException("Request is larger than "+LimitLengthASN1Reader.MAX_REQUEST_SIZE+" bytes.");
94                  }
95                  output.write(buf, 0, n);
96              }
97              service(output.toByteArray(), request.getRemoteAddr(), response, alias);
98          } catch (IOException | RuntimeException e) {
99          	// TODO: localize.
100             response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Bad request. ");
101             log.info(intres.getLocalizedMessage("cmp.errornoasn1"), e);
102         }
103         if (log.isTraceEnabled()) {
104             log.trace("<doPost()");
105         }
106     }

108     /**
109      * Handles HTTP get
110      * 
111      * @param request java standard arg
112      * @param response java standard arg
113      * 
114      * @throws IOException input/output error
115      */
116     @Override
117     public void doGet(final HttpServletRequest request, final HttpServletResponse response) throws IOException {
118         if (log.isTraceEnabled()) {
119             log.trace(">doGet()");
120         }
121         log.info("Received un-allowed method GET in CMP servlet: query string=" + request.getQueryString());
122         response.sendError(HttpServletResponse.SC_METHOD_NOT_ALLOWED, "You can only use POST!");
123         if (log.isTraceEnabled()) {
124             log.trace("<doGet()");
125         }
126     }

128     private void service(final byte[] pkiMessageBytes, final String remoteAddr, final HttpServletResponse response, String alias) throws IOException {
129         try {
130             log.info(intres.getLocalizedMessage("cmp.receivedmsg", remoteAddr, alias));
131             final long startTime = System.currentTimeMillis();
132             byte[] result = null;
133             try {
134                 final AuthenticationToken authenticationToken = new AlwaysAllowLocalAuthenticationToken(new WebPrincipal("CmpServlet", remoteAddr));
135                 result = raMasterApiProxyBean.cmpDispatch(authenticationToken, pkiMessageBytes, alias);
136             } catch (NoSuchAliasException e) {
137                 // The CMP alias does not exist
138                 response.sendError(HttpServletResponse.SC_NOT_FOUND, "Error in CmpServlet: No such alias");
139                 log.info(e.getMessage());
140                 return;
141             }
142             if (result == null) {
143                 // If resp is null, it means that the dispatcher failed to process the message.
144                 response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, intres.getLocalizedMessage("cmp.errornullresp"));
145                 return;
146             }
147             // Add no-cache headers as defined in 
148             // http://tools.ietf.org/html/draft-ietf-pkix-cmp-transport-protocols-14
149             ServletUtils.addCacheHeaders(response);
150             // Send back CMP response
151             RequestHelper.sendBinaryBytes(result, response, "application/pkixcmp", null);
152             final long endTime = System.currentTimeMillis();
153             log.info(intres.getLocalizedMessage("cmp.sentresponsemsg", remoteAddr, Long.valueOf(endTime - startTime)));
154         } catch (IOException | RuntimeException e) {
155             log.error("Error in CmpServlet:", e);
156             response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Error in CmpServlet.");
157         }
158     }
159     
160     private String getAlias(final String pathInfo) {
161         // PathInfo contains the alias used for CMP configuration. 
162         // The CMP URL for custom configuration looks like: http://HOST:PORT/ejbca/publicweb/cmp/*
163         // pathInfo contains what * is and should have the form "/<SOME IDENTIFYING TEXT>". We extract the "SOME IDENTIFYING 
164         // TEXT" and that will be the CMP configuration alias.
165         final String alias;
166         if (pathInfo!=null && pathInfo.length()>1) {
167             alias = pathInfo.substring(1);
168             if (log.isDebugEnabled()) {
169                 log.debug("Using CMP configuration alias: " + alias);
170             }
171         } else {
172             alias = DEFAULT_CMP_ALIAS;
173             if (log.isDebugEnabled()) {
174                 log.debug("No CMP alias specified in the URL. Using the default alias: " + DEFAULT_CMP_ALIAS);
175             }
176         }
177         return alias;
178     }
