80      @Override
81      public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
82          try {
83              if (log.isTraceEnabled()) {
84                  log.trace(">doGet()");
85              }
86              final String keyRenewalSignerDN =  request.getParameter("renewSigner");
87              final boolean performKeyRenewal = keyRenewalSignerDN!=null && keyRenewalSignerDN.length()>0;           
88              // We have a command to force reloading of keys that can only be run from localhost
89              final boolean doReload = StringUtils.equals(request.getParameter("reloadkeys"), "true");
90              final String newConfig = request.getParameter("newConfig");
91              final boolean doNewConfig = newConfig != null && newConfig.length() > 0;
92              final boolean doRestoreConfig = request.getParameter("restoreConfig") != null;
93              final String remote = request.getRemoteAddr();
94              if (doReload || doNewConfig || doRestoreConfig) {
95                  if (!StringUtils.equals(remote, "127.0.0.1")) {
96                      log.info("Got reloadkeys or updateConfig of restoreConfig command from unauthorized ip: " + remote);
97                      response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
98                      return;
99                  }
100             }
101             if (doReload) {
102                 log.info(intres.getLocalizedMessage("ocsp.reloadkeys", remote));
103                 // Reload CA certificates
104                 integratedOcspResponseGeneratorSession.reloadOcspSigningCache();
105                 return;
106             }
107             if (doNewConfig) {
108                 final String aConfig[] = newConfig.split("\\|\\|");
109                 for (int i = 0; i < aConfig.length; i++) {
110                     log.debug("Config change: " + aConfig[i]);
111                     final int separatorIx = aConfig[i].indexOf('=');
112                     if (separatorIx < 0) {
113                         ConfigurationHolder.updateConfiguration(aConfig[i], null);
114                         continue;
115                     }
116                     ConfigurationHolder.updateConfiguration(aConfig[i].substring(0, separatorIx),
117                             aConfig[i].substring(separatorIx + 1, aConfig[i].length()));
118                 }
119                 // This setting is cached and must be cleared on config update
120                 OcspConfiguration.clearAcceptedSignatureAlgorithmCache();
121                 OcspConfigurationCache.INSTANCE.reloadConfiguration();
122                 log.info("Call from " + remote + " to update configuration");
123                 return;
124             }
125             if (doRestoreConfig) {
126                 ConfigurationHolder.restoreConfiguration();
127                 OcspConfigurationCache.INSTANCE.reloadConfiguration();
128                 log.info("Call from " + remote + " to restore configuration.");
129                 return;
130             }
131             if ( performKeyRenewal ) {
132                 final Set<String> rekeyingTriggeringHosts = OcspConfiguration.getRekeyingTriggingHosts();
133                 if ( !rekeyingTriggeringHosts.contains(remote) ) {
134                     log.info( intres.getLocalizedMessage("ocsp.rekey.triggered.unauthorized.ip", remote) );
135                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
136                     return;
137                 }
138                 final String rekeyingTriggingPassword = OcspConfiguration.getRekeyingTriggingPassword();
139                 if ( rekeyingTriggingPassword==null ) {
140                     log.info( intres.getLocalizedMessage("ocsp.rekey.triggered.not.enabled",remote) );
141                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
142                     return;
143                 }
144                 final String requestPassword = request.getParameter("password");
145                 final String keyrenewalSignerDn =  request.getParameter("renewSigner");
146                 if ( !rekeyingTriggingPassword.equals(requestPassword) ) {
147                     log.info( intres.getLocalizedMessage("ocsp.rekey.triggered.wrong.password", remote) );
148                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
149                     return;
150                 }
151                 try {
152                     ocspKeyRenewalSession.renewKeyStores(keyrenewalSignerDn);
153                 } catch (KeyStoreException e) {
154                     log.info( intres.getLocalizedMessage("ocsp.rekey.keystore.notactivated", remote) );
155                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
156                     return;
157                 } catch (CryptoTokenOfflineException e) {
158                     log.info( intres.getLocalizedMessage("ocsp.rekey.cryptotoken.notactivated", remote) );
159                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
160                     return;
161                 } catch (InvalidKeyException e) {                   
162                     log.info( intres.getLocalizedMessage("ocsp.rekey.invalid.key", remote) );
163                     response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
164                     return;
165                 }
166                 return;
167             }
168             processOcspRequest(request, response, HttpMethod.GET);
169         } finally {
170             if (log.isTraceEnabled()) {
171                 log.trace("<doGet()");
172             }
173         }
174     }

176     @Override
177     public void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
178         boolean protocolEnabled = ((AvailableProtocolsConfiguration)globalConfigurationSession.getCachedConfiguration(AvailableProtocolsConfiguration.CONFIGURATION_ID)).
179                 getProtocolStatus(AvailableProtocols.OCSP.getName());
180         if (log.isTraceEnabled()) {
181             log.trace(">doPost()");
182         }
183         if (!protocolEnabled) {
184             log.info("OCSP Protocol is disabled");
185             response.sendError(HttpServletResponse.SC_FORBIDDEN, "OCSP is disabled");
186             return;
187         }
188         try {
189             final String contentType = request.getHeader("Content-Type");
190             if (contentType != null && contentType.equalsIgnoreCase("application/ocsp-request")) {
191                 processOcspRequest(request, response, HttpMethod.POST);
192                 return;
193             }
194             final String remoteAddr = request.getRemoteAddr();
195             // Legacy support for activation using ClientToolBox. We will only use this once for upgrading the installation.
196             final String activationPassword = request.getHeader("activate");
197             if ( activationPassword!=null && remoteAddr.equals("127.0.0.1")) {
198                 try {
199                     log.warn("'active' will only be used for initial one-time upgrade."+
200                             " Use regular CryptoToken activation in EJB CLI or Admin GUI to active your responder keystores.");
201                     integratedOcspResponseGeneratorSession.adhocUpgradeFromPre60(activationPassword.toCharArray());
202                 } catch (Exception e) {
203                     log.error("Problem loading keys.", e);
204                     response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Problem. See ocsp responder server log.");
205                 }
206                 return;
207             }
208             if (contentType != null) {
209                 final String sError = "Content-type is not application/ocsp-request. It is \'" + HTMLTools.htmlescape(contentType) + "\'.";
210                 log.debug(sError);
211                 response.sendError(HttpServletResponse.SC_BAD_REQUEST, sError);
212                 return;
213             }
214             if (!remoteAddr.equals("127.0.0.1")) {
215                 final String sError = "You have connected from \'" + remoteAddr + "\'. You may only connect from 127.0.0.1";
216                 log.debug(sError);
217                 response.sendError(HttpServletResponse.SC_UNAUTHORIZED, sError);
218                 return;
219             }
220         } finally {
221             if (log.isTraceEnabled()) {
222                 log.trace("<doPost()");
223             }
224         }
225     }

227     private void processOcspRequest(HttpServletRequest request, HttpServletResponse response, final HttpMethod httpMethod) throws ServletException {
228         final String remoteAddress = request.getRemoteAddr();
229         final String xForwardedFor = StringTools.getCleanXForwardedFor(request.getHeader("X-Forwarded-For"));
230         final StringBuffer requestUrl = request.getRequestURL();
231         final int localTransactionId = TransactionCounter.INSTANCE.getTransactionNumber();
232         // Create the transaction logger for this transaction.
233         TransactionLogger transactionLogger = new TransactionLogger(localTransactionId, GuidHolder.INSTANCE.getGlobalUid(), remoteAddress);
234         // Create the audit logger for this transaction.
235         AuditLogger auditLogger = new AuditLogger("", localTransactionId, GuidHolder.INSTANCE.getGlobalUid(), remoteAddress);
236         try {
237             if (auditLogger.isEnabled()) {
238                 auditLogger.paramPut(PatternLogger.LOG_ID, Integer.valueOf(localTransactionId));
239                 auditLogger.paramPut(PatternLogger.SESSION_ID, sessionID);
240                 auditLogger.paramPut(PatternLogger.CLIENT_IP, remoteAddress);
241             }
242             if (transactionLogger.isEnabled()) {
243                 transactionLogger.paramPut(PatternLogger.LOG_ID, Integer.valueOf(localTransactionId));
244                 transactionLogger.paramPut(PatternLogger.SESSION_ID, sessionID);
245                 transactionLogger.paramPut(PatternLogger.CLIENT_IP, remoteAddress);
246                 transactionLogger.paramPut(TransactionLogger.FORWARDED_FOR, xForwardedFor);
247             }
248             OCSPRespBuilder responseGenerator = new OCSPRespBuilder();
249             OcspResponseInformation ocspResponseInformation = null;
250             try {
251                 byte[] requestBytes = checkAndGetRequestBytes(request, httpMethod);
252                 X509Certificate[] requestCertificates = (X509Certificate[]) request.getAttribute("javax.servlet.request.X509Certificate");
253                 ocspResponseInformation = integratedOcspResponseGeneratorSession.getOcspResponse(
254                         requestBytes, requestCertificates, remoteAddress, xForwardedFor, requestUrl, auditLogger, transactionLogger);
255             } catch (MalformedRequestException e) {
256                 if (transactionLogger.isEnabled()) {
257                     transactionLogger.paramPut(PatternLogger.PROCESS_TIME, PatternLogger.PROCESS_TIME);
258                 }
259                 if (auditLogger.isEnabled()) {
260                     auditLogger.paramPut(PatternLogger.PROCESS_TIME, PatternLogger.PROCESS_TIME);
261                 }
262                 String errMsg = intres.getLocalizedMessage("ocsp.errorprocessreq", e.getMessage());
263                 log.info(errMsg);
264                 if (log.isDebugEnabled()) {
265                     log.debug(errMsg, e);
266                 }
267                 // RFC 2560: responseBytes are not set on error.
268                 ocspResponseInformation = new OcspResponseInformation(responseGenerator.build(OCSPRespBuilder.MALFORMED_REQUEST, null), OcspConfiguration.getMaxAge(CertificateProfileConstants.CERTPROFILE_NO_PROFILE), null);
269                 if (transactionLogger.isEnabled()) {
270                     transactionLogger.paramPut(TransactionLogger.STATUS, OCSPRespBuilder.MALFORMED_REQUEST);
271                     transactionLogger.writeln();
272                 }
273                 if (auditLogger.isEnabled()) {
274                     auditLogger.paramPut(AuditLogger.STATUS, OCSPRespBuilder.MALFORMED_REQUEST);
275                 }
276             } catch (Throwable e) { // NOPMD, we really want to catch everything here to return internal error on unexpected errors
277                 if (transactionLogger.isEnabled()) {
278                     transactionLogger.paramPut(IPatternLogger.PROCESS_TIME, IPatternLogger.PROCESS_TIME);
279                 }
280                 if (auditLogger.isEnabled()) {
281                     auditLogger.paramPut(IPatternLogger.PROCESS_TIME, IPatternLogger.PROCESS_TIME);
282                 }
283                 final String errMsg = intres.getLocalizedMessage("ocsp.errorprocessreq", e.getMessage());
284                 log.info(errMsg);
285                 if (log.isDebugEnabled()) {
286                     log.debug(errMsg, e);
287                 }
288                 // RFC 2560: responseBytes are not set on error.
289                 ocspResponseInformation = new OcspResponseInformation(responseGenerator.build(OCSPRespBuilder.INTERNAL_ERROR, null), OcspConfiguration.getMaxAge(CertificateProfileConstants.CERTPROFILE_NO_PROFILE), null);
290                 if (transactionLogger.isEnabled()) {
291                     transactionLogger.paramPut(TransactionLogger.STATUS, OCSPRespBuilder.INTERNAL_ERROR);
292                     transactionLogger.writeln();
293                 }
294                 if (auditLogger.isEnabled()) {
295                     auditLogger.paramPut(AuditLogger.STATUS, OCSPRespBuilder.INTERNAL_ERROR);
296                 }
297             }
298             byte[] ocspResponseBytes = ocspResponseInformation.getOcspResponse();    
299             response.setContentType("application/ocsp-response");
300             response.setContentLength(ocspResponseBytes.length);
301             if (HttpMethod.GET.equals(httpMethod)) {
302                 addRfc5019CacheHeaders(request, response, ocspResponseInformation);
303             } else {
304                 if (log.isDebugEnabled()) {
305                     log.debug("Will not add RFC 5019 cache headers: \"clients MUST use the GET method (to enable OCSP response caching)\"");
306                 }
307             }
308             
309             if (HttpMethod.POST.equals(httpMethod)) {
310                 addOscpPostHeaders(response, ocspResponseInformation);
311             }
312             
313             response.getOutputStream().write(ocspResponseBytes);
314             response.getOutputStream().flush();
315         } catch (Exception e) {
316             log.error("", e);
317             transactionLogger.flush();
318             auditLogger.flush();
319         }
320     }

322     private void addOscpPostHeaders(HttpServletResponse response, OcspResponseInformation ocspResponseInformation) {
323         
324         if (!ocspResponseInformation.shouldAddCacheHeaders()) {
325             return;
326         }
327         
328         final long thisUpdate = ocspResponseInformation.getThisUpdate();
329         long nextUpdate = ocspResponseInformation.getNextUpdate();
330         String responseHeader = ocspResponseInformation.getResponseHeader();
331         
332         // RFC 5019 6.2: Last-Modified: date and time at which the OCSP responder last modified the response. == thisUpdate
333         response.setDateHeader("Last-Modified", thisUpdate);
334         // RFC 5019 6.2: Expires: This date and time will be the same as the nextUpdate timestamp in the OCSP response itself.
335         response.setDateHeader("Expires", nextUpdate); // This is overridden by max-age on HTTP/1.1 compatible components
336         // RFC 5019 6.2: This profile RECOMMENDS that the ETag value be the ASCII HEX representation of the SHA1 hash of the OCSPResponse structure.
337         response.setHeader("ETag", "\"" + responseHeader + "\"");
338     }

342     /**
343      * RFC 2560 does not specify how cache headers should be used, but RFC 5019 does. Therefore we will only
344      * add the headers if the requirements of RFC 5019 is fulfilled: A GET-request, a single embedded response,
345      * the response contains a nextUpdate and no nonce is present.
346      * @param request is the incoming HttpServletRequest
347      * @param response is the outgoing HttpServletResponse
348      * @param ocspResponseInformation is the previously parsed data carrier that wraps the contents of an OCSPResp
349      * @throws IOException org.bouncycastle.cert.ocsp.OCSPException 
350      * @throws org.bouncycastle.cert.ocsp.OCSPException  fail
351      */
352     private void addRfc5019CacheHeaders(HttpServletRequest request, HttpServletResponse response, OcspResponseInformation ocspResponseInformation) 
353                 throws IOException, org.bouncycastle.cert.ocsp.OCSPException {
354         if (!ocspResponseInformation.shouldAddCacheHeaders()) {
355             return;
356         } 
357         if (ocspResponseInformation.getMaxAge() <= 0) {
358             if (log.isDebugEnabled()) {
359                 log.debug("Will not add RFC 5019 cache headers: RFC 5019 6.2: max-age should be 'later than thisUpdate but earlier than nextUpdate'.");
360             }
361             return;
362         }
363         final long now = System.currentTimeMillis();
364         final long thisUpdate = ocspResponseInformation.getThisUpdate();
365         long nextUpdate = ocspResponseInformation.getNextUpdate();
366         // Max-age and nextUpdate will be truncated if they point longer than the OCSP signer certificate is valid
367         // If we shrink nextUpdate, max-age will be automatically shrunk
368         X509Certificate signerCert = ocspResponseInformation.getSignerCert();
369         if (signerCert != null && signerCert.getNotAfter().getTime() < nextUpdate) {
370             nextUpdate = signerCert.getNotAfter().getTime();
371         }
372         // RFC 5019 6.2: Date: The date and time at which the OCSP server generated the HTTP response.
373         // On JBoss AS the "Date"-header is cached for 1 second, so this value will be overwritten and off by up to a second 
374         response.setDateHeader("Date", now);
375         // RFC 5019 6.2: Last-Modified: date and time at which the OCSP responder last modified the response. == thisUpdate
376         response.setDateHeader("Last-Modified", thisUpdate);
377         // RFC 5019 6.2: Expires: This date and time will be the same as the nextUpdate timestamp in the OCSP response itself.
378         response.setDateHeader("Expires", nextUpdate); // This is overridden by max-age on HTTP/1.1 compatible components
379         // RFC 5019 6.2: This profile RECOMMENDS that the ETag value be the ASCII HEX representation of the SHA1 hash of the OCSPResponse structure.
380         response.setHeader("ETag", "\"" + ocspResponseInformation.getResponseHeader() + "\"");
381         if (ocspResponseInformation.isExplicitNoCache()) {
382             // Note that using no-cache here is not conforming to RFC5019, but with more recent CABForum discussions it seems RFC5019 will not
383             // be followed, or will be changed. (See ECA-3289)
384             response.setHeader("Cache-Control", "no-cache, must-revalidate"); //HTTP 1.1
385             response.setHeader("Pragma", "no-cache"); //HTTP 1.0 
386         } else {
387             // Max age is retrieved in milliseconds, but it must be in seconds in the cache-control header
388             long maxAge = ocspResponseInformation.getMaxAge();
389             if (maxAge >= (nextUpdate - thisUpdate)) {
390                 if (log.isDebugEnabled()) {
391                     log.debug("max-age ("+maxAge+") is >= (nextUpdate - thisUpdate): ("+nextUpdate+" - "+thisUpdate+")");
392                 }
393                 maxAge = nextUpdate - thisUpdate - 1;
394                 log.warn(intres.getLocalizedMessage("ocsp.shrinkmaxage", maxAge));
395             } 
396             response.setHeader("Cache-Control", "max-age=" + (maxAge / 1000L) + ",public,no-transform,must-revalidate");
397         }

400     /**
401      * Reads the request bytes and verifies min and max size of the request. If an error occurs it throws a MalformedRequestException. 
402      * Can get request bytes both from a HTTP GET and POST request
403      * 
404      * @param request request
405      * @param httpMethod response
406      * @return the request bytes or null if an error occured.
407      * @throws IOException In case there is no stream to read
408      * @throws MalformedRequestException  fail
409      */
410     private byte[] checkAndGetRequestBytes(HttpServletRequest request, HttpMethod httpMethod) throws IOException, MalformedRequestException {
411         final byte[] ret;
412         // Get the request data
413         final int n = request.getContentLength();
414         // Expect n might be -1 for HTTP GET requests
415         if (log.isDebugEnabled()) {
416             log.debug(">checkAndGetRequestBytes. Received " + httpMethod.name() + " request with content length: " + n + " from " + request.getRemoteAddr());
417         }
418         if (n > LimitLengthASN1Reader.MAX_REQUEST_SIZE) {
419             String msg = intres.getLocalizedMessage("ocsp.toolarge", LimitLengthASN1Reader.MAX_REQUEST_SIZE, n);
420             log.info(msg);
421             throw new MalformedRequestException(msg);
422         }
423         // So we passed basic tests, now we can read the bytes, but still keep an eye on the size
424         // we can not fully trust the sent content length.
425         if (HttpMethod.POST.equals(httpMethod)) {
426             final ServletInputStream in = request.getInputStream(); // ServletInputStream does not have to be closed, container handles this
427             LimitLengthASN1Reader limitLengthASN1Reader = new LimitLengthASN1Reader(in, n);
428             try {
429                 ret = limitLengthASN1Reader.readFirstASN1Object();
430                 if (n > ret.length) {
431                     // The client is sending more data than the OCSP request. It might be slightly broken or trying to bog down the server on purpose.
432                     // In the interest of not breaking existing systems that might have slightly broken clients we just log for a warning for now.
433                     String msg = intres.getLocalizedMessage("ocsp.additionaldata", ret.length, n);
434                     log.warn(msg);
435                 }
436             } finally {
437                 limitLengthASN1Reader.close();
438             }
439         } else if (HttpMethod.GET.equals(httpMethod)) {
440             // GET request
441             final StringBuffer url = request.getRequestURL();
442             // RFC2560 A.1.1 says that request longer than 255 bytes SHOULD be sent by POST, we support GET for longer requests anyway.
443             if (url.length() <= LimitLengthASN1Reader.MAX_REQUEST_SIZE) {
444                 final String decodedRequest;
445                 try {
446                     // We have to extract the pathInfo manually, to avoid multiple slashes being converted to a single
447                     // According to RFC 2396 2.2 chars only have to encoded if they conflict with the purpose, so
448                     // we can for example expect both '/' and "%2F" in the request.
449                     final String fullServletpath = request.getContextPath() + request.getServletPath();
450                     final int paramIx = Math.max(url.indexOf(fullServletpath), 0) + fullServletpath.length() + 1;
451                     final String requestString = paramIx < url.length() ? url.substring(paramIx) : "";
452                     decodedRequest = URLDecoder.decode(requestString, "UTF-8").replaceAll(" ", "+");
453                 } catch (Exception e) {
454                     String msg = intres.getLocalizedMessage("ocsp.badurlenc");
455                     log.info(msg);
456                     throw new MalformedRequestException(e);
457                 }
458                 if (decodedRequest != null && decodedRequest.length() > 0) {
459                     if (log.isDebugEnabled()) {
460                         // Don't log the request if it's too long, we don't want to cause denial of service by filling log files or buffers.
461                         if (decodedRequest.length() < 2048) {
462                             log.debug("decodedRequest: " + decodedRequest);
463                         } else {
464                             log.debug("decodedRequest too long to log: " + decodedRequest.length());
465                         }
466                     }
467                     try {
468                         ret = Base64.decode(decodedRequest.getBytes());
469                     } catch (Exception e) {
470                         String msg = intres.getLocalizedMessage("ocsp.badurlenc");
471                         log.info(msg);
472                         throw new MalformedRequestException(e);
473                     }
474                 } else {
475                     String msg = intres.getLocalizedMessage("ocsp.missingreq");
476                     log.info(msg);
477                     throw new MalformedRequestException(msg);
478                 }
479             } else {
480                 String msg = intres.getLocalizedMessage("ocsp.toolarge", LimitLengthASN1Reader.MAX_REQUEST_SIZE, url.length());
481                 log.info(msg);
482                 throw new MalformedRequestException(msg);
483             }
484         } else {
485             // Strange, an unknown method
486             String msg = intres.getLocalizedMessage("ocsp.unknownmethod", request.getMethod());
487             log.info(msg);
488             throw new MalformedRequestException(msg);
489         }
490         // Make a final check that we actually received something
491         if (ret==null || ret.length==0) {
492             String msg = intres.getLocalizedMessage("ocsp.emptyreq", request.getRemoteAddr());
493             log.info(msg);
494             throw new MalformedRequestException(msg);
495         }
496         return ret;
497     }
