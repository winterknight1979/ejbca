29  /**
30   * Test reparation of XML serialized objects.
31   * 
32   * @version $Id: FixEndOfBrokenXMLTest.java 34163 2020-01-02 15:00:17Z samuellb $
33   */
34  public class FixEndOfBrokenXMLTest {
35  
36  	private static Logger log = Logger.getLogger(FixEndOfBrokenXMLTest.class);
37  	private static String CHAR_ENCODING = "UTF-8";
38  
39  	/**
40  	 * This test will take a XML serialized object and try to repair the end of it
41  	 * with a known String. It depends on the external file "test.xml".
42  	 * 
43  	 * This test does not reproduce the production failure that the fix was written
44  	 * for, but instead tests the general ability to repair XML under the current JDK.
45  	 * @throws Exception Fail
46  	 */ 
47  	@Test
48  	public void test01() throws Exception {
49  		log.trace(">test01");
50  		final int limit = "</string></void></object></java>".length(); // This is what we expect to be able to repair
51  		final byte testXml[] = readXmlFromFile();
52  		// We need to decode and encode it with the current JDK we are running, 
53  		// because the JDK version is in the XML and different JDKs add different amount of whitespace
54  		byte[] xml = decodeAndEncode(testXml);
55  		for ( int nrOfBytesMissing = 0; xml.length>nrOfBytesMissing; nrOfBytesMissing++ ) {
56  			final byte brokenXml[] = Arrays.copyOf(xml, xml.length-nrOfBytesMissing);
57  			final byte fixedXml[] = FixEndOfBrokenXML.fixXML(new String(brokenXml, CHAR_ENCODING), "string", "</void></object></java>").getBytes(CHAR_ENCODING);
58  
59  			final ByteArrayOutputStream baos = new ByteArrayOutputStream();
60              try (SecureXMLDecoder decoder = new SecureXMLDecoder(new ByteArrayInputStream(fixedXml), true)) {
61                  try (XMLEncoder encoder = new XMLEncoder(baos)) {
62                      encoder.writeObject(decoder.readObject());
63                  } catch (Throwable t) {
64                      log.error("Exception: ", t);
65                      notPossibleToRemoveMoreBytes(nrOfBytesMissing, brokenXml, limit);
66                      return;
67                  }
68              }
69  			final byte decodedXml[] = baos.toByteArray();
70  			if ( !Arrays.equals(xml,decodedXml) ) {
71  				if (nrOfBytesMissing < limit) {
72  					assertEquals("Only possible to fix "+nrOfBytesMissing+" missing bytes. We should be able to handle "+limit+" missing bytes.", new String(xml), new String(decodedXml));
73  				}
74  				notPossibleToRemoveMoreBytes(nrOfBytesMissing, brokenXml, limit);
75  				return;
76  			}
77  		}
78  		log.trace("<test01");
79  	}
80  
81  	private byte[] readXmlFromFile() throws IOException {
82  		final InputStream is = FixEndOfBrokenXMLTest.class.getResourceAsStream("/test.xml");
83  		final ByteArrayOutputStream baos = new ByteArrayOutputStream();
84  		while( true ) {
85  			final int available = is.available();
86  			if ( available>0 ) {
87  				final byte tmp[] = new byte[available];
88  				is.read(tmp);
89  				baos.write(tmp);
90  				continue;
91  			}
92  			final int tmp = is.read();
93  			if ( tmp<0 ) {
94  				break;
95  			}
96  			baos.write(tmp);
97  		}
98  		return baos.toByteArray();
99  	}
100 
101 	private void notPossibleToRemoveMoreBytes(int nrOfBytesMissing, byte brokenXml[], final int limit) throws UnsupportedEncodingException {
102 		log.info("Repair tool not able to mend xml with "+nrOfBytesMissing+" missing chars in the end:\n" + new String(brokenXml, CHAR_ENCODING));
103 		assertFalse("Only possible to fix "+nrOfBytesMissing+" missing bytes. We should be able to handle "+limit+" missing bytes.", nrOfBytesMissing<limit);
104 	}
105 
106     private byte[] decodeAndEncode(final byte[] testXml) throws IOException {
107         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
108         try (SecureXMLDecoder dec = new SecureXMLDecoder(new ByteArrayInputStream(testXml), true)) {
109             try (XMLEncoder encoder = new XMLEncoder(baos)) {
110                 encoder.writeObject(dec.readObject());
111             }
112         }
113         return baos.toByteArray();
114     }

116 	/**
117 	 * If the start of the last string-element is broken, we will remove all
118 	 * elements after the last valid closing string-element.
119 	 * 
120 	 * This might not be a good thing since we loose more data than necessary,
121 	 * but at least we want to have this behavior demonstrated and documented.
122 	 * @throws Exception Fail
123 	 */
124 	@Test
125 	public void test02MissingLastStartOfString() throws Exception {
126 		log.trace(">test02MissingLastStartOfString");
127 		final String BROKEN_XML_2_PART1 =
128 			"<?xml version=\"1.0\" encoding=\"UTF-8\"?><java version=\"1.6.0_18\" class=\"java.beans.XMLDecoder\">"
129 			+ "<object class=\"org.cesecore.certificates.endentity.EndEntityInformation\">"
130 			// ...
131 			+ "<void property=\"email\"><string>null@null.com</string></void>";
132 			// ...
133 		final String BROKEN_XML_2_PART2 =
134 			"<void property=\"type\"><int>1</int></void>"
135 			+ "<void property=\"username\"><strin";	// "Broken": "g>10022428256</string></void></object></java>";
136 		final String toFix = new String((BROKEN_XML_2_PART1+BROKEN_XML_2_PART2).getBytes(CHAR_ENCODING));
137 		final String fixed = new String(FixEndOfBrokenXML.fixXML(toFix, "string", "</void></object></java>").getBytes(CHAR_ENCODING), CHAR_ENCODING);
138 		final String expected = BROKEN_XML_2_PART1 + "</object></java>";
139 		log.info("toFix:    " + toFix);
140 		log.info("fixed:    " + fixed);
141 		log.info("expected: " + expected);
142 		assertEquals("XMLFix has chnaged behaviour. Did not remove objects as originally designed.", expected, fixed);
143 		log.trace("<test02MissingLastStartOfString");
144 	}
