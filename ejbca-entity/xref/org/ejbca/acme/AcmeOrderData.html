43  @Entity
44  @Table(name = "AcmeOrderData")
45  public class AcmeOrderData extends ProtectedData implements Serializable {
46  
47      private static final long serialVersionUID = 1L;
48      private static final Logger log = Logger.getLogger(AcmeOrderData.class);
49  
50      @Id
51      private String orderId;
52      
53      private String accountId;
54      private String fingerprint;
55      private String status;
56      private String rawData;
57      private int rowVersion = 0;
58      private String rowProtection;
59  
60      public AcmeOrderData() {}
61  
62      public AcmeOrderData(final String orderId, final String accountId, final String fingerprint, final String status, final LinkedHashMap<Object,Object> dataMap) {
63          this.orderId = orderId;
64          this.accountId = accountId;
65          this.fingerprint = fingerprint;
66          this.status = status;
67          setDataMap(dataMap);
68      }
69      
70      public String getOrderId() {
71          return orderId;
72      }
73      
74      public void setOrderId(String orderId) {
75          this.orderId = orderId;
76      }
77  
78      public String getAccountId() {
79          return accountId;
80      }
81  
82      public void setAccountId(String accountId) {
83          this.accountId = accountId;
84      }
85      
86      public String getFingerprint() {
87          return fingerprint;
88      }
89  
90      public void setFingerprint(String fingerprint) {
91          this.fingerprint = fingerprint;
92      }
93      
94      public String getStatus() {
95          return status;
96      }
97  
98      public void setStatus(String status) {
99          this.status = status;
100     }
101 
102     public String getRawData() {
103         return rawData;
104     }
105 
106     public void setRawData(String rawData) {
107         this.rawData = rawData;
108     }
109 
110     @Transient
111     @SuppressWarnings("unchecked")
112     public LinkedHashMap<Object,Object> getDataMap() {
113         try (final SecureXMLDecoder decoder = new SecureXMLDecoder(new ByteArrayInputStream(getRawData().getBytes(StandardCharsets.UTF_8)))) {
114             // Handle Base64 encoded string values
115             return new Base64GetHashMap((Map<?,?>)decoder.readObject());
116         } catch (IOException e) {
117             final String msg = "Failed to parse AcmeOrderData data map in database: " + e.getMessage();
118             if (log.isDebugEnabled()) {
119                 log.debug(msg + ". Data:\n" + getRawData());
120             }
121             throw new IllegalStateException(msg, e);
122         }
123     }
124 
125     @Transient
126     public void setDataMap(final LinkedHashMap<Object,Object> dataMap) {
127         // We must base64 encode string for UTF safety
128         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
129         try (final XMLEncoder encoder = new XMLEncoder(baos)) {
130             encoder.writeObject(new Base64PutHashMap(dataMap));
131         }
132         setRawData(new String(baos.toByteArray(), StandardCharsets.UTF_8));
133     }
134     
135     public int getRowVersion() { return rowVersion; }
136     public void setRowVersion(int rowVersion) { this.rowVersion = rowVersion; }
137 
138     
139     @Override
140     public String getRowProtection() { return rowProtection; }
141     
142     @Override
143     public void setRowProtection(String rowProtection) { this.rowProtection = rowProtection; }
144 
145     
146     //
147     // Start Database integrity protection methods
148     //
149 
150     @Transient
151     @Override
152     protected String getProtectString(final int version) {
153         // rowVersion is automatically updated by JPA, so it's not important, it is only used for optimistic locking so we will not include that in the database protection
154         return new ProtectionStringBuilder().append(getOrderId()).append(getAccountId()).append(getRawData()).toString();
155     }
156 
157     @Transient
158     @Override
159     protected int getProtectVersion() {
160         return 1;
161     }
162 
163     @PrePersist
164     @PreUpdate
165     @Override
166     protected void protectData() {
167         super.protectData();
168     }
169 
170     @PostLoad
171     @Override
172     protected void verifyData() {
173         super.verifyData();
174     }
175 
176     @Override
177     @Transient
178     protected String getRowId() {
179         return getAccountId();
180     }
181 
182     //
183     // End Database integrity protection methods
184     //
185 }
