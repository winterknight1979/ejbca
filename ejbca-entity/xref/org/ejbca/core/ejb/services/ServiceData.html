127     /**
128      * Method that returns the service configuration data and updates it if necessary.
129      * @return config
130      */
131     @Transient
132     public ServiceConfiguration getServiceConfiguration() {
133         final HashMap<?, ?> h;
134     	try (SecureXMLDecoder decoder = new SecureXMLDecoder(new ByteArrayInputStream(getData().getBytes(StandardCharsets.UTF_8)))) {
135     	    h = (HashMap<?, ?>) decoder.readObject();
136     	} catch (IOException e) {
137             final String msg = "Failed to parse ServiceData data map in database: " + e.getMessage();
138             if (log.isDebugEnabled()) {
139                 log.debug(msg + ". Data:\n" + getData());
140             }
141             throw new IllegalStateException(msg, e);
142         }
143     	// Handle Base64 encoded string values
144     	HashMap<?, ?> data = new Base64GetHashMap(h);
145     	float oldversion = ((Float) data.get(UpgradeableDataHashMap.VERSION)).floatValue();
146     	ServiceConfiguration serviceConfiguration = new ServiceConfiguration();
147     	serviceConfiguration.loadData(data);
148     	if (Float.compare(oldversion, serviceConfiguration.getVersion()) != 0) {
149     		// Upgrade in version 4 of ServiceConfiguration. If we do not have nextRunTimeStamp and runTimeStamp set in
150     		// the database, but we have them in serviceConfiguration, we will simply copy the values over.
151     		// After this we will not use the values in ServiceConfiguration any more 
152     		final String NEXTRUNTIMESTAMP = "NEXTRUNTIMESTAMP";
153     		final String OLDRUNTIMESTAMP = "OLDRUNTIMESTAMP";
154     		if ((getNextRunTimeStamp() == 0) && (data.get(NEXTRUNTIMESTAMP) != null)) {
155     			final long nextRunTs = ((Long) data.get(NEXTRUNTIMESTAMP)).longValue();
156     			log.debug("Upgrading nextRunTimeStamp to "+nextRunTs);
157     			setNextRunTimeStamp(nextRunTs);
158     		}
159     		if ((getRunTimeStamp() == 0) && (data.get(OLDRUNTIMESTAMP) != null)) {
160     			final long runTs = ((Long) data.get(OLDRUNTIMESTAMP)).longValue();
161     			log.debug("Upgrading runTimeStamp to "+runTs);
162     			setRunTimeStamp(runTs);
163     		}
164     		setServiceConfiguration(serviceConfiguration);
165     	}
166     	return serviceConfiguration;
167     }

169     /**
170      * Method that saves the service configuration data to database.
171      * @param serviceConfiguration config
172      */
173     @SuppressWarnings("unchecked")
174     public void setServiceConfiguration(ServiceConfiguration serviceConfiguration) {
175         // We must base64 encode string for UTF safety
176         HashMap<Object, Object> a = new Base64PutHashMap();
177         a.putAll((HashMap<Object, Object>)serviceConfiguration.saveData());
178         java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream();
179         try (XMLEncoder encoder = new XMLEncoder(baos)) {
180             encoder.writeObject(a);
181         }
182         try {
183             if (log.isDebugEnabled()) {
184                 log.debug("Service data: \n" + baos.toString("UTF8"));
185             }
186             setData(baos.toString("UTF8"));
187         } catch (UnsupportedEncodingException e) {
188             throw new RuntimeException(e);
189         }
190     }

196     @Transient
197     @Override
198     protected String getProtectString(final int version) {
199         final ProtectionStringBuilder build = new ProtectionStringBuilder();
200         // rowVersion is automatically updated by JPA, so it's not important, it is only used for optimistic locking
201         build.append(getId()).append(getName()).append(getData());
202         // runTimeStamp and nextRunTimeStamp are deliberately excluded from this so that they can be updated
203         // efficiently in method updateTimestamps below. 
204         // This causes a slight security risk of denial of service, since the runtimestamp can be modified to manipulate
205         // how services run. The EJB timer itself is not stored here though, and with other monitoring that the system/CRLs etc
206         // are working it should not be seen as a great security risk. No security vital parts can be modified by altering these values.
207         return build.toString();
208     }

239 	/** @param entityManager EM
240 	 * @param id  ID
241 	 * @param oldRunTimeStamp Time 
242 	 * @param oldNextRunTimeStamp Next time
243 	 * @param newRunTimeStamp Time
244 	 * @param newNextRunTimeStamp Next time
245 	 * @return true if a service with the old timestamps existed (a.runTimeStamp=:oldRunTimeStamp AND a.nextRunTimeStamp=:oldNextRunTimeStamp) and was updated */
246 	public static boolean updateTimestamps(EntityManager entityManager, Integer id, long oldRunTimeStamp, long oldNextRunTimeStamp, long newRunTimeStamp, long newNextRunTimeStamp) {
247 		Query query = entityManager.createQuery("UPDATE ServiceData a SET a.runTimeStamp=:newRunTimeStamp, a.nextRunTimeStamp=:newNextRunTimeStamp"
248 				+ " WHERE a.id=:id AND a.runTimeStamp=:oldRunTimeStamp AND a.nextRunTimeStamp=:oldNextRunTimeStamp");
249 		query.setParameter("newRunTimeStamp", newRunTimeStamp);
250 		query.setParameter("newNextRunTimeStamp", newNextRunTimeStamp);
251 		query.setParameter("id", id);
252 		query.setParameter("oldRunTimeStamp", oldRunTimeStamp);
253 		query.setParameter("oldNextRunTimeStamp", oldNextRunTimeStamp);
254 		return query.executeUpdate() == 1;
255 	}
