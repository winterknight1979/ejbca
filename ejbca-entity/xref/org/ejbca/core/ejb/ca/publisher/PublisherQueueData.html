45  /**
46   * Entity Bean representing publisher failure data. Data is stored here when
47   * publishing to a publisher fails. Using this data publishing can be tried
48   * again. This data bean should not duplicate data completely, but holds this:
49   *
50   * - Information needed for scheduling of republishing, such as publish dates,
51   * retry counter and last failure message. - Information which is volatile on
52   * other places in the database, and we need to publish this data as it was at
53   * the time of publishing. In this case it is UserData, which can change because
54   * every user can have several certificates with different DN, the password is
55   * re-set when a certificate is issued etc. - Foreign keys to information which
56   * is not volatile. In this case this is keys to CertificateData and CRLData.
57   * For CertificateData we always want to publish the latest information, even if
58   * it changed since we failed to publish. This is so there should be no chance
59   * that a revocation is overwritten with a good status if the publish events
60   * would happen out of order.
61   *
62   * @version $Id: PublisherQueueData.java 34163 2020-01-02 15:00:17Z samuellb $
63   */

91      public PublisherQueueData(int publisherId, int publishType, String fingerprint, PublisherQueueVolatileInformation queueData, int publishStatus) {
92          String pk = GUIDGenerator.generateGUID(this);
93          setPk(pk);
94          setTimeCreated(System.currentTimeMillis());
95          setLastUpdate(0);
96          setPublishStatus(publishStatus);
97          setTryCounter(0);
98          setPublishType(publishType);
99          setFingerprint(fingerprint);
100         setPublisherId(publisherId);
101         setPublisherQueueVolatileData(queueData);
102         if (log.isDebugEnabled()) {
103             log.debug("Created Publisher queue data " + pk);
104         }
105     }

171     /**
172      * Method that returns the PublisherQueueVolatileData data and updates it if
173      * necessary.
174      *
175      * @return VolatileData is optional in publisher queue data
176      */
177     @Transient
178     public PublisherQueueVolatileInformation getPublisherQueueVolatileData() {
179         PublisherQueueVolatileInformation ret = null;
180         String vd = getVolatileData();
181         if ((vd != null) && (vd.length() > 0)) {
182             final byte[] databytes = vd.getBytes(StandardCharsets.UTF_8);
183             final HashMap<?, ?> h;
184             try (SecureXMLDecoder decoder = new SecureXMLDecoder(new java.io.ByteArrayInputStream(databytes))) {
185                 h = (HashMap<?, ?>) decoder.readObject();
186             } catch (IOException e) {
187                 final String msg = "Failed to parse PublisherQueueVolatileInformation map in database: " + e.getMessage();
188                 if (log.isDebugEnabled()) {
189                     log.debug(msg + ". Data:\n" + vd);
190                 }
191                 throw new IllegalStateException(msg, e);
192             }
193             // Handle Base64 encoded string values
194             HashMap<?, ?> data = new Base64GetHashMap(h);
195             ret = new PublisherQueueVolatileInformation();
196             ret.loadData(data);
197             if (ret.isUpgraded()) {
198                 setPublisherQueueVolatileData(ret);
199             }
200         }
201         return ret;
202     }

204     /**
205      * Method that saves the PublisherQueueData data to database.
206      *
207      * @param qd
208      *            is optional in publisher queue data
209      */
210     @SuppressWarnings("unchecked")
211     public void setPublisherQueueVolatileData(PublisherQueueVolatileInformation qd) {
212         if (qd != null) {
213             // We must base64 encode string for UTF safety
214             HashMap<Object, Object> a = new Base64PutHashMap();
215             a.putAll((HashMap<Object, Object>) qd.saveData());
216 
217             // typical size of XML is something like 250-400 chars
218             java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream(400);
219             try (XMLEncoder encoder = new XMLEncoder(baos)) {
220                 encoder.writeObject(a);
221             }
222 
223             try {
224                 if (log.isDebugEnabled()) {
225                     log.debug("PublisherQueueVolatileData: \n" + baos.toString("UTF8"));
226                 }
227                 setVolatileData(baos.toString("UTF8"));
228             } catch (UnsupportedEncodingException e) {
229                 throw new RuntimeException(e);
230             }
231         }
232     }

238     @Transient
239     @Override
240     protected String getProtectString(final int version) {
241         final ProtectionStringBuilder build = new ProtectionStringBuilder();
242         // rowVersion is automatically updated by JPA, so it's not important, it is only used for optimistic locking
243         build.append(getPk()).append(getTimeCreated()).append(getLastUpdate()).append(getPublishStatus());
244         build.append(getTryCounter()).append(getPublishType()).append(getFingerprint()).append(getPublisherId()).append(getVolatileData());
245         return build.toString();
246     }

325 	/**
326 	 * @param entityManager EM
327 	 * @param publisherId ID
328 	 * @param lowerBounds Bound
329 	 * @param upperBounds Bound
330 	 * @return the count of pending entries for a publisher in the specified intervals.
331 	 */
332 	@SuppressWarnings("unchecked")
333     public static List<Integer> findCountOfPendingEntriesForPublisher(EntityManager entityManager, int publisherId, int[] lowerBounds, int[] upperBounds) {
334 	    if (lowerBounds.length == 0) {
335 	        throw new IllegalArgumentException("lowerBounds and upperBounds are mandatory parameters");
336 	    }
337 
338 	    final StringBuilder sql = new StringBuilder();
339 	    long now = System.currentTimeMillis();
340 
341         sql.append("select c from (");
342 
343         for(int i = 0; i < lowerBounds.length; i++) {
344             sql.append("SELECT " + i + " as ordering, COUNT(*) as c FROM PublisherQueueData where publisherId=");
345             sql.append(publisherId);
346             sql.append(" AND publishStatus=");
347             sql.append(PublisherConst.STATUS_PENDING);
348             if(lowerBounds[i] > 0) {
349                 sql.append(" AND timeCreated < ");
350                 sql.append(now - 1000L * lowerBounds[i]);
351             }
352             if(upperBounds[i] > 0) {
353                 sql.append(" AND timeCreated > ");
354                 sql.append(now - 1000L * upperBounds[i]);
355             }
356             if(i < lowerBounds.length-1) {
357                 sql.append(" UNION ALL ");
358             }
359         }
360         sql.append(") tmp ORDER BY tmp.ordering");
361 
362     	if (log.isDebugEnabled()) {
363     		log.debug("findCountOfPendingEntriesForPublisher executing SQL: "+sql.toString());
364 		}
365     	final Query query = entityManager.createNativeQuery(sql.toString());
366     	List<?> resultList = query.getResultList();
367         if (log.isDebugEnabled()) {
368             log.debug("findCountOfPendingEntriesForPublisher result: "+resultList.toString());
369         }
370     	List<Integer> returnList;
371     	// Derby returns Integers, MySQL returns BigIntegers, Oracle returns BigDecimal
372     	if (resultList.size()==0) {
373     		returnList = new ArrayList<Integer>();
374     	} else if (resultList.get(0) instanceof Integer) {
375     		returnList = (List<Integer>) resultList;	// This means we can return it in it's current format
376     	} else {
377     		returnList = new ArrayList<Integer>();
378     		for (Object o : resultList) {
379     			returnList.add(ValueExtractor.extractIntValue(o));
380     		}
381     	}
382 		return returnList;
383 	}
