13  package org.ejbca.util;
14  
15  import static org.junit.Assert.assertArrayEquals;
16  
17  import java.util.ArrayList;
18  import java.util.Arrays;
19  import java.util.Collections;
20  import java.util.List;
21  
22  import javax.faces.model.SelectItem;
23  
24  import org.apache.log4j.Logger;
25  import org.junit.Test;
26  
27  /**
28   * Test for SelectItemComparator
29   * @see SelectItemComparator
30   * @version $Id: SelectItemComparatorTest.java 29652 2018-08-15 12:05:16Z anatom $
31   */
32  public class SelectItemComparatorTest {
33      
34      private static final Logger log = Logger.getLogger(SelectItemComparatorTest.class);
35  
36      @Test
37      public void testBasic() {
38          log.debug(">testBasic");
39          List<SelectItem> items = new ArrayList<>();
40          items.add(new SelectItem(10, "A"));
41          items.add(new SelectItem(5));
42          items.add(new SelectItem(7, "1"));
43          items.add(new SelectItem(8, null));
44          items.add(new SelectItem(null, "B"));
45          Collections.sort(items, new SelectItemComparator());
46          final Object[] values = valuesToArray(items);
47          log.debug("After sorting: " + Arrays.toString(values));
48          assertArrayEquals("Wrong order of items.", new Object[] { 8, 7, 5, 10, null }, values);
49          log.debug("<testBasic");
50      }
51      
52      @Test
53      public void testWithSpecialItems() {
54          log.debug(">testWithSpecialItems");
55          List<SelectItem> items = new ArrayList<>();
56          items.add(new SelectItem(10, "A"));
57          items.add(new SelectItem(5));
58          items.add(new SelectItem(7, "1"));
59          items.add(new SelectItem(8, null));
60          items.add(new SelectItem(null, "B"));
61          Collections.sort(items, new SelectItemComparator(7, null));
62          final Object[] values = valuesToArray(items);
63          log.debug("After sorting: " + Arrays.toString(values));
64          assertArrayEquals("Wrong order of items.", new Object[] { 7, null, 8, 5, 10 }, values);
65          log.debug("<testWithSpecialItems");
66      }
67  
68      private Object[] valuesToArray(final List<SelectItem> items) {
69          final Object[] values = new Object[items.size()];
70          for (int i = 0; i < items.size(); i++) {
71              values[i] = items.get(i).getValue();
72          }
73          return values;
74      }

77      
78  }
