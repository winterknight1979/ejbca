63      /**
64       * Initializes the dynamic UI properties on a grid panel with two columns, label on the left, UI component on the right,
65       * and an optional help text below the UI component.
66       * @param panelGrid the panel grid instance to (re-)build.
67       * @param model DynamicUiModel to extract properties from
68       * @param i18nPrefix the name prefix for the I18N message resources.
69       * @throws DynamicUiModelException if the PSM could not be created by the dynamic UI model.
70       */
71      public static final void initGridInstance(final HtmlPanelGrid panelGrid, final DynamicUiModel model,
72              final String i18nPrefix) throws DynamicUiModelException {
73          if (log.isDebugEnabled()) {
74              log.debug("Build dynamic UI model PSM " + model + ", i18nPrefix is '" + i18nPrefix + "'.");
75          }
76          panelGrid.clearInitialState();
77          panelGrid.getFacets().clear();
78          panelGrid.getChildren().clear();
79          panelGrid.setColumns(2);
80          // Build PSM fields by PIM.
81          int index = 0;
82          for (final DynamicUiProperty<? extends Serializable> property : model.getProperties().values()) {
83              final HtmlOutputLabel label = new HtmlOutputLabel();
84              label.setValue(getText(i18nPrefix, property.getName()));
85              label.setStyleClass(STYLE_CLASS_SUB_ITEM);
86              if (index == 0) { // Re-factor: Set header bold.
87                  label.setStyle("font-weight: bold;");
88              }
89              panelGrid.getChildren().add(label);
90  
91              if (!property.isLabelOnly()) {
92                  final UIComponentBase component = createComponentInstance(i18nPrefix, property);
93                  final String helpText = getHelpText(i18nPrefix, property.getName());
94                  if (helpText != null) {
95                      final HtmlPanelGrid innerGrid = new HtmlPanelGrid();
96                      innerGrid.setColumns(1);
97                      final HtmlOutputLabel helpLabel = new HtmlOutputLabel();
98                      helpLabel.setValue(helpText);
99                      innerGrid.getChildren().add(component);
100                     innerGrid.getChildren().add(helpLabel);
101                     panelGrid.getChildren().add(innerGrid);
102                     if (log.isDebugEnabled()) {
103                         log.debug("Registered UIComponent with property name " + property.getName() + " + help text.");
104                     }
105                 } else {
106                     panelGrid.getChildren().add(component);
107                     if (log.isDebugEnabled()) {
108                         log.debug("Registered UIComponent with property name " + property.getName() + ".");
109                     }
110                 }
111             } else {
112                 label.setStyle("font-weight: bold;");
113                 label.setStyleClass(StringUtils.EMPTY);
114                 panelGrid.getChildren().add(new HtmlPanelGroup());
115             }
116             index++;
117         }
118     }

120     /**
121      * Create a concrete UIComponentBase instance based on the dynamic UI property.
122      *
123      * @param i18nPrefix the message resources prefix.
124      * @param property the dynamic UI property ({@link DynamicUiProperty}).
125      * @return the JSF component.
126      * @throws DynamicUiModelException if a component instance could not be created (i.e. does not exist).
127      */
128     public static final UIComponentBase createComponentInstance(final String i18nPrefix, final DynamicUiProperty<?> property)
129             throws DynamicUiModelException {
130         final String hint = property.getRenderingHint();
131         UIComponentBase component = null;
132         if (DynamicUiProperty.RENDER_NONE.equals(hint)) { // Insert dummy component.
133             component = new HtmlPanelGroup();
134         }
135         else if (property.isBooleanType()) {
136             if (DynamicUiProperty.RENDER_CHECKBOX.equals(hint)) {
137                 component = createCheckBoxInstance(property);
138             }
139         } else if (property.isStringType()) {
140             if (DynamicUiProperty.RENDER_TEXTFIELD.equals(hint)) {
141                 component = createTextFieldInstance(property);
142             } else if (DynamicUiProperty.RENDER_TEXTAREA.equals(hint)) {
143                 component = createTextAreaInstance(property);
144             } else if (DynamicUiProperty.RENDER_SELECT_ONE.equals(hint)) {
145                 component = createDropDownBoxInstance(property);
146             } else if (DynamicUiProperty.RENDER_SELECT_MANY.equals(hint)) {
147                 component = createListBoxInstance(property);
148             } else if (DynamicUiProperty.RENDER_BUTTON.equals(hint)) {
149                 component = createButtonInstance(i18nPrefix, property);
150             } else if (DynamicUiProperty.RENDER_LABEL.equals(hint)) {
151                 component = createLabelInstance(property);
152             }
153         } else if (property.isIntegerType()) {
154             if (DynamicUiProperty.RENDER_TEXTFIELD.equals(hint)) {
155                 component = createIntegerTextFieldInstance(property);
156             } else if (DynamicUiProperty.RENDER_SELECT_ONE.equals(hint)) {
157                 component = createIntegerDropDownBoxInstance(property);
158             }
159         } else if (property.isBigIntegerType()) {
160             if (DynamicUiProperty.RENDER_TEXTFIELD.equals(hint)) {
161                 component = createBigIntegerTextFieldInstance(property);
162             }
163         } else if (property.isFloatType()) {
164             if (DynamicUiProperty.RENDER_TEXTFIELD.equals(hint)) {
165                 component = createFloatTextFieldInstance(property);
166             }
167         } else if (property.isFileType()) {
168             if (DynamicUiProperty.RENDER_TEXTFIELD.equals(hint)) {
169                 component = createTextFieldInstance(property);
170             } else if (DynamicUiProperty.RENDER_FILE_CHOOSER.equals(hint)) {
171                 component = createFileChooserInstance(property);
172             }
173         } else if (property.isByteArrayType()) {
174             if (DynamicUiProperty.RENDER_FILE_CHOOSER.equals(hint)) {
175                 component = createFileChooserInstance(property);
176             }
177         }
178         if (component == null) {
179             throw new DynamicUiModelException("DynamicUiRendering component could not be found: " + property);
180         }
181         return component;
182     }

285      * Creates drop down box component by the given dynamic UI property.
286      * @param property the dynamic UI property.
287      * @return the drop down box instance.
288      */
289     public static final HtmlSelectOneMenu createDropDownBoxInstance(final DynamicUiProperty<?> property) {
290         final JsfDynamicUiHtmlSelectOneMenunu.html#JsfDynamicUiHtmlSelectOneMenu">JsfDynamicUiHtmlSelectOneMenu result = new JsfDynamicUiHtmlSelectOneMenu();
291         result.setDynamicUiProperty(property);
292         setUIInputAttributes(result, property);
293         result.setDisabled(property.getDynamicUiModel().isDisabled() || property.isDisabled());
294         final List<SelectItem> items = new ArrayList<>();
295         final Map<?, String> labels = property.getLabels();
296         for (Entry<?, String> entry : labels.entrySet()) {
297             items.add(new SelectItem(entry.getKey(),
298                     property.isI18NLabeled() ? EjbcaJSFHelper.getBean().getEjbcaWebBean().getText(entry.getValue()) : entry.getValue()));
299         }
300         final UISelectItems selectItems = new UISelectItems();
301         selectItems.setValue(items);
302         addAjaxListener(property, result, "change");
303         result.getChildren().add(selectItems);
304         return result;
305     }

394     /**
395      * Sets the common properties for a component.
396      * @param component the component.
397      * @param property the dynamic property.
398      */
399     private static final void setUIOutputAttributes(final UIOutput component, final DynamicUiProperty<? extends Serializable> property) {
400         setBaseProperties(component, property);
401         if (!property.getHasMultipleValues()) {
402             component.setValue(property.getValue());
403         } else {
404             component.setValue(property.getValues());
405         }
406     }

429     /**
430      * Gets an I18N help message.
431      * @param i18nPrefix the name prefix.
432      * @param name the name.
433      * @return the I18N help message if present, null otherwise
434      */
435     private static final String getHelpText(final String i18nPrefix, final String name) {
436         final String template = i18nPrefix.concat(name).toUpperCase().concat("HELP");
437         final String text = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText(template);
438         if (template.equals(text)) {
439             return null;
440         }
441         return text;
442     }

444     /**
445      * Adds an ajax behavior listener to the component if the dynamic UI properties action callback is not null.
446      * @param property the dynamic UI property.
447      * @param component the JSF UIInput component.
448      * @param eventName name
449      */
450     private static final void addAjaxListener(final DynamicUiProperty<? extends Serializable> property, final UIInput component, final String eventName) {
451         if (property.getActionCallback() != null) {
452             if (log.isDebugEnabled()) {
453                 log.debug("Registered dynamic UI model action callback for component " + property.getName() + ".");
454             }
455             final AjaxBehavior behavior = (AjaxBehavior) FacesContext.getCurrentInstance().getApplication().createBehavior(AjaxBehavior.BEHAVIOR_ID);
456             behavior.addAjaxBehaviorListener(new JsfDynamicUiAjaxBehaviorListener(property, component));
457             final List<String> render = new ArrayList<>();
458             if (CollectionUtils.isNotEmpty(property.getActionCallback().getRender())) {
459                 render.addAll(property.getActionCallback().getRender());
460             } else {
461                 render.add("@all");
462             }
463             behavior.setRender(render);
464             behavior.setTransient(false);
465             behavior.setImmediate(true);
466             component.addClientBehavior(eventName, behavior);
467         }
