149     /**Ë›
150      * Sets the selected validator type. This re-creates the entire validator, so only do it if it actually changed type.
151      * @param type the type {@link ValidatorBase#getValidatorTypeIdentifier()}.
152      */
153     public void setValidatorType(final String type) {
154         final String oldType = validator.getValidatorTypeIdentifier();
155         if (!oldType.equals(type)) {
156             if (log.isDebugEnabled()) {
157                 log.debug("Change key validator type from " + oldType + " to " + type);
158             }
159             setValidator(ValidatorFactory.INSTANCE.getArcheType(type));
160             getValidator().setDataMap(getValidator().getDataMap());
161             getValidator().setProfileId(validatorsBean.getValidatorId());
162             getValidator().setProfileName(validatorsBean.getValidatorName());
163         }
164     }

166     /**
167      * Gets the selected validator.
168      * @return the  validator.
169      */
170     public Validator getValidator() {
171         // @ViewScoped: If the back link was called it may happen that the same view is rendered again with another validator.
172         final int newId = ( validatorsBean.getValidatorId() != null ? validatorsBean.getValidatorId() : -1);
173         final int oldId = getValidatorId();
174         if (validator != null && oldId != newId) {
175             reset();
176         }
177         if (validator == null && newId != -1) {
178             if (log.isDebugEnabled()) {
179                 log.debug("Request validator with id " + newId);
180             }
181             setValidatorId(newId);         
182             setValidator(keyValidatorSession.getValidator(newId));
183         }
184         // (Re-)initialize dynamic UI PSM.
185         if (validator instanceof DynamicUiModelAware) {
186             if (uiModel == null || !uiModel.equals(((DynamicUiModelAware) validator).getDynamicUiModel())) {
187                 ((DynamicUiModelAware) validator).initDynamicUiModel();
188                 uiModel = ((DynamicUiModelAware) validator).getDynamicUiModel();
189                 if (log.isDebugEnabled()) {
190                     log.debug("Request dynamic UI properties for validator with (id=" + validator.getProfileId() + ") with properties " + validator.getDataMap());
191                 }
192                 try {
193                     initGrid(uiModel, validator.getClass().getSimpleName());
194                 } catch (DynamicUiModelException e) {
195                     log.warn("Could not initialize dynamic UI PSM: " + e.getMessage(), e);
196                 }
197             }
198         }
199         return validator;
200      }

266     /**
267      * Gets the available key validator types.
268      *
269      * @return List of the available key validator types
270      */
271     public List<SelectItem> getAvailableValidators() {
272         final List<Class<?>> excludeClasses = new ArrayList<>();
273         if (!((GlobalConfiguration) configurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID)).getEnableExternalScripts()) {
274             excludeClasses.add(ExternalCommandCertificateValidator.class);
275         }
276         final List<SelectItem> ret = new ArrayList<>();
277         for (final Validator validator : ValidatorFactory.INSTANCE.getAllImplementations(excludeClasses)) {
278             ret.add(new SelectItem(validator.getValidatorTypeIdentifier(), validator.getLabel()));
279         }
280         Collections.sort(ret, new Comparator<SelectItem>() {
281             @Override
282             public int compare(SelectItem o1, SelectItem o2) {
283                 return o1.getLabel().compareToIgnoreCase(o2.getLabel());
284             }
285         });
286         return ret;
287     }

328      * Validates the description field, see {@link ValidatorBase#getDescription()}.
329      * @param context the faces context.
330      * @param component the events source component
331      * @param value the source components value attribute
332      * @throws ValidatorException if the validation fails.
333      */
334     public void validateDescription(FacesContext context, UIComponent component, Object value) throws ValidatorException {
335         final String descripion = (String) value;
336         if (StringUtils.isNotBlank(descripion) && descripion.trim().length() > 256) {
337             final String message = "Description must not contain more than 256 characters.";
338             if (log.isDebugEnabled()) {
339                 log.debug(message);
340             }
341             throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message));
342         }
343     }

346      * Validates the BaseKeyValildator notBefore field, see {@link org.cesecore.keys.validation.ValidityAwareValidator#getNotBefore()}.
347      * @param context the faces context.
348      * @param component the events source component
349      * @param value the source components value attribute
350      * @throws ValidatorException if the validation fails.
351      */
352     public void validateNotBefore(FacesContext context, UIComponent component, Object value) throws ValidatorException {
353         final String string = (String) value;
354         try {
355             if (StringUtils.isNotBlank(string) && null == KeyValidatorBase.parseDate(string)) {
356                 final String message = "Key validator not before must be a valid ISO 8601 date or time " + value;
357                 throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message));
358             }
359         } catch (ParseException e) {
360             log.debug("Could not parse Date: " + string);
361         }
362     }

429      * Validates the BaseKeyValildator certificateProfileIds field, see {@link ValidatorBase#getCertificateProfileIds()}.
430      * @param context the faces context.
431      * @param component the events source component
432      * @param value the source components value attribute
433      * @throws ValidatorException if the validation fails.

435     public void validateCertificateProfileIds(FacesContext context, UIComponent component, Object value) throws ValidatorException {
436         @SuppressWarnings("unchecked")
437         final List<String> selectedIds = (List<String>) value;
438         final List<Integer> ids = new ArrayList<>(certificateProfileSession.getCertificateProfileIdToNameMap().keySet());
439         for (String id : selectedIds) {
440             if (!ids.contains(Integer.parseInt(id))) {
441                 final String message = "Key validator certificate profile id must be on of " + ids;
442                 throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message));
443             }
444         }

456     /**
457      * Save action.
458      * @return the navigation outcome defined in faces-config.xml.
459      */
460     public String save() {
461         final Validator validator = getValidator();
462         if (log.isDebugEnabled()) {
463             log.debug("Try to save validator: " + validator);
464         }
465         try {
466             if (validator instanceof DynamicUiModelAware) {
467                 ((DynamicUiModelAware) validator).getDynamicUiModel().writeProperties(((ValidatorBase) validator).getRawData());
468             }
469             keyValidatorSession.changeKeyValidator(getAdmin(), validator);
470             addInfoMessage("VALIDATORSAVED");
471             reset();
472             return "done";
473         } catch (AuthorizationDeniedException e) {
474             addNonTranslatedErrorMessage("Not authorized to edit validator " + validator.getProfileName());
475         } catch (KeyValidatorDoesntExistsException e) {
476             // NOPMD: ignore do nothing
477         }
478         return StringUtils.EMPTY;
479     }

481     /**
482      * Gets a list of select items of the available certificate profiles.
483      * @return the list.
484      */
485     public List<SelectItem> getAvailableCertificateProfiles() {
486         final List<SelectItem> result = new ArrayList<>();
487         List<Integer> authorizedCertificateProfiles = certificateProfileSession.getAuthorizedCertificateProfileIds(getAdmin(), CertificateConstants.CERTTYPE_UNKNOWN);
488         final Map<Integer, String> map = certificateProfileSession.getCertificateProfileIdToNameMap();
489         for(Integer certificateProfileId : authorizedCertificateProfiles) {
490             // Don't include fixed certificate profiles in validators, keep it clean and force usage of "real"
491             // profiles if you want to issue serious certificates.
492             if (certificateProfileId > CertificateProfileConstants.FIXED_CERTIFICATEPROFILE_BOUNDRY) {
493                 result.add(new SelectItem(certificateProfileId, map.get(certificateProfileId)));
494             }
495         }
496         Collections.sort(result, new Comparator<SelectItem>() {
497             @Override
498             public int compare(SelectItem o1, SelectItem o2) {
499                 return o1.getLabel().compareToIgnoreCase(o2.getLabel());
500             }
501         });
502         return result;
503     }
