90       public void authorizedToApprovalPages() throws AuthorizationDeniedException{
91  		  // Check Authorization
92          boolean approveendentity = getEjbcaWebBean().isAuthorizedNoLogSilent(AccessRulesConstants.REGULAR_APPROVEENDENTITY);
93          boolean approvecaaction = getEjbcaWebBean().isAuthorizedNoLogSilent(AccessRulesConstants.REGULAR_APPROVECAACTION);
94   		if (!approveendentity && !approvecaaction) {
95   			throw new AuthorizationDeniedException("Not authorized to view approval pages");
96   		}
97       }

103      public EjbcaWebBean getEjbcaWebBean(){
104          FacesContext ctx = FacesContext.getCurrentInstance();    		    	
105          HttpSession session = (HttpSession) ctx.getExternalContext().getSession(true);
106          synchronized (session) {
107              ejbcawebbean = (org.ejbca.ui.web.admin.configuration.EjbcaWebBean) session.getAttribute("ejbcawebbean");
108              if (ejbcawebbean == null){
109                  ejbcawebbean = new org.ejbca.ui.web.admin.configuration.EjbcaWebBean();
110                  try {
111                      ejbcawebbean.initialize((HttpServletRequest) ctx.getExternalContext().getRequest(), AccessRulesConstants.ROLE_ADMINISTRATOR);
112                      session.setAttribute("ejbcawebbean", ejbcawebbean);
113                  } catch (Exception e) {
114                      log.error(e);
115                  }
116              }
117          }
118          return ejbcawebbean;
119      }

132      /** @return true if the client browser has identified itself as a legacy Internet Explorer 10 (or earlier) */
133      public boolean isLegacyInternetExplorer() {
134          if (legacyInternetExplorer==null) {
135              final HttpServletRequest httpServletRequest = (HttpServletRequest) FacesContext.getCurrentInstance().getExternalContext().getRequest();
136              final String userAgent = httpServletRequest.getHeader("User-Agent");
137              if (log.isDebugEnabled()) {
138                  log.debug("User-Agent: " + userAgent);
139              }
140              // Check stolen from org.ejbca.ui.web.pub.ApplyBean.detectBrowser(HttpServletRequest)
141              // "Gecko"==Firefox, "MSIE"==Internet Exploder 10-, "Trident"==IE11
142              legacyInternetExplorer = Boolean.valueOf(userAgent != null && userAgent.contains("MSIE") && !userAgent.contains("Gecko"));
143          }
144          return legacyInternetExplorer.booleanValue();
145      }
