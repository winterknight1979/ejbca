78      @SuppressWarnings("deprecation")
79  	public void initialize(HttpServletRequest request, EjbcaWebBean ejbcawebbean) throws Exception {
80          if (!initialized) {
81              admin = ejbcawebbean.getAdminObject();
82              EjbLocalHelper ejbLocalHelper = new EjbLocalHelper();
83              hardtokensession = ejbLocalHelper.getHardTokenSession();
84              hardtokenbatchsession = ejbLocalHelper.getHardTokenBatchJobSession();
85              AuthorizationSessionLocal authorizationSession = ejbLocalHelper.getAuthorizationSession();
86              EndEntityManagementSessionLocal endEntityManagementSession = ejbLocalHelper.getEndEntityManagementSession();
87              CertificateProfileSession certificateProfileSession = ejbLocalHelper.getCertificateProfileSession();
88              keyrecoverysession = ejbLocalHelper.getKeyRecoverySession();
89              CaSessionLocal caSession = ejbLocalHelper.getCaSession();
90              roleSession = new EjbLocalHelper().getRoleSession();
91              initialized = true;
92              this.hardtokenprofiledatahandler = new HardTokenProfileDataHandler(admin, hardtokensession, certificateProfileSession,
93                      authorizationSession, endEntityManagementSession, caSession);
94          }
95      }

97      /** Returns the first found hard token for the given username. 
98       * @param username User
99       * @param includePUK PUK
100      * @return View */
101     public HardTokenView getHardTokenViewWithUsername(String username, boolean includePUK) {
102         this.result = null;
103         Collection<HardTokenInformation> res = hardtokensession.getHardTokens(admin, username, includePUK);
104         Iterator<HardTokenInformation> iter = res.iterator();
105         if (res.size() > 0) {
106             this.result = new HardTokenView[res.size()];
107             for (int i = 0; iter.hasNext(); i++) {
108                 this.result[i] = new HardTokenView(iter.next());
109             }
110             if (this.result != null && this.result.length > 0) {
111                 return this.result[0];
112             }
113         }
114         return null;
115     }
116 
117     public HardTokenView getHardTokenViewWithIndex(String username, int index, boolean includePUK) {
118         HardTokenView returnval = null;
119         if (result == null) {
120             getHardTokenViewWithUsername(username, includePUK);
121         }
122         if (result != null) {
123             if (index < result.length) {
124                 returnval = result[index];
125             }
126         }
127         return returnval;
128     }

187     public void addHardTokenIssuer(String alias, int roleId) throws HardTokenIssuerExistsException, AuthorizationDeniedException {
188         for (final Role role : getHardTokenIssuingRoles()) {
189             if (role.getRoleId()==roleId) {
190                 if (!hardtokensession.addHardTokenIssuer(admin, alias, roleId, new HardTokenIssuer())) {
191                     throw new HardTokenIssuerExistsException();
192                 }
193             }
194         }
195     }

197     public void changeHardTokenIssuer(String alias, HardTokenIssuer hardtokenissuer) throws HardTokenIssuerDoesntExistsException,
198             AuthorizationDeniedException {
199         if (hardtokensession.isAuthorizedToEditHardTokenIssuer(admin, alias)) {
200             if (!hardtokensession.changeHardTokenIssuer(admin, alias, hardtokenissuer)) {
201                 throw new HardTokenIssuerDoesntExistsException();
202             }
203         }
204     }

206     /** Returns false if profile is used by any user or in authorization rules. 
207      * @param alias Alias
208      * @return Issuer
209      * @throws AuthorizationDeniedException Fail */
210     public boolean removeHardTokenIssuer(String alias) throws AuthorizationDeniedException {
211         boolean issuerused = false;
212         if (hardtokensession.isAuthorizedToEditHardTokenIssuer(admin, alias)) {
213             int issuerid = hardtokensession.getHardTokenIssuerId(alias);
214             // Check if any users or authorization rule use the profile.
215             issuerused = hardtokenbatchsession.checkForHardTokenIssuerId(issuerid);
216             if (!issuerused) {
217                 hardtokensession.removeHardTokenIssuer(admin, alias);
218             }
219         }
220         return !issuerused;
221     }

241     /**
242      * Method that checks if a token is key recoverable and also check if the administrator is authorized to the action.
243      * @param tokensn SN
244      * @param username Iser
245      * @param rabean Bean
246      * @return bool
247      * @throws Exception fail
248      */
249     public boolean isTokenKeyRecoverable(String tokensn, String username, RAInterfaceBean rabean) throws Exception {
250         boolean retval = false;
251         X509Certificate keyRecCert = null;
252         for (final Certificate cert : hardtokensession.findCertificatesInHardToken(tokensn)) {
253             final X509Certificate x509cert = (X509Certificate) cert;
254             if (keyrecoverysession.existsKeys(EJBTools.wrap(x509cert))) {
255                 keyRecCert = x509cert;
256             }
257         }
258         if (keyRecCert != null) {
259             retval = rabean.keyRecoveryPossible(keyRecCert, username);
260         }
261         return retval;
262     }
263 
264     public void markTokenForKeyRecovery(String tokensn, String username, RAInterfaceBean rabean) throws Exception {
265         for (final Certificate cert : hardtokensession.findCertificatesInHardToken(tokensn)) {
266             final X509Certificate x509cert = (X509Certificate) cert;
267             if (keyrecoverysession.existsKeys(EJBTools.wrap(x509cert))) {
268                 rabean.markForRecovery(username, x509cert);
269             }
270         }
271     }
