70         /** Method to add a hard token profile. 
71       * @param name NAme
72       * @param profile Peofile
73          * 
74          * @return false, if the profile have a bad XML encoding.
75          * @throws HardTokenProfileExistsException if profile already exists  
76       * @throws AuthorizationDeniedException fail*/
77      public boolean addHardTokenProfile(String name, HardTokenProfile profile) throws HardTokenProfileExistsException, AuthorizationDeniedException {
78        boolean success = false;
79        if(authorizedToProfile(profile, true)){
80      	if(checkXMLEncoding(profile)){
81            hardtokensession.addHardTokenProfile(administrator, name, profile);
82            success=true;
83      	}  
84           
85        }else {
86          throw new AuthorizationDeniedException("Not authorized to add hard token profile");
87        }
88        return success;
89      }

113     /** Method to remove a hard token profile, returns true if deletion failed.
114      * @param name Name
115      * @return Ptofile
116      * @throws AuthorizationDeniedException fail */ 
117     public boolean removeHardTokenProfile(String name) throws AuthorizationDeniedException{
118       boolean returnval = true;  
119       
120 	  int profileid = getHardTokenProfileId(name);
121 	  
122       if(endEntityManagementSession.checkForHardTokenProfileId(profileid)) {
123         return true;
124       }
125 	  if(hardtokensession.existsHardTokenProfileInHardTokenIssuer(profileid)) {
126 		return true;  
127 	  }
128       if(authorizedToProfileName(name, true)){    
129 		hardtokensession.removeHardTokenProfile(administrator, name);
130 		returnval = false;
131       }else {
132         throw new AuthorizationDeniedException("Not authorized to remove hard token profile");
133       }
134       return returnval;          
135     }

208     /**
209      * Help function that checks if administrator is authorized to edit profile.
210      * @param profile Profile
211      * @param editcheck Check
212      * @return Bool
213      */    
214     private boolean authorizedToProfile(HardTokenProfile profile, boolean editcheck) {
215         boolean returnval = false;
216         if (authorizationSession.isAuthorizedNoLogging(administrator, StandardRules.ROLE_ROOT.resource())) {
217             returnval = true; // yes authorized to everything
218         } else {
219             if (editcheck && authorizationSession.isAuthorizedNoLogging(administrator, AccessRulesConstants.HARDTOKEN_EDITHARDTOKENPROFILES)) {      
220                 HashSet<Integer> authorizedcaids = new HashSet<>(caSession.getAuthorizedCaIds(administrator));
221                 HashSet<Integer> authorizedcertprofiles =
222                         new HashSet<>(certificateProfileSession.getAuthorizedCertificateProfileIds(administrator, CertificateConstants.CERTTYPE_HARDTOKEN));
223                 // It should be possible to indicate that a certificate should not be generated by not specifying a cert profile for this key. 
224                 authorizedcertprofiles.add(Integer.valueOf(CertificateProfileConstants.CERTPROFILE_NO_PROFILE));
225                 if (profile instanceof EIDProfile) {
226                     if (authorizedcertprofiles.containsAll(((EIDProfile) profile).getAllCertificateProfileIds())
227                             && authorizedcaids.containsAll(((EIDProfile) profile).getAllCAIds())) {
228                         returnval = true;
229                     }
230                 } else {
231                     // Implement for other profile types
232                 }
233 
234             }
235 
236         }
237         return returnval;
238     }

240     /**
241      * Method that test to XML encode and decode a profile.
242      * @param profile Profile
243      * @return false if something went wrong in the encoding process.
244      */
245     @SuppressWarnings({ "unchecked", "rawtypes" })
246     private boolean checkXMLEncoding(HardTokenProfile profile) {
247         boolean success = false;
248         try{
249             
250             java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream();
251             
252             // We must base64 encode string for UTF safety
253             HashMap<?, ?> a = new Base64PutHashMap();
254             a.putAll((HashMap)profile.saveData());
255             java.beans.XMLEncoder encoder = new java.beans.XMLEncoder(baos);
256             encoder.writeObject(a);
257             encoder.close();
258             String data = baos.toString("UTF8");
259             java.beans.XMLDecoder decoder = new java.beans.XMLDecoder(
260                         new java.io.ByteArrayInputStream(data.getBytes("UTF8")));
261             decoder.readObject();
262             decoder.close();
263             
264             success = true;
265         } catch (Exception e) {
266             success = false;  
267         }
268 
269 		return success;
270 	}
