56  /**
57   * Servlet used to export certificate profiles and end entity profiles in a downloadable zip file.<br>
58   *
59   * The servlet is called with method GET or POST and syntax
60   * <code>profileType=&lt;type&gt;</code>.
61   * <p>The following types are supported:<br>
62   * <ul>
63   * <li>cp - Certificate Profiles</li>
64   * <li>eep - End Entity Profiles </li>
65   * </ul>
66   *
67   * @version $Id: ProfilesExportServlet.java 34154 2019-12-23 13:38:17Z samuellb $
68   */
69  public class ProfilesExportServlet extends BaseAdminServlet {
70  
71      private static final long serialVersionUID = -8091852234056712787L;
72      private static final Logger log = Logger.getLogger(ProfilesExportServlet.class);
73  
74      @EJB
75      private AuthorizationSessionLocal authorizationSession;
76      @EJB
77      private CertificateProfileSessionLocal certificateProfileSession;
78      @EJB
79      private EndEntityProfileSessionLocal endEntityProfileSession;
80      @EJB
81      private KeyValidatorSessionLocal keyValidatorSession;
82      @EJB
83      private GlobalConfigurationSessionLocal globalConfigurationSession;
84      @EJB
85      private WebAuthenticationProviderSessionLocal authenticationSession;
86  
87      @Override
88      public void init(ServletConfig config) throws ServletException {
89          super.init(config);
90      }
91  
92      @Override
93      public void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
94          log.trace(">doPost()");
95          doGet(request, response);
96          log.trace("<doPost()");
97      }
98  
99      @Override
100     public void doGet(HttpServletRequest request,  HttpServletResponse response) throws IOException, ServletException {
101         log.trace(">doGet()");
102         final AuthenticationToken admin;
103         try {
104             admin = authenticateAdmin(request, response, AccessRulesConstants.ROLE_ADMINISTRATOR);
105         } catch (AdminWebAuthenticationException authExc) {
106         	// TODO: localize this.
107         	log.info("Authentication failed", authExc);
108             response.sendError(HttpServletResponse.SC_FORBIDDEN, "Authentication failed");
109             return;
110         }
111 
112         final String type = request.getParameter("profileType");
113         final boolean exportCertificateProfiles = StringUtils.equalsIgnoreCase(type, "cp");
114         final boolean exportEndEntityProfiles = StringUtils.equalsIgnoreCase(type, "eep");
115 
116         String zipfilename = null;
117         int exportedprofiles = 0;
118         int totalprofiles = 0;
119         int missingprofiles = 0;
120         ByteArrayOutputStream zbaos = new ByteArrayOutputStream();
121         ZipOutputStream zos = new ZipOutputStream(zbaos);
122 
123         if (exportCertificateProfiles) {
124             zipfilename = "certprofiles.zip";
125 
126             final List<Integer> certificateProfileTypes = new ArrayList<>();
127             certificateProfileTypes.add(CertificateConstants.CERTTYPE_ENDENTITY);
128             if (authorizationSession.isAuthorizedNoLogging(admin, StandardRules.ROLE_ROOT.resource())) {
129                 //Only root users may use CA profiles
130                 certificateProfileTypes.add(CertificateConstants.CERTTYPE_ROOTCA);
131                 certificateProfileTypes.add(CertificateConstants.CERTTYPE_SUBCA);
132             }
133             GlobalConfiguration globaConfiguration = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
134             if (globaConfiguration.getIssueHardwareTokens()) {
135                 certificateProfileTypes.add(CertificateConstants.CERTTYPE_HARDTOKEN);
136             }
137 
138             Collection<Integer> certprofids = new LinkedHashSet<>();
139             for(Integer certificateProfileType : certificateProfileTypes) {
140                 certprofids.addAll(certificateProfileSession.getAuthorizedCertificateProfileIds(admin, certificateProfileType));
141             }
142 
143             totalprofiles = certprofids.size();
144             log.info("Exporting non-fixed certificate profiles");
145             for (int profileid : certprofids) {
146                 if (profileid == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) { // Certificate profile not found i database.
147                     log.error("Couldn't find certificate profile '" + profileid + "' in database.");
148                 } else if (CertificateProfileConstants.isFixedCertificateProfile(profileid)) {
149                     if (log.isDebugEnabled()) {
150                         log.debug("Skipping export fixed certificate profile with id '" + profileid + "'.");
151                     }
152                 } else {
153                     String profilename = certificateProfileSession.getCertificateProfileName(profileid);
154                     CertificateProfile profile = certificateProfileSession.getCertificateProfile(profileid);
155                     if (profile == null) {
156                         missingprofiles++;
157                         log.error("Couldn't find certificate profile '" + profilename + "'-" + profileid + " in database.");
158                     } else {
159                         String profilenameEncoded;
160                         try {
161                             profilenameEncoded = URLEncoder.encode(profilename, "UTF-8");
162                         } catch (UnsupportedEncodingException e) {
163                             throw new IllegalStateException("UTF-8 was not a known encoding", e);
164                         }
165 
166                         byte[] ba = getProfileBytes(profile);
167                         String filename = "certprofile_" + profilenameEncoded + "-" + profileid + ".xml";
168                         ZipEntry ze = new ZipEntry(filename);
169                         zos.putNextEntry(ze);
170                         zos.write(ba);
171                         zos.closeEntry();
172                         exportedprofiles++;
173                     }
174                 }
175             }
176 
177         } else if (exportEndEntityProfiles) {
178 
179             zipfilename = "entityprofiles.zip";
180 
181             Collection<Integer> endentityprofids = endEntityProfileSession.getAuthorizedEndEntityProfileIds(admin, AccessRulesConstants.VIEW_END_ENTITY);
182             totalprofiles = endentityprofids.size();
183             log.info("Exporting non-fixed end entity profiles");
184             for (int profileid : endentityprofids) {
185                 if (profileid == EndEntityConstants.NO_END_ENTITY_PROFILE) { // Entity profile not found i database.
186                     missingprofiles++;
187                     log.error("Error : Couldn't find entity profile '" + profileid + "' in database.");
188                 } else if (profileid == EndEntityConstants.EMPTY_END_ENTITY_PROFILE) {
189                     if(log.isDebugEnabled()) {
190                         log.debug("Skipping export fixed end entity profile with id '"+profileid+"'.");
191                     }
192                 } else {
193                     String profilename = endEntityProfileSession.getEndEntityProfileName(profileid);
194                     EndEntityProfile profile = endEntityProfileSession.getEndEntityProfile(profileid);
195                     if (profile == null) {
196                         log.error("Error : Couldn't find entity profile '" + profilename + "'-" + profileid + " in database.");
197                     } else {
198                         String profilenameEncoded;
199                         try {
200                             profilenameEncoded = URLEncoder.encode(profilename, "UTF-8");
201                         } catch (UnsupportedEncodingException e) {
202                             throw new IllegalStateException("UTF-8 was not a known encoding", e);
203                         }
204 
205                         byte[] ba = getProfileBytes(profile);
206                         String filename = "entityprofile_" + profilenameEncoded + "-" + profileid + ".xml";
207                         ZipEntry ze = new ZipEntry(filename);
208                         zos.putNextEntry(ze);
209                         zos.write(ba);
210                         zos.closeEntry();
211 
212                         exportedprofiles++;
213                     }
214                 }
215             }
216         }
217         zos.close();
218 
219         final byte[] zipfile = zbaos.toByteArray();
220         zbaos.close();
221 
222         log.info("Found " + totalprofiles + " profiles. " + exportedprofiles + " profiles were exported to " + zipfilename + " and " + missingprofiles
223                 + " were not found in the database.");
224 
225         response.setContentType("application/octet-stream");
226         response.setHeader("Content-disposition", " attachment; filename=\"" + StringTools.stripFilename(zipfilename) + "\"");
227         response.getOutputStream().write(zipfile);
228         response.flushBuffer();
229 
230         log.trace("<doGet()");
231     } // doGet
232 
233     private byte[] getProfileBytes(UpgradeableDataHashMap profile) {
234         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
235         try (XMLEncoder encoder = new XMLEncoder(baos)) {
236             encoder.writeObject(profile.saveData());
237             encoder.close();
238             return baos.toByteArray();
239         }
240     }
