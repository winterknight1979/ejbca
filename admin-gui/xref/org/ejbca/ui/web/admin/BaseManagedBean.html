38  /**
39   * Base EJBCA JSF Managed Bean, all managed beans of EJBCA should inherit this class
40   *
41   * @version $Id: BaseManagedBean.java 29589 2018-08-08 12:02:53Z bastianf $
42   */
43  public abstract class BaseManagedBean implements Serializable {
44  
45      private static final long serialVersionUID = -8019234011853194880L;
46      private static final Logger log = Logger.getLogger(BaseManagedBean.class);
47  
48      private static final Map<String, Map<String, Object>> publicConstantCache = new ConcurrentHashMap<>();
49  
50  	protected EjbcaWebBean getEjbcaWebBean(){
51  		return EjbcaJSFHelper.getBean().getEjbcaWebBean();
52  	}
53  
54  	/** @param resources Resources
55  	 * @return true if the current admin is authorized to the resources or false otherwise */
56      protected boolean isAuthorizedTo(final String...resources) {
57          return getEjbcaWebBean().getEjb().getAuthorizationSession().isAuthorizedNoLogging(getAdmin(), resources);
58      }
59  
60      protected void addGlobalMessage(final Severity severity, final String messageResource, final Object... params) {
61          final String msg = getEjbcaWebBean().getText(messageResource, true, params);
62          FacesContext.getCurrentInstance().addMessage(null, new FacesMessage(severity, msg, msg));
63      }
64  
65  	protected void addErrorMessage(String messageResource, Object... params) {
66  		FacesContext ctx = FacesContext.getCurrentInstance();
67  		ctx.addMessage("error", new FacesMessage(FacesMessage.SEVERITY_ERROR,getEjbcaWebBean().getText(messageResource, true, params),getEjbcaWebBean().getText(messageResource, true, params)));
68  	}
69  
70  	protected void addNonTranslatedErrorMessage(String messageResource){
71  		FacesContext ctx = FacesContext.getCurrentInstance();
72  		ctx.addMessage("error", new FacesMessage(FacesMessage.SEVERITY_ERROR,messageResource,messageResource));
73  	}
74  
75  	protected void addInfoMessage(String messageResource, Object... params) {
76          FacesContext ctx = FacesContext.getCurrentInstance();
77          ctx.addMessage("error", new FacesMessage(FacesMessage.SEVERITY_INFO,getEjbcaWebBean().getText(messageResource, true, params),getEjbcaWebBean().getText(messageResource, true, params)));
78      }
79  
80      protected void addNonTranslatedInfoMessage(String messageResource){
81          FacesContext ctx = FacesContext.getCurrentInstance();
82          ctx.addMessage("error", new FacesMessage(FacesMessage.SEVERITY_INFO,messageResource,messageResource));
83      }
84  
85  	protected AuthenticationToken getAdmin(){
86  		return EjbcaJSFHelper.getBean().getAdmin();
87  	}
88  
89  	/**
90  	 * Return the public constants of classObject as a Map, so they can be referenced from the JSF page.
91  	 * (The implementation caches the Map for subsequent calls.)
92  	 * @param classObject Object
93  	 * @return Map
94  	 */
95  	protected Map<String, Object> getPublicConstantsAsMap(Class<?> classObject) {
96  		Map<String, Object> result = publicConstantCache.get(classObject.getName());
97  		if (result != null) {
98  			return result;
99  		}
100 		result = new HashMap<>();
101 		Field[] publicFields = classObject.getFields();
102 		for (int i = 0; i < publicFields.length; i++) {
103 			Field field = publicFields[i];
104 			String name = field.getName();
105 			try {
106 				result.put(name, field.get(null));
107 			} catch (IllegalArgumentException e) {
108 				throw new RuntimeException(e);
109 			} catch (IllegalAccessException e) {
110 				throw new RuntimeException(e);
111 			}
112 		}
113 		publicConstantCache.put(classObject.getName(), result);
114 		return result;
115 	}

117 	/** Sort the provided list by the SelectItems' labels. 
118 	 * @param selectItems Items*/
119 	protected void sortSelectItemsByLabel(List<SelectItem> selectItems) {
120 	    Collections.sort(selectItems, new SelectItemComparator());
121 	}

123     /**
124      * Perform a post-redirect-get if the requests is not invoked via AJAX to the current view id with the specified request string appended.
125      *
126      * It will try to preserve FacesMessages using the bug-riddled Flash scope.
127      * @param requestString String
128      */
129     protected void nonAjaxPostRedirectGet(final String requestString) {
130         final FacesContext facesContext = FacesContext.getCurrentInstance();
131         final PartialViewContext partialViewContext = facesContext.getPartialViewContext();
132         if (!partialViewContext.isAjaxRequest()) {
133             final String viewUrl = facesContext.getApplication().getViewHandler().getActionURL(facesContext, facesContext.getViewRoot().getViewId());
134             final Flash flash = facesContext.getExternalContext().getFlash();
135             flash.setKeepMessages(true);
136             final String url = viewUrl + (requestString==null ? "" : requestString);
137             if (log.isDebugEnabled()) {
138                 log.debug("Trying Post-Redirect-Get to '" + url + "'.");
139             }
140             try {
141                 facesContext.getExternalContext().redirect(url);
142             } catch (IOException e) {
143                 if (log.isDebugEnabled()) {
144                     log.debug("Post-Redirect-Get to '" + url + "' failed: " + e.getMessage());
145                 }
146             }
147         }
148     }
