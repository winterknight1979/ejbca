119 	public String save(){
120 		String retval = "listservices";
121 		ArrayList<String> errorMessages = new ArrayList<>();
122 		try {
123 			serviceConfigurationView.getServiceConfiguration(errorMessages);
124 			if(errorMessages.size() == 0){
125 				ejb.getServiceSession().changeService(getAdmin(), serviceName, serviceConfigurationView.getServiceConfiguration(errorMessages), false);
126 				ejb.getServiceSession().activateServiceTimer(getAdmin(), serviceName);
127 			}else{
128 				Iterator<String> iter = errorMessages.iterator();
129 				while(iter.hasNext()){
130 					addErrorMessage(iter.next());
131 				}
132 				retval = null;				
133 			}
134 		} catch (IOException e) {
135 			addErrorMessage(EjbcaJSFHelper.getBean().getText().get("ERROREDITINGSERVICE") + " " + e.getMessage());
136 		}
137 		return retval;
138 	}

208 		}
209 		log.trace("<getNotifyingType");
210 		return ret;
211 	}

213 	private String getTypeNameFromClassPath(String cp) {
214 		log.debug("ClassPath: "+cp);
215 		String ret = null;
216 		if ( (cp != null) && cp.equals(CertificateExpirationNotifierWorker.class.getName()) ) {
217 			ret = CertificateExpirationNotifierWorkerType.NAME;
218 		}			
219 		if ( (cp != null) && cp.equals(UserPasswordExpireWorker.class.getName()) ) {
220 			ret = UserPasswordExpireWorkerType.NAME;
221 		}			
222 		if ( (cp != null) && cp.equals(RenewCAWorker.class.getName()) ) {
223 			ret = RenewCAWorkerType.NAME;
224 		}			
225 		if ( (cp != null) && cp.equals(RolloverWorker.class.getName()) ) {
226             ret = RolloverWorkerType.NAME;
227         }
228 		if ( (cp != null) && cp.equals(PublishQueueProcessWorker.class.getName()) ) {
229 			ret = PublishQueueWorkerType.NAME;
230 		}			
231 		if ( (cp != null) && cp.equals(CRLUpdateWorker.class.getName()) ) {
232 			ret = CRLUpdateWorkerType.NAME;
233 		}
234         if ( (cp != null) && cp.equals(CRLDownloadWorker.class.getName()) ) {
235             ret = CRLDownloadWorkerType.NAME;
236         }
237         if ( (cp != null) && cp.equals(HsmKeepAliveWorker.class.getName()) ) {
238             ret = HsmKeepAliveWorkerType.NAME;
239         }
240         if (ret==null) {
241             ret = CustomWorkerType.NAME;
242         }
243 		return ret;
244 	}

293      * 
294      * @return List&lt;javax.faces.model.SelectItem&gt;(String, String) of CA id's (as String) and CA names
295      */
296     public List<SelectItem> getAvailableCAs() {
297         List<SelectItem> availableCANames = new ArrayList<>();
298         for (Integer caid : ejb.getCaSession().getAuthorizedCaIds(getAdmin())) {
299             try {
300                 availableCANames.add(new SelectItem(caid.toString(), ejb.getCaSession().getCAInfo(getAdmin(), caid).getName()));
301             } catch (AuthorizationDeniedException e) {
302                 log.debug("Not authorized to CA: " + caid);
303             }
304         }
305         return availableCANames;
306     }

319     /** Returns the list of available external X509 CAs, also including the special option 'Any CA'.
320      * 
321      * @return List&lt;javax.faces.model.SelectItem&gt;(String, String) of CA id's (as String) and CA names
322      */
323     public List<SelectItem> getAvailableExternalX509CAsWithAnyOption() {
324         final List<SelectItem> availableCANames = new ArrayList<>();
325         final String caname = EjbcaJSFHelper.getBean().getText().get("ANYCA");
326         availableCANames.add(new SelectItem(String.valueOf(SecConst.ALLCAS), caname));
327         for (final Integer caid : ejb.getCaSession().getAuthorizedCaIds(getAdmin())) {
328             try {
329                 CAInfo caInfo = ejb.getCaSession().getCAInfo(getAdmin(), caid);
330                 availableCANames.add(new SelectItem(caid.toString(), ejb.getCaSession().getCAInfo(getAdmin(), caid).getName(), null,
331                         caInfo.getCAType() != CAInfo.CATYPE_X509 || caInfo.getStatus() != CAConstants.CA_EXTERNAL));
332             } catch (AuthorizationDeniedException e) {
333                 log.debug("Not authorized to CA: " + caid);
334             }
335         }
336         return availableCANames;
337     }

339     /**
340      * 
341      * 
342      * @return a {@link List} of {@link SelectItem}s containing the ID's and names of all ENDENTITY, ROOTCA and SUBCA 
343      * (and HARDTOKEN if available) certificate profiles current admin is authorized to.
344      */
345     public Collection<SelectItem> getCertificateProfiles() {
346         TreeMap<String, SelectItem> certificateProfiles = new TreeMap<>();
347          
348         final List<Integer> certificateProfileTypes = new ArrayList<>();
349         certificateProfileTypes.add(CertificateConstants.CERTTYPE_ENDENTITY);
350         if (isAuthorizedTo(StandardRules.ROLE_ROOT.resource())) {
351             //Only root users may use CA profiles
352             certificateProfileTypes.add(CertificateConstants.CERTTYPE_ROOTCA);
353             certificateProfileTypes.add(CertificateConstants.CERTTYPE_SUBCA);
354         }
355         
356         if (getEjbcaWebBean().getGlobalConfiguration().getIssueHardwareTokens()) {
357             certificateProfileTypes.add(CertificateConstants.CERTTYPE_HARDTOKEN);
358         }
359              
360         for (Integer certificateProfileType : certificateProfileTypes) {
361             Collection<Integer> profiles = certificateProfileSession.getAuthorizedCertificateProfileIds(getAdmin(), certificateProfileType);
362             for (Integer certificateProfile : profiles) {
363                 String profileName = certificateProfileSession.getCertificateProfileName(
364                         certificateProfile);
365                 certificateProfiles.put(profileName.toLowerCase(), new SelectItem(certificateProfile.toString(), profileName));
366             }
367         }        
368         return certificateProfiles.values();
369     }

372 	public List<SelectItem> getAvailablePublishers(){
373 		List<SelectItem> availablePublisherNames = new ArrayList<>();
374         for (int next : ejb.getCaAdminSession().getAuthorizedPublisherIds(getAdmin())) {
375             // Display it in the list as "PublisherName (publisherId)" with publisherId as the value sent
376             availablePublisherNames.add(new SelectItem(String.valueOf(next), ejb.getPublisherSession().getPublisherName(next) + " (" + next + ")"));
377         }
378         Collections.sort(availablePublisherNames, new Comparator<SelectItem>() {
379             @Override
380             public int compare(SelectItem first, SelectItem second) {
381                 return first.getLabel().compareToIgnoreCase(second.getLabel());
382             }
383         });
384 		return availablePublisherNames;		
385 	}

387 	/** Return type used by getManualCustomActionItems */
388     public static class ManualCustomItems {
389         private final List<SelectItem> workers = new ArrayList<>();
390         private final List<SelectItem> intervals = new ArrayList<>();
391         private final List<SelectItem> actions = new ArrayList<>();
392         public List<SelectItem> getWorkers() { return workers; }
393         public List<SelectItem> getIntervals() { return intervals; }
394         public List<SelectItem> getActions() { return actions; }
395     }
396     
397     public ManualCustomItems getManualCustomItems() {
398         ManualCustomItems manual = new ManualCustomItems();
399         final String workerClass = getCustomWorkerType().getAutoClassPath();
400         if (!StringUtils.isEmpty(workerClass) && !CustomLoader.getCustomClasses(IWorker.class).contains(workerClass)) {
401             manual.getWorkers().add(new SelectItem(workerClass, workerClass+"*"));
402         }
403         final String intervalClass = getCustomIntervalType().getAutoClassPath();
404         if (!StringUtils.isEmpty(intervalClass) && !CustomLoader.getCustomClasses(IInterval.class).contains(intervalClass)) {
405             manual.getIntervals().add(new SelectItem(intervalClass, intervalClass+"*"));
406         }
407         final String actionClass = getCustomActionType().getAutoClassPath();
408         if (!StringUtils.isEmpty(actionClass) && !CustomLoader.getCustomClasses(IAction.class).contains(actionClass)) {
409             manual.getActions().add(new SelectItem(actionClass, actionClass+"*"));
410         }
411         return manual;
412     }
