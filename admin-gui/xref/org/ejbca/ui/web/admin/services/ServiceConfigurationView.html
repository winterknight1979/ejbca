74  	public ServiceConfigurationView(ServiceConfiguration serviceConfiguration) {
75  		
76  		typeManager = new ServiceTypeManager();
77  	
78  		this.serviceConfiguration = serviceConfiguration;
79  		WorkerType workerType = (WorkerType) typeManager.getServiceTypeByClassPath(serviceConfiguration.getWorkerClassPath());
80  		if (workerType == null) {
81  		    workerType = (WorkerType) typeManager.getServiceTypeByName(CustomWorkerType.NAME);
82  		    ((CustomWorkerType) workerType).setClassPath(serviceConfiguration.getWorkerClassPath());
83  		}			
84  	    setWorkerType(workerType);
85  		
86  		IntervalType intervalType = (IntervalType) typeManager.getServiceTypeByClassPath(serviceConfiguration.getIntervalClassPath());
87  		if (intervalType == null) {
88  		    if (workerType.getCompatibleIntervalTypeNames().contains(PeriodicalIntervalType.NAME)) {
89  	            // It seems most likely that the admin wants to configure a periodic interval even if custom interval are available
90  	            intervalType = (IntervalType) typeManager.getServiceTypeByName(PeriodicalIntervalType.NAME);
91  		    } else {
92  	            intervalType = (IntervalType) typeManager.getServiceTypeByName(CustomIntervalType.NAME);
93  	            ((CustomIntervalType) intervalType).setClassPath(serviceConfiguration.getIntervalClassPath());
94  		    }
95  		}						
96  		setIntervalType(intervalType);
97  		selectedInterval = intervalType.getName();
98  		
99  		ActionType actionType = (ActionType) typeManager.getServiceTypeByClassPath(serviceConfiguration.getActionClassPath());
100 		if (actionType == null) {
101             if (workerType.getCompatibleActionTypeNames().contains(NoActionType.NAME)) {
102                 // It seems most likely that the admin wants to configure a "no action" action even if custom actions are available
103                 actionType = (ActionType) typeManager.getServiceTypeByName(NoActionType.NAME);
104             } else {
105                 actionType = (ActionType) typeManager.getServiceTypeByName(CustomActionType.NAME);
106                 ((CustomActionType) actionType).setClassPath(serviceConfiguration.getActionClassPath());
107             }
108 		}						
109 	    setActionType(actionType);
110 	    selectedAction = actionType.getName();
111 		
112 		setDescription(serviceConfiguration.getDescription());
113 		setActive(serviceConfiguration.isActive());
114 		setHidden(serviceConfiguration.isHidden());
115 		setPinToNodes(serviceConfiguration.getPinToNodes());
116 		setRunOnAllNodes(serviceConfiguration.isRunOnAllNodes());
117 	}

234 		
235 		  if(selectedInterval != null && !workerType.getCompatibleIntervalTypeNames().contains(selectedInterval)){

237 			setIntervalType((IntervalType) typeManager.getServiceTypeByName(getSelectedInterval()));
238 		  }
239 		  
240 		
241 		  if(selectedAction != null && !workerType.getCompatibleActionTypeNames().contains(selectedAction)){
242 			setSelectedAction(workerType.getCompatibleActionTypeNames().iterator().next());
243 			setActionType((ActionType) typeManager.getServiceTypeByName(getSelectedAction()));
244 		  }
245 		  
246 		}catch(IOException e){
247 			log.error(e);
248 		}
249 	}

314 	public List<SelectItem> getAvailableWorkers(){
315 		final ArrayList<SelectItem> retval = new ArrayList<>();
316 		final Collection<ServiceType> available = typeManager.getAvailableWorkerTypes();
317 		for (final ServiceType next : available) {
318 			String label = next.getName();
319 			if (next.isTranslatable()) {
320 				label = EjbcaJSFHelper.getBean().getText().get(next.getName());
321 			}
322             retval.add(new SelectItem(next.getName(),label));
323 			if (next instanceof CustomWorkerType) {
324 		        List<String> customClasses = CustomLoader.getCustomClasses(IWorker.class);
325 		        for (final String customClass : customClasses) {
326 		            final String customClassSimpleName = customClass.substring(customClass.lastIndexOf('.')+1);
327 		            final String labelKey = customClassSimpleName.toUpperCase()+"_TITLE";
328 		            label = EjbcaJSFHelper.getBean().getText().get(labelKey);
329 		            if (label.equals(labelKey)) {
330 		                label = customClassSimpleName + " ("+EjbcaJSFHelper.getBean().getText().get(next.getName())+")";
331 		            }
332 		            retval.add(new SelectItem(next.getName()+"-"+customClass, label));
333 		        }
334 			}
335 		}
336 		// Sort by label
337 		Collections.sort(retval, new Comparator<SelectItem>() {
338             @Override
339             public int compare(SelectItem arg0, SelectItem arg1) {
340                 return arg0.getLabel().compareTo(arg1.getLabel());
341             }
342 		});
343 		return retval;
344 	}

348 		final WorkerType currentWorkerType = getWorkerType();
349 		for (final String name : currentWorkerType.getCompatibleIntervalTypeNames()) {
350 			final ServiceType next = typeManager.getServiceTypeByName(name);

354 			}
355 			retval.add(new SelectItem(name,label));

382 	public List<SelectItem> getAvailableCustomWorkerItems() {
383 	    final List<String> customClasses = CustomLoader.getCustomClasses(IWorker.class);
384 	    final List<String> customClassesWithoutUiSupport = new ArrayList<>();
385 	    for (final String classPath : customClasses) {
386 	    	// Exclude all the workers that have custom UI support and will be shown as any other worker
387 	        if (!CustomWorkerType.isCustomUiRenderingSupported(classPath)) {
388 	            customClassesWithoutUiSupport.add(classPath);
389 	        }
390 	    }
391 	    return stringsToItems(customClassesWithoutUiSupport);
392 	}

430 	public List<SelectItem> getNodesInCluster() {
431 		final List<SelectItem> ret = new LinkedList<>();
432 		final Set<String> nodes = EjbcaJSFHelper.getBean().getEjbcaWebBean().getGlobalConfiguration().getNodesInCluster();  
433 		for (String node : nodes) {
434 			ret.add(new SelectItem(node));
435 		}
436 		// Also add unknown nodes, that is nodes that has been removed but this service still is pinned to
437 		for (String node : getPinToNodes()) {
438 			if (!nodes.contains(node)) {
439 				ret.add(new SelectItem(node, node + " " + EjbcaJSFHelper.getBean().getText().get("PINTONODESUNKNOWNNODE")));
440 			}
441 		}
442 		return ret;
