127   /**
128    *  @param caname CA Name
129  * @param p12file PKCS12 File
130  * @param keystorepass Password
131  * @param privateSignatureKeyAlias Alias 
132  * @param privateEncryptionKeyAlias Alias
133  * @throws Exception Fail
134  * @see org.ejbca.core.ejb.ca.caadmin.CAAdminSessionBean
135    */
136   public void importCAFromKeyStore(String caname, byte[] p12file, String keystorepass, String privateSignatureKeyAlias, String privateEncryptionKeyAlias) throws Exception {
137       final KeyStore ks = KeyStore.getInstance("PKCS12", BouncyCastleProvider.PROVIDER_NAME);
138       ks.load(new ByteArrayInputStream(p12file), keystorepass.toCharArray());
139       if (privateSignatureKeyAlias.equals("")) {
140           Enumeration<String> aliases = ks.aliases();
141           if (aliases == null || !aliases.hasMoreElements()) {
142               throw new Exception("This file does not contain any aliases.");
143           }
144           privateSignatureKeyAlias = aliases.nextElement();
145           if (aliases.hasMoreElements()) {
146               while (aliases.hasMoreElements()) {
147                   privateSignatureKeyAlias += " " + aliases.nextElement();
148               }
149               throw new Exception("You have to specify any of the following aliases: " + privateSignatureKeyAlias);
150           }
151       }
152       if ( privateEncryptionKeyAlias.equals("") ) {
153           privateEncryptionKeyAlias = null;
154       }
155       caadminsession.importCAFromKeyStore(administrator, caname, p12file, keystorepass, keystorepass, privateSignatureKeyAlias, privateEncryptionKeyAlias);  
156   }

167   public void importCACertUpdate(int caId, byte[] certbytes) throws CertificateParsingException, CADoesntExistsException, AuthorizationDeniedException, CertificateImportException {
168       Collection<Certificate> certs = null;
169       try {
170           certs = CertTools.getCertsFromPEM(new ByteArrayInputStream(certbytes), Certificate.class);
171       } catch (CertificateException e) {
172           log.debug("Input stream is not PEM certificate(s): "+e.getMessage());
173           // See if it is a single binary certificate
174           certs = new ArrayList<>();
175           certs.add(CertTools.getCertfromByteArray(certbytes, Certificate.class));
176       }
177       caadminsession.updateCACertificate(administrator, caId, EJBTools.wrapCertCollection(certs));
178   }

232     /** @param caId CA ID
233      * @return success
234      * @throws AuthorizationDeniedException Fail 
235      * @see org.ejbca.core.ejb.ca.caadmin.CAAdminSessionBean */  
236     public boolean removeCA(final int caId) throws AuthorizationDeniedException{     
237         final boolean caIdIsPresent = this.endEntitySession.checkForCAId(caId) ||
238                 this.certificateProfileSession.existsCAIdInCertificateProfiles(caId) ||
239                 this.endEntityProfileSession.existsCAInEndEntityProfiles(caId) ||
240                 isCaIdInUseByRoleOrRoleMember(caId);   
241         if (!caIdIsPresent) {
242             caSession.removeCA(administrator, caId);
243         }
244         return !caIdIsPresent;
245     }

247     /** @param caId CA ID
248      * @return true if the CA ID is in use by any Role's access rule or as RoleMember.tokenIssuerId */
249     private boolean isCaIdInUseByRoleOrRoleMember(final int caId) {
250         for (final Role role : roleDataSession.getAllRoles()) {
251             if (role.getAccessRules().containsKey(AccessRulesHelper.normalizeResource(StandardRules.CAACCESS.resource() + caId))) {
252                 return true;
253             }
254             for (final RoleMember roleMember : roleMemberDataSession.findRoleMemberByRoleId(role.getRoleId())) {
255                 if (roleMember.getTokenIssuerId()==caId) {
256                     // Do more expensive checks if it is a potential match
257                     final AccessMatchValue accessMatchValue = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(
258                             roleMember.getTokenType()).getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey());
259                     if (accessMatchValue.isIssuedByCa()) {
260                         return true;
261                     }
262                 }
263             }
264         }
265         return false;
266     }

322   /** @param caid CA ID
323  * @param caChainBytes Chain
324  * @param nextSignKeyAlias Alias
325  * @return Reqiest
326  * @throws CADoesntExistsException Fail
327  * @throws AuthorizationDeniedException Fail 
328  * @throws CryptoTokenOfflineException Fail
329  * @see org.ejbca.core.ejb.ca.caadmin.CAAdminSessionBean */
330   public byte[] makeRequest(int caid, byte[] caChainBytes, String nextSignKeyAlias) throws CADoesntExistsException, AuthorizationDeniedException, CryptoTokenOfflineException {
331       List<Certificate> certChain = null;
332       if (caChainBytes != null) {
333           try {
334               certChain = CertTools.getCertsFromPEM(new ByteArrayInputStream(caChainBytes), Certificate.class);
335               if (certChain.size()==0) {
336                   throw new IllegalStateException("Certificate chain contained no certificates");
337               }
338           } catch (Exception e) {
339               // Maybe it's just a single binary CA cert
340               try {
341                   Certificate cert = CertTools.getCertfromByteArray(caChainBytes, Certificate.class);
342                   certChain = new ArrayList<>();
343                   certChain.add(cert);
344               } catch (CertificateParsingException e2) {
345                   // Ok.. so no chain was supplied.. we go ahead anyway..
346                   throw new CADoesntExistsException("Invalid CA chain file.");
347               }
348           }
349       }
350       try {
351           return caadminsession.makeRequest(administrator, caid, certChain, nextSignKeyAlias);
352       } catch (CertPathValidatorException e) {
353           throw new RuntimeException("Unexpected outcome.", e);
354       }
355   }

368   /** @param caid CA ID
369  * @param certBytes Certs
370  * @param nextSignKeyAlias Alias 
371  * @param futureRollover bool
372  * @throws Exception Fail
373  * @see org.ejbca.core.ejb.ca.caadmin.CAAdminSessionBean#receiveResponse */  
374   public void receiveResponse(int caid, byte[] certBytes, String nextSignKeyAlias, boolean futureRollover) throws Exception{
375 	  try {
376           final List<Certificate> certChain = new ArrayList<>();
377 		  try {
378 		      certChain.addAll(CertTools.getCertsFromPEM(new ByteArrayInputStream(certBytes), Certificate.class));
379           } catch (CertificateException e) {
380               log.debug("Input stream is not PEM certificate(s): "+e.getMessage());
381               // See if it is a single binary certificate
382               certChain.add(CertTools.getCertfromByteArray(certBytes, Certificate.class));
383           }
384 		  if (certChain.size()==0) {
385 		      throw new Exception("No certificate(s) could be read.");
386 		  }
387 		  Certificate caCertificate = certChain.get(0);
388 		  final X509ResponseMessage resmes = new X509ResponseMessage();
389 		  resmes.setCertificate(caCertificate);
390 		  caadminsession.receiveResponse(administrator, caid, resmes, certChain.subList(1, certChain.size()), nextSignKeyAlias, futureRollover);
391 	  } catch (Exception e) {
392 	      // log the error here, since otherwise it may be hidden by web pages...
393 		  log.error("Error receiving response: ", e);
394 		  throw e;
395 	  }
396   }

415   public boolean renewCA(int caid, String nextSignKeyAlias, boolean createLinkCertificate) throws Exception {
416       if (getCAInfo(caid).getCAInfo().getSignedBy() == CAInfo.SIGNEDBYEXTERNALCA) {
417           return false;
418       } else {
419           if (getCAInfo(caid).getCAInfo().getCAType()==CAInfo.CATYPE_CVC) {
420               // Force generation of link certificate for CVC CAs
421               createLinkCertificate = true;
422           }
423           if (nextSignKeyAlias == null || nextSignKeyAlias.length()==0) {
424               // Generate new keys
425               caadminsession.renewCA(administrator, caid, true, null, createLinkCertificate);
426           } else {
427               // Use existing keys
428               caadminsession.renewCA(administrator, caid, nextSignKeyAlias, null, createLinkCertificate);
429           }
430           return true;
431       }
432   }

458   /**
459    *  @param caid CA ID
460  * @throws AuthorizationDeniedException Fail
461  * @throws CADoesntExistsException Fail
462    */  
463  public void publishCA(int caid) throws AuthorizationDeniedException, CADoesntExistsException {
464  	CAInfo cainfo = caSession.getCAInfo(administrator, caid);
465  	Collection<Integer> publishers = cainfo.getCRLPublishers();
466  	// Publish ExtendedCAServices certificates as well
467 	Iterator<ExtendedCAServiceInfo> iter = cainfo.getExtendedCAServiceInfos().iterator();
468 	while(iter.hasNext()){
469 		ExtendedCAServiceInfo next = iter.next();	
470 		// Only publish certificates for active services
471 		if (next.getStatus() == ExtendedCAServiceInfo.STATUS_ACTIVE) {
472 			// The OCSP certificate is the same as the CA signing certificate
473 			if (next instanceof BaseSigningCAServiceInfo){
474 				List<Certificate> signingcert = ((BaseSigningCAServiceInfo) next).getCertificatePath();
475 				if (signingcert != null) {
476 					caadminsession.publishCACertificate(administrator, signingcert, publishers, cainfo.getSubjectDN());
477 				}
478 			}
479 		}
480 	}  
481     CertificateProfile certprofile = certificateProfileSession.getCertificateProfile(cainfo.getCertificateProfileId());
482     // A CA certificate is published where the CRL is published and if there is a publisher noted in the certificate profile 
483     // (which there is probably not) 
484     publishers.addAll(certprofile.getPublisherList());
485     caadminsession.publishCACertificate(administrator, cainfo.getCertificateChain(), publishers, cainfo.getSubjectDN());
486     caadminsession.publishCRL(administrator, cainfo.getCertificateChain().iterator().next(), publishers, cainfo.getSubjectDN(), cainfo.getDeltaCRLPeriod()>0);
487  }
