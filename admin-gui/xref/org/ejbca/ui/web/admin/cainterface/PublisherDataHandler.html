14  package org.ejbca.ui.web.admin.cainterface;
15  
16  import java.io.Serializable;
17  
18  import org.cesecore.authentication.tokens.AuthenticationToken;
19  import org.cesecore.authorization.AuthorizationDeniedException;
20  import org.cesecore.common.exception.ReferencesToItemExistException;
21  import org.ejbca.core.ejb.ca.publisher.PublisherSessionLocal;
22  import org.ejbca.core.model.ca.publisher.BasePublisher;
23  import org.ejbca.core.model.ca.publisher.PublisherConnectionException;
24  import org.ejbca.core.model.ca.publisher.PublisherDoesntExistsException;
25  import org.ejbca.core.model.ca.publisher.PublisherExistsException;
26  
27  /**
28   * A class handling the hardtoken profile data in the webinterface.
29   * 
30   * @deprecated since 6.12.0. Use PublisherSession directly instead
31   *
32   * @version $Id: PublisherDataHandler.java 30470 2018-11-12 11:02:11Z samuellb $
33   */
34  @Deprecated
35  public class PublisherDataHandler implements Serializable {
36  
37      private static final long serialVersionUID = -5646053740072121787L;
38  
39      private PublisherSessionLocal publishersession;
40      private AuthenticationToken administrator;
41  
42      /** Creates a new instance of PublisherDataHandler 
43       * @param administrator Admin
44       * @param publishersession Session */
45      public PublisherDataHandler(AuthenticationToken administrator, PublisherSessionLocal publishersession) {
46          this.publishersession = publishersession;
47          this.administrator = administrator;
48      }
49  
50      /** Method to add a publisher. Throws PublisherExitsException if profile already exists  
51       * @param name Name
52       * @param publisher Pub 
53       * @throws PublisherExistsException Fail 
54       * @throws AuthorizationDeniedException Fail */
55      public void addPublisher(String name, BasePublisher publisher) throws PublisherExistsException, AuthorizationDeniedException {
56          publishersession.addPublisher(administrator, name, publisher);
57  
58      }
59  
60      /** Method to change a publisher. 
61       * @param name Name
62       * @param publisher Pub 
63       * @throws AuthorizationDeniedException fail  */
64      public void changePublisher(String name, BasePublisher publisher) throws AuthorizationDeniedException {
65          publishersession.changePublisher(administrator, name, publisher);
66      }
67  
68      /**
69       * Removes a publisher
70       * @param name Name
71       * @throws AuthorizationDeniedException if not authorized
72       * @throws ReferencesToItemExistException if references exist.
73       */
74      public void removePublisher(String name) throws ReferencesToItemExistException, AuthorizationDeniedException {
75          publishersession.removePublisher(administrator, name);
76      }
77  
78      /** Metod to rename a publisher 
79       * @param oldname Name
80       * @param newname Name
81       * @throws PublisherExistsException fail
82       * @throws AuthorizationDeniedException fail */
83      public void renamePublisher(String oldname, String newname) throws PublisherExistsException, AuthorizationDeniedException {
84          publishersession.renamePublisher(administrator, oldname, newname);
85  
86      }
87  
88      public void clonePublisher(String originalname, String newname) throws AuthorizationDeniedException, PublisherDoesntExistsException,
89              PublisherExistsException {
90          publishersession.clonePublisher(administrator, originalname, newname);
91      }
92  
93      public void testConnection(String name) throws PublisherConnectionException {
94          publishersession.testConnection(publishersession.getPublisherId(name));
95  
96      }
97      public BasePublisher getPublisher(String name) {
98          return publishersession.getPublisher(name);
99      }
100 
101     public int getPublisherId(String name) {
102         return publishersession.getPublisherId(name);
103     }
104 
105 }
