35  /**
36   * Servlet used to distribute End Entity certificates through the "View Certificate" jsp page.
37   * Checks that the administrator is authorized to view the user before sending the certificate<br>
38   *
39   * cert - returns certificate in PEM-format
40   * nscert - returns certificate for Firefox
41   * iecert - returns certificate for Internet Explorer
42   *
43   * cert, nscert and iecert also takes  parameters issuer and certificate sn were issuer is the DN of issuer and certificate serienumber 
44   * is in hex format.
45   *
46   * @version $Id: EndEntityCertServlet.java 34154 2019-12-23 13:38:17Z samuellb $
47   */
48  public class EndEntityCertServlet extends BaseAdminServlet {
49  
50  	private static final long serialVersionUID = 1L;
51  	private static final Logger log = Logger.getLogger(EndEntityCertServlet.class);
52  
53      private static final String COMMAND_PROPERTY_NAME = "cmd";
54      private static final String COMMAND_NSCERT = "nscert";
55      private static final String COMMAND_IECERT = "iecert";
56      private static final String COMMAND_CERT = "cert";
57     
58      private static final String ISSUER_PROPERTY = "issuer";
59      private static final String CERTIFICATEDN_PROPERTY = "certificatesn";
60  
61      @Override
62      public void doPost(HttpServletRequest req, HttpServletResponse res) throws IOException, ServletException {
63          log.trace(">doPost()");
64          doGet(req, res);
65          log.trace("<doPost()");
66      }
67  
68      @Override
69      public void doGet(HttpServletRequest req,  HttpServletResponse res) throws IOException, ServletException {
70          log.trace(">doGet()");
71          try {
72              authenticateAdmin(req, res, AccessRulesConstants.REGULAR_VIEWCERTIFICATE);
73          } catch (AdminWebAuthenticationException authExc) {
74          	// TODO: localize this.
75          	log.info("Authentication failed", authExc);
76              res.sendError(HttpServletResponse.SC_FORBIDDEN, "Authentication failed");
77              return;
78          }
79          RequestHelper.setDefaultCharacterEncoding(req);
80          String issuerdn = req.getParameter(ISSUER_PROPERTY);        
81          String certificatesn = req.getParameter(CERTIFICATEDN_PROPERTY);
82  
83          String command;
84          // Keep this for logging.
85          log.debug("Got request from "+req.getRemoteAddr());
86          command = req.getParameter(COMMAND_PROPERTY_NAME);
87          if (command == null) {
88              command = "";
89          }
90          if ((command.equalsIgnoreCase(COMMAND_NSCERT) || command.equalsIgnoreCase(COMMAND_IECERT) || command.equalsIgnoreCase(COMMAND_CERT)) 
91          	 && issuerdn != null && certificatesn != null) {
92          	
93          	BigInteger certsn = new BigInteger(certificatesn,16);
94          	        	        
95          	// Fetch the certificate and at the same time check that the user is authorized to it.
96          	
97          	try {
98  				final RAInterfaceBean raBean = getRaBean(req);
99  				raBean.loadCertificates(certsn, issuerdn);
100 				CertificateView certview = raBean.getCertificate(0);
101 				
102 				Certificate cert = certview.getCertificate();
103 				byte[] enccert = cert.getEncoded();
104                 // We must remove cache headers for IE
105                 ServletUtils.removeCacheHeaders(res);
106 				if (command.equalsIgnoreCase(COMMAND_NSCERT)) {
107 					res.setContentType("application/x-x509-ca-cert");
108 					res.setContentLength(enccert.length);
109 					res.getOutputStream().write(enccert);
110 					log.debug("Sent CA cert to NS client, len="+enccert.length+".");
111 				} else if (command.equalsIgnoreCase(COMMAND_IECERT)) {
112 					res.setHeader("Content-disposition", "attachment; filename=" + URLEncoder.encode(certview.getUsername(),"UTF-8") + ".crt");
113 					res.setContentType("application/octet-stream");
114 					res.setContentLength(enccert.length);
115 					res.getOutputStream().write(enccert);
116 					log.debug("Sent CA cert to IE client, len="+enccert.length+".");
117 				} else if (command.equalsIgnoreCase(COMMAND_CERT)) {
118 					String out = CertTools.getPemFromCertificate(cert);
119 					res.setHeader("Content-disposition", "attachment; filename=" + URLEncoder.encode(certview.getUsername(),"UTF-8") + ".pem");
120 					res.setContentType("application/octet-stream");
121 					res.setContentLength(out.length());
122 					res.getOutputStream().write(out.getBytes());
123 					log.debug("Sent CA cert to client, len="+out.length()+".");
124 				} else {
125 					res.setContentType("text/plain");
126 					res.getOutputStream().println("Commands="+COMMAND_NSCERT+" || "+COMMAND_IECERT+" || "+COMMAND_CERT);
127 					return;
128 				}
129             } catch (Exception e) {
130                 log.error("Error getting certificates: ", e);
131                 res.sendError(HttpServletResponse.SC_NOT_FOUND, "Error getting certificates.");
132                 return;
133             }
134         }
135         else {
136             res.setContentType("text/plain");
137             res.sendError(HttpServletResponse.SC_BAD_REQUEST, "Bad Request format");
138             return;
139         }
140     } // doGet
