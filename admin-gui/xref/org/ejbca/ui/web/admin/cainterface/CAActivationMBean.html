144 	private final AuthenticationToken authenticationToken = EjbcaJSFHelper.getBean().getEjbcaWebBean().getAdminObject();
145     private final EjbLocalHelper ejbLocalhelper = new EjbLocalHelper();
146 	private final CAAdminSessionLocal caAdminSession = ejbLocalhelper.getCaAdminSession();
147 	private final CaSessionLocal caSession = ejbLocalhelper.getCaSession();

149     private final AuthorizationSessionLocal authorizationSession = ejbLocalhelper.getAuthorizationSession();

151 	private List<TokenAndCaActivationGuiComboInfo> authorizedTokensAndCas = null;
152 	private String authenticationcode;
153 
154 	public List<TokenAndCaActivationGuiComboInfo> getAuthorizedTokensAndCas() {
155         final Map<Integer,TokenAndCaActivationGuiInfo> sortMap = new HashMap<>();
156         for (final CAInfo caInfo : caSession.getAuthorizedAndEnabledCaInfos(authenticationToken)) {
157                 final Integer cryptoTokenId = Integer.valueOf(caInfo.getCAToken().getCryptoTokenId());
158                 if (sortMap.get(cryptoTokenId)==null) {
159                     // Perhaps not authorized to view the CryptoToken used by the CA, but we implicitly
160                     // allow this in the current context since we are authorized to the CA.
161                     final CryptoTokenInfo cryptoTokenInfo = cryptoTokenManagementSession.getCryptoTokenInfo(cryptoTokenId.intValue());
162                     if (cryptoTokenInfo==null) {
163                         sortMap.put(cryptoTokenId, new TokenAndCaActivationGuiInfo(cryptoTokenId));
164                     } else {
165                         final boolean allowedActivation = authorizationSession.isAuthorizedNoLogging(authenticationToken, CryptoTokenRules.ACTIVATE.resource() + '/' + cryptoTokenId);
166                         final boolean allowedDeactivation = authorizationSession.isAuthorizedNoLogging(authenticationToken, CryptoTokenRules.DEACTIVATE.resource() + '/' + cryptoTokenId);
167                         sortMap.put(cryptoTokenId, new TokenAndCaActivationGuiInfo(cryptoTokenInfo, allowedActivation, allowedDeactivation));
168                     }
169                 }
170                 sortMap.get(cryptoTokenId).add(new CaActivationGuiInfo(caInfo.getStatus(), caInfo.getIncludeInHealthCheck(), caInfo.getName(), caInfo.getCAId()));
171         }
172         final TokenAndCaActivationGuiInfo[] tokenAndCasArray = sortMap.values().toArray(new TokenAndCaActivationGuiInfo[0]);
173         // Sort array by CryptoToken name
174         Arrays.sort(tokenAndCasArray, new Comparator<TokenAndCaActivationGuiInfo>() {
175             @Override
176             public int compare(TokenAndCaActivationGuiInfo o1, TokenAndCaActivationGuiInfo o2) {
177                 return o1.getCryptoTokenName().compareToIgnoreCase(o2.getCryptoTokenName());
178             }
179         });
180         final List<TokenAndCaActivationGuiComboInfo> retValues = new ArrayList<>();
181         for (final TokenAndCaActivationGuiInfo value : tokenAndCasArray) {
182             boolean first = true;
183             final CaActivationGuiInfo[] casArray = value.getCas().toArray(new CaActivationGuiInfo[0]);
184             // Sort array by CA name
185             Arrays.sort(casArray, new Comparator<CaActivationGuiInfo>() {
186                 @Override
187                 public int compare(CaActivationGuiInfo o1, CaActivationGuiInfo o2) {
188                     return o1.getName().compareToIgnoreCase(o2.getName());
189                 }
190             });
191             for (final CaActivationGuiInfo value2 : casArray) {
192                 retValues.add(new TokenAndCaActivationGuiComboInfo(value, value2, first));
193                 first = false;
194             }
195         }
196         authorizedTokensAndCas = retValues;
197         return retValues;
198     }

200 	/**
201 	 * Tries to activate CryptoTokens (once for each), if authentication code is present and activation is requested.
202 	 * Set the CA service status to the requested state for each CA.
203 	 */
204 	public void applyChanges() {
205 	    if (authorizedTokensAndCas==null) {
206 	        return;
207 	    }
208 	    for (final TokenAndCaActivationGuiComboInfo tokenAndCaCombo : authorizedTokensAndCas) {
209             if (tokenAndCaCombo.isFirst()) {
210                 TokenAndCaActivationGuiInfo tokenAndCa = tokenAndCaCombo.getCryptoToken();
211                 if (log.isDebugEnabled()) {
212                     log.debug("isCryptoTokenActive(): " + tokenAndCa.isCryptoTokenActive() + " isCryptoTokenNewState(): " + tokenAndCa.isCryptoTokenNewState());
213                 }
214 	            if (tokenAndCa.isCryptoTokenActive() != tokenAndCa.isCryptoTokenNewState()) {
215 	                if (tokenAndCa.isCryptoTokenNewState()) {
216 	                    // Assert that authcode is present
217 	                    if (authenticationcode != null && authenticationcode.length()>0) {
218 	                        // Activate CA's CryptoToken
219 	                        try {
220 	                            cryptoTokenManagementSession.activate(authenticationToken, tokenAndCa.getCryptoTokenId(), authenticationcode.toCharArray());
221 	                            log.info(authenticationToken.toString() + " activated CryptoToken " + tokenAndCa.getCryptoTokenId());
222 	                        } catch (CryptoTokenAuthenticationFailedException e) {
223 	                            super.addNonTranslatedErrorMessage("Bad authentication code.");
224 	                        } catch (CryptoTokenOfflineException e) {
225 	                            super.addNonTranslatedErrorMessage("Crypto Token is offline and cannot be activated.");
226 	                        } catch (AuthorizationDeniedException e) {
227 	                            super.addNonTranslatedErrorMessage(e.getMessage());
228 	                        }
229 	                    } else {
230 	                        super.addNonTranslatedErrorMessage("Authentication code required.");
231 	                    }
232 	                } else {
233 	                    // Deactivate CA's CryptoToken
234 	                    try {
235 	                        cryptoTokenManagementSession.deactivate(authenticationToken, tokenAndCa.getCryptoTokenId());
236 	                        log.info(authenticationToken.toString() + " deactivated CryptoToken " + tokenAndCa.getCryptoTokenId());
237 	                    } catch (AuthorizationDeniedException e) {
238 	                        super.addNonTranslatedErrorMessage(e.getMessage());
239 	                    }
240 	                }
241 	            }
242 	        }
243 	        CaActivationGuiInfo ca = tokenAndCaCombo.getCa();
244 	        if (ca.isActive() != ca.isNewState()) {
245 	            // Valid transition 1: Currently offline, become active
246 	            if (ca.isNewState() && ca.getStatus()==CAConstants.CA_OFFLINE) {
247 	                try {
248 	                    caAdminSession.activateCAService(authenticationToken, ca.getCaId());
249 	                } catch (Exception e) {
250 	                    super.addNonTranslatedErrorMessage(e.getMessage());
251 	                }
252 	            } 
253 	            // Valid transition 2: Currently online, become offline
254 	            if (!ca.isNewState() && ca.getStatus()==CAConstants.CA_ACTIVE) {
255 	                try {
256 	                    caAdminSession.deactivateCAService(authenticationToken, ca.getCaId());
257 	                } catch (Exception e) {
258 	                    super.addNonTranslatedErrorMessage(e.getMessage());
259 	                }
260 	            }
261 	        }
262 	        if (ca.isMonitored() != ca.isMonitoredNewState()) {
263 	            // Only persist changes if there are any
264 	            try {
265 	                final CAInfo caInfo = caSession.getCAInfoInternal(ca.getCaId(), null, false);
266 	                caInfo.setIncludeInHealthCheck(ca.isMonitoredNewState());
267 	                caAdminSession.editCA(authenticationToken, caInfo);
268 	            } catch (AuthorizationDeniedException e) {
269 	                super.addNonTranslatedErrorMessage(e.getMessage());
270 	            }
271 	        }
272 	        if (log.isDebugEnabled()) {
273 	            log.debug("caId: " + ca.getCaId() + " monitored: " + ca.isMonitored() + " newCaStatus: " + ca.isNewState());
274 	        }
275 	    }
276 	}

278 	/** @return true when there is at least one CryptoToken that can be activated. */
279     public boolean isActivationCodeShown() {
280         if (authorizedTokensAndCas!=null) {
281             for (final TokenAndCaActivationGuiComboInfo tokenAndCa : authorizedTokensAndCas) {
282                 if (tokenAndCa.isFirst()) {
283                     if (!tokenAndCa.getCryptoToken().isCryptoTokenActive() && !tokenAndCa.getCryptoToken().isStateChangeDisabled()) {
284                         return true;
285                     }
286                 }
287             }
288         }
289         return false;
290     }
