135     /**
136      * Handles PKCS10 certificate request, these are constructed as:
137      * <pre><code>
138      * CertificationRequest ::= SEQUENCE {
139      * certificationRequestInfo  CertificationRequestInfo,
140      * signatureAlgorithm          AlgorithmIdentifier{{ SignatureAlgorithms }},
141      * signature                       BIT STRING
142      * }
143      * CertificationRequestInfo ::= SEQUENCE {
144      * version             INTEGER { v1(0) } (v1,...),
145      * subject             Name,
146      * subjectPKInfo   SubjectPublicKeyInfo{{ PKInfoAlgorithms }},
147      * attributes          [0] Attributes{{ CRIAttributes }}
148      * }
149      * SubjectPublicKeyInfo { ALGORITHM : IOSet} ::= SEQUENCE {
150      * algorithm           AlgorithmIdentifier {{IOSet}},
151      * subjectPublicKey    BIT STRING
152      * }
153      * </code></pre>
154      *
155      * PublicKey's encoded-format has to be RSA X.509.
156      */
157     @Override
158     public void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
159         final AuthenticationToken admin;
160         try {
161             admin = authenticateAdmin(request, response, AccessRulesConstants.REGULAR_CREATEENDENTITY);
162         } catch (AdminWebAuthenticationException authExc) {
163         	// TODO: localize this.
164         	log.info("Authentication failed", authExc);
165             response.sendError(HttpServletResponse.SC_FORBIDDEN, "Authentication failed");
166             return;
167         }
168 
169         RequestHelper.setDefaultCharacterEncoding(request);
170 
171         byte[] buffer = pkcs10Bytes(request.getParameter("pkcs10req"));
172         if (buffer == null) {
173             response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Invalid request, missing 'pkcs10req'!");
174             return;
175         }
176 
177         RAInterfaceBean rabean = getRaBean(request);
178 
179         // Decompose the PKCS#10 request, and create the user.
180         PKCS10RequestMessage p10 = new PKCS10RequestMessage(buffer);
181         String dn = p10.getCertificationRequest().getSubject().toString();
182 
183         String username = request.getParameter("username");
184         if (username == null || username.trim().length() == 0) {
185             username = dn;
186         }
187         // Strip dangerous chars
188         username = StringTools.stripUsername(username);
189         // need null check here?
190         // Before doing anything else, check if the user name is unique and ok.
191         username = checkUsername(rabean, username);
192 
193         UserViewinterface/UserView.html#UserView">UserView newuser = new UserView();
194         newuser.setUsername(username);
195 
196         newuser.setSubjectDN(dn);
197         newuser.setTokenType(SecConst.TOKEN_SOFT_BROWSERGEN);
198         newuser.setKeyRecoverable(false);
199 
200         String email = CertTools.getPartFromDN(dn, "E"); // BC says VeriSign
201         if (email == null) {
202             email = CertTools.getPartFromDN(dn, "EMAILADDRESS");
203         } else {
204             newuser.setEmail(email);
205         }
206 
207         String tmp = null;
208         int eProfileId = EndEntityConstants.EMPTY_END_ENTITY_PROFILE;
209         if ((tmp = request.getParameter("entityprofile")) != null) {
210             int reqId;
211             try {
212                 reqId = rabean.getEndEntityProfileId(tmp);
213             } catch (EndEntityProfileNotFoundException e) {
214                 throw new ServletException("No such end entity profile: " + tmp, e);
215             }
216             eProfileId = reqId;
217         }
218         newuser.setEndEntityProfileId(eProfileId);
219 
220         int cProfileId = CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER;
221         if ((tmp = request.getParameter("certificateprofile")) != null) {
222             CAInterfaceBean cabean = getCaBean(request);
223             int reqId = cabean.getCertificateProfileId(tmp);
224             if (reqId == 0) {
225                 throw new ServletException("No such certificate profile: " + tmp);
226             }
227             cProfileId = reqId;
228         }
229         newuser.setCertificateProfileId(cProfileId);
230 
231         int caid = 0;
232         if ((tmp = request.getParameter("ca")) != null) {
233             // TODO: get requested CA to sign with
234         }
235         newuser.setCAId(caid);
236 
237         String password = request.getParameter("password");
238         if (password == null) {
239             password = "";
240         }
241         newuser.setPassword(password);
242         newuser.setClearTextPassword(false);
243 
244         try {
245             rabean.addUser(newuser);
246         } catch (Exception e) {
247             throw new ServletException("Error adding user: " + e.toString(), e);
248         }
249 
250         byte[] pkcs7;
251         try {
252             p10.setUsername(username);
253             p10.setPassword(password);
254             ResponseMessage resp = signSession.createCertificate(admin, p10, X509ResponseMessage.class, null);
255             X509Certificate cert = CertTools.getCertfromByteArray(resp.getResponseMessage(), X509Certificate.class);
256             pkcs7 = signSession.createPKCS7(admin, cert, true);
257         } catch (EjbcaException e) {
258             // EJBCA did not accept any of all parameters in the request.
259             throw new ServletException(e);
260         } catch (CertificateEncodingException e) {
261             // Error in cert
262             throw new ServletException(e);
263         } catch (CertificateException e) {
264             // Error in cert
265             throw new ServletException(e);
266         } catch (CesecoreException e) {
267             // EJBCA did not accept any of all parameters in the request.
268             throw new ServletException(e);
269         } catch (AuthorizationDeniedException e) {
270             // Weird authorization error.
271             throw new ServletException(e);
272         } catch (CertificateExtensionException e) {
273             throw new ServletException(e);
274         }
275         if (log.isDebugEnabled()) {
276             log.debug("Created certificate (PKCS7) for " + username);
277         }
278         sendNewB64Cert(Base64.encode(pkcs7), response);
279 
280     }

290     private void sendNewB64Cert(byte[] b64cert, HttpServletResponse out) throws IOException {
291         out.setContentType("application/octet-stream");
292         out.setHeader("Content-Disposition", "filename=cert.pem");
293         out.setContentLength(b64cert.length + BEGIN_CERT_LENGTH + END_CERT_LENGTH + (3 * NL_LENGTH));
294 
295         ServletOutputStream os = out.getOutputStream();
296         os.write(BEGIN_CERT);
297         os.write(NL);
298         os.write(b64cert);
299         os.write(NL);
300         os.write(END_CERT);
301         os.write(NL);
302         out.flushBuffer();
303     }

305     /**
306      * @param pkcs10 PKCS 10 string
307      * @return bytes
308      *
309      */
310     private final static byte[] pkcs10Bytes(String pkcs10) {
311         byte[] bytes = null;
312         if (pkcs10 != null) {
313             byte[] reqBytes = pkcs10.getBytes();
314             try {
315                 // A real PKCS10 PEM request
316                 String beginKey = "-----BEGIN CERTIFICATE REQUEST-----";
317                 String endKey = "-----END CERTIFICATE REQUEST-----";
318                 bytes = FileTools.getBytesFromPEM(reqBytes, beginKey, endKey);
319             } catch (IOException e) {
320                 try {
321                     // Keytool PKCS10 PEM request
322                     String beginKey = "-----BEGIN NEW CERTIFICATE REQUEST-----";
323                     String endKey = "-----END NEW CERTIFICATE REQUEST-----";
324                     bytes = FileTools.getBytesFromPEM(reqBytes, beginKey, endKey);
325                 } catch (IOException e2) {
326                     // IE PKCS10 Base64 coded request
327                     bytes = Base64.decode(reqBytes);
328                 }
329             }
330         }
331         return bytes;
332     }

362     /**
363      * @param rabean Bean
364      * @param username User
365      * @return String
366      * @throws ServletException Fail
367      *
368      */
369     private final String checkUsername(RAInterfaceBean rabean, String username) throws ServletException {
370         if (username != null) {
371             username = username.trim();
372         }
373         if (username == null || username.length() == 0) {
374             throw new ServletException("Username must not be empty.");
375         }
376 
377         String msg = null;
378         try {
379             if (rabean.userExist(username)) {
380                 msg = "User '" + username + "' already exists.";
381             }
382         } catch (Exception e) {
383             throw new ServletException("Error checking username '" + username + ": " + e.toString(), e);
384         }
385         if (msg != null) {
386             throw new ServletException(msg);
387         }
388 
389         return username;
390     }
