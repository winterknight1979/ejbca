216     public CertificateView[] getCACertificates(int caid) {
217         final List<CertificateView> ret = new ArrayList<>();
218         for (final Certificate certificate : signsession.getCertificateChain(caid)) {
219             RevokedInfoView revokedinfo = null;
220             CertificateStatus revinfo = certificatesession.getStatus(CertTools.getIssuerDN(certificate), CertTools.getSerialNumber(certificate));
221             if (revinfo != null && revinfo.getRevocationReason() != RevokedCertInfo.NOT_REVOKED) {
222                 revokedinfo = new RevokedInfoView(revinfo, CertTools.getSerialNumber(certificate));
223             }
224             ret.add(new CertificateView(certificate, revokedinfo));
225         }
226         return ret.toArray(new CertificateView[0]);
227     }

250     public List<CaCrlStatusInfo> getAuthorizedInternalCaCrlStatusInfos() throws Exception {
251         final List<CaCrlStatusInfo> ret = new ArrayList<>();
252         final Collection<Integer> caIds = getAuthorizedCAs();
253         for (final Integer caId : caIds) {
254             final CAInfo cainfo = casession.getCAInfoInternal(caId.intValue());
255             if (cainfo == null || cainfo.getStatus() == CAConstants.CA_EXTERNAL) {
256                 continue;
257             }
258             final String caName = cainfo.getName();
259             boolean caTokenStatus = false;
260             final int cryptoTokenId = cainfo.getCAToken().getCryptoTokenId();
261             try {
262                 caTokenStatus = cryptoTokenManagementSession.isCryptoTokenStatusActive(cryptoTokenId);
263             } catch (Exception e) {
264                 final String msg = authenticationToken.toString() + " failed to load CryptoToken status for " + cryptoTokenId;
265                 if (log.isDebugEnabled()) {
266                     log.debug(msg, e);
267                 } else {
268                     log.info(msg);
269                 }
270             }
271             final boolean caService = (cainfo.getStatus() == CAConstants.CA_ACTIVE) && caTokenStatus;
272             boolean crlStatus = true;
273             final Date now = new Date();
274             final CRLInfo crlinfo = getLastCRLInfo(cainfo, false);
275             if ((crlinfo != null) && (now.after(crlinfo.getExpireDate()))) {
276                 crlStatus = false;
277             }
278             final CRLInfo deltacrlinfo = getLastCRLInfo(cainfo, true);
279             if ((deltacrlinfo != null) && (now.after(deltacrlinfo.getExpireDate()))) {
280                 crlStatus = false;
281             }
282             ret.add(new CaCrlStatusInfo(caName, caService, crlStatus));
283         }
284 
285         return ret;
286     }

493 	public String republish(CertificateView certificateView) throws AuthorizationDeniedException {
494 		String returnval = "CERTREPUBLISHFAILED";
495 		int certificateProfileId = CertificateProfileConstants.CERTPROFILE_NO_PROFILE;
496 		String password = null;
497 		ExtendedInformation ei = null;
498 		// Unescaped subjectDN is used to avoid causing issues in custom publishers (see ECA-6761)
499 		String dn = certificateView.getSubjectDNUnescaped(); 
500 		final CertReqHistory certreqhist = certreqhistorysession.retrieveCertReqHistory(certificateView.getSerialNumberBigInt(), certificateView.getIssuerDN());
501 		if (certreqhist != null) {
502 			// First try to look up all info using the Certificate Request History from when the certificate was issued
503 			// We need this since the certificate subjectDN might be a subset of the subjectDN in the template
504 			certificateProfileId = certreqhist.getEndEntityInformation().getCertificateProfileId();
505 			password = certreqhist.getEndEntityInformation().getPassword();
506 			ei = certreqhist.getEndEntityInformation().getExtendedInformation();
507 			dn = certreqhist.getEndEntityInformation().getCertificateDN();
508 		}
509 		final String fingerprint = certificateView.getSHA1Fingerprint().toLowerCase();
510 		final CertificateDataWrapper cdw = certificatesession.getCertificateData(fingerprint);
511 		if (cdw != null) {
512 			// If we are missing Certificate Request History for this certificate, we can at least recover some of this info
513 			if (certificateProfileId == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
514 				certificateProfileId = cdw.getCertificateData().getCertificateProfileId();
515 			}
516 		}
517 		if (certificateProfileId == CertificateProfileConstants.CERTPROFILE_NO_PROFILE) {
518 			// If there is no cert req history and the cert profile was not defined in the CertificateData row, so we can't do anything about it..
519 			returnval = "CERTREQREPUBLISHFAILED";
520 		} else {
521 			final CertificateProfile certprofile = certificateProfileSession.getCertificateProfile(certificateProfileId);
522 			if (certprofile != null) {
523 				if (certprofile.getPublisherList().size() > 0) {
524                     if (publishersession.storeCertificate(authenticationToken, certprofile.getPublisherList(), cdw, password, dn, ei)) {
525                         returnval = "CERTREPUBLISHEDSUCCESS";
526                     }
527 				} else {
528 					returnval = "NOPUBLISHERSDEFINED";
529 				}
530 			} else {
531 				returnval = "CERTPROFILENOTFOUND";
532 			}
533 		}
534 		return returnval; 
535 	}

989     public String isValidityTimeValid(String validityString) {
990         // Fixed end dates are not limited
991         if (ValidityDate.isValidIso8601Date(validityString)) {
992             //We have a valid date, let's just check that it's in the future as well. 
993             Date validityDate;
994             try {
995                 validityDate = ValidityDate.parseAsIso8601(validityString);
996             } catch (ParseException e) {
997                throw new IllegalStateException(validityString + " was an invalid date, but this should already have been checked.");
998             }
999             if (validityDate.before(new Date())) {
1000                 return ejbcawebbean.getText("INVALIDVALIDITY_PAST");
1001             }
1002         } else {
1003             //Only positive relative times allowed.
1004             try {
1005                 if (SimpleTime.parseMillies(validityString) <= 0) {
1006                     return ejbcawebbean.getText("INVALIDVALIDITYORCERTEND");
1007                 }
1008             } catch (NumberFormatException e) {
1009                 return ejbcawebbean.getText("INVALIDVALIDITYORCERTEND") + ": " + e.getMessage();
1010             }
1011         }
1012         return "";
1013     }
