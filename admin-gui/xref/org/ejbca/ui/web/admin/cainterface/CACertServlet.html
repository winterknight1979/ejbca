37  /**
38   * Servlet used to distribute CA certificates <br>
39   *
40   * cacert - returns ca certificate in PEM-format
41   * nscacert - returns ca certificate for Firefox
42   * iecacert - returns ca certificate for Internet Explorer
43   *
44   * cacert, nscacert and iecacert also takes optional parameter level=&lt;int 1,2,...&gt;, where the level is
45   * which ca certificate in a hierachy should be returned. 0=root (default), 1=sub to root etc.
46   *
47   * @version $Id: CACertServlet.java 34154 2019-12-23 13:38:17Z samuellb $
48   */
49  public class CACertServlet extends BaseAdminServlet {
50  
51  	private static final long serialVersionUID = 1L;
52  	private static final Logger log = Logger.getLogger(CACertServlet.class);
53  
54      private static final String COMMAND_PROPERTY_NAME = "cmd";
55      private static final String COMMAND_NSCACERT = "nscacert";
56      private static final String COMMAND_IECACERT = "iecacert";
57      private static final String COMMAND_CACERT = "cacert";
58      private static final String COMMAND_JKSTRUSTSTORE = "jkscert";
59  
60      private static final String LEVEL_PROPERTY = "level";
61      private static final String ISSUER_PROPERTY = "issuer";
62      private static final String JKSPASSWORD_PROPERTY = "password";
63  
64      @EJB
65      private SignSessionLocal signSession;
66  
67      @Override
68      public void init(ServletConfig config) throws ServletException {
69          super.init(config);
70      	if (signSession==null) {
71      		log.error("Local EJB injection failed.");
72      	}
73      }
74      
75      @Override
76      public void doPost(HttpServletRequest req, HttpServletResponse res) throws IOException, ServletException {
77          log.trace(">doPost()");
78          doGet(req, res);
79          log.trace("<doPost()");
80      }
81  
82      @Override
83      public void doGet(HttpServletRequest req,  HttpServletResponse res) throws java.io.IOException, ServletException {
84          log.trace(">doGet()");
85          try {
86              authenticateAdmin(req, res, AccessRulesConstants.REGULAR_VIEWCERTIFICATE);
87          } catch (AdminWebAuthenticationException authExc) {
88          	// TODO: localize this.
89          	log.info("Authentication failed", authExc);
90              res.sendError(HttpServletResponse.SC_FORBIDDEN, "Authentication failed");
91              return;
92          }
93          RequestHelper.setDefaultCharacterEncoding(req);
94  
95          // HttpServetRequets.getParameter URLDecodes the value for you
96          // No need to do it manually, that will cause problems with + characters
97          String issuerdn = req.getParameter(ISSUER_PROPERTY);
98          issuerdn = CertTools.stringToBCDNString(issuerdn);
99  
100         String command;
101         // Keep this for logging.
102         log.debug("Got request from "+req.getRemoteAddr());
103         command = req.getParameter(COMMAND_PROPERTY_NAME);
104         if (command == null) {
105             command = "";
106         }
107         if ((command.equalsIgnoreCase(COMMAND_NSCACERT) || command.equalsIgnoreCase(COMMAND_IECACERT) || command.equalsIgnoreCase(COMMAND_JKSTRUSTSTORE)
108         		|| command.equalsIgnoreCase(COMMAND_CACERT)) && issuerdn != null ) {
109             String lev = req.getParameter(LEVEL_PROPERTY);
110             int level = 0;
111             if (lev != null) {
112                 level = Integer.parseInt(lev);
113             }
114             // Root CA is level 0, next below root level 1 etc etc
115             try {
116                 Certificate[] chain = signSession.getCertificateChain(issuerdn.hashCode()).toArray(new Certificate[0]);
117                                                             
118                 // chain.length-1 is last cert in chain (root CA)
119                 if ( (chain.length-1-level) < 0 ) {
120                     PrintStream ps = new PrintStream(res.getOutputStream());
121                     ps.println("No CA certificate of level "+level+"exist.");
122                     log.error("No CA certificate of level "+level+"exist.");
123                     return;
124                 }
125                 Certificate cacert = chain[level];
126                 byte[] enccert = cacert.getEncoded();
127                 // Se if we can name the file as the CAs CN, if that does not exist try serialnumber, and if that does not exist, use the full O
128                 // and if that does not exist, use the fixed string CertificateAuthority. 
129                 String filename = RequestHelper.getFileNameFromCertNoEnding(cacert, "CertificateAuthority");
130                 // We must remove cache headers for IE
131                 ServletUtils.removeCacheHeaders(res);
132                 if (command.equalsIgnoreCase(COMMAND_NSCACERT)) {
133                     res.setContentType("application/x-x509-ca-cert");
134                     res.setContentLength(enccert.length);
135                     res.getOutputStream().write(enccert);
136                     log.debug("Sent CA cert to NS client, len="+enccert.length+".");
137                 } else if (command.equalsIgnoreCase(COMMAND_IECACERT)) {
138                 	String ending = ".cacert.crt";
139                 	if (cacert instanceof CardVerifiableCertificate) {
140                 		ending = ".cvcert";
141                 	}
142                     res.setHeader("Content-disposition", "attachment; filename=\"" + StringTools.stripFilename(filename + ending) + "\"");
143                     res.setContentType("application/octet-stream");
144                     res.setContentLength(enccert.length);
145                     res.getOutputStream().write(enccert);
146                     log.debug("Sent CA cert to IE client, len="+enccert.length+".");
147                 } else if (command.equalsIgnoreCase(COMMAND_CACERT)) {
148                     String out = CertTools.getPemFromCertificate(cacert);
149                     res.setHeader("Content-disposition", "attachment; filename=\"" + StringTools.stripFilename(filename + ".cacert.pem") + "\"");
150                     res.setContentType("application/octet-stream");
151                     res.setContentLength(out.length());
152                     res.getOutputStream().write(out.getBytes());
153                     log.debug("Sent CA cert to client, len="+out.length()+".");
154                 } else if (command.equalsIgnoreCase(COMMAND_JKSTRUSTSTORE)) {
155                     String jksPassword = req.getParameter(JKSPASSWORD_PROPERTY).trim();
156                     int passwordRequiredLength = 6;
157                     if ( jksPassword != null && jksPassword.length() >= passwordRequiredLength ) {
158                     	KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
159                     	ks.load(null, jksPassword.toCharArray());
160                     	ks.setCertificateEntry(filename, cacert);
161                         res.setHeader("Content-disposition", "attachment; filename=\"" + StringTools.stripFilename(filename + ".cacert.jks") + "\"");
162                         res.setContentType("application/octet-stream");
163                     	ks.store(res.getOutputStream(), jksPassword.toCharArray());
164                     } else {
165                         res.setContentType("text/plain");
166                         res.getOutputStream().println(COMMAND_JKSTRUSTSTORE + " requires " + JKSPASSWORD_PROPERTY +
167                         		" with a minimum of " + passwordRequiredLength+ " chars to be set");
168                         return;
169                     }
170                 } else {
171                     res.setContentType("text/plain");
172                     res.getOutputStream().println("Commands="+COMMAND_NSCACERT+" || "+COMMAND_IECACERT+" || "+COMMAND_CACERT+" || "+COMMAND_JKSTRUSTSTORE);
173                     return;
174                 }
175             } catch (Exception e) {
176                 log.error("Error getting CA certificates: ", e);
177                 res.sendError(HttpServletResponse.SC_NOT_FOUND, "Error getting CA certificates.");
178                 return;
179             }
180         }
181         else {
182             res.sendError(HttpServletResponse.SC_BAD_REQUEST, "Bad Request format");
183             return;
184         }
185     } // doGet
