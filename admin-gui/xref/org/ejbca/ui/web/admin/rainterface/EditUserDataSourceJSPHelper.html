128     public void initialize(HttpServletRequest request, EjbcaWebBean ejbcawebbean,
129             RAInterfaceBean rabean) throws  Exception{
130

132             initialized = true;

140     public String parseRequest(HttpServletRequest request) throws AuthorizationDeniedException{
141         String includefile = PAGE_USERDATASOURCES;
142         String userdatasource = null;        
143         String action = null;
144 
145         try {
146             RequestHelper.setDefaultCharacterEncoding(request);
147         } catch (UnsupportedEncodingException e1) {
148             // itgnore
149         }
150         action = request.getParameter(ACTION);
151         if( action != null){
152             if( action.equals(ACTION_EDIT_USERDATASOURCES)){
153                 if( request.getParameter(BUTTON_EDIT_USERDATASOURCE) != null){
154                     userdatasource = request.getParameter(SELECT_USERDATASOURCE);
155                     if(userdatasource != null){
156                         if(!userdatasource.trim().equals("")){
157                             includefile=PAGE_USERDATASOURCE;
158                             this.userdatasourcename = userdatasource;
159                             this.userdatasourcedata = userdatasourcesession.getUserDataSource(admin,userdatasourcename);
160                         }
161                         else{
162                             userdatasource= null;
163                         }
164                     }
165                     if(userdatasource == null){
166                         includefile=PAGE_USERDATASOURCES;
167                     }
168                 }
169                 if( request.getParameter(BUTTON_DELETE_USERDATASOURCE) != null) {
170                     userdatasource = request.getParameter(SELECT_USERDATASOURCE);
171                     if(userdatasource != null){
172                         if(!userdatasource.trim().equals("")){
173                             userdatasourcedeletefailed = !userdatasourcesession.removeUserDataSource(admin,userdatasource);
174                         }
175                     }
176                     includefile=PAGE_USERDATASOURCES;
177                 }
178                 if( request.getParameter(BUTTON_RENAME_USERDATASOURCE) != null){
179                     // Rename selected userdatasource and display profilespage.
180                     String newuserdatasourcename = request.getParameter(TEXTFIELD_USERDATASOURCENAME);
181                     String olduserdatasourcename = request.getParameter(SELECT_USERDATASOURCE);
182                     if(olduserdatasourcename != null && newuserdatasourcename != null){
183                         if(!newuserdatasourcename.trim().equals("") && !olduserdatasourcename.trim().equals("")){
184                             try{
185                             	userdatasourcesession.renameUserDataSource(admin,olduserdatasourcename.trim(),newuserdatasourcename.trim());
186                             }catch( UserDataSourceExistsException e){
187                                 userdatasourceexists=true;
188                             }
189                         }
190                     }
191                     includefile=PAGE_USERDATASOURCES;
192                 }
193                 if( request.getParameter(BUTTON_ADD_USERDATASOURCE) != null){
194                     userdatasource = request.getParameter(TEXTFIELD_USERDATASOURCENAME);
195                     if(userdatasource != null){
196                         if(!userdatasource.trim().equals("")){
197                             try{
198                             	userdatasourcesession.addUserDataSource(admin,userdatasource.trim(), new CustomUserDataSourceContainer());
199                             }catch( UserDataSourceExistsException e){
200                                 userdatasourceexists=true;
201                             }
202                         }
203                     }
204                     includefile=PAGE_USERDATASOURCES;
205                 }
206                 if( request.getParameter(BUTTON_CLONE_USERDATASOURCE) != null){
207                     String newuserdatasourcename = request.getParameter(TEXTFIELD_USERDATASOURCENAME);
208                     String olduserdatasourcename = request.getParameter(SELECT_USERDATASOURCE);
209                     if(olduserdatasourcename != null && newuserdatasourcename != null){
210                     	if(!newuserdatasourcename.trim().equals("") && !olduserdatasourcename.trim().equals("")){
211                     		try{
212                     			userdatasourcesession.cloneUserDataSource(admin,olduserdatasourcename.trim(),newuserdatasourcename.trim());
213                     		}catch( UserDataSourceExistsException e){
214                     			userdatasourceexists=true;
215                     		}
216                     	}
217                     }
218                     includefile=PAGE_USERDATASOURCES;
219                 }
220             }
221             if( action.equals(ACTION_EDIT_USERDATASOURCE)){
222                 // Display edit access rules page.
223                 userdatasource = request.getParameter(HIDDEN_USERDATASOURCENAME);
224                 this.userdatasourcename = userdatasource;
225                 if(userdatasource != null){
226                     if(!userdatasource.trim().equals("")){
227                         if(request.getParameter(BUTTON_SAVE) != null ||
228                                 request.getParameter(BUTTON_TESTCONNECTION) != null){
229 
230                             if(userdatasourcedata == null){
231                                 int tokentype = Integer.parseInt(request.getParameter(HIDDEN_USERDATASOURCETYPE));
232                                 if(tokentype == CustomUserDataSourceContainer.TYPE_CUSTOMUSERDATASOURCECONTAINER) {
233                                     userdatasourcedata = new CustomUserDataSourceContainer();
234                                 }
235                             }
236                             // Save changes.
237 
238                             // General settings
239                             String value = request.getParameter(TEXTAREA_DESCRIPTION);
240                             if(value != null){
241                                 value = value.trim();
242                                 userdatasourcedata.setDescription(value);
243                             }
244                             
245                             String[] values = request.getParameterValues(SELECT_MODIFYABLEFIELDS);
246                             if(values != null){
247                                 Set<Integer> modifyablefields = new HashSet<>();
248                                 for(int i=0;i< values.length;i++){
249                                 	modifyablefields.add(Integer.valueOf(values[i]));
250                                 }
251 
252                                 userdatasourcedata.setModifiableFields(modifyablefields);
253                             }else{
254                             	userdatasourcedata.setModifiableFields(new HashSet<Integer>());
255                             }
256 
257                             values = request.getParameterValues(SELECT_APPLICABLECAS);
258                             if(values != null){
259                                 List<Integer> useCAs = new ArrayList<>();
260                                 for(int i=0;i< values.length;i++){
261                                 	Integer caid = Integer.valueOf(values[i]);
262                                 	if(caid.intValue() == BaseUserDataSource.ANYCA){
263                                 		useCAs = new ArrayList<>();
264                                 		useCAs.add(caid);
265                                 		break;
266                                 	}
267                                 	useCAs.add(Integer.valueOf(values[i]));
268                                 }
269 
270                                 userdatasourcedata.setApplicableCAs(useCAs);
271                             }else{
272                             	userdatasourcedata.setApplicableCAs(new ArrayList<Integer>());
273                             }
274                             
275                            
276 
277 
278                             if(userdatasourcedata instanceof CustomUserDataSourceContainer){
279                                 value = request.getParameter(TEXTFIELD_CUSTOMCLASSPATH);
280                                 if(value != null){
281                                     value = value.trim();
282                                     ((CustomUserDataSourceContainer) userdatasourcedata).setClassPath(value);
283                                 }
284                                 value = request.getParameter(TEXTAREA_CUSTOMPROPERTIES);
285                                 if(value != null){
286                                     value = value.trim();
287                                     ((CustomUserDataSourceContainer) userdatasourcedata).setPropertyData(value);
288                                 }
289                             }
290 
291                             if(request.getParameter(BUTTON_SAVE) != null){
292                                 userdatasourcesession.changeUserDataSource(admin,userdatasource,userdatasourcedata);
293                                 includefile=PAGE_USERDATASOURCES;
294                             }
295                             if(request.getParameter(BUTTON_TESTCONNECTION)!= null){
296                                 connectionmessage = true;
297                                 userdatasourcesession.changeUserDataSource(admin, userdatasource,userdatasourcedata);
298                                 try{
299                                 	int userdatasourceid = userdatasourcesession.getUserDataSourceId(admin,userdatasource);
300                                 	userdatasourcesession.testConnection(admin,userdatasourceid);
301                                     connectionsuccessful = true;
302                                 }catch(UserDataSourceConnectionException pce){
303                                     connectionerrormessage = pce.getMessage();
304                                 }
305                                 includefile=PAGE_USERDATASOURCE;
306                             }
307 
308                         }
309                         if(request.getParameter(BUTTON_CANCEL) != null){
310                             // Don't save changes.
311                             includefile=PAGE_USERDATASOURCES;
312                         }
313 
314                     }
315                 }
316             }
317 
318             if( action.equals(ACTION_CHANGE_USERDATASOURCETYPE)){
319                 this.userdatasourcename = request.getParameter(HIDDEN_USERDATASOURCENAME);
320                 String value = request.getParameter(SELECT_USERDATASOURCETYPE);
321                 if(value!=null){
322                     int profiletype = Integer.parseInt(value);
323                     switch(profiletype){
324                     case CustomUserDataSourceContainer.TYPE_CUSTOMUSERDATASOURCECONTAINER :
325                         userdatasourcedata = new CustomUserDataSourceContainer();
326                         break;                    
327                     }
328                 }
329 
330                 includefile=PAGE_USERDATASOURCE;
331             }
332         }
333 
334         return includefile;
335     }

337     public int getUserDataSourceType(){
338         int retval = CustomUserDataSourceContainer.TYPE_CUSTOMUSERDATASOURCECONTAINER;
339 
340         if(userdatasourcedata instanceof CustomUserDataSourceContainer) {
341             retval = CustomUserDataSourceContainer.TYPE_CUSTOMUSERDATASOURCECONTAINER;
342         }
343 
344         return retval;
345     }

348     public TreeMap<String, Integer> getAuthorizedUserDataSourceNames(){
349     	TreeMap<String, Integer> retval = new TreeMap<>();
350     	
351     	Collection<Integer> authorizedsources = userdatasourcesession.getAuthorizedUserDataSourceIds(admin,false);
352         for(Integer id : authorizedsources) {
353     		retval.put(userdatasourcesession.getUserDataSourceName(admin,id.intValue()),id);
354     	}  	
355     	
356     	return retval;
357     }
