27  /**
28   * A class representing a set of users
29   * 
30   * @version $Id: UsersView.java 28844 2018-05-04 08:31:02Z samuellb $
31   */
32  public class UsersView implements Serializable {
33          
34      private static final long serialVersionUID = -7382135359016557800L;
35      /** Creates a new instance of UsersView */
36      public UsersView() {
37        users = new ArrayList<>();
38        sortby = new SortBy();
39      }
40      
41      public UsersView(EndEntityInformation importuser, HashMap<Integer, String> caidtonamemap){
42        users = new ArrayList<>();
43        sortby = new SortBy();        
44        users.add(new UserView(importuser, caidtonamemap)); 
45        
46        Collections.sort(users); 
47      }
48      
49      public UsersView(Collection<EndEntityInformation> importusers, HashMap<Integer, String>caidtonamemap){ 
50        users = new ArrayList<>();
51        sortby = new SortBy();
52        
53        setUsers(importusers, caidtonamemap);
54      }
55      // Public methods.
56      
57      public void sortBy(int sortby, int sortorder) {
58        this.sortby.setSortBy(sortby);
59        this.sortby.setSortOrder(sortorder);
60        
61        Collections.sort(users);
62      }
63      
64      public UserView[] getUsers(int index, int size) {       
65        int endindex;  
66        UserView[] returnval;
67     
68        if(index > users.size()) {
69      	  index = users.size()-1;
70        }
71        if(index < 0) {
72      	  index =0;
73        }
74        
75        // The below is used in order to return all the values in one page
76        // JasperReports has its own multiple page setings (right now it will print all pages inside a single web page, one after the other)
77        // i think this functions were first used in very specific places, where the user asks directly for a single page at once depending on the number of
78        // results in one page.  If this number is -1 then all results will be on one single page
79        if (size == -1)  {
80      	  endindex = users.size();
81        } else {
82      	  endindex = index + size;
83      	  if(endindex > users.size()) {
84      		  endindex = users.size();
85      	  }
86        }
87        
88        returnval = new UserView[endindex-index];  
89        
90        int end = endindex - index;
91        for(int i = 0; i < end; i++){
92          returnval[i] = users.get(index+i);   
93        }
94        
95        return returnval;
96      }
97      
98      public void setUsers(UserView[] users) {
99        this.users.clear();
100       if(users !=null && users.length > 0){       
101         for(int i=0; i < users.length; i++){
102           users[i].setSortBy(this.sortby);
103           this.users.add(users[i]);
104         }
105       }
106       Collections.sort(this.users);
107     }
108     
109     public void setUsers(EndEntityInformation[] users, Map<Integer, String> caidtonamemap) {
110       UserView user;  
111       this.users.clear();
112       if(users !=null && users.length > 0){ 
113         for(int i=0; i< users.length; i++){
114           user = new UserView(users[i], caidtonamemap); 
115           user.setSortBy(this.sortby);
116           this.users.add(user);
117         }
118         Collections.sort(this.users);
119       }
120     }
121 
122     public void setUsers(Collection<EndEntityInformation> importusers, Map<Integer, String> caidtonamemap) { 
123         
124       UserView user;  
125       Iterator<EndEntityInformation> i;  
126       this.users.clear();
127       if(importusers!=null && importusers.size() > 0){
128         i=importusers.iterator();
129         while(i.hasNext()){
130         	EndEntityInformation nextuser = i.next();  
131           user = new UserView(nextuser, caidtonamemap); 
132           user.setSortBy(this.sortby);
133           users.add(user);
134         }
135         Collections.sort(users);
136       }
137     }
138 
139     public void addUser(UserView user) {
140        user.setSortBy(this.sortby);        
141        users.add(user);
142     }
143     
144     public int size(){
145       return users.size();   
146     }
147     
148     public void clear(){
149       this.users.clear();   
150     }
151     // Private fields
152     private List<UserView> users;
153     private SortBy sortby;
154     
155 }
