103     /**
104      * Method that initialized the bean.
105      * @param ejbcawebbean bean
106      * @param rabean bean
107      * @param cabean bean
108      * @throws Exception fail
109      */
110     public void initialize(EjbcaWebBean ejbcawebbean,  
111     		               RAInterfaceBean rabean, CAInterfaceBean cabean) throws Exception {
112 
113       if(!initialized){
114 
115         this.rabean = rabean;
116         this.ejbcawebbean = ejbcawebbean;
117         this.cabean = cabean;
118         initialized = true;
119         
120         if (ejbcawebbean.getGlobalConfiguration().getIssueHardwareTokens()){
121             final TreeMap<String, Integer> hardtokenprofiles = ejbcawebbean.getHardTokenProfiles();
122             tokentexts = new String[RAInterfaceBean.tokentexts.length + hardtokenprofiles.keySet().size()];
123             tokenids   = new int[tokentexts.length];
124             for (int i=0; i < RAInterfaceBean.tokentexts.length; i++){
125               tokentexts[i]= RAInterfaceBean.tokentexts[i];
126               tokenids[i] = RAInterfaceBean.tokenids[i];
127             }
128             int index=0;
129             for (String name : hardtokenprofiles.keySet()) {
130               tokentexts[index+RAInterfaceBean.tokentexts.length] = name;
131               tokenids[index+RAInterfaceBean.tokentexts.length] = hardtokenprofiles.get(name).intValue();
132               index++;
133             }
134          }
135 		
136       }
137     }

139     public void parseRequest(HttpServletRequest request) throws AuthorizationDeniedException, Exception{
140     	  nouserparameter=true;
141     	  notauthorized = false;
142     	  profilenotfound = true;
143     	  
144           RequestHelper.setDefaultCharacterEncoding(request);
145     	  String action = request.getParameter(ACTION);
146     	  if( action == null  && request.getParameter(TIMESTAMP_PARAMETER) != null &&  request.getParameter(USER_PARAMETER) != null){    		  
147     		  username = java.net.URLDecoder.decode(request.getParameter(USER_PARAMETER),"UTF-8");
148     		  Date timestamp = new Date(Long.parseLong(request.getParameter(TIMESTAMP_PARAMETER)));
149     		      		      		  
150     	      notauthorized = !getUserDatas(username);
151     	      currentuserindex = this.getTimeStampIndex(timestamp);
152     	      if ( userdatas == null || userdatas.length < 1 ) {
153     	    	  // Make sure possibly cached value is removed
154     	          userdata = null;
155     			  throw new ServletException("Could not find any history for this user.");
156     	      }
157 			  userdata = userdatas[currentuserindex];
158     	      
159     		  nouserparameter = false;
160     		  if(userdata!=null) {
161     			  profile = rabean.getEndEntityProfile(userdata.getEndEntityProfileId());
162     		  }
163     	  }else{ 
164     		  if(action  == null && request.getParameter(USER_PARAMETER) != null){    		  
165     			  username = java.net.URLDecoder.decode(request.getParameter(USER_PARAMETER),"UTF-8");    			
166     			  notauthorized = !getUserDatas(username);
167     			  nouserparameter = false;
168     			  if ( (userdatas != null) && (userdatas.length > 0) ) {
169         			  userdata = userdatas[0];
170         			  currentuserindex = 0;
171         			  if(userdata!=null) {
172         				  profile = rabean.getEndEntityProfile(userdata.getEndEntityProfileId());
173         			  }
174     			  } else {
175     				  // Make sure possibly cached value is removed
176     			      userdata = null;
177     			  }
178     		  }else{
179     			  if( action != null && request.getParameter(USER_PARAMETER)!=null){
180         			  username = java.net.URLDecoder.decode(request.getParameter(USER_PARAMETER),"UTF-8");
181     				  if(request.getParameter(BUTTON_VIEW_NEWER)!=null){
182     					  if(currentuserindex>0){
183     						  currentuserindex--;
184     					  }	  
185     				  }
186     				  if(request.getParameter(BUTTON_VIEW_OLDER)!=null){
187     					  if(currentuserindex +1<userdatas.length){
188     						  currentuserindex++;
189     					  }	  
190     				  }
191     				  
192     				  notauthorized  = !getUserDatas(username);
193     				  userdata = userdatas[currentuserindex];
194     				  
195         			  nouserparameter = false;
196         			  if(userdata!=null) {
197         				  profile = rabean.getEndEntityProfile(userdata.getEndEntityProfileId());
198         			  }
199     			  }
200     		  }
201     	  }
202     	  
203     	  if(profile!=null){
204     		  profilenotfound=false;
205     	  }
206     }

209     /* returns false if the admin isn't authorized to view user
210      * Sets the available userdatas of current and previous values
211      */
212     
213     private boolean getUserDatas(String username) throws Exception{
214       boolean authorized = false;	
215     
216       try{
217     	  if(currentusername == null || !currentusername.equals(username)){
218     		  // fetch userdata and certreqdatas and order them by timestamp, newest first.
219     		  int currentexists = 0;
220     		  UserView currentuser = rabean.findUser(username);
221     		  if(currentuser != null){
222     			  currentexists = 1;  
223     		  }
224     		  List<CertReqHistory> hist = cabean.getCertReqUserDatas(username);
225     		  
226     		  userdatas = new UserView[hist.size() +currentexists];
227     		  
228     		  if(currentuser != null){
229     		    userdatas[0] = currentuser;    		  
230     		  }
231     		  for(int i=0; i< hist.size();i++){
232     			  CertReqHistory next = hist.get(i);
233     			  userdatas[i+currentexists] = new UserView(next.getEndEntityInformation(), ejbcawebbean.getCAIdToNameMap());
234     		  }
235 			  
236     	  }
237     	  authorized=true;
238 	  } catch(AuthorizationDeniedException e){ }            
239       return authorized;
240     }

242     /**
243      * Returns an Index to the user that related to a certain timestamp.
244      * 
245      * @param timestamp parameter sent from view log page
246      * @return index in user datas that should be shown.
247      */
248     private int getTimeStampIndex(Date timestamp){
249     	int i;
250     	
251     	for(i=0;i< userdatas.length;i++){
252     			if(timestamp.after(userdatas[i].getTimeModified())||
253     					timestamp.equals(userdatas[i].getTimeModified())){
254     				break;	
255     			}
256     	}
257     	
258     	return i;
259     }
