14  package org.ejbca.ui.web.admin;
15  
16  import javax.faces.application.FacesMessage;
17  import javax.faces.component.UIComponent;
18  import javax.faces.context.FacesContext;
19  import javax.faces.validator.Validator;
20  import javax.faces.validator.ValidatorException;
21  
22  import org.apache.log4j.Logger;
23  import org.cesecore.util.SimpleTime;
24  import org.cesecore.util.StringTools;
25  import org.cesecore.util.TimeUnitFormat;
26  import org.ejbca.ui.web.admin.configuration.EjbcaJSFHelper;
27  
28  /** JSF validator to check that the input does not contain any invalid characters and is a valid time unit format (i.e. '3y 6mo -10d 6h +30m 30s'). 
29   * 
30   * @version $Id: SimpleTimeValidator.java 25545 2017-03-20 16:50:07Z mikekushner $
31   */
32  public class SimpleTimeValidator implements Validator {
33  
34      private static final Logger log = Logger.getLogger(SimpleTimeValidator.class);
35  
36      @Override
37      public void validate(FacesContext facesContext, UIComponent component, Object object) throws ValidatorException {
38          final String value = (String) object;
39          final TimeUnitFormat format = SimpleTime.getTimeUnitFormatOrThrow( (String) component.getAttributes().get("precision"));
40          long minimumValue = Long.MIN_VALUE;
41          if (null != component.getAttributes().get("minimumValue")) {
42              minimumValue = Long.parseLong((String) component.getAttributes().get("minimumValue")); 
43          }
44          long maximumValue = Long.MAX_VALUE;
45          if (null != component.getAttributes().get("maximumValue")) {
46              maximumValue = Long.parseLong((String) component.getAttributes().get("maximumValue")); 
47          }
48          boolean failed = true;
49          if (StringTools.hasSqlStripChars(value).isEmpty()) {
50              try {
51                  final long millis = format.parseMillis(value);
52                  if (minimumValue <= millis && millis <= maximumValue) {
53                      failed = false;
54                  }
55              } catch (NumberFormatException e) {
56                  // NOOP
57              }
58          }
59          if (failed) {
60              if (log.isDebugEnabled()) {
61                  log.debug("Validating component " + component.getClientId(facesContext) + " with value \"" + value + "\" failed.");
62              }
63              final String message = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText("INVALIDTIMEFORMAT");
64              throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, message, null));
65          } else {
66              if (log.isDebugEnabled()) {
67                  log.debug("Validating component " + component.getClientId(facesContext) + " with value \"" + value + "\"");
68              }
69          }
70      }
71  }
