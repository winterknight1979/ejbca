106     /**
107      * Get the appropriate Access match type based on the access match value.
108      * @param accessMatchValue Value
109      * @return Equal Case sensitive unless access match value forces something else.
110      */
111     private AccessMatchType getAccessMatchType(AccessMatchValue accessMatchValue) {
112         if (accessMatchValue.equals(X500PrincipalAccessMatchValue.WITH_SERIALNUMBER)
113                 || accessMatchValue.equals(X500PrincipalAccessMatchValue.WITH_COUNTRY)
114                 || accessMatchValue.equals(X500PrincipalAccessMatchValue.WITH_DNEMAILADDRESS)
115                 || accessMatchValue.equals(X500PrincipalAccessMatchValue.WITH_STATEORPROVINCE)
116                 || accessMatchValue.equals(X500PrincipalAccessMatchValue.WITH_RFC822NAME)){
117             return AccessMatchType.TYPE_EQUALCASEINS;
118         }
119         return AccessMatchType.TYPE_EQUALCASE;
120     }

122     /** @return true when admin is authorized to edit members of this role */
123     public boolean isAuthorizedToEditRole() {
124         try {
125             if (authorizationSession.isAuthorizedNoLogging(getAdmin(), StandardRules.EDITROLES.resource()) && getRole()!=null) {
126                 roleSession.assertAuthorizedToRoleMembers(getAdmin(), getRole().getRoleId(), true);
127                 return true;
128             }
129         } catch (AuthorizationDeniedException e) {
130             // NOPMD: Ignore.. would be nicer with a API call returning a boolean
131         }
132         return false;
133     }

135     /** @return an authorized existing role based on the roleId HTTP param or null if no such role was found. */
136     public Role getRole() {
137         if (role==null && NumberUtils.isNumber(roleIdParam)) {
138             try {
139                 role = roleSession.getRole(getAdmin(), Integer.parseInt(roleIdParam));
140                 if (role==null && log.isDebugEnabled()) {
141                     log.debug("Admin '" + getAdmin() + "' failed to access non-existing role.");
142                 }
143             } catch (NumberFormatException | AuthorizationDeniedException e) {
144                 if (log.isDebugEnabled()) {
145                     log.debug("Admin '" + getAdmin() + "' failed to access a role: " + e.getMessage());
146                 }
147             }
148         }
149         return role;
150     }

152     /** @return a ListDataModel of all RoleMembers in this Role (sorted) */
153     public ListDataModel<RoleMember> getRoleMembers() {
154         if (roleMembers==null) {
155             final List<RoleMember> roleMembers = roleMemberDataSession.findRoleMemberByRoleId(role.getRoleId());
156             Collections.sort(roleMembers, new Comparator<RoleMember>() {
157                 @Override
158                 public int compare(final RoleMember roleMember1, final RoleMember roleMember2) {
159                     final String matchWithItemString1 = getMatchWithItemString(roleMember1);
160                     final String matchWithItemString2 = getMatchWithItemString(roleMember2);
161                     final int compareA = matchWithItemString1.compareTo(matchWithItemString2);
162                     if (compareA!=0) {
163                         return compareA;
164                     }
165                     final String tokenIssuerIdString1 = getTokenIssuerIdString(roleMember1);
166                     final String tokenIssuerIdString2 = getTokenIssuerIdString(roleMember2);
167                     final int compareB = tokenIssuerIdString1.compareTo(tokenIssuerIdString2);
168                     if (compareB!=0) {
169                         return compareB;
170                     }
171                     return roleMember1.getTokenMatchValue().compareTo(roleMember2.getTokenMatchValue());
172                 }
173             });
174             this.roleMembers = new ListDataModel<>(roleMembers);
175         }
176         return roleMembers;
177     }

179     /** @return a viewable list of 'match with'-texts */
180     public List<SelectItem> getMatchWithItems() {
181         if (matchWithItems == null) {
182             matchWithItems = new ArrayList<>();
183             final List<String> tokenTypes = new ArrayList<>(AccessMatchValueReverseLookupRegistry.INSTANCE.getAllTokenTypes());
184             Collections.sort(tokenTypes);
185             for (final String tokenType : tokenTypes) {
186                 final AuthenticationTokenMetaData authenticationTokenMetaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(tokenType);
187                 if (authenticationTokenMetaData.isUserConfigurable()) {
188                     for (final AccessMatchValue accessMatchValue : authenticationTokenMetaData.getAccessMatchValues()) {
189                         // Special exclusion of this rather useless match value that will never match anything
190                         if (!X500PrincipalAccessMatchValue.NONE.equals(accessMatchValue)) {
191                             matchWithItems.add(new SelectItem(tokenType + ":" + accessMatchValue.getNumericValue(), 
192                                     getEjbcaWebBean().getText(tokenType) + ": " + getEjbcaWebBean().getText(accessMatchValue.name())));
193                         }
194                     }
195                 }
196             }
197         }
198         return matchWithItems;
199     }

207     /** @param roleMember member
208      * @return a human readable version of the tokenType and tokenMatchKey values */
209     public String getMatchWithItemString(final RoleMember roleMember) {
210         final AuthenticationTokenMetaData authenticationTokenMetaData = AccessMatchValueReverseLookupRegistry.INSTANCE.getMetaData(roleMember.getTokenType());
211         final String tokenTypeString = getEjbcaWebBean().getText(authenticationTokenMetaData.getTokenType());
212         final String tokenMatchKeyString = getEjbcaWebBean().getText(authenticationTokenMetaData.getAccessMatchValueIdMap().get(roleMember.getTokenMatchKey()).name());
213         return tokenTypeString + ":" + tokenMatchKeyString;
214     }

238     /** @param roleMember Role
239      * @return a human readable version of the RoleMember's tokenIsserId (CA name) */
240     public String getTokenIssuerIdString(final RoleMember roleMember) {
241         final int tokenIssuerId = roleMember.getTokenIssuerId();
242         if (tokenIssuerId==RoleMember.NO_ISSUER) {
243             return "-";
244         }
245         if (getAccessMatchValue(roleMember.getTokenType(), roleMember.getTokenMatchKey()).isIssuedByCa()) {
246             final String caName = getCaIdToNameMap().get(tokenIssuerId);
247             if (caName==null) {
248                 return super.getEjbcaWebBean().getText("UNKNOWNCAID") + " " + tokenIssuerId;
249             } else {
250                 return caName;
251             }
252         } else {
253             return String.valueOf(tokenIssuerId);
254         }
255     }

311     /** Invoked by the admin when adding a new RoleMember. */
312     public void actionAddRoleMember() {
313         try {
314             final String tokenType = getSelectedTokenType();
315             final int tokenMatchKey = getSelectedTokenMatchKey();
316             final AccessMatchValue accessMatchValue = getAccessMatchValue(tokenType, tokenMatchKey);
317             final AccessMatchType accessMatchType;
318             if (accessMatchValue.getAvailableAccessMatchTypes().isEmpty()) {
319                 accessMatchType = AccessMatchType.TYPE_UNUSED;
320             } else {
321                 accessMatchType = getAccessMatchType(accessMatchValue);
322             }
323 
324             if (AccessMatchType.TYPE_UNUSED.equals(accessMatchType)) {
325                 tokenMatchValue = "";
326             } else {
327                 // Validate that a required tokenMatchValue has been entered
328                 if (StringUtils.isEmpty(tokenMatchValue)) {
329                     super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "MATCHVALUEREQUIRED");
330                     return;
331                 }
332             }
333             // Validate that the tokenMatchValue contains no illegal characters
334             // (Check this here instead of using an f:validator validatorId="legalCharsValidator" since we might need to do PRG later)
335             Set<String> invalidCharacters = StringTools.hasSqlStripChars(tokenMatchValue);
336             if (!invalidCharacters.isEmpty()) {
337                 StringBuilder sb = new StringBuilder("");
338                 for (String error : invalidCharacters) {
339                     sb.append(", " + error);
340                 }
341                 super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "INVALIDCHARS", sb.substring(2) );
342                 return;
343             }
344             // If the tokenMatchValue should be a hex number, validate that it is
345             if (X509CertificateAuthenticationTokenMetaData.TOKEN_TYPE.equals(tokenType) &&
346                     X500PrincipalAccessMatchValue.WITH_SERIALNUMBER.getNumericValue()==tokenMatchKey) {
347                 try {
348                     new BigInteger(tokenMatchValue, 16);
349                 } catch (NumberFormatException e) {
350                     super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "HEXREQUIRED");
351                     return;
352                 }
353             }
354             final int tokenIssuerId;
355             if (accessMatchValue.isIssuedByCa()) {
356                 tokenIssuerId = this.tokenIssuerId;
357             } else {
358                 tokenIssuerId = RoleMember.NO_ISSUER;
359             }
360             try {
361                 roleMemberSession.persist(getAdmin(), new RoleMember(tokenType, tokenIssuerId, tokenMatchKey,
362                         accessMatchType.getNumericValue(), tokenMatchValue, role.getRoleId(), description));
363             } catch (AuthorizationDeniedException e) {
364                 super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "AUTHORIZATIONDENIED");
365             }
366             // Only reset this one, since admin might want to add additional matches of the same type
367             tokenMatchValue = "";
368             // Trigger a reload of the RoleMember list (if this is an AJAX requests and no new viewscope will be created)
369             roleMembers = null;
370         } finally {
371             nonAjaxPostRedirectGet();
372         }
373     }
