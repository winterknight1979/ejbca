114     /** Advanced mode access rule tri-state */
115     private static enum AccessRuleState {
116         UNDEFINED, ALLOW, DENY;
117 
118         static AccessRuleState toAccessRuleState(final Boolean state) {
119             return state==null ? UNDEFINED : state ? ALLOW : DENY;
120         }

123     /** Advanced mode access rule representation */
124     public static class AccessRuleItem implements Comparable<AccessRuleItem>, Serializable {
125         private static final long serialVersionUID = 1L;
126 
127         private final String category;
128         private final String resource;
129         private final String resourceName;
130         private final String resourceMain;
131         private final String resourceSub;
132         private AccessRuleState state = AccessRuleState.UNDEFINED;
133 
134         public AccessRuleItem(final String category, final String resource, final String resourceName) {
135             this.category = category;
136             this.resource = AccessRulesHelper.normalizeResource(resource);
137             this.resourceName = resourceName==null ? this.resource : AccessRulesHelper.normalizeResource(resourceName);
138             final String[] resourceSplit = this.resourceName.split("/");
139             if (resourceSplit.length==0) {
140                 this.resourceSub = "/";
141                 this.resourceMain = "";
142             } else {
143                 this.resourceSub = resourceSplit[resourceSplit.length-1] + "/";
144                 String resourceMain = "/";
145                 for (int i=1; i<resourceSplit.length-1; i++) {
146                     resourceMain += resourceSplit[i] + "/";
147                 }
148                 this.resourceMain = resourceMain;
149             }
150         }
151 
152         /** @return the category this resource belongs to */
153         public String getCategory() { return category; }
154         /** @return the normalized resource with Id kept intact for objects */
155         public String getResource() { return resource; }
156         /** @return the normalized resource with Id replaced by names for objects */
157         public String getResourceName() { return resourceName; }
158         /** @return the first part(s) of the resourceName e.g.  '/mainrule/subrule/' from '/mainrule/subrule/subsubrule/' */
159         public String getResourceMain() { return resourceMain; }
160         /** @return the last part of the resourceName e.g. 'subsubrule/' from '/mainrule/subrule/subsubrule/' */
161         public String getResourceSub() { return resourceSub; }
162         /** @return one of the {@link AccessRuleState} enum names representing the current state of this rule */
163         public String getState() { return state.name(); }
164         /** Set one of the {@link AccessRuleState} enum names representing the current state of this rule 
165          * @param state state */
166         public void setState(String state) { this.state = AccessRuleState.valueOf(state); }
167         /** @return one of the {@link AccessRuleState} enum representing the current state of this rule */
168         private AccessRuleState getStateEnum() { return state; }
169         /** @return true if the resource in this istance is '/' */
170         public boolean isRootResource() { return StandardRules.ROLE_ROOT.resource().equals(resource); }
171 
172         @Override
173         public int compareTo(final AccessRuleItem other) {
174             // Sort by resource name (with IDs replaced by names)
175             return getResourceName().compareToIgnoreCase(other.getResourceName());
176         }
177     }

352     /** @return an authorized existing role based on the roleId HTTP param or null if no such role was found. */
353     public Role getRole() {
354         if (role==null && NumberUtils.isNumber(roleIdParam)) {
355             try {
356                 role = roleSession.getRole(getAdmin(), Integer.parseInt(roleIdParam));
357                 if (role==null && log.isDebugEnabled()) {
358                     log.debug("Admin '" + getAdmin() + "' failed to access non-existing role.");
359                 }
360             } catch (NumberFormatException | AuthorizationDeniedException e) {
361                 if (log.isDebugEnabled()) {
362                     log.debug("Admin '" + getAdmin() + "' failed to access a role: " + e.getMessage());
363                 }
364             }
365         }
366         return role;
367     }

369     /** Recalculate selection based on current state of template selection or loaded role's access rules */
370     private void reinitSelection() {
371         // Reset available lists that might be rendered differently depending on selected template
372         availableResourcesCa = null;
373         availableResourcesEe = null;
374         availableResourcesEep = null;
375         availableResourcesKeyValidators = null;
376         availableResourcesIkb = null;
377         availableResourcesOther = null;
378         // Calculate available templates and the current best match
379         getAvailableAccessRulesTemplates();
380         // Select access rules that are allowed by the role
381         final LinkedHashMap<String, Boolean> accessRules = getRole().getAccessRules();
382         // Find CA access resources allowed by this role
383         setResourcesCaSelected(getSelectedRulesFromIdentifiers(accessRules, StandardRules.CAACCESS.resource(), caIdToNameMap.keySet()));
384         // Find RA resources allowed by this role
385         setResourcesEeSelected(getSelectedRulesFromSelectItems(accessRules, getAvailableResourcesEe()));
386         // Find EEP resources allowed by this role
387         setResourcesEepSelected(getSelectedRulesFromIdentifiers(accessRules, AccessRulesConstants.ENDENTITYPROFILEPREFIX, eepIdToNameMap.keySet(), getAvailableResourcesEe()));
388         // Find KV resources allowed by this role
389         setResourcesKeyValidatorsSelected(getSelectedRulesFromIdentifiers(accessRules, StandardRules.VALIDATORACCESS.resource(), keyValidatorsIdToNameMap.keySet()));
390         // Find IKB resources allowed by this role
391         setResourcesIkbSelected(getSelectedRulesFromSelectItems(accessRules, getAvailableResourcesIkb()));
392         // Find Other resources allowed by this role
393         setResourcesOtherSelected(getSelectedRulesFromSelectItems(accessRules, getAvailableResourcesOther()));
394     }

396     /** @param accessRules Access Rules
397      * @param baseResource Base Resource
398      * @param ids IDs
399      * @return minimal list of resources that the provided access rules grants */
400     private List<String> getSelectedRulesFromIdentifiers(final LinkedHashMap<String, Boolean> accessRules, final String baseResource, final Set<Integer> ids) {
401         final List<String> ret = new ArrayList<>();
402         if (AccessRulesHelper.hasAccessToResource(accessRules, baseResource)) {
403             ret.add(baseResource);
404         } else {
405             for (final int id : ids) {
406                 final String resource = AccessRulesHelper.normalizeResource(baseResource + id);
407                 if (AccessRulesHelper.hasAccessToResource(accessRules, resource)) {
408                     ret.add(resource);
409                 }
410             }
411         }
412         return ret;
413     }

415     /**
416      * Like {@link #getSelectedRulesFromIdentifiers}, but also includes rules that have all enabled sub-rules present.
417      * (example: if "Create End Entity" and "Edit End Entity" are selected, then it will also include all profiles that have those items allowed)
418      * @param accessRules Access Rules
419      * @param baseResource Base Resource
420      * @param ids  IDs
421      * @param selectedSubRules sub rules 
422      * @return minimal list of resources that the provided access rules grants
423      */
424     private List<String> getSelectedRulesFromIdentifiers(final LinkedHashMap<String, Boolean> accessRules, final String baseResource, final Set<Integer> ids,
425             final List<SelectItem> selectedSubRules) {
426         // Get all items that are allowed recursively
427         final List<String> ret = getSelectedRulesFromIdentifiers(accessRules, baseResource, ids);
428         // Also add items where all allowed sub-rules are explicitly allowed for the item
429         for (final int id : ids) {
430             final String resource = AccessRulesHelper.normalizeResource(baseResource + id);
431             if (ret.contains(resource)) {
432                 continue; // Don't add twice
433             }
434             boolean allSubRulesAllowed = true;
435             for (final SelectItem selectedSubRule : selectedSubRules) {
436                 final String subRule = String.valueOf(selectedSubRule.getValue()).replace(AccessRulesConstants.REGULAR_RAFUNCTIONALITY + "/", "");
437                 if (!accessRules.containsKey(resource + subRule)) {
438                     allSubRulesAllowed = false;
439                     break;
440                 }
441             }
442             if (allSubRulesAllowed) {
443                 if (log.isDebugEnabled()) {
444                     log.debug("Item " + resource + " is considered to be allowed, since all allowed sub-rules are also allowed for this item");
445                 }
446                 ret.add(resource);
447             }
448         }
449         return ret;
450     }

452     /** @param accessRules Rules
453      * @param selectItems Items
454      * @return minimal list of resources that the provided access rules grants */
455     private List<String> getSelectedRulesFromSelectItems(final LinkedHashMap<String, Boolean> accessRules, final List<SelectItem> selectItems) {
456         final List<String> ret = new ArrayList<>();
457         for (final SelectItem selectItem : selectItems) {
458             final String resource = AccessRulesHelper.normalizeResource(String.valueOf(selectItem.getValue()));
459             if (AccessRulesHelper.hasAccessToResource(getRole().getAccessRules(), resource)) {
460                 ret.add(resource);
461             }
462         }
463         return ret;
464     }

510     /** @return a list of available access rules templates (and detects the best match to the existing role's rules for performance reasons at the same time) */
511     public List<SelectItem> getAvailableAccessRulesTemplates() {
512         if (availableAccessRulesTemplates==null) {
513             availableAccessRulesTemplates = new ArrayList<>();
514             for (final AccessRulesTemplate accessRulesTemplate : accessRulesTemplates) {
515                 // Ensure that current admin is authorized to all access rules implied by the template
516                 final List<String> allowedResources = new ArrayList<>();
517                 // Ensure that there is no access rule in role that is not covered by template to check for a match
518                 if (getRole() != null) {
519                     if (getRole().getAccessRules() != null) {
520                         final HashMap<String, Boolean> remainingAccessRulesInRole = new HashMap<>(getRole().getAccessRules());
521                         AccessRulesHelper.normalizeResources(remainingAccessRulesInRole);
522                         AccessRulesHelper.minimizeAccessRules(remainingAccessRulesInRole);
523                         filterOutSelectItems(remainingAccessRulesInRole, getAvailableResourcesCa());
524                         filterOutSelectItems(remainingAccessRulesInRole, getAvailableResourcesEep());
525                         filterOutSelectItems(remainingAccessRulesInRole, getAvailableResourcesKeyValidators());
526                         for (final Entry<String, Boolean> entry : accessRulesTemplate.getAccessRules().entrySet()) {

541                                     log.debug("Role '" + getRole().getRoleNameFull() + "' does not qualify as a '" + accessRulesTemplate.getName()
542                                             + "'. Extra rules: " + Arrays.toString(remainingAccessRulesInRole.keySet().toArray()));
543                                 }

553                         log.debug("Role with ID " + roleIdParam + " could not be retrieved from the database.");

778     /** Invoked by the user to save the current selection in Basic mode */
779     public void actionSaveAccessRules() {
780         // Add access rules from template that are not user configurable
781         final HashMap<String,Boolean> newAccessRules = new HashMap<>(getAccessRulesTemplate().getAccessRules());
782         filterOutSelectItems(newAccessRules, getAvailableResourcesCa());
783         filterOutSelectItems(newAccessRules, getAvailableResourcesEe());
784         filterOutSelectItems(newAccessRules, getAvailableResourcesEep());
785         filterOutSelectItems(newAccessRules, getAvailableResourcesKeyValidators());
786         filterOutSelectItems(newAccessRules, getAvailableResourcesIkb());
787         filterOutSelectItems(newAccessRules, getAvailableResourcesOther());
788         // Add access rules selected by the user
789         for (final String resource : getResourcesCaSelected()) {
790             newAccessRules.put(resource, Role.STATE_ALLOW);
791         }
792         for (final String resource : getResourcesEeSelected()) {
793             newAccessRules.put(resource, Role.STATE_ALLOW);
794         }
795         for (final String resource : getResourcesEepSelected()) {
796             newAccessRules.put(resource, Role.STATE_ALLOW);
797         }
798         if (isEnableEndEntityProfileLimitations()) {
799             /*
800              * To be authorized to an EEP, the authentication token needs to be authorized to both
801              *  '/ra_functionality/some_function/'
802              *  '/endentityprofilesrules/<eepId>/some_function/'
803              *
804              * So here in basic mode, we simply allow '/endentityprofilesrules/<eepId>/' and rely
805              * on the '/ra_functionality/' sub-rules to limit functions.
806              *
807              * Long story short: Do nothing here.
808              */
809         }
810         for (final String resource : getResourcesKeyValidatorsSelected()) {
811             newAccessRules.put(resource, Role.STATE_ALLOW);
812         }
813         for (final String resource : getResourcesIkbSelected()) {
814             newAccessRules.put(resource, Role.STATE_ALLOW);
815         }
816         for (final String resource : getResourcesOtherSelected()) {
817             newAccessRules.put(resource, Role.STATE_ALLOW);
818         }
819         // Replace access rules and persist
820         final Role role = getRole();
821         role.getAccessRules().clear();
822         role.getAccessRules().putAll(newAccessRules);
823         try {
824             this.role = roleSession.persistRole(getAdmin(), role);
825             super.addGlobalMessage(FacesMessage.SEVERITY_INFO, "ACCESSRULES_INFO_SAVED");
826         } catch (RoleExistsException e) {
827             throw new IllegalStateException(e);
828         } catch (AuthorizationDeniedException e) {
829             super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "ACCESSRULES_ERROR_UNAUTH", e.getMessage());
830         } finally {
831             nonAjaxPostRedirectGet();
832         }
833     }

835     /** Remove the resources of the SelectItem values from the access rule map if present 
836      * @param newAccessRules Rules
837      * @param selectItems Items */
838     private void filterOutSelectItems(final HashMap<String, Boolean> newAccessRules, final List<SelectItem> selectItems) {
839         for (final SelectItem selectItem : selectItems) {
840             final String selectedResource = String.valueOf(selectItem.getValue());
841             final Iterator<Map.Entry<String, Boolean>> iter = newAccessRules.entrySet().iterator();
842             while (iter.hasNext()) {
843                 final Map.Entry<String, Boolean> accessRuleEntry = iter.next();
844                 final boolean allowed = accessRuleEntry.getValue();
845                 final String rule = accessRuleEntry.getKey();
846                 if (allowed && (rule.equals(selectedResource) || rule.startsWith(selectedResource))) {
847                     iter.remove();
848                 }
849             }
850         }
851     }

866     public List<AccessRuleCollection> getAuthorizedResourcesByCategory() {
867         final Map<String, AccessRuleCollection> categoryToAccessRuleCollectionMap = new LinkedHashMap<>();
868         for (final AccessRuleItem accessRuleItem : getAuthorizedAccessRuleItems()) {
869             final String category = accessRuleItem.getCategory();
870             AccessRuleCollection accessRuleCollection = categoryToAccessRuleCollectionMap.get(category);
871             if (accessRuleCollection==null) {
872                 accessRuleCollection = new AccessRuleCollection(category, new ArrayList<AccessRuleItem>());
873                 categoryToAccessRuleCollectionMap.put(category, accessRuleCollection);
874             }
875             accessRuleCollection.getCollection().add(accessRuleItem);
876         }
877         for (final AccessRuleCollection accessRuleCollection : categoryToAccessRuleCollectionMap.values()) {
878             Collections.sort(accessRuleCollection.getCollection());
879         }
880         return new ArrayList<>(categoryToAccessRuleCollectionMap.values());

883     /** @return the advanced mode list of all authorized access rule items */
884     private List<AccessRuleItem> getAuthorizedAccessRuleItems() {
885         if (authorizedAccessRuleItems==null) {
886             final List<AccessRuleItem> allAccessRuleItems = getAllAccessRuleItems();
887             final LinkedHashMap<String, Boolean> rolesAccesssRules = getRole().getAccessRules();
888             AccessRulesHelper.minimizeAccessRules(rolesAccesssRules);
889             for (final AccessRuleItem accessRuleItem : new ArrayList<>(allAccessRuleItems)) {
890                 if (authorizationSession.isAuthorizedNoLogging(getAdmin(), accessRuleItem.getResource())) {
891                     // Check current Role' state of this rule
892                     accessRuleItem.setState(AccessRuleState.toAccessRuleState(rolesAccesssRules.get(accessRuleItem.getResource())).name());
893                 } else {
894                     // Note that for EEPs you are only "really" authorized to it if you also are authorized to all the CAs in it
895                     // Similar goes for UserDataSources which is super-inefficient to check..
896                     // BUT if the current admin is authorized to a rule he is also authorized to give the same access to others
897                     allAccessRuleItems.remove(accessRuleItem);
898                 }
899             }
900             authorizedAccessRuleItems = allAccessRuleItems;
901         }
902         return authorizedAccessRuleItems;
903     }

936     /** Invoked by the admin when saving access rules in advanced mode. */
937     public void actionSaveAccessRulesAdvanced() {
938         final Role role = getRole();
939         final LinkedHashMap<String, Boolean> accessRules = role.getAccessRules();
940         accessRules.clear();
941         for (final AccessRuleItem accessRuleItem : authorizedAccessRuleItems) {
942             if (!AccessRuleState.UNDEFINED.equals(accessRuleItem.getStateEnum())) {
943                 accessRules.put(accessRuleItem.getResource(), AccessRuleState.ALLOW.equals(accessRuleItem.getStateEnum()));
944             }
945         }
946         final int numberOfRulesBeforeSave = accessRules.size();
947         try {
948             this.role = roleSession.persistRole(getAdmin(), role);
949             final int numberOfRedundantRules = numberOfRulesBeforeSave-role.getAccessRules().size();
950             if (numberOfRedundantRules==0) {
951                 super.addGlobalMessage(FacesMessage.SEVERITY_INFO, "ACCESSRULES_INFO_SAVED");
952             } else {
953                 super.addGlobalMessage(FacesMessage.SEVERITY_INFO, "ACCESSRULES_INFO_SAVED_MIN", numberOfRedundantRules);
954             }
955         } catch (RoleExistsException e) {
956             throw new IllegalStateException(e);
957         } catch (AuthorizationDeniedException e) {
958             super.addGlobalMessage(FacesMessage.SEVERITY_ERROR, "ACCESSRULES_ERROR_UNAUTH", e.getMessage());
959         } finally {
960             nonAjaxPostRedirectGet();
961         }
962         authorizedAccessRuleItems = null;
