14  package org.ejbca.ui.web.admin;
15  
16  import java.util.Set;
17  
18  import javax.faces.application.FacesMessage;
19  import javax.faces.component.UIComponent;
20  import javax.faces.context.FacesContext;
21  import javax.faces.validator.Validator;
22  import javax.faces.validator.ValidatorException;
23  
24  import org.apache.log4j.Logger;
25  import org.cesecore.util.StringTools;
26  import org.ejbca.ui.web.admin.configuration.EjbcaJSFHelper;
27  
28  /** JSF validator to check that input fields do not contain characters that might be dangerous for SQL queries 
29   * (non parameterized queries that is).
30   * 
31   * @version $Id: LegalCharsValidator.java 29711 2018-08-21 16:29:40Z aminkh $
32   */
33  public class LegalCharsValidator implements Validator {
34  	private static final Logger log = Logger.getLogger(LegalCharsValidator.class);
35  
36  	@Override
37  	public void validate(FacesContext facesContext, UIComponent uIComponent, Object object) throws ValidatorException {
38          String textFieldValue = null;
39          if (object instanceof String) {
40              textFieldValue = (String) object;
41          }
42          if (log.isDebugEnabled()) {
43              log.debug("Validating component " + uIComponent.getClientId(facesContext) + " with value \"" + textFieldValue + "\"");
44          }
45  
46          Set<String> invalidCharacters = StringTools.hasSqlStripChars(textFieldValue);
47          if (!invalidCharacters.isEmpty()) {
48              StringBuilder sb = new StringBuilder("");
49              for (String error : invalidCharacters) {
50                  sb.append(", " + error);
51              }
52              final String msg = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText("INVALIDCHARS") + sb.substring(2);
53              throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, msg, null));
54          }
55      }
56  }
