210     @SuppressWarnings("unchecked")
211     public String save() {
212         try {
213             final ApprovalProfile approvalProfile = getApprovalProfile();
214             List<Integer> stepOrder = new ArrayList<>();
215             for (final ApprovalStepGuiObject approvalSequenceGuiObject : steps) {
216                 final int sequenceIdentifier = approvalSequenceGuiObject.getIdentifier();
217                 for (final ApprovalPartitionProfileGuiObject approvalPartitionGuiObject : approvalSequenceGuiObject.getPartitionGuiObjects()) {
218                     approvalProfile.addPropertiesToPartition(sequenceIdentifier, approvalPartitionGuiObject.getPartitionId(),
219                             (List<DynamicUiProperty<? extends Serializable>>) approvalPartitionGuiObject.getProfilePropertyList().getWrappedData());
220                 }
221                 stepOrder.add(sequenceIdentifier);
222             }
223             approvalProfileSession.changeApprovalProfile(getAdmin(), approvalProfile);
224             addInfoMessage("APPROVALPROFILESAVED");
225             return "done";
226         } catch (AuthorizationDeniedException e) {
227             addNonTranslatedErrorMessage("Not authorized to edit approval profiles.");
228         }
229         return "";
230     }

232     /**
233      * Saves the current approval profile to a temporary variable managed by this bean. Should be
234      * used to save the current state before the UI is refreshed. This prevents data entry in UI
235      * controls to be discarded, causing poor user experience. To permanently save the approval
236      * profile, see {@link #save}. In comparison, this method does not require any authorisation,
237      * and will not throw any exceptions, nor show any error messages to the user.
238      */
239     @SuppressWarnings("unchecked")
240     public void saveTemporary() {
241         final ApprovalProfile approvalProfile = getApprovalProfile();
242         final List<Integer> stepOrder = new ArrayList<>();
243         for (final ApprovalStepGuiObject step : steps) {
244             for (final ApprovalPartitionProfileGuiObject partition : step.getPartitionGuiObjects()) {
245                 final int stepId = step.getIdentifier();
246                 final int partitionId = partition.getPartitionId();
247                 final Object guiObject = partition.getProfilePropertyList().getWrappedData();
248                 approvalProfile.addPropertiesToPartition(stepId, partitionId, (List<DynamicUiProperty<? extends Serializable>>) guiObject);
249             }
250             stepOrder.add(step.getIdentifier());
251         }
252         steps = createStepListFromProfile(approvalProfile);
253     }

259     /**
260      * Adds the currently selected field to the partition specified by the parameter. The step that the partition resides in is retrieved from
261      * the steps class member.
262      *
263      * @param partitionId the ID of the partition to add the field to
264      * @return an empty string to keep the scope.
265      */
266     public String addField(int partitionId) {
267         final Integer currentStep = steps.getRowData().getIdentifier();
268 
269         saveTemporary();
270 
271         ApprovalProfile updatedApprovalProfile = getApprovalProfile();
272         DynamicUiProperty<? extends Serializable> property;
273         String fieldLabel = this.fieldLabel.get(partitionId);
274         FieldType fieldType = FieldType.getFromName(fieldToAdd.get(partitionId));
275         switch (fieldType) {
276         case TEXT:
277             property = new DynamicUiProperty<>(fieldLabel, new MultiLineString(""));
278             break;
279         case RADIOBUTTON:
280             property = new DynamicUiProperty<>(fieldLabel, null, new ArrayList<RadioButton>());
281             property.setType(RadioButton.class);
282             break;
283         case CHECKBOX:
284             property = new DynamicUiProperty<>(fieldLabel, Boolean.FALSE);
285             break;
286         case INTEGER:
287             property = new DynamicUiProperty<>(fieldLabel, Integer.valueOf(0));
288             break;
289         case LONG:
290             property = new DynamicUiProperty<>(fieldLabel, Long.valueOf(0L));
291             break;
292         default:
293             return "";
294         }
295 
296         if (updatedApprovalProfile.getStep(currentStep).getPartition(partitionId).getProperty(fieldLabel) != null) {
297             addErrorMessage("APPROVAL_PROFILE_FIELD_EXISTS");
298         } else {
299             updatedApprovalProfile.addPropertyToPartition(currentStep, partitionId, property);
300             steps = createStepListFromProfile(updatedApprovalProfile);
301             this.fieldLabel = new HashMap<>();
302             fieldToAdd = new HashMap<>();
303         }
304 
305         return "";
306     }

308     /**
309      * A special method for adding rows to radio button arrays. The ID of the partition is required, but the step identity and the radio button
310      * field identity can be derived from class members.
311      *
312      * @param partitionId the ID of the partition that the radio button resides in
313      * @param label the label for the new row
314      * @return an empty string to keep the scope.
315      */
316     public String addRowToRadioButton(int partitionId, String label) {
317         final Integer currentStep = steps.getRowData().getIdentifier();
318         final ApprovalProfile updatedApprovalProfile = getApprovalProfile();
319         final List<ApprovalPartitionProfileGuiObject> guiPartitions = steps.getRowData().getPartitionGuiObjects();
320         for (ApprovalPartitionProfileGuiObject approvalPartitionProfileGuiObject : guiPartitions) {
321             //find the right partition
322             if (approvalPartitionProfileGuiObject.getPartitionId() == partitionId) {
323                 @SuppressWarnings("unchecked")
324                 DynamicUiProperty<RadioButton> radioButtonProperty = (DynamicUiProperty<RadioButton>) approvalPartitionProfileGuiObject
325                         .getProfilePropertyList().getRowData();
326                 Collection<RadioButton> possibleValues = radioButtonProperty.getPossibleValues();
327                 RadioButton newRadio = new RadioButton(label);
328                 if(possibleValues.contains(newRadio)) {
329                     addErrorMessage("APPROVAL_PROFILE_FIELD_RADIO_EXISTS");
330                     return "";
331                 }
332                 if (possibleValues.size() == 0) {
333                     radioButtonProperty.setDefaultValue(newRadio);
334                 }
335                 possibleValues.add(newRadio);
336                 radioButtonProperty.setPossibleValues(possibleValues);
337                 saveTemporary();
338                 updatedApprovalProfile.addPropertyToPartition(currentStep, partitionId, radioButtonProperty);
339                 steps = createStepListFromProfile(updatedApprovalProfile);
340                 break;
341             }
342         }
343         return "";
344     }

346     /**
347      * Similar to the above, the below method removes a row from a radio button array.
348      *
349      * @param partitionId the ID of the partition that the radio button resides in
350      * @param encodedRadioButton the encoded radio button
351      * @return an empty string to keep the scope.
352      */
353     public String removeRowFromRadioButton(int partitionId, String encodedRadioButton) {
354         final Integer currentStep = steps.getRowData().getIdentifier();
355         final ApprovalProfile updatedApprovalProfile = getApprovalProfile();
356         final RadioButton radioButton = DynamicUiProperty.getAsObject(encodedRadioButton, RadioButton.class);
357         final List<ApprovalPartitionProfileGuiObject> guiPartitions = steps.getRowData().getPartitionGuiObjects();
358         for(ApprovalPartitionProfileGuiObject approvalPartitionProfileGuiObject : guiPartitions) {
359             //find the right partition
360             if(approvalPartitionProfileGuiObject.getPartitionId() == partitionId) {
361                 @SuppressWarnings("unchecked")
362                 DynamicUiProperty<RadioButton> radioButtonProperty = (DynamicUiProperty<RadioButton>) approvalPartitionProfileGuiObject.getProfilePropertyList().getRowData();
363                 List<RadioButton> oldValues = new ArrayList<>(radioButtonProperty.getPossibleValues());
364                 List<RadioButton> prunedValues = new ArrayList<>();
365                 for(RadioButton dynamicRadioButton : oldValues) {
366                     if(!dynamicRadioButton.equals(radioButton)) {
367                         prunedValues.add(dynamicRadioButton);
368                     }
369                 }
370                 if(radioButtonProperty.getDefaultValue().equals(radioButton) && !prunedValues.isEmpty()) {
371                     radioButtonProperty.setDefaultValue(prunedValues.get(0));
372                 }
373                 radioButtonProperty.setPossibleValues(prunedValues);
374                 saveTemporary();
375                 updatedApprovalProfile.addPropertyToPartition(currentStep, partitionId, radioButtonProperty);
376                 steps = createStepListFromProfile(updatedApprovalProfile);
377                 break;
378             }
379         }
380         return "";
381     }

478     private ListDataModel<ApprovalStepGuiObject> createStepListFromProfile(final ApprovalProfile approvalProfile) {
479         List<ApprovalStepGuiObject> steps = new ArrayList<>();
480         int ordinal = 1;
481         //Use the internal ordering for sequences, if one is predefined
482         ApprovalStep step = approvalProfile.getFirstStep();
483         Map<Integer, List<DynamicUiProperty<? extends Serializable>>> partitionProperties = getPartitionProperties(step);
484         steps.add(new ApprovalStepGuiObject(step, approvalProfile.getApprovalProfileTypeIdentifier(), ordinal, partitionProperties));
485         while (step.getNextStep() != null) {
486             step = approvalProfile.getStep(step.getNextStep());
487             partitionProperties = getPartitionProperties(step);
488             steps.add(new ApprovalStepGuiObject(step, approvalProfile.getApprovalProfileTypeIdentifier(), ++ordinal, partitionProperties));
489         }
490         return new ListDataModel<>(steps);
491     }

493     /**
494      * Take an approval step and extract its partitions and respective properties, filling in with values from the database where required.
495      *
496      * @param step an approval step
497      * @return a Map linking partitions IDs to lists of each partitions properties.
498      */
499     private Map<Integer, List<DynamicUiProperty<? extends Serializable>>> getPartitionProperties(ApprovalStep step) {
500         Map<Integer, List<DynamicUiProperty<? extends Serializable>>> partitionProperties = new LinkedHashMap<>();
501         for(ApprovalPartition approvalPartition : step.getPartitions().values() ) {
502             List<DynamicUiProperty<? extends Serializable>> propertyList = new ArrayList<>();
503             for(DynamicUiProperty<? extends Serializable> property : approvalPartition.getPropertyList().values()) {
504                 DynamicUiProperty<? extends Serializable> propertyClone = new DynamicUiProperty<>(property);
505                 switch (propertyClone.getPropertyCallback()) {
506                 case ROLES:
507                     final List<Role> allAuthorizedRoles = roleSession.getAuthorizedRoles(getAdmin());
508                     final List<RoleInformation> roleRepresentations = new ArrayList<>();
509                     for (final Role role : allAuthorizedRoles) {
510                         if (AccessRulesHelper.hasAccessToResource(role.getAccessRules(), AccessRulesConstants.REGULAR_APPROVEENDENTITY)
511                                 || AccessRulesHelper.hasAccessToResource(role.getAccessRules(), AccessRulesConstants.REGULAR_APPROVECAACTION)) {
512                             try {
513                                 final List<RoleMember> roleMembers = roleMemberSession.getRoleMembersByRoleId(getAdmin(), role.getRoleId());
514                                 roleRepresentations.add(RoleInformation.fromRoleMembers(role.getRoleId(), role.getNameSpace(), role.getRoleName(), roleMembers));
515                             } catch (AuthorizationDeniedException e) {
516                                 if (log.isDebugEnabled()) {
517                                     log.debug("Not authorized to members of authorized role '"+role.getRoleNameFull()+"' (?):" + e.getMessage());
518                                 }
519                             }
520                         }
521                     }
522                     if(!roleRepresentations.contains(propertyClone.getDefaultValue())) {
523                         //Add the default, because it makes no sense why it wouldn't be there. Also, it may be a placeholder for something else.
524                         roleRepresentations.add(0, (RoleInformation) propertyClone.getDefaultValue());
525                     }
526                     propertyClone.setPossibleValues(roleRepresentations);
527                     updateEncodedValues(propertyClone, property);
528                     break;
529                 case ROLES_VIEW:
530                     final List<Role> allAuthorizedRolesForViewing = roleSession.getAuthorizedRoles(getAdmin());
531                     final List<RoleInformation> viewingRoleRepresentations = new ArrayList<>();
532                     for (final Role role : allAuthorizedRolesForViewing) {
533                         if (AccessRulesHelper.hasAccessToResource(role.getAccessRules(), AccessRulesConstants.REGULAR_VIEWAPPROVALS)
534                                 || AccessRulesHelper.hasAccessToResource(role.getAccessRules(), AccessRulesConstants.REGULAR_APPROVEENDENTITY)
535                                 || AccessRulesHelper.hasAccessToResource(role.getAccessRules(), AccessRulesConstants.REGULAR_APPROVECAACTION)) {
536                             try {
537                                 final List<RoleMember> roleMembers = roleMemberSession.getRoleMembersByRoleId(getAdmin(), role.getRoleId());
538                                 viewingRoleRepresentations.add(RoleInformation.fromRoleMembers(role.getRoleId(), role.getNameSpace(), role.getRoleName(), roleMembers));
539                             } catch (AuthorizationDeniedException e) {
540                                 if (log.isDebugEnabled()) {
541                                     log.debug("Not authorized to members of authorized role '"+role.getRoleNameFull()+"' (?):" + e.getMessage());
542                                 }
543                             }
544                         }
545                     }
546                     if(!viewingRoleRepresentations.contains(propertyClone.getDefaultValue())) {
547                         viewingRoleRepresentations.add(0, (RoleInformation) propertyClone.getDefaultValue());
548                     }
549                     propertyClone.setPossibleValues(viewingRoleRepresentations);
550                     updateEncodedValues(propertyClone, property);
551                     break;    
552                 case NONE:
553                     break;
554                 default:
555                     break;
556                 }
557                 propertyList.add(propertyClone);
558             }
559             partitionProperties.put(approvalPartition.getPartitionIdentifier(), propertyList);
560         }
561         return partitionProperties;
562     }

600     public void addNotification(final int partitionIdentifier) {
601         final Integer currentStep = steps.getRowData().getIdentifier();
602         saveTemporary();
603         final ApprovalProfile approvalProfile = getApprovalProfile();
604         final ApprovalStep approvalStep = approvalProfile.getStep(currentStep);
605         final ApprovalPartition approvalPartition = approvalStep.getPartition(partitionIdentifier);
606         // Configure some nice defaults
607         final GlobalConfiguration globalConfiguration = (GlobalConfiguration) globalConfigurationSession.getCachedConfiguration(GlobalConfiguration.GLOBAL_CONFIGURATION_ID);
608         final HttpServletRequest httpServletRequest = ((HttpServletRequest) FacesContext.getCurrentInstance().getExternalContext().getRequest());

671     /**
672      * Updates the encoded values of propertyClone if
673      * there has been a change in the role members of the
674      * any of the roles which were selected in the list box
675      * before the change.
676      * Uses property identifier as a base for comparison.
677      *
678      * @param propertyClone updated property
679      * @param property current property
680      */
681     private void updateEncodedValues(
682             final DynamicUiProperty<? extends Serializable> propertyClone,
683             final DynamicUiProperty<? extends Serializable> property) {
684 
685         List<Integer> currentIds = new ArrayList<>();
686 
687         for (final String value : property.getEncodedValues()) {
688             RoleInformation roleInfo = DynamicUiProperty.getAsObject(value, RoleInformation.class);
689             currentIds.add(roleInfo.getIdentifier());
690         }
691 
692         List<String> finalListOfEncodedValues = new ArrayList<>();
693 
694         for (final Serializable value : propertyClone.getPossibleValues()) {
695             RoleInformation roleInformation = (RoleInformation) value;
696 
697             if (currentIds.contains(roleInformation.getIdentifier())) {
698                 finalListOfEncodedValues.add(property.getAsEncodedValue(property.getType().cast(value)));
699             }
700         }
701 
702         // Here we update the propertyClone set of encoded values.
703         try {
704             propertyClone.setEncodedValues(finalListOfEncodedValues);
705         } catch (PropertyValidationException e) {
706             log.error("Invalid propery value while setting the encoded values for property clone!" + e);
707         }
708     }
