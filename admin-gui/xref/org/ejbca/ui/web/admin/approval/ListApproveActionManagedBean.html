99  	public String list() {
100 		Query query = new Query(Query.TYPE_APPROVALQUERY);
101 		if (selectedStatus.equals(ALL_STATUSES)){			
102 			query.add(getStartDate(), new Date());			
103 		} else if (selectedStatus.equals(Integer.toString(ApprovalDataVO.STATUS_EXPIRED))) {
104 		    //Expired requests will remain set as Waiting in the database. 
105 		    query.add(ApprovalMatch.MATCH_WITH_STATUS, BasicMatch.MATCH_TYPE_EQUALS, Integer.toString(ApprovalDataVO.STATUS_WAITINGFORAPPROVAL), Query.CONNECTOR_AND);
106 		    query.add(TimeMatch.MATCH_WITH_EXPIRETIME, null, new Date(), Query.CONNECTOR_AND);
107             query.add(getStartDate(), new Date());
108 		} else if (selectedStatus.equals(Integer.toString(ApprovalDataVO.STATUS_WAITINGFORAPPROVAL))) {
109 		    query.add(ApprovalMatch.MATCH_WITH_STATUS, BasicMatch.MATCH_TYPE_EQUALS, Integer.toString(ApprovalDataVO.STATUS_WAITINGFORAPPROVAL), Query.CONNECTOR_ANDNOT);
110 		    query.add(TimeMatch.MATCH_WITH_EXPIRETIME, null, new Date(), Query.CONNECTOR_AND);
111             query.add(getStartDate(), new Date());
112 		} else {	
113 			query.add(ApprovalMatch.MATCH_WITH_STATUS, BasicMatch.MATCH_TYPE_EQUALS, selectedStatus, Query.CONNECTOR_AND);
114 			query.add(getStartDate(), new Date());
115 		}
116         List<ApprovalDataVO> result = new ArrayList<>();
117 		try {
118             RAAuthorization raAuthorization = new RAAuthorization(EjbcaJSFHelper.getBean().getAdmin(), ejbLocalHelper.getGlobalConfigurationSession(),
119             		ejbLocalHelper.getAuthorizationSession(), ejbLocalHelper.getCaSession(), ejbLocalHelper.getEndEntityProfileSession());
120 			result = ejbLocalHelper.getApprovalSession().query(query, 0, QUERY_MAX_NUM_ROWS, 
121 			        raAuthorization.getCAAuthorizationString(), raAuthorization.getEndEntityProfileAuthorizationString(AccessRulesConstants.APPROVE_END_ENTITY));
122 			if(result.size() == QUERY_MAX_NUM_ROWS){
123 				String messagestring = getEjbcaWebBean().getText("MAXAPPROVALQUERYROWS1", true) + " " + QUERY_MAX_NUM_ROWS + " " + getEjbcaWebBean().getText("MAXAPPROVALQUERYROWS2", true);
124 				FacesContext ctx = FacesContext.getCurrentInstance();
125 				ctx.addMessage("error", new FacesMessage(FacesMessage.SEVERITY_ERROR,messagestring,messagestring));
126 			}
127 		} catch (IllegalQueryException e) {
128            addErrorMessage("INVALIDQUERY");
129 		} catch (AuthorizationDeniedException e) {
130 			addErrorMessage("AUTHORIZATIONDENIED");
131 		}
132 		listData = new ApprovalDataVOViewList(result);
133 		return null;		
134 	}

157 	/**
158 	 * Help method to list.
159 	 * @return Date
160 	 */
161 	private Date getStartDate(){
162 		if(Integer.parseInt(selectedTimeSpan) == 0){
163 			return new Date(0);
164 		}
165 		return new Date(new Date().getTime() - Integer.parseInt(selectedTimeSpan));
166 	}

168 	public String getRowClasses(){		
169 		if(listData.size() == 0){
170 			return "";
171 		}
172 		if(listData.size() == 1){
173 			return "Row0";
174 		}
175 		return "Row0, Row1";
176 	}
