86      public String getCaName() {
87          EjbcaJSFHelper helpBean = EjbcaJSFHelper.getBean();
88          if (data.getCAId() == ApprovalDataVO.ANY_CA) {
89              return helpBean.getEjbcaWebBean().getText("ANYCA", true);
90          }
91          try {
92              CAInfo caInfo = ejbLocalHelper.getCaSession().getCAInfo(helpBean.getAdmin(), data.getCAId());
93              if(caInfo != null) {
94                  return caInfo.getName();
95              } else {
96                  log.error("Can not get CA with id: "+data.getCAId());
97              }
98  		} catch (AuthorizationDeniedException e) {
99  			log.error("Can not get CA with id: "+data.getCAId(), e);
100 		}
101 		return "Error";
102     }

125     public String getRequestAdminName() {
126         String retval;
127         final Certificate cert = data.getApprovalRequest().getRequestAdminCert();
128         final AuthenticationToken reqAdmin = data.getApprovalRequest().getRequestAdmin();
129         if (cert != null) {
130             String dn = CertTools.getSubjectDN(cert);
131             String o = CertTools.getPartFromDN(dn, "O");
132             if (o == null) {
133                 o = "";
134             } else {
135                 o = ", " + o;
136             }
137             retval = CertTools.getPartFromDN(dn, "CN") + o;
138         } else {
139             retval = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText("CLITOOL", true); // Assume CLI if not match
140             if (reqAdmin != null) {
141                 for (Principal principal : reqAdmin.getPrincipals()) {
142                     if (principal instanceof PublicAccessAuthenticationToken.PublicAccessPrincipal) {
143                         // Unauthenticated users accessing the RA
144                         retval = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText("RAWEB", true);
145                         break;
146                     } else if (principal instanceof PublicWebPrincipal) {
147                         // Mostly self-registration in the Public Web
148                         final String ipAddress = ((PublicWebPrincipal) principal).getClientIPAddress();
149                         retval = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText("PUBLICWEB", true) + ": " + ipAddress;
150                         break;
151                     } else if (principal instanceof WebPrincipal) {
152                         // Other things, such as CMP, etc. We probably shouldn't ever get here, unless something is miss-configured.
153                         retval = principal.getName(); // e.g. "NameOfServlet: 198.51.100.123"
154                         break;
155                     }
156                 }
157             }
158         }
159         log.debug("getRequestAdminName " + retval);
160         return retval;
161     }

212     /**
213      * Detect all certificate and user links from approval data based on the
214      * static translations variables.
215      * 
216      * @return An array of Link-objects
217      */
218     public boolean isContainingLink() {
219         final List<ApprovalDataText> newTextRows = getNewRequestDataAsText();
220         for (final ApprovalDataText row : newTextRows) {
221             if (row.getHeader().equals(CERTSERIALNUMBER) || row.getHeader().equals(ISSUERDN) || row.getHeader().equals(USERNAME)) {
222                 return true;
223             }
224         }
225         return false;
226     }

228     /**
229      * Extract all certificate and user links from approval data based on the
230      * static translations variables.
231      * 
232      * @return An array of Link-objects
233      */
234     public List<LinkView> getApprovalDataLinks() {
235         List<LinkView> certificateLinks = new ArrayList<>();
236         List<String> certificateSerialNumbers = new ArrayList<>();
237         List<String> certificateIssuerDN = new ArrayList<>();
238         List<ApprovalDataText> newTextRows = getNewRequestDataAsText();
239 
240         for (final ApprovalDataText row : newTextRows) {
241             if (row.getHeader().equals(CERTSERIALNUMBER)) {
242                 certificateSerialNumbers.add(row.getData());
243             }
244             if (row.getHeader().equals(ISSUERDN)) {
245                 certificateIssuerDN.add(row.getData());
246             }
247         }
248         if (certificateIssuerDN.size() != certificateSerialNumbers.size()) {
249             // Return an empty array if we have a mismatch
250             return certificateLinks;
251         }
252         String link = null;
253         for (int i = 0; i < certificateSerialNumbers.size(); i++) {
254             try {
255                 link = EjbcaJSFHelper.getBean().getEjbcaWebBean().getBaseUrl()
256                         + EjbcaJSFHelper.getBean().getEjbcaWebBean().getGlobalConfiguration().getAdminWebPath()
257                         + "viewcertificate.jsp?certsernoparameter="
258                         + java.net.URLEncoder.encode(certificateSerialNumbers.get(i) + "," + certificateIssuerDN.get(i), "UTF-8");
259             } catch (UnsupportedEncodingException e) {
260                 log.warn("UnsupportedEncoding creating approval data link. ", e);
261             }
262             certificateLinks.add(new LinkView(link, EjbcaJSFHelper.getBean().getEjbcaWebBean().getText(CERTSERIALNUMBER) + ": ",
263                     certificateSerialNumbers.get(i), ""));
264         }
265         return certificateLinks;
266     }

268     public List<TextComparisonView> getTextListExceptLinks() {
269         ArrayList<TextComparisonView> textComparisonList = new ArrayList<>();
270         List<ApprovalDataText> newTextRows = getNewRequestDataAsText();
271         for (final ApprovalDataText row : newTextRows) {
272             if (row.getHeader().equals(CERTSERIALNUMBER)
273                     || row.getHeader().equals(ISSUERDN)) {
274                 continue;
275             }
276             String newString = "";
277             try {
278                 newString = translateApprovalDataText(row);
279             } catch (ArrayIndexOutOfBoundsException e) {
280                 // Do nothing orgstring should be "";
281             }
282             textComparisonList.add(new TextComparisonView(null, newString));
283         }
284         return textComparisonList;
285     }

287     public List<TextComparisonView> getTextComparisonList() {
288         ArrayList<TextComparisonView> textComparisonList = new ArrayList<>();
289         if (data.getApprovalRequest().getApprovalRequestType() == ApprovalRequest.REQUESTTYPE_COMPARING) {
290             List<ApprovalDataText> newTextRows = getNewRequestDataAsText();
291             List<ApprovalDataText> orgTextRows = getOldRequestDataAsText();
292             int size = newTextRows.size();
293             if (orgTextRows.size() > size) {
294                 size = orgTextRows.size();
295             }
296             for (int i = 0; i < size; i++) {
297                 String orgString = "";
298                 try {
299                     orgString = translateApprovalDataText(orgTextRows.get(i));
300                 } catch (IndexOutOfBoundsException e) {
301                     // Do nothing orgstring should be "";
302                 }
303                 String newString = "";
304                 try {
305                     newString = translateApprovalDataText(newTextRows.get(i));
306                 } catch (IndexOutOfBoundsException e) {
307                     // Do nothing orgstring should be "";
308                 }
309                 textComparisonList.add(new TextComparisonView(orgString, newString));
310             }
311         } else {
312             for(ApprovalDataText approvalDataText : getNewRequestDataAsText()) {
313                 textComparisonList.add(new TextComparisonView(null, translateApprovalDataText(approvalDataText)));
314             }
315         }
316         return textComparisonList;
317     }

319     private String translateApprovalDataText(ApprovalDataText data) {
320         String retval = "";
321         if (data.isHeaderTranslateable()) {
322             retval = EjbcaJSFHelper.getBean().getEjbcaWebBean().getText(data.getHeader(), true);
323         } else {
324             retval = data.getHeader();
325         }
326         if (data.isDataTranslatable()) {
327             retval += " : " + EjbcaJSFHelper.getBean().getEjbcaWebBean().getText(data.getData(), true);
328         } else {
329             retval += " : " + data.getData();
330         }
331         return retval;
332     }

334     private List<ApprovalDataText> getNewRequestDataAsText() {
335     	ApprovalRequest approvalRequest = data.getApprovalRequest();
336     	AuthenticationToken admin = EjbcaJSFHelper.getBean().getAdmin();
337     	if (approvalRequest instanceof EditEndEntityApprovalRequest) {
338     		return ((EditEndEntityApprovalRequest)approvalRequest).getNewRequestDataAsText(ejbLocalHelper.getCaSession(),
339     				ejbLocalHelper.getEndEntityProfileSession(), ejbLocalHelper.getCertificateProfileSession(), ejbLocalHelper.getHardTokenSession());
340     	} else if (approvalRequest instanceof AddEndEntityApprovalRequest) {
341     		return ((AddEndEntityApprovalRequest)approvalRequest).getNewRequestDataAsText(ejbLocalHelper.getCaSession(),
342     				ejbLocalHelper.getEndEntityProfileSession(), ejbLocalHelper.getCertificateProfileSession(), ejbLocalHelper.getHardTokenSession());
343     	} else {
344     		return approvalRequest.getNewRequestDataAsText(admin);
345     	}
346     }
347 
348     private List<ApprovalDataText> getOldRequestDataAsText() {
349     	ApprovalRequest approvalRequest = data.getApprovalRequest();
350     	AuthenticationToken admin = EjbcaJSFHelper.getBean().getAdmin();
351     	if (approvalRequest instanceof EditEndEntityApprovalRequest) {
352     		return ((EditEndEntityApprovalRequest)approvalRequest).getOldRequestDataAsText(admin, ejbLocalHelper.getCaSession(),
353     				ejbLocalHelper.getEndEntityProfileSession(), ejbLocalHelper.getCertificateProfileSession(), ejbLocalHelper.getHardTokenSession());
354     	} else {
355     		return approvalRequest.getOldRequestDataAsText(admin);
356     	}
357     }
