13  package org.ejbca.util.owaspcsrfguard;
14  
15  import java.io.IOException;
16  import java.io.UnsupportedEncodingException;
17  import java.util.Enumeration;
18  
19  import javax.servlet.Filter;
20  import javax.servlet.FilterChain;
21  import javax.servlet.FilterConfig;
22  import javax.servlet.ServletException;
23  import javax.servlet.ServletRequest;
24  import javax.servlet.ServletResponse;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  
28  import org.apache.commons.lang.StringUtils;
29  import org.apache.log4j.Logger;
30  import org.ejbca.ui.web.RequestHelper;
31  
32  /**
33   *  When using some filters, namely the CSRFGuard filter, it will mess up character encoding (if using something else than US-ASCII).
34   *  We use this filter to set the correct, configured in EJBCA, (i.e. UTF-8 to support intl characters) character encoding.
35   *  
36   *  We don't use a WebFilter annotation here, because we must set it in web.xml in order to enforce the order filters are called.
37   *  
38   * @version $Id: EncodingFilter.java 34116 2019-12-18 10:32:41Z anatom $
39  */
40  public class EncodingFilter implements Filter {
41  
42      private static final Logger log = Logger.getLogger(EncodingFilter.class.getName());
43  
44      @Override
45      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws ServletException,IOException {
46          final HttpServletRequest httpreq = (HttpServletRequest) request;
47          
48          // The way we use OWASP CSRF Guard makes all POST requests to our form in the last remaining jsp pages protected against csrf attacks
49          // it does however not protect GET requests to jsp pages, and service methods in jsp pages will respond to both GET and POST
50          // therefore we will not allow any actions to jsp pages, that tries to perform anything (i.e. edituser etc) using anything else than POST
51          // GET is still required in order to link to the pages to display them 
52          log.trace("Using EncodingFilter to ensure that JSP actions can only be performed using HTTP POST");
53          if (!httpreq.getMethod().equalsIgnoreCase("post")) {
54              // It's not a POST request, check if it is an operation that tries to do something
55              final Enumeration<String> params = httpreq.getParameterNames();
56              while (params.hasMoreElements()) {
57                  final String param = (String) params.nextElement();
58                  if (StringUtils.contains(param, "button")) {
59                      // It is an action to a JSP page clicking a button, i.e. trying to perform some action not just viewing the page
60                      // for performing actions we require POST, so disallow this
61                      log.warn("Refusing HTTP GET request containing parameter named '" + param + "'. Requests with parameters matching *button* must be done with HTTP POST.");
62                      ((HttpServletResponse) response).sendError(HttpServletResponse.SC_METHOD_NOT_ALLOWED);
63                      return; // don't continue on down the chain
64                  }
65                  
66              }
67          }
68          log.trace("Using EncodingFilter to set HTTP request character encoding");
69          try {
70              RequestHelper.setDefaultCharacterEncoding(httpreq);
71          } catch (UnsupportedEncodingException e) {
72              log.error("UnsupportedEncodingException: ", e);
73          }
74          //WARNING: do NOT swallow any Exception!!! If container can not see any exception, it will not redirect client to related error page.
75          //You can catch Exceptions and print them, and you must throw them at last!
76          chain.doFilter(request, response);
77      }
78  
79      @Override
80      public void destroy() {}
81  
82      @Override
83      public void init(FilterConfig arg0) throws ServletException {}
84  
85  }
