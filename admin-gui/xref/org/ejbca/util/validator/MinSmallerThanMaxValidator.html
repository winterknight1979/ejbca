13  package org.ejbca.util.validator;
14  
15  import java.math.BigInteger;
16  
17  import javax.faces.application.FacesMessage;
18  import javax.faces.component.UIComponent;
19  import javax.faces.context.FacesContext;
20  import javax.faces.validator.Validator;
21  import javax.faces.validator.ValidatorException;
22  
23  import org.apache.commons.lang.StringUtils;
24  import org.apache.log4j.Logger;
25  import org.cesecore.internal.InternalResources;
26  
27  /**
28   * Validator which will validate that a BigInteger value is smaller than another input value.
29   * 
30   * @version $Id: MinSmallerThanMaxValidator.java 27466 2017-12-07 14:26:31Z antonh $
31   */
32  public class MinSmallerThanMaxValidator implements Validator {
33      
34      /** Internal localization of logs and errors */
35      private static final InternalResources intres = InternalResources.getInstance();
36      
37      /** Class logger. */
38      private static final Logger log = Logger.getLogger(MinSmallerThanMaxValidator.class);
39  
40      @Override
41      public void validate(FacesContext context, UIComponent component, Object value) throws ValidatorException {
42          final BigInteger minValue = (BigInteger) value;
43          final String maxInput = (String) component.getAttributes().get("maxFieldInput");
44          
45          if (StringUtils.isNotBlank(maxInput) && minValue != null && minValue.compareTo(new BigInteger(maxInput)) == 1) {
46              final String field = (String) component.getAttributes().get("fieldName");
47              final String message = intres.getLocalizedMessage("validator.error.minimum_bigger", field, minValue, maxInput);
48              if (log.isDebugEnabled()) {
49                  log.debug(message);
50              }
51              throw new ValidatorException(new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message));
52          }
53      }
54  }
