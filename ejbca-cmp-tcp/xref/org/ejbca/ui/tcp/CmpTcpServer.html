14  package org.ejbca.ui.tcp;
15  
16  import java.io.File;
17  import java.net.UnknownHostException;
18  import java.util.logging.FileHandler;
19  import java.util.logging.Level;
20  import java.util.logging.Logger;
21  
22  import org.apache.commons.lang.StringUtils;
23  import org.ejbca.config.CmpTcpConfiguration;
24  import org.quickserver.net.AppException;
25  import org.quickserver.net.server.QuickServer;
26  
27  /**
28   * Starts and stops the CMP TCP listener service
29   * 
30   * @version $Id: CmpTcpServer.java 19902 2014-09-30 14:32:24Z anatom $
31   */
32  public class CmpTcpServer {
33  
34  	private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(CmpTcpServer.class);
35  	private static final String VER = "1.0";
36  
37  	private transient QuickServer myServer = null;
38  	
39  	public void start() throws UnknownHostException	{
40  		final String cmdHandle = org.ejbca.ui.tcp.CmpTcpCommandHandler.class.getName();
41  
42  		myServer = new QuickServer();
43  		myServer.setClientAuthenticationHandler(null);
44  		myServer.setBindAddr(CmpTcpConfiguration.getTCPBindAdress());
45  		myServer.setPort(CmpTcpConfiguration.getTCPPortNumber());
46  		myServer.setName("CMP TCP Server v " + VER);
47  		if(QuickServer.getVersionNo() >= 1.2) {
48  			LOG.info("Using 1.2 feature");
49  			myServer.setClientBinaryHandler(cmdHandle);
50  			myServer.setClientEventHandler(cmdHandle);
51  
52  			//reduce info to Console
53  			myServer.setConsoleLoggingToMicro();
54  		}
55  
56  		//setup logger to log to file
57  		Logger logger = null;
58  		FileHandler txtLog = null;
59  		final String logDir = CmpTcpConfiguration.getTCPLogDir();
60  		final File logFile = new File(logDir + "/");
61  		if(!logFile.canRead()) {
62  			logFile.mkdir();
63  		}
64  		try	{
65  			logger = Logger.getLogger("");
66  			logger.setLevel(Level.INFO);
67  
68  			logger = Logger.getLogger("cmptcpserver");
69  			logger.setLevel(Level.FINEST); 
70  			txtLog = new FileHandler(logDir+"/cmptcpserver.log");
71  			//reduce info 
72  			txtLog.setFormatter(new org.quickserver.util.logging.MicroFormatter());
73  			logger.addHandler(txtLog);
74  
75  			myServer.setAppLogger(logger); //imp
76  
77  			//myServer.setConsoleLoggingToMicro();
78  			myServer.setConsoleLoggingFormatter("org.quickserver.util.logging.SimpleTextFormatter");
79  			myServer.setConsoleLoggingLevel(Level.INFO);
80  		} catch(Exception e){
81  			LOG.error("Could not create xmlLog FileHandler : ", e);
82  		}
83  		try	{
84  			final String confFile = CmpTcpConfiguration.getTCPConfigFile();
85  			if (!StringUtils.isEmpty(confFile)) {
86  				final Object config[] = new Object[] {confFile};
87  				if (!myServer.initService(config)) {
88  					LOG.error("Configuration from config file "+confFile+" failed!");
89  				}
90  			}
91  			myServer.startServer();	
92  			//myServer.getQSAdminServer().setShellEnable(true);
93  			//myServer.startQSAdminServer();			
94  		} catch(AppException e){
95  			LOG.error("Error in server : ", e);
96  		}
97  	}
98  
99  	public void stop() {
100 		if (myServer != null) {
101 			try {
102 				myServer.stopService();
103 				myServer.closeAllPools();
104 			} catch (AppException e) {
105 				LOG.error("Error in server : ", e);
106 			} catch (Exception e) {
107 				LOG.error("Error in server : ", e);
108 			}
109 		}
110 	}
111 }
