14  package org.ejbca.ui.tcp;
15  
16  import java.io.ByteArrayOutputStream;
17  import java.io.DataOutputStream;
18  import java.io.IOException;
19  
20  import org.apache.log4j.Logger;
21  
22  /**
23   * Encodes a TCP messages to a client.
24   * 
25   * @author lars
26   * @version $Id: TcpReturnMessage.java 19902 2014-09-30 14:32:24Z anatom $
27   *
28   */
29  public class TcpReturnMessage {
30  	private static final Logger log = Logger.getLogger(TcpReceivedMessage.class.getName());
31  	/**
32  	 * The message to be sent to be returned to the client.
33  	 */
34  	final public byte message[];
35  	/**
36  	 * true if the socket should be closed after returning the message to the client.
37  	 */
38  	final public boolean doClose;
39  
40  	private TcpReturnMessage(byte m[], boolean c) {
41  		this.message = m;
42  		this.doClose = c;
43  	}
44  	/**
45  	 * @param inMsg resp message to be returned to the client.
46  	 * @param close true if the session to the client should be closed.
47  	 * @return the message to be sent to the client
48  	 * @throws IOException Fail
49  	 */
50  	public static TcpReturnMessage createMessage(byte inMsg[], boolean close) throws IOException {
51  		final byte[] msg;
52  		final int msgType;
53  		final boolean doClose;
54  		if ( inMsg!=null && inMsg.length>0 ) {
55  			msg = inMsg;
56  			doClose = close;
57  			msgType = 5;
58  		} else {
59  			msg = new byte[1];
60  			msg[0] = 0;
61  			msgType = 3;
62  			doClose = true;
63  		}
64  		// 5 is pkiRep, 6 is errorMsgRep, 3 is finRep
65  		// errorMsgRep should only be used for TCP protocol errors, see 3.5.6 in cmp-transport-protocols
66  		//if (resp.getStatus() != ResponseStatus.SUCCESS) {
67  		//  msgType = 6;
68  		//}
69  		final ByteArrayOutputStream bao = new ByteArrayOutputStream();
70  		final DataOutputStream dos = new DataOutputStream(bao); 
71  		// return msg length = msg.length + 3; 1 byte version, 1 byte flags and 1 byte message type
72  		dos.writeInt(msg.length+3);
73  		dos.writeByte(10);
74  		final int flags = doClose ? 1 : 0; // 1 if we should close, 0 otherwise
75  		dos.writeByte(flags); 
76  		dos.writeByte(msgType); 
77  		dos.write(msg);
78  		dos.flush();
79  		if (log.isDebugEnabled()) {
80  			log.debug("Wrote length: '"+msg.length+3+"' Wrote version: '10' Wrote flags: '"+flags+"' Wrote msgType: '"+msgType+"' Wrote msg with length: "+msg.length);
81  		}
82  		return new TcpReturnMessage( bao.toByteArray(), doClose );
83  	}
84  }
