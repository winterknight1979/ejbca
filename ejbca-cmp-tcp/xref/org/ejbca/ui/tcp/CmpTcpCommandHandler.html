14  package org.ejbca.ui.tcp;
15  
16  import java.io.IOException;
17  import java.net.SocketTimeoutException;
18  
19  import org.apache.log4j.Logger;
20  import org.cesecore.authentication.tokens.AlwaysAllowLocalAuthenticationToken;
21  import org.cesecore.authentication.tokens.AuthenticationToken;
22  import org.cesecore.authentication.tokens.WebPrincipal;
23  import org.ejbca.core.model.InternalEjbcaResources;
24  import org.ejbca.core.model.util.EjbLocalHelper;
25  import org.ejbca.core.protocol.NoSuchAliasException;
26  import org.quickserver.net.server.ClientBinaryHandler;
27  import org.quickserver.net.server.ClientEventHandler;
28  import org.quickserver.net.server.ClientHandler;
29  import org.quickserver.net.server.DataMode;
30  import org.quickserver.net.server.DataType;
31  
32  /**
33   * Class receiving TCP messages from QuickServer (receives quickserver events) and routing them to the correct CMP handler class.
34   * 
35   * @version $Id: CmpTcpCommandHandler.java 28716 2018-04-13 15:22:48Z samuellb $
36   */
37  public class CmpTcpCommandHandler implements ClientEventHandler, ClientBinaryHandler  {
38  
39  	private static final Logger LOG = Logger.getLogger(CmpTcpCommandHandler.class.getName());
40      private static final InternalEjbcaResources INTRES = InternalEjbcaResources.getInstance();
41      private static EjbLocalHelper ejb = null;
42  	
43  	private static synchronized EjbLocalHelper getEjb() {
44  		if (ejb == null) {
45  			ejb = new EjbLocalHelper();
46  		}
47  		return ejb;
48  	}
49  	
50      @Override
51  	public void gotConnected(final ClientHandler handler) throws SocketTimeoutException, IOException {
52  		if (LOG.isDebugEnabled()) {
53  			LOG.debug("CMP connection opened: "+handler.getHostAddress());
54  		}
55  		handler.setDataMode(DataMode.BINARY, DataType.IN);
56  		handler.setDataMode(DataMode.BINARY, DataType.OUT);
57  	}
58  
59      @Override
60  	public void lostConnection(final ClientHandler handler) throws IOException {
61  		LOG.debug("Connection lost: "+handler.getHostAddress());
62  	}
63  
64      @Override
65  	public void closingConnection(final ClientHandler handler) throws IOException {
66  		LOG.debug("Connection closed: "+handler.getHostAddress());
67  	}
68  
69      @Override
70  	public void handleBinary(final ClientHandler handler, final byte command[])	throws SocketTimeoutException, IOException {
71  		LOG.info(INTRES.getLocalizedMessage("cmp.receivedmsg", handler.getHostAddress()));
72  		long startTime = System.currentTimeMillis();
73  		final TcpReceivedMessage cmpTcpMessage = TcpReceivedMessage.getTcpMessage(command);
74  		if (cmpTcpMessage.message == null) {
75  			handler.closeConnection();
76  		} else {
77  		    final AuthenticationToken authenticationToken = new AlwaysAllowLocalAuthenticationToken(new WebPrincipal("CmpTcp", handler.getHostAddress()));
78  		    byte[] result = null;
79  			try {
80  			    result = getEjb().getRaMasterApiProxyBean().cmpDispatch(authenticationToken, cmpTcpMessage.message, "tcp");
81  			} catch (NoSuchAliasException e) {
82                  LOG.info(e.getMessage());
83                  handler.closeConnection();
84                  return;
85              }
86  			if (LOG.isDebugEnabled()) {
87  				LOG.debug("Sending back CMP response to client.");
88  			}
89  			// Send back reply
90  			final TcpReturnMessage sendBack = TcpReturnMessage.createMessage(result, cmpTcpMessage.doClose);
91  			if (LOG.isDebugEnabled()) {
92  				LOG.debug("Sending "+sendBack.message.length+" bytes to client");
93  			}
94  			handler.sendClientBinary(sendBack.message);
95  			long endTime = System.currentTimeMillis();
96  			final String iMsg = INTRES.getLocalizedMessage("cmp.sentresponsemsg", handler.getHostAddress(), Long.valueOf(endTime - startTime));
97  			LOG.info(iMsg);
98  			if ( cmpTcpMessage.doClose || sendBack.doClose ) {
99  				handler.closeConnection(); // It's time to say good bye			
100 			}
101 		}
102 	}
103 }
