20  /**
21   * Configuration used by specifying the configurationId as part of the request URL path.
22   *
23   * @version $Id: AcmeConfiguration.java 29902 2018-09-26 11:32:56Z samuellb $
24   */
25  public class AcmeConfiguration extends UpgradeableDataHashMap implements Serializable {
26  
27      private static final long serialVersionUID = 1L;
28  
29      private String configurationId = null;
30      private List<String> caaIdentities = new ArrayList<String>();
31  
32      private static final String KEY_REQUIRE_EXTERNAL_ACCOUNT_BINDING = "requireExternalAccountBinding";
33      private static final String KEY_PRE_AUTHORIZATION_ALLOWED = "preAuthorizationAllowed";
34      private static final String KEY_END_ENTITY_PROFILE_ID = "endEntityProfileId";
35      private static final String KEY_VALIDATION_HTTP_CALLBACK_URL_TEMPLATE = "valiationHttpCallbackUrlTemplate";
36     // private static final String KEY_TERMS_OF_SERVICE_VERSION = "termsOfServiceVersion";
37      private static final String KEY_TERMS_OF_SERVICE_URL = "termsOfServiceUrl";
38      private static final String KEY_WEB_SITE_URL = "webSiteUrl";
39      private static final String KEY_ORDER_VALIDITY = "orderValidity";
40      private static final String KEY_PRE_AUTHORIZATION_VALIDITY = "preAuthorizationValidity";
41      private static final String KEY_WILDCARD_CERTIFICATE_ISSUANCE_ALLOWED = "wildcardCertificateIssuanceAllowed";
42      private static final String KEY_DNS_RESOLVER = "dnsResolver";
43      private static final String KEY_DNSSEC_TRUST_ANCHOR = "dnssecTrustAnchor";
44      private static final String KEY_DNS_PORT = "dnsPort";
45      private static final String KEY_USE_DNSSEC_VALIDATION = "useDnssecValidation";
46      private static final String KEY_TERMS_OF_SERVICE_REQUIRE_NEW_APPROVAL = "termsOfServiceRequireNewApproval";
47      private static final String DNS_RESOLVER_DEFAULT = "8.8.8.8";
48      private static final int DNS_SERVER_PORT_DEFAULT = 53;
49  
50  
51      private static final int DEFAULT_END_ENTITY_PROFILE_ID = EndEntityConstants.NO_END_ENTITY_PROFILE;
52      private static final boolean DEFAULT_REQUIRE_EXTERNAL_ACCOUNT_BINDING = false;
53      private static final boolean DEFAULT_PRE_AUTHORIZATION_ALLOWED = false;
54      private static final boolean DEFAULT_REQUIRE_NEW_APPROVAL = true;
55      private static final boolean DEFAULT__WILDCARD_CERTIFICATE_ISSUANCE_ALLOWED = false;
56      private static final String DEFAULT_TERMS_OF_SERVICE_URL = "https://example.com/acme/terms";
57      private static final String DEFAULT_WEBSITE_URL = "https://www.example.com/";
58      private static final boolean DEFAULT_USE_DNSSEC_VALIDATION = true;

61      public AcmeConfiguration() {}
62  
63      public AcmeConfiguration(final Object upgradeableDataHashMapData) {
64          super.loadData(upgradeableDataHashMapData);
65      }
66  
67      @Override
68      public float getLatestVersion() {
69          return 0;
70      }
71  
72      @Override
73      public void upgrade() {}
74  
75      /** @return the configuration ID as used in the request URL path */
76      public String getConfigurationId() { return configurationId; }
77      public void setConfigurationId(final String configurationId) { this.configurationId = configurationId; }
78  
79      /**
80       * https://tools.ietf.org/html/draft-ietf-acme-acme-12#section-7.3.5
81       *
82       * NOTE: Don't expose this as client configuration yet. The current implementation has the code for validation
83       *       using a dummy HMAC, but will strip this info anyway from the actual account.
84       *
85       * @return true if an the server should enforce external account bindings.
86       */
87      public boolean isRequireExternalAccountBinding() {
88          return Boolean.valueOf((String)super.data.get(KEY_REQUIRE_EXTERNAL_ACCOUNT_BINDING));
89      }
90      public void setRequireExternalAccountBinding(final boolean requireExternalAccountBinding) {
91          super.data.put(KEY_REQUIRE_EXTERNAL_ACCOUNT_BINDING, String.valueOf(requireExternalAccountBinding));
92      }
93  
94      /**
95       * https://tools.ietf.org/html/draft-ietf-acme-acme-12#section-7.4.1
96       * "If a CA wishes to allow pre-authorization within ACME, it can offer a "new authorization" resource in its
97       * directory by adding the field "newAuthz" with a URL for the new authorization resource."
98       * @return boolean
99       */
100     public boolean isPreAuthorizationAllowed() {
101         return Boolean.valueOf((String)super.data.get(KEY_PRE_AUTHORIZATION_ALLOWED));
102     }
103     public void setPreAuthorizationAllowed(final boolean preAuthorizationAllowed) {
104         super.data.put(KEY_PRE_AUTHORIZATION_ALLOWED, String.valueOf(preAuthorizationAllowed));
105     }
106 
107     /** @return the End Entity Profile ID whos default Certificate Profile and CA will be used for issuing */
108     public int getEndEntityProfileId() {
109         final Integer endEntityProfileId = (Integer) super.data.get(KEY_END_ENTITY_PROFILE_ID);
110         return endEntityProfileId==null ? EndEntityConstants.NO_END_ENTITY_PROFILE : endEntityProfileId.intValue();
111     }
112     public void setEndEntityProfileId(final int endEntityProfileId) {
113         super.data.put(KEY_END_ENTITY_PROFILE_ID, Integer.valueOf(endEntityProfileId));
114     }
115 
116     /** @return the pattern we will use for "http-01" challenge validation. Defaults to example from RFC draft 06. */
117     public String getValidationHttpCallBackUrlTemplate() {
118         final String urlTemplate = (String) super.data.get(KEY_VALIDATION_HTTP_CALLBACK_URL_TEMPLATE);
119         return urlTemplate==null ? "http://{identifer}:80/.well-known/acme-challenge/{token}" : urlTemplate;
120     }
121     public void setValidationHttpCallBackUrlTemplate(final String urlTemplate) {
122         super.data.put(KEY_VALIDATION_HTTP_CALLBACK_URL_TEMPLATE, urlTemplate);
123     }
124 
125     /** @return an URL of where the current Terms Of Services can be located. */
126     public String getTermsOfServiceUrl() {
127         return (String) super.data.get(KEY_TERMS_OF_SERVICE_URL);
128     }
129     
130     public void setTermsOfServiceUrl(final String termsOfServiceUrl) {
131         super.data.put(KEY_TERMS_OF_SERVICE_URL, termsOfServiceUrl);
132     }
133     
134     /** @return the web site URL presented in the directory meta data */
135     public String getWebSiteUrl() {
136         return (String) super.data.get(KEY_WEB_SITE_URL);
137     }
138     public void setWebSiteUrl(final String webSiteUrl) {
139         super.data.put(KEY_WEB_SITE_URL, webSiteUrl);
140     }
141 
142     public List<String> getCaaIdentities() {
143         return caaIdentities;
144     }
145 
146     public void setCaaIdentities(final List<String> caaIdentities) {
147         this.caaIdentities = caaIdentities;
148     }
149 
150     /** @return how long a new order will be valid for in milliseconds */
151     public long getOrderValidity() {
152         final Long orderValidity = (Long) super.data.get(KEY_ORDER_VALIDITY);
153         return orderValidity==null ? 3600000L : orderValidity.intValue();
154     }
155     public void setOrderValidity(final int orderValidity) {
156         super.data.put(KEY_ORDER_VALIDITY, Long.valueOf(orderValidity));
157     }
158 
159     /** @return how long a new pre-authorizations will be valid for in milliseconds */
160     public long getPreAuthorizationValidity() {
161         final Long preAuthorizationValidity = (Long) super.data.get(KEY_PRE_AUTHORIZATION_VALIDITY);
162         return preAuthorizationValidity==null ? 24*3600000L : preAuthorizationValidity.intValue();
163     }
164     public void setPreAuthorizationValidity(final int preAuthorizationValidity) {
165         super.data.put(KEY_PRE_AUTHORIZATION_VALIDITY, Long.valueOf(preAuthorizationValidity));
166     }
167 
168     /** @return the number of required challenges that needs to be fulfilled in order to grant authorization for an identifier */
169     public int getValidChallengesPerAuthorization() {
170         return 1;
171     }

179     }

198         final Integer dnsPort = (Integer) super.data.get(KEY_DNS_PORT);

202     public void setDnsPort(final int dnsPort) {
203         super.data.put(KEY_DNS_PORT, dnsPort);
204     }

207         return Boolean.valueOf((String) super.data.get(KEY_TERMS_OF_SERVICE_REQUIRE_NEW_APPROVAL));
208     }
209     
210     public void setTermsOfServiceRequireNewApproval(boolean termsOfServiceRequireNewApproval) {
211         super.data.put(KEY_TERMS_OF_SERVICE_REQUIRE_NEW_APPROVAL, String.valueOf(termsOfServiceRequireNewApproval));
212     }

216         return Boolean.valueOf((String) super.data.get(KEY_USE_DNSSEC_VALIDATION));
217     }

220         super.data.put(KEY_USE_DNSSEC_VALIDATION, String.valueOf(useDnsSecValidation));

223     /** Initializes a new acme configuration with default values. 
224      * @param alias Alias*/
225     public void initialize(String alias) {
226         alias += ".";
227         setEndEntityProfileId(DEFAULT_END_ENTITY_PROFILE_ID);
228         setRequireExternalAccountBinding(DEFAULT_REQUIRE_EXTERNAL_ACCOUNT_BINDING);
229         setPreAuthorizationAllowed(DEFAULT_PRE_AUTHORIZATION_ALLOWED);
230         setTermsOfServiceUrl(DEFAULT_TERMS_OF_SERVICE_URL);
231         setTermsOfServiceRequireNewApproval(DEFAULT_REQUIRE_NEW_APPROVAL);
232         setWildcardCertificateIssuanceAllowed(DEFAULT__WILDCARD_CERTIFICATE_ISSUANCE_ALLOWED);
233         setWebSiteUrl(DEFAULT_WEBSITE_URL);
234         setDnsResolver(DNS_RESOLVER_DEFAULT);
235         setDnssecTrustAnchor(DnsSecDefaults.IANA_ROOT_ANCHORS_DEFAULT);
236         setDnsPort(DNS_SERVER_PORT_DEFAULT);
237         setUseDnsSecValidation(DEFAULT_USE_DNSSEC_VALIDATION);
238     }
