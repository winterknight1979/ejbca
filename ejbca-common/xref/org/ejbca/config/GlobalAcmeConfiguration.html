25  /**
26   * Global Configuration object which holds all the AcmeConfigurations and generic protocol settings.
27   * 
28   * @version $Id: GlobalAcmeConfiguration.java 29424 2018-07-02 10:08:09Z jekaterina_b_helmes $
29   */
30  public class GlobalAcmeConfiguration extends ConfigurationBase {
31  
32      private static final long serialVersionUID = 1L;
33      //private static final Logger log = Logger.getLogger(GlobalAcmeConfiguration.class);
34  
35      public static final String ACME_CONFIGURATION_ID = "ACME";
36      
37      private static final String KEY_INITIALIZED = "initialized";
38      private static final String KEY_DEFAULT_ACME_CONFIGURATION_ID = "defaultAcmeConfigurationId";
39      private static final String KEY_ACME_CONFIGURATION_PREFIX = "acmeConfiguration_";
40      private static final String KEY_REPLAY_NONCE_VALIDITY = "replayNonceValiditity";
41      private static final String KEY_REPLAY_NONCE_SECRETS_HMAC_WITH_SHA256 = "replayNonceSecretsHmacWithSHA256";
42  
43      @Override
44      public void upgrade() {}
45  
46      @Override
47      public String getConfigurationId() {
48          return ACME_CONFIGURATION_ID;
49      }
50  
51      public Properties getAsProperties() {
52          final Properties properties = new Properties();
53          properties.put(KEY_INITIALIZED, isInitialized());
54          properties.put(KEY_DEFAULT_ACME_CONFIGURATION_ID, getDefaultAcmeConfigurationId());
55          properties.put(KEY_REPLAY_NONCE_VALIDITY, getReplayNonceValidity());
56          properties.put("acmeConfigurationIds", Arrays.toString(getAcmeConfigurationIds().toArray()));
57          properties.put(KEY_REPLAY_NONCE_SECRETS_HMAC_WITH_SHA256,
58                  getReplayNonceSharedSecrets(PKCSObjectIdentifiers.id_hmacWithSHA256.getId())==null ? "not configured" : "configured");
59          return properties;
60      }
61  
62      /** @return true if this global configuration has been initialized */
63      public boolean isInitialized() {
64          return Boolean.valueOf((String)super.data.get(KEY_INITIALIZED));
65      }
66      /** @param initialized Set to true when this global configuration has been initialized */
67      public void setInitialized(final boolean initialized) {
68          super.data.put(KEY_INITIALIZED, String.valueOf(initialized));
69      }
70  
71      /** @return the default configured AcmeConfiguration IDs that will be used if none is specified in the request path */
72      public String getDefaultAcmeConfigurationId() {
73          return (String) data.get(KEY_DEFAULT_ACME_CONFIGURATION_ID);
74      }
75      /** @param defaultAcmeConfigurationId Set the default configured AcmeConfiguration IDs that will be used if none is specified in the request path */
76      public void setDefaultAcmeConfigurationId(final String defaultAcmeConfigurationId) {
77          data.put(KEY_DEFAULT_ACME_CONFIGURATION_ID, defaultAcmeConfigurationId);
78      }
79  
80      /** @return all the configured AcmeConfiguration IDs */
81      public List<String> getAcmeConfigurationIds() {
82          final Set<String> acmeConfigurationIds = new HashSet<>();
83          for (final Object key : data.keySet()) {
84              if (key instanceof String && ((String)key).startsWith(KEY_ACME_CONFIGURATION_PREFIX)) {
85                  acmeConfigurationIds.add(((String)key).substring(KEY_ACME_CONFIGURATION_PREFIX.length()));
86              }
87          }
88          return new ArrayList<String>(acmeConfigurationIds);
89      }
90  
91      /** @param configurationId ID
92       * @return the AcmeConfiguration that will be based on the configurationId specified in the request path */
93      public AcmeConfiguration getAcmeConfiguration(final String configurationId) {
94          final Object upgradeableDataHashMapData = data.get(KEY_ACME_CONFIGURATION_PREFIX + configurationId);
95          if (upgradeableDataHashMapData==null) {
96              return null;
97          }
98          final AcmeConfigurationtion">AcmeConfiguration acmeConfiguration = new AcmeConfiguration(upgradeableDataHashMapData);
99          acmeConfiguration.setConfigurationId(configurationId);
100         return acmeConfiguration;
101     }
102     /** @param acmeConfiguration Overwrite and update the persisted AcmeConfiguration */
103     public void updateAcmeConfiguration(final AcmeConfiguration acmeConfiguration) {
104         data.put(KEY_ACME_CONFIGURATION_PREFIX + acmeConfiguration.getConfigurationId(), acmeConfiguration.saveData());
105     }
106 
107     /** @return the validity period of newly generated nonces */
108     public long getReplayNonceValidity() {
109         final Long replayNonceValidity = (Long)data.get(KEY_REPLAY_NONCE_VALIDITY);
110         return replayNonceValidity==null ? 600*1000L : replayNonceValidity.longValue();
111     }
112     public void setReplayNonceValidity(final long replayNonceValidity) {
113         data.put(KEY_REPLAY_NONCE_VALIDITY, replayNonceValidity);
114     }
115 
116     /** @param hmacOid OID
117      * @return all replay-nonce secrets for the specified algorithm that have been configured 
118      * @throws IllegalArgumentException Bad arg 
119      * @throws IllegalStateException BAd state */
120     @SuppressWarnings("unchecked")
121     public ArrayList<String> getReplayNonceSharedSecrets(final String hmacOid) throws IllegalArgumentException, IllegalStateException {
122         final ArrayList<String> sharedSecretStrings;
123         if (PKCSObjectIdentifiers.id_hmacWithSHA256.getId().equals(hmacOid)) {
124             sharedSecretStrings = (ArrayList<String>)data.get(KEY_REPLAY_NONCE_SECRETS_HMAC_WITH_SHA256);
125         } else {
126             throw new IllegalArgumentException("Unsupported hmac algorithm '" + hmacOid + "'.");
127         }
128         if (sharedSecretStrings==null || sharedSecretStrings.isEmpty()) {
129             throw new IllegalStateException("No secret for replay protection has been configured yet for this algorithm '" + hmacOid + "'.");
130         }
131         return sharedSecretStrings;
132     }
133 
134     /** @param hmacOid OID
135      * @return the latest replay-nonce secret for the specified algorithm that have been configured 
136      * @throws IllegalArgumentException Bad arg 
137      * @throws IllegalStateException  Bad state */
138     public byte[] getReplayNonceSharedSecretCurrent(final String hmacOid) throws IllegalArgumentException, IllegalStateException {
139         final ArrayList<String> replayNonceSharedSecrets = getReplayNonceSharedSecrets(hmacOid);
140         return Hex.decode(replayNonceSharedSecrets.get(replayNonceSharedSecrets.size()-1));
141     }
142 
143     /** Add a new replay-nonce secret for the specified algorithm to use for all new generated replay-nonces 
144      * @param hmacOid OID
145      * @param secret Secret
146      * @throws IllegalArgumentException BAd arg 
147      * @throws IllegalStateException Bad state*/
148     public void addReplayNonceSharedSecret(final String hmacOid, byte[] secret) throws IllegalArgumentException, IllegalStateException {
149         // TODO: Use obfuscation during serialized transfer, but StringTools.encrypt when data is at rest
150         // TODO: Consider also adding a timestamp to when the new secret was added, to allow smooth purge of old secret during roll-over
151         final String sharedSecretString = new String(Hex.encode(secret), StandardCharsets.UTF_8);
152         if (PKCSObjectIdentifiers.id_hmacWithSHA256.getId().equals(hmacOid)) {
153             @SuppressWarnings("unchecked")
154             ArrayList<String> sharedSecretStrings = (ArrayList<String>)data.get(KEY_REPLAY_NONCE_SECRETS_HMAC_WITH_SHA256);
155             if (sharedSecretStrings==null) {
156                 sharedSecretStrings = new ArrayList<>();
157             }
158             sharedSecretStrings.add(sharedSecretString);
159             data.put(KEY_REPLAY_NONCE_SECRETS_HMAC_WITH_SHA256, sharedSecretStrings);
160         } else {
161             throw new IllegalArgumentException("Unsupported hmac algorithm '" + hmacOid + "'.");
162         }
163     }

165     public boolean aliasExists(String configurationId) {
166         if(StringUtils.isNotEmpty(configurationId)) {
167             AcmeConfiguration acmeConfiguration = getAcmeConfiguration(configurationId);
168             return acmeConfiguration != null;
169         }
170         return false;
171     }
172 
173     public void renameConfigId(String newConfigId, String oldConfigId){
174         AcmeConfiguration acmeConfiguration = getAcmeConfiguration(oldConfigId);
175         acmeConfiguration.setConfigurationId(newConfigId);
176         removeConfigId(oldConfigId);
177         updateAcmeConfiguration(acmeConfiguration);
178     }
179 
180     public void removeConfigId(String configId){
181         data.remove(KEY_ACME_CONFIGURATION_PREFIX + configId);
182     }
