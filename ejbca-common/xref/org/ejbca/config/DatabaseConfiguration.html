14  package org.ejbca.config;
15  
16  /**
17   * Parses embedded or overridden database.properties for info.
18   * 
19   * @version $Id: DatabaseConfiguration.java 22117 2015-10-29 10:53:42Z mikekushner $
20   */
21  public class DatabaseConfiguration {
22  
23  	public static final String CONFIG_DATASOURCENAME = "datasource.jndi-name";
24      public static final String CONFIG_DATABASENAME = "database.name";
25  
26  	public static String getFullDataSourceJndiName(){
27  		return InternalConfiguration.getDataSourceJndiNamePrefix() + EjbcaConfigurationHolder.getString(CONFIG_DATASOURCENAME);
28  	}
29  
30      public static String getDatabaseName(){
31          final String ret = EjbcaConfigurationHolder.getString(CONFIG_DATABASENAME);
32          if (ret==null) {
33              return "hsqldb";
34          }
35          return ret;
36      }
37  }
