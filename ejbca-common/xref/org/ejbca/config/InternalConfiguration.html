14  package org.ejbca.config;
15  
16  /**
17   * This file handles configuration from internal.properties
18   *
19   * @version $Id: InternalConfiguration.java 22117 2015-10-29 10:53:42Z mikekushner $
20   */
21  public class InternalConfiguration {
22  
23      public static final String CONFIG_APPNAME_CAPITAL = "app.name.capital";
24  	public static final String CONFIG_DATASOURCENAMEPREFIX = "datasource.jndi-name-prefix";
25  
26  	/**
27  	 * Lower case application name
28  	 * @return String
29  	 */
30  	public static String getAppNameLower() {
31  		return "ejbca";
32  	}
33  
34  	/**
35  	 * Dynamic version of getAppNameLower() for use from JSP/JSF
36  	 * @return String
37  	 */
38  	public String getAppNameLowerDynamic() {
39  		return InternalConfiguration.getAppNameLower();
40  	}
41  
42  	/**
43  	 * Upper case application name
44  	 * @return String
45  	 */
46  	public static String getAppNameCapital() {
47  		return EjbcaConfigurationHolder.getExpandedString(CONFIG_APPNAME_CAPITAL);
48  	}
49  
50  	/**
51  	 * Application version number
52  	 * @return String
53  	 */
54  	public static String getAppVersionNumber() {
55  		return EjbcaConfigurationHolder.getExpandedString("app.version.number");
56  	}
57  
58  	/**
59  	 * SVN revision
60  	 * @return String
61  	 */
62  	public static String getSvnRevision() {
63  		return EjbcaConfigurationHolder.getExpandedString("svn.revision");
64  	}
65  
66  	/**
67  	 * Full application version
68  	 * @return String
69  	 */
70  	public static String getAppVersion() {
71  		return EjbcaConfigurationHolder.getExpandedString("app.version");
72  	}
73  
74  	public static String getDataSourceJndiNamePrefix(){
75  		return EjbcaConfigurationHolder.getString(CONFIG_DATASOURCENAMEPREFIX);	// We need to return an empty string for WebLogic. "java:/" will be set anyway on JBoss.
76  	}
77  }
