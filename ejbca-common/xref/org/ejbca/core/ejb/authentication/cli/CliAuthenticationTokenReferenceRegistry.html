13  package org.ejbca.core.ejb.authentication.cli;
14  
15  import java.util.Collections;
16  import java.util.HashMap;
17  import java.util.Map;
18  
19  import org.apache.log4j.Logger;
20  import org.cesecore.authentication.AuthenticationFailedException;
21  
22  /**
23   * Due to being sent over remote, this token needs to protect against two threats: replay and interception.
24   * 
25   * Replay is stopped via the CliAuthenticationTokenReferenceRegistry, where each authentication token must register prior to its first authorization.
26   * Once authorized the token is removed from that registry, and further attempts to authorize the same token will fail. In order to allow the same
27   * token to be used several times within the local environment, the transient 'isVerified'-flag is set upon the first authorization.
28   * 
29   * Interception of this token being transmitted risks making the password hash known, which can be used to forge future tokens. To guard against this
30   * threat, the password hash is never transmitted in cleartext. Instead a SHA1 hash is calculated on server side, registered here but not transmitted
31   * back (see CliAuthenticationToken's clone method). Meanwhile, on the client side the SHA1 hash is recalculated and stored in the token, and at
32   * authorization the two are compared to make sure that the token is valid. Note that this implies that true athentication (password comparison) is
33   * not performed until the first authorization act.
34   * 
35   * 
36   * @version $Id: CliAuthenticationTokenReferenceRegistry.java 22117 2015-10-29 10:53:42Z mikekushner $
37   * 
38   */
39  public enum CliAuthenticationTokenReferenceRegistry {
40      INSTANCE;
41  
42      private static final Logger log = Logger.getLogger(CliAuthenticationTokenReferenceRegistry.class);
43  
44      private Map<Long, CliAuthenticationToken> tokenRegistry;
45  
46      private CliAuthenticationTokenReferenceRegistry() {
47          tokenRegistry = Collections.synchronizedMap(new HashMap<Long, CliAuthenticationToken>());
48      }
49  
50      /**
51       * Verifies that the sha1Hash supplied as a parameter matches the one in the token registered by the given reference number.
52       * 
53       * @param referenceNumber a reference number to a registered token.
54       * @param sha1Hash the hash to check against.
55       * @return true if the hash matches.
56       * @throws AuthenticationFailedException if an attempt is made to verify password on an non-existent or already used token
57       */
58      public boolean verifySha1Hash(Long referenceNumber, String sha1Hash) throws AuthenticationFailedException {
59          if (tokenRegistry.containsKey(referenceNumber)) {
60              return sha1Hash.equals(tokenRegistry.get(referenceNumber).getSha1Hash());
61          } else {
62              throw new AuthenticationFailedException("Attempt was made to verify password on an non-existent or already used token.");
63          }
64      }
65  
66      /**
67       * Register an authentication token to this registry.
68       * 
69       * @param token the reference number to that token.
70       */
71      public void registerToken(final CliAuthenticationToken token) {
72          CliAuthenticationToken safetyCopy = token.clone();
73          safetyCopy.setSha1Hash(token.getSha1Hash());
74          tokenRegistry.put(token.getReferenceNumber(), safetyCopy);
75          if (log.isTraceEnabled()) {
76              log.trace("Registered new CliAuthenticationToken: "+safetyCopy+", with reference number: "+safetyCopy.getReferenceNumber());
77          }
78      }
79  
80      /**
81       * Unregisters a token from the registry. Note that this method is safe from querying, since it's only available from inside the same process.
82       * 
83       * @param referenceNumber a reference number to a CLI authentication token.
84       * 
85       * @return true if token was successfully removed, false otherwise
86       */
87      public boolean unregisterToken(Long referenceNumber) {
88          if (tokenRegistry.containsKey(referenceNumber)) {
89              tokenRegistry.remove(referenceNumber);
90              if (log.isTraceEnabled()) {
91                  log.trace("Unregistered CliAuthenticationToken with reference number: "+referenceNumber);
92              }
93              return true;
94          } else {
95              return false;
96          }
97      }
98  
99  }
