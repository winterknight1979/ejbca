162     public void setDN(String dn) {
163     	final StringBuilder removedAllEmpties = new StringBuilder(dn.length());
164     	final StringBuilder removedTrailingEmpties = DNFieldsUtil.removeEmpties(dn, removedAllEmpties, true);
165     	if (removedTrailingEmpties == null) {
166         	this.subjectDNClean=StringTools.putBase64String(removedAllEmpties.toString());
167         	this.subjectDN=this.subjectDNClean;
168     	} else {
169         	this.subjectDNClean=StringTools.putBase64String(removedAllEmpties.toString());
170         	this.subjectDN=StringTools.putBase64String(removedTrailingEmpties.toString());
171     	}
172     }

278     /**
279      * Help Method used to create an ExtendedInformation from String representation.
280      * Used when creating an ExtendedInformation from queries.
281      * @param extendedinfostring Info
282      * @return Admin
283      */
284     public static ExtendedInformation getExtendedInformation(String extendedinfostring) {
285         ExtendedInformation returnval = null;
286         if ( (extendedinfostring != null) && (extendedinfostring.length() > 0) ) {
287             try (SecureXMLDecoder decoder = new SecureXMLDecoder(new java.io.ByteArrayInputStream(extendedinfostring.getBytes(StandardCharsets.UTF_8)))) {            	
288             	@SuppressWarnings("rawtypes")
289                 HashMap h = (HashMap) decoder.readObject();
290                 // Handle Base64 encoded string values
291                 @SuppressWarnings("rawtypes")
292                 HashMap data = new Base64GetHashMap(h);
293             	int type = ((Integer) data.get(ExtendedInformation.TYPE)).intValue();
294             	switch(type){
295             	  case ExtendedInformation.TYPE_BASIC :
296               		returnval = new ExtendedInformation();            	
297               		returnval.loadData(data);
298               		break;
299             	}            	
300             } catch (IOException e) {
301             	throw new RuntimeException("Problems generating extended information from String",e);
302             }
303         }
304         return returnval;
305     }

307     @SuppressWarnings({ "unchecked", "rawtypes" })
308     public static String extendedInformationToStringData(ExtendedInformation extendedinformation) throws UnsupportedEncodingException {
309     	String ret = null;
310     	if(extendedinformation != null){
311             // We must base64 encode string for UTF safety
312             HashMap a = new Base64PutHashMap();
313             a.putAll((HashMap)extendedinformation.saveData());
314             java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream();
315     	    try (XMLEncoder encoder = new XMLEncoder(baos)) {
316                 encoder.writeObject(a);
317     	    }
318             ret = baos.toString("UTF8");
319     	}
320     	return ret;
321     }

323     /**
324      * Returns the DN to be used when creating a certificate (without empty fields).
325      * If the registered DN has unused DN fields the empty ones are kept, i.e.
326      * CN=Tomas,OU=,OU=PrimeKey,C=SE. See ECA-1841 for an explanation of this.
327      * Use method getCertificateDN() to get the DN stripped from empty fields, getDN() returns DN with empty fields.
328      * @see #getDN()
329      * @return String with DN, with no empty fields, use getDN to get including empty fields
330      */
331     public String getCertificateDN() {
332     	if (subjectDNClean == null) {
333     		// This might be fetched from database serialization so we need to perform the cleaning all over again
334     		return DNFieldsUtil.removeAllEmpties(getDN());
335     	} else {
336             return StringTools.getBase64String(subjectDNClean);
337     	}
338     }

340     /** Helper method to convert this old deprecated type to the new EndEntityInformation
341      * 
342      * @return EndEntityInformation 
343      */
344     public EndEntityInformation toEndEntityInformation() {
345     	org.cesecore.certificates.endentity.ExtendedInformation newee = null;
346     	if (extendedinformation != null) {
347         	newee = new org.cesecore.certificates.endentity.ExtendedInformation();
348         	newee.loadData(extendedinformation.saveData());    		
349     	}
350     	EndEntityInformation eei = new EndEntityInformation(getUsername(), getDN(), caid, getSubjectAltName(), getEmail(), type, 
351     			endentityprofileid, certificateprofileid, tokentype, hardtokenissuerid, newee);
352     	eei.setPassword(getPassword());
353     	eei.setCardNumber(getCardNumber());
354     	eei.setStatus(status);
355     	eei.setTimeCreated(timecreated);
356     	eei.setTimeModified(timemodified);
357     	eei.setTokenType(tokentype);
358     	eei.setHardTokenIssuerId(hardtokenissuerid);
359     	return eei;
360     }
