13  package org.ejbca.core.model.approval.profile;
14  
15  import java.util.Collection;
16  import java.util.HashMap;
17  import java.util.Map;
18  import java.util.ServiceLoader;
19  
20  /**
21   * Reads in the implementations of the ApprovalProfile interface. Mainly used by GUI and CLI to list the available types of approvals. 
22   * 
23   * @version $Id: ApprovalProfilesFactory.java 24946 2016-12-21 15:17:53Z mikekushner $
24   *
25   */
26  public enum ApprovalProfilesFactory {
27      INSTANCE;
28  
29      private Map<String, ApprovalProfile> identifierToImplementationMap = new HashMap<>();
30  
31      private ApprovalProfilesFactory() {
32          ServiceLoader<ApprovalProfile> svcloader = ServiceLoader.load(ApprovalProfile.class);
33          for(ApprovalProfile type : svcloader) {
34              type.initialize();
35              identifierToImplementationMap.put(type.getApprovalProfileTypeIdentifier(), type);
36          }
37      }
38      
39      public Collection<ApprovalProfile> getAllImplementations() {
40          return identifierToImplementationMap.values();
41      }
42      
43      public ApprovalProfile getArcheType(String identifier) {
44          return identifierToImplementationMap.get(identifier).clone();
45      }
46      
47     
48  }
