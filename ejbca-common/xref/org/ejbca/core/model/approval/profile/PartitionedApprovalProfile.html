103     @Override
104     public boolean canApprovalExecute(final Collection<Approval> approvalsPerformed) throws ApprovalException, AuthenticationFailedException {
105         // Walk through all steps and their respective partitions, verify that the collection of approvals satisfies them.
106         ApprovalStep step = getFirstStep();
107         while(step != null) {
108             if(!isStepSatisfied(step, approvalsPerformed)) {
109                 return false;
110             }
111             step = getStep(step.getNextStep());
112             if(step != null && step.equals(getFirstStep())) {
113                 throw new IllegalStateException("Approval steps have begun referencing each other as a circular array.");
114             }
115         }
116         return true;
117     }

119     @Override
120     public int getOrdinalOfStepBeingEvaluated(Collection<Approval> approvalsPerformed) throws AuthenticationFailedException {
121         ApprovalStep step = getFirstStep();
122         int i = 1;
123         while(step != null) {
124             if(!isStepSatisfied(step, approvalsPerformed)) {
125                 return i;
126             } else {
127                 i++;
128                 step = getStep(step.getNextStep());
129                 if (step != null && step.equals(getFirstStep())) {
130                     throw new IllegalStateException("Approval steps have begun referencing each other as a circular array.");
131                 }
132             }
133         }
134         return -1;
135     }

153     @Override
154     public boolean canApprovePartition(final AuthenticationToken authenticationToken, final ApprovalPartition approvalPartition) throws AuthenticationFailedException {
155         if(approvalPartition != null) {
156             @SuppressWarnings("unchecked")
157             List<RoleInformation> roles = (List<RoleInformation>) approvalPartition.getProperty(PROPERTY_ROLES_WITH_APPROVAL_RIGHTS).getValues();
158             for (RoleInformation role : roles) {
159                 if (log.isTraceEnabled()) {
160                     log.trace("Checking if authenticationToken '"+authenticationToken+"' matches role "+role.getName());
161                 }
162                 if (role.equals(ANYBODY)) {
163                     return true;
164                 } else {
165                     // Check if authenticationToken matches any of the AccessUserAspects that existed in the Role when the ApprovalProfile was saved.
166                     for (final AccessUserAspect accessUserAspect : role.getAccessUserAspects()) {
167                         if (authenticationToken.matches(accessUserAspect)) {
168                             return true;
169                         }
170                     }
171                 }
172             }
173         } else {
174             if (log.isTraceEnabled()) {
175                 log.trace("Approval partition is null, canApprovePartition returns false for authenticationToken "+authenticationToken);
176             }
177             return false;
178         }
179         if (log.isDebugEnabled()) {
180             log.debug("Administrator '"+authenticationToken+"' does not belong to a role that can approve partition "+approvalPartition.getPartitionIdentifier());
181         }
182         return false;
183     }

208     @Override
209     public boolean canViewPartition(AuthenticationToken authenticationToken, ApprovalPartition approvalPartition)
210             throws AuthenticationFailedException {
211         boolean result = false;
212         @SuppressWarnings("unchecked")
213         List<RoleInformation> roles = (List<RoleInformation>) approvalPartition.getProperty(PROPERTY_ROLES_WITH_VIEW_RIGHTS).getValues();
214         for (RoleInformation role : roles) {
215             if (role.equals(ANYBODY)) {
216                 result = true;
217             } else {
218                 // Check if authenticationToken matches any of the AccessUserAspects that existed in the Role when the ApprovalProfile was saved.
219                 for (final AccessUserAspect accessUserAspect : role.getAccessUserAspects()) {
220                     if (authenticationToken.matches(accessUserAspect)) {
221                         result = true;
222                     }
223                 }
224             }
225         }
226         return result || canApprovePartition(authenticationToken, approvalPartition);
227     }

229     @Override
230     public int getRemainingApprovals(Collection<Approval> approvalsPerformed) {
231         //Return the total number of partitions lacking approval, minus the number of approvals performed (presume that no approvals performed overlap)
232         int remainingApprovalsInAllPartitions = 0;
233         for (final ApprovalStep approvalStep : getSteps().values()) {
234             for (final ApprovalPartition approvalPartition : approvalStep.getPartitions().values()) {
235                 int remainingApprovalsInPartition = getRemainingApprovalsInPartition(approvalsPerformed, approvalStep.getStepIdentifier(),
236                         approvalPartition.getPartitionIdentifier());
237                 if (remainingApprovalsInPartition >= 0) {
238                     remainingApprovalsInAllPartitions += remainingApprovalsInPartition;
239                 } else {
240                     return ApprovalDataVO.STATUS_REJECTED;
241                 }
242             }
243         }
244         return remainingApprovalsInAllPartitions;
245     }

266         //Will make this property into a multi-select instead of single select.

268         //Tell whatever bean is using this property to fill it with authorized roles.

272         //Add roles with view rights, with "Anybody" as the default Role.

274                 new HashSet<RoleInformation>());
275         //Will make this property into a multi-select instead of single select.

277         //Tell whatever bean is using this property to fill it with authorized roles.

279         approvalPartition.addProperty(viewRoles);
