143         ApprovalProfile clone;
144         try {
145             clone = (ApprovalProfile) getType().getConstructor().newInstance();
146         } catch (InstantiationException | IllegalAccessException | NoSuchMethodException | InvocationTargetException e) {
147             throw new IllegalStateException("Could not instansiate class of type " + getType().getCanonicalName());
148         }
149         clone.setProfileName(getProfileName());
150         clone.setProfileId(getProfileId());
151 
152         // We need to make a deep copy of the hashmap here
153         LinkedHashMap<Object, Object> dataMap = new LinkedHashMap<>(data.size());
154         for (final Entry<Object, Object> entry : data.entrySet()) {
155             Object value = entry.getValue();
156             if (value instanceof ArrayList<?>) {
157                 // We need to make a clone of this object, but the stored immutables can still be referenced
158                 value = ((ArrayList<?>) value).clone();
159             }
160             dataMap.put(entry.getKey(), value);
161         }
162         clone.setDataMap(dataMap);
163         Map<Integer, ApprovalStep> stepClone = new HashMap<>();
164         for (ApprovalStep approvalStep : getSteps().values()) {
165             stepClone.put(approvalStep.getStepIdentifier(), new ApprovalStep(approvalStep));
166         }
167         clone.setSteps(stepClone);
168         clone.setFirstStep(getFirstStepId());
169         return clone;

307         if (isStepSizeFixed()) {

322         saveTransientObjects();

365     @Override
366     public void deleteStep(final int approvalStepIdentifier) {
367         if(isStepSizeFixed()) {
368             throw new NonModifiableApprovalProfileException("Cannot delete an approval step in a profile with fixed step size");
369         }
370         ApprovalStep stepToDelete = getStep(approvalStepIdentifier);
371         ApprovalStep previousStep = getStep(stepToDelete.getPreviousStep());
372         ApprovalStep nextStep = getStep(stepToDelete.getNextStep());
373         if(previousStep == null) {
374             // Handle deleting the last sequence, in which case there is no next step. In this case we can't set "first step" here,
375             // but in the end of this method if there are no steps we initialize, which will recreate the first step in a default manner.
376             if (nextStep != null) {
377                 //This step was first, so set the next one first
378                 setFirstStep(nextStep.getStepIdentifier());
379             }
380         }
381         if(nextStep == null && previousStep != null) {
382             //This was the last step, so make sure the previous step knows it's now last
383             previousStep.setNextStep(null);
384         }
385         if(nextStep != null && previousStep != null) {
386             previousStep.setNextStep(nextStep.getStepIdentifier());
387             nextStep.setPreviousStep(previousStep.getStepIdentifier());
388         }
389         getSteps().remove(approvalStepIdentifier);
390         if (getSteps().isEmpty()) {
391             // We have removed all steps, re-initialize to default
392             initialize();
393         }
394         saveTransientObjects();
395     }

475     @Override
476     public boolean isApprovalAuthorized(Collection<Approval> approvalsPerformed, Approval approval) throws AuthenticationFailedException {
477         ApprovalStep previousStep = getFirstStep();
478         ApprovalStep relevantStep = getStep(approval.getStepId());
479         while(previousStep != null) {
480             if(!previousStep.equals(relevantStep)) {
481                 if(!isStepSatisfied(previousStep, approvalsPerformed)) {
482                     return false;
483                 } else {
484                     previousStep = getStep(previousStep.getNextStep());
485                 }
486             } else {
487               //Verify that all previous steps are good
488                 ApprovalPartition approvalPartition = relevantStep.getPartition(approval.getPartitionId());
489                 if(approvalPartition == null) {
490                     return false;
491                 }
492                 return canApprovePartition(approval.getAdmin(), approvalPartition);
493 
494             }
495         }
496         return false;
497     }

506             throws AuthenticationFailedException {

517             return false;
518         }

520         return true;

543     public int getRemainingApprovalsInPartition(final Collection<Approval> approvalsPerformed, final int stepIdentifier, final int partitionIdentifier) {
544         int partitionApprovalsRequired = getNumberOfApprovalsRequired(stepIdentifier, partitionIdentifier);
545         int partitionApprovalsPerformed = 0;
546         for (Approval approval : approvalsPerformed) {
547             if (!approval.isApproved()) {
548                 return -1;
549             }
550             if (approval.getStepId() == stepIdentifier && approval.getPartitionId() == partitionIdentifier) {
551                 partitionApprovalsPerformed++;
552             }
553         }
554         // Don't return a negative number, could happen if the partition has been approved multiple times
555         int diff = partitionApprovalsRequired - partitionApprovalsPerformed;
556         return diff < 0 ? 0 : diff;
557     }

573     @Override
574     public void switchStepOrder(Integer firstStepIdentifier, Integer secondStepIdentifier) {
575         if(firstStepIdentifier == null || secondStepIdentifier == null) {
576             return;
577         }
578         ApprovalStep firstStep = getStep(firstStepIdentifier);
579         ApprovalStep secondStep = getStep(secondStepIdentifier);
580         Integer firstStepPrevious = firstStep.getPreviousStep();
581         Integer secondStepNext = secondStep.getNextStep();
582         if(firstStepPrevious != null) {
583             ApprovalStep previousStep = getStep(firstStepPrevious);
584             previousStep.setNextStep(secondStepIdentifier);
585         }
586         secondStep.setPreviousStep(firstStepPrevious);
587         secondStep.setNextStep(firstStepIdentifier);
588         if(secondStepNext != null) {
589             ApprovalStep nextStep = getStep(secondStepNext);
590             nextStep.setPreviousStep(firstStepIdentifier);
591         }
592         firstStep.setPreviousStep(secondStepIdentifier);
593         firstStep.setNextStep(secondStepNext);
594         if(getFirstStepId() == firstStepIdentifier) {
595             setFirstStep(secondStepIdentifier);
596         }
597         saveTransientObjects();
598     }

602         boolean changed = false;
603         final Map<Integer,ApprovalStep> steps = getSteps();
604         for (final ApprovalStep step : new ArrayList<>(steps.values())) {
605             final Map<Integer,ApprovalPartition> partitions = step.getPartitions();
606             for (final ApprovalPartition partition : new ArrayList<>(partitions.values())) {
607                 // Check if the role user aspect datas need updating
608                 final DynamicUiProperty<? extends Serializable> prop = partition.getProperty(PartitionedApprovalProfile.PROPERTY_ROLES_WITH_APPROVAL_RIGHTS);
609                 if (prop != null) {
610                     @SuppressWarnings("unchecked")
611                     final List<RoleInformation> values = (List<RoleInformation>)prop.getValues();
612                     boolean propertyChanged = false;
613                     for (final RoleInformation role : values) {
614                         final List<AccessUserAspectData> userAspects = role.getAccessUserAspects();
615                         for (final AccessUserAspectData userAspect : userAspects) {
616                             if (userAspect.getCaId() == fromId) {
617                                 userAspect.setCaId(toId);
618                                 propertyChanged = true;
619                             }
620                         }
621                     }
622                     if (propertyChanged) {
623                         // Update the property
624                         addPropertyToPartition(step.getStepIdentifier(), partition.getPartitionIdentifier(), prop);
625                         changed = true;
626                     }
627                 }
628             }
629         }
630         return changed;
631     }
