87      /**
88      * Clone has to be implemented instead of a copy constructor due to the fact that we'll be referring to implementations by this interface only.

92      ApprovalProfile clone();

94      /**
95       * Method to give an approval implementation a chance to skip out if it's not configured to perform any actions.
96       *
97       * @return true if approval is relevant and should be used.
98       */
99      boolean isApprovalRequired();
100 
101     /**
102      * @return true if an administrator is able to approve a request after editing it, without approval from an additional administrator.
103      */
104     boolean getAllowSelfEdit();
105 
106     /**
107      * @param allowSelfEdit an administrator should be able to approve a request after editing it, without approval from an additional administrator
108      */
109     void setAllowSelfEdit(boolean allowSelfEdit);
110 
111     /**
112      * @param approvalsPerformed a Collection of approvals already performed.
113      *
114      * @return true if this approval profile's criteria are fulfilled, allowing the approval to pass
115      * @throws ApprovalException if approval fails
116      * @throws AuthenticationFailedException if any of the authentication tokens in the approval collection were faulty
117      */
118     boolean canApprovalExecute(final Collection<Approval> approvalsPerformed) throws ApprovalException, AuthenticationFailedException;
119 
120     /**
121      *
122      * @param approvalsPerformed the approvals performed against this profile
123      * @return the number of remaining approvals, or -1 if any approval is denied
124      */
125     int getRemainingApprovals(final Collection<Approval> approvalsPerformed);
126 
127     /**
128      * @return true if the amount of sequences of this profile is fixed, false if it's dynamic
129      */
130     boolean isStepSizeFixed();
131 
132     /**
133      * @return true if it's possible to add fields to the partitions of this profile
134      */
135     boolean arePartitionsFixed();

137     Map<Integer, ApprovalStep> getSteps();

139     /**
140      * Adds a step without modifying any order. Without setting order, this step will not be handled.
141      *
142      * @param step an ApprovalStep
143      */
144     void addStep(final ApprovalStep step);
145 
146     /**
147      * Creates a new step and adds it first.
148      *
149      * @return the new step
150      */
151     ApprovalStep addStepFirst();
152 
153     /**
154      * Creates a new step and adds it last.
155      *
156      * @return the new step
157      */
158     ApprovalStep addStepLast();
159 
160     /**
161      * Deletes a step and attaches the steps before and after to each other in order.
162      *
163      * @param approvalStepIdentifier the identifier of the approval step
164      */
165     void deleteStep(final int approvalStepIdentifier);
166 
167     void setSteps(final Map<Integer, ApprovalStep> steps);
168 
169     /**
170      * Switches position betwen two steps
171      *
172      * @param firstStepIdentifier the ID of the first step
173      * @param secondStepIdentifier the ID of the second step
174      */
175     void switchStepOrder(Integer firstStepIdentifier, Integer secondStepIdentifier);
176 
177     /**
178     * Adds a property to a specific partition in a specific sequence in this approval profile, for display in the UI. If the property already
179     * exists, it will be overwritten
180     *
181     * @param stepId the identifier of the step
182     * @param partitionId the ID of a partition in the step
183     * @param property a DynamicUiProperty
184      * @throws NoSuchApprovalStepException if the step specified by stepId wasn't found.
185     */
186     void addPropertyToPartition(final int stepId, final int partitionId, final DynamicUiProperty<? extends Serializable> property) throws NoSuchApprovalStepException;
187 
188     /**
189      * Removes a property from a partition. Will do nothing if property was predefined in the template.
190      *
191      * @param stepId the identifier of the step
192      * @param partitionId the ID of a partition in the step
193      * @param propertyName the name of the property.
194      */
195     void removePropertyFromPartition(final int stepId, final int partitionId, final String propertyName);
196 
197     /**
198      * Adds a partition to this sequence
199      *
200      * @param stepIdentifier the identifier of the sequence
201      *
202      * @return the partition, with a generated ID
203      */
204      ApprovalPartition addPartition(final int stepIdentifier);
205 
206      /**
207       *
208       * @param stepId the identifier of the step
209       * @param partitionId the ID of a partition in the step
210       * @param properties a list of DynamicUiProperties
211       * @throws NoSuchApprovalStepException if the step identified by stepId didn't exist
212       */
213      void addPropertiesToPartition(final Integer stepId, final int partitionId, final Collection< DynamicUiProperty<? extends Serializable>> properties) throws NoSuchApprovalStepException;

242     /**

244      *

248      * @throws AuthenticationFailedException if the authentication token in the approval wasn't valid
249      */
250     boolean isApprovalAuthorized(final Collection<Approval> approvalsPerformed, final Approval approval) throws AuthenticationFailedException;

253      * @return the number of steps in this profile
254      */
255     int getNumberOfSteps();

260      * @return the ordinal of the step currently being evaluated, given the performed approvals
261      * @throws AuthenticationFailedException if the authentication of the approvals failed
262      */
263     int getOrdinalOfStepBeingEvaluated(final Collection<Approval> approvalsPerformed) throws AuthenticationFailedException;
264 
265     /**
266      * Returns the first step which hasn't been fully evaluated by the given collection of approvals, or null if all steps
267      * have been evaluated.
268      *
269      * @param approvalsPerformed approvalsPerformed a list of performed approvals
270      * @return the step currently being evaluated, given the performed approvals, or null if all steps have been evaluated.
271      * @throws AuthenticationFailedException if the authentication of the approvals failed
272      */
273     ApprovalStep getStepBeingEvaluated(final Collection<Approval> approvalsPerformed) throws AuthenticationFailedException;
274 
275     /**
276      * Tests if an administrator can approve a particular partition
277      *
278      * @param authenticationToken an authentication token
279      * @param approvalPartition an approval partition from an approval step
280      * @return true if administrator has approval rights
281      * @throws AuthenticationFailedException if the authentication token in the approval doesn't check out
282      */
283     boolean canApprovePartition(final AuthenticationToken authenticationToken, final ApprovalPartition approvalPartition) throws AuthenticationFailedException;
284 
285     /**
286      * Tests if an administrator can view a particular partition. Approval rights automatically count as view rights.
287      *
288      * @param authenticationToken an authentication token
289      * @param approvalPartition an approval partition from an approval step
290      * @return true if administrator has view or approval rights
291      * @throws AuthenticationFailedException if the authentication token in the approval doesn't check out
292      */
293     boolean canViewPartition(final AuthenticationToken authenticationToken, final ApprovalPartition approvalPartition) throws AuthenticationFailedException;
294 
295     /**
296      * Returns true if the given partition has been configured to allow any administrator to approve it.

301 
302     /**
303      * Returns the list of roles which have been configured in the given partition to be allowed to approve it.

355     /**
356      * Allows for querying a partition of a certain property was defined procedurally.

370     /**
371      * Return the number of required approvals of the specified partition that have not yet been approved
372      *

381     /**

388      */
389     boolean updateCAIds(final int fromId, final int toId, final String toSubjectDN);
