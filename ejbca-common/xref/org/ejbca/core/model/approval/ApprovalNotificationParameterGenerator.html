13  package org.ejbca.core.model.approval;
14  
15  import org.ejbca.core.model.approval.profile.ApprovalPartitionWorkflowState;
16  import org.ejbca.util.NotificationParamGen;
17  
18  
19  /**
20   * Parameters for notification of partition owners in the approval request work flow.
21   * 
22   * ${approvalRequest.ID}            The approval request identifier.
23   * ${approvalRequest.STEP_ID}       The approval step that this notification concerns.
24   * ${approvalRequest.PARTITION_ID}  The approval partition in the step that this notification concerns.
25   * ${approvalRequest.PARTITION_NAME} The approval partition in the step that this notification concerns.
26   * ${approvalRequest.TYPE}          The type of approval request.
27   * ${approvalRequest.WORKFLOWSTATE} The work flow state from the perspective of the one(s) responsible for handling the partition.
28   * ${approvalRequest.REQUESTOR}     The human readable version of the authentication token that was used to create the request.
29   * ${approvalRequest.APPROVALADMIN} The human readable version of the authentication token that was used to last approved the request, if any.
30   * 
31   * @see ApprovalPartitionWorkflowState
32   * @version $Id: ApprovalNotificationParameterGenerator.java 24431 2016-09-28 12:10:34Z anatom $
33   */
34  public class ApprovalNotificationParameterGenerator extends NotificationParamGen {
35  
36      /**
37       * 
38       * @param approvalRequestID approval request ID, is the ID of the ApprovalData
39       * @param approvalStepId Step
40       * @param approvalPartitionId Partition
41       * @param approvalPartitionName Name
42       * @param approvalType one of the ApprovalDataVO.APPROVALTYPENAMES
43       * @param workflowState "APPROVAL_WFSTATE_" + approvalPartitionWorkflowState.name(), where approvalPartitionWorkflowState.name() typically is "approved", "rejected"
44       * @param requestor AuthenticationToken.toString() of the admin who created the approval request
45       * @param lastApprovedBy AuthenticationToken.toString() of the admin who last approved the request, or empty string/null if none
46       */
47      public ApprovalNotificationParameterGenerator(final int approvalRequestID, final int approvalStepId, final int approvalPartitionId,
48              final String approvalPartitionName, final String approvalType, final String workflowState, final String requestor, final String lastApprovedBy) {
49          paramPut("approvalRequest.ID", approvalRequestID);
50          paramPut("approvalRequest.STEP_ID", approvalStepId);
51          paramPut("approvalRequest.PARTITION_ID", approvalPartitionId);
52          paramPut("approvalRequest.PARTITION_NAME", approvalPartitionName);
53          paramPut("approvalRequest.TYPE", approvalType);
54          paramPut("approvalRequest.WORKFLOWSTATE", workflowState);
55          paramPut("approvalRequest.REQUESTOR", requestor);
56          paramPut("approvalRequest.APPROVALADMIN", lastApprovedBy);
57      }
58  
59  }
