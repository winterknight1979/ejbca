115     /**
116      * Writes certificate to temporary file and executes an external command with
117      * the full pathname of the temporary file as argument. The temporary file
118      * is the encoded form of the certificate e.g. X.509 certificates would be
119      * encoded as ASN.1 DER. All parameters but incert are ignored.
120      * 
121      * @param incert
122      *            The certificate
123      * @param username
124      *            The username
125      * @param type
126      *            The certificate type
127      * 
128      * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#storeCertificate
129      */
130     @Override
131     public boolean storeCertificate(AuthenticationToken admin, Certificate incert, String username, String password, String userDN, String cafp, int status, int type, long revocationDate,
132             int revocationReason, String tag, int certificateProfileId, long lastUpdate, ExtendedInformation extendedinformation) throws PublisherException {
133         if (log.isTraceEnabled()) {
134             log.trace(">storeCertificate, Storing Certificate for user: " + username);
135         }
136 
137         if (status == CertificateConstants.CERT_REVOKED) {
138             // Call separate script for revocation
139             revokeCertificate(admin, incert, revocationReason);
140         } else if (status == CertificateConstants.CERT_ACTIVE) {
141             // Don't publish non-active certificates
142             // Make sure that an external command was specified
143             if (certExternalCommandFileName == null) {
144                 String msg = intres.getLocalizedMessage("publisher.errormissingproperty", CERT_EXTERNAL_COMMAND_PROPERTY_NAME);
145                 log.error(msg);
146                 throw new PublisherException(msg);
147             }
148             // Run internal method to create tempfile and run the command
149             List<String> arguments = new ArrayList<>();
150             arguments.add(String.valueOf(type));
151             try {
152                 arguments.add(CertTools.getSubjectDN(incert));
153                 arguments.add(CertTools.getIssuerDN(incert));
154                 arguments.add(CertTools.getSerialNumberAsString(incert));
155                 ExternalProcessTools.launchExternalCommand(certExternalCommandFileName, incert.getEncoded(), 
156                         certFailOnErrorCode, certFailOnStandardError, arguments, "GeneralPurposeCustomPublisher");
157             } catch (CertificateEncodingException e) {
158                 String msg = intres.getLocalizedMessage("publisher.errorcertconversion");
159                 log.error(msg);
160                 throw new PublisherException(msg);
161             } catch (ExternalProcessException e) {
162                 throw new PublisherException(e.getMessage());
163             }
164         }
165         if (log.isTraceEnabled()) {
166             log.trace("<storeCertificate");
167         }
168         return true;
169     } // storeCertificate

171     /**
172      * Writes the CRL to a temporary file and executes an external command with
173      * the temporary file as argument. By default, a PublisherException is
174      * thrown if the external command returns with an errorlevel or outputs to
175      * stderr.
176      * 
177      * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#storeCRL
178      */
179     @Override
180     public boolean storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String userDN) throws PublisherException {
181         if (log.isTraceEnabled()) {
182         	log.trace(">storeCRL, Storing CRL");
183         }
184         // Verify initialization
185         if (crlExternalCommandFileName == null) {
186             String msg = intres.getLocalizedMessage("publisher.errormissingproperty", CRL_EXTERNAL_COMMAND_PROPERTY_NAME);
187             log.error(msg);
188             throw new PublisherException(msg);
189         }
190 
191         List<String> additionalArguments = new ArrayList<>();
192 
193         if (calclulateDeltaCrlLocally) {
194             X509CRL crl;
195             try {
196                 crl = CertTools.getCRLfromByteArray(incrl);
197                 additionalArguments.add(Boolean.toString(crl.getExtensionValue(Extension.deltaCRLIndicator.getId()) != null));
198             } catch (CRLException e) {
199                 log.error("Byte array does not contain a correct CRL.", e);
200             }
201         }
202 
203         // Write temporary file and run the external script / command.
204         try {
205             ExternalProcessTools.launchExternalCommand(crlExternalCommandFileName, incrl, 
206                 crlFailOnErrorCode, crlFailOnStandardError, additionalArguments, "GeneralPurposeCustomPublisher");
207         } catch (ExternalProcessException e) {
208             throw new PublisherException(e.getMessage());
209         }
210         if (log.isTraceEnabled()) {
211         	log.trace("<storeCRL");
212         }
213         return true;
214     }

216     /**
217      * Writes certificate to temporary file and executes an external command
218      * with the full pathname of the temporary file as argument. The temporary
219      * file is the encoded form of the certificate e.g. X.509 certificates would
220      * be encoded as ASN.1 DER. All parameters but cert are ignored.
221      * @param admin Token
222      * 
223      * @param cert
224      *            The certificate
225      * @param reason Reason
226      * @throws PublisherException On error
227      * 
228      */
229     public void revokeCertificate(AuthenticationToken admin, Certificate cert, int reason) throws PublisherException {
230         if (log.isTraceEnabled()) {
231         	log.trace(">revokeCertificate, Rekoving Certificate");
232         }
233         // Verify initialization
234         if (revokeExternalCommandFileName == null) {
235             String msg = intres.getLocalizedMessage("publisher.errormissingproperty", REVOKE_EXTERNAL_COMMAND_PROPERTY_NAME);
236             log.error(msg);
237             throw new PublisherException(msg);
238         }
239         // Run internal method to create tempfile and run the command
240         List<String> arguments = new ArrayList<>();
241         arguments.add(String.valueOf(reason));
242         try {
243             arguments.add(CertTools.getSubjectDN(cert));
244             arguments.add(CertTools.getIssuerDN(cert));
245             arguments.add(CertTools.getSerialNumberAsString(cert));
246             ExternalProcessTools.launchExternalCommand(revokeExternalCommandFileName, cert.getEncoded(), 
247                     revokeFailOnErrorCode, revokeFailOnStandardError, arguments, "GeneralPurposeCustomPublisher");
248         } catch (CertificateEncodingException e) {
249             String msg = intres.getLocalizedMessage("publisher.errorcertconversion");
250             log.error(msg);
251             throw new PublisherException(msg);
252         } catch (ExternalProcessException e) {
253             throw new PublisherException(e.getMessage());
254         }
255         if (log.isTraceEnabled()) {
256         	log.trace("<revokeCertificate");
257         }
258     } // revokeCertificate

260     /**
261      * Check if the specified external executable file(s) exist.
262      * 
263      * @see org.ejbca.core.model.ca.publisher.ICustomPublisher#testConnection()
264      */
265     @Override
266     public void testConnection() throws PublisherConnectionException {
267         if (log.isTraceEnabled()) {
268         	log.trace("testConnection, Testing connection");
269         }
270         // Test if specified commands exist
271         if (StringUtils.isNotBlank(crlExternalCommandFileName)) {
272             if (!(new File(crlExternalCommandFileName)).exists()) {
273                 String msg = intres.getLocalizedMessage("publisher.commandnotfound", crlExternalCommandFileName);
274                 log.error(msg);
275                 throw new PublisherConnectionException(msg);
276             }
277         }
278         if (StringUtils.isNotBlank(certExternalCommandFileName)) {
279             if (!(new File(certExternalCommandFileName)).exists()) {
280                 String msg = intres.getLocalizedMessage("publisher.commandnotfound", certExternalCommandFileName);
281                 log.error(msg);
282                 throw new PublisherConnectionException(msg);
283             }
284         }
285         if (StringUtils.isNotBlank(revokeExternalCommandFileName)) {
286             if (!(new File(revokeExternalCommandFileName)).exists()) {
287                 String msg = intres.getLocalizedMessage("publisher.commandnotfound", revokeExternalCommandFileName);
288                 log.error(msg);
289                 throw new PublisherConnectionException(msg);
290             }
291         }
292     } // testConnection
