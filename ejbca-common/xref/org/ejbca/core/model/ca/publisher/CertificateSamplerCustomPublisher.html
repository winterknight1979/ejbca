27  /**
28   * Publisher storing certificates to a specified folder.
29   * 
30   * The configured sampling method and its parameters (if any) determines which 
31   * certificates that should be sampled.
32   * 
33   * @see SamplingMethod
34   * @version $Id: CertificateSamplerCustomPublisher.java 23625 2016-06-07 19:32:11Z anatom $
35   */
36  public class CertificateSamplerCustomPublisher implements ICustomPublisher {
37      private static final Logger LOG = Logger.getLogger(CertificateSamplerCustomPublisher.class);
38      private static final InternalEjbcaResources intres = InternalEjbcaResources.getInstance();
39  
40      private static final String PROPERTY_OUTPUTFOLDER = "outputfolder";
41      
42      private static final String PROPERTYPREFIX_PROFILEID = "profileid.";
43      
44      private static final String PROPERTYSUFFIX_PVALUE = ".pvalue";
45      private static final String PROPERTYSUFFIX_SAMPLINGMETHOD = ".samplingmethod";
46      
47      private static final String PROPERTY_DEFAULT_SAMPLINGMETHOD = "default" + PROPERTYSUFFIX_SAMPLINGMETHOD;
48      private static final String PROPERTY_DEFAULT_PVALUE = "default" + PROPERTYSUFFIX_PVALUE;
49      
50      private Properties config;
51      
52      @Override
53      public void init(Properties config) {
54      	if (LOG.isTraceEnabled()) {
55      		LOG.trace(">init: " + this);
56      	}
57          this.config = config;

60      @Override
61      public void testConnection() throws PublisherConnectionException {
62          if (LOG.isTraceEnabled()) {
63          	LOG.trace(">testConnection, Testing connection");
64          }
65          
66          // Test output folder
67          File outputFolder;
68          try {
69              outputFolder = getOutputFolder();
70          } catch (PublisherException ex) {
71              LOG.error(null, ex);
72              throw new PublisherConnectionException(ex.getLocalizedMessage());
73          }
74          if (!outputFolder.exists() || !outputFolder.isDirectory()) {
75              final String msg = intres.getLocalizedMessage("publisher.erroroutputpath", outputFolder.getAbsolutePath());
76              LOG.error(msg);
77              throw new PublisherConnectionException(msg);
78          }
79          
80          // Test default samplingmethod
81          SamplingMethod defaultMethod = null;
82          try {
83              String methodString = config.getProperty(PROPERTY_DEFAULT_SAMPLINGMETHOD);
84              if (methodString == null) {
85                  final String msg = intres.getLocalizedMessage("publisher.errormissingproperty", PROPERTY_DEFAULT_SAMPLINGMETHOD);
86                  LOG.error(msg);
87                  throw new PublisherConnectionException(msg);
88              } else {
89                  defaultMethod = SamplingMethod.valueOf(methodString);
90              }
91          } catch (IllegalArgumentException ex) {
92              final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", PROPERTY_DEFAULT_SAMPLINGMETHOD, ex.getLocalizedMessage());
93              LOG.error(msg, ex);
94              throw new PublisherConnectionException(msg);
95          }
96          
97          // Test default pvalue
98          final String pvalueString = config.getProperty(PROPERTY_DEFAULT_PVALUE);
99          if (pvalueString == null) {
100             if (SamplingMethod.SAMPLE_PROBABILISTIC.equals(defaultMethod)) {
101                 final String msg = intres.getLocalizedMessage("publisher.errormissingproperty", PROPERTY_DEFAULT_PVALUE);
102                 LOG.error(msg);
103                 throw new PublisherConnectionException(msg);
104             }
105         } else {
106             try {
107                 final double defaultPvalue = Double.parseDouble(pvalueString);
108                 if (defaultPvalue < 0.0 || defaultPvalue > 1.0) {
109                     final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", PROPERTY_DEFAULT_PVALUE, intres.getLocalizedMessage("publisher.pvalueinterval"));
110                     LOG.error(msg);
111                     throw new PublisherConnectionException(msg);
112                 }
113             } catch (NumberFormatException ex) {
114                 final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", PROPERTY_DEFAULT_PVALUE, ex.getLocalizedMessage());
115                 LOG.error(msg, ex);
116                 throw new PublisherConnectionException(msg);
117             }
118         }
119         
120         // Test every profile specific values
121         for (String name : config.stringPropertyNames()) {
122             if (name.startsWith(PROPERTYPREFIX_PROFILEID)) {
123                 if (name.endsWith(PROPERTYSUFFIX_SAMPLINGMETHOD) && name.length() > PROPERTYPREFIX_PROFILEID.length() + PROPERTYSUFFIX_SAMPLINGMETHOD.length()) {
124                     final String profileIdString = name.substring(PROPERTYPREFIX_PROFILEID.length(), name.indexOf(PROPERTYSUFFIX_SAMPLINGMETHOD));
125                     int profileId;
126                     try {
127                         profileId = Integer.parseInt(profileIdString);
128                     } catch (NumberFormatException ex) {
129                         final String msg = intres.getLocalizedMessage("publisher.errorinvalidkey", name);
130                         LOG.error(msg, ex);
131                         throw new PublisherConnectionException(msg);
132                     }
133                     try {
134                         SamplingMethod profileMethod = SamplingMethod.valueOf(config.getProperty(name));
135                         
136                         // Check that there is an pvalue for this profile
137                         if (SamplingMethod.SAMPLE_PROBABILISTIC.equals(profileMethod)) {
138                             getPvalue(profileId); // Throws exception if the value (or default value) does not exist
139                         }
140                     } catch (PublisherException ex) {
141                         throw new PublisherConnectionException(ex.getLocalizedMessage());
142                     } catch (IllegalArgumentException ex) {
143                         final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", name, ex.getLocalizedMessage());
144                         LOG.error(msg, ex);
145                         throw new PublisherConnectionException(msg);
146                     }
147                 } else if (name.endsWith(PROPERTYSUFFIX_PVALUE) && name.length() > PROPERTYPREFIX_PROFILEID.length() + PROPERTYSUFFIX_PVALUE.length()) {
148                     final String profileIdString = name.substring(PROPERTYPREFIX_PROFILEID.length(), name.indexOf(PROPERTYSUFFIX_PVALUE));
149                     try {
150                         Integer.parseInt(profileIdString);
151                     } catch (NumberFormatException ex) {
152                         final String msg = intres.getLocalizedMessage("publisher.errorinvalidkey", name);
153                         LOG.error(msg, ex);
154                         throw new PublisherConnectionException(msg);
155                     }
156                     try {
157                         final double pvalue = Double.parseDouble(config.getProperty(name));
158                         if (pvalue < 0.0 || pvalue > 1.0) {
159                             final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", name, intres.getLocalizedMessage("publisher.pvalueinterval"));
160                             LOG.error(msg);
161                             throw new PublisherConnectionException(msg);
162                         }
163                     } catch (NumberFormatException ex) {
164                         final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", name, ex.getLocalizedMessage());
165                         LOG.error(msg, ex);
166                         throw new PublisherConnectionException(msg);
167                     }
168                 }
169             }
170         }
171     }

173     @Override
174     public boolean storeCRL(AuthenticationToken admin, byte[] incrl, String cafp, int number, String userDN) throws PublisherException {
175         if (LOG.isTraceEnabled()) {
176         	LOG.trace("CRL sampling is not supported");
177         }
178         return true; // This ICustomPublisher does not care about CRLs and chooses to simply return SUCCESS for those.
179     }

181     @Override
182     public boolean storeCertificate(AuthenticationToken admin, Certificate incert, String username, String password, String userDN, String cafp, int status, int type, long revocationDate, int revocationReason, String tag, int certificateProfileId, long lastUpdate, org.cesecore.certificates.endentity.ExtendedInformation extendedinformation) throws PublisherException {
183         if (LOG.isTraceEnabled()) {
184             LOG.trace(">storeCertificate, Storing Certificate of profileid: " + certificateProfileId);
185         }
186         if (status == CertificateConstants.CERT_ACTIVE) {
187             // Sample the certificate if it is time for that
188             if (isTimeToSample(certificateProfileId)) {
189                 if (LOG.isTraceEnabled()) {
190                     LOG.trace("Sample");
191                 }
192                 writeCertificate(incert, getOutputFolder(), username + "-", ".crt");
193             } else {
194                 if (LOG.isTraceEnabled()) {
195                     LOG.trace("No sampling");
196                 }
197             }
198         }
199         if (LOG.isTraceEnabled()) {
200             LOG.trace("<storeCertificate");
201         }
202         return true; // If we got this far either we chose to not store the certificate or it was stored correctly
203     }

205     protected void writeCertificate(Certificate cert, File outFolder, String prefix, String suffix) throws PublisherException {
206         try {
207             final File destFile = File.createTempFile(prefix, suffix, outFolder);
208             try (FileOutputStream fos = new FileOutputStream(destFile)) {
209                 fos.write(cert.getEncoded());
210             }
211         } catch (CertificateEncodingException ex) {
212             final String msg = intres.getLocalizedMessage("publisher.errorcertconversion");
213             LOG.error(msg);
214             throw new PublisherException(msg);
215         } catch (IOException e) {
216             final String msg = intres.getLocalizedMessage("publisher.errortempfile");
217             LOG.error(msg, e);
218             throw new PublisherException(msg);
219         }
220     }

222     /**
223      * Checks if the certificate should be sampled according to the sampling 
224      * method.
225      * @param profileId The profileId the certificate was issued for
226      * @return True if it is "time" to take a sample.
227      * @throws PublisherException in case of configuration errors
228      */
229     private boolean isTimeToSample(int profileId) throws PublisherException {
230         final boolean result;
231         
232         SamplingMethod theMethod = getMethod(profileId);
233         if (LOG.isTraceEnabled()) {
234             LOG.trace("samplingMethod: " + theMethod);
235         }
236         
237         switch (theMethod) {
238             case SAMPLE_NONE: {
239                 result = false;
240                 break;
241             } 
242             case SAMPLE_ALL: {
243                 result = true;
244                 break;
245             }
246             case SAMPLE_PROBABILISTIC: {
247                 // Get the ratio
248                 Double p = getPvalue(profileId);
249                 if (LOG.isTraceEnabled()) {
250                     LOG.trace("pvalue: " + p);
251                 }
252 
253                 // Gets an pseudorandomly value between 0 and 1 from an approximately uniform distribution
254                 // Notice: This method call is thread-safe, however uses synchronization which might decrease performance
255                 //         We can not use a random object as instance variable as a new instance of this class is created for every request and requests at the same time would then be initialized with the same seed
256                 //         When we switch to Java 7 we could possibly use ThreadLocalRandom instead.
257                 final double r = Math.random();
258 
259                 // r will be less than our p with the probability of p
260                 result = r < p;
261                     break;
262                 }
263             default: {
264                 final String msg = intres.getLocalizedMessage("publisher.errorsamplingmethod");
265                 LOG.error(msg);
266                 throw new PublisherException(msg);
267             }
268         }
269         return result;
270     }
271     
272     private SamplingMethod getMethod(int profileId) throws PublisherException {
273         final SamplingMethod result;
274         
275         String propertyKey = PROPERTYPREFIX_PROFILEID + profileId + PROPERTYSUFFIX_SAMPLINGMETHOD;
276         String propertyString = config.getProperty(propertyKey);
277         if (propertyString == null) {
278             propertyKey = PROPERTY_DEFAULT_SAMPLINGMETHOD;
279             propertyString = config.getProperty(propertyKey);
280         }
281         if (propertyString == null) {
282             final String msg = intres.getLocalizedMessage("publisher.errormissingproperty", propertyKey);
283             LOG.error(msg);
284             throw new PublisherException(msg);
285         }
286         try {
287             result = SamplingMethod.valueOf(propertyString);
288         } catch (IllegalArgumentException ex) {
289             final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", PROPERTY_DEFAULT_PVALUE, ex.getLocalizedMessage());
290             LOG.debug(msg, ex);
291             throw new PublisherException(msg);
292         }
293         return result;
294     }
295     
296     private Double getPvalue(int profileId) throws PublisherException {
297         final Double result;
298         
299         String propertyKey = PROPERTYPREFIX_PROFILEID + profileId + PROPERTYSUFFIX_PVALUE;
300         String propertyString = config.getProperty(propertyKey);
301         if (propertyString == null) {
302             propertyKey = PROPERTY_DEFAULT_PVALUE;
303             propertyString = config.getProperty(propertyKey);
304         }
305         if (propertyString == null) {
306             final String msg = intres.getLocalizedMessage("publisher.errormissingproperty", propertyKey);
307             LOG.error(msg);
308             throw new PublisherException(msg);
309         }
310         try {
311             result = Double.parseDouble(propertyString);
312         } catch (NumberFormatException ex) {
313             final String msg = intres.getLocalizedMessage("publisher.errorinvalidvalue", PROPERTY_DEFAULT_PVALUE, ex.getLocalizedMessage());
314             LOG.debug(msg, ex);
315             throw new PublisherException(msg);
316         }
317         return result;
318     }
319     
320     private File getOutputFolder() throws PublisherException {
321         final File result;
322         String outputFolderString = config.getProperty(PROPERTY_OUTPUTFOLDER, "").trim();
323         if (outputFolderString.length() < 1) {
324             final String msg = intres.getLocalizedMessage("publisher.errormissingproperty", PROPERTY_OUTPUTFOLDER);
325             LOG.error(msg);
326             throw new PublisherException(msg);
327         } else {
328             result = new File(outputFolderString);
329         }
330         return result;
331     }
332     
333     public enum SamplingMethod {
334         /** No certificates are sampled. */
335         SAMPLE_NONE,
336         
337         /** All certificates are sampled. */ 
338         SAMPLE_ALL,
339         
340         /** Every certificate is sampled with the probability as specified as the pvalue. */
341         SAMPLE_PROBABILISTIC,
342     }
