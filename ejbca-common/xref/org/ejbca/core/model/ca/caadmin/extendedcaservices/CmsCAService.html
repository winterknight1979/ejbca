142 	public CmsCAService(final HashMap<Object, Object> data) throws IllegalArgumentException {
143 		super(data);
144 		CryptoProviderTools.installBCProviderIfNotAvailable();
145 		loadData(data);
146 		if (this.data.get(KEYSTORE) != null) {    
147 			// lookup keystore passwords      
148 			final String keystorepass = StringTools.passwordDecryption(EjbcaConfiguration.getCaCmsKeyStorePass(), "ca.cmskeystorepass");
149 			int status = ExtendedCAServiceInfo.STATUS_INACTIVE;
150 			try {
151 		        if (log.isDebugEnabled()) {
152 	                log.debug("Loading CMS keystore");
153 		        }
154 				final KeyStore keystore = KeyStore.getInstance("PKCS12", "BC");
155 				keystore.load(new ByteArrayInputStream(Base64.decode(((String) this.data.get(KEYSTORE)).getBytes())),keystorepass.toCharArray());
156                 if (log.isDebugEnabled()) {
157                     log.debug("Finished loading CMS keystore");
158                 }
159                 String alias = PRIVATESIGNKEYALIAS;
160 				privKey = (PrivateKey) keystore.getKey(alias, null);
161 				if (privKey == null) {
162                     if (log.isDebugEnabled()) {
163                         log.debug("privKey was null for alias "+alias+", trying with "+OLDPRIVATESIGNKEYALIAS);
164                     }
165 	                alias = OLDPRIVATESIGNKEYALIAS;
166 	                privKey = (PrivateKey) keystore.getKey(alias, null);
167 				}
168 				// Due to a bug in Glassfish v1 (fixed in v2), we used to have to make sure all certificates in this 
169 				// Array were of SUNs own provider, using CertTools.SYSTEM_SECURITY_PROVIDER.
170 				// As of EJBCA 3.9.3 we decided that we don't have to support Glassfish v1 anymore.
171 				Collection<Certificate> coll = CertTools.getCertCollectionFromArray(keystore.getCertificateChain(alias), null);
172 				this.certificatechain = new ArrayList<Certificate>(coll);  
173 				status = getStatus();
174 			} catch (Exception e) {
175 				log.error("Could not load keystore or certificate for CA CMS service. Perhaps the password was changed? " + e.getMessage(), e);
176 			} finally {
177 				info = new CmsCAServiceInfo(status, getSubjectDN(), getSubjectAltName(), (String)this.data.get(KEYSPEC), 
178 						(String) this.data.get(KEYALGORITHM), this.certificatechain);
179 			}
180 			data.put(EXTENDEDCASERVICETYPE, Integer.valueOf(ExtendedCAServiceTypes.TYPE_CMSEXTENDEDSERVICE));        
181 		} else {
182             if (log.isDebugEnabled()) {
183                 log.debug("KEYSTORE is null when creating CmsCAService");
184             }
185 		}
186 	}

189 	public void init(final CryptoToken cryptoToken, final CA ca, final AvailableCustomCertificateExtensionsConfiguration cceConfig) throws Exception {
190         if (log.isTraceEnabled()) {
191             log.trace(">init");
192         }
193         if (info.getStatus() != ExtendedCAServiceInfo.STATUS_ACTIVE) {
194             if (log.isDebugEnabled()) {
195                 log.debug("Not generating certificates for inactive service");
196             }
197         } else {
198     		// lookup keystore passwords      
199     	    final String keystorepass = StringTools.passwordDecryption(EjbcaConfiguration.getCaCmsKeyStorePass(), "ca.cmskeystorepass");
200     		// Currently only RSA keys are supported
201     	    final CmsCAServiceInfo/../../../../org/ejbca/core/model/ca/caadmin/extendedcaservices/CmsCAServiceInfo.html#CmsCAServiceInfo">CmsCAServiceInfo info = (CmsCAServiceInfo) getExtendedCAServiceInfo();
202     		// Create KeyStore	    
203     	    final KeyStore keystore = KeyStore.getInstance("PKCS12", "BC");
204     		keystore.load(null, null);                              
205     		final KeyPair cmskeys = KeyTools.genKeys(info.getKeySpec(), info.getKeyAlgorithm());
206     		// A simple hard coded certificate profile that works for the CMS CA service
207     		final CertificateProfile certProfile = new CertificateProfile(CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER);
208     		certProfile.setUseKeyUsage(true);
209     		certProfile.setKeyUsage(new boolean[9]);
210     		certProfile.setKeyUsage(CertificateConstants.DIGITALSIGNATURE,true);
211     		certProfile.setKeyUsage(CertificateConstants.KEYENCIPHERMENT,true);
212     		certProfile.setKeyUsage(CertificateConstants.DATAENCIPHERMENT,true);
213     		certProfile.setKeyUsageCritical(true);
214             final EndEntityInformation eeInformation = new EndEntityInformation("NOUSERNAME", info.getSubjectDN(), 0, info.getSubjectAltName(), "NOEMAIL", 0,new EndEntityType(),0,0, null,null,0,0,null);
215             final Certificate certificate = ca.generateCertificate(cryptoToken, eeInformation,
216             		cmskeys.getPublic(), -1, null, ca.getEncodedValidity(), certProfile, null, cceConfig);
217     		certificatechain = new ArrayList<Certificate>();
218     		certificatechain.add(certificate);
219     		certificatechain.addAll(ca.getCertificateChain());
220     		privKey = cmskeys.getPrivate(); 
221     		keystore.setKeyEntry(PRIVATESIGNKEYALIAS,cmskeys.getPrivate(),null,(Certificate[]) certificatechain.toArray(new Certificate[certificatechain.size()]));              
222     		final ByteArrayOutputStream baos = new ByteArrayOutputStream();
223     		keystore.store(baos, keystorepass.toCharArray());
224     		data.put(KEYSTORE, new String(Base64.encode(baos.toByteArray())));      
225 	    }
226 		setStatus(info.getStatus());
227 		this.info = new CmsCAServiceInfo(info.getStatus(), getSubjectDN(), getSubjectAltName(), (String)data.get(KEYSPEC), (String) data.get(KEYALGORITHM), certificatechain);
228         if (log.isTraceEnabled()) {
229             log.trace("<init");
230         }
231 	}

234 	public void update(final CryptoToken cryptoToken, final ExtendedCAServiceInfo serviceinfo, final CA ca, final AvailableCustomCertificateExtensionsConfiguration cceConfig) {
235 	    final boolean missingCert = (!data.containsKey(KEYSTORE) && serviceinfo.getStatus() == ExtendedCAServiceInfo.STATUS_ACTIVE);
236 		final CmsCAServiceInfo/../../../../org/ejbca/core/model/ca/caadmin/extendedcaservices/CmsCAServiceInfo.html#CmsCAServiceInfo">CmsCAServiceInfo info = (CmsCAServiceInfo) serviceinfo; 
237         if (log.isDebugEnabled()) {
238             log.debug("CmsCAService : update " + serviceinfo.getStatus());
239         }
240 		setStatus(serviceinfo.getStatus());
241 		data.put(KEYSPEC, info.getKeySpec());
242 		data.put(KEYALGORITHM, info.getKeyAlgorithm());
243 		// We only updated the status, and keyspec/keyalg which can be edited in uninitialized CAs
244 		this.info = new CmsCAServiceInfo(serviceinfo.getStatus(), getSubjectDN(), getSubjectAltName(), info.getKeySpec(), info.getKeyAlgorithm(), certificatechain);
245 		if (info.getRenewFlag() || missingCert) {
246 			// Renew The Signers certificate.
247 			try {
248 				this.init(cryptoToken, ca, cceConfig);
249 			} catch (Exception e) {
250 				log.error("Error initilizing Extended CA service during upgrade: ", e);
251 			}
252 		}

255     @Override
256     public ExtendedCAServiceResponse extendedService(final CryptoToken cryptoToken, final ExtendedCAServiceRequest request)
257             throws ExtendedCAServiceRequestException, IllegalExtendedCAServiceRequestException, ExtendedCAServiceNotActiveException {
258         if (log.isTraceEnabled()) {
259             log.trace(">extendedService");
260         }
261 		if (!(request instanceof CmsCAServiceRequest)) {
262 			throw new IllegalExtendedCAServiceRequestException();            
263 		}
264 		if (getStatus() != ExtendedCAServiceInfo.STATUS_ACTIVE) {
265 			final String msg = intres.getLocalizedMessage("caservice.notactive", "CMS");
266 			log.error(msg);
267 			throw new ExtendedCAServiceNotActiveException(msg);                            
268 		}
269 		ExtendedCAServiceResponse returnval = null;
270 		final X509Certificate signerCert = (X509Certificate) certificatechain.get(0);
271 		final CmsCAServiceRequest/../org/ejbca/core/model/ca/caadmin/extendedcaservices/CmsCAServiceRequest.html#CmsCAServiceRequest">CmsCAServiceRequest serviceReq = (CmsCAServiceRequest)request;
272 		// Create the signed data
273 		final CMSSignedDataGenerator gen1 = new CMSSignedDataGenerator();
274         try {
275             byte[] resp = serviceReq.getDoc();
276             // Add our signer info and sign the message
277             if ((serviceReq.getMode() & CmsCAServiceRequest.MODE_SIGN) != 0) {
278                 final List<X509Certificate> x509CertChain = new ArrayList<X509Certificate>();
279                 for (Certificate certificate : certificatechain) {
280                     x509CertChain.add((X509Certificate) certificate);
281                 }
282                 gen1.addCertificates(new CollectionStore<>(CertTools.convertToX509CertificateHolder(x509CertChain)));
283                 JcaDigestCalculatorProviderBuilder calculatorProviderBuilder = new JcaDigestCalculatorProviderBuilder().setProvider(BouncyCastleProvider.PROVIDER_NAME);
284                 JcaSignerInfoGeneratorBuilder builder = new JcaSignerInfoGeneratorBuilder(calculatorProviderBuilder.build());
285                 ASN1ObjectIdentifier oid = AlgorithmTools.getSignAlgOidFromDigestAndKey(CMSSignedGenerator.DIGEST_SHA1, privKey.getAlgorithm());
286                 String signatureAlgorithmName = AlgorithmTools.getAlgorithmNameFromOID(oid);
287                 JcaContentSignerBuilder signerBuilder = new JcaContentSignerBuilder(signatureAlgorithmName).setProvider(BouncyCastleProvider.PROVIDER_NAME);
288                 ContentSigner contentSigner = signerBuilder.build(privKey);
289                 gen1.addSignerInfoGenerator(builder.build(contentSigner, signerCert));
290                 final CMSTypedData msg = new CMSProcessableByteArray(resp);
291                 final CMSSignedData s = gen1.generate(msg, true);
292                 resp = s.getEncoded();
293             }
294             if ((serviceReq.getMode() & CmsCAServiceRequest.MODE_ENCRYPT) != 0) {
295                 CMSEnvelopedDataGenerator edGen = new CMSEnvelopedDataGenerator();
296                 edGen.addRecipientInfoGenerator(new JceKeyTransRecipientInfoGenerator(getCMSCertificate()).setProvider(BouncyCastleProvider.PROVIDER_NAME));
297                 JceCMSContentEncryptorBuilder jceCMSContentEncryptorBuilder = new JceCMSContentEncryptorBuilder(PKCSObjectIdentifiers.des_EDE3_CBC).setProvider(BouncyCastleProvider.PROVIDER_NAME);
298                 CMSEnvelopedData ed = edGen.generate(new CMSProcessableByteArray(resp), jceCMSContentEncryptorBuilder.build());
299                 resp = ed.getEncoded();
300 			}
301 			if ((serviceReq.getMode() & CmsCAServiceRequest.MODE_DECRYPT) != 0) {
302 				final CMSEnvelopedData ed = new CMSEnvelopedData(resp);
303 				final RecipientInformationStore  recipients = ed.getRecipientInfos();
304 				final X500Name issuer = X500Name.getInstance(getCMSCertificate().getIssuerX500Principal().getEncoded());
305 				final KeyTransRecipientId id = new KeyTransRecipientId(issuer, getCMSCertificate().getSerialNumber());
306 				final RecipientInformation recipient = recipients.get(id);
307 				if (recipient != null) {
308 	                JceKeyTransEnvelopedRecipient rec = new JceKeyTransEnvelopedRecipient(this.privKey);
309 	                // Provider for decrypting the symmetric key 
310                     // We can use a different provider for decrypting the content, for example of we used a PKCS#11 provider above we could use the BC provider below
311 	                rec.setContentProvider(BouncyCastleProvider.PROVIDER_NAME);
312 	                rec.setProvider(cryptoToken.getSignProviderName());
313 	                // Option we must set to prevent Java PKCS#11 provider to try to make the symmetric decryption in the HSM, 
314 	                // even though we set content provider to BC. Symm decryption in HSM varies between different HSMs and at least for this case is known 
315 	                // to not work in SafeNet Luna (JDK behavior changed in JDK 7_75 where they introduced imho a buggy behavior)
316 	                rec.setMustProduceEncodableUnwrappedKey(true);	                
317 	                resp = recipient.getContent(rec);
318 				}
319 			}
320 			returnval = new CmsCAServiceResponse(resp);
321         } catch (CMSException e) {
322         	log.error("Error in CmsCAService", e);
323         	throw new ExtendedCAServiceRequestException(e);
324 		} catch (IOException e) {
325         	log.error("Error in CmsCAService", e);
326         	throw new ExtendedCAServiceRequestException(e);
327 		} catch (OperatorCreationException e) {
328 		    log.error("Error in CmsCAService", e);
329             throw new ExtendedCAServiceRequestException(e);
330 		} catch (CertificateEncodingException e) {
331 		    log.error("Error in CmsCAService", e);
332             throw new ExtendedCAServiceRequestException(e);
333         }
334         if (log.isTraceEnabled()) {
335             log.trace("<extendedService");              
336         }
337 		return returnval;
338 	}

340 	private X509Certificate getCMSCertificate() {
341 		if (cmsCertificate == null) {
342 		    for (final Certificate current : certificatechain) {
343 				final X509Certificate cert = (X509Certificate) current;
344 				if (cert.getBasicConstraints() == -1) {
345 					cmsCertificate = cert;
346 					break;
347 				}
348 			}
349 		}
350 		return cmsCertificate;
351 	}

391     private String getSubjectXName(final String key) {
392         String ret = null;
393         final String value = (String) data.get(key);
394         try {
395             ret = new String(Base64.decode((value).getBytes("UTF-8")));
396         } catch (UnsupportedEncodingException e) {
397             log.error("Could not decode data from Base64", e);
398         } catch (DecoderException e) {
399             // This is an old CA, where it's not Base64encoded
400             if (log.isDebugEnabled()) {
401                 log.debug("Old non base64 encoded " + key + ": " + value);
402             }
403             ret = value; 
404         }
405         return ret;       
406     }

408     private void setSubjectXName(final String key, final String value) {
409         try {
410             data.put(key, new String(Base64.encode(value.getBytes("UTF-8"), false)));
411         } catch (UnsupportedEncodingException e) {
412             log.error("Could not encode data to Base64", e);
413         }
