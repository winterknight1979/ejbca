20  /**
21   * SwedishEIDHardToken is a class defining data stored in database for a Swedish EID token.
22   *
23   * @version $Id: SwedishEIDHardToken.java 19901 2014-09-30 14:29:38Z anatom $
24   */
25  public class SwedishEIDHardToken extends HardToken {
26      /**
27       * Determines if a de-serialized file is compatible with this class.
28       *
29       * Maintainers must change this value if and only if the new version
30       * of this class is not compatible with old versions. See Sun docs
31       * for <a href=http://java.sun.com/products/jdk/1.1/docs/guide
32       * /serialization/spec/version.doc.html> details. </a>
33       *
34       */
35      private static final long serialVersionUID = 5695294040446656470L;
36  
37      // Public Constants
38  	public static final int THIS_TOKENTYPE = SecConst.TOKEN_SWEDISHEID;
39  	
40      public static final String INITIALAUTHENCPIN   = "INITIALAUTHENCPIN";
41      public static final String AUTHENCPUK          = "AUTHENCPUK";
42      public static final String INITIALSIGNATUREPIN = "INITIALSIGNATUREPIN";
43      public static final String SIGNATUREPUK        = "SIGNATUREPUK";   
44            
45      
46      public static final String[] FIELDSWITHPUK = new String[] {INITIALAUTHENCPIN, AUTHENCPUK, EMPTYROW_FIELD, INITIALSIGNATUREPIN, SIGNATUREPUK};
47      public static final int[] DATATYPESWITHPUK = new int[] { STRING, STRING, EMPTYROW, STRING, STRING };
48      public static final String[] FIELDTEXTSWITHPUK = new String[] { INITIALAUTHENCPIN, AUTHENCPUK, EMPTYROW_FIELD, INITIALSIGNATUREPIN, SIGNATUREPUK};   
49      
50      public static final String[] FIELDSWITHOUTPUK = new String[] {};
51      public static final int[] DATATYPESWITHOUTPUK = new int[] {};
52      public static final String[] FIELDTEXTSWITHOUTPUK = new String[] {};    	 
53  
54  
55      // Public Methods
56      /** Constructor to use. 
57       *  @param initialsignaturepin PIN
58   * @param signaturepuk PUK
59   * @param initialauthencpin PIN 
60   * @param authencpuk PUK
61   * @param hardtokenprofileid int */
62      public SwedishEIDHardToken(String initialauthencpin,
63                                 String authencpuk,
64                                 String initialsignaturepin,
65                                 String signaturepuk,
66                                 int hardtokenprofileid) {
67      	super(true);
68          setInitialAuthEncPIN(initialauthencpin);
69          setAuthEncPUK(authencpuk);
70          setInitialSignaturePIN(initialsignaturepin);
71          setSignaturePUK(signaturepuk);
72          setTokenProfileId(hardtokenprofileid);        
73          
74          data.put(TOKENTYPE, Integer.valueOf(THIS_TOKENTYPE));
75      } 
76      
77      /** Constructor only to be used internally. 
78       * @param includePUK PUK*/
79      public SwedishEIDHardToken(boolean includePUK) {
80      	super(includePUK);
81      	data.put(TOKENTYPE, Integer.valueOf(THIS_TOKENTYPE));
82      	if(!includePUK){
83      	  setInitialAuthEncPIN("");
84      	  setAuthEncPUK("");
85      	  setInitialSignaturePIN("");
86      	  setSignaturePUK("");
87      	}
88      }
89  
90      
91      // Public Methods.
92      public String getInitialAuthEncPIN() {
93          return (String) data.get(INITIALAUTHENCPIN);
94      }
95  
96      public void setInitialAuthEncPIN(String initialbasicpin) {
97          data.put(INITIALAUTHENCPIN, initialbasicpin);
98      }
99      
100 
101     public String getAuthEncPUK() {
102         return (String) data.get(AUTHENCPUK);
103     }
104 
105     public void setAuthEncPUK(String basicpuk) {
106         data.put(AUTHENCPUK, basicpuk);
107     }
108     
109 
110     public String getInitialSignaturePIN() {
111         return (String) data.get(INITIALSIGNATUREPIN);
112     }
113 
114     public void setInitialSignaturePIN(String initialsignaturepin) {
115         data.put(INITIALSIGNATUREPIN, initialsignaturepin);
116     }
117     
118 
119 
120     public String getSignaturePUK() {
121         return (String) data.get(SIGNATUREPUK);
122     }
123 
124     public void setSignaturePUK(String signaturepuk) {
125         data.put(SIGNATUREPUK, signaturepuk);
126     }
127 
128 
129 	
130 	public int[] getDataTypes(boolean includePUK) {
131 		if(includePUK){
132 			return DATATYPESWITHPUK;	
133 		}
134 		return DATATYPESWITHOUTPUK;
135 	}
136 
137 	public String[] getFieldTexts(boolean includePUK) {
138 		if(includePUK){
139 			return FIELDTEXTSWITHPUK;	
140 		}
141 		return FIELDTEXTSWITHOUTPUK;
142 	}
143 
144 	public String[] getFields(boolean includePUK) {
145 		if(includePUK){
146 			return FIELDSWITHPUK;	
147 		}
148 		return FIELDSWITHOUTPUK;
149 	}
150 
151     // Private fields.
152 }
