14  package org.ejbca.core.model.hardtoken;
15  
16  import java.io.Serializable;
17  import java.util.Collection;
18  import java.util.Date;
19  
20  import org.cesecore.util.StringTools;
21  import org.ejbca.core.model.hardtoken.types.HardToken;
22  
23  
24  /**
25   *  This is a value class containing the data relating to a hard token sent between
26   *  server and clients.
27   *
28   * @version $Id: HardTokenInformation.java 26797 2017-10-12 04:45:37Z anatom $
29   */
30  
31  public class HardTokenInformation implements Serializable, Comparable<HardTokenInformation> {
32  
33      private static final long serialVersionUID = 2801790818906276161L;
34  
35      // Public Constructors
36  	/** 
37  	 * Constructor of a hard token data.
38  	 * 
39  	 * @param tokensn the tokensn
40  	 * @param username the username owning the token
41  	 * @param createtime time the token was created
42  	 * @param modifytime time whem token was modified or a copy was made.
43  	 * @param tokentype the hardtokenprofile used to create the token
44  	 * @param significantIssuerDN of the CA that the card belongs to
45  	 * @param hardtoken the actual hardtoken data
46  	 * @param copyof tokenSN of original or null of this is an original
47  	 * @param copies Collection of tokensn of tokens copied from this token, null if no copies have been made.
48  	 * 
49  	 */
50      public HardTokenInformation(String tokensn, String username, Date createtime,  Date modifytime, 
51                           int tokentype, String significantIssuerDN, HardToken hardtoken, String copyof,
52                           Collection<String> copies){
53        this.tokensn=tokensn;
54        this.username=StringTools.stripUsername(username);
55        this.createtime=createtime;
56        this.modifytime=modifytime;
57        this.tokentype=tokentype;
58        this.significantIssuerDN = significantIssuerDN;
59        this.hardtoken=hardtoken;
60        this.copyof=copyof;
61        this.copies=copies;
62      }
63  
64      public HardTokenInformation(){
65      }
66  
67      // Public Methods
68  
69      public String getTokenSN(){ return this.tokensn; }
70      public void setTokenSN(String tokensn){ this.tokensn=tokensn; }
71  
72      public String getUsername(){ return this.username; }
73      public void setUsername(String username){ this.username=StringTools.stripUsername(username); }
74  
75      public Date getCreateTime(){ return this.createtime; }
76      public void setCreateTime(Date createtime){ this.createtime=createtime; }
77  
78      public Date getModifyTime(){ return this.modifytime; }
79      public void setModifyTime(Date modifytime){ this.modifytime=modifytime; }
80  
81      public int getTokenType(){ return this.tokentype; }
82      public void setTokenType(int tokentype){ this.tokentype=tokentype; }
83  
84      public HardToken getHardToken(){ return this.hardtoken; }
85      public void setHardToken(HardToken hardtoken){ this.hardtoken=hardtoken; }
86      
87      public boolean isOriginal(){
88        return copyof==null;	
89      }
90      
91      public String getCopyOf(){
92        return copyof;	
93      }
94  
95  	public String getSignificantIssuerDN() {
96  		return significantIssuerDN;
97  	}
98      
99      /** 
100      * Returns a collection of (Strings) containing the tokenSN of all copies made
101      * of this token.
102      * 
103      * @return A Collection of tokenSN or null of no copies have been made.
104      */
105     public Collection<String> getCopies(){
106 		return copies;
107     }
108 
109     // Private fields
110     private    String          tokensn;
111     private    String          username;
112     private    Date            createtime;
113     private    Date            modifytime;
114     private    int             tokentype;
115     private    String          significantIssuerDN;
116     private    HardToken       hardtoken;
117     private    String          copyof;
118     private    Collection<String> copies;
119 
120 	/** 
121 	 * When viewing the tokens in the GUI they should come in the order that they were created.
122 	 * @see java.lang.Comparable#compareTo(java.lang.Object)
123 	 */
124 	@Override
125 	public int compareTo(HardTokenInformation o) {
126 		if ( this.createtime.equals(o.getCreateTime())) {
127 			return 0;
128 		}
129 		return this.createtime.after(o.getCreateTime()) ? -1 : 1;
130 	}
131 
132 }
