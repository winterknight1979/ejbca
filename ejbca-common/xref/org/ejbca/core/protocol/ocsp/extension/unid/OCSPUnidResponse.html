14  package org.ejbca.core.protocol.ocsp.extension.unid;
15  
16  import java.util.Date;
17  
18  import org.bouncycastle.cert.ocsp.BasicOCSPResp;
19  import org.bouncycastle.cert.ocsp.OCSPException;
20  import org.bouncycastle.cert.ocsp.OCSPResp;
21  import org.bouncycastle.cert.ocsp.RevokedStatus;
22  import org.bouncycastle.cert.ocsp.SingleResp;
23  import org.bouncycastle.cert.ocsp.UnknownStatus;
24  
25  /** Class holding data returned by the OCSPUnidExtension
26   * 
27   * @version $Id: OCSPUnidResponse.java 28652 2018-04-09 14:35:21Z aminkh $
28   *
29   */
30  public class OCSPUnidResponse {
31  	
32      /** Constants capturing the OCSP response status. 
33       * These are the return codes defined in the RFC. 
34       * The codes are just used for simple access to the OCSP return value. 
35       */
36      public static final int OCSP_GOOD = 0;
37      public static final int OCSP_REVOKED = 1;
38      public static final int OCSP_UNKNOWN = 2;
39  
40      //
41      // Constants for error status
42      //
43      /**
44       * This is the standard code when no error occurred. Ideally this should always be the returned value.
45       */
46      public static final int ERROR_NO_ERROR = 0;
47      /**
48       * An unknown error has occurred (for example internal server error on the OCSP responder) .
49       */
50  	public static final int ERROR_UNKNOWN = 1;
51      /**
52       * You are not authorized to perform a FNR/UNID lookup.
53       */
54  	public static final int ERROR_UNAUTHORIZED = 2;
55      /**
56       * There was no response from the server.
57       */
58  	public static final int ERROR_NO_RESPONSE = 3;
59      /**
60       * This error is returned when the signature of the OCSP-response sent by the server has an invalid 
61       * signature. This should typically never happen unless the OCSP-server is compromised in someway, 
62       * a fake OCSP-server is installed or something went wrong with the communication so the response 
63       * was truncated.
64       */
65  	public static final int ERROR_INVALID_SIGNATURE = 4;
66      /**
67       * This error is returned when the signerId in the OCSP-response sent by the server does not match
68       * the first certificate in the chain in the response. 
69       * This should typically never happen unless the OCSP-server is broken. 
70       */
71  	public static final int ERROR_INVALID_SIGNERID = 5;
72      /**
73       * This error is returned when the OCSP signers certificate can not be verified using the CA-certificate.
74       * This should typically never happen unless the OCSP-server is broken or compromised. 
75       */
76  	public static final int ERROR_INVALID_SIGNERCERT = 6;
77      /**
78       * You did not specify a URI in the call, and there is not one embedded in the certificate.
79       */
80      public static final int ERROR_NO_OCSP_URI = 7;
81      /**
82       * The nonce in the response did not match the nonce sent in the request.
83       */
84      public static final int ERROR_INVALID_NONCE = 8;
85  
86  	/*
87  	 * Private variables
88  	 */
89  	private OCSPResp resp = null;
90  	private String fnr = null;
91  	private int httpReturnCode = 200;
92  	private int errCode = OCSPUnidResponse.ERROR_NO_ERROR;
93  	
94  	public OCSPUnidResponse() {
95  	}
96  	public OCSPUnidResponse(OCSPResp ocspresp) {
97  		this.resp = ocspresp;
98  	}
99  	public int getHttpReturnCode() {
100 		return httpReturnCode;
101 	}
102 	public void setHttpReturnCode(int code) {
103 		httpReturnCode = code;
104 	}
105 	public int getErrorCode() {
106 		return errCode;
107 	}
108 	public void setErrorCode(int code) {
109 		errCode = code;
110 	}
111 	public String getFnr() {
112 		return fnr;
113 	}
114 	public void setFnr(String fnr) {
115 		this.fnr = fnr;
116 	}
117 	public OCSPResp getResp() {
118 		return resp;
119 	}
120 	public void setResp(OCSPResp resp) {
121 		this.resp = resp;
122 	}
123 	public int getStatus() {
124         if (resp == null) {
125             return OCSPUnidResponse.OCSP_UNKNOWN;
126         }
127 		try {
128 			BasicOCSPResp brep;
129 			brep = (BasicOCSPResp) resp.getResponseObject();
130 			SingleResp[] singleResps = brep.getResponses();
131 			SingleResp singleResp = singleResps[0];
132 			Object status = singleResp.getCertStatus();
133 			if (status == null) {
134 				return OCSPUnidResponse.OCSP_GOOD;
135 			}
136 			if (status instanceof RevokedStatus) {
137 				return OCSPUnidResponse.OCSP_REVOKED;
138 			}
139 			if (status instanceof UnknownStatus) {
140 				return OCSPUnidResponse.OCSP_UNKNOWN;
141 			}
142 		} catch (OCSPException e) {
143 			// Ignore, default return
144 		}
145 		return OCSPUnidResponse.OCSP_UNKNOWN;
146 		
147 	}
148 	
149 	/** Returns the OCSP response status
150 	 * 
151 	 * @return the response code of the OCSP message, OCSPRespBuilder.XX for example OCSPRespBuilder.SIG_REQUIRED
152 	 */
153 	public int getResponseStatus() {
154         if (resp == null) {
155             return OCSPUnidResponse.ERROR_UNKNOWN;
156         }
157         return resp.getStatus();
158 	}
159 	
160 	public Date getProducedAt() throws OCSPException {
161 		return ((BasicOCSPResp)resp.getResponseObject()).getProducedAt();
162 	}
163 	
164 	public Date getThisUpdate() throws OCSPException {
165 		return ((BasicOCSPResp)resp.getResponseObject()).getResponses()[0].getThisUpdate();
166 	}
167 	
168 	public Date getNextUpdate() throws OCSPException {
169 		return ((BasicOCSPResp)resp.getResponseObject()).getResponses()[0].getNextUpdate();
170 	}
171 	
172 }
