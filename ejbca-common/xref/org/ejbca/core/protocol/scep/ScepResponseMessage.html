172         } catch (CertificateEncodingException e) {

174         } catch (CertificateException e) {

229     @Override
230     public boolean create() throws CertificateEncodingException, CRLException {
231       boolean ret = false;
232         try {
233             if (status.equals(ResponseStatus.SUCCESS)) {
234                 log.debug("Creating a STATUS_OK message.");
235             } else {
236             	if (status.equals(ResponseStatus.FAILURE)) {
237                     log.debug("Creating a STATUS_FAILED message (or returning false).");
238                     if (failInfo.equals(FailInfo.WRONG_AUTHORITY)) {
239                     	return false;      
240                     }
241                     if (failInfo.equals(FailInfo.INCORRECT_DATA)) {
242                         return false;     
243                     }
244                 } else {
245                     log.debug("Creating a STATUS_PENDING message.");
246                 }               
247             }
248 
249             CMSTypedData msg;
250             // Create encrypted response if this is success and NOT a CRL response message
251             if (status.equals(ResponseStatus.SUCCESS)) {
252 
253                 CMSEnvelopedDataGenerator edGen = new CMSEnvelopedDataGenerator();
254                 // Add the issued certificate to the signed portion of the CMS (as signer, degenerate case)
255                 List<X509Certificate> certList = new ArrayList<X509Certificate>();
256                 if (cert != null) {
257                     log.debug("Adding certificates to response message");
258                     certList.add((X509Certificate) cert);
259                     // Add the CA cert, it's optional but Cisco VPN client complains if it isn't there
260                     if (includeCACert) {
261                     	if (caCert != null) {
262                     		// If we have an explicit CAcertificate
263                     		log.debug("Including explicitly set CA certificate in SCEP response.");
264                     		certList.add((X509Certificate) caCert);
265                     	} else {
266                     		// If we don't have an explicit caCert, we think that the signCert is the CA cert
267                     		// If we have an explicit caCert, the signCert is probably the RA certificate, and we don't include that one
268                     		log.debug("Including message signer certificate in SCEP response.");
269                     		certList.add((X509Certificate) signCertChain.iterator().next());
270                     	}
271                     }
272                 }
273                 // Create the signed CMS message to be contained inside the envelope
274                 // this message does not contain any message, and no signerInfo
275                 CMSSignedDataGenerator gen = new CMSSignedDataGenerator();
276                 gen.addCertificates(new CollectionStore<>(CertTools.convertToX509CertificateHolder(certList)));
277                 if(crl != null) {
278                     gen.addCRL(new JcaX509CRLHolder((X509CRL) crl));
279                 }
280 
281                 CMSSignedData s = gen.generate(new CMSAbsentContent(), false);
282 
283                 // Envelope the CMS message
284                 if (recipientKeyInfo != null) {
285                     try {
286                         X509Certificate rec = CertTools.getCertfromByteArray(recipientKeyInfo, X509Certificate.class);
287                         log.debug("Added recipient information - issuer: '" + CertTools.getIssuerDN(rec) + "', serno: '" + CertTools.getSerialNumberAsString(rec));
288                         edGen.addRecipientInfoGenerator(new JceKeyTransRecipientInfoGenerator(rec).setProvider(BouncyCastleProvider.PROVIDER_NAME));
289                     } catch (CertificateParsingException e) {
290                         throw new IllegalArgumentException("Can not decode recipients self signed certificate!", e);
291                     }
292                 } else {
293                     edGen.addRecipientInfoGenerator(new JceKeyTransRecipientInfoGenerator((X509Certificate) cert).setProvider(BouncyCastleProvider.PROVIDER_NAME));
294                 }
295                 try {
296                     JceCMSContentEncryptorBuilder jceCMSContentEncryptorBuilder = new JceCMSContentEncryptorBuilder(SMIMECapability.dES_CBC).setProvider(BouncyCastleProvider.PROVIDER_NAME);
297                     CMSEnvelopedData ed = edGen.generate(new CMSProcessableByteArray(s.getEncoded()), jceCMSContentEncryptorBuilder.build());
298                     if (log.isDebugEnabled()) {
299                         log.debug("Enveloped data is " + ed.getEncoded().length + " bytes long");
300                     }
301                     msg = new CMSProcessableByteArray(ed.getEncoded());
302                 } catch (IOException e) {
303                     throw new IllegalStateException("Unexpected IOException caught", e);
304                 }
305             } else {
306                 // Create an empty message here
307                 //msg = new CMSProcessableByteArray("PrimeKey".getBytes());
308                 msg = new CMSProcessableByteArray(new byte[0]);
309             }
310 
311             // Create the outermost signed data
312             CMSSignedDataGenerator gen1 = new CMSSignedDataGenerator();
313 
314             // add authenticated attributes...status, transactionId, sender- and recipientNonce and more...
315             Hashtable<ASN1ObjectIdentifier, Attribute> attributes = new Hashtable<ASN1ObjectIdentifier, Attribute>();
316             ASN1ObjectIdentifier oid;
317             Attribute attr;
318             DERSet value;
319 
320             // Message type (certrep)
321             oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_messageType);
322             value = new DERSet(new DERPrintableString("3"));
323             attr = new Attribute(oid, value);
324             attributes.put(attr.getAttrType(), attr);
325 
326             // TransactionId
327             if (transactionId != null) {
328                 oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_transId);
329                 log.debug("Added transactionId: " + transactionId);
330                 value = new DERSet(new DERPrintableString(transactionId));
331                 attr = new Attribute(oid, value);
332                 attributes.put(attr.getAttrType(), attr);
333             }
334 
335             // status
336             oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_pkiStatus);
337             value = new DERSet(new DERPrintableString(status.getStringValue()));
338             attr = new Attribute(oid, value);
339             attributes.put(attr.getAttrType(), attr);
340 
341             if (status.equals(ResponseStatus.FAILURE)) {
342                 oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_failInfo);
343                 log.debug("Added failInfo: " + failInfo.getValue());
344                 value = new DERSet(new DERPrintableString(failInfo.getValue()));
345                 attr = new Attribute(oid, value);
346                 attributes.put(attr.getAttrType(), attr);
347             }
348 
349             // senderNonce
350             if (senderNonce != null) {
351                 oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_senderNonce);
352                 log.debug("Added senderNonce: " + senderNonce);
353                 value = new DERSet(new DEROctetString(Base64.decode(senderNonce.getBytes())));
354                 attr = new Attribute(oid, value);
355                 attributes.put(attr.getAttrType(), attr);
356             }
357 
358             // recipientNonce
359             if (recipientNonce != null) {
360                 oid = new ASN1ObjectIdentifier(ScepRequestMessage.id_recipientNonce);
361                 log.debug("Added recipientNonce: " + recipientNonce);
362                 value = new DERSet(new DEROctetString(Base64.decode(recipientNonce.getBytes())));
363                 attr = new Attribute(oid, value);
364                 attributes.put(attr.getAttrType(), attr);
365             }
366 
367             // Add our signer info and sign the message
368             Certificate cacert = signCertChain.iterator().next();
369             log.debug("Signing SCEP message with cert: "+CertTools.getSubjectDN(cacert));
370             String signatureAlgorithmName = AlgorithmTools.getAlgorithmNameFromDigestAndKey(digestAlg, signKey.getAlgorithm());
371             try {
372                 ContentSigner contentSigner = new JcaContentSignerBuilder(signatureAlgorithmName).setProvider(provider).build(signKey);
373                 JcaDigestCalculatorProviderBuilder calculatorProviderBuilder = new JcaDigestCalculatorProviderBuilder().setProvider(BouncyCastleProvider.PROVIDER_NAME);
374                 JcaSignerInfoGeneratorBuilder builder = new JcaSignerInfoGeneratorBuilder(calculatorProviderBuilder.build());
375                 builder.setSignedAttributeGenerator(new DefaultSignedAttributeTableGenerator(new AttributeTable(attributes)));
376                 gen1.addSignerInfoGenerator(builder.build(contentSigner, (X509Certificate) cacert));
377             } catch (OperatorCreationException e) {
378                 throw new IllegalStateException("BouncyCastle failed in creating signature provider.", e);
379             }
380             // The un-encoded response message itself
381             final CMSSignedData signedData = gen1.generate(msg, true);
382             try {
383                 responseMessage = signedData.getEncoded();
384             } catch (IOException e) {
385                 throw new IllegalStateException("Unexpected IOException caught.", e);
386             }
387             if (responseMessage != null) {
388                 ret = true;
389             }
390         } catch (CMSException e) {
391             log.error("Error creating CMS message: ", e);
392         }
393 
394         return ret;
395     }
