13  package org.ejbca.core.protocol.acme;
14  
15  import org.cesecore.certificates.endentity.EndEntityConstants;
16  
17  /**
18   * @version $Id: AcmeOrderStatus.java 29587 2018-08-07 15:25:52Z mikekushner $
19   *
20   */
21  public enum AcmeOrderStatus {
22      /** "The certificate will not be issued. Consider this order process abandoned." */
23      INVALID(EndEntityConstants.STATUS_FAILED),
24      /** 
25       * "The server does not believe that the client has fulfilled the requirements.
26       * Check the "authorizations" array for entries that are still pending."
27       */
28      PENDING(EndEntityConstants.STATUS_INITIALIZED),
29      /** 
30       * "The server agrees that the requirements have been fulfilled, and is awaiting finalization. 
31       * Submit a finalization request." 
32       */
33      READY(EndEntityConstants.STATUS_INPROCESS),
34      /**
35       * "The certificate is being issued. Send a GET request after the time given in the "Retry-After" 
36       * header field of the response, if any."
37       */
38      PROCESSING(EndEntityConstants.STATUS_NEW),
39      /** 
40       * "The server has issued the certificate and provisioned its URL to the "certificate" field of the order. 
41       * Download the certificate." 
42       */
43      VALID(EndEntityConstants.STATUS_GENERATED);
44      
45      private final int endEntityStatus;
46      
47      private AcmeOrderStatus(final int endEntityStatus) {
48          this.endEntityStatus = endEntityStatus;
49      }
50  
51      /** @return the corresponding EJBCA end entity status */
52      public int getEndEntityStatus() { return endEntityStatus; }
53      public static AcmeOrderStatus fromEndEntityStatus(final int endEntityStatus) {
54          for (final AcmeOrderStatus current : AcmeOrderStatus.values()) {
55              if (current.getEndEntityStatus()==endEntityStatus) {
56                  return current;
57              }
58          }
59          return null;
60      }
61      public String getJsonValue() { return this.name().toLowerCase(); }
62      public static AcmeOrderStatus fromJsonValue(final String status) { return AcmeOrderStatus.valueOf(status.toUpperCase()); }
63  }
