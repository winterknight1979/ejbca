14

16   * @version $Id: AcmeProblem.java 29845 2018-09-08 21:49:10Z aminkh $
17   *

19  public enum AcmeProblem {
20      PROBLEM_BAD_CSR("badCSR", "The CSR is unacceptable (e.g., due to a short key)"),
21      PROBLEM_BAD_NONCE("badNonce", "The client sent an unacceptable anti-replay nonce"),
22      PROBLEM_BAD_SIGNATURE_ALGORITHM("badSignatureAlgorithm",
23                      "The JWS was signed with an algorithm the server does not support"),
24      PROBLEM_INVALID_CONTACT("invalidContact", "The contact URI for an account was invalid"),
25      PROBLEM_MALFORMED("malformed", "The request message was malformed"),
26      PROBLEM_RATE_LIMITED("rateLimited", "The request exceeds a rate limit"),
27      PROBLEM_REJECTED_INDENTIFIER("rejectedIdentifier", "The server will not issue certificate for the identifier"),
28      PROBLEM_SERVER_INTERNAL("serverInternal", "The server experienced an internal error"),
29      PROBLEM_UNAUTHORIZED("unauthorized", "The client lacks sufficient authorization"),
30      PROBLEM_UNSUPPORTED_INDENTIFIER("unsupportedIdentifier", "Identifier is not supported, but may be in future"),
31      PROBLEM_USER_ACTION_REQUIRED("userActionRequired", "Visit the \"instance\" URL and take actions specified there"),
32      PROBLEM_BAD_REVOCATION_REASON("badRevocationReason", "The revocation reason provided is not allowed by the server"),
33      PROBLEM_CAA("caa", "CAA records forbid the CA from issuing"),
34      PROBLEM_DNS("dns", "There was a problem with a DNS query"),
35      PROBLEM_CONNECTION("connection", "The server could not connect to validation target"),
36      PROBLEM_TLS("tls", "The server received a TLS error during validation"),
37      PROBLEM_INCORRECT_RESPONSE("incorrectResponse", "Response received didn't match the challenge's requirements"),
38      PROBLEM_ACCOUNT_DOES_NOT_EXIST("accountDoesNotExist", "The request specified an account that does not exist");
39      
40      
41      private final String type;
42      private final String detail;
43  
44      private AcmeProblem(final String subType, final String description) {
45          this.type = "urn:ietf:params:acme:error:" + subType;
46          this.detail = description;
47      }
48  
49      public String getType() {
50          return type;
51      }
52  
53      public String getDetail() {
54          return detail;
55      }
56  
57  }
