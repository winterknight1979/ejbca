13  package org.ejbca.util;
14  
15  import java.util.ArrayList;
16  import java.util.List;
17  
18  /** Tools used in command line handling
19   * @version $Id: CliTools.java 22117 2015-10-29 10:53:42Z mikekushner $
20   */
21  public class CliTools {
22  
23      /**
24       * Using Arrays.asList will return an AbstractList that you cannot remove items from. This List is backed by a new ArrayList.
25       * 
26       * Example usage for parsing argument switches:
27       *  public void method(String[] args) {
28       *   List&lt;String&gt; argsList = getAsModifyableList(args);
29       *   boolean switch1 = argsList.remove("-switch1");
30       *   boolean switch2 = argsList.remove("-switch2");
31       *   args = argsList.toArray(new String[0]);
32       *   ...
33       *  }
34       * @param stringArray Strings
35       * @return List of strings
36       */
37      public static List<String> getAsModifyableList(String[] stringArray) {
38      	List<String> list = new ArrayList<String>();
39      	for (String string : stringArray) {
40      		list.add(string);
41      	}
42      	return list;
43      }
44      
45      /** 
46       * @return the next string after the specified parameter if the switch exists and null otherwise.
47       * The switch and the following value is removed from the supplied list.
48       * @param parameter Param
49       * @param modifyableList List
50       *  
51       */
52      public static String getAndRemoveParameter(String parameter, List<String> modifyableList) {
53          String ret = null;
54          int index = modifyableList.indexOf(parameter);
55          if (index > -1) {
56              ret = modifyableList.get(index + 1);
57              modifyableList.remove(index + 1);
58              modifyableList.remove(index);
59          }
60          return ret;
61      }
62      
63      /** @param parameter Param
64       * @param modifyableList List
65       * @return true if the parameter was present and removes it from the list. */
66      public static boolean getAndRemoveSwitch(final String parameter, List<String> modifyableList) {
67          final int index = modifyableList.indexOf(parameter);
68          if (index > -1) {
69              modifyableList.remove(index);
70              return true;
71          }
72          return false;
73      }
74      
75      /** @param modifyableList Param
76       * @return the remaining list as a regular args[] array */
77      public static String[] getAsArgs(List<String> modifyableList) {
78          return modifyableList.toArray(new String[modifyableList.size()]);
79      }
80  }
