13  package org.ejbca.util;
14  
15  import java.util.TreeSet;
16  import java.util.regex.Matcher;
17  import java.util.regex.Pattern;
18  
19  /**
20   * A list of slots numbers and indexes. Can contain ranges and individual entries.
21   * 
22   * @version $Id: SlotList.java 22117 2015-10-29 10:53:42Z mikekushner $
23   */
24  public class SlotList {
25      
26      private static final class Range implements Comparable<Range> { 
27          public final int min, max;
28          public Range(int min, int max) {
29              if (min > max) {
30                  throw new IllegalArgumentException("Minimum value ("+min+") in slot range is greater than maximum value ("+max+")");
31              }
32              this.min = min;
33              this.max = max;
34          }
35          
36          @Override
37          public int compareTo(Range o) {
38              if (min < o.min) return -1;
39              if (min > o.min) return 1;
40              else return 0;
41          }
42      }
43      
44      private final TreeSet<Range> ranges = new TreeSet<Range>();
45      private final TreeSet<Range> indexRanges = new TreeSet<Range>();
46      
47      
48      private int intval(String s, int defaultValue) {
49          return s != null ? Integer.valueOf(s) : defaultValue;
50      }
51      
52      private void addRange(Matcher m, int groupMin, int groupMax) {
53          final int min = intval(m.group(groupMin), Integer.MIN_VALUE);
54          final int max = intval(m.group(groupMax), Integer.MAX_VALUE);
55          addRangeTo(ranges, min, max);
56          
57      }
58      
59      private void addIndexRange(Matcher m, int groupMin, int groupMax) {
60          final int min = intval(m.group(groupMin), Integer.MIN_VALUE);
61          final int max = intval(m.group(groupMax), Integer.MAX_VALUE);
62          addRangeTo(indexRanges, min, max);
63      }
64      
65      private void addRangeTo(TreeSet<Range> r, int min, int max) {
66          Range toAdd = new Range(min, max);
67          
68          // Check for overlap with lower numbers
69          final Range lower = r.floor(toAdd);
70          if (lower != null && (lower.max >= min-1 || lower.max >= min)) { // special handling for integer overflows
71              if (lower.max >= max) return; // complete overlap
72              min = lower.min; // expand self
73              toAdd = new Range(min, max);
74              r.remove(lower);
75          }
76          
77          // Check for overlap with higher numbers
78          final Range higher = r.ceiling(toAdd);
79          if (higher != null && (higher.min <= max+1 || higher.min <= max)) {
80              if (higher.min <= min) return; // complete overlap
81              max = higher.max;
82              toAdd = new Range(min, max); // expand self
83              r.remove(higher);
84          }
85          
86          r.add(toAdd);
87      }
88      
89      
90      private static final Pattern slotListSingle = Pattern.compile("^([0-9]+)$");
91      private static final Pattern slotListRange = Pattern.compile("^([0-9]+)?-([0-9]+)?$");
92      private static final Pattern slotListISingle = Pattern.compile("^i([0-9]+)$");
93      private static final Pattern slotListIRange = Pattern.compile("^i([0-9]+)?-i([0-9]+)?$");
94      
95      public static SlotList fromString(String s) {
96          if (s == null) return null;
97          final SlotListSlotList">SlotList sl = new SlotList();
98          if (s.trim().isEmpty()) return sl;
99          
100         for (String piece : s.split(",")) {
101             piece = piece.trim();
102             Matcher m;
103             
104             // Single entries
105             m = slotListSingle.matcher(piece);
106             // create a range from the first matcher group
107             if (m.find()) { sl.addRange(m, 1, 1); continue; }
108             
109             m = slotListISingle.matcher(piece);
110             if (m.find()) { sl.addIndexRange(m, 1, 1); continue; }
111             
112             
113             // Range entries
114             m = slotListRange.matcher(piece);
115             // create a range from the matcher groups 1 and 2
116             if (m.find()) { sl.addRange(m, 1, 2); continue; }
117             
118             m = slotListIRange.matcher(piece);
119             if (m.find()) { sl.addIndexRange(m, 1, 2); continue; }
120             
121             throw new IllegalArgumentException("Invalid syntax of slot number or range: "+piece);
122         }
123         
124         return sl;
125     }
126     
127     public boolean contains(String slot) {
128         final boolean isIndexed = slot.startsWith("i");
129         final int number = Integer.valueOf(isIndexed ? slot.substring(1) : slot);
130         final TreeSet<Range> tree = (isIndexed ? indexRanges : ranges);
131         
132         Range lower = tree.floor(new Range(number, number));
133         return lower != null && number <= lower.max;
134     }
135     
136     @Override
137     public String toString() {
138         StringBuilder sb = new StringBuilder();
139         
140         // Slot numbers
141         for (Range r : ranges) {
142             if (r.min != Integer.MIN_VALUE) {
143                 sb.append(r.min);
144             }
145             if (r.min != r.max) {
146                 sb.append('-');
147                 if (r.max != Integer.MAX_VALUE) {
148                     sb.append(r.max);
149                 }
150             }
151             sb.append(", ");
152         }
153         
154         // Slot index numbers
155         for (Range r : indexRanges) {
156             sb.append('i');
157             if (r.min != Integer.MIN_VALUE) {
158                 sb.append(r.min);
159             }
160             if (r.min != r.max) {
161                 sb.append("-i");
162                 if (r.max != Integer.MAX_VALUE) {
163                     sb.append(r.max);
164                 }
165             }
166             sb.append(", ");
167         }
168         
169         // Remove trailing comma
170         if (sb.length() > 1) {
171             sb.delete(sb.length()-2, sb.length());
172         }
173         
174         return sb.toString();
175     }
