13  package org.ejbca.util;
14  
15  import java.util.Date;
16  import java.util.HashMap;
17  import java.util.regex.Matcher;
18  import java.util.regex.Pattern;
19  
20  import org.cesecore.util.ValidityDate;
21  
22  /**
23   * Base class generating parameter data for email notifications. Derived classes can add additional parameters.
24   * 
25   * The following parameters are set in this class
26   * ${NL}                           = New Line in message
27   * ${DATE} or ${current.DATE}      = The current date
28   * 
29   *
30   * @version $Id: NotificationParamGen.java 24437 2016-09-29 12:52:09Z anatom $
31   */
32  
33  public class NotificationParamGen {
34  
35    private HashMap<String, String> params = new HashMap<String, String>();	
36    
37    /** regexp pattern to match ${identifier} patterns */
38    private final static Pattern PATTERN = Pattern.compile("\\$\\{(.+?)\\}");
39    
40    protected NotificationParamGen() {
41  	  paramPut("NL", System.getProperty("line.separator"));
42        paramPut("DATE", fastDateFormat(new Date()));
43    }
44  
45    /**
46     * Method used to retrieve the populated parameter HashMap with the notification text.
47   * @return params
48     */
49    public HashMap<String, String> getParams(){
50  	  return params;
51    }
52  
53    /**
54     * method that makes sure that a "" is inserted instead of null
55     * @param key key
56     * @param value value
57     */
58    protected void paramPut(String key, String value){
59  	  if(value == null){
60  		  params.put(key, "");
61  	  }else{
62  		  params.put(key, value);
63  	  }
64    }
65    
66    /**
67     * method that makes sure that a "" is inserted instead of null
68     * @param key key
69     * @param value value
70     */
71    protected void paramPut(String key, Integer value){
72  	  if(value == null){
73  		  params.put(key, "");
74  	  }else{
75  		  params.put(key, value.toString());
76  	  }
77    }
78  	
79    // Help method used to populate a message 
80    /**
81     * Interpolate the patterns that exists on the input on the form '${pattern}'.
82     * @param input the input content to be interpolated
83     * @return the interpolated content
84     */
85    public String interpolate(String input) {
86  	  return interpolate(getParams(), input);
87    }
88  
89    /**
90     * Interpolate the patterns that exists on the input on the form '${pattern}'.
91   * @param patterns patterns
92     * @param input the input content to be interpolated
93     * @return the interpolated content
94     */
95    public static String interpolate(HashMap<String, String> patterns, String input) {
96        final Matcher m = PATTERN.matcher(input);
97        final StringBuffer sb = new StringBuffer(input.length());
98        while (m.find()) {
99            // when the pattern is ${identifier}, group 0 is 'identifier'
100           String key = m.group(1);
101           String value = patterns.get(key);
102           // if the pattern does exists, replace it by its value
103           // otherwise keep the pattern ( it is group(0) )
104           if (value != null) {
105               // $ is a group symbol in regexp replacement, see:
106               // http://stackoverflow.com/questions/11913709/why-does-replaceall-fail-with-illegal-group-reference
107               // since we can generate passwords etc with $ in them we need to escape $ signs in the value. The same applies for backslash.
108               m.appendReplacement(sb, Matcher.quoteReplacement(value));
109           } else {
110               // I'm doing this to avoid the backreference problem as there will be a $
111               // if I replace directly with the group 0 (which is also a pattern)
112               m.appendReplacement(sb, "");
113               String unknown = m.group(0);
114               sb.append(unknown);
115           }
116       }
117       m.appendTail(sb);
118       return sb.toString();
119   }
120   
121   protected String fastDateFormat(Date date) {
122 	  return ValidityDate.formatAsISO8601(date, ValidityDate.TIMEZONE_SERVER);
123   }
124 }
