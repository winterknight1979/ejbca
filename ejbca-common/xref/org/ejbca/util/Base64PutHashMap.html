14  package org.ejbca.util;
15  
16  import java.util.HashMap;
17  import java.util.Map;
18  
19  import org.cesecore.util.StringTools;
20  
21  /** Only used for backwards compatibility with earlier versions of EJBCA
22   * @see org.cesecore.util.Base64PutHashMap
23   * @deprecated now
24   * 
25   * @version $Id: Base64PutHashMap.java 23091 2016-03-29 12:04:12Z samuellb $
26   */
27  @Deprecated
28  public class Base64PutHashMap extends HashMap<Object, Object> {
29      private static final long serialVersionUID = 4700506858751520533L;
30  
31      public Base64PutHashMap() {
32          super();
33      }
34  
35      public Base64PutHashMap(Map<?, ?> m) {
36          super(m);
37      }
38  
39      public Object put(Object key, Object value) {
40          if (value == null) {
41              return super.put(key, value);
42          }
43          if (value instanceof String) {
44              String s = StringTools.putBase64String((String) value);
45              return super.put(key, s);
46          }
47          return super.put(key, value);
48      }
49  
50  }
