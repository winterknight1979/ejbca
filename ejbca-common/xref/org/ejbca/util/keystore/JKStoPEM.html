40  /**
41   * JKStoPEM is used to export PEM files from a single jks file. The class exports the user
42   * certificate, user private key in seperated files and the chain of sub ca and ca certifikate in
43   * a third file. The PEM files will have the names <i>common name</i>.pem, <i>common
44   * name</i>Key.pem and <i>common name</i>CA.pem derived from the DN in user certificate.
45   *
46   * @version $Id: JKStoPEM.java 22117 2015-10-29 10:53:42Z mikekushner $
47   */
48  public class JKStoPEM {
49      String exportpath = "./p12/pem/";
50      String jksFile;
51      String password;
52      String keypass;
53      KeyStore ks = null;
54  
55      boolean overwrite = false;
56      byte[] beginCertificate = "-----BEGIN CERTIFICATE-----".getBytes();
57      byte[] endCertificate = "-----END CERTIFICATE-----".getBytes();
58      byte[] beginPrivateKey = "-----BEGIN PRIVATE KEY-----".getBytes();
59      byte[] endPrivateKey = "-----END PRIVATE KEY-----".getBytes();
60      byte[] NL = "\n".getBytes();
61  
62      /**
63       * DOCUMENT ME!
64       *
65       * @param args DOCUMENT ME!
66       */
67      public static void main(String[] args) {
68  
69          // Bouncy Castle security provider
70          CryptoProviderTools.installBCProvider();
71  
72          JKStoPEM jks = null;
73  
74          try {
75              if (args.length > 4) {
76                  boolean overwrite = false;
77  
78                  if (args[4].equalsIgnoreCase("true")) {
79                      overwrite = true;
80                  }
81  
82                  jks = new JKStoPEM(args[0], args[1], args[2], args[3], overwrite);
83              } else if (args.length > 3) {
84                  jks = new JKStoPEM(args[0], args[1], args[2], args[3]);
85              } else {
86                  System.out.println(
87                      "Usage: JKStoPEM <jksFile> <jkspassword> <keypassword> <outpath> [overwrite (true/false)(default false)]");
88                  System.exit(0); // NOPMD this is a cli command
89              }
90  
91              jks.createPEM();
92          } catch (Exception e) {
93              e.printStackTrace();
94          }
95      }

100      * @param jksFile jksFile The (path +) name of the input jks file.

102      * @param keypass Key

105      */

111     }

143      * @param jksFile jksFile The (path +) name of the input jks file.

145      * @param keypass Key
146      * @param outpath Out

149     public JKStoPEM(String jksFile, String password,  String keypass, String outpath, boolean overwrite) {
150         this.jksFile = jksFile;
151         this.password = password;
152         this.overwrite = overwrite;
153         this.keypass = keypass;
154         exportpath = outpath;

157     /**
158      * DOCUMENT ME!
159      *
160      * @throws KeyStoreException DOCUMENT ME!
161      * @throws FileNotFoundException DOCUMENT ME!
162      * @throws IOException DOCUMENT ME!
163      * @throws NoSuchAlgorithmException DOCUMENT ME!
164      * @throws CertificateEncodingException DOCUMENT ME!
165      * @throws CertificateException DOCUMENT ME!
166      * @throws UnrecoverableKeyException DOCUMENT ME!
167      */
168     public void createPEM()
169         throws KeyStoreException, FileNotFoundException, IOException,
170             NoSuchAlgorithmException, CertificateEncodingException, CertificateException,
171             UnrecoverableKeyException {
172 
173          if(this.ks == null){
174             ks = KeyStore.getInstance("JKS");
175             InputStream in = new FileInputStream(jksFile);
176             ks.load(in, password.toCharArray());
177             in.close();
178         }
179         // Find the key private key entry in the keystore
180         Enumeration<String> e = ks.aliases();
181         Object o = null;
182         PrivateKey serverPrivKey = null;
183 
184         while (e.hasMoreElements()) {
185             o = e.nextElement();
186 
187             if (o instanceof String) {
188                 if ((ks.isKeyEntry((String) o)) &&
189                         ((serverPrivKey = (PrivateKey) ks.getKey((String) o, keypass.toCharArray())) != null)) {
190                     break;
191                 }
192             }
193         }
194 
195 
196         byte[] privKeyEncoded = "".getBytes();
197 
198         if (serverPrivKey != null) {
199             privKeyEncoded = serverPrivKey.getEncoded();
200         }
201 
202         //Certificate chain[] = ks.getCertificateChain((String) o);
203         Certificate[] chain = KeyTools.getCertChain(ks, (String) o);
204  
205         X509Certificate userX509Certificate = (X509Certificate) chain[0];
206 
207         byte[] output = userX509Certificate.getEncoded();
208         String sn = CertTools.getSubjectDN(userX509Certificate);
209         String userFile = CertTools.getPartFromDN(sn, "CN");
210         String filetype = ".pem";
211 
212         File path = new File(exportpath);
213         path.mkdir();
214 
215         File tmpFile = new File(path, userFile + filetype);
216 
217         if (!overwrite) {
218             if (tmpFile.exists()) {
219                 System.out.println("File '" + tmpFile + "' already exists, don't overwrite.");
220 
221                 return;
222             }
223         }
224 
225         OutputStream out = new FileOutputStream(tmpFile);
226         out.write(beginCertificate);
227         out.write(NL);
228 
229         byte[] userCertB64 = Base64.encode(output);
230         out.write(userCertB64);
231         out.write(NL);
232         out.write(endCertificate);
233         out.close();
234 
235         tmpFile = new File(path, userFile + "-Key" + filetype);
236 
237         if (!overwrite) {
238             if (tmpFile.exists()) {
239                 System.out.println("File '" + tmpFile + "' already exists, don't overwrite.");
240 
241                 return;
242             }
243         }
244 
245         out = new FileOutputStream(tmpFile);
246         out.write(beginPrivateKey);
247         out.write(NL);
248 
249         byte[] privKey = Base64.encode(privKeyEncoded);
250         out.write(privKey);
251         out.write(NL);
252         out.write(endPrivateKey);
253         out.close();
254 
255         tmpFile = new File(path, userFile + "-CA" + filetype);
256 
257         if (!overwrite) {
258             if (tmpFile.exists()) {
259                 System.out.println("File '" + tmpFile + "' already exists, don't overwrite.");
260 
261                 return;
262             }
263         }
264 
265         if (CertTools.isSelfSigned(userX509Certificate)) {
266             System.out.println(
267                 "User certificate is selfsigned, this is a RootCA, no CA certificates written.");
268         } else {
269             out = new FileOutputStream(tmpFile);
270 
271             for (int num = 1; num < chain.length; num++) {
272                 X509Certificate tmpX509Cert = (X509Certificate) chain[num];
273                 byte[] tmpOutput = tmpX509Cert.getEncoded();
274                 out.write(beginCertificate);
275                 out.write(NL);
276 
277                 byte[] tmpCACertB64 = Base64.encode(tmpOutput);
278                 out.write(tmpCACertB64);
279                 out.write(NL);
280                 out.write(endCertificate);
281                 out.write(NL);
282             }
283             out.close();
284         }
285     } // createPEM
