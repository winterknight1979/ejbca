32  /**
33   * Helper class for reading the index meta data from the database using direct JDBC.
34   * 
35   * @version $Id: DatabaseIndexUtil.java 31557 2019-02-22 12:26:04Z anatom $
36   */
37  public abstract class DatabaseIndexUtil {
38  
39      private static final Logger log = Logger.getLogger(DatabaseIndexUtil.class);
40  
41      /** Private helper class to help sorting the columns in the right order even if the database would return them in a different order than the ordinal */
42      private static class OrdinalColumn implements Comparable<OrdinalColumn> {
43          final short ordinalPosition;
44          final String columnName;
45          
46          private OrdinalColumn(final short ordinalPosition, final String columnName) {
47              this.ordinalPosition = ordinalPosition;
48              this.columnName = columnName;
49          }
50  
51          @Override
52          public int compareTo(final OrdinalColumn other) {
53              return this.ordinalPosition-other.ordinalPosition;
54          }
55      }
56      
57      /** Database index representation. */
58      public static class DatabaseIndex implements Serializable {
59          private static final long serialVersionUID = 1L;
60          private final String indexName;
61          private final transient List<OrdinalColumn> ordinalColumns = new ArrayList<>();
62          private List<String> columnNames = new ArrayList<>();
63          private final boolean nonUnique;
64  
65          public DatabaseIndex(final String indexName, final boolean nonUnique) {
66              this.indexName = indexName;
67              this.nonUnique = nonUnique;
68          }
69  
70          private void appendOrdinalColumn(final OrdinalColumn ordinalColumn) {
71              ordinalColumns.add(ordinalColumn);
72              Collections.sort(ordinalColumns);
73              columnNames.clear();
74              for (final OrdinalColumn current : ordinalColumns) {
75                  columnNames.add(current.columnName);
76              }
77          }
78  
79          /** @return the name of the index as reported by the database */
80          public String getIndexName() { return indexName; }
81  
82          /** @return the column names in the correct order as reported by the database */
83          public List<String> getColumnNames() { return columnNames; }
84  
85          /** @return true if the index is reported as unique */
86          public boolean isUnique() { return !nonUnique; }
87  
88          /** Case insensitive check if all columns present in the argument is also exactly present in this index. 
89           * @param columnNames columns
90           * @return bool */
91          public boolean isExactlyOverColumns(final List<String> columnNames) {
92              final List<String> indexColumnNames = new ArrayList<>();
93              for (final String indexColumnName : getColumnNames()) {
94                  indexColumnNames.add(indexColumnName.toLowerCase());
95              }
96              for (final String columnName : columnNames) {
97                  if (!indexColumnNames.remove(columnName.toLowerCase())) {
98                      return false;
99                  }
100             }
101             return indexColumnNames.isEmpty();
102         }
103     }
104 
105     /** @param dataSource DS
106      * @param tableName Table
107      * @param columnNames Columns
108      * @param requireUnique bool
109      * @return true if there exists an index on the specified table exactly matches the requested columns and optionally is unique. null if the check was inconclusive. */
110     public static Boolean isIndexPresentOverColumns(final DataSource dataSource, final String tableName, final List<String> columnNames, final boolean requireUnique) {
111         if (dataSource!=null) {
112             try {
113                 final List<DatabaseIndex> databaseIndexes = getDatabaseIndexFromTable(dataSource, tableName, requireUnique);
114                 if (databaseIndexes.isEmpty()) {
115                     if (log.isDebugEnabled()) {
116                         log.debug("Failed to read any index meta data from the database for table '" + tableName + "'. At least a primary key index was expected.");
117                     }
118                 } else {
119                     for (final DatabaseIndex databaseIndex : databaseIndexes) {
120                         if (databaseIndex.isExactlyOverColumns(columnNames)) {
121                             return Boolean.TRUE;
122                         }
123                     }
124                     return Boolean.FALSE;
125                 }
126             } catch (SQLException e) {
127                 if (log.isDebugEnabled()) {
128                     log.debug("Failed to read index meta data from the database for table '" + tableName + "'.", e);
129                 }
130             }
131         }
132         return null;
133     }

135     /** @param dataSource DS
136      * @param tableName Table
137      * @param requireUnique bool
138      * @return a list of representations of each database index present for a table 
139      * @throws SQLException fail */
140     public static List<DatabaseIndex> getDatabaseIndexFromTable(final DataSource dataSource, final String tableName, final boolean requireUnique) throws SQLException {
141         final List<DatabaseIndex> ret = new ArrayList<>();
142         try (final Connection connection = dataSource.getConnection();) {
143             final DatabaseMetaData databaseMetaData = connection.getMetaData();
144             /*
145              * Table names are case sensitive on at least Oracle XE (upper case) and MySQL 5.5 (camel case).
146              * 
147              * On MySQL the "catalog" is the database.
148              * On Oracle XE the username used to access the db is the schema.
149              * 
150              * This is an attempt at a very defensive version where we assume as little as possible about the database and it's configuration.
151              */
152             final Map<String,DatabaseIndex> tableIndexMap = new HashMap<>();
153             // First try the simple case that has been shown to work on MariaDB 5.5 (but where the returned table name apparently does not work)
154             tableIndexMap.putAll(getDatabaseIndexMap(databaseMetaData, null, null, tableName, requireUnique));
155             // If this failed, try the searching for the table as returned by the database meta data
156             if (tableIndexMap.isEmpty()) {
157                 log.trace("Looking up all available tables available in the datasource to find a matching table.");
158                 try (final ResultSet resultSetSchemas = databaseMetaData.getTables(null, null, null, null)) {
159                     while (resultSetSchemas.next()) {
160                         final String tableCatalog = resultSetSchemas.getString("TABLE_CAT");
161                         final String tableSchema = resultSetSchemas.getString("TABLE_SCHEM");
162                         final String tableName2 = resultSetSchemas.getString("TABLE_NAME");
163                         final String tableType = resultSetSchemas.getString("TABLE_TYPE");
164                         if (log.isTraceEnabled()) {
165                             log.trace(" catalog: " + tableCatalog + " tableSchema: " + tableSchema + " tableName: " + tableName2 + " tableType: " + tableType);
166                         }
167                         if ("TABLE".equals(tableType.toUpperCase(Locale.ENGLISH)) && tableName.toUpperCase(Locale.ENGLISH).equals(tableName2.toUpperCase(Locale.ENGLISH))) {
168                             if (log.isDebugEnabled()) {
169                                 log.debug("Will perform index detection on "+tableType+" in catalog " + null + " schema " + null + " table '" + tableName2 + "'.");
170                             }
171                             tableIndexMap.putAll(getDatabaseIndexMap(databaseMetaData, null, null, tableName2, requireUnique));
172                             if (tableIndexMap.isEmpty()) {
173                                 // Fall-back since null arguments apparently did not match the table and assume that this will find the correct one
174                                 if (log.isDebugEnabled()) {
175                                     log.debug("Will perform index detection on "+tableType+" in catalog '" + tableCatalog + "' schema '" + tableSchema + "' table '" + tableName2 + "'.");
176                                 }
177                                 tableIndexMap.putAll(getDatabaseIndexMap(databaseMetaData, tableCatalog, tableSchema, tableName2, requireUnique));
178                             }
179                             break;
180                         }
181                     }
182                 }
183             }
184             ret.addAll(tableIndexMap.values());
185         }
186         return ret;
187     }

189     /** @param databaseMetaData Metadata
190      * @param catalog Catalog 
191      * @param schemaName Schema
192      * @param tableName Table
193      * @param requireUnique Bool
194      * @return a Map of index name and the index representations of each database index present for a schema and table 
195      * @throws SQLException Error */
196     private static Map<String, DatabaseIndex> getDatabaseIndexMap(final DatabaseMetaData databaseMetaData, final String catalog, final String schemaName, final String tableName, final boolean requireUnique) throws SQLException {
197         final Map<String,DatabaseIndex> tableIndexMap = new HashMap<>();
198         // http://docs.oracle.com/javase/7/docs/api/java/sql/DatabaseMetaData.html#getIndexInfo(java.lang.String,%20java.lang.String,%20java.lang.String,%20boolean,%20boolean)
199         try (final ResultSet resultSet = databaseMetaData.getIndexInfo(catalog, schemaName, tableName, requireUnique, /*approximate=*/true);) {
200             while (resultSet.next()) {
201                 final String indexName = resultSet.getString("INDEX_NAME");
202                 if (indexName==null) {
203                     log.trace("Ignoring index of type tableIndexStatistic.");
204                     continue;
205                 }
206                 final boolean nonUnique = resultSet.getBoolean("NON_UNIQUE");
207                 if (!tableIndexMap.containsKey(indexName)) {
208                     tableIndexMap.put(indexName, new DatabaseIndex(indexName, nonUnique));
209                 }
210                 final DatabaseIndex databaseIndex = tableIndexMap.get(indexName);
211                 final String columnName = resultSet.getString("COLUMN_NAME");
212                 final short ordinalPosition = resultSet.getShort("ORDINAL_POSITION");
213                 databaseIndex.appendOrdinalColumn(new OrdinalColumn(ordinalPosition, columnName));
214                 if (log.isDebugEnabled()) {
215                     // Extract additional info if we are debug logging
216                     final short type = resultSet.getShort("TYPE");
217                     final String typeString;
218                     switch (type) {
219                     case DatabaseMetaData.tableIndexStatistic: typeString = "tableIndexStatistic"; break;
220                     case DatabaseMetaData.tableIndexClustered: typeString = "tableIndexClustered"; break;
221                     case DatabaseMetaData.tableIndexHashed: typeString = "tableIndexHashed"; break;
222                     case DatabaseMetaData.tableIndexOther: typeString = "tableIndexOther"; break;
223                     default: typeString = "unknown";
224                     }
225                     log.debug("Detected part of index on table '" + tableName + "' indexName: " + indexName + " column["+ordinalPosition+"]: " + columnName +
226                             " unique: " + !nonUnique + " type: " + typeString + " current columns: " + Arrays.toString(databaseIndex.getColumnNames().toArray()));
227                 }
228             }
229         }
230         return tableIndexMap;
231     }
