13  package org.ejbca.util.query;
14  
15  import java.io.Serializable;
16  import java.util.ArrayList;
17  import java.util.Date;
18  import java.util.List;
19  
20  import org.apache.log4j.Logger;
21  import org.cesecore.util.StringTools;
22  
23  /**
24   * A class used to produce advanced queries from the user data and approval tables.
25   * 
26   * It's main function is getQueryString which returns a string which should be placed in the 'WHERE' clause of a SQL
27   * statement.
28   *
29   * @version $Id: Query.java 25704 2017-04-18 14:42:43Z jeklund $
30   */
31  public class Query implements Serializable {
32  
33      private static final long serialVersionUID = 2020709050743730820L;
34   
35      private static Logger log = Logger.getLogger(Query.class);
36  
37      /** @deprecated Type of query is no longer support and only kept as a reminder not to re-reuse the value 0 */
38      @Deprecated
39      private static final int TYPE_LOGQUERY = 0;
40      public static final int TYPE_USERQUERY = 1;
41      public static final int TYPE_APPROVALQUERY = 2;
42      public static final int CONNECTOR_AND = 0;
43      public static final int CONNECTOR_OR = 1;
44      public static final int CONNECTOR_ANDNOT = 2;
45      public static final int CONNECTOR_ORNOT = 3;
46  
47      static final String[] CONNECTOR_SQL_NAMES = { " AND ", " OR ", " AND NOT ", " OR NOT " };
48  
49      private final List<BasicMatch> matches = new ArrayList<>();
50      private final List<Integer> connectors = new ArrayList<Integer>();
51      protected final int type;
52      private boolean hasIllegalSqlChars = false;
53      
54      /**
55       * Creates a new instance of Query
56       *
57       * @param type is the type of query to produce. Should be one of the 'TYPE' constants of this class.
58       */
59      public Query(final int type) {
60          this.type = type;
61      }
62  
63      /**
64       * Adds a time restraint to the query. Both parameter cannot be null This method should only be
65       * used in ra user queries.
66       *
67       * @param startdate gives the start date of the query or null if it no startdate.
68       * @param enddate gives the end date of the query or null if it no startdate.
69       */
70      public void add(Date startdate, Date enddate) {
71          matches.add(new TimeMatch(type, startdate, enddate));
72      }
73  
74      /**
75       * Adds a time restraint to the query. Both start and enddate parameters cannot be null This
76       * method should only be used in ra user queries.
77       *
78       * @param matchwith should indicate which field to match with, on of the TimeMatch.MATCH_WITH
79       *        constants.
80       * @param startdate gives the start date of the query or null if it no startdate.
81       * @param enddate gives the end date of the query or null if it no startdate.
82       */
83      public void add(int matchwith, Date startdate, Date enddate) {
84          matches.add(new TimeMatch(type, matchwith, startdate, enddate));
85      }
86  
87      /**
88       * Adds a time restraint and a connector to the query. Both parameter cannot be null. This
89       * method should only be used in log queries.
90       *
91       * @param startdate gives the start date of the query or null if it no startdate.
92       * @param enddate gives the end date of the query or null if it no startdate.
93       * @param connector should be one of the 'CONNECTOR' constants.
94       */
95      public void add(Date startdate, Date enddate, int connector) {
96          matches.add(new TimeMatch(type, startdate, enddate));
97          connectors.add(Integer.valueOf(connector));
98      }
99  
100     /**
101      * Adds a time restraint and a connector to the query. Both start and enddate parameters cannot
102      * be null. This method should only be used in ra user queries.
103      *
104      * @param matchwith should indicate which field to match with, on of the TimeMatch.MATCH_WITH
105      *        constants.
106      * @param startdate gives the start date of the query or null if it no startdate.
107      * @param enddate gives the end date of the query or null if it no startdate.
108      * @param connector should be one of the 'CONNECTOR' constants.
109      */
110     public void add(int matchwith, Date startdate, Date enddate, int connector) {
111         matches.add(new TimeMatch(type, matchwith, startdate, enddate));
112         connectors.add(Integer.valueOf(connector));
113     }
114 
115     /**
116      * Adds a match of type UserMatch or ApprovalMatch to the query.
117      *
118      * @param matchwith should be one of the the UserMatch.MATCH_WITH or LogMatch.MATCH_WITH
119      *        constants depending on query type.
120      * @param matchtype should be one of BasicMatch.MATCH_TYPE constants.
121      * @param matchvalue should be a string representation to match against.
122      *
123      * @throws NumberFormatException if there is an illegal character in matchvalue string.
124      */
125     public void add(int matchwith, int matchtype, String matchvalue) throws NumberFormatException {
126         switch (this.type) {
127         case TYPE_USERQUERY:
128             matches.add(new UserMatch(matchwith, matchtype, matchvalue));
129             break;
130         case TYPE_APPROVALQUERY:
131             matches.add(new ApprovalMatch(matchwith, matchtype, matchvalue));
132             break;
133         case TYPE_LOGQUERY:
134         default:
135             throw new IllegalArgumentException("Query type '"+type+"' is unknown.");
136         }
137         hasIllegalSqlChars |= !StringTools.hasSqlStripCharsAssumingSingleQuoteEscape(matchvalue).isEmpty();
138     }
139 
140     /**
141      * Adds a match of type UserMatch or ApprovalMatch and a connector to the query.
142      *
143      * @param matchwith should be one of the the UserMatch.MATCH_WITH or ApprovalMatch.MATCH_WITH depending on query type.
144      * @param matchtype should be one of BasicMatch.MATCH_TYPE constants.
145      * @param matchvalue should be a string representation to match against.
146      * @param connector should be one of the 'CONNECTOR' constants.
147      *
148      * @throws NumberFormatException if there is an illegal character in matchvalue string.
149      */
150     public void add(int matchwith, int matchtype, String matchvalue, int connector) throws NumberFormatException {
151         add(matchwith,matchtype,matchvalue);
152         connectors.add(Integer.valueOf(connector));
153     }
154 
155     /**
156      * Adds a connector to the query.
157      *
158      * @param connector should be one of the 'CONNECTOR' constants.
159      *
160      * @throws NumberFormatException if there is an illegal character in matchvalue string.
161      */
162     public void add(int connector) {
163         connectors.add(Integer.valueOf(connector));
164     }
165 
166     /**
167      * Gives the string to be used in the 'WHERE' clause int the SQL-statement.
168      *
169      * @return the string to be used in the 'WHERE'-clause.
170      */
171     public String getQueryString() {
172         String returnval = "";
173         for (int i = 0; i < (matches.size() - 1); i++) {
174             returnval += matches.get(i).getQueryString();
175             returnval += CONNECTOR_SQL_NAMES[connectors.get(i).intValue()];
176         }
177         returnval += matches.get(matches.size() - 1).getQueryString();
178         return returnval;
179     }
180 
181     /**
182      * Checks if the present query is legal by checking if every match is legal and that the number
183      * of connectors is one less than matches.
184      *
185      * @return true if the query is legal, false otherwise
186      */
187     public boolean isLegalQuery() {
188         boolean returnval = true;
189         for (final BasicMatch match : matches) {
190             returnval &= match.isLegalQuery();
191             if (!returnval) {
192                 log.info("Query is illegal due to match: " + match.getQueryString());
193             }
194         }
195         returnval &= (matches.size() - 1) == connectors.size();
196         returnval &= matches.size() > 0;
197         returnval &= !hasIllegalSqlChars();
198         return returnval;
199     }
200 
201     /**
202      * @return true if any of the match values added so far contain illegal SQL string characters.
203      * @see org.cesecore.util.StringTools#hasSqlStripChars(String)
204      * @see org.ejbca.util.query.BasicMatch#escapeSql(String)
205      */
206     private boolean hasIllegalSqlChars() {
207         if (log.isDebugEnabled()) {
208             log.debug("hasIllegalSqlChars: "+hasIllegalSqlChars);
209         }
210         return hasIllegalSqlChars;
211     }
212 
213     /** @return the current query string */
214     @Override
215     public String toString() {
216         return getQueryString();
217     }
218 }
