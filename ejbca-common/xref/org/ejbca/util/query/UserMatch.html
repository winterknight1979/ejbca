13  package org.ejbca.util.query;
14  
15  import org.apache.commons.lang.StringUtils;
16  
17  /**
18   * A class used by Query class to build a query for EJBCA RA module.
19   *
20   * @version $Id: UserMatch.java 26481 2017-09-04 11:43:58Z henriks $
21   */
22  public class UserMatch extends BasicMatch {
23  
24      private static final long serialVersionUID = 5458563135026714888L;
25  
26      public static final int MATCH_NONE                     = -1;
27  
28      public static final int MATCH_WITH_USERNAME            = 0;
29      public static final int MATCH_WITH_EMAIL               = 1;
30      public static final int MATCH_WITH_STATUS              = 2; // Value must the number representation.
31      public static final int MATCH_WITH_ENDENTITYPROFILE    = 3; // Matches the profile id not profilename.
32      public static final int MATCH_WITH_CERTIFICATEPROFILE  = 4; // Matches the certificatetype id not name.
33      public static final int MATCH_WITH_CA                  = 5; // Matches the CA id not CA name.
34  	public static final int MATCH_WITH_TOKEN               = 6;
35  	public static final int MATCH_WITH_DN                  = 7;
36      // Subject DN fields.
37      public static final int MATCH_WITH_UID                 = 100;
38      public static final int MATCH_WITH_COMMONNAME          = 101;
39      public static final int MATCH_WITH_DNSERIALNUMBER      = 102;
40      public static final int MATCH_WITH_GIVENNAME           = 103;
41      public static final int MATCH_WITH_INITIALS            = 104;
42      public static final int MATCH_WITH_SURNAME             = 105;
43      public static final int MATCH_WITH_TITLE               = 106;
44      public static final int MATCH_WITH_ORGANIZATIONALUNIT  = 107;
45      public static final int MATCH_WITH_ORGANIZATION        = 108;
46      public static final int MATCH_WITH_LOCALITY            = 109;
47      public static final int MATCH_WITH_STATEORPROVINCE     = 110;
48      public static final int MATCH_WITH_DOMAINCOMPONENT     = 111;
49      public static final int MATCH_WITH_COUNTRY             = 112;
50      // Subject Altname Fields
51      public static final int MATCH_WITH_RFC822NAME          = 200;
52      public static final int MATCH_WITH_DNSNAME             = 201;
53      public static final int MATCH_WITH_IPADDRESS           = 202;
54      public static final int MATCH_WITH_X400ADDRESS         = 203;
55      public static final int MATCH_WITH_DIRECTORYNAME       = 204;
56      public static final int MATCH_WITH_EDIPARTYNAME        = 205;
57      public static final int MATCH_WITH_URI                 = 206;
58      public static final int MATCH_WITH_REGISTEREDID        = 207;
59      public static final int MATCH_WITH_UPN                 = 208;
60      public static final int MATCH_WITH_GUID                = 209;
61  
62      static final String[] MATCH_WITH_SQLNAMES = {
63          "username", "subjectEmail", "status", "endEntityProfileId", "certificateProfileId", "cAId", "tokenType"
64      };
65  
66      // Represents the column names in ra userdata table.
67      private static final String MATCH_WITH_USERNAMESTRING = "UPPER(username)";
68      private static final String MATCH_WITH_SUBJECTDN = "UPPER(subjectDN)";
69      private static final String[] MATCH_WITH_SUBJECTDN_NAMES = {
70          "UID=", "CN=", "SN=", "GIVENNAME=", "INITIALS=", "SURNAME=", "T=", "OU=", "O=", "L=", "ST=",
71          "DC", "C="
72      };
73      
74      private static final String MATCH_WITH_SUBJECTALTNAME = "subjectAltName";
75      private static final String[] MATCH_WITH_SUBJECTALTNAME_NAMES = {
76          "RFC822NAME=", "DNSNAME=", "IPADDRESS=", "X400ADDRESS=", "DIRECTORYNAME=",
77          "EDIPARTYNAME=", "UNIFORMRESOURCEIDENTIFIER=", "REGISTEREDID=", "UPN=",  "GUID="
78      };
79  
80      /** For example UserMatch.MATCH_WITH_USERNAME */
81      private int matchwith;
82      /** For example BasicMatch.MATCH_TYPE_EQUALS */
83      private int matchtype;
84      /** For example a username */
85      private String matchvalue;
86  
87      /**
88       * Creates a new instance of UserMatch.
89       *
90       * @param matchwith determines which field i userdata table to match with.
91       * @param matchtype determines how to match the field. SubjectDN fields can only be matched with 'begins with'.
92       * @param matchvalue the value to match with.
93       *
94       * @throws NumberFormatException if matchvalue contains illegal numbervalue when matching number field.
95       */
96      public UserMatch(int matchwith, int matchtype, String matchvalue) throws NumberFormatException {
97          this.matchwith = matchwith;
98          this.matchtype = matchtype;
99          this.matchvalue = matchvalue;
100         if (matchwith >= MATCH_WITH_STATUS && matchwith <= MATCH_WITH_CA) {
101             Integer.valueOf(matchvalue);
102         }
103     }
104 
105     @Override
106     public String getQueryString() {
107         String returnval = "";
108         final String matchvalue = super.escapeSql(this.matchvalue).toUpperCase();
109         if (isSubjectDNMatch()) {
110             // Ignore MATCH_TYPE_EQUALS.
111             returnval = MATCH_WITH_SUBJECTDN + " LIKE '%" +
112             MATCH_WITH_SUBJECTDN_NAMES[matchwith - 100] + matchvalue + "%'";
113         } else if (isSubjectAltNameMatch()) {
114             returnval = MATCH_WITH_SUBJECTALTNAME + " LIKE '%" + MATCH_WITH_SUBJECTALTNAME_NAMES[matchwith - 200] + matchvalue + "%'";           
115         } else if (matchwith == MATCH_WITH_DN) {
116             if (matchtype == BasicMatch.MATCH_TYPE_EQUALS) {
117                 returnval = MATCH_WITH_SUBJECTDN + " = '" + matchvalue.trim() + "'";
118             } else if (matchtype == BasicMatch.MATCH_TYPE_BEGINSWITH) {
119                 returnval = MATCH_WITH_SUBJECTDN + " LIKE '" + matchvalue + "%'";
120             } else if (matchtype == BasicMatch.MATCH_TYPE_CONTAINS) {
121                 returnval = MATCH_WITH_SUBJECTDN + " LIKE '%" + matchvalue + "%'";
122             } 
123         } else if (matchwith == MATCH_WITH_USERNAME) {
124             if (matchtype == BasicMatch.MATCH_TYPE_EQUALS) {
125                 returnval = MATCH_WITH_USERNAMESTRING + " = '" + matchvalue.trim() + "'";
126             } else if (matchtype == BasicMatch.MATCH_TYPE_BEGINSWITH) {
127                 returnval = MATCH_WITH_USERNAMESTRING + " LIKE '" + matchvalue + "%'";
128             } else if (matchtype == BasicMatch.MATCH_TYPE_CONTAINS) {
129                 returnval = MATCH_WITH_USERNAMESTRING + " LIKE '%" + matchvalue + "%'";
130             }
131         } else if (matchtype == BasicMatch.MATCH_TYPE_EQUALS) {
132             // Because some databases (read JavaDB/Derby) does not allow matching of integer with a string expression
133             // like "where status='10'" instead of "where status=10", we have to have some special handling here.
134             String stringChar = "'";
135             if (matchwith == MATCH_WITH_STATUS || matchwith == MATCH_WITH_CA || matchwith == MATCH_WITH_CERTIFICATEPROFILE || 
136                     matchwith == MATCH_WITH_ENDENTITYPROFILE || matchwith == MATCH_WITH_TOKEN) {
137                 stringChar = "";
138             }
139             returnval = MATCH_WITH_SQLNAMES[matchwith] + " = "+stringChar + matchvalue.trim() + stringChar;
140         } else if (matchtype == BasicMatch.MATCH_TYPE_BEGINSWITH) {
141             returnval = MATCH_WITH_SQLNAMES[matchwith] + " LIKE '" + matchvalue + "%'";
142         }
143         return returnval;
144     }
145 
146     @Override
147     public boolean isLegalQuery() {
148         return StringUtils.isNotBlank(matchvalue);
149     }
150 
151     private boolean isSubjectDNMatch() {
152         return this.matchwith >= 100 && this.matchwith < 200;
153     }
154     
155     private boolean isSubjectAltNameMatch() {
156         return this.matchwith >= 200 && this.matchwith < 300;
157     }
158 }
