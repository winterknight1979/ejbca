13  package org.ejbca.util.query;
14  
15  import org.apache.commons.lang.StringUtils;
16  
17  /**
18   * A class used by Query class to build a custom query for EJBCA approval module.
19   *
20   * @version $Id: ApprovalMatch.java 25704 2017-04-18 14:42:43Z jeklund $
21   */
22  public class ApprovalMatch extends BasicMatch {
23  
24      private static final long serialVersionUID = -4891299802473333801L;
25  
26      /** Match with the id column (the primary key), for looking up a particular approval */
27      public static final int MATCH_WITH_UNIQUEID                      = 0;
28      /** Match with the computed id for the approval, for searching for identical approvals */
29      public static final int MATCH_WITH_APPROVALID                    = 1;
30      public static final int MATCH_WITH_APPROVALTYPE                  = 2;
31      public static final int MATCH_WITH_ENDENTITYPROFILEID            = 3;
32      public static final int MATCH_WITH_CAID                          = 4;
33      public static final int MATCH_WITH_REQUESTADMINCERTISSUERDN      = 5;
34      public static final int MATCH_WITH_REQUESTADMINCERTSERIALNUMBER  = 6;
35      public static final int MATCH_WITH_STATUS                        = 7; 
36      public static final int MATCH_WITH_REMAININGAPPROVALS            = 8;
37  
38      /** These refer to column names in the database and are used for native SQL querying. */
39      private static final String[] MATCH_WITH_SQLNAMES = {
40          "id", "approvalId", "approvalType", "endEntityProfileId", "cAId", "reqAdminCertIssuerDn", "reqAdminCertSn", "status", "remainingApprovals"
41      };
42  
43      private int matchwith;
44      private int matchtype;
45      private String matchvalue;
46  
47      /**
48       * Creates a new instance.
49       *
50       * @param matchwith determines which field in approval table to match with.
51       * @param matchtype determines how to match the field..
52       * @param matchvalue the value to match with.
53       *
54       * @throws NumberFormatException if matchvalue contains illegal numbervalue when matching number field.
55       */
56      public ApprovalMatch(final int matchwith, final int matchtype, final String matchvalue)
57          throws NumberFormatException {
58          this.matchwith = matchwith;
59          this.matchtype = matchtype;
60          this.matchvalue = matchvalue;
61          // The row below does not do anything but check that matchvalue contains
62          // a legal number value when matching number field. See @throws clause.
63          if (matchwith != MATCH_WITH_REQUESTADMINCERTISSUERDN &&
64              matchwith != MATCH_WITH_REQUESTADMINCERTSERIALNUMBER){
65              Integer.valueOf(matchvalue);
66          }
67      }
68  
69      @Override
70      public String getQueryString() {
71          String returnval = "";
72          final String matchvalue = super.escapeSql(this.matchvalue);
73          if (matchtype == BasicMatch.MATCH_TYPE_EQUALS) {
74              // Because some databases (read JavaDB/Derby) does not allow matching of integer with a string expression
75              // like "where status='10'" instead of "where status=10", we have to hav e some special handling here.
76              String stringChar = "'";
77              if ((matchwith >= MATCH_WITH_UNIQUEID && matchwith <= MATCH_WITH_CAID) || (matchwith == MATCH_WITH_STATUS) || (matchwith == MATCH_WITH_REMAININGAPPROVALS)) {
78                  stringChar = "";
79              }
80              returnval = MATCH_WITH_SQLNAMES[matchwith] + " = "+stringChar + matchvalue.trim() + stringChar;
81          }
82          if (matchtype == BasicMatch.MATCH_TYPE_BEGINSWITH) {
83              returnval = MATCH_WITH_SQLNAMES[matchwith] + " LIKE '" + matchvalue + "%'";
84          }
85          if (matchtype == BasicMatch.MATCH_TYPE_CONTAINS) {
86              returnval = MATCH_WITH_SQLNAMES[matchwith] + " LIKE '%" + matchvalue + "%'";
87          }
88          return returnval;
89      }
90  
91      @Override
92      public boolean isLegalQuery() {
93          return StringUtils.isNotBlank(matchvalue);
94      }
95  }
