14  package org.ejbca.util.passgen;
15  
16  import java.util.ArrayList;
17  import java.util.Collection;
18  
19  /**
20   * Factory class creating PasswordGenerators.
21   * Usage:
22   * <pre>
23   *  IPasswordGenerator pwdgen = PasswordGeneratorFactory.getInstance(PasswordGeneratorFactory.PASSWORDTYPE_ALLPRINTABLE);
24   *  String pwd = pwdgen.getNewPassword(12, 16);
25   * </pre>
26   *
27   * @version $Id: PasswordGeneratorFactory.java 22117 2015-10-29 10:53:42Z mikekushner $
28   */
29  public class PasswordGeneratorFactory {
30      
31      
32      public static final String PASSWORDTYPE_DIGITS                       = DigitPasswordGenerator.NAME;
33      public static final String PASSWORDTYPE_LETTERSANDDIGITS             = LettersAndDigitsPasswordGenerator.NAME;
34  	public static final String PASSWORDTYPE_ALLPRINTABLE                 = AllPrintableCharPasswordGenerator.NAME;	
35  	public static final String PASSWORDTYPE_NOLOOKALIKELD                 = NoLookALikeLDPasswordGenerator.NAME;	
36  	public static final String PASSWORDTYPE_NOSOUNDALIKEENLD                 = NoSoundALikeENLDPasswordGenerator.NAME;	
37  	public static final String PASSWORDTYPE_NOTALIKEENLD                 = NoLookOrSoundALikeENLDPasswordGenerator.NAME;	
38      
39      static final IPasswordGenerator[] classes = { new DigitPasswordGenerator(),
40      	                                          new LettersAndDigitsPasswordGenerator(),
41      	                                          new AllPrintableCharPasswordGenerator(),
42      	                                          new NoLookALikeLDPasswordGenerator(),
43      	                                          new NoSoundALikeENLDPasswordGenerator(),
44      	                                          new NoLookOrSoundALikeENLDPasswordGenerator()};
45      
46      /**
47       *  Method returning an instance of the specified IPasswordGenerator class.
48       *      
49       *  @param type should be on of the PasswordGeneratorFactory constants.
50       * @return generator
51       */
52      
53      public static IPasswordGenerator getInstance(String type){
54      	IPasswordGenerator ret = null;
55      	for (int i=0; i<classes.length; i++) {
56      		if (classes[i].getName().equals(type)) {
57      			ret = classes[i];
58      		}
59      	}
60  		return ret;
61      }
62  
63  	public static Collection<String> getAvailablePasswordTypes() {
64  		ArrayList<String> al = new ArrayList<String>();
65      	for (int i=0; i<classes.length; i++) {
66      		al.add(classes[i].getName());
67      	}
68  		return al;
69  	}
70     
71  }
