16  /**
17   * This class allows all letters and digits except those that sound similar in english like aj and eg.
18   *
19   * @version $Id: NoSoundALikeENLDPasswordGenerator.java 22117 2015-10-29 10:53:42Z mikekushner $
20   */
21  public class NoSoundALikeENLDPasswordGenerator extends BasePasswordGenerator {
22      
23      private static final char[] USEDCHARS = {'1','2','3','4','5','6','7','8','9','0',
24      																		'q','Q','w','W','r','R','t','T',
25      																		'y','Y','u','U','i','I','o','O','p','P',
26      																		's','S','d','D','f','F','h','H',
27      																		'k','K','l','L','z','Z','x','X','c','C',
28      																		'v','V','b','B','n','N','m','M'};
29          
30  	protected static final String NAME = "PWGEN_NOSOUNDALIKEENLD";
31      
32  	public String getName() { return NAME; }
33  	
34      public NoSoundALikeENLDPasswordGenerator(){
35      	super(USEDCHARS);
36      }
37        
38  }
