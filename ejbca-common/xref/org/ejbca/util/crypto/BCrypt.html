19  /**
20   * BCrypt implements OpenBSD-style Blowfish password hashing using
21   * the scheme described in "A Future-Adaptable Password Scheme" by
22   * Niels Provos and David Mazieres.
23   * <p>
24   * This password hashing system tries to thwart off-line password
25   * cracking using a computationally-intensive hashing algorithm,
26   * based on Bruce Schneier's Blowfish cipher. The work factor of
27   * the algorithm is parameterised, so it can be increased as
28   * computers get faster.
29   * <p>
30   * Usage is really simple. To hash a password for the first time,
31   * call the hashpw method with a random salt, like this:
32   * <p>
33   * <code>
34   * String pw_hash = BCrypt.hashpw(plain_password, BCrypt.gensalt()); <br>
35   * </code>
36   * <p>
37   * To check whether a plaintext password matches one that has been
38   * hashed previously, use the checkpw method:
39   * <p>
40   * <code>
41   * if (BCrypt.checkpw(candidate_password, stored_hash))<br>
42   * &nbsp;&nbsp;&nbsp;&nbsp;System.out.println("It matches");<br>
43   * else<br>
44   * &nbsp;&nbsp;&nbsp;&nbsp;System.out.println("It does not match");<br>
45   * </code>
46   * <p>
47   * The gensalt() method takes an optional parameter (log_rounds)
48   * that determines the computational complexity of the hashing:
49   * <p>
50   * <code>
51   * String strong_salt = BCrypt.gensalt(10)<br>
52   * String stronger_salt = BCrypt.gensalt(12)<br>
53   * </code>
54   * <p>
55   * The amount of work increases exponentially (2**log_rounds), so 
56   * each increment is twice as much work. The default log_rounds is
57   * 10, and the valid range is 1 to 31.
58   * 
59   * The original source from DM only allowed rounds in the interval 4-31. The
60   * ability to lower security further is a modification by Johan Eklund and is
61   * in no way a reflection on the opinion of DM.
62   *
63   * @author Damien Miller
64   * @version 0.3
65   * @version $Id: BCrypt.java 12654 2011-09-21 11:29:47Z mikekushner $
66   */

379 	/**
380 	 * Encode a byte array using bcrypt's slightly-modified base64
381 	 * encoding scheme. Note that this is *not* compatible with
382 	 * the standard MIME-base64 encoding.
383 	 *
384 	 * @param d	the byte array to encode
385 	 * @param len	the number of bytes to encode
386 	 * @return	base64-encoded string
387 	 * @exception IllegalArgumentException if the length is invalid
388 	 */
389 	private static String encodeBase64(final byte d[], final int len)
390 		throws IllegalArgumentException {
391 		int off = 0;
392 		final StringBuilder rs = new StringBuilder();
393 		int c1, c2;
394 
395 		if (len <= 0 || len > d.length) {
396 			throw new IllegalArgumentException ("Invalid len");
397 		}
398 		while (off < len) {
399 			c1 = d[off++] & 0xff;
400 			rs.append(BASE64_CODE[(c1 >> 2) & 0x3f]);
401 			c1 = (c1 & 0x03) << 4;
402 			if (off >= len) {
403 				rs.append(BASE64_CODE[c1 & 0x3f]);
404 				break;
405 			}
406 			c2 = d[off++] & 0xff;
407 			c1 |= (c2 >> 4) & 0x0f;
408 			rs.append(BASE64_CODE[c1 & 0x3f]);
409 			c1 = (c2 & 0x0f) << 2;
410 			if (off >= len) {
411 				rs.append(BASE64_CODE[c1 & 0x3f]);
412 				break;
413 			}
414 			c2 = d[off++] & 0xff;
415 			c1 |= (c2 >> 6) & 0x03;
416 			rs.append(BASE64_CODE[c1 & 0x3f]);
417 			rs.append(BASE64_CODE[c2 & 0x3f]);
418 		}
419 		return rs.toString();
420 	}

435 	/**
436 	 * Decode a string encoded using bcrypt's base64 scheme to a
437 	 * byte array. Note that this is *not* compatible with
438 	 * the standard MIME-base64 encoding.
439 	 * @param s	the string to decode
440 	 * @param maxolen	the maximum number of bytes to decode
441 	 * @return	an array containing the decoded bytes
442 	 * @throws IllegalArgumentException if maxolen is invalid
443 	 */
444 	private static byte[] decodeBase64(final String s, final int maxolen)
445 		throws IllegalArgumentException {
446 		final StringBuilder rs = new StringBuilder();
447 		int off = 0;
448 		final int slen = s.length();
449 		int olen = 0;
450 		byte ret[];
451 		byte c1, c2, c3, c4, o;
452 
453 		if (maxolen <= 0) {
454 			throw new IllegalArgumentException ("Invalid maxolen");
455 		}
456 		while (off < slen - 1 && olen < maxolen) {
457 			c1 = char64(s.charAt(off++));
458 			c2 = char64(s.charAt(off++));
459 			if (c1 == -1 || c2 == -1) {
460 				break;
461 			}
462 			o = (byte)(c1 << 2);
463 			o |= (c2 & 0x30) >> 4;
464 			rs.append((char)o);
465 			if (++olen >= maxolen || off >= slen) {
466 				break;
467 			}
468 			c3 = char64(s.charAt(off++));
469 			if (c3 == -1) {
470 				break;
471 			}
472 			o = (byte)((c2 & 0x0f) << 4);
473 			o |= (c3 & 0x3c) >> 2;
474 			rs.append((char)o);
475 			if (++olen >= maxolen || off >= slen) {
476 				break;
477 			}
478 			c4 = char64(s.charAt(off++));
479 			o = (byte)((c3 & 0x03) << 6);
480 			o |= c4;
481 			rs.append((char)o);
482 			++olen;
483 		}
484 
485 		ret = new byte[olen];
486 		for (off = 0; off < olen; off++) {
487 			ret[off] = (byte)rs.charAt(off);
488 		}
489 		return ret;
490 	}

492 	/**
493 	 * Blowfish encipher a single 64-bit block encoded as
494 	 * two 32-bit halves
495 	 * @param lr	an array containing the two 32-bit half blocks
496 	 * @param off	the position in the array of the blocks
497 	 */
498 	private final void encipher(final int lr[], final int off) {
499 		int i, n, l = lr[off], r = lr[off + 1];
500 
501 		l ^= bfP[0];
502 		for (i = 0; i <= BLOWFISH_NUM_ROUNDS - 2;) {
503 			// Feistel substitution on left word
504 			n = bfS[(l >> 24) & 0xff];
505 			n += bfS[0x100 | ((l >> 16) & 0xff)];
506 			n ^= bfS[0x200 | ((l >> 8) & 0xff)];
507 			n += bfS[0x300 | (l & 0xff)];
508 			r ^= n ^ bfP[++i];
509 
510 			// Feistel substitution on right word
511 			n = bfS[(r >> 24) & 0xff];
512 			n += bfS[0x100 | ((r >> 16) & 0xff)];
513 			n ^= bfS[0x200 | ((r >> 8) & 0xff)];
514 			n += bfS[0x300 | (r & 0xff)];
515 			l ^= n ^ bfP[++i];
516 		}
517 		lr[off] = r ^ bfP[BLOWFISH_NUM_ROUNDS + 1];
518 		lr[off + 1] = l;
519 	}

522 	 * Cycically extract a word of key material
523 	 * @param data	the string to extract the data from
524 	 * @param offp	a "pointer" (as a one-entry array) to the
525 	 * current offset into data
526 	 * @return	the next word of material from data
527 	 */
528 	private static int streamtoword(final byte data[], final int offp[]) {
529 		int i;
530 		int word = 0;
531 		int off = offp[0];
532 
533 		for (i = 0; i < 4; i++) {
534 			word = (word << 8) | (data[off] & 0xff);
535 			off = (off + 1) % data.length;
536 		}
537 
538 		offp[0] = off;
539 		return word;
540 	}

551 	 * Key the Blowfish cipher
552 	 * @param key	an array containing the key
553 	 */
554 	private void key(final byte key[]) {
555 		int i;
556 		final int koffp[] = { 0 };
557 		final int lr[] = { 0, 0 };
558 		final int plen = bfP.length;
559 		final int slen = bfS.length;
560 
561 		for (i = 0; i < plen; i++) {

569

571 			encipher(lr, 0);
572 			bfS[i] = lr[0];
573 			bfS[i + 1] = lr[1];
574 		}

577 	/**
578 	 * Perform the "enhanced key schedule" step described by
579 	 * Provos and Mazieres in "A Future-Adaptable Password Scheme"
580 	 * http://www.openbsd.org/papers/bcrypt-paper.ps
581 	 * @param data	salt information
582 	 * @param key	password information
583 	 */
584 	private void ekskey(final byte data[], final byte key[]) {
585 		int i;
586 		final int koffp[] = { 0 };
587 		final int doffp[] = { 0 };
588 		int lr[] = { 0, 0 };
589 		final int plen = bfP.length;
590 		final int slen = bfS.length;
591 
592 		for (i = 0; i < plen; i++) {
593 			bfP[i] = bfP[i] ^ streamtoword(key, koffp);
594 		}
595 		for (i = 0; i < plen; i += 2) {
596 			lr[0] ^= streamtoword(data, doffp);
597 			lr[1] ^= streamtoword(data, doffp);
598 			encipher(lr, 0);
599 			bfP[i] = lr[0];
600 			bfP[i + 1] = lr[1];
601 		}
602 
603 		for (i = 0; i < slen; i += 2) {
604 			lr[0] ^= streamtoword(data, doffp);
605 			lr[1] ^= streamtoword(data, doffp);
606 			encipher(lr, 0);
607 			bfS[i] = lr[0];
608 			bfS[i + 1] = lr[1];
609 		}
610 	}

612 	/**
613 	 * Perform the central password hashing step in the
614 	 * bcrypt scheme
615 	 * @param password	the password to hash
616 	 * @param salt	the binary salt to hash with the password
617 	 * @param log_rounds	the binary logarithm of the number
618 	 * of rounds of hashing to apply
619 	 * @return	an array containing the binary hashed password
620 	 */
621 	private byte[] cryptRaw(final byte password[], final byte salt[], final int log_rounds) {
622 		int rounds, i, j;
623 		final int cdata[] = (int[])BF_CRYPT_CIPHERTEXT.clone();
624 		final int clen = cdata.length;
625 		byte ret[];
626 
627 		// Original code modified to allow a minimum of 1 round, instead of the original 4
628 		if (log_rounds < 1 || log_rounds > 31) {
629 			throw new IllegalArgumentException ("Bad number of rounds");
630 		}
631 		rounds = 1 << log_rounds;
632 		if (salt.length != BCRYPT_SALT_LEN) {
633 			throw new IllegalArgumentException ("Bad salt length");
634 		}
635 		initKey();
636 		ekskey(salt, password);
637 		for (i = 0; i < rounds; i++) {
638 			key(password);
639 			key(salt);
640 		}
641 
642 		for (i = 0; i < 64; i++) {
643 			for (j = 0; j < (clen >> 1); j++) {
644 				encipher(cdata, j << 1);
645 			}
646 		}
647 
648 		ret = new byte[clen * 4];
649 		for (i = 0, j = 0; i < clen; i++) {
650 			ret[j++] = (byte)((cdata[i] >> 24) & 0xff);
651 			ret[j++] = (byte)((cdata[i] >> 16) & 0xff);
652 			ret[j++] = (byte)((cdata[i] >> 8) & 0xff);
653 			ret[j++] = (byte)(cdata[i] & 0xff);
654 		}
655 		return ret;
656 	}

658 	/**
659 	 * Hash a password using the OpenBSD bcrypt scheme
660 	 * @param password	the password to hash
661 	 * @param salt	the salt to hash with (perhaps generated
662 	 * using BCrypt.gensalt)
663 	 * @return	the hashed password
664 	 */
665 	public static String hashpw(final String password, final String salt) {
666 		BCrypt B;
667 		String real_salt;
668 		byte passwordb[], saltb[], hashed[];
669 		char minor = (char)0;
670 		int rounds, off = 0;
671 		if (salt.charAt(0) != '$' || salt.charAt(1) != '2') {
672 			throw new IllegalArgumentException ("Invalid salt version");
673 		}
674 		if (salt.charAt(2) == '$') {
675 			off = 3;
676 		} else {
677 			minor = salt.charAt(2);
678 			if (minor != 'a' || salt.charAt(3) != '$') {
679 				throw new IllegalArgumentException ("Invalid salt revision");
680 			}
681 			off = 4;
682 		}
683 
684 		// Extract number of rounds
685 		if (salt.charAt(off + 2) > '$') {
686 			throw new IllegalArgumentException ("Missing salt rounds");
687 		}
688 		rounds = Integer.parseInt(salt.substring(off, off + 2));
689 
690 		real_salt = salt.substring(off + 3, off + 25);
691 		try {
692 			passwordb = (password + (minor >= 'a' ? "\000" : "")).getBytes("UTF-8");
693 		} catch (UnsupportedEncodingException uee) {
694 			throw new AssertionError("UTF-8 is not supported");
695 		}
696 
697 		saltb = decodeBase64(real_salt, BCRYPT_SALT_LEN);
698 
699 		B = new BCrypt();
700 		hashed = B.cryptRaw(passwordb, saltb, rounds);
701 
702 		final StringBuilder rs = new StringBuilder("$2");
703 		if (minor >= 'a') {
704 			rs.append(minor);
705 		}
706 		rs.append('$');
707 		if (rounds < 10) {
708 			rs.append('0');
709 		}
710 		rs.append(Integer.toString(rounds));
711 		rs.append('$');
712 		rs.append(encodeBase64(saltb, saltb.length));
713 		rs.append(encodeBase64(hashed,
714 		    BF_CRYPT_CIPHERTEXT.length * 4 - 1));
715 		return rs.toString();
716 	}
