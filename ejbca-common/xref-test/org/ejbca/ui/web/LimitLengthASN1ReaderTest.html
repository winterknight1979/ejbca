107 	/**
108 	 * Test happy path implementation of readFirstASN1Object.
109 	 * @throws MalformedRequestException Fail
110 	 * @throws IOException Fail
111 	 */
112     @Test
113 	public void testReadFirstASN1ObjectHappyPath() throws MalformedRequestException, IOException {
114 		inputStreamStub = new InputStreamStub(VALID_STREAM);
115 		limitLengthASN1Reader = new LimitLengthASN1Reader(inputStreamStub, VALID_STREAM.length);
116 		byte[] result = limitLengthASN1Reader.readFirstASN1Object();
117 		for (int i = 0; i < result.length; i++) {
118 			assertEquals("Value #" + i + " did not match expected output.", VALID_STREAM[i], result[i]);
119 		}
120 	}

201 	/**
202 	 * Private inner class which represents an input stream with controlled output.
203 	 * 
204 	 * TODO: When our test runner is reformed to only run tests matching
205 	 * **\*Test.class, make this an inner class to LimitLengthASN1ReaderTest
206 	 * 
207 	 */
208 	class InputStreamStub extends InputStream {
209 
210 		private int[] contents;
211 		private int counter = 0;
212 
213 		public InputStreamStub(int[] contents) {
214 			super();
215 			this.contents = contents;
216 		}
217 
218 		public int read() throws IOException {
219 			if (contents == null) {
220 				throw new NullPointerException(
221 						"Class member contents must be set for anonymous inner class.");
222 			}
223 			if (counter < contents.length) {
224 				return contents[counter++];
225 			} else {
226 				return -1;
227 			}
228 		}
229 	}
