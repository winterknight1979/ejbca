42      /**
43       * Make sure that each token is single-use only. This test tests both CliAuthenticationToken and the CliAuthenticationTokenRegistry
44       * @throws AuthenticationFailedException  fail
45       */
46      @Test
47      public void testUseTokenWorksTwiceInternally() throws AuthenticationFailedException {
48          final Long referenceNumber = 0L;
49          final String passwordHash = "kittens";
50          CliAuthenticationToken authenticationToken = new CliAuthenticationToken(new UsernamePrincipal("TEST"), passwordHash, BCRYPT_SALT, referenceNumber,
51                  SupportedPasswordHashAlgorithm.SHA1_OLD);
52          CliAuthenticationTokenReferenceRegistry.INSTANCE.registerToken(authenticationToken);
53          AccessUserAspect accessUser = EasyMock.createMock(AccessUserAspect.class);
54          EasyMock.expect(accessUser.getMatchValue()).andReturn("TEST");
55          EasyMock.expect(accessUser.getTokenType()).andReturn(CliAuthenticationTokenMetaData.TOKEN_TYPE);
56          EasyMock.replay(accessUser);
57          assertTrue(authenticationToken.matches(accessUser));
58          // The token should still work
59          assertTrue(authenticationToken.matches(accessUser));
60  
61          EasyMock.verify(accessUser);
62      }
63      @Test
64      public void testUseTokenDoesNotWorkAfterSerialization() throws IOException, ClassNotFoundException, AuthenticationFailedException {
65          final Long referenceNumber = 0L;
66          final String passwordHash = "kittens";
67          CliAuthenticationToken authenticationToken = new CliAuthenticationToken(new UsernamePrincipal("TEST"), passwordHash, BCRYPT_SALT, referenceNumber,
68                  SupportedPasswordHashAlgorithm.SHA1_OLD);
69          CliAuthenticationTokenReferenceRegistry.INSTANCE.registerToken(authenticationToken);
70          AccessUserAspect accessUser = EasyMock.createMock(AccessUserAspect.class);
71          EasyMock.expect(accessUser.getTokenType()).andReturn(CliAuthenticationTokenMetaData.TOKEN_TYPE).times(2);
72          EasyMock.expect(accessUser.getMatchValue()).andReturn("TEST").times(2);
73          EasyMock.replay(accessUser);
74          assertTrue(authenticationToken.matches(accessUser));
75          // Serialize the token and pick it up again.
76          ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
77          ObjectOutputStream out = new ObjectOutputStream(byteArrayOutputStream);
78          out.writeObject(authenticationToken);
79          out.close();
80          byte[] serializedObject = byteArrayOutputStream.toByteArray();
81  
82          ObjectInputStream in = new ObjectInputStream(new ByteArrayInputStream(serializedObject));
83          CliAuthenticationToken deserializedObject = (CliAuthenticationToken) in.readObject();
84          in.close();
85  
86          // The token should be spent now.
87          try {
88              deserializedObject.matches(accessUser);
89              fail("CliAuthenticationFailedException should have been thrown");
90          } catch (AuthenticationFailedException e) {
91              //NOPMD
92          }
93  
94          EasyMock.verify(accessUser);
95      }

97      @Test
98      public void testReplayAttack() throws AuthenticationFailedException, SecurityException, NoSuchFieldException, IllegalArgumentException, IllegalAccessException  {
99          final Long referenceNumber = 0L;
100         final String passwordHash = "kittens";
101         CliAuthenticationToken authenticationToken = new CliAuthenticationToken(new UsernamePrincipal("TEST"), passwordHash, BCRYPT_SALT, referenceNumber,
102                 SupportedPasswordHashAlgorithm.SHA1_OLD);
103         CliAuthenticationTokenReferenceRegistry.INSTANCE.registerToken(authenticationToken);
104         AccessUserAspect accessUser = EasyMock.createMock(AccessUserAspect.class);
105         EasyMock.expect(accessUser.getMatchValue()).andReturn("TEST").times(2);
106         EasyMock.expect(accessUser.getTokenType()).andReturn(CliAuthenticationTokenMetaData.TOKEN_TYPE).times(2);
107         EasyMock.replay(accessUser);
108         assertTrue(authenticationToken.matches(accessUser));        
109         // Modify the token, so that it becomes "spent"
110         Field f = CliAuthenticationToken.class.getDeclaredField("isVerified");
111         f.setAccessible(true);
112         f.setBoolean(authenticationToken, false);
113         boolean caught = false;
114         try {
115             authenticationToken.matches(accessUser);
116         } catch (AuthenticationFailedException e) {
117             caught = true;
118         }
119         assertTrue("Authentication failure did not occur for replay attack.", caught);
120         EasyMock.verify(accessUser);
121     }

123     @Test 
124     public void testIncorrectPasswordBehaviour() throws Exception {
125         final Long referenceNumber = 0L;
126         final String passwordHash = "kittens";
127         CliAuthenticationToken authenticationToken = new CliAuthenticationToken(new UsernamePrincipal("TEST"), passwordHash, BCRYPT_SALT, referenceNumber,
128                 SupportedPasswordHashAlgorithm.SHA1_OLD);
129         CliAuthenticationTokenReferenceRegistry.INSTANCE.registerToken(authenticationToken);
130         AccessUserAspect accessUser = EasyMock.createMock(AccessUserAspect.class);
131         EasyMock.expect(accessUser.getMatchValue()).andReturn("TEST");
132         EasyMock.expect(accessUser.getTokenType()).andReturn(CliAuthenticationTokenMetaData.TOKEN_TYPE);
133         EasyMock.replay(accessUser);
134         authenticationToken.setSha1Hash("You're a kitty!");
135         boolean caught = false;
136         try {
137             authenticationToken.matches(accessUser);  
138         } catch(AuthenticationFailedException e) {
139             caught = true;
140         }
141         assertTrue("Authentication failure did not occur for incorrectly set password.", caught);
142         
143         EasyMock.verify(accessUser);
144     }
