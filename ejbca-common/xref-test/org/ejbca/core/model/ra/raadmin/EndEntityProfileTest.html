50      /**
51       * Tests the validation system, and the regex validator in particular.
52       * @throws EndEntityFieldValidatorException Fail
53       */
54      @Test
55      public void testRegexValidation() throws EndEntityFieldValidatorException {
56          // The regex validator ignored the dn component, so that can be set to anything
57          EndEntityValidationHelper.checkValidator(DnComponents.COMMONNAME, RegexFieldValidator.class.getName(), "[0-9-]*");
58          EndEntityValidationHelper.checkValidator(DnComponents.COMMONNAME, RegexFieldValidator.class.getName(), "[0-9-]*@([a-z.-]+|localhost)");
59          try {
60              EndEntityValidationHelper.checkValidator(DnComponents.COMMONNAME, RegexFieldValidator.class.getName(), "*");
61              fail("should throw EndEntityFieldValidatorException on invalid regex");
62          } catch (EndEntityFieldValidatorException e) {
63              // NOPMD should throw
64          }
65          
66          // Test some values
67          EndEntityValidationHelper.checkValue(DnComponents.COMMONNAME, makeRegexValidator("[0-9]*"), "123");
68          try {
69              EndEntityValidationHelper.checkValue(DnComponents.COMMONNAME, makeRegexValidator("[0-9]*"), "abc");
70             fail("should throw EndEntityFieldValidatorException on invalid value");
71          } catch (EndEntityFieldValidatorException e) {
72              // NOPMD should throw
73          }
74      }

89          profile.setUse(EndEntityProfile.CLEARTEXTPASSWORD, 0, true);
90          profile.setUse(EndEntityProfile.ISSUANCEREVOCATIONREASON, 0, true);

98      @Test
99      public void testUserFulfillEndEntityProfile() {
100         EndEntityProfile profile = new EndEntityProfile();
101         profile.addField(EndEntityProfile.CARDNUMBER);
102         profile.setRequired(EndEntityProfile.CARDNUMBER, 0, true);
103         profile.setValue(EndEntityProfile.AVAILCAS, 0, Integer.toString(SecConst.ALLCAS));
104 
105         // Test generic that required fields are required
106         EndEntityInformation userdata = new EndEntityInformation("foo", "CN=foo", 123, "", "", new EndEntityType(EndEntityTypes.ENDUSER),
107                 123, CertificateProfileConstants.CERTPROFILE_FIXED_ENDUSER,
108                 SecConst.TOKEN_SOFT_PEM, 0, null);
109         userdata.setPassword("foo123");
110         try {
111             profile.doesUserFulfillEndEntityProfile(userdata, false);
112             fail("cardnumber should be required");
113         } catch (EndEntityProfileValidationException e) {
114             // NOPMD: ignore this is what we want
115         }
116         userdata.setCardNumber("123456789");
117         try {
118             profile.doesUserFulfillEndEntityProfile(userdata, false);
119         } catch (EndEntityProfileValidationException e) {
120             fail("cardnumber was in and should be ok");
121         }
122         
123         // Test that email address can be required as well, and that it does not require an @ sign in it 
124         // (see ECA-5650 about Cisco ISE using the rfc822Name field for MAC address)
125         profile.addField(DnComponents.RFC822NAME);
126         profile.setRequired(DnComponents.RFC822NAME, 0, true);
127         try {
128             profile.doesUserFulfillEndEntityProfile(userdata, false);
129             fail("rfc822Name should be required");
130         } catch (EndEntityProfileValidationException e) {
131             // NOPMD: ignore this is what we want
132         }
133         userdata.setSubjectAltName("rfc822Name=foo@bar.com");
134         try {
135             profile.doesUserFulfillEndEntityProfile(userdata, false);
136         } catch (EndEntityProfileValidationException e) {
137             fail("rfc822Name was in and should be ok");
138         }
139         userdata.setSubjectAltName("rfc822Name=AB:CD:32:45");
140         try {
141             profile.doesUserFulfillEndEntityProfile(userdata, false);
142         } catch (EndEntityProfileValidationException e) {
143             fail("rfc822Name was not an email address, but it should be ok");
144         }
145     }

147     private static Map<String,Serializable> makeRegexValidator(final String regex) {
148         final Map<String,Serializable> map = new LinkedHashMap<String,Serializable>();
149         map.put(RegexFieldValidator.class.getName(), regex);
150         return map;
151     }
