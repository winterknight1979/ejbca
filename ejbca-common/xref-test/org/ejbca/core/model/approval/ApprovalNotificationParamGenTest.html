13  package org.ejbca.core.model.approval;

15  import static org.junit.Assert.assertEquals;
16  import static org.junit.Assert.assertFalse;
17  import static org.junit.Assert.assertNotNull;
18  
19  import org.junit.Test;
20  
21  /** Tests some substitution variables for approval profile notifications
22   * @version $Id: ApprovalNotificationParamGenTest.java 24472 2016-10-06 11:05:28Z aveen4711 $
23   */
24  public class ApprovalNotificationParamGenTest {
25  
26  	@Test
27  	public void testInterpolate(){
28          final ApprovalNotificationParameterGenerator paramGen = new ApprovalNotificationParameterGenerator(123, 3, 47,
29                  "Approval Step $1", ApprovalDataVO.APPROVALTYPENAMES[0], "approved", "CN=requestor\\me", "CN=Last approved by");
30          assertNotNull("paramGen is null", paramGen);
31          
32          String msg = paramGen.interpolate("${approvalRequest.ID} ${approvalRequest.STEP_ID} ${approvalRequest.PARTITION_ID} ${approvalRequest.PARTITION_NAME} ${approvalRequest.TYPE} ${approvalRequest.WORKFLOWSTATE} ${approvalRequest.REQUESTOR} ${approvalRequest.APPROVALADMIN}");
33          assertFalse("Interpolating message failed", (msg==null || msg.length()==0));
34          assertEquals("123 3 47 Approval Step $1 APDUMMY approved CN=requestor\\me CN=Last approved by", msg);
35  		
36  	}
