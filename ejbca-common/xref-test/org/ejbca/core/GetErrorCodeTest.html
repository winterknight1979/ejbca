13  package org.ejbca.core;
14  
15  
16  import org.cesecore.CesecoreException;
17  import org.cesecore.ErrorCode;
18  import org.junit.Test;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.junit.Assert.assertNull;
22  
23  /**
24   * Tests the static EjbcaException.getErrorCode(Throwable). Although it is placed in the EjbcaException
25   * this static method can be invoked for every other Throwable. Result from this method can determine
26   * how the purging of the exception message and stacktrace can be done over peers. 
27   * See PeerRaSerialization.SerializationExceptionHolder for more details about purging the possible
28   * sensitive information with Exceptions over peers
29   * 
30   * @version $Id: GetErrorCodeTest.java 24934 2016-12-20 10:51:26Z mikekushner $
31   */
32  public class GetErrorCodeTest {
33  
34  	@Test
35      public void testEjbcaExceptionDefaultConstructor() {
36          assertNull("EjbcaException.getErrorCode of the EjbcaException() is not null", EjbcaException.getErrorCode(new EjbcaException()));
37  	}
38  	
39  	@Test
40      public void testEjbcaExceptionMessageConstructor() {
41          assertNull("EjbcaException.getErrorCode of the EjbcaException(String message) is not null", EjbcaException.getErrorCode(new EjbcaException("Exception message...")));
42      }
43  	
44  	@Test
45      public void testEjbcaExceptionErrorCodeConstructor() {
46          assertEquals("EjbcaException.getErrorCode of the EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS) is not ErrorCode.APPROVAL_ALREADY_EXISTS",
47                  EjbcaException.getErrorCode(new EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS)), ErrorCode.APPROVAL_ALREADY_EXISTS);
48      }
49  
50      @Test
51      public void testCesecoreExceptionDefaultConstructor() {
52          assertNull("EjbcaException.getErrorCode of the CesecoreException() is not null", EjbcaException.getErrorCode(new CesecoreException()));
53      }
54  
55      @Test
56      public void testCesecoreExceptionMessageConstructor() {
57          assertNull("EjbcaException.getErrorCode of the CesecoreException(String message) is not null", EjbcaException.getErrorCode(new CesecoreException("Exception message")));
58      }
59  
60      @Test
61      public void testCesecoreExceptionErrorCodeConstructor() {
62          assertEquals("EjbcaException.getErrorCode of the CesecoreException(ErrorCode.APPROVAL_ALREADY_EXISTS) is not ErrorCode.APPROVAL_ALREADY_EXISTS",
63                  EjbcaException.getErrorCode(new CesecoreException(ErrorCode.APPROVAL_ALREADY_EXISTS)), ErrorCode.APPROVAL_ALREADY_EXISTS);
64      }
65  
66      @Test
67      public void testEjbcaExceptionAsCauseOfAnException() {
68          assertEquals("EjbcaException.getErrorCode of Exception with the cause EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS)) is not ErrorCode.APPROVAL_ALREADY_EXISTS",
69                  EjbcaException.getErrorCode(new Exception(new EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS))), ErrorCode.APPROVAL_ALREADY_EXISTS);
70      }
71  
72      @Test
73      public void testEjbcaExceptionAsCauseOfAnEjbcaException() {
74          assertEquals("EjbcaException.getErrorCode of EjbcaException(ErrorCode.APPROVAL_REQUEST_ID_NOT_EXIST)) with the cause EjbcaException with an error code is not ErrorCode.APPROVAL_REQUEST_ID_NOT_EXIST",
75                  EjbcaException.getErrorCode(new EjbcaException(ErrorCode.APPROVAL_REQUEST_ID_NOT_EXIST, new EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS))), ErrorCode.APPROVAL_REQUEST_ID_NOT_EXIST);
76      }
77  
78      @Test
79      public void testExceptionAsCauseOfAnEjbcaException() {
80          assertEquals("EjbcaException.getErrorCode of EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS)) with the cause Exception is not ErrorCode.APPROVAL_ALREADY_EXISTS",
81                  EjbcaException.getErrorCode(new Exception(new EjbcaException(ErrorCode.APPROVAL_ALREADY_EXISTS))), ErrorCode.APPROVAL_ALREADY_EXISTS);
82      }
83  
84      @Test
85      public void testNestedExceptions() {
86          assertNull("EjbcaException.getErrorCode of the Exception(new Exception()) is not null", EjbcaException.getErrorCode(new Exception(new Exception())));
87      }
88  }
