13  package org.ejbca.ui.cli.infrastructure.command;
14  
15  import java.util.Set;
16  
17  /**
18   * A marker interface (mostly)
19   * 
20   * @version $Id: CliCommandPlugin.java 19902 2014-09-30 14:32:24Z anatom $
21   *
22   */
23  public interface CliCommandPlugin extends CliCommand {
24      
25      /**
26       * @return the main entrance path to the implementing command.
27       */
28      String getMainCommand();
29      
30      /**
31       * 
32       * @return a set of aliases to the command
33       */
34      Set<String> getMainCommandAliases();
35      
36      /**
37       * 
38       * @return a path of super commands leading to this command. Conflicts will be resolved at runtime.
39       */
40      String[] getCommandPath();
41      
42      /**
43       * 
44       * @return aliases for the super commands leading to this path. Conflicts will be resolved at runtime.
45       */
46      Set<String[]> getCommandPathAliases();
47  }
