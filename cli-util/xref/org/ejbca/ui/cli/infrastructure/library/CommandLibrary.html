31  /**
32   * A library class to register and interface with commands. 
33   * 
34   * @version $Id: CommandLibrary.java 26134 2017-07-06 06:12:16Z oskareriksson $
35   *
36   */
37  
38  public enum CommandLibrary {
39      INSTANCE;
40  
41      private static final String TAB = "    ";
42      private static final String DELIMITER = "--------------------------------\n";
43  
44      private final Logger log = Logger.getLogger(CommandLibrary.class);
45  
46      private Branch root;
47  
48      private CommandLibrary() {
49          synchronized (this) {
50              if (root == null) {
51                  root = new Branch("");
52                  ServiceLoader<? extends CliCommandPlugin> serviceLoader = ServiceLoader.load(CliCommandPlugin.class);
53                  try {
54                      for (Iterator<? extends CliCommandPlugin> iterator = serviceLoader.iterator(); iterator.hasNext();) {
55                          CliCommandPlugin command = iterator.next();
56                          root.addChild(command, command.getCommandPath());
57                          if (!command.getCommandPathAliases().isEmpty()) {
58                              Iterator<String[]> aliasIterator = command.getCommandPathAliases().iterator();
59                              while (aliasIterator.hasNext()) {
60                                  root.addChild(command, true, aliasIterator.next());
61                              }
62                          }
63                      }
64                  } catch (ServiceConfigurationError e) {
65                      if (e.getCause() instanceof IllegalStateException && e.getCause().getLocalizedMessage().contains("No EJB receiver")) {
66                          log.error("Error: CLI could not contact EJBCA instance. Either your application server is not up and running,"
67                                  + " EJBCA has not been deployed successfully, or some firewall rule is blocking the CLI from the application server.");
68                          System.exit(1);
69                      } else {
70                          throw e;
71                      }
72                  }
73              }
74          }
75      }
76  
77      public CommandResult findAndExecuteCommandFromParameters(String... parameters) {
78          return root.execute(parameters);
79      }
80  
81      public void listRootCommands() {
82          root.printManPage();
83      }

85      /**
86       * 
87       * @param args The full argument list from a command. Just pass the whole thing in there and the library will try to figure out if a command is 
88       *              hidden in there somewhere.
89       * @return true if a command can be found that matches the path in the args
90       */
91      public boolean doesCommandExist(String... args) {
92          return root.doesCommandExist(args);
93      }

95      /**
96       * A private subclass which describes a branch in the command tree. A branch can contain both a number of subbranches as well as a number of 
97       * direct commands. 
98       * 
99       * @version $Id: CommandLibrary.java 26134 2017-07-06 06:12:16Z oskareriksson $
100      *
101      */
102     private static class Branch {
103         private final String name;
104         //Denotes that a branch is an alias branch, i.e it won't show up in the man page. 
105         private final boolean isAlternate;
106         private Map<String, Branch> subBranches = new HashMap<String, Branch>();
107         private Map<String, CliCommandPlugin> commands = new HashMap<String, CliCommandPlugin>();
108         private Map<String, CliCommandPlugin> alternateCommands = new HashMap<String, CliCommandPlugin>();
109 
110         public Branch(final String name) {
111             this.name = name;
112             this.isAlternate = false;
113         }
114 
115         public Branch(final String name, final boolean isAlternate) {
116             this.name = name;
117             this.isAlternate = isAlternate;
118         }
119 
120         /**
121          * Adds a child branch 
122          * 
123          * @param command the command to add
124          * @param path the remaining path elements
125          */
126         public void addChild(CliCommandPlugin command, String... path) {
127             addChild(command, false, path);
128         }
129 
130         /**
131          * Recursively adds a child branch 
132          * 
133          * @param command the command
134          * @param isAlternatePath true if the path is an alternate
135          * @param path the remaining path elements
136          */
137         public void addChild(CliCommandPlugin command, boolean isAlternatePath, String... path) {
138             if (path.length == 0) {
139                 Iterator<String> aliases = command.getMainCommandAliases().iterator();
140                 while (aliases.hasNext()) {
141                     String alias = aliases.next();             
142                     if (commands.containsKey(alias)) {
143                         throw new CliCommandLibraryConflictException("Command alias" + alias + " for command " + command.getMainCommand()
144                                 + " has been added twice.");
145                     } else if (alternateCommands.containsKey(alias)) {
146                         throw new CliCommandLibraryConflictException("Command alias" + alias + " for command " + command.getMainCommand()
147                                 + " has been added twice.");
148                     } else {
149                         alternateCommands.put(alias, command);
150                     }
151                 }
152                 if (commands.containsKey(command.getMainCommand())) {
153                     throw new CliCommandLibraryConflictException("Command " + command.getMainCommand() + "(" + command.getClass()
154                             + ") has been added twice. Conflicts with " + commands.get(command.getMainCommand()).getClass());
155                 } else {
156                     commands.put(command.getMainCommand(), command);
157                 }
158             } else {
159                 Branch childBranch;
160                 final String childName = path[0];
161                 if (childName == command.getMainCommand() || command.getMainCommandAliases().contains(childName)) {
162                     throw new CliCommandLibraryConflictException("Naming conflict when attempting to construct command library: " + childName);
163                 }
164                 if (subBranches.containsKey(childName)) {
165                     childBranch = (Branch) subBranches.get(path[0]);
166                 } else {
167                     childBranch = new Branch(childName, isAlternatePath);
168                     subBranches.put(childBranch.getKey(), childBranch);
169                 }
170                 childBranch.addChild(command, Arrays.copyOfRange(path, 1, path.length));
171             }
172         }
173 
174         /**
175          * 
176          * @param args a list of arguments
177          * @return true if command exists, or if a request to print the man page of a subbranch has come in. 
178          */
179         public boolean doesCommandExist(String... args) {
180             if (args.length == 0) {
181                 return false;
182             } else {
183                 String key = args[0].toLowerCase(Locale.ENGLISH);
184                 if (commands.containsKey(key)) {
185                     return true;
186                 } else if (alternateCommands.containsKey(key)) {
187                     return true;
188                 } else {
189                     if (subBranches.containsKey(key)) {
190                         String[] subset = Arrays.copyOfRange(args, 1, args.length);
191                         if (subset.length == 0) {
192                             //Counts as a request for a man page
193                             return true;
194                         } else {
195                             return subBranches.get(key).doesCommandExist(subset);
196                         }
197                     } else {
198                         return false;
199                     }
200                 }
201             }
202 
203         }
204 
205         public void printManPage() {
206             StringBuffer stringBuffer = new StringBuffer();
207             List<String> sortedSubCommandList = new ArrayList<String>();
208             for (String subBranchKey : subBranches.keySet()) {
209                 Branch subBranch = subBranches.get(subBranchKey);
210                 if (!subBranch.isAlternate()) {
211                     sortedSubCommandList.add(subBranch.getName());
212                 }
213             }
214 
215             Collections.sort(sortedSubCommandList, new Comparator<String>() {
216                 @Override
217                 public int compare(String o1, String o2) {
218                     return o1.compareToIgnoreCase(o2);
219                 }
220             });
221             if (sortedSubCommandList.size() > 0) {
222                 stringBuffer.append(DELIMITER);
223                 stringBuffer.append("The following " + (name.length() > 0 ? "sub" : "") + "categories are available"
224                         + (name.length() > 0 ? " for the command '" + name + "'" : "") + ":\n");
225                 stringBuffer.append(TAB + "[ ");
226                 final String subcommandDelimiter = " | ";
227                 for (String subCommand : sortedSubCommandList) {
228                     stringBuffer.append(subCommand + subcommandDelimiter);
229                 }
230                 stringBuffer.replace(stringBuffer.length() - subcommandDelimiter.length(), stringBuffer.length(), " ]\n");
231             }
232             List<String> sortedCommandList = new ArrayList<String>();
233             sortedCommandList.addAll(commands.keySet());
234             Collections.sort(sortedCommandList, new Comparator<String>() {
235                 @Override
236                 public int compare(String o1, String o2) {
237                     return o1.compareToIgnoreCase(o2);
238                 }
239             });
240             int longestCommandLength = 0;
241             for (String command : sortedCommandList) {
242                 if (command.length() > longestCommandLength) {
243                     longestCommandLength = command.length();
244                 }
245             }
246             String indent = tab(longestCommandLength / TAB.length() + 2);
247             if (sortedCommandList.size() > 0) {
248                 stringBuffer.append(DELIMITER);
249                 stringBuffer.append((sortedSubCommandList.size() > 0 ? "And the " : "The ") + "following commands are available:\n");
250                 for (String command : sortedCommandList) {
251                     stringBuffer.append(TAB + command + indent.substring(command.length()) + commands.get(command).getCommandDescription() + "\n");
252                 }
253             }
254             stringBuffer.append("\nType a command and \"--help\" for more information.\n");
255             INSTANCE.log.info(stringBuffer.toString());
256         }
257 
258         private final String tab(int indentation) {
259             StringBuilder sb = new StringBuilder();
260             for (int i = 0; i < indentation; i++) {
261                 sb.append(TAB);
262             }
263             return sb.toString();
264         }

266         public CommandResult execute(String... parameters) {
267             if (parameters.length == 0) {
268                 //We only got to a branch
269                 if (isAlternate) {
270                     INSTANCE.log.warn("WARNING: The path used is an unlisted alternate path and may be deprecated, and may cease to exist at any point."
271                             + " Please start using the updated path as soon as possible.\n");
272                 }
273                 printManPage();
274                 return CommandResult.SUCCESS;
275             } else {
276                 String key = parameters[0].toLowerCase(Locale.ENGLISH);
277                 if (commands.containsKey(key) || alternateCommands.containsKey(key)) {
278                     if (isAlternate) {
279                         INSTANCE.log.warn("WARNING: The path used is an unlisted alternate path and may be deprecated, and may cease to exist at any point."
280                                 + " Please start using the updated path as soon as possible.\n");
281                     }
282                     if(alternateCommands.containsKey(key)) {
283                         INSTANCE.log.warn("WARNING: The command \"" + key + "\" used is an unlisted alternate command and may be deprecated, and may cease to exist at any point."
284                                 + " Please start using the updated command \"" + alternateCommands.get(key).getMainCommand() + "\" as soon as possible.\n");
285                         return alternateCommands.get(key).execute(Arrays.copyOfRange(parameters, 1, parameters.length));
286                     } else {
287                         return commands.get(key).execute(Arrays.copyOfRange(parameters, 1, parameters.length));
288                     }
289                 } else if (!subBranches.containsKey(key)) {
290                     if (isAlternate) {
291                         INSTANCE.log.warn("WARNING: The path used is an unlisted alternate path and may be deprecated, and may cease to exist at any point."
292                                 + " Please start using the updated path as soon as possible.\n");
293                     }
294                     printManPage();
295                     return CommandResult.SUCCESS;
296                 } else {
297                     return subBranches.get(key).execute(Arrays.copyOfRange(parameters, 1, parameters.length));
298                 }
299             }
300         }
301 
302         /**
303          * 
304          * @return the name, but in lowercase. 
305          */
306         private final String getKey() {
307             return name.toLowerCase(Locale.ENGLISH);
308         }
309 
310         private final String getName() {
311             return name;
312         }
313 
314         private final boolean isAlternate() {
315             return isAlternate;
316         }

318     }
