13  package org.ejbca.ui.cli.infrastructure.parameter;
14  
15  import org.ejbca.ui.cli.infrastructure.parameter.enums.MandatoryMode;
16  import org.ejbca.ui.cli.infrastructure.parameter.enums.ParameterMode;
17  import org.ejbca.ui.cli.infrastructure.parameter.enums.StandaloneMode;
18  
19  /**
20   * Wrapper class for a command parameter
21   * 
22   * @version $Id: Parameter.java 26057 2017-06-22 08:08:34Z anatom $
23   *
24   */
25  public class Parameter {
26  
27      private final String keyWord;
28      private final StandaloneMode allowStandAlone;
29      private final ParameterMode parameterMode;
30      private final MandatoryMode mandatoryMode;
31      private final String instruction;
32      private final String name;
33      //Uncommon value set as false if this Parameter shouldn't be listed as part of the man page.
34      private boolean allowList = true;
35  
36      /**
37       * Constructor for defining a parameter
38       * 
39       * @param keyWord The keyword used to identify this parameter. Commonly prefixed with a dash ('-')
40       * @param name What this parameter denotes. Used for documentation purposes.
41       * @param mandatoryMode Defines whether this parameter is mandatory or not. 
42       * @param allowStandAlone true if this parameter can be inputed without its keyword. 
43       * @param parameterMode the type of parameter, if it requires an argument, if it's a flag, or a password etc
44       * @param instruction Instructiion
45       */
46      public Parameter(String keyWord, String name, MandatoryMode mandatoryMode, StandaloneMode allowStandAlone, ParameterMode parameterMode,
47              String instruction) {
48          //Perform validation
49          if (allowStandAlone.equals(StandaloneMode.ALLOW) && !parameterMode.equals(ParameterMode.ARGUMENT)) {
50              throw new IllegalStateException("A non argument parameter can not be set to standalone.");
51          }
52          this.keyWord = keyWord;
53          this.allowStandAlone = allowStandAlone;
54          this.parameterMode = parameterMode;
55          this.mandatoryMode = mandatoryMode;
56          this.instruction = instruction;
57          this.name = name;
58      }
59  
60      public String getName() {
61          return name;
62      }
63  
64      public boolean isMandatory() {
65          return mandatoryMode.isMandatory();
66      }
67      
68      public boolean isStandAlone() {
69          return allowStandAlone.isStandAlone();
70      }
71  
72      @Override
73      public int hashCode() {
74          final int prime = 31;
75          int result = 1;
76          result = prime * result + ((keyWord == null) ? 0 : keyWord.hashCode());
77          return result;
78      }
79  
80      @Override
81      public boolean equals(Object obj) {
82          if (this == obj)
83              return true;
84          if (obj == null)
85              return false;
86          if (getClass() != obj.getClass())
87              return false;

89          if (keyWord == null) {
90              if (other.keyWord != null)
91                  return false;
92          } else if (!keyWord.equals(other.keyWord))
93              return false;
94          return true;
95      }
96  
97      public ParameterMode getParameterMode() {
98          return parameterMode;
99      }
100 
101     public StandaloneMode allowStandAlone() {
102         return allowStandAlone;
103     }
104 
105     /**
106      * @return the keyWord
107      */
108     public String getKeyWord() {
109         return keyWord;
110     }
111 
112     public String getInstruction() {
113         return instruction;
114     }
115     
116     /**
117      * Quick factory method for creating flags.
118      * @param keyWord Key word
119      * @param instruction Instruction
120      * @return Parameter
121      * 
122      */
123     public static Parameter createFlag(String keyWord, String instruction) {
124         return new Parameter(keyWord, "", MandatoryMode.OPTIONAL, StandaloneMode.FORBID, ParameterMode.FLAG, instruction);
125     }
126 
127     public boolean allowList() {
128         return allowList;
129     }
130 
131     public void setAllowList(boolean allowList) {
132         this.allowList = allowList;
133     }
134 }
