13  package org.cesecore.authorization.user;
14  
15  import static org.junit.Assert.assertTrue;
16  
17  import org.cesecore.authorization.user.matchvalues.X500PrincipalAccessMatchValue;
18  import org.junit.Test;
19  
20  /**
21   * @version $Id: AccessUserAspectDataTest.java 28007 2018-01-18 12:49:29Z mikekushner $
22   *
23   */
24  public class AccessUserAspectDataTest {
25  
26      @SuppressWarnings("deprecation")
27      @Test
28      public void testGeneratePrimaryKeyWhereRoleNameAndMatchValueAreTheSame() {
29          int alpha = AccessUserAspectData.generatePrimaryKey("foo", 1337, X500PrincipalAccessMatchValue.WITH_COMMONNAME,
30                  AccessMatchType.TYPE_EQUALCASE, "bar1");
31          int beta = AccessUserAspectData.generatePrimaryKey("foo", 1337, X500PrincipalAccessMatchValue.WITH_COMMONNAME,
32                  AccessMatchType.TYPE_EQUALCASE, "bar2");
33          assertTrue("Two identical primary keys were produced for different values", alpha != beta);
34      }
35      
36      @SuppressWarnings("deprecation")
37      @Test
38      public void testGeneratePrimaryKeyWhereRoleNameAndMatchValueAreSwitched() {
39          int alpha = AccessUserAspectData.generatePrimaryKey("foo", 1337, X500PrincipalAccessMatchValue.WITH_COMMONNAME,
40                  AccessMatchType.TYPE_EQUALCASE, "bar");
41          int beta = AccessUserAspectData.generatePrimaryKey("bar", 1337, X500PrincipalAccessMatchValue.WITH_COMMONNAME,
42                  AccessMatchType.TYPE_EQUALCASE, "foo");
43          assertTrue("Two identical primary keys were produced for different values", alpha != beta);
44      }
