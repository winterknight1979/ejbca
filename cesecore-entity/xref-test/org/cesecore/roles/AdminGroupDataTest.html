31  @Deprecated
32  public class AdminGroupDataTest {
33      
34      @Test
35      public void testHasAccessToRule() {
36          final String roleName = "role";
37          AdminGroupData testRole = new AdminGroupData(1, roleName);
38          Map<Integer, AccessRuleData> accessRules = new HashMap<Integer, AccessRuleData>();
39          accessRules.put(1, new AccessRuleData(roleName, "/fuu", AccessRuleState.RULE_ACCEPT, true));
40          accessRules.put(2, new AccessRuleData(roleName, "/foo/bar", AccessRuleState.RULE_DECLINE, true));
41          accessRules.put(3, new AccessRuleData(roleName, "/xyz", AccessRuleState.RULE_DECLINE, true));
42          accessRules.put(4, new AccessRuleData(roleName, "/xyz_abc", AccessRuleState.RULE_ACCEPT, true));
43          accessRules.put(5, new AccessRuleData(roleName, "/recursive", AccessRuleState.RULE_ACCEPT, true));
44          accessRules.put(6, new AccessRuleData(roleName, "/recursivewithdenied", AccessRuleState.RULE_ACCEPT, true));
45          accessRules.put(7, new AccessRuleData(roleName, "/recursivewithdenied/denied", AccessRuleState.RULE_DECLINE, false));
46          accessRules.put(8, new AccessRuleData(roleName, "/non_recursive", AccessRuleState.RULE_ACCEPT, false));
47  
48  
49          testRole.setAccessRules(accessRules);
50          assertFalse("Was incorrectly given access to a rule that should have state unknown", testRole.hasAccessToRule("/", false));
51          assertTrue("Should have been given access to a rule which was explicitly accepted.", testRole.hasAccessToRule("/fuu"));
52          assertFalse("Was incorrectly given access to a rule which was explicitly denied.",testRole.hasAccessToRule("/foo/bar"));
53          assertFalse("Was incorrectly given access to a rule which was denied in a subrule.",testRole.hasAccessToRule("/foo/bar/xyz"));  
54          assertTrue("Should have been given access to a rule with a similar name as one denied", testRole.hasAccessToRule("/xyz_abc"));
55          assertTrue("Should have been given access to a rule with recursive required", testRole.hasAccessToRule("/recursive", true));
56          assertTrue("Should have been given access to a rule with recursive allowed by parent", testRole.hasAccessToRule("/recursive/subrule", false));
57          assertTrue("Should have been given access to a rule with recursive required", testRole.hasAccessToRule("/recursive/subrule", true));
58          assertFalse("[Privilege escalation] Was incorrectly given access to a rule which was has declined in a subrule", testRole.hasAccessToRule("/recursivewithdenied", true));
59          assertFalse("Was incorrectly given access to a rule which was explicitly denied.", testRole.hasAccessToRule("/recursivewithdenied/denied", false));
60          assertFalse("Was incorrectly given access to a rule which was explicitly denied.", testRole.hasAccessToRule("/recursivewithdenied/denied", true));
61          assertFalse("Should not have been given access to a rule ", testRole.hasAccessToRule("/non_recursive", true));
62          assertFalse("Should not have been given access to a rule with recursive required", testRole.hasAccessToRule("/non_recursive", true));
63      }
64      
65      /**
66       * Make sure that access to root is never given by mistake. 
67       */
68      @Test
69      public void testHasAccessToRoot() {
70          final String roleName = "role";
71          AdminGroupData testRole = new AdminGroupData(1, roleName);
72          Map<Integer, AccessRuleData> accessRules = new HashMap<Integer, AccessRuleData>();
73          accessRules.put(1, new AccessRuleData(roleName, "/fuu", AccessRuleState.RULE_ACCEPT, true));
74  
75          testRole.setAccessRules(accessRules);
76          assertFalse("Was incorrectly given access to root.", testRole.hasAccessToRule("/", false));
77          
78      }
