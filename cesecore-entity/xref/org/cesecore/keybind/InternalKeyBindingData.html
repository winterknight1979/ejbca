51      private int id;                 // Internal and static over time representation when referencing this object
52      private String name;            // A human friendly representation of this object
53      private String status;          // The status as a String constant of InternalKeyBindingStatus
54      private String keyBindingType;  // Mapped to implementation class
55      private String certificateId;   // Reference to a Certificate currently in use by the implementation
56      private int cryptoTokenId;      // Reference to a CryptoToken currently in use by the implementation
57      private String keyPairAlias;    // Reference to an alias in the CryptoToken currently in use by the implementation
58      private String rawData;         // Raw data like a with implementation specific details and configuration
59      private long lastUpdate = 0;    // Last update to database, Unix epoch milliseconds

63      public InternalKeyBindingData(final int id, final String name, final InternalKeyBindingStatus status, final String keyBindingType, final String certificateId,
64              final int cryptoTokenId, final String keyPairAlias, final LinkedHashMap<Object, Object> dataMap) {
65          setId(id);
66          setName(name);
67          setStatusEnum(status);
68          setKeyBindingType(keyBindingType);
69          if (certificateId != null) {
70              setCertificateId(certificateId.toLowerCase(Locale.ENGLISH));            
71          } else {
72              setCertificateId(null);
73          }
74          setCryptoTokenId(cryptoTokenId);
75          setKeyPairAlias(keyPairAlias);
76          setDataMap(dataMap);
77          setLastUpdate(System.currentTimeMillis());
78      }

145     @Override
146     public String getProtectString(final int version) {
147         final ProtectionStringBuilderer.html#ProtectionStringBuilder">ProtectionStringBuilder build = new ProtectionStringBuilder(1024);
148         // What is important to protect here is the data that we define

184     @Transient
185     @SuppressWarnings("unchecked")
186     public LinkedHashMap<Object, Object> getDataMap() {
187         try (SecureXMLDecoder decoder = new SecureXMLDecoder(new java.io.ByteArrayInputStream(getRawData().getBytes(StandardCharsets.UTF_8)))) {
188             final Map<?, ?> h = (Map<?, ?>)decoder.readObject();
189             // Handle Base64 encoded string values
190             final LinkedHashMap<Object, Object> dataMap = new Base64GetHashMap(h);
191             return dataMap;
192         } catch (IOException e) {
193             final String msg = "Failed to parse InternalKeyBindingData data map in database: " + e.getMessage();
194             if (log.isDebugEnabled()) {
195                 log.debug(msg + ". Data:\n" + getRawData());
196             }
197             throw new IllegalStateException(msg, e);
198         }
199     }
200 
201     @Transient
202     @SuppressWarnings({"rawtypes", "unchecked"})
203     public void setDataMap(final LinkedHashMap<Object, Object> dataMap) {
204         // We must base64 encode string for UTF safety
205         final LinkedHashMap<?, ?> a = new Base64PutHashMap();
206         a.putAll((LinkedHashMap)dataMap);
207         final java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream();
208         try {
209             try (XMLEncoder encoder = new XMLEncoder(baos)) {
210                 encoder.writeObject(a);
211             }
212             final String data = baos.toString("UTF8");
213             setRawData(data);
214         } catch (UnsupportedEncodingException e) {
215             throw new IllegalStateException(e);
216         }
217     }
