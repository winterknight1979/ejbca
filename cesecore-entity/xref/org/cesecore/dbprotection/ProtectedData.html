13  package org.cesecore.dbprotection;
14  
15  import java.lang.reflect.InvocationTargetException;
16  import org.apache.log4j.Logger;
17  
18  /**
19   * Used as base class for JPA data beans that has a rowProtection column. The
20   * JPA class should extend this class and implement the simple methods:
21   *
22   * <pre>
23   * &#064;Transient
24   * &#064;Override
25   * String getProtectString(int version) {
26   *   return &quot;concatenation of fields to be integrity protected. Must be
27   *   deterministic and can change with different version of
28   *    rowprotection.&quot;;
29   *   // Example from CertificateProfileData
30   *   // StringBuilder build = new StringBuilder();
31   *   // What is important to protect here is the data that we define, id,
32   *    name and certificate profile data
33   *   // rowVersion is automatically updated by JPA, so it's not important,
34   *    it is only used for optimistic locking
35   *   // build.append(getId()).append(getCertificateProfileName())
36   *           .append(getData());
37   *   // return build.toString();
38   * }
39   *
40   * &#064;Transient
41   * &#064;Override
42   * int getProtectVersion() {
43   *   return 1;
44   * }
45   *
46   * &#064;PrePersist
47   * &#064;PreUpdate
48   * &#064;Transient
49   * &#064;Override
50   * void protectData() {
51   *   super.protectData();
52   * }
53   *
54   * &#064;PostLoad
55   * &#064;Transient
56   * &#064;Override
57   * void verifyData() {
58   *   super.verifyData();
59   * }
60   *
61   * &#064;Override
62   * &#064;Transient
63   * protected String getRowId() {
64   *   return String.valueOf(getPrimaryKey());
65   * }
66   * </pre>
67   *
68   * @version $Id: ProtectedData.java 29874 2018-09-13 09:38:35Z samuellb $
69   */
70  public abstract class ProtectedData {
71  
72      /** Logger. */
73    private static final Logger LOG = Logger.getLogger(ProtectedData.class);
74  
75    /** Implementation for the database protection method used. */
76    protected ProtectedDataImpl impl;
77  
78    /** Definition of the optional database integrity
79     *  protection implementation. */
80    private static final String IMPL_CLASS_NAME =
81        "org.cesecore.dbprotection.ProtectedDataIntegrityImpl";
82    /**
83     * Cache class so we don't have to do Class.forName for every entity object
84     * created.
85     */
86    private static volatile Class<?> implClass = null;
87  
88    /**
89     * Optimization variable so we don't have to check for existence of implClass
90     * for every construction of an entity object.
91     */
92    private static volatile boolean integrityExists = true;
93  
94    /**
95     * A default constructor is needed by JPA. This constructor initializes the
96     * available database integrity protection module, if any is available
97     */
98    public ProtectedData() {
99      if (integrityExists) {
100       try {
101         if (implClass == null) {
102           // We only end up here once, if the class does not exist, we will
103           // never end up here again (ClassNotFoundException)
104           // and if the class exists we will never end up here again (it will
105           // not be null)
106           implClass = Class.forName(IMPL_CLASS_NAME);
107           LOG.debug(
108               "ProtectedDataIntegrityImpl is available, and used, in this"
109                   + " version of EJBCA.");
110         }
111         impl = (ProtectedDataImpl) implClass.getConstructor().newInstance();
112         impl.setTableName(getTableName());
113       } catch (ClassNotFoundException | NoSuchMethodException e) {
114         // We only end up here once, if the class does not exist, we will never
115         // end up here again
116         integrityExists = false;
117         LOG.info(
118             "No database integrity protection available in this version of"
119                 + " EJBCA.");
120         impl = new ProtectedDataNoopImpl();
121       } catch (InstantiationException | InvocationTargetException e) {
122         LOG.error("Error intitilizing database integrity protection: ", e);
123       } catch (IllegalAccessException e) {
124         LOG.error("Error intitilizing database integrity protection: ", e);
125       }
126     } else {
127       impl = new ProtectedDataNoopImpl();
128     }
129   }
130 
131   /**
132    * asks the data class for the string to be protected. Version is -1 for a new
133    * row to be protected, and otherwise a version given earlier from the data
134    * class when storing the row.
135    *
136    * @param rowversion the version of the string that is protected, used as
137    *     input when verifying data. -1 when getting protection string for data
138    *     to be inserted or updated. -1 means that the data class should use it's
139    *     latest version of protect string
140    * @return String to be integrity protected, i.e. input to hmac.
141    */
142   protected abstract String getProtectString(int rowversion);
143 
144   /**
145    * asks the data class for the version of the string that is protected, used
146    * as input to getProtectString() when verifying data. This is used so that
147    * the data class can alter itself with new fields, but still be backwards
148    * compatible and verify older database data. Called when getting the version
149    * for inserts or updates.
150    *
151    * @return int version the latest version of protection string.
152    */
153   protected abstract int getProtectVersion();
154 
155   /**
156    * The extending class must have a database column "rowProtection" that can be
157    * read and set.
158    *
159    * @param rowProtection protect
160    */
161   public abstract void setRowProtection(String rowProtection);
162 
163   /**
164    * @return Protection
165    */
166   public abstract String getRowProtection();
167 
168   /**
169    * Returns id of the row in the database, in case of failure we can see in the
170    * log which row failed to verify.
171    *
172    * @return id of database row, specific for implementing class.
173    */
174   protected abstract String getRowId();
175 
176   /**
177    * @return the database table name. Should be overridden by classes that does
178    *     not share the same name as the database table it maps to (for example
179    *     by subclassing).
180    */
181   protected String getTableName() {
182     return this.getClass().getSimpleName();
183   }
184 
185   /**
186    * Overridden by extending class to be able to use @PrePersist, overriding
187    * class calls super.protectData(). This method creates integrity protection
188    * for the specific entity in the database.
189    *
190    * @throws DatabaseProtectionException (RuntimeException) on error creating
191    *     integrity protection.
192    */
193   protected void protectData() {
194     impl.protectData(this);
195   }
196 
197   /**
198    * Overridden by extending class to be able to use @PostLoad, overriding class
199    * calls super.verifyData(). This method verifies integrity protection for the
200    * specific entity in the database. If the data verification failed, it
201    * invokes {@link #onDataVerificationError(DatabaseProtectionException)}.
202    */
203   protected void verifyData() {
204     try {
205       impl.verifyData(this);
206     } catch (final DatabaseProtectionException e) {
207       onDataVerificationError(e);
208     }
209   }
210 
211   /**
212    * Method that calculates integrity protection of an entity, but does not
213    * store it anywhere. Used primarily to make test protection in order to
214    * exercise the CryptoToken.
215    *
216    * @return the calculated protection string
217    */
218   public String calculateProtection() {
219     return impl.calculateProtection(this);
220   }
221 
222   /**
223    * Throws DatabaseProtectionException if erroronverifyfail is enabled in
224    * databaseprotection.properties and logs a "row protection failed" message on
225    * ERROR level.
226    *
227    * @param e exception
228    * @throws DatabaseProtectionException the exception given as parameter if
229    *     erroronverifyfail is enabled
230    */
231   protected void onDataVerificationError(final DatabaseProtectionException e) {
232     impl.onDataVerificationError(e);
233   }
234 }
