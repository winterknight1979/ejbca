54      /** 
55       * Static helper to create new instance and still allowing deserialization of this class on EJBCA instances that
56       * don't have access to the RoleMember class.
57       * 
58       * Note that this it is in general a bad idea to keep a list of RoleMembers that were part of the Role at a point
59       * in time in this class.
60       * @param identifier ID
61       * @param nameSpace NS
62       * @param roleName Name
63       * @param roleMembers Members
64       * @return Info
65       */
66      public static RoleInformation fromRoleMembers(final int identifier, final String nameSpace, final String roleName, final List<RoleMember> roleMembers) {
67          final String nameSpaceToUse = StringUtils.isEmpty(nameSpace) ? null : nameSpace;
68          final List<AccessUserAspectData> accessUserAspects = new ArrayList<>();
69          if (roleMembers!=null) {
70              for (final RoleMember roleMember : roleMembers) {
71                  final String roleNameForAspect = StringUtils.isEmpty(nameSpace) ? roleName : nameSpace+";"+roleName;
72                  accessUserAspects.add(new AccessUserAspectData(roleNameForAspect, roleMember.getTokenIssuerId(), roleMember.getTokenMatchKey(),
73                          roleMember.getTokenType(), roleMember.getAccessMatchType(), roleMember.getTokenMatchValue()));
74              }
75          }
76          return new RoleInformation(identifier, nameSpaceToUse, roleName, accessUserAspects);
77      }

102         final int prime = 37;
103         int result = 1;
