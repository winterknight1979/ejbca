30  /**
31   * Entity bean for Role members. Does not correspond to a physical entity, but rather to an individual credential linked to an entity. The same 
32   * individuals may share the same credential (such as belonging to the same organization, or sharing an account, while one individual may have access
33   * to several credentials (such as a user using several different certificates for identification depending on location). 
34   * 
35   *  Each member is linked to a Role, though not intrinsically so via foreign keys 
36   * 
37   * @version $Id: RoleMemberData.java 25634 2017-04-03 12:02:08Z jeklund $
38   */

40  @Table(name = "RoleMemberData")

68      /**
69       * Slightly more verbose constructor used for upgrades.
70       * 
71       * @param primaryKey the primary key for this object. It's required to check the database for any objects with the same key, otherwise that 
72       *  object will be overridden
73       * @param tokenType a string which defined the implementation of AcceessMatchValue used by this member
74       * @param tokenIssuerId the issuer of token if relevant or 0 (RoleMember.NO_ISSUER) otherwise
75       * @param tokenMatchKey the integer value determining how to interpret the tokenMatchValue, defined in a class that inherits the interface AcceessMatchValue
76       * @param tokenMatchOperator how to perform the match. 0 (AccessMatchType.UNUSED.getNumericValue())to let the determine this from tokenSubType.
77       * @param tokenMatchValue the actual value with which to match
78       * @param roleId the ID of the role to which this member belongs. May be null.
79       * @param description a human readable description of this role member. Null will be treated as an empty String.
80       */
81      public RoleMemberData(final int primaryKey, final String tokenType, final int tokenIssuerId, final int tokenMatchKey, final int tokenMatchOperator,
82              final String tokenMatchValue, final int roleId, String description) {
83          this.primaryKey = primaryKey;
84          this.tokenType = tokenType;
85          this.tokenIssuerId = tokenIssuerId;
86          this.tokenMatchKey = tokenMatchKey;
87          this.tokenMatchOperator = tokenMatchOperator;
88          this.tokenMatchValueColumn = tokenMatchValue;
89          this.roleId = roleId;
90          this.setDescription(description);
91      }

268     @Override
269     public int compareTo(RoleMemberData o) {
270         return new CompareToBuilder().append(this.tokenType, o.tokenType).append(this.tokenIssuerId, o.tokenIssuerId).append(this.tokenMatchKey, o.tokenMatchKey)
271                 .append(this.tokenMatchOperator, o.tokenMatchOperator).append(this.tokenMatchValueColumn, o.tokenMatchValueColumn).toComparison();
272     }
