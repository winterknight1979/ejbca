102     @Transient
103     @SuppressWarnings("unchecked")
104     public LinkedHashMap<Object, Object> getDataMap() {
105         try (final SecureXMLDecoder decoder = new SecureXMLDecoder(new ByteArrayInputStream(getRawData().getBytes(StandardCharsets.UTF_8)));) {
106             // Handle Base64 encoded string values
107             return new Base64GetHashMap((Map<?, ?>)decoder.readObject());
108         } catch (IOException e) {
109             final String msg = "Failed to parse data map for role '" + roleName + "': " + e.getMessage();
110             if (log.isDebugEnabled()) {
111                 log.debug(msg + ". Data:\n" + getRawData());
112             }
113             throw new IllegalStateException(msg, e);
114         }
115     }
