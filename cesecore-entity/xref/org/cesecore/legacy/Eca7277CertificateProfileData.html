14  package org.cesecore.legacy;
15  
16  import org.apache.log4j.Logger;
17  import org.cesecore.certificates.certificateprofile.CertificateProfileData;
18  
19  /**
20   * CertificateProfileData was one of the old classes that used serialized binary data in a blob in the database.
21   * In order for CertificateProfileData.getProtectString to produce a String, the object was read in and then the
22   * getData() method returned a LinkedHashMap. StringBuilder.build() called toString() on the LinkedHashMap, which
23   * in turn called toString() on the objects in it. One of the objects was an ApprovalRequestType. In ECA-6518 the
24   * toString() method was overridden in this enum, causing the string representation to be different. This caused
25   * rowProtection signature verification to fail when you upgraded from a version &lt;=6.11 to a version &gt;=6.12.
26   *
27   * The aim of Eca7277CertificateProfileData is to fix this problem. Eca7277CertificateProfileData behaves exactly
28   * like CertificateProfileData, but replaces any faulty substrings found in the protect string returned by
29   * getProtectString() before returning it to the caller.
30   *
31   * This class and all references to it can be safely removed when we drop support for EJBCA 6.14.
32   *
33   * @version $Id: Eca7277CertificateProfileData.java 29814 2018-09-05 15:03:36Z bastianf $
34   */
35  @SuppressWarnings("serial")
36  @Deprecated
37  public class Eca7277CertificateProfileData extends CertificateProfileData {
38      private final static Logger log = Logger.getLogger(Eca7277CertificateProfileData.class);
39  
40      public Eca7277CertificateProfileData(final CertificateProfileData data) {
41          super(data.getId(), data.getCertificateProfileName(), data.getCertificateProfile());
42          this.setRowVersion(data.getRowVersion());
43          this.setRowProtection(data.getRowProtection());
44      }
45  
46      @Override
47      protected String getProtectString(final int rowversion) {
48          if (log.isDebugEnabled()) {
49              log.debug("Verification of row protected data for Certificate Profile " + getCertificateProfileName()
50                      + " failed. Trying to fix the protect string now.");
51          }
52          // Signature was computer over "Add or Edit End Entity" ect. in EJBCA 6.11-6.14
53          return super.getProtectString(rowversion)
54              .replace("ADDEDITENDENTITY=", "Add or Edit End Entity=")
55              .replace("KEYRECOVER=", "Key Recovery=")
56              .replace("REVOCATION=", "Revocation=")
57              .replace("ACTIVATECA=", "CA Activation=");
58      }
59  }
