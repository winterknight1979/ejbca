14  package org.cesecore.legacy;
15  
16  import org.apache.log4j.Logger;
17  import org.cesecore.certificates.certificateprofile.CertificateProfileData;
18  
19  /**
20   * CertificateProfileData was one of the old classes that used serialized binary
21   * data in a blob in the database. In order for
22   * CertificateProfileData.getProtectString to produce a String, the object was
23   * read in and then the getData() method returned a LinkedHashMap.
24   * StringBuilder.build() called toString() on the LinkedHashMap, which in turn
25   * called toString() on the objects in it. One of the objects was an
26   * ApprovalRequestType. In ECA-6518 the toString() method was overridden in this
27   * enum, causing the string representation to be different. This caused
28   * rowProtection signature verification to fail when you upgraded from a version
29   * &lt;=6.11 to a version &gt;=6.12.
30   *
31   * <p>The aim of Eca7277CertificateProfileData is to fix this problem.
32   * Eca7277CertificateProfileData behaves exactly like CertificateProfileData,
33   * but replaces any faulty substrings found in the protect string returned by
34   * getProtectString() before returning it to the caller.
35   *
36   * <p>This class and all references to it can be safely removed when we drop
37   * support for EJBCA 6.14.
38   *
39   * @version $Id: Eca7277CertificateProfileData.java 29814 2018-09-05 15:03:36Z
40   *     bastianf $
41   */
42  @SuppressWarnings("serial")
43  @Deprecated
44  public class Eca7277CertificateProfileData extends CertificateProfileData {
45    /** Logger. */
46      private static final Logger LOG =
47        Logger.getLogger(Eca7277CertificateProfileData.class);
48  
49    /**
50     * @param data data
51     */
52    public Eca7277CertificateProfileData(final CertificateProfileData data) {
53      super(
54          data.getId(),
55          data.getCertificateProfileName(),
56          data.getCertificateProfile());
57      this.setRowVersion(data.getRowVersion());
58      this.setRowProtection(data.getRowProtection());
59    }
60  
61    @Override
62    protected String getProtectString(final int rowversion) {
63      if (LOG.isDebugEnabled()) {
64        LOG.debug(
65            "Verification of row protected data for Certificate Profile "
66                + getCertificateProfileName()
67                + " failed. Trying to fix the protect string now.");
68      }
69      // Signature was computer over "Add or Edit End Entity" ect. in EJBCA
70      // 6.11-6.14
71      return super.getProtectString(rowversion)
72          .replace("ADDEDITENDENTITY=", "Add or Edit End Entity=")
73          .replace("KEYRECOVER=", "Key Recovery=")
74          .replace("REVOCATION=", "Revocation=")
75          .replace("ACTIVATECA=", "CA Activation=");
76    }
77  }
