151   /**
152    * @return map
153    */
154   @Transient
155   @SuppressWarnings("unchecked")
156   public LinkedHashMap<Object, Object> getDataMap() {
157     try (SecureXMLDecoder decoder =
158         new SecureXMLDecoder(
159             new ByteArrayInputStream(
160                 getRawData().getBytes(StandardCharsets.UTF_8)))) {
161       final Map<?, ?> h = (Map<?, ?>) decoder.readObject();
162       // Handle Base64 encoded string values
163       final LinkedHashMap<Object, Object> dataMap = new Base64GetHashMap(h);
164       return dataMap;
165     } catch (IOException e) {
166       final String msg =
167           "Failed to parse data map for "
168               + profileType
169               + " '"
170               + profileName
171               + "': "
172               + e.getMessage();
173       if (LOG.isDebugEnabled()) {
174         LOG.debug(msg + ". Data:\n" + getRawData());
175       }
176       throw new IllegalStateException(msg, e);
177     }
178   }
179 
180   /**
181    * @param dataMap map
182    */
183   @Transient
184   @SuppressWarnings({"rawtypes", "unchecked"})
185   public void setDataMap(final LinkedHashMap<Object, Object> dataMap) {
186     try {
187       // We must base64 encode string for UTF safety
188       final LinkedHashMap<?, ?> a = new Base64PutHashMap();
189       a.putAll((LinkedHashMap) dataMap);
190       final ByteArrayOutputStream baos = new ByteArrayOutputStream();
191       final XMLEncoder encoder = new XMLEncoder(baos);
192       encoder.writeObject(a);
193       encoder.close();
194       final String data = baos.toString("UTF8");
195       setRawData(data);
196     } catch (UnsupportedEncodingException e) {
197       throw new IllegalStateException(e);
198     }
199   }

250   /** @return the value object representation of this database row */
251   @SuppressWarnings("unchecked")
252   @Transient
253   public Profile getProfile() {
254     LinkedHashMap<Object, Object> datamap =
255         (LinkedHashMap<Object, Object>) getDataMap();
256     Class<? extends Profile> implementationClass =
257         (Class<? extends Profile>) datamap.get(Profile.PROFILE_TYPE);
258     if (implementationClass == null) {
259       throw new IllegalStateException(
260           "No implementation class available for profile '"
261               + profileName
262               + "'");
263     }
264     Profile returnValue;
265     try {
266       returnValue = implementationClass.getConstructor().newInstance();
267     } catch (InstantiationException
268         | IllegalAccessException
269         | NoSuchMethodException
270         | InvocationTargetException e) {
271       throw new IllegalStateException(
272           "Could not instansiate class of type "
273               + implementationClass.getCanonicalName()
274               + " for profile '"
275               + profileName
276               + "'",
277           e);
278     }
279     returnValue.setProfileName(profileName);
280     returnValue.setProfileId(id);
281     returnValue.setDataMap(datamap);
282     return returnValue;
283   }
